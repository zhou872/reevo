import numpy as np
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray, 
                  sample_size: int = 100,
                  learning_rate: float = 0.1) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with dynamic weight adaptation and graph attention.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        sample_size: Number of random tours to sample for weight adaptation
        learning_rate: Rate for updating component weights based on sampled tours
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive Neighborhood Selection ====================
    adjacency_mask = np.zeros((n, n), dtype=bool)
    base_k = max(2, int(np.sqrt(n)))
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Density-aware neighborhood
        sorted_dists = np.sort(distances)
        avg_near_dist = np.mean(sorted_dists[:base_k])
        cv = np.std(sorted_dists[:base_k]) / (avg_near_dist + 1e-10)
        
        adaptive_k = min(n-1, int(base_k * (1.0 + 0.5 * cv)))
        nn_indices = np.argpartition(distances, adaptive_k)[:adaptive_k]
        adjacency_mask[i, nn_indices] = True
        
        # Expand to similar edges
        if len(nn_indices) > 0:
            threshold = 1.3 * np.max(distances[nn_indices])
            adjacency_mask[i, distances <= threshold] = True
    
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Distance-Based Initial Scores ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    for i in range(n):
        row_dists = distance_matrix[i].copy()
        row_dists[i] = np.inf
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            valid_dists = row_dists[valid_mask]
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            weights = 1.0 / ranks
            weights /= np.sum(weights)
            inv_vals = 1.0 / (valid_dists + 1e-10)
            inv_dist[i, valid_mask] = inv_vals * weights
    
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    
    # Normalize
    inv_dist_max = np.max(inv_dist)
    if inv_dist_max > 0:
        inv_dist = inv_dist / inv_dist_max
    
    # ==================== 3. Minimum Spanning Tree Components ====================
    mst_mask = np.zeros((n, n), dtype=bool)
    visited = np.zeros(n, dtype=bool)
    min_edge = np.full(n, np.inf)
    parent = np.full(n, -1, dtype=int)
    min_edge[0] = 0
    
    for _ in range(n):
        u = -1
        min_val = np.inf
        for i in range(n):
            if not visited[i] and min_edge[i] < min_val:
                min_val = min_edge[i]
                u = i
        if u == -1:
            break
        visited[u] = True
        if parent[u] != -1:
            mst_mask[u, parent[u]] = True
            mst_mask[parent[u], u] = True
        for v in range(n):
            if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                min_edge[v] = distance_matrix[u, v]
                parent[v] = u
    
    mst_degrees = np.sum(mst_mask, axis=1)
    odd_degree_nodes = np.where(mst_degrees % 2 == 1)[0]
    
    # ==================== 4. Efficient Two-Hop Analysis ====================
    two_hop_score = np.zeros_like(distance_matrix)
    neighbor_sets = [set(np.where(adjacency_mask[i])[0]) for i in range(n)]
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                candidates = (neighbor_sets[i] | neighbor_sets[j]) - {i, j}
                if candidates:
                    candidate_list = list(candidates)
                    two_hop_dists = distance_matrix[i, candidate_list] + distance_matrix[candidate_list, j]
                    min_two_hop = np.min(two_hop_dists) if len(two_hop_dists) > 0 else np.inf
                    direct_dist = distance_matrix[i, j]
                    if direct_dist > 1e-10 and min_two_hop < np.inf:
                        # Efficiency score: how much better is direct edge
                        efficiency = min_two_hop / (direct_dist + 1e-10)
                        two_hop_score[i, j] = two_hop_score[j, i] = np.exp(-efficiency + 1)
    
    # Normalize two-hop score
    two_hop_max = np.max(two_hop_score)
    if two_hop_max > 0:
        two_hop_score = two_hop_score / two_hop_max
    
    # ==================== 5. Graph Attention Mechanism ====================
    # Compute node importance using PageRank-inspired approach
    node_importance = np.zeros(n)
    damping = 0.85
    
    # Build transition matrix based on inverse distance
    transition = np.zeros((n, n))
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            inv_vals = 1.0 / (row[valid_mask] + 1e-10)
            transition[i, valid_mask] = inv_vals / np.sum(inv_vals)
    
    # Power iteration for importance scores
    node_importance = np.ones(n) / n
    for _ in range(20):  # Fixed number of iterations
        new_importance = damping * transition.T @ node_importance + (1 - damping) / n
        node_importance = new_importance / np.sum(new_importance)
    
    # Attention score for edges
    attention_score = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Attention based on node importance and edge efficiency
                importance_factor = node_importance[i] * node_importance[j]
                efficiency_factor = two_hop_score[i, j]
                attention_score[i, j] = attention_score[j, i] = importance_factor * (1.0 + efficiency_factor)
    
    attention_max = np.max(attention_score)
    if attention_max > 0:
        attention_score = attention_score / attention_max
    
    # ==================== 6. Spectral Partition Bonus ====================
    partition_bonus = np.zeros_like(distance_matrix)
    
    if n >= 10:  # Only compute for reasonably sized graphs
        # Simple spectral clustering using power iteration
        try:
            # Build similarity matrix
            sigma = np.median(distance_matrix[adjacency_mask])
            similarity = np.exp(-distance_matrix**2 / (2 * sigma**2))
            similarity[~adjacency_mask] = 0
            np.fill_diagonal(similarity, 0)
            
            # Compute normalized Laplacian's second eigenvector
            degree = similarity.sum(axis=1)
            D_inv_sqrt = np.diag(1.0 / np.sqrt(degree + 1e-10))
            L_sym = np.eye(n) - D_inv_sqrt @ similarity @ D_inv_sqrt
            
            # Power iteration for Fiedler vector
            v = np.random.randn(n)
            v = v / np.linalg.norm(v)
            for _ in range(30):
                v = L_sym @ v
                v = v / np.linalg.norm(v)
            
            # Assign partition labels based on sign
            partition_labels = np.sign(v)
            
            # Bonus for edges crossing partitions
            for i in range(n):
                for j in range(i + 1, n):
                    if adjacency_mask[i, j] and partition_labels[i] * partition_labels[j] < 0:
                        partition_bonus[i, j] = partition_bonus[j, i] = 1.0
        
        except:
            # Fallback if spectral computation fails
            pass
    
    # ==================== 7. Dynamic Weight Adaptation ====================
    # Initial component weights
    if n < 20:
        w_inv = 0.25
        w_mst = 0.20
        w_two_hop = 0.25
        w_attention = 0.20
        w_partition = 0.10
    elif n < 100:
        w_inv = 0.20
        w_mst = 0.20
        w_two_hop = 0.25
        w_attention = 0.25
        w_partition = 0.10
    else:
        w_inv = 0.15
        w_mst = 0.15
        w_two_hop = 0.25
        w_attention = 0.30
        w_partition = 0.15
    
    # Sample random tours to learn better weights
    if sample_size > 0 and n > 10:
        edge_counts = defaultdict(int)
        total_samples = 0
        
        for _ in range(sample_size):
            try:
                # Generate random tour using nearest neighbor with randomization
                tour = [np.random.randint(n)]
                visited = set(tour)
                
                while len(visited) < n:
                    current = tour[-1]
                    candidates = [j for j in range(n) if j not in visited and adjacency_mask[current, j]]
                    if not candidates:
                        candidates = [j for j in range(n) if j not in visited]
                    
                    if candidates:
                        # Weighted selection based on inverse distance
                        dists = distance_matrix[current, candidates]
                        probs = 1.0 / (dists + 1e-10)
                        probs = probs / np.sum(probs)
                        next_node = np.random.choice(candidates, p=probs)
                        
                        # Count edge usage
                        edge_counts[(min(current, next_node), max(current, next_node))] += 1
                        total_samples += 1
                        
                        tour.append(next_node)
                        visited.add(next_node)
                
                # Close the tour
                edge_counts[(min(tour[-1], tour[0]), max(tour[-1], tour[0]))] += 1
                total_samples += 1
                
            except:
                continue
        
        # Update weights based on edge usage in good tours
        if total_samples > 0:
            for (i, j), count in edge_counts.items():
                edge_frequency = count / total_samples
                
                # Components that should correlate with edge frequency
                inv_val = inv_dist[i, j]
                mst_val = 1.0 if mst_mask[i, j] else 0.0
                two_hop_val = two_hop_score[i, j]
                attention_val = attention_score[i, j]
                
                # Gradient updates (simplified)
                if edge_frequency > 0.1:  # Frequently used edges
                    if inv_val > 0.5:
                        w_inv += learning_rate * edge_frequency
                    if mst_val > 0.5:
                        w_mst += learning_rate * edge_frequency
                    if two_hop_val > 0.5:
                        w_two_hop += learning_rate * edge_frequency
                    if attention_val > 0.5:
                        w_attention += learning_rate * edge_frequency
        
        # Normalize weights to sum to 1
        total_w = w_inv + w_mst + w_two_hop + w_attention + w_partition
        w_inv, w_mst, w_two_hop, w_attention, w_partition = [
            w/total_w for w in [w_inv, w_mst, w_two_hop, w_attention, w_partition]
        ]
    
    # ==================== 8. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                score = 0.0
                
                # Add weighted components
                score += w_inv * inv_dist[i, j]
                
                if mst_mask[i, j]:
                    score += w_mst * 1.0
                
                score += w_two_hop * two_hop_score[i, j]
                score += w_attention * attention_score[i, j]
                score += w_partition * partition_bonus[i, j]
                
                # Christofides-inspired bonus for odd-degree MST nodes
                if i in odd_degree_nodes and j in odd_degree_nodes:
                    other_odd = [node for node in odd_degree_nodes if node not in (i, j)]
                    if other_odd:
                        avg_dist = np.mean([distance_matrix[i, k] for k in other_odd])
                        if avg_dist > 0:
                            matching_quality = avg_dist / (distance_matrix[i, j] + 1e-10)
                            score *= (1.0 + 0.3 * min(matching_quality, 2.0))
                
                heuristics[i, j] = heuristics[j, i] = score
    
    # ==================== 9. Final Processing ====================
    np.fill_diagonal(heuristics, 0)
    
    # Apply softmax normalization per node
    for i in range(n):
        row = heuristics[i]
        exp_row = np.exp(row - np.max(row))
        heuristics[i] = exp_row / np.sum(exp_row)
    
    # Symmetrize
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # Ensure all valid edges have non-zero probability
    heuristics[adjacency_mask] = np.maximum(heuristics[adjacency_mask], 1e-8)
    
    # Final normalization
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    return heuristics
