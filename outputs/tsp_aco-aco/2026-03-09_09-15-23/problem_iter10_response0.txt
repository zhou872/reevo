import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with global graph properties,
    adaptive multi-hop analysis, and dynamic parameter tuning.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Global Graph Analysis & Adaptive Parameters ====================
    # Compute global statistics for parameter adaptation
    flat_dists = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(flat_dists)
    dist_std = np.std(flat_dists) + 1e-10
    dist_cv = dist_std / dist_mean  # Coefficient of variation
    
    # Adaptive parameters based on problem characteristics
    base_k = max(3, int(np.log2(n)))
    adaptive_k = base_k + int(dist_cv * 3)  # More neighbors for variable distances
    lookahead_depth = min(3, max(2, n // 25))  # Adaptive multi-hop depth
    temperature = 0.5 + dist_cv * 0.3  # Adaptive smoothing
    
    # ==================== 2. Multi-scale Centrality Computation ====================
    # Compute closeness centrality (inverse of average distance)
    inv_dists = 1.0 / (distance_matrix + np.eye(n))
    closeness = n / (np.sum(distance_matrix, axis=1) + 1e-10)
    closeness = closeness / np.max(closeness)
    
    # Compute betweenness approximation via MST connectivity
    mst = minimum_spanning_tree(distance_matrix)
    mst_adj = mst.toarray()
    mst_betweenness = np.zeros(n)
    
    # Simple betweenness approximation: count shortest path neighbors in MST
    for i in range(n):
        mst_betweenness[i] = np.sum(mst_adj[i] > 0) + np.sum(mst_adj[:, i] > 0)
    mst_betweenness = mst_betweenness / np.max(mst_betweenness)
    
    # Combined centrality score
    centrality = 0.6 * closeness + 0.4 * mst_betweenness
    
    # ==================== 3. Adaptive Symmetric Sparsification ====================
    mask = np.zeros((n, n), dtype=bool)
    
    # Get adaptive nearest neighbors
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argpartition(distances, adaptive_k)[:adaptive_k]
        mask[i, nn_indices] = True
    
    # Ensure symmetry with centrality-weighted reinforcement
    for i in range(n):
        for j in range(i+1, n):
            if mask[i, j] or mask[j, i]:
                # Reinforce connection if nodes have complementary centrality
                centrality_score = centrality[i] * centrality[j]
                if centrality_score > 0.25:  # Threshold for reinforcement
                    mask[i, j] = mask[j, i] = True
    
    np.fill_diagonal(mask, False)
    
    # ==================== 4. Structural Enhancement via MST ====================
    mst_bonus = np.ones_like(distance_matrix)
    
    # Boost edges that are part of MST or close substitutes
    mst_edges = np.where(mst_adj > 0)
    mst_bonus[mst_edges] = 1.3  # Direct MST edges
    
    # Also boost edges that connect MST nodes with high centrality
    for i, j in zip(*mst_edges):
        # Find substitutes for (i,j) in MST
        for k in range(n):
            if k != i and k != j and mask[i, k] and mask[j, k]:
                # If k connects both i and j well, boost edge (i,j)
                alternative_path = distance_matrix[i, k] + distance_matrix[k, j]
                if alternative_path < 1.5 * distance_matrix[i, j]:
                    mst_bonus[i, j] = mst_bonus[j, i] = max(mst_bonus[i, j], 1.2)
    
    # ==================== 5. Multi-hop Lookahead Analysis ====================
    penalty = np.ones_like(distance_matrix)
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    # Determine candidate pool size adaptively
    candidate_pool = max(5, int(np.sqrt(n) * (1 + dist_cv)))
    
    for i in range(n):
        # Get i's top candidates for multi-hop analysis
        i_top = sorted_indices[i, 1:candidate_pool+1]
        
        for j in range(i+1, n):
            if not mask[i, j]:
                continue
                
            # Get j's top candidates
            j_top = sorted_indices[j, 1:candidate_pool+1]
            
            # Combine and filter candidates
            candidates = np.unique(np.concatenate([i_top, j_top]))
            candidates = candidates[(candidates != i) & (candidates != j)]
            
            if len(candidates) == 0:
                continue
            
            # Evaluate 2-hop alternatives
            two_hop_dists = distance_matrix[i, candidates] + distance_matrix[candidates, j]
            min_two_hop = np.min(two_hop_dists)
            direct_dist = distance_matrix[i, j]
            
            if direct_dist > 1e-10:
                two_hop_ratio = min_two_hop / direct_dist
                
                # Adaptive penalty based on lookahead depth
                if lookahead_depth >= 3 and n <= 100:
                    # Evaluate 3-hop alternatives for smaller instances
                    three_hop_dists = []
                    for k in candidates:
                        # Get k's top candidates
                        k_top = sorted_indices[k, 1:min(5, candidate_pool)]
                        for l in k_top:
                            if l not in (i, j, k):
                                three_hop = (distance_matrix[i, k] + 
                                            distance_matrix[k, l] + 
                                            distance_matrix[l, j])
                                three_hop_dists.append(three_hop)
                    
                    if three_hop_dists:
                        min_three_hop = np.min(three_hop_dists)
                        three_hop_ratio = min_three_hop / direct_dist
                        
                        # Combined penalty considering both 2-hop and 3-hop
                        if three_hop_ratio < 1.8:
                            penalty_val = 0.2 + 0.6 * (two_hop_ratio / 2.0) + 0.2 * (three_hop_ratio / 2.0)
                        else:
                            penalty_val = 0.3 + 0.7 * (two_hop_ratio / 1.5)
                    else:
                        penalty_val = 0.3 + 0.7 * (two_hop_ratio / 1.5)
                else:
                    # Only 2-hop analysis for larger instances
                    penalty_val = 0.3 + 0.7 * (two_hop_ratio / 1.5)
                
                penalty[i, j] = penalty_val
                penalty[j, i] = penalty_val
    
    # ==================== 6. Enhanced Inverse Distance with Centrality Weighting ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = mask[i]
        if not np.any(valid_mask):
            continue
            
        valid_dists = distance_matrix[i, valid_mask]
        ranks = np.argsort(np.argsort(valid_dists)) + 1
        
        # Adaptive exponential decay based on distance distribution
        adaptive_decay = 0.3 + 0.4 * (1 - dist_cv)
        weights = np.exp(-adaptive_decay * (ranks - 1))
        weights = weights / np.sum(weights)
        
        # Base inverse distance
        inv_vals = 1.0 / (valid_dists + 1e-10)
        
        # Centrality-aware augmentation
        centrality_bonus = 1.0 + temperature * (centrality[i] * 0.5 + 
                                               np.mean(centrality[valid_mask]) * 0.5)
        
        inv_dist[i, valid_mask] = inv_vals * weights * centrality_bonus
    
    # Symmetrize with geometric mean for smoother blending
    inv_dist_sym = np.sqrt(np.maximum(inv_dist, 1e-10) * np.maximum(inv_dist.T, 1e-10))
    np.fill_diagonal(inv_dist_sym, 0)
    
    # ==================== 7. Dynamic Combination & Normalization ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Combine all components with adaptive weights
    base_scores = inv_dist_sym * mst_bonus
    base_scores[~mask] = 0
    heuristics = base_scores * penalty
    
    # Add small centrality-based smoothing
    for i in range(n):
        for j in range(i+1, n):
            if mask[i, j]:
                smooth_factor = 1.0 + 0.1 * (centrality[i] + centrality[j])
                heuristics[i, j] *= smooth_factor
                heuristics[j, i] *= smooth_factor
    
    # Row-normalize with temperature-based smoothing
    row_sums = heuristics.sum(axis=1, keepdims=True)
    valid_rows = row_sums.flatten() > 0
    
    if np.any(valid_rows):
        # Apply temperature scaling before normalization
        heuristics[valid_rows] = np.power(heuristics[valid_rows], 1/temperature)
        row_sums = heuristics.sum(axis=1, keepdims=True)
        heuristics[valid_rows] = heuristics[valid_rows] / row_sums[valid_rows]
    
    # Final symmetrization with harmonic mean for stability
    heuristics = 2 * heuristics * heuristics.T / (heuristics + heuristics.T + 1e-10)
    np.fill_diagonal(heuristics, 0)
    
    # Ensure connectivity of isolated nodes
    min_nonzero = np.min(heuristics[heuristics > 0]) if np.any(heuristics > 0) else 1e-8
    heuristics[mask & (heuristics == 0)] = min_nonzero * 0.1
    
    return heuristics
