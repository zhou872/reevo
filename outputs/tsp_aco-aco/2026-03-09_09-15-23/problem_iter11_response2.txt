import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic function for TSP edge selection with context-aware
    two-hop penalty and density-adaptive scaling for robust performance.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Enhanced Logarithmic Symmetric Sparsification ====================
    # Use logarithmic scaling with adaptive k
    k = max(3, int(np.log2(n)))
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Precompute nearest neighbors for all nodes efficiently
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        # Get k+1 to exclude self (first element is i itself)
        nn_indices = sorted_indices[i, 1:k+1]
        adjacency_mask[i, nn_indices] = True
    
    # Ensure bidirectionality by taking union
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Context-Sensitive Distance Scoring ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    # Precompute instance density metric
    instance_density = np.mean([np.mean(np.sort(row)[1:k+1]) for row in distance_matrix])
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            # Get distances to neighbors
            valid_dists = distance_matrix[i, valid_mask]
            
            # Enhanced rank-based weighting with exponential decay
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            # Exponential weighting: closest gets highest weight
            weights = np.exp(-0.5 * (ranks - 1)) / np.sum(np.exp(-0.5 * (ranks - 1)))
            
            # Density-aware scaling of inverse distance
            density_factor = 1.0 / (1.0 + instance_density)
            inv_vals = 1.0 / (valid_dists + 1e-10)
            
            # Apply adaptive scaling
            scaled_inv = inv_vals * (1.0 + density_factor * weights)
            inv_dist[i, valid_mask] = scaled_inv / np.sum(valid_mask)
    
    # Symmetrize by harmonic mean (more robust than max)
    non_zero_mask = (inv_dist > 0) & (inv_dist.T > 0)
    inv_dist = np.where(non_zero_mask, 
                        2 * inv_dist * inv_dist.T / (inv_dist + inv_dist.T + 1e-10),
                        np.maximum(inv_dist, inv_dist.T))
    
    # ==================== 3. Optimized Context-Sensitive Two-Hop Analysis ====================
    two_hop_penalty = np.ones_like(distance_matrix)
    base_penalty = 0.3
    
    # Precompute common neighbors efficiently using adjacency mask
    for i in range(n):
        i_neighbors = np.where(adjacency_mask[i])[0]
        if len(i_neighbors) == 0:
            continue
            
        for j_idx in range(len(i_neighbors)):
            j = i_neighbors[j_idx]
            if i >= j:
                continue
                
            # Find common neighbors using vectorized operations
            common_mask = adjacency_mask[i] & adjacency_mask[j]
            common_mask[i] = False
            common_mask[j] = False
            common_nodes = np.where(common_mask)[0]
            
            if len(common_nodes) > 0:
                # Vectorized two-hop distance computation
                two_hop_dists = distance_matrix[i, common_nodes] + distance_matrix[common_nodes, j]
                min_two_hop = np.min(two_hop_dists)
                direct_dist = distance_matrix[i, j]
                
                if direct_dist > 1e-10:
                    # Context-sensitive penalty scaling
                    ratio = min_two_hop / direct_dist
                    
                    # Adaptive scale factor based on instance density and edge quality
                    edge_quality = inv_dist[i, j] * np.mean(inv_dist[i, i_neighbors])
                    scale_factor = 0.5 + 0.5 * np.tanh(instance_density / edge_quality)
                    
                    # Enhanced penalty function
                    if ratio < 1.1:  # Strong penalty for very competitive alternatives
                        penalty = base_penalty
                    elif ratio < 1.5:
                        penalty = base_penalty + (0.7 - base_penalty) * (ratio - 1.1) / 0.4
                    else:
                        penalty = 0.7 + 0.3 * np.tanh((ratio - 1.5) * scale_factor)
                    
                    two_hop_penalty[i, j] = two_hop_penalty[j, i] = penalty
    
    # ==================== 4. Adaptive Component Combination ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Only process edges in adjacency mask
    for i in range(n):
        valid_j = np.where(adjacency_mask[i] & (np.arange(n) > i))[0]
        for j in valid_j:
            # Base score with edge quality consideration
            edge_score = inv_dist[i, j]
            
            # Apply two-hop penalty with adaptive weighting
            penalty_weight = 0.7 + 0.3 * np.tanh(instance_density)
            combined_score = edge_score * (penalty_weight * two_hop_penalty[i, j] + 
                                          (1 - penalty_weight))
            
            heuristics[i, j] = heuristics[j, i] = combined_score
    
    # ==================== 5. Robust Final Processing ====================
    np.fill_diagonal(heuristics, 0)
    
    # Adaptive smoothing to avoid zeros
    min_nonzero = np.min(heuristics[heuristics > 0]) if np.any(heuristics > 0) else 1e-8
    heuristics[adjacency_mask] += 0.1 * min_nonzero
    
    # Row-normalize with stability
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Final symmetrization with geometric mean for stability
    geometric_mean = np.sqrt(heuristics * heuristics.T)
    heuristics = np.where((heuristics > 0) & (heuristics.T > 0), 
                          geometric_mean, 
                          0.5 * (heuristics + heuristics.T))
    
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
