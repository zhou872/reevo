import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristic function with adaptive geometric fusion,
    smooth regret-based scoring, density-aware spatial coherence,
    and optimized MST safeguard.
    
    Args:
        distance_matrix: Square matrix of pairwise distances between n nodes
        
    Returns:
        Heuristic matrix of same shape, with higher values indicating more promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. ENHANCED ADAPTIVE DENSITY ANALYSIS ====================
    # Multi-scale density estimation
    k1 = min(5, max(2, n // 15))
    k2 = min(15, max(5, n // 5))
    
    sorted_distances = np.sort(distance_matrix, axis=1)
    local_density_fine = 1.0 / (np.mean(sorted_distances[:, 1:k1+1], axis=1) + 1e-10)
    local_density_coarse = 1.0 / (np.mean(sorted_distances[:, 1:k2+1], axis=1) + 1e-10)
    
    # Fuse multi-scale densities using geometric mean
    local_density = np.sqrt(local_density_fine * local_density_coarse)
    
    # Adaptive normalization with robust statistics
    density_median = np.median(local_density)
    density_mad = np.median(np.abs(local_density - density_median))
    if density_mad > 1e-10:
        # Robust z-score normalization
        density_z = (local_density - density_median) / (density_mad + 1e-10)
        local_density = 1.0 + np.tanh(density_z * 0.8)  # More gradual scaling
    
    # Adaptive neighbor selection with non-linear scaling
    base_k = max(3, int(np.log2(n)) + 3)
    # Higher density → focus on closer neighbors, but more of them
    density_factor = 1.0 + 0.5 * (local_density - 1.0)  # [0.75, 1.5]
    adaptive_k = np.clip(base_k * (1.5 - 0.5 * density_factor), 
                        max(3, base_k // 2), 
                        min(4 * base_k, n - 1)).astype(int)
    
    # Efficient neighbor selection with argpartition
    adjacency_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        k_i = adaptive_k[i]
        # Use argpartition for efficient selection
        indices = np.argpartition(distance_matrix[i], k_i + 1)[:k_i + 1]
        neighbors = indices[indices != i]
        adjacency_mask[i, neighbors] = True
    
    # Symmetrize adjacency
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. DENSITY-ADAPTIVE RANK WEIGHTING ====================
    base_scores = np.zeros_like(distance_matrix, dtype=float)
    
    # Precompute sorted neighbor lists for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if not np.any(valid_mask):
            continue
            
        valid_dists = distance_matrix[i, valid_mask]
        neighbor_indices = np.where(valid_mask)[0]
        
        # Rank neighbors by distance
        ranks = np.argsort(np.argsort(valid_dists)) + 1
        
        # Non-linear rank weighting with density adaptation
        # In dense areas, prioritize top ranks more aggressively
        density_factor_i = np.clip(local_density[i], 0.7, 1.3)
        # Sigmoid-shaped rank weighting
        normalized_ranks = (ranks - 1) / (len(ranks) - 1) if len(ranks) > 1 else np.zeros_like(ranks)
        rank_weights = 1.0 / (1.0 + np.exp(5.0 * density_factor_i * (normalized_ranks - 0.3)))
        rank_weights = rank_weights / np.sum(rank_weights)
        
        # Adaptive inverse distance with curvature
        inv_vals = 1.0 / (valid_dists + 1e-10)
        # Curvature increases with density: higher density → stronger emphasis on short edges
        curvature = 0.8 + 0.4 * (density_factor_i - 0.7) / (1.3 - 0.7)  # [0.8, 1.2]
        inv_vals = np.power(inv_vals, curvature)
        
        base_scores[i, neighbor_indices] = inv_vals * rank_weights
    
    # Geometric mean symmetrization (preserves scale better)
    with np.errstate(divide='ignore'):
        log_base = np.log(base_scores + 1e-10)
        sym_log = 0.5 * (log_base + log_base.T)
        base_scores = np.exp(sym_log)
    
    # ==================== 3. SMOOTH REGRET-BASED SCORING ====================
    regret_scores = np.ones_like(distance_matrix, dtype=float)
    
    # Adaptive candidate selection based on problem size
    candidate_count = min(20, int(np.sqrt(n) * 1.5))
    
    # Precompute top candidates for each node
    top_candidates = sorted_indices[:, 1:candidate_count+1]
    
    # Vectorized regret computation for edges in adjacency mask
    edge_indices = np.where(np.triu(adjacency_mask, 1))
    
    if len(edge_indices[0]) > 0:
        i_list, j_list = edge_indices
        direct_dists = distance_matrix[i_list, j_list]
        
        # Prepare arrays for two-hop analysis
        best_alts = np.full(len(i_list), np.inf)
        
        # Check each candidate position
        for pos in range(candidate_count):
            # Get candidates for i and j
            k_i = top_candidates[i_list, pos]
            k_j = top_candidates[j_list, pos]
            
            # Compute two-hop distances
            alt1 = distance_matrix[i_list, k_i] + distance_matrix[k_i, j_list]
            alt2 = distance_matrix[i_list, k_j] + distance_matrix[k_j, j_list]
            
            # Update best alternative
            best_alts = np.minimum(best_alts, alt1)
            best_alts = np.minimum(best_alts, alt2)
        
        # Compute smooth regret scores
        valid_mask = direct_dists > 1e-10
        if np.any(valid_mask):
            regret = (best_alts[valid_mask] - direct_dists[valid_mask]) / direct_dists[valid_mask]
            
            # Adaptive sigmoid with scale based on problem characteristics
            # Larger n → more sensitive to small regret differences
            scale_factor = 2.0 + min(1.0, n / 50.0)  # [2.0, 3.0]
            
            # Smooth scoring function
            # Positive regret (direct is better) → score > 1
            # Negative regret (two-hop is better) → score < 1
            scores = 1.0 + 0.4 * np.tanh(regret * scale_factor)
            
            # Apply scores symmetrically
            valid_i = i_list[valid_mask]
            valid_j = j_list[valid_mask]
            regret_scores[valid_i, valid_j] = scores
            regret_scores[valid_j, valid_i] = scores
    
    # ==================== 4. DENSITY-AWARE SPATIAL COHERENCE ====================
    coherence_bonus = np.ones_like(distance_matrix, dtype=float)
    
    # Only compute for edges in adjacency mask
    edge_mask = np.triu(adjacency_mask, 1)
    i_coords, j_coords = np.where(edge_mask)
    
    if len(i_coords) > 0:
        # Compute density similarity with distance weighting
        density_ratio = local_density[i_coords] / (local_density[j_coords] + 1e-10)
        density_sim = np.minimum(density_ratio, 1.0 / density_ratio)  # [0, 1]
        
        # Include geometric information: edges connecting similar density regions
        # that are also geometrically aligned get higher bonus
        edge_dists = distance_matrix[i_coords, j_coords]
        
        # Compute average distance to common neighbors as geometric alignment measure
        # (simplified: use distance ratio to median)
        median_dist = np.median(distance_matrix[np.triu_indices(n, 1)])
        dist_factor = np.exp(-np.abs(edge_dists - median_dist) / (median_dist + 1e-10))
        
        # Combined bonus: density similarity weighted by geometric alignment
        bonus = 0.9 + 0.2 * density_sim * dist_factor
        
        coherence_bonus[i_coords, j_coords] = bonus
        coherence_bonus[j_coords, i_coords] = bonus
    
    # ==================== 5. ADAPTIVE GEOMETRIC FUSION ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Adaptive weights based on problem characteristics
    # Larger problems benefit more from regret analysis
    size_factor = min(1.0, n / 100.0)
    
    w_base = 0.5 - 0.1 * size_factor      # Base importance decreases slightly with size
    w_regret = 0.3 + 0.2 * size_factor    # Regret importance increases with size
    w_coherence = 0.2 - 0.1 * size_factor  # Coherence decreases slightly
    
    # Normalize weights
    total_w = w_base + w_regret + w_coherence
    w_base, w_regret, w_coherence = w_base/total_w, w_regret/total_w, w_coherence/total_w
    
    # Compute weighted geometric mean for edges in adjacency mask
    edge_mask_flat = adjacency_mask.copy()
    np.fill_diagonal(edge_mask_flat, False)
    
    # Prepare components with safety margins
    base_comp = np.maximum(base_scores, 1e-12)
    regret_comp = np.maximum(regret_scores, 1e-12)
    coherence_comp = np.maximum(coherence_bonus, 1e-12)
    
    # Weighted geometric mean in log space
    log_score = (w_base * np.log(base_comp) +
                 w_regret * np.log(regret_comp) +
                 w_coherence * np.log(coherence_comp))
    
    heuristics = np.exp(log_score)
    heuristics[~edge_mask_flat] = 0  # Zero out non-adjacent edges
    np.fill_diagonal(heuristics, 0)
    
    # ==================== 6. OPTIMIZED MST SAFEGUARD ====================
    if n > 3 and np.sum(edge_mask_flat) > 0:
        # Efficient Prim's algorithm implementation
        mst_mask = np.zeros((n, n), dtype=bool)
        in_mst = np.zeros(n, dtype=bool)
        min_edge = np.full(n, np.inf)
        parent = np.full(n, -1, dtype=int)
        
        # Start from node with median density
        start_node = np.argsort(local_density)[n // 2]
        min_edge[start_node] = 0
        
        for _ in range(n):
            # Find minimum edge to unvisited node
            u = -1
            min_val = np.inf
            for v in range(n):
                if not in_mst[v] and min_edge[v] < min_val:
                    min_val = min_edge[v]
                    u = v
            
            if u == -1:
                break
            
            in_mst[u] = True
            if parent[u] != -1:
                mst_mask[u, parent[u]] = mst_mask[parent[u], u] = True
            
            # Update distances to unvisited nodes
            for v in range(n):
                if not in_mst[v] and distance_matrix[u, v] < min_edge[v]:
                    min_edge[v] = distance_matrix[u, v]
                    parent[v] = u
        
        # Boost MST edges that are significantly underrated
        mst_edges = heuristics[mst_mask]
        if len(mst_edges) > 0:
            mst_median = np.median(mst_edges)
            boost_threshold = mst_median * 0.4
            
            # Find underrated MST edges
            boost_i, boost_j = np.where(mst_mask & (heuristics < boost_threshold))
            for i, j in zip(boost_i, boost_j):
                # Boost to median level
                heuristics[i, j] = heuristics[j, i] = mst_median
    
    # ==================== 7. ADAPTIVE NORMALIZATION ====================
    np.fill_diagonal(heuristics, 0)
    
    # Row-wise normalization with adaptive temperature
    for i in range(n):
        row = heuristics[i]
        if np.sum(row) > 0:
            # Adaptive temperature based on row statistics
            non_zero = row[row > 0]
            if len(non_zero) > 1:
                # Higher variance → lower temperature (more peaky distribution)
                row_var = np.var(non_zero)
                row_mean = np.mean(non_zero)
                if row_mean > 1e-10:
                    cv = np.sqrt(row_var) / row_mean  # Coefficient of variation
                    temperature = max(0.3, min(2.0, 1.0 / (cv + 0.5)))
                else:
                    temperature = 1.0
            else:
                temperature = 1.0
            
            # Softmax with temperature
            exp_row = np.exp(row / (temperature + 1e-10))
            heuristics[i] = exp_row / np.sum(exp_row)
    
    # Final symmetrization with harmonic mean (preserves sparsity)
    non_zero_mask = (heuristics > 0) | (heuristics.T > 0)
    with np.errstate(divide='ignore', invalid='ignore'):
        harmonic = np.zeros_like(heuristics)
        mask = non_zero_mask & ((heuristics > 0) & (heuristics.T > 0))
        harmonic[mask] = 2.0 / (1.0/(heuristics[mask] + 1e-10) + 1.0/(heuristics.T[mask] + 1e-10))
        harmonic[~mask] = np.maximum(heuristics, heuristics.T)[~mask]
    
    heuristics = harmonic
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
