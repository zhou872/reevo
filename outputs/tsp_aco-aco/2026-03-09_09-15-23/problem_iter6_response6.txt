import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with adaptive components.
    
    Features:
    1. Density-adaptive sparsification using instance metrics
    2. Multi-hop analysis (two and three-hop paths)
    3. Size-dependent component blending
    4. Rank-weighted inverse distance scoring
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Density-Adaptive Sparsification ====================
    # Calculate instance density metrics
    upper_tri = distance_matrix[np.triu_indices(n, 1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    density_factor = 1.0 / (1.0 + dist_std / (dist_mean + 1e-10))
    
    # Adaptive neighbor count based on size and density
    k_base = int(np.sqrt(n))
    k_sparse = max(3, min(k_base, int(k_base * (0.8 + 0.4 * density_factor))))
    
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Pre-sort distances for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        # Select adaptive number of nearest neighbors
        nearest = sorted_indices[i, 1:k_sparse+1]
        adjacency_mask[i, nearest] = True
        
        # Mutual neighbor validation for symmetry
        for j in nearest:
            j_nearest = sorted_indices[j, 1:k_sparse+1]
            if i in j_nearest[:k_sparse//2 + 1]:  # Check only close mutual neighbors
                adjacency_mask[j, i] = True
    
    # Ensure connectivity
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Rank-Weighted Inverse Distance ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            neighbor_count = np.sum(valid_mask)
            valid_dists = distance_matrix[i, valid_mask]
            
            # Exponential rank weighting (emphasizes closer edges)
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            weights = np.exp(-0.5 * (ranks - 1))
            weights = weights / np.sum(weights)
            
            # Inverse distance with adaptive scaling
            inv_vals = 1.0 / (valid_dists + 1e-10)
            scaling = 1.0 / np.log1p(neighbor_count)
            inv_dist[i, valid_mask] = inv_vals * weights * scaling
    
    # Symmetrize using geometric mean
    inv_dist = np.sqrt(inv_dist * inv_dist.T)
    inv_dist[np.isnan(inv_dist)] = 0
    
    # ==================== 3. Multi-Hop Alternative Analysis ====================
    # Size-dependent analysis depth
    hop_depth = 2 if n < 100 else 3
    
    multi_hop_penalty = np.ones_like(distance_matrix)
    neighbor_limit = min(10, max(4, int(np.cbrt(n) * 2)))
    
    # Precompute neighbor sets
    neighbor_sets = []
    for i in range(n):
        valid_mask = adjacency_mask[i]
        neighbors = np.where(valid_mask)[0]
        if len(neighbors) > neighbor_limit:
            # Take closest neighbors up to limit
            neighbor_dists = distance_matrix[i, neighbors]
            top_idx = np.argsort(neighbor_dists)[:neighbor_limit]
            neighbor_sets.append(neighbors[top_idx])
        else:
            neighbor_sets.append(neighbors)
    
    edge_indices = np.where(np.triu(adjacency_mask, 1))
    
    for idx in range(len(edge_indices[0])):
        i, j = edge_indices[0][idx], edge_indices[1][idx]
        direct_dist = distance_matrix[i, j]
        
        if direct_dist < 1e-10:
            continue
            
        # Collect candidate intermediate nodes
        i_neighbors = neighbor_sets[i]
        j_neighbors = neighbor_sets[j]
        
        # Remove endpoints
        i_neighbors = i_neighbors[(i_neighbors != i) & (i_neighbors != j)]
        j_neighbors = j_neighbors[(j_neighbors != i) & (j_neighbors != j)]
        
        if len(i_neighbors) == 0 or len(j_neighbors) == 0:
            continue
            
        # Two-hop analysis
        two_hop_dists = []
        for k in i_neighbors:
            if k in j_neighbors:
                # Direct two-hop: i -> k -> j
                two_hop_dists.append(distance_matrix[i, k] + distance_matrix[k, j])
            else:
                # Consider common neighbors
                for l in j_neighbors:
                    if l == k:
                        continue
                    # Check if k and l are connected
                    if adjacency_mask[k, l]:
                        two_hop_dists.append(
                            distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j]
                        )
        
        if two_hop_dists:
            min_two_hop = np.min(two_hop_dists)
            ratio_2hop = min_two_hop / direct_dist
        else:
            ratio_2hop = np.inf
        
        # Three-hop analysis (if enabled and worthwhile)
        ratio_3hop = np.inf
        if hop_depth >= 3 and ratio_2hop < 2.0:  # Only compute if two-hop is competitive
            three_hop_dists = []
            for k in i_neighbors:
                for l in j_neighbors:
                    if k == l:
                        continue
                    # Find intermediate node m connecting k and l
                    k_neighbors = neighbor_sets[k]
                    l_neighbors = neighbor_sets[l]
                    
                    common = np.intersect1d(k_neighbors, l_neighbors)
                    common = common[(common != i) & (common != j) & (common != k) & (common != l)]
                    
                    for m in common:
                        path_dist = (distance_matrix[i, k] + distance_matrix[k, m] +
                                   distance_matrix[m, l] + distance_matrix[l, j])
                        three_hop_dists.append(path_dist)
            
            if three_hop_dists:
                min_three_hop = np.min(three_hop_dists)
                ratio_3hop = min_three_hop / direct_dist
        
        # Combine multi-hop ratios
        valid_ratios = [r for r in [ratio_2hop, ratio_3hop] if r < np.inf]
        if valid_ratios:
            min_ratio = np.min(valid_ratios)
            # Size-dependent penalty function
            if n < 50:
                penalty = 0.2 + 0.8 * np.clip(min_ratio / 2.5, 0, 1)
            elif n < 200:
                penalty = 0.3 + 0.7 * np.clip(min_ratio / 3.0, 0, 1)
            else:
                penalty = 0.4 + 0.6 * np.clip(min_ratio / 4.0, 0, 1)
            
            multi_hop_penalty[i, j] = multi_hop_penalty[j, i] = penalty
    
    # ==================== 4. Size-Dependent Component Blending ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Blend weights based on problem size
    if n < 30:
        inv_weight, penalty_weight = 0.6, 0.4
    elif n < 100:
        inv_weight, penalty_weight = 0.5, 0.5
    else:
        inv_weight, penalty_weight = 0.4, 0.6
    
    # Normalize components for blending
    inv_norm = inv_dist.copy()
    inv_norm[inv_norm > 0] = (inv_norm[inv_norm > 0] - inv_norm[inv_norm > 0].min()) / \
                             (inv_norm[inv_norm > 0].max() - inv_norm[inv_norm > 0].min() + 1e-10)
    
    penalty_norm = multi_hop_penalty.copy()
    penalty_norm = (penalty_norm - penalty_norm.min()) / \
                   (penalty_norm.max() - penalty_norm.min() + 1e-10)
    
    # Blend components
    heuristics = inv_weight * inv_norm + penalty_weight * penalty_norm
    heuristics[~adjacency_mask] = 0
    np.fill_diagonal(heuristics, 0)
    
    # ==================== 5. Adaptive Normalization ====================
    # Row-normalize with size-dependent temperature
    temperature = max(0.3, 1.0 - n / 1000)  # Decrease temperature with size
    
    row_max = heuristics.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    normalized = heuristics / row_max
    
    # Softmax-like transformation
    exp_scores = np.exp(normalized / temperature)
    row_sums = exp_scores.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = exp_scores / row_sums
    
    # Final symmetrization
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
