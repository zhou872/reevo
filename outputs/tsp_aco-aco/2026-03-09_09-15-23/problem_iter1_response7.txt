import numpy as np
from scipy.spatial.distance import cdist

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors.
    Returns a matrix of edge promises with same shape as input.
    """
    n = distance_matrix.shape[0]
    
    # Safety check: handle zero distances (self-connections)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base inverse distance weighting (as in v1)
    inv_dist = 1.0 / safe_dist
    
    # 2. Nearest neighbor sparsity factor
    # For each node, identify top k nearest neighbors
    k = max(2, min(5, n // 4))  # Adaptive k based on problem size
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k+1 smallest distances (skip self)
        idx = np.argpartition(safe_dist[i], k+1)[:k+1]
        idx = idx[idx != i]  # Remove self
        nn_mask[i, idx] = True
    
    # Symmetrize the nearest neighbor mask
    nn_mask = np.logical_or(nn_mask, nn_mask.T)
    
    # 3. Cluster connectivity bonus
    # Identify natural clusters in the data
    cluster_bonus = np.zeros_like(distance_matrix)
    if n > 10:
        # Use farthest point sampling to find representative nodes
        n_clusters = max(2, n // 10)
        centroids = [np.random.randint(0, n)]
        
        for _ in range(1, n_clusters):
            # Find point farthest from all current centroids
            min_dists = np.min(safe_dist[np.ix_(centroids, range(n))], axis=0)
            new_centroid = np.argmax(min_dists)
            centroids.append(new_centroid)
        
        # Assign nodes to nearest centroid
        cluster_assignments = np.argmin(safe_dist[centroids, :], axis=0)
        
        # Boost intra-cluster edges and selected inter-cluster edges
        for i in range(n):
            for j in range(i+1, n):
                if cluster_assignments[i] == cluster_assignments[j]:
                    # Intra-cluster edges get bonus
                    cluster_bonus[i, j] = 0.3
                elif nn_mask[i, j]:
                    # Inter-cluster edges that are in nearest neighbors
                    cluster_bonus[i, j] = 0.1
        
        # Make symmetric
        cluster_bonus = cluster_bonus + cluster_bonus.T
    
    # 4. Regional hub enhancement
    # Identify nodes that serve as natural "hubs" (well-connected)
    hub_scores = np.zeros(n)
    for i in range(n):
        # Count how many times node i appears in others' nearest neighbors
        hub_scores[i] = np.sum(nn_mask[:, i])
    
    # Normalize hub scores
    if hub_scores.max() > 0:
        hub_scores = hub_scores / hub_scores.max()
    
    hub_bonus = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            # Edges involving hubs get bonus
            hub_factor = (hub_scores[i] + hub_scores[j]) / 2
            if hub_factor > 0.5:  # Both nodes are reasonably hub-like
                hub_bonus[i, j] = 0.2 * hub_factor
    
    # Make symmetric
    hub_bonus = hub_bonus + hub_bonus.T
    
    # 5. Combine all factors with weights
    combined = (
        0.6 * inv_dist +          # Base inverse distance
        0.8 * nn_mask.astype(float) +  # Nearest neighbor emphasis
        0.4 * cluster_bonus +     # Cluster connectivity
        0.3 * hub_bonus           # Hub enhancement
    )
    
    # 6. Sparsification: zero out unpromising edges
    # Use percentile-based thresholding
    if n > 4:
        # Get upper triangle values (excluding diagonal)
        triu_vals = combined[np.triu_indices(n, k=1)]
        if len(triu_vals) > 0:
            # Keep top 30% of edges (or at least n edges per node)
            threshold = np.percentile(triu_vals, 70)
            
            # Create sparse mask
            sparse_mask = combined > threshold
            
            # Ensure each node has at least 2 connections
            for i in range(n):
                connected = np.where(sparse_mask[i])[0]
                if len(connected) < 2:
                    # Connect to 2 nearest nodes not already connected
                    candidates = np.argsort(safe_dist[i])[1:6]  # Check 5 nearest
                    for cand in candidates:
                        if cand not in connected and len(connected) < 2:
                            sparse_mask[i, cand] = True
                            sparse_mask[cand, i] = True
                            connected = np.append(connected, cand)
            
            # Apply sparsification
            combined = combined * sparse_mask
    
    # Zero out diagonal
    np.fill_diagonal(combined, 0)
    
    return combined
