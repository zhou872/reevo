import numpy as np
from scipy.sparse.csgraph import connected_components
from scipy.sparse import csr_matrix
from typing import Tuple, List, Optional

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristic with Lin-Kernighan inspired analysis, GNN-style feature embedding,
    and meta-heuristic optimization of hyperparameters.
    
    Args:
        distance_matrix: Square matrix of pairwise distances between n nodes
        
    Returns:
        Heuristic matrix of same shape, with higher values indicating more promising edges
    """
    n = len(distance_matrix)
    
    # Handle trivial cases
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. ENHANCED CHRISTOFIDES COMPONENTS ====================
    christofides_edges = _compute_enhanced_christofides(distance_matrix)
    
    # ==================== 2. GNN-STYLE GRAPH FEATURE EMBEDDING ====================
    node_embeddings = _compute_gnn_features(distance_matrix)
    
    # ==================== 3. ADAPTIVE MULTI-SCALE NEIGHBORHOOD ====================
    adjacency_mask = _build_adaptive_adjacency(
        distance_matrix, 
        christofides_edges,
        node_embeddings
    )
    
    # ==================== 4. LK-INSPIRED PENALTY ANALYSIS ====================
    lk_penalties = _compute_lk_penalties(distance_matrix, adjacency_mask, n)
    
    # ==================== 5. META-OPTIMIZED MULTI-HOP REGRET ====================
    optimal_hops = _optimize_hop_limit(distance_matrix, n)
    regret_scores = _compute_meta_optimized_regret(
        distance_matrix, 
        adjacency_mask,
        optimal_hops
    )
    
    # ==================== 6. BAYESIAN COMPONENT FUSION ====================
    heuristics = _bayesian_fusion(
        regret_scores,
        lk_penalties,
        christofides_edges,
        node_embeddings,
        distance_matrix,
        n
    )
    
    return heuristics


def _compute_enhanced_christofides(distance_matrix: np.ndarray) -> np.ndarray:
    """Enhanced Christofides with 2-opt local search initialization."""
    n = len(distance_matrix)
    edge_scores = np.ones_like(distance_matrix, dtype=float)
    
    # 1. Compute MST with Prim's algorithm
    mst_mask = _compute_mst_mask(distance_matrix)
    
    # 2. Find odd-degree vertices
    degrees = np.sum(mst_mask, axis=1)
    odd_mask = (degrees % 2 == 1)
    odd_vertices = np.where(odd_mask)[0]
    
    # 3. Enhanced perfect matching with blossom algorithm approximation
    matching_mask = _enhanced_perfect_matching(distance_matrix, odd_vertices)
    
    # 4. Apply 2-opt improvement to Christofides edges
    improved_mask = _apply_2opt_improvement(distance_matrix, mst_mask, matching_mask)
    
    # 5. Assign scores with gradient
    for i in range(n):
        for j in range(i + 1, n):
            if improved_mask[i, j]:
                # Higher score for edges in improved Christofides
                edge_scores[i, j] = edge_scores[j, i] = 2.5
            elif mst_mask[i, j]:
                edge_scores[i, j] = edge_scores[j, i] = 2.0
            elif matching_mask[i, j]:
                edge_scores[i, j] = edge_scores[j, i] = 1.7
            else:
                edge_scores[i, j] = edge_scores[j, i] = 1.0
    
    return edge_scores


def _compute_mst_mask(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute Minimum Spanning Tree using Prim's algorithm."""
    n = len(distance_matrix)
    mst_mask = np.zeros((n, n), dtype=bool)
    
    if n == 0:
        return mst_mask
    
    visited = np.zeros(n, dtype=bool)
    min_edge_weights = np.full(n, np.inf)
    min_edge_sources = np.full(n, -1, dtype=int)
    
    min_edge_weights[0] = 0
    
    for _ in range(n):
        current = -1
        min_weight = np.inf
        for v in range(n):
            if not visited[v] and min_edge_weights[v] < min_weight:
                min_weight = min_edge_weights[v]
                current = v
        
        if current == -1:
            break
        
        visited[current] = True
        
        if min_edge_sources[current] != -1:
            source = min_edge_sources[current]
            mst_mask[current, source] = mst_mask[source, current] = True
        
        for neighbor in range(n):
            if (not visited[neighbor] and 
                distance_matrix[current, neighbor] < min_edge_weights[neighbor]):
                min_edge_weights[neighbor] = distance_matrix[current, neighbor]
                min_edge_sources[neighbor] = current
    
    return mst_mask


def _enhanced_perfect_matching(distance_matrix: np.ndarray, 
                               odd_vertices: np.ndarray) -> np.ndarray:
    """Improved perfect matching with blossom algorithm approximation."""
    n = len(distance_matrix)
    matching_mask = np.zeros((n, n), dtype=bool)
    
    if len(odd_vertices) % 2 != 0:
        return matching_mask
    
    # Sort vertices by isolation (average distance to other odd vertices)
    avg_dists = []
    for v in odd_vertices:
        other_odds = odd_vertices[odd_vertices != v]
        if len(other_odds) > 0:
            avg_dist = np.mean(distance_matrix[v, other_odds])
        else:
            avg_dist = 0
        avg_dists.append(avg_dist)
    
    sorted_indices = np.argsort(avg_dists)[::-1]
    paired = set()
    
    # Greedy matching starting from most isolated vertices
    for idx in range(0, len(odd_vertices), 2):
        if idx >= len(odd_vertices):
            break
            
        v1_idx = sorted_indices[idx]
        v1 = odd_vertices[v1_idx]
        
        if v1 in paired:
            continue
            
        # Find best match among remaining unpaired vertices
        min_dist = np.inf
        best_match = -1
        
        for jdx in range(idx + 1, len(odd_vertices)):
            v2_idx = sorted_indices[jdx]
            v2 = odd_vertices[v2_idx]
            
            if v2 not in paired:
                dist = distance_matrix[v1, v2]
                if dist < min_dist:
                    min_dist = dist
                    best_match = v2
        
        if best_match != -1:
            matching_mask[v1, best_match] = matching_mask[best_match, v1] = True
            paired.update([v1, best_match])
    
    return matching_mask


def _apply_2opt_improvement(distance_matrix: np.ndarray, 
                           mst_mask: np.ndarray, 
                           matching_mask: np.ndarray) -> np.ndarray:
    """Apply 2-opt local search to improve Christofides edges."""
    n = len(distance_matrix)
    combined_mask = mst_mask | matching_mask
    improved_mask = combined_mask.copy()
    
    # For each pair of edges in the combined graph, try 2-opt swap
    edges = np.where(np.triu(combined_mask, k=1))
    edge_list = list(zip(edges[0], edges[1]))
    
    for i in range(min(len(edge_list), 100)):  # Limit for performance
        for j in range(i + 1, min(len(edge_list), 100)):
            a, b = edge_list[i]
            c, d = edge_list[j]
            
            # Check if 2-opt swap would improve
            current_cost = distance_matrix[a, b] + distance_matrix[c, d]
            swapped_cost = distance_matrix[a, c] + distance_matrix[b, d]
            
            if swapped_cost < current_cost - 1e-6:
                # Perform the swap
                improved_mask[a, b] = improved_mask[b, a] = False
                improved_mask[c, d] = improved_mask[d, c] = False
                improved_mask[a, c] = improved_mask[c, a] = True
                improved_mask[b, d] = improved_mask[d, b] = True
    
    return improved_mask


def _compute_gnn_features(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute GNN-style node embeddings using multi-hop graph convolutions."""
    n = len(distance_matrix)
    
    # Initial node features: distance statistics
    features = np.zeros((n, 5))
    
    for i in range(n):
        distances = distance_matrix[i]
        sorted_dists = np.sort(distances)
        
        # Feature 1: inverse of mean distance (centrality)
        features[i, 0] = 1.0 / (np.mean(distances) + 1e-10)
        
        # Feature 2: distance variance (spread)
        features[i, 1] = np.var(distances)
        
        # Feature 3: ratio of nearest to average distance
        if n > 1:
            features[i, 2] = sorted_dists[1] / (np.mean(distances) + 1e-10)
        
        # Feature 4: distance skewness
        features[i, 3] = np.mean((distances - np.mean(distances))**3) / (np.std(distances)**3 + 1e-10)
        
        # Feature 5: local density (k-nearest average)
        k = min(5, n-1)
        features[i, 4] = np.mean(sorted_dists[1:k+1])
    
    # Normalize features
    for j in range(5):
        col = features[:, j]
        if np.std(col) > 1e-10:
            features[:, j] = (col - np.mean(col)) / np.std(col)
    
    # Simple 2-layer "convolution" - aggregate neighbor features
    for _ in range(2):
        new_features = features.copy()
        for i in range(n):
            # Get top 5 nearest neighbors
            k = min(5, n-1)
            neighbor_indices = np.argsort(distance_matrix[i])[1:k+1]
            
            if len(neighbor_indices) > 0:
                neighbor_features = features[neighbor_indices]
                # Weighted average based on inverse distance
                weights = 1.0 / (distance_matrix[i, neighbor_indices] + 1e-10)
                weights = weights / np.sum(weights)
                
                # Update features with neighbor information
                new_features[i] = 0.7 * features[i] + 0.3 * np.sum(
                    weights[:, np.newaxis] * neighbor_features, axis=0
                )
        
        features = new_features
    
    return features


def _build_adaptive_adjacency(distance_matrix: np.ndarray,
                             christofides_edges: np.ndarray,
                             node_embeddings: np.ndarray) -> np.ndarray:
    """Build adjacency matrix using node embeddings for adaptive scaling."""
    n = len(distance_matrix)
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Always include enhanced Christofides edges
    adjacency_mask[christofides_edges > 1.5] = True
    
    # Compute adaptive k for each node based on embedding similarity
    base_k = max(3, int(np.log2(n)))
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        # Compute embedding similarity with other nodes
        similarities = np.zeros(n)
        for j in range(n):
            if i != j:
                # Cosine similarity between embeddings
                emb_i = node_embeddings[i]
                emb_j = node_embeddings[j]
                similarity = np.dot(emb_i, emb_j) / (
                    np.linalg.norm(emb_i) * np.linalg.norm(emb_j) + 1e-10
                )
                similarities[j] = similarity
        
        # Combine distance and similarity for neighbor selection
        combined_scores = np.zeros(n)
        for j in range(n):
            if i != j:
                # Normalize distance
                norm_dist = distance_matrix[i, j] / (np.mean(distance_matrix[i]) + 1e-10)
                combined_scores[j] = similarities[j] / (norm_dist + 1e-10)
        
        # Select top neighbors based on combined score
        k_adaptive = min(n - 1, base_k + int(5 * np.std(similarities)))
        top_neighbors = np.argsort(combined_scores)[-k_adaptive:]
        
        for j in top_neighbors:
            if j != i:
                adjacency_mask[i, j] = True
    
    # Ensure symmetry and connectivity
    adjacency_mask = adjacency_mask | adjacency_mask.T
    
    # Check connectivity
    sparse_graph = csr_matrix(adjacency_mask.astype(int))
    n_components, labels = connected_components(sparse_graph, directed=False, return_labels=True)
    
    if n_components > 1:
        # Connect components with nearest neighbors
        for comp in range(n_components):
            nodes_in_comp = np.where(labels == comp)[0]
            if len(nodes_in_comp) == 0:
                continue
            
            # Connect each component to its nearest neighbor in other components
            for node in nodes_in_comp[:min(3, len(nodes_in_comp))]:
                distances = distance_matrix[node].copy()
                distances[nodes_in_comp] = np.inf
                
                if np.min(distances) < np.inf:
                    closest = np.argmin(distances)
                    adjacency_mask[node, closest] = adjacency_mask[closest, node] = True
    
    np.fill_diagonal(adjacency_mask, False)
    return adjacency_mask


def _compute_lk_penalties(distance_matrix: np.ndarray,
                         adjacency_mask: np.ndarray,
                         n: int) -> np.ndarray:
    """Compute Lin-Kernighan inspired penalties by analyzing k-opt move potential."""
    lk_penalties = np.ones_like(distance_matrix)
    
    # Precompute sorted neighbors for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    neighbor_limit = min(10, n - 1)
    
    # Analyze each promising edge for potential LK improvements
    edge_mask = np.triu(adjacency_mask, k=1)
    edge_indices = np.where(edge_mask)
    
    if len(edge_indices[0]) == 0:
        return lk_penalties
    
    i_vals, j_vals = edge_indices[0], edge_indices[1]
    
    for idx in range(len(i_vals)):
        i, j = i_vals[idx], j_vals[idx]
        
        # Skip if not a promising connection
        if not adjacency_mask[i, j]:
            continue
        
        direct_dist = distance_matrix[i, j]
        if direct_dist < 1e-10:
            continue
        
        # Find candidate LK moves (2-opt and 3-opt approximations)
        best_improvement = 0
        
        # 2-opt candidate analysis
        i_neighbors = sorted_indices[i, 1:neighbor_limit+1]
        j_neighbors = sorted_indices[j, 1:neighbor_limit+1]
        
        # Check 2-opt swaps
        for k in i_neighbors:
            if k == j or not adjacency_mask[i, k]:
                continue
            
            for l in j_neighbors:
                if l == i or l == k or not adjacency_mask[j, l]:
                    continue
                
                # Check if (k,l) exists
                if adjacency_mask[k, l]:
                    current = direct_dist + distance_matrix[k, l]
                    swapped = distance_matrix[i, k] + distance_matrix[j, l]
                    
                    if swapped < current:
                        improvement = (current - swapped) / current
                        best_improvement = max(best_improvement, improvement)
        
        # 3-opt approximation (limited search)
        if n > 10:
            for k in i_neighbors[:5]:
                if k == j or not adjacency_mask[i, k]:
                    continue
                
                k_neighbors = sorted_indices[k, 1:min(5, neighbor_limit)+1]
                
                for l in k_neighbors:
                    if l == i or l == j or not adjacency_mask[k, l]:
                        continue
                    
                    for m in j_neighbors[:5]:
                        if m == i or m == k or m == l or not adjacency_mask[j, m]:
                            continue
                        
                        # Check if (l,m) exists
                        if adjacency_mask[l, m]:
                            # 3-opt gain estimation
                            gain = (direct_dist + distance_matrix[k, l] + distance_matrix[m, j] -
                                   (distance_matrix[i, k] + distance_matrix[l, m] + distance_matrix[j, m]))
                            
                            if gain > 0:
                                improvement = gain / (direct_dist + 1e-10)
                                best_improvement = max(best_improvement, improvement)
        
        # Apply penalty based on LK improvement potential
        if best_improvement > 0.1:  # Significant improvement possible
            # Stronger penalty for edges with high improvement potential
            penalty = 1.0 - 0.7 * np.tanh(best_improvement * 3)
            lk_penalties[i, j] = lk_penalties[j, i] = penalty
        elif best_improvement > 0.01:  # Minor improvement
            penalty = 1.0 - 0.3 * np.tanh(best_improvement * 5)
            lk_penalties[i, j] = lk_penalties[j, i] = penalty
    
    return lk_penalties


def _optimize_hop_limit(distance_matrix: np.ndarray, n: int) -> int:
    """Meta-optimize hop limit based on problem characteristics."""
    if n <= 20:
        return 4  # Small instances: deeper search
    elif n <= 50:
        return 3  # Medium instances: moderate search
    elif n <= 100:
        return 2  # Larger instances: shallower search
    else:
        return 2  # Very large: focus on 2-hop
    
    # Alternative: adaptive based on distance variance
    avg_dist = np.mean(distance_matrix)
    std_dist = np.std(distance_matrix)
    uniformity = std_dist / (avg_dist + 1e-10)
    
    if uniformity < 0.2:
        return 2  # Uniform distances: simpler analysis
    elif uniformity < 0.5:
        return 3  # Moderate variation: deeper analysis
    else:
        return 4  # High variation: thorough analysis


def _compute_meta_optimized_regret(distance_matrix: np.ndarray,
                                  adjacency_mask: np.ndarray,
                                  max_hops: int) -> np.ndarray:
    """Compute regret scores with meta-optimized hop limit."""
    n = len(distance_matrix)
    regret_scores = np.ones_like(distance_matrix, dtype=float)
    
    # Precompute sorted neighbors for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    neighbor_limit = min(15, n - 1)
    
    for i in range(n):
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                regret_scores[i, j] = regret_scores[j, i] = 0.0
                continue
            
            direct_dist = distance_matrix[i, j]
            if direct_dist < 1e-10:
                regret_scores[i, j] = regret_scores[j, i] = 1.0
                continue
            
            # Get candidate intermediate nodes
            i_neighbors = sorted_indices[i, 1:neighbor_limit+1]
            j_neighbors = sorted_indices[j, 1:neighbor_limit+1]
            
            best_alternative = np.inf
            
            # 2-hop alternatives
            for k in i_neighbors:
                if k == j:
                    continue
                two_hop = distance_matrix[i, k] + distance_matrix[k, j]
                best_alternative = min(best_alternative, two_hop)
            
            # 3-hop alternatives
            if max_hops >= 3:
                for k in i_neighbors[:10]:
                    if k == j:
                        continue
                    
                    k_neighbors = sorted_indices[k, 1:min(8, neighbor_limit)+1]
                    for l in k_neighbors:
                        if l == i or l == j:
                            continue
                        
                        three_hop = (distance_matrix[i, k] + 
                                    distance_matrix[k, l] + 
                                    distance_matrix[l, j])
                        best_alternative = min(best_alternative, three_hop)
            
            # 4-hop alternatives (for max_hops >= 4)
            if max_hops >= 4 and n <= 50:  # Only for smaller instances
                for k in i_neighbors[:6]:
                    if k == j:
                        continue
                    
                    k_neighbors = sorted_indices[k, 1:min(6, neighbor_limit)+1]
                    for l in k_neighbors:
                        if l == i or l == j:
                            continue
                        
                        l_neighbors = sorted_indices[l, 1:min(4, neighbor_limit)+1]
                        for m in l_neighbors:
                            if m == i or m == j or m == k:
                                continue
                            
                            four_hop = (distance_matrix[i, k] + 
                                       distance_matrix[k, l] + 
                                       distance_matrix[l, m] + 
                                       distance_matrix[m, j])
                            best_alternative = min(best_alternative, four_hop)
            
            # Compute regret-based score
            if best_alternative < np.inf:
                regret_ratio = best_alternative / direct_dist
                
                # Adaptive scoring based on ratio
                if regret_ratio > 1.3:
                    score = 1.0  # Direct edge is much better
                elif regret_ratio > 1.1:
                    score = 0.8 + 0.2 * np.tanh(regret_ratio - 1.1)
                elif regret_ratio > 1.0:
                    score = 0.6 + 0.4 * np.tanh(regret_ratio - 1.0)
                else:
                    # Alternative is better
                    alt_advantage = 1.0 - regret_ratio
                    score = 0.3 + 0.3 * np.tanh(alt_advantage * 5)
                
                regret_scores[i, j] = regret_scores[j, i] = np.clip(score, 0.1, 1.0)
    
    return regret_scores


def _bayesian_fusion(regret_scores: np.ndarray,
                    lk_penalties: np.ndarray,
                    christofides_edges: np.ndarray,
                    node_embeddings: np.ndarray,
                    distance_matrix: np.ndarray,
                    n: int) -> np.ndarray:
    """Bayesian fusion of multiple heuristic components."""
    heuristics = np.zeros_like(regret_scores, dtype=float)
    
    # Compute problem characteristics for adaptive weighting
    avg_distance = np.mean(distance_matrix)
    std_distance = np.std(distance_matrix)
    uniformity = std_distance / (avg_distance + 1e-10)
    
    # Compute embedding coherence
    embedding_coherence = np.mean([
        np.corrcoef(node_embeddings[i], node_embeddings[j])[0, 1]
        for i in range(min(n, 20))
        for j in range(i + 1, min(n, 20))
    ])
    
    # Bayesian prior weights based on problem characteristics
    if uniformity < 0.25:
        # Uniform distances: trust Christofides and embeddings
        w_regret = 0.3
        w_lk = 0.2
        w_christofides = 0.3
        w_embedding = 0.2
    elif uniformity < 0.6:
        # Moderate variation: balanced approach
        w_regret = 0.4
        w_lk = 0.25
        w_christofides = 0.2
        w_embedding = 0.15
    else:
        # High variation: focus on regret and LK
        w_regret = 0.5
        w_lk = 0.35
        w_christofides = 0.1
        w_embedding = 0.05
    
    # Adjust based on embedding coherence
    if embedding_coherence > 0.5:
        w_embedding *= 1.5
        w_regret *= 0.9
    elif embedding_coherence < 0.2:
        w_embedding *= 0.5
        w_regret *= 1.2
    
    # Normalize weights
    total = w_regret + w_lk + w_christofides + w_embedding
    w_regret /= total
    w_lk /= total
    w_christofides /= total
    w_embedding /= total
    
    # Compute embedding similarity matrix
    embedding_similarity = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            sim = np.dot(node_embeddings[i], node_embeddings[j]) / (
                np.linalg.norm(node_embeddings[i]) * np.linalg.norm(node_embeddings[j]) + 1e-10
            )
            embedding_similarity[i, j] = embedding_similarity[j, i] = (sim + 1) / 2  # Scale to [0,1]
    
    # Combine all components
    for i in range(n):
        for j in range(i + 1, n):
            # Get normalized Christofides score (0-1 range)
            christofides_norm = min(1.0, (christofides_edges[i, j] - 1.0) / 1.5)
            
            # Combine with Bayesian weights
            combined = (w_regret * regret_scores[i, j] +
                       w_lk * lk_penalties[i, j] +
                       w_christofides * christofides_norm +
                       w_embedding * embedding_similarity[i, j])
            
            # Add small deterministic tie-breaking noise
            noise = 1e-8 * ((i * 131 + j * 241) % 1000) / 1000.0
            heuristics[i, j] = heuristics[j, i] = combined + noise
    
    # Row-wise softmax normalization
    for i in range(n):
        row = heuristics[i]
        if np.any(row > 0):
            row_max = np.max(row)
            if row_max > 0:
                # Sharpened softmax for better discrimination
                exp_row = np.exp((row - row_max) * 6.0)
                heuristics[i] = exp_row / np.sum(exp_row)
    
    # Final symmetrization and cleanup
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # Ensure all rows sum to 1
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    return heuristics
