import numpy as np
from scipy.sparse.csgraph import connected_components
from scipy.sparse import csr_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic for TSP edge prioritization with enhanced regret analysis.
    
    Features:
    1. Regret-augmented inverse distance with vectorized computation
    2. Efficient two-hop analysis using candidate screening
    3. Adaptive sparsification based on local regret patterns
    4. Mutual rank reinforcement for promising reciprocal edges
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive Regret-Based Sparsification ====================
    # Determine connectivity based on local regret patterns
    base_k = max(3, int(np.log2(n)))
    
    # Precompute sorted neighbors for all nodes
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_distances = np.sort(distance_matrix, axis=1)
    
    # Compute local regret statistics for each node
    regret_stats = np.zeros(n)
    for i in range(n):
        if sorted_distances[i, 1] > 0:  # Ensure non-zero minimum distance
            # Compute regret for top candidates
            top_k = min(base_k + 2, n - 1)
            distances = sorted_distances[i, 1:top_k+1]
            min_dist = distances[0]
            regrets = (distances - min_dist) / (min_dist + 1e-10)
            
            # Higher regret dispersion indicates more competitive alternatives
            regret_stats[i] = np.std(regrets) if len(regrets) > 1 else 0
    
    # Normalize regret statistics
    if regret_stats.max() > 0:
        regret_stats = regret_stats / regret_stats.max()
    
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Build adaptive neighbor sets based on regret patterns
    for i in range(n):
        # Adaptive k based on regret dispersion
        regret_factor = 0.7 + 0.6 * regret_stats[i]  # Range [0.7, 1.3]
        adaptive_k = min(n - 1, max(3, int(base_k * regret_factor)))
        
        # Get nearest neighbors
        nn_indices = sorted_indices[i, 1:adaptive_k+1]
        adjacency_mask[i, nn_indices] = True
        
        # Ensure reciprocal connections for high-regret nodes
        if regret_stats[i] > 0.5:
            for j in nn_indices:
                if i in sorted_indices[j, 1:adaptive_k+1]:
                    adjacency_mask[j, i] = True
    
    # Ensure graph connectivity
    adjacency_mask = adjacency_mask | adjacency_mask.T
    sparse_graph = csr_matrix(adjacency_mask.astype(int))
    n_components, labels = connected_components(sparse_graph, directed=False, return_labels=True)
    
    if n_components > 1:
        # Connect components with their nearest neighbors
        for comp in range(n_components):
            nodes_in_comp = np.where(labels == comp)[0]
            if len(nodes_in_comp) == 0:
                continue
            
            # Find closest node from other components
            for node in nodes_in_comp:
                distances = distance_matrix[node].copy()
                distances[node] = np.inf
                other_nodes = np.where(labels != comp)[0]
                if len(other_nodes) > 0:
                    closest_other = other_nodes[np.argmin(distances[other_nodes])]
                    adjacency_mask[node, closest_other] = adjacency_mask[closest_other, node] = True
    
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Enhanced Regret-Augmented Inverse Distance ====================
    regret_scores = np.zeros_like(distance_matrix, dtype=float)
    
    # Vectorized computation of regret scores
    for i in range(n):
        mask_i = adjacency_mask[i]
        if np.any(mask_i):
            valid_dists = distance_matrix[i, mask_i]
            min_dist = np.min(valid_dists)
            
            if min_dist > 0:
                # Enhanced regret computation with stronger penalty
                inv_vals = 1.0 / (valid_dists + 1e-10)
                regret_penalty = np.exp(-2.0 * (valid_dists - min_dist) / (min_dist + 1e-10))
                
                # Additional penalty for edges much longer than alternatives
                alt_factor = np.clip((valid_dists - min_dist) / min_dist, 0, 2)
                length_penalty = np.exp(-0.5 * alt_factor)
                
                regret_scores[i, mask_i] = inv_vals * regret_penalty * length_penalty
    
    # Symmetrize scores using geometric mean for better consistency
    nonzeros = (regret_scores > 0) & (regret_scores.T > 0)
    regret_scores[nonzeros] = np.sqrt(regret_scores[nonzeros] * regret_scores.T[nonzeros])
    regret_scores = np.maximum(regret_scores, regret_scores.T)
    
    # ==================== 3. Efficient Two-Hop Analysis ====================
    two_hop_penalty = np.ones_like(distance_matrix)
    
    # Vectorized two-hop analysis for all promising edges
    edge_mask = np.triu(adjacency_mask, k=1)
    edge_indices = np.where(edge_mask)
    
    if len(edge_indices[0]) > 0:
        i_vals, j_vals = edge_indices
        
        # Precompute candidate nodes for two-hop paths
        candidate_depth = max(5, int(np.sqrt(n)))
        top_neighbors = sorted_indices[:, 1:candidate_depth+1]
        
        for idx in range(len(i_vals)):
            i, j = i_vals[idx], j_vals[idx]
            direct_dist = distance_matrix[i, j]
            
            if direct_dist < 1e-10:
                continue
            
            # Get candidate intermediate nodes (neighbors of either endpoint)
            i_candidates = top_neighbors[i]
            j_candidates = top_neighbors[j]
            candidates = np.union1d(i_candidates, j_candidates)
            candidates = candidates[(candidates != i) & (candidates != j)]
            
            if len(candidates) > 0:
                # Compute two-hop distances
                two_hop_dists = distance_matrix[i, candidates] + distance_matrix[candidates, j]
                if len(two_hop_dists) > 0:
                    min_two_hop = np.min(two_hop_dists)
                    
                    if min_two_hop > 0:
                        ratio = direct_dist / min_two_hop
                        # Adaptive penalty based on ratio: stronger penalty when direct edge is much worse
                        if ratio > 1.2:  # Direct edge is at least 20% worse
                            penalty = 0.3 + 0.7 * np.exp(-(ratio - 1.2))
                        else:
                            penalty = 1.0
                        
                        two_hop_penalty[i, j] = two_hop_penalty[j, i] = penalty
    
    # ==================== 4. Enhanced Mutual Rank Reinforcement ====================
    rank_bonus = np.ones_like(distance_matrix)
    
    # Compute rank-based bonuses for edges where both endpoints rank each other highly
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Find ranks in each other's sorted neighbor lists
                i_rank = np.where(sorted_indices[i] == j)[0][0] if j in sorted_indices[i] else n
                j_rank = np.where(sorted_indices[j] == i)[0][0] if i in sorted_indices[j] else n
                
                # Enhanced bonus based on mutual ranking
                if i_rank <= 2 and j_rank <= 2:
                    rank_bonus[i, j] = rank_bonus[j, i] = 1.5
                elif i_rank <= 3 and j_rank <= 3:
                    rank_bonus[i, j] = rank_bonus[j, i] = 1.3
                elif i_rank <= 5 or j_rank <= 5:
                    rank_bonus[i, j] = rank_bonus[j, i] = 1.1
    
    # ==================== 5. Combine Components with Adaptive Weights ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Simple adaptive weighting based on graph connectivity
    edge_density = adjacency_mask.sum() / (n * (n - 1))
    
    # Denser graphs benefit more from two-hop analysis
    two_hop_weight = 0.5 + 0.5 * edge_density
    base_weight = 1.0
    
    # Combine scores
    mask_combined = adjacency_mask & (distance_matrix > 0)
    heuristics[mask_combined] = (
        base_weight * regret_scores[mask_combined] * 
        (two_hop_weight * two_hop_penalty[mask_combined] + (1 - two_hop_weight)) * 
        rank_bonus[mask_combined]
    )
    
    # ==================== 6. Connectivity Assurance ====================
    # Ensure all nodes have at least some promising connections
    for i in range(n):
        if heuristics[i].sum() == 0:
            # Add connections to nearest neighbors
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            nn_indices = sorted_indices[i, 1:4]  # Top 3 neighbors
            for j in nn_indices:
                inv_val = 1.0 / (distance_matrix[i, j] + 1e-10)
                heuristics[i, j] = heuristics[j, i] = max(heuristics[i, j], inv_val)
    
    # ==================== 7. Final Normalization ====================
    np.fill_diagonal(heuristics, 0)
    
    # Row normalization with smoothing
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Symmetrize using arithmetic mean
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
