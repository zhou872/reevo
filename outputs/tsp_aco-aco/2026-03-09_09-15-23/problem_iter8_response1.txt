import numpy as np
from scipy.sparse.csgraph import connected_components
from scipy.sparse import csr_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with multi-hop regret analysis,
    spectral clustering integration, and adaptive component weighting.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Density-Adaptive Sparsification ====================
    # Use local density to determine sparsification level
    local_densities = np.zeros(n)
    base_k = max(3, int(np.log2(n)))
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nearest = np.partition(distances, min(base_k, n-1))[:min(base_k, n-1)]
        local_densities[i] = 1.0 / (np.mean(nearest) + 1e-10)
    
    # Normalize densities
    density_min, density_max = local_densities.min(), local_densities.max()
    if density_max > density_min:
        normalized_densities = (local_densities - density_min) / (density_max - density_min)
    else:
        normalized_densities = np.ones(n)
    
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Adaptive k based on local density and global statistics
        density_factor = 0.5 + 1.5 * normalized_densities[i]  # Range [0.5, 2.0]
        adaptive_k = min(n-1, int(base_k * density_factor))
        
        # Get nearest neighbors
        nn_indices = np.argpartition(distances, adaptive_k)[:adaptive_k]
        adjacency_mask[i, nn_indices] = True
        
        # Ensure connectivity for low-density nodes
        if normalized_densities[i] < 0.3:
            # Add reciprocal connections to maintain connectivity
            for j in nn_indices:
                if not adjacency_mask[j, i]:
                    j_distances = distance_matrix[j].copy()
                    j_distances[j] = np.inf
                    j_k = min(adaptive_k, n-1)
                    j_nn = np.argpartition(j_distances, j_k)[:j_k]
                    if i in j_nn:
                        adjacency_mask[j, i] = True
    
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Spectral Clustering Integration ====================
    # Use connected components of sparsified graph as natural clusters
    sparse_graph = csr_matrix(adjacency_mask.astype(int))
    n_components, labels = connected_components(sparse_graph, directed=False, return_labels=True)
    
    # Compute cluster statistics
    cluster_sizes = np.bincount(labels)
    cluster_centroids = np.zeros((n_components, n))
    for c in range(n_components):
        nodes_in_cluster = np.where(labels == c)[0]
        if len(nodes_in_cluster) > 0:
            # Use average distance to all other nodes as centroid feature
            cluster_centroids[c] = distance_matrix[nodes_in_cluster].mean(axis=0)
    
    # Compute cluster-based scores
    cluster_scores = np.ones_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                if labels[i] == labels[j]:
                    # Same cluster: boost edges within small clusters
                    cluster_size = cluster_sizes[labels[i]]
                    size_factor = 2.0 / (1.0 + np.log1p(cluster_size))
                    cluster_scores[i, j] = cluster_scores[j, i] = size_factor
                else:
                    # Different clusters: penalize unless clusters are close
                    c1, c2 = labels[i], labels[j]
                    cluster_dist = np.linalg.norm(cluster_centroids[c1] - cluster_centroids[c2])
                    avg_cluster_dist = distance_matrix.mean()
                    
                    # Normalize and apply penalty
                    norm_dist = cluster_dist / (avg_cluster_dist + 1e-10)
                    penalty = 0.5 + 0.5 * np.exp(-norm_dist)
                    cluster_scores[i, j] = cluster_scores[j, i] = penalty
    
    # ==================== 3. Multi-Hop Regret Analysis ====================
    # Precompute sorted neighbors for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_distances = np.sort(distance_matrix, axis=1)
    
    multi_hop_scores = np.ones_like(distance_matrix)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        valid_indices = np.where(valid_mask)[0]
        
        for j in valid_indices:
            if i >= j:
                continue
            
            direct = distance_matrix[i, j]
            if direct < 1e-10:
                continue
            
            # 2-hop analysis
            best_2hop = np.inf
            # Check common neighbors
            common_neighbors = np.where(adjacency_mask[i] & adjacency_mask[j])[0]
            common_neighbors = common_neighbors[(common_neighbors != i) & (common_neighbors != j)]
            
            if len(common_neighbors) > 0:
                two_hop_dists = distance_matrix[i, common_neighbors] + distance_matrix[common_neighbors, j]
                best_2hop = np.min(two_hop_dists)
            
            # 3-hop analysis (sampled for efficiency)
            best_3hop = np.inf
            sample_size = min(10, n)
            
            # Sample intermediate nodes
            for _ in range(min(sample_size, n-2)):
                k = np.random.choice([x for x in range(n) if x not in (i, j)])
                if not adjacency_mask[i, k]:
                    continue
                
                # Find l that connects k to j
                possible_l = np.where(adjacency_mask[k] & adjacency_mask[j])[0]
                possible_l = possible_l[(possible_l != i) & (possible_l != j) & (possible_l != k)]
                
                if len(possible_l) > 0:
                    for l in possible_l[:2]:  # Check at most 2 options
                        three_hop = distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j]
                        if three_hop < best_3hop:
                            best_3hop = three_hop
            
            # Compute multi-hop regret scores
            regret_2hop = 0.0
            if best_2hop < np.inf:
                regret_2hop = max(0, (direct - best_2hop) / (direct + 1e-10))
            
            regret_3hop = 0.0
            if best_3hop < np.inf:
                regret_3hop = max(0, (direct - best_3hop) / (direct + 1e-10))
            
            # Combined multi-hop score
            if best_2hop < np.inf or best_3hop < np.inf:
                # Weighted combination of 2-hop and 3-hop regrets
                weight_2hop = 0.7 if best_2hop < np.inf else 0.0
                weight_3hop = 0.3 if best_3hop < np.inf else 0.0
                total_weight = weight_2hop + weight_3hop
                
                if total_weight > 0:
                    avg_regret = (weight_2hop * regret_2hop + weight_3hop * regret_3hop) / total_weight
                    # Convert regret to score (higher regret -> higher score)
                    multi_hop_scores[i, j] = multi_hop_scores[j, i] = 1.0 + 0.5 * avg_regret
    
    # ==================== 4. Adaptive Inverse Distance Scoring ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            valid_dists = distance_matrix[i, valid_mask]
            
            # Rank-based weighting with exponential decay
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            weights = np.exp(-0.3 * (ranks - 1))
            weights = weights / np.sum(weights)
            
            # Adaptive inverse distance based on local density
            density_weight = 0.5 + 0.5 * normalized_densities[i]
            inv_vals = 1.0 / (valid_dists ** density_weight + 1e-10)
            
            inv_dist[i, valid_mask] = inv_vals * weights
    
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    
    # ==================== 5. Adaptive Component Weighting ====================
    # Dynamically weight different heuristic components
    component_weights = np.ones(3)
    
    # Adjust weights based on problem characteristics
    avg_distance = distance_matrix.mean()
    distance_variance = distance_matrix.var()
    
    if distance_variance > 0:
        cv = np.sqrt(distance_variance) / avg_distance
        # Higher variance -> more emphasis on multi-hop analysis
        component_weights[0] = 1.0  # Base inverse distance weight
        component_weights[1] = 0.5 + 0.5 * min(1.0, cv)  # Multi-hop weight
        component_weights[2] = 1.0 - 0.3 * min(1.0, cv)  # Cluster weight
    
    # Normalize weights
    component_weights = component_weights / component_weights.sum()
    
    # ==================== 6. Combine All Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Weighted combination of components
                score = (component_weights[0] * inv_dist[i, j] +
                        component_weights[1] * multi_hop_scores[i, j] +
                        component_weights[2] * cluster_scores[i, j])
                
                # Apply local density adjustment
                avg_density = 0.5 * (normalized_densities[i] + normalized_densities[j])
                density_boost = 0.8 + 0.4 * avg_density  # Range [0.8, 1.2]
                
                heuristics[i, j] = heuristics[j, i] = score * density_boost
    
    # ==================== 7. Connectivity Enhancement ====================
    # Ensure all nodes are reachable in the heuristic graph
    for i in range(n):
        row_nonzero = np.where(heuristics[i] > 0)[0]
        if len(row_nonzero) == 0:
            # Node has no promising edges, add connections to nearest neighbors
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            nn_indices = np.argpartition(distances, min(3, n-1))[:min(3, n-1)]
            for j in nn_indices:
                inv_val = 1.0 / (distance_matrix[i, j] + 1e-10)
                heuristics[i, j] = heuristics[j, i] = max(heuristics[i, j], inv_val)
    
    # ==================== 8. Final Processing ====================
    np.fill_diagonal(heuristics, 0)
    
    # Row normalization with smoothing
    row_max = heuristics.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    heuristics_normalized = np.exp(heuristics / row_max - 1)
    heuristics_normalized[np.diag_indices(n)] = 0
    
    # Add small epsilon to avoid division by zero
    heuristics_normalized = heuristics_normalized + 1e-8
    
    row_sums = heuristics_normalized.sum(axis=1, keepdims=True)
    heuristics = heuristics_normalized / row_sums
    
    # Final symmetrization
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
