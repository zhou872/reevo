import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges combining multiple factors:
    1. Two-step lookahead spatial scoring
    2. Cluster density bonus for edges within dense regions
    3. Penalty for edges with high distance variance to other nodes
    4. Sparsification via k-nearest neighbors
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Avoid division by zero and self-loops
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    inv_dist = 1.0 / safe_dist
    
    # Factor 1: Two-step lookahead spatial scoring
    # For each edge (i,j), consider best connections from i and j to other nodes
    nearest_to_i = np.partition(safe_dist, 2, axis=1)[:, :2]  # Two nearest per node
    nearest_to_j = np.partition(safe_dist, 2, axis=0)[:2, :]
    
    lookahead_scores = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Best alternative connections excluding each other
            alt_i = nearest_to_i[i, 1] if j == np.argmin(safe_dist[i]) else nearest_to_i[i, 0]
            alt_j = nearest_to_j[1, j] if i == np.argmin(safe_dist[:, j]) else nearest_to_j[0, j]
            lookahead_scores[i, j] = (alt_i + alt_j) / (2 * safe_dist[i, j])
    
    # Factor 2: Cluster density bonus
    # Compute local density around each node (inverse of average distance to k neighbors)
    k_cluster = min(5, n-1)
    densities = np.zeros(n)
    for i in range(n):
        k_nearest = np.partition(safe_dist[i], k_cluster)[:k_cluster]
        densities[i] = 1.0 / np.mean(k_nearest)
    
    density_bonus = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            density_bonus[i, j] = (densities[i] + densities[j]) / 2
    
    # Factor 3: Distance variance penalty
    # Penalize edges where distances to other nodes are very different
    var_penalty = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Variance of distances from i and j to all other nodes
            var_i = np.var([safe_dist[i, k] for k in range(n) if k not in (i, j)])
            var_j = np.var([safe_dist[j, k] for k in range(n) if k not in (i, j)])
            var_penalty[i, j] = 1.0 / (1.0 + (var_i + var_j) / 2)
    
    # Combine factors with weights
    base_score = inv_dist * 0.4
    lookahead_component = lookahead_scores * 0.3
    density_component = density_bonus * 0.2
    variance_component = var_penalty * 0.1
    
    combined = base_score + lookahead_component + density_component + variance_component
    
    # Sparsification: Keep only edges in k-nearest neighbor graph
    k_sparse = min(3, n-1)
    mask = np.zeros_like(combined, dtype=bool)
    
    for i in range(n):
        # Get indices of k-nearest neighbors for node i
        indices = np.argpartition(safe_dist[i], k_sparse)[:k_sparse]
        mask[i, indices] = True
    
    # Symmetrize the mask (undirected graph)
    mask = mask | mask.T
    
    # Apply sparsification
    result = combined * mask
    
    # Normalize to reasonable range
    max_val = np.max(result[result < np.inf])
    if max_val > 0:
        result = result / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(result, 0.0)
    
    return result
