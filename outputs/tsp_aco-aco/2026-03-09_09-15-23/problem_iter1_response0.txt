import numpy as np
from scipy.spatial import distance_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization.
    Combines multiple factors and sparsifies the matrix by setting
    unpromising edges to zero.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (handle zeros for diagonal)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.where(distance_matrix > 0, 1.0 / distance_matrix, 0)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Nearest neighbor bonus (exponential decay)
    k = min(5, n // 4)  # Adaptive k for nearest neighbors
    nn_bonus = np.zeros((n, n))
    for i in range(n):
        sorted_idx = np.argsort(distance_matrix[i])
        # Exponential decay bonus for k-nearest neighbors
        decay = np.exp(-np.arange(k))
        nn_bonus[i, sorted_idx[1:k+1]] = decay / decay.sum()
    
    # 3. Degree-based centrality weighting
    # Nodes that are centrally located (closer to all others) get higher weight
    centrality = 1.0 / np.sum(distance_matrix, axis=1, keepdims=True)
    centrality_weight = centrality + centrality.T  # Symmetric
    
    # 4. Two-hop connection density
    # Reward edges that connect to nodes with overlapping neighborhoods
    two_hop_density = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Find common neighbors within threshold distance
            threshold = np.percentile(distance_matrix[i], 25)
            neighbors_i = set(np.where(distance_matrix[i] < threshold)[0])
            neighbors_j = set(np.where(distance_matrix[j] < threshold)[0])
            common = len(neighbors_i.intersection(neighbors_j))
            two_hop_density[i, j] = two_hop_density[j, i] = common / max(len(neighbors_i), 1)
    
    # 5. Combine factors with weights
    weights = [0.4, 0.2, 0.2, 0.2]  # Tuned weights for each factor
    combined = (weights[0] * inv_dist + 
                weights[1] * nn_bonus + 
                weights[2] * centrality_weight + 
                weights[3] * two_hop_density)
    
    # 6. Sparsification: Keep only top edges per node
    sparsified = np.zeros((n, n))
    edges_per_node = max(3, int(np.log2(n)))  # Adaptive sparsity
    
    for i in range(n):
        # Get top edges for this node (excluding diagonal)
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        top_indices = np.argpartition(row, -edges_per_node)[-edges_per_node:]
        sparsified[i, top_indices] = row[top_indices]
    
    # 7. Symmetrize (since TSP edges are undirected)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Normalize to [0, 1] range
    max_val = sparsified.max()
    if max_val > 0:
        sparsified /= max_val
    
    return sparsified


# Example usage and validation
if __name__ == "__main__":
    # Generate a sample distance matrix
    np.random.seed(42)
    n = 10
    coords = np.random.rand(n, 2)
    dist_mat = distance_matrix(coords, coords)
    
    # Test the heuristic function
    heuristics = heuristics_v2(dist_mat)
    
    print(f"Original distance matrix shape: {dist_mat.shape}")
    print(f"Heuristics matrix shape: {heuristics.shape}")
    print(f"Sparsity: {np.mean(heuristics == 0):.2%}")
    print(f"Non-zero mean: {heuristics[heuristics > 0].mean():.4f}")
    print(f"Top 5 edges: {np.sort(heuristics.flatten())[-5:][::-1]}")
