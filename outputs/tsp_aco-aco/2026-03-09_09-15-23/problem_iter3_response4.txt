import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with adaptive sparsification
    and local competition normalization.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive Neighbor Sparsification ====================
    # Use adaptive k based on problem size: max(3, log2(n)) nearest neighbors
    k = max(3, int(np.log2(n)))
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        # Get k nearest neighbors
        nn_indices = np.argpartition(distances, k)[:k]
        adjacency_mask[i, nn_indices] = True
    
    # Ensure symmetry
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Inverse Distance with Rank Weighting ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        # Get distances for this node's neighbors
        row_dists = distance_matrix[i].copy()
        row_dists[i] = np.inf
        valid_mask = adjacency_mask[i]
        
        if np.any(valid_mask):
            valid_dists = row_dists[valid_mask]
            # Rank by distance (closest = rank 1)
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            # Weight inversely with rank (closer neighbors get higher weights)
            weights = 1.0 / ranks
            weights = weights / np.sum(weights)  # Normalize
            
            # Apply weighted inverse distance
            inv_vals = 1.0 / (valid_dists + 1e-10)
            inv_dist[i, valid_mask] = inv_vals * weights
    
    # Symmetrize
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    
    # ==================== 3. Local Competition Normalization ====================
    # Scale by reciprocal of number of neighbors to account for local competition
    local_norm = np.zeros_like(distance_matrix, dtype=float)
    neighbor_counts = adjacency_mask.sum(axis=1)
    
    for i in range(n):
        if neighbor_counts[i] > 0:
            local_norm[i] = 1.0 / neighbor_counts[i]
    
    # Apply normalization symmetrically
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Use geometric mean of local normalization factors
                norm_factor = np.sqrt(local_norm[i, i] * local_norm[j, j])
                inv_dist[i, j] *= norm_factor
                inv_dist[j, i] = inv_dist[i, j]
    
    # ==================== 4. Two-Hop Shortcut Bonus ====================
    # Bonus for edges that are significantly shorter than two-hop alternatives
    two_hop_bonus = np.ones_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                direct_dist = distance_matrix[i, j]
                # Find minimum two-hop path through any intermediate node
                min_two_hop = np.inf
                for k in range(n):
                    if k != i and k != j:
                        two_hop_dist = distance_matrix[i, k] + distance_matrix[k, j]
                        if two_hop_dist < min_two_hop:
                            min_two_hop = two_hop_dist
                
                if min_two_hop < np.inf and direct_dist > 1e-10:
                    ratio = min_two_hop / direct_dist
                    # Stronger bonus for edges that are much shorter than alternatives
                    if ratio > 1.5:
                        bonus = min(2.0, ratio / 2.0)
                        two_hop_bonus[i, j] = two_hop_bonus[j, i] = bonus
    
    # ==================== 5. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                score = inv_dist[i, j] * two_hop_bonus[i, j]
                heuristics[i, j] = heuristics[j, i] = score
    
    # ==================== 6. Final Processing ====================
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Add small constant to avoid zeros
    heuristics[adjacency_mask] += 1e-8
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1  # Avoid division by zero
    heuristics = heuristics / row_sums
    
    # Symmetrize final matrix
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
