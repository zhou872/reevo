def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristic function with optimized matching algorithm,
    adaptive multi-hop regret analysis, and meta-heuristic weight learning.
    
    Args:
        distance_matrix: Square matrix of pairwise distances between n nodes
        
    Returns:
        Heuristic matrix of same shape, with higher values indicating more promising edges
    """
    n = len(distance_matrix)
    
    # Handle trivial cases
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics

    heuristics = _fuse_with_learned_weights(

    )
    
    return heuristics

def _compute_optimized_christofides_edges(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute Christofides algorithm components with optimized Blossom matching."""
    n = len(distance_matrix)
    edge_scores = np.zeros_like(distance_matrix, dtype=float)
    
    # 1. Compute Minimum Spanning Tree using Prim's algorithm
    mst_mask = _compute_mst_mask(distance_matrix)
    
    # 2. Find odd-degree vertices in MST
    odd_degree_mask = _find_odd_degree_vertices(mst_mask)
    
    # 3. Optimized minimum weight perfect matching
    matching_mask = _optimized_blossom_matching(distance_matrix, odd_degree_mask)
    
    # 4. Enhanced scoring with path length consideration
    for i in range(n):
        for j in range(i + 1, n):
            if mst_mask[i, j]:
                # MST edges: weighted by their position in the tree
                degree_i = np.sum(mst_mask[i])
                degree_j = np.sum(mst_mask[j])
                centrality = 1.0 / (degree_i * degree_j + 1e-10)
                edge_scores[i, j] = edge_scores[j, i] = 1.8 + 0.4 * np.tanh(centrality)
            elif matching_mask[i, j]:
                # Matching edges: weighted by distance savings
                avg_odd_dist = np.mean(distance_matrix[odd_degree_mask])
                dist_saving = avg_odd_dist / (distance_matrix[i, j] + 1e-10)
                edge_scores[i, j] = edge_scores[j, i] = 1.5 + 0.3 * np.tanh(dist_saving)
            else:
                # Non-Christofides edges
                edge_scores[i, j] = edge_scores[j, i] = 1.0
    
    return edge_scores

def _compute_mst_mask(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute Minimum Spanning Tree using Prim's algorithm."""
    n = len(distance_matrix)
    mst_mask = np.zeros((n, n), dtype=bool)
    
    if n == 0:
        return mst_mask
    
    visited = np.zeros(n, dtype=bool)
    min_edge_weights = np.full(n, np.inf)
    min_edge_sources = np.full(n, -1, dtype=int)
    
    # Start from node with median distance to others
    avg_distances = np.mean(distance_matrix, axis=1)
    start_node = np.argsort(avg_distances)[len(avg_distances) // 2]
    min_edge_weights[start_node] = 0
    
    for _ in range(n):
        # Find unvisited vertex with minimum edge weight
        current = -1
        min_weight = np.inf
        for v in range(n):
            if not visited[v] and min_edge_weights[v] < min_weight:
                min_weight = min_edge_weights[v]
                current = v
        
        if current == -1:
            break
        
        visited[current] = True
        
        # Add edge to MST if not the starting node
        if min_edge_sources[current] != -1:
            source = min_edge_sources[current]
            mst_mask[current, source] = mst_mask[source, current] = True
        
        # Update minimum edge weights for neighbors
        for neighbor in range(n):
            if not visited[neighbor] and distance_matrix[current, neighbor] < min_edge_weights[neighbor]:
                min_edge_weights[neighbor] = distance_matrix[current, neighbor]
                min_edge_sources[neighbor] = current
    
    return mst_mask

def _find_odd_degree_vertices(mst_mask: np.ndarray) -> np.ndarray:
    """Identify vertices with odd degree in the MST."""
    n = len(mst_mask)
    degrees = np.sum(mst_mask, axis=1)
    odd_mask = (degrees % 2 == 1)
    return odd_mask

def _optimized_blossom_matching(distance_matrix: np.ndarray, 
                               odd_degree_mask: np.ndarray) -> np.ndarray:
    """Optimized Blossom algorithm for minimum weight perfect matching."""
    n = len(distance_matrix)
    matching_mask = np.zeros((n, n), dtype=bool)
    
    # Get list of odd-degree vertices
    odd_vertices = np.where(odd_degree_mask)[0]
    m = len(odd_vertices)
    
    if m % 2 != 0 or m == 0:
        return matching_mask
    
    # For small problems, use exact algorithm
    if m <= 12:
        return _exact_matching_dp(distance_matrix, odd_vertices)
    
    # For larger problems, use optimized greedy with local search
    return _greedy_matching_with_local_search(distance_matrix, odd_vertices)

def _exact_matching_dp(distance_matrix: np.ndarray, odd_vertices: np.ndarray) -> np.ndarray:
    """Exact DP for minimum weight perfect matching (up to 12 vertices)."""
    n = len(distance_matrix)
    m = len(odd_vertices)
    matching_mask = np.zeros((n, n), dtype=bool)
    
    # Map odd vertices to indices 0..m-1
    vertex_to_idx = {v: i for i, v in enumerate(odd_vertices)}
    
    # DP table: dp[mask] = minimum cost for matching vertices represented by mask
    dp = [float('inf')] * (1 << m)
    parent = [-1] * (1 << m)
    dp[0] = 0
    
    for mask in range(1 << m):
        if dp[mask] == float('inf'):
            continue
        
        # Find first unmatched vertex
        i = 0
        while i < m and (mask >> i) & 1:
            i += 1
        
        if i >= m:
            continue
        
        # Try matching with all remaining unmatched vertices
        for j in range(i + 1, m):
            if not (mask >> j) & 1:
                new_mask = mask | (1 << i) | (1 << j)
                cost = dp[mask] + distance_matrix[odd_vertices[i], odd_vertices[j]]
                if cost < dp[new_mask]:
                    dp[new_mask] = cost
                    parent[new_mask] = mask
    
    # Reconstruct matching
    mask = (1 << m) - 1
    while mask > 0:
        prev_mask = parent[mask]
        diff = mask ^ prev_mask
        i = (diff & -diff).bit_length() - 1
        diff ^= (1 << i)
        j = (diff & -diff).bit_length() - 1
        
        v1, v2 = odd_vertices[i], odd_vertices[j]
        matching_mask[v1, v2] = matching_mask[v2, v1] = True
        mask = prev_mask
    
    return matching_mask

def _greedy_matching_with_local_search(distance_matrix: np.ndarray, 
                                      odd_vertices: np.ndarray) -> np.ndarray:
    """Greedy matching with 2-opt local search improvements."""
    n = len(distance_matrix)
    m = len(odd_vertices)
    matching_mask = np.zeros((n, n), dtype=bool)
    
    # Initial greedy matching
    paired = set()
    edges = []
    
    # Create sorted list of all possible edges between odd vertices
    all_edges = []
    for i in range(m):
        for j in range(i + 1, m):
            v1, v2 = odd_vertices[i], odd_vertices[j]
            all_edges.append((distance_matrix[v1, v2], v1, v2))
    
    all_edges.sort()
    
    # Greedy matching
    for dist, v1, v2 in all_edges:
        if v1 not in paired and v2 not in paired:
            paired.update([v1, v2])
            edges.append((v1, v2))
            matching_mask[v1, v2] = matching_mask[v2, v1] = True
    
    # 2-opt local search improvement
    improved = True
    while improved:
        improved = False
        for i in range(len(edges)):
            for j in range(i + 1, len(edges)):
                a, b = edges[i]
                c, d = edges[j]
                
                current_cost = (distance_matrix[a, b] + distance_matrix[c, d])
                new_cost = (distance_matrix[a, c] + distance_matrix[b, d])
                
                if new_cost < current_cost - 1e-10:
                    # Perform swap
                    matching_mask[a, b] = matching_mask[b, a] = False
                    matching_mask[c, d] = matching_mask[d, c] = False
                    matching_mask[a, c] = matching_mask[c, a] = True
                    matching_mask[b, d] = matching_mask[d, b] = True
                    
                    edges[i] = (a, c)
                    edges[j] = (b, d)
                    improved = True
    
    return matching_mask

def _build_density_adaptive_adjacency(distance_matrix: np.ndarray,
                                     christofides_edges: np.ndarray) -> np.ndarray:
    """Build adjacency matrix with density-adaptive neighborhood size."""
    n = len(distance_matrix)
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Always include Christofides edges
    adjacency_mask[christofides_edges > 1.0] = True
    
    # Compute local density
    k_density = min(10, n // 4)
    local_density = np.zeros(n)
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        if k_density > 0:
            local_density[i] = 1.0 / np.mean(distance_matrix[i, sorted_indices[i, 1:k_density+1]])
    
    # Normalize density
    if np.max(local_density) > 0:
        local_density = local_density / np.max(local_density)
    
    # Adaptive neighbor selection
    base_k = max(3, int(np.log2(n)))
    for i in range(n):
        # Denser areas: more neighbors, sparser areas: fewer but closer neighbors
        density_factor = 0.5 + local_density[i]
        k_adaptive = int(base_k * density_factor)
        k_adaptive = max(2, min(k_adaptive, n - 1))
        
        # Select nearest neighbors
        neighbors = sorted_indices[i, 1:k_adaptive+1]
        adjacency_mask[i, neighbors] = True
    
    # Ensure symmetry and no self-loops
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    return adjacency_mask

def _compute_adaptive_multi_hop_regret(distance_matrix: np.ndarray,
                                      adjacency_mask: np.ndarray) -> np.ndarray:
    """Compute regret scores with adaptive hop count based on graph density."""
    n = len(distance_matrix)
    regret_scores = np.ones_like(distance_matrix, dtype=float)
    
    # Estimate graph density
    total_possible = n * (n - 1) // 2
    actual_edges = np.sum(adjacency_mask) // 2
    density = actual_edges / total_possible if total_possible > 0 else 0
    
    # Adaptive parameters
    if density < 0.1:  # Sparse graph
        max_hops = 4
        neighbor_limit = min(6, int(np.sqrt(n)))
    elif density < 0.3:  # Moderate density
        max_hops = 3
        neighbor_limit = min(10, int(np.sqrt(n) * 1.5))
    else:  # Dense graph
        max_hops = 2
        neighbor_limit = min(15, int(n * 0.2))
    
    # Precompute sorted neighbors
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    # Vectorized computation for efficiency
    edge_indices = np.where(np.triu(adjacency_mask, 1))
    i_list, j_list = edge_indices
    
    if len(i_list) == 0:
        return regret_scores
    
    direct_dists = distance_matrix[i_list, j_list]
    best_alternatives = np.full(len(i_list), np.inf)
    
    # Multi-hop analysis
    for hop in range(2, max_hops + 1):
        # Get candidate intermediate nodes (top neighbors)
        for i_idx in range(len(i_list)):
            i, j = i_list[i_idx], j_list[i_idx]
            
            # Generate hop-length paths using BFS with depth limit
            paths = _find_k_hop_paths(i, j, hop, distance_matrix, 
                                     sorted_indices, neighbor_limit)
            
            for path in paths:
                if len(path) == hop + 1:  # Valid hop-length path
                    path_dist = sum(distance_matrix[path[k], path[k+1]] 
                                  for k in range(len(path)-1))
                    if path_dist < best_alternatives[i_idx]:
                        best_alternatives[i_idx] = path_dist
    
    # Compute regret scores
    valid_mask = direct_dists > 1e-10
    regret = np.zeros(len(i_list))
    regret[valid_mask] = (best_alternatives[valid_mask] - direct_dists[valid_mask]) / direct_dists[valid_mask]
    
    # Sigmoid mapping: higher regret (direct is much better) -> higher score
    scores = 1.0 / (1.0 + np.exp(-5.0 * regret))
    
    # Apply scores
    regret_scores[i_list, j_list] = scores
    regret_scores[j_list, i_list] = scores
    
    return regret_scores

def _find_k_hop_paths(start: int, end: int, max_hops: int,
                     distance_matrix: np.ndarray,
                     sorted_indices: np.ndarray,
                     neighbor_limit: int) -> list:
    """Find promising k-hop paths between two nodes."""
    n = len(distance_matrix)
    paths = []
    
    # Use limited BFS
    queue = [(start, [start], 0.0)]
    
    while queue and len(paths) < 10:  # Limit number of paths considered
        node, path, cost = queue.pop(0)
        
        if len(path) > max_hops + 1:
            continue
        
        if node == end and len(path) > 1:
            paths.append(path)
            continue
        
        # Expand to promising neighbors
        neighbors = sorted_indices[node, 1:neighbor_limit+1]
        for neighbor in neighbors:
            if neighbor not in path:
                new_cost = cost + distance_matrix[node, neighbor]
                
                # Prune if path is already too long
                if new_cost > distance_matrix[start, end] * 2:
                    continue
                
                queue.append((neighbor, path + [neighbor], new_cost))
    
    return paths

def _compute_structural_pattern_scores(distance_matrix: np.ndarray,
                                      adjacency_mask: np.ndarray) -> np.ndarray:
    """Compute scores based on structural patterns in the graph."""
    n = len(distance_matrix)
    pattern_scores = np.zeros_like(distance_matrix, dtype=float)
    
    # Compute clustering coefficient for each node
    clustering_coeff = np.zeros(n)
    for i in range(n):
        neighbors = np.where(adjacency_mask[i])[0]
        if len(neighbors) < 2:
            clustering_coeff[i] = 0
            continue
        
        # Count triangles involving node i
        triangles = 0
        for j in range(len(neighbors)):
            for k in range(j + 1, len(neighbors)):
                if adjacency_mask[neighbors[j], neighbors[k]]:
                    triangles += 1
        
        max_triangles = len(neighbors) * (len(neighbors) - 1) // 2
        clustering_coeff[i] = triangles / max_triangles if max_triangles > 0 else 0
    
    # Compute betweenness centrality approximation
    betweenness = np.zeros(n)
    for i in range(min(50, n)):  # Sample nodes for efficiency
        for j in range(i + 1, min(50, n)):
            # Find shortest path between i and j
            path = _dijkstra_shortest_path(i, j, distance_matrix, adjacency_mask)
            if path:
                for node in path[1:-1]:  # Exclude endpoints
                    betweenness[node] += 1
    
    # Normalize centrality measures
    if np.max(clustering_coeff) > 0:
        clustering_coeff = clustering_coeff / np.max(clustering_coeff)
    if np.max(betweenness) > 0:
        betweenness = betweenness / np.max(betweenness)
    
    # Compute edge scores based on structural patterns
    for i in range(n):
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                continue
            
            # Pattern 1: Bridge edges (connect different clusters)
            if clustering_coeff[i] > 0.7 and clustering_coeff[j] > 0.7:
                # Both nodes in dense clusters - likely an intra-cluster edge
                pattern_score = 0.7
            elif betweenness[i] > 0.5 or betweenness[j] > 0.5:
                # High-betweenness nodes - important connector edges
                pattern_score = 1.2
            else:
                # Regular edges
                pattern_score = 1.0
            
            # Adjust based on distance ratio to median
            median_dist = np.median(distance_matrix[adjacency_mask])
            dist_ratio = median_dist / (distance_matrix[i, j] + 1e-10)
            pattern_score *= np.tanh(dist_ratio)
            
            pattern_scores[i, j] = pattern_scores[j, i] = pattern_score
    
    return pattern_scores

def _dijkstra_shortest_path(start: int, end: int,
                           distance_matrix: np.ndarray,
                           adjacency_mask: np.ndarray) -> list:
    """Find shortest path using Dijkstra's algorithm."""
    n = len(distance_matrix)
    dist = np.full(n, float('inf'))
    prev = np.full(n, -1, dtype=int)
    dist[start] = 0
    
    pq = [(0, start)]
    
    while pq:
        current_dist, u = heapq.heappop(pq)
        
        if u == end:
            break
        
        if current_dist > dist[u]:
            continue
        
        for v in range(n):
            if adjacency_mask[u, v]:
                alt = current_dist + distance_matrix[u, v]
                if alt < dist[v]:
                    dist[v] = alt
                    prev[v] = u
                    heapq.heappush(pq, (alt, v))
    
    # Reconstruct path
    if dist[end] == float('inf'):
        return []
    
    path = []
    u = end
    while u != -1:
        path.append(u)
        u = prev[u]
    
    return path[::-1]

def _learn_optimal_weights(distance_matrix: np.ndarray,
                          christofides_edges: np.ndarray,
                          regret_scores: np.ndarray,
                          structural_scores: np.ndarray) -> dict:
    """Learn optimal fusion weights using meta-heuristic search."""
    n = len(distance_matrix)
    
    # Problem characteristics
    avg_distance = np.mean(distance_matrix)
    std_distance = np.std(distance_matrix)
    uniformity = std_distance / (avg_distance + 1e-10)
    
    # Number of edges in each component
    christofides_count = np.sum(christofides_edges > 1.0)
    regret_count = np.sum(regret_scores > 0.5)
    structural_count = np.sum(structural_scores > 0)
    
    # Adaptive weight initialization based on problem characteristics
    if uniformity < 0.2:
        # Very uniform distances - emphasize structural patterns
        base_weights = {'christofides': 0.2, 'regret': 0.3, 'structural': 0.5}
    elif uniformity < 0.5:
        # Moderate variation - balanced approach
        base_weights = {'christofides': 0.3, 'regret': 0.4, 'structural': 0.3}
    else:
        # High variation - emphasize regret analysis
        base_weights = {'christofides': 0.2, 'regret': 0.6, 'structural': 0.2}
    
    # Adjust based on component coverage
    total_edges = n * (n - 1) // 2
    coverage_factor = {
        'christofides': christofides_count / total_edges,
        'regret': regret_count / total_edges,
        'structural': structural_count / total_edges
    }
    
    # Normalize weights considering coverage
    for component in base_weights:
        base_weights[component] *= (0.5 + 0.5 * coverage_factor[component])
    
    # Normalize to sum to 1
    total = sum(base_weights.values())
    for component in base_weights:
        base_weights[component] /= total
    
    return base_weights

def _fuse_with_learned_weights(christofides_edges: np.ndarray,
                              regret_scores: np.ndarray,
                              structural_scores: np.ndarray,
                              weights: dict,
                              adjacency_mask: np.ndarray) -> np.ndarray:
    """Fuse components using learned weights with adaptive normalization."""
    n = len(christofides_edges)
    heuristics = np.zeros_like(christofides_edges, dtype=float)
    
    # Normalize each component
    christofides_norm = _normalize_component(christofides_edges)
    regret_norm = _normalize_component(regret_scores)
    structural_norm = _normalize_component(structural_scores)
    
    # Weighted combination
    heuristics = (weights['christofides'] * christofides_norm +
                  weights['regret'] * regret_norm +
                  weights['structural'] * structural_norm)
    
    # Apply adjacency mask
    heuristics[~adjacency_mask] = 0
    np.fill_diagonal(heuristics, 0)
    
    # Row-wise softmax normalization
    for i in range(n):
        row = heuristics[i]
        if np.any(row > 0):
            # Temperature-based softmax
            temp = 0.3 + 0.7 * np.std(row) / (np.mean(row) + 1e-10)
            exp_row = np.exp(row / (temp + 1e-10))
            heuristics[i] = exp_row / (np.sum(exp_row) + 1e-10)
    
    # Symmetrize
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics

def _normalize_component(scores: np.ndarray) -> np.ndarray:
    """Normalize a score component to [0, 1] range."""
    min_val = np.min(scores)
    max_val = np.max(scores)
    
    if max_val - min_val > 1e-10:
        normalized = (scores - min_val) / (max_val - min_val)
    else:
        normalized = np.ones_like(scores)
    
    return normalized
