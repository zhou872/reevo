import numpy as np
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection prioritizing computational efficiency
    through aggressive sparsification while maintaining edge quality assessment.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Logarithmic Edge Sparsification ====================
    # Use logarithmic scaling to retain promising edges efficiently
    k_edges = max(3, int(np.log2(n)))  # Log-based sparsification
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Build initial candidate edges from nearest neighbors
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argpartition(distances, k_edges)[:k_edges]
        adjacency_mask[i, nn_indices] = True
    
    # Ensure symmetry for undirected graph
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Adaptive Neighbor Normalization ====================
    # Normalize edge priorities by local competition
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            # Adaptive neighbor count for normalization (sqrt of problem size)
            k_norm = max(2, int(np.sqrt(n)))
            
            # Get distances to neighbors
            valid_dists = distance_matrix[i, valid_mask]
            
            # Compute rank-based weights emphasizing closer neighbors
            ranks = np.argsort(np.argsort(valid_dists)) + 1  # Closest gets rank 1
            weights = 1.0 / ranks  # Higher weight for closer ranks
            weights = weights / np.sum(weights)  # Normalize
            
            # Apply weighted inverse distance with adaptive neighbor normalization
            inv_vals = 1.0 / (valid_dists + 1e-10)
            inv_dist[i, valid_mask] = (inv_vals * weights) / k_norm
    
    # Symmetrize by taking maximum of both directions
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    
    # ==================== 3. Two-hop Alternative Penalty ====================
    # Penalize edges that have good two-hop alternatives
    two_hop_penalty = np.ones_like(distance_matrix)
    
    # Pre-compute sorted distances for each node for efficient lookups
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        valid_indices = np.where(valid_mask)[0]
        
        for j in valid_indices:
            if i >= j:
                continue
                
            # Consider top neighbors from both endpoints
            check_limit = max(3, int(np.sqrt(n)))
            i_neighbors = sorted_indices[i, 1:check_limit+1]
            j_neighbors = sorted_indices[j, 1:check_limit+1]
            
            # Candidate intermediate nodes from both neighborhoods
            candidate_nodes = np.unique(np.concatenate([i_neighbors, j_neighbors]))
            candidate_nodes = candidate_nodes[candidate_nodes != i]
            candidate_nodes = candidate_nodes[candidate_nodes != j]
            
            if len(candidate_nodes) > 0:
                # Vectorized two-hop distance computation
                two_hop_dists = distance_matrix[i, candidate_nodes] + distance_matrix[candidate_nodes, j]
                min_two_hop = np.min(two_hop_dists)
                direct_dist = distance_matrix[i, j]
                
                if direct_dist > 1e-10 and min_two_hop < np.inf:
                    # Compute penalty based on quality of two-hop alternative
                    ratio = min_two_hop / direct_dist
                    
                    # Strong penalty if two-hop is comparable or better
                    if ratio <= 1.0:
                        # Two-hop path is better - heavily penalize direct edge
                        penalty = 0.2 + 0.3 * ratio  # 0.2 to 0.5 when ratio <= 1
                    else:
                        # Two-hop is worse, but if ratio < 1.5, still moderate penalty
                        penalty = 0.5 + 0.5 * np.clip(2.0 - ratio, 0.0, 1.0)  # 0.5 to 1.0
                    
                    two_hop_penalty[i, j] = two_hop_penalty[j, i] = penalty
    
    # ==================== 4. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Base score from normalized inverse distance
                score = inv_dist[i, j]
                
                # Apply two-hop penalty
                score *= two_hop_penalty[i, j]
                
                heuristics[i, j] = heuristics[j, i] = score
    
    # ==================== 5. Final Processing ====================
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Add small constant to avoid zeros in selected edges
    heuristics[adjacency_mask] += 1e-8
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Symmetrize final matrix (average of both directions)
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # Final sparsification: zero out entries below threshold
    threshold = np.percentile(heuristics[heuristics > 0], 30) if np.any(heuristics > 0) else 0
    heuristics[heuristics < threshold] = 0
    
    return heuristics
