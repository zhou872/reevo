import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with logarithmic adaptive sparsification
    and adaptive neighbor normalization.
    
    Args:
        distance_matrix: Square matrix of pairwise distances (n x n)
    
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Logarithmic Adaptive Sparsification ====================
    # Determine k adaptively using logarithmic scaling
    k = max(3, int(np.log2(n)))
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argpartition(distances, k)[:k]
        adjacency_mask[i, nn_indices] = True
    
    # Ensure symmetry and remove self-loops
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Adaptive Neighbor Normalization ====================
    # Compute adaptive k for normalization (using sqrt(n) as suggested)
    k_norm = max(2, int(np.sqrt(n)))
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        row_dists = distance_matrix[i].copy()
        row_dists[i] = np.inf
        valid_mask = adjacency_mask[i]
        
        if np.any(valid_mask):
            valid_dists = row_dists[valid_mask]
            if len(valid_dists) > 0:
                # Inverse distance with adaptive normalization
                inv_vals = 1.0 / (valid_dists + 1e-10)
                # Normalize by reciprocal of number of neighbors
                inv_vals = inv_vals / k_norm
                inv_dist[i, valid_mask] = inv_vals
    
    # Symmetrize inverse distance scores
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    
    # ==================== 3. Minimum Spanning Tree Priority ====================
    mst_mask = np.zeros((n, n), dtype=bool)
    
    # Prim's algorithm for MST
    visited = np.zeros(n, dtype=bool)
    min_edge = np.full(n, np.inf)
    parent = np.full(n, -1, dtype=int)
    min_edge[0] = 0
    
    for _ in range(n):
        u = -1
        min_val = np.inf
        for i in range(n):
            if not visited[i] and min_edge[i] < min_val:
                min_val = min_edge[i]
                u = i
        
        if u == -1:
            break
            
        visited[u] = True
        
        if parent[u] != -1:
            mst_mask[u, parent[u]] = mst_mask[parent[u], u] = True
        
        for v in range(n):
            if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                min_edge[v] = distance_matrix[u, v]
                parent[v] = u
    
    # ==================== 4. Two-Hop Gap Analysis ====================
    two_hop_gap = np.zeros_like(distance_matrix)
    
    # Precompute sorted neighbors for efficiency
    sorted_neighbors = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_idx = np.argsort(dists)
        sorted_neighbors.append(sorted_idx)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Find minimum two-hop path through any intermediate node
                min_two_hop = np.inf
                direct_dist = distance_matrix[i, j]
                
                # Check top k_norm neighbors of both nodes
                candidates = set(sorted_neighbors[i][:k_norm]) | set(sorted_neighbors[j][:k_norm])
                candidates.discard(i)
                candidates.discard(j)
                
                for k_node in candidates:
                    two_hop_dist = distance_matrix[i, k_node] + distance_matrix[k_node, j]
                    if two_hop_dist < min_two_hop:
                        min_two_hop = two_hop_dist
                
                if min_two_hop < np.inf and direct_dist > 1e-10:
                    # Gap ratio: how much better is direct edge than best detour
                    gap_ratio = (min_two_hop - direct_dist) / direct_dist
                    two_hop_gap[i, j] = two_hop_gap[j, i] = max(0, gap_ratio)
    
    # Normalize two-hop gaps
    max_gap = np.max(two_hop_gap)
    if max_gap > 0:
        two_hop_gap = two_hop_gap / max_gap
    
    # ==================== 5. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                score = inv_dist[i, j]
                
                # MST bonus (edges in MST are prioritized)
                if mst_mask[i, j]:
                    score *= 2.0
                
                # Two-hop gap bonus (edges with large direct advantage get bonus)
                if two_hop_gap[i, j] > 0:
                    score *= (1.0 + two_hop_gap[i, j])
                
                # Degree penalty (avoid edges that are the only connection for a node)
                degree_i = np.sum(adjacency_mask[i])
                degree_j = np.sum(adjacency_mask[j])
                if degree_i <= 3 or degree_j <= 3:
                    score *= 0.8  # Penalize edges to low-degree nodes
                
                heuristics[i, j] = heuristics[j, i] = score
    
    # ==================== 6. Final Processing ====================
    np.fill_diagonal(heuristics, 0)
    
    # Ensure all promising edges have non-zero values
    heuristics[adjacency_mask] = np.maximum(heuristics[adjacency_mask], 1e-8)
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Symmetrize final matrix
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
