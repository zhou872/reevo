import numpy as np

def heuristics_v2(distance_matrix: np.ndarray, 
                  depth: int = 4,
                  use_embedding: bool = True,
                  pattern_weight: float = 0.15) -> np.ndarray:
    """
    Advanced TSP edge prioritization heuristic with deep multi-hop analysis,
    learned feature embeddings, reinforcement-inspired weight adaptation,
    and structural pattern matching.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        depth: Maximum depth for multi-hop critical path analysis (2-5)
        use_embedding: Use neural-inspired feature embeddings
        pattern_weight: Weight for structural pattern matching component
        
    Returns:
        Heuristic matrix with higher values for promising edges
    """
    n = distance_matrix.shape[0]
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Enhanced Adjacency Construction ====================
    adjacency_mask, neighbor_lists = _build_enhanced_adjacency(distance_matrix, n)
    
    # ==================== 2. Deep Multi-Hop Critical Path Analysis ====================
    criticality_scores = _deep_multi_hop_analysis(
        distance_matrix, adjacency_mask, neighbor_lists, n, depth
    )
    
    # ==================== 3. Learnable Feature Embeddings ====================
    if use_embedding:
        edge_features = _compute_learnable_embeddings(
            distance_matrix, adjacency_mask, neighbor_lists, n
        )
    else:
        edge_features = _compute_basic_features(distance_matrix, adjacency_mask, n)
    
    # ==================== 4. Structural Pattern Matching ====================
    pattern_scores = _structural_pattern_matching(distance_matrix, adjacency_mask, n)
    
    # ==================== 5. Reinforcement-Inspired Weight Adaptation ====================
    weights = _reinforcement_weight_adaptation(distance_matrix, n)
    
    # ==================== 6. Integrated Edge Scoring ====================
    heuristics = _integrated_edge_scoring(
        distance_matrix, adjacency_mask, criticality_scores,

    )
    
    # ==================== 7. Adaptive Normalization ====================
    heuristics = _adaptive_normalization(heuristics, distance_matrix, n)
    
    return heuristics

def _build_enhanced_adjacency(distance_matrix: np.ndarray, n: int):
    """Build adaptive adjacency structure with density-aware connectivity."""
    adjacency_mask = np.zeros((n, n), dtype=bool)
    neighbor_lists = [[] for _ in range(n)]
    
    # Adaptive k based on instance characteristics
    base_k = max(3, int(np.sqrt(n)))
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Compute local statistics for adaptive thresholding
        sorted_dists = np.sort(distances)[:min(10, n-1)]
        mean_dist = np.mean(sorted_dists)
        std_dist = np.std(sorted_dists)
        
        # Dynamic threshold based on local statistics
        if std_dist > mean_dist * 0.3:
            threshold = mean_dist + 0.5 * std_dist
        else:
            threshold = mean_dist * 1.2
        
        # Select neighbors based on adaptive criteria
        near_neighbors = np.where(distances <= threshold)[0]
        k = min(len(near_neighbors), base_k)
        
        if len(near_neighbors) > k:
            # Get k closest within threshold
            near_distances = distances[near_neighbors]
            closest_idx = np.argsort(near_distances)[:k]
            neighbors = near_neighbors[closest_idx]
        else:
            neighbors = near_neighbors
        
        # Update adjacency structures
        adjacency_mask[i, neighbors] = True
        neighbor_lists[i] = list(neighbors)
    
    # Ensure symmetry and connectivity
    adjacency_mask = adjacency_mask | adjacency_mask.T
    
    # Ensure each node has at least 2 connections
    for i in range(n):
        connections = np.where(adjacency_mask[i])[0]
        if len(connections) < 2:
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            candidates = np.argsort(distances)[:max(2, base_k//2)]
            for j in candidates:
                if j != i and not adjacency_mask[i, j]:
                    adjacency_mask[i, j] = True
                    adjacency_mask[j, i] = True
                    if j not in neighbor_lists[i]:
                        neighbor_lists[i].append(j)
                    if i not in neighbor_lists[j]:
                        neighbor_lists[j].append(i)
    
    return adjacency_mask, neighbor_lists

def _deep_multi_hop_analysis(distance_matrix, adjacency_mask, neighbor_lists, n, depth):
    """Perform deep multi-hop analysis to identify critical edges."""
    criticality_scores = np.zeros((n, n))
    
    # Precompute shortest paths for efficiency
    shortest_paths = _precompute_shortest_paths(distance_matrix, adjacency_mask, n, depth)
    
    for i in range(n):
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                continue
            
            direct_dist = distance_matrix[i, j]
            if direct_dist < 1e-10:
                continue
            
            # Find best alternative paths for each depth
            alternative_ratios = []
            
            for d in range(2, depth + 1):
                if d in shortest_paths[i][j]:
                    alternative_dist = shortest_paths[i][j][d]
                    if alternative_dist < np.inf:
                        ratio = alternative_dist / direct_dist
                        alternative_ratios.append(ratio)
            
            # Compute criticality based on alternative path quality
            if alternative_ratios:
                best_ratio = min(alternative_ratios)
                if best_ratio > 1.0:
                    # Edge is critical (no good alternative)
                    criticality = min(3.0, best_ratio - 1.0)
                else:
                    # Edge has good alternatives
                    criticality = max(0.1, 1.0 - (1.0 / best_ratio))
            else:
                # No alternative path found
                criticality = 2.0
            
            criticality_scores[i, j] = criticality_scores[j, i] = criticality
    
    # Normalize to [0, 1]
    max_score = criticality_scores.max()
    if max_score > 0:
        criticality_scores = criticality_scores / max_score
    
    return criticality_scores

def _precompute_shortest_paths(distance_matrix, adjacency_mask, n, max_depth):
    """Precompute shortest paths up to max_depth for efficiency."""
    # Initialize data structure
    shortest_paths = [defaultdict(lambda: defaultdict(lambda: float('inf'))) 
                      for _ in range(n)]
    
    # Initialize with direct neighbors
    for i in range(n):
        for j in range(n):
            if adjacency_mask[i, j]:
                shortest_paths[i][j][1] = distance_matrix[i, j]
    
    # BFS-like propagation for increasing path lengths
    for d in range(2, max_depth + 1):
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                best_dist = float('inf')
                # Try all intermediate nodes for paths of length d
                for k in range(n):
                    if k == i or k == j:
                        continue
                    
                    # Check if there's a path from i to k of length d-1
                    if d-1 in shortest_paths[i][k]:
                        # Check if k is connected to j
                        if adjacency_mask[k, j]:
                            candidate = shortest_paths[i][k][d-1] + distance_matrix[k, j]
                            if candidate < best_dist:
                                best_dist = candidate
                
                if best_dist < float('inf'):
                    shortest_paths[i][j][d] = best_dist
    
    return shortest_paths

def _compute_learnable_embeddings(distance_matrix, adjacency_mask, neighbor_lists, n):
    """Compute neural-inspired feature embeddings for edges."""
    # Compute node-level features
    node_features = _compute_node_features(distance_matrix, adjacency_mask, n)
    
    # Initialize edge feature matrix
    edge_embeddings = np.zeros((n, n, 6))  # 6-dimensional embedding
    
    for i in range(n):
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                continue
            
            # Feature 1: Normalized inverse distance
            if distance_matrix[i, j] > 0:
                inv_dist = 1.0 / distance_matrix[i, j]
            else:
                inv_dist = 0
            
            # Feature 2: Rank among neighbors
            i_neighbors = neighbor_lists[i]
            if len(i_neighbors) > 1:
                i_distances = [distance_matrix[i, k] for k in i_neighbors]
                i_rank = np.argsort(np.argsort(i_distances))[i_neighbors.index(j)] / len(i_neighbors)
            else:
                i_rank = 0.5
            
            # Feature 3: Common neighbors ratio
            common = len(set(neighbor_lists[i]) & set(neighbor_lists[j]))
            union = len(set(neighbor_lists[i]) | set(neighbor_lists[j]))
            if union > 0:
                jaccard = common / union
            else:
                jaccard = 0
            
            # Feature 4: Triangle potential
            triangle_potential = 0
            for k in set(neighbor_lists[i]) & set(neighbor_lists[j]):
                if k != i and k != j:
                    triangle_potential += 1.0 / (distance_matrix[i, k] + distance_matrix[j, k] + 1e-10)
            
            # Feature 5: Local density product
            density_product = node_features['local_density'][i] * node_features['local_density'][j]
            
            # Feature 6: Structural hole measure
            if len(neighbor_lists[i]) > 0 and len(neighbor_lists[j]) > 0:
                structural_hole = 1.0 - (common / min(len(neighbor_lists[i]), len(neighbor_lists[j])))
            else:
                structural_hole = 0
            
            # Store features
            edge_embeddings[i, j] = edge_embeddings[j, i] = [
                inv_dist, i_rank, jaccard, triangle_potential, density_product, structural_hole
            ]
    
    # Normalize each feature dimension
    for d in range(6):
        feature_values = edge_embeddings[:, :, d]
        max_val = feature_values.max()
        if max_val > 0:
            edge_embeddings[:, :, d] = feature_values / max_val
    
    return edge_embeddings

def _compute_node_features(distance_matrix, adjacency_mask, n):
    """Compute node-level features for embedding construction."""
    features = {}
    
    # Local density
    local_density = np.zeros(n)
    for i in range(n):
        neighbors = np.where(adjacency_mask[i])[0]
        if len(neighbors) > 0:
            avg_dist = np.mean([distance_matrix[i, j] for j in neighbors])
            if avg_dist > 0:
                local_density[i] = 1.0 / avg_dist
    
    # Clustering coefficient
    clustering = np.zeros(n)
    for i in range(n):
        neighbors = list(np.where(adjacency_mask[i])[0])
        if len(neighbors) >= 2:
            triangles = 0
            for a in range(len(neighbors)):
                for b in range(a + 1, len(neighbors)):
                    if adjacency_mask[neighbors[a], neighbors[b]]:
                        triangles += 1
            possible = len(neighbors) * (len(neighbors) - 1) / 2
            clustering[i] = triangles / possible if possible > 0 else 0
    
    # Degree centrality
    degrees = adjacency_mask.sum(axis=1)
    if degrees.max() > 0:
        degree_centrality = degrees / degrees.max()
    else:
        degree_centrality = np.zeros(n)
    
    features['local_density'] = local_density / (local_density.max() + 1e-10)
    features['clustering'] = clustering
    features['degree_centrality'] = degree_centrality
    
    return features

def _compute_basic_features(distance_matrix, adjacency_mask, n):
    """Fallback to basic features if embedding is disabled."""
    edge_embeddings = np.zeros((n, n, 3))
    
    for i in range(n):
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                continue
            
            # Simple features
            if distance_matrix[i, j] > 0:
                inv_dist = 1.0 / distance_matrix[i, j]
            else:
                inv_dist = 0
            
            # Degree product
            deg_i = adjacency_mask[i].sum()
            deg_j = adjacency_mask[j].sum()
            deg_product = deg_i * deg_j
            
            # Symmetry measure
            common_neighbors = np.sum(adjacency_mask[i] & adjacency_mask[j])
            
            edge_embeddings[i, j] = edge_embeddings[j, i] = [inv_dist, deg_product, common_neighbors]
    
    # Normalize
    for d in range(3):
        feature_values = edge_embeddings[:, :, d]
        max_val = feature_values.max()
        if max_val > 0:
            edge_embeddings[:, :, d] = feature_values / max_val
    
    return edge_embeddings

def _structural_pattern_matching(distance_matrix, adjacency_mask, n):
    """Identify edges that match optimal TSP structural patterns."""
    pattern_scores = np.zeros((n, n))
    
    # Pattern 1: MST edges
    mst_edges = _compute_mst_edges(distance_matrix, n)
    
    # Pattern 2: High-degree connecting edges
    degrees = adjacency_mask.sum(axis=1)
    median_degree = np.median(degrees)
    
    # Pattern 3: Bridge edges between clusters
    bridge_scores = _detect_bridge_edges(distance_matrix, adjacency_mask, n)
    
    for i in range(n):
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                continue
            
            score = 0.0
            
            # MST pattern bonus
            if mst_edges[i, j]:
                score += 0.4
            
            # High-degree connection bonus
            if degrees[i] > median_degree and degrees[j] > median_degree:
                score += 0.3
            
            # Bridge edge bonus
            score += 0.3 * bridge_scores[i, j]
            
            pattern_scores[i, j] = pattern_scores[j, i] = score
    
    # Normalize
    max_score = pattern_scores.max()
    if max_score > 0:
        pattern_scores = pattern_scores / max_score
    
    return pattern_scores

def _compute_mst_edges(distance_matrix, n):
    """Compute Minimum Spanning Tree edges using Prim's algorithm."""
    mst_mask = np.zeros((n, n), dtype=bool)
    
    visited = np.zeros(n, dtype=bool)
    min_edge = np.full(n, np.inf)
    parent = np.full(n, -1, dtype=int)
    
    min_edge[0] = 0
    
    for _ in range(n):
        # Find minimum edge from visited to unvisited
        u = -1
        min_val = np.inf
        for v in range(n):
            if not visited[v] and min_edge[v] < min_val:
                min_val = min_edge[v]
                u = v
        
        if u == -1:
            break
        
        visited[u] = True
        if parent[u] != -1:
            mst_mask[u, parent[u]] = True
            mst_mask[parent[u], u] = True
        
        # Update minimum edges
        for v in range(n):
            if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                min_edge[v] = distance_matrix[u, v]
                parent[v] = u
    
    return mst_mask

def _detect_bridge_edges(distance_matrix, adjacency_mask, n):
    """Detect edges that connect different clusters."""
    bridge_scores = np.zeros((n, n))
    
    # Simple approach: edges that connect nodes with different neighbor sets
    for i in range(n):
        neighbors_i = set(np.where(adjacency_mask[i])[0])
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                continue
            
            neighbors_j = set(np.where(adjacency_mask[j])[0])
            
            # Jaccard distance between neighbor sets
            intersection = len(neighbors_i & neighbors_j)
            union = len(neighbors_i | neighbors_j)
            
            if union > 0:
                jaccard_dist = 1.0 - (intersection / union)
            else:
                jaccard_dist = 1.0
            
            # Distance ratio to nearest common neighbor
            common_neighbors = list(neighbors_i & neighbors_j)
            if common_neighbors:
                min_common_dist = min(distance_matrix[i, k] + distance_matrix[j, k] 
                                     for k in common_neighbors)
                direct_dist = distance_matrix[i, j]
                if direct_dist > 0:
                    ratio = min_common_dist / direct_dist
                    bridge_measure = min(1.0, max(0.0, (ratio - 1.0) / 2.0))
                else:
                    bridge_measure = 0
            else:
                bridge_measure = 1.0
            
            # Combine measures
            bridge_scores[i, j] = bridge_scores[j, i] = 0.6 * jaccard_dist + 0.4 * bridge_measure
    
    return bridge_scores

def _reinforcement_weight_adaptation(distance_matrix, n):
    """Dynamically adapt component weights based on instance characteristics."""
    weights = {}
    
    # Analyze instance characteristics
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    mean_dist = np.mean(upper_tri)
    std_dist = np.std(upper_tri)
    cv_dist = std_dist / (mean_dist + 1e-10)
    
    # Size-based adjustments
    size_factor = min(1.0, 50.0 / n)
    
    # Variation-based adjustments
    if cv_dist > 1.5:
        # High variation: emphasize criticality
        variation_weights = {'criticality': 0.4, 'embedding': 0.3, 'distance': 0.2, 'pattern': 0.1}
    elif cv_dist < 0.3:
        # Low variation: emphasize patterns and embeddings
        variation_weights = {'criticality': 0.2, 'embedding': 0.4, 'distance': 0.2, 'pattern': 0.2}
    else:
        # Moderate variation: balanced weights
        variation_weights = {'criticality': 0.3, 'embedding': 0.3, 'distance': 0.25, 'pattern': 0.15}
    
    # Apply size factor
    for key in variation_weights:
        weights[key] = variation_weights[key] * (0.8 + 0.2 * size_factor)
    
    # Ensure sum = 1
    total = sum(weights.values())
    for key in weights:
        weights[key] = weights[key] / total
    
    return weights

def _integrated_edge_scoring(distance_matrix, adjacency_mask, criticality_scores,
                            edge_embeddings, pattern_scores, weights, pattern_weight, n):
    """Integrate all components into final edge scores."""
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Precompute MST for additional bonuses
    mst_edges = _compute_mst_edges(distance_matrix, n)
    mst_degrees = mst_edges.sum(axis=1)
    
    for i in range(n):
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                continue
            
            # 1. Distance-based component
            if distance_matrix[i, j] > 0:
                dist_score = 1.0 / distance_matrix[i, j]
            else:
                dist_score = 0
            
            # 2. Criticality component
            crit_score = criticality_scores[i, j]
            
            # 3. Embedding component (non-linear combination)
            if edge_embeddings.shape[2] > 0:
                embedding_vec = edge_embeddings[i, j]
                # Neural-inspired combination
                embed_score = (embedding_vec[0] * 0.4 +  # Inverse distance
                              embedding_vec[1] * 0.2 +  # Rank
                              embedding_vec[2] * 0.1 +  # Jaccard
                              embedding_vec[3] * 0.1 +  # Triangle potential
                              embedding_vec[4] * 0.1 +  # Density product
                              embedding_vec[5] * 0.1)   # Structural hole
            else:
                embed_score = 0
            
            # 4. Pattern component
            pat_score = pattern_scores[i, j]
            
            # 5. MST-based bonuses
            mst_bonus = 1.0
            if mst_edges[i, j]:
                mst_bonus *= 1.2
            
            # Christofides-inspired bonus for odd-degree MST nodes
            if mst_degrees[i] % 2 == 1 and mst_degrees[j] % 2 == 1:
                mst_bonus *= 1.15
            
            # Combine components with learned weights
            combined_score = (weights['distance'] * dist_score +
                             weights['criticality'] * crit_score +
                             weights['embedding'] * embed_score +
                             pattern_weight * pat_score) * mst_bonus
            
            heuristics[i, j] = heuristics[j, i] = combined_score
    
    return heuristics

def _adaptive_normalization(heuristics, distance_matrix, n):
    """Apply adaptive normalization based on instance characteristics."""
    # Zero out diagonal
    np.fill_diagonal(heuristics, 0)
    
    # Symmetrize
    heuristics = 0.5 * (heuristics + heuristics.T)
    
    # Row-wise softmax for probabilistic interpretation
    for i in range(n):
        row = heuristics[i]
        if row.sum() > 0:
            # Apply softmax with temperature
            max_val = row.max()
            if max_val > 0:
                # Temperature based on row variance
                row_std = np.std(row[row > 0])
                if row_std > 0:
                    temperature = max(0.1, min(2.0, 1.0 / (row_std + 0.1)))
                else:
                    temperature = 1.0
                
                exp_row = np.exp((row - max_val) / temperature)
                heuristics[i] = exp_row / exp_row.sum()
    
    # Ensure symmetry is preserved
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
