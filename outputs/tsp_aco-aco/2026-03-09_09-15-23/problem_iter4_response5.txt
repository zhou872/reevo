import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.cluster.hierarchy import linkage, fcluster
from scipy.spatial.distance import squareform
from typing import Tuple, Optional

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge prioritization with three-step lookahead evaluation,
    meta-heuristic parameter adaptation, and exponential priority amplification.
    
    Features:
    1. Three-step lookahead path evaluation for deeper future state assessment
    2. Meta-heuristic parameter adaptation based on problem characteristics
    3. Exponential transformation to amplify differences between high/low priority edges
    4. Connectivity-preserving sparsification with adaptive thresholding
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes.
        
    Returns:
        Priority matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        heuristics = np.ones_like(distance_matrix) / (distance_matrix + 1e-12)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # Meta-heuristic parameter adaptation based on problem characteristics
    params = _adapt_parameters(distance_matrix)
    
    # 1. Base inverse distance with exponential amplification
    inv_dist = _compute_exponential_inverse_distance(distance_matrix, params['inv_dist_exp'])
    
    # 2. Minimum Spanning Tree structural bonus
    mst_bonus = _compute_mst_structural_bonus(distance_matrix, params['mst_weight'])
    
    # 3. Enhanced regret factor with node importance weighting
    regret_factor = _compute_enhanced_regret_factor(distance_matrix, params['regret_power'])
    
    # 4. Three-step lookahead path evaluation
    lookahead_score = _compute_three_step_lookahead(distance_matrix, params['lookahead_depth'])
    
    # 5. Adaptive cluster density reward
    density_bonus = _compute_adaptive_density_bonus(distance_matrix, params['density_sensitivity'])
    
    # 6. Combine components with exponential multiplicative integration
    priorities = inv_dist * mst_bonus * regret_factor * lookahead_score * density_bonus
    
    # 7. Priority amplification through exponential transformation
    priorities = _exponential_amplification(priorities, params['amplification_factor'])
    
    # 8. Connectivity-preserving adaptive sparsification
    priorities = _connectivity_preserving_sparsify(priorities, distance_matrix, params['sparsity_factor'])
    
    # Ensure symmetry and zero diagonal
    priorities = (priorities + priorities.T) / 2
    np.fill_diagonal(priorities, 0)
    
    return priorities


def _adapt_parameters(distance_matrix: np.ndarray) -> dict:
    """Adapt heuristic parameters based on problem characteristics."""
    n = distance_matrix.shape[0]
    
    # Analyze distance distribution
    distances = distance_matrix[np.triu_indices_from(distance_matrix, k=1)]
    dist_mean = np.mean(distances)
    dist_std = np.std(distances)
    dist_variation = dist_std / (dist_mean + 1e-12)
    
    # Base parameters adjusted by problem size and variation
    base_params = {
        'inv_dist_exp': 1.2 + 0.3 * min(1.0, dist_variation),
        'mst_weight': 1.5 + 0.5 * np.log1p(n) / n,
        'regret_power': 1.3 + 0.2 * min(1.0, n / 50),
        'lookahead_depth': min(3, max(2, int(np.log2(n)))),
        'density_sensitivity': 0.8 + 0.4 * min(1.0, dist_variation),
        'amplification_factor': 1.5 + 0.5 * min(1.0, n / 100),
        'sparsity_factor': max(0.05, min(0.3, 15.0 / n))
    }
    
    return base_params


def _compute_exponential_inverse_distance(dist_matrix: np.ndarray, exponent: float) -> np.ndarray:
    """Compute exponentially amplified inverse distance."""
    n = dist_matrix.shape[0]
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / dist_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    inv_dist[np.eye(n, dtype=bool)] = 0
    
    # Apply exponential amplification to enhance contrast
    inv_dist_max = np.max(inv_dist)
    if inv_dist_max > 0:
        inv_dist = (inv_dist / inv_dist_max) ** exponent
    
    return inv_dist


def _compute_mst_structural_bonus(dist_matrix: np.ndarray, weight: float) -> np.ndarray:
    """Compute structural bonus based on Minimum Spanning Tree."""
    n = dist_matrix.shape[0]
    bonus = np.ones_like(dist_matrix)
    
    if n <= 1:
        return bonus
    
    # Compute MST
    mst = minimum_spanning_tree(dist_matrix)
    mst_mask = mst.toarray() > 0
    
    # Add bonus for MST edges
    bonus += weight * mst_mask
    
    # Add reduced bonus for edges connecting to MST nodes
    for i in range(n):
        mst_neighbors = np.where(mst_mask[i])[0]
        if len(mst_neighbors) > 0:
            # Add connections to k-nearest neighbors of MST nodes
            k = max(2, int(np.sqrt(n)))
            for neighbor in mst_neighbors:
                row = dist_matrix[neighbor].copy()
                row[neighbor] = np.inf
                nearest_indices = np.argsort(row)[:k]
                for idx in nearest_indices:
                    if idx != neighbor:
                        bonus[neighbor, idx] += weight * 0.5
                        bonus[idx, neighbor] += weight * 0.5
    
    return bonus


def _compute_enhanced_regret_factor(dist_matrix: np.ndarray, power: float) -> np.ndarray:
    """Compute enhanced regret factor with node importance weighting."""
    n = dist_matrix.shape[0]
    if n < 3:
        return np.ones_like(dist_matrix)
    
    # Compute node regrets (difference between second best and best neighbor)
    node_regret = np.zeros(n)
    node_importance = np.zeros(n)
    
    for i in range(n):
        sorted_dists = np.sort(dist_matrix[i])[1:3]  # Best and second best
        if len(sorted_dists) >= 2:
            best, second = sorted_dists[0], sorted_dists[1]
            node_regret[i] = (second - best) / (best + 1e-12)
        
        # Node importance based on centrality (inverse of average distance)
        node_importance[i] = 1.0 / (np.mean(dist_matrix[i]) + 1e-12)
    
    # Normalize
    if node_regret.max() > 0:
        node_regret = node_regret / node_regret.max()
    if node_importance.max() > 0:
        node_importance = node_importance / node_importance.max()
    
    # Combine regrets of endpoints with importance weighting
    regret_factor = np.ones_like(dist_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if dist_matrix[i, j] > 0:
                # Weighted geometric mean of endpoint regrets
                w_i = 0.5 * (1 + node_importance[i])
                w_j = 0.5 * (1 + node_importance[j])
                
                if node_regret[i] > 0 and node_regret[j] > 0:
                    regret_val = (node_regret[i] ** w_i) * (node_regret[j] ** w_j)
                else:
                    regret_val = 0
                
                # Apply power transformation
                regret_factor[i, j] = 1.0 + 0.8 * (regret_val ** power)
                regret_factor[j, i] = regret_factor[i, j]
    
    return regret_factor


def _compute_three_step_lookahead(dist_matrix: np.ndarray, depth: int) -> np.ndarray:
    """Evaluate edges using three-step lookahead path assessment."""
    n = dist_matrix.shape[0]
    lookahead_score = np.ones_like(dist_matrix)
    
    if n < 4:
        return lookahead_score
    
    # Precompute sorted neighbors for efficiency
    sorted_neighbors = []
    for i in range(n):
        indices = np.argsort(dist_matrix[i])[1:depth+2]  # Exclude self
        sorted_neighbors.append(indices)
    
    for i in range(n):
        # Get distances from node i
        i_dists = dist_matrix[i].copy()
        i_dists[i] = np.inf
        
        for j in range(n):
            if i == j:
                continue
            
            # Three-step path evaluation: i -> j -> k -> l
            current_cost = dist_matrix[i, j]
            best_continuation = np.inf
            
            # Consider k as the next step after j
            for k in sorted_neighbors[j][:depth]:
                if k == i:
                    continue
                
                k_dists = dist_matrix[k].copy()
                k_dists[i] = k_dists[j] = k_dists[k] = np.inf
                
                if np.any(k_dists < np.inf):
                    best_l = np.min(k_dists)
                    continuation_cost = dist_matrix[j, k] + best_l
                    best_continuation = min(best_continuation, continuation_cost)
            
            if best_continuation < np.inf:
                current_three_step = current_cost + best_continuation
                
                # Find best alternative three-step path from i (not using j)
                best_alternative = np.inf
                for k2 in sorted_neighbors[i][:depth]:
                    if k2 == j:
                        continue
                    
                    k2_dists = dist_matrix[k2].copy()
                    k2_dists[i] = k2_dists[j] = k2_dists[k2] = np.inf
                    
                    if np.any(k2_dists < np.inf):
                        best_l2 = np.min(k2_dists)
                        alternative_cost = dist_matrix[i, k2] + dist_matrix[k2, best_l2]
                        best_alternative = min(best_alternative, alternative_cost)
                
                if best_alternative < np.inf:
                    # Score based on ratio, favoring current path when it's better
                    ratio = best_alternative / (current_three_step + 1e-12)
                    # Sigmoid transformation to get smooth scoring
                    lookahead_score[i, j] = 1.0 / (1.0 + np.exp(-2.0 * (ratio - 1.0)))
    
    # Symmetrize
    lookahead_score = (lookahead_score + lookahead_score.T) / 2
    return lookahead_score


def _compute_adaptive_density_bonus(dist_matrix: np.ndarray, sensitivity: float) -> np.ndarray:
    """Compute adaptive cluster density bonus."""
    n = dist_matrix.shape[0]
    if n <= 3:
        return np.ones_like(dist_matrix)
    
    # Compute local density for each node
    densities = np.zeros(n)
    for i in range(n):
        # Consider k nearest neighbors for density estimation
        k = min(5, n-1)
        neighbor_dists = np.sort(dist_matrix[i])[1:k+1]
        densities[i] = 1.0 / (np.mean(neighbor_dists) + 1e-12)
    
    # Normalize densities
    if densities.max() > 0:
        densities = densities / densities.max()
    
    # Apply hierarchical clustering to identify clusters
    if n >= 10:
        try:
            condensed_dist = squareform(dist_matrix, checks=False)
            Z = linkage(condensed_dist, method='average')
            
            # Determine optimal number of clusters
            n_clusters = max(2, min(5, n // 10))
            clusters = fcluster(Z, n_clusters, criterion='maxclust')
            
            # Intra-cluster bonus: edges within same cluster get density bonus
            intra_cluster_bonus = np.zeros_like(dist_matrix)
            for i in range(n):
                for j in range(i+1, n):
                    if clusters[i] == clusters[j]:
                        # Bonus proportional to geometric mean of node densities
                        density_factor = np.sqrt(densities[i] * densities[j])
                        intra_cluster_bonus[i, j] = sensitivity * density_factor
                        intra_cluster_bonus[j, i] = intra_cluster_bonus[i, j]
            
            # Inter-cluster penalty: edges between different clusters get reduced bonus
            inter_cluster_bonus = 1.0 - 0.3 * sensitivity
            bonus = np.where(intra_cluster_bonus > 0, 
                            1.0 + intra_cluster_bonus, 
                            inter_cluster_bonus)
            
            return bonus
        except:
            pass
    
    # Fallback: simple density-based bonus
    bonus = np.ones_like(dist_matrix)
    for i in range(n):
        for j in range(i+1, n):
            density_factor = np.sqrt(densities[i] * densities[j])
            bonus[i, j] = 1.0 + sensitivity * density_factor / (dist_matrix[i, j] + 1e-12)
            bonus[j, i] = bonus[i, j]
    
    if bonus.max() > 0:
        bonus = bonus / bonus.max()
    
    return bonus


def _exponential_amplification(priorities: np.ndarray, factor: float) -> np.ndarray:
    """Apply exponential transformation to amplify priority differences."""
    result = priorities.copy()
    
    # Shift to positive range for exponential
    min_val = np.min(result)
    if min_val < 0:
        result = result - min_val + 1e-12
    
    # Apply exponential amplification
    result = result ** factor
    
    # Normalize
    max_val = np.max(result)
    if max_val > 0:
        result = result / max_val
    
    return result


def _connectivity_preserving_sparsify(priorities: np.ndarray, dist_matrix: np.ndarray, 
                                      sparsity_factor: float) -> np.ndarray:
    """Adaptive sparsification that preserves graph connectivity."""
    n = priorities.shape[0]
    result = priorities.copy()
    
    if n <= 10:
        # Keep more edges for small graphs
        threshold = np.percentile(result[result > 0], 30) if np.any(result > 0) else 0
    else:
        # Adaptive threshold based on sparsity factor
        keep_ratio = sparsity_factor
        if np.any(result > 0):
            threshold = np.percentile(result[result > 0], 100 * (1 - keep_ratio))
        else:
            threshold = 0
    
    # Apply threshold
    result[result < threshold] = 0
    
    # Ensure minimum connectivity: each node connected to at least 2 others
    for i in range(n):
        non_zero = np.where(result[i] > 0)[0]
        if len(non_zero) < 2:
            # Find top 2 edges by priority, excluding self
            top_indices = np.argsort(priorities[i])[-3:]  # Get top 3 (might include self)
            top_indices = [idx for idx in top_indices if idx != i][:2]
            
            for idx in top_indices:
                if idx != i:
                    result[i, idx] = priorities[i, idx]
                    result[idx, i] = priorities[idx, i]
    
    # Ensure the graph remains connected
    if n > 1:
        # Add MST edges if they were removed
        mst = minimum_spanning_tree(dist_matrix)
        mst_mask = mst.toarray() > 0
        
        for i in range(n):
            for j in range(i+1, n):
                if mst_mask[i, j] and result[i, j] == 0:
                    result[i, j] = priorities[i, j]
                    result[j, i] = priorities[j, i]
    
    return result
