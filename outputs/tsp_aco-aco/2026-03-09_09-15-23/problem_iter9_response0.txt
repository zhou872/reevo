import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with improved computational
    efficiency through vectorized operations and density-aware scoring.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive Sparsification ====================
    # Logarithmic scaling for sparsification
    k = max(3, int(np.log2(n)))
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Precompute sorted indices for each row
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    # Mark k-nearest neighbors
    for i in range(n):
        nn_indices = sorted_indices[i, 1:k+1]
        adjacency_mask[i, nn_indices] = True
    
    # Ensure symmetry and no self-loops
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Density-Augmented Rank-Weighted Scoring ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if not np.any(valid_mask):
            continue
            
        # Get distances to neighbors
        neighbor_dists = distance_matrix[i, valid_mask]
        
        # Compute rank-based weights with exponential decay
        ranks = np.argsort(neighbor_dists) + 1  # Rank 1 for closest
        alpha = 0.5  # Exponential decay factor
        weights = np.exp(-alpha * (ranks - 1))
        weights = weights / np.sum(weights)  # Normalize
        
        # Compute inverse distances with neighbor density factor
        inv_vals = 1.0 / (neighbor_dists + 1e-10)
        
        # Density factor: saturation from mean inverse neighbor distances
        mean_inv_neighbor = np.mean(inv_vals)
        density_factor = 1.0 + 0.5 * np.tanh(mean_inv_neighbor - 1.0)  # Centered around 1
        
        # Apply weighted scoring with density augmentation
        inv_dist[i, valid_mask] = (inv_vals * weights) * density_factor
    
    # Symmetrize by taking average
    inv_dist = 0.5 * (inv_dist + inv_dist.T)
    
    # ==================== 3. Vectorized Two-Hop Analysis ====================
    two_hop_penalty = np.ones_like(distance_matrix)
    
    # Precompute sorted distances and indices
    sorted_distances = np.sort(distance_matrix, axis=1)
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    # Use top m neighbors for two-hop analysis
    m = max(3, int(np.sqrt(n)))
    
    # For each node, get its m nearest neighbors (excluding self)
    top_m_neighbors = sorted_indices[:, 1:m+1]
    
    # Compute two-hop distances efficiently
    for i in range(n):
        # Get neighbors to consider for two-hop paths
        i_neighbors = top_m_neighbors[i]
        
        for j in range(i+1, n):
            if not adjacency_mask[i, j]:
                continue
                
            j_neighbors = top_m_neighbors[j]
            
            # Consider intermediate nodes from both i and j's neighbor sets
            candidate_nodes = np.unique(np.concatenate([i_neighbors, j_neighbors]))
            candidate_nodes = candidate_nodes[(candidate_nodes != i) & (candidate_nodes != j)]
            
            if len(candidate_nodes) > 0:
                # Vectorized two-hop distance computation
                two_hop_dists = distance_matrix[i, candidate_nodes] + distance_matrix[candidate_nodes, j]
                min_two_hop = np.min(two_hop_dists)
                direct_dist = distance_matrix[i, j]
                
                if direct_dist > 1e-10:
                    # Compute penalty based on two-hop alternative quality
                    ratio = min_two_hop / direct_dist
                    # Penalty increases as ratio approaches 1 (good alternative exists)
                    penalty = 0.3 + 0.7 * np.clip(ratio, 0.5, 2.0) / 2.0
                    two_hop_penalty[i, j] = two_hop_penalty[j, i] = penalty
    
    # ==================== 4. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Apply scores only to promising edges
    mask_idx = np.where(adjacency_mask)
    heuristics[mask_idx] = inv_dist[mask_idx] * two_hop_penalty[mask_idx]
    
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Add small constant to avoid zeros and normalize rows
    heuristics[adjacency_mask] += 1e-8
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Final symmetrization
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
