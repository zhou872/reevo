import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with improved balancing of
    local and global considerations through rank-based exponential decay and
    dynamic sparsification strategies.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Dynamic Reciprocity-Based Sparsification ====================
    # Adapt sparsity level based on instance characteristics
    k_base = max(3, int(np.log2(n)))
    # Increase connectivity for small instances, reduce for large ones
    k_adj = k_base if n < 50 else max(3, int(k_base * 0.8))
    
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Get k_adj nearest neighbors
        nn_indices = np.argpartition(distances, k_adj)[:k_adj]
        adjacency_mask[i, nn_indices] = True
        
        # Enhanced reciprocal checking: include edges where either endpoint
        # considers the other among its top (k_adj + 2) neighbors
        for j in nn_indices:
            j_distances = distance_matrix[j].copy()
            j_distances[j] = np.inf
            j_nn = np.argpartition(j_distances, k_adj + 2)[:k_adj + 2]
            if i in j_nn:
                adjacency_mask[j, i] = True
    
    # Ensure connectivity and symmetry
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Rank-Aware Exponential Distance Scoring ====================
    # Use exponential decay based on rank rather than inverse distance
    exp_scores = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            # Get distances and sort them
            valid_dists = distance_matrix[i, valid_mask]
            sorted_indices = np.argsort(valid_dists)
            sorted_dists = valid_dists[sorted_indices]
            
            # Convert ranks to exponential decay weights
            ranks = np.arange(1, len(sorted_dists) + 1)
            # Base decay factor adapts to neighborhood size
            base_decay = 0.7 + 0.3 * np.exp(-len(ranks) / 10)
            weights = base_decay ** (ranks - 1)
            weights = weights / np.sum(weights)  # Normalize
            
            # Apply normalized distances with cluster-aware scaling
            # Scale distances by their relative position in the sorted list
            scaled_dists = sorted_dists / (np.median(sorted_dists) + 1e-10)
            exp_vals = np.exp(-scaled_dists)
            
            # Restore original order and apply weights
            restored_weights = np.zeros_like(weights)
            restored_weights[sorted_indices] = weights
            restored_exp = np.zeros_like(exp_vals)
            restored_exp[sorted_indices] = exp_vals
            
            exp_scores[i, valid_mask] = restored_exp * restored_weights
    
    # Symmetrize by geometric mean for better consistency
    exp_scores_sym = np.sqrt(np.maximum(exp_scores, 1e-10) * np.maximum(exp_scores.T, 1e-10))
    
    # ==================== 3. Enhanced Two-Hop Analysis ====================
    # More sophisticated alternative route assessment
    two_hop_factor = np.ones_like(distance_matrix)
    
    # Pre-compute sorted neighbors for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    check_limit = max(3, int(np.sqrt(n)))
    
    # Vectorized computation for valid edges
    valid_edges = np.where(np.triu(adjacency_mask, 1))
    
    for i, j in zip(valid_edges[0], valid_edges[1]):
        # Consider both endpoints' neighborhoods
        i_neighbors = sorted_indices[i, 1:check_limit+1]
        j_neighbors = sorted_indices[j, 1:check_limit+1]
        
        # Combine neighborhoods, excluding endpoints
        candidate_nodes = np.unique(np.concatenate([i_neighbors, j_neighbors]))
        candidate_nodes = candidate_nodes[(candidate_nodes != i) & (candidate_nodes != j)]
        
        if len(candidate_nodes) > 0:
            # Compute two-hop distances
            two_hop_dists = distance_matrix[i, candidate_nodes] + distance_matrix[candidate_nodes, j]
            min_two_hop = np.min(two_hop_dists) if len(two_hop_dists) > 0 else np.inf
            
            direct_dist = distance_matrix[i, j]
            
            if direct_dist > 1e-10 and min_two_hop < np.inf:
                ratio = min_two_hop / direct_dist
                
                # Dynamic penalty/bonus based on ratio:
                # - Strong penalty if two-hop is significantly better (ratio < 0.8)
                # - Mild penalty if slightly better (0.8 <= ratio < 1.0)
                # - Bonus if direct is better (ratio > 1.0)
                if ratio < 0.8:
                    factor = 0.2 + 0.8 * (ratio / 0.8)
                elif ratio < 1.0:
                    factor = 0.5 + 0.5 * (ratio - 0.8) / 0.2
                else:  # ratio >= 1.0
                    # Diminishing returns for large ratios
                    factor = 1.0 + 0.5 * (1 - np.exp(-(ratio - 1.0)))
                
                two_hop_factor[i, j] = two_hop_factor[j, i] = np.clip(factor, 0.1, 2.0)
    
    # ==================== 4. Combined Scoring with Cluster Awareness ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Apply exponential scores with two-hop adjustment
    heuristics[adjacency_mask] = exp_scores_sym[adjacency_mask] * two_hop_factor[adjacency_mask]
    
    # Add cluster density consideration: boost edges between nodes with
    # similar distance profiles to encourage intra-cluster connections
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Compare distance profiles to other nodes
                i_profile = np.sort(distance_matrix[i])
                j_profile = np.sort(distance_matrix[j])
                profile_sim = 1.0 / (1.0 + np.mean(np.abs(i_profile - j_profile)))
                
                # Moderate boost for similar profiles
                heuristics[i, j] *= (0.7 + 0.3 * profile_sim)
                heuristics[j, i] = heuristics[i, j]
    
    # ==================== 5. Adaptive Normalization ====================
    np.fill_diagonal(heuristics, 0)
    
    # Softmax-like normalization per node
    for i in range(n):
        row_vals = heuristics[i]
        if np.any(row_vals > 0):
            # Shift to positive values for stable exponential
            min_val = np.min(row_vals[row_vals > 0])
            shifted = row_vals - min_val + 1e-8
            exp_vals = np.exp(shifted / (np.std(shifted) + 1e-10))
            heuristics[i] = exp_vals / np.sum(exp_vals)
    
    # Final symmetrization with harmonic mean for consistency
    heuristics = np.where(adjacency_mask,
                          2 * heuristics * heuristics.T / (heuristics + heuristics.T + 1e-10),
                          0)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
