import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization focusing on global 
    connectivity patterns through MST integration and multi-hop lookahead.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes.
        
    Returns:
    --------
    np.ndarray
        Matrix of edge priorities with same shape as input.
    """
    n = distance_matrix.shape[0]
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # 1. Adaptive Neighborhood Selection
    adjacency_mask = _adaptive_neighborhood_selection(distance_matrix, n)
    
    # 2. Rank-Weighted Inverse Distance
    inv_dist = _rank_weighted_inverse_distance(distance_matrix, adjacency_mask, n)
    
    # 3. MST Analysis with Dynamic Bonuses
    mst_mask, odd_degree_nodes = _mst_analysis(distance_matrix, n)
    
    # 4. Multi-Hop Regret Analysis
    regret_scores = _multi_hop_regret(distance_matrix, adjacency_mask, n)
    
    # 5. Lightweight Multi-Hop Lookahead
    lookahead_scores = _multi_hop_lookahead(distance_matrix, adjacency_mask, n)
    
    # 6. Combine Components
    heuristics = _combine_components(
        distance_matrix, adjacency_mask, inv_dist, mst_mask, 
        odd_degree_nodes, regret_scores, lookahead_scores, n
    )
    
    # 7. Final Processing
    heuristics = _final_processing(heuristics, adjacency_mask, n)
    
    return heuristics


def _adaptive_neighborhood_selection(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Select adaptive neighborhoods based on local density."""
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Base neighborhood size
    base_k = max(2, int(np.sqrt(n)))
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Sort distances
        sorted_dists = np.sort(distances)
        
        # Calculate local density
        if len(sorted_dists) >= base_k:
            local_avg = np.mean(sorted_dists[:base_k])
            local_std = np.std(sorted_dists[:base_k])
            
            # Adjust k based on coefficient of variation
            if local_avg > 1e-10:
                cv = local_std / local_avg
                adaptive_k = min(n-1, int(base_k * (1.0 + 0.5 * cv)))
            else:
                adaptive_k = base_k
        else:
            adaptive_k = base_k
        
        # Get nearest neighbors
        nn_indices = np.argpartition(distances, adaptive_k)[:adaptive_k]
        adjacency_mask[i, nn_indices] = True
        
        # Include nodes within adaptive threshold
        if len(nn_indices) > 0:
            threshold = 1.3 * np.max(distances[nn_indices])
            within_threshold = distances <= threshold
            adjacency_mask[i, within_threshold] = True
    
    # Ensure symmetry
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    return adjacency_mask


def _rank_weighted_inverse_distance(distance_matrix: np.ndarray, 
                                   adjacency_mask: np.ndarray, 
                                   n: int) -> np.ndarray:
    """Compute inverse distance with rank-based weighting."""
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        row_dists = distance_matrix[i].copy()
        row_dists[i] = np.inf
        
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            valid_dists = row_dists[valid_mask]
            if len(valid_dists) > 0:
                # Rank by distance (closest = rank 1)
                ranks = np.argsort(np.argsort(valid_dists)) + 1
                
                # Exponential decay based on rank
                decay_factor = 0.7
                weights = np.exp(-decay_factor * (ranks - 1))
                weights = weights / np.sum(weights)  # Normalize
                
                # Apply weighted inverse distance
                inv_vals = 1.0 / (valid_dists + 1e-10)
                inv_dist[i, valid_mask] = inv_vals * weights
    
    # Symmetrize
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    
    return inv_dist


def _mst_analysis(distance_matrix: np.ndarray, n: int):
    """Analyze Minimum Spanning Tree for edge bonuses and odd-degree nodes."""
    mst_mask = np.zeros((n, n), dtype=bool)
    
    try:
        # Compute MST using Prim's algorithm
        visited = np.zeros(n, dtype=bool)
        min_edge = np.full(n, np.inf)
        parent = np.full(n, -1, dtype=int)
        
        min_edge[0] = 0
        
        for _ in range(n):
            # Find minimum edge to unvisited node
            u = -1
            min_val = np.inf
            for i in range(n):
                if not visited[i] and min_edge[i] < min_val:
                    min_val = min_edge[i]
                    u = i
            
            if u == -1:
                break
                
            visited[u] = True
            
            # Add edge to MST if not the root
            if parent[u] != -1:
                mst_mask[u, parent[u]] = True
                mst_mask[parent[u], u] = True
            
            # Update distances to unvisited nodes
            for v in range(n):
                if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                    min_edge[v] = distance_matrix[u, v]
                    parent[v] = u
        
        # Compute MST degrees for Christofides bonus
        mst_degrees = np.sum(mst_mask, axis=1)
        odd_degree_nodes = np.where(mst_degrees % 2 == 1)[0]
        
    except Exception as e:
        warnings.warn(f"MST computation failed: {e}")
        # Fallback: use nearest neighbor heuristic
        for i in range(n):
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            nearest = np.argmin(distances)
            mst_mask[i, nearest] = mst_mask[nearest, i] = True
        
        mst_degrees = np.sum(mst_mask, axis=1)
        odd_degree_nodes = np.where(mst_degrees % 2 == 1)[0]
    
    return mst_mask, odd_degree_nodes


def _multi_hop_regret(distance_matrix: np.ndarray, 
                     adjacency_mask: np.ndarray, 
                     n: int) -> np.ndarray:
    """Compute regret scores using multi-hop analysis."""
    regret = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                direct_dist = distance_matrix[i, j]
                
                # Find best 2-hop and 3-hop alternatives
                min_alternative = np.inf
                
                # 2-hop alternatives
                for k in range(n):
                    if k != i and k != j:
                        two_hop_dist = distance_matrix[i, k] + distance_matrix[k, j]
                        if two_hop_dist < min_alternative:
                            min_alternative = two_hop_dist
                
                # 3-hop alternatives (limited search for efficiency)
                if n > 10:
                    # Sample a few intermediate nodes
                    candidates = np.random.choice([x for x in range(n) if x not in [i, j]], 
                                                 size=min(10, n-2), replace=False)
                    for k1 in candidates:
                        for k2 in candidates:
                            if k1 != k2:
                                three_hop_dist = (distance_matrix[i, k1] + 
                                                 distance_matrix[k1, k2] + 
                                                 distance_matrix[k2, j])
                                if three_hop_dist < min_alternative:
                                    min_alternative = three_hop_dist
                
                if min_alternative < np.inf and direct_dist > 1e-10:
                    # Regret ratio: how much worse is the best alternative?
                    regret_ratio = min_alternative / direct_dist
                    # Exponential transformation to emphasize good edges
                    regret[i, j] = regret[j, i] = np.exp(regret_ratio - 1)
    
    # Normalize
    max_regret = regret.max()
    if max_regret > 0:
        regret = regret / max_regret
    
    return regret


def _multi_hop_lookahead(distance_matrix: np.ndarray, 
                        adjacency_mask: np.ndarray, 
                        n: int) -> np.ndarray:
    """Lightweight multi-hop lookahead for edge evaluation."""
    lookahead = np.zeros_like(distance_matrix, dtype=float)
    
    # For each edge, evaluate 2-step lookahead paths
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                score = 0
                count = 0
                
                # Look at promising 3-node chains
                for k in range(n):
                    if k != i and k != j and adjacency_mask[j, k]:
                        # Evaluate path i->j->k and alternative i->k directly
                        chain_length = distance_matrix[i, j] + distance_matrix[j, k]
                        direct_length = distance_matrix[i, k]
                        
                        if direct_length > 1e-10:
                            ratio = chain_length / direct_length
                            # Penalty if the chain is much longer than direct connection
                            penalty = max(0, 2.0 - ratio) / 2.0
                            score += penalty
                            count += 1
                
                if count > 0:
                    lookahead[i, j] = lookahead[j, i] = score / count
    
    # Normalize
    max_lookahead = lookahead.max()
    if max_lookahead > 0:
        lookahead = lookahead / max_lookahead
    
    return lookahead


def _combine_components(distance_matrix: np.ndarray,
                       adjacency_mask: np.ndarray,
                       inv_dist: np.ndarray,
                       mst_mask: np.ndarray,
                       odd_degree_nodes: np.ndarray,
                       regret_scores: np.ndarray,
                       lookahead_scores: np.ndarray,
                       n: int) -> np.ndarray:
    """Combine all heuristic components."""
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Base score from inverse distance
                score = inv_dist[i, j]
                
                # Dynamic MST bonus
                if mst_mask[i, j]:
                    # Stronger bonus for critical MST edges
                    mst_bonus = 2.5  # Increased from previous versions
                    score *= mst_bonus
                
                # Matching bonus for odd-degree nodes (Christofides-inspired)
                if i in odd_degree_nodes and j in odd_degree_nodes:
                    # Bonus based on distance relative to other odd nodes
                    other_odd = [node for node in odd_degree_nodes if node not in (i, j)]
                    if other_odd:
                        avg_dist_to_other = np.mean([distance_matrix[i, k] for k in other_odd])
                        if avg_dist_to_other > 0:
                            matching_quality = avg_dist_to_other / (distance_matrix[i, j] + 1e-10)
                            # Cap the bonus to avoid extreme values
                            bonus_factor = 1.0 + 0.5 * min(matching_quality, 2.5)
                            score *= bonus_factor
                
                # Multi-hop regret bonus
                score *= (1.0 + 0.3 * regret_scores[i, j])
                
                # Multi-hop lookahead adjustment
                score *= (1.0 + 0.2 * (1.0 - lookahead_scores[i, j]))
                
                # Local density adjustment
                # Compute average distance to nearest neighbors for both endpoints
                k = max(2, int(np.sqrt(n)))
                i_dists = np.partition(distance_matrix[i].copy(), k)[:k]
                j_dists = np.partition(distance_matrix[j].copy(), k)[:k]
                density_i = np.mean(i_dists[i_dists > 0])
                density_j = np.mean(j_dists[j_dists > 0])
                
                # Normalize and adjust
                max_density = max(density_i, density_j, 1e-10)
                density_factor = 0.7 + 0.3 * (1.0 - (density_i + density_j) / (2 * max_density))
                score *= density_factor
                
                heuristics[i, j] = heuristics[j, i] = score
    
    return heuristics


def _final_processing(heuristics: np.ndarray, 
                     adjacency_mask: np.ndarray, 
                     n: int) -> np.ndarray:
    """Final normalization and processing."""
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Add small constant to promising edges
    heuristics[adjacency_mask] += 1e-8
    
    # Row-normalize
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Symmetrize
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
