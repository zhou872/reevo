import numpy as np
from scipy.spatial import distance_matrix as scipy_distance_matrix
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray, 
                  density_aware: bool = True,
                  christofides_bonus: bool = True) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection combining graph-theoretic insights
    with adaptive sparsification and global connectivity considerations.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        density_aware: If True, adapt k-neighborhood size based on local density
        christofides_bonus: If True, apply Christofides-inspired matching bonus
    
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive Neighborhood Selection ====================
    # Determine k for each node based on local density if enabled
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    if density_aware:
        # Adaptive k selection based on distance distribution
        base_k = max(2, int(np.sqrt(n)))
        for i in range(n):
            # Compute distance statistics for node i
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            
            # Sort distances
            sorted_dists = np.sort(distances)
            
            # Calculate local density metrics
            nearest_avg = np.mean(sorted_dists[:base_k])
            dist_std = np.std(sorted_dists[:base_k])
            
            # Adaptive threshold: include more neighbors in sparse regions
            if dist_std > 1e-10:
                # Use coefficient of variation to adjust k
                cv = dist_std / nearest_avg
                adaptive_k = min(n-1, int(base_k * (1.0 + 0.5 * cv)))
            else:
                adaptive_k = base_k
                
            # Get adaptive_k nearest neighbors
            nn_indices = np.argpartition(distances, adaptive_k)[:adaptive_k]
            adjacency_mask[i, nn_indices] = True
            
            # Also include any nodes within 1.2x of the maximum nearest neighbor distance
            if len(nn_indices) > 0:
                threshold = 1.2 * np.max(distances[nn_indices])
                within_threshold = distances <= threshold
                adjacency_mask[i, within_threshold] = True
    else:
        # Fixed k selection (fallback)
        k = max(2, int(np.sqrt(n)))
        for i in range(n):
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            nn_indices = np.argpartition(distances, k)[:k]
            adjacency_mask[i, nn_indices] = True
    
    # Ensure symmetry
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Distance-Based Initial Scores ====================
    # Compute inverse distance scores with rank-based weighting
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        # Get distances and indices for this node's neighbors
        row_dists = distance_matrix[i].copy()
        row_dists[i] = np.inf
        
        # Create ranked weights: closer neighbors get higher weights
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            valid_dists = row_dists[valid_mask]
            if len(valid_dists) > 0:
                # Rank by distance (closest = rank 1)
                ranks = np.argsort(np.argsort(valid_dists)) + 1
                # Weight inversely with rank
                weights = 1.0 / ranks
                weights = weights / np.sum(weights)  # Normalize
                
                # Apply weighted inverse distance
                inv_vals = 1.0 / (valid_dists + 1e-10)
                inv_dist[i, valid_mask] = inv_vals * weights
    
    # Symmetrize
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    
    # ==================== 3. Minimum Spanning Tree Components ====================
    # Prim's algorithm for MST
    mst_mask = np.zeros((n, n), dtype=bool)
    
    visited = np.zeros(n, dtype=bool)
    min_edge = np.full(n, np.inf)
    parent = np.full(n, -1, dtype=int)
    
    # Start from node 0
    min_edge[0] = 0
    
    for _ in range(n):
        # Find minimum edge to unvisited node
        u = -1
        min_val = np.inf
        for i in range(n):
            if not visited[i] and min_edge[i] < min_val:
                min_val = min_edge[i]
                u = i
        
        if u == -1:
            break
            
        visited[u] = True
        
        # Add edge to MST if not the root
        if parent[u] != -1:
            mst_mask[u, parent[u]] = True
            mst_mask[parent[u], u] = True
        
        # Update distances to unvisited nodes
        for v in range(n):
            if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                min_edge[v] = distance_matrix[u, v]
                parent[v] = u
    
    # Compute MST degrees for Christofides bonus
    mst_degrees = np.sum(mst_mask, axis=1)
    odd_degree_nodes = np.where(mst_degrees % 2 == 1)[0]
    
    # ==================== 4. Enhanced Two-Hop Analysis ====================
    # Find minimum two-hop alternative for each edge
    two_hop_ratio = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Find best intermediate node (minimum two-hop path)
                # Exclude i and j themselves
                other_nodes = [k for k in range(n) if k not in (i, j)]
                if other_nodes:
                    # Vectorized computation of two-hop distances
                    two_hop_dists = distance_matrix[i, other_nodes] + distance_matrix[other_nodes, j]
                    min_two_hop = np.min(two_hop_dists) if len(two_hop_dists) > 0 else np.inf
                    
                    direct_dist = distance_matrix[i, j]
                    if direct_dist > 1e-10 and min_two_hop < np.inf:
                        # Ratio > 1 means direct edge is shorter than best detour
                        two_hop_ratio[i, j] = two_hop_ratio[j, i] = min_two_hop / direct_dist
    
    # Normalize two-hop ratios
    if np.max(two_hop_ratio) > 0:
        two_hop_ratio = two_hop_ratio / np.max(two_hop_ratio)
    
    # ==================== 5. Local Density Measures ====================
    # Compute local density for normalization
    local_density = np.zeros(n)
    
    for i in range(n):
        # Average distance to sqrt(n) nearest neighbors
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        k = max(2, int(np.sqrt(n)))
        nearest_dists = np.partition(distances, k)[:k]
        local_density[i] = np.mean(nearest_dists)
    
    # Normalize density scores (invert so sparse regions get higher values)
    if np.max(local_density) > 0:
        density_norm = local_density / np.max(local_density)
        # Invert: sparse regions (high avg distance) get boost
        density_factor = 1.0 + 0.3 * (1.0 - density_norm)
    else:
        density_factor = np.ones(n)
    
    # ==================== 6. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Base score from inverse distance
                score = inv_dist[i, j]
                
                # MST bonus (stronger than in v1)
                if mst_mask[i, j]:
                    score *= 2.0  # Increased from 1.5
                
                # Christofides matching bonus for edges between odd-degree nodes
                if christofides_bonus and (i in odd_degree_nodes and j in odd_degree_nodes):
                    # Bonus proportional to how "good" this matching edge would be
                    # based on distance relative to other potential matching edges
                    other_odd = [node for node in odd_degree_nodes if node not in (i, j)]
                    if other_odd:
                        # Estimate quality by comparing to average distance to other odd nodes
                        avg_dist_to_other = np.mean([distance_matrix[i, k] for k in other_odd])
                        if avg_dist_to_other > 0:
                            matching_quality = avg_dist_to_other / (distance_matrix[i, j] + 1e-10)
                            score *= (1.0 + 0.4 * min(matching_quality, 2.0))
                
                # Two-hop ratio bonus
                if two_hop_ratio[i, j] > 0:
                    # Stronger bonus for edges that are much shorter than alternatives
                    score *= (1.0 + 0.5 * two_hop_ratio[i, j])
                
                # Local density adjustment
                score *= 0.5 * (density_factor[i] + density_factor[j])
                
                heuristics[i, j] = heuristics[j, i] = score
    
    # ==================== 7. Final Processing ====================
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Add small constant to all promising edges to avoid zeros
    heuristics[adjacency_mask] += 1e-8
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1  # Avoid division by zero
    heuristics = heuristics / row_sums
    
    # Symmetrize final matrix
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
