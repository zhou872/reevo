import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristic function integrating Christofides algorithm components,
    multi-hop regret analysis, and adaptive structural learning for optimal edge prioritization.
    
    Args:
        distance_matrix: Square matrix of pairwise distances between n nodes
        
    Returns:
        Heuristic matrix of same shape, with higher values indicating more promising edges
    """
    n = len(distance_matrix)
    
    # Handle trivial cases
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. CHRISTOFIDES-BASED EDGE PRIORITIZATION ====================
    christofides_edges = _compute_christofides_edges(distance_matrix)
    
    # ==================== 2. ADAPTIVE MULTI-SCALE NEIGHBORHOOD CONSTRUCTION ====================
    adjacency_mask = _build_multi_scale_adjacency(
        distance_matrix, 
        christofides_edges
    )
    
    # ==================== 3. MULTI-HOP REGRET ANALYSIS ====================
    regret_scores = _compute_multi_hop_regret(
        distance_matrix, 
        adjacency_mask,
        max_hops=3
    )
    
    # ==================== 4. TOUR COMPLETION POTENTIAL SCORING ====================
    completion_scores = _compute_tour_completion_potential(
        distance_matrix,
        adjacency_mask,
        christofides_edges
    )
    
    # ==================== 5. DYNAMIC COMPONENT FUSION ====================
    heuristics = _fuse_components_dynamically(
        regret_scores,
        completion_scores,
        christofides_edges,
        adjacency_mask,
        distance_matrix
    )
    
    return heuristics

def _compute_christofides_edges(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute Christofides algorithm components: MST and minimum weight perfect matching."""
    n = len(distance_matrix)
    edge_scores = np.zeros_like(distance_matrix, dtype=float)
    
    # 1. Compute Minimum Spanning Tree using Prim's algorithm
    mst_mask = _compute_mst_mask(distance_matrix)
    
    # 2. Find odd-degree vertices in MST
    odd_degree_mask = _find_odd_degree_vertices(mst_mask)
    
    # 3. Compute minimum weight perfect matching on odd-degree vertices
    matching_mask = _minimum_weight_perfect_matching(
        distance_matrix, 
        odd_degree_mask
    )
    
    # 4. Combine MST and matching edges with different weights
    for i in range(n):
        for j in range(i + 1, n):
            if mst_mask[i, j]:
                # MST edges get highest priority
                edge_scores[i, j] = edge_scores[j, i] = 2.0
            elif matching_mask[i, j]:
                # Matching edges get medium priority
                edge_scores[i, j] = edge_scores[j, i] = 1.5
            else:
                # Non-Christofides edges start with neutral score
                edge_scores[i, j] = edge_scores[j, i] = 1.0
    
    return edge_scores

def _compute_mst_mask(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute Minimum Spanning Tree using Prim's algorithm."""
    n = len(distance_matrix)
    mst_mask = np.zeros((n, n), dtype=bool)
    
    if n == 0:
        return mst_mask
    
    visited = np.zeros(n, dtype=bool)
    min_edge_weights = np.full(n, np.inf)
    min_edge_sources = np.full(n, -1, dtype=int)
    
    # Start from node 0
    min_edge_weights[0] = 0
    
    for _ in range(n):
        # Find unvisited vertex with minimum edge weight
        current = -1
        min_weight = np.inf
        for v in range(n):
            if not visited[v] and min_edge_weights[v] < min_weight:
                min_weight = min_edge_weights[v]
                current = v
        
        if current == -1:
            break
        
        visited[current] = True
        
        # Add edge to MST if not the starting node
        if min_edge_sources[current] != -1:
            source = min_edge_sources[current]
            mst_mask[current, source] = mst_mask[source, current] = True
        
        # Update minimum edge weights for neighbors
        for neighbor in range(n):
            if (not visited[neighbor] and 
                distance_matrix[current, neighbor] < min_edge_weights[neighbor]):
                min_edge_weights[neighbor] = distance_matrix[current, neighbor]
                min_edge_sources[neighbor] = current
    
    return mst_mask

def _find_odd_degree_vertices(mst_mask: np.ndarray) -> np.ndarray:
    """Identify vertices with odd degree in the MST."""
    n = len(mst_mask)
    degrees = np.sum(mst_mask, axis=1)
    odd_mask = (degrees % 2 == 1)
    return odd_mask

def _minimum_weight_perfect_matching(distance_matrix: np.ndarray, 
                                     odd_degree_mask: np.ndarray) -> np.ndarray:
    """Greedy minimum weight perfect matching on odd-degree vertices."""
    n = len(distance_matrix)
    matching_mask = np.zeros((n, n), dtype=bool)
    
    # Get list of odd-degree vertices
    odd_vertices = np.where(odd_degree_mask)[0]
    
    if len(odd_vertices) % 2 != 0:
        # Shouldn't happen for MST, but handle gracefully
        return matching_mask
    
    # Sort odd vertices by their average distance to other odd vertices
    avg_dists = []
    for v in odd_vertices:
        other_odds = odd_vertices[odd_vertices != v]
        avg_dist = np.mean(distance_matrix[v, other_odds])
        avg_dists.append(avg_dist)
    
    # Pair vertices greedily starting from most isolated
    sorted_indices = np.argsort(avg_dists)[::-1]
    paired = set()
    
    for i in range(0, len(odd_vertices), 2):
        if i + 1 >= len(odd_vertices):
            break
            
        idx1 = sorted_indices[i]
        v1 = odd_vertices[idx1]
        
        # Find closest unpaired odd vertex
        min_dist = np.inf
        best_v2 = -1
        
        for j in range(i + 1, len(odd_vertices)):
            idx2 = sorted_indices[j]
            v2 = odd_vertices[idx2]
            
            if v2 not in paired:
                dist = distance_matrix[v1, v2]
                if dist < min_dist:
                    min_dist = dist
                    best_v2 = v2
        
        if best_v2 != -1:
            matching_mask[v1, best_v2] = matching_mask[best_v2, v1] = True
            paired.update([v1, best_v2])
    
    return matching_mask

def _build_multi_scale_adjacency(distance_matrix: np.ndarray,
                                 christofides_edges: np.ndarray) -> np.ndarray:
    """Build adaptive adjacency matrix using multi-scale neighborhood construction."""
    n = len(distance_matrix)
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Always include Christofides edges
    adjacency_mask[christofides_edges > 1.0] = True
    
    # Compute local density statistics
    base_k = max(3, int(np.log2(n)))
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    # First pass: core connections
    for i in range(n):
        # Get k nearest neighbors
        k_core = max(2, base_k // 2)
        nn_core = sorted_indices[i, 1:k_core + 1]
        adjacency_mask[i, nn_core] = True
    
    # Second pass: adaptive expansion based on local structure
    for i in range(n):
        # Count current connections
        current_degree = np.sum(adjacency_mask[i])
        
        # Determine expansion factor based on local density
        avg_core_dist = np.mean(distance_matrix[i, sorted_indices[i, 1:base_k + 1]])
        std_core_dist = np.std(distance_matrix[i, sorted_indices[i, 1:base_k + 1]])
        
        if std_core_dist > 1e-10:
            # In dense regions, connect to more neighbors
            density_factor = avg_core_dist / std_core_dist
            k_expand = min(n - 1, int(current_degree * (1.0 + 0.5 * np.tanh(density_factor))))
            
            # Add expanded neighbors
            nn_expand = sorted_indices[i, 1:k_expand + 1]
            adjacency_mask[i, nn_expand] = True
    
    # Ensure symmetry and no self-loops
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    return adjacency_mask

def _compute_multi_hop_regret(distance_matrix: np.ndarray,
                              adjacency_mask: np.ndarray,
                              max_hops: int = 3) -> np.ndarray:
    """Compute regret scores considering alternative paths of up to max_hops."""
    n = len(distance_matrix)
    regret_scores = np.ones_like(distance_matrix, dtype=float)
    
    # Precompute sorted neighbors for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    neighbor_limit = max(8, int(np.sqrt(n)))
    
    for i in range(n):
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                regret_scores[i, j] = regret_scores[j, i] = 0.0
                continue
            
            direct_dist = distance_matrix[i, j]
            if direct_dist < 1e-10:
                regret_scores[i, j] = regret_scores[j, i] = 1.0
                continue
            
            # Get candidate intermediate nodes (top neighbors)
            i_neighbors = sorted_indices[i, 1:neighbor_limit + 1]
            j_neighbors = sorted_indices[j, 1:neighbor_limit + 1]
            
            best_alternative = np.inf
            
            # 2-hop alternatives
            for k in i_neighbors:
                if k == j:
                    continue
                two_hop = distance_matrix[i, k] + distance_matrix[k, j]
                best_alternative = min(best_alternative, two_hop)
            
            # 3-hop alternatives (if enabled)
            if max_hops >= 3:
                for k in i_neighbors:
                    if k == j:
                        continue
                    # Get k's top neighbors
                    k_neighbors = sorted_indices[k, 1:min(neighbor_limit, 6) + 1]
                    for l in k_neighbors:
                        if l == i or l == j:
                            continue
                        three_hop = (distance_matrix[i, k] + 
                                    distance_matrix[k, l] + 
                                    distance_matrix[l, j])
                        best_alternative = min(best_alternative, three_hop)
            
            # Compute regret-based score
            if best_alternative < np.inf:
                regret_ratio = best_alternative / direct_dist
                
                # Sigmoid mapping: higher ratio (worse alternative) → higher score
                if regret_ratio > 1.5:
                    score = 1.0  # Direct edge is much better
                elif regret_ratio > 1.1:
                    score = 0.8 + 0.2 * np.tanh(regret_ratio - 1.1)
                elif regret_ratio > 1.0:
                    score = 0.6 + 0.4 * np.tanh(regret_ratio - 1.0)
                else:
                    # Alternative is better than direct edge
                    score = 0.3 + 0.3 * np.tanh(1.0 - regret_ratio)
                
                regret_scores[i, j] = regret_scores[j, i] = np.clip(score, 0.1, 1.0)
    
    return regret_scores

def _compute_tour_completion_potential(distance_matrix: np.ndarray,
                                       adjacency_mask: np.ndarray,
                                       christofides_edges: np.ndarray) -> np.ndarray:
    """Score edges based on their potential to complete high-quality tours."""
    n = len(distance_matrix)
    completion_scores = np.zeros_like(distance_matrix, dtype=float)
    
    # Precompute node centrality using average distance to other nodes
    node_centrality = np.zeros(n)
    for i in range(n):
        node_centrality[i] = 1.0 / (np.mean(distance_matrix[i]) + 1e-10)
    
    # Normalize centrality
    if np.max(node_centrality) > 0:
        node_centrality = node_centrality / np.max(node_centrality)
    
    for i in range(n):
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                continue
            
            base_score = 0.0
            
            # 1. Distance-based component
            avg_dist = np.mean(distance_matrix)
            dist_ratio = avg_dist / (distance_matrix[i, j] + 1e-10)
            base_score += 0.4 * np.tanh(dist_ratio)
            
            # 2. Centrality-based component
            centrality_bonus = (node_centrality[i] + node_centrality[j]) / 2.0
            base_score += 0.3 * centrality_bonus
            
            # 3. Christofides reinforcement
            if christofides_edges[i, j] > 1.0:
                base_score += 0.3 * (christofides_edges[i, j] - 1.0)
            
            # 4. Connectivity potential
            i_degree = np.sum(adjacency_mask[i])
            j_degree = np.sum(adjacency_mask[j])
            connectivity = (i_degree + j_degree) / (2.0 * (n - 1))
            base_score += 0.2 * connectivity
            
            completion_scores[i, j] = completion_scores[j, i] = np.clip(base_score, 0.0, 1.0)
    
    return completion_scores

def _fuse_components_dynamically(regret_scores: np.ndarray,
                                 completion_scores: np.ndarray,
                                 christofides_edges: np.ndarray,
                                 adjacency_mask: np.ndarray,
                                 distance_matrix: np.ndarray) -> np.ndarray:
    """Dynamically combine all heuristic components with learned weighting."""
    n = len(regret_scores)
    heuristics = np.zeros_like(regret_scores, dtype=float)
    
    # Compute dynamic weights based on problem characteristics
    n = len(distance_matrix)
    
    # Measure problem structure
    avg_distance = np.mean(distance_matrix)
    std_distance = np.std(distance_matrix)
    uniformity = std_distance / (avg_distance + 1e-10)
    
    # Dynamic weighting based on problem uniformity
    if uniformity < 0.3:  # Very uniform distances
        w_regret, w_completion, w_christofides = 0.4, 0.4, 0.2
    elif uniformity < 0.6:  # Moderately varied
        w_regret, w_completion, w_christofides = 0.5, 0.3, 0.2
    else:  # Highly varied distances
        w_regret, w_completion, w_christofides = 0.6, 0.2, 0.2
    
    # Combine scores for each edge
    for i in range(n):
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                continue
            
            # Base combination
            combined = (w_regret * regret_scores[i, j] +
                       w_completion * completion_scores[i, j] +
                       w_christofides * (christofides_edges[i, j] - 1.0) / 1.0)
            
            # Add small deterministic noise for tie-breaking
            noise = 1e-6 * ((i * 131 + j * 241) % 1000) / 1000.0
            heuristics[i, j] = heuristics[j, i] = combined + noise
    
    # Row-wise normalization using softmax
    for i in range(n):
        row = heuristics[i]
        if np.any(row > 0):
            # Shift for numerical stability
            row_max = np.max(row)
            if row_max > 0:
                exp_row = np.exp((row - row_max) * 5.0)  # Sharpened distribution
                heuristics[i] = exp_row / np.sum(exp_row)
    
    # Ensure symmetry and zero diagonal
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
