import numpy as np
from scipy.sparse import csr_matrix
from scipy.sparse.csgraph import minimum_spanning_tree, dijkstra

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with extended multi-hop analysis,
    graph-theoretic features, and learned component combination.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Dual-Scale Adaptive Sparsification ====================
    # Global k based on problem size, local k adapts to cluster density
    global_k = max(3, min(int(np.log2(n)) * 2, int(np.sqrt(n))))
    adjacency_mask = np.zeros((n, n), dtype=bool)
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    # Precompute local density metrics
    local_density = np.zeros(n)
    for i in range(n):
        k_local = min(5, n-1)
        nearest_dists = distance_matrix[i, sorted_indices[i, 1:k_local+1]]
        local_density[i] = 1.0 / (np.mean(nearest_dists) + 1e-10)
    
    # Adaptive neighbor selection with density awareness
    for i in range(n):
        base_k = max(3, int(global_k * (0.5 + 0.5 * local_density[i] / np.max(local_density))))
        neighbors = sorted_indices[i, 1:base_k+1]
        adjacency_mask[i, neighbors] = True
        
        # Reciprocal connection reinforcement in dense regions
        if local_density[i] > np.median(local_density):
            for j in neighbors:
                if local_density[j] > np.median(local_density):
                    adjacency_mask[j, i] = True
    
    # Ensure symmetry and no self-loops
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Enhanced Regret-Aware Scoring ====================
    # Multi-level regret considering nearest, second nearest, and cluster-level alternatives
    regret_scores = np.ones_like(distance_matrix)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if not np.any(valid_mask):
            continue
            
        # Get distances to neighbors
        valid_dists = distance_matrix[i, valid_mask]
        valid_indices = np.where(valid_mask)[0]
        
        # Compute base regret against nearest neighbor
        min_dist_i = np.min(valid_dists) if len(valid_dists) > 0 else np.inf
        
        for idx, j in enumerate(valid_indices):
            dist_ij = distance_matrix[i, j]
            
            if abs(dist_ij - min_dist_i) < 1e-10:
                # This is the nearest neighbor
                # Regret: how much better than second nearest?
                second_min = np.partition(valid_dists, 1)[1] if len(valid_dists) > 1 else np.inf
                if second_min > 0:
                    regret = 1.0 + min_dist_i / second_min
                else:
                    regret = 2.0
            else:
                # Regret: how much worse than nearest?
                regret = min_dist_i / (dist_ij + 1e-10)
            
            # Cluster-level regret: compare to average distance in local neighborhood
            i_neighbors = np.where(adjacency_mask[i])[0]
            j_neighbors = np.where(adjacency_mask[j])[0]
            common_neighbors = np.intersect1d(i_neighbors, j_neighbors)
            
            if len(common_neighbors) > 0:
                avg_common_dist = np.mean([distance_matrix[i, c] + distance_matrix[c, j] 
                                         for c in common_neighbors])
                if avg_common_dist > 0:
                    cluster_regret = dist_ij / avg_common_dist
                    # Combine base and cluster regret
                    regret = 0.7 * regret + 0.3 * cluster_regret
            
            regret_scores[i, j] = regret
    
    # Symmetrize regret scores
    regret_scores = 0.5 * (regret_scores + regret_scores.T)
    
    # ==================== 3. Extended Multi-Hop Analysis ====================
    shortcut_quality = np.ones_like(distance_matrix)
    
    for i in range(n):
        i_neighbors = np.where(adjacency_mask[i])[0]
        if len(i_neighbors) == 0:
            continue
            
        for j in i_neighbors:
            if i >= j:
                continue
                
            direct_dist = distance_matrix[i, j]
            if direct_dist < 1e-10:
                continue
            
            # Get 2-hop alternatives through common neighbors
            j_neighbors = np.where(adjacency_mask[j])[0]
            common_neighbors = np.intersect1d(i_neighbors, j_neighbors)
            
            two_hop_alts = []
            if len(common_neighbors) > 0:
                two_hop_dists = distance_matrix[i, common_neighbors] + distance_matrix[common_neighbors, j]
                two_hop_alts.extend(two_hop_dists)
            
            # Get 3-hop alternatives for larger instances (n > 20)
            if n > 20 and len(i_neighbors) > 2 and len(j_neighbors) > 2:
                # Sample promising 3-hop paths through two intermediate nodes
                sample_size = min(10, len(i_neighbors) * len(j_neighbors))
                i_sample = np.random.choice(i_neighbors, min(5, len(i_neighbors)), replace=False)
                j_sample = np.random.choice(j_neighbors, min(5, len(j_neighbors)), replace=False)
                
                for u in i_sample:
                    if u == j:
                        continue
                    u_neighbors = np.where(adjacency_mask[u])[0]
                    for v in j_sample:
                        if v == i or v == u:
                            continue
                        if np.any(adjacency_mask[u, v]):
                            three_hop_dist = (distance_matrix[i, u] + 
                                            distance_matrix[u, v] + 
                                            distance_matrix[v, j])
                            two_hop_alts.append(three_hop_dist)
            
            if len(two_hop_alts) > 0:
                best_alt = np.min(two_hop_alts)
                if best_alt > 0:
                    # Quality: direct vs best alternative
                    quality = direct_dist / best_alt
                    # Transform to [0.3, 1.7] range with better direct edges getting higher scores
                    if quality < 1.0:
                        # Direct edge is shorter than alternative
                        shortcut_quality[i, j] = 1.0 + 0.7 * (1.0 - quality)
                    else:
                        # Alternative is shorter
                        shortcut_quality[i, j] = 0.3 + 0.7 * (1.0 / quality)
                else:
                    shortcut_quality[i, j] = 1.0
            else:
                shortcut_quality[i, j] = 1.0
    
    # Symmetrize
    shortcut_quality = 0.5 * (shortcut_quality + shortcut_quality.T)
    
    # ==================== 4. Graph-Theoretic Edge Criticality ====================
    # Compute approximate edge betweenness in k-NN graph
    edge_criticality = np.ones_like(distance_matrix)
    
    if n <= 100:  # Only compute for manageable sizes
        # Build sparse adjacency matrix from mask
        row_indices, col_indices = np.where(adjacency_mask)
        data = distance_matrix[row_indices, col_indices]
        
        if len(data) > 0:
            # Compute all-pairs shortest paths in k-NN graph
            try:
                # Use Dijkstra on the k-NN graph
                dist_matrix = np.full((n, n), np.inf)
                dist_matrix[row_indices, col_indices] = data
                np.fill_diagonal(dist_matrix, 0)
                
                # Sample nodes for betweenness approximation
                sample_size = min(20, n)
                sample_nodes = np.random.choice(n, sample_size, replace=False)
                
                betweenness_counts = np.zeros((n, n))
                
                for s in sample_nodes:
                    # Run Dijkstra from source s
                    dist_from_s = np.full(n, np.inf)
                    dist_from_s[s] = 0
                    prev = {s: -1}
                    
                    # Simple Dijkstra implementation
                    unvisited = set(range(n))
                    
                    while unvisited:
                        # Find unvisited node with minimum distance
                        current = min(unvisited, key=lambda x: dist_from_s[x])
                        unvisited.remove(current)
                        
                        if dist_from_s[current] == np.inf:
                            break
                        
                        # Update neighbors
                        for neighbor in np.where(adjacency_mask[current])[0]:
                            if neighbor in unvisited:
                                new_dist = dist_from_s[current] + distance_matrix[current, neighbor]
                                if new_dist < dist_from_s[neighbor]:
                                    dist_from_s[neighbor] = new_dist
                                    prev[neighbor] = current
                
                # Count edge usage in shortest paths (simplified approximation)
                for i in range(n):
                    for j in range(np.where(adjacency_mask[i])[0]):
                        if i < j:
                            # Simple heuristic: count how many sampled nodes have i and j as important connectors
                            count = 0
                            for s in sample_nodes:
                                if (abs(dist_from_s[i] - dist_from_s[j]) < 
                                    distance_matrix[i, j] + 1e-6):
                                    count += 1
                            edge_criticality[i, j] = edge_criticality[j, i] = 1.0 + 0.5 * (count / sample_size)
            except:
                # Fallback to degree-based criticality
                degrees = adjacency_mask.sum(axis=1)
                for i in range(n):
                    for j in range(i+1, n):
                        if adjacency_mask[i, j]:
                            crit = 1.0 + 0.3 * ((degrees[i] + degrees[j]) / (2 * global_k))
                            edge_criticality[i, j] = edge_criticality[j, i] = min(crit, 1.5)
    
    # ==================== 5. Meta-Heuristic Component Fusion ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Compute adaptive weights based on problem characteristics
    avg_degree = adjacency_mask.sum() / n
    density_variation = np.std(local_density) / (np.mean(local_density) + 1e-10)
    
    # Weight adaptation: more weight to regret in sparse graphs, to criticality in dense clusters
    w_regret = 0.5 + 0.3 * (1.0 - min(avg_degree / 10, 1.0))
    w_shortcut = 0.4 + 0.3 * min(density_variation, 1.0)
    w_criticality = 0.3 + 0.4 * min(avg_degree / 15, 1.0)
    
    # Normalize weights
    total_weight = w_regret + w_shortcut + w_criticality
    w_regret /= total_weight
    w_shortcut /= total_weight
    w_criticality /= total_weight
    
    # Combine components
    for i in range(n):
        for j in range(i+1, n):
            if adjacency_mask[i, j]:
                # Base inverse distance component
                inv_dist = 1.0 / (distance_matrix[i, j] + 1e-10)
                
                # Weighted combination
                score = (inv_dist * 
                        (w_regret * regret_scores[i, j] + 
                         w_shortcut * shortcut_quality[i, j] + 
                         w_criticality * edge_criticality[i, j]))
                
                heuristics[i, j] = heuristics[j, i] = score
    
    # ==================== 6. Final Refinement ====================
    np.fill_diagonal(heuristics, 0)
    
    # Ensure all edges in mask have non-zero scores
    min_nonzero = heuristics[heuristics > 0].min() if np.any(heuristics > 0) else 1e-8
    heuristics[adjacency_mask] = np.maximum(heuristics[adjacency_mask], min_nonzero)
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Symmetrize with geometric mean for consistency
    heuristics_sym = np.sqrt(np.maximum(heuristics * heuristics.T, 1e-16))
    np.fill_diagonal(heuristics_sym, 0)
    
    # Final normalization
    row_sums = heuristics_sym.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics_sym = heuristics_sym / row_sums
    
    return heuristics_sym
