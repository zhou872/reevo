import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with dynamic sparsity 
    and local cluster-aware two-hop analysis. Combines lightweight regret
    metrics with cluster-based penalty adaptation for robust performance.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Dynamic Sparsity with Cluster Awareness ====================
    # Adaptive k based on problem size and instance characteristics
    base_k = max(3, int(np.log2(n)))
    
    # Precompute nearest neighbors for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_dists = np.sort(distance_matrix, axis=1)
    
    # Compute local density using adaptive neighborhood
    local_density = np.zeros(n)
    density_k = min(5, n-1)
    for i in range(n):
        if density_k > 0:
            local_density[i] = 1.0 / (np.mean(sorted_dists[i, 1:density_k+1]) + 1e-10)
    
    # Adaptive sparsification based on local density
    adjacency_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        # Adjust k based on local density
        density_factor = 0.5 + 1.5 * (local_density[i] / np.max(local_density))
        adaptive_k = min(n-1, max(3, int(base_k * density_factor)))
        
        # Select neighbors
        neighbors = sorted_indices[i, 1:adaptive_k+1]
        adjacency_mask[i, neighbors] = True
    
    # Ensure bidirectionality for robustness
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Lightweight Regret Scoring ====================
    regret_scores = np.ones((n, n))
    
    # Compute nearest and second-nearest distances
    nearest_dists = sorted_dists[:, 1]
    second_nearest = np.zeros(n)
    for i in range(n):
        if n > 2:
            second_nearest[i] = sorted_dists[i, 2] if len(sorted_dists[i]) > 2 else np.inf
        else:
            second_nearest[i] = np.inf
    
    # Compute regret for each edge in mask
    for i in range(n):
        i_neighbors = np.where(adjacency_mask[i])[0]
        if len(i_neighbors) == 0:
            continue
            
        for j in i_neighbors:
            if i >= j:
                continue
                
            dist_ij = distance_matrix[i, j]
            
            # Base regret: compare to nearest neighbor
            regret_i = 1.0 if abs(dist_ij - nearest_dists[i]) < 1e-10 else nearest_dists[i] / (dist_ij + 1e-10)
            regret_j = 1.0 if abs(dist_ij - nearest_dists[j]) < 1e-10 else nearest_dists[j] / (dist_ij + 1e-10)
            
            # Combined regret with symmetry
            regret_ij = 0.5 * (regret_i + regret_j)
            
            # Additional regret for sparse regions
            if local_density[i] < np.median(local_density) or local_density[j] < np.median(local_density):
                # Boost regret in sparse regions
                regret_ij = min(1.5, regret_ij * 1.2)
            
            regret_scores[i, j] = regret_scores[j, i] = regret_ij
    
    # ==================== 3. Cluster-Aware Two-Hop Analysis ====================
    two_hop_quality = np.ones((n, n))
    
    # Precompute cluster membership approximation
    cluster_scores = np.zeros(n)
    for i in range(n):
        i_neighbors = np.where(adjacency_mask[i])[0]
        if len(i_neighbors) > 1:
            # Count mutual connections among neighbors
            mutual_count = 0
            for a in range(len(i_neighbors)):
                for b in range(a+1, len(i_neighbors)):
                    if adjacency_mask[i_neighbors[a], i_neighbors[b]]:
                        mutual_count += 1
            
            max_possible = len(i_neighbors) * (len(i_neighbors) - 1) // 2
            cluster_scores[i] = mutual_count / (max_possible + 1e-10)
    
    # Analyze two-hop alternatives
    for i in range(n):
        i_neighbors = np.where(adjacency_mask[i])[0]
        if len(i_neighbors) == 0:
            continue
            
        for j in i_neighbors:
            if i >= j:
                continue
                
            direct_dist = distance_matrix[i, j]
            
            # Find common neighbors efficiently
            common_mask = adjacency_mask[i] & adjacency_mask[j]
            common_mask[i] = False
            common_mask[j] = False
            common_nodes = np.where(common_mask)[0]
            
            if len(common_nodes) > 0:
                # Compute best two-hop alternative
                two_hop_dists = distance_matrix[i, common_nodes] + distance_matrix[common_nodes, j]
                best_alt = np.min(two_hop_dists)
                
                if best_alt > 0:
                    ratio = direct_dist / best_alt
                    
                    # Adjust penalty based on local clustering
                    cluster_factor = 0.5 * (cluster_scores[i] + cluster_scores[j])
                    
                    if ratio < 0.9:
                        # Direct edge is a good shortcut
                        quality = 1.2 + 0.3 * (1.0 - ratio) * (1.0 + cluster_factor)
                    elif ratio < 1.1:
                        # Competitive alternative exists
                        if cluster_factor > 0.5:
                            # In dense clusters, penalize competitive edges more
                            quality = 0.8 - 0.2 * cluster_factor
                        else:
                            quality = 0.9
                    else:
                        # Direct edge is significantly better
                        quality = 1.0 + 0.2 * np.tanh(ratio - 1.1)
                    
                    two_hop_quality[i, j] = two_hop_quality[j, i] = quality
            else:
                # No common neighbors - edge might be a bridge
                # Use density to assess bridge importance
                density_factor = 0.5 * (local_density[i] + local_density[j]) / np.max(local_density)
                two_hop_quality[i, j] = two_hop_quality[j, i] = 1.0 + 0.3 * density_factor
    
    # ==================== 4. Inverse Distance with Density Weighting ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    # Instance-level statistics
    instance_density = np.mean(local_density)
    density_std = np.std(local_density)
    
    for i in range(n):
        neighbors = np.where(adjacency_mask[i])[0]
        if len(neighbors) == 0:
            continue
            
        # Adaptive weighting based on local vs global density
        density_ratio = local_density[i] / instance_density
        weight_factor = 0.5 + 0.5 * np.tanh(density_ratio - 1.0)
        
        # Compute weighted inverse distances
        for j in neighbors:
            dist = distance_matrix[i, j]
            # Density-aware scaling
            scale = 1.0 + weight_factor * (1.0 - np.exp(-density_std))
            inv_dist[i, j] = scale / (dist + 1e-10)
    
    # Symmetrize
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    
    # ==================== 5. Adaptive Component Fusion ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Dynamic weighting based on instance characteristics
    if n < 50:
        # Small instances: focus on regret and two-hop quality
        w_regret = 0.4
        w_two_hop = 0.4
        w_inv_dist = 0.2
    else:
        # Larger instances: balance based on density variation
        density_variation = density_std / instance_density
        w_inv_dist = 0.3 + 0.2 * (1.0 - density_variation)
        w_regret = 0.3 + 0.2 * density_variation
        w_two_hop = 0.4
    
    # Normalize weights
    total = w_inv_dist + w_regret + w_two_hop
    w_inv_dist /= total
    w_regret /= total
    w_two_hop /= total
    
    # Combine components for edges in mask
    for i in range(n):
        for j in range(i+1, n):
            if adjacency_mask[i, j]:
                combined = (
                    w_inv_dist * inv_dist[i, j] +
                    w_regret * regret_scores[i, j] +
                    w_two_hop * two_hop_quality[i, j]
                )
                
                # Local adjustment based on cluster structure
                cluster_adj = 0.5 * (cluster_scores[i] + cluster_scores[j])
                if cluster_adj > 0.7:
                    # In dense clusters, slightly boost intra-cluster edges
                    combined *= 1.1
                
                heuristics[i, j] = heuristics[j, i] = combined
    
    # ==================== 6. Final Processing ====================
    np.fill_diagonal(heuristics, 0)
    
    # Ensure minimum values for edges in mask
    mask_values = heuristics[adjacency_mask]
    if len(mask_values) > 0:
        min_val = np.min(mask_values[mask_values > 0]) if np.any(mask_values > 0) else 1e-8
        heuristics[adjacency_mask] = np.maximum(heuristics[adjacency_mask], min_val)
    
    # Row normalization with stability
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Final symmetrization
    geometric_mean = np.sqrt(np.maximum(heuristics * heuristics.T, 1e-16))
    heuristics = np.where((heuristics > 0) & (heuristics.T > 0), 
                          geometric_mean, 
                          0.5 * (heuristics + heuristics.T))
    
    np.fill_diagonal(heuristics, 0)
    
    # Final row normalization
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    return heuristics
