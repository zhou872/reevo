import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Sparsified nearest-neighbor ratios (inverse distance weighted by relative rank)
    2. Minimum Spanning Tree edge bonus
    3. Two-hop path density scoring
    4. Degree-constrained clustering consideration
    """
    n = len(distance_matrix)
    if n < 3:
        return np.ones_like(distance_matrix) / (distance_matrix + np.eye(n))
    
    # Initialize heuristic matrix
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # 1. Compute sparsified nearest-neighbor ratios
    # For each node, find k nearest neighbors (k ~ sqrt(n))
    k = max(2, int(np.sqrt(n)))
    
    # Create adjacency mask for promising edges
    adjacency_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        distances = distance_matrix[i].copy()
        distances[i] = np.inf  # Exclude self
        nearest_indices = np.argpartition(distances, k)[:k]
        adjacency_mask[i, nearest_indices] = True
    
    # Make adjacency symmetric (undirected graph)
    adjacency_mask = adjacency_mask | adjacency_mask.T
    
    # Calculate inverse distance weighted by rank
    inv_dist = np.zeros_like(distance_matrix)
    inv_dist[adjacency_mask] = 1.0 / (distance_matrix[adjacency_mask] + 1e-10)
    
    # Normalize by the number of competing edges for each node
    node_normalization = adjacency_mask.sum(axis=1, keepdims=True)
    inv_dist = inv_dist / (node_normalization + 1e-10)
    inv_dist = inv_dist + inv_dist.T  # Symmetrize
    
    # 2. Minimum Spanning Tree edge bonus using Prim's algorithm
    mst_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Prim's algorithm for MST
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    min_dist = distance_matrix[0].copy()
    min_dist[0] = np.inf
    
    for _ in range(n - 1):
        # Find minimum edge from visited to unvisited
        u = np.argmin(min_dist)
        visited[u] = True
        
        # Find the visited node that connects to u
        for v in range(n):
            if visited[v] and np.isclose(distance_matrix[v, u], min_dist[u]):
                mst_mask[v, u] = True
                mst_mask[u, v] = True
                break
        
        # Update min_dist for unvisited nodes
        for v in range(n):
            if not visited[v] and distance_matrix[u, v] < min_dist[v]:
                min_dist[v] = distance_matrix[u, v]
    
    # MST bonus factor
    mst_bonus = 1.5
    
    # 3. Two-hop path density scoring
    # For each edge (i,j), compute how many two-hop paths i-k-j exist
    two_hop_density = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Find common neighbors
                common_neighbors = np.where(adjacency_mask[i] & adjacency_mask[j])[0]
                if len(common_neighbors) > 0:
                    # Average two-hop distance through common neighbors
                    two_hop_dist = []
                    for k in common_neighbors:
                        two_hop_dist.append(distance_matrix[i, k] + distance_matrix[k, j])
                    avg_two_hop = np.mean(two_hop_dist) if two_hop_dist else np.inf
                    
                    # Score is better if direct edge is much shorter than two-hop
                    direct_dist = distance_matrix[i, j]
                    if avg_two_hop > 0:
                        two_hop_density[i, j] = two_hop_density[j, i] = avg_two_hop / direct_dist
    
    # Normalize two-hop density
    if two_hop_density.max() > 0:
        two_hop_density = two_hop_density / two_hop_density.max()
    
    # 4. Degree-constrained clustering consideration
    # Nodes that are far from others get higher penalty
    node_isolation = np.zeros(n)
    for i in range(n):
        # Average distance to k nearest neighbors
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nearest_dists = np.partition(distances, k)[:k]
        node_isolation[i] = np.mean(nearest_dists)
    
    # Normalize isolation scores
    if node_isolation.max() > 0:
        node_isolation = node_isolation / node_isolation.max()
    
    # Combine all factors
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Base score from inverse distance
                score = inv_dist[i, j]
                
                # Apply MST bonus
                if mst_mask[i, j]:
                    score *= mst_bonus
                
                # Apply two-hop density factor
                score *= (1.0 + two_hop_density[i, j] * 0.3)
                
                # Apply isolation penalty (edges between isolated nodes get boost)
                isolation_factor = 1.0 + 0.2 * (node_isolation[i] + node_isolation[j])
                score *= isolation_factor
                
                heuristics[i, j] = heuristics[j, i] = score
    
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Normalize final heuristics
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
