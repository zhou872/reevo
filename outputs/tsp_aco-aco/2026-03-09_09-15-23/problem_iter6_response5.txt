import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic function for TSP edge selection with adaptive sparsification,
    regret-based scoring, and efficient multi-hop analysis.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive K-NN Sparsification ====================
    # Base k scales logarithmically but adapts to local density variance
    base_k = max(3, int(np.log2(n)))
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Calculate local density statistics
        nearest_dists = np.partition(distances, min(base_k, n-1))[:min(base_k, n-1)]
        mean_dist = np.mean(nearest_dists)
        var_dist = np.var(nearest_dists) if len(nearest_dists) > 1 else 0
        
        # Adaptive k based on density variance (more neighbors in varied regions)
        if mean_dist > 0:
            cv = np.sqrt(var_dist) / mean_dist
            adaptive_k = min(n-1, int(base_k * (1.0 + 0.5 * cv)))
        else:
            adaptive_k = base_k
        
        # Get adaptive_k nearest neighbors
        nn_indices = np.argpartition(distances, adaptive_k)[:adaptive_k]
        adjacency_mask[i, nn_indices] = True
        
        # Ensure reciprocal connections for dense regions
        if cv > 0.5:  # High variance indicates boundary region
            for j in nn_indices:
                if not adjacency_mask[j, i]:
                    # Check if i is in top adaptive_k of j's neighbors
                    j_distances = distance_matrix[j].copy()
                    j_distances[j] = np.inf
                    j_k = min(adaptive_k, n-1)
                    j_nn = np.argpartition(j_distances, j_k)[:j_k]
                    if i in j_nn:
                        adjacency_mask[j, i] = True
    
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Regret-Based Inverse Distance Scoring ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    regret_scores = np.zeros_like(distance_matrix, dtype=float)
    
    # Precompute nearest neighbor distances for each node
    nearest_dists = np.zeros(n)
    second_nearest_dists = np.zeros(n)
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        sorted_dists = np.sort(distances)[:2]
        nearest_dists[i] = sorted_dists[0] if len(sorted_dists) > 0 else np.inf
        second_nearest_dists[i] = sorted_dists[1] if len(sorted_dists) > 1 else np.inf
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            # Get distances to neighbors
            valid_dists = distance_matrix[i, valid_mask]
            
            # Compute regret: how much worse is this edge than the best alternative?
            regret_i = np.zeros_like(valid_dists)
            for idx, d in enumerate(valid_dists):
                if nearest_dists[i] > 0:
                    # Regret relative to nearest neighbor (except if this IS the nearest)
                    is_nearest = abs(d - nearest_dists[i]) < 1e-10
                    if is_nearest:
                        # For the nearest edge, regret is how much better it is than second best
                        regret_i[idx] = 1.0 + (nearest_dists[i] / (second_nearest_dists[i] + 1e-10))
                    else:
                        # For other edges, regret is how much worse than the nearest
                        regret_i[idx] = 1.0 / (1.0 + (d - nearest_dists[i]) / nearest_dists[i])
                else:
                    regret_i[idx] = 1.0
            
            # Rank-based weights with regret adjustment
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            weights = 1.0 / ranks
            weights = weights * regret_i  # Boost weights for high-regret edges
            weights = weights / np.sum(weights)
            
            # Weighted inverse distance
            inv_vals = 1.0 / (valid_dists + 1e-10)
            inv_dist[i, valid_mask] = inv_vals * weights
            
            # Store regret scores for combination
            regret_scores[i, valid_mask] = regret_i
    
    # Symmetrize inverse distance
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    
    # ==================== 3. Fast Multi-Hop Shortcut Detection ====================
    shortcut_quality = np.ones_like(distance_matrix)
    
    # Precompute sorted neighbors for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        valid_indices = np.where(valid_mask)[0]
        
        for j in valid_indices:
            if i >= j:
                continue
            
            direct_dist = distance_matrix[i, j]
            if direct_dist < 1e-10:
                continue
            
            # Check 2-hop alternatives through common neighbors
            i_neighbors = set(np.where(adjacency_mask[i])[0])
            j_neighbors = set(np.where(adjacency_mask[j])[0])
            common = list(i_neighbors.intersection(j_neighbors))
            
            if common:
                # Vectorized computation of 2-hop distances
                two_hop_dists = distance_matrix[i, common] + distance_matrix[common, j]
                if len(two_hop_dists) > 0:
                    min_two_hop = np.min(two_hop_dists)
                    
                    # Quality measure: how good is direct edge compared to best 2-hop?
                    if min_two_hop > direct_dist:
                        # Direct edge is a shortcut
                        shortcut_ratio = direct_dist / min_two_hop
                        shortcut_quality[i, j] = shortcut_quality[j, i] = 1.0 + 0.5 * shortcut_ratio
                    else:
                        # Direct edge is not a shortcut
                        penalty = 0.7 + 0.3 * (direct_dist / (min_two_hop + 1e-10))
                        shortcut_quality[i, j] = shortcut_quality[j, i] = np.clip(penalty, 0.3, 1.0)
    
    # ==================== 4. Local Density Variance Penalty ====================
    density_penalty = np.ones_like(distance_matrix)
    
    # Compute local density for each node
    local_density = np.zeros(n)
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        k = min(5, n-1)
        nearest = np.partition(distances, k)[:k]
        local_density[i] = 1.0 / (np.mean(nearest) + 1e-10)
    
    # Normalize densities
    if local_density.max() > 0:
        local_density = local_density / local_density.max()
    
    # Apply penalty based on density mismatch
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                density_diff = abs(local_density[i] - local_density[j])
                # Penalize edges between nodes with very different densities
                penalty = 1.0 - 0.3 * density_diff
                density_penalty[i, j] = density_penalty[j, i] = penalty
    
    # ==================== 5. Combine All Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Base score from regret-weighted inverse distance
                score = inv_dist[i, j]
                
                # Boost by regret (edges that are much better than alternatives)
                regret_boost = 1.0 + 0.2 * (regret_scores[i, j] - 1.0)
                
                # Apply shortcut quality and density penalty
                score = score * regret_boost * shortcut_quality[i, j] * density_penalty[i, j]
                
                heuristics[i, j] = heuristics[j, i] = score
    
    # ==================== 6. Final Normalization ====================
    np.fill_diagonal(heuristics, 0)
    
    # Add small constant to avoid zeros
    min_nonzero = heuristics[heuristics > 0].min() if np.any(heuristics > 0) else 1e-8
    heuristics[adjacency_mask] = np.maximum(heuristics[adjacency_mask], min_nonzero)
    
    # Row-normalize for probability distribution
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Symmetrize
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
