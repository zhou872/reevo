import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristic function with adaptive sparsification and smooth
    regret-based scoring for efficient stochastic sampling.
    
    Args:
        distance_matrix: Square matrix of pairwise distances between n nodes
        
    Returns:
        Heuristic matrix of same shape, with higher values indicating more promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. ADAPTIVE LOGARITHMIC SPARSIFICATION ====================
    # Base neighborhood size grows logarithmically with problem size
    k = max(4, int(np.log2(n)) + 2)
    
    # Precompute sorted indices for efficient neighbor lookups
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_distances = np.sort(distance_matrix, axis=1)
    
    # Build adjacency mask with reciprocal connections for better connectivity
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nn_indices = sorted_indices[i, 1:k+1]
        adjacency_mask[i, nn_indices] = True
        
        # Ensure reciprocal connections when i is in j's top k neighborhood
        for j in nn_indices:
            j_nn = sorted_indices[j, 1:k+1]
            if i in j_nn:
                adjacency_mask[j, i] = True
    
    # Ensure symmetry and remove self-loops
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. VECTORIZED RANK-WEIGHTED DISTANCE SCORING ====================
    inv_dist_scores = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if not np.any(valid_mask):
            continue
            
        # Get distances to valid neighbors
        valid_dists = distance_matrix[i, valid_mask]
        neighbor_indices = np.where(valid_mask)[0]
        
        # Compute rank-based weights with exponential decay
        ranks = np.argsort(valid_dists)  # 0 for closest, 1 for next, etc.
        weights = np.exp(-0.5 * ranks)   # Exponential decay by rank
        weights = weights / np.sum(weights)  # Normalize
        
        # Inverse distance with rank weighting
        inv_vals = 1.0 / (valid_dists + 1e-10)
        
        # Incorporate local density using smooth tanh transformation
        neighbor_density = np.mean(inv_vals)
        density_factor = 1.0 + 0.3 * np.tanh(neighbor_density)  # Smooth saturation
        
        inv_dist_scores[i, neighbor_indices] = inv_vals * weights * density_factor
    
    # Symmetrize by averaging both directions
    inv_dist_scores = 0.5 * (inv_dist_scores + inv_dist_scores.T)
    
    # ==================== 3. VECTORIZED TWO-HOP REGRET ANALYSIS ====================
    regret_penalties = np.ones_like(distance_matrix, dtype=float)
    
    # Efficient regret computation using top neighbor intersection
    m = max(5, int(np.sqrt(n) * 1.5))  # Number of top neighbors to consider
    top_neighbors = sorted_indices[:, 1:m+1]
    
    # Process each edge in the adjacency mask
    edge_indices = np.where(np.triu(adjacency_mask))
    for i, j in zip(edge_indices[0], edge_indices[1]):
        # Find common neighbors in top neighborhoods
        i_top_set = set(top_neighbors[i])
        j_top_set = set(top_neighbors[j])
        common = list(i_top_set.intersection(j_top_set) - {i, j})
        
        if len(common) > 0:
            common_arr = np.array(common)
            # Vectorized two-hop distance computation
            two_hop_dists = (distance_matrix[i, common_arr] + 
                           distance_matrix[common_arr, j])
            min_two_hop = np.min(two_hop_dists)
            direct_dist = distance_matrix[i, j]
            
            if direct_dist > 1e-10:
                # Compute normalized regret
                regret = (min_two_hop - direct_dist) / direct_dist
                
                # Smooth sigmoid-based penalty mapping
                # Higher regret (better direct edge) → lower penalty
                penalty = 0.6 + 0.4 / (1.0 + np.exp(-2.0 * regret))
                penalty = np.clip(penalty, 0.2, 1.0)
                
                regret_penalties[i, j] = regret_penalties[j, i] = penalty
    
    # ==================== 4. PERCENTILE-BASED CLUSTER DETECTION ====================
    cluster_factors = np.ones_like(distance_matrix, dtype=float)
    
    if n > 10:
        # Use 30th percentile as local distance threshold (more stable than median)
        local_thresholds = np.percentile(distance_matrix, 30, axis=1)
        
        # Process edges in adjacency mask
        for i, j in zip(edge_indices[0], edge_indices[1]):
            direct_dist = distance_matrix[i, j]
            # Check if edge is significantly shorter than local thresholds
            is_local = (direct_dist < 0.8 * local_thresholds[i] and 
                       direct_dist < 0.8 * local_thresholds[j])
            
            if is_local:
                # Exponential boost for local connections
                avg_threshold = 0.5 * (local_thresholds[i] + local_thresholds[j])
                boost = 1.0 + 0.3 * np.exp(-direct_dist / (0.6 * avg_threshold))
                cluster_factors[i, j] = cluster_factors[j, i] = boost
    
    # ==================== 5. SMOOTH COMPONENT FUSION ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Combine components with multiplication (preserves relative scales)
    for i, j in zip(edge_indices[0], edge_indices[1]):
        score = inv_dist_scores[i, j] * regret_penalties[i, j] * cluster_factors[i, j]
        
        # Add minimal random component for tie-breaking (deterministic seed)
        score += 1e-9 * ((i * 37 + j * 79) % 100) / 100.0
        
        heuristics[i, j] = heuristics[j, i] = score
    
    # ==================== 6. FINAL NORMALIZATION ====================
    np.fill_diagonal(heuristics, 0)
    
    # Row-wise softmax for probability distribution
    for i in range(n):
        row = heuristics[i]
        if np.any(row > 0):
            # Stable softmax
            row_max = np.max(row)
            exp_row = np.exp((row - row_max) / (row_max + 1e-10))
            heuristics[i] = exp_row / np.sum(exp_row)
    
    # Final symmetrization
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
