import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with improved local consistency
    and adaptive edge competition analysis.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive Symmetric Sparsification ====================
    # Use logarithmic scaling for sparsification while ensuring mutual consistency
    k = max(3, int(np.log2(n)))
    
    # Create initial adjacency from mutual nearest neighbors
    adjacency_mask = np.zeros((n, n), dtype=bool)
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        # Get k nearest neighbors for each node
        i_nn = sorted_indices[i, 1:k+1]
        
        # Mark edges where both endpoints consider each other among their k nearest
        for j in i_nn:
            j_nn = sorted_indices[j, 1:k+1]
            if i in j_nn:
                adjacency_mask[i, j] = True
    
    # Ensure symmetry and no self-loops
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Enhanced Distance Scoring ====================
    # Compute distance-based scores with neighbor consistency weighting
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            # Get distances to neighbors
            valid_dists = distance_matrix[i, valid_mask]
            
            # Compute enhanced scoring using squared rank weighting
            # This gives higher emphasis to closer neighbors
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            # Square weights for stronger emphasis on top ranks
            weights = 1.0 / (ranks ** 1.5)
            weights = weights / np.sum(weights)
            
            # Apply weighted inverse distance with soft minimum scaling
            inv_vals = 1.0 / (valid_dists + 1e-10)
            inv_dist[i, valid_mask] = inv_vals * weights
    
    # Symmetrize by geometric mean (better than max for consistency)
    inv_dist_sym = np.sqrt(np.maximum(inv_dist, 1e-10) * np.maximum(inv_dist.T, 1e-10))
    inv_dist = inv_dist_sym
    
    # ==================== 3. Enhanced Two-Hop Competition ====================
    # More comprehensive two-hop analysis with adaptive neighbor selection
    two_hop_penalty = np.ones_like(distance_matrix)
    
    # Pre-compute k2 nearest neighbors for two-hop analysis
    k2 = max(4, int(np.sqrt(n)))
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        valid_indices = np.where(valid_mask)[0]
        
        for j in valid_indices:
            if i >= j:
                continue
                
            # Get top neighbors for both endpoints
            i_neighbors = sorted_indices[i, 1:k2+1]
            j_neighbors = sorted_indices[j, 1:k2+1]
            
            # Consider all potential intermediate nodes
            candidate_nodes = np.unique(np.concatenate([i_neighbors, j_neighbors]))
            candidate_nodes = candidate_nodes[candidate_nodes != i]
            candidate_nodes = candidate_nodes[candidate_nodes != j]
            
            if len(candidate_nodes) > 0:
                # Vectorized two-hop distance computation
                two_hop_dists = distance_matrix[i, candidate_nodes] + distance_matrix[candidate_nodes, j]
                
                if len(two_hop_dists) > 0:
                    min_two_hop = np.min(two_hop_dists)
                    direct_dist = distance_matrix[i, j]
                    
                    if direct_dist > 1e-10 and min_two_hop < np.inf:
                        ratio = min_two_hop / direct_dist
                        
                        # Enhanced penalty function with smoother transition
                        # Strongly penalize edges with good two-hop alternatives
                        if ratio < 1.0:
                            # Two-hop is better than direct
                            penalty = 0.2 + 0.4 * ratio
                        elif ratio < 1.5:
                            # Two-hop is slightly worse
                            penalty = 0.6 + 0.4 * (ratio - 1.0) / 0.5
                        else:
                            # Two-hop is significantly worse
                            penalty = 1.0
                        
                        two_hop_penalty[i, j] = two_hop_penalty[j, i] = penalty
    
    # ==================== 4. Combined Scoring with Local Normalization ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        valid_indices = np.where(valid_mask)[0]
        
        if len(valid_indices) > 0:
            # Compute combined scores for all valid edges from i
            base_scores = inv_dist[i, valid_indices]
            penalties = two_hop_penalty[i, valid_indices]
            combined = base_scores * penalties
            
            # Apply local normalization (softmax-like scaling)
            # Add small constant and scale by maximum in neighborhood
            combined = combined + 1e-8
            max_score = np.max(combined)
            if max_score > 0:
                combined = combined / max_score
            
            heuristics[i, valid_indices] = combined
    
    # ==================== 5. Final Symmetrization and Refinement ====================
    # Ensure symmetry by taking average
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # Apply final sparsification: keep only edges with score above threshold
    threshold = np.percentile(heuristics[heuristics > 0], 30) if np.any(heuristics > 0) else 0
    heuristics[heuristics < threshold] = 0
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Final symmetrization
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
