import numpy as np
from scipy.spatial import KDTree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection probability.
    Returns a sparse matrix with non-zero values only for promising edges.
    """
    n = distance_matrix.shape[0]
    heuristics = np.zeros((n, n))
    
    # Basic inverse distance component
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
    
    # 1. Nearest neighbor regret component
    # For each node, find regret of not choosing the nearest neighbor
    regret = np.zeros((n, n))
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf  # Ignore self
        sorted_idx = np.argsort(row)
        
        # Regret is difference between edge distance and nearest neighbor distance
        nearest_dist = row[sorted_idx[0]]
        for j in range(n):
            if i != j:
                regret[i, j] = nearest_dist / max(distance_matrix[i, j], 1e-10)
    
    # 2. Local connectivity component - favor edges that connect to nodes with few good options
    degree_penalty = np.zeros(n)
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        # Count how many edges are within 1.5x of the minimum distance
        min_dist = np.min(row)
        threshold = 1.5 * min_dist
        degree_penalty[i] = np.sum(row <= threshold)
    
    # 3. Triangle inequality bonus - edges that complete promising triangles
    triangle_bonus = np.zeros((n, n))
    if n >= 3:
        for i in range(n):
            for j in range(i + 1, n):
                if i != j:
                    # Find the best third node that forms a triangle
                    other_nodes = [k for k in range(n) if k not in (i, j)]
                    if other_nodes:
                        # Look for triangles where this edge is part of a good local tour
                        triangle_scores = []
                        for k in other_nodes:
                            # Triangle inequality ratio
                            detour = distance_matrix[i, k] + distance_matrix[k, j]
                            direct = distance_matrix[i, j]
                            if direct > 0:
                                ratio = detour / direct
                                triangle_scores.append(1.0 / ratio if ratio > 0 else 0)
                        if triangle_scores:
                            triangle_bonus[i, j] = np.max(triangle_scores)
                            triangle_bonus[j, i] = triangle_bonus[i, j]
    
    # Combine components with weighted sum
    heuristics = (
        0.5 * inv_dist +                    # Inverse distance (base component)
        0.3 * regret +                      # Regret component
        0.1 * triangle_bonus                # Triangle bonus
    )
    
    # Apply degree-based sparsification
    for i in range(n):
        if degree_penalty[i] > 0:
            # Keep more edges for isolated nodes, fewer for well-connected nodes
            k = max(2, min(5, n // (degree_penalty[i] + 1)))
            
            # Get the top k edges for this node
            row_heuristics = heuristics[i].copy()
            row_heuristics[i] = -np.inf  # Ignore self
            
            # Find indices of top k values
            if k < n - 1:
                top_k_indices = np.argpartition(row_heuristics, -k)[-k:]
                
                # Create mask for this row
                mask = np.zeros(n, dtype=bool)
                mask[top_k_indices] = True
                mask[i] = False  # Ensure diagonal remains zero
                
                # Zero out non-promising edges
                heuristics[i, ~mask] = 0
    
    # Ensure symmetry and zero diagonal
    heuristics = np.maximum(heuristics, heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # Normalize to avoid numerical issues
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
