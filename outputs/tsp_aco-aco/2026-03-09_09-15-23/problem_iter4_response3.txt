import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse import csr_matrix
import heapq

def heuristics_v2(distance_matrix: np.ndarray,
                  density_aware: bool = True,
                  use_centrality: bool = True,
                  approx_two_hop: bool = True) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge prioritization combining graph-theoretic
    insights with adaptive weighting and scalable approximations.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        density_aware: If True, adapt neighborhood sizes based on local density
        use_centrality: If True, compute betweenness centrality approximation
        approx_two_hop: If True, use approximation for large instances (>200 nodes)
    
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Multi-Scale Adaptive Neighborhood ====================
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    if density_aware:
        # Dual-scale neighborhood: small k for immediate neighbors, larger for sparse regions
        base_k = max(2, int(np.sqrt(n)))
        extended_k = min(n-1, base_k * 3)
        
        for i in range(n):
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            
            # Primary neighborhood for dense connectivity
            primary_indices = np.argpartition(distances, base_k)[:base_k]
            adjacency_mask[i, primary_indices] = True
            
            # Extended neighborhood for sparse region connectivity
            sorted_dists = np.sort(distances)
            dist_std = np.std(sorted_dists[:base_k])
            dist_mean = np.mean(sorted_dists[:base_k])
            
            if dist_std > 1e-10:
                cv = dist_std / (dist_mean + 1e-10)
                if cv > 0.5:  # High variability indicates sparse regions
                    secondary_indices = np.argpartition(distances, extended_k)[:extended_k]
                    adjacency_mask[i, secondary_indices] = True
    else:
        # Fallback to fixed neighborhood
        k = max(2, int(np.sqrt(n)))
        for i in range(n):
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            nn_indices = np.argpartition(distances, k)[:k]
            adjacency_mask[i, nn_indices] = True
    
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Betweenness Centrality Approximation ====================
    centrality_scores = np.zeros((n, n), dtype=float)
    
    if use_centrality and n > 10:
        # Approximate betweenness using random sampling of shortest paths
        # For each node, find shortest paths to k random targets
        sample_size = min(20, max(5, n // 10))
        targets = np.random.choice(n, size=min(sample_size, n), replace=False)
        
        for source in range(n):
            # Dijkstra's algorithm for shortest paths
            dist = np.full(n, np.inf)
            dist[source] = 0
            prev = [None] * n
            visited = np.zeros(n, dtype=bool)
            
            heap = [(0, source)]
            while heap:
                d, u = heapq.heappop(heap)
                if visited[u]:
                    continue
                visited[u] = True
                
                for v in range(n):
                    if adjacency_mask[u, v] and not visited[v]:
                        new_dist = d + distance_matrix[u, v]
                        if new_dist < dist[v]:
                            dist[v] = new_dist
                            prev[v] = u
                            heapq.heappush(heap, (new_dist, v))
            
            # Count edge usage for paths to sampled targets
            for target in targets:
                if target == source or dist[target] == np.inf:
                    continue
                    
                # Reconstruct path and count edges
                path = []
                current = target
                while current is not None:
                    path.append(current)
                    current = prev[current]
                path.reverse()
                
                # Increment centrality for edges in this path
                for u, v in zip(path[:-1], path[1:]):
                    centrality_scores[u, v] += 1
                    centrality_scores[v, u] += 1
        
        # Normalize centrality scores
        if np.max(centrality_scores) > 0:
            centrality_scores = centrality_scores / np.max(centrality_scores)
    
    # ==================== 3. Scalable Two-Hop Analysis ====================
    two_hop_ratio = np.zeros_like(distance_matrix)
    
    if approx_two_hop and n > 200:
        # Approximate two-hop analysis using sampled intermediate nodes
        for i in range(n):
            # Sample intermediate nodes for efficiency
            sample_size = min(50, n - 2)
            other_nodes = np.random.choice([j for j in range(n) if j != i], 
                                         size=sample_size, replace=False)
            
            for j in range(i + 1, n):
                if adjacency_mask[i, j]:
                    direct_dist = distance_matrix[i, j]
                    
                    # Use only sampled intermediate nodes
                    mask = other_nodes != j
                    if np.any(mask):
                        inter_nodes = other_nodes[mask]
                        two_hop_dists = distance_matrix[i, inter_nodes] + distance_matrix[inter_nodes, j]
                        min_two_hop = np.min(two_hop_dists)
                        
                        if direct_dist > 1e-10:
                            two_hop_ratio[i, j] = two_hop_ratio[j, i] = min_two_hop / direct_dist
    else:
        # Exact two-hop analysis for smaller instances
        for i in range(n):
            for j in range(i + 1, n):
                if adjacency_mask[i, j]:
                    other_nodes = [k for k in range(n) if k not in (i, j)]
                    if other_nodes:
                        two_hop_dists = distance_matrix[i, other_nodes] + distance_matrix[other_nodes, j]
                        min_two_hop = np.min(two_hop_dists) if len(two_hop_dists) > 0 else np.inf
                        
                        direct_dist = distance_matrix[i, j]
                        if direct_dist > 1e-10 and min_two_hop < np.inf:
                            two_hop_ratio[i, j] = two_hop_ratio[j, i] = min_two_hop / direct_dist
    
    # Normalize two-hop ratios
    max_ratio = np.max(two_hop_ratio)
    if max_ratio > 0:
        two_hop_ratio = two_hop_ratio / max_ratio
    
    # ==================== 4. MST and Christofides Components ====================
    mst_mask = np.zeros((n, n), dtype=bool)
    
    if n <= 1000:  # MST feasible for reasonable sizes
        try:
            mst = minimum_spanning_tree(csr_matrix(distance_matrix))
            mst_dense = mst.toarray()
            mst_mask = (mst_dense > 0) | (mst_dense.T > 0)
        except:
            # Fallback to Prim's algorithm for large instances
            visited = np.zeros(n, dtype=bool)
            min_edge = np.full(n, np.inf)
            parent = np.full(n, -1, dtype=int)
            min_edge[0] = 0
            
            for _ in range(n):
                u = np.argmin(np.where(visited, np.inf, min_edge))
                visited[u] = True
                
                if parent[u] != -1:
                    mst_mask[u, parent[u]] = mst_mask[parent[u], u] = True
                
                for v in range(n):
                    if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                        min_edge[v] = distance_matrix[u, v]
                        parent[v] = u
    
    mst_degrees = np.sum(mst_mask, axis=1)
    odd_degree_nodes = np.where(mst_degrees % 2 == 1)[0]
    
    # ==================== 5. Dynamic Component Weighting ====================
    # Analyze problem structure to weight components
    if n < 50:
        # Small problems: emphasize exact calculations
        weights = {
            'distance': 0.4,
            'centrality': 0.25,
            'two_hop': 0.25,
            'mst': 0.1
        }
    else:
        # Larger problems: emphasize scalable approximations
        # Compute problem "regularity"
        triu_dists = distance_matrix[np.triu_indices(n, k=1)]
        dist_cv = np.std(triu_dists) / (np.mean(triu_dists) + 1e-10)
        
        if dist_cv < 0.3:
            # Regular problem: emphasize distance and MST
            weights = {
                'distance': 0.5,
                'centrality': 0.2,
                'two_hop': 0.2,
                'mst': 0.1
            }
        else:
            # Irregular problem: emphasize centrality and two-hop
            weights = {
                'distance': 0.3,
                'centrality': 0.35,
                'two_hop': 0.25,
                'mst': 0.1
            }
    
    # ==================== 6. Rank-Weighted Inverse Distance ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        row_dists = distance_matrix[i].copy()
        row_dists[i] = np.inf
        
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            valid_dists = row_dists[valid_mask]
            if len(valid_dists) > 0:
                ranks = np.argsort(np.argsort(valid_dists)) + 1
                weights_dist = 1.0 / np.sqrt(ranks)  # Square root for smoother decay
                weights_dist = weights_dist / np.sum(weights_dist)
                
                inv_vals = 1.0 / (valid_dists + 1e-10)
                inv_dist[i, valid_mask] = inv_vals * weights_dist
    
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    
    # ==================== 7. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Base distance component
                score = weights['distance'] * inv_dist[i, j]
                
                # Centrality component
                if use_centrality:
                    score += weights['centrality'] * centrality_scores[i, j]
                
                # Two-hop detour component
                if two_hop_ratio[i, j] > 0:
                    score += weights['two_hop'] * two_hop_ratio[i, j]
                
                # MST and Christofides component
                if mst_mask[i, j]:
                    mst_bonus = 2.0 if i in odd_degree_nodes and j in odd_degree_nodes else 1.5
                    score += weights['mst'] * mst_bonus
                
                # Apply local density adjustment
                # Compute simplified density factor
                local_density_i = np.mean(np.sort(distance_matrix[i])[1:4]) if n >= 4 else 1.0
                local_density_j = np.mean(np.sort(distance_matrix[j])[1:4]) if n >= 4 else 1.0
                density_factor = 0.5 + 0.5 * (local_density_i + local_density_j) / 2
                score *= density_factor
                
                heuristics[i, j] = heuristics[j, i] = score
    
    # ==================== 8. Final Processing ====================
    np.fill_diagonal(heuristics, 0)
    
    # Softmax normalization per row
    for i in range(n):
        row_max = np.max(heuristics[i])
        if row_max > 0:
            # Apply softmax with temperature
            exp_vals = np.exp(heuristics[i] / (row_max + 1e-10))
            heuristics[i] = exp_vals / np.sum(exp_vals)
    
    # Symmetrize
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # Ensure all values are non-negative
    heuristics = np.maximum(heuristics, 0)
    
    return heuristics
