import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge prioritization heuristic with adaptive sparsification,
    multi-path regret analysis, and learned embedding proxies.
    
    Key Innovations:
    1. Adaptive k-NN sparsification based on local/global density
    2. Multi-path regret penalty using top-m alternative routes
    3. Global features (MST, centrality, clustering) integration
    4. Pseudo-embedding via multi-scale distance statistics
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive Density-Based Sparsification ====================
    # Compute local and global density measures
    local_densities = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Median distance to 5 nearest neighbors as local density proxy
        k_local = min(5, n-1)
        nn_dists = np.partition(dists, k_local)[:k_local]
        local_densities[i] = np.median(nn_dists)
    
    global_density = np.mean(local_densities)
    density_variation = np.std(local_densities) / (global_density + 1e-10)
    
    # Adaptive k for each node based on local density
    base_k = max(3, int(np.log2(n)))
    k_values = np.zeros(n, dtype=int)
    for i in range(n):
        density_ratio = local_densities[i] / (global_density + 1e-10)
        # Higher k for dense regions, lower for sparse regions
        k_values[i] = max(3, min(int(base_k * (1.5 - 0.5 * density_ratio)), n-1))
    
    # Build adaptive mask
    mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        # Get k_i nearest neighbors
        nn_indices = np.argpartition(distances, k_values[i])[:k_values[i]]
        mask[i, nn_indices] = True
    
    # Ensure symmetry
    mask = mask | mask.T
    np.fill_diagonal(mask, False)
    
    # ==================== 2. Enhanced Global Features ====================
    # 2.1 Minimum Spanning Tree with edge importance weighting
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = mst > 0
    
    # Compute MST edge importance based on alternative costs
    mst_importance = np.ones_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if mst_mask[i, j]:
                # Find minimal alternative connecting the two components
                # Simple proxy: minimum distance through any third node
                mask_ij = (mst_mask[i] | mst_mask[j]) & ~np.array([i, j], dtype=int)
                if np.any(mask_ij):
                    alt_costs = distance_matrix[i, mask_ij] + distance_matrix[mask_ij, j]
                    min_alt = np.min(alt_costs) if len(alt_costs) > 0 else np.inf
                    direct = distance_matrix[i, j]
                    if min_alt < np.inf and direct > 0:
                        importance = 1.0 + (min_alt / direct - 1.0) * 0.5
                        mst_importance[i, j] = mst_importance[j, i] = min(importance, 2.0)
    
    # 2.2 Advanced centrality using resistance distance approximation
    centrality = np.zeros(n)
    for i in range(n):
        # Use effective resistance approximation: sum of inverse distances
        valid_dists = distance_matrix[i, mask[i]]
        if len(valid_dists) > 0:
            # Weight by distance rank
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            weights = 1.0 / np.sqrt(ranks)
            inv_dists = 1.0 / (valid_dists + 1e-10)
            centrality[i] = np.sum(inv_dists * weights)
    
    # Normalize centrality
    if np.max(centrality) > 0:
        centrality = centrality / np.max(centrality)
    
    # 2.3 Pseudo-embeddings via multi-scale distance statistics
    embedding_features = np.zeros((n, 4))
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Feature 1: Mean of closest 10%
        k_close = max(1, n // 10)
        close_dists = np.partition(dists, k_close)[:k_close]
        embedding_features[i, 0] = np.mean(close_dists)
        
        # Feature 2: Standard deviation of distances
        embedding_features[i, 1] = np.std(dists[dists < np.inf])
        
        # Feature 3: Distance to centroid (approximate)
        centroid_dist = np.mean(distance_matrix[i])
        embedding_features[i, 2] = centroid_dist
        
        # Feature 4: Skewness of distance distribution
        valid_dists = dists[dists < np.inf]
        if len(valid_dists) > 0:
            mean_dist = np.mean(valid_dists)
            std_dist = np.std(valid_dists)
            if std_dist > 0:
                embedding_features[i, 3] = np.mean(((valid_dists - mean_dist) / std_dist) ** 3)
    
    # ==================== 3. Multi-Path Regret Analysis ====================
    sorted_indices = np.argsort(distance_matrix, axis=1)
    lookahead_depth = max(4, int(np.sqrt(n)))
    top_neighbors = sorted_indices[:, 1:lookahead_depth+1]
    
    regret_penalty = np.ones_like(distance_matrix)
    avg_alt_ratio = np.ones_like(distance_matrix)
    
    for i in range(n):
        valid_j = np.where(mask[i] & (i < np.arange(n)))[0]
        if len(valid_j) == 0:
            continue
            
        i_neighbors = top_neighbors[i]
        
        for j in valid_j:
            j_neighbors = top_neighbors[j]
            
            # Collect alternative paths
            alt_paths = []
            
            # Two-hop paths through common neighbors
            common = np.intersect1d(i_neighbors, j_neighbors)
            common = common[(common != i) & (common != j)]
            for k in common[:5]:  # Limit to top 5
                alt_paths.append(distance_matrix[i, k] + distance_matrix[k, j])
            
            # Two-hop paths through top neighbors
            for k in i_neighbors[:3]:
                if k != j and k != i:
                    # Find best connection from k to j
                    k_to_j = distance_matrix[k, j]
                    # Also consider k->l->j for l in j's neighbors
                    for l in j_neighbors[:2]:
                        if l != i and l != j and l != k:
                            alt_paths.append(distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j])
                    alt_paths.append(distance_matrix[i, k] + k_to_j)
            
            # Three-hop paths (limited)
            for k in i_neighbors[:2]:
                if k == i or k == j:
                    continue
                for l in j_neighbors[:2]:
                    if l == i or l == j or l == k:
                        continue
                    # Path: i -> k -> l -> j
                    alt_paths.append(distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j])
            
            if alt_paths:
                alt_paths = np.array(alt_paths)
                # Take top 3 alternative paths
                top_m = min(3, len(alt_paths))
                top_alts = np.partition(alt_paths, top_m-1)[:top_m]
                
                # Compute regret metrics
                direct = distance_matrix[i, j]
                avg_alt = np.mean(top_alts)
                min_alt = np.min(alt_paths)
                
                # Store average alternative ratio
                avg_alt_ratio[i, j] = avg_alt_ratio[j, i] = avg_alt / (direct + 1e-10)
                
                # Multi-path regret penalty
                if direct > 1e-10:
                    # Base penalty from best alternative
                    ratio = min_alt / direct
                    
                    # Enhanced penalty considering average of top alternatives
                    avg_ratio = avg_alt / direct
                    
                    # Combined penalty (weighted towards worst-case)
                    combined_ratio = 0.7 * ratio + 0.3 * avg_ratio
                    
                    if combined_ratio < 1.5:
                        # Strong penalty when alternatives are competitive
                        penalty_val = 0.2 + 0.8 * (combined_ratio / 1.5) ** 1.5
                    elif combined_ratio < 3.0:
                        # Moderate penalty
                        penalty_val = 0.7 + 0.3 * ((combined_ratio - 1.5) / 1.5)
                    else:
                        penalty_val = 1.0
                    
                    # Adjust based on centrality (central nodes are more critical)
                    centrality_adj = 0.3 + 0.7 * (centrality[i] + centrality[j]) / 2
                    penalty_val = penalty_val ** centrality_adj
                    
                    regret_penalty[i, j] = regret_penalty[j, i] = penalty_val
    
    # ==================== 4. Edge Score Computation ====================
    edge_scores = np.zeros_like(distance_matrix, dtype=float)
    
    # Precompute base inverse distances with adaptive weighting
    inv_dist_base = np.zeros_like(distance_matrix)
    for i in range(n):
        valid_mask = mask[i]
        if np.any(valid_mask):
            valid_dists = distance_matrix[i, valid_mask]
            
            # Adaptive weighting based on local density
            density_weight = 1.0 + 0.5 * (1.0 - local_densities[i] / (global_density + 1e-10))
            
            # Rank-based weights with exponential decay
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            weights = np.exp(-0.3 * (ranks - 1)) * density_weight
            weights = weights / np.sum(weights)
            
            # Base inverse distance
            inv_vals = 1.0 / (valid_dists + 1e-10)
            inv_dist_base[i, valid_mask] = inv_vals * weights
    
    # Symmetrize
    inv_dist_base = np.maximum(inv_dist_base, inv_dist_base.T)
    
    # Combine features for edge scoring
    for i in range(n):
        for j in range(i+1, n):
            if mask[i, j]:
                # Base score from inverse distance
                score = inv_dist_base[i, j]
                
                # Apply MST importance
                if mst_mask[i, j]:
                    score *= mst_importance[i, j]
                
                # Apply regret penalty
                score *= regret_penalty[i, j]
                
                # Embedding similarity bonus (nodes with similar profiles)
                emb_sim = 1.0 / (1.0 + np.linalg.norm(embedding_features[i] - embedding_features[j]))
                score *= (0.9 + 0.1 * emb_sim)
                
                # Centrality-based adjustment (edges between central nodes are more important)
                centrality_bonus = 0.8 + 0.2 * (centrality[i] + centrality[j]) / 2
                score *= centrality_bonus
                
                edge_scores[i, j] = edge_scores[j, i] = score
    
    # ==================== 5. Adaptive Normalization ====================
    # Row normalization with density adaptation
    heuristics = np.zeros_like(distance_matrix)
    
    for i in range(n):
        row_scores = edge_scores[i]
        if np.sum(row_scores) > 0:
            # Adaptive smoothing based on local density
            smooth_factor = 0.1 + 0.1 * (local_densities[i] / (global_density + 1e-10))
            
            # Add minimal uniform probability for exploration
            uniform_component = smooth_factor / max(1, np.sum(mask[i]))
            heuristics[i] = (1 - smooth_factor) * (row_scores / np.sum(row_scores)) + uniform_component
        else:
            heuristics[i] = 1.0 / (n - 1)
            heuristics[i, i] = 0
    
    # Final symmetrization and cleanup
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # Ensure all rows sum to 1
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    return heuristics
