import numpy as np
from scipy.spatial import KDTree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection.
    Returns a sparse matrix with high values for promising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (similar to v1)
    with np.errstate(divide='ignore'):
        inverse_dist = 1 / (distance_matrix + 1e-10)
    np.fill_diagonal(inverse_dist, 0)
    
    # 2. Regret-based node priority
    # Compute how "urgent" each node is based on its best alternative connections
    regrets = np.zeros(n)
    for i in range(n):
        # Get two smallest distances (excluding self)
        row = distance_matrix[i].copy()
        row[i] = np.inf
        sorted_dists = np.partition(row, 2)[:2]
        if sorted_dists[1] > 0:
            regrets[i] = (sorted_dists[1] - sorted_dists[0]) / sorted_dists[0]
    
    # Combine node regrets for each edge
    regret_matrix = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            regret_matrix[i, j] = regret_matrix[j, i] = (regrets[i] + regrets[j]) / 2
    
    # 3. Two-hop neighborhood analysis
    # Build KD-tree for efficient spatial queries
    coords = np.random.randn(n, 2)  # Placeholder - in practice would use actual coordinates
    tree = KDTree(coords)
    
    # For each node, consider its k-nearest neighbors
    k = min(10, n-1)
    neighbor_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        distances, indices = tree.query(coords[i], k=k+1)
        neighbor_mask[i, indices[1:]] = True
    
    # Make mask symmetric (undirected graph)
    neighbor_mask = neighbor_mask | neighbor_mask.T
    
    # 4. Cluster density penalty
    # Penalize edges between dense clusters
    density_penalty = np.zeros((n, n))
    avg_dist = np.mean(distance_matrix[neighbor_mask])
    
    for i in range(n):
        for j in range(i+1, n):
            if neighbor_mask[i, j]:
                # Count common neighbors
                common_neighbors = np.sum(neighbor_mask[i] & neighbor_mask[j])
                if common_neighbors > 0:
                    # Penalize edges in dense clusters
                    density_penalty[i, j] = density_penalty[j, i] = 1.0 / (1.0 + common_neighbors)
    
    # 5. Combine all factors
    heuristic = inverse_dist * (1.0 + regret_matrix) * density_penalty
    
    # 6. Sparsify: keep only promising edges
    # Keep top edges for each node and global top edges
    heuristic_sparse = np.zeros_like(heuristic)
    
    # Per-node sparsification
    per_node_k = min(5, n-1)
    for i in range(n):
        row = heuristic[i].copy()
        row[i] = -np.inf  # Exclude diagonal
        top_indices = np.argpartition(row, -per_node_k)[-per_node_k:]
        heuristic_sparse[i, top_indices] = row[top_indices]
    
    # Symmetrize (since TSP is undirected)
    heuristic_sparse = np.maximum(heuristic_sparse, heuristic_sparse.T)
    
    # Global sparsification: keep only top 20% of edges by value
    if n > 10:
        flat_heuristic = heuristic_sparse.flatten()
        threshold = np.percentile(flat_heuristic[flat_heuristic > 0], 80)
        heuristic_sparse[heuristic_sparse < threshold] = 0
    
    # Final normalization
    max_val = np.max(heuristic_sparse)
    if max_val > 0:
        heuristic_sparse = heuristic_sparse / max_val
    
    return heuristic_sparse
