import numpy as np
from scipy.spatial import KDTree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristic function with dynamic regret tuning, multi-hop analysis,
    and adaptive graph properties for robust edge prioritization.
    
    Args:
        distance_matrix: Square matrix of pairwise distances between n nodes
        
    Returns:
        Heuristic matrix of same shape, with higher values indicating more promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. ENHANCED DENSITY-AWARE SPARSIFICATION ====================
    # Dynamically determine sparsity based on graph properties
    base_k = max(3, int(np.log2(n)))
    
    # Compute distance statistics for adaptive parameters
    sorted_distances = np.sort(distance_matrix, axis=1)
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    # Calculate local and global density metrics
    local_density = sorted_distances[:, base_k]
    global_density_mean = np.mean(local_density)
    global_density_std = np.std(local_density)
    
    # Enhanced adaptive neighborhood sizing
    density_ratio = (global_density_mean + 1e-10) / (local_density + 1e-10)
    adaptive_k = np.clip(
        base_k * density_ratio,
        max(2, base_k // 2),
        min(3 * base_k, n - 1)
    ).astype(int)
    
    # Build sparse adjacency with density-aware mutual inclusion
    adjacency_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        k_i = adaptive_k[i]
        neighbors = sorted_indices[i, 1:min(k_i + 1, n)]
        adjacency_mask[i, neighbors] = True
    
    # Enhanced mutual inclusion with asymmetric tolerance
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j] or adjacency_mask[j, i]:
                # Asymmetric tolerance: accept if j is in top k_i*1.5 or i in top k_j*1.5
                if (adjacency_mask[i, j] and j in sorted_indices[i, :int(adaptive_k[i] * 1.5)]) or \
                   (adjacency_mask[j, i] and i in sorted_indices[j, :int(adaptive_k[j] * 1.5)]):
                    adjacency_mask[i, j] = adjacency_mask[j, i] = True
    
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. DYNAMIC REGRET QUANTIFICATION ====================
    # Compute global regret statistics for dynamic tuning
    all_regrets = []
    for i in range(n):
        valid_mask = adjacency_mask[i]
        valid_indices = np.where(valid_mask)[0]
        
        if len(valid_indices) >= 2:
            neighbor_dists = distance_matrix[i, valid_indices]
            sorted_dists = np.sort(neighbor_dists)
            if len(sorted_dists) >= 2:
                regret = sorted_dists[1] - sorted_dists[0]
                normalized_regret = regret / (sorted_dists[0] + 1e-10)
                all_regrets.append(normalized_regret)
    
    # Dynamic regret parameters based on global statistics
    if all_regrets:
        mean_regret = np.mean(all_regrets)
        std_regret = np.std(all_regrets)
    else:
        mean_regret = 0.1
        std_regret = 0.05
    
    regret_boost = np.ones_like(distance_matrix)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        valid_indices = np.where(valid_mask)[0]
        
        if len(valid_indices) >= 2:
            neighbor_dists = distance_matrix[i, valid_indices]
            
            # Find best and second-best
            best_idx = np.argmin(neighbor_dists)
            best_dist = neighbor_dists[best_idx]
            second_best_dist = np.partition(neighbor_dists, 1)[1]
            
            # Dynamic regret calculation
            regret = second_best_dist - best_dist
            normalized_regret = regret / (best_dist + 1e-10)
            
            # Adaptive boost based on global regret distribution
            if normalized_regret > mean_regret + std_regret:
                # High regret edge: strong boost
                regret_factor = 1.0 + 0.4 * min(1.0, (normalized_regret - mean_regret) / (2 * std_regret))
            elif normalized_regret > mean_regret:
                # Moderate regret: moderate boost
                regret_factor = 1.0 + 0.2 * (normalized_regret - mean_regret) / std_regret
            else:
                # Low regret: neutral
                regret_factor = 1.0
            
            # Apply to all edges from this node
            for idx, j in enumerate(valid_indices):
                if i < j:
                    if idx == best_idx:
                        # Best edge gets the regret boost
                        boost = min(regret_factor, 1.6)
                    else:
                        # Non-best edges: penalty based on relative distance
                        penalty = 0.8 + 0.2 * (neighbor_dists[idx] - best_dist) / (second_best_dist - best_dist + 1e-10)
                        boost = min(penalty, 1.0)
                    
                    regret_boost[i, j] = regret_boost[j, i] = boost
    
    # ==================== 3. ENHANCED MULTI-HOP ANALYSIS ====================
    # Multi-hop lookahead with adaptive depth
    multi_hop_penalty = np.ones_like(distance_matrix)
    
    # Determine analysis depth based on graph size
    analysis_depth = min(3, int(np.log2(n)) + 1)
    top_m = min(15, int(np.sqrt(n) * 1.5))
    
    if analysis_depth > 1:
        # Precompute top neighbors for efficiency
        top_neighbors = sorted_indices[:, 1:top_m + 1]
        
        for i in range(n):
            valid_mask = adjacency_mask[i]
            valid_indices = np.where(valid_mask)[0]
            
            for j in valid_indices:
                if i >= j:
                    continue
                
                direct_dist = distance_matrix[i, j]
                if direct_dist < 1e-10:
                    continue
                
                # Collect candidate intermediate nodes
                candidates = set(top_neighbors[i]) | set(top_neighbors[j])
                candidates.discard(i)
                candidates.discard(j)
                
                if not candidates:
                    continue
                
                # Evaluate multi-hop paths
                best_multi_hop = np.inf
                
                for k in candidates:
                    # Two-hop path
                    two_hop = distance_matrix[i, k] + distance_matrix[k, j]
                    
                    # Three-hop path if depth allows
                    if analysis_depth >= 3:
                        # Find best intermediate node for three-hop
                        k_candidates = set(top_neighbors[k]) - {i, j, k}
                        if k_candidates:
                            for l in k_candidates:
                                three_hop = distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j]
                                best_multi_hop = min(best_multi_hop, three_hop)
                    
                    best_multi_hop = min(best_multi_hop, two_hop)
                
                if best_multi_hop < np.inf:
                    ratio = best_multi_hop / direct_dist
                    
                    # Adaptive penalty based on ratio
                    if ratio < 1.0:
                        # Very competitive multi-hop alternative
                        penalty = 0.4 + 0.3 * ratio
                    elif ratio < 1.5:
                        # Moderately competitive
                        penalty = 0.7 + 0.2 * (ratio - 1.0) / 0.5
                    else:
                        # Not competitive
                        penalty = 0.9 + 0.1 * min(1.0, (ratio - 1.5) / 2.0)
                    
                    multi_hop_penalty[i, j] = multi_hop_penalty[j, i] = penalty
    
    # ==================== 4. DENSITY-AWARE WEIGHTED SCORING ====================
    # Compute base scores with adaptive decay based on local density
    base_scores = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            valid_dists = distance_matrix[i, valid_mask]
            
            # Adaptive decay factor based on local density
            density_factor = local_density[i] / (global_density_mean + 1e-10)
            decay_rate = 0.4 + 0.3 * min(2.0, density_factor)  # Varies between 0.4 and 1.0
            
            # Rank-based weighting with adaptive decay
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            rank_weights = np.exp(-decay_rate * (ranks - 1))
            rank_weights = rank_weights / np.sum(rank_weights)
            
            # Inverse distance with adaptive scaling
            inv_vals = 1.0 / (valid_dists + 1e-10)
            base_scores[i, valid_mask] = inv_vals * rank_weights
    
    # Symmetrize base scores
    base_scores = 0.5 * (base_scores + base_scores.T)
    
    # ==================== 5. COMBINE COMPONENTS WITH ADAPTIVE WEIGHTS ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Determine component weights based on graph properties
    if global_density_std / (global_density_mean + 1e-10) > 0.5:
        # High variance graph: emphasize multi-hop analysis
        components = [0.4, 0.3, 0.3]  # base, regret, multi-hop
    else:
        # Uniform graph: balanced approach
        components = [0.5, 0.3, 0.2]
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                score = 0.0
                if components[0] > 0:
                    score += components[0] * base_scores[i, j]
                if components[1] > 0:
                    score += components[1] * regret_boost[i, j]
                if components[2] > 0:
                    score += components[2] * multi_hop_penalty[i, j]
                
                # Normalize by sum of weights
                heuristics[i, j] = heuristics[j, i] = score / sum(components)
    
    # ==================== 6. ENHANCED CONNECTIVITY ENSURANCE ====================
    # Use Kruskal's algorithm for more robust MST selection
    mst_edges = []
    edge_list = []
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                edge_list.append((distance_matrix[i, j], i, j))
    
    if edge_list:
        edge_list.sort()
        
        # Union-find for Kruskal's algorithm
        parent = list(range(n))
        rank = [0] * n
        
        def find(x):
            while parent[x] != x:
                parent[x] = parent[parent[x]]
                x = parent[x]
            return x
        
        def union(x, y):
            rx, ry = find(x), find(y)
            if rx == ry:
                return False
            if rank[rx] < rank[ry]:
                parent[rx] = ry
            elif rank[rx] > rank[ry]:
                parent[ry] = rx
            else:
                parent[ry] = rx
                rank[rx] += 1
            return True
        
        # Add edges until graph is connected or we have enough edges
        for dist, u, v in edge_list:
            if union(u, v):
                mst_edges.append((u, v))
                if len(mst_edges) >= min(n - 1, max(3, n // 4)):
                    break
        
        # Boost MST edges with low heuristic values
        for u, v in mst_edges:
            if heuristics[u, v] < np.percentile(heuristics[adjacency_mask], 30):
                boost_value = np.percentile(heuristics[adjacency_mask], 60)
                heuristics[u, v] = heuristics[v, u] = max(heuristics[u, v], boost_value)
    
    # ==================== 7. FINAL REFINEMENT ====================
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Adaptive smoothing based on graph size
    smoothing_factor = min(0.2, 1.0 / np.sqrt(n))
    row_max = heuristics.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    heuristics = heuristics / row_max
    
    # Row-normalize for probability distribution
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Final symmetrization
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
