import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection using improved regret-based
    penalty and refined scoring mechanisms.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Enhanced Adaptive Sparsification ====================
    k = max(3, int(np.log2(n)))
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argpartition(distances, k)[:k]
        adjacency_mask[i, nn_indices] = True
    
    # Mutual nearest neighbor for better connectivity
    for i in range(n):
        for j in range(n):
            if adjacency_mask[i, j]:
                j_distances = distance_matrix[j].copy()
                j_distances[j] = np.inf
                j_nn = np.argpartition(j_distances, k)[:k]
                if i in j_nn:
                    adjacency_mask[j, i] = True
    
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Improved Rank-Weighted Inverse Distance ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            valid_dists = distance_matrix[i, valid_mask]
            
            # Enhanced rank weighting with exponential decay
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            weights = np.exp(-0.5 * (ranks - 1))  # Exponential decay
            weights = weights / np.sum(weights)
            
            # Logarithmic inverse distance for better scaling
            inv_vals = 1.0 / np.log1p(valid_dists)  # log(1 + d)
            inv_dist[i, valid_mask] = inv_vals * weights
    
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    
    # ==================== 3. Regret-Based Two-Hop Penalty ====================
    two_hop_penalty = np.ones_like(distance_matrix)
    
    # Pre-compute sorted neighbors for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        valid_indices = np.where(valid_mask)[0]
        
        for j in valid_indices:
            if i >= j:
                continue
                
            direct = distance_matrix[i, j]
            if direct < 1e-10:
                two_hop_penalty[i, j] = two_hop_penalty[j, i] = 1.0
                continue
            
            # Dynamic neighbor selection based on graph density
            check_limit = max(3, int(np.sqrt(n)))
            i_neighbors = sorted_indices[i, 1:check_limit+1]
            j_neighbors = sorted_indices[j, 1:check_limit+1]
            candidate_nodes = np.unique(np.concatenate([i_neighbors, j_neighbors]))
            candidate_nodes = candidate_nodes[(candidate_nodes != i) & (candidate_nodes != j)]
            
            if len(candidate_nodes) > 0:
                # Vectorized two-hop computation
                two_hop_dists = distance_matrix[i, candidate_nodes] + distance_matrix[candidate_nodes, j]
                best_alt = np.min(two_hop_dists)
                
                # True regret computation with normalization
                regret = max(0, (direct - best_alt) / direct)
                
                # Sigmoid penalty: high regret -> low penalty
                penalty = 1.0 / (1.0 + np.exp(5.0 * (regret - 0.5)))
                two_hop_penalty[i, j] = two_hop_penalty[j, i] = penalty
    
    # ==================== 4. Integrated Scoring with Sparsification ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i+1, n):
            if adjacency_mask[i, j]:
                # Combine base score with penalty
                base_score = inv_dist[i, j]
                
                # Additional density factor: prefer edges in sparse regions
                i_degree = np.sum(adjacency_mask[i])
                j_degree = np.sum(adjacency_mask[j])
                density_factor = 2.0 / (i_degree + j_degree) if (i_degree + j_degree) > 0 else 1.0
                
                score = base_score * two_hop_penalty[i, j] * density_factor
                heuristics[i, j] = heuristics[j, i] = score
    
    # ==================== 5. Final Normalization ====================
    np.fill_diagonal(heuristics, 0)
    
    # Softmax normalization per row for probability distribution
    row_max = heuristics.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    heuristics_normalized = np.exp(heuristics / row_max - 1)
    heuristics_normalized[np.diag_indices(n)] = 0
    
    row_sums = heuristics_normalized.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics_normalized / row_sums
    
    # Final symmetrization
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
