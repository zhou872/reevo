import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with degree-aware sparsification
    and multi-hop lookahead penalties.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Degree-aware Adaptive Sparsification ====================
    # Use count of edges within 1.5x of minimum distance to determine neighbors per node
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get distances excluding self
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Find minimum distance from i
        min_dist = np.min(distances)
        
        # Count edges within 1.5x of minimum distance
        within_factor = distances <= 1.5 * min_dist
        k_i = np.sum(within_factor)
        
        # Ensure reasonable bounds: at least 2, at most log2(n) + 3
        k_i = max(2, min(k_i, int(np.log2(n)) + 3))
        
        # Get k_i nearest neighbors
        nn_indices = np.argpartition(distances, k_i)[:k_i]
        adjacency_mask[i, nn_indices] = True
        
        # Ensure symmetry: if i is in j's top neighbors, include edge
        for j in nn_indices:
            if not adjacency_mask[j, i]:
                # Check if i is within 1.5x of j's minimum distance
                j_distances = distance_matrix[j].copy()
                j_distances[j] = np.inf
                j_min_dist = np.min(j_distances)
                if distance_matrix[j, i] <= 1.5 * j_min_dist:
                    adjacency_mask[j, i] = True
    
    # Final symmetry enforcement
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Rank-weighted Inverse Distance Scoring ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            # Count neighbors for normalization
            neighbor_count = np.sum(valid_mask)
            
            # Get distances to neighbors
            valid_dists = distance_matrix[i, valid_mask]
            
            # Compute rank-based weights (favor closer ranks more strongly)
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            weights = 1.0 / (ranks ** 0.75)  # Stronger decay than linear
            weights = weights / np.sum(weights)
            
            # Apply weighted inverse distance
            inv_vals = 1.0 / (valid_dists + 1e-10)
            inv_dist[i, valid_mask] = (inv_vals * weights) / np.sqrt(neighbor_count)  # Soft normalization
    
    # Symmetrize by taking geometric mean
    inv_dist = np.sqrt(inv_dist * inv_dist.T)
    
    # ==================== 3. Multi-hop Lookahead Penalty ====================
    # Compute penalties based on two-hop and three-hop alternatives
    penalty_matrix = np.ones_like(distance_matrix)
    
    # Pre-compute sorted neighbor indices for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        valid_indices = np.where(valid_mask)[0]
        
        for j in valid_indices:
            if i >= j:
                continue
                
            direct_dist = distance_matrix[i, j]
            if direct_dist < 1e-10:
                continue
                
            # Get candidate intermediate nodes (top L neighbors of i and j)
            L = max(3, int(np.sqrt(n)))
            i_neighbors = sorted_indices[i, 1:L+1]
            j_neighbors = sorted_indices[j, 1:L+1]
            candidate_nodes = np.unique(np.concatenate([i_neighbors, j_neighbors]))
            candidate_nodes = candidate_nodes[(candidate_nodes != i) & (candidate_nodes != j)]
            
            if len(candidate_nodes) > 0:
                # Compute best two-hop path
                two_hop_dists = distance_matrix[i, candidate_nodes] + distance_matrix[candidate_nodes, j]
                min_two_hop = np.min(two_hop_dists) if len(two_hop_dists) > 0 else np.inf
                
                # Compute best three-hop path (i -> k -> l -> j)
                min_three_hop = np.inf
                for k in candidate_nodes[:min(10, len(candidate_nodes))]:  # Limit for efficiency
                    # Get top neighbors of k
                    k_neighbors = sorted_indices[k, 1:min(5, L)]
                    k_neighbors = k_neighbors[(k_neighbors != i) & (k_neighbors != j)]
                    
                    if len(k_neighbors) > 0:
                        three_hop_dists = (distance_matrix[i, k] + 
                                          distance_matrix[k, k_neighbors] + 
                                          distance_matrix[k_neighbors, j])
                        min_k_three_hop = np.min(three_hop_dists) if len(three_hop_dists) > 0 else np.inf
                        min_three_hop = min(min_three_hop, min_k_three_hop)
                
                # Use piecewise penalty based on best alternative
                best_alt = min(min_two_hop, min_three_hop)
                if best_alt < np.inf:
                    ratio = best_alt / direct_dist
                    
                    # Piecewise penalty function
                    if ratio < 0.8:  # Much better alternative exists
                        penalty = 0.2
                    elif ratio < 1.0:  # Slightly better alternative
                        penalty = 0.5
                    elif ratio < 1.2:  # Comparable alternative
                        penalty = 0.8
                    elif ratio < 1.5:  # Direct edge is somewhat better
                        penalty = 0.95
                    else:  # Direct edge is significantly better
                        penalty = 1.1
                    
                    penalty_matrix[i, j] = penalty_matrix[j, i] = penalty
    
    # ==================== 4. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Combine inverse distance with penalties
    for i in range(n):
        valid_mask = adjacency_mask[i]
        for j in np.where(valid_mask)[0]:
            if i < j:  # Process each edge once
                base_score = inv_dist[i, j]
                # Apply penalty and ensure non-negative
                score = max(0, base_score * penalty_matrix[i, j])
                heuristics[i, j] = heuristics[j, i] = score
    
    # ==================== 5. Final Processing ====================
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Add small constant to avoid zeros and enhance sparsity
    heuristics[adjacency_mask] += 1e-8
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Symmetrize using arithmetic mean
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # Final sparsification: keep only edges in adjacency mask
    heuristics[~adjacency_mask] = 0
    
    return heuristics
