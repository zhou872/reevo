import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection combining multiple sophisticated strategies.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Degree-aware Adaptive Sparsification ====================
    # Determine number of neighbors per node based on local edge density
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Find edges within 1.5x of minimum distance
        min_dist = np.min(distances)
        threshold = 1.5 * min_dist
        promising_edges = distances <= threshold
        
        # Count promising edges and set k adaptively
        promising_count = np.sum(promising_edges)
        
        if promising_count > 0:
            # Adaptive k based on promising edge count and logarithmic scaling
            base_k = max(3, int(np.log2(n)))
            k = min(base_k, promising_count)
            
            # Get k closest among promising edges
            if promising_count > k:
                # Use partition to get k smallest distances among promising ones
                promising_indices = np.where(promising_edges)[0]
                promising_dists = distances[promising_indices]
                # Get indices of k smallest distances
                k_smallest_indices = np.argpartition(promising_dists, k)[:k]
                nn_indices = promising_indices[k_smallest_indices]
            else:
                # All promising edges are neighbors
                nn_indices = np.where(promising_edges)[0]
                
            adjacency_mask[i, nn_indices] = True
    
    # ==================== 2. Neighbor-aware Symmetric Sparsification ====================
    # Ensure symmetry while considering neighbor relationships
    for i in range(n):
        for j in np.where(adjacency_mask[i])[0]:
            if i != j and not adjacency_mask[j, i]:
                # Check if j considers i among its neighbors
                # Count neighbors of j to see if we should add i
                j_neighbors = np.where(adjacency_mask[j])[0]
                if len(j_neighbors) > 0:
                    # Check if i is closer to j than its current farthest neighbor
                    farthest_j_neighbor = j_neighbors[np.argmax(distance_matrix[j, j_neighbors])]
                    if distance_matrix[j, i] < distance_matrix[j, farthest_j_neighbor]:
                        adjacency_mask[j, i] = True
                else:
                    adjacency_mask[j, i] = True
    
    # Final symmetry enforcement and cleanup
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 3. Enhanced Distance Scoring ====================
    # Compute inverse distance with adaptive weighting
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            # Get distances to neighbors
            valid_dists = distance_matrix[i, valid_mask]
            
            # Compute adaptive weights based on distance distribution
            if len(valid_dists) > 1:
                # Exponential decay based on rank
                ranks = np.argsort(np.argsort(valid_dists)) + 1
                # Stronger preference for closer nodes
                weights = np.exp(-ranks / 2)  # Exponential decay
                weights = weights / np.sum(weights)
            else:
                weights = np.array([1.0])
            
            # Apply weighted inverse distance
            inv_vals = 1.0 / (valid_dists + 1e-10)
            inv_dist[i, valid_mask] = inv_vals * weights
    
    # Symmetrize using geometric mean (preserves scale relationships better)
    sym_inv_dist = np.sqrt(inv_dist * inv_dist.T)
    # Where both are zero, keep zero
    zero_mask = (inv_dist == 0) & (inv_dist.T == 0)
    sym_inv_dist[zero_mask] = 0
    inv_dist = sym_inv_dist
    
    # ==================== 4. Multi-hop Lookahead Penalty ====================
    # Compute penalty considering 2-hop and 3-hop alternatives
    multi_hop_penalty = np.ones_like(distance_matrix)
    
    # Pre-compute sorted distances for efficient neighbor lookup
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        valid_indices = np.where(valid_mask)[0]
        
        for j in valid_indices:
            if i >= j:
                continue
                
            direct_dist = distance_matrix[i, j]
            if direct_dist < 1e-10:
                continue
                
            # Find top neighbors for multi-hop analysis
            neighbor_limit = max(3, int(np.sqrt(n)))
            
            # Get candidate intermediate nodes (neighbors of i and j)
            i_neighbors = sorted_indices[i, 1:neighbor_limit+1]
            j_neighbors = sorted_indices[j, 1:neighbor_limit+1]
            candidate_nodes = np.unique(np.concatenate([i_neighbors, j_neighbors]))
            candidate_nodes = candidate_nodes[candidate_nodes != i]
            candidate_nodes = candidate_nodes[candidate_nodes != j]
            
            if len(candidate_nodes) > 0:
                # Vectorized 2-hop distance computation
                two_hop_dists = distance_matrix[i, candidate_nodes] + distance_matrix[candidate_nodes, j]
                min_two_hop = np.min(two_hop_dists) if len(two_hop_dists) > 0 else np.inf
                
                # Vectorized 3-hop distance computation
                if len(candidate_nodes) >= 2:
                    # Consider pairs of intermediate nodes
                    three_hop_candidates = []
                    for k in candidate_nodes:
                        # Find best l != k that connects k to j
                        remaining = candidate_nodes[candidate_nodes != k]
                        if len(remaining) > 0:
                            three_hop_dists = distance_matrix[i, k] + distance_matrix[k, remaining] + distance_matrix[remaining, j]
                            if len(three_hop_dists) > 0:
                                three_hop_candidates.append(np.min(three_hop_dists))
                    
                    if three_hop_candidates:
                        min_three_hop = np.min(three_hop_candidates)
                    else:
                        min_three_hop = np.inf
                else:
                    min_three_hop = np.inf
                
                # Compute penalty based on best alternative
                if min_two_hop < np.inf or min_three_hop < np.inf:
                    best_alternative = min(min_two_hop, min_three_hop)
                    
                    if best_alternative < np.inf:
                        ratio = best_alternative / direct_dist
                        
                        # Piecewise penalty function
                        if ratio < 0.8:
                            # Strong penalty if alternative is much better
                            penalty = 0.2
                        elif ratio < 1.2:
                            # Moderate penalty if alternative is competitive
                            penalty = 0.5
                        elif ratio < 1.8:
                            # Mild penalty if alternative is somewhat worse
                            penalty = 0.8
                        else:
                            # No penalty if alternative is much worse
                            penalty = 1.0
                        
                        multi_hop_penalty[i, j] = multi_hop_penalty[j, i] = penalty
    
    # ==================== 5. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Apply inverse distance scoring with multi-hop penalty
    heuristics = inv_dist * multi_hop_penalty
    
    # Apply sparsification mask
    heuristics[~adjacency_mask] = 0
    
    # ==================== 6. Final Processing ====================
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Add small constant to avoid zeros in sparse regions
    min_nonzero = heuristics[heuristics > 0].min() if np.any(heuristics > 0) else 1e-8
    heuristics[heuristics == 0] = min_nonzero * 0.01
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Symmetrize final matrix using arithmetic mean
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
