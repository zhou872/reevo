import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection incorporating density
    augmentation and vectorized two-hop analysis for improved edge quality assessment.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive Symmetric Sparsification ====================
    # Use logarithmic scaling for neighbor selection, ensuring symmetry
    k = max(3, int(np.log2(n)))
    mask = np.zeros((n, n), dtype=bool)
    
    # Get k nearest neighbors for each node
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argpartition(distances, k)[:k]
        mask[i, nn_indices] = True
    
    # Ensure symmetry: if i is in j's top-k, include edge in both directions
    mask = mask | mask.T
    np.fill_diagonal(mask, False)
    
    # ==================== 2. Density-Augmented Rank-Weighted Inverse Distance ====================
    # Compute inverse distances with exponential rank weighting and density augmentation
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = mask[i]
        if np.any(valid_mask):
            # Get distances to neighbors
            valid_dists = distance_matrix[i, valid_mask]
            
            # Rank neighbors by distance (closest gets rank 1)
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            
            # Exponential decay weights based on rank
            weights = np.exp(-0.5 * (ranks - 1))
            weights = weights / np.sum(weights)  # Normalize
            
            # Base inverse distance with smoothing
            inv_vals = 1.0 / (valid_dists + 1e-10)
            
            # Density augmentation: compute saturation factor from neighbor inverse distances
            neighbor_inv_dists = 1.0 / (distance_matrix[i] + 1e-10)
            neighbor_inv_dists[i] = 0
            density_factor = np.mean(neighbor_inv_dists[mask[i]]) if np.any(mask[i]) else 1.0
            
            # Apply density augmentation and rank weighting
            inv_dist[i, valid_mask] = (inv_vals * weights) * (1.0 + density_factor)
    
    # Symmetrize by taking maximum
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    np.fill_diagonal(inv_dist, 0)
    
    # ==================== 3. Vectorized Two-Hop Alternative Penalty ====================
    # Efficient computation of two-hop alternatives using vectorized operations
    penalty = np.ones_like(distance_matrix)
    
    # Pre-compute sorted neighbor indices for efficient two-hop analysis
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    # For each pair of nodes with edge in mask
    for i in range(n):
        valid_indices = np.where(mask[i] & (i < np.arange(n)))[0]
        if len(valid_indices) == 0:
            continue
            
        # Get relevant neighbors for two-hop analysis
        check_limit = max(3, int(np.sqrt(n)))
        i_neighbors = sorted_indices[i, 1:check_limit+1]
        
        # Vectorized two-hop computation for all valid edges from i
        for j in valid_indices:
            j_neighbors = sorted_indices[j, 1:check_limit+1]
            
            # Combine candidate intermediate nodes
            candidates = np.unique(np.concatenate([i_neighbors, j_neighbors]))
            candidates = candidates[(candidates != i) & (candidates != j)]
            
            if len(candidates) > 0:
                # Vectorized two-hop distance computation
                two_hop_dists = distance_matrix[i, candidates] + distance_matrix[candidates, j]
                min_two_hop = np.min(two_hop_dists)
                direct_dist = distance_matrix[i, j]
                
                if direct_dist > 1e-10:
                    # Penalize edges with good two-hop alternatives
                    ratio = min_two_hop / direct_dist
                    # Stronger penalty for edges with nearly as good two-hop alternatives
                    penalty_val = 0.2 + 0.8 * np.clip(ratio, 0.5, 2.0) / 2.0
                    penalty[i, j] = penalty_val
                    penalty[j, i] = penalty_val
    
    # ==================== 4. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Apply base heuristic scores and penalties only to edges in mask
    base_scores = inv_dist.copy()
    base_scores[~mask] = 0
    heuristics = base_scores * penalty
    
    # Normalize non-zero entries to create probability distribution
    row_sums = heuristics.sum(axis=1, keepdims=True)
    non_zero_rows = row_sums.flatten() > 0
    heuristics[non_zero_rows] = heuristics[non_zero_rows] / row_sums[non_zero_rows]
    
    # Add small constant to avoid exact zeros and ensure connectivity
    heuristics[mask] += 1e-8
    
    # Final symmetrization
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
