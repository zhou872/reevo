import numpy as np
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization that balances
    computational efficiency with sophisticated edge assessment.
    
    Key improvements:
    1. Adaptive density-based sparsification with local variance adjustment
    2. Rank-weighted inverse distance scoring for effective edge prioritization
    3. Lightweight MST-based bridge detection for structural importance
    4. Efficient two-hop alternative analysis with adaptive penalty
    
    Args:
        distance_matrix: Square symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for
        more promising edges. Matrix is symmetric and row-normalized.
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive Density-Based Sparsification ====================
    # Logarithmic base connectivity with local density adjustment
    base_k = max(3, int(np.log2(n)))
    k_values = np.full(n, base_k, dtype=int)
    
    # Precompute sorted distances for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    # Adjust k based on local density variance
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Get distances to base_k nearest neighbors
        k_nearest = min(base_k, n-1)
        nearest_dists = np.partition(distances, k_nearest)[:k_nearest]
        nearest_dists = np.maximum(nearest_dists, 1e-10)
        
        # Compute local density metric using log variance
        log_variance = np.var(np.log(nearest_dists))
        density_factor = 1.0 + np.tanh(log_variance)  # Between 1 and 2
        k_values[i] = max(3, int(base_k * density_factor))
    
    # Build adjacency mask with density-adaptive connectivity
    adjacency_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        k_i = min(k_values[i], n-1)
        nn_indices = sorted_indices[i, 1:k_i+1]
        adjacency_mask[i, nn_indices] = True
    
    # Ensure symmetry by including mutual nearest neighbors
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Rank-Weighted Inverse Distance Scoring ====================
    # Efficient scoring using rank-based weights
    rank_scores = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if not np.any(valid_mask):
            continue
        
        # Get distances to neighbors
        neighbor_dists = distance_matrix[i, valid_mask]
        
        # Compute rank-based weights (closer neighbors get higher weight)
        ranks = np.argsort(neighbor_dists) + 1  # Closest gets rank 1
        weights = 1.0 / ranks
        weights = weights / np.sum(weights)  # Normalize to sum to 1
        
        # Apply weighted inverse distance
        inv_dists = 1.0 / (neighbor_dists + 1e-10)
        rank_scores[i, valid_mask] = inv_dists * weights
    
    # Symmetrize by taking geometric mean
    rank_scores = np.sqrt(rank_scores * rank_scores.T)
    
    # ==================== 3. Lightweight MST Bridge Detection ====================
    # Compute MST using Prim's algorithm for structural importance
    mst_boost = np.ones_like(distance_matrix, dtype=float)
    
    if n > 3:
        # Prim's algorithm for MST
        visited = np.zeros(n, dtype=bool)
        min_edge = np.full(n, np.inf)
        parent = np.full(n, -1, dtype=int)
        min_edge[0] = 0
        
        for _ in range(n):
            # Find minimum edge to unvisited node
            u = np.argmin(np.where(~visited, min_edge, np.inf))
            visited[u] = True
            
            if parent[u] != -1:
                # Mark MST edges with initial boost
                mst_boost[u, parent[u]] = mst_boost[parent[u], u] = 1.2
            
            # Update distances to unvisited neighbors
            for v in range(n):
                if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                    min_edge[v] = distance_matrix[u, v]
                    parent[v] = u
        
        # Additional boost for edges connecting different MST branches
        # Simple approximation: boost edges where both endpoints have high MST degree
        mst_degree = np.zeros(n)
        for v in range(1, n):
            u = parent[v]
            if u != -1:
                mst_degree[u] += 1
                mst_degree[v] += 1
        
        # Apply degree-based boost to original adjacency
        if np.max(mst_degree) > 0:
            norm_degree = mst_degree / np.max(mst_degree)
            for i in range(n):
                if norm_degree[i] > 0.5:  # Boost high-degree MST nodes
                    mask = adjacency_mask[i]
                    mst_boost[i, mask] = np.maximum(mst_boost[i, mask], 
                                                   1.0 + 0.2 * norm_degree[i])
                    mst_boost[mask, i] = mst_boost[i, mask]
    
    # ==================== 4. Efficient Two-Hop Alternative Analysis ====================
    # Penalize edges with good alternative paths
    two_hop_penalty = np.ones_like(distance_matrix, dtype=float)
    
    # Precompute top neighborhoods for efficiency
    analysis_depth = min(10, max(3, int(np.sqrt(n))))
    neighborhoods = []
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn = np.argpartition(distances, analysis_depth)[:analysis_depth]
        neighborhoods.append(set(nn))
    
    # Analyze two-hop alternatives for sparse edges
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                direct_dist = distance_matrix[i, j]
                
                # Find common neighbors among top neighborhoods
                common = neighborhoods[i].intersection(neighborhoods[j])
                common.discard(i)
                common.discard(j)
                
                if common:
                    common_list = list(common)
                    two_hop_dists = (distance_matrix[i, common_list] + 
                                   distance_matrix[common_list, j])
                    min_two_hop = np.min(two_hop_dists)
                    
                    if direct_dist > 1e-10:
                        ratio = min_two_hop / direct_dist
                        
                        # Adaptive penalty based on ratio
                        if ratio < 1.1:
                            # Strong penalty if two-hop is almost as good
                            penalty = 0.4 + 0.3 * ratio
                        elif ratio < 1.5:
                            # Moderate penalty
                            penalty = 0.6 + 0.2 * ratio
                        else:
                            # Minimal penalty
                            penalty = 0.8 + 0.1 * ratio
                        
                        two_hop_penalty[i, j] = two_hop_penalty[j, i] = penalty
    
    # ==================== 5. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Multiplicative combination of all components for sparse edges
    mask_indices = np.where(adjacency_mask)
    if len(mask_indices[0]) > 0:
        i_indices, j_indices = mask_indices
        
        rank_vals = rank_scores[mask_indices]
        penalty_vals = two_hop_penalty[mask_indices]
        boost_vals = mst_boost[mask_indices]
        
        # Geometric mean of components (ensures balanced combination)
        epsilon = 1e-10
        combined = np.cbrt((rank_vals + epsilon) * 
                         (penalty_vals + epsilon) * 
                         (boost_vals + epsilon))
        
        heuristics[i_indices, j_indices] = combined
    
    # ==================== 6. Final Processing ====================
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Add small constant to avoid division by zero
    heuristics[adjacency_mask] += 1e-8
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Final symmetrization
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
