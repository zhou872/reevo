import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic function for TSP edge selection that combines multiple 
    mathematical strategies to better identify promising edges.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive Logarithmic Sparsification ====================
    k = max(3, int(np.log2(n)))  # Logarithmic neighbor count
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Find k-nearest neighbors for each node
    for i in range(n):
        # Get distances to all other nodes
        distances = distance_matrix[i].copy()
        distances[i] = np.inf  # Exclude self
        
        # Find k nearest neighbors using argpartition for efficiency
        nn_indices = np.argpartition(distances, k)[:k]
        adjacency_mask[i, nn_indices] = True
        
        # Ensure symmetry by checking reciprocal relationships
        for j in nn_indices:
            j_distances = distance_matrix[j].copy()
            j_distances[j] = np.inf
            j_nn = np.argpartition(j_distances, k)[:k]
            if i in j_nn:
                adjacency_mask[j, i] = True
    
    # Final symmetry and diagonal cleanup
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Rank-Weighted Inverse Distance Scoring ====================
    inv_dist_scores = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            # Get distances to valid neighbors
            valid_dists = distance_matrix[i, valid_mask]
            neighbor_indices = np.where(valid_mask)[0]
            
            # Sort by distance and assign ranks (closest = rank 1)
            sorted_indices = np.argsort(valid_dists)
            ranks = np.zeros_like(valid_dists)
            ranks[sorted_indices] = np.arange(1, len(valid_dists) + 1)
            
            # Compute rank-based weights (inverse rank, normalized)
            weights = 1.0 / ranks
            weights = weights / np.sum(weights)
            
            # Apply weighted inverse distance scoring
            inv_vals = 1.0 / (valid_dists + 1e-10)
            weighted_inv = inv_vals * weights
            
            # Store results
            inv_dist_scores[i, neighbor_indices] = weighted_inv
    
    # Symmetrize by taking maximum of both directions
    inv_dist_scores = np.maximum(inv_dist_scores, inv_dist_scores.T)
    
    # ==================== 3. Dynamic Two-Hop Regret Penalty ====================
    regret_penalties = np.ones_like(distance_matrix)
    
    # Pre-sort indices for efficient neighbor lookups
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        # Get neighbors from adjacency mask
        i_neighbors = np.where(adjacency_mask[i])[0]
        
        for j in i_neighbors:
            if i >= j:  # Process each edge once
                continue
                
            # Get top neighbors of both endpoints for candidate intermediates
            check_limit = max(3, int(np.sqrt(n)))
            i_top = sorted_indices[i, 1:check_limit+1]
            j_top = sorted_indices[j, 1:check_limit+1]
            
            # Combine and filter candidate intermediate nodes
            candidates = np.unique(np.concatenate([i_top, j_top]))
            candidates = candidates[(candidates != i) & (candidates != j)]
            
            if len(candidates) > 0:
                # Compute two-hop distances through all candidates
                two_hop_dists = distance_matrix[i, candidates] + distance_matrix[candidates, j]
                best_alt = np.min(two_hop_dists)
                direct_cost = distance_matrix[i, j]
                
                if direct_cost > 1e-10:
                    # Calculate regret: (direct - best_alternative) / direct
                    regret = (direct_cost - best_alt) / direct_cost
                    
                    # Convert regret to penalty (higher regret = lower penalty)
                    # Use sigmoid-like function to map regret to [0.2, 1.0]
                    penalty = 0.6 + 0.4 * np.tanh(-2.0 * regret)
                    regret_penalties[i, j] = regret_penalties[j, i] = np.clip(penalty, 0.2, 1.0)
    
    # ==================== 4. Combined Scoring ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Apply multiplicative fusion of components only to promising edges
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Base score from inverse distance
                base_score = inv_dist_scores[i, j]
                
                # Apply regret penalty multiplicatively
                penalized_score = base_score * regret_penalties[i, j]
                
                # Additional boost for very short edges
                if distance_matrix[i, j] < np.percentile(distance_matrix[adjacency_mask], 10):
                    penalized_score *= 1.5
                
                heuristics[i, j] = heuristics[j, i] = penalized_score
    
    # ==================== 5. Final Processing ====================
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Add small epsilon to avoid exact zeros
    heuristics[adjacency_mask] += 1e-8
    
    # Row-normalize to create probability-like distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Symmetrize final matrix
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
