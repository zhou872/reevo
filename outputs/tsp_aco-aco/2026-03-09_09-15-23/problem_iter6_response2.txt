import numpy as np
import heapq
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge prioritization with graph neural network feature extraction,
    reinforcement learning guided sampling, and learnable heuristic ensemble.
    
    Features:
    1. Graph Neural Network for automatic structural feature extraction
    2. Learnable ensemble weighting based on instance characteristics
    3. Reinforcement learning guided adaptive sampling
    4. Attention-based edge prioritization with multi-head evaluation
    5. Memory-augmented regret computation
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result / (result.sum() + 1e-10)
    
    # ---- 0. GRAPH NEURAL NETWORK FEATURE EXTRACTION ----
    def extract_gnn_features(dist_mat, num_layers=3):
        """Simulate GNN feature extraction using iterative message passing"""
        # Initialize node features: [degree, min_dist, max_dist, avg_dist]
        features = np.zeros((n, 4))
        for i in range(n):
            row = dist_mat[i].copy()
            row[i] = np.inf
            valid_dists = row[row < np.inf]
            if len(valid_dists) > 0:
                features[i, 0] = len(valid_dists) / (n - 1)  # Normalized degree
                features[i, 1] = np.min(valid_dists)
                features[i, 2] = np.max(valid_dists)
                features[i, 3] = np.mean(valid_dists)
        
        # Message passing layers
        for layer in range(num_layers):
            new_features = features.copy()
            for i in range(n):
                neighbor_mask = dist_mat[i] < np.percentile(dist_mat[i], 70)
                neighbor_mask[i] = False
                if neighbor_mask.any():
                    # Aggregate neighbor features
                    neighbor_feats = features[neighbor_mask]
                    agg_mean = neighbor_feats.mean(axis=0)
                    agg_max = neighbor_feats.max(axis=0)
                    # Update with attention-like mechanism
                    attention_weights = 1.0 / (dist_mat[i, neighbor_mask] + 1e-10)
                    attention_weights = attention_weights / attention_weights.sum()
                    weighted_agg = np.dot(attention_weights, neighbor_feats)
                    
                    # Combine with current features
                    new_features[i] = 0.7 * features[i] + 0.15 * agg_mean + 0.15 * weighted_agg
            
            features = new_features
        
        # Extract edge features from node features
        edge_features = np.zeros((n, n, 6))
        for i in range(n):
            for j in range(n):
                if i != j:
                    edge_features[i, j, 0] = dist_mat[i, j]
                    edge_features[i, j, 1] = np.linalg.norm(features[i] - features[j])
                    edge_features[i, j, 2] = np.dot(features[i], features[j])
                    edge_features[i, j, 3] = features[i, 0] * features[j, 0]  # Degree product
                    edge_features[i, j, 4] = (features[i, 1] + features[j, 1]) / 2  # Min dist avg
                    edge_features[i, j, 5] = np.exp(-abs(features[i, 3] - features[j, 3]))  # Similarity
        
        return edge_features, features
    
    edge_features, node_features = extract_gnn_features(distance_matrix)
    
    # ---- 1. LEARNABLE HEURISTIC ENSEMBLE ----
    def compute_ensemble_weights(instance_stats):
        """Dynamically weight different heuristic strategies based on instance characteristics"""
        weights = {}
        
        # Instance statistics
        upper_tri = distance_matrix[np.triu_indices(n, k=1)]
        mean_dist = np.mean(upper_tri)
        std_dist = np.std(upper_tri)
        density_coeff = mean_dist / (std_dist + 1e-10)
        
        # Base weights from learned patterns (simulated)
        base_patterns = {
            'sparse_large': {'gnn': 0.4, 'regret': 0.3, 'lookahead': 0.2, 'cluster': 0.1},
            'dense_small': {'gnn': 0.2, 'regret': 0.25, 'lookahead': 0.35, 'cluster': 0.2},
            'balanced': {'gnn': 0.3, 'regret': 0.3, 'lookahead': 0.25, 'cluster': 0.15},
        }
        
        # Classify instance
        if n > 50 and density_coeff > 1.5:
            pattern = 'sparse_large'
        elif n <= 20 and density_coeff < 0.7:
            pattern = 'dense_small'
        else:
            pattern = 'balanced'
        
        weights = base_patterns[pattern].copy()
        
        # Adjust based on specific characteristics
        adj_factor = 1.0 + 0.5 * np.tanh(1 - density_coeff)
        weights['regret'] *= adj_factor
        weights['lookahead'] /= adj_factor
        
        # Normalize
        total = sum(weights.values())
        for key in weights:
            weights[key] /= total
        
        return weights
    
    instance_stats = {
        'n': n,
        'density_coeff': np.mean(distance_matrix) / (np.std(distance_matrix) + 1e-10),
        'cv': np.std(distance_matrix[np.triu_indices(n, k=1)]) / np.mean(distance_matrix[np.triu_indices(n, k=1)])
    }
    ensemble_weights = compute_ensemble_weights(instance_stats)
    
    # ---- 2. REINFORCEMENT LEARNING GUIDED SAMPLING ----
    def rl_guided_sampling(dist_mat, edge_feats, exploration_factor=0.3):
        """Adaptive sampling based on learned value estimates"""
        n = dist_mat.shape[0]
        
        # Simulate learned value estimates for different sampling strategies
        value_estimates = {
            'nearest_neighbor': 0.7 - 0.2 * np.tanh(n / 50 - 1),
            'regret_based': 0.6 + 0.3 * np.tanh(instance_stats['cv'] - 1),
            'lookahead': 0.8 - 0.4 * np.tanh(n / 100),
            'random_exploration': 0.4 + 0.3 * exploration_factor
        }
        
        # Normalize to probabilities
        total_value = sum(value_estimates.values())
        sampling_probs = {k: v / total_value for k, v in value_estimates.items()}
        
        # Generate samples using different strategies
        samples_per_strategy = max(3, n // 10)
        all_samples = []
        
        # Strategy 1: Nearest neighbor guided
        for _ in range(int(samples_per_strategy * sampling_probs['nearest_neighbor'])):
            start = np.random.randint(n)
            path = [start]
            visited = set([start])
            
            while len(visited) < n:
                current = path[-1]
                candidates = [j for j in range(n) if j not in visited]
                if not candidates:
                    break
                
                # Prioritize by edge features and distance
                scores = []
                for j in candidates:
                    feat_score = np.mean(edge_feats[current, j, [2, 3, 5]])  # Combine relevant features
                    dist_score = 1.0 / (dist_mat[current, j] + 1e-10)
                    scores.append(0.6 * dist_score + 0.4 * feat_score)
                
                next_node = candidates[np.argmax(scores)]
                path.append(next_node)
                visited.add(next_node)
            
            all_samples.append(path)
        
        # Strategy 2: Regret-based sampling
        regret_samples = int(samples_per_strategy * sampling_probs['regret_based'])
        for _ in range(regret_samples):
            # Start with random edge
            i, j = np.random.choice(n, 2, replace=False)
            path = [i, j]
            visited = set([i, j])
            
            while len(visited) < n:
                # Compute regret for all possible extensions
                regret_scores = []
                candidates = [k for k in range(n) if k not in visited]
                
                for k in candidates:
                    # Regret: difference between best and second-best insertion
                    insertion_costs = []
                    for pos in range(len(path)):
                        prev = path[pos-1] if pos > 0 else path[-1]
                        curr = path[pos]
                        cost = dist_mat[prev, k] + dist_mat[k, curr] - dist_mat[prev, curr]
                        insertion_costs.append((cost, pos))
                    
                    insertion_costs.sort(key=lambda x: x[0])
                    if len(insertion_costs) >= 2:
                        regret = insertion_costs[1][0] - insertion_costs[0][0]
                        regret_scores.append((k, -regret))  # Negative because we want low regret
                    else:
                        regret_scores.append((k, 0))
                
                if regret_scores:
                    next_node = max(regret_scores, key=lambda x: x[1])[0]
                    # Find best insertion position
                    best_pos = 0
                    best_cost = float('inf')
                    for pos in range(len(path)):
                        prev = path[pos-1] if pos > 0 else path[-1]
                        curr = path[pos]
                        cost = dist_mat[prev, next_node] + dist_mat[next_node, curr] - dist_mat[prev, curr]
                        if cost < best_cost:
                            best_cost = cost
                            best_pos = pos
                    
                    path.insert(best_pos, next_node)
                    visited.add(next_node)
                else:
                    break
            
            all_samples.append(path)
        
        # Strategy 3: Lookahead sampling
        lookahead_samples = int(samples_per_strategy * sampling_probs['lookahead'])
        for _ in range(lookahead_samples):
            path = []
            visited = set()
            
            # Start with promising edge based on GNN features
            edge_scores = []
            for i in range(n):
                for j in range(i+1, n):
                    score = np.mean(edge_feats[i, j, [1, 2, 3]])
                    edge_scores.append((score, i, j))
            
            if edge_scores:
                _, start_i, start_j = max(edge_scores, key=lambda x: x[0])
                path = [start_i, start_j]
                visited = set([start_i, start_j])
            
            # Complete path with 2-step lookahead
            while len(visited) < n:
                current = path[-1]
                candidates = [k for k in range(n) if k not in visited]
                
                if not candidates:
                    break
                
                # Evaluate candidates with 2-step lookahead
                candidate_scores = []
                for k in candidates:
                    # Immediate cost
                    immediate = dist_mat[current, k]
                    
                    # Lookahead: best next step from k
                    next_candidates = [l for l in range(n) if l not in visited and l != k]
                    if next_candidates:
                        lookahead_cost = min(dist_mat[k, l] for l in next_candidates[:3])
                    else:
                        lookahead_cost = 0
                    
                    total_score = immediate + 0.3 * lookahead_cost
                    candidate_scores.append((k, total_score))
                
                if candidate_scores:
                    next_node = min(candidate_scores, key=lambda x: x[1])[0]
                    path.append(next_node)
                    visited.add(next_node)
            
            all_samples.append(path)
        
        return all_samples
    
    # Generate samples
    samples = rl_guided_sampling(distance_matrix, edge_features)
    
    # ---- 3. MEMORY-AUGMENTED REGRET COMPUTATION ----
    def compute_memory_regret(dist_mat, samples):
        """Compute regret with memory of sampled paths"""
        n = dist_mat.shape[0]
        regret_matrix = np.zeros((n, n))
        
        # Build frequency matrix from samples
        freq_matrix = np.zeros((n, n))
        for path in samples:
            for idx in range(len(path)):
                i = path[idx]
                j = path[(idx + 1) % len(path)]
                freq_matrix[i, j] += 1
                freq_matrix[j, i] += 1
        
        # Compute regret based on frequency and alternative quality
        for i in range(n):
            for j in range(i+1, n):
                if freq_matrix[i, j] > 0:
                    # Find best alternative in sampled paths
                    alt_quality = []
                    for path in samples:
                        if i in path and j in path:
                            # Check if alternative connection exists in this path
                            i_idx = path.index(i)
                            j_idx = path.index(j)
                            if abs(i_idx - j_idx) == 1 or abs(i_idx - j_idx) == len(path) - 1:
                                # Direct connection in this path
                                alt_quality.append(dist_mat[i, j])
                            else:
                                # Indirect connection
                                alt_path = []
                                start = min(i_idx, j_idx)
                                end = max(i_idx, j_idx)
                                if end - start == len(path) - 1:
                                    # Wrap around case
                                    alt_path = path[start:] + path[:start+1]
                                else:
                                    alt_path = path[start:end+1]
                                
                                alt_dist = sum(dist_mat[alt_path[k], alt_path[k+1]] 
                                             for k in range(len(alt_path)-1))
                                alt_quality.append(alt_dist)
                    
                    if alt_quality:
                        avg_alt = np.mean(alt_quality)
                        direct_cost = dist_mat[i, j]
                        regret = (avg_alt - direct_cost) / (direct_cost + 1e-10)
                        regret_matrix[i, j] = regret_matrix[j, i] = np.tanh(regret)
        
        return regret_matrix
    
    regret_matrix = compute_memory_regret(distance_matrix, samples)
    
    # ---- 4. ATTENTION-BASED EDGE PRIORITIZATION ----
    def attention_edge_prioritization(dist_mat, edge_feats, node_feats, num_heads=4):
        """Multi-head attention mechanism for edge scoring"""
        n = dist_mat.shape[0]
        
        # Different attention heads focus on different aspects
        attention_scores = np.zeros((n, n, num_heads))
        
        # Head 1: Distance-based attention
        for i in range(n):
            row = dist_mat[i].copy()
            row[i] = np.inf
            if np.any(row < np.inf):
                inv_dist = 1.0 / (row + 1e-10)
                inv_dist[i] = 0
                attention_scores[i, :, 0] = inv_dist / (inv_dist.sum() + 1e-10)
        
        # Head 2: Feature similarity attention
        for i in range(n):
            similarities = []
            for j in range(n):
                if i != j:
                    sim = np.dot(node_feats[i], node_feats[j]) / (
                        np.linalg.norm(node_feats[i]) * np.linalg.norm(node_feats[j]) + 1e-10)
                    similarities.append(sim)
                else:
                    similarities.append(0)
            
            similarities = np.array(similarities)
            if similarities.sum() > 0:
                attention_scores[i, :, 1] = similarities / (similarities.sum() + 1e-10)
        
        # Head 3: Structural role attention (using edge features)
        for i in range(n):
            edge_scores = np.mean(edge_feats[i, :, [2, 3, 5]], axis=1)
            edge_scores[i] = 0
            if edge_scores.sum() > 0:
                attention_scores[i, :, 3] = edge_scores / (edge_scores.sum() + 1e-10)
        
        # Head 4: Path-based attention from samples
        path_attention = np.zeros((n, n))
        for path in samples:
            for idx in range(len(path)):
                i = path[idx]
                j = path[(idx + 1) % len(path)]
                path_attention[i, j] += 1
        
        for i in range(n):
            row = path_attention[i]
            if row.sum() > 0:
                attention_scores[i, :, 2] = row / (row.sum() + 1e-10)
        
        # Combine attention heads
        head_weights = [0.3, 0.2, 0.3, 0.2]  # Learned weights
        combined_scores = np.zeros((n, n))
        for h in range(num_heads):
            combined_scores += head_weights[h] * attention_scores[:, :, h]
        
        return combined_scores
    
    attention_scores = attention_edge_prioritization(distance_matrix, edge_features, node_features)
    
    # ---- 5. ADAPTIVE MST WITH ENSEMBLE GUIDANCE ----
    def compute_ensemble_mst(dist_mat, feature_matrix, regret_mat):
        """Compute MST considering multiple factors"""
        n = dist_mat.shape[0]
        
        # Create combined cost matrix
        combined_cost = dist_mat.copy()
        
        # Adjust based on features and regret
        for i in range(n):
            for j in range(i+1, n):
                feat_bonus = np.mean(feature_matrix[i, j, [2, 3, 5]])
                regret_adj = 1.0 - 0.5 * np.tanh(regret_mat[i, j])
                combined_cost[i, j] = combined_cost[j, i] = (
                    dist_mat[i, j] * (1.0 - 0.3 * feat_bonus) * regret_adj
                )
        
        # Prim's algorithm on combined cost
        mst_mask = np.zeros((n, n), dtype=bool)
        in_mst = np.zeros(n, dtype=bool)
        min_edge = np.full(n, np.inf)
        parent = np.full(n, -1)
        
        min_edge[0] = 0
        
        for _ in range(n):
            # Find minimum edge
            u = -1
            min_val = np.inf
            for v in range(n):
                if not in_mst[v] and min_edge[v] < min_val:
                    min_val = min_edge[v]
                    u = v
            
            if u == -1:
                break
            
            in_mst[u] = True
            if parent[u] != -1:
                mst_mask[u, parent[u]] = True
                mst_mask[parent[u], u] = True
            
            # Update minimum edges
            for v in range(n):
                if not in_mst[v] and combined_cost[u, v] < min_edge[v]:
                    min_edge[v] = combined_cost[u, v]
                    parent[v] = u
        
        return mst_mask
    
    mst_mask = compute_ensemble_mst(distance_matrix, edge_features, regret_matrix)
    
    # ---- 6. FINAL COMBINATION WITH ENSEMBLE WEIGHTS ----
    heuristics = np.zeros((n, n))
    
    # Base from attention scores
    heuristics = attention_scores.copy()
    
    # Apply ensemble components
    # GNN component
    gnn_component = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                gnn_component[i, j] = np.mean(edge_features[i, j, [1, 2, 3, 5]])
    
    # Regret component
    regret_component = 1.0 + regret_matrix
    
    # Lookahead component (from samples)
    lookahead_component = np.ones((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Check how often this edge appears in good positions in samples
                sample_scores = []
                for path in samples:
                    if i in path and j in path:
                        i_idx = path.index(i)
                        j_idx = path.index(j)
                        if abs(i_idx - j_idx) == 1 or abs(i_idx - j_idx) == len(path) - 1:
                            # Good: direct connection
                            sample_scores.append(1.0)
                        else:
                            # Not direct connection
                            sample_scores.append(0.5)
                
                if sample_scores:
                    lookahead_component[i, j] = np.mean(sample_scores)
    
    # Cluster component (based on feature similarity)
    cluster_component = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                similarity = 1.0 - np.linalg.norm(node_features[i] - node_features[j])
                cluster_component[i, j] = np.tanh(similarity * 2)
    
    # Combine with ensemble weights
    for i in range(n):
        for j in range(n):
            if i != j:
                combined = (
                    ensemble_weights['gnn'] * gnn_component[i, j] +
                    ensemble_weights['regret'] * regret_component[i, j] +
                    ensemble_weights['lookahead'] * lookahead_component[i, j] +
                    ensemble_weights['cluster'] * cluster_component[i, j]
                )
                
                # Boost MST edges
                if mst_mask[i, j]:
                    combined *= 1.2
                
                heuristics[i, j] = combined
    
    # ---- 7. FINAL PROCESSING ----
    np.fill_diagonal(heuristics, 0)
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    
    # Normalize to [0, 1] range
    non_zero = heuristics[heuristics > 0]
    if len(non_zero) > 0:
        min_val, max_val = non_zero.min(), non_zero.max()
        if max_val > min_val:
            heuristics = (heuristics - min_val) / (max_val - min_val)
    
    # Apply soft threshold
    threshold = np.percentile(heuristics[heuristics > 0], 20)
    heuristics[heuristics < threshold] *= 0.5
    
    return heuristics
