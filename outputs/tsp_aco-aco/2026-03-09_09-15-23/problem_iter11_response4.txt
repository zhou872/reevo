import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic function for TSP edge selection with context-sensitive
    two-hop penalties and dynamic local density adjustments.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Logarithmic Symmetric Sparsification ====================
    k = max(3, int(np.log2(n)))
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Get k nearest neighbors for each node
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argpartition(distances, k)[:k]
        adjacency_mask[i, nn_indices] = True
    
    # Ensure bidirectionality for undirected graph
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Dynamic Local Density Scoring ====================
    # Compute local density for each node based on its neighborhood
    local_density = np.zeros(n)
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            # Average distance to k nearest neighbors as density measure
            avg_dist = np.mean(distance_matrix[i, valid_mask])
            local_density[i] = 1.0 / (avg_dist + 1e-10)
    
    # Normalize local densities
    density_std = np.std(local_density)
    if density_std > 1e-10:
        local_density = (local_density - np.mean(local_density)) / density_std
    
    # Compute base scores with density-aware adjustment
    base_scores = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if not np.any(valid_mask):
            continue
            
        valid_indices = np.where(valid_mask)[0]
        valid_dists = distance_matrix[i, valid_mask]
        
        # Rank-based weights (closer neighbors get higher weight)
        ranks = np.argsort(np.argsort(valid_dists)) + 1
        weights = 1.0 / ranks
        weights = weights / np.sum(weights)
        
        # Density-aware scaling: favor edges in dense regions
        for idx, j in enumerate(valid_indices):
            density_factor = 1.0 + 0.5 * np.tanh(local_density[i] + local_density[j])
            inv_dist = 1.0 / (valid_dists[idx] + 1e-10)
            base_scores[i, j] = inv_dist * weights[idx] * density_factor
    
    # Symmetrize by taking maximum
    base_scores = np.maximum(base_scores, base_scores.T)
    
    # ==================== 3. Context-Sensitive Two-Hop Penalty ====================
    # Compute instance density for scaling
    instance_density = n / (np.mean(distance_matrix[distance_matrix > 0]) + 1e-10)
    base_penalty = 0.3 + 0.2 * np.tanh(instance_density / n)
    
    # Pre-compute sorted neighbors for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    # Initialize penalty matrix
    penalty_matrix = np.ones_like(distance_matrix)
    
    # Check common neighbors efficiently using vectorization
    for i in range(n):
        # Get valid neighbors from mask
        i_neighbors = np.where(adjacency_mask[i])[0]
        if len(i_neighbors) == 0:
            continue
            
        for j in i_neighbors:
            if i >= j:
                continue
                
            # Find common neighbors between i and j
            j_neighbors = np.where(adjacency_mask[j])[0]
            common_neighbors = np.intersect1d(i_neighbors, j_neighbors)
            common_neighbors = common_neighbors[(common_neighbors != i) & (common_neighbors != j)]
            
            if len(common_neighbors) == 0:
                continue
                
            # Compute two-hop distances through common neighbors
            i_to_common = distance_matrix[i, common_neighbors]
            common_to_j = distance_matrix[common_neighbors, j]
            two_hop_dists = i_to_common + common_to_j
            
            if len(two_hop_dists) == 0:
                continue
                
            min_two_hop = np.min(two_hop_dists)
            direct_dist = distance_matrix[i, j]
            
            if direct_dist > 1e-10:
                ratio = min_two_hop / direct_dist
                
                # Context-sensitive scaling based on common neighbor count
                scale_factor = 1.0 + 0.5 * np.log1p(len(common_neighbors)) / np.log1p(k)
                
                # Apply penalty with density-dependent scaling
                if ratio < 1.5:
                    penalty = base_penalty + (1.0 - base_penalty) * (ratio / 1.5) * scale_factor
                else:
                    penalty = 1.0  # No penalty for poor two-hop alternatives
                    
                penalty_matrix[i, j] = penalty_matrix[j, i] = np.clip(penalty, 0.1, 1.0)
    
    # ==================== 4. Combine Components with Adaptive Weights ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Use penalty matrix only for edges in adjacency mask
    heuristics[adjacency_mask] = base_scores[adjacency_mask] * penalty_matrix[adjacency_mask]
    
    # Add small epsilon to avoid zeros and enable normalization
    heuristics[adjacency_mask] += 1e-12
    
    # ==================== 5. Final Processing ====================
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Symmetrize final matrix
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
