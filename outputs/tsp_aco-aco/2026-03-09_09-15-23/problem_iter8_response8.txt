import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Efficient heuristic for TSP edge prioritization with simplified design.
    
    Features:
    1. Inverse distance with adaptive regret penalty for competing alternatives
    2. Lightweight two-hop analysis using vectorized candidate screening
    3. Rank-based symmetric normalization without complex temperature scaling
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Simplified Adaptive Sparsification ====================
    # Logarithmic connectivity based on problem size
    k = max(3, int(np.log2(n)))
    
    # Efficient nearest neighbor computation using argsort
    sorted_indices = np.argsort(distance_matrix, axis=1)
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Get k-nearest neighbors for each node
    for i in range(n):
        nn_indices = sorted_indices[i, 1:k+1]
        adjacency_mask[i, nn_indices] = True
        
        # Ensure reciprocal connections
        for j in nn_indices:
            if i in sorted_indices[j, 1:k+1]:
                adjacency_mask[j, i] = True
    
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Regret-Augmented Inverse Distance ====================
    # Vectorized regret computation for all nodes simultaneously
    regret_scores = np.zeros_like(distance_matrix, dtype=float)
    
    # Compute minimal distances within neighborhood for each node
    for i in range(n):
        mask_i = adjacency_mask[i]
        if np.any(mask_i):
            valid_dists = distance_matrix[i, mask_i]
            if len(valid_dists) > 0:
                min_dist = np.min(valid_dists)
                
                # Inverse distance with regret penalty
                inv_vals = 1.0 / (distance_matrix[i, mask_i] + 1e-10)
                regret_penalty = np.exp(-(distance_matrix[i, mask_i] - min_dist) / (min_dist + 1e-10))
                
                regret_scores[i, mask_i] = inv_vals * regret_penalty
    
    # Symmetrize scores using maximum
    regret_scores = np.maximum(regret_scores, regret_scores.T)
    
    # ==================== 3. Efficient Two-Hop Analysis ====================
    two_hop_penalty = np.ones_like(distance_matrix)
    
    # Vectorized two-hop assessment for all eligible edges
    edge_mask = np.triu(adjacency_mask, k=1)
    edge_indices = np.where(edge_mask)
    
    if len(edge_indices[0]) > 0:
        i_vals, j_vals = edge_indices
        
        # Precompute top neighbors for all nodes (excluding self)
        check_depth = max(3, int(np.sqrt(n)))
        top_neighbors = sorted_indices[:, 1:check_depth+1]
        
        # Find common candidate nodes for two-hop analysis
        i_candidates = top_neighbors[i_vals]
        j_candidates = top_neighbors[j_vals]
        
        # Vectorized two-hop distance computation
        for idx, (i, j) in enumerate(zip(i_vals, j_vals)):
            candidates = np.unique(np.concatenate([i_candidates[idx], j_candidates[idx]]))
            candidates = candidates[(candidates != i) & (candidates != j)]
            
            if len(candidates) > 0:
                # Compute two-hop distances efficiently
                two_hop_dists = distance_matrix[i, candidates] + distance_matrix[candidates, j]
                if len(two_hop_dists) > 0:
                    min_two_hop = np.min(two_hop_dists)
                    direct_dist = distance_matrix[i, j]
                    
                    if direct_dist > 1e-10:
                        ratio = min_two_hop / direct_dist
                        # Simple penalty curve: high penalty when two-hop is nearly as good
                        penalty = 0.3 + 0.7 * (ratio ** 2) if ratio < 1.5 else 1.0
                        two_hop_penalty[i, j] = two_hop_penalty[j, i] = penalty
    
    # ==================== 4. Rank-Based Weighting ====================
    # Apply mutual rank reinforcement
    rank_bonus = np.ones_like(distance_matrix)
    
    for i in range(n):
        for j in range(i+1, n):
            if adjacency_mask[i, j]:
                i_rank = np.where(sorted_indices[i] == j)[0][0] if j in sorted_indices[i] else n
                j_rank = np.where(sorted_indices[j] == i)[0][0] if i in sorted_indices[j] else n
                
                # Bonus for mutual high ranking
                if i_rank <= 2 and j_rank <= 2:
                    rank_bonus[i, j] = rank_bonus[j, i] = 1.25
                elif i_rank <= 3 or j_rank <= 3:
                    rank_bonus[i, j] = rank_bonus[j, i] = 1.1
    
    # ==================== 5. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Combine scores only for connected edges
    mask_combined = adjacency_mask & (distance_matrix > 0)
    heuristics[mask_combined] = (
        regret_scores[mask_combined] * 
        two_hop_penalty[mask_combined] * 
        rank_bonus[mask_combined]
    )
    
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Row-normalize to create probability distribution
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Symmetrize using arithmetic mean
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
