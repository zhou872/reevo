import numpy as np
from scipy.spatial.distance import squareform, pdist

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic function for TSP edge prioritization that combines:
    1. Adaptive nearest-neighbor sparsification for efficiency
    2. Lightweight regret scoring for edge quality assessment
    3. Dynamic two-hop penalty with instance-aware scaling
    4. Balanced edge selection through quality-normalized ranking
    
    Args:
        distance_matrix: Square symmetric matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for more promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive Sparsification ====================
    # Logarithmic scaling with instance-size awareness
    base_k = max(3, int(np.log2(n)))
    # Adjust based on instance density
    avg_dist = np.mean(distance_matrix[np.triu_indices(n, k=1)])
    std_dist = np.std(distance_matrix[np.triu_indices(n, k=1)])
    density_factor = avg_dist / (std_dist + 1e-10)
    
    if density_factor > 1.5:  # Sparse instance
        k = base_k + 1
    elif density_factor < 0.8:  # Dense instance
        k = max(2, base_k - 1)
    else:
        k = base_k
    
    # Build adjacency mask using k-nearest neighbors with reciprocity
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Pre-sort distances for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nn_indices = sorted_indices[i, 1:k+1]
        adjacency_mask[i, nn_indices] = True
        
        # Ensure mutual connections for symmetry
        for j in nn_indices:
            if i in sorted_indices[j, 1:k+1]:
                adjacency_mask[j, i] = True
    
    # Force symmetry and remove self-loops
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Regret-Weighted Distance Scoring ====================
    inv_scores = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if not np.any(valid_mask):
            continue
            
        # Get distances to valid neighbors
        valid_dists = distance_matrix[i, valid_mask]
        
        # Compute regret scores: difference from best alternative
        if len(valid_dists) >= 2:
            min_dist = np.min(valid_dists)
            # Regret: normalized difference from minimum (higher regret = worse edge)
            regrets = (valid_dists - min_dist) / (min_dist + 1e-10)
            # Convert to weight: lower regret (closer to minimum) gets higher weight
            regret_weights = 1.0 / (1.0 + regrets)
        else:
            regret_weights = np.ones_like(valid_dists)
        
        # Combine inverse distance with regret weights
        inv_vals = 1.0 / (valid_dists + 1e-10)
        weighted_scores = inv_vals * regret_weights
        
        # Normalize by number of neighbors and apply
        neighbor_count = np.sum(valid_mask)
        inv_scores[i, valid_mask] = weighted_scores / (neighbor_count + 1e-10)
    
    # Symmetrize scores
    inv_scores = np.maximum(inv_scores, inv_scores.T)
    
    # ==================== 3. Dynamic Two-Hop Analysis ====================
    two_hop_penalty = np.ones_like(distance_matrix)
    
    # Adaptive scaling of two-hop analysis based on instance characteristics
    if n > 10:
        # For larger instances, scale analysis intensity
        analysis_depth = max(3, int(np.log2(n)))
        
        # Pre-compute neighborhood information
        neighborhood_sets = []
        for i in range(n):
            # Get top neighbors for two-hop analysis
            top_neighbors = sorted_indices[i, 1:analysis_depth+1]
            neighborhood_sets.append(set(top_neighbors))
        
        for i in range(n):
            valid_mask = adjacency_mask[i]
            valid_indices = np.where(valid_mask)[0]
            
            for j in valid_indices:
                if i >= j:
                    continue
                
                # Find overlapping neighbors for efficient two-hop analysis
                common_neighbors = neighborhood_sets[i].intersection(neighborhood_sets[j])
                common_neighbors.discard(i)
                common_neighbors.discard(j)
                
                if common_neighbors:
                    # Vectorized computation of two-hop distances
                    common_list = list(common_neighbors)
                    two_hop_dists = (distance_matrix[i, common_list] + 
                                    distance_matrix[common_list, j])
                    
                    if len(two_hop_dists) > 0:
                        min_two_hop = np.min(two_hop_dists)
                        direct_dist = distance_matrix[i, j]
                        
                        if direct_dist > 1e-10 and min_two_hop < np.inf:
                            # Adaptive penalty based on ratio
                            ratio = min_two_hop / direct_dist
                            # Scale penalty based on instance density
                            base_penalty = 0.4 if density_factor > 1.0 else 0.2
                            scale_factor = 1.0 if density_factor > 1.0 else 0.8
                            penalty = base_penalty + scale_factor * np.clip(ratio, 0.5, 2.0) / 2.0
                            two_hop_penalty[i, j] = two_hop_penalty[j, i] = penalty
    
    # ==================== 4. Edge Quality Balancing ====================
    # Combine components with quality-aware normalization
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Base quality score
                quality = inv_scores[i, j]
                
                # Apply adaptive penalties
                quality *= two_hop_penalty[i, j]
                
                # Additional quality boost for mutual top-rank edges
                i_rank = np.where(sorted_indices[i] == j)[0][0] if j in sorted_indices[i] else n
                j_rank = np.where(sorted_indices[j] == i)[0][0] if i in sorted_indices[j] else n
                
                if i_rank <= 2 and j_rank <= 2:
                    quality *= 1.2  # Mutual top preference bonus
                
                heuristics[i, j] = heuristics[j, i] = quality
    
    # ==================== 5. Final Normalization ====================
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Add small constant to maintain connectivity
    heuristics[adjacency_mask] += 1e-8
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    mask = row_sums == 0
    row_sums[mask] = 1
    heuristics = heuristics / row_sums
    heuristics[mask.flatten(), :] = 0
    
    # Final symmetrization
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
