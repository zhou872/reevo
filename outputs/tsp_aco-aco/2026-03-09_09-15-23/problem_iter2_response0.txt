import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection combining:
    1. MST-based global structure
    2. Dynamic regret adjustment (distance-normalized)
    3. 2-opt lookahead scoring
    4. Cluster-aware density weighting with adaptive sparsification
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes.
        
    Returns:
        Priority matrix of same shape, where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) / (distance_matrix + 1e-12)
    
    # 1. Base inverse distance (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    inv_dist[np.eye(n, dtype=bool)] = 0
    
    # 2. Minimum Spanning Tree edges with adaptive weight
    mst_mask = _compute_mst_mask(distance_matrix)
    mst_weight = 3.0 + 2.0 * np.log1p(n) / n  # Decays with n
    
    # 3. Dynamic regret component (distance-normalized)
    regret_matrix = _compute_dynamic_regret(distance_matrix)
    
    # 4. 2-opt lookahead scoring
    two_opt_scores = _compute_two_opt_scores(distance_matrix)
    
    # 5. Cluster density with geometric mean
    density_bonus = _compute_cluster_density(distance_matrix)
    
    # Combine components with instance-adaptive weights
    regret_weight = 0.25 * (1 + np.log(n) / n)  # Higher for small instances
    two_opt_weight = 0.15 * min(1.0, 100 / n)   # Focus on smaller graphs
    density_weight = 0.8
    
    priorities = inv_dist * (
        1.0 + 
        mst_weight * mst_mask + 
        regret_weight * regret_matrix +
        two_opt_weight * two_opt_scores + 
        density_weight * density_bonus
    )
    
    # 6. Adaptive sparsification with degree constraints
    priorities = _adaptive_sparsify(priorities, n, distance_matrix)
    
    # Ensure symmetry and zero diagonal
    priorities = (priorities + priorities.T) / 2
    priorities[np.eye(n, dtype=bool)] = 0
    
    # Final normalization
    max_val = priorities.max()
    if max_val > 0:
        priorities = priorities / max_val
    
    return priorities


def _compute_mst_mask(dist_matrix: np.ndarray) -> np.ndarray:
    """Return binary mask of edges in Minimum Spanning Tree."""
    n = dist_matrix.shape[0]
    mask = np.zeros_like(dist_matrix, dtype=float)
    
    if n <= 1:
        return mask
    
    # Prim's algorithm for MST
    in_mst = np.zeros(n, dtype=bool)
    min_edge = np.full(n, np.inf)
    parent = np.full(n, -1, dtype=int)
    
    # Start from node 0
    min_edge[0] = 0
    
    for _ in range(n):
        # Find minimum edge to growing MST
        u = -1
        min_val = np.inf
        for v in range(n):
            if not in_mst[v] and min_edge[v] < min_val:
                min_val = min_edge[v]
                u = v
        
        if u == -1:
            break
        
        in_mst[u] = True
        if parent[u] != -1:
            mask[u, parent[u]] = 1
            mask[parent[u], u] = 1
        
        # Update min edges
        for v in range(n):
            if not in_mst[v] and dist_matrix[u, v] < min_edge[v]:
                min_edge[v] = dist_matrix[u, v]
                parent[v] = u
    
    return mask


def _compute_dynamic_regret(dist_matrix: np.ndarray) -> np.ndarray:
    """
    Compute dynamic regret matrix where:
    regret[i,j] = (min_dist_i + min_dist_j) / (2 * dist[i,j])
    Normalized to [0, 1] range.
    """
    n = dist_matrix.shape[0]
    regret = np.zeros_like(dist_matrix)
    
    if n <= 1:
        return regret
    
    # Find each node's minimum distance (excluding self)
    min_dists = np.full(n, np.inf)
    for i in range(n):
        row = dist_matrix[i].copy()
        row[i] = np.inf
        min_dists[i] = np.min(row)
    
    # Compute regret for each edge
    for i in range(n):
        for j in range(i + 1, n):
            if dist_matrix[i, j] > 0:
                # Symmetric regret: average of both nodes' regret
                avg_min_dist = (min_dists[i] + min_dists[j]) / 2
                regret[i, j] = avg_min_dist / dist_matrix[i, j]
    
    # Symmetrize and normalize
    regret = regret + regret.T
    max_val = regret.max()
    if max_val > 0:
        regret = regret / max_val
    
    return regret


def _compute_two_opt_scores(dist_matrix: np.ndarray, k_neighbors: int = 8) -> np.ndarray:
    """
    Compute 2-opt lookahead scores for edges.
    For edge (i,j), consider potential 2-opt swaps with edges (i,k) and (j,l)
    and compute the best possible improvement ratio.
    """
    n = dist_matrix.shape[0]
    scores = np.zeros_like(dist_matrix)
    
    if n < 4:
        return scores
    
    # For each edge (i,j)
    for i in range(n):
        # Get k_neighbors nearest neighbors for i and j
        row_i = np.argsort(dist_matrix[i])[1:k_neighbors+1]  # Exclude self
        row_j = np.argsort(dist_matrix[j])[1:k_neighbors+1]
        
        for j in range(i + 1, n):
            if i == j:
                continue
                
            best_ratio = 0
            ij_dist = dist_matrix[i, j]
            
            # Consider 2-opt swaps: (i,j) + (k,l) vs (i,k) + (j,l)
            for k in row_i:
                if k == j:
                    continue
                for l in row_j:
                    if l == i or l == k:
                        continue
                    
                    current = ij_dist + dist_matrix[k, l]
                    alternative = dist_matrix[i, k] + dist_matrix[j, l]
                    
                    if current > 0 and alternative > 0:
                        ratio = current / alternative
                        if ratio > best_ratio:
                            best_ratio = ratio
            
            # Score is improvement ratio (higher is better)
            scores[i, j] = best_ratio if best_ratio > 1 else 0
    
    # Symmetrize and normalize
    scores = scores + scores.T
    max_val = scores.max()
    if max_val > 0:
        scores = scores / max_val
    
    return scores


def _compute_cluster_density(dist_matrix: np.ndarray) -> np.ndarray:
    """Compute density-based bonus using local neighborhood statistics."""
    n = dist_matrix.shape[0]
    if n <= 3:
        return np.zeros_like(dist_matrix)
    
    # Adaptive neighborhood size
    k = min(5, max(2, n // 10))
    
    # Compute local density for each node using k-nearest neighbors
    densities = np.zeros(n)
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        distances = np.sort(dist_matrix[i])[1:k+1]
        if len(distances) > 0:
            # Density is inverse of median distance
            densities[i] = 1.0 / (np.median(distances) + 1e-12)
    
    # Normalize densities
    if densities.max() > 0:
        densities = densities / densities.max()
    
    # Compute bonus as geometric mean of endpoint densities
    # weighted by inverse distance
    bonus = np.zeros_like(dist_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            if dist_matrix[i, j] > 0:
                geo_mean = np.sqrt(densities[i] * densities[j])
                bonus[i, j] = geo_mean / dist_matrix[i, j]
    
    # Symmetrize and normalize
    bonus = bonus + bonus.T
    max_val = bonus.max()
    if max_val > 0:
        bonus = bonus / max_val
    
    return bonus


def _adaptive_sparsify(priorities: np.ndarray, n: int, 
                       dist_matrix: np.ndarray) -> np.ndarray:
    """
    Adaptively sparsify the priority matrix based on:
    1. Global percentile threshold
    2. Minimum degree constraint per node
    3. Connection quality threshold
    """
    result = priorities.copy()
    
    if n <= 10:
        # Keep more edges for small graphs
        threshold = np.percentile(result[result > 0], 30) if np.any(result > 0) else 0
    else:
        # Adaptive threshold based on graph size
        keep_ratio = max(0.08, 15.0 / n)
        if np.any(result > 0):
            threshold = np.percentile(result[result > 0], 100 * (1 - keep_ratio))
        else:
            threshold = 0
    
    # Apply global threshold
    result[result < threshold] = 0
    
    # Ensure each node has at least 2 connections
    for i in range(n):
        non_zero = np.where(result[i] > 0)[0]
        if len(non_zero) < 2:
            # Find best edges based on combination of priority and distance
            candidates = np.argsort(priorities[i])[-5:]
            for idx in candidates:
                if idx != i and len(non_zero) < 2:
                    result[i, idx] = priorities[i, idx]
                    result[idx, i] = priorities[idx, i]
                    non_zero = np.where(result[i] > 0)[0]
    
    # Additional pruning: remove edges that are too long relative to local optimum
    for i in range(n):
        row_nonzero = np.where(result[i] > 0)[0]
        if len(row_nonzero) > 2:
            # Find minimum distance among kept edges
            local_min = np.min(dist_matrix[i, row_nonzero])
            threshold_dist = 2.5 * local_min
            
            # Remove edges significantly longer than local minimum
            for j in row_nonzero:
                if dist_matrix[i, j] > threshold_dist:
                    result[i, j] = 0
                    result[j, i] = 0
    
    return result
