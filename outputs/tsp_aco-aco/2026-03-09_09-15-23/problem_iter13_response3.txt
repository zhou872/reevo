import numpy as np
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray, 
                  sparsity_factor: float = 1.5,
                  three_hop_limit: int = 10) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with improved sparsification
    and multi-hop analysis.
    
    Features:
    1. Degree-aware adaptive sparsification based on local connectivity
    2. Two-hop and three-hop alternative path analysis
    3. Cluster density bonus for edges within dense regions
    4. Inverse distance weighting with rank-based normalization
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        sparsity_factor: Factor for determining sparsification threshold (default: 1.5)
        three_hop_limit: Maximum number of neighbors to consider for three-hop analysis
    
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    
    # Handle trivial cases
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Degree-aware Adaptive Sparsification ====================
    adjacency_mask = np.zeros((n, n), dtype=bool)
    min_edges_per_node = max(3, int(np.log2(n)))
    max_edges_per_node = min(n-1, int(np.sqrt(n) * 2))
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Find minimum distance from node i
        min_dist = np.min(distances)
        threshold = sparsity_factor * min_dist
        
        # Count edges within threshold
        within_threshold = distances <= threshold
        count_within = np.sum(within_threshold)
        
        # Determine number of edges to keep
        k = min(max_edges_per_node, max(min_edges_per_node, count_within))
        
        # Get k nearest neighbors
        nn_indices = np.argpartition(distances, k-1)[:k]
        adjacency_mask[i, nn_indices] = True
    
    # Ensure symmetry and remove self-loops
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Base Heuristic Computation ====================
    # Compute inverse distance with rank-based weighting
    base_heuristic = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            valid_dists = distance_matrix[i, valid_mask]
            
            # Compute ranks (1 for closest)
            ranks = np.argsort(np.argsort(valid_dists)) + 1
            
            # Rank-based weights (closer ranks get higher weights)
            weights = 1.0 / np.sqrt(ranks)
            weights = weights / np.sum(weights)
            
            # Inverse distance weighted by rank
            inv_dists = 1.0 / (valid_dists + 1e-10)
            base_heuristic[i, valid_mask] = inv_dists * weights
    
    # Symmetrize base heuristic
    base_heuristic = np.maximum(base_heuristic, base_heuristic.T)
    
    # ==================== 3. Multi-hop Lookahead Analysis ====================
    penalty_matrix = np.ones_like(distance_matrix)
    
    # Pre-sort distances for efficient neighbor access
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        # Get active neighbors
        active_neighbors = np.where(adjacency_mask[i])[0]
        
        for j in active_neighbors:
            if i >= j:
                continue
                
            direct_cost = distance_matrix[i, j]
            if direct_cost < 1e-10:
                continue
            
            # ========== Two-hop Analysis ==========
            # Consider top neighbors of both nodes for intermediate nodes
            check_limit = max(3, int(np.sqrt(n)))
            i_neighbors = sorted_indices[i, 1:check_limit+1]
            j_neighbors = sorted_indices[j, 1:check_limit+1]
            
            candidate_nodes = np.unique(np.concatenate([i_neighbors, j_neighbors]))
            candidate_nodes = candidate_nodes[(candidate_nodes != i) & (candidate_nodes != j)]
            
            if len(candidate_nodes) > 0:
                # Vectorized two-hop distance computation
                two_hop_dists = distance_matrix[i, candidate_nodes] + distance_matrix[candidate_nodes, j]
                min_two_hop = np.min(two_hop_dists)
                
                if min_two_hop < np.inf:
                    ratio = min_two_hop / direct_cost
                    # Piecewise penalty based on ratio
                    if ratio <= 1.0:
                        # Two-hop is better - strong penalty
                        penalty_2hop = 0.2
                    elif ratio <= 1.2:
                        # Two-hop is competitive - moderate penalty
                        penalty_2hop = 0.3 + 0.5 * (ratio - 1.0) / 0.2
                    elif ratio <= 1.5:
                        # Two-hop is somewhat competitive
                        penalty_2hop = 0.8 + 0.2 * (ratio - 1.2) / 0.3
                    else:
                        # Two-hop is not competitive
                        penalty_2hop = 1.0
                else:
                    penalty_2hop = 1.0
            else:
                penalty_2hop = 1.0
            
            # ========== Three-hop Analysis ==========
            # For large graphs, limit three-hop analysis to top neighbors
            if n > 20 and three_hop_limit > 0:
                # Get top neighbors for intermediate nodes
                top_i_neighbors = sorted_indices[i, 1:min(three_hop_limit, n-2)+1]
                top_j_neighbors = sorted_indices[j, 1:min(three_hop_limit, n-2)+1]
                
                min_three_hop = np.inf
                
                # Sample promising three-hop paths
                for k in top_i_neighbors:
                    if k == j:
                        continue
                    
                    # Get top neighbors of k
                    top_k_neighbors = sorted_indices[k, 1:min(5, n-3)+1]
                    
                    for l in top_k_neighbors:
                        if l == i or l == j:
                            continue
                        
                        three_hop_cost = (distance_matrix[i, k] + 
                                         distance_matrix[k, l] + 
                                         distance_matrix[l, j])
                        
                        if three_hop_cost < min_three_hop:
                            min_three_hop = three_hop_cost
                
                if min_three_hop < np.inf:
                    ratio_3 = min_three_hop / direct_cost
                    # Penalize if three-hop is competitive
                    if ratio_3 <= 1.3:
                        penalty_3hop = 0.7 + 0.3 * np.clip(ratio_3, 1.0, 1.3) / 1.3
                    else:
                        penalty_3hop = 1.0
                else:
                    penalty_3hop = 1.0
            else:
                penalty_3hop = 1.0
            
            # Combine penalties
            combined_penalty = penalty_2hop * penalty_3hop
            penalty_matrix[i, j] = penalty_matrix[j, i] = combined_penalty
    
    # ==================== 4. Cluster Density Bonus ====================
    # Reward edges that connect to nodes with many short connections
    density_bonus = np.ones_like(distance_matrix)
    
    # Compute local density for each node
    local_density = np.zeros(n)
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            # Average of inverse distances to neighbors
            inv_dists = 1.0 / (distance_matrix[i, valid_mask] + 1e-10)
            local_density[i] = np.mean(inv_dists)
    
    # Normalize densities
    if np.max(local_density) > 0:
        local_density = local_density / np.max(local_density)
    
    # Apply density bonus to edges
    for i in range(n):
        for j in range(i+1, n):
            if adjacency_mask[i, j]:
                # Bonus proportional to geometric mean of local densities
                density_score = 0.5 + 0.5 * np.sqrt(local_density[i] * local_density[j])
                density_bonus[i, j] = density_bonus[j, i] = density_score
    
    # ==================== 5. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Only compute for edges in the sparsified graph
    for i in range(n):
        for j in range(i+1, n):
            if adjacency_mask[i, j]:
                score = base_heuristic[i, j]
                score *= penalty_matrix[i, j]
                score *= density_bonus[i, j]
                heuristics[i, j] = heuristics[j, i] = score
    
    # ==================== 6. Final Processing ====================
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Add small constant to avoid zeros in sparsified edges
    heuristics[adjacency_mask] += 1e-8
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Symmetrize final matrix
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
