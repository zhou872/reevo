import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic function for TSP edge selection balancing computational efficiency
    with adaptive sparsity and robust edge quality assessment.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        # Handle trivial cases
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Adaptive Local Degree Sparsification ====================
    # Base logarithmic sparsity with degree-based adaptation
    k_base = max(3, int(np.log2(n)))
    
    # Compute local density indicators using nearest neighbor distances
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Average distance to k_base nearest neighbors as density proxy
        nn_dists = np.partition(dists, k_base)[:k_base]
        local_density[i] = np.mean(nn_dists)
    
    # Normalize density scores and compute adaptive k per node
    if np.max(local_density) > np.min(local_density):
        density_norm = (local_density - np.min(local_density)) / (np.max(local_density) - np.min(local_density))
        # Scale k between 0.7*k_base and 1.3*k_base based on density
        k_values = np.clip(k_base * (0.7 + 0.6 * density_norm), k_base//2, 2*k_base).astype(int)
    else:
        k_values = np.full(n, k_base, dtype=int)
    
    # Build sparse adjacency mask with reciprocal preference
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        k_i = min(k_values[i], n-1)
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_indices = np.argpartition(dists, k_i)[:k_i]
        adjacency_mask[i, nn_indices] = True
        
        # Mark reciprocal connections
        for j in nn_indices:
            k_j = min(k_values[j], n-1)
            j_dists = distance_matrix[j].copy()
            j_dists[j] = np.inf
            j_nn = np.argpartition(j_dists, k_j)[:k_j]
            if i in j_nn:
                adjacency_mask[j, i] = True
    
    # Ensure symmetry and no self-loops
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Reciprocal Rank Weighting ====================
    # Compute edge scores using reciprocal rank weighting
    edge_scores = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if np.any(valid_mask):
            valid_dists = distance_matrix[i, valid_mask]
            valid_indices = np.where(valid_mask)[0]
            
            # Sort neighbors by distance
            sorted_idx = np.argsort(valid_dists)
            sorted_indices = valid_indices[sorted_idx]
            
            # Reciprocal rank weights: 1/rank for closest neighbor, etc.
            ranks = np.arange(1, len(sorted_indices) + 1)
            weights = 1.0 / ranks
            weights = weights / np.sum(weights)  # Normalize to sum to 1
            
            # Apply weights to inverse distances
            inv_dists = 1.0 / (valid_dists[sorted_idx] + 1e-10)
            edge_scores[i, sorted_indices] = inv_dists * weights
            
            # Adjust for degree: divide by sqrt(degree_i * degree_j) to balance hub nodes
            for idx, j in enumerate(sorted_indices):
                deg_i = np.sum(adjacency_mask[i])
                deg_j = np.sum(adjacency_mask[j])
                if deg_i > 0 and deg_j > 0:
                    edge_scores[i, j] /= np.sqrt(deg_i * deg_j)
    
    # Symmetrize by geometric mean
    edge_scores = np.sqrt(edge_scores * edge_scores.T)
    np.fill_diagonal(edge_scores, 0)
    
    # ==================== 3. Minimal Two-Hop Regret Ratio ====================
    # Fast two-hop alternative analysis
    two_hop_penalty = np.ones_like(distance_matrix)
    
    # Precompute sorted neighbor indices for efficiency
    sorted_neighbors = np.argsort(distance_matrix, axis=1)
    
    # Analyze edges in the sparse graph
    for i in range(n):
        for j in range(i+1, n):
            if adjacency_mask[i, j]:
                # Check limited set of alternative paths
                check_limit = max(3, int(np.sqrt(n)))
                candidates_i = sorted_neighbors[i, 1:check_limit+1]
                candidates_j = sorted_neighbors[j, 1:check_limit+1]
                candidates = np.unique(np.concatenate([candidates_i, candidates_j]))
                candidates = candidates[(candidates != i) & (candidates != j)]
                
                if len(candidates) > 0:
                    # Compute best two-hop alternative
                    two_hop_costs = distance_matrix[i, candidates] + distance_matrix[candidates, j]
                    best_two_hop = np.min(two_hop_costs) if len(two_hop_costs) > 0 else np.inf
                    
                    # Compute regret ratio
                    direct_cost = distance_matrix[i, j]
                    if direct_cost > 1e-10 and best_two_hop < np.inf:
                        ratio = best_two_hop / direct_cost
                        # Simple penalty: 0.5 if excellent alternative exists, 1.0 if none
                        if ratio < 1.1:
                            penalty = 0.5
                        elif ratio < 1.5:
                            penalty = 0.5 + 0.5 * (ratio - 1.1) / 0.4
                        else:
                            penalty = 1.0
                        two_hop_penalty[i, j] = two_hop_penalty[j, i] = penalty
    
    # ==================== 4. Edge Stability Pruning ====================
    # Remove edges that are outliers in local context
    stable_mask = adjacency_mask.copy()
    
    for i in range(n):
        if np.sum(adjacency_mask[i]) >= 3:
            local_edges = distance_matrix[i, adjacency_mask[i]]
            q1 = np.percentile(local_edges, 25)
            q3 = np.percentile(local_edges, 75)
            iqr = q3 - q1
            if iqr > 0:
                # Mark edges more than 1.5 IQR above Q3 as unstable
                threshold = q3 + 1.5 * iqr
                for j in range(n):
                    if adjacency_mask[i, j] and distance_matrix[i, j] > threshold:
                        stable_mask[i, j] = stable_mask[j, i] = False
    
    # ==================== 5. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Apply scores only to stable edges
    heuristics[stable_mask] = edge_scores[stable_mask] * two_hop_penalty[stable_mask]
    
    # Ensure no self-loops
    np.fill_diagonal(heuristics, 0)
    
    # Add epsilon to avoid zero scores
    heuristics[stable_mask] = np.maximum(heuristics[stable_mask], 1e-12)
    
    # Row-normalize to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    valid_rows = row_sums.flatten() > 0
    heuristics[valid_rows] = heuristics[valid_rows] / row_sums[valid_rows]
    
    # Symmetrize by averaging
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
