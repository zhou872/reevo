import numpy as np
from scipy.spatial import KDTree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors.
    
    Features:
    1. Spatial clustering density bonus for edges within dense regions
    2. Distance-regret penalty to avoid suboptimal local connections
    3. Two-step lookahead scoring considering subsequent edge potential
    4. Adaptive sparsification keeping only the most promising edges
    
    Returns a matrix of edge priorities with the same shape as input.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    # 1. Base inverse distance matrix (avoid division by zero)
    epsilon = 1e-10
    inv_distance = 1.0 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_distance, 0)
    
    # 2. Spatial clustering density analysis
    # Convert distance matrix to coordinates (using MDS approximation)
    from sklearn.manifold import MDS
    mds = MDS(n_components=2, dissimilarity='precomputed', random_state=42)
    coords = mds.fit_transform(distance_matrix)
    
    # Use KDTree for efficient density estimation
    tree = KDTree(coords)
    # Adaptive radius based on average nearest neighbor distance
    avg_nn_dist = np.mean(np.partition(distance_matrix, 2, axis=1)[:, 1])
    radius = avg_nn_dist * 1.5
    
    density_bonus = np.zeros((n, n))
    for i in range(n):
        # Find points within radius of point i
        indices = tree.query_ball_point(coords[i], radius)
        density_i = len(indices) / (n * np.pi * radius**2 + 1)
        
        for j in range(n):
            if i != j:
                # Bonus for edges within dense clusters
                if j in indices:
                    # Both points are in the same dense region
                    indices_j = tree.query_ball_point(coords[j], radius)
                    density_j = len(indices_j) / (n * np.pi * radius**2 + 1)
                    density_bonus[i, j] = (density_i + density_j) / 2
    
    # 3. Distance-regret calculation
    regret_penalty = np.zeros((n, n))
    for i in range(n):
        # Get sorted distances from node i
        distances_i = distance_matrix[i].copy()
        distances_i[i] = np.inf
        sorted_idx = np.argsort(distances_i)
        
        for j in range(n):
            if i != j:
                # Find position of edge (i,j) in sorted order
                pos = np.where(sorted_idx == j)[0][0]
                
                # Calculate regret: how much worse is this edge compared to the best?
                best_dist = distances_i[sorted_idx[0]]
                current_dist = distances_i[j]
                
                if pos > 0:
                    # Also compare with second best alternative
                    second_best = distances_i[sorted_idx[1]] if n > 2 else best_dist
                    regret = (current_dist - best_dist) / (second_best - best_dist + epsilon)
                    regret_penalty[i, j] = np.exp(-regret)
    
    # 4. Two-step lookahead scoring
    lookahead_score = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Consider the best next step from j (excluding i)
                distances_from_j = distance_matrix[j].copy()
                distances_from_j[i] = np.inf  # Don't go back immediately
                distances_from_j[j] = np.inf
                
                if np.all(np.isinf(distances_from_j)):
                    best_next = 0
                else:
                    best_next = 1.0 / (np.min(distances_from_j) + epsilon)
                
                lookahead_score[i, j] = best_next
    
    # 5. Combine all factors with weights
    # Weights determined empirically
    w_inv = 1.0      # Inverse distance weight
    w_density = 0.3  # Density bonus weight  
    w_regret = 0.4   # Regret penalty weight
    w_lookahead = 0.2 # Lookahead weight
    
    combined_score = (
        w_inv * inv_distance +
        w_density * density_bonus +
        w_regret * regret_penalty +
        w_lookahead * lookahead_score
    )
    
    # 6. Adaptive sparsification
    # For each node, keep top-k promising edges, zero out others
    sparsified = np.zeros_like(combined_score)
    
    # Adaptive k: smaller for large graphs, larger for small graphs
    k = max(3, min(10, n // 3))
    
    for i in range(n):
        # Get scores for node i (excluding self)
        node_scores = combined_score[i].copy()
        node_scores[i] = -np.inf
        
        # Find top k edges
        if len(node_scores) > k:
            top_k_indices = np.argpartition(node_scores, -k)[-k:]
            threshold = np.min(node_scores[top_k_indices])
            
            # Keep edges above threshold
            mask = node_scores >= threshold
            sparsified[i, mask] = combined_score[i, mask]
        else:
            sparsified[i, :] = combined_score[i, :]
    
    # Ensure symmetry for undirected TSP
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Normalize to reasonable range
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
