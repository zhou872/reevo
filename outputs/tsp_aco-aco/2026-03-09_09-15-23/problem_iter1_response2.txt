import numpy as np
from scipy.spatial.distance import squareform
from scipy.cluster.hierarchy import linkage, fcluster
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Sparsified nearest-neighbor connectivity
    2. Local density-weighted inverse distance
    3. Edge connectivity entropy
    4. Inter-cluster penalty for distant nodes
    
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    np.fill_diagonal(distance_matrix, np.inf)
    
    # 1. Base inverse distance (normalized)
    inv_dist = 1 / distance_matrix
    inv_dist = inv_dist / (np.max(inv_dist) + 1e-8)
    
    # 2. Sparsified nearest-neighbor connectivity
    k = max(2, int(np.sqrt(n)))  # Adaptive k-nearest
    nearest_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        indices = np.argpartition(distance_matrix[i], k)[:k]
        nearest_mask[i, indices] = True
    
    # Symmetrize nearest-neighbor relationships
    nearest_mask = nearest_mask | nearest_mask.T
    
    # 3. Local density weighting
    local_density = np.zeros(n)
    for i in range(n):
        # Median distance to other nodes as density estimator
        local_density[i] = 1 / (np.median(distance_matrix[i]) + 1e-8)
    
    density_weight = np.outer(local_density, local_density)
    density_weight = density_weight / (np.max(density_weight) + 1e-8)
    
    # 4. Edge connectivity entropy (edges connecting diverse neighborhoods)
    edge_entropy = np.zeros_like(distance_matrix)
    if n > 4:
        # Cluster nodes hierarchically
        condensed_dist = squareform(distance_matrix, checks=False)
        Z = linkage(condensed_dist, method='average')
        clusters = fcluster(Z, t=2, criterion='maxclust')
        
        # Entropy bonus for edges connecting different clusters
        for i in range(n):
            for j in range(n):
                if i != j:
                    edge_entropy[i, j] = 1.0 if clusters[i] != clusters[j] else 0.3
    
    # 5. Minimum spanning tree prior
    mst_prior = np.zeros_like(distance_matrix)
    if n > 2:
        mst = minimum_spanning_tree(distance_matrix)
        mst_prior[mst.toarray() > 0] = 1.0
    
    # Combine factors with weights
    heuristics = np.zeros_like(distance_matrix)
    
    # Core formula
    heuristics = (
        0.4 * inv_dist * density_weight +  # Density-weighted inverse distance
        0.3 * edge_entropy +                # Connectivity diversity bonus
        0.3 * mst_prior                     # MST-based prior
    )
    
    # Apply sparsification: zero out non-promising edges
    heuristics = heuristics * nearest_mask
    
    # Additional penalty for extremely long edges
    max_dist = np.max(distance_matrix)
    long_edge_penalty = np.exp(-distance_matrix / (0.3 * max_dist + 1e-8))
    heuristics = heuristics * long_edge_penalty
    
    # Normalize to [0, 1]
    heuristics_max = np.max(heuristics)
    if heuristics_max > 0:
        heuristics = heuristics / heuristics_max
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
