import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.cluster import KMeans

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge prioritization with clustering,
    lookahead evaluation, and dynamic parameter adjustment.
    Returns a matrix of edge priorities with the same shape as distance_matrix.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        inv_dist = np.zeros_like(distance_matrix)
        np.fill_diagonal(inv_dist, 0)
        return inv_dist
    
    # 1. Dynamic parameter adjustment based on problem size and density
    base_k = max(3, min(8, n // 10))
    lookahead_depth = 1 if n < 50 else 2
    use_clustering = n >= 100
    
    # 2. Base inverse distance (regularized)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # 3. Two-step lookahead edge evaluation
    lookahead_scores = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get potential next nodes sorted by distance
        sorted_indices = np.argsort(distance_matrix[i])
        valid_indices = [idx for idx in sorted_indices if idx != i][:base_k*2]
        
        for j in valid_indices:
            if i == j:
                continue
                
            # First step: edge (i,j)
            current_cost = distance_matrix[i, j]
            
            # Look ahead to potential third nodes
            third_nodes = [k for k in range(n) 
                          if k not in (i, j)][:base_k]
            
            if third_nodes:
                # Evaluate potential two-step sequences
                sequences = []
                for k in third_nodes:
                    seq_cost = current_cost + distance_matrix[j, k]
                    
                    # Optionally look further ahead
                    if lookahead_depth > 1 and n >= 10:
                        # Look for a fourth node
                        fourth_nodes = [l for l in range(n) 
                                       if l not in (i, j, k)][:base_k//2]
                        if fourth_nodes:
                            best_fourth = min(distance_matrix[k, l] 
                                            for l in fourth_nodes)
                            seq_cost += best_fourth * 0.5  # Discounted future
                    
                    sequences.append(seq_cost)
                
                if sequences:
                    # Score based on best possible continuation
                    best_sequence = min(sequences)
                    lookahead_scores[i, j] = 1 / (best_sequence + 1e-10)
    
    # 4. Adaptive clustering for large instances
    cluster_bonus = np.zeros_like(distance_matrix)
    
    if use_clustering:
        try:
            # Use KMeans on normalized distances for clustering
            n_clusters = max(2, min(8, n // 20))
            
            # Create feature matrix from distances
            features = distance_matrix.copy()
            row_sums = features.sum(axis=1, keepdims=True)
            features = features / (row_sums + 1e-10)
            
            kmeans = KMeans(n_clusters=n_clusters, random_state=42, n_init=10)
            labels = kmeans.fit_predict(features)
            
            # Boost intra-cluster edges
            for i in range(n):
                for j in range(i+1, n):
                    if labels[i] == labels[j]:
                        # Nodes in same cluster get bonus
                        intra_bonus = 1.5 - (distance_matrix[i, j] / 
                                          distance_matrix[i].mean())
                        cluster_bonus[i, j] = max(0.5, intra_bonus)
                        cluster_bonus[j, i] = cluster_bonus[i, j]
                    else:
                        # Inter-cluster edges get reduced penalty if they're bridges
                        # between close clusters
                        cluster_dist = np.linalg.norm(
                            kmeans.cluster_centers_[labels[i]] - 
                            kmeans.cluster_centers_[labels[j]]
                        )
                        if cluster_dist < np.percentile(distance_matrix, 25):
                            cluster_bonus[i, j] = 0.3
                            cluster_bonus[j, i] = 0.3
        except:
            # Fallback if clustering fails
            pass
    
    # 5. MST influence (sparse computation for large n)
    mst_weight = np.zeros_like(distance_matrix)
    if n < 500:  # MST becomes expensive for very large n
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            
            # Boost MST edges
            for i in range(n):
                for j in range(i+1, n):
                    if mst_dense[i, j] > 0 or mst_dense[j, i] > 0:
                        mst_weight[i, j] = 1.2
                        mst_weight[j, i] = mst_weight[i, j]
        except:
            pass
    
    # 6. Dynamic weight combination based on problem characteristics
    avg_dist = np.mean(distance_matrix[np.triu_indices(n, k=1)])
    std_dist = np.std(distance_matrix[np.triu_indices(n, k=1)])
    density_factor = std_dist / (avg_dist + 1e-10)
    
    # Adjust weights based on problem characteristics
    if n < 50:
        weights = [0.4, 0.35, 0.15, 0.1]  # [inv_dist, lookahead, cluster, mst]
    elif density_factor > 1.5:  # Irregular problem
        weights = [0.3, 0.4, 0.2, 0.1]  # Emphasize lookahead and clustering
    else:  # Regular problem
        weights = [0.5, 0.3, 0.1, 0.1]  # Emphasize base distances
    
    # Combine components
    priorities = (
        weights[0] * inv_dist +
        weights[1] * lookahead_scores +
        weights[2] * cluster_bonus +
        weights[3] * mst_weight
    )
    
    # 7. Hierarchical sparsification
    # First level: Keep top k edges per node
    k1 = max(3, min(10, n // 5))
    temp_priorities = priorities.copy()
    
    for i in range(n):
        row = priorities[i, :].copy()
        row[i] = -np.inf
        
        if n > k1 + 1:
            threshold = np.partition(row, -k1)[-k1]
            mask = row < threshold
            temp_priorities[i, mask] = 0
            temp_priorities[i, i] = 0
    
    # Second level: Global sparsification for large instances
    if n > 100:
        # Keep only edges in top percentiles
        non_zero = temp_priorities[temp_priorities > 0]
        if len(non_zero) > 0:
            percentile_75 = np.percentile(non_zero, 75)
            temp_priorities[temp_priorities < percentile_75] = 0
    
    priorities = temp_priorities
    
    # Ensure symmetry
    priorities = (priorities + priorities.T) / 2
    np.fill_diagonal(priorities, 0)
    
    # Normalize for consistency
    max_val = priorities.max()
    if max_val > 0:
        priorities = priorities / max_val
    
    return priorities
