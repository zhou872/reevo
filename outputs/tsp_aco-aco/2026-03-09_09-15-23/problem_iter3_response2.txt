import numpy as np
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Heuristic function for TSP edge selection using adaptive sparsification and
    two-hop analysis to identify promising edges.
    
    Args:
        distance_matrix: Square matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for more promising edges.
        Unpromising edges are set to zero through sparsification.
    """
    n = len(distance_matrix)
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. Adaptive Sparsification ====================
    # Determine adaptive number of nearest neighbors per node
    k = max(3, int(np.log2(n)))  # Logarithmic scaling
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get distances for this node (excluding self)
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Get indices of k nearest neighbors
        if k < n - 1:
            nn_indices = np.argpartition(distances, k)[:k]
        else:
            nn_indices = np.arange(n)
            nn_indices = nn_indices[nn_indices != i]
        
        adjacency_mask[i, nn_indices] = True
    
    # Ensure symmetry and remove self-connections
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Base Inverse Distance Scoring ====================
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        # Only consider edges in adjacency mask
        mask_i = adjacency_mask[i]
        if np.any(mask_i):
            # Inverse distance with small epsilon to avoid division by zero
            inv_vals = 1.0 / (distance_matrix[i, mask_i] + 1e-10)
            
            # Normalize by number of neighbors to balance edge priorities
            # This gives advantage to nodes with fewer good options
            neighbor_count = np.sum(mask_i)
            inv_dist[i, mask_i] = inv_vals / neighbor_count
    
    # Symmetrize the scores
    inv_dist = np.maximum(inv_dist, inv_dist.T)
    
    # ==================== 3. Two-Hop Path Analysis ====================
    # Compare direct edges with best two-hop alternatives
    two_hop_score = np.zeros_like(distance_matrix)
    
    # For efficiency, precompute nearest neighbors for each node
    nearest_neighbors = []
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        # Get k nearest neighbors
        nn_count = min(k, n-1)
        nn_indices = np.argpartition(distances, nn_count)[:nn_count]
        nearest_neighbors.append(nn_indices)
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                direct_dist = distance_matrix[i, j]
                
                # Find minimum two-hop path through common neighbors
                min_two_hop = np.inf
                
                # Check neighbors of i and j
                for k in nearest_neighbors[i]:
                    if k != j:
                        two_hop_dist = distance_matrix[i, k] + distance_matrix[k, j]
                        min_two_hop = min(min_two_hop, two_hop_dist)
                
                for k in nearest_neighbors[j]:
                    if k != i:
                        two_hop_dist = distance_matrix[i, k] + distance_matrix[k, j]
                        min_two_hop = min(min_two_hop, two_hop_dist)
                
                # If we found a valid two-hop path
                if min_two_hop < np.inf and direct_dist > 1e-10:
                    # Ratio > 1 means direct edge is shorter than best detour
                    ratio = min_two_hop / direct_dist
                    # Apply sigmoid-like transformation to emphasize clear advantages
                    two_hop_score[i, j] = two_hop_score[j, i] = np.tanh(ratio - 1)
    
    # ==================== 4. Minimum Spanning Tree Boost ====================
    # Simple MST computation using Prim's algorithm
    mst_boost = np.ones_like(distance_matrix)
    
    if n > 2:
        # Prim's algorithm for MST
        visited = np.zeros(n, dtype=bool)
        min_edge = np.full(n, np.inf)
        parent = np.full(n, -1, dtype=int)
        min_edge[0] = 0
        
        for _ in range(n):
            # Find minimum edge to unvisited node
            u = -1
            min_val = np.inf
            for i in range(n):
                if not visited[i] and min_edge[i] < min_val:
                    min_val = min_edge[i]
                    u = i
            
            if u == -1:
                break
                
            visited[u] = True
            
            # Add edge to MST if not the root
            if parent[u] != -1:
                mst_boost[u, parent[u]] = 1.5  # Boost MST edges
                mst_boost[parent[u], u] = 1.5
            
            # Update distances to unvisited nodes
            for v in range(n):
                if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                    min_edge[v] = distance_matrix[u, v]
                    parent[v] = u
    
    # ==================== 5. Combine Components ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Apply adjacency mask and combine scores
    mask_indices = np.where(adjacency_mask)
    i_indices, j_indices = mask_indices
    
    # Base inverse distance score
    base_scores = inv_dist[mask_indices]
    
    # Two-hop score (add 1 to avoid zeroing out)
    two_hop_scores = 1.0 + two_hop_score[mask_indices]
    
    # MST boost
    mst_boosts = mst_boost[mask_indices]
    
    # Combine all components
    combined_scores = base_scores * two_hop_scores * mst_boosts
    
    # Apply to heuristic matrix
    heuristics[i_indices, j_indices] = combined_scores
    
    # Ensure symmetry
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # ==================== 6. Final Processing ====================
    # Set diagonal to zero
    np.fill_diagonal(heuristics, 0)
    
    # Row-normalize to create probability-like distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1  # Avoid division by zero
    heuristics = heuristics / row_sums
    
    # Apply final sparsification: keep only top edges per node
    # This ensures the matrix is sparse as requested
    for i in range(n):
        row = heuristics[i].copy()
        if np.sum(row > 0) > k:
            # Find threshold for top k edges
            threshold = np.partition(row, -k)[-k]
            row[row < threshold] = 0
            heuristics[i] = row
    
    # Re-normalize after sparsification
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Final symmetrization
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
