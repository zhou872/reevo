import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic function for TSP edge prioritization that combines:
    1. Lightweight global structure via Minimum Spanning Tree edge bonus
    2. Adaptive sparsification based on local density metrics
    3. Dynamic two-hop regret penalty for edge competitiveness
    4. Rank-weighted inverse distance scoring for edge desirability
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for promising edges
    """
    n = len(distance_matrix)
    
    # Handle trivial cases
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Compute Minimum Spanning Tree ====================
    mst_mask = _compute_mst_mask(distance_matrix)
    
    # ==================== 2. Adaptive Density-Based Sparsification ====================
    # Analyze local density patterns for adaptive connectivity
    local_density_stats = _compute_local_density_stats(distance_matrix)
    
    # Compute adaptive k for each node based on local density
    k_values = _compute_adaptive_k(local_density_stats, n)
    
    # Build adjacency mask with adaptive k-nearest neighbors
    adjacency_mask = _build_adaptive_adjacency(distance_matrix, k_values, mst_mask)
    
    # ==================== 3. Rank-Weighted Inverse Distance Scoring ====================
    base_scores = _compute_rank_weighted_scores(distance_matrix, adjacency_mask, k_values)
    
    # ==================== 4. Dynamic Two-Hop Regret Penalty ====================
    regret_penalties = _compute_regret_penalties(distance_matrix, adjacency_mask)
    
    # ==================== 5. Combine Components with MST Bonus ====================
    heuristics = _combine_components(
        base_scores, 
        regret_penalties, 
        mst_mask, 
        adjacency_mask
    )
    
    return heuristics


def _compute_mst_mask(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute Minimum Spanning Tree using Prim's algorithm."""
    n = len(distance_matrix)
    mst_mask = np.zeros((n, n), dtype=bool)
    
    if n == 0:
        return mst_mask
    
    # Prim's algorithm for MST
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    min_edge_weights = np.full(n, np.inf)
    min_edge_sources = np.full(n, -1, dtype=int)
    
    for _ in range(n - 1):
        # Find minimum edge from visited to unvisited
        min_weight = np.inf
        min_target = -1
        min_source = -1
        
        for i in range(n):
            if visited[i]:
                for j in range(n):
                    if not visited[j] and i != j:
                        weight = distance_matrix[i, j]
                        if weight < min_weight:
                            min_weight = weight
                            min_target = j
                            min_source = i
        
        if min_target != -1:
            visited[min_target] = True
            mst_mask[min_source, min_target] = True
            mst_mask[min_target, min_source] = True
    
    return mst_mask


def _compute_local_density_stats(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute local density statistics for each node."""
    n = len(distance_matrix)
    
    # For each node, compute average distance to k nearest neighbors
    base_k = max(2, int(np.log2(n)))
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    density_stats = np.zeros(n)
    for i in range(n):
        # Get distances to base_k nearest neighbors (excluding self)
        nn_distances = distance_matrix[i, sorted_indices[i, 1:base_k+1]]
        density_stats[i] = np.mean(nn_distances)
    
    return density_stats


def _compute_adaptive_k(density_stats: np.ndarray, n: int) -> np.ndarray:
    """Compute adaptive k values for each node based on local density."""
    base_k = max(3, int(np.log2(n)))
    mean_density = np.mean(density_stats)
    std_density = np.std(density_stats)
    
    k_values = np.full(n, base_k, dtype=int)
    
    if std_density > 1e-10:
        # Adjust k based on z-score of local density
        z_scores = (density_stats - mean_density) / std_density
        
        for i in range(n):
            # In dense regions (low distance to neighbors), use smaller k
            # In sparse regions (high distance to neighbors), use larger k
            adjustment = int(np.round(z_scores[i] * -0.5))  # Negative correlation
            k_values[i] = max(2, min(base_k + adjustment, n - 1))
    
    return k_values


def _build_adaptive_adjacency(distance_matrix: np.ndarray, 
                             k_values: np.ndarray,
                             mst_mask: np.ndarray) -> np.ndarray:
    """Build adjacency mask with adaptive k-nearest neighbors and MST edges."""
    n = len(distance_matrix)
    adjacency_mask = mst_mask.copy()  # Start with MST edges
    
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    # Add k-nearest neighbors for each node
    for i in range(n):
        k_i = k_values[i]
        nn_indices = sorted_indices[i, 1:k_i+1]
        adjacency_mask[i, nn_indices] = True
        
        # Ensure reciprocal connections for symmetry
        for j in nn_indices:
            adjacency_mask[j, i] = True
    
    np.fill_diagonal(adjacency_mask, False)
    return adjacency_mask


def _compute_rank_weighted_scores(distance_matrix: np.ndarray,
                                 adjacency_mask: np.ndarray,
                                 k_values: np.ndarray) -> np.ndarray:
    """Compute rank-weighted inverse distance scores."""
    n = len(distance_matrix)
    scores = np.zeros_like(distance_matrix, dtype=float)
    
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if not np.any(valid_mask):
            continue
            
        valid_dists = distance_matrix[i, valid_mask]
        valid_indices = np.where(valid_mask)[0]
        
        # Compute ranks within valid neighbors
        dist_order = np.argsort(valid_dists)
        ranks = np.zeros_like(valid_dists)
        ranks[dist_order] = np.arange(len(valid_dists)) + 1
        
        # Exponential rank weighting
        rank_weights = np.exp(-0.5 * (ranks - 1))
        rank_weights = rank_weights / np.sum(rank_weights)
        
        # Inverse distance with smoothing
        inv_dists = 1.0 / (valid_dists + 1e-10)
        
        # Local scaling based on adaptive k
        scale_factor = 1.0 / np.sqrt(k_values[i])
        
        scores[i, valid_indices] = inv_dists * rank_weights * scale_factor
    
    # Symmetrize scores
    scores_sym = np.sqrt(np.maximum(scores * scores.T, 1e-16))
    np.fill_diagonal(scores_sym, 0)
    
    return scores_sym


def _compute_regret_penalties(distance_matrix: np.ndarray,
                             adjacency_mask: np.ndarray) -> np.ndarray:
    """Compute dynamic two-hop regret penalties."""
    n = len(distance_matrix)
    penalties = np.ones_like(distance_matrix)
    
    sorted_indices = np.argsort(distance_matrix, axis=1)
    check_limit = max(4, int(np.sqrt(n)))
    
    # Precompute nearest neighbors for efficiency
    nearest_neighbors = []
    for i in range(n):
        nearest_neighbors.append(sorted_indices[i, 1:check_limit+1])
    
    # Vectorized computation for upper triangle
    for i in range(n):
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                continue
                
            direct_dist = distance_matrix[i, j]
            if direct_dist < 1e-10:
                continue
            
            # Get top neighbors (excluding each other)
            i_neighbors = [x for x in nearest_neighbors[i] if x != j]
            j_neighbors = [x for x in nearest_neighbors[j] if x != i]
            
            # Combine candidate intermediate nodes
            candidates = np.unique(np.concatenate([i_neighbors, j_neighbors]))
            candidates = candidates[(candidates != i) & (candidates != j)]
            
            if len(candidates) > 0:
                # Compute two-hop distances through candidates
                two_hop_dists = distance_matrix[i, candidates] + distance_matrix[candidates, j]
                min_two_hop = np.min(two_hop_dists)
                
                # Compute regret ratio
                regret_ratio = min_two_hop / direct_dist
                
                # Apply adaptive penalty based on regret
                if regret_ratio < 1.0:
                    # Two-hop is better - strong penalty
                    penalty = 0.2 + 0.8 * np.exp(-3.0 * (1.0 - regret_ratio))
                elif regret_ratio < 1.5:
                    # Competitive - moderate penalty
                    penalty = 0.5 + 0.5 * np.exp(-2.0 * (regret_ratio - 1.0))
                else:
                    # Direct is much better - light penalty
                    penalty = 0.8 + 0.2 * np.exp(-1.0 * (regret_ratio - 1.5))
                
                penalties[i, j] = penalties[j, i] = np.clip(penalty, 0.1, 1.0)
    
    return penalties


def _combine_components(base_scores: np.ndarray,
                       regret_penalties: np.ndarray,
                       mst_mask: np.ndarray,
                       adjacency_mask: np.ndarray) -> np.ndarray:
    """Combine all components with MST bonus and normalize."""
    n = len(base_scores)
    
    # Initialize heuristics
    heuristics = np.zeros_like(base_scores)
    
    # Apply scoring only to adjacency mask edges
    for i in range(n):
        for j in range(i + 1, n):
            if not adjacency_mask[i, j]:
                continue
                
            # Base score with regret penalty
            score = base_scores[i, j] * regret_penalties[i, j] + 1e-8
            
            # Apply MST bonus
            if mst_mask[i, j]:
                score *= 1.3  # Bonus for MST edges
            
            heuristics[i, j] = heuristics[j, i] = score
    
    # Row normalization
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Final symmetrization and cleaning
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
