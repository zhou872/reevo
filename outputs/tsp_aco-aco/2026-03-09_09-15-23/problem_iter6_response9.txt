import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization combining:
    1. Regret-weighted inverse distance scoring with local density adaptation
    2. MST edge-betweenness centrality for structural importance
    3. Two-hop alternative path analysis with adaptive penalty scaling
    4. Instance-adaptive sparsification and parameter tuning
    
    Args:
        distance_matrix: Square symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for more promising edges.
        Matrix is symmetric, row-normalized, and sparse.
    """
    n = len(distance_matrix)
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. Instance Characterization ====================
    # Compute global statistics for adaptive parameter tuning
    triu_indices = np.triu_indices(n, k=1)
    global_distances = distance_matrix[triu_indices]
    
    avg_dist = np.mean(global_distances)
    std_dist = np.std(global_distances)
    cv_dist = std_dist / (avg_dist + 1e-10)  # Coefficient of variation
    
    # Classify instance type
    if cv_dist < 0.3:
        instance_type = 'uniform'
        sparsity_factor = 0.8
        regret_weight = 0.7
    elif cv_dist > 0.7:
        instance_type = 'clustered'
        sparsity_factor = 1.2
        regret_weight = 0.9
    else:
        instance_type = 'mixed'
        sparsity_factor = 1.0
        regret_weight = 0.8
    
    # ==================== 2. Adaptive Sparsification ====================
    # Base connectivity per node with instance-aware scaling
    base_k = max(3, int(np.log2(n) * sparsity_factor))
    k_values = np.full(n, base_k, dtype=int)
    
    # Adjust k based on local density variations
    if n > 10:
        for i in range(n):
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            
            # Compute local distance variance metric
            k_nearest = min(base_k, n-1)
            nearest_dists = np.partition(distances, k_nearest)[:k_nearest]
            nearest_dists = np.maximum(nearest_dists, 1e-10)
            
            log_variance = np.var(np.log(nearest_dists))
            density_factor = 1.0 + np.tanh(log_variance)
            k_values[i] = max(3, int(base_k * density_factor))
    
    # Build adjacency mask with adaptive connectivity
    adjacency_mask = np.zeros((n, n), dtype=bool)
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        k_i = min(k_values[i], n-1)
        if k_i < n-1:
            nn_indices = sorted_indices[i, 1:k_i+1]
        else:
            nn_indices = np.arange(n)
            nn_indices = nn_indices[nn_indices != i]
        
        adjacency_mask[i, nn_indices] = True
    
    # Ensure symmetry
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 3. Regret-Weighted Inverse Distance ====================
    regret_scores = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if not np.any(valid_mask):
            continue
            
        valid_dists = distance_matrix[i, valid_mask]
        
        # Compute regret scores
        min_dist = np.min(valid_dists)
        if min_dist > 1e-10:
            # Normalized regret: (distance - best_distance) / best_distance
            regrets = (valid_dists - min_dist) / (min_dist + 1e-10)
            # Convert to weights: lower regret gets higher weight
            regret_weights = 1.0 / (1.0 + regrets)
            
            # Combine inverse distance with regret weights
            inv_dists = 1.0 / (valid_dists + 1e-10)
            combined = inv_dists * (regret_weight + (1 - regret_weight) * regret_weights)
            
            # Normalize by local connectivity
            neighbor_count = np.sum(valid_mask)
            regret_scores[i, valid_mask] = combined / (neighbor_count + 1e-10)
        else:
            regret_scores[i, valid_mask] = 1.0 / (np.sum(valid_mask) + 1e-10)
    
    # Symmetrize
    regret_scores = np.maximum(regret_scores, regret_scores.T)
    
    # ==================== 4. Two-Hop Alternative Path Analysis ====================
    two_hop_penalty = np.ones_like(distance_matrix)
    
    if n > 5:
        # Adaptive analysis depth based on instance type
        if instance_type == 'clustered':
            analysis_depth = min(5, max(2, int(np.log2(n))))
        else:
            analysis_depth = min(8, max(3, int(np.log2(n) * 1.5)))
        
        # Precompute neighborhoods for efficiency
        neighborhoods = []
        for i in range(n):
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            k_analysis = min(analysis_depth, n-1)
            if k_analysis < n-1:
                nn = np.argpartition(distances, k_analysis)[:k_analysis]
            else:
                nn = np.arange(n)
                nn = nn[nn != i]
            neighborhoods.append(set(nn))
        
        for i in range(n):
            for j in range(i + 1, n):
                if adjacency_mask[i, j]:
                    direct_dist = distance_matrix[i, j]
                    
                    # Find common neighbors for two-hop paths
                    common = neighborhoods[i].intersection(neighborhoods[j])
                    common.discard(i)
                    common.discard(j)
                    
                    if common:
                        common_list = list(common)
                        two_hop_dists = distance_matrix[i, common_list] + distance_matrix[common_list, j]
                        min_two_hop = np.min(two_hop_dists)
                        
                        if direct_dist > 1e-10:
                            ratio = min_two_hop / direct_dist
                            
                            # Instance-aware penalty scaling
                            if instance_type == 'uniform':
                                # Less penalty for uniform instances
                                penalty = 0.5 + 0.3 * np.clip(ratio, 0.8, 2.0)
                            elif instance_type == 'clustered':
                                # Stronger penalty for clustered instances
                                penalty = 0.3 + 0.4 * np.clip(ratio, 0.5, 1.5)
                            else:
                                penalty = 0.4 + 0.35 * np.clip(ratio, 0.6, 1.8)
                            
                            two_hop_penalty[i, j] = two_hop_penalty[j, i] = penalty
    
    # ==================== 5. MST Edge-Betweenness Centrality ====================
    centrality_boost = np.ones_like(distance_matrix, dtype=float)
    
    if n > 2:
        # Compute MST using Prim's algorithm
        visited = np.zeros(n, dtype=bool)
        min_edge = np.full(n, np.inf)
        parent = np.full(n, -1, dtype=int)
        min_edge[0] = 0
        
        for _ in range(n):
            # Find minimum edge to unvisited node
            u = -1
            min_val = np.inf
            for i in range(n):
                if not visited[i] and min_edge[i] < min_val:
                    min_val = min_edge[i]
                    u = i
            
            if u == -1:
                break
                
            visited[u] = True
            
            if parent[u] != -1:
                centrality_boost[u, parent[u]] = 1.5
                centrality_boost[parent[u], u] = 1.5
            
            # Update distances
            for v in range(n):
                if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                    min_edge[v] = distance_matrix[u, v]
                    parent[v] = u
        
        # Compute edge betweenness centrality on MST
        # For tree: betweenness of edge (u,v) = size(u_component) * size(v_component)
        if n > 3:
            # Build MST adjacency
            mst_adj = [[] for _ in range(n)]
            for v in range(1, n):
                u = parent[v]
                if u != -1:
                    mst_adj[u].append(v)
                    mst_adj[v].append(u)
            
            # Compute subtree sizes via DFS
            subtree_size = np.ones(n, dtype=int)
            
            def dfs_size(node, par):
                for child in mst_adj[node]:
                    if child != par:
                        dfs_size(child, node)
                        subtree_size[node] += subtree_size[child]
            
            dfs_size(0, -1)
            
            # Compute edge betweenness scores
            for v in range(1, n):
                u = parent[v]
                if u != -1:
                    size_v = subtree_size[v]
                    size_u = n - size_v
                    edge_betweenness = size_v * size_u
                    
                    # Normalize and apply boost
                    max_possible = (n // 2) * ((n + 1) // 2)  # Max when split is balanced
                    if max_possible > 0:
                        norm_betweenness = edge_betweenness / max_possible
                        # Stronger boost for high-betweenness edges
                        boost_factor = 1.0 + 0.6 * norm_betweenness
                        centrality_boost[u, v] *= boost_factor
                        centrality_boost[v, u] = centrality_boost[u, v]
        
        # Boost edges incident to high-degree MST nodes
        mst_degree = np.zeros(n)
        for v in range(1, n):
            u = parent[v]
            if u != -1:
                mst_degree[u] += 1
                mst_degree[v] += 1
        
        if np.max(mst_degree) > 0:
            norm_degree = mst_degree / np.max(mst_degree)
            degree_threshold = np.median(norm_degree)
            
            for i in range(n):
                if norm_degree[i] > degree_threshold:
                    # Boost connections from high-degree nodes in original graph
                    mask = adjacency_mask[i]
                    if np.any(mask):
                        centrality_boost[i, mask] = np.maximum(
                            centrality_boost[i, mask], 1.0 + 0.3 * norm_degree[i]
                        )
                        centrality_boost[mask, i] = centrality_boost[i, mask]
    
    # ==================== 6. Multiplicative Combination ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Apply multiplicative scoring only to sparse edges
    mask_indices = np.where(adjacency_mask)
    
    if len(mask_indices[0]) > 0:
        i_indices, j_indices = mask_indices
        
        # Get component scores
        regret_vals = regret_scores[mask_indices]
        penalty_vals = two_hop_penalty[mask_indices]
        centrality_vals = centrality_boost[mask_indices]
        
        # Multiplicative combination with instance-aware weights
        if instance_type == 'uniform':
            weights = [0.4, 0.3, 0.3]  # Equal emphasis
        elif instance_type == 'clustered':
            weights = [0.3, 0.4, 0.3]  # Stronger two-hop penalty
        else:
            weights = [0.35, 0.35, 0.3]
        
        # Weighted geometric mean
        epsilon = 1e-10
        combined = (
            np.power(regret_vals + epsilon, weights[0]) *
            np.power(penalty_vals + epsilon, weights[1]) *
            np.power(centrality_vals + epsilon, weights[2])
        )
        
        heuristics[i_indices, j_indices] = combined
    
    # Ensure symmetry
    heuristics = np.maximum(heuristics, heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # ==================== 7. Final Processing ====================
    # Row-normalize
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Final adaptive sparsification: keep top edges per node
    for i in range(n):
        row = heuristics[i].copy()
        nonzero_mask = row > 0
        
        if np.sum(nonzero_mask) > k_values[i]:
            threshold = np.partition(row, -k_values[i])[-k_values[i]]
            row[row < threshold] = 0
            heuristics[i] = row
    
    # Re-normalize after final sparsification
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Final symmetrization
    heuristics = 0.5 * (heuristics + heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
