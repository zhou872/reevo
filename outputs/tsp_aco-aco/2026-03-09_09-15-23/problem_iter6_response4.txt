import numpy as np
from typing import Tuple, List

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge prioritization that combines:
    1. Cluster-augmented sparsification for locality-aware connectivity
    2. Regret-augmented distance scoring with lookahead awareness
    3. Two-step lookahead edge assessment for multi-consequence evaluation
    4. Adaptive instance-aware normalization with reinforcement-inspired feedback
    
    Args:
        distance_matrix: Square symmetric matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for more promising edges
    """
    n = len(distance_matrix)
    if n < 3:
        heuristics = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ==================== 1. Cluster-Augmented Sparsification ====================
    # Adaptive connectivity based on instance size and local density
    base_k = max(3, int(np.log2(n)))
    
    # Detect instance structure through local density analysis
    avg_local_dists = np.mean(np.sort(distance_matrix, axis=1)[:, 1:4], axis=1)
    global_avg = np.mean(avg_local_dists)
    local_variation = np.std(avg_local_dists) / (global_avg + 1e-10)
    
    # Adjust connectivity based on structure uniformity
    if local_variation > 0.5:  # High variation suggests clustered structure
        k = base_k + 2
        use_clustering = True
    else:
        k = base_k
        use_clustering = False
    
    # Build adjacency mask with k-nearest neighbors
    adjacency_mask = np.zeros((n, n), dtype=bool)
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        nn_indices = sorted_indices[i, 1:k+1]
        adjacency_mask[i, nn_indices] = True
        
        # Ensure reciprocal connections for symmetry
        for j in nn_indices:
            if i in sorted_indices[j, 1:k+1]:
                adjacency_mask[j, i] = True
    
    adjacency_mask = adjacency_mask | adjacency_mask.T
    np.fill_diagonal(adjacency_mask, False)
    
    # ==================== 2. Regret-Augmented Distance Scoring ====================
    quality_scores = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_mask = adjacency_mask[i]
        if not np.any(valid_mask):
            continue
            
        valid_dists = distance_matrix[i, valid_mask]
        
        # Enhanced regret scoring with lookahead awareness
        if len(valid_dists) >= 2:
            min_dist = np.min(valid_dists)
            second_min = np.partition(valid_dists, 1)[1] if len(valid_dists) > 1 else min_dist
            
            # Dual regret: distance to best and second-best alternatives
            regret_primary = (valid_dists - min_dist) / (min_dist + 1e-10)
            regret_secondary = (valid_dists - second_min) / (second_min + 1e-10)
            
            # Combine regrets with exponential decay for near-optimal edges
            regret_combined = 0.7 * regret_primary + 0.3 * regret_secondary
            regret_weights = np.exp(-regret_combined)
        else:
            regret_weights = np.ones_like(valid_dists)
        
        # Inverse distance with adaptive scaling
        inv_vals = 1.0 / (valid_dists + 1e-10)
        
        # Local density adjustment
        local_density = len(valid_dists) / k
        density_factor = 0.8 + 0.4 * local_density  # Range: 0.8-1.2
        
        quality_scores[i, valid_mask] = inv_vals * regret_weights * density_factor
    
    # Symmetrize and incorporate clustering bonus if applicable
    quality_scores = 0.5 * (quality_scores + quality_scores.T)
    
    if use_clustering:
        # Simple geometric clustering: nodes with similar nearest-neighbor profiles
        cluster_labels = np.zeros(n, dtype=int)
        nn_profiles = sorted_indices[:, 1:min(5, n)]
        
        for i in range(n):
            if cluster_labels[i] == 0:
                cluster_labels[i] = i + 1
                for j in range(i + 1, n):
                    if np.intersect1d(nn_profiles[i], nn_profiles[j]).size >= 2:
                        cluster_labels[j] = i + 1
        
        # Apply intra-cluster bonus
        for i in range(n):
            for j in range(i + 1, n):
                if adjacency_mask[i, j] and cluster_labels[i] == cluster_labels[j]:
                    quality_scores[i, j] *= 1.3
                    quality_scores[j, i] *= 1.3
    
    # ==================== 3. Two-Step Lookahead Edge Assessment ====================
    lookahead_scores = np.ones_like(distance_matrix)
    
    # Dynamic analysis depth based on instance characteristics
    lookahead_depth = min(8, max(3, int(np.sqrt(n))))
    
    for i in range(n):
        i_neighbors = sorted_indices[i, 1:lookahead_depth+1]
        
        for j in i_neighbors:
            if i >= j or not adjacency_mask[i, j]:
                continue
            
            # Get top neighbors for both endpoints (excluding each other)
            i_top = [nbr for nbr in sorted_indices[i, 1:lookahead_depth+1] if nbr != j]
            j_top = [nbr for nbr in sorted_indices[j, 1:lookahead_depth+1] if nbr != i]
            
            # Evaluate two-step consequences through best intermediate nodes
            best_consequence = np.inf
            
            for k in i_top:
                if k == j:
                    continue
                    
                # Find best continuation from k to j (avoiding i)
                k_to_j_options = [l for l in sorted_indices[k, 1:lookahead_depth+1] 
                                 if l != i and l != k]
                
                if k_to_j_options:
                    best_k_continuation = np.min([distance_matrix[k, l] + distance_matrix[l, j] 
                                                 for l in k_to_j_options[:3]])
                    consequence = distance_matrix[i, k] + best_k_continuation
                    best_consequence = min(best_consequence, consequence)
            
            direct_cost = distance_matrix[i, j]
            
            if best_consequence < np.inf and direct_cost > 1e-10:
                # Adaptive penalty based on consequence analysis
                consequence_ratio = best_consequence / direct_cost
                
                if consequence_ratio < 1.2:
                    penalty = 0.3 + 0.3 * consequence_ratio
                elif consequence_ratio < 1.8:
                    penalty = 0.6 + 0.2 * (consequence_ratio - 1.2) / 0.6
                else:
                    penalty = 0.8 + 0.2 * min(1.0, (consequence_ratio - 1.8) / 2.0)
                
                # Adjust based on local competition
                competition_factor = min(len(i_top), len(j_top)) / lookahead_depth
                penalty = penalty * (0.7 + 0.3 * competition_factor)
                
                lookahead_scores[i, j] = lookahead_scores[j, i] = np.clip(penalty, 0.2, 1.0)
    
    # ==================== 4. Integrated Edge Scoring ====================
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Combine components with instance-aware weighting
    instance_density = np.mean(adjacency_mask) / n
    lookahead_weight = 0.5 if instance_density > 0.2 else 0.3
    
    for i in range(n):
        for j in range(i + 1, n):
            if adjacency_mask[i, j]:
                # Base quality from regret-augmented scoring
                base_quality = quality_scores[i, j]
                
                # Apply lookahead assessment
                quality = base_quality * (lookahead_weight + 
                                         (1 - lookahead_weight) * lookahead_scores[i, j])
                
                # Rank-based reinforcement: edges that are mutually top-ranked get bonus
                i_rank = np.where(sorted_indices[i] == j)[0][0] if j in sorted_indices[i] else n
                j_rank = np.where(sorted_indices[j] == i)[0][0] if i in sorted_indices[j] else n
                
                if i_rank <= 2 and j_rank <= 2:
                    quality *= 1.25
                elif i_rank <= 3 or j_rank <= 3:
                    quality *= 1.1
                
                heuristics[i, j] = heuristics[j, i] = quality
    
    # ==================== 5. Adaptive Instance-Aware Normalization ====================
    np.fill_diagonal(heuristics, 0)
    
    # Add minimal value to maintain connectivity
    heuristics[adjacency_mask] += 1e-8
    
    # Instance-aware normalization
    row_max = heuristics.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    normalized = heuristics / row_max
    
    # Adaptive softmax with instance-dependent temperature
    if n > 20:
        temperature = 0.4  # Cooler for larger instances
    else:
        temperature = 0.6  # Warmer for smaller instances
    
    exp_scores = np.exp(normalized / temperature)
    row_sums = exp_scores.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = exp_scores / row_sums
    
    # Final symmetrization with geometric mean
    heuristics = np.sqrt(heuristics * heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
