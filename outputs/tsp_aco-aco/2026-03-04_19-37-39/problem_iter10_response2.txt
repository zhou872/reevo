import numpy as np
from scipy.sparse import csr_matrix, minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform
from typing import List, Tuple, Dict
import heapq


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with reinforcement learning weight adaptation,
    hybrid tour construction, and hierarchical heuristic fusion.
    
    Key innovations:
    1. Reinforcement learning-based weight adaptation using Q-learning principles
    2. Multi-strategy hybrid tour construction with adaptive selection
    3. Hierarchical graph sampling with landmark-based compression
    4. Hierarchical heuristic fusion with attention mechanisms
    
    Args:
        distance_matrix: Square matrix of pairwise distances (n x n)
        
    Returns:
        Matrix of edge promise indicators (same shape as input)
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. INSTANCE CHARACTERIZATION ====================
    
    # Compute instance statistics
    triu_indices = np.triu_indices(n, k=1)
    edge_lengths = distance_matrix[triu_indices]
    length_mean = np.mean(edge_lengths)
    length_std = np.std(edge_lengths)
    length_cv = length_std / (length_mean + 1e-10)
    
    # Instance complexity classification
    if n <= 50:
        complexity = "small"
    elif n <= 500:
        complexity = "medium"
    elif n <= 5000:
        complexity = "large"
    else:
        complexity = "ultra_large"
    
    # ==================== 2. HIERARCHICAL GRAPH SAMPLING ====================
    
    def hierarchical_sampling(nodes: np.ndarray, target_size: int) -> np.ndarray:
        """Hierarchical sampling with landmark selection and coverage optimization."""
        if len(nodes) <= target_size:
            return nodes
        
        # First level: landmark selection via farthest-first traversal
        selected = []
        # Start with random node
        current = np.random.choice(nodes)
        selected.append(current)
        
        # Select landmarks using farthest-first traversal
        while len(selected) < min(target_size // 2, len(nodes)):
            distances = np.min(distance_matrix[nodes[:, None], selected], axis=1)
            farthest_idx = np.argmax(distances)
            selected.append(nodes[farthest_idx])
        
        # Second level: coverage optimization
        remaining_nodes = np.setdiff1d(nodes, selected)
        if len(remaining_nodes) > 0 and len(selected) < target_size:
            # Select nodes that best cover the remaining space
            coverage_scores = []
            for node in remaining_nodes:
                # Compute how well this node covers remaining nodes
                node_distances = distance_matrix[node, remaining_nodes]
                coverage = np.sum(np.exp(-node_distances / (length_mean + 1e-10)))
                coverage_scores.append((coverage, node))
            
            # Select nodes with highest coverage scores
            coverage_scores.sort(reverse=True)
            additional_count = min(len(coverage_scores), target_size - len(selected))
            for i in range(additional_count):
                selected.append(coverage_scores[i][1])
        
        return np.array(selected)
    
    # Determine sampling parameters based on complexity
    if complexity == "small":
        sample_size = n
        use_sampling = False
    elif complexity == "medium":
        sample_size = min(300, n)
        use_sampling = n > 300
    elif complexity == "large":
        sample_size = min(200, n // 4)
        use_sampling = True
    else:  # ultra_large
        sample_size = min(100, n // 10)
        use_sampling = True
    
    if use_sampling:
        sample_nodes = hierarchical_sampling(np.arange(n), sample_size)
    else:
        sample_nodes = np.arange(n)
    
    # ==================== 3. REINFORCEMENT LEARNING WEIGHT ADAPTATION ====================
    
    class RLWeightAdaptor:
        """Reinforcement learning-based weight adaptation for heuristic factors."""
        
        def __init__(self, n_factors: int, instance_stats: Dict):
            self.n_factors = n_factors
            self.instance_stats = instance_stats
            
            # Initialize Q-values based on instance characteristics
            if instance_stats['complexity'] == "small":
                self.q_values = np.ones(n_factors) * 0.5
            elif instance_stats['complexity'] == "medium":
                self.q_values = np.array([0.4, 0.6, 0.5, 0.7, 0.3, 0.5])
            elif instance_stats['complexity'] == "large":
                self.q_values = np.array([0.3, 0.4, 0.6, 0.8, 0.2, 0.4])
            else:
                self.q_values = np.array([0.2, 0.3, 0.7, 0.9, 0.1, 0.3])
            
            self.learning_rate = 0.1
            self.discount_factor = 0.9
            self.epsilon = 0.1  # Exploration rate
            
        def get_weights(self) -> np.ndarray:
            """Get current weights with softmax exploration."""
            # Apply softmax with temperature
            temperature = max(0.1, 1.0 - self.instance_stats['length_cv'])
            exp_q = np.exp(self.q_values / temperature)
            weights = exp_q / exp_q.sum()
            
            # Add epsilon-greedy exploration
            if np.random.random() < self.epsilon:
                weights = np.ones_like(weights) / len(weights)
            
            return weights
        
        def update(self, factor_performances: np.ndarray, reward: float):
            """Update Q-values based on performance and reward."""
            # TD update
            target = reward + self.discount_factor * np.max(self.q_values)
            td_error = target - self.q_values
            
            # Weight updates proportional to factor performance
            performance_weights = factor_performances / (factor_performances.sum() + 1e-10)
            self.q_values += self.learning_rate * performance_weights * td_error
            
            # Normalize Q-values to prevent divergence
            self.q_values = np.clip(self.q_values, 0, 1)
    
    # ==================== 4. HYBRID TOUR CONSTRUCTION ====================
    
    def hybrid_tour_construction(method: str, start_node: int = None) -> np.ndarray:
        """Hybrid tour construction combining multiple strategies."""
        if start_node is None:
            start_node = np.random.randint(0, n)
        
        if method == "enhanced_nn":
            # Enhanced nearest neighbor with lookahead
            tour = [start_node]
            unvisited = set(range(n)) - {start_node}
            
            while unvisited:
                current = tour[-1]
                
                # Lookahead: consider next two steps
                best_score = float('inf')
                best_next = None
                
                for candidate in list(unvisited)[:min(20, len(unvisited))]:
                    # First step score
                    step1 = distance_matrix[current, candidate]
                    
                    # Second step lookahead
                    remaining = unvisited - {candidate}
                    if remaining:
                        next_candidates = list(remaining)[:min(10, len(remaining))]
                        step2_options = [distance_matrix[candidate, next_cand] 
                                        for next_cand in next_candidates]
                        step2 = np.mean(step2_options) if step2_options else step1
                    else:
                        step2 = distance_matrix[candidate, start_node]
                    
                    score = step1 * 0.7 + step2 * 0.3
                    if score < best_score:
                        best_score = score
                        best_next = candidate
                
                tour.append(best_next)
                unvisited.remove(best_next)
            
            return np.array(tour)
        
        elif method == "savings_enhanced":
            # Enhanced savings algorithm with randomization
            if n < 3:
                return np.arange(n)
            
            # Start with dummy tours from depot (node 0)
            depot = start_node
            routes = [[depot, i, depot] for i in range(n) if i != depot]
            
            # Compute savings with randomization
            savings = []
            for i in range(n):
                if i == depot:
                    continue
                for j in range(i + 1, n):
                    if j == depot:
                        continue
                    saving = (distance_matrix[i, depot] + 
                             distance_matrix[depot, j] - 
                             distance_matrix[i, j])
                    # Add small noise for diversity
                    saving += np.random.normal(0, saving * 0.1)
                    savings.append((saving, i, j))
            
            savings.sort(reverse=True, key=lambda x: x[0])
            
            # Merge routes with randomization
            for _, i, j in savings:
                route_i = next((r for r in routes if (r[1] == i or r[-2] == i)), None)
                route_j = next((r for r in routes if (r[1] == j or r[-2] == j)), None)
                
                if route_i and route_j and route_i != route_j:
                    # Randomly choose merge direction
                    if np.random.random() < 0.5:
                        if route_i[1] == i and route_j[-2] == j:
                            new_route = route_j[:-1] + route_i[1:]
                        elif route_i[-2] == i and route_j[1] == j:
                            new_route = route_i[:-1] + route_j[1:]
                        else:
                            continue
                    else:
                        if route_i[1] == i and route_j[1] == j:
                            new_route = list(reversed(route_j[:-1])) + route_i[1:]
                        elif route_i[-2] == i and route_j[-2] == j:
                            new_route = route_i[:-1] + list(reversed(route_j[1:]))
                        else:
                            continue
                    
                    routes.remove(route_i)
                    routes.remove(route_j)
                    routes.append(new_route)
            
            # Return the largest route
            final_route = max(routes, key=len)
            return np.array(final_route[1:-1])
        
        elif method == "randomized_insertion":
            # Randomized insertion heuristic
            if n < 4:
                return np.random.permutation(n)
            
            # Start with triangle of farthest nodes
            # Find three nodes maximizing perimeter
            best_perim = -1
            triangle = [0, 1, 2]
            
            for i in range(min(n, 50)):
                for j in range(i + 1, min(n, 50)):
                    for k in range(j + 1, min(n, 50)):
                        perim = (distance_matrix[i, j] + 
                                distance_matrix[j, k] + 
                                distance_matrix[k, i])
                        if perim > best_perim:
                            best_perim = perim
                            triangle = [i, j, k]
            
            tour = triangle.copy()
            unvisited = set(range(n)) - set(tour)
            
            # Randomized insertion
            while unvisited:
                # Select random node to insert
                node = np.random.choice(list(unvisited))
                
                # Find best insertion position with randomness
                best_pos = 0
                best_increase = float('inf')
                
                for i in range(len(tour)):
                    j = (i + 1) % len(tour)
                    increase = (distance_matrix[tour[i], node] + 
                               distance_matrix[node, tour[j]] - 
                               distance_matrix[tour[i], tour[j]])
                    # Add random perturbation
                    increase *= (1 + np.random.normal(0, 0.1))
                    
                    if increase < best_increase:
                        best_increase = increase
                        best_pos = i + 1
                
                tour.insert(best_pos, node)
                unvisited.remove(node)
            
            return np.array(tour)
        
        else:  # Random as fallback
            return np.random.permutation(n)
    
    def adaptive_local_search(tour: np.ndarray, max_iter: int = 20) -> np.ndarray:
        """Adaptive local search combining multiple move types."""
        current_tour = tour.copy()
        current_len = sum(distance_matrix[current_tour[i], current_tour[(i+1)%n]] 
                         for i in range(n))
        
        for _ in range(max_iter):
            improved = False
            
            # Randomly select move type based on instance size
            move_type = np.random.choice(['2-opt', '3-opt', 'or-opt'], 
                                        p=[0.6, 0.3, 0.1]) if n > 10 else '2-opt'
            
            if move_type == '2-opt':
                # Random 2-opt move
                i = np.random.randint(0, n-1)
                j = np.random.randint(i+2, n)
                if j == n-1 and i == 0:
                    continue
                
                a, b = current_tour[i], current_tour[(i+1)%n]
                c, d = current_tour[j], current_tour[(j+1)%n]
                
                old = distance_matrix[a, b] + distance_matrix[c, d]
                new = distance_matrix[a, c] + distance_matrix[b, d]
                
                if new < old:
                    current_tour[i+1:j+1] = current_tour[i+1:j+1][::-1]
                    current_len = current_len - old + new
                    improved = True
            
            elif move_type == '3-opt' and n >= 6:
                # Random 3-opt move
                i = np.random.randint(0, n-4)
                j = np.random.randint(i+2, n-2)
                k = np.random.randint(j+2, n)
                
                a, b = current_tour[i], current_tour[(i+1)%n]
                c, d = current_tour[j], current_tour[(j+1)%n]
                e, f = current_tour[k], current_tour[(k+1)%n]
                
                old = distance_matrix[a, b] + distance_matrix[c, d] + distance_matrix[e, f]
                
                # Try different 3-opt reconnections
                new_options = [
                    distance_matrix[a, c] + distance_matrix[b, e] + distance_matrix[d, f],
                    distance_matrix[a, d] + distance_matrix[e, b] + distance_matrix[c, f],
                    distance_matrix[a, d] + distance_matrix[e, c] + distance_matrix[b, f]
                ]
                best_new = min(new_options)
                
                if best_new < old:
                    # Apply the best reconnection
                    idx = np.argmin(new_options)
                    if idx == 0:
                        current_tour[i+1:j+1] = current_tour[i+1:j+1][::-1]
                        current_tour[j+1:k+1] = current_tour[j+1:k+1][::-1]
                    elif idx == 1:
                        segment = current_tour[i+1:k+1]
                        current_tour[i+1:j+1] = segment[j-i:]
                        current_tour[j+1:k+1] = segment[:j-i]
                    else:
                        segment = current_tour[i+1:k+1]
                        current_tour[i+1:i+1+(k-j)] = segment[j-i:k-i]
                        current_tour[i+1+(k-j):k+1] = segment[:j-i]
                    
                    current_len = current_len - old + best_new
                    improved = True
            
            elif move_type == 'or-opt' and n >= 8:
                # Or-opt move: relocate a segment
                seg_len = np.random.randint(1, min(4, n//4))
                i = np.random.randint(0, n-seg_len)
                segment = current_tour[i:i+seg_len]
                
                # Try inserting at different positions
                for j in range(n-seg_len+1):
                    if j == i or j == i-seg_len:
                        continue
                    
                    # Calculate delta
                    a = current_tour[(i-1)%n]
                    b = current_tour[i]
                    c = current_tour[i+seg_len-1]
                    d = current_tour[(i+seg_len)%n]
                    
                    e = current_tour[(j-1)%n]
                    f = current_tour[j]
                    
                    old = (distance_matrix[a, b] + 
                          distance_matrix[c, d])
                    new = (distance_matrix[a, d] + 
                          distance_matrix[e, segment[0]] + 
                          distance_matrix[segment[-1], f])
                    
                    if j > i:
                        old += distance_matrix[e, f]
                        new += distance_matrix[e, f]
                    else:
                        old += distance_matrix[e, f]
                        new += distance_matrix[e, f]
                    
                    if new < old:
                        # Perform relocation
                        new_tour = list(current_tour)
                        # Remove segment
                        del new_tour[i:i+seg_len]
                        # Insert at new position
                        insert_pos = j if j < i else j - seg_len
                        new_tour[insert_pos:insert_pos] = segment
                        current_tour = np.array(new_tour)
                        current_len = current_len - old + new
                        improved = True
                        break
            
            if not improved:
                break
        
        return current_tour
    
    # Generate diverse tours for weight learning
    n_tours = min(50, max(15, n // 5))
    tours = []
    tour_lengths = []
    
    for _ in range(n_tours):
        # Select construction method based on instance characteristics
        if np.random.random() < 0.4:
            method = "enhanced_nn"
        elif np.random.random() < 0.7:
            method = "savings_enhanced"
        else:
            method = "randomized_insertion"
        
        tour = hybrid_tour_construction(method)
        
        # Apply adaptive local search
        if n >= 4:
            ls_iter = min(30, max(5, n // 10))
            tour = adaptive_local_search(tour, ls_iter)
        
        tours.append(tour)
        tour_lengths.append(sum(distance_matrix[tour[i], tour[(i+1)%n]] 
                               for i in range(n)))
    
    # ==================== 5. HIERARCHICAL HEURISTIC FUSION ====================
    
    def compute_heuristic_factors(nodes: np.ndarray) -> List[np.ndarray]:
        """Compute multiple heuristic factors on given nodes."""
        sub_n = len(nodes)
        sub_dist = distance_matrix[np.ix_(nodes, nodes)]
        
        factors = []
        
        # 1. Inverse distance factor with attention
        with np.errstate(divide='ignore', invalid='ignore'):
            inv_dist = 1.0 / (sub_dist + 1e-10)
        np.fill_diagonal(inv_dist, 0)
        
        # Apply attention mechanism: weight by node importance
        row_sums = np.sum(sub_dist, axis=1)
        node_importance = 1.0 / (row_sums + 1e-10)
        attention = np.outer(node_importance, node_importance)
        inv_dist_attended = inv_dist * attention
        
        if inv_dist_attended.max() > 0:
            inv_dist_attended = inv_dist_attended / inv_dist_attended.max()
        
        # Map to full matrix
        full_factor = np.zeros((n, n))
        idx = np.ix_(nodes, nodes)
        full_factor[idx] = inv_dist_attended
        factors.append(full_factor)
        
        # 2. MST-based factor
        mst_factor = np.zeros((sub_n, sub_n))
        if sub_n <= 500:
            mst = minimum_spanning_tree(csr_matrix(sub_dist)).toarray()
            mst_mask = (mst > 0) | (mst.T > 0)
            mst_factor[mst_mask] = 1.0
        else:
            # Approximate MST for large instances
            for i in range(sub_n):
                # Connect to nearest neighbors
                nearest = np.argsort(sub_dist[i])[1:4]
                mst_factor[i, nearest] = 0.8
        
        mst_factor = np.maximum(mst_factor, mst_factor.T)
        
        full_factor = np.zeros((n, n))
        full_factor[idx] = mst_factor
        factors.append(full_factor)
        
        # 3. Triangle closure factor
        if sub_n >= 4:
            # Find promising triangles
            triangle_factor = np.zeros((sub_n, sub_n))
            
            # For each node, consider its k-nearest neighbors
            k_tri = min(10, sub_n-1)
            for i in range(sub_n):
                neighbors = np.argsort(sub_dist[i])[1:k_tri+1]
                
                for j in neighbors:
                    # Find common neighbors to form triangles
                    common = np.intersect1d(
                        np.argsort(sub_dist[i])[1:k_tri+1],
                        np.argsort(sub_dist[j])[1:k_tri+1]
                    )
                    
                    if len(common) > 0:
                        # Score based on triangle quality
                        triangle_scores = []
                        for k in common:
                            if k != i and k != j:
                                # Triangle inequality score
                                direct = sub_dist[i, j]
                                via_k = sub_dist[i, k] + sub_dist[k, j]
                                score = max(0, via_k - direct) / (direct + 1e-10)
                                triangle_scores.append(np.exp(-score))
                        
                        if triangle_scores:
                            triangle_factor[i, j] = np.mean(triangle_scores)
            
            triangle_factor = np.maximum(triangle_factor, triangle_factor.T)
            full_factor = np.zeros((n, n))
            full_factor[idx] = triangle_factor
            factors.append(full_factor)
        else:
            factors.append(np.zeros((n, n)))
        
        # 4. Christofides-inspired factor
        christofides_factor = np.zeros((sub_n, sub_n))
        if sub_n >= 4:
            # Simplified Christofides: connect odd-degree MST nodes
            mst_degrees = (mst_factor > 0).sum(axis=1)
            odd_nodes = np.where(mst_degrees % 2 == 1)[0]
            
            if len(odd_nodes) >= 2:
                odd_dist = sub_dist[np.ix_(odd_nodes, odd_nodes)]
                
                # Greedy matching
                matched = set()
                for _ in range(len(odd_nodes) // 2):
                    # Find minimum unmatched edge
                    min_val = np.inf
                    min_pair = (-1, -1)
                    
                    for i in range(len(odd_nodes)):
                        if odd_nodes[i] not in matched:
                            for j in range(i+1, len(odd_nodes)):
                                if odd_nodes[j] not in matched:
                                    if odd_dist[i, j] < min_val:
                                        min_val = odd_dist[i, j]
                                        min_pair = (odd_nodes[i], odd_nodes[j])
                    
                    if min_pair[0] != -1:
                        christofides_factor[min_pair[0], min_pair[1]] = 1.0
                        christofides_factor[min_pair[1], min_pair[0]] = 1.0
                        matched.update(min_pair)
        
        full_factor = np.zeros((n, n))
        full_factor[idx] = christofides_factor
        factors.append(full_factor)
        
        # 5. Node centrality factor
        centrality_factor = np.zeros((sub_n, sub_n))
        
        # Compute betweenness approximation
        betweenness = np.zeros(sub_n)
        for i in range(sub_n):
            # Count shortest paths through i (approximation)
            for j in range(min(20, sub_n)):
                for k in range(j+1, min(20, sub_n)):
                    if j != i and k != i:
                        direct = sub_dist[j, k]
                        via_i = sub_dist[j, i] + sub_dist[i, k]
                        if abs(via_i - direct) < 1e-6:
                            betweenness[i] += 1
        
        if betweenness.max() > 0:
            betweenness = betweenness / betweenness.max()
        
        centrality_factor = np.outer(betweenness, betweenness)
        
        full_factor = np.zeros((n, n))
        full_factor[idx] = centrality_factor
        factors.append(full_factor)
        
        # 6. Edge frequency factor from sampled tours
        freq_factor = np.zeros((n, n))
        if tours:
            # Create edge frequency matrix from top tours
            sorted_indices = np.argsort(tour_lengths)
            top_k = min(10, len(tours))
            top_tours = [tours[i] for i in sorted_indices[:top_k]]
            
            for tour in top_tours:
                for i in range(n):
                    j = (i + 1) % n
                    u, v = tour[i], tour[j]
                    freq_factor[u, v] += 1
                    freq_factor[v, u] += 1
            
            if freq_factor.max() > 0:
                freq_factor = freq_factor / freq_factor.max()
        
        factors.append(freq_factor)
        
        return factors
    
    # ==================== 6. REINFORCEMENT LEARNING WEIGHT OPTIMIZATION ====================
    
    # Initialize RL weight adaptor
    instance_stats = {
        'complexity': complexity,
        'n': n,
        'length_cv': length_cv,
        'length_mean': length_mean
    }
    
    rl_adaptor = RLWeightAdaptor(n_factors=6, instance_stats=instance_stats)
    
    # Compute factors on sampled nodes
    factors = compute_heuristic_factors(sample_nodes)
    
    # Normalize factors
    normalized_factors = []
    for factor in factors:
        f_max = factor.max()
        if f_max > 0:
            normalized_factors.append(factor / f_max)
        else:
            normalized_factors.append(factor)
    
    # Get initial weights and combine factors
    weights = rl_adaptor.get_weights()
    combined = np.zeros((n, n))
    for w, factor in zip(weights, normalized_factors):
        combined += w * factor
    
    # Evaluate factor performance on top tours
    if tours:
        sorted_indices = np.argsort(tour_lengths)
        top_k = min(5, len(tours) // 3)
        top_tours = [tours[i] for i in sorted_indices[:top_k]]
        
        # Create edge mask for top tours
        edge_mask = np.zeros((n, n), dtype=bool)
        for tour in top_tours:
            for i in range(n):
                j = (i + 1) % n
                u, v = tour[i], tour[j]
                edge_mask[u, v] = True
                edge_mask[v, u] = True
        
        # Compute factor performances
        factor_performances = np.zeros(len(factors))
        for i, factor in enumerate(normalized_factors):
            if edge_mask.any():
                factor_performances[i] = factor[edge_mask].mean()
            else:
                factor_performances[i] = 0.5
        
        # Compute reward based on tour quality improvement
        if len(tour_lengths) >= 2:
            best_len = min(tour_lengths)
            avg_len = np.mean(tour_lengths)
            reward = 1.0 - (best_len / (avg_len + 1e-10))
            reward = np.clip(reward, 0, 1)
        else:
            reward = 0.5
        
        # Update RL weights
        rl_adaptor.update(factor_performances, reward)
        
        # Recombine with updated weights
        weights = rl_adaptor.get_weights()
        combined = np.zeros((n, n))
        for w, factor in zip(weights, normalized_factors):
            combined += w * factor
    
    # ==================== 7. ADAPTIVE SPARSIFICATION ====================
    
    # Determine target edges per node
    if complexity == "small":
        target_edges = min(10, n-1)
    elif complexity == "medium":
        target_edges = min(8, max(3, n//8))
    elif complexity == "large":
        target_edges = min(6, max(3, n//15))
    else:
        target_edges = min(4, max(2, n//30))
    
    sparsity_mask = np.zeros_like(combined, dtype=bool)
    
    for i in range(n):
        node_scores = combined[i].copy()
        node_scores[i] = -np.inf
        
        # Select top edges with probabilistic sampling
        n_select = min(target_edges, n-1)
        
        # Use softmax for probabilistic selection
        valid_scores = node_scores[node_scores > -np.inf]
        if len(valid_scores) > 0:
            temperature = np.std(valid_scores) if np.std(valid_scores) > 0 else 1.0
            exp_scores = np.exp(node_scores / temperature)
            exp_scores[i] = 0
            probs = exp_scores / exp_scores.sum()
            
            # Sample edges
            selected = np.random.choice(n, size=min(n_select, n-1), 
                                       replace=False, p=probs)
            sparsity_mask[i, selected] = True
    
    # Ensure symmetry and connectivity
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Ensure minimum degree of 2
    for i in range(n):
        connections = np.sum(sparsity_mask[i])
        if connections < 2:
            # Add connections to nearest neighbors
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            nearest = np.argsort(distances)[:2]
            for j in nearest:
                sparsity_mask[i, j] = True
                sparsity_mask[j, i] = True
    
    # Apply sparsification
    result = np.zeros_like(combined)
    result[sparsity_mask] = combined[sparsity_mask]
    
    # ==================== 8. POST-PROCESSING ====================
    
    # Ensure symmetry and zero diagonal
    result = (result + result.T) / 2
    np.fill_diagonal(result, 0)
    
    # Apply adaptive scaling
    non_zero = result[result > 0]
    if len(non_zero) > 0:
        # Use percentile-based scaling
        q_low, q_high = np.percentile(non_zero, [10, 90])
        result = np.clip(result, q_low, q_high)
        
        # Normalize to [0, 1]
        min_val, max_val = result.min(), result.max()
        if max_val > min_val:
            result = (result - min_val) / (max_val - min_val)
    
    # Apply sigmoid for better contrast
    result = 1 / (1 + np.exp(-8 * (result - 0.5)))
    
    # Final normalization
    result_min, result_max = result.min(), result.max()
    if result_max - result_min > 1e-10:
        result = (result - result_min) / (result_max - result_min)
    
    return result
