import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with dynamic sampling, multi-resolution analysis,
    and adaptive parameter optimization.
    
    Key Innovations:
    1. Dynamic edge betweenness sampling with path diversity optimization
    2. Multi-resolution MST analysis with hierarchical clustering
    3. Local pattern recognition via triangle inequality analysis
    4. Adaptive sparsification with runtime-aware parameter tuning
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, where higher values
        indicate more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. DYNAMIC INSTANCE ANALYSIS =====
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    # Comprehensive statistical analysis
    dist_mean = np.mean(distances)
    dist_std = np.std(distances)
    cv = dist_std / (dist_mean + 1e-10)
    dist_median = np.median(distances)
    q25, q75 = np.percentile(distances, [25, 75])
    iqr = q75 - q25
    
    # Instance complexity score
    complexity = 0.4 * cv + 0.3 * np.log1p(n) / 5 + 0.3 * (iqr / (dist_median + 1e-10))
    
    # ===== 2. DYNAMIC EDGE BETWEENNESS SAMPLING =====
    edge_betweenness = np.zeros_like(distance_matrix, dtype=float)
    
    # Adaptive sampling parameters
    base_samples = max(20, min(100, n // 3))
    diversity_factor = 1.0 + 0.5 * np.tanh(complexity)
    sample_size = int(base_samples * diversity_factor)
    
    # Pathfinding strategies based on instance characteristics
    strategies = []
    if cv < 0.5:
        strategies.append('greedy')
    if n > 20:
        strategies.append('beam')
    if complexity > 0.7:
        strategies.append('diversified')
    
    # Dynamic sampling with multiple strategies
    samples_per_strategy = max(5, sample_size // len(strategies))
    
    for strategy in strategies:
        for _ in range(samples_per_strategy):
            src, tgt = np.random.choice(n, 2, replace=False)
            
            if strategy == 'greedy':
                # Greedy shortest path
                path = [src]
                current = src
                visited = set([src])
                
                while current != tgt and len(visited) < n:
                    mask = np.ones(n, dtype=bool)
                    mask[list(visited)] = False
                    mask[tgt] = True
                    
                    if not np.any(mask):
                        break
                    
                    # Lookahead greedy: consider both distance to current and to target
                    lookahead_scores = 0.6 * distance_matrix[current] + 0.4 * distance_matrix[tgt]
                    lookahead_scores[~mask] = np.inf
                    next_node = np.argmin(lookahead_scores)
                    
                    path.append(next_node)
                    visited.add(next_node)
                    current = next_node
            
            elif strategy == 'beam':
                # Beam search with width 2
                beam = [(src, [src], 0)]
                visited_global = set([src])
                
                for _ in range(min(n, 10)):
                    new_beam = []
                    for node, path_so_far, cost in beam:
                        if node == tgt:
                            new_beam.append((node, path_so_far, cost))
                            continue
                        
                        # Expand node
                        mask = np.ones(n, dtype=bool)
                        mask[list(set(path_so_far))] = False
                        mask[tgt] = True
                        
                        if not np.any(mask):
                            new_beam.append((node, path_so_far, cost))
                            continue
                        
                        # Score potential next nodes
                        scores = 0.5 * distance_matrix[node] + 0.5 * distance_matrix[tgt]
                        scores[~mask] = np.inf
                        
                        # Keep top 2 candidates
                        top_candidates = np.argsort(scores)[:2]
                        for cand in top_candidates:
                            if scores[cand] < np.inf:
                                new_beam.append((cand, path_so_far + [cand], 
                                               cost + distance_matrix[node, cand]))
                    
                    # Keep best 2 paths
                    beam = sorted(new_beam, key=lambda x: x[2])[:2]
                    
                    # Check if target reached
                    if any(node == tgt for node, _, _ in beam):
                        break
                
                # Use best path found
                path = min(beam, key=lambda x: (x[0] != tgt, x[2]))[1]
            
            elif strategy == 'diversified':
                # Randomized exploration with restart
                best_path = None
                best_cost = np.inf
                
                for restart in range(3):
                    path = [src]
                    current = src
                    visited = set([src])
                    
                    while current != tgt and len(visited) < min(n, 15):
                        mask = np.ones(n, dtype=bool)
                        mask[list(visited)] = False
                        mask[tgt] = True
                        
                        if not np.any(mask):
                            break
                        
                        # Boltzmann exploration
                        scores = 0.5 * distance_matrix[current] + 0.5 * distance_matrix[tgt]
                        scores[~mask] = np.inf
                        
                        valid_scores = scores[mask]
                        if len(valid_scores) == 0:
                            break
                        
                        # Temperature decreases with path length
                        temperature = 0.3 * dist_mean * (1 - len(path) / 15)
                        exp_scores = np.exp(-valid_scores / (temperature + 1e-10))
                        probs = exp_scores / np.sum(exp_scores)
                        
                        valid_indices = np.where(mask)[0]
                        next_idx = np.random.choice(valid_indices, p=probs)
                        path.append(next_idx)
                        visited.add(next_idx)
                        current = next_idx
                    
                    # Complete path if target not reached
                    if current != tgt:
                        path.append(tgt)
                    
                    path_cost = sum(distance_matrix[path[i], path[i+1]] 
                                  for i in range(len(path)-1))
                    
                    if path_cost < best_cost:
                        best_cost = path_cost
                        best_path = path
                
                path = best_path
            
            # Update edge betweenness for edges in the path
            for i in range(len(path) - 1):
                u, v = path[i], path[i+1]
                if u < v:
                    edge_betweenness[u, v] += 1
                else:
                    edge_betweenness[v, u] += 1
    
    # Normalize edge betweenness
    edge_betweenness = edge_betweenness + edge_betweenness.T
    if np.max(edge_betweenness) > 0:
        edge_betweenness = edge_betweenness / np.max(edge_betweenness)
    
    # ===== 3. MULTI-RESOLUTION MST ENHANCEMENT =====
    mst_components = []
    
    # Base MST
    try:
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0) | (mst.T > 0)
        mst_scores = np.zeros_like(distance_matrix)
        mst_edges = np.where(mst_mask)
        if len(mst_edges[0]) > 0:
            mst_values = 1 / (distance_matrix[mst_edges] + 1e-10)
            mst_scores[mst_edges] = mst_values
            if np.max(mst_scores) > 0:
                mst_scores = mst_scores / np.max(mst_scores)
        mst_components.append(mst_scores)
    except:
        mst_components.append(np.ones_like(distance_matrix))
    
    # Hierarchical clustering for large instances
    if n > 30:
        # Cluster-based MST approximation
        n_clusters = max(2, min(10, n // 10))
        
        # Simple geometric clustering (assuming roughly Euclidean-like structure)
        if dist_std > 0:
            # Normalize distances
            norm_dist = distance_matrix / (dist_mean + 1e-10)
            
            # Initialize cluster centers randomly
            centers = np.random.choice(n, n_clusters, replace=False)
            
            # Assign clusters
            cluster_assignments = np.argmin(norm_dist[centers], axis=0)
            
            # Compute intra-cluster MST scores
            cluster_mst = np.zeros_like(distance_matrix)
            
            for c in range(n_clusters):
                cluster_nodes = np.where(cluster_assignments == c)[0]
                if len(cluster_nodes) > 1:
                    cluster_dist = distance_matrix[np.ix_(cluster_nodes, cluster_nodes)]
                    try:
                        cluster_mst_local = minimum_spanning_tree(cluster_dist).toarray()
                        for i_idx, i in enumerate(cluster_nodes):
                            for j_idx, j in enumerate(cluster_nodes):
                                if i < j and cluster_mst_local[i_idx, j_idx] > 0:
                                    cluster_mst[i, j] = 1 / (distance_matrix[i, j] + 1e-10)
                    except:
                        pass
            
            cluster_mst = cluster_mst + cluster_mst.T
            if np.max(cluster_mst) > 0:
                cluster_mst = cluster_mst / np.max(cluster_mst)
            mst_components.append(cluster_mst)
    
    # Combine MST components
    if len(mst_components) > 1:
        mst_component = np.mean(mst_components, axis=0)
    else:
        mst_component = mst_components[0]
    
    # ===== 4. LOCAL PATTERN RECOGNITION =====
    local_pattern = np.zeros_like(distance_matrix)
    
    # Triangle inequality analysis
    if n >= 3:
        # Normalize distances for consistent scaling
        norm_dist = distance_matrix / (dist_median + 1e-10)
        
        # Sample intermediate nodes for efficiency
        k_samples = min(100, n)
        if n > 100:
            sample_indices = np.random.choice(n, k_samples, replace=False)
        else:
            sample_indices = np.arange(n)
        
        # Vectorized triangle analysis
        for k in sample_indices:
            # Triangle inequality: d(i,j) <= d(i,k) + d(k,j)
            triangle_sum = norm_dist[:, k:k+1] + norm_dist[k:k+1, :]
            violation = np.maximum(0, norm_dist - triangle_sum)
            
            # Edges that violate triangle inequality (are longer than sum)
            # get penalized, edges that satisfy it well get rewarded
            satisfaction = np.exp(-violation * 5)
            local_pattern += satisfaction
        
        local_pattern = local_pattern / len(sample_indices)
    
    # Nearest neighbor patterns
    k_neigh = max(2, min(8, int(np.log2(n)) + 2))
    nn_pattern = np.zeros_like(distance_matrix)
    
    for i in range(n):
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        nn_i = np.argsort(dist_i)[:k_neigh]
        nn_pattern[i, nn_i] = 1.0 / (1 + np.arange(k_neigh))
    
    nn_pattern = (nn_pattern + nn_pattern.T) / 2
    
    # ===== 5. INVERSE DISTANCE WITH ADAPTIVE TRANSFORM =====
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(inv_dist, 0)
    
    # Adaptive power transform based on instance characteristics
    power = 0.8 + 0.4 * np.tanh(complexity - 0.5)
    inv_dist = np.power(inv_dist, power)
    
    if np.max(inv_dist) > 0:
        inv_dist = inv_dist / np.max(inv_dist)
    
    # ===== 6. ADAPTIVE COMPONENT FUSION =====
    # Dynamic weights based on instance analysis
    if n < 15:
        weights = [0.25, 0.25, 0.25, 0.15, 0.1]  # [betweenness, mst, local, nn, inv]
    elif complexity < 0.3:
        # Uniform instances: emphasize structure
        weights = [0.2, 0.3, 0.2, 0.2, 0.1]
    elif complexity > 0.7:
        # Complex instances: emphasize sampling and patterns
        weights = [0.3, 0.2, 0.25, 0.15, 0.1]
    else:
        # Balanced approach
        weights = [0.25, 0.25, 0.2, 0.2, 0.1]
    
    # Adjust weights based on n
    size_factor = min(1.0, np.log10(n) / 2)
    weights[0] *= (0.8 + 0.4 * size_factor)  # Increase sampling for larger instances
    weights[1] *= (1.2 - 0.4 * size_factor)  # Decrease MST for larger instances
    
    weights = np.array(weights) / np.sum(weights)
    
    # Combine components
    heuristic_matrix = (
        weights[0] * edge_betweenness +
        weights[1] * mst_component +
        weights[2] * local_pattern +
        weights[3] * nn_pattern +
        weights[4] * inv_dist
    )
    
    # Ensure MST edges are prominent
    if 'mst_mask' in locals():
        heuristic_matrix[mst_mask] *= 1.3
    
    # ===== 7. DYNAMIC SPARSIFICATION =====
    if n > 10:
        # Adaptive edges per node
        edges_per_node = max(3, min(10, int(3 + np.log2(n))))
        
        # For complex instances, keep more edges
        if complexity > 0.7:
            edges_per_node = int(edges_per_node * 1.3)
        
        sparsity_mask = np.zeros_like(heuristic_matrix, dtype=bool)
        
        for i in range(n):
            row_scores = heuristic_matrix[i].copy()
            row_scores[i] = -np.inf
            
            # Keep top edges with adaptive threshold
            threshold = np.partition(row_scores, -edges_per_node)[-edges_per_node]
            sparsity_mask[i, row_scores >= threshold] = True
        
        # Ensure connectivity
        if 'mst_mask' in locals():
            sparsity_mask = sparsity_mask | mst_mask
        
        # Ensure symmetry
        sparsity_mask = sparsity_mask | sparsity_mask.T
        
        # Apply sparsification
        heuristic_matrix = heuristic_matrix * sparsity_mask
    
    # ===== 8. FINAL REFINEMENT =====
    np.fill_diagonal(heuristic_matrix, 0)
    
    # Symmetrize
    heuristic_matrix = 0.5 * (heuristic_matrix + heuristic_matrix.T)
    
    # Normalize to [0, 1]
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
    
    # Apply soft threshold for very small values
    heuristic_matrix[heuristic_matrix < 0.05] = 0
    
    return heuristic_matrix
