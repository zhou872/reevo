import numpy as np
from scipy.spatial.distance import pdist, squareform
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Inverse distance component (like v1)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Nearest neighbor emphasis
    k = max(2, min(5, n // 4))  # Adaptive k based on problem size
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        nn_indices = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, nn_indices] = True
    
    # Make mask symmetric (if j is in i's k-NN, also mark i->j)
    nn_mask = nn_mask | nn_mask.T
    
    # 3. Minimum spanning tree component
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # 4. Node centrality measure (approximation of betweenness)
    centrality = np.zeros(n)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Count how many shortest paths might go through edge i-j
                # Simple approximation: inverse of sum of distances to other nodes
                centrality[i] += 1 / (np.sum(distance_matrix[i]) - distance_matrix[i, i] + 1e-10)
    
    # Normalize centrality
    centrality = centrality / (np.sum(centrality) + 1e-10)
    
    # 5. Edge potential: combine centrality of endpoints
    edge_potential = np.outer(centrality, centrality)
    
    # 6. Combine all factors with weights
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base: inverse distance (strongest factor)
    heuristics_matrix = 0.6 * inv_dist
    
    # Boost edges that are in nearest neighbors
    heuristics_matrix[nn_mask] *= 1.5
    
    # Strong boost for MST edges
    heuristics_matrix[mst_mask] *= 2.0
    
    # Add edge potential component
    heuristics_matrix += 0.3 * edge_potential
    
    # 7. Sparsification: zero out clearly unpromising edges
    # Keep only top edges for each node
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    keep_per_node = max(3, min(8, n // 3))
    
    for i in range(n):
        # Get indices of top edges from node i
        top_indices = np.argsort(heuristics_matrix[i])[-keep_per_node:]
        sparsity_mask[i, top_indices] = True
    
    # Ensure symmetry
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # 8. Final normalization
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
