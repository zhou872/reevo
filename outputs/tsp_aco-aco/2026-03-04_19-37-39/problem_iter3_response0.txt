import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection using betweenness centrality,
    adaptive neighborhood, and multi-criteria sparsification.
    
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Inverse distance with stabilized division
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    mask = distance_matrix > 0
    inv_dist[mask] = 1.0 / distance_matrix[mask]
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Adaptive nearest neighbor with symmetric emphasis
    # Dynamic k based on node density
    k = max(2, min(int(np.sqrt(n)) + 1, n // 2))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nn_indices = np.argpartition(distance_matrix[i], k+1)[1:k+1]
        nn_mask[i, nn_indices] = True
    
    # Ensure symmetry and add small boost for reciprocal connections
    reciprocal = nn_mask & nn_mask.T
    nn_mask = nn_mask | nn_mask.T
    nn_boost = np.where(reciprocal, 1.8, 1.3)  # Higher boost for mutual NN
    
    # 3. Betweenness centrality approximation using inverse average distances
    # Compute node importance based on accessibility to all other nodes
    centrality = np.zeros(n)
    for i in range(n):
        # Use harmonic centrality (inverse of average distance)
        valid_dist = distance_matrix[i, distance_matrix[i] > 0]
        if len(valid_dist) > 0:
            centrality[i] = len(valid_dist) / np.sum(valid_dist)
        else:
            centrality[i] = 0
    
    # Normalize and add small constant to avoid zero centrality
    centrality = (centrality - centrality.min()) / (centrality.max() - centrality.min() + 1e-10)
    centrality = centrality + 0.1  # Ensure minimum importance
    
    # 4. Edge betweenness potential (product of node centralities)
    edge_betweenness = np.outer(centrality, centrality)
    
    # 5. Minimum spanning tree component
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # 6. Combine factors with adaptive weights
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base component: inverse distance weighted by node importance
    heuristics_matrix = 0.4 * inv_dist * (centrality[:, np.newaxis] + centrality[np.newaxis, :])
    
    # Apply nearest neighbor boost
    heuristics_matrix[nn_mask] *= nn_boost[nn_mask]
    
    # Strong emphasis on MST edges with centrality weighting
    heuristics_matrix[mst_mask] += 0.3 * edge_betweenness[mst_mask]
    
    # Add pure betweenness component for global connectivity
    heuristics_matrix += 0.3 * edge_betweenness
    
    # 7. Multi-criteria sparsification with connectivity guarantee
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    # Strategy 1: Keep top edges by combined score
    for i in range(n):
        # Adaptive number of edges to keep per node
        keep_count = max(3, min(n // 3, 10))
        
        # Get top indices by heuristics score
        top_indices = np.argsort(heuristics_matrix[i])[-keep_count:]
        sparsity_mask[i, top_indices] = True
        
        # Additionally keep 2 nearest neighbors if not already included
        nearest_indices = np.argsort(distance_matrix[i])[1:3]
        sparsity_mask[i, nearest_indices] = True
    
    # Strategy 2: Ensure symmetric connections
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Strategy 3: Force inclusion of MST edges for connectivity
    sparsity_mask = sparsity_mask | mst_mask
    
    # Strategy 4: Ensure each node has at least 2 connections
    for i in range(n):
        if np.sum(sparsity_mask[i]) < 2:
            # Add next best edges to reach minimum connectivity
            remaining = ~sparsity_mask[i]
            remaining[i] = False  # Exclude self
            if np.any(remaining):
                additional_count = 2 - np.sum(sparsity_mask[i])
                best_remaining = np.argsort(heuristics_matrix[i][remaining])[-additional_count:]
                actual_indices = np.where(remaining)[0][best_remaining]
                sparsity_mask[i, actual_indices] = True
                # Maintain symmetry
                sparsity_mask[actual_indices, i] = True
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # 8. Final normalization with softmax-like scaling
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        # Use exponential scaling to emphasize strong edges
        exp_matrix = np.exp(heuristics_matrix / (max_val + 1e-10))
        heuristics_matrix = exp_matrix / np.max(exp_matrix)
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
