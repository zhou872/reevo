import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple strategic factors.
    Returns a matrix where higher values indicate more promising edges for inclusion.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Inverse distance component (fundamental)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Adaptive Nearest Neighbor emphasis with symmetric boosting
    k = max(2, min(5, n // 4))  # Dynamic k based on problem size
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nn_indices = np.argsort(distance_matrix[i])[1:k+1]  # k nearest excluding self
        nn_mask[i, nn_indices] = True
    nn_mask = nn_mask | nn_mask.T  # Ensure symmetry
    
    # 3. Minimum Spanning Tree connectivity (structural importance)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # 4. Centrality-Based Betweenness Approximation
    # Compute node centrality as inverse of average distance to other nodes
    centrality = np.zeros(n)
    for i in range(n):
        # Average distance to all other nodes (excluding self)
        avg_dist = np.sum(distance_matrix[i]) / (n - 1) if n > 1 else 0
        centrality[i] = 1 / (avg_dist + 1e-10) if avg_dist > 0 else 0
    
    # Normalize centrality scores
    centrality = centrality / (np.sum(centrality) + 1e-10)
    
    # Edge potential as product of endpoint centralities (betweenness approximation)
    edge_potential = np.outer(centrality, centrality)
    
    # 5. Multi-factor combination with tuned weights
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base component: inverse distance (strongest individual predictor)
    heuristics_matrix = 0.5 * inv_dist
    
    # Boost nearest neighbor connections (adaptive local structure)
    heuristics_matrix[nn_mask] *= 1.8
    
    # Strong emphasis on MST edges (ensures connectivity)
    heuristics_matrix[mst_mask] *= 2.2
    
    # Add global centrality component
    heuristics_matrix += 0.4 * edge_potential
    
    # 6. Multi-Criteria Sparsification with Minimum Connectivity
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    # Determine edges to keep based on multiple criteria
    for i in range(n):
        # Criterion 1: Keep top edges by inverse distance (connectivity guarantee)
        inv_top_indices = np.argsort(inv_dist[i])[-max(2, n//10):]  # At least 2 connections
        sparsity_mask[i, inv_top_indices] = True
        
        # Criterion 2: Keep top edges by combined heuristics score
        heur_top_indices = np.argsort(heuristics_matrix[i])[-max(3, n//8):]
        sparsity_mask[i, heur_top_indices] = True
        
        # Criterion 3: Ensure all MST edges are retained
        sparsity_mask[i, mst_mask[i]] = True
    
    # Ensure symmetric connectivity
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsification - zero out non-promising edges
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # 7. Final normalization
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
