import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge heuristics with adaptive sampling and intelligent edge scoring.
    
    Key Improvements:
    1. Adaptive sampling strategy based on instance characteristics
    2. Variable-intensity local search with 2-opt improvements
    3. Probabilistic edge scoring with uncertainty modeling
    4. Connectivity-preserving sparsification with MST guarantee
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # === 1. INSTANCE ANALYSIS ===
    distances = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances)
    dist_std = np.std(distances)
    cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    dist_q25 = np.percentile(distances, 25)
    
    # Instance-based parameter adaptation
    log_n = np.log(n + 1)
    sqrt_n = np.sqrt(n)
    
    # === 2. ADAPTIVE TOUR GENERATION ===
    def generate_tour(method: str, start: int = None) -> np.ndarray:
        """Generate initial tour using specified method."""
        if start is None:
            start = np.random.randint(0, n)
            
        if method == 'nearest':
            # Probabilistic nearest neighbor
            tour = [start]
            unvisited = set(range(n)) - {start}
            while unvisited:
                current = tour[-1]
                dists = distance_matrix[current].copy()
                dists[list(tour)] = np.inf
                
                # Temperature-based selection
                temperature = max(0.1, 0.5 * (1 - cv))
                inv_dists = 1 / (dists + 1e-10)
                probs = np.exp(-inv_dists / temperature)
                probs[dists == np.inf] = 0
                if probs.sum() > 0:
                    probs /= probs.sum()
                    next_node = np.random.choice(n, p=probs)
                else:
                    next_node = np.random.choice(list(unvisited))
                
                tour.append(next_node)
                unvisited.remove(next_node)
            return np.array(tour)
        
        elif method == 'random':
            tour = np.random.permutation(n)
            return tour
        
        elif method == 'insertion':
            if n <= 3:
                return np.random.permutation(n)
            
            # Random selection for first 3 nodes
            nodes = list(np.random.choice(n, 3, replace=False))
            tour = nodes[:3] + [nodes[0]]
            remaining = set(range(n)) - set(tour[:-1])
            
            while remaining:
                # Randomly select next node to insert
                node = np.random.choice(list(remaining))
                best_pos = -1
                best_cost = float('inf')
                
                # Find best insertion position
                for i in range(len(tour)-1):
                    cost = (distance_matrix[tour[i], node] +
                           distance_matrix[node, tour[i+1]] -
                           distance_matrix[tour[i], tour[i+1]])
                    if cost < best_cost:
                        best_cost = cost
                        best_pos = i + 1
                
                tour.insert(best_pos, node)
                remaining.remove(node)
            
            return np.array(tour[:-1])
        
        return np.random.permutation(n)
    
    def adaptive_2opt(tour: np.ndarray, budget: int) -> np.ndarray:
        """Apply 2-opt with adaptive search intensity."""
        n_nodes = len(tour)
        if n_nodes <= 3:
            return tour
        
        current = tour.copy()
        improved = True
        iterations = 0
        
        while improved and iterations < budget:
            improved = False
            # Sample candidate edges based on their distance
            for _ in range(min(50, n_nodes)):
                i = np.random.randint(0, n_nodes - 2)
                j = np.random.randint(i + 2, n_nodes)
                
                if j == n_nodes - 1 and i == 0:
                    continue
                
                a, b = current[i], current[(i + 1) % n_nodes]
                c, d = current[j], current[(j + 1) % n_nodes]
                
                # Check if swap improves
                current_cost = distance_matrix[a, b] + distance_matrix[c, d]
                new_cost = distance_matrix[a, c] + distance_matrix[b, d]
                
                if new_cost < current_cost - 1e-10:
                    current[i+1:j+1] = current[i+1:j+1][::-1]
                    improved = True
                    iterations += 1
                    break
            
            if not improved:
                break
        
        return current
    
    # Adaptive sampling strategy
    if n <= 20:
        num_tours = min(50, 3 * n)
        methods_dist = {'nearest': 0.5, 'insertion': 0.3, 'random': 0.2}
        search_budget = min(30, n * 2)
    else:
        num_tours = min(80, int(n * 1.5))
        methods_dist = {'nearest': 0.4, 'insertion': 0.4, 'random': 0.2}
        search_budget = min(50, int(n * 1.2))
    
    # Generate diverse tours with local search
    tours = []
    tour_lengths = []
    
    for _ in range(num_tours):
        method = np.random.choice(
            list(methods_dist.keys()),
            p=list(methods_dist.values())
        )
        tour = generate_tour(method)
        
        # Apply adaptive 2-opt
        budget = int(search_budget * (1 + cv))  # More search for varied instances
        tour = adaptive_2opt(tour, budget)
        
        # Calculate tour length
        tour_len = sum(distance_matrix[tour[i], tour[(i+1)%n]] for i in range(n))
        tours.append(tour)
        tour_lengths.append(tour_len)
    
    # === 3. PROBABILISTIC EDGE SCORING ===
    # Normalize tour lengths to weights (better tours get higher weight)
    if tour_lengths:
        min_len = min(tour_lengths)
        max_len = max(tour_lengths)
        if max_len > min_len:
            weights = 1 - (np.array(tour_lengths) - min_len) / (max_len - min_len)
        else:
            weights = np.ones(len(tour_lengths))
    else:
        weights = np.ones(num_tours)
    
    # Edge frequency with weighted contributions
    edge_freq = np.zeros_like(distance_matrix, dtype=float)
    for tour, weight in zip(tours, weights):
        for i in range(n):
            u, v = tour[i], tour[(i + 1) % n]
            edge_freq[u, v] += weight
            edge_freq[v, u] += weight
    
    if edge_freq.max() > 0:
        edge_freq = edge_freq / edge_freq.max()
    
    # === 4. COMPONENT INTEGRATION ===
    # Inverse distance component
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # MST component for connectivity
    try:
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0) | (mst.T > 0)
        # MST edges get scores based on inverse distance
        mst_scores = np.zeros_like(distance_matrix)
        mst_scores[mst_mask] = 1 / (distance_matrix[mst_mask] + 1e-10)
        if mst_scores.max() > 0:
            mst_scores = mst_scores / mst_scores.max()
    except:
        mst_mask = np.ones_like(distance_matrix, dtype=bool)
        np.fill_diagonal(mst_mask, False)
        mst_scores = np.ones_like(distance_matrix)
    
    # Adaptive weighting based on instance characteristics
    if n < 10:
        w_freq, w_dist, w_mst = 0.6, 0.2, 0.2
    elif cv < 0.2:  # Very uniform distances
        w_freq, w_dist, w_mst = 0.3, 0.5, 0.2
    elif cv > 0.8:  # Highly varied distances
        w_freq, w_dist, w_mst = 0.6, 0.2, 0.2
    else:  # Moderate variation
        w_freq, w_dist, w_mst = 0.5, 0.3, 0.2
    
    # Combine components
    heuristic = w_freq * edge_freq + w_dist * inv_dist + w_mst * mst_scores
    
    # Boost edges that appear both in good tours and MST
    boost_mask = (edge_freq > np.percentile(edge_freq[edge_freq > 0], 75)) & mst_mask
    heuristic[boost_mask] *= 1.4
    
    # === 5. CONNECTIVITY-PRESERVING SPARSIFICATION ===
    if n > 5:
        # Adaptive edge retention
        base_edges = max(2, min(8, int(2 + log_n)))
        edges_per_node = min(n-1, int(base_edges * (1 + 0.3 * cv)))
        
        sparsity_mask = np.zeros_like(heuristic, dtype=bool)
        for i in range(n):
            scores = heuristic[i].copy()
            scores[i] = -np.inf
            
            # Keep top edges_per_node edges
            threshold_idx = n - edges_per_node - 1
            if threshold_idx >= 0:
                threshold = np.partition(scores, threshold_idx)[threshold_idx]
                sparsity_mask[i] = scores >= threshold
            else:
                sparsity_mask[i] = scores > -np.inf
        
        # Always include MST edges for connectivity
        sparsity_mask = sparsity_mask | mst_mask
        sparsity_mask = sparsity_mask | sparsity_mask.T  # Ensure symmetry
        
        heuristic = heuristic * sparsity_mask
    
    # === 6. FINAL PROCESSING ===
    np.fill_diagonal(heuristic, 0)
    heuristic = 0.5 * (heuristic + heuristic.T)  # Ensure symmetry
    
    # Normalize to [0, 1] with outlier protection
    h_max = heuristic.max()
    if h_max > 0:
        heuristic = heuristic / h_max
    
    # Apply soft threshold
    threshold = 0.01 + 0.04 * cv  # Higher threshold for more uniform instances
    heuristic[heuristic < threshold] = 0
    
    return heuristic
