import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with refined stochastic sampling,
    adaptive weighting based on instance complexity, and improved sparsification.
    
    Key improvements over v1:
    1. More sophisticated tour generation using regret-based insertion
    2. Better edge centrality estimation with directed random walks
    3. Instance complexity-aware parameter tuning
    4. Enhanced sparsification preserving tour connectivity
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    cv_dist = dist_std / (dist_mean + 1e-10)
    
    # Instance size dependent parameters
    log_n = np.log(n + 1)
    size_factor = min(1.0, n / 80.0)
    
    # ===== 2. ENHANCED SOLUTION SAMPLING =====
    def generate_tour_v2(method: str = 'regret') -> np.ndarray:
        """Generate tour using enhanced construction methods."""
        if method == 'regret':  # Regret-2 insertion
            if n <= 3:
                return np.random.permutation(n)
            
            # Start with a small random tour
            start_nodes = np.random.choice(n, min(4, n), replace=False)
            tour = list(start_nodes)
            tour.append(start_nodes[0])
            remaining = set(range(n)) - set(tour[:-1])
            
            while remaining:
                best_gain = -np.inf
                best_node = None
                best_pos = -1
                
                for node in remaining:
                    # Calculate insertion costs for all positions
                    insert_costs = []
                    for i in range(len(tour)-1):
                        cost = (distance_matrix[tour[i], node] + 
                               distance_matrix[node, tour[i+1]] - 
                               distance_matrix[tour[i], tour[i+1]])
                        insert_costs.append(cost)
                    
                    # Regret calculation: difference between best and second-best
                    sorted_costs = np.sort(insert_costs)
                    regret = sorted_costs[1] - sorted_costs[0] if len(sorted_costs) > 1 else 0
                    gain = -sorted_costs[0] + 0.5 * regret  # Balance immediate cost and regret
                    
                    if gain > best_gain:
                        best_gain = gain
                        best_node = node
                        best_pos = np.argmin(insert_costs) + 1
                
                tour.insert(best_pos, best_node)
                remaining.remove(best_node)
            
            return np.array(tour[:-1])
        
        elif method == 'cheapest':  # Cheapest insertion
            if n <= 3:
                return np.random.permutation(n)
            
            start_nodes = np.random.choice(n, 3, replace=False)
            tour = list(start_nodes)
            tour.append(start_nodes[0])
            remaining = set(range(n)) - set(tour[:-1])
            
            while remaining:
                best_cost = np.inf
                best_node = None
                best_pos = -1
                
                for node in remaining:
                    for i in range(len(tour)-1):
                        cost_inc = (distance_matrix[tour[i], node] + 
                                   distance_matrix[node, tour[i+1]] - 
                                   distance_matrix[tour[i], tour[i+1]])
                        if cost_inc < best_cost:
                            best_cost = cost_inc
                            best_node = node
                            best_pos = i + 1
                
                tour.insert(best_pos, best_node)
                remaining.remove(best_node)
            
            return np.array(tour[:-1])
        
        else:  # Random with nearest-neighbor bias
            tour = np.random.permutation(n)
            # Apply a few nearest-neighbor swaps to improve
            for _ in range(min(10, n)):
                i = np.random.randint(n)
                candidates = np.argsort(distance_matrix[tour[i]])[:3]
                j = np.random.choice([idx for idx in candidates if idx != i])
                tour[i], tour[j] = tour[j], tour[i]
            return tour
    
    def fast_two_opt(tour: np.ndarray, max_iter: int = 20) -> np.ndarray:
        """Optimized 2-opt local search."""
        improved = True
        iteration = 0
        current = tour.copy()
        
        while improved and iteration < max_iter:
            improved = False
            # Random order of edges to check
            edges_i = np.random.permutation(n)
            edges_j = np.random.permutation(n)
            
            for i in edges_i:
                if improved:
                    break
                i_next = (i + 1) % n
                a, b = current[i], current[i_next]
                
                for j in edges_j:
                    if j == i or j == i_next or (j + 1) % n == i:
                        continue
                    
                    j_next = (j + 1) % n
                    c, d = current[j], current[j_next]
                    
                    # Quick check if swap could be beneficial
                    if (distance_matrix[a, b] + distance_matrix[c, d] > 
                        distance_matrix[a, c] + distance_matrix[b, d] + 1e-6):
                        
                        # Perform the swap
                        if i < j:
                            current[i+1:j+1] = current[i+1:j+1][::-1]
                        else:
                            current[j+1:i+1] = current[j+1:i+1][::-1]
                        improved = True
                        break
            
            iteration += 1
        
        return current
    
    # Generate diverse ensemble of tours
    num_tours = min(15, max(5, int(n * 0.3)))
    tours = []
    
    for _ in range(num_tours // 3):
        tour = generate_tour_v2('regret')
        tours.append(fast_two_opt(tour, max_iter=int(20 * size_factor)))
    
    for _ in range(num_tours // 3):
        tour = generate_tour_v2('cheapest')
        tours.append(fast_two_opt(tour, max_iter=int(20 * size_factor)))
    
    for _ in range(num_tours - 2*(num_tours//3)):
        tour = generate_tour_v2('random')
        tours.append(fast_two_opt(tour, max_iter=int(20 * size_factor)))
    
    # Edge frequency from sampled tours
    edge_freq = np.zeros_like(distance_matrix)
    for tour in tours:
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            edge_freq[u, v] += 1
            edge_freq[v, u] += 1
    
    # Normalize edge frequency
    if edge_freq.max() > 0:
        edge_freq = edge_freq / edge_freq.max()
    
    # ===== 3. IMPROVED EDGE BETWEENNESS SAMPLING =====
    edge_betweenness = np.zeros_like(distance_matrix)
    
    # Adaptive number of random paths
    num_paths = min(300, max(80, int(n * 2.5)))
    
    for _ in range(num_paths):
        s, t = np.random.choice(n, 2, replace=False)
        current = s
        path_length = 0
        max_path_length = min(2 * n, 50)
        
        while current != t and path_length < max_path_length:
            # Create candidate list: prefer shorter edges and progress towards target
            candidates = list(range(n))
            candidates.remove(current)
            
            # Score candidates
            scores = np.zeros(len(candidates))
            for idx, v in enumerate(candidates):
                # Distance component (prefer shorter)
                dist_score = 1.0 / (distance_matrix[current, v] + 1e-10)
                
                # Progress component (prefer moving toward target)
                if distance_matrix[current, t] > distance_matrix[v, t]:
                    progress_score = 2.0
                else:
                    progress_score = 0.5
                
                # Exploration component (prefer less visited nodes in this path)
                path_visit_bonus = 1.0 / (edge_betweenness[current, v] + 1)
                
                scores[idx] = dist_score * progress_score * path_visit_bonus
            
            # Select next node with probability proportional to score
            if scores.sum() > 0:
                probs = scores / scores.sum()
                next_node = np.random.choice(candidates, p=probs)
            else:
                next_node = np.random.choice(candidates)
            
            # Update edge betweenness
            edge_betweenness[current, next_node] += 1
            edge_betweenness[next_node, current] += 1
            
            current = next_node
            path_length += 1
    
    # Normalize edge betweenness with logarithmic scaling
    if edge_betweenness.max() > 0:
        edge_betweenness = np.log1p(edge_betweenness) / np.log1p(edge_betweenness.max())
    
    # ===== 4. REFINED ADAPTIVE WEIGHTING =====
    # Compute component statistics
    freq_values = edge_freq[np.triu_indices(n, k=1)]
    bet_values = edge_betweenness[np.triu_indices(n, k=1)]
    
    freq_cv = np.std(freq_values) / (np.mean(freq_values) + 1e-10)
    bet_cv = np.std(bet_values) / (np.mean(bet_values) + 1e-10)
    
    # Adaptive weights based on instance characteristics
    # Higher CV indicates more informative component
    w_freq = 0.25 + 0.15 * min(1.0, freq_cv * 2)
    w_bet = 0.25 + 0.15 * min(1.0, bet_cv * 2)
    w_dist = 0.25 + 0.15 * (1 - min(1.0, cv_dist))  # Lower distance CV means distances less informative
    w_mst = 0.25 * size_factor  # MST more important for larger instances
    
    # Normalize to sum to 1
    total = w_freq + w_bet + w_dist + w_mst
    w_freq /= total
    w_bet /= total
    w_dist /= total
    w_mst /= total
    
    # ===== 5. COMPONENT COMPUTATION =====
    # Inverse distance component
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + np.eye(n))  # Avoid division by zero on diagonal
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # MST component
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_scores = np.zeros_like(distance_matrix)
    mst_edges = np.where(mst_mask)
    if len(mst_edges[0]) > 0:
        mst_values = 1.0 / (distance_matrix[mst_edges] + 1e-10)
        mst_scores[mst_edges] = mst_values
        if mst_values.max() > 0:
            mst_scores[mst_edges] = mst_values / mst_values.max()
    
    # ===== 6. COMBINE COMPONENTS =====
    heuristics_matrix = (
        w_freq * edge_freq +
        w_bet * edge_betweenness +
        w_dist * inv_dist +
        w_mst * mst_scores
    )
    
    # ===== 7. ENHANCED SPARSIFICATION =====
    # Determine edges to keep per node
    base_edges = max(2, int(1.8 * log_n))
    max_edges = min(n-1, int(base_edges * 3.5))
    
    # Tour connectivity preservation
    tour_edge_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    for tour in tours[:min(3, len(tours))]:  # Use best few tours
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            tour_edge_mask[u, v] = True
            tour_edge_mask[v, u] = True
    
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        # Adaptive edge count based on instance characteristics
        n_keep = min(max_edges, max(base_edges, 
                  int((n-1) * (0.1 + 0.15 * size_factor * (0.5 + cv_dist)))))
        
        # Keep top edges
        top_indices = np.argpartition(scores, -n_keep)[-n_keep:]
        sparsity_mask[i, top_indices] = True
    
    # Ensure critical edges are kept
    sparsity_mask = sparsity_mask | mst_mask | tour_edge_mask
    sparsity_mask = sparsity_mask | sparsity_mask.T  # Ensure symmetry
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 8. FINAL ENHANCEMENTS =====
    # Ensure symmetry and zero diagonal
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Boost edges that appear in multiple components
    component_mask = (edge_freq > 0).astype(float) + (edge_betweenness > 0).astype(float) + \
                     (inv_dist > np.percentile(inv_dist[inv_dist > 0], 70)).astype(float) + \
                     mst_mask.astype(float)
    
    heuristics_matrix = heuristics_matrix * (1.0 + 0.2 * np.minimum(component_mask, 3) / 3)
    
    # Final normalization
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0 and non_zero.max() > non_zero.min():
        heuristics_matrix[heuristics_matrix > 0] = (
            (non_zero - non_zero.min()) / (non_zero.max() - non_zero.min())
        )
    
    return heuristics_matrix
