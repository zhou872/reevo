import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List, Tuple


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with improved sampling and adaptive components.
    
    Key improvements:
    1. Multiple construction heuristics for diverse tour sampling
    2. Edge betweenness computed via strategic random walks
    3. Adaptive component weighting based on instance characteristics
    4. Intelligent sparsification preserving connectivity
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    
    # Size-dependent parameters
    density_factor = min(1.0, n / 100.0)
    log_n = np.log(n + 1)
    
    # ===== 2. ENHANCED TOUR SAMPLING WITH MULTIPLE HEURISTICS =====
    def generate_tour(method: str = 'nn', start_node: int = None) -> np.ndarray:
        """Generate tour using specified construction method."""
        if start_node is None:
            start_node = np.random.randint(0, n)
            
        if method == 'nn':  # Nearest neighbor
            tour = [start_node]
            unvisited = set(range(n)) - {start_node}
            while unvisited:
                current = tour[-1]
                nearest = min(unvisited, key=lambda x: distance_matrix[current, x])
                tour.append(nearest)
                unvisited.remove(nearest)
            return np.array(tour)
        
        elif method == 'greedy':  # Greedy insertion
            if n <= 3:
                return np.random.permutation(n)
            
            # Start with triangle of random nodes
            nodes = list(range(n))
            np.random.shuffle(nodes)
            tour = nodes[:3].tolist()
            tour.append(tour[0])  # Close the triangle
            remaining = set(nodes[3:])
            
            while remaining:
                best_cost = float('inf')
                best_node = None
                best_pos = -1
                
                for node in remaining:
                    for i in range(len(tour)-1):
                        # Cost of inserting node between i and i+1
                        cost_inc = (distance_matrix[tour[i], node] + 
                                   distance_matrix[node, tour[i+1]] - 
                                   distance_matrix[tour[i], tour[i+1]])
                        if cost_inc < best_cost:
                            best_cost = cost_inc
                            best_node = node
                            best_pos = i + 1
                
                tour.insert(best_pos, best_node)
                remaining.remove(best_node)
            
            return np.array(tour[:-1])  # Remove duplicate start node
        
        elif method == 'farthest':  # Farthest insertion
            tour = [start_node]
            unvisited = list(set(range(n)) - {start_node})
            
            # Initialize with farthest node from start
            farthest = max(unvisited, key=lambda x: distance_matrix[start_node, x])
            tour.append(farthest)
            unvisited.remove(farthest)
            tour.append(start_node)  # Close partial tour
            
            while unvisited:
                # Find node farthest from current tour
                max_dist = -1
                farthest_node = -1
                for node in unvisited:
                    # Minimum distance from node to tour
                    min_dist_to_tour = min(distance_matrix[node, t] for t in tour[:-1])
                    if min_dist_to_tour > max_dist:
                        max_dist = min_dist_to_tour
                        farthest_node = node
                
                # Insert at position that minimizes cost increase
                best_cost = float('inf')
                best_pos = -1
                for i in range(len(tour)-1):
                    cost_inc = (distance_matrix[tour[i], farthest_node] + 
                               distance_matrix[farthest_node, tour[i+1]] - 
                               distance_matrix[tour[i], tour[i+1]])
                    if cost_inc < best_cost:
                        best_cost = cost_inc
                        best_pos = i + 1
                
                tour.insert(best_pos, farthest_node)
                unvisited.remove(farthest_node)
            
            return np.array(tour[:-1])
        
        else:  # Random permutation
            return np.random.permutation(n)
    
    def two_opt_improve(tour: np.ndarray, max_iter: int = 50) -> np.ndarray:
        """Apply 2-opt local search."""
        improved = True
        iteration = 0
        current = tour.copy()
        
        while improved and iteration < max_iter:
            improved = False
            for i in range(n-1):
                for j in range(i+2, n):
                    # Avoid swapping the last edge with the first
                    if j == n-1 and i == 0:
                        continue
                    
                    a, b = current[i], current[(i+1) % n]
                    c, d = current[j], current[(j+1) % n]
                    
                    old_len = distance_matrix[a, b] + distance_matrix[c, d]
                    new_len = distance_matrix[a, c] + distance_matrix[b, d]
                    
                    if new_len < old_len - 1e-10:
                        # Reverse segment between i+1 and j
                        current[i+1:j+1] = current[i+1:j+1][::-1]
                        improved = True
                        break
                if improved:
                    break
            iteration += 1
        
        return current
    
    # Generate diverse tours using different heuristics
    num_tours = min(20, max(8, n // 5))
    tours = []
    
    # Use multiple construction methods
    construction_methods = ['nn', 'greedy', 'farthest', 'random']
    method_weights = [0.35, 0.3, 0.2, 0.15]  # Prefer NN and greedy
    
    for _ in range(num_tours):
        method = np.random.choice(construction_methods, p=method_weights)
        start = np.random.randint(0, n) if np.random.random() < 0.7 else 0
        tour = generate_tour(method, start)
        
        # Apply 2-opt improvement with probability based on tour quality
        if method != 'random' or np.random.random() < 0.5:
            improvement_iter = int(30 * density_factor)
            tour = two_opt_improve(tour, max_iter=improvement_iter)
        
        tours.append(tour)
    
    # Edge frequency from tours
    edge_freq = np.zeros_like(distance_matrix)
    for tour in tours:
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            edge_freq[u, v] += 1
            edge_freq[v, u] += 1
    
    if edge_freq.max() > 0:
        edge_freq = edge_freq / edge_freq.max()
    
    # ===== 3. ENHANCED EDGE BETWEENNESS SAMPLING =====
    edge_betweenness = np.zeros_like(distance_matrix)
    
    # Adaptive number of random walks based on instance
    num_walks = min(300, max(80, n * 3))
    
    for walk_idx in range(num_walks):
        s, t = np.random.choice(n, 2, replace=False)
        current = s
        path = [s]
        recent_visited = set([s])
        
        # Adaptive walk length
        max_steps = min(2 * n, max(10, int(n * 1.5)))
        
        for step in range(max_steps):
            if current == t:
                break
            
            # Get candidate next nodes
            candidates = list(range(n))
            
            # Filter candidates to avoid very recent visits (avoid cycles)
            if len(recent_visited) > 3:
                candidates = [c for c in candidates if c not in recent_visited]
            
            if not candidates:
                candidates = list(range(n))
            
            # Score candidates: balance edge quality and progress toward target
            scores = np.zeros(len(candidates))
            for idx, cand in enumerate(candidates):
                edge_score = 1.0 / (distance_matrix[current, cand] + 1e-10)
                progress = 1.0 / (distance_matrix[cand, t] + 1e-10)
                
                # Add small randomness to encourage exploration
                randomness = 0.1 * np.random.random()
                
                scores[idx] = edge_score + 0.3 * progress + randomness
            
            # Softmax selection
            scores = np.exp(scores - np.max(scores))
            if scores.sum() > 0:
                probs = scores / scores.sum()
                next_node = np.random.choice(candidates, p=probs)
            else:
                next_node = np.random.choice(candidates)
            
            # Update edge betweenness
            edge_betweenness[current, next_node] += 1
            edge_betweenness[next_node, current] += 1
            
            current = next_node
            path.append(current)
            
            # Maintain recent visited set (size limited to avoid cycles)
            recent_visited.add(current)
            if len(recent_visited) > 5:
                # Remove oldest element
                recent_visited.remove(path[-6] if len(path) >= 6 else list(recent_visited)[0])
    
    # Normalize edge betweenness
    if edge_betweenness.max() > 0:
        edge_betweenness = edge_betweenness / edge_betweenness.max()
    
    # ===== 4. ENHANCED COMPONENT WEIGHTING =====
    # Compute component statistics
    freq_std = np.std(edge_freq[np.triu_indices(n, k=1)])
    betweenness_std = np.std(edge_betweenness[np.triu_indices(n, k=1)])
    
    # Adaptive weights based on instance characteristics
    w_freq = 0.35 + 0.15 * min(1.0, freq_std * 2.5)  # Higher variance -> trust frequency more
    w_bet = 0.25 + 0.15 * min(1.0, betweenness_std * 2)  # Higher variance -> trust betweenness more
    w_dist = 0.25 * (1 - 0.5 * min(1.0, cv))  # Uniform distances -> less weight on inverse distance
    w_mst = 0.25 * density_factor  # MST more important for larger instances
    
    # Adjust for very small instances
    if n <= 10:
        w_freq *= 0.8
        w_dist *= 1.2
    
    # Normalize weights
    total = w_freq + w_bet + w_dist + w_mst
    w_freq /= total
    w_bet /= total
    w_dist /= total
    w_mst /= total
    
    # ===== 5. COMPONENT COMPUTATION =====
    # Distance component (inverse, normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # MST component
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_scores = np.zeros_like(distance_matrix)
    mst_scores[mst_mask] = 1.0 / (distance_matrix[mst_mask] + 1e-10)
    if mst_scores.max() > 0:
        mst_scores = mst_scores / mst_scores.max()
    
    # ===== 6. COMBINE COMPONENTS =====
    heuristics_matrix = (
        w_freq * edge_freq +
        w_bet * edge_betweenness +
        w_dist * inv_dist +
        w_mst * mst_scores
    )
    
    # ===== 7. ENHANCED SPARSIFICATION =====
    # Determine edges to keep per node
    base_edges = max(2, int(1.8 * log_n))
    max_edges = min(n-1, base_edges * 5)
    
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        # Adaptive number of edges
        n_edges = min(max_edges, max(base_edges, int((n-1) * (0.1 + 0.15 * density_factor * (1 + cv)))))
        
        # Keep top edges, but also ensure we keep some edges from each component
        top_indices = np.argsort(scores)[-n_edges:]
        sparsity_mask[i, top_indices] = True
    
    # Ensure connectivity: keep MST edges and their neighbors
    sparsity_mask = sparsity_mask | mst_mask
    
    # Ensure each node has at least 2 connections
    for i in range(n):
        if sparsity_mask[i].sum() < 2:
            # Add top edges until we have at least 2
            scores = heuristics_matrix[i].copy()
            scores[i] = -np.inf
            additional = 2 - sparsity_mask[i].sum()
            if additional > 0:
                top_indices = np.argsort(scores)[-additional:]
                sparsity_mask[i, top_indices] = True
    
    # Ensure symmetry
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 8. FINAL PROCESSING =====
    # Ensure symmetry
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Apply soft thresholding
    threshold = 0.05 * np.max(heuristics_matrix)
    heuristics_matrix[heuristics_matrix < threshold] = 0
    
    # Normalize non-zero elements to [0, 1]
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0 and non_zero.max() > non_zero.min():
        heuristics_matrix[heuristics_matrix > 0] = (
            (non_zero - non_zero.min()) / (non_zero.max() - non_zero.min())
        )
    
    return heuristics_matrix
