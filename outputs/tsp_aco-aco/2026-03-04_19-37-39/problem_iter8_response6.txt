import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List, Tuple
import itertools

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristic function incorporating ML-inspired edge features,
    diverse sampling strategies, and parallel-optimized computations.
    
    Key innovations:
    1. Feature-based edge prediction using multiple sampling methods
    2. Ant Colony Optimization (ACO) simulation for path exploration
    3. Parallel-inspired vectorized computations
    4. Adaptive ensemble weighting with instance-specific tuning
    5. Efficient sparse graph construction with connectivity guarantees
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE ANALYSIS AND PARAMETER TUNING =====
    # Compute instance characteristics for adaptive behavior
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    cv = dist_std / (dist_mean + 1e-10)
    
    # Size-adaptive parameters
    log_n = np.log(n + 1)
    size_factor = min(1.0, n / 100.0)
    
    # Sampling parameters
    n_samples = min(50, max(10, int(5 * log_n)))
    n_ants = min(20, max(5, n // 10))
    n_iterations = min(15, max(3, int(2 * log_n)))
    
    # ===== 2. FEATURE-BASED EDGE CHARACTERIZATION =====
    def compute_edge_features() -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
        """Compute multiple edge features efficiently."""
        # Feature 1: Inverse distance (normalized)
        eps = 1e-10
        inv_dist = 1.0 / (distance_matrix + eps)
        np.fill_diagonal(inv_dist, 0)
        inv_dist_norm = inv_dist / (inv_dist.max() + eps)
        
        # Feature 2: MST-based importance
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0) | (mst.T > 0)
        mst_feature = np.zeros_like(distance_matrix)
        mst_feature[mst_mask] = 1.0 / (distance_matrix[mst_mask] + eps)
        if mst_feature.max() > 0:
            mst_feature /= mst_feature.max()
        
        # Feature 3: Local neighborhood density
        k_neighbors = min(8, max(3, int(np.sqrt(n))))
        density_feature = np.zeros_like(distance_matrix)
        for i in range(n):
            neighbor_dists = np.sort(distance_matrix[i])[1:k_neighbors+1]
            local_density = 1.0 / (np.mean(neighbor_dists) + eps)
            density_feature[i] = local_density
        density_feature = (density_feature + density_feature.T) / 2
        density_feature /= density_feature.max() + eps
        
        return inv_dist_norm, mst_feature, density_feature
    
    inv_dist_feature, mst_feature, density_feature = compute_edge_features()
    
    # ===== 3. DIVERSE SOLUTION SAMPLING =====
    def generate_aco_tours(n_ants: int, n_iterations: int) -> List[np.ndarray]:
        """Generate tours using Ant Colony Optimization simulation."""
        # Initialize pheromones with distance-based heuristic
        alpha = 1.0  # Pheromone importance
        beta = 2.0   # Distance importance
        rho = 0.1    # Evaporation rate
        Q = 100.0    # Pheromone deposit constant
        
        # Initialize pheromone matrix
        pheromones = np.ones_like(distance_matrix)
        np.fill_diagonal(pheromones, 0)
        
        best_tours = []
        
        for _ in range(n_iterations):
            ant_tours = []
            ant_lengths = []
            
            for ant in range(n_ants):
                # Start from random city
                start = np.random.randint(0, n)
                tour = [start]
                unvisited = set(range(n)) - {start}
                
                while unvisited:
                    current = tour[-1]
                    # Calculate probabilities for unvisited cities
                    unvisited_list = list(unvisited)
                    if not unvisited_list:
                        break
                    
                    # Compute attractiveness
                    pheromone_row = pheromones[current, unvisited_list]
                    distance_row = distance_matrix[current, unvisited_list]
                    attractiveness = (pheromone_row ** alpha) * ((1.0 / (distance_row + eps)) ** beta)
                    
                    # Select next city
                    if np.sum(attractiveness) > 0:
                        probabilities = attractiveness / np.sum(attractiveness)
                        next_city = np.random.choice(unvisited_list, p=probabilities)
                    else:
                        next_city = np.random.choice(unvisited_list)
                    
                    tour.append(next_city)
                    unvisited.remove(next_city)
                
                # Complete tour
                tour = np.array(tour)
                ant_tours.append(tour)
                ant_lengths.append(calculate_tour_length(tour))
            
            # Update pheromones
            pheromones *= (1 - rho)  # Evaporation
            
            # Deposit pheromones on best tours
            sorted_indices = np.argsort(ant_lengths)[:min(5, len(ant_tours))]
            for idx in sorted_indices:
                tour = ant_tours[idx]
                deposit = Q / (ant_lengths[idx] + eps)
                for i in range(n):
                    j = (i + 1) % n
                    u, v = tour[i], tour[j]
                    pheromones[u, v] += deposit
                    pheromones[v, u] += deposit
            
            # Keep best tour
            best_idx = np.argmin(ant_lengths)
            best_tours.append(ant_tours[best_idx])
        
        return best_tours
    
    def generate_diverse_tours(n_samples: int) -> List[np.ndarray]:
        """Generate diverse tours using multiple strategies."""
        tours = []
        
        # Strategy 1: ACO tours (40%)
        aco_tours = generate_aco_tours(n_ants, n_iterations)
        tours.extend(aco_tours[:int(0.4 * n_samples)])
        
        # Strategy 2: Nearest neighbor with random starts (30%)
        nn_samples = int(0.3 * n_samples)
        for _ in range(nn_samples):
            start = np.random.randint(0, n)
            tour = [start]
            unvisited = set(range(n)) - {start}
            while unvisited:
                current = tour[-1]
                nearest = min(unvisited, key=lambda x: distance_matrix[current, x])
                tour.append(nearest)
                unvisited.remove(nearest)
            tours.append(np.array(tour))
        
        # Strategy 3: Random permutations with 2-opt improvement (30%)
        random_samples = int(0.3 * n_samples)
        for _ in range(random_samples):
            tour = np.random.permutation(n)
            # Apply quick 2-opt improvement
            improved_tour = local_search_2opt(tour, max_iter=int(10 * log_n))
            tours.append(improved_tour)
        
        return tours[:n_samples]
    
    def calculate_tour_length(tour: np.ndarray) -> float:
        """Calculate total length of a tour."""
        return sum(distance_matrix[tour[i], tour[(i+1)%n]] for i in range(n))
    
    def local_search_2opt(tour: np.ndarray, max_iter: int = 20) -> np.ndarray:
        """Apply 2-opt local search to improve tour."""
        n = len(tour)
        best_tour = tour.copy()
        best_length = calculate_tour_length(tour)
        
        for _ in range(max_iter):
            improved = False
            for i in range(n-1):
                for j in range(i+2, n):
                    if j == n-1 and i == 0:
                        continue
                    
                    # Calculate potential improvement
                    a, b = best_tour[i], best_tour[(i+1)%n]
                    c, d = best_tour[j], best_tour[(j+1)%n]
                    
                    old_len = distance_matrix[a, b] + distance_matrix[c, d]
                    new_len = distance_matrix[a, c] + distance_matrix[b, d]
                    
                    if new_len < old_len:
                        # Perform 2-opt swap
                        best_tour[i+1:j+1] = best_tour[i+1:j+1][::-1]
                        best_length = best_length - old_len + new_len
                        improved = True
                        break
                if improved:
                    break
            if not improved:
                break
        
        return best_tour
    
    # Generate diverse tours
    all_tours = generate_diverse_tours(n_samples)
    
    # ===== 4. EDGE FREQUENCY WITH QUALITY WEIGHTING =====
    edge_frequency = np.zeros_like(distance_matrix)
    tour_lengths = [calculate_tour_length(tour) for tour in all_tours]
    
    if len(tour_lengths) > 0:
        # Weight tours by quality (shorter tours get higher weight)
        min_length = min(tour_lengths)
        max_length = max(tour_lengths)
        if max_length > min_length:
            weights = 1.0 - (np.array(tour_lengths) - min_length) / (max_length - min_length)
        else:
            weights = np.ones_like(tour_lengths)
        
        # Normalize weights
        weights = weights / (weights.sum() + 1e-10)
        
        # Accumulate weighted edge frequencies
        for tour, weight in zip(all_tours, weights):
            for i in range(n):
                j = (i + 1) % n
                u, v = tour[i], tour[j]
                edge_frequency[u, v] += weight
                edge_frequency[v, u] += weight
        
        # Normalize edge frequency
        if edge_frequency.max() > 0:
            edge_frequency /= edge_frequency.max()
    
    # ===== 5. ADAPTIVE ENSEMBLE COMBINATION =====
    # Compute adaptive weights based on instance characteristics
    freq_std = np.std(edge_frequency[np.triu_indices(n, k=1)]) if n > 1 else 0
    
    # Weight tuning based on instance properties
    w_freq = 0.4 + 0.2 * min(1.0, freq_std * 3)  # Higher when edge frequencies show clear patterns
    w_dist = 0.25 - 0.1 * min(1.0, cv)  # Lower when distances are uniform
    w_mst = 0.15 + 0.1 * size_factor  # Higher for larger instances
    w_density = 0.2 + 0.1 * min(1.0, cv)  # Higher when distances vary
    
    # Normalize weights
    total = w_freq + w_dist + w_mst + w_density
    w_freq /= total
    w_dist /= total
    w_mst /= total
    w_density /= total
    
    # Combine features
    heuristic_matrix = (
        w_freq * edge_frequency +
        w_dist * inv_dist_feature +
        w_mst * mst_feature +
        w_density * density_feature
    )
    
    # ===== 6. INTELLIGENT SPARSIFICATION =====
    # Determine edges to keep per node
    min_edges = max(2, int(1.5 * log_n))
    max_edges = min(n-1, int(3 * log_n))
    
    # Create sparse mask
    sparse_mask = np.zeros_like(heuristic_matrix, dtype=bool)
    
    for i in range(n):
        # Get scores for this node
        scores = heuristic_matrix[i].copy()
        scores[i] = -np.inf
        
        # Determine number of edges to keep
        n_keep = min(max_edges, max(min_edges, int((n-1) * (0.15 + 0.1 * size_factor))))
        
        # Select top edges
        top_indices = np.argsort(scores)[-n_keep:]
        sparse_mask[i, top_indices] = True
    
    # Ensure MST edges are included for connectivity
    sparse_mask = sparse_mask | (mst_feature > 0)
    
    # Ensure symmetry
    sparse_mask = sparse_mask | sparse_mask.T
    
    # Apply sparsification
    heuristic_matrix = heuristic_matrix * sparse_mask
    
    # ===== 7. FINAL PROCESSING =====
    # Ensure symmetry and zero diagonal
    heuristic_matrix = 0.5 * (heuristic_matrix + heuristic_matrix.T)
    np.fill_diagonal(heuristic_matrix, 0)
    
    # Robust normalization
    non_zero = heuristic_matrix[heuristic_matrix > 0]
    if len(non_zero) > 0:
        # Clip extreme values
        q10, q90 = np.percentile(non_zero, [10, 90])
        heuristic_matrix = np.clip(heuristic_matrix, q10, q90)
        # Normalize to [0, 1]
        if heuristic_matrix.max() > 0:
            heuristic_matrix /= heuristic_matrix.max()
    
    return heuristic_matrix
