import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with improved stochastic sampling,
    adaptive component balancing, and refined sparsification.
    
    Key improvements over v1:
    1. More diverse and higher-quality solution sampling with improved 2-opt
    2. Better edge betweenness estimation with weighted random walks
    3. More robust adaptive weighting based on instance statistics
    4. Enhanced sparsification preserving critical connectivity
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    cv_dist = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    
    # Size-dependent parameters
    log_n = np.log(n + 1)
    density_factor = min(1.0, n / 75.0)  # Adjusted for larger instances
    
    # ===== 2. ENHANCED SOLUTION SAMPLING =====
    def generate_initial_tour(method: str = 'nn') -> np.ndarray:
        """Generate initial tour using specified construction method."""
        if method == 'nn':  # Improved nearest neighbor
            start = np.random.randint(0, n)
            tour = [start]
            unvisited = set(range(n)) - {start}
            
            while unvisited:
                current = tour[-1]
                # Consider k-nearest neighbors for more diversity
                k = min(5, len(unvisited))
                candidates = sorted(unvisited, key=lambda x: distance_matrix[current, x])[:k]
                # Weighted random selection favoring closer nodes
                weights = 1.0 / (distance_matrix[current, candidates] + 1e-10)
                weights = weights / weights.sum()
                next_node = np.random.choice(candidates, p=weights)
                tour.append(next_node)
                unvisited.remove(next_node)
            return np.array(tour)
        
        elif method == 'greedy':  # Farthest insertion
            if n <= 3:
                return np.random.permutation(n)
            
            # Start with three farthest nodes for better dispersion
            # Find two farthest nodes
            max_dist = -1
            start_nodes = (0, 1)
            for i in range(n):
                for j in range(i+1, n):
                    if distance_matrix[i, j] > max_dist:
                        max_dist = distance_matrix[i, j]
                        start_nodes = (i, j)
            
            # Find third node maximizing minimum distance to existing
            candidates = set(range(n)) - set(start_nodes)
            third_node = max(candidates, key=lambda x: min(
                distance_matrix[x, start_nodes[0]],
                distance_matrix[x, start_nodes[1]]
            ))
            
            tour = list(start_nodes) + [third_node]
            remaining = set(range(n)) - set(tour)
            
            while remaining:
                # Farthest unvisited node from current tour
                farthest_node = max(remaining, key=lambda x: min(
                    distance_matrix[x, node] for node in tour
                ))
                
                # Find best insertion position
                best_cost = float('inf')
                best_pos = -1
                
                for i in range(len(tour)-1):
                    cost_inc = (distance_matrix[tour[i], farthest_node] +
                               distance_matrix[farthest_node, tour[i+1]] -
                               distance_matrix[tour[i], tour[i+1]])
                    if cost_inc < best_cost:
                        best_cost = cost_inc
                        best_pos = i + 1
                
                # Also consider insertion at end
                cost_inc_end = (distance_matrix[tour[-1], farthest_node] +
                               distance_matrix[farthest_node, tour[0]] -
                               distance_matrix[tour[-1], tour[0]])
                if cost_inc_end < best_cost:
                    best_pos = len(tour)
                
                tour.insert(best_pos, farthest_node)
                remaining.remove(farthest_node)
            
            return np.array(tour)
        
        else:  # Randomized construction
            tour = np.random.permutation(n)
            return tour
    
    def improved_two_opt(tour: np.ndarray, max_iter: int = 50) -> np.ndarray:
        """Enhanced 2-opt local search with better neighborhood exploration."""
        best_tour = tour.copy()
        best_length = sum(distance_matrix[best_tour[i], best_tour[(i+1)%n]] 
                         for i in range(n))
        
        improved = True
        iteration = 0
        
        while improved and iteration < max_iter:
            improved = False
            
            # Generate random order of indices for more diverse exploration
            indices = list(range(n))
            np.random.shuffle(indices)
            
            for i_idx in range(n-1):
                i = indices[i_idx]
                for j_idx in range(i_idx+1, n):
                    j = indices[j_idx]
                    
                    if j == (i + 1) % n or i == (j + 1) % n:
                        continue
                    
                    # Calculate gain more efficiently
                    a, b = tour[i], tour[(i+1)%n]
                    c, d = tour[j], tour[(j+1)%n]
                    
                    gain = distance_matrix[a, b] + distance_matrix[c, d] - \
                           distance_matrix[a, c] - distance_matrix[b, d]
                    
                    if gain > 1e-6:  # Significant improvement
                        # Reverse the segment between i+1 and j
                        if i < j:
                            tour[i+1:j+1] = tour[i+1:j+1][::-1]
                        else:
                            # Handle wrap-around case
                            segment = np.concatenate([tour[i+1:], tour[:j+1]])
                            segment = segment[::-1]
                            tour[i+1:] = segment[:n-i-1]
                            tour[:j+1] = segment[n-i-1:]
                        
                        improved = True
                        break
                if improved:
                    break
            
            # Keep track of best tour found
            current_length = sum(distance_matrix[tour[i], tour[(i+1)%n]] 
                               for i in range(n))
            if current_length < best_length:
                best_tour = tour.copy()
                best_length = current_length
            
            iteration += 1
        
        return best_tour
    
    # Generate diverse and high-quality tours
    num_tours = min(15, max(6, n // 3))
    tours = []
    
    # Mix of construction methods
    method_weights = [0.4, 0.4, 0.2]  # NN, greedy, random
    
    for _ in range(num_tours):
        method_choice = np.random.choice(['nn', 'greedy', 'random'], p=method_weights)
        tour = generate_initial_tour(method_choice)
        # Adaptive number of iterations based on instance size
        max_improve_iter = int(25 + 15 * density_factor)
        improved_tour = improved_two_opt(tour, max_iter=max_improve_iter)
        tours.append(improved_tour)
    
    # Edge frequency from tours with quality weighting
    edge_freq = np.zeros_like(distance_matrix)
    tour_lengths = []
    
    for tour in tours:
        length = sum(distance_matrix[tour[i], tour[(i+1)%n]] for i in range(n))
        tour_lengths.append(length)
    
    # Weight tours by quality (shorter tours get higher weight)
    if len(tour_lengths) > 0:
        min_len = min(tour_lengths)
        max_len = max(tour_lengths)
        if max_len > min_len:
            weights = [(max_len - l) / (max_len - min_len) for l in tour_lengths]
        else:
            weights = [1.0] * len(tour_lengths)
    else:
        weights = [1.0] * len(tours)
    
    for tour, weight in zip(tours, weights):
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            edge_freq[u, v] += weight
            edge_freq[v, u] += weight
    
    if edge_freq.max() > 0:
        edge_freq = edge_freq / edge_freq.max()
    
    # ===== 3. ENHANCED EDGE BETWEENNESS ESTIMATION =====
    edge_betweenness = np.zeros_like(distance_matrix)
    
    # Adaptive number of random walks based on instance size
    num_walks = min(300, max(80, n * 3))
    
    for _ in range(num_walks):
        # Random source and target
        s, t = np.random.choice(n, 2, replace=False)
        
        current = s
        path = [s]
        # Use tabu list to avoid short cycles
        tabu_size = max(3, n // 10)
        tabu_list = [s]
        
        while current != t and len(path) < n * 1.5:
            # Get all possible next nodes
            candidates = list(range(n))
            candidates.remove(current)
            
            # Score candidates based on multiple factors
            scores = []
            for v in candidates:
                # Base score: inverse distance (prefer shorter edges)
                dist_score = 1.0 / (distance_matrix[current, v] + 1e-10)
                
                # Progress score: proximity to target
                progress_to_target = 1.0 / (distance_matrix[v, t] + 1e-10)
                
                # Avoidance score: penalize recently visited nodes
                recency_penalty = 1.0
                if v in tabu_list[-tabu_size:]:
                    # Reduce probability of revisiting recent nodes
                    recency_penalty = 0.3
                
                # Combine scores
                total_score = dist_score + 0.4 * progress_to_target
                total_score *= recency_penalty
                
                # Small random component for exploration
                total_score *= (0.9 + 0.2 * np.random.random())
                
                scores.append(total_score)
            
            # Choose next node with probability proportional to score
            scores = np.array(scores)
            if scores.sum() > 0:
                probs = scores / scores.sum()
                next_node = np.random.choice(candidates, p=probs)
            else:
                next_node = np.random.choice(candidates)
            
            # Update edge betweenness
            edge_betweenness[current, next_node] += 1
            edge_betweenness[next_node, current] += 1
            
            # Update path and tabu list
            current = next_node
            path.append(current)
            tabu_list.append(current)
            
            # Trim tabu list if too long
            if len(tabu_list) > tabu_size * 2:
                tabu_list = tabu_list[-tabu_size:]
            
            # Reset if stuck (too many steps without progress)
            if len(path) > n * 1.2 and np.random.random() < 0.1:
                # Random restart near target
                current = t if np.random.random() < 0.7 else np.random.randint(0, n)
                path = [current]
                tabu_list = [current]
    
    # Normalize edge betweenness
    if edge_betweenness.max() > 0:
        edge_betweenness = edge_betweenness / edge_betweenness.max()
    
    # ===== 4. IMPROVED ADAPTIVE WEIGHTING =====
    # Compute component statistics
    freq_vals = edge_freq[np.triu_indices(n, k=1)]
    bet_vals = edge_betweenness[np.triu_indices(n, k=1)]
    
    freq_std = np.std(freq_vals)
    bet_std = np.std(bet_vals)
    
    # Compute entropy-like measures of component distributions
    def normalized_entropy(vals):
        """Compute normalized entropy of values."""
        hist, _ = np.histogram(vals, bins=min(20, len(vals)//10), density=True)
        hist = hist[hist > 0]
        if len(hist) > 1:
            entropy = -np.sum(hist * np.log(hist))
            max_entropy = np.log(len(hist))
            return entropy / max_entropy
        return 0.5
    
    freq_entropy = normalized_entropy(freq_vals)
    bet_entropy = normalized_entropy(bet_vals)
    
    # Determine weights based on instance characteristics and component quality
    # Base weights with adjustments based on entropy and variance
    w_freq_base = 0.35
    w_bet_base = 0.30
    w_dist_base = 0.25
    w_mst_base = 0.10
    
    # Adjust based on entropy (higher entropy -> more uniform -> less weight)
    freq_weight = w_freq_base * (1.0 + 0.3 * (1 - freq_entropy) - 0.2 * freq_entropy)
    bet_weight = w_bet_base * (1.0 + 0.3 * (1 - bet_entropy) - 0.2 * bet_entropy)
    
    # Adjust distance weight based on distance variability
    dist_weight = w_dist_base * (0.8 + 0.4 * min(1.0, cv_dist))
    
    # MST weight increases with instance size
    mst_weight = w_mst_base * (0.5 + density_factor)
    
    # Normalize weights
    total = freq_weight + bet_weight + dist_weight + mst_weight
    w_freq = freq_weight / total
    w_bet = bet_weight / total
    w_dist = dist_weight / total
    w_mst = mst_weight / total
    
    # ===== 5. COMPONENT COMPUTATION =====
    # Distance component (inverse, normalized with smoothing)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + 0.01 * dist_mean)  # Smoothing
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # MST component with importance weighting
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_scores = np.zeros_like(distance_matrix)
    
    # MST edges get higher scores for shorter distances
    mst_distances = distance_matrix[mst_mask]
    if len(mst_distances) > 0:
        min_mst_dist = mst_distances.min()
        max_mst_dist = mst_distances.max()
        if max_mst_dist > min_mst_dist:
            normalized_mst = 1.0 - (mst_distances - min_mst_dist) / (max_mst_dist - min_mst_dist)
            mst_scores[mst_mask] = normalized_mst
        else:
            mst_scores[mst_mask] = 1.0
    
    if mst_scores.max() > 0:
        mst_scores = mst_scores / mst_scores.max()
    
    # ===== 6. COMBINE COMPONENTS =====
    heuristics_matrix = (
        w_freq * edge_freq +
        w_bet * edge_betweenness +
        w_dist * inv_dist +
        w_mst * mst_scores
    )
    
    # ===== 7. REFINED SPARSIFICATION =====
    # Determine edges to keep with adaptive thresholds
    base_edges_per_node = max(2, int(1.8 * log_n))
    max_edges_per_node = min(n-1, base_edges_per_node * 3)
    
    # Global threshold for promising edges
    all_scores = heuristics_matrix[np.triu_indices(n, k=1)]
    if len(all_scores) > 0:
        score_threshold = np.percentile(all_scores, max(20, 100 - 15 * density_factor * 100))
    else:
        score_threshold = 0
    
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        # Determine number of edges to keep for this node
        # More edges for nodes with high degree centrality (estimated by score sum)
        node_score_sum = scores.sum()
        if node_score_sum > 0:
            node_importance = min(1.5, node_score_sum / np.median(scores[scores > 0]))
        else:
            node_importance = 1.0
        
        n_edges = min(max_edges_per_node, 
                     max(base_edges_per_node, 
                         int(base_edges_per_node * node_importance * (1 + 0.5 * cv_dist))))
        
        # Keep top n_edges AND edges above global threshold
        top_indices = np.argsort(scores)[-n_edges:]
        above_threshold = np.where(scores >= score_threshold)[0]
        
        keep_indices = np.union1d(top_indices, above_threshold)
        sparsity_mask[i, keep_indices] = True
    
    # Ensure connectivity: add MST edges and ensure symmetry
    sparsity_mask = sparsity_mask | mst_mask
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # For each node, ensure at least 2 connections
    for i in range(n):
        if sparsity_mask[i].sum() < 2:
            # Add the best missing edges
            missing = np.where(~sparsity_mask[i])[0]
            missing = missing[missing != i]
            if len(missing) > 0:
                scores = heuristics_matrix[i, missing]
                n_needed = 2 - sparsity_mask[i].sum()
                top_missing = missing[np.argsort(-scores)[:n_needed]]
                sparsity_mask[i, top_missing] = True
                sparsity_mask[top_missing, i] = True
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 8. FINAL PROCESSING =====
    # Ensure symmetry and zero diagonal
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Apply softmax-like normalization per node for better relative scoring
    for i in range(n):
        row = heuristics_matrix[i]
        if row.max() > 0:
            # Softmax with temperature parameter
            temperature = 0.3 + 0.7 * density_factor
            exp_row = np.exp(row / temperature)
            heuristics_matrix[i] = exp_row / exp_row.sum()
    
    # Final global normalization
    if heuristics_matrix.max() > 0:
        heuristics_matrix = heuristics_matrix / heuristics_matrix.max()
    
    return heuristics_matrix
