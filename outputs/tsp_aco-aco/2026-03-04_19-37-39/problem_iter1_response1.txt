import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Sparsifies by keeping only nearest neighbor edges
    2. Rewards mutual nearest neighbor relationships
    3. Normalizes heuristics by node degree
    """
    n = distance_matrix.shape[0]
    
    # Avoid division by zero for diagonal (self-distances)
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    
    # 1. Determine nearest neighbors for each node
    # Keep top k = min(5, n-1) nearest neighbors for sparsity
    k = min(5, n - 1)
    nearest_counts = np.zeros((n, n), dtype=int)
    
    # For each node, find its k nearest neighbors
    for i in range(n):
        # Get indices of k smallest distances (excluding self)
        nearest_idx = np.argsort(dist[i])[:k]
        nearest_counts[i, nearest_idx] = 1
    
    # 2. Create mutual nearest neighbor mask
    # An edge is more promising if it's a nearest neighbor for BOTH endpoints
    mutual_mask = (nearest_counts + nearest_counts.T) == 2
    one_way_mask = (nearest_counts + nearest_counts.T) == 1
    
    # 3. Compute base heuristics (inverse distance)
    heuristics = np.zeros_like(dist, dtype=float)
    valid_mask = dist > 0  # Exclude diagonal and any zero distances
    heuristics[valid_mask] = 1.0 / dist[valid_mask]
    
    # 4. Apply sparsification and weighting
    # Zero out edges not in nearest neighbor graph
    heuristics *= (nearest_counts + nearest_counts.T > 0)
    
    # Boost mutual nearest neighbors
    heuristics[mutual_mask] *= 3.0  # Strong preference
    heuristics[one_way_mask] *= 1.5  # Moderate preference
    
    # 5. Normalize by node connectivity (degree in nearest neighbor graph)
    # Nodes with fewer good connections should have stronger edge preferences
    node_degree = (nearest_counts + nearest_counts.T > 0).sum(axis=1)
    # Avoid division by zero (isolated nodes)
    node_degree[node_degree == 0] = 1
    degree_weight = 1.0 / np.sqrt(node_degree)
    
    # Apply degree-based normalization
    for i in range(n):
        heuristics[i, :] *= degree_weight[i]
        heuristics[:, i] *= degree_weight[i]  # Symmetric normalization
    
    # 6. Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0.0)
    
    # 7. Optional: Softmax normalization for probabilistic interpretation
    # Reshape for softmax computation
    flat_heuristics = heuristics.flatten()
    # Apply softmax to non-zero entries only
    non_zero_mask = flat_heuristics > 0
    if np.any(non_zero_mask):
        non_zero_values = flat_heuristics[non_zero_mask]
        # Stable softmax
        exp_values = np.exp(non_zero_values - np.max(non_zero_values))
        softmax_values = exp_values / np.sum(exp_values)
        # Rescale to maintain magnitude while preserving relative ordering
        flat_heuristics[non_zero_mask] = softmax_values * np.mean(non_zero_values)
        heuristics = flat_heuristics.reshape(n, n)
    
    return heuristics
