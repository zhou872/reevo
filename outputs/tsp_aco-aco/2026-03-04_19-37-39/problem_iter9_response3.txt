import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with improved efficiency and adaptive strategies.
    Focuses on faster computations and better integration of sampling and centrality.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    
    # Size-adaptive parameters
    density_factor = min(1.0, n / 50.0)
    log_n = np.log(n + 1)
    
    # ===== 2. EFFICIENT SOLUTION SAMPLING =====
    def generate_initial_tour(method: str = 'nn', start_node: int = 0) -> np.ndarray:
        if method == 'nn':  # Nearest neighbor
            tour = [start_node]
            unvisited = set(range(n)) - {start_node}
            current = start_node
            while unvisited:
                # Use argmin for speed
                unvisited_list = list(unvisited)
                nearest_idx = np.argmin([distance_matrix[current, j] for j in unvisited_list])
                nearest = unvisited_list[nearest_idx]
                tour.append(nearest)
                unvisited.remove(nearest)
                current = nearest
            return np.array(tour)
        
        elif method == 'random':
            return np.random.permutation(n)
        
        elif method == 'greedy':
            if n <= 3:
                return np.random.permutation(n)
            # Start with a random triangle
            start_nodes = np.random.choice(n, 3, replace=False)
            tour = list(start_nodes[:3])
            tour.append(tour[0])  # Close the triangle
            remaining = set(range(n)) - set(tour[:-1])
            
            while remaining:
                best_increase = float('inf')
                best_node = None
                best_pos = -1
                
                for node in remaining:
                    for i in range(len(tour)-1):
                        inc = (distance_matrix[tour[i], node] +
                               distance_matrix[node, tour[i+1]] -
                               distance_matrix[tour[i], tour[i+1]])
                        if inc < best_increase:
                            best_increase = inc
                            best_node = node
                            best_pos = i + 1
                
                tour.insert(best_pos, best_node)
                remaining.remove(best_node)
            
            return np.array(tour[:-1])
        else:
            return np.random.permutation(n)
    
    def fast_2opt_improvement(tour: np.ndarray, max_iter: int = 30) -> np.ndarray:
        """Apply 2-opt with early stopping for efficiency."""
        improved = True
        iter_count = 0
        current_tour = tour.copy()
        current_len = sum(distance_matrix[current_tour[i], current_tour[(i+1)%n]] for i in range(n))
        
        while improved and iter_count < max_iter:
            improved = False
            for i in range(n-1):
                for j in range(i+2, n):
                    if j == n-1 and i == 0:
                        continue
                    
                    a, b = current_tour[i], current_tour[(i+1)%n]
                    c, d = current_tour[j], current_tour[(j+1)%n]
                    
                    old = distance_matrix[a, b] + distance_matrix[c, d]
                    new = distance_matrix[a, c] + distance_matrix[b, d]
                    
                    if new < old - 1e-10:  # Small tolerance
                        current_tour = np.concatenate([
                            current_tour[:i+1],
                            current_tour[i+1:j+1][::-1],
                            current_tour[j+1:]
                        ])
                        current_len = current_len - old + new
                        improved = True
                        break
                if improved:
                    break
            iter_count += 1
        
        return current_tour
    
    # Generate diverse tours with adaptive counts
    num_tours = min(20, max(5, int(n * 0.4)))
    tours = []
    
    # Adjust method distribution based on instance size
    if n < 30:
        nn_ratio, greedy_ratio, random_ratio = 0.5, 0.3, 0.2
    else:
        nn_ratio, greedy_ratio, random_ratio = 0.4, 0.4, 0.2
    
    nn_count = int(num_tours * nn_ratio)
    greedy_count = int(num_tours * greedy_ratio)
    random_count = num_tours - nn_count - greedy_count
    
    for _ in range(nn_count):
        start = np.random.randint(0, n)
        tour = generate_initial_tour('nn', start)
        tour = fast_2opt_improvement(tour, max_iter=int(15 * density_factor))
        tours.append(tour)
    
    for _ in range(greedy_count):
        tour = generate_initial_tour('greedy')
        tour = fast_2opt_improvement(tour, max_iter=int(15 * density_factor))
        tours.append(tour)
    
    for _ in range(random_count):
        tour = generate_initial_tour('random')
        tour = fast_2opt_improvement(tour, max_iter=int(15 * density_factor))
        tours.append(tour)
    
    # Compute edge frequency matrix efficiently
    edge_frequency = np.zeros_like(distance_matrix, dtype=float)
    for tour in tours:
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            edge_frequency[u, v] += 1
            edge_frequency[v, u] += 1
    
    if edge_frequency.max() > 0:
        edge_frequency /= edge_frequency.max()
    
    # ===== 3. EFFICIENT EDGE BETWEENNESS APPROXIMATION =====
    # Use random path sampling for betweenness to reduce computation
    betweenness = np.zeros(n)
    sample_size = min(20, max(5, int(n * 0.5)))  # Adaptive sample size
    
    for _ in range(sample_size):
        # Sample a random simple path using greedy neighbor selection
        path = []
        current = np.random.randint(0, n)
        visited = set([current])
        path.append(current)
        
        while len(visited) < min(n, 10):  # Limit path length for efficiency
            # Choose next node from unvisited based on inverse distance probability
            unvisited = set(range(n)) - visited
            if not unvisited:
                break
            probs = np.array([1.0 / (distance_matrix[current, j] + 1e-10) for j in unvisited])
            probs /= probs.sum()
            next_node = np.random.choice(list(unvisited), p=probs)
            path.append(next_node)
            visited.add(next_node)
            current = next_node
        
        # Update betweenness for nodes in the path
        for node in path:
            betweenness[node] += 1
    
    # Normalize betweenness
    if betweenness.max() > betweenness.min():
        betweenness = (betweenness - betweenness.min()) / (betweenness.max() - betweenness.min() + 1e-10)
    
    # Edge potential from centrality
    centrality = betweenness / (betweenness.sum() + 1e-10)
    edge_potential = np.outer(centrality, centrality)
    
    # ===== 4. DYNAMIC WEIGHT BALANCING =====
    # Adjust weights based on instance statistics and sample quality
    edge_freq_std = np.std(edge_frequency[np.triu_indices(n, k=1)])
    freq_weight = 0.5 * (1 + min(1.0, edge_freq_std))  # Higher if clear patterns
    
    dist_weight = 0.3 * (1 - min(1.0, cv))  # Lower for uniform distances
    cent_weight = 0.2 * min(1.0, cv)  # Higher for varied distances
    
    # MST weight for connectivity assurance
    mst_weight = 0.2 * density_factor
    
    # Normalize weights
    total = freq_weight + dist_weight + cent_weight + mst_weight
    freq_weight /= total
    dist_weight /= total
    cent_weight /= total
    mst_weight /= total
    
    # ===== 5. COMPONENT COMPUTATION =====
    # Inverse distance component
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist /= inv_dist.max()
    
    # MST component
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_scores = np.zeros_like(distance_matrix)
    mst_scores[mst_mask] = 1.0
    # Normalize by distance for MST edges
    mst_scores = mst_scores / (distance_matrix + 1e-10)
    if mst_scores.max() > 0:
        mst_scores /= mst_scores.max()
    
    # ===== 6. COMBINE COMPONENTS =====
    heuristics_matrix = (
        freq_weight * edge_frequency +
        dist_weight * inv_dist +
        cent_weight * edge_potential +
        mst_weight * mst_scores
    )
    
    # Boost edges that are in both high-frequency tours and MST
    high_freq_mask = edge_frequency > np.percentile(edge_frequency[edge_frequency > 0], 70)
    overlap_mask = mst_mask & high_freq_mask
    heuristics_matrix[overlap_mask] *= 1.4
    
    # ===== 7. SMART GRAPH SPARSIFICATION =====
    # Determine edges to keep per node adaptively
    min_edges = max(2, int(1.5 * log_n))
    max_edges = min(n-1, int(3 * min_edges))
    
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        # Dynamic edge count based on node connectivity
        base_edges = int(min_edges + (max_edges - min_edges) * density_factor)
        keep_edges = min(n-1, base_edges)
        
        top_indices = np.argpartition(scores, -keep_edges)[-keep_edges:]
        sparsity_mask[i, top_indices] = True
    
    # Ensure connectivity by including MST edges
    sparsity_mask = sparsity_mask | mst_mask
    # Ensure symmetry
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # Additional boost for nearest neighbors within sparse graph
    k_nn = max(2, min(5, int(np.sqrt(n))))
    for i in range(n):
        nn_indices = np.argsort(distance_matrix[i])[1:k_nn+1]
        heuristics_matrix[i, nn_indices] *= 1.3
        heuristics_matrix[nn_indices, i] *= 1.3
    
    # ===== 8. FINAL PROCESSING =====
    # Ensure symmetry
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1] range robustly
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        min_val, max_val = non_zero.min(), non_zero.max()
        if max_val > min_val:
            heuristics_matrix = (heuristics_matrix - min_val) / (max_val - min_val)
    
    return heuristics_matrix
