import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge heuristics with focus on practical sampling and clear adaptive mechanisms.
    
    Key improvements:
    1. Efficient tour sampling using construction heuristics (NN, Greedy, Random)
    2. Edge importance estimation via 2-opt improved tours
    3. Adaptive sparsification based on instance characteristics
    4. Minimum spanning tree backbone for connectivity
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE ANALYSIS =====
    # Analyze distance distribution
    triu_distances = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(triu_distances)
    dist_std = np.std(triu_distances)
    cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    
    # Size and complexity factors
    log_n = np.log(n + 1)
    size_factor = min(1.0, np.log10(n) / 2)
    
    # ===== 2. SOLUTION SAMPLING =====
    def construct_nn_tour(start: int) -> np.ndarray:
        """Construct tour using nearest neighbor from given start."""
        tour = [start]
        unvisited = set(range(n))
        unvisited.remove(start)
        
        while unvisited:
            current = tour[-1]
            # Find nearest unvisited city
            distances_to_current = distance_matrix[current].copy()
            for visited in tour:
                distances_to_current[visited] = np.inf
            next_city = np.argmin(distances_to_current)
            tour.append(next_city)
            unvisited.remove(next_city)
        
        return np.array(tour)
    
    def construct_greedy_tour() -> np.ndarray:
        """Construct tour using greedy edge selection."""
        edges = []
        edge_set = set()
        
        # Get all edges sorted by distance
        for i in range(n):
            for j in range(i+1, n):
                edges.append((distance_matrix[i, j], i, j))
        edges.sort()
        
        # Greedy selection while maintaining degree ≤ 2
        degree = np.zeros(n, dtype=int)
        tour_edges = []
        
        for dist, i, j in edges:
            if degree[i] < 2 and degree[j] < 2:
                # Check if adding edge would create a cycle prematurely
                if degree[i] == 1 and degree[j] == 1:
                    # Trace from i and j to see if they're already connected
                    visited_i = set()
                    node = i
                    for _ in range(n):
                        visited_i.add(node)
                        next_nodes = [n for n, deg in enumerate(degree) if deg > 0 and n not in visited_i]
                        if not next_nodes or len(visited_i) > 2:
                            break
                        # Find connected node
                        for edge in tour_edges:
                            if edge[0] == node and edge[1] not in visited_i:
                                node = edge[1]
                                break
                            elif edge[1] == node and edge[0] not in visited_i:
                                node = edge[0]
                                break
                    if j in visited_i and len(visited_i) < n:
                        continue  # Would create cycle
                
                tour_edges.append((i, j))
                degree[i] += 1
                degree[j] += 1
                
            if len(tour_edges) == n - 1:
                break
        
        # Convert edges to tour
        # Find nodes with degree 1 to start
        start = np.where(degree == 1)[0][0]
        tour = [start]
        visited = set([start])
        
        for _ in range(n-1):
            current = tour[-1]
            for edge in tour_edges:
                if edge[0] == current and edge[1] not in visited:
                    tour.append(edge[1])
                    visited.add(edge[1])
                    break
                elif edge[1] == current and edge[0] not in visited:
                    tour.append(edge[0])
                    visited.add(edge[0])
                    break
        
        return np.array(tour)
    
    def two_opt(tour: np.ndarray, max_iter: int = 20) -> np.ndarray:
        """Apply 2-opt local search to improve tour."""
        improved = True
        iteration = 0
        current = tour.copy()
        
        while improved and iteration < max_iter:
            improved = False
            for i in range(n-1):
                for j in range(i+2, n):
                    if j == n-1 and i == 0:
                        continue
                    
                    a, b = current[i], current[(i+1)%n]
                    c, d = current[j], current[(j+1)%n]
                    
                    old_len = distance_matrix[a, b] + distance_matrix[c, d]
                    new_len = distance_matrix[a, c] + distance_matrix[b, d]
                    
                    if new_len < old_len:
                        # Reverse segment between i+1 and j
                        current[i+1:j+1] = current[i+1:j+1][::-1]
                        improved = True
                        break
                if improved:
                    break
            iteration += 1
        
        return current
    
    # Generate diverse tours
    num_tours = min(15, max(5, n // 3))
    tours = []
    
    # NN tours from different starting points
    nn_starts = np.random.choice(n, min(5, n), replace=False)
    for start in nn_starts:
        tour = construct_nn_tour(start)
        tours.append(two_opt(tour, max_iter=int(15 * size_factor)))
    
    # Greedy tours
    for _ in range(min(5, num_tours - len(nn_starts))):
        tour = construct_greedy_tour()
        tours.append(two_opt(tour, max_iter=int(15 * size_factor)))
    
    # Random tours with 2-opt
    for _ in range(num_tours - len(tours)):
        tour = np.random.permutation(n)
        tours.append(two_opt(tour, max_iter=int(10 * size_factor)))
    
    # Edge frequency from tours
    edge_freq = np.zeros_like(distance_matrix)
    for tour in tours:
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            edge_freq[u, v] += 1
            edge_freq[v, u] += 1
    
    if edge_freq.max() > 0:
        edge_freq = edge_freq / edge_freq.max()
    
    # ===== 3. MINIMUM SPANNING TREE COMPONENT =====
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_scores = np.zeros_like(distance_matrix)
    mst_scores[mst_mask] = 1.0 / (distance_matrix[mst_mask] + 1e-10)
    if mst_scores.max() > 0:
        mst_scores = mst_scores / mst_scores.max()
    
    # ===== 4. DISTANCE-BASED COMPONENT =====
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n))
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # ===== 5. ADAPTIVE WEIGHTING =====
    # Analyze edge frequency distribution
    freq_nonzero = edge_freq[edge_freq > 0]
    if len(freq_nonzero) > 0:
        freq_std = np.std(freq_nonzero)
        freq_mean = np.mean(freq_nonzero)
        freq_cv = freq_std / (freq_mean + 1e-10)
    else:
        freq_cv = 0
    
    # Base weights
    w_freq = 0.4 + 0.2 * min(1.0, freq_cv)  # More weight if frequency has high variance
    w_mst = 0.3 - 0.1 * size_factor  # Less weight for larger instances
    w_dist = 0.3 - 0.1 * cv  # Less weight if distances are uniform
    
    # Normalize
    total = w_freq + w_mst + w_dist
    w_freq /= total
    w_mst /= total
    w_dist /= total
    
    # ===== 6. COMBINE COMPONENTS =====
    heuristics_matrix = (
        w_freq * edge_freq +
        w_mst * mst_scores +
        w_dist * inv_dist
    )
    
    # ===== 7. ADAPTIVE SPARSIFICATION =====
    if n > 10:
        # Determine edges to keep per node
        base_edges = max(3, int(1.5 * log_n))
        # More edges for complex instances, fewer for uniform
        target_edges = int(base_edges * (1.0 + 0.5 * cv - 0.2 * size_factor))
        target_edges = max(base_edges, min(n-1, target_edges))
        
        sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
        
        for i in range(n):
            scores = heuristics_matrix[i].copy()
            scores[i] = -np.inf
            
            # Keep top edges
            threshold = np.partition(scores, -target_edges)[-target_edges]
            sparsity_mask[i, scores >= threshold] = True
        
        # Ensure MST edges are always included
        sparsity_mask = sparsity_mask | mst_mask
        sparsity_mask = sparsity_mask | sparsity_mask.T  # Ensure symmetry
        
        # Apply sparsification
        heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 8. FINAL PROCESSING =====
    # Ensure symmetry and zero diagonal
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1]
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        h_min, h_max = non_zero.min(), non_zero.max()
        if h_max > h_min:
            heuristics_matrix[heuristics_matrix > 0] = (non_zero - h_min) / (h_max - h_min)
    
    return heuristics_matrix
