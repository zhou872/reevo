import numpy as np
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristic combining Christofides algorithm insights with
    reinforcement learning adaptation and efficient graph processing.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Matrix of edge promise indicators (same shape as input)
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ========== 1. EFFICIENT MST COMPUTATION ==========
    def prim_mst(dist_mat):
        """Prim's algorithm for MST using priority queue approach."""
        mst_edges = np.zeros((n, n), dtype=bool)
        visited = np.zeros(n, dtype=bool)
        min_edge = np.full(n, np.inf)
        parent = np.full(n, -1, dtype=int)
        
        min_edge[0] = 0
        
        for _ in range(n):
            # Find minimum edge vertex
            u = -1
            min_val = np.inf
            for i in range(n):
                if not visited[i] and min_edge[i] < min_val:
                    min_val = min_edge[i]
                    u = i
            
            if u == -1:
                break
                
            visited[u] = True
            
            # Add edge to MST if not root
            if parent[u] != -1:
                mst_edges[u, parent[u]] = True
                mst_edges[parent[u], u] = True
            
            # Update minimum edges
            for v in range(n):
                if not visited[v] and dist_mat[u, v] < min_edge[v]:
                    min_edge[v] = dist_mat[u, v]
                    parent[v] = u
        
        return mst_edges
    
    mst_mask = prim_mst(distance_matrix)
    
    # ========== 2. ADAPTIVE PARAMETER INITIALIZATION ==========
    # Parameters based on problem characteristics
    problem_density = np.mean(distance_matrix) / np.max(distance_matrix) if np.max(distance_matrix) > 0 else 1.0
    
    base_params = {
        'mst_weight': 2.5 + 0.5 * np.log1p(n) / 10,
        'matching_weight': 1.8,
        'local_weight': 1.2 - 0.2 * problem_density,
        'centrality_weight': 0.8,
        'learn_rate': 0.1,
        'exploration_rate': 0.3
    }
    
    # ========== 3. REINFORCEMENT LEARNING ADAPTATION ==========
    def learn_edge_weights(n_episodes=5):
        """Simple Q-learning style adaptation of edge weights."""
        # Initialize Q-values with inverse distance
        with np.errstate(divide='ignore', invalid='ignore'):
            q_values = 1.0 / distance_matrix
        np.fill_diagonal(q_values, 0)
        
        # Normalize
        q_max = np.max(q_values)
        if q_max > 0:
            q_values = q_values / q_max
        
        # Simple tour sampling for weight updates
        for _ in range(n_episodes):
            # Greedy randomized construction
            tour = [np.random.randint(n)]
            visited = set(tour)
            
            while len(visited) < n:
                current = tour[-1]
                
                # Epsilon-greedy selection
                if np.random.random() < base_params['exploration_rate']:
                    # Explore: choose random unvisited
                    candidates = [i for i in range(n) if i not in visited]
                    if candidates:
                        next_node = np.random.choice(candidates)
                else:
                    # Exploit: choose best based on current Q-values
                    mask = np.array([i not in visited for i in range(n)], dtype=bool)
                    mask[current] = False
                    available_q = q_values[current] * mask
                    if np.sum(available_q) > 0:
                        # Softmax selection
                        exp_q = np.exp(available_q - np.max(available_q))
                        exp_q[~mask] = 0
                        if np.sum(exp_q) > 0:
                            exp_q = exp_q / np.sum(exp_q)
                            next_node = np.random.choice(n, p=exp_q)
                        else:
                            next_node = np.random.choice(np.where(mask)[0])
                    else:
                        candidates = np.where(mask)[0]
                        next_node = np.random.choice(candidates) if len(candidates) > 0 else -1
                
                if next_node != -1:
                    # Update Q-value with reward (negative distance)
                    reward = -distance_matrix[current, next_node]
                    max_future_q = np.max(q_values[next_node][mask]) if np.sum(mask) > 0 else 0
                    
                    q_values[current, next_node] = (1 - base_params['learn_rate']) * q_values[current, next_node] + \
                                                  base_params['learn_rate'] * (reward + 0.9 * max_future_q)
                    
                    tour.append(next_node)
                    visited.add(next_node)
        
        return q_values
    
    learned_weights = learn_edge_weights()
    
    # ========== 4. CHRISTOFIDES COMPONENT ENHANCEMENT ==========
    # Identify odd-degree nodes in MST
    degrees = np.sum(mst_mask, axis=0)
    odd_nodes = np.where(degrees % 2 == 1)[0]
    
    # Approximate minimum perfect matching on odd nodes
    matching_mask = np.zeros((n, n), dtype=bool)
    if len(odd_nodes) >= 2:
        # Create distance matrix for odd nodes only
        odd_dist = distance_matrix[np.ix_(odd_nodes, odd_nodes)]
        np.fill_diagonal(odd_dist, np.inf)
        
        # Greedy matching
        matched = np.zeros(len(odd_nodes), dtype=bool)
        
        for i in range(len(odd_nodes)):
            if not matched[i]:
                # Find closest unmatched odd node
                available = np.where(~matched)[0]
                if len(available) <= 1:
                    break
                    
                # Exclude self
                mask = available != i
                if not np.any(mask):
                    break
                    
                available = available[mask]
                closest_idx = available[np.argmin(odd_dist[i, available])]
                
                # Add matching edge
                u, v = odd_nodes[i], odd_nodes[closest_idx]
                matching_mask[u, v] = True
                matching_mask[v, u] = True
                
                matched[i] = True
                matched[closest_idx] = True
    
    # ========== 5. LOCAL OPTIMALITY WITH ADAPTIVE THRESHOLDS ==========
    # Compute adaptive thresholds per node
    local_stats = np.zeros((n, 2))
    for i in range(n):
        non_zero_dists = distance_matrix[i, distance_matrix[i] > 0]
        if len(non_zero_dists) > 0:
            local_stats[i, 0] = np.percentile(non_zero_dists, 25)  # Q1
            local_stats[i, 1] = np.percentile(non_zero_dists, 50)  # Median
    
    local_optimality = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            dist = distance_matrix[i, j]
            threshold_i = 0.6 * local_stats[i, 0] + 0.4 * local_stats[i, 1]
            threshold_j = 0.6 * local_stats[j, 0] + 0.4 * local_stats[j, 1]
            avg_threshold = (threshold_i + threshold_j) / 2
            
            # Sigmoid-like scoring
            if dist < avg_threshold:
                score = 1.0 / (1.0 + np.exp(5.0 * (dist / avg_threshold - 0.8)))
            else:
                score = 0.2 / (1.0 + np.exp(-2.0 * (dist / avg_threshold - 1.5)))
            
            local_optimality[i, j] = local_optimality[j, i] = score
    
    # ========== 6. BETWEENNESS CENTRALITY APPROXIMATION ==========
    # Simple approximation using MST structure
    centrality = np.zeros(n)
    
    # Nodes that are hubs in MST (high degree)
    mst_degrees = np.sum(mst_mask, axis=0)
    max_mst_degree = np.max(mst_degrees) if np.max(mst_degrees) > 0 else 1
    centrality = mst_degrees / max_mst_degree
    
    # Boost centrality for nodes in matching
    matching_nodes = np.unique(np.where(matching_mask))
    if len(matching_nodes) > 0:
        centrality[matching_nodes] *= 1.2
    
    edge_centrality = np.outer(centrality, centrality)
    np.fill_diagonal(edge_centrality, 0)
    
    # ========== 7. ADAPTIVE ENSEMBLE COMBINATION ==========
    # Dynamic weight adjustment based on problem characteristics
    if n < 20:
        ensemble_weights = [0.4, 0.3, 0.2, 0.1]  # Learned, MST/matching, local, centrality
    elif n < 100:
        ensemble_weights = [0.35, 0.35, 0.2, 0.1]
    else:
        ensemble_weights = [0.3, 0.4, 0.2, 0.1]
    
    # Combine components
    combined = (
        ensemble_weights[0] * learned_weights +
        ensemble_weights[1] * (mst_mask * base_params['mst_weight'] + 
                              matching_mask * base_params['matching_weight']) +
        ensemble_weights[2] * local_optimality * base_params['local_weight'] +
        ensemble_weights[3] * edge_centrality * base_params['centrality_weight']
    )
    
    # ========== 8. GRAPH SPARSIFICATION ==========
    # Determine edges to keep based on problem size
    if n < 50:
        keep_per_node = max(4, min(12, n // 2))
        top_pct = 85
    elif n < 200:
        keep_per_node = max(3, min(8, n // 3))
        top_pct = 80
    else:
        keep_per_node = max(2, min(6, n // 4))
        top_pct = 75
    
    # Create sparsity mask
    sparsity_mask = np.zeros_like(combined, dtype=bool)
    
    for i in range(n):
        # Get node's edge values
        node_values = combined[i].copy()
        node_values[i] = -np.inf
        
        # Always keep MST and matching edges
        sparsity_mask[i, mst_mask[i]] = True
        sparsity_mask[i, matching_mask[i]] = True
        
        # Keep top additional edges
        non_mst_mask = ~mst_mask[i] & ~matching_mask[i]
        if np.any(non_mst_mask):
            non_mst_values = node_values[non_mst_mask]
            if len(non_mst_values) > 0:
                threshold = np.percentile(non_mst_values, top_pct)
                top_additional = np.where(non_mst_mask & (node_values >= threshold))[0]
                
                # Limit number of additional edges
                if len(top_additional) > keep_per_node:
                    sorted_idx = np.argsort(node_values[top_additional])[-keep_per_node:]
                    top_additional = top_additional[sorted_idx]
                
                sparsity_mask[i, top_additional] = True
    
    # Ensure symmetry
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply mask
    heuristics_matrix = combined * sparsity_mask
    
    # ========== 9. FINAL ENHANCEMENTS ==========
    # Boost consensus edges (in multiple important sets)
    consensus = (mst_mask & matching_mask) | (mst_mask & (local_optimality > 0.7))
    heuristics_matrix[consensus] *= 1.3
    
    # Smooth with exponential transformation
    min_val = np.min(heuristics_matrix[heuristics_matrix > 0])
    if min_val > 0:
        heuristics_matrix = np.exp(heuristics_matrix / min_val) - 1
    
    # Final normalization
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
