import numpy as np
from scipy.sparse import csr_matrix, lil_matrix
from scipy.sparse.csgraph import minimum_spanning_tree
import math
from typing import List, Tuple, Dict, Optional

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with deep heuristic prediction, 
    reinforcement learning enhanced meta-learning, and scalable multi-resolution strategies.
    
    Key Innovations:
    1. Deep heuristic prediction via neural feature extraction
    2. Reinforcement learning guided meta-learning with experience replay
    3. Hierarchical multi-resolution sampling with adaptive granularity
    4. Transformer-inspired attention for edge importance estimation
    
    Args:
        distance_matrix: Square matrix of pairwise distances (n x n)
        
    Returns:
        Matrix of edge promise indicators (same shape as input)
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. ADVANCED INSTANCE CHARACTERIZATION ====================
    
    # Extract comprehensive instance features for deep prediction
    instance_features = {}
    triu_indices = np.triu_indices(n, k=1)
    edge_lengths = distance_matrix[triu_indices]
    
    # Basic statistics
    instance_features['n_nodes'] = n
    instance_features['dist_mean'] = np.mean(edge_lengths)
    instance_features['dist_std'] = np.std(edge_lengths)
    instance_features['dist_median'] = np.median(edge_lengths)
    instance_features['dist_skew'] = (3 * (instance_features['dist_mean'] - instance_features['dist_median']) / 
                                     (instance_features['dist_std'] + 1e-10))
    instance_features['uniformity'] = 1.0 / (1.0 + instance_features['dist_std'] / (instance_features['dist_mean'] + 1e-10))
    
    # Graph structure features
    instance_features['density'] = n * (n - 1) / 2
    instance_features['aspect_ratio'] = np.log1p(n) / 5 if n > 0 else 0
    
    # Spatial distribution features
    if n >= 5:
        # Compute approximate clustering coefficient
        k_neighbors = min(5, n-1)
        clustering_scores = []
        for i in range(min(100, n)):
            node = np.random.randint(0, n)
            neighbor_dists = distance_matrix[node].copy()
            neighbor_dists[node] = np.inf
            neighbors = np.argsort(neighbor_dists)[:k_neighbors]
            
            # Compute local clustering
            if len(neighbors) >= 2:
                neighbor_matrix = distance_matrix[np.ix_(neighbors, neighbors)]
                triangle_potential = len(neighbors) * (len(neighbors) - 1) / 2
                if triangle_potential > 0:
                    threshold = np.percentile(neighbor_matrix, 50)
                    triangles = np.sum(neighbor_matrix < threshold) / 2 - len(neighbors)
                    clustering_scores.append(triangles / triangle_potential)
        
        instance_features['avg_clustering'] = np.mean(clustering_scores) if clustering_scores else 0.5
    else:
        instance_features['avg_clustering'] = 0.5
    
    # ==================== 2. DEEP HEURISTIC PREDICTION ENGINE ====================
    
    class DeepHeuristicPredictor:
        """Neural-inspired feature extractor for edge importance prediction."""
        
        def __init__(self, n_nodes: int, instance_features: Dict):
            self.n = n_nodes
            self.features = instance_features
            
            # Learnable parameters (simulated neural network weights)
            self.layer1_weights = self._initialize_weights(8, 16)  # Input: 8 features, Output: 16
            self.layer2_weights = self._initialize_weights(16, 8)   # 16 -> 8
            self.layer3_weights = self._initialize_weights(8, 4)    # 8 -> 4
            self.output_weights = self._initialize_weights(4, 1)    # 4 -> 1
            
            # Adaptive learning rates based on instance complexity
            self.learning_rate = 0.1 * (1 + instance_features['aspect_ratio'])
            
        def _initialize_weights(self, in_dim: int, out_dim: int) -> np.ndarray:
            """Xavier-inspired weight initialization."""
            scale = np.sqrt(2.0 / (in_dim + out_dim))
            return np.random.randn(in_dim, out_dim) * scale
        
        def extract_edge_features(self, i: int, j: int) -> np.ndarray:
            """Extract comprehensive features for edge (i, j)."""
            features = []
            
            # Distance-based features
            features.append(distance_matrix[i, j] / (self.features['dist_mean'] + 1e-10))
            features.append(np.log1p(distance_matrix[i, j]))
            
            # Neighborhood features
            k_neighbors = min(10, self.n - 1)
            i_neighbors = np.argsort(distance_matrix[i])[1:k_neighbors+1]
            j_neighbors = np.argsort(distance_matrix[j])[1:k_neighbors+1]
            
            # Common neighbors
            common_neighbors = len(set(i_neighbors) & set(j_neighbors)) / k_neighbors
            features.append(common_neighbors)
            
            # Triangle completion score
            if len(i_neighbors) >= 2 and len(j_neighbors) >= 2:
                triangle_score = 0
                for k in set(i_neighbors) & set(j_neighbors):
                    if k != i and k != j:
                        triangle_score += (distance_matrix[i, k] + distance_matrix[j, k]) / (2 * distance_matrix[i, j] + 1e-10)
                features.append(triangle_score / (len(set(i_neighbors) & set(j_neighbors)) + 1e-10))
            else:
                features.append(0)
            
            # Centrality features
            i_centrality = 1.0 / np.sum(distance_matrix[i]) if np.sum(distance_matrix[i]) > 0 else 0
            j_centrality = 1.0 / np.sum(distance_matrix[j]) if np.sum(distance_matrix[j]) > 0 else 0
            features.append(i_centrality + j_centrality)
            
            # Spatial distribution features
            mean_dist_i = np.mean([distance_matrix[i, k] for k in range(self.n) if k != i])
            mean_dist_j = np.mean([distance_matrix[j, k] for k in range(self.n) if k != j])
            features.append(np.abs(mean_dist_i - mean_dist_j) / (self.features['dist_mean'] + 1e-10))
            
            # Instance-level features
            features.append(self.features['uniformity'])
            features.append(self.features['avg_clustering'])
            
            return np.array(features).reshape(1, -1)
        
        def forward(self, edge_features: np.ndarray) -> float:
            """Forward pass through the neural network."""
            # Layer 1 with ReLU
            z1 = edge_features @ self.layer1_weights
            a1 = np.maximum(z1, 0)
            
            # Layer 2 with ReLU
            z2 = a1 @ self.layer2_weights
            a2 = np.maximum(z2, 0)
            
            # Layer 3 with ReLU
            z3 = a2 @ self.layer3_weights
            a3 = np.maximum(z3, 0)
            
            # Output layer with sigmoid
            z4 = a3 @ self.output_weights
            output = 1 / (1 + np.exp(-z4[0, 0]))
            
            return output
        
        def update_weights(self, edge_features: np.ndarray, target: float, importance: float = 1.0):
            """Update weights using gradient descent with importance weighting."""
            # Forward pass to get activations
            z1 = edge_features @ self.layer1_weights
            a1 = np.maximum(z1, 0)
            da1 = (z1 > 0).astype(float)
            
            z2 = a1 @ self.layer2_weights
            a2 = np.maximum(z2, 0)
            da2 = (z2 > 0).astype(float)
            
            z3 = a2 @ self.layer3_weights
            a3 = np.maximum(z3, 0)
            da3 = (z3 > 0).astype(float)
            
            z4 = a3 @ self.output_weights
            prediction = 1 / (1 + np.exp(-z4[0, 0]))
            
            # Compute gradient
            error = (prediction - target) * importance
            d_output = error * prediction * (1 - prediction)
            
            # Backpropagation
            grad_w3 = a3.T @ d_output
            grad_a2 = d_output @ self.output_weights.T * da3
            
            grad_w2 = a2.T @ grad_a2
            grad_a1 = grad_a2 @ self.layer2_weights.T * da2
            
            grad_w1 = edge_features.T @ grad_a1
            
            # Update weights with adaptive learning rate
            lr = self.learning_rate * importance
            self.output_weights -= lr * grad_w3
            self.layer3_weights -= lr * grad_w2
            self.layer2_weights -= lr * grad_w1
        
        def predict_edge_matrix(self) -> np.ndarray:
            """Predict heuristic values for all edges."""
            predictions = np.zeros((self.n, self.n))
            
            # Use importance sampling for efficiency
            n_samples = min(1000, self.n * (self.n - 1) // 2)
            sampled_edges = []
            
            # Sample edges with probability proportional to inverse distance
            for _ in range(n_samples):
                i = np.random.randint(0, self.n)
                j = np.random.randint(0, self.n)
                while j == i:
                    j = np.random.randint(0, self.n)
                sampled_edges.append((i, j))
            
            # Predict for sampled edges and interpolate
            for i, j in sampled_edges:
                features = self.extract_edge_features(i, j)
                pred = self.forward(features)
                predictions[i, j] = pred
                predictions[j, i] = pred
            
            # Fill in missing values using nearest neighbors in feature space
            for i in range(self.n):
                row_values = predictions[i]
                zero_mask = row_values == 0
                if np.any(zero_mask):
                    # Find similar nodes based on distance profile
                    dist_profile = distance_matrix[i]
                    for j in np.where(zero_mask)[0]:
                        if j != i:
                            # Weighted average of predictions for similar nodes
                            similar_nodes = np.argsort(np.abs(dist_profile - distance_matrix[i, j]))[1:6]
                            weights = 1 / (1 + np.abs(dist_profile[similar_nodes] - distance_matrix[i, j]))
                            weights = weights / weights.sum()
                            predictions[i, j] = np.sum(predictions[similar_nodes, j] * weights)
                            predictions[j, i] = predictions[i, j]
            
            np.fill_diagonal(predictions, 0)
            return predictions
    
    # Initialize deep predictor
    deep_predictor = DeepHeuristicPredictor(n, instance_features)
    
    # ==================== 3. REINFORCEMENT LEARNING ENHANCED META-LEARNING ====================
    
    class MetaLearner:
        """Reinforcement learning enhanced meta-learner for adaptive weight optimization."""
        
        def __init__(self, n_factors: int, instance_features: Dict):
            self.n_factors = n_factors
            self.features = instance_features
            
            # Q-learning parameters
            self.q_values = np.ones((5, n_factors))  # 5 states, n_factors actions
            self.learning_rate = 0.1
            self.discount_factor = 0.9
            self.exploration_rate = 0.3
            
            # Experience replay buffer
            self.replay_buffer = []
            self.buffer_size = 100
            self.batch_size = 32
            
            # State representation: [uniformity, clustering, n_nodes_norm, aspect_ratio, cv]
            self.current_state = self._extract_state()
            
            # Weight history for momentum
            self.weight_history = []
            self.momentum = 0.8
            
        def _extract_state(self) -> int:
            """Convert instance features to discrete state."""
            # Discretize features into 5 states
            uniformity = self.features['uniformity']
            if uniformity < 0.3:
                state = 0
            elif uniformity < 0.5:
                state = 1
            elif uniformity < 0.7:
                state = 2
            elif uniformity < 0.9:
                state = 3
            else:
                state = 4
            return state
        
        def select_action(self, weights: np.ndarray) -> Tuple[int, np.ndarray]:
            """Select action using epsilon-greedy policy."""
            if np.random.random() < self.exploration_rate:
                # Explore: random perturbation
                action = np.random.randint(0, 4)
            else:
                # Exploit: choose best action based on Q-values
                action = np.argmax(self.q_values[self.current_state])
            
            # Apply action to weights
            new_weights = weights.copy()
            
            if action == 0:  # Boost distance-based factors
                new_weights[0] *= 1.2
                new_weights[1] *= 1.2
            elif action == 1:  # Boost structure-based factors
                new_weights[2] *= 1.2
                new_weights[3] *= 1.2
            elif action == 2:  # Boost neural predictions
                new_weights[-1] *= 1.2
            elif action == 3:  # Equalize weights
                new_weights = np.ones_like(new_weights) / len(new_weights)
            
            # Renormalize
            new_weights = np.clip(new_weights, 0.01, 0.5)
            new_weights = new_weights / new_weights.sum()
            
            return action, new_weights
        
        def update(self, action: int, reward: float, next_state_features: Dict):
            """Update Q-values using experience."""
            next_state = self._extract_state()
            
            # Q-learning update
            best_next_action = np.max(self.q_values[next_state])
            td_target = reward + self.discount_factor * best_next_action
            td_error = td_target - self.q_values[self.current_state, action]
            self.q_values[self.current_state, action] += self.learning_rate * td_error
            
            # Store experience in replay buffer
            experience = (self.current_state, action, reward, next_state)
            self.replay_buffer.append(experience)
            
            if len(self.replay_buffer) > self.buffer_size:
                self.replay_buffer.pop(0)
            
            # Experience replay
            if len(self.replay_buffer) >= self.batch_size:
                batch = np.random.choice(len(self.replay_buffer), self.batch_size, replace=False)
                for idx in batch:
                    s, a, r, ns = self.replay_buffer[idx]
                    best_next = np.max(self.q_values[ns])
                    td_target = r + self.discount_factor * best_next
                    td_error = td_target - self.q_values[s, a]
                    self.q_values[s, a] += self.learning_rate * td_error
            
            # Update current state
            self.current_state = next_state
        
        def compute_reward(self, old_tour_lengths: List[float], new_tour_lengths: List[float]) -> float:
            """Compute reward based on improvement in tour quality."""
            if not old_tour_lengths or not new_tour_lengths:
                return 0
            
            old_best = np.min(old_tour_lengths)
            new_best = np.min(new_tour_lengths)
            
            if old_best == 0:
                return 0
            
            improvement = (old_best - new_best) / old_best
            reward = np.tanh(improvement * 10)  # Scale and bound reward
            
            # Additional reward for consistency
            old_std = np.std(old_tour_lengths) if len(old_tour_lengths) > 1 else 0
            new_std = np.std(new_tour_lengths) if len(new_tour_lengths) > 1 else 0
            
            if old_std > 0:
                consistency_improvement = (old_std - new_std) / old_std
                reward += 0.1 * np.tanh(consistency_improvement * 5)
            
            return np.clip(reward, -1, 1)
    
    # ==================== 4. SCALABLE MULTI-RESOLUTION SAMPLING ====================
    
    def hierarchical_multiresolution_sampling(nodes: np.ndarray, base_size: int, 
                                             n_levels: int = 3) -> List[np.ndarray]:
        """
        Multi-resolution sampling with adaptive granularity.
        Returns samples at different resolution levels.
        """
        samples = []
        
        if len(nodes) <= base_size:
            samples.append(nodes)
            for _ in range(1, n_levels):
                samples.append(nodes)
            return samples
        
        # Level 0: Coarsest resolution
        sample_size_l0 = base_size
        sample_l0 = _adaptive_sampling(nodes, sample_size_l0, resolution=0.3)
        samples.append(sample_l0)
        
        # Level 1: Medium resolution
        sample_size_l1 = min(base_size * 2, len(nodes))
        # Include all nodes from level 0 plus new ones
        remaining = np.array([n for n in nodes if n not in sample_l0])
        if len(remaining) > 0:
            n_new = sample_size_l1 - len(sample_l0)
            if n_new > 0:
                new_samples = _adaptive_sampling(remaining, n_new, resolution=0.6)
                sample_l1 = np.concatenate([sample_l0, new_samples])
            else:
                sample_l1 = sample_l0
        else:
            sample_l1 = sample_l0
        samples.append(sample_l1)
        
        # Level 2: Finest resolution (if needed)
        if n_levels > 2:
            sample_size_l2 = min(base_size * 3, len(nodes))
            remaining = np.array([n for n in nodes if n not in sample_l1])
            if len(remaining) > 0:
                n_new = sample_size_l2 - len(sample_l1)
                if n_new > 0:
                    new_samples = _adaptive_sampling(remaining, n_new, resolution=0.9)
                    sample_l2 = np.concatenate([sample_l1, new_samples])
                else:
                    sample_l2 = sample_l1
            else:
                sample_l2 = sample_l1
            samples.append(sample_l2)
        
        return samples
    
    def _adaptive_sampling(nodes: np.ndarray, target_size: int, resolution: float) -> np.ndarray:
        """Adaptive sampling based on resolution parameter."""
        if len(nodes) <= target_size:
            return nodes
        
        selected = []
        
        # Phase 1: Diversity sampling (higher weight for higher resolution)
        n_diverse = int(target_size * resolution)
        
        if n_diverse > 0:
            # Use farthest point sampling for diversity
            selected.append(np.random.choice(nodes))
            
            for _ in range(1, n_diverse):
                # Find node farthest from already selected
                max_min_dist = -1
                farthest_node = None
                
                for node in nodes:
                    if node not in selected:
                        min_dist = np.min([distance_matrix[node, s] for s in selected])
                        if min_dist > max_min_dist:
                            max_min_dist = min_dist
                            farthest_node = node
                
                if farthest_node is not None:
                    selected.append(farthest_node)
        
        # Phase 2: Structure-preserving sampling
        n_structural = target_size - len(selected)
        
        if n_structural > 0:
            remaining = [n for n in nodes if n not in selected]
            
            # Sample based on local structural importance
            structural_scores = np.zeros(len(remaining))
            for idx, node in enumerate(remaining):
                # Local clustering score
                k_neighbors = min(5, len(remaining))
                neighbor_dists = distance_matrix[node, remaining]
                neighbor_indices = np.argsort(neighbor_dists)[1:k_neighbors+1]
                
                if len(neighbor_indices) >= 2:
                    # Compute local density
                    local_dists = distance_matrix[node, np.array(remaining)[neighbor_indices]]
                    structural_scores[idx] = 1.0 / (np.mean(local_dists) + 1e-10)
                else:
                    structural_scores[idx] = 1.0
            
            if structural_scores.sum() > 0:
                probs = structural_scores / structural_scores.sum()
                n_select = min(n_structural, len(remaining))
                indices = np.random.choice(len(remaining), n_select, p=probs, replace=False)
                selected.extend([remaining[i] for i in indices])
            else:
                # Random sampling as fallback
                additional = np.random.choice(remaining, min(n_structural, len(remaining)), replace=False)
                selected.extend(additional)
        
        return np.array(selected)
    
    # ==================== 5. TRANSFORMER-INSPIRED EDGE ATTENTION ====================
    
    def compute_edge_attention(distance_matrix: np.ndarray, 
                              node_features: Optional[np.ndarray] = None) -> np.ndarray:
        """
        Compute edge attention scores using transformer-inspired mechanism.
        """
        n = distance_matrix.shape[0]
        
        # If node features not provided, create from distance statistics
        if node_features is None:
            node_features = np.zeros((n, 4))
            for i in range(n):
                dists = distance_matrix[i]
                node_features[i, 0] = np.mean(dists)
                node_features[i, 1] = np.std(dists)
                node_features[i, 2] = np.median(dists)
                node_features[i, 3] = np.sum(1.0 / (dists + 1e-10))
        
        # Normalize features
        feature_means = np.mean(node_features, axis=0, keepdims=True)
        feature_stds = np.std(node_features, axis=0, keepdims=True) + 1e-10
        node_features = (node_features - feature_means) / feature_stds
        
        # Simple attention mechanism
        attention_scores = np.zeros((n, n))
        
        # Learnable query, key projections (simulated)
        W_q = np.random.randn(node_features.shape[1], 4)
        W_k = np.random.randn(node_features.shape[1], 4)
        
        # Project features
        Q = node_features @ W_q  # [n, 4]
        K = node_features @ W_k  # [n, 4]
        
        # Compute attention scores
        for i in range(n):
            for j in range(n):
                if i != j:
                    # Dot-product attention
                    attention = np.dot(Q[i], K[j])
                    
                    # Add distance-based bias
                    dist_bias = -np.log1p(distance_matrix[i, j])
                    attention += dist_bias
                    
                    # Scale
                    attention /= np.sqrt(4)  # sqrt(d_k)
                    
                    attention_scores[i, j] = attention
        
        # Apply softmax
        np.fill_diagonal(attention_scores, -np.inf)
        exp_scores = np.exp(attention_scores - np.max(attention_scores, axis=1, keepdims=True))
        attention_scores = exp_scores / np.sum(exp_scores, axis=1, keepdims=True)
        
        # Symmetrize
        attention_scores = (attention_scores + attention_scores.T) / 2
        np.fill_diagonal(attention_scores, 0)
        
        return attention_scores
    
    # ==================== 6. HEURISTIC FACTOR COMPUTATION ====================
    
    # Define heuristic factors

    def compute_heuristic_factors(sample_nodes: np.ndarray = None) -> List[np.ndarray]:
        """Compute heuristic factors, optionally on sampled subgraph."""
        if sample_nodes is None:
            nodes = np.arange(n)
            sub_n = n
            sub_dist = distance_matrix
        else:
            nodes = sample_nodes
            sub_n = len(nodes)
            sub_dist = distance_matrix[np.ix_(nodes, nodes)]
        
        factors = []
        
        # Factor 1: Inverse distance with adaptive scaling
        with np.errstate(divide='ignore', invalid='ignore'):
            inv_dist = 1.0 / sub_dist
        np.fill_diagonal(inv_dist, 0)
        
        # Adaptive scaling based on distance distribution
        power = 0.8 + 0.4 * (1 - instance_features['uniformity'])
        inv_dist = np.power(inv_dist, power)
        
        row_sums = inv_dist.sum(axis=1, keepdims=True)
        row_sums[row_sums == 0] = 1
        inv_dist_norm = inv_dist / row_sums
        
        if sample_nodes is not None:
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = inv_dist_norm
            factors.append(full_factor)
        else:
            factors.append(inv_dist_norm)
        
        # Factor 2: MST-based connectivity
        mst_factor = np.zeros((sub_n, sub_n))
        try:
            if sub_n <= 1000:
                mst = minimum_spanning_tree(csr_matrix(sub_dist)).toarray()
                mst_mask = (mst > 0) | (mst.T > 0)
                mst_factor[mst_mask] = 1.0
            else:
                # Approximate MST for large instances
                for i in range(sub_n):
                    neighbors = np.argsort(sub_dist[i])[1:min(4, sub_n//100)]
                    mst_factor[i, neighbors] = 0.8
        except:
            pass
        
        mst_factor = np.maximum(mst_factor, mst_factor.T)
        
        if sample_nodes is not None:
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = mst_factor
            factors.append(full_factor)
        else:
            factors.append(mst_factor)
        
        # Factor 3: Triangle completion
        triangle_factor = np.zeros((sub_n, sub_n))
        if sub_n >= 4:
            # Find promising triangles
            threshold = np.percentile(sub_dist, 40)
            short_edges = sub_dist <= threshold
            np.fill_diagonal(short_edges, False)
            
            # Count common short neighbors
            common_short = short_edges @ short_edges.T
            triangle_factor = common_short.astype(float) / (sub_n - 2)
        
        if sample_nodes is not None:
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = triangle_factor
            factors.append(full_factor)
        else:
            factors.append(triangle_factor)
        
        # Factor 4: Edge attention scores
        attention_factor = compute_edge_attention(sub_dist)
        
        if sample_nodes is not None:
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = attention_factor
            factors.append(full_factor)
        else:
            factors.append(attention_factor)
        
        # Factor 5: Deep prediction
        deep_factor = deep_predictor.predict_edge_matrix()
        if sample_nodes is not None:
            # Extract relevant portion
            deep_factor_sub = deep_factor[np.ix_(nodes, nodes)]
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = deep_factor_sub
            factors.append(full_factor)
        else:
            factors.append(deep_factor)
        
        # Factor 6: Local search evidence (initialize as zeros, will be updated)
        evidence_factor = np.zeros((sub_n, sub_n))
        
        if sample_nodes is not None:
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = evidence_factor
            factors.append(full_factor)
        else:
            factors.append(evidence_factor)
        
        return factors
    
    # ==================== 7. ADAPTIVE LOCAL SEARCH WITH META-LEARNING ====================
    
    def adaptive_local_search(tours: List[np.ndarray], 
                            n_iterations: int = 3) -> Tuple[List[np.ndarray], List[float]]:
        """Apply adaptive local search with meta-learning."""
        improved_tours = []
        tour_lengths = []
        
        for tour in tours:
            current_tour = tour.copy()
            current_len = _tour_length(current_tour)
            
            # Adaptive local search based on instance characteristics
            if instance_features['uniformity'] > 0.7:
                # For uniform instances, focus on 3-opt
                methods = ['3opt', '2opt', 'or_opt']
                weights = [0.5, 0.3, 0.2]
            else:
                # For non-uniform instances, focus on 2-opt
                methods = ['2opt', '3opt', 'double_bridge']
                weights = [0.6, 0.2, 0.2]
            
            for _ in range(n_iterations):
                method = np.random.choice(methods, p=weights)
                new_tour = _apply_local_search(current_tour, method)
                new_len = _tour_length(new_tour)
                
                if new_len < current_len - 1e-10:
                    current_tour = new_tour
                    current_len = new_len
            
            improved_tours.append(current_tour)
            tour_lengths.append(current_len)
        
        return improved_tours, tour_lengths
    
    def _tour_length(tour: np.ndarray) -> float:
        """Calculate tour length."""
        n_nodes = len(tour)
        return sum(distance_matrix[tour[i], tour[(i+1)%n_nodes]] for i in range(n_nodes))
    
    def _apply_local_search(tour: np.ndarray, method: str) -> np.ndarray:
        """Apply specific local search method."""
        n_nodes = len(tour)
        
        if method == '2opt':
            best_tour = tour.copy()
            best_len = _tour_length(tour)
            
            for i in range(n_nodes-1):
                for j in range(i+2, n_nodes):
                    if j == n_nodes-1 and i == 0:
                        continue
                    
                    # Check if swap improves
                    a, b = tour[i], tour[(i+1)%n_nodes]
                    c, d = tour[j], tour[(j+1)%n_nodes]
                    
                    old = distance_matrix[a, b] + distance_matrix[c, d]
                    new = distance_matrix[a, c] + distance_matrix[b, d]
                    
                    if new < old - 1e-10:
                        # Perform swap
                        new_tour = tour.copy()
                        new_tour[i+1:j+1] = tour[i+1:j+1][::-1]
                        new_len = best_len - old + new
                        
                        if new_len < best_len:
                            best_tour = new_tour
                            best_len = new_len
            
            return best_tour
        
        elif method == '3opt':
            # Simplified 3-opt
            best_tour = tour.copy()
            best_len = _tour_length(tour)
            
            # Limited search for efficiency
            for _ in range(min(50, n_nodes)):
                i, j, k = sorted(np.random.choice(n_nodes, 3, replace=False))
                
                # Try different reconnections
                segments = [
                    tour[:i+1], tour[i+1:j+1], tour[j+1:k+1], tour[k+1:]
                ]
                
                # Try 2-opt swaps between segments
                for perm in [(0, 2, 1, 3), (0, 3, 2, 1), (0, 3, 1, 2)]:
                    new_tour = np.concatenate([segments[perm[0]], segments[perm[1]], 
                                             segments[perm[2]], segments[perm[3]]])
                    new_len = _tour_length(new_tour)
                    
                    if new_len < best_len:
                        best_tour = new_tour
                        best_len = new_len
            
            return best_tour
        
        elif method == 'or_opt':
            # OR-opt: relocate a segment
            best_tour = tour.copy()
            best_len = _tour_length(tour)
            
            for length in [1, 2, 3]:
                for start in range(n_nodes):
                    end = (start + length) % n_nodes
                    
                    if start < end:
                        segment = tour[start:end]
                        remaining = np.concatenate([tour[:start], tour[end:]])
                    else:
                        segment = np.concatenate([tour[start:], tour[:end]])
                        remaining = tour[end:start]
                    
                    # Try all insertion positions
                    for insert_pos in range(len(remaining)+1):
                        new_tour = np.concatenate([
                            remaining[:insert_pos],
                            segment,
                            remaining[insert_pos:]
                        ])
                        new_len = _tour_length(new_tour)
                        
                        if new_len < best_len:
                            best_tour = new_tour
                            best_len = new_len
            
            return best_tour
        
        else:  # double_bridge
            if n_nodes >= 8:
                i, j, k, l = sorted(np.random.choice(n_nodes, 4, replace=False))
                new_tour = np.concatenate([
                    tour[:i+1],
                    tour[k+1:l+1],
                    tour[j+1:k+1],
                    tour[i+1:j+1],
                    tour[l+1:]
                ])
                return new_tour
            else:
                return tour
    
    # ==================== 8. MAIN ALGORITHM LOOP ====================
    
    # Determine computational strategy based on instance size
    if n <= 100:
        compute_strategy = "full"
        n_tours = min(30, n * 2)
        sample_size = n
    elif n <= 1000:
        compute_strategy = "balanced"
        n_tours = min(20, n // 5)
        sample_size = min(200, n // 2)
    else:
        compute_strategy = "sampled"
        n_tours = min(15, n // 10)
        sample_size = min(100, n // 4)
    
    # Initialize weights
    weights = np.ones(n_factors) / n_factors
    
    # Main learning loop
    for iteration in range(3):  # 3 meta-learning iterations
        # Generate initial tours using different methods
        tours = []
        for _ in range(n_tours):
            method = np.random.choice(['nearest_neighbor', 'random', 'insertion'])
            if method == 'nearest_neighbor':
                start = np.random.randint(0, n)
                tour = [start]
                unvisited = set(range(n)) - {start}
                while unvisited:
                    current = tour[-1]
                    nearest = min(unvisited, key=lambda x: distance_matrix[current, x])
                    tour.append(nearest)
                    unvisited.remove(nearest)
                tours.append(np.array(tour))
            elif method == 'random':
                tours.append(np.random.permutation(n))
            else:  # insertion
                if n <= 3:
                    tours.append(np.random.permutation(n))
                else:
                    nodes = list(np.random.choice(n, 3, replace=False))
                    tour = nodes
                    remaining = set(range(n)) - set(tour)
                    while remaining:
                        best_node = None
                        best_pos = -1
                        best_increase = float('inf')
                        for node in remaining:
                            for i in range(len(tour)):
                                j = (i + 1) % len(tour)
                                inc = (distance_matrix[tour[i], node] + 
                                       distance_matrix[node, tour[j]] - 
                                       distance_matrix[tour[i], tour[j]])
                                if inc < best_increase:
                                    best_increase = inc
                                    best_node = node
                                    best_pos = i + 1
                        tour.insert(best_pos, best_node)
                        remaining.remove(best_node)
                    tours.append(np.array(tour))
        
        # Apply local search
        old_tour_lengths = [_tour_length(t) for t in tours]
        tours, new_tour_lengths = adaptive_local_search(tours)
        
        # Compute reward and update meta-learner
        reward = meta_learner.compute_reward(old_tour_lengths, new_tour_lengths)
        action, new_weights = meta_learner.select_action(weights)
        meta_learner.update(action, reward, instance_features)
        
        # Update weights with momentum
        if iteration > 0:
            weights = meta_learner.momentum * weights + (1 - meta_learner.momentum) * new_weights
        else:
            weights = new_weights
        
        # Update deep predictor with elite tours
        elite_indices = np.argsort(new_tour_lengths)[:max(3, len(tours)//4)]
        for idx in elite_indices:
            tour = tours[idx]
            tour_len = new_tour_lengths[idx]
            
            # Update deep predictor for edges in elite tour
            for i in range(n):
                j = (i + 1) % n
                u, v = tour[i], tour[j]
                
                # Extract features and update
                features = deep_predictor.extract_edge_features(u, v)
                target = 1.0 / (1.0 + tour_len / instance_features['dist_mean'])
                importance = 1.0 / (1 + idx)  # Higher importance for better tours
                
                deep_predictor.update_weights(features, target, importance)
    
    # ==================== 9. FINAL HEURISTIC COMPUTATION ====================
    
    # Compute heuristic factors
    if compute_strategy == "full":
        factors = compute_heuristic_factors()
    else:
        # Multi-resolution sampling
        sample_sets = hierarchical_multiresolution_sampling(
            np.arange(n), 
            sample_size, 
            n_levels=2 if compute_strategy == "balanced" else 3
        )
        
        # Compute factors at different resolutions and combine
        all_factors = []
        for sample_nodes in sample_sets:
            sample_factors = compute_heuristic_factors(sample_nodes)
            all_factors.append(sample_factors)
        
        # Combine multi-resolution factors
        factors = []
        for factor_idx in range(n_factors):
            combined = np.zeros((n, n))
            for resolution_idx, sample_factors in enumerate(all_factors):
                weight = 1.0 / (1 + resolution_idx)  # Higher weight for coarser resolutions
                combined += weight * sample_factors[factor_idx]
            factors.append(combined / len(sample_sets))
    
    # Combine factors with learned weights
    heuristic = np.zeros((n, n))
    for w, factor in zip(weights, factors):
        # Normalize factor
        f_min, f_max = factor.min(), factor.max()
        if f_max - f_min > 1e-10:
            factor_norm = (factor - f_min) / (f_max - f_min)
        else:
            factor_norm = factor
        
        heuristic += w * factor_norm
    
    # Ensure MST connectivity
    try:
        if n <= 10000:
            mst = minimum_spanning_tree(csr_matrix(distance_matrix)).toarray()
            mst_mask = (mst > 0) | (mst.T > 0)
            heuristic[mst_mask] *= 1.3
    except:
        pass
    
    # Adaptive sparsification
    if n > 10:
        edges_per_node = max(3, min(10, int(4 + np.log2(n) * (1 - instance_features['uniformity']))))
        
        sparsity_mask = np.zeros_like(heuristic, dtype=bool)
        for i in range(n):
            node_scores = heuristic[i].copy()
            node_scores[i] = -np.inf
            
            n_select = min(edges_per_node, n-1)
            top_indices = np.argsort(node_scores)[-n_select:]
            sparsity_mask[i, top_indices] = True
        
        sparsity_mask = sparsity_mask | sparsity_mask.T
        heuristic = heuristic * sparsity_mask
    
    # Final processing
    heuristic = (heuristic + heuristic.T) / 2
    np.fill_diagonal(heuristic, 0)
    
    # Adaptive scaling
    h_min, h_max = heuristic.min(), heuristic.max()
    if h_max - h_min > 1e-10:
        heuristic = (heuristic - h_min) / (h_max - h_min)
    
    # Apply sigmoid for better contrast
    alpha = 3.0 + 2.0 * instance_features['uniformity']
    heuristic = 1 / (1 + np.exp(-alpha * (heuristic - 0.5)))
    
    # Final normalization
    h_min, h_max = heuristic.min(), heuristic.max()
    if h_max - h_min > 1e-10:
        heuristic = (heuristic - h_min) / (h_max - h_min)
    
    # Soft threshold
    threshold = 0.01 + 0.04 * (1 - instance_features['uniformity'])
    heuristic[heuristic < threshold] = 0
    
    return heuristic
