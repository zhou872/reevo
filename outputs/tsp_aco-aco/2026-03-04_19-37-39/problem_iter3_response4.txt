import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with improved centrality 
    calculation and sparsification strategy.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Inverse distance component
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Adaptive nearest neighbor emphasis
    k = max(2, min(5, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nn_indices = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, nn_indices] = True
    nn_mask = nn_mask | nn_mask.T
    
    # 3. Minimum spanning tree component
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # 4. Improved centrality measure (vectorized)
    # Remove self-distances and calculate centrality as inverse average distance
    row_sums = np.sum(distance_matrix, axis=1) - np.diag(distance_matrix)
    centrality = 1 / (row_sums / (n - 1) + 1e-10)
    centrality = centrality / (np.sum(centrality) + 1e-10)
    
    # 5. Edge betweenness approximation using centrality product
    edge_potential = np.outer(centrality, centrality)
    np.fill_diagonal(edge_potential, 0)
    
    # 6. Combine factors with refined weights
    heuristic_scores = np.zeros_like(distance_matrix)
    
    # Base component: inverse distance with reduced weight
    heuristic_scores = 0.5 * inv_dist
    
    # Boost nearest neighbor edges
    heuristic_scores[nn_mask] *= 1.3
    
    # Strong boost for MST edges (connectivity guarantee)
    heuristic_scores[mst_mask] *= 2.0
    
    # Add centrality-based component
    heuristic_scores += 0.4 * edge_potential
    
    # 7. Multi-criteria sparsification with minimum connectivity
    # Keep edges that rank high by multiple criteria
    sparsity_mask = np.zeros_like(heuristic_scores, dtype=bool)
    
    # Determine edge ranks by different criteria
    inv_dist_ranks = np.argsort(inv_dist, axis=1)[:, ::-1]
    combined_ranks = np.argsort(heuristic_scores, axis=1)[:, ::-1]
    
    # Percentage thresholds for keeping edges
    inv_dist_keep = max(2, int(0.15 * n))  # Keep top 15% by inverse distance
    combined_keep = max(3, int(0.25 * n))  # Keep top 25% by combined score
    
    for i in range(n):
        # Keep top edges by inverse distance
        top_by_inv_dist = inv_dist_ranks[i, :inv_dist_keep]
        sparsity_mask[i, top_by_inv_dist] = True
        
        # Keep top edges by combined score
        top_by_combined = combined_ranks[i, :combined_keep]
        sparsity_mask[i, top_by_combined] = True
        
        # Always keep MST edges (already in heuristic_scores)
        sparsity_mask[i, mst_mask[i]] = True
        
        # Ensure at least 2 connections per node for feasibility
        current_connections = np.sum(sparsity_mask[i])
        if current_connections < 2:
            # Add next best edges from combined ranking
            needed = 2 - current_connections
            candidates = combined_ranks[i, combined_keep:combined_keep + needed]
            sparsity_mask[i, candidates] = True
    
    # Make mask symmetric
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # 8. Apply sparsification
    heuristic_scores = heuristic_scores * sparsity_mask
    
    # 9. Final normalization
    max_val = np.max(heuristic_scores)
    if max_val > 0:
        heuristic_scores = heuristic_scores / max_val
    
    return heuristic_scores
