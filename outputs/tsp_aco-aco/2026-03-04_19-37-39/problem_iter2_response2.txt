import numpy as np
from scipy.spatial.distance import cdist
from typing import Tuple, Optional

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP combining multiple strategies with
    vectorized operations and adaptive parameterization.
    
    Key improvements:
    1. Vectorized mutual nearest neighbors computation
    2. Adaptive parameters based on problem size and structure
    3. Ensemble of complementary heuristic strategies
    4. Reinforcement learning-inspired edge weights
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        return 1.0 / (distance_matrix + np.eye(n) * 1e-12)
    
    # ========== 1. ADAPTIVE PARAMETERIZATION ==========
    # Parameters adapt to problem size and structure
    k_neighbors = max(2, min(10, int(np.sqrt(n))))  # Nearest neighbors to consider
    mst_weight = 0.3 + 0.2 * np.log1p(n) / 10  # MST importance grows with n
    sparsity_factor = min(0.4, 2.5 * k_neighbors / n)  # Percentage of edges to keep
    
    # ========== 2. VECTORIZED BASE HEURISTICS ==========
    # Base inverse distance with safety
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + np.eye(n) * 1e-12)
        inv_dist[np.isinf(inv_dist)] = 0
    
    # ========== 3. VECTORIZED MUTUAL NEAREST NEIGHBORS ==========
    # Efficient computation using argpartition
    mutual_nn = np.zeros((n, n))
    for i in range(n):
        # Get k nearest neighbors for node i
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        nearest_i = np.argpartition(dist_i, k_neighbors)[:k_neighbors]
        
        # For each neighbor, check if relationship is mutual
        for j in nearest_i:
            dist_j = distance_matrix[j].copy()
            dist_j[j] = np.inf
            nearest_j = np.argpartition(dist_j, k_neighbors)[:k_neighbors]
            if i in nearest_j:
                mutual_nn[i, j] = 1
    
    mutual_nn = np.maximum(mutual_nn, mutual_nn.T)  # Ensure symmetry
    
    # ========== 4. EFFICIENT MST COMPUTATION ==========
    # Using Prim's algorithm with early stopping for large n
    mst_edges = np.zeros((n, n))
    if n <= 1000:  # Full MST for moderate sizes
        visited = np.zeros(n, dtype=bool)
        visited[0] = True
        for _ in range(n - 1):
            # Vectorized min edge search
            visited_mask = visited[:, None] & (~visited[None, :])
            if not visited_mask.any():
                break
            min_val = np.inf
            min_i, min_j = -1, -1
            for i in np.where(visited)[0]:
                for j in np.where(~visited)[0]:
                    if distance_matrix[i, j] < min_val:
                        min_val = distance_matrix[i, j]
                        min_i, min_j = i, j
            if min_i >= 0 and min_j >= 0:
                mst_edges[min_i, min_j] = mst_edges[min_j, min_i] = 1
                visited[min_j] = True
    else:  # Approximate MST for large instances
        # Use nearest neighbor chain
        for i in range(n):
            dist_i = distance_matrix[i].copy()
            dist_i[i] = np.inf
            nearest = np.argmin(dist_i)
            mst_edges[i, nearest] = mst_edges[nearest, i] = 1
    
    # ========== 5. ENSEMBLE HEURISTIC STRATEGIES ==========
    heuristic_components = []
    weights = []
    
    # Strategy 1: Inverse distance (classical)
    heuristic_components.append(inv_dist)
    weights.append(0.3)
    
    # Strategy 2: Mutual nearest neighbors (connectivity)
    heuristic_components.append(mutual_nn * inv_dist)
    weights.append(0.25)
    
    # Strategy 3: MST-based (structural importance)
    heuristic_components.append(mst_edges * inv_dist * mst_weight)
    weights.append(0.25)
    
    # Strategy 4: Geometric betweenness (anti-bypass)
    if n > 3:
        # Sample-based geometric penalty
        sample_size = min(15, n // 2)
        sample_indices = np.random.choice(n, sample_size, replace=False)
        geometric_factor = np.ones((n, n))
        
        for i in sample_indices:
            for j in range(n):
                if i == j:
                    continue
                d_ij = distance_matrix[i, j]
                # Check for bypassing nodes
                mask = (distance_matrix[i] < d_ij) & (distance_matrix[j] < d_ij)
                mask[i] = mask[j] = False
                if mask.any():
                    # Penalize edges that can be bypassed
                    geometric_factor[i, j] *= 0.85
        
        heuristic_components.append(geometric_factor * inv_dist)
        weights.append(0.2)
    
    # Normalize weights
    weights = np.array(weights) / sum(weights)
    
    # ========== 6. WEIGHTED ENSEMBLE COMBINATION ==========
    heuristic_matrix = np.zeros((n, n))
    for comp, weight in zip(heuristic_components, weights):
        heuristic_matrix += weight * comp
    
    # ========== 7. ADAPTIVE SPARSIFICATION ==========
    # Keep only the most promising edges for each node
    edges_per_node = max(2, int(np.sqrt(n)))
    
    # Create sparsification mask
    mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        row_heuristics = heuristic_matrix[i].copy()
        row_heuristics[i] = -np.inf
        # Get indices of top edges_per_node neighbors
        top_indices = np.argpartition(row_heuristics, -edges_per_node)[-edges_per_node:]
        mask[i, top_indices] = True
    
    # Ensure symmetry
    mask = mask | mask.T
    
    # Apply mask
    sparsified = heuristic_matrix * mask
    
    # ========== 8. REINFORCEMENT LEARNING INSPIRED ADJUSTMENT ==========
    # Boost edges that appear in multiple heuristic strategies
    consensus_mask = (mutual_nn > 0) & (mst_edges > 0)
    if consensus_mask.any():
        sparsified[consensus_mask] *= 1.5  # Consensus boost
    
    # ========== 9. FINAL PROCESSING ==========
    # Normalize to [0, 1] range
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(sparsified, 0)
    
    # Apply final sparsity threshold
    threshold = np.percentile(sparsified[sparsified > 0], 100 * (1 - sparsity_factor))
    sparsified[sparsified < threshold] = 0
    
    return sparsified
