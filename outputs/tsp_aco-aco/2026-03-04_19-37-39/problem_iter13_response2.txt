import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with improved adaptive weighting,
    refined probabilistic edge betweenness, and smarter sparsification.
    
    Key improvements over v1:
    1. More sophisticated adaptive weighting that better balances components
    2. Enhanced random walk with teleportation for better betweenness estimation
    3. Refined sparsification with dynamic k-nearest neighbors
    4. Improved tour generation with randomized construction heuristics
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    cv = dist_std / (dist_mean + 1e-10)
    
    # Size-dependent parameters
    log_n = np.log(n + 1)
    size_factor = min(1.0, n / 100.0)  # Normalize by 100 nodes
    
    # ===== 2. IMPROVED SOLUTION SAMPLING =====
    def generate_diverse_tour() -> np.ndarray:
        """Generate tour using randomized construction heuristics."""
        method = np.random.choice(['nn', 'greedy', 'random', 'farthest'], 
                                 p=[0.35, 0.3, 0.2, 0.15])
        
        if method == 'nn':  # Randomized nearest neighbor
            start = np.random.randint(0, n)
            tour = [start]
            unvisited = set(range(n)) - {start}
            
            while unvisited:
                current = tour[-1]
                # Consider top 3 nearest with probability
                distances_to_unvisited = [(v, distance_matrix[current, v]) 
                                        for v in unvisited]
                distances_to_unvisited.sort(key=lambda x: x[1])
                
                if len(distances_to_unvisited) >= 3:
                    candidates = distances_to_unvisited[:3]
                    probs = np.array([1.0/(d+1e-10) for _, d in candidates])
                    probs = probs / probs.sum()
                    selected_idx = np.random.choice(3, p=probs)
                    next_node = candidates[selected_idx][0]
                else:
                    next_node = distances_to_unvisited[0][0]
                
                tour.append(next_node)
                unvisited.remove(next_node)
            
            return np.array(tour)
        
        elif method == 'greedy':  # Randomized greedy insertion
            if n <= 3:
                return np.random.permutation(n)
            
            # Start with random triangle
            indices = np.random.choice(n, 3, replace=False)
            tour = list(indices[:3])
            tour.append(tour[0])
            remaining = set(range(n)) - set(tour[:-1])
            
            while remaining:
                # Randomly select candidate nodes
                candidates = list(remaining)
                if len(candidates) > 5:
                    candidates = np.random.choice(candidates, 5, replace=False)
                
                best_cost = float('inf')
                best_node = None
                best_pos = -1
                
                for node in candidates:
                    for i in range(len(tour)-1):
                        cost_inc = (distance_matrix[tour[i], node] + 
                                   distance_matrix[node, tour[i+1]] - 
                                   distance_matrix[tour[i], tour[i+1]])
                        # Add small noise for randomization
                        cost_inc *= np.random.uniform(0.95, 1.05)
                        if cost_inc < best_cost:
                            best_cost = cost_inc
                            best_node = node
                            best_pos = i + 1
                
                tour.insert(best_pos, best_node)
                remaining.remove(best_node)
            
            return np.array(tour[:-1])
        
        elif method == 'farthest':  # Farthest insertion
            start = np.random.randint(0, n)
            tour = [start]
            unvisited = set(range(n)) - {start}
            
            while unvisited:
                # Find farthest node from tour
                farthest = max(unvisited, 
                             key=lambda x: min(distance_matrix[x, t] for t in tour))
                
                # Find best insertion position
                best_cost = float('inf')
                best_pos = -1
                
                for i in range(len(tour)):
                    if i == len(tour) - 1:
                        cost_inc = (distance_matrix[tour[i], farthest] + 
                                   distance_matrix[farthest, tour[0]] - 
                                   distance_matrix[tour[i], tour[0]])
                    else:
                        cost_inc = (distance_matrix[tour[i], farthest] + 
                                   distance_matrix[farthest, tour[i+1]] - 
                                   distance_matrix[tour[i], tour[i+1]])
                    
                    if cost_inc < best_cost:
                        best_cost = cost_inc
                        best_pos = i + 1
                
                tour.insert(best_pos, farthest)
                unvisited.remove(farthest)
            
            return np.array(tour)
        
        else:  # Random permutation
            return np.random.permutation(n)
    
    def fast_two_opt(tour: np.ndarray, iterations: int = 20) -> np.ndarray:
        """Optimized 2-opt with limited iterations."""
        current = tour.copy()
        n = len(current)
        
        for _ in range(iterations):
            improved = False
            
            # Random sampling of edges to check
            for _ in range(min(100, n*n//4)):
                i = np.random.randint(0, n)
                j = np.random.randint(0, n)
                
                if abs(i - j) < 2 or (i == 0 and j == n-1) or (j == 0 and i == n-1):
                    continue
                
                if i > j:
                    i, j = j, i
                
                a, b = current[i], current[(i+1)%n]
                c, d = current[j], current[(j+1)%n]
                
                old_len = distance_matrix[a, b] + distance_matrix[c, d]
                new_len = distance_matrix[a, c] + distance_matrix[b, d]
                
                if new_len < old_len * 0.99:  # Slight threshold to avoid minor improvements
                    # Reverse segment i+1 to j
                    current[i+1:j+1] = current[i+1:j+1][::-1]
                    improved = True
            
            if not improved:
                break
        
        return current
    
    # Generate diverse tours with adaptive count
    num_tours = min(15, max(5, int(n * 0.3)))
    tours = []
    
    for _ in range(num_tours):
        tour = generate_diverse_tour()
        # Apply 2-opt with probability based on tour quality
        tour_length = sum(distance_matrix[tour[i], tour[(i+1)%n]] for i in range(n))
        avg_edge_length = tour_length / n
        
        # More improvement for worse tours
        if avg_edge_length > dist_mean * 1.2:
            tour = fast_two_opt(tour, iterations=int(25 * size_factor))
        else:
            tour = fast_two_opt(tour, iterations=int(10 * size_factor))
        
        tours.append(tour)
    
    # Edge frequency from tours
    edge_freq = np.zeros_like(distance_matrix)
    for tour in tours:
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            edge_freq[u, v] += 1
            edge_freq[v, u] += 1
    
    if edge_freq.max() > 0:
        edge_freq = edge_freq / edge_freq.max()
    
    # ===== 3. ENHANCED PROBABILISTIC EDGE BETWEENNESS =====
    edge_betweenness = np.zeros_like(distance_matrix)
    
    # Adaptive number of random paths
    num_paths = min(250, max(60, int(n * 2.5)))
    
    for _ in range(num_paths):
        s, t = np.random.choice(n, 2, replace=False)
        
        # Enhanced random walk with teleportation
        current = s
        path = [s]
        teleport_prob = 0.05  # Small chance to teleport to random node
        
        steps = 0
        max_steps = min(3 * n, 100)  # Limit steps
        
        while current != t and steps < max_steps:
            steps += 1
            
            # Teleport with small probability
            if np.random.random() < teleport_prob:
                current = np.random.randint(0, n)
                path.append(current)
                continue
            
            # Get all possible next nodes (except current)
            neighbors = list(range(n))
            neighbors.remove(current)
            
            # Score neighbors based on multiple factors
            scores = []
            for v in neighbors:
                # 1. Edge quality (shorter is better)
                edge_quality = 1.0 / (distance_matrix[current, v] + 1e-10)
                
                # 2. Progress toward target
                progress = 1.0 / (distance_matrix[v, t] + 1e-10)
                
                # 3. Avoid going back (penalize recently visited)
                recency_penalty = 1.0
                if v in path[-min(5, len(path)):]:
                    recency_penalty = 0.3
                
                # 4. Explore new areas (nodes far from path)
                if len(path) > 3:
                    avg_dist_to_path = np.mean([distance_matrix[v, p] for p in path[-3:]])
                    exploration = 1.0 / (avg_dist_to_path + 1e-10)
                else:
                    exploration = 1.0
                
                total_score = (edge_quality * 1.5 + 
                             progress * 1.0 + 
                             exploration * 0.5) * recency_penalty
                scores.append(total_score)
            
            scores = np.array(scores)
            
            # Softmax selection
            if scores.sum() > 0:
                exp_scores = np.exp(scores - np.max(scores))  # Numerical stability
                probs = exp_scores / exp_scores.sum()
                next_node = np.random.choice(neighbors, p=probs)
            else:
                next_node = np.random.choice(neighbors)
            
            # Update edge betweenness
            edge_betweenness[current, next_node] += 1
            edge_betweenness[next_node, current] += 1
            
            current = next_node
            path.append(current)
    
    # Normalize and apply logarithmic scaling to emphasize differences
    if edge_betweenness.max() > 0:
        edge_betweenness = edge_betweenness / edge_betweenness.max()
        # Apply soft amplification
        edge_betweenness = np.log1p(edge_betweenness * 10) / np.log1p(10)
    
    # ===== 4. IMPROVED ADAPTIVE COMPONENT WEIGHTING =====
    # Compute additional statistics
    freq_nonzero = edge_freq[edge_freq > 0]
    freq_variation = np.std(freq_nonzero) / (np.mean(freq_nonzero) + 1e-10) if len(freq_nonzero) > 0 else 0
    
    # Base weights that adapt to instance size and characteristics
    if n < 15:
        # Small instances: rely more on distance and MST
        w_freq_base = 0.2
        w_bet_base = 0.2
        w_dist_base = 0.4
        w_mst_base = 0.2
    elif n < 50:
        # Medium instances: balanced approach
        w_freq_base = 0.3
        w_bet_base = 0.25
        w_dist_base = 0.25
        w_mst_base = 0.2
    else:
        # Large instances: rely more on sampling-based methods
        w_freq_base = 0.35
        w_bet_base = 0.3
        w_dist_base = 0.15
        w_mst_base = 0.2
    
    # Adjust based on instance characteristics
    # More uniform distances -> less weight on raw distance
    dist_weight_factor = max(0.5, 1.0 - cv)
    
    # High frequency variation -> trust frequency more
    freq_weight_factor = min(1.5, 1.0 + freq_variation)
    
    # Final weights
    w_freq = w_freq_base * freq_weight_factor
    w_bet = w_bet_base * (1.0 + 0.5 * size_factor)  # Betweenness more important for larger instances
    w_dist = w_dist_base * dist_weight_factor
    w_mst = w_mst_base * (0.5 + 0.5 * size_factor)  # MST more important for larger instances
    
    # Normalize
    total = w_freq + w_bet + w_dist + w_mst
    w_freq /= total
    w_bet /= total
    w_dist /= total
    w_mst /= total
    
    # ===== 5. COMPONENT COMPUTATION =====
    # Distance component with adaptive scaling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # Apply distance-based scaling
    dist_quantiles = np.percentile(distance_matrix[np.triu_indices(n, k=1)], [25, 50, 75])
    dist_scale = dist_quantiles[2] / (dist_quantiles[0] + 1e-10)
    
    # Enhance contrast for non-uniform distances
    if dist_scale > 3:
        inv_dist = np.power(inv_dist, 1.2)
    
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # MST component with edge importance
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # Compute MST edge scores considering both distance and centrality
    mst_scores = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if mst_mask[i, j]:
                # Score based on inverse distance and edge betweenness
                dist_score = 1.0 / (distance_matrix[i, j] + 1e-10)
                bet_score = edge_betweenness[i, j] if edge_betweenness.max() > 0 else 0.5
                mst_scores[i, j] = dist_score * 0.7 + bet_score * 0.3
                mst_scores[j, i] = mst_scores[i, j]
    
    if mst_scores.max() > 0:
        mst_scores = mst_scores / mst_scores.max()
    
    # ===== 6. COMBINE COMPONENTS =====
    heuristics_matrix = (
        w_freq * edge_freq +
        w_bet * edge_betweenness +
        w_dist * inv_dist +
        w_mst * mst_scores
    )
    
    # ===== 7. ENHANCED INTELLIGENT SPARSIFICATION =====
    # Determine k-nearest neighbors for each node
    k_base = max(2, int(1.2 * log_n))
    k_max = min(n-1, int(k_base * (1.5 + 0.5 * size_factor)))
    
    # Dynamic k based on distance distribution
    if cv > 0.5:  # High variation in distances
        k = min(k_max, max(k_base, int(n * 0.15)))
    else:  # Uniform distances
        k = min(k_max, max(k_base, int(n * 0.25)))
    
    # Build k-NN mask
    knn_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    for i in range(n):
        # Get distances to all other nodes
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Get k nearest neighbors
        nn_indices = np.argsort(distances)[:k]
        knn_mask[i, nn_indices] = True
    
    # Symmetrize k-NN mask
    knn_mask = knn_mask | knn_mask.T
    
    # Top heuristic edges per node
    top_edge_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    edges_per_node = min(n-1, int(k * (1.2 + 0.3 * size_factor)))
    
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        # Keep top edges
        top_indices = np.argsort(scores)[-edges_per_node:]
        top_edge_mask[i, top_indices] = True
    
    # Combined mask: keep k-NN edges, MST edges, and top heuristic edges
    combined_mask = knn_mask | mst_mask | top_edge_mask
    
    # Ensure symmetry
    combined_mask = combined_mask | combined_mask.T
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * combined_mask
    
    # ===== 8. FINAL ENHANCEMENTS =====
    # Ensure symmetry
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Boost edges that appear in multiple components
    component_mask = (edge_freq > 0).astype(int) + \
                    (edge_betweenness > 0.3).astype(int) + \
                    (inv_dist > np.percentile(inv_dist[inv_dist>0], 70)).astype(int) + \
                    (mst_scores > 0).astype(int)
    
    boost_factor = 1.0 + 0.15 * (component_mask - 2).clip(min=0)
    heuristics_matrix *= boost_factor
    
    # Final normalization
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        if non_zero.max() > non_zero.min():
            # Apply softmax-like normalization
            heuristics_matrix[heuristics_matrix > 0] = (
                np.tanh((non_zero - non_zero.mean()) / (non_zero.std() + 1e-10)) * 0.5 + 0.5
            )
    
    return heuristics_matrix
