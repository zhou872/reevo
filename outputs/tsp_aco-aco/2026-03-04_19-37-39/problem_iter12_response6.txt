import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List, Tuple
import math

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with ensemble optimization, feature-based 
    weight adaptation, and efficient sparse graph construction.
    
    Key improvements:
    1. Multi-method tour ensemble with parallelization readiness
    2. Feature-based adaptive component weighting using instance characteristics
    3. Hybrid edge scoring with geometric neighborhood analysis
    4. Optimized sparse graph construction preserving solution diversity
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE FEATURE EXTRACTION =====
    # Extract geometric and statistical features for adaptive parameter tuning
    triu_indices = np.triu_indices(n, k=1)
    distances_flat = distance_matrix[triu_indices]
    
    dist_stats = {
        'mean': np.mean(distances_flat),
        'std': np.std(distances_flat),
        'cv': np.std(distances_flat) / (np.mean(distances_flat) + 1e-10),
        'skewness': np.mean((distances_flat - np.mean(distances_flat))**3) / 
                   (np.std(distances_flat)**3 + 1e-10),
        'kurtosis': np.mean((distances_flat - np.mean(distances_flat))**4) / 
                   (np.std(distances_flat)**4 + 1e-10) - 3
    }
    
    # Size-based parameters
    log_n = np.log(n + 1)
    size_factor = min(1.0, n / 100.0)
    density_factor = n * (n - 1) / 2  # Number of possible edges
    
    # ===== 2. ENSEMBLE TOUR GENERATION =====
    def generate_tour_ensemble(num_tours: int) -> Tuple[List[np.ndarray], List[float]]:
        """Generate diverse high-quality tours using multiple construction methods."""
        tours = []
        tour_lengths = []
        
        # Determine method distribution based on instance features
        if n <= 20:
            methods = ['nn', 'fi', 'random', 'savings'] * 2
        elif n <= 100:
            methods = ['nn', 'fi', 'christofides', 'random']
        else:
            methods = ['nn', 'fi', 'random']
        
        for method in methods[:min(len(methods), num_tours)]:
            if method == 'nn':
                tour = _nearest_neighbor_tour(distance_matrix)
            elif method == 'fi':
                tour = _farthest_insertion_tour(distance_matrix)
            elif method == 'christofides' and n <= 1000:
                tour = _christofides_approx(distance_matrix)
            elif method == 'savings':
                tour = _savings_algorithm(distance_matrix)
            else:
                tour = np.random.permutation(n)
            
            # Apply local search with adaptive intensity
            ls_intensity = min(50, max(10, int(20 * size_factor)))
            tour = _adaptive_local_search(tour, distance_matrix, max_iter=ls_intensity)
            
            tours.append(tour)
            tour_lengths.append(_tour_length(tour, distance_matrix))
        
        return tours, tour_lengths
    
    # Generate tour ensemble
    num_tours = min(20, max(5, int(5 * log_n)))
    tours, tour_lengths = generate_tour_ensemble(num_tours)
    
    # ===== 3. HYBRID EDGE SCORING =====
    # Component 1: Quality-weighted edge frequency
    edge_freq = np.zeros_like(distance_matrix)
    if tour_lengths:
        # Exponential weighting based on tour quality
        min_len = min(tour_lengths)
        quality_weights = np.exp(-(np.array(tour_lengths) - min_len) / 
                                (dist_stats['mean'] + 1e-10))
        quality_weights = quality_weights / quality_weights.sum()
        
        for tour, weight in zip(tours, quality_weights):
            for i in range(n):
                j = (i + 1) % n
                u, v = tour[i], tour[j]
                edge_freq[u, v] += weight
                edge_freq[v, u] += weight
        
        if edge_freq.max() > 0:
            edge_freq = edge_freq / edge_freq.max()
    
    # Component 2: Geometric neighborhood similarity
    neighborhood_sim = np.zeros_like(distance_matrix)
    if n > 3:
        # For each node, compare its distance profile to others
        for i in range(n):
            dist_i = distance_matrix[i]
            for j in range(i + 1, n):
                dist_j = distance_matrix[j]
                # Pearson correlation of distance profiles
                corr = np.corrcoef(dist_i, dist_j)[0, 1]
                if corr > 0:
                    neighborhood_sim[i, j] = corr
                    neighborhood_sim[j, i] = corr
        
        if neighborhood_sim.max() > 0:
            neighborhood_sim = neighborhood_sim / neighborhood_sim.max()
    
    # Component 3: Edge centrality via betweenness approximation
    edge_centrality = np.zeros_like(distance_matrix)
    # Use tour segments to approximate edge betweenness
    num_segments = min(200, max(50, n * 2))
    
    for tour in tours[:min(5, len(tours))]:
        tour_array = np.array(tour)
        # Sample random segments from tour
        for _ in range(num_segments // min(5, len(tours))):
            if n <= 2:
                continue
                
            start = np.random.randint(0, n)
            length = np.random.randint(3, min(n, max(5, int(log_n * 3))))
            path_indices = [(start + k) % n for k in range(length)]
            path = tour_array[path_indices]
            
            # Count edges in this path with decay for longer paths
            decay = 1.0
            for k in range(len(path) - 1):
                u, v = path[k], path[k + 1]
                edge_centrality[u, v] += decay
                edge_centrality[v, u] += decay
                decay *= 0.9  # Exponential decay for longer paths
    
    if edge_centrality.max() > 0:
        edge_centrality = edge_centrality / edge_centrality.max()
    
    # Component 4: MST-based connectivity
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_component = np.zeros_like(distance_matrix)
    mst_component[mst_mask] = 1.0
    
    # Component 5: Inverse distance (classical heuristic)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # ===== 4. FEATURE-BASED COMPONENT WEIGHTING =====
    # Compute component quality metrics
    components = {
        'freq': edge_freq,
        'sim': neighborhood_sim,
        'cent': edge_centrality,
        'mst': mst_component,
        'dist': inv_dist
    }
    
    # Base weights adjusted by instance features
    base_weights = {
        'freq': 0.30,  # Tour frequency most important
        'sim': 0.10,   # Geometric similarity
        'cent': 0.20,  # Edge centrality
        'mst': 0.15,   # MST connectivity
        'dist': 0.25   # Inverse distance
    }
    
    # Adjust weights based on instance characteristics
    # Higher CV → distances more varied → give more weight to distance-based components
    if dist_stats['cv'] > 0.5:
        base_weights['dist'] += 0.10
        base_weights['freq'] -= 0.05
        base_weights['sim'] -= 0.05
    
    # Large instances → emphasize MST and centrality
    if n > 100:
        base_weights['mst'] += 0.05
        base_weights['cent'] += 0.05
        base_weights['freq'] -= 0.10
    
    # Normalize weights to sum to 1
    total_weight = sum(base_weights.values())
    weights = {k: v / total_weight for k, v in base_weights.items()}
    
    # ===== 5. COMPONENT COMBINATION =====
    heuristics_matrix = np.zeros_like(distance_matrix)
    for comp_name, comp_matrix in components.items():
        heuristics_matrix += weights[comp_name] * comp_matrix
    
    # ===== 6. ADAPTIVE SPARSE GRAPH CONSTRUCTION =====
    # Determine edges to keep per node adaptively
    min_edges = max(3, int(1.5 * log_n))
    max_edges = min(n - 1, int(5 * log_n + 10))
    
    # Base edge count based on instance characteristics
    base_edges = int(min_edges + (max_edges - min_edges) * 
                    (0.3 + 0.7 * min(1.0, dist_stats['cv'])))
    
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    for i in range(n):
        # Get scores for all edges from node i
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf  # Exclude self-loop
        
        # Calculate edge count for this node
        if n <= 20:
            n_edges = min(n - 1, base_edges + 2)
        else:
            # More edges for nodes with high degree in good tours
            tour_degree = np.sum([edge_freq[i] > 0 for _ in range(3)]) / 3 if len(tours) >= 3 else 0
            n_edges = min(max_edges, max(min_edges, 
                      int(base_edges * (0.8 + 0.4 * tour_degree))))
        
        # Select top edges using threshold-based approach
        if n_edges < n - 1:
            threshold = np.partition(scores, -n_edges)[-n_edges]
            sparsity_mask[i, scores >= threshold] = True
        else:
            sparsity_mask[i, :] = True
    
    # Ensure critical edges are preserved
    sparsity_mask = sparsity_mask | mst_mask
    
    # Ensure symmetry
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 7. FINAL ENHANCEMENT AND NORMALIZATION =====
    # Boost edges that appear in multiple high-quality tours
    if len(tours) > 0:
        tour_quality = np.array(tour_lengths)
        quality_threshold = np.percentile(tour_quality, 30)  # Top 70% tours
        high_quality_tours = [tours[i] for i in range(len(tours)) 
                             if tour_lengths[i] <= quality_threshold]
        
        if high_quality_tours:
            quality_mask = np.zeros_like(heuristics_matrix, dtype=bool)
            for tour in high_quality_tours[:min(3, len(high_quality_tours))]:
                for i in range(n):
                    j = (i + 1) % n
                    u, v = tour[i], tour[j]
                    quality_mask[u, v] = True
                    quality_mask[v, u] = True
            
            heuristics_matrix[quality_mask] *= 1.2
    
    # Ensure symmetry and zero diagonal
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1] range
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        h_min, h_max = non_zero.min(), non_zero.max()
        if h_max > h_min:
            heuristics_matrix[heuristics_matrix > 0] = (
                (heuristics_matrix[heuristics_matrix > 0] - h_min) / 
                (h_max - h_min + 1e-10)
            )
    
    return heuristics_matrix

def _nearest_neighbor_tour(dist_matrix: np.ndarray) -> np.ndarray:
    """Fast nearest neighbor tour construction."""
    n = dist_matrix.shape[0]
    start = np.random.randint(0, n)
    tour = [start]
    unvisited = set(range(n))
    unvisited.remove(start)
    
    current = start
    while unvisited:
        # Find nearest unvisited neighbor
        nearest = min(unvisited, key=lambda x: dist_matrix[current, x])
        tour.append(nearest)
        unvisited.remove(nearest)
        current = nearest
    
    return np.array(tour)

def _farthest_insertion_tour(dist_matrix: np.ndarray) -> np.ndarray:
    """Farthest insertion tour construction."""
    n = dist_matrix.shape[0]
    if n <= 2:
        return np.arange(n)
    
    # Start with two farthest nodes
    flat_idx = np.argmax(dist_matrix)
    i, j = divmod(flat_idx, n)
    
    tour = [i, j]
    unvisited = set(range(n)) - {i, j}
    
    while unvisited:
        # Find farthest node from current tour
        farthest = max(unvisited, 
                      key=lambda x: min(dist_matrix[x, y] for y in tour))
        
        # Find best insertion position
        best_pos = 1
        best_cost = float('inf')
        
        for k in range(len(tour)):
            l = (k + 1) % len(tour)
            cost = (dist_matrix[tour[k], farthest] + 
                   dist_matrix[farthest, tour[l]] - 
                   dist_matrix[tour[k], tour[l]])
            if cost < best_cost:
                best_cost = cost
                best_pos = l
        
        tour.insert(best_pos, farthest)
        unvisited.remove(farthest)
    
    return np.array(tour)

def _christofides_approx(dist_matrix: np.ndarray) -> np.ndarray:
    """Christofides algorithm approximation for TSP."""
    n = dist_matrix.shape[0]
    if n < 4:
        return np.arange(n)
    
    # 1. Find MST
    mst = minimum_spanning_tree(dist_matrix)
    
    # 2. Find nodes with odd degree in MST
    mst_array = mst.toarray()
    degree = np.sum((mst_array > 0) | (mst_array.T > 0), axis=1)
    odd_nodes = np.where(degree % 2 == 1)[0]
    
    # 3. Minimum weight perfect matching on odd nodes (simplified greedy)
    matched = set()
    matching_edges = []
    
    odd_node_list = list(odd_nodes)
    np.random.shuffle(odd_node_list)
    
    while odd_node_list:
        u = odd_node_list.pop()
        if not odd_node_list:
            break
        
        # Find closest unmatched odd node
        candidates = [v for v in odd_node_list if v not in matched]
        if not candidates:
            break
        
        v = min(candidates, key=lambda x: dist_matrix[u, x])
        matching_edges.append((u, v))
        matched.add(v)
        odd_node_list.remove(v)
    
    # 4. Combine MST and matching, find Eulerian tour
    # Simplified: use double MST edges + matching edges, find Hamiltonian tour
    combined_graph = mst_array.copy()
    for u, v in matching_edges:
        combined_graph[u, v] = dist_matrix[u, v]
        combined_graph[v, u] = dist_matrix[v, u]
    
    # 5. Find Hamiltonian tour (simplified DFS)
    visited = [False] * n
    tour = []
    
    def dfs(node):
        visited[node] = True
        tour.append(node)
        # Visit neighbors in order of increasing distance
        neighbors = np.argsort(dist_matrix[node])
        for neighbor in neighbors:
            if not visited[neighbor] and (combined_graph[node, neighbor] > 0 or 
                                         combined_graph[neighbor, node] > 0):
                dfs(neighbor)
    
    dfs(0)
    return np.array(tour)

def _savings_algorithm(dist_matrix: np.ndarray) -> np.ndarray:
    """Clark & Wright savings algorithm for TSP."""
    n = dist_matrix.shape[0]
    if n <= 2:
        return np.arange(n)
    
    depot = 0
    routes = [[i] for i in range(1, n)]
    
    # Calculate savings
    savings = []
    for i in range(1, n):
        for j in range(i + 1, n):
            saving = dist_matrix[depot, i] + dist_matrix[depot, j] - dist_matrix[i, j]
            savings.append((saving, i, j))
    
    savings.sort(reverse=True, key=lambda x: x[0])
    
    # Merge routes
    route_map = {i: i-1 for i in range(1, n)}  # Map node to route index
    
    for saving, i, j in savings:
        route_i = route_map[i]
        route_j = route_map[j]
        
        if route_i == route_j:
            continue
        
        route1 = routes[route_i]
        route2 = routes[route_j]
        
        # Check if merge is possible
        if route1[0] == i and route2[-1] == j:
            # route2 + route1
            new_route = route2 + route1
        elif route1[-1] == i and route2[0] == j:
            # route1 + route2
            new_route = route1 + route2
        elif route1[0] == i and route2[0] == j:
            # reversed(route2) + route1
            new_route = route2[::-1] + route1
        elif route1[-1] == i and route2[-1] == j:
            # route1 + reversed(route2)
            new_route = route1 + route2[::-1]
        else:
            continue
        
        # Update routes
        routes[route_i] = new_route
        routes.pop(route_j)
        
        # Update route map
        for node in new_route:
            route_map[node] = route_i
        for idx in range(route_j, len(routes)):
            for node in routes[idx]:
                route_map[node] = idx
    
    # Final tour: depot + only route + depot
    final_route = [depot] + routes[0] + [depot]
    return np.array(final_route[:-1])  # Remove closing depot

def _tour_length(tour: np.ndarray, dist_matrix: np.ndarray) -> float:
    """Calculate tour length."""
    n = len(tour)
    idx1 = tour
    idx2 = np.roll(tour, -1)
    return np.sum(dist_matrix[idx1, idx2])
