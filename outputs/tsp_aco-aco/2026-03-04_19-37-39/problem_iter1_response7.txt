import numpy as np
from scipy.spatial import distance_matrix
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors.
    
    Combines:
    1. Inverse distance (base factor)
    2. Local neighborhood structure (nearest neighbors)
    3. Global connectivity patterns (betweenness approximation)
    4. Adaptive thresholding to sparsify unpromising edges
    
    Returns: Heuristic matrix of same shape as input, with zero for unpromising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance (handle zeros to avoid division by infinity)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)  # Avoid self-connections
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0)
    
    # 2. Local neighborhood factor - edges to/from nearest neighbors
    k = min(5, n - 1)  # Number of nearest neighbors to consider
    nn_factor = np.zeros_like(distance_matrix)
    
    # For each node, boost edges to its k nearest neighbors
    for i in range(n):
        # Get indices of k+1 smallest distances (including self, which we'll exclude)
        nearest = np.argsort(distance_matrix[i])[:k+1]
        nearest = nearest[nearest != i]  # Remove self
        nn_factor[i, nearest] = 1.0
        nn_factor[nearest, i] = 1.0  # Symmetric
    
    # 3. Global connectivity factor - approximate edge betweenness
    # Use sum of distances to all other nodes as centrality measure
    centrality = np.zeros(n)
    for i in range(n):
        # Node centrality: inverse of average distance to other nodes
        other_dists = distance_matrix[i]
        other_dists = other_dists[other_dists > 0]  # Exclude zero distance to self
        if len(other_dists) > 0:
            centrality[i] = 1.0 / np.mean(other_dists)
    
    # Normalize centrality
    if np.max(centrality) > 0:
        centrality = centrality / np.max(centrality)
    
    # Edge betweenness approximation: product of node centralities
    betweenness_factor = np.outer(centrality, centrality)
    np.fill_diagonal(betweenness_factor, 0)
    
    # 4. Triangle closure factor - if i-j completes many triangles
    triangle_factor = np.zeros_like(distance_matrix)
    if n >= 3:
        # For each edge (i,j), count how many nodes k such that
        # both (i,k) and (j,k) are short edges
        for i in range(n):
            for j in range(i+1, n):
                # Find nodes that are close to both i and j
                close_to_i = distance_matrix[i] < np.percentile(distance_matrix[i], 25)
                close_to_j = distance_matrix[j] < np.percentile(distance_matrix[j], 25)
                common_close = np.sum(close_to_i & close_to_j)
                triangle_factor[i, j] = triangle_factor[j, i] = common_close / (n - 2)
    
    # 5. Combine all factors with weights
    weights = {
        'base': 2.0,           # Inverse distance is most important
        'nn': 1.5,             # Nearest neighbors are promising
        'betweenness': 1.0,    # Central connections help
        'triangle': 0.8        # Triangle completion is good but less important
    }
    
    combined = (
        weights['base'] * base_heuristic +
        weights['nn'] * nn_factor +
        weights['betweenness'] * betweenness_factor +
        weights['triangle'] * triangle_factor
    )
    
    # 6. Adaptive sparsification - keep only promising edges
    # Use percentile-based thresholding
    if n > 2:
        # Get upper triangle values (excluding diagonal)
        upper_tri = combined[np.triu_indices(n, k=1)]
        
        if len(upper_tri) > 0:
            # Keep top 40% of edges based on combined score
            threshold = np.percentile(upper_tri, 60)
            
            # Create mask for promising edges
            promising_mask = combined >= threshold
            
            # Also always include edges from nearest neighbors
            promising_mask = promising_mask | (nn_factor > 0)
            
            # Apply mask: zero out unpromising edges
            result = np.zeros_like(combined)
            result[promising_mask] = combined[promising_mask]
            
            # Ensure symmetry
            result = np.maximum(result, result.T)
            
            return result
    
    return combined

    # Test with a simple 4-city TSP
    cities = np.array([[0, 0], [1, 0], [1, 1], [0, 1]])
    dist_mat = distance_matrix(cities, cities)

    print(dist_mat)
    print("\nHeuristics v2:")

    result = heuristics_v2(dist_mat)
