import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge promisingness heuristics combining inverse distance,
    MST connectivity, local neighborhood analysis, and adaptive sparsification.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between nodes
        
    Returns:
    --------
    np.ndarray
        Matrix of the same shape indicating edge promisingness
    """
    n = distance_matrix.shape[0]
    
    # Handle small instances directly
    if n <= 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result
    
    # Ensure no zeros for division
    eps = 1e-10
    dist_safe = np.maximum(distance_matrix, eps)
    
    # 1. Base heuristic: inverse distance (fundamental TSP principle)
    inv_dist = 1 / dist_safe
    np.fill_diagonal(inv_dist, 0)
    
    # 2. MST connectivity (Christofides-inspired)
    # Edges in MST are often part of optimal TSP solutions
    mst = minimum_spanning_tree(dist_safe).toarray()
    mst_binary = ((mst + mst.T) > 0).astype(float)
    
    # 3. Local connectivity analysis
    # Consider both individual edge quality and neighborhood structure
    local_quality = np.zeros_like(distance_matrix)
    
    # Adaptive neighborhood size based on problem scale
    k_neighbors = max(2, min(5, n // 4))
    
    for i in range(n):
        # Get distances to other nodes
        distances_i = distance_matrix[i].copy()
        distances_i[i] = np.inf  # Exclude self
        
        # Find k nearest neighbors
        neighbor_indices = np.argsort(distances_i)[:k_neighbors]
        
        # Edge quality: compare to average of alternatives
        avg_dist_i = np.mean(distances_i[distances_i < np.inf])
        
        for j in range(n):
            if i != j:
                # Quality from i's perspective
                dist_ij = distance_matrix[i, j]
                
                # Compare to average connection from i
                if dist_ij < 0.7 * avg_dist_i:
                    quality_i = 1.0
                elif dist_ij < 0.9 * avg_dist_i:
                    quality_i = 0.7
                else:
                    quality_i = 0.3
                
                # Bonus for being in neighborhood
                if j in neighbor_indices:
                    rank = np.where(neighbor_indices == j)[0][0]
                    neighbor_bonus = 1.0 - 0.2 * rank  # Closer neighbors get higher bonus
                    quality_i *= (1.0 + 0.3 * neighbor_bonus)
                
                local_quality[i, j] = quality_i
    
    # 4. Symmetric edge consideration
    # Good edges should be good from both endpoints
    symmetric_factor = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            # Average of both directions' local quality
            sym_val = (local_quality[i, j] + local_quality[j, i]) / 2
            symmetric_factor[i, j] = symmetric_factor[j, i] = sym_val
    
    # 5. Adaptive combination weights based on problem size
    # Larger problems benefit more from MST guidance
    mst_weight = min(0.4, 0.2 + 0.1 * np.log10(n))
    base_weight = 0.4
    local_weight = 0.6 - mst_weight  # Remaining weight for local analysis
    
    # Combine factors
    combined = (
        base_weight * inv_dist +
        mst_weight * mst_binary +
        local_weight * symmetric_factor
    )
    
    # 6. Adaptive sparsification
    # Keep only the most promising connections per node
    # More aggressive sparsification for larger problems
    if n <= 10:
        keep_per_node = n - 1
    elif n <= 50:
        keep_per_node = max(5, n // 3)
    else:
        keep_per_node = max(8, n // 4)
    
    sparsified = np.zeros_like(combined)
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self from top selection
        
        # Keep top connections
        if keep_per_node < n - 1:
            threshold = np.sort(row)[-keep_per_node]
            row[row < threshold] = 0
        else:
            # For very small n, keep all connections
            pass
        
        sparsified[i] = row
    
    # Ensure symmetry (TSP is undirected)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Zero diagonal
    np.fill_diagonal(sparsified, 0)
    
    # Normalize to reasonable scale while preserving relative differences
    max_val = np.max(sparsified)
    if max_val > eps:
        sparsified = sparsified / max_val
    
    return sparsified
