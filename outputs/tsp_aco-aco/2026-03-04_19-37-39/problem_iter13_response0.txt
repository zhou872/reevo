import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Mutated TSP edge heuristics with enhanced adaptive weighting, improved edge betweenness
    sampling, and refined sparsification for better performance and accuracy.
    
    Key mutations from v1:
    1. More adaptive tour generation and edge betweenness sampling based on instance size.
    2. Enhanced weighting strategy incorporating correlation between components.
    3. Optimized sparsification parameters for connectivity and focus on promising edges.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    
    density_factor = min(1.0, n / 50.0)
    log_n = np.log(n + 1)
    
    # ===== 2. ENHANCED SOLUTION SAMPLING FOR EDGE FREQUENCY =====
    def generate_tour(method: str = 'nn') -> np.ndarray:
        """Generate tour using specified construction method."""
        if method == 'nn':  # Nearest neighbor
            start = np.random.randint(0, n)
            tour = [start]
            unvisited = set(range(n)) - {start}
            while unvisited:
                current = tour[-1]
                nearest = min(unvisited, key=lambda x: distance_matrix[current, x])
                tour.append(nearest)
                unvisited.remove(nearest)
            return np.array(tour)
        
        elif method == 'greedy':  # Greedy insertion
            if n <= 3:
                return np.random.permutation(n)
            indices = np.random.choice(n, 3, replace=False)
            tour = list(indices[:3])
            tour.append(tour[0])
            remaining = set(range(n)) - set(tour[:-1])
            
            while remaining:
                best_cost = float('inf')
                best_node = None
                best_pos = -1
                
                for node in remaining:
                    for i in range(len(tour)-1):
                        cost_inc = (distance_matrix[tour[i], node] + 
                                   distance_matrix[node, tour[i+1]] - 
                                   distance_matrix[tour[i], tour[i+1]])
                        if cost_inc < best_cost:
                            best_cost = cost_inc
                            best_node = node
                            best_pos = i + 1
                
                tour.insert(best_pos, best_node)
                remaining.remove(best_node)
            
            return np.array(tour[:-1])
        
        else:  # Random
            return np.random.permutation(n)
    
    def two_opt_improve(tour: np.ndarray, max_iter: int = 30) -> np.ndarray:
        """Apply 2-opt local search."""
        improved = True
        iteration = 0
        current = tour.copy()
        
        while improved and iteration < max_iter:
            improved = False
            for i in range(n-1):
                for j in range(i+2, n):
                    if j == n-1 and i == 0:
                        continue
                    
                    a, b = current[i], current[(i+1)%n]
                    c, d = current[j], current[(j+1)%n]
                    
                    old_len = distance_matrix[a, b] + distance_matrix[c, d]
                    new_len = distance_matrix[a, c] + distance_matrix[b, d]
                    
                    if new_len < old_len:
                        current = np.concatenate([
                            current[:i+1],
                            current[i+1:j+1][::-1],
                            current[j+1:]
                        ])
                        improved = True
                        break
                if improved:
                    break
            iteration += 1
        
        return current
    
    # Adaptive number of tours
    num_tours = min(15, max(5, int(n * 0.5)))
    tours = []
    
    for _ in range(num_tours // 3):  # NN tours
        tour = generate_tour('nn')
        tours.append(two_opt_improve(tour, max_iter=int(20 * density_factor)))
    
    for _ in range(num_tours // 3):  # Greedy tours
        tour = generate_tour('greedy')
        tours.append(two_opt_improve(tour, max_iter=int(20 * density_factor)))
    
    for _ in range(num_tours - 2*(num_tours//3)):  # Random tours
        tour = generate_tour('random')
        tours.append(two_opt_improve(tour, max_iter=int(20 * density_factor)))
    
    # Edge frequency from tours
    edge_freq = np.zeros_like(distance_matrix)
    for tour in tours:
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            edge_freq[u, v] += 1
            edge_freq[v, u] += 1
    
    if edge_freq.max() > 0:
        edge_freq = edge_freq / edge_freq.max()
    
    # ===== 3. IMPROVED SAMPLED EDGE BETWEENNESS =====
    edge_betweenness = np.zeros_like(distance_matrix)
    
    # Adaptive number of random paths
    num_paths = min(300, max(100, n * 3))
    
    for _ in range(num_paths):
        s, t = np.random.choice(n, 2, replace=False)
        
        current = s
        visited = set([s])
        path = [s]
        
        while current != t and len(path) < n * 2:
            neighbors = list(range(n))
            np.random.shuffle(neighbors)
            candidates = [v for v in neighbors if v not in visited]
            if not candidates:
                candidates = neighbors[:min(5, len(neighbors))]
            
            scores = []
            for v in candidates:
                dist_score = 1.0 / (distance_matrix[current, v] + 1e-10)
                progress_score = 1.0 / (distance_matrix[v, t] + 1e-10) if distance_matrix[v, t] > 0 else 0
                scores.append(dist_score + 0.7 * progress_score)
            
            scores = np.array(scores)
            if scores.sum() > 0:
                probs = scores / scores.sum()
                next_node = np.random.choice(candidates, p=probs)
            else:
                next_node = np.random.choice(candidates)
            
            edge_betweenness[current, next_node] += 1
            edge_betweenness[next_node, current] += 1
            
            current = next_node
            visited.add(current)
            path.append(current)
            
            if len(visited) > n * 0.7:
                visited = set([current])
    
    if edge_betweenness.max() > 0:
        edge_betweenness = edge_betweenness / edge_betweenness.max()
    
    # ===== 4. ENHANCED ADAPTIVE COMPONENT WEIGHTING =====
    freq_values = edge_freq[np.triu_indices(n, k=1)]
    bet_values = edge_betweenness[np.triu_indices(n, k=1)]
    
    freq_std = np.std(freq_values)
    bet_std = np.std(bet_values)
    
    inv_dist_flat = 1 / distances_flat
    inv_dist_flat[np.isinf(inv_dist_flat)] = 0
    corr_freq_dist = np.corrcoef(freq_values, inv_dist_flat)[0,1] if len(freq_values) > 1 else 0
    corr_bet_dist = np.corrcoef(bet_values, inv_dist_flat)[0,1] if len(bet_values) > 1 else 0
    
    # Base weights
    w_freq_base = 0.3
    w_bet_base = 0.3
    w_dist_base = 0.2
    w_mst_base = 0.2
    
    # Adjust based on statistics
    w_freq = w_freq_base * (1 + min(1.0, freq_std)) * (1 - abs(corr_freq_dist) * 0.5)
    w_bet = w_bet_base * (1 + min(1.0, bet_std)) * (1 - abs(corr_bet_dist) * 0.5)
    w_dist = w_dist_base * (1 + (1 - min(1.0, cv)))
    w_mst = w_mst_base * density_factor
    
    total = w_freq + w_bet + w_dist + w_mst
    w_freq /= total
    w_bet /= total
    w_dist /= total
    w_mst /= total
    
    # ===== 5. COMPONENT COMPUTATION =====
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_scores = np.zeros_like(distance_matrix)
    mst_scores[mst_mask] = 1.0 / (distance_matrix[mst_mask] + 1e-10)
    if mst_scores.max() > 0:
        mst_scores = mst_scores / mst_scores.max()
    
    # ===== 6. COMBINE COMPONENTS =====
    heuristics_matrix = (
        w_freq * edge_freq +
        w_bet * edge_betweenness +
        w_dist * inv_dist +
        w_mst * mst_scores
    )
    
    # ===== 7. REFINED INTELLIGENT SPARSIFICATION =====
    base_edges = max(2, int(2.0 * log_n))
    max_edges = min(n-1, base_edges * 3)
    
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        n_edges = min(max_edges, max(base_edges, int((n-1) * (0.1 + 0.1 * density_factor * (1 + cv)))))
        top_indices = np.argsort(scores)[-n_edges:]
        sparsity_mask[i, top_indices] = True
    
    sparsity_mask = sparsity_mask | mst_mask
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 8. FINAL PROCESSING =====
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0 and non_zero.max() > non_zero.min():
        heuristics_matrix[heuristics_matrix > 0] = (non_zero - non_zero.min()) / (non_zero.max() - non_zero.min())
    
    return heuristics_matrix
