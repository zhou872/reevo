import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse import csr_matrix
import heapq

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with refined betweenness approximation via 
    landmark-based sampling and meta-heuristic weight adaptation.
    
    Key innovations:
    1. Landmark-based betweenness approximation with efficient Dijkstra sampling
    2. Meta-heuristic weight tuning using instance characteristics
    3. Multi-scale edge evaluation with local and global considerations
    4. Connectivity-preserving sparsification with dynamic thresholds
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes.
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    # Analyze distance matrix properties
    distances = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances)
    dist_std = np.std(distances)
    cv = dist_std / (dist_mean + 1e-10)
    
    # Instance complexity metrics
    size_factor = min(1.0, n / 50)
    uniformity = 1.0 / (1.0 + cv)
    
    # ===== 2. EFFICIENT BETWEENNESS APPROXIMATION =====
    # Landmark-based betweenness using Dijkstra from sampled nodes
    betweenness = np.zeros(n)
    
    # Adaptive landmark selection
    num_landmarks = min(20, max(5, n // 5))
    if n <= 20:
        landmarks = np.arange(n)
    else:
        # Select landmarks that are spread out in the space
        # Start with random node, then iteratively pick farthest from selected
        landmarks = []
        start = np.random.randint(0, n)
        landmarks.append(start)
        
        for _ in range(1, num_landmarks):
            # Find node farthest from all selected landmarks
            min_dists = np.full(n, np.inf)
            for l in landmarks:
                min_dists = np.minimum(min_dists, distance_matrix[l])
            next_landmark = np.argmax(min_dists)
            landmarks.append(next_landmark)
    
    # Run Dijkstra from each landmark and count shortest paths
    for source in landmarks:
        # Dijkstra from source
        dist = np.full(n, np.inf)
        dist[source] = 0
        prev = [-1] * n
        visited = np.zeros(n, dtype=bool)
        
        # Priority queue
        heap = [(0, source)]
        
        while heap:
            d, u = heapq.heappop(heap)
            if visited[u]:
                continue
            visited[u] = True
            
            for v in range(n):
                if v == u:
                    continue
                new_dist = d + distance_matrix[u, v]
                if new_dist < dist[v]:
                    dist[v] = new_dist
                    prev[v] = u
                    heapq.heappush(heap, (new_dist, v))
        
        # Count paths for betweenness
        # Process nodes in order of distance from source
        node_order = np.argsort(dist)
        sigma = np.zeros(n)
        sigma[source] = 1
        
        # Forward pass to count shortest paths
        for v in node_order:
            if v == source:
                continue
            for u in range(n):
                if u == v:
                    continue
                if prev[v] == u:
                    sigma[v] += sigma[u]
        
        # Backward pass to accumulate betweenness
        delta = np.zeros(n)
        for v in reversed(node_order):
            if v == source:
                continue
            for u in range(n):
                if u == v:
                    continue
                if prev[v] == u and sigma[u] > 0:
                    delta[u] += (sigma[u] / sigma[v]) * (1 + delta[v])
        
        betweenness += delta
    
    # Normalize betweenness
    betweenness = betweenness / (len(landmarks) + 1e-10)
    if np.max(betweenness) > np.min(betweenness):
        betweenness = (betweenness - np.min(betweenness)) / (np.max(betweenness) - np.min(betweenness) + 1e-10)
    
    # ===== 3. META-HEURISTIC WEIGHT TUNING =====
    # Adaptive weights based on instance characteristics
    if cv > 1.0:  # High variability
        w_inv_dist = 0.8
        w_centrality = 0.1
        w_nn = 0.1
    elif cv < 0.3:  # Low variability
        w_inv_dist = 0.4
        w_centrality = 0.4
        w_nn = 0.2
    else:  # Moderate variability
        w_inv_dist = 0.6
        w_centrality = 0.2
        w_nn = 0.2
    
    # Adjust for problem size
    size_adjust = min(1.0, np.log10(n + 1) / 2)
    w_centrality *= (1.0 + 0.3 * size_adjust)
    w_nn *= (1.0 + 0.2 * size_adjust)
    weight_sum = w_inv_dist + w_centrality + w_nn
    w_inv_dist /= weight_sum
    w_centrality /= weight_sum
    w_nn /= weight_sum
    
    # ===== 4. MULTI-SCALE EDGE EVALUATION =====
    # Base inverse distance component
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + np.eye(n) * 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # Normalize inverse distance
    inv_max = np.max(inv_dist)
    if inv_max > 0:
        inv_dist = inv_dist / inv_max
    
    # Centrality-based edge potential
    closeness = (n - 1) / (np.sum(distance_matrix, axis=1) - np.diag(distance_matrix) + 1e-10)
    closeness = closeness / (np.sum(closeness) + 1e-10)
    
    # Combine betweenness and closeness
    node_centrality = 0.7 * betweenness + 0.3 * closeness
    node_centrality = node_centrality / (np.sum(node_centrality) + 1e-10)
    
    # Edge centrality as geometric mean
    edge_centrality = np.sqrt(np.outer(node_centrality, node_centrality))
    np.fill_diagonal(edge_centrality, 0)
    
    # Nearest neighbor component with adaptive k
    k = max(2, min(8, int(5 * (1 + 0.5 * cv))))
    nn_scores = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get k nearest neighbors
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argpartition(distances, k-1)[:k]
        nn_scores[i, nn_indices] = 1.0 / (distances[nn_indices] + 1e-10)
    
    # Symmetrize and normalize
    nn_scores = (nn_scores + nn_scores.T) / 2
    nn_max = np.max(nn_scores)
    if nn_max > 0:
        nn_scores = nn_scores / nn_max
    
    # ===== 5. ENHANCED MST COMPONENT =====
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # MST edge importance based on distance
    mst_importance = np.zeros_like(distance_matrix)
    if np.any(mst_mask):
        mst_distances = distance_matrix[mst_mask]
        if len(mst_distances) > 0:
            mst_inv = 1.0 / (mst_distances + 1e-10)
            mst_importance[mst_mask] = mst_inv
            mst_max = np.max(mst_importance)
            if mst_max > 0:
                mst_importance = mst_importance / mst_max
    
    # ===== 6. INTEGRATED FUSION =====
    # Weighted combination of components
    heuristics_matrix = (
        w_inv_dist * inv_dist +
        w_centrality * edge_centrality +
        w_nn * nn_scores
    )
    
    # Boost MST edges with adaptive factor
    mst_boost = 1.5 + 0.8 * size_factor
    heuristics_matrix[mst_mask] *= (mst_boost + mst_importance[mst_mask])
    
    # Local neighborhood enhancement
    for i in range(n):
        # Boost edges connecting to high-betweenness nodes
        betweenness_threshold = np.percentile(betweenness, 70)
        if betweenness[i] > betweenness_threshold:
            mask = distance_matrix[i] < np.percentile(distance_matrix[i], 30)
            mask[i] = False
            heuristics_matrix[i, mask] *= 1.2
            heuristics_matrix[mask, i] *= 1.2
    
    # ===== 7. DYNAMIC SPARSIFICATION =====
    # Adaptive threshold based on score distribution
    positive_scores = heuristics_matrix[heuristics_matrix > 0]
    
    if len(positive_scores) > 0:
        # Dynamic percentile calculation
        if cv > 0.8:  # High variability -> keep more edges
            keep_percentile = 40
        elif cv < 0.3:  # Low variability -> keep fewer edges
            keep_percentile = 60
        else:
            keep_percentile = 50
        
        threshold = np.percentile(positive_scores, keep_percentile)
        
        # Initial sparsification
        sparsity_mask = heuristics_matrix >= threshold
        
        # Ensure MST edges are included
        sparsity_mask = sparsity_mask | mst_mask
        
        # Connectivity guarantee with logarithmic growth
        min_connections = max(2, int(np.log2(n + 1)))
        
        for i in range(n):
            connected = np.sum(sparsity_mask[i])
            if connected < min_connections:
                # Add top missing edges by heuristic score
                missing = min_connections - connected
                candidates = np.argsort(heuristics_matrix[i])[::-1]
                added = 0
                
                for cand in candidates:
                    if cand != i and not sparsity_mask[i, cand]:
                        sparsity_mask[i, cand] = True
                        sparsity_mask[cand, i] = True
                        added += 1
                        if added >= missing:
                            break
    else:
        sparsity_mask = np.ones_like(heuristics_matrix, dtype=bool)
        np.fill_diagonal(sparsity_mask, False)
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 8. FINAL NORMALIZATION =====
    # Symmetrize
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    
    # Robust scaling
    max_val = np.max(heuristics_matrix)
    min_val = np.min(heuristics_matrix[heuristics_matrix > 0])
    
    if max_val > min_val:
        heuristics_matrix = (heuristics_matrix - min_val) / (max_val - min_val + 1e-10)
    elif max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    # Zero diagonal
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
