import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with improved
    centrality measures and efficient sparsification.
    
    Key improvements:
    1. Scale-adaptive parameter selection for different problem sizes
    2. Robust centrality computation using distance percentiles
    3. Multi-criteria sparsification with connectivity guarantees
    4. Efficient combination of MST, nearest neighbors, and inverse distance
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges.
        Same shape as input, zero on diagonal.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ========== 1. SCALE-ADAPTIVE PARAMETERS ==========
    # Parameters adjusted based on problem size
    if n <= 10:
        inv_dist_weight = 0.65
        centrality_weight = 0.35
        nn_k = min(4, n-1)
        keep_factor = 0.4
        min_connections = 3
    elif n <= 30:
        inv_dist_weight = 0.55
        centrality_weight = 0.45
        nn_k = min(5, n//2)
        keep_factor = 0.3
        min_connections = 4
    elif n <= 100:
        inv_dist_weight = 0.50
        centrality_weight = 0.30
        nn_k = min(6, n//3)
        keep_factor = 0.2
        min_connections = 5
    else:
        inv_dist_weight = 0.45
        centrality_weight = 0.25
        nn_k = min(8, n//4)
        keep_factor = 0.15
        min_connections = max(5, n//20)
    
    # ========== 2. BASE COMPONENTS ==========
    # Inverse distance with epsilon for numerical stability
    eps = 1e-10
    inv_dist = 1.0 / (distance_matrix + np.eye(n) * eps)
    np.fill_diagonal(inv_dist, 0)
    
    # Nearest neighbor mask
    nn_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argpartition(distances, nn_k)[:nn_k]
        nn_mask[i, nn_indices] = True
    nn_mask = nn_mask | nn_mask.T  # Symmetrize
    
    # Minimum spanning tree for connectivity
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # ========== 3. ENHANCED CENTRALITY MEASURE ==========
    # Compute robust centrality using percentiles
    centrality = np.zeros(n)
    for i in range(n):
        # Use distances excluding self
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Calculate trimmed mean (excluding extremes)
        if n >= 10:
            lower = np.percentile(distances, 25)
            upper = np.percentile(distances, 75)
            mask = (distances >= lower) & (distances <= upper)
            if np.any(mask):
                robust_avg = np.mean(distances[mask])
            else:
                robust_avg = np.mean(distances[distances < np.inf])
        else:
            robust_avg = np.mean(distances[distances < np.inf])
        
        # Inverse of robust average distance (more central = higher value)
        centrality[i] = 1.0 / (robust_avg + eps)
    
    # Normalize centrality
    if np.max(centrality) > 0:
        centrality = centrality / np.max(centrality)
    
    # Edge centrality potential (product of node centralities)
    edge_centrality = np.outer(centrality, centrality)
    np.fill_diagonal(edge_centrality, 0)
    
    # ========== 4. WEIGHTED COMBINATION ==========
    # Base heuristic scores
    heuristic = (inv_dist_weight * inv_dist + 
                 centrality_weight * edge_centrality)
    
    # Boost important edges
    heuristic[nn_mask] *= 1.5
    heuristic[mst_mask] *= 2.0
    
    # ========== 5. MULTI-CRITERIA SPARSIFICATION ==========
    # Determine number of edges to keep per node
    edges_to_keep = max(min_connections, int(keep_factor * n))
    
    # Create selection mask
    selection_mask = np.zeros((n, n), dtype=bool)
    
    # Ensure each node has at least edges_to_keep connections
    for i in range(n):
        # Get heuristic values for this node (excluding self)
        node_scores = heuristic[i].copy()
        node_scores[i] = -np.inf
        
        # Prioritize MST edges first (connectivity guarantee)
        mst_neighbors = np.where(mst_mask[i])[0]
        selection_mask[i, mst_neighbors] = True
        
        # Add top remaining edges by heuristic score
        remaining_needed = edges_to_keep - len(mst_neighbors)
        if remaining_needed > 0:
            # Temporarily mask already selected MST edges
            temp_scores = node_scores.copy()
            temp_scores[mst_neighbors] = -np.inf
            
            # Select top remaining edges
            top_indices = np.argpartition(-temp_scores, remaining_needed)[:remaining_needed]
            selection_mask[i, top_indices] = True
    
    # Ensure symmetry
    selection_mask = selection_mask | selection_mask.T
    
    # Additional connectivity safeguard
    # Ensure each node has at least min_connections
    for i in range(n):
        current_conn = np.sum(selection_mask[i])
        if current_conn < min_connections:
            # Add closest nodes not already connected
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            distances[selection_mask[i]] = np.inf
            needed = min_connections - current_conn
            closest = np.argpartition(distances, needed)[:needed]
            selection_mask[i, closest] = True
            selection_mask[closest, i] = True
    
    # Apply sparsification
    heuristic = heuristic * selection_mask
    
    # ========== 6. FINAL PROCESSING ==========
    # Normalize to [0, 1]
    max_val = np.max(heuristic)
    if max_val > eps:
        heuristic = heuristic / max_val
    
    # Ensure symmetry and zero diagonal
    heuristic = (heuristic + heuristic.T) / 2.0
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
