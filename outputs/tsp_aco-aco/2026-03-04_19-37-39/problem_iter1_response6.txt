import numpy as np
from scipy import sparse
from scipy.sparse.csgraph import minimum_spanning_tree, connected_components

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP that combines multiple factors to assess
    edge promisingness and sparsifies the matrix by setting unpromising edges to zero.
    """
    n = distance_matrix.shape[0]
    if n <= 3:
        return 1 / np.maximum(distance_matrix, 1e-10)
    
    # Avoid division by zero and handle very small distances
    eps = 1e-10
    dist = np.maximum(distance_matrix, eps)
    
    # 1. Base inverse distance heuristic
    inv_dist = 1 / dist
    
    # 2. Nearest neighbor factor: edges to k-nearest neighbors get bonus
    k_neighbors = max(3, n // 10)  # Adaptive k
    nn_bonus = np.zeros_like(dist)
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        neighbors = np.argsort(dist[i])[1:k_neighbors+1]
        nn_bonus[i, neighbors] = 1.0
    
    # 3. Minimum spanning tree indicator
    mst = minimum_spanning_tree(dist).toarray()
    mst_indicator = (mst + mst.T) > 0  # Symmetric MST edges
    
    # 4. Edge betweenness approximation: edges connecting different clusters
    # Create clusters based on distance threshold
    threshold = np.percentile(dist[dist > eps], 30)  # 30th percentile as threshold
    clusters = np.zeros((n, n), dtype=bool)
    for i in range(n):
        clusters[i] = dist[i] < threshold
    # Edge betweenness: connect nodes from different clusters
    edge_betweenness = np.zeros_like(dist)
    for i in range(n):
        for j in range(i+1, n):
            # If nodes share few common neighbors, edge might connect clusters
            common_neighbors = np.sum(clusters[i] & clusters[j])
            if common_neighbors < max(2, n // 20):
                edge_betweenness[i, j] = edge_betweenness[j, i] = 1.0
    
    # 5. Local clustering coefficient factor
    clustering_factor = np.zeros_like(dist)
    for i in range(n):
        # Find neighbors within 1.5x the median distance from i
        median_dist = np.median(dist[i, dist[i] > eps])
        neighbors_mask = dist[i] < 1.5 * median_dist
        neighbors = np.where(neighbors_mask)[0]
        neighbors = neighbors[neighbors != i]
        
        if len(neighbors) >= 2:
            # Calculate clustering coefficient for this node
            triangles = 0
            possible = 0
            for a_idx, a in enumerate(neighbors[:-1]):
                for b in neighbors[a_idx+1:]:
                    possible += 1
                    if dist[a, b] < 1.5 * max(median_dist, np.median(dist[a]), np.median(dist[b])):
                        triangles += 1
            if possible > 0:
                cc = triangles / possible
                # Distribute clustering factor to edges based on distance
                for j in neighbors:
                    weight = (1 - dist[i, j] / (2 * median_dist))
                    clustering_factor[i, j] = cc * max(0, weight)
    
    # 6. Distance rank factor: penalize very long edges
    rank_penalty = np.ones_like(dist)
    for i in range(n):
        sorted_indices = np.argsort(dist[i])
        ranks = np.zeros(n)
        ranks[sorted_indices] = np.arange(n)
        # Normalize ranks to [0, 1]
        normalized_ranks = ranks / (n - 1) if n > 1 else np.zeros(n)
        # Penalize edges with high rank (long distance)
        rank_penalty[i] = 1 - 0.7 * normalized_ranks  # Keep some weight even for long edges
    
    # Combine all factors
    heuristics_matrix = inv_dist * (
        1.0 +  # Base weight
        0.8 * nn_bonus +  # Nearest neighbor bonus
        1.2 * mst_indicator +  # MST edges are very promising
        0.6 * edge_betweenness +  # Between-cluster edges
        0.4 * clustering_factor  # Local clustering
    ) * rank_penalty  # Apply distance rank penalty
    
    # 7. Sparsification: keep only promising edges
    # Calculate threshold based on statistical properties
    upper_triangle = heuristics_matrix[np.triu_indices_from(heuristics_matrix, k=1)]
    if len(upper_triangle) > 0:
        mean_val = np.mean(upper_triangle)
        std_val = np.std(upper_triangle)
        threshold_val = mean_val + 0.5 * std_val
        
        # Also ensure we keep at least n edges per node
        for i in range(n):
            row_vals = heuristics_matrix[i]
            row_vals[row_vals < threshold_val] = 0
            # Ensure each node has at least min(5, n-1) connections
            non_zero = np.sum(row_vals > 0)
            min_connections = min(5, n-1)
            if non_zero < min_connections:
                # Keep top min_connections edges
                top_indices = np.argsort(row_vals)[-min_connections:]
                mask = np.zeros(n, dtype=bool)
                mask[top_indices] = True
                row_vals[~mask] = 0
    
    # Ensure symmetry
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    
    # Zero out diagonal
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to similar scale as input
    max_val = np.max(heuristics_matrix)
    if max_val > eps:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
