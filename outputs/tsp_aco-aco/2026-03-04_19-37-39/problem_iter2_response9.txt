import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection with dynamic weight tuning,
    problem symmetry exploitation, and adaptive sparsification.
    
    Args:
        distance_matrix: Symmetric matrix of distances between n nodes
        
    Returns:
        Heuristic matrix of same shape indicating edge promisingness
    """
    n = distance_matrix.shape[0]
    
    # Handle small instances
    if n <= 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # Safe distance matrix (avoid division by zero)
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    eps = np.finfo(dist.dtype).eps
    
    # ==================== DYNAMIC WEIGHT COMPUTATION ====================
    # Compute problem characteristics for adaptive weight tuning
    problem_stats = _compute_problem_stats(dist, n)
    weights = _compute_dynamic_weights(problem_stats)
    
    # ==================== SYMMETRIC FACTOR COMPUTATION ====================
    # Compute factors that respect TSP symmetry
    factors = {}
    
    # 1. Base inverse distance (symmetric by nature)
    factors['inv_dist'] = 1 / np.maximum(dist, eps)
    
    # 2. Local connectivity with symmetry preservation
    factors['local'] = _compute_symmetric_local_connectivity(dist, n)
    
    # 3. Edge centrality using graph theory concepts
    factors['centrality'] = _compute_edge_centrality(dist, n)
    
    # 4. Structural importance from Minimum Spanning Tree
    factors['mst'] = _compute_mst_importance(dist)
    
    # 5. Neighborhood overlap (cluster indicator)
    factors['overlap'] = _compute_neighborhood_overlap(dist, n)
    
    # ==================== ADAPTIVE COMBINATION ====================
    # Combine factors using dynamically computed weights
    combined = np.zeros_like(dist)
    for i, (key, factor) in enumerate(factors.items()):
        combined += weights[i] * factor
    
    # ==================== INTELLIGENT SPARSIFICATION ====================
    # Adaptive sparsification based on problem characteristics
    heuristics_matrix = _adaptive_sparsification(combined, dist, n, problem_stats)
    
    # Ensure symmetry and clean diagonal
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Final normalization
    max_val = np.max(heuristics_matrix)
    if max_val > eps:
        heuristics_matrix /= max_val
    
    return heuristics_matrix

def _compute_problem_stats(dist: np.ndarray, n: int) -> dict:
    """Compute statistical characteristics of the problem instance."""
    upper_tri = dist[np.triu_indices_from(dist, k=1)]
    
    stats = {
        'n': n,
        'mean_dist': np.mean(upper_tri),
        'std_dist': np.std(upper_tri),
        'skewness': 0.0,
        'density_variation': 0.0
    }
    
    if len(upper_tri) > 0:
        # Compute skewness of distances
        mean = stats['mean_dist']
        std = max(stats['std_dist'], eps)
        stats['skewness'] = np.mean(((upper_tri - mean) / std) ** 3)
        
        # Estimate spatial density variation
        local_densities = []
        for i in range(n):
            # Median distance to nearest 30% of nodes
            k = max(1, int(0.3 * n))
            nearest = np.partition(dist[i], k)[:k+1]
            local_densities.append(np.median(nearest[nearest > 0]))
        
        stats['density_variation'] = np.std(local_densities) / np.mean(local_densities)
    
    return stats

def _compute_dynamic_weights(stats: dict) -> list:
    """Compute adaptive weights based on problem characteristics."""
    n = stats['n']
    
    # Base weights that adapt to problem size
    if n < 20:
        # Small problems: emphasize global structure
        base_weights = [0.25, 0.20, 0.25, 0.20, 0.10]
    elif n < 100:
        # Medium problems: balanced approach
        base_weights = [0.30, 0.25, 0.20, 0.15, 0.10]
    else:
        # Large problems: emphasize local connectivity for efficiency
        base_weights = [0.35, 0.30, 0.15, 0.10, 0.10]
    
    # Adjust based on distance distribution skewness
    skew = stats['skewness']
    if skew > 0.5:  # Right-skewed (many short distances)
        # Emphasize local connectivity more
        base_weights[1] += 0.05
        base_weights[0] -= 0.05
    elif skew < -0.5:  # Left-skewed (many long distances)
        # Emphasize global structure more
        base_weights[2] += 0.05
        base_weights[4] -= 0.05
    
    # Normalize weights
    total = sum(base_weights)
    return [w/total for w in base_weights]

def _compute_symmetric_local_connectivity(dist: np.ndarray, n: int) -> np.ndarray:
    """Compute local connectivity factor with symmetry preservation."""
    connectivity = np.zeros_like(dist)
    
    # Adaptive k based on problem size
    k = max(2, min(8, n // 5))
    
    for i in range(n):
        # Get distances and sort
        distances = dist[i].copy()
        distances[i] = np.inf
        
        # Find k-nearest neighbors
        nearest_indices = np.argpartition(distances, k)[:k]
        nearest_dists = distances[nearest_indices]
        
        if len(nearest_dists) > 0:
            max_dist = np.max(nearest_dists)
            for idx, j in enumerate(nearest_indices):
                # Score based on relative distance within neighborhood
                if max_dist > eps:
                    score = 1.0 - (distances[j] / max_dist)
                    # Ensure symmetry by setting both directions
                    connectivity[i, j] = score
                    connectivity[j, i] = score
    
    return connectivity

def _compute_edge_centrality(dist: np.ndarray, n: int) -> np.ndarray:
    """Compute edge centrality using betweenness approximation."""
    centrality = np.zeros_like(dist)
    
    # For each node, compute its importance based on distance distribution
    node_importance = np.zeros(n)
    for i in range(n):
        # Node importance inversely related to average distance to others
        avg_dist = np.mean(dist[i, dist[i] < np.inf])
        node_importance[i] = 1.0 / (avg_dist + eps)
    
    # Normalize node importance
    max_imp = np.max(node_importance)
    if max_imp > eps:
        node_importance /= max_imp
    
    # Edge centrality combines importance of endpoints and their distance
    for i in range(n):
        for j in range(i+1, n):
            # Combine node importance and distance
            score = (node_importance[i] + node_importance[j]) / 2
            # Penalize long edges more
            penalty = np.exp(-dist[i, j] / (2 * np.mean(dist)))
            centrality[i, j] = centrality[j, i] = score * penalty
    
    return centrality

def _compute_mst_importance(dist: np.ndarray) -> np.ndarray:
    """Compute importance based on Minimum Spanning Tree membership."""
    mst_importance = np.zeros_like(dist)
    
    try:
        # Compute MST
        mst = minimum_spanning_tree(dist).toarray()
        mst_importance = (mst + mst.T) > 0
        
        # Give MST edges higher weight
        mst_importance = mst_importance.astype(float) * 0.8
    except:
        # Fallback if MST computation fails
        pass
    
    return mst_importance

def _compute_neighborhood_overlap(dist: np.ndarray, n: int) -> np.ndarray:
    """Compute neighborhood overlap indicating cluster structure."""
    overlap = np.zeros_like(dist)
    
    # Adaptive neighborhood radius
    radius_multiplier = 1.5
    
    for i in range(n):
        # Determine local distance threshold
        local_dists = dist[i, dist[i] < np.inf]
        if len(local_dists) > 0:
            threshold_i = np.median(local_dists) * radius_multiplier
            neighbors_i = set(np.where(dist[i] < threshold_i)[0])
            neighbors_i.discard(i)
            
            for j in range(i+1, n):
                threshold_j = np.median(dist[j, dist[j] < np.inf]) * radius_multiplier
                neighbors_j = set(np.where(dist[j] < threshold_j)[0])
                neighbors_j.discard(j)
                
                # Compute Jaccard similarity of neighborhoods
                if len(neighbors_i) > 0 and len(neighbors_j) > 0:
                    intersection = len(neighbors_i & neighbors_j)
                    union = len(neighbors_i | neighbors_j)
                    if union > 0:
                        similarity = intersection / union
                        
                        # High overlap suggests nodes are in same cluster
                        # Low overlap suggests bridge between clusters (valuable)
                        if similarity < 0.3:  # Bridge edge
                            overlap[i, j] = overlap[j, i] = 0.6
                        elif similarity > 0.7:  # Same cluster
                            overlap[i, j] = overlap[j, i] = 0.3
                        else:  # Moderate overlap
                            overlap[i, j] = overlap[j, i] = 0.1
    
    return overlap

def _adaptive_sparsification(combined: np.ndarray, dist: np.ndarray, 
                           n: int, stats: dict) -> np.ndarray:
    """Intelligent sparsification based on problem characteristics."""
    sparsified = np.zeros_like(combined)
    
    # Adaptive parameters based on problem size and density variation
    if n < 30:
        min_edges = max(3, n // 3)
        max_edges = max(5, n // 2)
    elif n < 100:
        min_edges = max(3, n // 4)
        max_edges = max(5, n // 3)
    else:
        min_edges = max(3, n // 5)
        max_edges = max(5, n // 4)
    
    # Adjust based on density variation
    density_var = stats['density_variation']
    if density_var > 0.5:  # High variation: keep more edges
        min_edges = int(min_edges * 1.2)
        max_edges = int(max_edges * 1.2)
    
    for i in range(n):
        # Get node's connection scores
        scores = combined[i].copy()
        scores[i] = -np.inf
        
        # Determine number of edges to keep for this node
        if density_var > 0.3:
            # Variable sparsity based on local density
            local_dists = dist[i, dist[i] < np.inf]
            local_density = np.median(local_dists) if len(local_dists) > 0 else stats['mean_dist']
            
            if local_density < stats['mean_dist']:
                # Dense area: keep more edges
                k = min(max_edges, n-1)
            else:
                # Sparse area: keep fewer edges
                k = min(min_edges, n-1)
        else:
            # Uniform sparsity
            k = min((min_edges + max_edges) // 2, n-1)
        
        # Keep top k edges
        if k > 0:
            top_indices = np.argpartition(scores, -k)[-k:]
            sparsified[i, top_indices] = scores[top_indices]
    
    return sparsified
