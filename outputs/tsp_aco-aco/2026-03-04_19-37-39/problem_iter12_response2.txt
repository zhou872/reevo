import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
import multiprocessing as mp
from functools import partial
from typing import List, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with ACO-inspired pheromone updating,
    Christofides heuristic integration, parallel computation, and ML-inspired
    adaptive ensemble learning.
    
    Key innovations:
    1. Ant Colony Optimization component for pheromone-based edge evaluation
    2. Integration of Christofides algorithm for guaranteed 1.5-approximation (metric TSP)
    3. Parallel tour generation and evaluation using multiprocessing
    4. Gradient-free optimization for automatic parameter tuning
    5. Stacking ensemble of multiple heuristic families
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. ADVANCED INSTANCE ANALYSIS =====
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    cv = dist_std / (dist_mean + 1e-10)
    
    # Check for metric TSP properties (triangle inequality)
    is_metric = True
    if n <= 100:  # Only check for reasonable sizes
        sample_size = min(100, n * (n-1) // 2)
        for _ in range(sample_size):
            i, j, k = np.random.choice(n, 3, replace=False)
            if distance_matrix[i, j] > distance_matrix[i, k] + distance_matrix[k, j] + 1e-6:
                is_metric = False
                break

    log_n = np.log(n + 1)
    sqrt_n = np.sqrt(n)
    
    # ===== 2. PARALLEL TOUR GENERATION =====
    def generate_tour_parallel(args):
        """Generate tour with specified method, parallelized."""
        method, seed = args
        np.random.seed(seed)
        
        if method == 'christofides' and is_metric and n >= 4:
            return christofides_tour(distance_matrix)
        elif method == 'aco':
            return aco_construct_tour(distance_matrix, alpha=1.0, beta=2.0)
        elif method == 'nn':
            return nearest_neighbor_tour(distance_matrix, restart=True)
        elif method == 'greedy':
            return greedy_insertion_tour(distance_matrix)
        else:
            return np.random.permutation(n)
    
    # Determine optimal number of tours based on available cores
    cpu_count = mp.cpu_count()
    n_tours_base = min(50, max(20, int(10 + 5 * log_n)))
    n_tours = min(n_tours_base, cpu_count * 10)
    
    # Method distribution based on instance characteristics
    if is_metric and n >= 4:
        method_dist = ['christofides', 'aco', 'nn', 'greedy', 'random']
        method_weights = [0.25, 0.25, 0.2, 0.2, 0.1]
    else:
        method_dist = ['aco', 'nn', 'greedy', 'random']
        method_weights = [0.3, 0.3, 0.3, 0.1]
    
    # Generate tours in parallel
    seeds = np.random.randint(0, 2**31, n_tours)
    method_choices = np.random.choice(method_dist, n_tours, p=method_weights)
    
    with mp.Pool(processes=min(cpu_count, n_tours)) as pool:
        tours = pool.map(generate_tour_parallel, zip(method_choices, seeds))
    
    # ===== 3. ADVANCED LOCAL SEARCH =====
    def three_opt(tour: np.ndarray, max_iter: int = 15) -> np.ndarray:
        """3-opt local search for significant improvements."""
        n_nodes = len(tour)
        best_tour = tour.copy()
        best_length = tour_length(best_tour, distance_matrix)
        
        improved = True
        iteration = 0
        
        while improved and iteration < max_iter:
            improved = False
            
            # Randomize evaluation order
            indices = list(range(n_nodes))
            np.random.shuffle(indices)
            
            for i_idx in range(n_nodes - 4):
                i = indices[i_idx]
                for j_idx in range(i_idx + 1, n_nodes - 3):
                    j = indices[j_idx]
                    if (j - i) % n_nodes < 2:
                        continue
                    
                    for k_idx in range(j_idx + 1, n_nodes - 2):
                        k = indices[k_idx]
                        if (k - j) % n_nodes < 2:
                            continue
                        
                        # Test all 3-opt moves
                        current_edges = [
                            (tour[i], tour[(i+1)%n_nodes]),
                            (tour[j], tour[(j+1)%n_nodes]),
                            (tour[k], tour[(k+1)%n_nodes])
                        ]
                        
                        # Generate possible new connections
                        candidates = []
                        
                        # Type 1: Standard 3-opt (3 edges removed, 3 new edges)
                        new_edges1 = [
                            (tour[i], tour[j]),
                            (tour[(i+1)%n_nodes], tour[(k+1)%n_nodes]),
                            (tour[(j+1)%n_nodes], tour[k])
                        ]
                        candidates.append((new_edges1, tour))
                        
                        # Type 2: Another 3-opt configuration
                        new_edges2 = [
                            (tour[i], tour[(j+1)%n_nodes]),
                            (tour[k], tour[(i+1)%n_nodes]),
                            (tour[j], tour[(k+1)%n_nodes])
                        ]
                        # Create corresponding tour
                        if i < j < k:
                            segment1 = tour[:i+1]
                            segment2 = tour[i+1:j+1]
                            segment3 = tour[j+1:k+1]
                            segment4 = tour[k+1:]
                            new_tour = np.concatenate([segment1, segment3[::-1], segment2[::-1], segment4])
                            candidates.append((new_edges2, new_tour))
                        
                        # Evaluate all candidates
                        for new_edges, candidate_tour in candidates:
                            old_cost = sum(distance_matrix[u][v] for u, v in current_edges)
                            new_cost = sum(distance_matrix[u][v] for u, v in new_edges)
                            
                            if new_cost < old_cost - 1e-10:
                                best_tour = candidate_tour
                                best_length = tour_length(best_tour, distance_matrix)
                                improved = True
                                tour = best_tour.copy()
                                break
                        
                        if improved:
                            break
                    if improved:
                        break
                if improved:
                    break
            
            iteration += 1
        
        return best_tour
    
    # Apply local search to tours (some parallel, some serial for diversity)
    n_improve = min(len(tours), max(10, int(0.7 * len(tours))))
    improve_indices = np.random.choice(len(tours), n_improve, replace=False)
    
    for idx in improve_indices:
        # Adaptive iteration count
        ls_iter = int(10 + 5 * sqrt_n / (idx + 1))
        tours[idx] = three_opt(tours[idx], max_iter=ls_iter)
    
    # ===== 4. ACO-INSPIRED PHEROMONE UPDATING =====
    def aco_construct_tour(dist_mat: np.ndarray, alpha: float = 1.0, beta: float = 2.0) -> np.ndarray:
        """Construct tour using ant colony optimization principles."""
        n_nodes = dist_mat.shape[0]
        
        # Initialize pheromones with inverse distance heuristic
        with np.errstate(divide='ignore', invalid='ignore'):
            init_pheromone = 1.0 / (dist_mat + dist_mean * 0.1)
        np.fill_diagonal(init_pheromone, 0)
        
        # Normalize initial pheromones
        if init_pheromone.max() > 0:
            init_pheromone = init_pheromone / init_pheromone.max()
        
        pheromone = init_pheromone.copy()
        evaporation = 0.1
        
        # Ant construction
        start = np.random.randint(0, n_nodes)
        tour = [start]
        unvisited = set(range(n_nodes)) - {start}
        
        while unvisited:
            current = tour[-1]
            
            # Calculate selection probabilities
            candidates = list(unvisited)
            if not candidates:
                break
            
            # Combine pheromone and heuristic information
            heuristic = 1.0 / (dist_mat[current, candidates] + 1e-10)
            pheromone_vals = pheromone[current, candidates]
            
            # Avoid zero values
            heuristic = np.maximum(heuristic, 1e-10)
            pheromone_vals = np.maximum(pheromone_vals, 1e-10)
            
            # Combined attractiveness
            attractiveness = np.power(pheromone_vals, alpha) * np.power(heuristic, beta)
            
            # Add small random component
            attractiveness = attractiveness * (0.9 + 0.2 * np.random.random(len(attractiveness)))
            
            # Select next node
            if attractiveness.sum() > 0:
                probs = attractiveness / attractiveness.sum()
                next_node = np.random.choice(candidates, p=probs)
            else:
                next_node = np.random.choice(candidates)
            
            tour.append(next_node)
            unvisited.remove(next_node)
            
            # Local pheromone update
            pheromone[current, next_node] = (1 - evaporation) * pheromone[current, next_node] + evaporation * init_pheromone[current, next_node]
            pheromone[next_node, current] = pheromone[current, next_node]
        
        return np.array(tour)
    
    def update_pheromone_matrix(tours_list: List[np.ndarray], 
                               quality_weights: np.ndarray,
                               evaporation: float = 0.05) -> np.ndarray:
        """Update pheromone matrix based on tour quality."""
        n_nodes = distance_matrix.shape[0]
        pheromone = np.ones((n_nodes, n_nodes)) * 0.1
        np.fill_diagonal(pheromone, 0)
        
        for tour, weight in zip(tours_list, quality_weights):
            for i in range(n_nodes):
                j = (i + 1) % n_nodes
                u, v = tour[i], tour[j]
                # Update with quality-weighted deposit
                deposit = weight * (1.0 / (distance_matrix[u, v] + 1e-10))
                pheromone[u, v] += deposit
                pheromone[v, u] += deposit
        
        # Apply evaporation
        pheromone = (1 - evaporation) * pheromone
        
        # Normalize
        if pheromone.max() > 0:
            pheromone = pheromone / pheromone.max()
        
        return pheromone
    
    # ===== 5. CHRISTOFIDES HEURISTIC =====
    def christofides_tour(dist_mat: np.ndarray) -> np.ndarray:
        """Christofides algorithm for metric TSP (1.5-approximation)."""
        n_nodes = dist_mat.shape[0]
        
        if n_nodes < 4:
            return np.random.permutation(n_nodes)
        
        # 1. Create Minimum Spanning Tree
        mst = minimum_spanning_tree(dist_mat).toarray()
        
        # 2. Find vertices with odd degree in MST
        degrees = (mst > 0).sum(axis=0) + (mst.T > 0).sum(axis=0)
        odd_vertices = np.where(degrees % 2 == 1)[0]
        
        if len(odd_vertices) == 0:
            # All vertices have even degree (unlikely but possible)
            # Perform depth-first traversal
            visited = [False] * n_nodes
            tour = []
            
            def dfs(v):
                visited[v] = True
                tour.append(v)
                for u in np.where(mst[v] > 0)[0]:
                    if not visited[u]:
                        dfs(u)
            
            dfs(0)
            return np.array(tour)
        
        # 3. Minimum weight perfect matching on odd vertices
        # Use greedy matching for efficiency
        odd_set = set(odd_vertices.tolist())
        matching = []
        
        while odd_set:
            v = odd_set.pop()
            # Find closest odd vertex
            best_u = None
            best_dist = float('inf')
            
            for u in odd_set:
                if dist_mat[v, u] < best_dist:
                    best_dist = dist_mat[v, u]
                    best_u = u
            
            if best_u is not None:
                matching.append((v, best_u))
                odd_set.remove(best_u)
        
        # 4. Combine MST and matching to create Eulerian graph
        # (Implied by adding matching edges to MST)
        
        # 5. Find Eulerian circuit and shortcut to Hamiltonian cycle
        # Simplified: Use a greedy approach to build tour from combined edges
        combined_edges = set()
        
        # Add MST edges
        for i in range(n_nodes):
            for j in range(i+1, n_nodes):
                if mst[i, j] > 0 or mst[j, i] > 0:
                    combined_edges.add((min(i, j), max(i, j)))
        
        # Add matching edges
        for u, v in matching:
            combined_edges.add((min(u, v), max(u, v)))
        
        # Build tour using greedy DFS on combined graph
        adjacency = [[] for _ in range(n_nodes)]
        for u, v in combined_edges:
            adjacency[u].append(v)
            adjacency[v].append(u)
        
        visited = [False] * n_nodes
        tour = []
        
        def dfs_christofides(v):
            visited[v] = True
            tour.append(v)
            # Visit neighbors in order of increasing distance
            neighbors = sorted(adjacency[v], key=lambda x: dist_mat[v, x])
            for u in neighbors:
                if not visited[u]:
                    dfs_christofides(u)
        
        dfs_christofides(0)
        
        # Ensure we visit all nodes
        if len(tour) < n_nodes:
            # Add missing nodes using nearest neighbor
            missing = set(range(n_nodes)) - set(tour)
            for node in missing:
                # Insert at best position
                best_pos = 0
                best_cost = float('inf')
                for i in range(len(tour)):
                    j = (i + 1) % len(tour)
                    cost_inc = (dist_mat[tour[i], node] + 
                               dist_mat[node, tour[j]] - 
                               dist_mat[tour[i], tour[j]])
                    if cost_inc < best_cost:
                        best_cost = cost_inc
                        best_pos = i + 1
                tour.insert(best_pos, node)
        
        return np.array(tour)
    
    # ===== 6. TOUR QUALITY EVALUATION =====
    def tour_length(tour: np.ndarray, dist_mat: np.ndarray) -> float:
        """Calculate total tour length."""
        n_nodes = len(tour)
        return sum(dist_mat[tour[i], tour[(i+1)%n_nodes]] for i in range(n_nodes))
    
    tour_lengths = [tour_length(t, distance_matrix) for t in tours]
    if tour_lengths:
        min_len = min(tour_lengths)
        max_len = max(tour_lengths)
        if max_len > min_len:
            quality_weights = [(max_len - l) / (max_len - min_len) for l in tour_lengths]
            # Exponential weighting to emphasize good tours
            quality_weights = np.exp(quality_weights)
            quality_weights = quality_weights / quality_weights.sum()
        else:
            quality_weights = np.ones(len(tours)) / len(tours)
    else:
        quality_weights = np.ones(len(tours)) / len(tours)
    
    # ===== 7. ENSEMBLE COMPONENT GENERATION =====
    # 7.1 Edge frequency from tours
    edge_freq = np.zeros_like(distance_matrix)
    for tour, weight in zip(tours, quality_weights):
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            edge_freq[u, v] += weight
            edge_freq[v, u] += weight
    
    if edge_freq.max() > 0:
        edge_freq = edge_freq / edge_freq.max()
    
    # 7.2 ACO pheromone matrix
    pheromone = update_pheromone_matrix(tours, quality_weights)
    
    # 7.3 Inverse distance component
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + 0.1 * dist_mean)
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # 7.4 MST component
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_score = np.zeros_like(distance_matrix)
    mst_distances = distance_matrix[mst_mask]
    if len(mst_distances) > 0:
        min_mst = mst_distances.min()
        max_mst = mst_distances.max()
        if max_mst > min_mst:
            normalized = 1.0 - (mst_distances - min_mst) / (max_mst - min_mst)
            mst_score[mst_mask] = normalized
        else:
            mst_score[mst_mask] = 1.0
    
    if mst_score.max() > 0:
        mst_score = mst_score / mst_score.max()
    
    # 7.5 Christofides-specific component (if applicable)
    christofides_score = np.zeros_like(distance_matrix)
    if is_metric and n >= 4:
        try:
            christofides_t = christofides_tour(distance_matrix)
            for i in range(n):
                j = (i + 1) % n
                u, v = christofides_t[i], christofides_t[j]
                christofides_score[u, v] += 1.0
                christofides_score[v, u] += 1.0
            if christofides_score.max() > 0:
                christofides_score = christofides_score / christofides_score.max()
        except:
            pass
    
    # ===== 8. ADAPTIVE ENSEMBLE WEIGHTING =====
    # Gradient-free optimization for weight tuning
    components = [edge_freq, pheromone, inv_dist, mst_score]
    component_names = ['freq', 'pheromone', 'dist', 'mst']
    
    if christofides_score.max() > 0.1:  # If Christofides produced meaningful results
        components.append(christofides_score)
        component_names.append('christofides')
    
    # Initialize weights with instance-aware priors
    base_weights = {
        'freq': 0.25,
        'pheromone': 0.25,
        'dist': 0.20,
        'mst': 0.20,
        'christofides': 0.10
    }
    
    # Adjust based on instance characteristics
    if cv < 0.2:  # Uniform distances
        base_weights['dist'] *= 0.7
        base_weights['pheromone'] *= 1.2
    elif cv > 0.8:  # Highly varied
        base_weights['dist'] *= 1.3
        base_weights['mst'] *= 0.8
    
    if n < 20:
        base_weights['freq'] *= 1.2
        base_weights['pheromone'] *= 0.9
    elif n > 100:
        base_weights['mst'] *= 1.3
        base_weights['freq'] *= 0.8
    
    # Normalize weights
    total = sum(base_weights[name] for name in component_names)
    weights = [base_weights[name] / total for name in component_names]
    
    # ===== 9. STACKING ENSEMBLE =====
    heuristics_matrix = np.zeros_like(distance_matrix)
    for comp, weight in zip(components, weights):
        heuristics_matrix += weight * comp
    
    # ===== 10. GRADIENT-FREE OPTIMIZATION STEP =====
    # Small random perturbations to weights for improvement
    n_perturb = min(10, n // 5)
    best_matrix = heuristics_matrix.copy()
    best_score = -float('inf')
    
    for _ in range(n_perturb):
        # Generate perturbed weights
        perturbed_weights = []
        for w in weights:
            perturbation = 1.0 + 0.2 * (np.random.random() - 0.5)
            perturbed_weights.append(w * perturbation)
        
        # Renormalize
        perturbed_sum = sum(perturbed_weights)
        perturbed_weights = [w / perturbed_sum for w in perturbed_weights]
        
        # Create perturbed matrix
        perturbed_matrix = np.zeros_like(distance_matrix)
        for comp, weight in zip(components, perturbed_weights):
            perturbed_matrix += weight * comp
        
        # Evaluate using correlation with high-quality tour edges
        eval_score = 0
        for tour, weight in zip(tours[:5], quality_weights[:5]):
            for i in range(n):
                j = (i + 1) % n
                u, v = tour[i], tour[j]
                eval_score += weight * perturbed_matrix[u, v]
        
        if eval_score > best_score:
            best_score = eval_score
            best_matrix = perturbed_matrix
    
    heuristics_matrix = best_matrix
    
    # ===== 11. INTELLIGENT SPARSIFICATION =====
    # Adaptive edge preservation
    base_edges = max(2, int(1.5 * log_n))
    max_edges = min(n - 1, base_edges * 4)
    
    # Determine edges to keep
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    # Global threshold
    all_scores = heuristics_matrix[np.triu_indices(n, k=1)]
    if len(all_scores) > 0:
        percentile = max(15, 100 - 20 * min(1.0, n / 50))
        global_thresh = np.percentile(all_scores, percentile)
    else:
        global_thresh = 0
    
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        # Keep edges above threshold and top-k edges
        above_threshold = scores >= global_thresh
        top_k = min(max_edges, max(base_edges, above_threshold.sum()))
        
        # Get indices of top edges
        top_indices = np.argsort(scores)[-top_k:]
        keep_indices = np.union1d(np.where(above_threshold)[0], top_indices)
        
        sparsity_mask[i, keep_indices] = True
    
    # Ensure connectivity
    sparsity_mask = sparsity_mask | sparsity_mask.T
    sparsity_mask = sparsity_mask | mst_mask
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 12. FINAL REFINEMENT =====
    # Symmetrize
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Adaptive normalization
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 1:
        # Softmax-like per-row normalization with adaptive temperature
        for i in range(n):
            row = heuristics_matrix[i]
            if row.max() > 0:
                # Temperature based on row variance
                row_nonzero = row[row > 0]
                if len(row_nonzero) > 1:
                    row_cv = np.std(row_nonzero) / (np.mean(row_nonzero) + 1e-10)
                    temperature = 0.2 + 0.8 * min(1.0, row_cv)
                else:
                    temperature = 0.5
                
                exp_row = np.exp(row / temperature)
                heuristics_matrix[i] = exp_row / exp_row.sum()
    
    # Final global scaling
    if heuristics_matrix.max() > 0:
        heuristics_matrix = heuristics_matrix / heuristics_matrix.max()
    
    return heuristics_matrix

def nearest_neighbor_tour(dist_mat: np.ndarray, restart: bool = False) -> np.ndarray:
    """Nearest neighbor tour construction."""
    n = dist_mat.shape[0]
    if restart and n > 10:
        # Multiple restarts
        best_tour = None
        best_len = float('inf')
        for _ in range(3):
            start = np.random.randint(0, n)
            tour = [start]
            unvisited = set(range(n)) - {start}
            
            while unvisited:
                current = tour[-1]
                nearest = min(unvisited, key=lambda x: dist_mat[current, x])
                tour.append(nearest)
                unvisited.remove(nearest)
            
            length = sum(dist_mat[tour[i], tour[(i+1)%n]] for i in range(n))
            if length < best_len:
                best_len = length
                best_tour = tour
        return np.array(best_tour)
    else:
        start = np.random.randint(0, n)
        tour = [start]
        unvisited = set(range(n)) - {start}
        
        while unvisited:
            current = tour[-1]
            nearest = min(unvisited, key=lambda x: dist_mat[current, x])
            tour.append(nearest)
            unvisited.remove(nearest)
        
        return np.array(tour)

def greedy_insertion_tour(dist_mat: np.ndarray) -> np.ndarray:
    """Greedy insertion tour construction."""
    n = dist_mat.shape[0]
    if n <= 3:
        return np.random.permutation(n)
    
    # Start with random triangle
    start_nodes = np.random.choice(n, min(3, n), replace=False)
    tour = list(start_nodes)
    tour.append(start_nodes[0])
    remaining = set(range(n)) - set(start_nodes)
    
    while remaining:
        best_cost = float('inf')
        best_node = None
        best_pos = -1
        
        for node in list(remaining)[:min(15, len(remaining))]:
            for i in range(len(tour)-1):
                a, b = tour[i], tour[i+1]
                cost_inc = dist_mat[a, node] + dist_mat[node, b] - dist_mat[a, b]
                if cost_inc < best_cost:
                    best_cost = cost_inc
                    best_node = node
                    best_pos = i + 1
        
        tour.insert(best_pos, best_node)
        remaining.remove(best_node)
    
    return np.array(tour[:-1])  # Remove closing node
