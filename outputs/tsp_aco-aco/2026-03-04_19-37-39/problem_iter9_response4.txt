import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with improved edge centrality sampling
    and more adaptive parameter tuning based on instance characteristics.
    
    Key improvements over v1:
    1. More efficient edge betweenness sampling with adaptive sample sizes
    2. Better integration of MST connectivity information
    3. Refined sparsification strategy based on edge quality distribution
    4. Adaptive parameter tuning based on instance size and distance patterns
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    cv = dist_std / (dist_mean + 1e-10)
    
    # Size-dependent parameters
    log_n = np.log(n + 1)
    density_factor = min(1.0, n / 50.0)
    
    # ===== 2. ADAPTIVE SAMPLING PARAMETERS =====
    # Adjust sample sizes based on instance characteristics
    num_tours = max(5, min(20, int(5 + n * 0.3)))
    num_betweenness_samples = max(3, min(10, int(3 + n * 0.2)))
    
    # ===== 3. IMPROVED EDGE FREQUENCY SAMPLING =====
    def generate_tour(method: str) -> np.ndarray:
        """Generate tour using specified construction method."""
        n = distance_matrix.shape[0]
        
        if method == 'nn':
            start = np.random.randint(0, n)
            tour = [start]
            unvisited = set(range(n)) - {start}
            while unvisited:
                current = tour[-1]
                nearest = min(unvisited, key=lambda x: distance_matrix[current, x])
                tour.append(nearest)
                unvisited.remove(nearest)
            return np.array(tour)
        
        elif method == 'greedy':
            if n <= 3:
                return np.random.permutation(n)
            
            # Start with 2 nodes
            start_nodes = np.random.choice(n, 2, replace=False)
            tour = list(start_nodes)
            tour.append(start_nodes[0])  # Close initial cycle
            remaining = set(range(n)) - set(start_nodes)
            
            while remaining:
                best_cost = float('inf')
                best_node = None
                best_pos = -1
                
                for node in remaining:
                    for i in range(len(tour)-1):
                        cost_inc = (distance_matrix[tour[i], node] + 
                                   distance_matrix[node, tour[i+1]] - 
                                   distance_matrix[tour[i], tour[i+1]])
                        if cost_inc < best_cost:
                            best_cost = cost_inc
                            best_node = node
                            best_pos = i + 1
                
                if best_node is not None:
                    tour.insert(best_pos, best_node)
                    remaining.remove(best_node)
            
            return np.array(tour[:-1])
        
        else:  # random
            return np.random.permutation(n)
    
    def two_opt_improve(tour: np.ndarray, max_iter: int = 30) -> np.ndarray:
        """Quick 2-opt improvement."""
        improved = True
        iteration = 0
        n = len(tour)
        current_tour = tour.copy()
        
        while improved and iteration < max_iter:
            improved = False
            for i in range(n-1):
                for j in range(i+2, n):
                    if j == n-1 and i == 0:
                        continue
                    
                    a, b = current_tour[i], current_tour[(i+1)%n]
                    c, d = current_tour[j], current_tour[(j+1)%n]
                    
                    old_seg = distance_matrix[a, b] + distance_matrix[c, d]
                    new_seg = distance_matrix[a, c] + distance_matrix[b, d]
                    
                    if new_seg < old_seg:
                        current_tour = np.concatenate([
                            current_tour[:i+1],
                            current_tour[i+1:j+1][::-1],
                            current_tour[j+1:]
                        ])
                        improved = True
                        break
                if improved:
                    break
            iteration += 1
        
        return current_tour
    
    # Generate diverse tours
    edge_frequency = np.zeros_like(distance_matrix)
    
    for _ in range(num_tours):
        # Select method based on adaptive distribution
        if np.random.random() < 0.5:
            tour = generate_tour('nn')
        elif np.random.random() < 0.7:
            tour = generate_tour('greedy')
        else:
            tour = generate_tour('random')
        
        # Quick improvement
        tour = two_opt_improve(tour, max_iter=int(15 * density_factor))
        
        # Record edges
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            edge_frequency[u, v] += 1
            edge_frequency[v, u] += 1
    
    # Normalize edge frequency
    if edge_frequency.max() > 0:
        edge_frequency = edge_frequency / edge_frequency.max()
    
    # ===== 4. IMPROVED EDGE BETWEENNESS SAMPLING =====
    edge_betweenness = np.zeros_like(distance_matrix)
    
    for _ in range(num_betweenness_samples):
        # Start with a random node
        start = np.random.randint(0, n)
        
        # Build a greedy path
        visited = {start}
        path = [start]
        
        while len(visited) < n:
            current = path[-1]
            # Find nearest unvisited neighbor
            unvisited = [i for i in range(n) if i not in visited]
            if not unvisited:
                break
            next_node = min(unvisited, key=lambda x: distance_matrix[current, x])
            path.append(next_node)
            visited.add(next_node)
        
        # Return to start
        path.append(start)
        
        # Record edges in this path
        for i in range(len(path)-1):
            u, v = path[i], path[i+1]
            edge_betweenness[u, v] += 1
            edge_betweenness[v, u] += 1
        
        # Also create a random path for diversity
        if n > 5:
            random_path = np.random.permutation(n).tolist()
            random_path.append(random_path[0])
            for i in range(len(random_path)-1):
                u, v = random_path[i], random_path[i+1]
                edge_betweenness[u, v] += 0.5
                edge_betweenness[v, u] += 0.5
    
    # Normalize edge betweenness
    if edge_betweenness.max() > 0:
        edge_betweenness = edge_betweenness / edge_betweenness.max()
    
    # ===== 5. COMPUTE BASIC COMPONENTS =====
    # Inverse distance component
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # MST component
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_scores = np.zeros_like(distance_matrix)
    mst_scores[mst_mask] = 1
    
    # Nearest neighbor mask
    k_nn = max(2, min(5, int(np.log2(n + 1))))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nn_indices = np.argsort(distance_matrix[i])[1:k_nn+1]
        nn_mask[i, nn_indices] = True
    nn_mask = nn_mask | nn_mask.T
    
    # ===== 6. ADAPTIVE HYBRID WEIGHTING =====
    # Analyze component variances for weighting
    freq_var = np.var(edge_frequency[np.triu_indices(n, k=1)])
    betw_var = np.var(edge_betweenness[np.triu_indices(n, k=1)])
    
    # Adaptive weights based on component characteristics
    w_freq = 0.4 + 0.2 * min(1.0, freq_var * 3)
    w_betw = 0.2 + 0.2 * min(1.0, betw_var * 3)
    w_dist = 0.3 * (1 - min(1.0, cv))  # Less weight if distances are uniform
    w_mst = 0.1 + 0.1 * density_factor
    
    # Normalize weights
    total = w_freq + w_betw + w_dist + w_mst
    w_freq /= total
    w_betw /= total
    w_dist /= total
    w_mst /= total
    
    # ===== 7. COMBINE COMPONENTS =====
    heuristics_matrix = (
        w_freq * edge_frequency +
        w_betw * edge_betweenness +
        w_dist * inv_dist +
        w_mst * mst_scores
    )
    
    # ===== 8. INTELLIGENT SPARSIFICATION =====
    # Determine edges to keep per node
    edges_per_node = max(2, min(n-1, int(2 * log_n + 3)))
    
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        # Keep top edges
        threshold = np.sort(scores)[-edges_per_node]
        sparsity_mask[i, scores >= threshold] = True
    
    # Ensure connectivity: include MST edges
    sparsity_mask = sparsity_mask | mst_mask
    
    # Ensure symmetry
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Boost edges that appear in multiple good components
    strong_edges = ((edge_frequency > np.percentile(edge_frequency, 70)) & 
                    (edge_betweenness > np.percentile(edge_betweenness, 70)))
    heuristics_matrix[strong_edges] *= 1.3
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # Boost nearest neighbors within sparse graph
    heuristics_matrix[nn_mask & sparsity_mask] *= 1.2
    
    # ===== 9. FINAL PROCESSING =====
    # Ensure symmetry
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    
    # Zero diagonal
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        heuristics_matrix = heuristics_matrix / (heuristics_matrix.max() + 1e-10)
    
    return heuristics_matrix
