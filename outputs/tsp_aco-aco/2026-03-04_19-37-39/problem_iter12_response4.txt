import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Dict, Tuple, List
import math

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristics with learning-based parameter optimization and
    sophisticated edge evaluation via graph neural-inspired scoring.
    
    Key innovations:
    1. Multi-armed bandit for adaptive construction method selection
    2. Candidate edge lists with learned pruning thresholds
    3. Ensemble of complementary local search operators
    4. Graph neural network-inspired edge scoring with learned filters
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. LEARNED INSTANCE ANALYSIS =====
    # Extract comprehensive instance features
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    cv_dist = dist_std / (dist_mean + 1e-10)
    
    # Skewness and kurtosis for distribution shape
    if len(distances_flat) > 3:
        dist_skew = np.mean(((distances_flat - dist_mean) / dist_std) ** 3)
        dist_kurt = np.mean(((distances_flat - dist_mean) / dist_std) ** 4) - 3
    else:
        dist_skew = 0
        dist_kurt = 0
    
    # Graph density features
    median_dist = np.median(distances_flat)
    quartile_25 = np.percentile(distances_flat, 25)
    quartile_75 = np.percentile(distances_flat, 75)
    iqr = quartile_75 - quartile_25
    
    # Normalized instance features
    size_factor = min(1.0, n / 100.0)
    log_n = np.log(n + 1)
    
    # ===== 2. MULTI-ARMED BANDIT FOR CONSTRUCTION METHOD SELECTION =====
    class ConstructionBandit:
        def __init__(self, n_methods: int = 4, epsilon: float = 0.2, decay: float = 0.99):
            self.q_values = np.ones(n_methods)  # Initial quality estimates
            self.counts = np.zeros(n_methods)
            self.epsilon = epsilon
            self.decay = decay
            self.method_names = ['cheapest', 'regret2', 'regret3', 'farthest']
            
        def select_method(self) -> Tuple[int, str]:
            if np.random.random() < self.epsilon:
                method_idx = np.random.randint(len(self.method_names))
            else:
                method_idx = np.argmax(self.q_values)
            return method_idx, self.method_names[method_idx]
        
        def update(self, method_idx: int, reward: float):
            self.counts[method_idx] += 1
            alpha = 1.0 / (self.counts[method_idx] ** 0.8)  # Decaying learning rate
            self.q_values[method_idx] = (1 - alpha) * self.q_values[method_idx] + alpha * reward
            self.epsilon *= self.decay  # Reduce exploration over time
    
    bandit = ConstructionBandit(epsilon=0.3, decay=0.95)
    
    def construct_tour(method: str) -> np.ndarray:
        """Construct initial tour using specified method."""
        if method == 'cheapest':
            # Cheapest insertion
            if n <= 3:
                return np.random.permutation(n)
            
            start_nodes = np.random.choice(n, min(4, n), replace=False)
            tour = list(start_nodes)
            tour.append(start_nodes[0])
            remaining = set(range(n)) - set(tour[:-1])
            
            while remaining:
                best_cost = np.inf
                best_node = None
                best_pos = -1
                
                for node in remaining:
                    for i in range(len(tour)-1):
                        cost_inc = (distance_matrix[tour[i], node] + 
                                   distance_matrix[node, tour[i+1]] - 
                                   distance_matrix[tour[i], tour[i+1]])
                        if cost_inc < best_cost:
                            best_cost = cost_inc
                            best_node = node
                            best_pos = i + 1
                
                tour.insert(best_pos, best_node)
                remaining.remove(best_node)
            
            return np.array(tour[:-1])
        
        elif method == 'regret2':
            # Regret-2 insertion
            if n <= 3:
                return np.random.permutation(n)
            
            start_nodes = np.random.choice(n, min(4, n), replace=False)
            tour = list(start_nodes)
            tour.append(start_nodes[0])
            remaining = set(range(n)) - set(tour[:-1])
            
            while remaining:
                best_regret = -np.inf
                best_node = None
                best_pos = -1
                
                for node in remaining:
                    insert_costs = []
                    for i in range(len(tour)-1):
                        cost = (distance_matrix[tour[i], node] + 
                               distance_matrix[node, tour[i+1]] - 
                               distance_matrix[tour[i], tour[i+1]])
                        insert_costs.append(cost)
                    
                    if len(insert_costs) >= 2:
                        sorted_costs = np.sort(insert_costs)
                        regret = sorted_costs[1] - sorted_costs[0]
                        if regret > best_regret:
                            best_regret = regret
                            best_node = node
                            best_pos = np.argmin(insert_costs) + 1
                    else:
                        if insert_costs[0] < best_regret:
                            best_regret = insert_costs[0]
                            best_node = node
                            best_pos = 1
                
                tour.insert(best_pos, best_node)
                remaining.remove(best_node)
            
            return np.array(tour[:-1])
        
        elif method == 'regret3':
            # Regret-3 insertion
            if n <= 3:
                return np.random.permutation(n)
            
            start_nodes = np.random.choice(n, min(4, n), replace=False)
            tour = list(start_nodes)
            tour.append(start_nodes[0])
            remaining = set(range(n)) - set(tour[:-1])
            
            while remaining:
                best_weighted_regret = -np.inf
                best_node = None
                best_pos = -1
                
                for node in remaining:
                    insert_costs = []
                    for i in range(len(tour)-1):
                        cost = (distance_matrix[tour[i], node] + 
                               distance_matrix[node, tour[i+1]] - 
                               distance_matrix[tour[i], tour[i+1]])
                        insert_costs.append(cost)
                    
                    if len(insert_costs) >= 3:
                        sorted_costs = np.sort(insert_costs)
                        # Weighted regret: 2*(2nd - 1st) + (3rd - 1st)
                        weighted_regret = 2 * (sorted_costs[1] - sorted_costs[0]) + \
                                         (sorted_costs[2] - sorted_costs[0])
                        if weighted_regret > best_weighted_regret:
                            best_weighted_regret = weighted_regret
                            best_node = node
                            best_pos = np.argmin(insert_costs) + 1
                    elif len(insert_costs) >= 2:
                        sorted_costs = np.sort(insert_costs)
                        regret = sorted_costs[1] - sorted_costs[0]
                        if regret > best_weighted_regret:
                            best_weighted_regret = regret
                            best_node = node
                            best_pos = np.argmin(insert_costs) + 1
                    else:
                        if insert_costs[0] < best_weighted_regret:
                            best_weighted_regret = insert_costs[0]
                            best_node = node
                            best_pos = 1
                
                tour.insert(best_pos, best_node)
                remaining.remove(best_node)
            
            return np.array(tour[:-1])
        
        else:  # 'farthest'
            # Farthest insertion
            if n <= 3:
                return np.random.permutation(n)
            
            start = np.random.randint(0, n)
            tour = [start]
            remaining = set(range(n)) - {start}
            
            while remaining:
                # Find farthest node from current tour
                farthest_dist = -1
                farthest_node = None
                
                for node in remaining:
                    min_dist_to_tour = min(distance_matrix[node, t] for t in tour)
                    if min_dist_to_tour > farthest_dist:
                        farthest_dist = min_dist_to_tour
                        farthest_node = node
                
                # Insert at position with minimal increase
                best_cost = np.inf
                best_pos = -1
                for i in range(len(tour)):
                    cost_inc = (distance_matrix[tour[i-1], farthest_node] + 
                               distance_matrix[farthest_node, tour[i]] - 
                               distance_matrix[tour[i-1], tour[i]])
                    if cost_inc < best_cost:
                        best_cost = cost_inc
                        best_pos = i
                
                tour.insert(best_pos, farthest_node)
                remaining.remove(farthest_node)
            
            return np.array(tour)
    
    # ===== 3. ADAPTIVE CANDIDATE LIST CONSTRUCTION =====
    def build_candidate_lists(k_factor: float = 1.5) -> List[np.ndarray]:
        """Build candidate lists for each node using adaptive k."""
        k = max(3, min(n-1, int(k_factor * log_n)))
        candidates = []
        
        for i in range(n):
            # Sort neighbors by distance
            sorted_indices = np.argsort(distance_matrix[i])
            # Skip self
            valid_indices = sorted_indices[sorted_indices != i]
            # Take top k
            candidates.append(valid_indices[:k])
        
        return candidates
    
    # Learn optimal k_factor based on instance
    if cv_dist < 0.2:
        k_factor = 1.2  # Uniform distances, fewer candidates needed
    elif cv_dist > 0.8:
        k_factor = 2.0  # Varied distances, more candidates
    else:
        k_factor = 1.5
    
    candidate_lists = build_candidate_lists(k_factor)
    
    # ===== 4. ENSEMBLE LOCAL SEARCH =====
    def two_opt_fast(tour: np.ndarray, max_iter: int = 30) -> np.ndarray:
        """Optimized 2-opt using candidate lists."""
        improved = True
        current = tour.copy()
        iteration = 0
        
        while improved and iteration < max_iter:
            improved = False
            
            for i in range(n):
                if improved:
                    break
                
                a, b = current[i], current[(i + 1) % n]
                
                # Only check candidates of a and b
                candidates_ij = set(candidate_lists[a]).union(candidate_lists[b])
                
                for j_idx in candidates_ij:
                    j = np.where(current == j_idx)[0][0]
                    if abs(i - j) < 2 or (i == 0 and j == n-1) or (j == 0 and i == n-1):
                        continue
                    
                    c, d = current[j], current[(j + 1) % n]
                    
                    gain = (distance_matrix[a, b] + distance_matrix[c, d] -
                           distance_matrix[a, c] - distance_matrix[b, d])
                    
                    if gain > 1e-6:
                        # Perform swap
                        if i < j:
                            current[i+1:j+1] = current[i+1:j+1][::-1]
                        else:
                            current[j+1:i+1] = current[j+1:i+1][::-1]
                        improved = True
                        break
            
            iteration += 1
        
        return current
    
    def or_opt(tour: np.ndarray, max_iter: int = 20) -> np.ndarray:
        """OR-opt: relocate small segments (1-3 nodes)."""
        current = tour.copy()
        improved = True
        iteration = 0
        
        while improved and iteration < max_iter:
            improved = False
            
            segment_size = min(3, n // 10)
            if segment_size < 1:
                break
            
            for start in np.random.permutation(n):
                if improved:
                    break
                
                for length in range(1, segment_size + 1):
                    end = (start + length) % n
                    
                    if start < end:
                        segment = current[start:end]
                        before_segment = current[:start]
                        after_segment = current[end:]
                    else:
                        # Wrap around
                        segment = np.concatenate([current[start:], current[:end]])
                        before_segment = current[end:start]
                        after_segment = np.array([])
                    
                    # Try inserting at different positions
                    for insert_pos in range(len(before_segment) + len(after_segment)):
                        if insert_pos == start or (insert_pos > start and insert_pos < end):
                            continue
                        
                        # Build new tour
                        if insert_pos <= len(before_segment):
                            new_tour = np.concatenate([
                                before_segment[:insert_pos],
                                segment,
                                before_segment[insert_pos:],
                                after_segment
                            ])
                        else:
                            adj_pos = insert_pos - len(before_segment)
                            new_tour = np.concatenate([
                                before_segment,
                                after_segment[:adj_pos],
                                segment,
                                after_segment[adj_pos:]
                            ])
                        
                        # Calculate cost difference
                        old_cost = sum(
                            distance_matrix[current[i], current[(i+1)%n]] 
                            for i in range(n)
                        )
                        new_cost = sum(
                            distance_matrix[new_tour[i], new_tour[(i+1)%n]] 
                            for i in range(n)
                        )
                        
                        if new_cost < old_cost - 1e-6:
                            current = new_tour
                            improved = True
                            break
                    
                    if improved:
                        break
            
            iteration += 1
        
        return current
    
    def double_bridge_kick(tour: np.ndarray) -> np.ndarray:
        """Double-bridge move for diversification."""
        if n < 8:
            return tour
        
        # Choose 4 random cut points
        points = sorted(np.random.choice(n, 4, replace=False))
        a, b, c, d = points
        
        # Reorder segments: [0,a], [a,b], [b,c], [c,d], [d,end]
        segment1 = tour[:a]
        segment2 = tour[a:b]
        segment3 = tour[b:c]
        segment4 = tour[c:d]
        segment5 = tour[d:]
        
        # New order: 1, 4, 3, 2, 5
        return np.concatenate([segment1, segment4, segment3, segment2, segment5])
    
    # ===== 5. TOUR GENERATION WITH ADAPTIVE LEARNING =====
    num_tours = min(25, max(8, int(n * 0.4)))
    tours = []
    tour_qualities = []
    
    for _ in range(num_tours):
        # Select construction method using bandit
        method_idx, method_name = bandit.select_method()
        
        # Construct initial tour
        tour = construct_tour(method_name)
        
        # Apply local search ensemble
        for _ in range(2):  # Multiple local search applications
            tour = two_opt_fast(tour, max_iter=int(25 * size_factor))
            tour = or_opt(tour, max_iter=int(15 * size_factor))
        
        # Occasionally apply diversification
        if np.random.random() < 0.3:
            tour = double_bridge_kick(tour)
            tour = two_opt_fast(tour, max_iter=15)
        
        tours.append(tour)
        
        # Calculate tour quality
        tour_length = sum(
            distance_matrix[tour[i], tour[(i+1)%n]] for i in range(n)
        )
        quality = 1.0 / (1.0 + tour_length / (dist_mean * n))
        tour_qualities.append(quality)
        
        # Update bandit with normalized reward
        normalized_quality = (quality - 0.5) * 2  # Scale to [-1, 1]
        bandit.update(method_idx, max(0, normalized_quality))
    
    # ===== 6. GRAPH NEURAL-INSPIRED EDGE SCORING =====
    # Edge frequency with quality weighting
    edge_freq = np.zeros_like(distance_matrix)
    max_quality = max(tour_qualities) if tour_qualities else 1.0
    
    for tour, quality in zip(tours, tour_qualities):
        weight = quality / max_quality
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            edge_freq[u, v] += weight
            edge_freq[v, u] += weight
    
    if edge_freq.max() > 0:
        edge_freq = edge_freq / edge_freq.max()
    
    # ===== 7. LEARNED EDGE CENTRALITY =====
    edge_centrality = np.zeros_like(distance_matrix)
    
    # Adaptive number of walks based on instance
    num_walks = min(400, max(100, int(n * 3)))
    
    for _ in range(num_walks):
        s, t = np.random.choice(n, 2, replace=False)
        current = s
        
        # Biased random walk
        for step in range(min(2 * n, 60)):
            if current == t:
                break
            
            # Get candidate neighbors
            candidates = candidate_lists[current]
            if len(candidates) == 0:
                break
            
            # Score candidates using multi-criteria
            scores = []
            for v in candidates:
                if v == current:
                    scores.append(-np.inf)
                    continue
                
                # Distance score (inverse)
                dist_score = 1.0 / (distance_matrix[current, v] + 1e-10)
                
                # Progress toward target
                progress = 0
                if distance_matrix[current, t] > 0:
                    progress = max(0, distance_matrix[current, t] - distance_matrix[v, t])
                    progress = progress / (distance_matrix[current, t] + 1e-10)
                
                # Edge frequency influence
                freq_influence = edge_freq[current, v] if edge_freq[current, v] > 0 else 0.1
                
                # Combined score
                score = dist_score * (1.0 + 0.7 * progress) * (1.0 + 0.5 * freq_influence)
                scores.append(score)
            
            scores = np.array(scores)
            if scores.sum() > 0:
                probs = scores / scores.sum()
                next_node = np.random.choice(candidates, p=probs)
            else:
                next_node = np.random.choice(candidates)
            
            # Update centrality with decay
            decay = 1.0 / (step + 2)
            edge_centrality[current, next_node] += decay
            edge_centrality[next_node, current] += decay
            
            current = next_node
    
    # Normalize centrality
    if edge_centrality.max() > 0:
        edge_centrality = np.log1p(edge_centrality) / np.log1p(edge_centrality.max())
    
    # ===== 8. ADVANCED COMPONENT INTEGRATION =====
    # MST component with edge weights
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    mst_scores = np.zeros_like(distance_matrix)
    mst_edges = np.where(mst_mask)
    if len(mst_edges[0]) > 0:
        # Weight MST edges by inverse distance
        mst_distances = distance_matrix[mst_edges]
        inv_mst_dist = 1.0 / (mst_distances + 1e-10)
        mst_scores[mst_edges] = inv_mst_dist
        if inv_mst_dist.max() > 0:
            mst_scores[mst_edges] = inv_mst_dist / inv_mst_dist.max()
    
    # Distance-based component with learned transformation
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + np.eye(n) * dist_mean * 0.1)
    np.fill_diagonal(inv_dist, 0)
    
    # Apply learned distance transformation based on distribution
    if dist_skew > 0.5:
        # Right-skewed: emphasize small distances more
        inv_dist = np.power(inv_dist, 1.2)
    elif dist_skew < -0.5:
        # Left-skewed: flatten distribution
        inv_dist = np.power(inv_dist, 0.8)
    
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # ===== 9. LEARNED WEIGHT OPTIMIZATION =====
    # Compute component statistics
    comp_stats = []
    components = [edge_freq, edge_centrality, inv_dist, mst_scores]
    
    for comp in components:
        comp_values = comp[np.triu_indices(n, k=1)]
        if len(comp_values) > 0:
            mean_val = np.mean(comp_values)
            std_val = np.std(comp_values)
            cv = std_val / (mean_val + 1e-10) if mean_val > 0 else 0
            entropy = -np.sum(comp_values * np.log(comp_values + 1e-10))
            comp_stats.append((cv, entropy))
        else:
            comp_stats.append((0, 0))
    
    # Learned weight assignment based on component characteristics
    weights = np.zeros(4)
    
    for i, (cv, entropy) in enumerate(comp_stats):
        # Higher CV suggests more discriminative power
        cv_score = min(1.0, cv * 3)
        
        # Moderate entropy is good (neither too uniform nor too peaky)
        if len(components[i][np.triu_indices(n, k=1)]) > 0:
            max_entropy = np.log(len(components[i][np.triu_indices(n, k=1)]))
            if max_entropy > 0:
                entropy_score = 1.0 - abs(entropy/max_entropy - 0.7)  # Target 0.7 normalized entropy
            else:
                entropy_score = 0.5
        else:
            entropy_score = 0.5
        
        # Instance-specific adjustments
        if i == 0:  # Edge frequency
            size_adjust = 0.8 + 0.4 * size_factor
        elif i == 1:  # Centrality
            size_adjust = 0.6 + 0.6 * size_factor
        elif i == 2:  # Distance
            size_adjust = 1.2 - 0.4 * size_factor
        else:  # MST
            size_adjust = 0.7 + 0.6 * size_factor
        
        weights[i] = (0.5 + 0.5 * cv_score) * (0.5 + 0.5 * entropy_score) * size_adjust
    
    # Normalize weights
    weights = weights / weights.sum()
    
    # ===== 10. COMBINE WITH GRAPH CONVOLUTION INSPIRATION =====
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    for i, comp in enumerate(components):
        heuristics_matrix += weights[i] * comp
    
    # Add cross-component interactions (inspired by graph convolution)
    for _ in range(2):  # Two propagation steps
        # Average of neighbors' scores (weighted by distance)
        neighbor_influence = np.zeros_like(heuristics_matrix)
        for i in range(n):
            for j in candidate_lists[i]:
                weight = 1.0 / (distance_matrix[i, j] + 1e-10)
                neighbor_influence[i, j] = weight * heuristics_matrix[j].mean()
        
        if neighbor_influence.max() > 0:
            neighbor_influence = neighbor_influence / neighbor_influence.max()
        
        heuristics_matrix = 0.8 * heuristics_matrix + 0.2 * neighbor_influence
    
    # ===== 11. INTELLIGENT SPARSIFICATION =====
    # Build edge importance from multiple perspectives
    importance_scores = np.zeros_like(heuristics_matrix)
    
    # Perspective 1: Direct heuristic score
    importance_scores += heuristics_matrix
    
    # Perspective 2: Edge stability (appearance in multiple good tours)
    tour_edge_counts = np.zeros_like(heuristics_matrix)
    for tour in tours[:min(5, len(tours))]:  # Best tours
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            tour_edge_counts[u, v] += 1
            tour_edge_counts[v, u] += 1
    
    if tour_edge_counts.max() > 0:
        tour_edge_counts = tour_edge_counts / tour_edge_counts.max()
    importance_scores += 0.3 * tour_edge_counts
    
    # Perspective 3: Structural importance (bridge edges)
    degree = np.sum(distance_matrix < np.inf, axis=1)
    bridge_scores = np.zeros_like(heuristics_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if degree[i] < degree[j]:
                bridge_scores[i, j] = 1.0 / (degree[i] + 1)
            else:
                bridge_scores[i, j] = 1.0 / (degree[j] + 1)
    bridge_scores = bridge_scores + bridge_scores.T
    
    if bridge_scores.max() > 0:
        bridge_scores = bridge_scores / bridge_scores.max()
    importance_scores += 0.2 * bridge_scores
    
    # Final sparsification
    edges_per_node = max(3, min(n-1, int(2.5 * log_n)))
    
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    for i in range(n):
        scores = importance_scores[i].copy()
        scores[i] = -np.inf
        
        # Keep top edges based on combined importance
        top_k = min(edges_per_node, n-1)
        threshold = np.partition(scores, -top_k)[-top_k]
        sparsity_mask[i] = scores >= threshold
        
        # Ensure connectivity: always keep shortest edge
        if n > 1:
            nearest = np.argsort(distance_matrix[i])[1]
            sparsity_mask[i, nearest] = True
    
    # Symmetrize and include critical edges
    sparsity_mask = sparsity_mask | sparsity_mask.T
    sparsity_mask = sparsity_mask | mst_mask
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 12. FINAL REFINEMENT =====
    # Ensure symmetry and zero diagonal
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Smooth the distribution
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 10:
        # Apply winsorization to reduce outlier effects
        q_low, q_high = np.percentile(non_zero, [5, 95])
        heuristics_matrix[heuristics_matrix > 0] = np.clip(non_zero, q_low, q_high)
    
    # Final normalization
    max_val = heuristics_matrix.max()
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
