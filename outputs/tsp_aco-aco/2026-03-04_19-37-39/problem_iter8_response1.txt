import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import ConvexHull
import warnings
warnings.filterwarnings('ignore')

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with adaptive learning and geometric intelligence.
    
    Key innovations:
    1. Multi-strategy tour sampling with adaptive strategy selection
    2. Reinforcement learning mechanism for edge weight updates
    3. Geometric clustering and hull-based structural analysis
    4. Hybrid sparsification combining greedy and connectivity constraints
    5. Dynamic parameter adaptation based on solution quality feedback
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ========== 1. ADVANCED GEOMETRIC ANALYSIS ==========
    # Extended feature detection for adaptive parameter tuning
    distances_flat = distance_matrix[~np.eye(n, dtype=bool)].flatten()
    mean_dist = np.mean(distances_flat)
    std_dist = np.std(distances_flat)
    cv = std_dist / mean_dist if mean_dist > 0 else 0.5
    
    # Higher moment analysis
    skewness = 0
    kurtosis = 0
    if len(distances_flat) > 3:
        if std_dist > 0:
            z_scores = (distances_flat - mean_dist) / std_dist
            skewness = np.mean(z_scores ** 3)
            kurtosis = np.mean(z_scores ** 4) - 3
    
    # Estimate clustering tendency via distance quartiles
    q1, q3 = np.percentile(distances_flat, [25, 75])
    iqr = q3 - q1
    cluster_tendency = iqr / mean_dist if mean_dist > 0 else 0
    
    # ========== 2. REINFORCEMENT LEARNING SETUP ==========
    # Initialize edge confidence with base heuristics
    edge_confidence = np.ones_like(distance_matrix, dtype=np.float32)
    np.fill_diagonal(edge_confidence, 0)
    
    # Learning parameters based on instance characteristics
    if n < 30:
        n_iterations = max(3, n // 5)
        n_tours_per_iter = 5
        learning_rate = 0.4
        exploration_rate = 0.3
    elif n < 100:
        n_iterations = 4
        n_tours_per_iter = max(4, n // 20)
        learning_rate = 0.3
        exploration_rate = 0.2
    else:
        n_iterations = 3
        n_tours_per_iter = max(3, n // 30)
        learning_rate = 0.2
        exploration_rate = 0.15
    
    # Adjust based on geometric features
    if cluster_tendency > 0.6:
        exploration_rate *= 1.5  # More exploration for clustered instances
    if cv < 0.3:
        learning_rate *= 0.7  # Slower learning for uniform instances
    
    # ========== 3. MULTI-STRATEGY TOUR GENERATION ==========
    def generate_tour_strategy1(start_idx: int) -> np.ndarray:
        """Nearest neighbor with edge confidence bias"""
        tour = [start_idx]
        unvisited = set(range(n)) - {start_idx}
        current = start_idx
        
        while unvisited:
            # Combine distance and confidence for selection
            candidates = list(unvisited)
            dist_scores = distance_matrix[current, candidates]
            conf_scores = edge_confidence[current, candidates]
            
            # Adaptive weighting
            if cv > 0.8:
                # For variable distances, trust confidence more
                weights = (0.3 / (dist_scores + 1e-10) + 0.7 * conf_scores)
            else:
                weights = (0.6 / (dist_scores + 1e-10) + 0.4 * conf_scores)
            
            # Epsilon-greedy exploration
            if np.random.random() < exploration_rate:
                next_node = np.random.choice(candidates)
            else:
                weights = weights / (weights.sum() + 1e-10)
                next_node = np.random.choice(candidates, p=weights)
            
            tour.append(next_node)
            unvisited.remove(next_node)
            current = next_node
        
        return np.array(tour)
    
    def generate_tour_strategy2() -> np.ndarray:
        """Greedy insertion with geometric awareness"""
        # Start with convex hull if feasible
        if n >= 4:
            try:
                points = np.random.randn(n, 2)  # Proxy coordinates for hull
                hull = ConvexHull(points)
                start_tour = list(hull.vertices)
                remaining = set(range(n)) - set(start_tour)
            except:
                start_tour = [0, np.argmax(distance_matrix[0]), 
                             np.argmax(distance_matrix[0] + distance_matrix[np.argmax(distance_matrix[0])])]
                remaining = set(range(n)) - set(start_tour)
        else:
            start_tour = [0, np.argmax(distance_matrix[0])]
            remaining = set(range(n)) - set(start_tour)
        
        # Greedy insertion with confidence-weighted cost
        while remaining:
            best_node = -1
            best_position = -1
            best_cost = np.inf
            
            for node in remaining:
                for i in range(len(start_tour)):
                    j = (i + 1) % len(start_tour)
                    a, b = start_tour[i], start_tour[j]
                    
                    # Confidence-weighted insertion cost
                    base_cost = (distance_matrix[a, node] + distance_matrix[node, b] - 
                                distance_matrix[a, b])
                    conf_boost = 1.0 - 0.5 * (edge_confidence[a, node] + edge_confidence[node, b])
                    total_cost = base_cost * conf_boost
                    
                    if total_cost < best_cost:
                        best_cost = total_cost
                        best_node = node
                        best_position = i + 1
            
            start_tour.insert(best_position, best_node)
            remaining.remove(best_node)
        
        return np.array(start_tour)
    
    def generate_tour_strategy3(start_idx: int) -> np.ndarray:
        """Randomized Christofides-like approach"""
        # Simplified MST-based approach
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = np.column_stack(np.where(mst > 0))
        
        # Build tour from MST with shortcuts
        tour = [start_idx]
        visited = set([start_idx])
        
        while len(tour) < n:
            current = tour[-1]
            # Find connected nodes in MST
            connected = [j for j in range(n) if (current, j) in map(tuple, mst_edges) or 
                        (j, current) in map(tuple, mst_edges)]
            unvisited_connected = [j for j in connected if j not in visited]
            
            if unvisited_connected:
                # Choose based on confidence and distance
                candidates = unvisited_connected
                scores = []
                for cand in candidates:
                    dist_score = 1.0 / (distance_matrix[current, cand] + 1e-10)
                    conf_score = edge_confidence[current, cand]
                    scores.append(0.6 * dist_score + 0.4 * conf_score)
                
                if np.random.random() < 0.7:
                    next_node = candidates[np.argmax(scores)]
                else:
                    next_node = np.random.choice(candidates)
            else:
                # Find nearest unvisited
                distances = distance_matrix[current].copy()
                distances[list(visited)] = np.inf
                next_node = np.argmin(distances)
            
            tour.append(next_node)
            visited.add(next_node)
        
        return np.array(tour)
    
    # ========== 4. REINFORCEMENT LEARNING LOOP ==========
    best_tour_length = np.inf
    stagnation_counter = 0
    
    for iteration in range(n_iterations):
        iteration_tours = []
        iteration_lengths = []
        
        # Generate diverse tours
        for _ in range(n_tours_per_iter):
            # Strategy selection based on iteration and instance
            strategy_choice = np.random.choice([1, 2, 3], p=[0.4, 0.3, 0.3])
            
            if strategy_choice == 1:
                start = np.random.randint(0, n)
                tour = generate_tour_strategy1(start)
            elif strategy_choice == 2:
                tour = generate_tour_strategy2()
            else:
                start = np.random.randint(0, n)
                tour = generate_tour_strategy3(start)
            
            # Simple 2-opt improvement
            improved = True
            while improved:
                improved = False
                for i in range(n - 1):
                    for j in range(i + 2, n):
                        a, b = tour[i], tour[(i + 1) % n]
                        c, d = tour[j], tour[(j + 1) % n]
                        
                        if distance_matrix[a, c] + distance_matrix[b, d] < \
                           distance_matrix[a, b] + distance_matrix[c, d] - 1e-10:
                            tour[i + 1:j + 1] = tour[i + 1:j + 1][::-1]
                            improved = True
                            break
                    if improved:
                        break
            
            # Calculate tour length
            tour_length = sum(distance_matrix[tour[i], tour[(i + 1) % n]] for i in range(n))
            iteration_tours.append(tour)
            iteration_lengths.append(tour_length)
        
        # Update edge confidence based on tour quality
        min_length = np.min(iteration_lengths)
        if min_length < best_tour_length:
            best_tour_length = min_length
            stagnation_counter = 0
        else:
            stagnation_counter += 1
        
        # Adjust learning rate based on stagnation
        current_lr = learning_rate * (0.8 ** stagnation_counter)
        
        # Update edge weights
        for tour, length in zip(iteration_tours, iteration_lengths):
            # Weight based on relative quality
            weight = np.exp(-(length - min_length) / (mean_dist + 1e-10))
            
            for i in range(n):
                j = (i + 1) % n
                u, v = tour[i], tour[j]
                edge_confidence[u, v] += current_lr * weight
                edge_confidence[v, u] += current_lr * weight
    
    # Normalize confidence scores
    if edge_confidence.max() > 0:
        edge_confidence = edge_confidence / edge_confidence.max()
    
    # ========== 5. STRUCTURAL COMPONENTS ==========
    # Inverse distance with geometric adaptation
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(inv_dist, 0)
    
    # Geometric centrality with clustering awareness
    if cluster_tendency > 0.5:
        # For clustered instances, use median-based centrality
        median_dists = np.median(distance_matrix, axis=1)
        centrality = 1 / (median_dists + 1e-10)
    else:
        # For uniform instances, use mean-based centrality
        mean_dists = np.mean(distance_matrix, axis=1)
        centrality = 1 / (mean_dists + 1e-10)
    
    centrality = centrality / centrality.sum()
    edge_potential = np.outer(centrality, centrality)
    np.fill_diagonal(edge_potential, 0)
    
    # MST component with importance weighting
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # KNN adaptation based on geometric features
    if cluster_tendency > 0.6:
        k = max(2, min(4, n // 15))
    elif cv < 0.3:
        k = max(3, min(6, n // 8))
    else:
        k = max(2, min(5, n // 10))
    
    knn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argpartition(distances, k)[:k]
        knn_mask[i, nn_indices] = True
    knn_mask = knn_mask | knn_mask.T
    
    # ========== 6. ADAPTIVE COMPONENT WEIGHTING ==========
    # Dynamic weights based on learned confidence quality
    conf_quality = np.mean(edge_confidence[knn_mask]) if np.any(knn_mask) else 0.5
    
    if conf_quality > 0.7:
        # Good confidence scores, trust them more
        w_conf = 0.6
        w_dist = 0.2
        w_cent = 0.1
        w_mst = 0.1
    elif conf_quality < 0.3:
        # Poor confidence, rely more on structural components
        w_conf = 0.2
        w_dist = 0.4
        w_cent = 0.2
        w_mst = 0.2
    else:
        # Balanced approach
        w_conf = 0.4
        w_dist = 0.3
        w_cent = 0.2
        w_mst = 0.1
    
    # Adjust for instance size
    if n > 100:
        w_conf *= 0.8
        w_mst *= 1.2
    
    # Combine components
    heuristic_scores = (w_conf * edge_confidence +
                       w_dist * inv_dist +
                       w_cent * edge_potential)
    
    # Boost structural edges
    heuristic_scores[knn_mask] *= 1.4
    heuristic_scores[mst_mask] *= 2.0
    
    # Penalize extremely long edges
    long_edge_thresh = np.percentile(distances_flat, 90)
    long_edges = distance_matrix > long_edge_thresh
    heuristic_scores[long_edges] *= 0.2
    
    # ========== 7. HYBRID SPARSIFICATION ==========
    # Phase 1: Greedy selection per node
    min_edges = max(2, int(np.log2(n + 1)))
    max_edges = min(n - 1, int(3 * np.sqrt(n)))
    
    selected_mask = np.zeros_like(heuristic_scores, dtype=bool)
    
    for i in range(n):
        # Always include nearest neighbors
        k_nn = min(3, n - 1)
        if k_nn > 0:
            nn_indices = np.argsort(distance_matrix[i])[1:k_nn + 1]
            selected_mask[i, nn_indices] = True
        
        # Include MST edges
        selected_mask[i] = selected_mask[i] | mst_mask[i]
        
        # Greedy selection of additional edges
        current_score = 0
        available = np.where(~selected_mask[i])[0]
        available = available[available != i]
        
        while len(available) > 0 and np.sum(selected_mask[i]) < max_edges:
            best_idx = available[np.argmax(heuristic_scores[i, available])]
            selected_mask[i, best_idx] = True
            available = available[available != best_idx]
    
    # Phase 2: Ensure connectivity via BFS
    def is_connected(mask):
        visited = np.zeros(n, dtype=bool)
        stack = [0]
        visited[0] = True
        
        while stack:
            node = stack.pop()
            neighbors = np.where(mask[node])[0]
            for neighbor in neighbors:
                if not visited[neighbor]:
                    visited[neighbor] = True
                    stack.append(neighbor)
        
        return np.all(visited)
    
    # Add minimal edges to ensure connectivity if needed
    if not is_connected(selected_mask):
        # Use MST edges as backbone
        selected_mask = selected_mask | mst_mask
    
    # Ensure symmetry
    selected_mask = selected_mask | selected_mask.T
    
    # Apply sparsification
    heuristic_scores = heuristic_scores * selected_mask
    
    # ========== 8. FINAL REFINEMENT ==========
    # Smooth the scores
    for _ in range(2):
        for i in range(n):
            neighbors = np.where(selected_mask[i])[0]
            if len(neighbors) > 0:
                avg_neighbor_score = np.mean(heuristic_scores[neighbors, i])
                heuristic_scores[i, neighbors] = 0.7 * heuristic_scores[i, neighbors] + \
                                               0.3 * avg_neighbor_score
    
    # Final normalization
    max_val = np.max(heuristic_scores)
    if max_val > 0:
        heuristic_scores = heuristic_scores / max_val
    
    # Zero diagonal
    np.fill_diagonal(heuristic_scores, 0)
    
    return heuristic_scores
