import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Streamlined TSP edge heuristics with adaptive sampling and efficient computation.
    
    Key Features:
    1. Efficient diverse tour sampling with adaptive method selection
    2. Fast local optimization using limited 2-opt improvements
    3. Smart sparsification preserving connectivity via MST
    4. Balanced component integration without over-engineering
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # === 1. INSTANCE ANALYSIS ===
    distances = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances)
    dist_std = np.std(distances)
    cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    
    # === 2. EFFICIENT TOUR GENERATION ===
    def generate_tour(method: str, start: int = 0) -> np.ndarray:
        """Generate initial tour using specified method."""
        if method == 'nearest':
            # Nearest neighbor from random start
            tour = [start]
            unvisited = set(range(n)) - {start}
            while unvisited:
                current = tour[-1]
                # Find nearest unvisited node
                nearest = min(unvisited, key=lambda x: distance_matrix[current, x])
                tour.append(nearest)
                unvisited.remove(nearest)
            return np.array(tour)
        
        elif method == 'random':
            return np.random.permutation(n)
        
        elif method == 'insertion':
            # Cheapest insertion heuristic
            if n <= 3:
                return np.random.permutation(n)
            
            # Start with triangle
            nodes = list(np.random.choice(n, 3, replace=False))
            tour = nodes[:3] + [nodes[0]]  # Close triangle
            
            remaining = set(range(n)) - set(tour[:-1])
            while remaining:
                best_node = None
                best_pos = -1
                best_increase = float('inf')
                
                for node in remaining:
                    # Find best position to insert
                    for i in range(len(tour)-1):
                        inc = (distance_matrix[tour[i], node] +
                               distance_matrix[node, tour[i+1]] -
                               distance_matrix[tour[i], tour[i+1]])
                        if inc < best_increase:
                            best_increase = inc
                            best_node = node
                            best_pos = i + 1
                
                tour.insert(best_pos, best_node)
                remaining.remove(best_node)
            
            return np.array(tour[:-1])
        
        return np.random.permutation(n)
    
    def fast_2opt(tour: np.ndarray, max_iter: int = 20) -> np.ndarray:
        """Apply limited 2-opt local search for efficiency."""
        n_nodes = len(tour)
        current = tour.copy()
        improved = True
        iter_count = 0
        
        while improved and iter_count < max_iter:
            improved = False
            # Random sampling of edges for efficiency
            for _ in range(min(50, n_nodes)):
                i = np.random.randint(0, n_nodes - 2)
                j = np.random.randint(i + 2, n_nodes)
                
                if j == n_nodes - 1 and i == 0:
                    continue
                
                # Check if swap improves
                a, b = current[i], current[(i + 1) % n_nodes]
                c, d = current[j], current[(j + 1) % n_nodes]
                
                if distance_matrix[a, b] + distance_matrix[c, d] > distance_matrix[a, c] + distance_matrix[b, d] + 1e-10:
                    # Perform 2-opt swap
                    current[i+1:j+1] = current[i+1:j+1][::-1]
                    improved = True
                    break
            
            iter_count += 1
        
        return current
    
    # Adaptive tour generation based on instance size
    if n <= 10:
        num_tours = min(20, 5 * n)
        methods = ['nearest'] * 6 + ['insertion'] * 3 + ['random'] * 1
    elif n <= 50:
        num_tours = min(30, n)
        methods = ['nearest'] * 5 + ['insertion'] * 3 + ['random'] * 2
    else:
        num_tours = min(40, int(n * 0.7))
        methods = ['nearest'] * 4 + ['insertion'] * 4 + ['random'] * 2
    
    # Generate diverse tours
    tours = []
    for _ in range(num_tours):
        method = np.random.choice(methods)
        start = np.random.randint(0, n) if method == 'nearest' else 0
        tour = generate_tour(method, start)
        # Apply fast 2-opt with adaptive iterations
        max_iter = min(30, max(5, int(20 * (1 - cv))))
        tour = fast_2opt(tour, max_iter)
        tours.append(tour)
    
    # === 3. EDGE FREQUENCY COMPUTATION ===
    edge_freq = np.zeros_like(distance_matrix, dtype=float)
    for tour in tours:
        for i in range(n):
            u, v = tour[i], tour[(i + 1) % n]
            edge_freq[u, v] += 1
            edge_freq[v, u] += 1
    
    if edge_freq.max() > 0:
        edge_freq /= edge_freq.max()
    
    # === 4. COMPONENT INTEGRATION ===
    # Inverse distance component (simple and effective)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist /= inv_dist.max()
    
    # MST component for connectivity
    try:
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0) | (mst.T > 0)
        mst_score = np.zeros_like(distance_matrix)
        mst_score[mst_mask] = 1 / (distance_matrix[mst_mask] + 1e-10)
        if mst_score.max() > 0:
            mst_score /= mst_score.max()
    except:
        mst_mask = np.ones_like(distance_matrix, dtype=bool)
        mst_score = np.ones_like(distance_matrix)
        np.fill_diagonal(mst_mask, False)
    
    # Dynamic weighting based on instance characteristics
    if n < 15:
        w_freq, w_dist, w_mst = 0.5, 0.3, 0.2
    elif cv < 0.3:  # Uniform distances
        w_freq, w_dist, w_mst = 0.4, 0.4, 0.2
    else:  # Varied distances
        w_freq, w_dist, w_mst = 0.5, 0.3, 0.2
    
    # Combine components
    heuristic = w_freq * edge_freq + w_dist * inv_dist + w_mst * mst_score
    
    # Boost edges appearing in multiple components
    high_freq_mask = edge_freq > np.percentile(edge_freq[edge_freq > 0], 75)
    heuristic[high_freq_mask & mst_mask] *= 1.3
    
    # === 5. SMART SPARSIFICATION ===
    if n > 8:
        # Adaptive edge retention per node
        base_edges = max(3, min(8, int(3 + np.log2(n))))
        edges_per_node = int(base_edges * (1 + 0.5 * cv))  # More edges for varied instances
        
        sparsity_mask = np.zeros_like(heuristic, dtype=bool)
        for i in range(n):
            scores = heuristic[i].copy()
            scores[i] = -np.inf
            threshold = np.partition(scores, -edges_per_node)[-edges_per_node]
            sparsity_mask[i, scores >= threshold] = True
        
        # Ensure connectivity through MST
        sparsity_mask = sparsity_mask | mst_mask
        sparsity_mask = sparsity_mask | sparsity_mask.T
        
        heuristic = heuristic * sparsity_mask
    
    # === 6. FINAL PROCESSING ===
    np.fill_diagonal(heuristic, 0)
    heuristic = 0.5 * (heuristic + heuristic.T)  # Ensure symmetry
    
    # Normalize to [0, 1]
    h_max = heuristic.max()
    if h_max > 0:
        heuristic /= h_max
    
    # Soft threshold for very small values
    heuristic[heuristic < 0.02] = 0
    
    return heuristic
