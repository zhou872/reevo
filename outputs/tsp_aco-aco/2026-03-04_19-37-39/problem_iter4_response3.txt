import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from collections import defaultdict
import random

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics v2 with Christofides-inspired MST,
    local search optimization, adaptive weight learning, and enhanced scalability.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Matrix of edge promise indicators (same shape as input), higher = more promising
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Minimum Spanning Tree (Christofides foundation)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # 2. Inverse distance with adaptive normalization
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # Robust normalization per node
    for i in range(n):
        row = inv_dist[i].copy()
        row[i] = -np.inf
        max_val = np.max(row)
        if max_val > 0:
            inv_dist[i] = row / max_val
    np.fill_diagonal(inv_dist, 0)
    
    # 3. Adaptive nearest neighbors with connectivity analysis
    k = max(2, min(15, n // 2))  # More aggressive scaling for larger instances
    nn_strength = np.zeros_like(distance_matrix)
    node_density = np.zeros(n)
    
    for i in range(n):
        # Adaptive k based on local density
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        avg_dist = np.mean(distances[distances < np.inf])
        
        # Nodes within 1.5x average distance are considered "local"
        local_mask = distances <= 1.5 * avg_dist
        node_density[i] = np.sum(local_mask)
        
        # Get k nearest neighbors
        nn_indices = np.argsort(distances)[:k]
        
        if len(nn_indices) > 0:
            # Exponential decay with rank
            for rank, j in enumerate(nn_indices):
                decay = np.exp(-rank / 3.0)
                # Adjust based on how "exclusive" this connection is
                exclusivity = 1.0 if rank == 0 else 0.7
                nn_strength[i, j] = decay * exclusivity
    
    # Make symmetric and enhance mutual connections
    nn_strength = (nn_strength + nn_strength.T) / 2
    for i in range(n):
        for j in range(i+1, n):
            if nn_strength[i, j] > 0 and nn_strength[j, i] > 0:
                nn_strength[i, j] = nn_strength[j, i] = np.max([nn_strength[i, j], nn_strength[j, i]])
    
    # 4. Local search optimization hints (k-opt inspired)
    # Precompute sorted neighbors for each node
    sorted_neighbors = []
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        sorted_idx = np.argsort(distances)
        sorted_neighbors.append(sorted_idx)
    
    opt_potential = np.zeros_like(distance_matrix)
    
    # Quick 2-opt potential estimation
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] < np.inf:
                # Look for alternative paths that might replace (i,j)
                # Check if there exists k,l such that d(i,k)+d(j,l) < d(i,j)+d(k,l)
                # We'll check a limited number of candidates
                best_alternative = distance_matrix[i, j] * 1.2  # Start with slight improvement threshold
                
                # Check top neighbors of i and j
                top_k = min(5, n-1)
                for ki in range(top_k):
                    k = sorted_neighbors[i][ki]
                    if k == j:
                        continue
                    for lj in range(top_k):
                        l = sorted_neighbors[j][lj]
                        if l == i or l == k:
                            continue
                        
                        current = distance_matrix[i, j] + distance_matrix[k, l]
                        alternative = distance_matrix[i, k] + distance_matrix[j, l]
                        
                        if alternative < current * 0.95:  # 5% improvement potential
                            # This edge might be replaceable
                            improvement = (current - alternative) / current
                            opt_potential[i, j] += improvement
                            opt_potential[j, i] += improvement
    
    # Normalize and invert (edges with low replacement potential are better)
    if np.max(opt_potential) > 0:
        opt_potential = 1.0 - (opt_potential / np.max(opt_potential))
    
    # 5. Dynamic weight learning based on instance characteristics
    # Analyze problem structure
    flat_dists = distance_matrix[np.triu_indices(n, k=1)]
    cv = np.std(flat_dists) / np.mean(flat_dists) if np.mean(flat_dists) > 0 else 0
    
    # Determine instance type
    if cv < 0.2:
        # Uniform distances
        weights = [0.25, 0.35, 0.40, 0.0]  # inv_dist, nn, mst, opt
    elif cv < 0.5:
        # Moderately varying
        weights = [0.30, 0.30, 0.30, 0.10]
    else:
        # Highly clustered or varying
        weights = [0.20, 0.25, 0.40, 0.15]
    
    # Adjust based on size
    size_factor = min(1.0, n / 100.0)
    weights = [w * (1.0 - size_factor * 0.3) for w in weights]
    weights[2] += size_factor * 0.3  # Increase MST weight for larger instances
    
    # Normalize weights
    weight_sum = sum(weights)
    weights = [w / weight_sum for w in weights]
    
    # 6. Combine components
    # MST component (boost factor based on edge criticality)
    mst_component = np.zeros_like(distance_matrix)
    mst_edges = np.argwhere(mst_mask)
    for i, j in mst_edges:
        if i != j:
            # Compute edge criticality in MST
            # For now, use degree in MST as proxy
            mst_degree_i = np.sum(mst_mask[i])
            mst_degree_j = np.sum(mst_mask[j])
            criticality = (mst_degree_i + mst_degree_j) / (2 * (n - 1))
            mst_component[i, j] = 0.5 + 0.5 * criticality
    
    # Weighted combination
    heuristic = (
        weights[0] * inv_dist +
        weights[1] * nn_strength +
        weights[2] * mst_component +
        weights[3] * opt_potential
    )
    
    # 7. Enhanced MST boost with hierarchical structure
    # Strong boost for MST edges, especially those connecting different clusters
    boost_factor = 3.0 + size_factor * 2.0  # 3-5x boost
    heuristic[mst_mask] *= boost_factor
    
    # Additional boost for MST edges that are also in top k neighbors
    mutual_mst_nn = mst_mask & (nn_strength > 0)
    heuristic[mutual_mst_nn] *= 1.5
    
    # 8. Scalable adaptive sparsification
    # Determine edges to keep per node (logarithmic scaling)
    base_edges = max(3, int(np.log2(n)) * 2)
    if n <= 20:
        keep_per_node = min(base_edges, n-1)
    elif n <= 100:
        keep_per_node = min(base_edges + 2, n//3)
    else:
        keep_per_node = min(base_edges + 4, n//4)
    
    # Create sparsity mask
    sparsity_mask = np.zeros_like(heuristic, dtype=bool)
    
    # Keep top edges per node
    for i in range(n):
        node_vals = heuristic[i].copy()
        node_vals[i] = -np.inf
        
        # Always include MST edges
        mst_neighbors = np.where(mst_mask[i])[0]
        sparsity_mask[i, mst_neighbors] = True
        
        # Include top remaining edges
        remaining_slots = keep_per_node - len(mst_neighbors)
        if remaining_slots > 0:
            # Get indices of top values (excluding already selected MST edges)
            temp_vals = node_vals.copy()
            temp_vals[mst_neighbors] = -np.inf
            top_indices = np.argsort(temp_vals)[-remaining_slots:]
            sparsity_mask[i, top_indices] = True
    
    # Ensure symmetry and connectivity
    sparsity_mask = sparsity_mask | sparsity_mask.T | mst_mask
    
    # Additional connectivity check: ensure each node connects to at least 2 others
    for i in range(n):
        connections = np.where(sparsity_mask[i])[0]
        if len(connections) < 2:
            # Add closest nodes not already connected
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            distances[connections] = np.inf
            needed = 2 - len(connections)
            closest = np.argsort(distances)[:needed]
            sparsity_mask[i, closest] = True
            sparsity_mask[closest, i] = True
    
    # Apply sparsification
    heuristic = heuristic * sparsity_mask
    
    # 9. Final refinement
    np.fill_diagonal(heuristic, 0)
    
    # Smooth with local averaging for clustered instances
    if cv > 0.4 and n > 10:
        smoothed = heuristic.copy()
        for i in range(n):
            for j in range(n):
                if i != j and heuristic[i, j] > 0:
                    # Average with neighbors' connections
                    i_neighbors = np.where(heuristic[i] > 0)[0]
                    j_neighbors = np.where(heuristic[j] > 0)[0]
                    if len(i_neighbors) > 0 and len(j_neighbors) > 0:
                        avg_i = np.mean(heuristic[i, i_neighbors]) if len(i_neighbors) > 0 else 0
                        avg_j = np.mean(heuristic[j, j_neighbors]) if len(j_neighbors) > 0 else 0
                        smoothed[i, j] = 0.7 * heuristic[i, j] + 0.15 * avg_i + 0.15 * avg_j
        heuristic = smoothed
    
    # Final normalization
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    # Ensure symmetry
    heuristic = (heuristic + heuristic.T) / 2
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
