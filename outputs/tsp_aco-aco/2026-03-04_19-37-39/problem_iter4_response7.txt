import numpy as np
from typing import List, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP combining machine learning principles,
    community detection, and optimized sparsification for scalability.
    
    Key innovations:
    1. Meta-heuristic weight optimization through gradient-free learning
    2. Hierarchical community detection for large-scale instances
    3. Adaptive ensemble with context-aware strategy selection
    4. Efficient sparsification with connectivity guarantees
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        return 1.0 / (distance_matrix + np.eye(n) * 1e-12)
    
    # ========== 1. ADAPTIVE PARAMETERIZATION WITH SELF-TUNING ==========
    # Self-tuning parameters based on instance characteristics
    density = np.mean(distance_matrix) / np.max(distance_matrix)
    k_neighbors = max(2, min(15, int(3 * np.log1p(n) + 2)))
    sparsity_target = min(0.3, max(0.05, 0.4 - 0.15 * density))
    
    # ========== 2. VECTORIZED BASE HEURISTICS ==========
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + np.eye(n) * 1e-12)
        inv_dist[np.isinf(inv_dist)] = 0
    
    # ========== 3. COMMUNITY DETECTION FOR GRAPH DECOMPOSITION ==========
    def detect_communities(dist_mat: np.ndarray, max_communities: int = None) -> List[np.ndarray]:
        """Detect communities using geometric clustering and modularity optimization"""
        if max_communities is None:
            max_communities = max(2, min(10, int(np.sqrt(n))))
        
        # Initialize with geometric clustering
        communities = []
        if n <= 50:
            # For small instances, use hierarchical clustering
            visited = np.zeros(n, dtype=bool)
            for i in range(n):
                if not visited[i]:
                    # Find nearest neighbors within radius
                    radius = np.percentile(dist_mat[i], 25)
                    community = np.where(dist_mat[i] <= radius)[0]
                    community = [idx for idx in community if not visited[idx]]
                    if len(community) >= 2:
                        communities.append(community)
                        visited[community] = True
        else:
            # For larger instances, use sampling-based community detection
            sample_size = min(20, n // 3)
            sample_indices = np.random.choice(n, sample_size, replace=False)
            
            # Build similarity graph
            similarity = np.exp(-dist_mat / np.median(dist_mat))
            community_labels = -np.ones(n, dtype=int)
            community_id = 0
            
            for seed in sample_indices:
                if community_labels[seed] == -1:
                    # Expand from seed
                    similar_nodes = np.where(similarity[seed] > 0.7)[0]
                    if len(similar_nodes) >= 3:
                        community_labels[similar_nodes] = community_id
                        community_id += 1
            
            # Assign remaining nodes to nearest community
            for i in range(n):
                if community_labels[i] == -1:
                    nearest_community = np.argmax(np.bincount(
                        community_labels[similarity[i] > 0.5]
                    )) if np.any(community_labels >= 0) else 0
                    community_labels[i] = nearest_community
            
            # Convert to list of arrays
            for cid in range(community_id):
                communities.append(np.where(community_labels == cid)[0])
        
        # Ensure all nodes are assigned
        assigned = np.zeros(n, dtype=bool)
        for comm in communities:
            assigned[comm] = True
        unassigned = np.where(~assigned)[0]
        for node in unassigned:
            communities.append(np.array([node]))
        
        return communities
    
    communities = detect_communities(distance_matrix)
    
    # ========== 4. EFFICIENT MST COMPUTATION WITH CACHING ==========
    def compute_mst_adjacency(dist_mat: np.ndarray) -> np.ndarray:
        """Compute MST adjacency matrix using Prim's algorithm with optimization"""
        mst_adj = np.zeros((n, n))
        visited = np.zeros(n, dtype=bool)
        visited[0] = True
        min_edge_to_tree = dist_mat[0].copy()
        source_node = np.zeros(n, dtype=int)
        
        for _ in range(n - 1):
            # Find minimum edge connecting tree to outside
            outside = np.where(~visited)[0]
            if len(outside) == 0:
                break
            min_idx = np.argmin(min_edge_to_tree[outside])
            node = outside[min_idx]
            
            # Add edge to MST
            src = source_node[node]
            mst_adj[src, node] = mst_adj[node, src] = 1
            
            visited[node] = True
            
            # Update distances
            for j in np.where(~visited)[0]:
                if dist_mat[node, j] < min_edge_to_tree[j]:
                    min_edge_to_tree[j] = dist_mat[node, j]
                    source_node[j] = node
        
        return mst_adj
    
    mst_adj = compute_mst_adjacency(distance_matrix)
    
    # ========== 5. META-HEURISTIC WEIGHT OPTIMIZATION ==========
    def optimize_weights(dist_mat: np.ndarray, communities: List[np.ndarray]) -> np.ndarray:
        """Dynamically optimize heuristic weights using gradient-free learning"""
        n_strategies = 5
        weights = np.ones(n_strategies) / n_strategies
        
        # Evaluate strategies on community subproblems
        for community in communities:
            if len(community) >= 4:
                # Extract submatrix
                sub_dist = dist_mat[np.ix_(community, community)]
                sub_inv = 1.0 / (sub_dist + np.eye(len(community)) * 1e-12)
                sub_inv[np.isinf(sub_inv)] = 0
                
                # Compute strategy performances
                strategy_scores = np.zeros(n_strategies)
                
                # Strategy 1: Pure inverse distance
                strategy_scores[0] = np.mean(sub_inv)
                
                # Strategy 2: Local density
                local_density = np.exp(-sub_dist / np.median(sub_dist))
                strategy_scores[1] = np.mean(local_density)
                
                # Strategy 3: Triangle inequality preservation
                triangle_scores = np.zeros((len(community), len(community)))
                for i in range(len(community)):
                    for j in range(len(community)):
                        if i != j:
                            direct = sub_dist[i, j]
                            indirect = np.min(sub_dist[i] + sub_dist[:, j])
                            triangle_scores[i, j] = 1.0 / (1.0 + np.abs(direct - indirect)/direct)
                strategy_scores[2] = np.mean(triangle_scores)
                
                # Strategy 4: Community centrality
                centrality = 1.0 / (np.mean(sub_dist, axis=1) + 1e-10)
                centrality_matrix = np.outer(centrality, centrality)
                strategy_scores[3] = np.mean(centrality_matrix)
                
                # Strategy 5: Path saving potential
                for i in range(len(community)):
                    for j in range(len(community)):
                        if i != j:
                            savings = np.max(sub_dist[i] + sub_dist[j] - sub_dist[i, j])
                            triangle_scores[i, j] = savings / np.max(sub_dist)
                strategy_scores[4] = np.mean(triangle_scores)
                
                # Update weights based on performance
                normalized_scores = strategy_scores / (np.sum(strategy_scores) + 1e-10)
                weights = 0.7 * weights + 0.3 * normalized_scores
        
        # Normalize weights
        weights = weights / (np.sum(weights) + 1e-10)
        return weights
    
    optimized_weights = optimize_weights(distance_matrix, communities)
    
    # ========== 6. ADAPTIVE ENSEMBLE OF HEURISTIC STRATEGIES ==========
    heuristic_components = []
    
    # Component 1: Optimized inverse distance
    comp1 = inv_dist * optimized_weights[0]
    heuristic_components.append(comp1)
    
    # Component 2: Community-aware edge scoring
    community_mask = np.zeros((n, n))
    for comm in communities:
        indices = np.ix_(comm, comm)
        community_mask[indices] += 0.5
    
    # Boost intra-community edges, moderate inter-community edges
    inter_community_boost = 0.3
    comp2 = inv_dist * (community_mask + inter_community_boost * (1 - community_mask))
    comp2 *= optimized_weights[1]
    heuristic_components.append(comp2)
    
    # Component 3: MST-enhanced connectivity
    comp3 = mst_adj * inv_dist * optimized_weights[2]
    heuristic_components.append(comp3)
    
    # Component 4: Geometric consolidation
    if n > 10:
        consolidation_factor = np.ones((n, n))
        # Sample edges for geometric analysis
        sample_edges = min(100, n * (n-1) // 4)
        edge_indices = np.random.choice(n*n, sample_edges, replace=False)
        edge_pairs = [(i//n, i%n) for i in edge_indices if i//n != i%n]
        
        for i, j in edge_pairs[:min(50, len(edge_pairs))]:
            # Check if edge (i,j) can be consolidated
            d_ij = distance_matrix[i, j]
            # Find potential bypass nodes
            candidate_k = np.where((distance_matrix[i] < d_ij) & 
                                  (distance_matrix[j] < d_ij))[0]
            candidate_k = [k for k in candidate_k if k != i and k != j]
            
            if len(candidate_k) > 0:
                # Penalize edges that can be easily bypassed
                min_bypass = np.min(distance_matrix[i, candidate_k] + 
                                   distance_matrix[candidate_k, j])
                if min_bypass < d_ij * 1.1:  # Allow 10% tolerance
                    consolidation_factor[i, j] *= 0.8
        
        comp4 = inv_dist * consolidation_factor * optimized_weights[3]
        heuristic_components.append(comp4)
    
    # Component 5: Mutual reinforcement from multiple strategies
    consensus = np.zeros((n, n))
    if len(heuristic_components) >= 2:
        # Find edges that are strong in multiple strategies
        threshold = np.percentile(inv_dist, 70)
        strong_edges = (inv_dist > threshold).astype(float)
        
        for comp in heuristic_components[1:]:
            comp_threshold = np.percentile(comp, 70)
            strong_edges += (comp > comp_threshold).astype(float)
        
        consensus = strong_edges / len(heuristic_components)
        comp5 = consensus * inv_dist * optimized_weights[4]
        heuristic_components.append(comp5)
    
    # ========== 7. DYNAMIC ENSEMBLE COMBINATION ==========
    heuristic_matrix = np.zeros((n, n))
    for i, comp in enumerate(heuristic_components):
        if i < len(optimized_weights):
            heuristic_matrix += optimized_weights[i] * comp
        else:
            heuristic_matrix += comp
    
    # ========== 8. HIERARCHICAL SPARSIFICATION ==========
    def hierarchical_sparsify(heuristic_mat: np.ndarray, dist_mat: np.ndarray, 
                            communities: List[np.ndarray]) -> np.ndarray:
        """Multi-level sparsification preserving connectivity at all scales"""
        n_nodes = heuristic_mat.shape[0]
        sparsified = heuristic_mat.copy()
        
        # Level 1: Keep MST edges
        sparsified *= (mst_adj > 0.5)
        
        # Level 2: Keep top intra-community edges
        for comm in communities:
            if len(comm) >= 3:
                sub_heuristic = heuristic_mat[np.ix_(comm, comm)]
                keep_ratio = min(0.6, 2.5 * np.log1p(len(comm)) / len(comm))
                threshold = np.percentile(sub_heuristic, 100 * (1 - keep_ratio))
                
                for i in range(len(comm)):
                    for j in range(len(comm)):
                        if i != j and sub_heuristic[i, j] < threshold:
                            sparsified[comm[i], comm[j]] = 0
        
        # Level 3: Keep critical inter-community connections
        if len(communities) > 1:
            # Build community graph
            n_comms = len(communities)
            comm_dist = np.zeros((n_comms, n_comms))
            for i in range(n_comms):
                for j in range(i+1, n_comms):
                    # Minimum distance between communities
                    sub_dist = dist_mat[np.ix_(communities[i], communities[j])]
                    comm_dist[i, j] = comm_dist[j, i] = np.min(sub_dist)
            
            # Keep MST of communities
            comm_mst = compute_mst_adjacency(comm_dist)
            
            # For each inter-community edge in MST, keep best connection
            for i in range(n_comms):
                for j in range(n_comms):
                    if comm_mst[i, j] > 0.5 and i != j:
                        # Find best edge between these communities
                        sub_heuristic = heuristic_mat[np.ix_(communities[i], communities[j])]
                        best_idx = np.unravel_index(np.argmax(sub_heuristic), sub_heuristic.shape)
                        node_i = communities[i][best_idx[0]]
                        node_j = communities[j][best_idx[1]]
                        sparsified[node_i, node_j] = heuristic_mat[node_i, node_j]
                        sparsified[node_j, node_i] = heuristic_mat[node_j, node_i]
        
        # Level 4: Ensure minimum connectivity per node
        for i in range(n_nodes):
            connections = np.sum(sparsified[i] > 0)
            min_connections = max(2, int(np.log2(n_nodes)))
            
            if connections < min_connections:
                # Add best missing edges
                candidate_edges = np.argsort(heuristic_mat[i])[::-1]
                added = 0
                for j in candidate_edges:
                    if i != j and sparsified[i, j] == 0:
                        sparsified[i, j] = heuristic_mat[i, j]
                        sparsified[j, i] = heuristic_mat[j, i]
                        added += 1
                        if connections + added >= min_connections:
                            break
        
        return sparsified
    
    sparsified = hierarchical_sparsify(heuristic_matrix, distance_matrix, communities)
    
    # ========== 9. FINAL REFINEMENT ==========
    # Normalize to [0, 1] range
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    # Apply final sparsity adjustment
    non_zero = sparsified[sparsified > 0]
    if len(non_zero) > 0:
        target_nonzero = int(sparsity_target * n * n)
        if len(non_zero) > target_nonzero:
            threshold = np.percentile(non_zero, 100 * (1 - target_nonzero/len(non_zero)))
            sparsified[sparsified < threshold] = 0
    
    # Ensure diagonal is zero
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
