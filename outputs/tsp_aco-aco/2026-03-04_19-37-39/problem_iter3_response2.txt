import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics combining centrality-based edge potential,
    adaptive nearest neighbors, and multi-criteria sparsification.
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Inverse distance component
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Adaptive nearest neighbor emphasis
    k = max(2, min(6, n // 3))  # Slightly more aggressive adaptation
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nn_indices = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, nn_indices] = True
    nn_mask = nn_mask | nn_mask.T  # Symmetrize
    
    # 3. Centrality-based edge potential (betweenness approximation)
    # Compute node centralities as inverse average distances
    row_sums = np.sum(distance_matrix, axis=1) - np.diag(distance_matrix)
    centrality = (n - 1) / (row_sums + 1e-10)  # Inverse average distance
    
    # Normalize and compute edge potential as product
    centrality = centrality / (np.sum(centrality) + 1e-10)
    edge_potential = np.outer(centrality, centrality)
    
    # 4. Minimum spanning tree component
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # 5. Combine factors with refined weights
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base: inverse distance with centrality modulation
    heuristics_matrix = 0.7 * inv_dist * (1 + 0.3 * edge_potential)
    
    # Boost for nearest neighbors with centrality awareness
    nn_boost = 1.4 + 0.2 * edge_potential[nn_mask]
    heuristics_matrix[nn_mask] *= nn_boost
    
    # Strong boost for MST edges
    heuristics_matrix[mst_mask] *= 2.2
    
    # 6. Multi-criteria sparsification
    # First criterion: keep top edges by inverse distance percentile
    inv_dist_rank = np.zeros_like(heuristics_matrix, dtype=bool)
    inv_dist_threshold = np.percentile(inv_dist[inv_dist > 0], 70) if n > 2 else 0
    inv_dist_rank = inv_dist > inv_dist_threshold
    
    # Second criterion: keep top edges by combined score percentile
    score_threshold = np.percentile(heuristics_matrix[heuristics_matrix > 0], 60) if n > 2 else 0
    score_rank = heuristics_matrix > score_threshold
    
    # Combine criteria
    sparsity_mask = inv_dist_rank | score_rank | mst_mask
    
    # Ensure minimum connectivity: each node has at least 2 edges
    for i in range(n):
        connected_edges = np.sum(sparsity_mask[i])
        if connected_edges < 2:
            # Add top edges from heuristics_matrix to reach minimum
            needed = 2 - connected_edges
            candidate_indices = np.argsort(heuristics_matrix[i])[::-1]
            for idx in candidate_indices:
                if idx != i and not sparsity_mask[i, idx] and needed > 0:
                    sparsity_mask[i, idx] = True
                    sparsity_mask[idx, i] = True
                    needed -= 1
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # 7. Final normalization
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
