import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic for TSP edge selection with efficient computation and adaptive elements.
    
    Features:
    1. Base inverse distance for direct edge quality.
    2. Nearest neighbor boost to favor local connections.
    3. Centrality-based factor using normalized closeness centrality.
    4. Vectorized triangle closure factor to assess edge participation in local clusters.
    5. Adaptive sparsification to retain only promising edges.
    
    Args:
        distance_matrix: Symmetric square matrix of distances between nodes.
    
    Returns:
        Heuristic matrix of same shape, with higher values indicating more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create a copy with diagonal set to infinity to avoid self-connections
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    
    # 1. Base heuristic: inverse distance
    base_heuristic = 1.0 / dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Nearest neighbor factor
    k = min(5, n - 1)  # Number of nearest neighbors to consider
    nn_factor = np.zeros((n, n))
    for i in range(n):
        nearest = np.argsort(dist[i])[:k]  # dist has diag inf, so self excluded
        nn_factor[i, nearest] = 1.0
        nn_factor[nearest, i] = 1.0  # Ensure symmetry
    
    # 3. Centrality factor: closeness centrality approximation
    centrality = np.zeros(n)
    for i in range(n):
        valid_dists = dist[i][dist[i] < np.inf]  # Exclude inf (self)
        if len(valid_dists) > 0:
            centrality[i] = 1.0 / np.mean(valid_dists)
    if np.max(centrality) > 0:
        centrality /= np.max(centrality)  # Normalize to [0, 1]
    betweenness_factor = np.outer(centrality, centrality)
    np.fill_diagonal(betweenness_factor, 0.0)
    
    # 4. Triangle closure factor: vectorized computation for efficiency
    triangle_factor = np.zeros((n, n))
    if n >= 3:
        # Compute per-node distance thresholds (25th percentile)
        dist_for_thresh = distance_matrix.copy()
        np.fill_diagonal(dist_for_thresh, np.inf)
        thresholds = np.array([np.percentile(dist_for_thresh[i], 25) for i in range(n)])
        
        # Boolean matrix indicating close nodes (distance < threshold, excluding self)
        close_indicator = distance_matrix < thresholds[:, None]
        np.fill_diagonal(close_indicator, False)
        
        # Count common close nodes for each edge via matrix multiplication
        close_indicator_float = close_indicator.astype(float)
        common_close = np.dot(close_indicator_float, close_indicator_float.T)
        triangle_factor = common_close / (n - 2)  # Normalize by max possible
        np.fill_diagonal(triangle_factor, 0.0)
    
    # 5. Combine factors with fixed weights (tuned for balance)
    weights = {'base': 2.0, 'nn': 1.5, 'betweenness': 1.0, 'triangle': 0.8}
    combined = (
        weights['base'] * base_heuristic +
        weights['nn'] * nn_factor +
        weights['betweenness'] * betweenness_factor +
        weights['triangle'] * triangle_factor
    )
    
    # 6. Adaptive sparsification: keep top edges and ensure nearest neighbors are retained
    if n > 2:
        upper_tri = combined[np.triu_indices(n, k=1)]
        if len(upper_tri) > 0:
            threshold = np.percentile(upper_tri, 60)  # Keep top 40% of edges
            promising_mask = (combined >= threshold) | (nn_factor > 0)
            result = np.zeros_like(combined)
            result[promising_mask] = combined[promising_mask]
            result = np.maximum(result, result.T)  # Ensure symmetry
            return result
    
    return combined

    cities = np.array([[0, 0], [1, 0], [1, 1], [0, 1]])
