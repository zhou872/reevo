import numpy as np
from scipy.sparse import csr_matrix
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:

    4. Efficient edge importance sampling via importance-weighted updates

        distance_matrix: Square matrix of pairwise distances (n x n)

        Matrix of edge promise indicators (same shape as input)

        compute_strategy = "full"

    elif n <= 1000:

    # ==================== 2. HIERARCHICAL GRAPH SAMPLING ====================
    
    def hierarchical_node_sampling(nodes: np.ndarray, target_size: int) -> np.ndarray:
        """
        Hierarchical sampling with variance control and structural preservation.
        Uses multi-resolution sampling to preserve both local and global structure.
        """
        if len(nodes) <= target_size:
            return nodes
            
        # Multi-resolution sampling strategy
        selected = []
        
        # Phase 1: Seed nodes - ensure spatial coverage
        n_seeds = max(2, target_size // 3)
        
        # Use k-means++ like initialization for seeds
        selected.append(np.random.choice(nodes))
        
        for _ in range(1, n_seeds):
            # Compute minimum distance to selected nodes
            min_distances = np.full(len(nodes), np.inf)
            for i, node in enumerate(nodes):
                if node not in selected:
                    dist_to_selected = distance_matrix[node, selected].min()
                    min_distances[i] = dist_to_selected
            
            # Weight by distance^2
            probs = min_distances ** 2
            probs[~np.isfinite(probs)] = 0
            if probs.sum() == 0:
                probs = np.ones(len(probs))
            probs = probs / probs.sum()
            
            next_idx = np.random.choice(len(nodes), p=probs)
            selected.append(nodes[next_idx])
        
        # Phase 2: Structural nodes - preserve graph connectivity
        n_structural = min(target_size - n_seeds, len(nodes) - n_seeds)
        
        if n_structural > 0:
            # Sample based on local structural importance
            structural_scores = np.zeros(len(nodes))
            remaining_nodes = [node for node in nodes if node not in selected]
            
            # Simple centrality measure for remaining nodes
            for i, node in enumerate(remaining_nodes):
                # Local degree approximation
                k_neighbors = max(2, min(5, n // 20))
                dist_to_node = distance_matrix[node]
                close_neighbors = np.argsort(dist_to_node)[1:k_neighbors+1]
                structural_scores[i] = np.sum(1 / (dist_to_node[close_neighbors] + 1e-10))
            
            if structural_scores.sum() > 0:
                probs = structural_scores / structural_scores.sum()
                structural_indices = np.random.choice(
                    len(remaining_nodes), 
                    size=n_structural, 
                    p=probs, 
                    replace=False
                )
                selected.extend([remaining_nodes[i] for i in structural_indices])
            else:
                # Fallback random sampling
                remaining = [node for node in nodes if node not in selected]
                additional = np.random.choice(remaining, min(n_structural, len(remaining)), replace=False)
                selected.extend(additional)
        
        return np.array(selected)
    
    # ==================== 3. DIVERSE LOCAL SEARCH PORTFOLIO ====================
    
    def tour_length(tour: np.ndarray) -> float:
        """Calculate length of a tour."""
        return sum(distance_matrix[tour[i], tour[(i+1)%n]] for i in range(n))
    
    def generate_initial_tour(method: str, start_node: int = None) -> np.ndarray:
        """Generate initial tour using various methods."""
        if start_node is None:
            start_node = np.random.randint(0, n)
        
        if method == 'nearest_neighbor':
            tour = [start_node]
            unvisited = set(range(n)) - {start_node}
            while unvisited:
                current = tour[-1]
                nearest = min(unvisited, key=lambda x: distance_matrix[current, x])
                tour.append(nearest)
                unvisited.remove(nearest)
            return np.array(tour)
            
        elif method == 'random_insertion':
            tour = [start_node, np.argmin(distance_matrix[start_node])]
            unvisited = set(range(n)) - set(tour)
            while unvisited:
                # Choose random unvisited node
                node = np.random.choice(list(unvisited))
                
                # Find best insertion position
                best_pos = 0
                best_increase = float('inf')
                
                for i in range(len(tour)):
                    j = (i + 1) % len(tour)
                    increase = (distance_matrix[tour[i], node] + 
                               distance_matrix[node, tour[j]] - 
                               distance_matrix[tour[i], tour[j]])
                    if increase < best_increase:
                        best_increase = increase
                        best_pos = i + 1
                
                tour.insert(best_pos, node)
                unvisited.remove(node)
            
            return np.array(tour)
            
        elif method == 'cheapest_insertion':
            tour = [start_node]
            # Start with closest node
            dist_to_start = distance_matrix[start_node].copy()
            dist_to_start[start_node] = np.inf
            tour.append(np.argmin(dist_to_start))
            unvisited = set(range(n)) - set(tour)
            
            while unvisited:
                best_node = None
                best_pos = -1
                best_increase = float('inf')
                
                for node in unvisited:
                    for i in range(len(tour)):
                        j = (i + 1) % len(tour)
                        increase = (distance_matrix[tour[i], node] + 
                                   distance_matrix[node, tour[j]] - 
                                   distance_matrix[tour[i], tour[j]])
                        if increase < best_increase:
                            best_increase = increase
                            best_node = node
                            best_pos = i + 1
                
                tour.insert(best_pos, best_node)
                unvisited.remove(best_node)
            
            return np.array(tour)
            
        else:  # Random tour
            return np.random.permutation(n)
    
    def two_opt_swap(tour: np.ndarray, i: int, j: int) -> np.ndarray:
        """Perform 2-opt swap between positions i and j."""
        new_tour = tour.copy()
        if i < j:
            new_tour[i+1:j+1] = tour[i+1:j+1][::-1]
        else:
            # Handle wrap-around case
            segment = np.concatenate([tour[i+1:], tour[:j+1]])
            segment = segment[::-1]
            new_tour[i+1:] = segment[:len(tour)-i-1]
            new_tour[:j+1] = segment[len(tour)-i-1:]
        return new_tour
    
    def three_opt_swap(tour: np.ndarray, i: int, j: int, k: int) -> np.ndarray:
        """Perform 3-opt swap."""
        # There are 7 possible reconnections for 3-opt
        # We'll implement the 4 most common ones
        a, b = tour[i], tour[(i+1)%n]
        c, d = tour[j], tour[(j+1)%n]
        e, f = tour[k], tour[(k+1)%n]
        
        current = distance_matrix[a, b] + distance_matrix[c, d] + distance_matrix[e, f]
        best_tour = tour
        best_gain = 0
        
        # Try different reconnections
        # Type 1: 2-opt on [i+1, j] and [j+1, k]
        new1 = distance_matrix[a, c] + distance_matrix[b, e] + distance_matrix[d, f]
        if new1 - current < best_gain:
            best_gain = new1 - current
            new_tour = tour.copy()
            new_tour[i+1:j+1] = tour[i+1:j+1][::-1]
            new_tour[j+1:k+1] = tour[j+1:k+1][::-1]
            best_tour = new_tour
        
        # Type 2: 2-opt on [i+1, k] and reverse [j+1, k]
        new2 = distance_matrix[a, e] + distance_matrix[d, b] + distance_matrix[c, f]
        if new2 - current < best_gain:
            best_gain = new2 - current
            new_tour = tour.copy()
            new_tour[i+1:k+1] = tour[i+1:k+1][::-1]
            new_tour[j+1:k+1] = tour[j+1:k+1][::-1]  # Double reverse restores part
            best_tour = new_tour
        
        return best_tour if best_gain < -1e-10 else tour
    
    def or_opt_move(tour: np.ndarray, start: int, length: int) -> np.ndarray:
        """OR-opt move: relocate a segment of specified length."""
        if length >= n-1:
            return tour
        
        end = (start + length) % n
        segment = []
        if start < end:
            segment = tour[start:end]
            remaining = np.concatenate([tour[:start], tour[end:]])
        else:
            segment = np.concatenate([tour[start:], tour[:end]])
            remaining = tour[end:start]
        
        # Try all insertion positions
        best_tour = tour
        best_length = tour_length(tour)
        
        for insert_pos in range(len(remaining)+1):
            new_tour = np.concatenate([
                remaining[:insert_pos],
                segment,
                remaining[insert_pos:]
            ])
            new_len = tour_length(new_tour)
            if new_len < best_length:
                best_length = new_len
                best_tour = new_tour
        
        return best_tour
    
    def local_search(tour: np.ndarray, method: str, budget: int = None) -> np.ndarray:
        """Apply specified local search method."""
        if budget is None:
            budget = max(100, n * 2)
        
        current_tour = tour.copy()
        current_len = tour_length(current_tour)
        improved = True
        iterations = 0
        
        if method == '2opt':
            while improved and iterations < budget:
                improved = False
                for i in range(n-1):
                    for j in range(i+2, n):
                        if j == n-1 and i == 0:
                            continue
                        
                        a, b = current_tour[i], current_tour[(i+1)%n]
                        c, d = current_tour[j], current_tour[(j+1)%n]
                        
                        current = distance_matrix[a, b] + distance_matrix[c, d]
                        new = distance_matrix[a, c] + distance_matrix[b, d]
                        
                        if new < current - 1e-10:
                            current_tour = two_opt_swap(current_tour, i, j)
                            current_len = current_len - current + new
                            improved = True
                            break
                    if improved:
                        break
                iterations += 1
        
        elif method == '3opt':
            while improved and iterations < budget:
                improved = False
                # Limited 3-opt search for efficiency
                for i in range(0, n-5, max(1, n//20)):
                    for j in range(i+2, n-3, max(1, n//20)):
                        for k in range(j+2, n-1, max(1, n//20)):
                            if k == n-1 and i == 0:
                                continue
                            
                            new_tour = three_opt_swap(current_tour, i, j, k)
                            new_len = tour_length(new_tour)
                            
                            if new_len < current_len - 1e-10:
                                current_tour = new_tour
                                current_len = new_len
                                improved = True
                                break
                        if improved:
                            break
                    if improved:
                        break
                iterations += 1
        
        elif method == 'or_opt':
            segment_lengths = [1, 2, 3] if n > 10 else [1, 2]
            while improved and iterations < budget:
                improved = False
                for length in segment_lengths:
                    for i in range(n):
                        new_tour = or_opt_move(current_tour, i, length)
                        new_len = tour_length(new_tour)
                        if new_len < current_len - 1e-10:
                            current_tour = new_tour
                            current_len = new_len
                            improved = True
                            break
                    if improved:
                        break
                iterations += 1
        
        elif method == 'double_bridge':
            # Double-bridge kick for diversification
            if n >= 8:
                i, j, k, l = sorted(np.random.choice(n, 4, replace=False))
                # Perform double-bridge move
                new_tour = np.concatenate([
                    current_tour[:i+1],
                    current_tour[k+1:l+1],
                    current_tour[j+1:k+1],
                    current_tour[i+1:j+1],
                    current_tour[l+1:]
                ])
                current_tour = new_tour
        
        return current_tour

    # Initialize heuristic factors

    elif n > 500:
        base_weights = np.array([0.10, 0.10, 0.25, 0.30, 0.08, 0.10, 0.07])

    if uniformity > 0.8:
        base_weights[2] += 0.05  # Boost triangle completion
        base_weights[0] -= 0.03  # Reduce inverse distance
    
    base_weights = np.clip(base_weights, 0.02, 0.35)
    base_weights = base_weights / base_weights.sum()

    # ==================== 5. HEURISTIC FACTOR COMPUTATION ====================
    
    def compute_heuristic_factors(sample_nodes: np.ndarray = None) -> List[np.ndarray]:
        """Compute all heuristic factors, optionally on sampled subgraph."""
        if sample_nodes is None:
            nodes = np.arange(n)
            sub_n = n
            sub_dist = distance_matrix
        else:
            nodes = sample_nodes
            sub_n = len(nodes)
            sub_dist = distance_matrix[np.ix_(nodes, nodes)]
        
        factors = []
        
        # 5a. Inverse distance factor (with adaptive scaling)
        with np.errstate(divide='ignore', invalid='ignore'):
            inv_dist = 1.0 / sub_dist
        np.fill_diagonal(inv_dist, 0)
        
        # Adaptive power scaling based on distance distribution
        power = 0.7 + 0.3 * np.tanh(1 - uniformity)
        inv_dist = np.power(inv_dist, power)
        
        row_sums = inv_dist.sum(axis=1, keepdims=True)
        row_sums[row_sums == 0] = 1
        inv_dist_norm = inv_dist / row_sums
        
        if sample_nodes is not None:
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = inv_dist_norm
            factors.append(full_factor)
        else:
            factors.append(inv_dist_norm)
        
        # 5b. Nearest neighbor factor (adaptive k)
        k_nn = max(2, min(15, sub_n // 4))
        if uniformity < 0.5:
            k_nn = max(k_nn, sub_n // 3)  # More neighbors for non-uniform
        
        nn_factor = np.zeros((sub_n, sub_n))
        if k_nn > 0:
            partitioned = np.argpartition(sub_dist, k_nn+1, axis=1)[:, :k_nn+1]
            for i in range(sub_n):
                neighbors = partitioned[i][partitioned[i] != i][:k_nn]
                if len(neighbors) > 0:
                    ranks = np.arange(len(neighbors))
                    weights = np.exp(-ranks / (1.5 + 0.5 * uniformity))
                    nn_factor[i, neighbors] = weights
        
        nn_factor = np.maximum(nn_factor, nn_factor.T)
        if sample_nodes is not None:
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = nn_factor
            factors.append(full_factor)
        else:
            factors.append(nn_factor)
        
        # 5c. Triangle completion factor
        triangle_factor = np.zeros((sub_n, sub_n))
        if sub_n >= 4:
            # Adaptive threshold based on distance distribution
            threshold_percentile = 30 + 20 * uniformity  # Higher threshold for uniform instances
            percentiles = np.percentile(sub_dist, threshold_percentile, axis=1, keepdims=True)
            short_edges = sub_dist <= percentiles
            np.fill_diagonal(short_edges, False)
            
            common_short = short_edges @ short_edges.T
            max_possible = sub_n - 2
            if max_possible > 0:
                triangle_factor = common_short.astype(float) / max_possible
                triangle_factor = triangle_factor * inv_dist_norm
        
        if sample_nodes is not None:
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = triangle_factor
            factors.append(full_factor)
        else:
            factors.append(triangle_factor)
        
        # 5d. MST-based factor
        mst_factor = np.zeros((sub_n, sub_n))
        if sub_n <= 500 or sample_nodes is not None:
            try:
                mst = minimum_spanning_tree(csr_matrix(sub_dist)).toarray()
                mst_mask = (mst > 0) | (mst.T > 0)
                mst_factor[mst_mask] = 1.0
            except:
                # Fallback: use nearest neighbors as MST approximation
                for i in range(sub_n):
                    neighbors = np.argsort(sub_dist[i])[1:3]
                    mst_factor[i, neighbors] = 0.8
        else:
            # Approximate MST for large instances
            for i in range(sub_n):
                neighbors = np.argsort(sub_dist[i])[1:min(4, sub_n//10)]
                mst_factor[i, neighbors] = 0.7
        
        mst_factor = np.maximum(mst_factor, mst_factor.T)
        
        if sample_nodes is not None:
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = mst_factor
            factors.append(full_factor)
        else:
            factors.append(mst_factor)
        
        # 5e. Christofides-inspired factor
        christofides_factor = np.zeros((sub_n, sub_n))
        if sub_n >= 4:
            mst_degrees = (mst_factor > 0.5).sum(axis=1)
            odd_nodes = np.where(mst_degrees % 2 == 1)[0]
            
            if len(odd_nodes) >= 2:
                odd_dist = sub_dist[np.ix_(odd_nodes, odd_nodes)]
                k_odd = min(3, len(odd_nodes)-1)
                if k_odd > 0:
                    for idx, i in enumerate(odd_nodes):
                        odd_indices = np.argsort(odd_dist[idx])
                        selected_odds = odd_indices[1:k_odd+1]
                        original_odds = odd_nodes[selected_odds]
                        for rank, j in enumerate(original_odds):
                            christofides_factor[i, j] = np.exp(-rank / (1.0 + uniformity))
        
        christofides_factor = np.maximum(christofides_factor, christofides_factor.T)
        
        if sample_nodes is not None:
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = christofides_factor
            factors.append(full_factor)
        else:
            factors.append(christofides_factor)
        
        # 5f. Node centrality factor
        centrality_factor = np.zeros((sub_n, sub_n))
        # Use harmonic centrality for better robustness
        with np.errstate(divide='ignore', invalid='ignore'):
            inv_dist_sum = np.sum(1.0 / (sub_dist + np.eye(sub_n) * 1e-10), axis=1)
        inv_dist_sum[~np.isfinite(inv_dist_sum)] = 0
        node_centrality = inv_dist_sum / (sub_n - 1)
        centrality_factor = np.outer(node_centrality, node_centrality)
        
        if sample_nodes is not None:
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = centrality_factor
            factors.append(full_factor)
        else:
            factors.append(centrality_factor)
        
        # 5g. Edge betweenness approximation
        betweenness_factor = np.zeros((sub_n, sub_n))
        if sub_n >= 4:
            # Simple approximation using random paths
            n_paths = min(100, sub_n * 2)
            for _ in range(n_paths):
                src, tgt = np.random.choice(sub_n, 2, replace=False)
                # Greedy path construction
                path = [src]
                current = src
                visited = set([src])
                
                while current != tgt and len(visited) < sub_n:
                    mask = np.ones(sub_n, dtype=bool)
                    mask[list(visited)] = False
                    mask[tgt] = True
                    
                    if not np.any(mask):
                        break
                    
                    # Choose next node balancing distance to current and target
                    scores = 0.6 * sub_dist[current] + 0.4 * sub_dist[tgt]
                    scores[~mask] = np.inf
                    next_node = np.argmin(scores)
                    
                    path.append(next_node)
                    visited.add(next_node)
                    current = next_node
                
                # Add tgt if not already in path
                if path[-1] != tgt:
                    path.append(tgt)
                
                # Update betweenness counts
                for i in range(len(path)-1):
                    u, v = path[i], path[i+1]
                    betweenness_factor[u, v] += 1
                    betweenness_factor[v, u] += 1
            
            # Normalize
            if betweenness_factor.max() > 0:
                betweenness_factor = betweenness_factor / betweenness_factor.max()
        
        if sample_nodes is not None:
            full_factor = np.zeros((n, n))
            idx = np.ix_(nodes, nodes)
            full_factor[idx] = betweenness_factor
            factors.append(full_factor)
        else:
            factors.append(betweenness_factor)
        
        return factors
    
    # ==================== 6. SOLUTION GENERATION AND WEIGHT LEARNING ====================
    
    # Generate diverse initial tours
    n_tours = min(local_search_budget, max(10, n // 5))
    tours = []
    tour_lengths = []
    
    # Portfolio of local search methods with adaptive allocation
    ls_methods = ['2opt', '3opt', 'or_opt', 'double_bridge']
    ls_weights = [0.4, 0.3, 0.2, 0.1]

    if n > 200:

    elif n < 30:
        ls_weights = [0.3, 0.4, 0.2, 0.1]  # More 3-opt for smaller instances
    
    for tour_idx in range(n_tours):
        # Mix tour generation methods
        if tour_idx < n_tours * 0.4:
            init_method = 'nearest_neighbor'
        elif tour_idx < n_tours * 0.7:
            init_method = 'cheapest_insertion'
        else:
            init_method = 'random_insertion'
        
        start_node = np.random.randint(0, n)
        tour = generate_initial_tour(init_method, start_node)
        
        # Apply local search with method selection
        ls_method = np.random.choice(ls_methods, p=ls_weights)
        budget_factor = 0.5 + 0.5 * (tour_idx / n_tours)  # Increase budget for later tours
        
        if n <= 100:
            ls_budget = max(50, int(100 * budget_factor))
        elif n <= 500:
            ls_budget = max(20, int(50 * budget_factor))
        else:
            ls_budget = max(10, int(20 * budget_factor))
        
        tour = local_search(tour, ls_method, ls_budget)
        
        # Optional: apply a second local search with different method
        if tour_idx < n_tours * 0.3 and n <= 200:
            second_method = np.random.choice([m for m in ls_methods if m != ls_method])
            tour = local_search(tour, second_method, ls_budget // 2)
        
        tours.append(tour)
        tour_lengths.append(tour_length(tour))
    
    # Select elite tours for weight learning
    elite_ratio = 0.3 + 0.2 * uniformity  # More elite tours for uniform instances
    n_elite = max(3, int(n_tours * elite_ratio))
    elite_indices = np.argsort(tour_lengths)[:n_elite]
    elite_tours = [tours[i] for i in elite_indices]
    
    # Create elite edge mask
    elite_edge_mask = np.zeros((n, n), dtype=bool)
    elite_edge_weights = np.zeros((n, n))
    
    for tour in elite_tours:
        tour_len = tour_length(tour)
        weight = 1.0 / (tour_len + 1e-10)  # Better tours get higher weight
        
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            elite_edge_mask[u, v] = True
            elite_edge_mask[v, u] = True
            elite_edge_weights[u, v] += weight
            elite_edge_weights[v, u] += weight
    
    # Normalize elite edge weights
    if elite_edge_weights.sum() > 0:
        elite_edge_weights = elite_edge_weights / elite_edge_weights.sum()
    
    # ==================== 7. META-LEARNING WEIGHT UPDATE ====================
    
    # Compute factors (with sampling for large instances)
    if compute_strategy == "full":
        factors = compute_heuristic_factors()
    elif compute_strategy == "balanced":
        sample_size = min(200, max(50, n // 5))
        sample_nodes = hierarchical_node_sampling(np.arange(n), sample_size)
        factors = compute_heuristic_factors(sample_nodes)
    else:  # "sampled"
        sample_size = min(100, max(30, n // 10))
        sample_nodes = hierarchical_node_sampling(np.arange(n), sample_size)
        factors = compute_heuristic_factors(sample_nodes)
    
    # Normalize factors
    normalized_factors = []
    for factor in factors:
        f_max = factor.max()
        if f_max > 0:
            normalized_factors.append(factor / f_max)
        else:
            normalized_factors.append(factor)
    
    # Compute factor scores using importance-weighted sampling
    factor_scores = np.zeros(n_factors)
    total_weight = 0
    
    if elite_edge_mask.any():
        # Sample edges according to their importance weights
        elite_edges = np.where(elite_edge_mask)
        n_edge_samples = min(1000, len(elite_edges[0]))
        
        if len(elite_edges[0]) > 0:
            # Importance sampling based on elite_edge_weights
            edge_indices = np.arange(len(elite_edges[0]))
            sample_weights = elite_edge_weights[elite_edges]
            sample_weights = sample_weights / sample_weights.sum()
            
            sampled_indices = np.random.choice(
                edge_indices, 
                size=n_edge_samples, 
                p=sample_weights,
                replace=True
            )
            
            # Evaluate factors on sampled edges
            for i, factor in enumerate(normalized_factors):
                sampled_values = factor[elite_edges[0][sampled_indices], 
                                       elite_edges[1][sampled_indices]]
                factor_scores[i] = sampled_values.mean()
    
    # Adaptive learning rate adjustment
    if factor_scores.sum() > 0:
        normalized_scores = factor_scores / factor_scores.sum()
        
        # Adjust learning rate based on score variance
        score_variance = np.var(normalized_scores)
        adaptive_lr = learning_rate * (0.5 + 0.5 * np.tanh(1 - score_variance * 5))
        
        # Momentum update
        velocity = momentum * velocity + adaptive_lr * (normalized_scores - base_weights)
        base_weights = base_weights + velocity
        
        # Clamp and normalize
        base_weights = np.clip(base_weights, 0.02, 0.35)
        base_weights = base_weights / base_weights.sum()
    
    # ==================== 8. FACTOR COMBINATION AND SPARSIFICATION ====================
    
    # Combine factors with learned weights
    combined = np.zeros((n, n))
    for w, factor in zip(base_weights, normalized_factors):
        combined += w * factor
    
    # Ensure MST connectivity is preserved
    if compute_strategy != "sampled" or n <= 1000:
        try:
            mst = minimum_spanning_tree(csr_matrix(distance_matrix)).toarray()
            mst_mask = (mst > 0) | (mst.T > 0)
            combined[mst_mask] *= 1.5  # Boost MST edges
        except:
            pass
    
    # Adaptive sparsification
    if n > 10:
        # Dynamic edges per node based on instance characteristics
        if uniformity > 0.7:
            edges_per_node = max(3, min(8, int(4 + np.log2(n))))
        else:
            edges_per_node = max(4, min(12, int(5 + np.log2(n) * 1.5)))
        
        sparsity_mask = np.zeros_like(combined, dtype=bool)
        
        for i in range(n):
            node_scores = combined[i].copy()
            node_scores[i] = -np.inf
            
            n_select = min(edges_per_node, n-1)
            top_indices = np.argsort(node_scores)[-n_select:]
            sparsity_mask[i, top_indices] = True
        
        # Ensure symmetry and minimum connectivity
        sparsity_mask = sparsity_mask | sparsity_mask.T
        
        for i in range(n):
            connections = np.sum(sparsity_mask[i])
            if connections < 3:  # Ensure minimum degree 3
                candidate_scores = combined[i].copy()
                candidate_scores[i] = -np.inf
                candidate_scores[sparsity_mask[i]] = -np.inf
                
                needed = 3 - connections
                if needed > 0:
                    best_indices = np.argsort(candidate_scores)[-needed:]
                    for idx in best_indices:
                        if candidate_scores[idx] > -np.inf:
                            sparsity_mask[i, idx] = True
                            sparsity_mask[idx, i] = True
        
        # Apply sparsification
        result = np.zeros_like(combined)
        result[sparsity_mask] = combined[sparsity_mask]
    else:
        result = combined
    
    # ==================== 9. FINAL POST-PROCESSING ====================
    
    # Final symmetrization
    result = (result + result.T) / 2
    np.fill_diagonal(result, 0)
    
    # Adaptive sigmoid scaling
    max_val = result.max()
    if max_val > 0:
        scaled = result / max_val
        # Parameterized sigmoid for contrast adjustment
        alpha = 4.0 + 2.0 * uniformity  # More contrast for uniform instances
        result = 1 / (1 + np.exp(-alpha * (scaled - 0.5)))
    
    # Normalize to [0, 1]
    result_min, result_max = result.min(), result.max()
    if result_max - result_min > 1e-10:
        result = (result - result_min) / (result_max - result_min)
    
    # Apply soft threshold
    threshold = 0.02 + 0.03 * (1 - uniformity)  # Lower threshold for non-uniform
    result[result < threshold] = 0
    
    return result
