import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors
    to determine edge promise and sparsifying unpromising connections.
    """
    n = distance_matrix.shape[0]
    
    # Initialize base heuristics matrix
    with np.errstate(divide='ignore', invalid='ignore'):
        # Inverse distance as base (like v1)
        base_heuristic = 1 / distance_matrix
        base_heuristic[np.isinf(base_heuristic)] = 0
        base_heuristic[np.isnan(base_heuristic)] = 0
    
    # Factor 1: Local connectivity strength
    # For each node, consider its k-nearest neighbors
    k = max(2, min(5, n // 4))  # Adaptive k based on problem size
    
    local_strength = np.zeros_like(distance_matrix)
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        distances = distance_matrix[i].copy()
        distances[i] = np.inf  # Exclude self
        nearest_indices = np.argsort(distances)[:k]
        
        # Assign higher value to closer connections
        for j in nearest_indices:
            local_strength[i, j] = 1 - (distances[j] / distances[nearest_indices[-1]])
    
    # Factor 2: Edge value aggregation
    # Consider how good an edge is relative to both endpoint alternatives
    edge_value = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Compare to average distance from i and to j
                avg_from_i = np.mean(distance_matrix[i, [k for k in range(n) if k != i]])
                avg_to_j = np.mean(distance_matrix[[k for k in range(n) if k != j], j])
                
                # Edge gets higher value if it's better than average connections
                if distance_matrix[i, j] < 0.5 * (avg_from_i + avg_to_j):
                    edge_value[i, j] = 1
                elif distance_matrix[i, j] < 0.8 * (avg_from_i + avg_to_j):
                    edge_value[i, j] = 0.5
                else:
                    edge_value[i, j] = 0.1
    
    # Factor 3: Competitive edge ranking
    # Edges that are competitive in both directions
    competitive = np.zeros_like(distance_matrix)
    for i in range(n):
        # For node i, rank edges by quality
        row_distances = distance_matrix[i].copy()
        row_distances[i] = np.inf
        row_rank = np.argsort(row_distances)
        
        # For each potential target j, check if edge (i,j) is good from j's perspective too
        for rank_pos, j in enumerate(row_rank):
            if j == i:
                continue
                
            # Check how edge (j,i) ranks from j's perspective
            col_distances = distance_matrix[j].copy()
            col_distances[j] = np.inf
            col_rank = np.argsort(col_distances)
            col_pos = np.where(col_rank == i)[0][0]
            
            # Both ranks matter - edges good from both sides are promising
            avg_rank = (rank_pos + col_pos) / 2
            max_rank = n - 1
            competitive[i, j] = 1 - (avg_rank / max_rank)
    
    # Combine factors with weighted aggregation
    combined = (
        0.4 * base_heuristic +
        0.25 * local_strength +
        0.20 * edge_value +
        0.15 * competitive
    )
    
    # Sparsification: Only keep top connections
    # For each node, keep connections to at most m other nodes
    m = max(3, min(8, n // 3))  # Adaptive m
    
    sparsified = np.zeros_like(combined)
    for i in range(n):
        # Get indices of top m connections (excluding self)
        node_connections = combined[i].copy()
        node_connections[i] = -np.inf  # Exclude self
        top_indices = np.argsort(node_connections)[-m:]
        
        # Only keep top m connections, zero out others
        for j in top_indices:
            sparsified[i, j] = combined[i, j]
    
    # Symmetrize the matrix (TSP is symmetric)
    sparsified = (sparsified + sparsified.T) / 2
    
    # Ensure diagonal is zero
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
