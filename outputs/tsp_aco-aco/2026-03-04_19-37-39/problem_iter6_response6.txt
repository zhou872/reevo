import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics using adaptive multi-resolution analysis
    with learned structural priors and probabilistic edge scoring.
    
    Key innovations:
    1. Multi-resolution analysis: Combine global MST structure with local clustering patterns
    2. Learned fusion: Adaptive combination weights based on instance characteristics
    3. Probabilistic scoring: Convert heuristics to edge selection probabilities
    4. Structural priors: Incorporate degree constraints and triangle inequality information
    
    Args:
        distance_matrix: Square symmetric matrix of pairwise distances
        
    Returns:
        Matrix of edge promise indicators where higher values suggest better edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # ========== 1. INSTANCE CHARACTERIZATION ==========
    # Compute instance characteristics for adaptive behavior
    triu_idx = np.triu_indices(n, k=1)
    distances_flat = distance_matrix[triu_idx]
    
    # Statistical features
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    dist_cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    dist_skew = np.mean((distances_flat - dist_mean) ** 3) / (dist_std ** 3 + 1e-10)
    
    # Estimate instance difficulty (correlation with known TSP hardness factors)
    hardness = 0.3 * dist_cv + 0.2 * np.log(n) / 10 + 0.1 * dist_skew
    
    # ========== 2. MULTI-RESOLUTION COMPONENTS ==========
    # A. Global structural component (MST-based)
    mst_component = np.zeros_like(distance_matrix)
    if n <= 5000:  # Practical limit for MST computation
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_mask = (mst > 0) | (mst.T > 0)
            # MST edges get base score, non-MST edges get distance-weighted score
            mst_component[mst_mask] = 1.0
            # For non-MST edges, score inversely proportional to distance gap to MST
            if np.any(~mst_mask):
                # Find minimum distance to connect each non-MST edge through MST
                for i in range(n):
                    for j in range(i+1, n):
                        if not mst_mask[i, j]:
                            # Minimum path through MST (approximate via triangle inequality)
                            min_path = np.min(distance_matrix[i] + distance_matrix[j])
                            mst_component[i, j] = np.exp(-(distance_matrix[i, j] - min_path) / (dist_mean + 1e-10))
                mst_component = (mst_component + mst_component.T) / 2
        except:
            # Fallback: use nearest neighbors
            mst_component = np.ones_like(distance_matrix)
            np.fill_diagonal(mst_component, 0)
    
    # B. Local clustering component (triangle completion patterns)
    cluster_component = np.zeros_like(distance_matrix)
    if n >= 3:
        # Normalized distance matrix for triangle inequality analysis
        norm_dist = distance_matrix / (np.median(distances_flat) + 1e-10)
        
        # For each edge (i,j), compute how many triangles it completes
        # Use vectorized computation for efficiency
        for k in range(min(50, n)):  # Sample limited intermediate nodes for speed
            # Triangle inequality violation measure
            triangle_viol = np.maximum(0, norm_dist[:, k:k+1] + norm_dist[k:k+1, :] - norm_dist)
            # Edges that help complete triangles get higher scores
            cluster_component += np.exp(-triangle_viol)
        
        # Normalize by number of sampled intermediate nodes
        cluster_component = cluster_component / min(50, n)
    
    # C. Neighborhood coherence component
    neigh_component = np.zeros_like(distance_matrix)
    k_neigh = max(2, min(10, n // 20))  # Adaptive neighborhood size
    
    # Compute shared nearest neighbors
    for i in range(n):
        # Find k nearest neighbors
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        neigh_idx = np.argpartition(dist_i, k_neigh)[:k_neigh]
        
        # Score edges based on neighbor overlap
        for j in range(i+1, n):
            dist_j = distance_matrix[j].copy()
            dist_j[j] = np.inf
            neigh_j = np.argpartition(dist_j, k_neigh)[:k_neigh]
            
            # Jaccard similarity of neighborhoods
            overlap = len(set(neigh_idx) & set(neigh_j))
            neigh_component[i, j] = overlap / (2 * k_neigh - overlap + 1e-10)
    
    neigh_component = (neigh_component + neigh_component.T) / 2
    
    # D. Inverse distance with adaptive scaling
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + np.eye(n) * 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # Adaptive scaling based on instance characteristics
    scale_factor = 1.0 + 0.5 * np.tanh(hardness - 1.0)  # Sigmoid adjustment
    inv_dist = np.power(inv_dist, scale_factor)
    
    # Row-normalize
    row_sums = inv_dist.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    inv_dist = inv_dist / row_sums
    
    # ========== 3. ADAPTIVE FUSION ==========
    # Dynamic weights based on instance characteristics
    if n < 20:
        # Small instances: emphasize local structure
        weights = [0.2, 0.3, 0.3, 0.2]  # [mst, cluster, neigh, inv_dist]
    elif n < 100:
        # Medium instances: balanced approach
        if dist_cv > 1.0:
            weights = [0.3, 0.25, 0.25, 0.2]  # High variability: emphasize MST
        else:
            weights = [0.2, 0.3, 0.3, 0.2]
    else:
        # Large instances: emphasize clustering and neighborhoods
        if hardness > 1.5:
            weights = [0.25, 0.35, 0.25, 0.15]  # Hard instances: more clustering
        else:
            weights = [0.2, 0.3, 0.3, 0.2]
    
    # Normalize weights
    weights = np.array(weights) / np.sum(weights)
    
    # Combine components
    combined = (
        weights[0] * mst_component +
        weights[1] * cluster_component +
        weights[2] * neigh_component +
        weights[3] * inv_dist
    )
    
    # ========== 4. PROBABILISTIC EDGE SCORING ==========
    # Convert to edge selection probabilities
    edge_probs = np.zeros_like(combined)
    
    # For each node, compute softmax over edges (excluding self)
    for i in range(n):
        node_scores = combined[i].copy()
        node_scores[i] = -np.inf  # Exclude self
        
        # Temperature scaling based on node degree requirements
        # Nodes in dense areas need higher temperature (more exploration)
        node_density = np.mean(combined[i])
        temperature = 0.5 + 0.5 * np.tanh(node_density * 5)
        
        # Softmax with temperature
        exp_scores = np.exp(node_scores / temperature)
        exp_scores[i] = 0  # Set diagonal to 0 after softmax
        row_sum = np.sum(exp_scores)
        if row_sum > 0:
            edge_probs[i] = exp_scores / row_sum
    
    # Symmetrize probabilities
    edge_probs = (edge_probs + edge_probs.T) / 2
    
    # ========== 5. STRUCTURAL CONSTRAINTS ENFORCEMENT ==========
    # Ensure minimum connectivity while preserving probabilities
    min_connections = max(2, min(5, n // 10))
    
    for i in range(n):
        # Get current probabilities for this node
        probs_i = edge_probs[i].copy()
        probs_i[i] = 0
        
        # Find top connections
        top_idx = np.argsort(probs_i)[-min_connections:]
        
        # Boost probabilities of essential connections
        boost_factor = 1.0 + 0.5 / (min_connections + 1)
        edge_probs[i, top_idx] *= boost_factor
        
        # Renormalize
        row_sum = np.sum(edge_probs[i])
        if row_sum > 0:
            edge_probs[i] = edge_probs[i] / row_sum
    
    # Final symmetrization and diagonal zeroing
    edge_probs = (edge_probs + edge_probs.T) / 2
    np.fill_diagonal(edge_probs, 0)
    
    # ========== 6. QUALITY-BASED SPARSIFICATION ==========
    # Adaptive sparsification based on edge quality distribution
    if n > 10:
        # Compute edge quality percentiles
        edge_vals = edge_probs[triu_idx]
        p75 = np.percentile(edge_vals, 75)
        p90 = np.percentile(edge_vals, 90)
        
        # Progressive sparsification: keep fewer edges for larger instances
        keep_ratio = 0.3 + 0.5 * np.exp(-n / 100)  # Decays from 0.8 to 0.3
        
        if n > 100:
            # For large instances, be more aggressive
            threshold = p90
        else:
            threshold = p75 * 0.7 + p90 * 0.3
        
        # Apply threshold with ratio constraint
        above_threshold = edge_probs >= threshold
        total_above = np.sum(above_threshold) / 2  # Undirected edges
        
        # Ensure we keep reasonable number of edges
        target_edges = int(keep_ratio * n * (n - 1) / 2)
        
        if total_above > target_edges:
            # Need to be more selective
            final_threshold = np.percentile(edge_vals, 100 * (1 - 2 * target_edges / (n * (n - 1))))
            final_mask = edge_probs >= final_threshold
        else:
            final_mask = above_threshold
        
        # Always include MST edges
        if n <= 5000:
            final_mask = final_mask | mst_mask
        
        # Apply mask
        result = np.zeros_like(edge_probs)
        result[final_mask] = edge_probs[final_mask]
    else:
        result = edge_probs
    
    # Final normalization
    max_val = np.max(result)
    if max_val > 0:
        result = result / max_val
    
    return result
