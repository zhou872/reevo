import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import squareform


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with edge betweenness centrality, 
    adaptive weight tuning based on instance characteristics, and
    efficient sparsification with connectivity preservation.
    
    Key Improvements:
    1. Edge betweenness centrality computed via efficient sampling
    2. Adaptive component weighting based on distance distribution
    3. Graph sparsification preserving essential edges
    4. Multi-factor fusion with improved normalization
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, where higher values
        indicate more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    # Extract upper triangle distances for statistics
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    # Compute distribution statistics
    dist_mean = np.mean(distances)
    dist_std = np.std(distances)
    cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    
    # ===== 2. ADAPTIVE PARAMETER TUNING =====
    # Base weights adapt based on instance characteristics
    # High cv → emphasize inverse distance more
    # Low cv → emphasize centrality and structure more
    inv_weight = 0.5 + 0.3 * np.tanh(cv)
    edge_betweenness_weight = 0.4 - 0.2 * np.tanh(cv)
    mst_weight = 0.3 + 0.1 * (1 - np.tanh(cv))
    
    # Normalize weights
    total_weight = inv_weight + edge_betweenness_weight + mst_weight
    inv_weight /= total_weight
    edge_betweenness_weight /= total_weight
    mst_weight /= total_weight
    
    # ===== 3. EDGE BETWEENNESS CENTRALITY =====
    # Approximate edge betweenness via random path sampling
    edge_betweenness = np.zeros_like(distance_matrix)
    
    # Determine sampling parameters adaptively
    sample_size = min(50, max(10, n // 2))
    
    for _ in range(sample_size):
        # Random source-target pair
        src, tgt = np.random.choice(n, 2, replace=False)
        
        # Build a greedy path from src to tgt
        path = [src]
        current = src
        visited = set([src])
        
        while current != tgt and len(visited) < n:
            # Find unvisited node with minimum distance to current
            mask = np.ones(n, dtype=bool)
            mask[list(visited)] = False
            mask[tgt] = True  # Always include target
            
            # Consider both distance to target and to current node
            # Weighted combination: 70% towards target, 30% towards current
            scores = 0.7 * distance_matrix[current] + 0.3 * distance_matrix[tgt]
            scores[~mask] = np.inf
            
            if np.all(~mask):
                break
                
            next_node = np.argmin(scores)
            path.append(next_node)
            visited.add(next_node)
            current = next_node
        
        # Update edge betweenness for edges in the path
        for i in range(len(path) - 1):
            u, v = path[i], path[i + 1]
            if u < v:
                edge_betweenness[u, v] += 1
            else:
                edge_betweenness[v, u] += 1
    
    # Symmetrize and normalize edge betweenness
    edge_betweenness = edge_betweenness + edge_betweenness.T
    if np.max(edge_betweenness) > 0:
        edge_betweenness = edge_betweenness / np.max(edge_betweenness)
    
    # ===== 4. INVERSE DISTANCE COMPONENT =====
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(inv_dist, 0)
    
    # Normalize inverse distance
    if np.max(inv_dist) > 0:
        inv_dist = inv_dist / np.max(inv_dist)
    
    # ===== 5. MINIMUM SPANNING TREE COMPONENT =====
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # MST edge scores based on inverse edge weight
    mst_scores = np.zeros_like(distance_matrix)
    mst_edges = np.where(mst_mask)
    if len(mst_edges[0]) > 0:
        mst_values = 1 / (distance_matrix[mst_edges] + 1e-10)
        mst_scores[mst_edges] = mst_values
        if np.max(mst_scores) > 0:
            mst_scores = mst_scores / np.max(mst_scores)
    
    # ===== 6. MULTI-FACTOR FUSION =====
    # Combine components with adaptive weights
    heuristic_matrix = (
        inv_weight * inv_dist +
        edge_betweenness_weight * edge_betweenness +
        mst_weight * mst_scores
    )
    
    # Boost MST edges (connectivity guarantee)
    heuristic_matrix[mst_mask] *= 1.5
    
    # ===== 7. INTELLIGENT SPARSIFICATION =====
    # Determine sparsification threshold adaptively
    if n > 10:
        # For larger instances, keep top edges per node
        edges_per_node = max(3, min(8, int(np.log2(n)) + 2))
        
        sparsity_mask = np.zeros_like(heuristic_matrix, dtype=bool)
        for i in range(n):
            # Get top edges for node i (excluding self)
            row_scores = heuristic_matrix[i].copy()
            row_scores[i] = -np.inf
            top_indices = np.argpartition(row_scores, -edges_per_node)[-edges_per_node:]
            sparsity_mask[i, top_indices] = True
        
        # Ensure MST edges are included
        sparsity_mask = sparsity_mask | mst_mask
        
        # Ensure symmetry
        sparsity_mask = sparsity_mask | sparsity_mask.T
        
        # Apply sparsification
        heuristic_matrix = heuristic_matrix * sparsity_mask
    
    # ===== 8. FINAL NORMALIZATION =====
    # Zero out diagonal
    np.fill_diagonal(heuristic_matrix, 0)
    
    # Normalize non-zero values to [0, 1]
    non_zero_vals = heuristic_matrix[heuristic_matrix > 0]
    if len(non_zero_vals) > 0:
        min_val = np.min(non_zero_vals)
        max_val = np.max(non_zero_vals)
        if max_val > min_val:
            heuristic_matrix = (heuristic_matrix - min_val) / (max_val - min_val + 1e-10)
    
    # Ensure symmetry (numerical stability)
    heuristic_matrix = 0.5 * (heuristic_matrix + heuristic_matrix.T)
    
    return heuristic_matrix
