import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with adaptive weighting
    and lightweight learning from instance characteristics.
    
    Key improvements over v1:
    1. Lightweight instance adaptation without complex analysis
    2. Learned weight combination from scale-based rules
    3. Efficient centrality computation with node importance
    4. Balanced sparsification ensuring connectivity
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ========== 1. LIGHTWEIGHT INSTANCE ANALYSIS ==========
    # Simple scale-based adaptation without complex analysis
    if n < 10:
        # Small instances: keep more edges, rely less on sparsification
        inv_dist_weight = 0.6
        centrality_weight = 0.4
        nn_boost = 1.2
        mst_boost = 1.8
        inv_dist_keep_pct = 0.25
        combined_keep_pct = 0.35
    elif n < 50:
        # Medium instances: balanced approach
        inv_dist_weight = 0.5
        centrality_weight = 0.4
        nn_boost = 1.3
        mst_boost = 2.0
        inv_dist_keep_pct = 0.15
        combined_keep_pct = 0.25
    else:
        # Large instances: aggressive sparsification
        inv_dist_weight = 0.4
        centrality_weight = 0.3
        nn_boost = 1.4
        mst_boost = 2.2
        inv_dist_keep_pct = 0.10
        combined_keep_pct = 0.20
    
    # ========== 2. BASE COMPONENTS ==========
    # Inverse distance with regularization
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(inv_dist, 0)
    
    # Nearest neighbors with adaptive k
    k = max(2, min(5, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argpartition(distances, k)[:k]
        nn_mask[i, nn_indices] = True
    nn_mask = nn_mask | nn_mask.T  # Make symmetric
    
    # Minimum spanning tree (connectivity guarantee)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # ========== 3. ENHANCED CENTRALITY MEASURE ==========
    # Node importance based on distance distribution
    row_sums = np.sum(distance_matrix, axis=1) - np.diag(distance_matrix)
    
    # Robust centrality: inverse of average distance with outlier handling
    avg_dist = row_sums / (n - 1)
    # Reduce influence of extreme distances
    robust_avg = np.clip(avg_dist, np.percentile(avg_dist, 10), 
                         np.percentile(avg_dist, 90))
    centrality = 1 / (robust_avg + 1e-10)
    
    # Normalize and apply softmax-like transformation
    centrality = np.exp(centrality - np.max(centrality))
    centrality = centrality / (np.sum(centrality) + 1e-10)
    
    # Edge potential as product of node centralities
    edge_potential = np.outer(centrality, centrality)
    np.fill_diagonal(edge_potential, 0)
    
    # ========== 4. ADAPTIVE COMBINATION ==========
    heuristic_scores = np.zeros_like(distance_matrix)
    
    # Base combination with learned weights
    base_component = inv_dist_weight * inv_dist + centrality_weight * edge_potential
    heuristic_scores = base_component.copy()
    
    # Boost for important edges
    heuristic_scores[nn_mask] *= nn_boost
    heuristic_scores[mst_mask] *= mst_boost
    
    # ========== 5. INTELLIGENT SPARSIFICATION ==========
    sparsity_mask = np.zeros_like(heuristic_scores, dtype=bool)
    
    # Adaptive keep counts
    inv_dist_keep = max(2, int(inv_dist_keep_pct * n))
    combined_keep = max(3, int(combined_keep_pct * n))
    
    # Pre-compute rankings
    inv_dist_ranks = np.argsort(inv_dist, axis=1)[:, ::-1]
    combined_ranks = np.argsort(heuristic_scores, axis=1)[:, ::-1]
    
    # Multi-criteria edge selection
    for i in range(n):
        # Keep top edges by inverse distance (local connectivity)
        top_by_inv_dist = inv_dist_ranks[i, :inv_dist_keep]
        sparsity_mask[i, top_by_inv_dist] = True
        
        # Keep top edges by combined score (global quality)
        top_by_combined = combined_ranks[i, :combined_keep]
        sparsity_mask[i, top_by_combined] = True
        
        # Always keep MST edges for connectivity guarantee
        sparsity_mask[i, mst_mask[i]] = True
        
        # Ensure minimum connectivity
        current_connections = np.sum(sparsity_mask[i])
        min_connections = max(2, int(0.1 * n))  # At least 2, up to 10% of nodes
        if current_connections < min_connections:
            needed = min_connections - current_connections
            # Add next best edges from combined ranking
            next_best = combined_ranks[i, combined_keep:combined_keep + needed]
            sparsity_mask[i, next_best] = True
    
    # Ensure symmetry
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # ========== 6. FINAL PROCESSING ==========
    # Apply sparsification
    heuristic_scores = heuristic_scores * sparsity_mask
    
    # Final normalization
    max_val = np.max(heuristic_scores)
    if max_val > 0:
        heuristic_scores = heuristic_scores / max_val
    
    # Zero out diagonal
    np.fill_diagonal(heuristic_scores, 0)
    
    return heuristic_scores
