import numpy as np
from scipy.spatial import distance_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP that combines multiple factors to identify
    promising edges, then sparsifies the result matrix.
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes
    
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n <= 2:
        return 1.0 / (distance_matrix + np.eye(n) * 1e-12)
    
    # 1. Base inverse distance heuristic
    with np.errstate(divide='ignore', invalid='ignore'):
        inverse_dist = 1.0 / (distance_matrix + np.eye(n) * 1e-12)
        inverse_dist[np.isinf(inverse_dist)] = 0
    
    # 2. Mutual nearest neighbors factor
    k = max(2, int(np.sqrt(n)))  # Adaptive neighbor count
    mutual_nn = np.zeros((n, n))
    
    for i in range(n):
        # Find k nearest neighbors for node i
        row_distances = distance_matrix[i].copy()
        row_distances[i] = np.inf  # Exclude self
        nearest_indices = np.argpartition(row_distances, k)[:k]
        
        # Mark mutual nearest neighbors
        for j in nearest_indices:
            if j != i:
                # Check if i is among j's k nearest neighbors
                col_distances = distance_matrix[j].copy()
                col_distances[j] = np.inf
                j_nearest = np.argpartition(col_distances, k)[:k]
                if i in j_nearest:
                    mutual_nn[i, j] = 1
    
    # 3. Connectivity heuristic based on node degrees in a Minimum Spanning Tree approximation
    # Construct approximate MST using Prim's algorithm
    mst_mask = np.zeros((n, n))
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    
    for _ in range(n - 1):
        # Find minimum edge between visited and unvisited nodes
        min_edge = (np.inf, -1, -1)
        for i in np.where(visited)[0]:
            for j in np.where(~visited)[0]:
                if distance_matrix[i, j] < min_edge[0]:
                    min_edge = (distance_matrix[i, j], i, j)
        
        _, u, v = min_edge
        mst_mask[u, v] = mst_mask[v, u] = 1
        visited[v] = True
    
    # 4. Edge betweenness approximation (geometric heuristic)
    # For nodes forming obtuse angles, prefer shorter edges
    edge_betweenness = np.ones((n, n))
    
    # Only compute for a sample of nodes to reduce complexity
    sample_size = min(20, n)
    if sample_size >= 3:
        sample_indices = np.random.choice(n, sample_size, replace=False)
        
        for i in sample_indices:
            for j in range(n):
                if i == j:
                    continue
                for k in range(n):
                    if k == i or k == j:
                        continue
                    
                    # Triangle inequality check
                    d_ij = distance_matrix[i, j]
                    d_ik = distance_matrix[i, k]
                    d_jk = distance_matrix[j, k]
                    
                    # Penalize edges that are part of "thin" triangles
                    if d_ij > max(d_ik, d_jk):
                        # This edge might be bypassable
                        penalty = 0.8
                        edge_betweenness[i, j] *= penalty
                        edge_betweenness[j, i] *= penalty
    
    # 5. Combine all factors
    heuristic_matrix = np.zeros((n, n))
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            base_score = inverse_dist[i, j]
            
            # Apply multiplicative factors
            score = base_score
            score *= (1.0 + 0.5 * mutual_nn[i, j])  # Boost mutual nearest neighbors
            score *= (1.0 + 0.3 * mst_mask[i, j])   # Boost MST edges
            score *= edge_betweenness[i, j]          # Apply geometric factor
            
            heuristic_matrix[i, j] = score
    
    # 6. Sparsification: keep only the most promising edges
    # For each node, keep connections to its top m neighbors
    m = max(3, int(np.sqrt(n)))
    
    sparsified = np.zeros((n, n))
    for i in range(n):
        # Get heuristic values for this node
        node_heuristics = heuristic_matrix[i].copy()
        node_heuristics[i] = -np.inf  # Exclude self
        
        # Find top m indices
        top_indices = np.argpartition(node_heuristics, -m)[-m:]
        
        # Keep only top m connections
        for j in top_indices:
            if node_heuristics[j] > 0:
                sparsified[i, j] = heuristic_matrix[i, j]
    
    # Ensure symmetry (undirected graph)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Normalize to [0, 1] range for better numerical stability
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
