import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP that combines multiple factors to score edge promise.
    Returns a sparsified matrix where unpromising edges are set to zero.
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return 1 / distance_matrix
    
    # 1. Basic inverse distance (higher for shorter edges)
    inv_dist = np.divide(1, distance_matrix, where=distance_matrix != 0)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Local cluster score: edges connecting to nodes with similar nearest neighbors
    k = min(5, n - 1)  # Number of nearest neighbors to consider
    nn_matrix = np.zeros((n, n))
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        sorted_idx = np.argsort(distance_matrix[i])
        nearest = sorted_idx[1:k+1]
        nn_matrix[i, nearest] = 1
    
    # Symmetric nearest neighbor overlap
    nn_overlap = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            overlap = np.sum(nn_matrix[i] * nn_matrix[j])
            nn_overlap[i, j] = nn_overlap[j, i] = overlap / k
    
    # 3. Density-based score: favor edges in dense regions
    densities = np.zeros(n)
    for i in range(n):
        # Use median distance as density measure
        distances = distance_matrix[i, np.arange(n) != i]
        densities[i] = 1 / (np.median(distances) + 1e-10)
    
    density_score = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Geometric mean of endpoint densities
            density_score[i, j] = density_score[j, i] = np.sqrt(densities[i] * densities[j])
    
    # 4. Triangle inequality score: measure of "directness"
    triangle_score = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] == 0:
                continue
            # Find alternative paths through third nodes
            alternatives = []
            for k in range(n):
                if k != i and k != j:
                    alt_dist = distance_matrix[i, k] + distance_matrix[k, j]
                    if alt_dist > 0:
                        alternatives.append(distance_matrix[i, j] / alt_dist)
            if alternatives:
                # Higher score when direct edge is much shorter than alternatives
                triangle_score[i, j] = triangle_score[j, i] = np.median(alternatives)
    
    # 5. Combine factors with adaptive weighting
    nn_weight = 0.3
    density_weight = 0.25
    triangle_weight = 0.15
    inv_weight = 0.3
    
    combined = (inv_weight * inv_dist + 
                nn_weight * nn_overlap + 
                density_weight * density_score + 
                triangle_weight * triangle_score)
    
    # 6. Sparsification: keep only promising edges
    # Strategy: keep edges that are in top percentiles by multiple criteria
    
    # Top edges by inverse distance
    inv_rank = np.zeros((n, n))
    for i in range(n):
        row_vals = inv_dist[i]
        if np.any(row_vals > 0):
            threshold = np.percentile(row_vals[row_vals > 0], 70)
            inv_rank[i] = row_vals >= threshold
    
    # Top edges by combined score
    comb_rank = np.zeros((n, n))
    for i in range(n):
        row_vals = combined[i]
        if np.any(row_vals > 0):
            threshold = np.percentile(row_vals[row_vals > 0], 60)
            comb_rank[i] = row_vals >= threshold
    
    # Create sparsity mask: keep edge if promising by multiple criteria
    sparsity_mask = (inv_rank.astype(int) + comb_rank.astype(int)) >= 1
    
    # Additional constraint: each node should have at least 2 connections
    for i in range(n):
        if np.sum(sparsity_mask[i]) < 2:
            # Add back the best missing edges for this node
            row_vals = combined[i].copy()
            row_vals[sparsity_mask[i]] = -np.inf
            top_indices = np.argsort(row_vals)[-2:]
            sparsity_mask[i, top_indices] = True
            sparsity_mask[top_indices, i] = True
    
    # Apply sparsity mask
    result = combined * sparsity_mask
    
    # 7. Normalize to reasonable scale (preserve relative differences)
    max_val = np.max(result)
    if max_val > 0:
        result = result / max_val
    
    return result
