import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse import csr_matrix
from typing import Tuple, List

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with adaptive weighting,
    intelligent sparsification, and vectorized efficiency.
    
    Returns a sparsified matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n < 3:
        # For trivial cases, return simple reciprocal
        result = np.divide(1, distance_matrix, where=distance_matrix != 0)
        np.fill_diagonal(result, 0)
        return result
    
    # Avoid division by zero
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # ==================== 1. CORE FACTORS (OPTIMIZED VECTORIZED) ====================
    
    # 1a. Reciprocal distance (fundamental TSP heuristic)
    inv_dist = 1.0 / safe_dist
    np.fill_diagonal(inv_dist, 0)
    
    # 1b. Nearest neighbor overlap (optimized vectorized)
    k = min(max(3, int(np.log2(n))), n - 1)
    
    # Vectorized nearest neighbors using argpartition
    nearest_indices = np.argpartition(safe_dist, k+1, axis=1)[:, 1:k+1]
    
    # Create binary matrix efficiently using advanced indexing
    row_indices = np.repeat(np.arange(n), k)
    col_indices = nearest_indices.ravel()
    nn_binary = np.zeros((n, n))
    nn_binary[row_indices, col_indices] = 1
    nn_overlap = (nn_binary @ nn_binary.T) / k
    
    # 1c. Efficient triangle inequality score using sampling
    triangle_scores = np.zeros((n, n))
    if n <= 1000:  # Full calculation for moderate sizes
        for i in range(n):
            for j in range(n):
                if i != j:
                    # Use vectorized computation for triangle inequality
                    k_indices = np.setdiff1d(np.arange(n), [i, j])
                    if len(k_indices) > 0:
                        alt_paths = safe_dist[i, k_indices] + safe_dist[k_indices, j]
                        direct = safe_dist[i, j]
                        ratios = direct / (alt_paths + 1e-10)
                        triangle_scores[i, j] = np.median(ratios)
    else:  # Sampling for large instances
        sample_size = min(50, n-2)
        for i in range(n):
            sample_nodes = np.random.choice(np.setdiff1d(np.arange(n), [i]), 
                                           size=min(sample_size, n-1), replace=False)
            for j in range(n):
                if i != j:
                    k_samples = np.setdiff1d(sample_nodes, [j])
                    if len(k_samples) > 0:
                        alt_paths = safe_dist[i, k_samples] + safe_dist[k_samples, j]
                        direct = safe_dist[i, j]
                        ratios = direct / (alt_paths + 1e-10)
                        triangle_scores[i, j] = np.median(ratios)
    
    # 1d. Minimum Spanning Tree importance
    mst = minimum_spanning_tree(csr_matrix(distance_matrix))
    mst_importance = (mst.toarray() != 0).astype(float)
    
    # 1e. Betweenness centrality approximation (optimized)
    # Use harmonic centrality (inverse of average distance)
    valid_counts = (distance_matrix > 0).sum(axis=1)
    sum_dists = distance_matrix.sum(axis=1)
    centrality = np.where(valid_counts > 0, valid_counts / (sum_dists + 1e-10), 0)
    
    # Normalize centrality
    c_min, c_max = centrality.min(), centrality.max()
    if c_max - c_min > 1e-10:
        centrality = (centrality - c_min) / (c_max - c_min)
    
    betweenness_approx = np.outer(centrality, centrality)
    
    # ==================== 2. ADAPTIVE WEIGHT OPTIMIZATION ====================
    
    # Analyze problem characteristics
    dist_mean = distance_matrix[np.triu_indices_from(distance_matrix, k=1)].mean()
    dist_std = distance_matrix[np.triu_indices_from(distance_matrix, k=1)].std()
    variability = dist_std / (dist_mean + 1e-10)
    sparsity = 2.0 * (n * (n-1) // 2) / (n * n)  # Edge density
    
    # Dynamic weight calculation based on problem features
    if variability > 1.5:
        # High variability: emphasize distance-based heuristics
        weights = np.array([0.40, 0.25, 0.15, 0.10, 0.10])
    elif variability < 0.3:
        # Low variability: emphasize structural heuristics
        weights = np.array([0.25, 0.30, 0.20, 0.15, 0.10])
    else:
        # Balanced approach for medium variability
        weights = np.array([0.35, 0.25, 0.15, 0.15, 0.10])
    
    # Adjust based on problem size and sparsity
    size_factor = min(1.0, n / 100)
    weights = weights * (0.7 + 0.6 * size_factor)
    
    if sparsity < 0.5:  # Sparse distance matrix
        weights[0] *= 1.2  # Boost distance importance
        weights[2] *= 0.8  # Reduce triangle importance
    
    weights = weights / weights.sum()  # Normalize
    
    # ==================== 3. INTELLIGENT COMBINATION ====================
    
    # Normalize each factor
    factors = [inv_dist, nn_overlap, triangle_scores, mst_importance, betweenness_approx]
    normalized_factors = []
    
    for factor in factors:
        f_max = factor.max()
        if f_max > 0:
            normalized_factors.append(factor / f_max)
        else:
            normalized_factors.append(factor)
    
    # Weighted combination
    combined = np.zeros_like(distance_matrix)
    for w, f in zip(weights, normalized_factors):
        combined += w * f
    
    # ==================== 4. SMART SPARSIFICATION ====================
    
    # Determine number of edges to keep (adaptive)
    if n <= 100:
        n_edges_to_keep = min(n * (n-1) // 3, 4 * n)
    else:
        n_edges_to_keep = min(n * (n-1) // 4, 5 * n)
    
    # Multi-criteria edge selection
    edge_scores = []
    edge_indices = []
    
    # Get upper triangle indices (excluding diagonal)
    for i in range(n):
        for j in range(i+1, n):
            edge_scores.append(combined[i, j])
            edge_indices.append((i, j))
    
    edge_scores = np.array(edge_scores)
    edge_indices = np.array(edge_indices)
    
    if len(edge_scores) > 0:
        # Select edges using adaptive threshold
        if len(edge_scores) > n_edges_to_keep:
            threshold = np.sort(edge_scores)[-n_edges_to_keep]
        else:
            threshold = edge_scores.min() - 1  # Keep all edges
        
        # Create selection mask
        selected_mask = edge_scores >= threshold
        
        # Create result matrix with selected edges
        result = np.zeros_like(distance_matrix)
        result[edge_indices[selected_mask, 0], edge_indices[selected_mask, 1]] = \
            combined[edge_indices[selected_mask, 0], edge_indices[selected_mask, 1]]
        result = result + result.T  # Make symmetric
    else:
        result = combined.copy()
    
    # Ensure connectivity through MST edges
    result = np.maximum(result, 0.5 * mst_importance * combined)
    
    # Ensure minimum node degree of 2
    for i in range(n):
        connected_edges = np.where(result[i] > 0)[0]
        if len(connected_edges) < 2:
            # Find best additional connections
            node_scores = combined[i].copy()
            node_scores[i] = -np.inf  # Exclude self
            node_scores[connected_edges] = -np.inf  # Exclude already connected
            
            # Select top edges to reach degree 2
            needed = 2 - len(connected_edges)
            if needed > 0:
                best_indices = np.argsort(node_scores)[-needed:]
                for idx in best_indices:
                    if node_scores[idx] > -np.inf:
                        result[i, idx] = combined[i, idx]
                        result[idx, i] = combined[idx, i]
    
    # ==================== 5. FINAL NORMALIZATION ====================
    
    # Zero diagonal
    np.fill_diagonal(result, 0)
    
    # Apply softmax-like scaling for better differentiation
    max_val = result.max()
    if max_val > 0:
        # Use sigmoid scaling for better numerical stability
        scaled = result / (max_val + 1e-10)
        result = 1 / (1 + np.exp(-5 * (scaled - 0.5)))  # Centered sigmoid
    
    # Normalize to [0, 1] range
    result_min, result_max = result.min(), result.max()
    if result_max - result_min > 1e-10:
        result = (result - result_min) / (result_max - result_min)
    
    return result
