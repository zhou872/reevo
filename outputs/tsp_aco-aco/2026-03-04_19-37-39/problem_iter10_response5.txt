import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with iterative refinement and intelligent path sampling.
    
    Key innovations:
    1. Two-phase betweenness computation with iterative refinement
    2. Biased random walks guided by evolving edge potentials
    3. Topology-aware sparsification preserving solution structure
    4. Adaptive component fusion based on solution quality feedback
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    cv = dist_std / (dist_mean + 1e-10)
    density_factor = min(1.0, n / 50.0)
    log_n = np.log(n + 1)
    
    # ===== 2. SOLUTION SAMPLING & EDGE FREQUENCY =====
    def construct_diverse_tours() -> Tuple[np.ndarray, list]:
        """Generate diverse tours using multiple heuristics with local optimization."""
        tours = []
        tour_costs = []
        
        # Construction methods with adaptive probabilities
        methods = ['nn', 'greedy', 'random', 'christofides']
        weights = [0.35, 0.3, 0.2, 0.15]
        
        num_tours = min(15, max(6, n // 3))
        
        for _ in range(num_tours):
            method = np.random.choice(methods, p=weights)
            
            if method == 'nn':
                tour = _nearest_neighbor_tour()
            elif method == 'greedy':
                tour = _greedy_insertion_tour()
            elif method == 'christofides':
                tour = _approximate_christofides()
            else:  # random
                tour = np.random.permutation(n)
            
            # Apply limited local search
            improved_tour = _two_opt_improve(tour, max_iter=int(20 * density_factor))
            tours.append(improved_tour)
            tour_costs.append(_tour_length(improved_tour))
        
        # Edge frequency from best tours
        edge_freq = np.zeros_like(distance_matrix)
        top_k = min(5, len(tours))
        best_idx = np.argsort(tour_costs)[:top_k]
        
        for idx in best_idx:
            tour = tours[idx]
            for i in range(n):
                u, v = tour[i], tour[(i+1)%n]
                edge_freq[u, v] += 1
                edge_freq[v, u] += 1
        
        if edge_freq.max() > 0:
            edge_freq = edge_freq / edge_freq.max()
        
        return edge_freq, tours, tour_costs
    
    # Helper construction functions
    def _nearest_neighbor_tour() -> np.ndarray:
        start = np.random.randint(0, n)
        tour = [start]
        unvisited = set(range(n)) - {start}
        
        while unvisited:
            current = tour[-1]
            # Consider k nearest neighbors
            k = min(5, len(unvisited))
            nearest = sorted(unvisited, key=lambda x: distance_matrix[current, x])[:k]
            if len(nearest) > 1 and np.random.random() < 0.3:
                next_node = np.random.choice(nearest[1:])
            else:
                next_node = nearest[0]
            tour.append(next_node)
            unvisited.remove(next_node)
        
        return np.array(tour)
    
    def _greedy_insertion_tour() -> np.ndarray:
        if n <= 3:
            return np.random.permutation(n)
        
        # Start with a random triangle
        indices = np.random.choice(n, 3, replace=False)
        tour = list(indices[:3])
        remaining = set(range(n)) - set(tour)
        
        while remaining:
            best_cost = float('inf')
            best_node = None
            best_pos = -1
            
            for node in remaining:
                # Fast heuristic: only consider insertion at promising positions
                for i in range(0, len(tour), max(1, len(tour)//10)):
                    cost_inc = (distance_matrix[tour[i-1], node] +
                               distance_matrix[node, tour[i]] -
                               distance_matrix[tour[i-1], tour[i]])
                    if cost_inc < best_cost:
                        best_cost = cost_inc
                        best_node = node
                        best_pos = i
            
            tour.insert(best_pos, best_node)
            remaining.remove(best_node)
        
        return np.array(tour)
    
    def _approximate_christofides() -> np.ndarray:
        """Simplified Christofides-like algorithm for diverse tours."""
        if n < 4:
            return np.random.permutation(n)
        
        # 1. Build MST
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = np.stack(np.where(mst > 0), axis=1)
        
        # 2. Find odd-degree vertices (approximate)
        degrees = np.sum(mst > 0, axis=0) + np.sum(mst > 0, axis=1)
        odd_vertices = np.where(degrees % 2 == 1)[0]
        
        # 3. Simple greedy matching for odd vertices
        matching = []
        remaining = set(odd_vertices)
        
        while len(remaining) > 1:
            u = remaining.pop()
            # Find nearest odd vertex
            nearest = min(remaining, key=lambda x: distance_matrix[u, x])
            matching.append((u, nearest))
            remaining.remove(nearest)
        
        # 4. Combine MST and matching to create Eulerian graph (simplified)
        # Start from random node and do depth-first traversal
        start = np.random.randint(0, n)
        visited = set()
        tour = []
        
        def dfs(node):
            visited.add(node)
            tour.append(node)
            neighbors = np.where(mst[node] > 0)[0]
            for neighbor in neighbors:
                if neighbor not in visited:
                    dfs(neighbor)
        
        dfs(start)
        
        # Add matching edges if not already in tour
        for u, v in matching:
            if u not in visited:
                tour.append(u)
            if v not in visited:
                tour.append(v)
        
        return np.array(tour[:n])  # Ensure proper length
    
    def _two_opt_improve(tour: np.ndarray, max_iter: int = 30) -> np.ndarray:
        current = tour.copy()
        improved = True
        iter_count = 0
        
        while improved and iter_count < max_iter:
            improved = False
            # Randomize evaluation order for diversity
            indices = np.random.permutation(n)
            
            for i_idx in range(n-1):
                i = indices[i_idx]
                for j_idx in range(i_idx+2, min(i_idx+10, n)):
                    j = indices[j_idx]
                    if j == n-1 and i == 0:
                        continue
                    
                    a, b = current[i], current[(i+1)%n]
                    c, d = current[j], current[(j+1)%n]
                    
                    old = distance_matrix[a, b] + distance_matrix[c, d]
                    new = distance_matrix[a, c] + distance_matrix[b, d]
                    
                    if new < old - 1e-10:
                        # Perform 2-opt swap
                        segment = current[(i+1)%n:(j+1)%n]
                        if len(segment) > 0:
                            current[(i+1)%n:(j+1)%n] = segment[::-1]
                        improved = True
                        break
                if improved:
                    break
            
            iter_count += 1
        
        return current
    
    def _tour_length(tour: np.ndarray) -> float:
        return sum(distance_matrix[tour[i], tour[(i+1)%n]] for i in range(n))
    
    # Generate initial edge frequency and tours
    edge_freq, tours, tour_costs = construct_diverse_tours()
    
    # ===== 3. ITERATIVE BETWEENNESS REFINEMENT =====
    def compute_edge_betweenness(edge_weights: np.ndarray = None, 
                                 num_paths: int = None) -> np.ndarray:
        """Compute edge betweenness using biased random walks."""
        if num_paths is None:
            num_paths = min(300, max(80, n * 3))
        
        betweenness = np.zeros_like(distance_matrix)
        
        # Precompute transition probabilities if edge weights are provided
        transition_matrix = None
        if edge_weights is not None:
            transition_matrix = np.zeros_like(distance_matrix)
            for i in range(n):
                # Combine edge weight with distance information
                weights_i = edge_weights[i].copy()
                weights_i[i] = 0
                inv_dist_i = 1 / (distance_matrix[i] + 1e-10)
                inv_dist_i[i] = 0
                # Softmax combination
                combined = 0.7 * weights_i + 0.3 * inv_dist_i
                # Normalize for probabilities
                if combined.sum() > 0:
                    transition_matrix[i] = combined / combined.sum()
        
        for _ in range(num_paths):
            # Sample source and target
            s, t = np.random.choice(n, 2, replace=False)
            
            # Biased random walk
            current = s
            path = [s]
            visited = set([s])
            
            # Maximum path length to prevent infinite loops
            max_path_length = n * 2
            
            for step in range(max_path_length):
                if current == t:
                    break
                
                if transition_matrix is not None and transition_matrix[current].sum() > 0:
                    # Use biased transition probabilities
                    probs = transition_matrix[current]
                else:
                    # Distance-based probabilities
                    inv_dist = 1 / (distance_matrix[current] + 1e-10)
                    inv_dist[current] = 0
                    # Bias towards target
                    bias_to_target = 1 / (distance_matrix[:, t] + 1e-10)
                    combined = 0.8 * inv_dist + 0.2 * bias_to_target
                    if combined.sum() > 0:
                        probs = combined / combined.sum()
                    else:
                        probs = np.ones(n) / n
                
                # Sample next node
                next_node = np.random.choice(n, p=probs)
                
                # Update betweenness
                betweenness[current, next_node] += 1
                betweenness[next_node, current] += 1
                
                current = next_node
                visited.add(current)
                path.append(current)
                
                # Reset visited occasionally to avoid getting stuck
                if len(visited) > max(0.7 * n, 10):
                    visited = set([current])
        
        if betweenness.max() > 0:
            betweenness = betweenness / betweenness.max()
        
        return betweenness
    
    # Two-phase betweenness computation
    # Phase 1: Initial betweenness using distance information only
    betweenness_phase1 = compute_edge_betweenness(num_paths=max(50, n))
    
    # Phase 2: Compute initial heuristic to guide second phase
    # Basic components for initial heuristic
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # MST component
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_scores = np.zeros_like(distance_matrix)
    mst_scores[mst_mask] = 1.0 / (distance_matrix[mst_mask] + 1e-10)
    if mst_scores.max() > 0:
        mst_scores = mst_scores / mst_scores.max()
    
    # Initial heuristic (simple combination)
    initial_heuristic = 0.4 * edge_freq + 0.3 * betweenness_phase1 + 0.2 * inv_dist + 0.1 * mst_scores
    
    # Phase 2: Refined betweenness guided by initial heuristic
    betweenness_phase2 = compute_edge_betweenness(edge_weights=initial_heuristic, 
                                                  num_paths=min(200, max(100, n*2)))
    
    # ===== 4. ADAPTIVE COMPONENT FUSION =====
    # Analyze component correlations with tour quality
    component_scores = {
        'freq': edge_freq,
        'bet': betweenness_phase2,
        'dist': inv_dist,
        'mst': mst_scores
    }
    
    # Estimate component quality by correlation with good tours
    component_weights = np.zeros(4)
    component_names = ['freq', 'bet', 'dist', 'mst']
    
    if len(tours) >= 3:
        # Create binary edge presence matrix for tours
        edge_presence = np.zeros_like(distance_matrix)
        for tour in tours[:5]:  # Top 5 tours
            for i in range(n):
                u, v = tour[i], tour[(i+1)%n]
                edge_presence[u, v] = 1
                edge_presence[v, u] = 1
        
        # Compute correlation between each component and edge presence
        for idx, name in enumerate(component_names):
            comp = component_scores[name]
            # Flatten upper triangular matrices
            comp_flat = comp[np.triu_indices(n, k=1)]
            presence_flat = edge_presence[np.triu_indices(n, k=1)]
            
            # Simple correlation-like measure
            if comp_flat.std() > 1e-10 and presence_flat.std() > 1e-10:
                correlation = np.corrcoef(comp_flat, presence_flat)[0, 1]
                component_weights[idx] = max(0.1, abs(correlation))
            else:
                component_weights[idx] = 0.25
    
    # Normalize weights with instance-specific adjustments
    if component_weights.sum() > 0:
        base_weights = component_weights / component_weights.sum()
    else:
        base_weights = np.array([0.35, 0.3, 0.2, 0.15])
    
    # Adjust based on instance characteristics
    w_freq = base_weights[0] * (1 + 0.3 * min(1.0, np.std(edge_freq)*2))
    w_bet = base_weights[1] * (1 + 0.2 * min(1.0, np.std(betweenness_phase2)*1.5))
    w_dist = base_weights[2] * (1 - 0.2 * min(1.0, cv))
    w_mst = base_weights[3] * (1 + 0.3 * density_factor)
    
    total = w_freq + w_bet + w_dist + w_mst
    w_freq, w_bet, w_dist, w_mst = w_freq/total, w_bet/total, w_dist/total, w_mst/total
    
    # ===== 5. COMBINE COMPONENTS WITH TOPOLOGY AWARENESS =====
    heuristics_matrix = (
        w_freq * edge_freq +
        w_bet * betweenness_phase2 +
        w_dist * inv_dist +
        w_mst * mst_scores
    )
    
    # Boost edges that appear in multiple promising structures
    consensus_mask = (
        (edge_freq > np.percentile(edge_freq, 75)) &
        (betweenness_phase2 > np.percentile(betweenness_phase2, 60)) &
        mst_mask
    )
    heuristics_matrix[consensus_mask] *= 1.5
    
    # ===== 6. TOPOLOGY-AWARE SPARSIFICATION =====
    # Determine edges to keep based on local and global structure
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    # Per-node sparsification
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        # Adaptive number of edges: consider both local quality and global structure
        base_edges = max(3, int(2 * log_n))
        max_edges = min(n-1, int(5 * log_n))
        
        # Consider density and solution quality
        n_edges = int(np.clip(
            base_edges * (1 + 0.5 * density_factor + 0.3 * cv),
            base_edges, max_edges
        ))
        
        # Keep top edges plus any MST edges
        top_indices = np.argsort(scores)[-n_edges:]
        sparsity_mask[i, top_indices] = True
    
    # Ensure connectivity and symmetry
    sparsity_mask = sparsity_mask | mst_mask
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Additional connectivity check
    for i in range(n):
        if not np.any(sparsity_mask[i]):
            # Connect to nearest neighbor
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            nearest = np.argmin(distances)
            sparsity_mask[i, nearest] = True
            sparsity_mask[nearest, i] = True
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 7. FINAL REFINEMENT =====
    # Ensure symmetry
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Smooth the heuristic values
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        # Apply softmax-like scaling
        exp_values = np.exp(non_zero - np.mean(non_zero))
        if exp_values.max() > 0:
            heuristics_matrix[heuristics_matrix > 0] = exp_values / exp_values.max()
    
    # Final normalization
    if heuristics_matrix.max() > 0:
        heuristics_matrix = heuristics_matrix / heuristics_matrix.max()
    
    return heuristics_matrix
