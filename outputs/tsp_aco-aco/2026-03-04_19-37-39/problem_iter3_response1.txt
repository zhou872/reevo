import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    Returns a matrix where higher values indicate more promising edges.
    
    Improvements over v1:
    1. Improved centrality measure using inverse average distances
    2. Multi-criteria edge ranking with percentile-based thresholding
    3. Connectivity-preserving sparsification with minimum degree guarantee
    4. Adaptive neighborhood size based on graph density
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Inverse distance component - base measure
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Improved centrality measure: inverse average distance to all other nodes
    # This better captures node importance than simple sum of inverses
    centrality = np.zeros(n)
    for i in range(n):
        # Exclude self-distance
        other_distances = distance_matrix[i, np.arange(n) != i]
        if len(other_distances) > 0:
            # Inverse of average distance to all other nodes
            avg_distance = np.mean(other_distances)
            centrality[i] = 1.0 / (avg_distance + 1e-10)
    
    # Normalize centrality to [0, 1]
    if np.max(centrality) > 0:
        centrality = centrality / np.max(centrality)
    
    # 3. Edge betweenness approximation: product of node centralities
    # Edges connecting central nodes get higher scores
    edge_betweenness = np.outer(centrality, centrality)
    
    # 4. Adaptive nearest neighbor emphasis with symmetric masking
    # Use adaptive k based on problem size and graph density
    avg_dist = np.mean(distance_matrix[distance_matrix > 0])
    density_factor = np.clip(avg_dist / np.max(distance_matrix), 0.1, 0.9)
    k = max(2, min(8, int(n * (1 - density_factor) / 3)))
    
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        nn_indices = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, nn_indices] = True
    
    # Make mask symmetric
    nn_mask = nn_mask | nn_mask.T
    
    # 5. Minimum spanning tree component - ensure connectivity
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # 6. Combined edge scoring with multiple criteria
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base: inverse distance (scaled by log to reduce extreme values)
    heuristics_matrix = np.log1p(0.8 * inv_dist)
    
    # Boost edges in nearest neighbor graph
    heuristics_matrix[nn_mask] += 0.3
    
    # Strong boost for MST edges
    heuristics_matrix[mst_mask] += 0.5
    
    # Add edge betweenness component
    heuristics_matrix += 0.4 * edge_betweenness
    
    # 7. Multi-criteria sparsification with minimum connectivity guarantee
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    # Calculate combined score percentiles
    flat_scores = heuristics_matrix[np.triu_indices(n, 1)]
    if len(flat_scores) > 0:
        # Use dynamic threshold based on distribution
        score_threshold = np.percentile(flat_scores, 70)  # Keep top 30%
        
        # First pass: keep edges above threshold
        above_threshold = heuristics_matrix >= score_threshold
        sparsity_mask = above_threshold.copy()
        
        # Ensure each node has at least 2 connections for feasibility
        for i in range(n):
            connections = np.sum(sparsity_mask[i])
            if connections < 2:
                # Find best edges not already included
                node_scores = heuristics_matrix[i].copy()
                node_scores[sparsity_mask[i]] = -np.inf  # Exclude already selected
                node_scores[i] = -np.inf  # Exclude self
                
                # Add best edges until minimum connectivity reached
                needed = 2 - connections
                if needed > 0:
                    best_indices = np.argsort(node_scores)[-needed:]
                    for j in best_indices:
                        if node_scores[j] > -np.inf:
                            sparsity_mask[i, j] = True
                            sparsity_mask[j, i] = True
        
        # Also ensure graph remains connected by keeping MST edges
        sparsity_mask = sparsity_mask | mst_mask
    
    # 8. Apply sparsification and final normalization
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
