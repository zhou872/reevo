import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List, Tuple, Dict
import random

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with reinforcement-guided weight optimization,
    population-based diversity enhancement, and adaptive constraint integration.
    
    Key innovations:
    1. Reinforcement learning-inspired weight adaptation based on edge performance
    2. Population-based sampling with quality-biased selection
    3. Constraint-aware edge promotion for diverse problem structures
    4. Dynamic exploration-exploitation balance
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. REINFORCEMENT LEARNING PARAMETERS =====
    # Q-learning inspired parameters for edge quality tracking
    edge_q_values = np.ones_like(distance_matrix) * 0.5
    np.fill_diagonal(edge_q_values, 0)
    learning_rate = 0.3
    exploration_rate = 0.2
    
    # ===== 2. POPULATION-BASED TOUR GENERATION =====
    def generate_population_diverse(size: int) -> List[np.ndarray]:
        """Generate diverse population using multiple strategies."""
        population = []
        
        # Strategy 1: Farthest insertion (good for structured instances)
        for _ in range(size // 3):
            if n <= 3:
                tour = np.random.permutation(n)
            else:
                # Start with random node
                start = np.random.randint(0, n)
                tour = [start]
                remaining = set(range(n)) - {start}
                
                while remaining:
                    # Find farthest node from current tour
                    farthest = max(remaining, 
                                 key=lambda x: min(distance_matrix[x, y] for y in tour))
                    tour.append(farthest)
                    remaining.remove(farthest)
                
                tour = np.array(tour)
            population.append(tour)
        
        # Strategy 2: Christofides-like approach
        for _ in range(size // 3):
            if n <= 3:
                tour = np.random.permutation(n)
            else:
                # Compute minimum spanning tree
                mst = minimum_spanning_tree(distance_matrix)
                # Simplified approach: start from random node and do DFS
                start = np.random.randint(0, n)
                visited = [False] * n
                tour = []
                
                stack = [start]
                while stack:
                    node = stack.pop()
                    if not visited[node]:
                        visited[node] = True
                        tour.append(node)
                        # Add neighbors in random order
                        neighbors = np.where(mst[node] > 0)[0]
                        np.random.shuffle(neighbors)
                        stack.extend(neighbors)
                
                tour = np.array(tour)
            population.append(tour)
        
        # Strategy 3: Randomized nearest neighbor with exploration
        for _ in range(size - len(population)):
            start = np.random.randint(0, n)
            tour = [start]
            unvisited = set(range(n)) - {start}
            
            while unvisited:
                current = tour[-1]
                
                # Use softmax selection based on edge Q-values
                candidates = list(unvisited)
                if np.random.random() < exploration_rate:
                    # Exploration: random choice
                    next_node = random.choice(candidates)
                else:
                    # Exploitation: weighted by Q-values and distances
                    scores = []
                    for node in candidates:
                        base_score = 1.0 / (distance_matrix[current, node] + 1e-10)
                        q_bonus = edge_q_values[current, node]
                        scores.append(base_score * (1.0 + q_bonus))
                    
                    scores = np.array(scores)
                    if scores.sum() > 0:
                        probs = scores / scores.sum()
                        next_node = np.random.choice(candidates, p=probs)
                    else:
                        next_node = random.choice(candidates)
                
                tour.append(next_node)
                unvisited.remove(next_node)
            
            population.append(np.array(tour))
        
        return population
    
    # ===== 3. QUALITY-AWARE LOCAL IMPROVEMENT =====
    def population_improve(population: List[np.ndarray], 
                          elite_ratio: float = 0.3) -> List[np.ndarray]:
        """Improve population with varying intensities based on quality."""
        # Evaluate current population
        qualities = []
        for tour in population:
            length = sum(distance_matrix[tour[i], tour[(i+1)%n]] for i in range(n))
            qualities.append(1.0 / (length + 1e-10))
        
        # Sort by quality
        sorted_indices = np.argsort(qualities)[::-1]
        elite_size = max(1, int(len(population) * elite_ratio))
        
        improved_population = []
        
        for idx, tour_idx in enumerate(sorted_indices):
            tour = population[tour_idx].copy()
            
            # Apply local search with intensity based on rank
            if idx < elite_size:
                # Elite tours get intensive search
                max_iter = min(50, n * 2)
                improvement_factor = 1.5
            else:
                # Others get moderate search
                max_iter = min(20, n)
                improvement_factor = 1.0
            
            improved = True
            iter_count = 0
            
            while improved and iter_count < max_iter:
                improved = False
                
                # Try multiple 2-opt moves
                for _ in range(min(n, 10)):
                    i, j = np.random.choice(n, 2, replace=False)
                    if abs(i - j) < 2 or (i == 0 and j == n-1):
                        continue
                    
                    i, j = sorted([i, j])
                    a, b = tour[i], tour[(i+1)%n]
                    c, d = tour[j], tour[(j+1)%n]
                    
                    current_cost = distance_matrix[a, b] + distance_matrix[c, d]
                    new_cost = distance_matrix[a, c] + distance_matrix[b, d]
                    
                    if new_cost < current_cost * improvement_factor:
                        # Perform the swap
                        if j > i + 1:
                            tour[i+1:j+1] = tour[i+1:j+1][::-1]
                        improved = True
                        break
                
                iter_count += 1
            
            improved_population.append(tour)
        
        return improved_population
    
    # ===== 4. REINFORCEMENT-BASED WEIGHT OPTIMIZATION =====
    def update_q_values(population: List[np.ndarray], 
                       qualities: List[float]) -> np.ndarray:
        """Update edge Q-values based on population performance."""
        global learning_rate
        
        # Normalize qualities
        max_quality = max(qualities)
        if max_quality > 0:
            normalized_qualities = [q / max_quality for q in qualities]
        else:
            normalized_qualities = [0] * len(qualities)
        
        # Update Q-values
        for tour, quality in zip(population, normalized_qualities):
            reward = quality * 2.0  # Scale reward
            
            for i in range(n):
                u, v = tour[i], tour[(i+1)%n]
                
                # Q-learning update
                current_q = edge_q_values[u, v]
                edge_q_values[u, v] = (1 - learning_rate) * current_q + learning_rate * reward
                edge_q_values[v, u] = edge_q_values[u, v]  # Symmetry
        
        # Decay learning rate
        learning_rate *= 0.95
        
        return edge_q_values
    
    # ===== 5. MAIN EXECUTION =====
    # Generate and improve population
    pop_size = min(30, max(5, n * 2))
    population = generate_population_diverse(pop_size)
    improved_population = population_improve(population)
    
    # Evaluate population and update Q-values
    population_qualities = []
    for tour in improved_population:
        length = sum(distance_matrix[tour[i], tour[(i+1)%n]] for i in range(n))
        population_qualities.append(1.0 / (length + 1e-10))
    
    edge_q_values = update_q_values(improved_population, population_qualities)
    
    # ===== 6. COMPONENT COMPUTATION =====
    # Edge frequency from population (weighted by quality)
    edge_freq = np.zeros_like(distance_matrix)
    max_quality = max(population_qualities) if population_qualities else 1.0
    
    for tour, quality in zip(improved_population, population_qualities):
        weight = quality / max_quality
        for i in range(n):
            u, v = tour[i], tour[(i+1)%n]
            edge_freq[u, v] += weight
            edge_freq[v, u] += weight
    
    if edge_freq.max() > 0:
        edge_freq = edge_freq / edge_freq.max()
    
    # Q-value component
    q_component = edge_q_values.copy()
    if q_component.max() > 0:
        q_component = q_component / q_component.max()
    
    # Distance component
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + np.eye(n) * 1e-10)
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > inv_dist.min():
        inv_dist = (inv_dist - inv_dist.min()) / (inv_dist.max() - inv_dist.min())
    
    # MST component (always included for connectivity)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_scores = np.zeros_like(distance_matrix)
    mst_scores[mst_mask] = 1.0
    
    # ===== 7. ADAPTIVE ENSEMBLE WEIGHTING =====
    # Analyze component correlations for weight adjustment
    components = {
        'freq': edge_freq,
        'q_val': q_component,
        'dist': inv_dist,
        'mst': mst_scores
    }
    
    # Dynamic weight adjustment based on component performance
    # More weight to components that correlate with good tours
    component_weights = {
        'freq': 0.3,  # Baseline frequency weight
        'q_val': 0.3,  # Q-values from reinforcement learning
        'dist': 0.2,   # Distance-based component
        'mst': 0.2     # Connectivity component
    }
    
    # Adjust weights based on variance in quality correlation
    if len(improved_population) >= 3:
        # Compute component-edge scores for top vs bottom tours
        top_idx = np.argsort(population_qualities)[-3:]  # Top 3 tours
        bottom_idx = np.argsort(population_qualities)[:3]  # Bottom 3 tours
        
        for comp_name, comp_matrix in components.items():
            top_scores = []
            bottom_scores = []
            
            for idx in top_idx:
                tour = improved_population[idx]
                score = sum(comp_matrix[tour[i], tour[(i+1)%n]] for i in range(n)) / n
                top_scores.append(score)
            
            for idx in bottom_idx:
                tour = improved_population[idx]
                score = sum(comp_matrix[tour[i], tour[(i+1)%n]] for i in range(n)) / n
                bottom_scores.append(score)
            
            if len(top_scores) > 0 and len(bottom_scores) > 0:
                mean_top = np.mean(top_scores)
                mean_bottom = np.mean(bottom_scores)
                
                if mean_top > mean_bottom:
                    # Component is better in good tours - increase weight
                    component_weights[comp_name] *= 1.2
                elif mean_top < mean_bottom:
                    # Component is worse in good tours - decrease weight
                    component_weights[comp_name] *= 0.8
    
    # Normalize weights
    total_weight = sum(component_weights.values())
    for key in component_weights:
        component_weights[key] /= total_weight
    
    # ===== 8. COMBINE COMPONENTS =====
    heuristics_matrix = np.zeros_like(distance_matrix)
    for comp_name, comp_matrix in components.items():
        heuristics_matrix += component_weights[comp_name] * comp_matrix
    
    # Ensure symmetry
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # ===== 9. CONNECTIVITY-PRESERVING SPARSIFICATION =====
    # Enhanced sparsification with connectivity guarantees
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    # Ensure each node connects to promising neighbors
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        # Adaptive edge selection
        base_edges = max(2, int(np.log2(n + 1)))
        if n < 20:
            target_edges = min(n-1, base_edges + 1)
        else:
            target_edges = min(n-1, base_edges + 2)
        
        # Keep top edges
        if scores.sum() > 0 and target_edges < n-1:
            threshold = np.partition(scores, -target_edges)[-target_edges]
            sparsity_mask[i, scores >= threshold] = True
        else:
            sparsity_mask[i, :] = True
            sparsity_mask[i, i] = False
        
        # Always include shortest edge for connectivity
        if n > 1:
            nearest = np.argsort(distance_matrix[i])[1]
            sparsity_mask[i, nearest] = True
    
    # Ensure MST edges are included and maintain symmetry
    sparsity_mask = sparsity_mask | mst_mask
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 10. FINAL NORMALIZATION AND REFINEMENT =====
    # Normalize non-zero values to [0, 1]
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        if non_zero.max() > non_zero.min():
            heuristics_matrix[heuristics_matrix > 0] = (
                (non_zero - non_zero.min()) / (non_zero.max() - non_zero.min())
            )
    
    # Smooth the matrix for better gradient
    heuristics_matrix = np.sqrt(heuristics_matrix)  # Apply square root for smoother distribution
    
    # Final symmetry check
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
