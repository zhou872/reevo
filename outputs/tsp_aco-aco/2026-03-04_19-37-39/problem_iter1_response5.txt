import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP that combines multiple factors to determine edge promise.
    Returns a sparse matrix where promising edges have positive values and others are zero.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (similar to v1 but with smoothing)
    # Add small epsilon to avoid division by zero and extreme values
    epsilon = 1e-10
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)  # Zero out diagonal
    
    # 2. Local clustering factor: edges that connect to nodes with similar neighborhoods
    # For each node, identify its k nearest neighbors
    k = max(3, min(10, n // 4))  # Adaptive k based on problem size
    cluster_factor = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        neighbors = np.argsort(distance_matrix[i])[1:k+1]
        
        # For each neighbor j, check how many of j's nearest neighbors are also i's nearest neighbors
        for j in neighbors:
            if i < j:  # Process each edge only once
                # Find common neighbors in top 2k neighbors
                i_top = set(np.argsort(distance_matrix[i])[1:2*k+1])
                j_top = set(np.argsort(distance_matrix[j])[1:2*k+1])
                common = len(i_top.intersection(j_top))
                cluster_factor[i, j] = cluster_factor[j, i] = common / (2*k)
    
    # 3. Path saving heuristic (inspired by Clarke-Wright)
    savings = np.zeros_like(distance_matrix)
    if n >= 3:
        for i in range(n):
            for j in range(i+1, n):
                if i != j:
                    # Classic savings formula: save = d(i,0) + d(0,j) - d(i,j)
                    # Using average of all nodes as "depot" for robustness
                    depot_avg = np.mean([distance_matrix[i].mean(), distance_matrix[j].mean()])
                    savings[i, j] = savings[j, i] = max(0, 2*depot_avg - distance_matrix[i, j])
    
    # 4. Normalize and combine factors with weighted sum
    # Normalize each factor to [0, 1] range
    def safe_normalize(arr):
        arr_min, arr_max = arr.min(), arr.max()
        if arr_max - arr_min > 1e-10:
            return (arr - arr_min) / (arr_max - arr_min)
        return np.zeros_like(arr)
    
    norm_inv = safe_normalize(inv_dist)
    norm_cluster = safe_normalize(cluster_factor)
    norm_savings = safe_normalize(savings)
    
    # Combined heuristic with weights
    weights = [0.5, 0.3, 0.2]  # Weights for inverse distance, clustering, savings
    heuristic = (weights[0] * norm_inv + 
                 weights[1] * norm_cluster + 
                 weights[2] * norm_savings)
    
    # 5. Adaptive sparsification: keep only promising edges
    # Use percentile-based threshold that adapts to distribution
    flat_heuristic = heuristic.flatten()
    
    # Remove zeros (diagonal) from threshold calculation
    non_zero = flat_heuristic[flat_heuristic > 0]
    if len(non_zero) > 0:
        # Keep top edges: threshold at 60th percentile (adjustable)
        threshold = np.percentile(non_zero, 60)
        heuristic[heuristic < threshold] = 0
    
    # 6. Additional local sparsification: for each node, keep only top m edges
    m = max(2, min(8, n // 5))  # Adaptive m
    for i in range(n):
        row = heuristic[i].copy()
        if np.count_nonzero(row) > m:
            # Get indices of top m values
            top_indices = np.argsort(row)[-m:]
            mask = np.zeros(n, dtype=bool)
            mask[top_indices] = True
            heuristic[i][~mask] = 0
    
    # Ensure symmetry (since TSP is symmetric)
    heuristic = np.maximum(heuristic, heuristic.T)
    
    # Final smoothing: apply mild Gaussian-like smoothing to non-zero edges
    non_zero_mask = heuristic > 0
    if np.any(non_zero_mask):
        # Small enhancement to promising edges
        heuristic[non_zero_mask] *= 1.1
        heuristic[non_zero_mask] = np.minimum(heuristic[non_zero_mask], 1.0)
    
    return heuristic
