import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
import random

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with ensemble learning
    and solution-quality feedback loops. Returns a matrix where higher values
    indicate more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # Handle zero distances to avoid division errors
    eps = 1e-10
    dist = np.maximum(distance_matrix, eps)
    
    # 1. Dynamic weight initialization based on problem characteristics
    avg_dist = np.mean(dist[dist > eps])
    std_dist = np.std(dist[dist > eps])
    cv_dist = std_dist / (avg_dist + eps)
    
    # Adaptive initial weights
    if cv_dist < 0.3:
        weights = {'inv_dist': 0.7, 'nn': 0.5, 'mst': 1.5, 'centrality': 0.2, 'candidate': 0.8}
    elif cv_dist < 0.7:
        weights = {'inv_dist': 0.8, 'nn': 0.6, 'mst': 1.3, 'centrality': 0.3, 'candidate': 0.9}
    else:
        weights = {'inv_dist': 0.9, 'nn': 0.7, 'mst': 1.1, 'centrality': 0.4, 'candidate': 1.0}
    
    # 2. Ensemble of base heuristics
    # Base inverse distance component
    inv_dist = 1 / dist
    np.fill_diagonal(inv_dist, 0)
    
    # Adaptive nearest neighbor
    base_k = max(2, min(6, n // 5))
    k = int(base_k * (1 + 0.5 * cv_dist))
    nn_mask = np.zeros_like(dist, dtype=bool)
    for i in range(n):
        nn_indices = np.argsort(dist[i])[1:k+1]
        nn_mask[i, nn_indices] = True
    nn_mask = nn_mask | nn_mask.T
    
    # Minimum spanning tree component
    mst = minimum_spanning_tree(dist).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # Enhanced node centrality using multiple measures
    centrality_harmonic = np.zeros(n)
    centrality_betweenness = np.zeros(n)
    
    # Harmonic centrality
    for i in range(n):
        valid_dists = dist[i, dist[i] > eps]
        if len(valid_dists) > 0:
            centrality_harmonic[i] = np.sum(1 / valid_dists)
    
    # Approximate betweenness centrality using random paths
    if n <= 100:  # Only compute for manageable sizes
        num_paths = min(100, n * 10)
        for _ in range(num_paths):
            i, j = random.sample(range(n), 2)
            # Find shortest path using nearest neighbor approximation
            current = i
            visited = {i}
            while current != j:
                # Find nearest unvisited neighbor to j
                unvisited = [x for x in range(n) if x not in visited]
                if not unvisited:
                    break
                # Prefer nodes that are close to both current and j
                scores = dist[current, unvisited] * 0.7 + dist[j, unvisited] * 0.3
                next_node = unvisited[np.argmin(scores)]
                centrality_betweenness[next_node] += 1
                visited.add(next_node)
                current = next_node
    
    # Combine centrality measures
    max_harmonic = np.max(centrality_harmonic)
    if max_harmonic > eps:
        centrality_harmonic = centrality_harmonic / max_harmonic
    
    max_betweenness = np.max(centrality_betweenness)
    if max_betweenness > eps:
        centrality_betweenness = centrality_betweenness / max_betweenness
    
    centrality = 0.7 * centrality_harmonic + 0.3 * centrality_betweenness
    centrality = (centrality - np.min(centrality)) / (np.max(centrality) - np.min(centrality) + eps)
    
    edge_potential = np.outer(centrality, centrality)
    
    # 3. Solution quality feedback loop
    candidate_scores = np.zeros_like(dist)
    
    if n <= 200:  # Only generate candidate solutions for manageable instances
        num_candidates = min(50, n * 5)
        candidate_tours = []
        candidate_lengths = []
        
        # Generate diverse candidate solutions
        for _ in range(num_candidates):
            if random.random() < 0.7:
                # Nearest neighbor tour
                start = random.randint(0, n-1)
                tour = [start]
                visited = set([start])
                while len(tour) < n:
                    current = tour[-1]
                    unvisited = [i for i in range(n) if i not in visited]
                    # Use weighted selection: prefer shorter edges and nodes with high centrality
                    weights_nn = 0.8 * (1 / (dist[current, unvisited] + eps)) + 0.2 * centrality[unvisited]
                    next_node = unvisited[np.random.choice(len(unvisited), p=weights_nn/weights_nn.sum())]
                    tour.append(next_node)
                    visited.add(next_node)
            else:
                # Random tour
                tour = random.sample(range(n), n)
            
            # Calculate tour length
            length = sum(dist[tour[i], tour[(i+1)%n]] for i in range(n))
            candidate_tours.append(tour)
            candidate_lengths.append(length)
        
        # Normalize lengths and select top 30% of tours
        if candidate_lengths:
            candidate_lengths = np.array(candidate_lengths)
            length_scores = 1 - (candidate_lengths - np.min(candidate_lengths)) / (np.max(candidate_lengths) - np.min(candidate_lengths) + eps)
            
            # Weight edges by their appearance in good tours
            for tour, score in zip(candidate_tours, length_scores):
                for i in range(n):
                    j = tour[(i+1)%n]
                    candidate_scores[tour[i], j] += score
                    candidate_scores[j, tour[i]] += score
            
            # Normalize candidate scores
            max_candidate = np.max(candidate_scores)
            if max_candidate > eps:
                candidate_scores = candidate_scores / max_candidate
    
    # 4. Dynamic weight refinement based on component correlation
    # Calculate component correlations and adjust weights
    components = {
        'inv_dist': inv_dist,
        'nn': nn_mask.astype(float) * inv_dist,
        'mst': mst_mask.astype(float),
        'centrality': edge_potential,
        'candidate': candidate_scores
    }
    
    # If we have candidate scores, adjust weights based on correlation
    if np.max(candidate_scores) > eps:
        for key in weights:
            if np.max(components[key]) > eps:
                # Calculate correlation with candidate scores (flattened arrays)
                flat_comp = components[key].flatten()
                flat_cand = candidate_scores.flatten()
                mask = flat_cand > 0
                if np.sum(mask) > 10:
                    comp_values = flat_comp[mask]
                    cand_values = flat_cand[mask]
                    if np.std(comp_values) > eps and np.std(cand_values) > eps:
                        corr = np.corrcoef(comp_values, cand_values)[0, 1]
                        # Adjust weight based on correlation (positive correlation increases weight)
                        if not np.isnan(corr):
                            weights[key] *= (1 + 0.5 * max(0, corr))
    
    # Normalize weights
    weight_sum = sum(weights.values())
    for key in weights:
        weights[key] /= weight_sum
    
    # 5. Combine all factors with refined weights
    heuristics_matrix = weights['inv_dist'] * inv_dist
    heuristics_matrix[nn_mask] += weights['nn'] * inv_dist[nn_mask]
    heuristics_matrix[mst_mask] += weights['mst']
    heuristics_matrix += weights['centrality'] * edge_potential
    heuristics_matrix += weights['candidate'] * candidate_scores
    
    # 6. Adaptive sparsification with connectivity guarantee
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    for i in range(n):
        row_vals = heuristics_matrix[i].copy()
        row_vals[i] = -np.inf
        
        # Dynamic edge selection based on value distribution
        sorted_vals = np.sort(row_vals)[::-1]
        if len(sorted_vals) > 3:
            # Find natural elbow in the distribution
            diffs = np.diff(sorted_vals[:min(15, n)])
            if len(diffs) > 0:
                rel_diffs = diffs / (sorted_vals[:len(diffs)] + eps)
                # Look for significant drops
                significant_drops = np.where(rel_diffs > 0.2)[0]
                if len(significant_drops) > 0:
                    keep_count = min(max(3, n//4), significant_drops[0] + 1)
                else:
                    keep_count = min(max(3, n//4), int(5 + n * 0.1 * cv_dist))
            else:
                keep_count = max(3, min(8, n//4))
        else:
            keep_count = max(1, n-1)
        
        # Keep top edges
        top_indices = np.argsort(row_vals)[-keep_count:]
        sparsity_mask[i, top_indices] = True
    
    # Ensure symmetry and minimum connectivity
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Force connectivity through MST if needed
    if n > 2:
        # Check if graph is connected (approximation)
        visited = set([0])
        stack = [0]
        while stack:
            node = stack.pop()
            neighbors = np.where(sparsity_mask[node])[0]
            for neighbor in neighbors:
                if neighbor not in visited:
                    visited.add(neighbor)
                    stack.append(neighbor)
        
        if len(visited) < n:
            # Add MST edges to ensure connectivity
            sparsity_mask = sparsity_mask | mst_mask
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # 7. Row-wise exponential normalization with temperature
    for i in range(n):
        row = heuristics_matrix[i]
        max_val = np.max(row)
        if max_val > eps:
            # Adaptive temperature based on row statistics
            row_mean = np.mean(row[row > 0])
            if row_mean > eps:
                temperature = max_val / (row_mean + eps)
                temperature = min(max(temperature, 0.5), 2.0)
            else:
                temperature = 1.0
            
            exp_row = np.exp(row / (max_val * temperature + eps))
            heuristics_matrix[i] = exp_row / (np.sum(exp_row) + eps)
    
    # Ensure symmetry and zero diagonal
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Global normalization to [0,1]
    max_val = np.max(heuristics_matrix)
    if max_val > eps:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
