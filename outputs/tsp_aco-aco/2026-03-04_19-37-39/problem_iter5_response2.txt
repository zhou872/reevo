import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with adaptive weighting
    and lightweight learning from instance characteristics.
    
    Key improvements in v2:
    1. Adaptive weight tuning based on distance variability (coefficient of variation)
    2. Multi-factor edge evaluation with adjusted component contributions
    3. Top-m sparsification with adaptive connectivity preservation
    4. Enhanced edge importance scoring
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ========== 1. ADAPTIVE PARAMETER TUNING ==========
    # Compute coefficient of variation of distances for adaptive tuning
    distances_no_diag = distance_matrix[~np.eye(n, dtype=bool)].flatten()
    if np.mean(distances_no_diag) > 0:
        cv = np.std(distances_no_diag) / np.mean(distances_no_diag)
    else:
        cv = 0.5  # Default value
    
    # Adaptive weights based on CV and problem size
    if cv > 1.0:
        # High variability: emphasize inverse distance more
        inv_dist_weight = 0.7
        centrality_weight = 0.2
        nn_boost = 1.3
        mst_boost = 2.0
    elif cv > 0.5:
        # Medium variability: balanced approach
        inv_dist_weight = 0.5
        centrality_weight = 0.4
        nn_boost = 1.4
        mst_boost = 2.2
    else:
        # Low variability: emphasize centrality more
        inv_dist_weight = 0.3
        centrality_weight = 0.6
        nn_boost = 1.5
        mst_boost = 2.4
    
    # Adjust for problem size
    if n < 10:
        inv_dist_weight *= 1.2
    elif n > 100:
        centrality_weight *= 1.1
    
    # ========== 2. BASE COMPONENTS ==========
    # Inverse distance with regularization
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(inv_dist, 0)
    
    # Nearest neighbors with adaptive k
    k = max(2, min(5, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argpartition(distances, k)[:k]
        nn_mask[i, nn_indices] = True
    nn_mask = nn_mask | nn_mask.T  # Make symmetric
    
    # Minimum spanning tree
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # ========== 3. ENHANCED CENTRALITY MEASURE ==========
    # Row sums excluding diagonal
    row_sums = np.sum(distance_matrix, axis=1) - np.diag(distance_matrix)
    
    # Robust centrality with percentile clipping
    avg_dist = row_sums / (n - 1)
    robust_avg = np.clip(avg_dist, 
                        np.percentile(avg_dist, 20), 
                        np.percentile(avg_dist, 80))
    centrality = 1 / (robust_avg + 1e-10)
    
    # Apply softmax normalization
    centrality = np.exp(centrality - np.max(centrality))
    centrality = centrality / (np.sum(centrality) + 1e-10)
    
    # Edge potential as product of node centralities
    edge_potential = np.outer(centrality, centrality)
    np.fill_diagonal(edge_potential, 0)
    
    # ========== 4. MULTI-FACTOR COMBINATION ==========
    heuristic_scores = np.zeros_like(distance_matrix)
    
    # Base combination with adaptive weights
    base_component = inv_dist_weight * inv_dist + centrality_weight * edge_potential
    
    # Add distance-based term for edges that are both short and connect central nodes
    distance_factor = np.exp(-distance_matrix / np.median(distance_matrix[~np.eye(n, dtype=bool)]))
    np.fill_diagonal(distance_factor, 0)
    
    heuristic_scores = base_component + 0.3 * distance_factor * edge_potential
    
    # Boost important edges
    heuristic_scores[nn_mask] *= nn_boost
    heuristic_scores[mst_mask] *= mst_boost
    
    # ========== 5. TOP-M SPARSIFICATION ==========
    # Adaptive m: retain only top m connections per node
    m = max(3, min(8, n // 3))
    sparsity_mask = np.zeros_like(heuristic_scores, dtype=bool)
    
    # Pre-compute rankings
    combined_ranks = np.argsort(heuristic_scores, axis=1)[:, ::-1]
    
    # Keep top m edges per node
    for i in range(n):
        top_m_indices = combined_ranks[i, :m]
        sparsity_mask[i, top_m_indices] = True
        
        # Always include MST edges for connectivity
        sparsity_mask[i, mst_mask[i]] = True
    
    # Ensure minimum connectivity for small instances
    min_connections = max(2, n // 10)
    for i in range(n):
        current_connections = np.sum(sparsity_mask[i])
        if current_connections < min_connections:
            needed = min_connections - current_connections
            next_best = combined_ranks[i, m:m + needed]
            sparsity_mask[i, next_best] = True
    
    # Ensure symmetry
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # ========== 6. FINAL PROCESSING ==========
    # Apply sparsification
    heuristic_scores = heuristic_scores * sparsity_mask
    
    # Final normalization with outlier handling
    non_zero = heuristic_scores[heuristic_scores > 0]
    if len(non_zero) > 0:
        mean_val = np.mean(non_zero)
        std_val = np.std(non_zero)
        heuristic_scores = np.clip(heuristic_scores, 0, mean_val + 2 * std_val)
    
    max_val = np.max(heuristic_scores)
    if max_val > 0:
        heuristic_scores = heuristic_scores / max_val
    
    # Zero out diagonal
    np.fill_diagonal(heuristic_scores, 0)
    
    return heuristic_scores
