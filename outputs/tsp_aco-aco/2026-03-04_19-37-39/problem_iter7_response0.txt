import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics using centrality-based edge potential approximation,
    adaptive multi-factor fusion, and intelligent graph sparsification.
    
    Core strategies:
    1. Node centrality approximation via inverse sum of distances
    2. Edge potential approximation as product of incident node centralities
    3. Adaptive fusion with inverse distance and solution sampling components
    4. Size-aware sparsification preserving promising edges
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    # Adaptive parameters based on instance properties
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    
    # Size-dependent factors
    log_n = np.log(n + 1)
    size_factor = min(1.0, n / 100.0)
    
    # ===== 2. NODE CENTRALITY APPROXIMATION =====
    # Compute node centrality as inverse of sum of distances to other nodes
    row_sums = np.sum(distance_matrix, axis=1) - np.diag(distance_matrix)
    centrality = 1.0 / (row_sums + 1e-10)
    
    # Normalize centrality
    if centrality.max() > centrality.min():
        centrality = (centrality - centrality.min()) / (centrality.max() - centrality.min())
    
    # ===== 3. EDGE POTENTIAL VIA CENTRALITY PRODUCT =====
    # Edge potential as product of incident node centralities
    edge_potential = np.outer(centrality, centrality)
    np.fill_diagonal(edge_potential, 0)
    
    # ===== 4. SOLUTION SAMPLING (SIMPLIFIED) =====
    # Generate diverse tours quickly without intensive local search
    num_tours = min(10, max(3, n // 5))
    edge_frequency = np.zeros_like(distance_matrix)
    
    for _ in range(num_tours):
        # Simple nearest neighbor from random start
        start = np.random.randint(0, n)
        tour = [start]
        unvisited = set(range(n)) - {start}
        
        while unvisited:
            current = tour[-1]
            # Consider top 3 nearest unvisited nodes
            unvisited_list = list(unvisited)
            distances_to_current = distance_matrix[current, unvisited_list]
            top_k = min(3, len(unvisited_list))
            top_indices = np.argpartition(distances_to_current, top_k)[:top_k]
            next_node = unvisited_list[np.random.choice(top_indices)]
            
            tour.append(next_node)
            unvisited.remove(next_node)
        
        # Record edges
        for i in range(n):
            j = (i + 1) % n
            u, v = tour[i], tour[j]
            edge_frequency[u, v] += 1
            edge_frequency[v, u] += 1
    
    # Normalize edge frequency
    if edge_frequency.max() > 0:
        edge_frequency = edge_frequency / edge_frequency.max()
    
    # ===== 5. INVERSE DISTANCE COMPONENT =====
    inv_dist = 1.0 / (distance_matrix + 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # ===== 6. MST COMPONENT =====
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # ===== 7. ADAPTIVE WEIGHT COMBINATION =====
    # Weights based on instance characteristics
    if n <= 20:
        # Small instances: emphasize distance information
        w_dist = 0.5
        w_cent = 0.3
        w_freq = 0.2
    elif n <= 100:
        # Medium instances: balanced approach
        w_dist = 0.4
        w_cent = 0.4
        w_freq = 0.2
    else:
        # Large instances: emphasize centrality structure
        w_dist = 0.3
        w_cent = 0.5
        w_freq = 0.2
    
    # Adjust based on distance variability
    if cv > 0.5:  # High variability in distances
        w_dist = min(0.6, w_dist * 1.3)  # Boost distance importance
        w_cent = max(0.2, w_cent * 0.8)  # Reduce centrality importance
    
    # Normalize weights
    total = w_dist + w_cent + w_freq
    w_dist /= total
    w_cent /= total
    w_freq /= total
    
    # ===== 8. COMBINE COMPONENTS =====
    heuristics_matrix = (
        w_dist * inv_dist +
        w_cent * edge_potential +
        w_freq * edge_frequency
    )
    
    # Boost MST edges
    heuristics_matrix[mst_mask] *= 1.5
    
    # ===== 9. INTELLIGENT SPARSIFICATION =====
    # Determine edges to keep per node
    min_edges = max(2, int(1.5 * log_n))
    max_edges = min(n-1, int(3.5 * log_n))
    
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        # Adaptive number of edges based on instance size and variability
        if cv > 0.5:  # More edges when distances are variable
            n_edges = min(max_edges, max(min_edges, int((n-1) * 0.3)))
        else:
            n_edges = min(max_edges, max(min_edges, int((n-1) * 0.2)))
        
        # Keep top edges
        top_indices = np.argsort(scores)[-n_edges:]
        sparsity_mask[i, top_indices] = True
    
    # Ensure MST edges are always included
    sparsity_mask = sparsity_mask | mst_mask
    
    # Ensure symmetry
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 10. FINAL PROCESSING =====
    # Ensure symmetry
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    
    # Zero diagonal
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        q10 = np.percentile(non_zero, 10)
        q90 = np.percentile(non_zero, 90)
        heuristics_matrix = np.clip(heuristics_matrix, q10, q90)
        if heuristics_matrix.max() > 0:
            heuristics_matrix = heuristics_matrix / heuristics_matrix.max()
    
    return heuristics_matrix
