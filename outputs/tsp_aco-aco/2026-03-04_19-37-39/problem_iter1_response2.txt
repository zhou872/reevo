import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection.
    
    Combines multiple factors to estimate edge promise:
    1. Reciprocal of distance (baseline)
    2. Sparsification via k-nearest neighbors
    3. Savings heuristic (Clarke-Wright inspired)
    4. Node centrality adjustment
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape, with unpromising edges set to zero
    """
    n = distance_matrix.shape[0]
    
    # Avoid division by zero
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Baseline: reciprocal of distance
    baseline = 1.0 / safe_dist
    
    # 2. Sparsification: keep only k-nearest neighbors for each node
    k = max(3, int(np.sqrt(n)))  # Adaptive k based on problem size
    
    # Create mask for k-nearest neighbors
    mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        # Get indices of k+1 smallest distances (including self at index 0)
        nearest = np.argsort(distance_matrix[i])[:k+1]
        # Exclude self
        nearest = nearest[nearest != i]
        mask[i, nearest] = True
    
    # Make symmetric (undirected TSP)
    mask = mask | mask.T
    
    # 3. Savings heuristic (inspired by Clarke-Wright)
    # Compute potential savings for each edge (i,j)
    # savings = d(0,i) + d(0,j) - d(i,j), using node with highest centrality as "hub"
    
    # Find central node (minimum average distance to all others)
    avg_distances = distance_matrix.mean(axis=1)
    hub_idx = np.argmin(avg_distances)
    
    savings = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if i != hub_idx and j != hub_idx:
                # Classic savings formula
                s = (distance_matrix[hub_idx, i] + 
                     distance_matrix[hub_idx, j] - 
                     distance_matrix[i, j])
                savings[i, j] = savings[j, i] = max(s, 0)  # Non-negative
    
    # Normalize savings
    if savings.max() > 0:
        savings = savings / savings.max()
    
    # 4. Node centrality adjustment
    # Nodes with lower average distance are more central
    centrality = 1.0 / (avg_distances + 1e-10)
    centrality = centrality / centrality.max()
    
    # Combine factors with weighted geometric mean
    alpha, beta, gamma = 0.4, 0.3, 0.3  # Weighting factors
    
    # Adjust baseline by centrality (edges connecting central nodes get boost)
    centrality_factor = np.outer(centrality, centrality)
    adjusted_baseline = baseline * (0.5 + 0.5 * centrality_factor)
    
    # Combine factors
    heuristics = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if mask[i, j]:  # Only consider promising edges
                # Geometric mean of factors
                factors = [
                    adjusted_baseline[i, j] ** alpha,
                    (1.0 + savings[i, j]) ** beta,
                    centrality_factor[i, j] ** gamma
                ]
                score = np.prod(factors)
                heuristics[i, j] = heuristics[j, i] = score
    
    # Additional sparsification: keep only top m% of edges
    m_percent = min(30, 100 * (2*k) / n)  # Adaptive threshold
    threshold = np.percentile(heuristics[heuristics > 0], 100 - m_percent)
    heuristics[heuristics < threshold] = 0
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
