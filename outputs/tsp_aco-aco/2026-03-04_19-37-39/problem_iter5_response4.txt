import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with adaptive weighting based on 
    coefficient of variation and top-m sparsification.
    
    Key improvements over v1:
    1. Adaptive weight tuning using coefficient of variation (CV) of distances.
    2. Top-m sparsification per node with adaptive m = max(3, min(8, n//3)).
    3. Ensures connectivity via MST inclusion and symmetric masking.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for TSP solutions. Unpromising edges are set to zero.
    """
    n = distance_matrix.shape[0]
    
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ========== 1. COEFFICIENT OF VARIATION COMPUTATION ==========
    # Use upper triangle distances excluding diagonal for CV calculation
    dist_flat = distance_matrix[np.triu_indices(n, k=1)]
    mean_dist = np.mean(dist_flat)
    std_dist = np.std(dist_flat)
    if mean_dist == 0:
        cv = 0.0
    else:
        cv = std_dist / mean_dist
    
    # Normalize CV for weight adjustment (assumed range 0.1 to 1.0 for typical TSP instances)
    cv_min = 0.1
    cv_max = 1.0
    cv_norm = np.clip((cv - cv_min) / (cv_max - cv_min), 0.0, 1.0)
    
    # ========== 2. ADAPTIVE WEIGHT TUNING BASED ON CV ==========
    # Weights vary linearly with CV: higher CV gives more weight to inverse distance
    inv_dist_weight = 0.4 + 0.3 * cv_norm  # Ranges from 0.4 to 0.7
    centrality_weight = 0.6 - 0.3 * cv_norm  # Ranges from 0.6 to 0.3
    
    # Boost factors based on problem size (unchanged from v1 for simplicity)
    if n < 10:
        nn_boost = 1.2
        mst_boost = 1.8
    elif n < 50:
        nn_boost = 1.3
        mst_boost = 2.0
    else:
        nn_boost = 1.4
        mst_boost = 2.2
    
    # ========== 3. BASE COMPONENTS COMPUTATION ==========
    # Inverse distance with regularization
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(inv_dist, 0)
    
    # Nearest neighbors with adaptive k
    k = max(2, min(5, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argpartition(distances, k)[:k]
        nn_mask[i, nn_indices] = True
    nn_mask = nn_mask | nn_mask.T  # Make symmetric
    
    # Minimum spanning tree for connectivity guarantee
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # Enhanced centrality measure
    row_sums = np.sum(distance_matrix, axis=1) - np.diag(distance_matrix)
    avg_dist = row_sums / (n - 1)
    robust_avg = np.clip(avg_dist, np.percentile(avg_dist, 10), np.percentile(avg_dist, 90))
    centrality = 1 / (robust_avg + 1e-10)
    centrality = np.exp(centrality - np.max(centrality))
    centrality = centrality / (np.sum(centrality) + 1e-10)
    edge_potential = np.outer(centrality, centrality)
    np.fill_diagonal(edge_potential, 0)
    
    # ========== 4. ADAPTIVE COMBINATION WITH BOOSTS ==========
    base_component = inv_dist_weight * inv_dist + centrality_weight * edge_potential
    heuristic_scores = base_component.copy()
    heuristic_scores[nn_mask] *= nn_boost
    heuristic_scores[mst_mask] *= mst_boost
    
    # ========== 5. TOP-M SPARSIFICATION ==========
    m = max(3, min(8, n // 3))  # Adaptive m as per suggestion
    sparsity_mask = np.zeros_like(heuristic_scores, dtype=bool)
    
    for i in range(n):
        # Get heuristic scores for node i, excluding self
        scores_i = heuristic_scores[i].copy()
        scores_i[i] = -np.inf  # Exclude diagonal from top-m selection
        # Select top m edges based on heuristic scores
        top_indices = np.argpartition(scores_i, -m)[-m:]
        sparsity_mask[i, top_indices] = True
        # Always include MST edges for connectivity
        sparsity_mask[i, mst_mask[i]] = True
    
    # Ensure symmetry
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsification
    heuristic_scores = heuristic_scores * sparsity_mask
    
    # ========== 6. FINAL PROCESSING ==========
    max_val = np.max(heuristic_scores)
    if max_val > 0:
        heuristic_scores = heuristic_scores / max_val
    
    np.fill_diagonal(heuristic_scores, 0)
    
    return heuristic_scores
