import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with improved edge betweenness sampling
    and optimized adaptive weighting strategies.
    
    Key improvements over v1:
    1. Direct edge betweenness via path sampling (replaces node centrality)
    2. More sophisticated adaptive weight tuning using instance metrics
    3. Refined sparsification with distance-aware edge selection
    4. Reduced computational overhead while maintaining quality
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of pairwise distances between nodes (n x n)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n < 4:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    dist_cv = dist_std / (dist_mean + 1e-10)
    
    # Size-adaptive parameters
    log_n = np.log(n + 1)
    density_factor = min(1.0, 2 * np.log(n + 1) / np.log(50))
    
    # ===== 2. SOLUTION SAMPLING FOR EDGE FREQUENCY =====
    def generate_initial_tour(method='nn', start_node=0):
        """Fast tour construction methods."""
        if method == 'nn':
            tour = [start_node]
            unvisited = set(range(n)) - {start_node}
            while unvisited:
                current = tour[-1]
                nearest = min(unvisited, key=lambda x: distance_matrix[current, x])
                tour.append(nearest)
                unvisited.remove(nearest)
            return np.array(tour)
        
        elif method == 'greedy':
            if n <= 3:
                return np.random.permutation(n)
            tour = list(np.random.choice(n, 3, replace=False))
            remaining = set(range(n)) - set(tour)
            
            while remaining:
                best_cost = float('inf')
                best_node = None
                best_pos = -1
                
                for node in remaining:
                    for i in range(len(tour)):
                        j = (i + 1) % len(tour)
                        cost_inc = (distance_matrix[tour[i], node] + 
                                   distance_matrix[node, tour[j]] - 
                                   distance_matrix[tour[i], tour[j]])
                        if cost_inc < best_cost:
                            best_cost = cost_inc
                            best_node = node
                            best_pos = i + 1
                
                tour.insert(best_pos, best_node)
                remaining.remove(best_node)
            
            return np.array(tour)
        
        else:  # random
            return np.random.permutation(n)
    
    def fast_2opt(tour, max_iter=20):
        """Simplified 2-opt local search."""
        improved = True
        current_tour = tour.copy()
        iteration = 0
        
        while improved and iteration < max_iter:
            improved = False
            for i in range(n-2):
                for j in range(i+2, min(n, i + int(np.sqrt(n)) + 3)):
                    if j == n-1 and i == 0:
                        continue
                    
                    a, b = current_tour[i], current_tour[(i+1)%n]
                    c, d = current_tour[j], current_tour[(j+1)%n]
                    
                    if distance_matrix[a, b] + distance_matrix[c, d] > \
                       distance_matrix[a, c] + distance_matrix[b, d]:
                        
                        current_tour[i+1:j+1] = current_tour[i+1:j+1][::-1]
                        improved = True
                        break
                if improved:
                    break
            iteration += 1
        
        return current_tour
    
    # Generate diverse tours
    num_tours = min(12, max(3, n // 4))
    tours = []
    
    for _ in range(num_tours):
        method = np.random.choice(['nn', 'greedy', 'random'], 
                                  p=[0.5, 0.3, 0.2])
        start = np.random.randint(0, n) if method == 'nn' else 0
        tour = generate_initial_tour(method, start)
        tour = fast_2opt(tour, max_iter=int(15 * density_factor))
        tours.append(tour)
    
    # Edge frequency matrix
    edge_frequency = np.zeros_like(distance_matrix)
    for tour in tours:
        for i in range(n):
            u, v = tour[i], tour[(i+1)%n]
            edge_frequency[u, v] += 1
            edge_frequency[v, u] += 1
    
    if edge_frequency.max() > 0:
        edge_frequency /= edge_frequency.max()
    
    # ===== 3. SAMPLED EDGE BETWEENNESS =====
    edge_betweenness = np.zeros_like(distance_matrix)
    
    # Adaptive sample size based on instance characteristics
    num_samples = min(100, max(20, int(n * (1 + dist_cv) * 2)))
    
    for _ in range(num_samples):
        # Random source-destination pair
        s, t = np.random.choice(n, 2, replace=False)
        path = [s]
        visited = {s}
        
        # Greedy path construction
        current = s
        while current != t:
            # Consider k nearest neighbors
            k = min(5, n-1)
            neighbors = np.argsort(distance_matrix[current])[1:k+1]
            
            # Filter unvisited neighbors
            candidates = [nbr for nbr in neighbors if nbr not in visited]
            
            if not candidates:
                # If all neighbors visited, choose closest to target
                candidates = [np.argmin([distance_matrix[current, j] + distance_matrix[j, t] 
                                       for j in range(n) if j != current and j not in visited])]
            
            # Choose neighbor minimizing distance to target
            next_node = min(candidates, key=lambda x: distance_matrix[current, x] + distance_matrix[x, t])
            
            # Update path
            path.append(next_node)
            visited.add(next_node)
            current = next_node
        
        # Count edges in path
        for i in range(len(path)-1):
            u, v = path[i], path[i+1]
            edge_betweenness[u, v] += 1
            edge_betweenness[v, u] += 1
    
    # Normalize edge betweenness
    if edge_betweenness.max() > 0:
        edge_betweenness /= edge_betweenness.max()
    
    # ===== 4. DISTANCE-BASED COMPONENTS =====
    # Inverse distance component
    inv_dist = 1 / (distance_matrix + 1e-10)
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > 0:
        inv_dist /= inv_dist.max()
    
    # MST component
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_scores = np.zeros_like(distance_matrix)
    mst_scores[mst_mask] = 1 / (distance_matrix[mst_mask] + 1e-10)
    if mst_scores.max() > 0:
        mst_scores /= mst_scores.max()
    
    # ===== 5. ADAPTIVE HYBRID WEIGHTING =====
    # Analyze component characteristics
    edge_freq_std = np.std(edge_frequency[np.triu_indices(n, k=1)])
    betweenness_std = np.std(edge_betweenness[np.triu_indices(n, k=1)])
    
    # Dynamic weight adjustment
    w_freq = 0.3 + 0.3 * min(1.0, edge_freq_std * 3)  # More weight if clear patterns
    w_between = 0.25 + 0.2 * min(1.0, betweenness_std * 2)  # More weight if betweenness varies
    w_dist = 0.25 + 0.15 * min(1.0, dist_cv)  # More weight if distances vary
    w_mst = 0.2 + 0.15 * density_factor  # More weight for larger instances
    
    # Normalize weights
    total = w_freq + w_between + w_dist + w_mst
    w_freq /= total
    w_between /= total
    w_dist /= total
    w_mst /= total
    
    # ===== 6. COMBINE COMPONENTS =====
    heuristics_matrix = (
        w_freq * edge_frequency +
        w_between * edge_betweenness +
        w_dist * inv_dist +
        w_mst * mst_scores
    )
    
    # ===== 7. INTELLIGENT GRAPH SPARSIFICATION =====
    # Dynamic edge selection per node
    min_edges = max(2, int(1.5 * log_n))
    max_edges = min(n-1, min_edges * 4)
    
    sparsity_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        # Determine number of edges based on distance distribution
        node_distances = distance_matrix[i]
        distance_q75 = np.percentile(node_distances[node_distances > 0], 75)
        
        # Keep more edges for nodes with uniform distances
        if dist_cv < 0.3:
            n_edges = min(max_edges, int((n-1) * 0.3))
        else:
            n_edges = min(max_edges, max(min_edges, int((n-1) * (0.15 + 0.1 * density_factor))))
        
        # Prioritize edges with good heuristic scores and reasonable distances
        good_edges = (distance_matrix[i] <= distance_q75)
        good_indices = np.where(good_edges)[0]
        good_indices = good_indices[good_indices != i]
        
        if len(good_indices) > 0:
            # Select top edges among good candidates
            if len(good_indices) > n_edges:
                top_good = good_indices[np.argsort(scores[good_indices])[-n_edges:]]
            else:
                top_good = good_indices
            
            sparsity_mask[i, top_good] = True
    
    # Ensure MST connectivity
    sparsity_mask = sparsity_mask | mst_mask
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparsity_mask
    
    # ===== 8. FINAL ENHANCEMENTS =====
    # Ensure symmetry
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    
    # Zero diagonal
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize non-zero values
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        heuristics_matrix[heuristics_matrix > 0] = (
            heuristics_matrix[heuristics_matrix > 0] / non_zero.max()
        )
    
    return heuristics_matrix
