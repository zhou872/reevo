```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse import csr_matrix
from typing import List, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with adaptive solution sampling and ML-inspired edge prediction.
    
    Core innovations:
    1. Adaptive multi-strategy tour generation with reinforcement
    2. Ensemble learning from sampled tours with feature importance
    3. Scalable centrality approximations using random walks
    4. Intelligent sparsification with connectivity guarantees
    5. Machine learning-inspired edge prediction
    
    Args:
        distance_matrix: Square symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    # Compute instance statistics for adaptive behavior
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    dist_mean = np.mean(distances)
    dist_std = np.std(distances)
    cv = dist_std / (dist_mean + 1e-10)
    log_n = np.log(n + 1)
    
    # Adaptive parameters
    density_factor = min(1.0, n / 50.0)
    if n < 50:
        base_samples = 20
    elif n < 200:
        base_samples = 15
    else:
        base_samples = max(5, int(20 * np.sqrt(200 / n)))
    
    # ===== 2. ADAPTIVE TOUR GENERATION WITH REINFORCEMENT =====
    
    def generate_tour_with_method(method: str, start_node: int = None) -> np.ndarray:
        """Generate tours using various construction heuristics."""
        if method == 'nn':
            # Nearest neighbor with random start
            if start_node is None:
                start_node = np.random.randint(0, n)
            tour = [start_node]
            unvisited = set(range(n)) - {start_node}
            while unvisited:
                current = tour[-1]
                nearest = min(unvisited, key=lambda x: distance_matrix[current, x])
                tour.append(nearest)
                unvisited.remove(nearest)
            return np.array(tour)
        
        elif method == 'greedy_insertion':
            # Greedy insertion heuristic
            if n <= 3:
                return np.random.permutation(n)
            
            # Start with a random triangle
            start_nodes = np.random.choice(n, 3, replace=False)
            tour = list(start_nodes)
            tour.append(start_nodes[0])  # Close triangle
            remaining = list(set(range(n)) - set(start_nodes))
            
            while remaining:
                best_improvement = float('inf')
                best_node = None
                best_pos = -1
                
                for node in remaining:
                    for i in range(len(tour)-1):
                        # Cost increase = d(tour[i], node) + d(node, tour[i+1]) - d(tour[i], tour[i+1])
                        cost_inc = (distance_matrix[tour[i], node] + 
                                   distance_matrix[node, tour[i+1]] - 
                                   distance_matrix[tour[i], tour[i+1]])
                        if cost_inc < best_improvement:
                            best_improvement = cost_inc
                            best_node = node
                            best_pos = i + 1
                
                if best_node is not None:
                    tour.insert(best_pos, best_node)
                    remaining.remove(best_node)
            
            return np.array(tour[:-1])
        
        elif method == 'farthest_insertion':
            # Farthest insertion heuristic
            if n <= 3:
                return np.random.permutation(n)
            
            # Start with two farthest apart nodes
            max_dist = -1
            start_pair = (0, 1)
            for i in range(n):
                for j in range(i+1, n):
                    if distance_matrix[i, j] > max_dist:
                        max_dist = distance_matrix[i, j]
                        start_pair = (i, j)
            
            tour = list(start_pair)
            tour.append(start_pair[0])  # Close initial edge
            remaining = list(set(range(n)) - set(start_pair))
            
            while remaining:
                # Find farthest node from current tour
                farthest_dist = -1
                farthest_node = None
                
                for node in remaining:
                    min_dist_to_tour = min(distance_matrix[node, t] for t in tour[:-1])
                    if min_dist_to_tour > farthest_dist:
                        farthest_dist = min_dist_to_tour
                        farthest_node = node
                
                # Insert at best position
                best_improvement = float('inf')
                best_pos = -1
                
                for i in range(len(tour)-1):
                    cost_inc = (distance_matrix[tour[i], farthest_node] + 
                               distance_matrix[farthest_node, tour[i+1]] - 
                               distance_matrix[tour[i], tour[i+1]])
                    if cost_inc < best_improvement:
                        best_improvement = cost_inc
                        best_pos = i + 1
                
                tour.insert(best_pos, farthest_node)
                remaining.remove(farthest_node)
            
            return np.array(tour[:-1])
        
        elif method == 'random':
            return np.random.permutation(n)
        
        elif method == 'spanning_tree':
            # Christofides-inspired: MST + minimum weight matching approximation
            # Compute MST
            mst = minimum_spanning_tree(csr_matrix(distance_matrix)).toarray()
            mst_degrees = (mst > 0).sum(axis=0) + (mst > 0).sum(axis=1)
            odd_nodes = np.where(mst_degrees % 2 == 1)[0]
            
            if len(odd_nodes) <= 1:
                # Fall back to MST traversal
                return np.random.permutation(n)
            
            # Greedy matching on odd nodes
            matched = np.zeros(len(odd_nodes), dtype=bool)
            matching_edges = []
            
            for i in range(len(odd_nodes)):
                if not matched[i]:
                    best_j = -1
                    best_dist = float('inf')
                    for j in range(i+1, len(odd_nodes)):
                        if not matched[j]:
                            dist = distance_matrix[odd_nodes[i], odd_nodes[j]]
                            if dist < best_dist:
                                best_dist = dist
                                best_j = j
                    if best_j != -1:
                        matching_edges.append((odd_nodes[i], odd_nodes[j]))
                        matched[i] = matched[best_j] = True
            
            # Combine MST and matching
            combined_graph = mst.copy()
            for i, j in matching_edges:
                combined_graph[i, j] = max(combined_graph[i, j], distance_matrix[i, j])
                combined_graph[j, i] = max(combined_graph[j, i], distance_matrix[j, i])
            
            # Hierholzer's algorithm for Eulerian circuit
            # Simplified: Use DFS on combined graph
            adj_list = [list(np.where(combined_graph[i] > 0)[0]) for i in range(n)]
            visited = np.zeros(n, dtype=bool)
            tour = []
            
            def dfs(node):
                visited[node] = True
                tour.append(node)
                for neighbor in adj_list[node]:
                    if not visited[neighbor]:
                        dfs(neighbor)
            
            dfs(0)
            return np.array(tour)
        
        else:
            return np.random.permutation(n)
    
    def local_search_2opt(tour: np.ndarray, max_iter: int = 30) -> np.ndarray:
        """Apply 2-opt local search."""
        n_tour = len(tour)
        best_tour = tour.copy()
        best_len = sum(distance_matrix[best_tour[i], best_tour[(i+1)%n_tour]] for i in range(n_tour))
        
        improved = True
        iteration = 0
        
        while improved and iteration < max_iter:
            improved = False
            
            for i in range(n_tour - 1):
                for j in range(i + 2, n_tour):
                    if j == n_tour - 1 and i == 0:
                        continue
                    
                    # Calculate gain
                    a, b = best_tour[i], best_tour[(i+1)%n_tour]
                    c, d = best_tour[j], best_tour[(j+1)%n_tour]
                    
                    old_cost = distance_matrix[a, b] + distance_matrix[c, d]
                    new_cost = distance_matrix[a, c] + distance_matrix[b, d]
                    
                    if new_cost < old_cost:
                        # Perform 2-opt swap
                        best_tour = np.concatenate([
                            best_tour[:i+1],
                            best_tour[i+1:j+1][::-1],
                            best_tour[j+1:]
                        ])
                        best_len = best_len - old_cost + new_cost
                        improved = True
                        break
                if improved:
                    break
            iteration += 1
        
        return best_tour
    
    # Generate diverse tours
    tours = []
    method_weights = {
        'nn': 0.3,
        'greedy_insertion': 0.25,
        'farthest_insertion': 0.2,
        'spanning_tree': 0.15 if n < 100 else 0.0,  # Skip for large instances
        'random': 0.1
    }
    
    total_samples = base_samples
    samples_per_method = {}
    remaining = total_samples
    
    for method, weight in method_weights.items():
        if method == 'spanning_tree' and n >= 100:
            continue
        count = max(1, int(total_samples * weight))
        samples_per_method[method] = count
        remaining -= count
    
    # Adjust for rounding
    if remaining > 0:
        samples_per_method['nn'] += remaining
    
    # Generate tours
    for method, count in samples_per_method.items():
        for _ in range(count):
            if method == 'nn':
                start = np.random.randint(0, n)
                tour = generate_tour_with_method(method, start)
            else:
                tour = generate_tour_with_method(method)
            
            # Apply local search
            ls_iter = max(10, int(30 * density_factor))
            tour = local_search_2opt(tour, max_iter=ls_iter)
            tours.append(tour)
    
    # ===== 3. ENSEMBLE LEARNING FROM SAMPLES =====
    
    # Compute edge frequencies
    edge_frequency = np.zeros_like(distance_matrix)
    tour_lengths = []
    
    for tour in tours:
        # Calculate tour length
        tour_len = sum(distance_matrix[tour[i], tour[(i+1)%n]] for i in range(n))
        tour_lengths.append(tour_len)
        
        # Add edge counts with weight based on tour quality
        for i in range(n):
            u, v = tour[i], tour[(i+1)%n]
            # Weight edges from shorter tours more
            weight = 1.0 / (tour_len / dist_mean + 1e-5)
            edge_frequency[u, v] += weight
            edge_frequency[v, u] += weight
    
    # Normalize edge frequencies
    if edge_frequency.max() > 0:
        edge_frequency = edge_frequency / edge_frequency.max()
    
    # Extract quality indicators from tours
    if tour_lengths:
        min_len = min(tour_lengths)
        max_len = max(tour_lengths)
        tour_quality = [(max_len - l) / (max_len - min_len + 1e-10) for l in tour_lengths]
    else:
        tour_quality = [1.0] * len(tours)
    
    # ===== 4. SCALABLE CENTRALITY APPROXIMATIONS =====
    
    # Fast approximate betweenness centrality using random walks
    betweenness = np.zeros(n)
    num_walks = min(100, max(20, n // 5))
    walk_length = min(15, max(5, int(np.log2(n) * 3)))
    
    for _ in range(num_walks):
        start = np.random.randint(0, n)
        visited = set()
        
        # Perform random walk with edge weight bias
        current = start
        visited.add(current)
        
        for step in range(walk_length):
            # Transition probabilities based on inverse distance
            weights = 1.0 / (distance_matrix[current] + 1e-10)
            weights[current] = 0
            weights = weights / weights.sum()
            
            if weights.sum() > 0:
                next_node = np.random.choice(n, p=weights)
                visited.add(next_node)
                current = next_node
        
        # Update betweenness: nodes visited more often in short walks are central
        for node in visited:
            betweenness[node] += 1
    
    # Normalize betweenness
    if betweenness.max() > 0:
        betweenness = betweenness / betweenness.max()
    
    # Degree centrality (simplified)
    inv_dist = 1.0 / (distance_matrix + np.eye(n))
    np.fill_diagonal(inv_dist, 0)
    degree_centrality = inv_dist.sum(axis=1)
    if degree_centrality.max() > 0:
        degree_centrality = degree_centrality / degree_centrality.max()
    
    # Combined centrality
    centrality = 0.6 * betweenness + 0.4 * degree_centrality
    if centrality.max() > 0:
        centrality = centrality / centrality.max()
    
    # Edge centrality potential
    edge_centrality = np.outer(centrality, centrality)
    
    # ===== 5. EDGE FEATURE COMPUTATION =====
    
    # Inverse distance feature
    inv_dist_feature = 1.0 / (distance_matrix + np.eye(n) * 1e10)
    np.fill_diagonal(inv_dist_feature, 0)
    if inv_dist_feature.max() > 0:
        inv_dist_feature = inv_dist_feature / inv_dist_feature.max()
    
    # MST feature
    mst = minimum_spanning_tree(csr_matrix(distance_matrix)).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    mst_feature = np.zeros_like(distance_matrix)
    mst_feature[mst_mask] = 1.0
    
    # Nearest neighbor feature
    k_nn = max(2, min(8, int(np.sqrt(n))))
    nn_feature = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get k nearest neighbors
        neighbors = np.argsort(distance_matrix[i])[1:k_nn+1]
        nn_feature[i, neighbors] = 1.0
        # Reciprocal connections
        for j in neighbors:
            if i in np.argsort(distance_matrix[j])[1:k_nn+1]:
                nn_feature[i, j] = 1.5  # Boost reciprocal connections
    
    nn_feature = np.maximum(nn_feature, nn_feature.T)
    
    # Triangle completion feature
    if n >= 4:
        # Find edges that are part of many triangles
        threshold = np.percentile(distance_matrix, 25)
        short_mask = distance_matrix <= threshold
        np.fill_diagonal(short_mask, False)
        
        # Count common neighbors via matrix multiplication
        common_neighbors = short_mask @ short_mask.T
        triangle_feature = common_neighbors / (n - 2 + 1e-10)
    else:
        triangle_feature = np.zeros_like(distance_matrix)
    
    # ===== 6. ADAPTIVE ENSEMBLE COMBINATION =====
    
    # Determine feature weights based on instance characteristics
    edge_freq_std = np.std(edge_frequency[triu_indices])
    triangle_std = np.std(triangle_feature[triu_indices]) if n >= 4 else 0
    
    # Adaptive weights based on statistics
    w_freq = 0.35 + 0.2 * min(1.0, edge_freq_std * 3)  # More weight when frequencies vary
    w_dist = 0.25 - 0.1 * min(1.0, cv)  # Less weight when distances are uniform
    w_cent = 0.15 + 0.1 * min(1.0, cv)  # More centrality when distances vary
    w_mst = 0.10 + 0.05 * (1 - density_factor)  # More MST for sparse problems
    w_tri = 0.08 + 0.12 * min(1.0, triangle_std * 2) if n >= 4 else 0
    w_nn = 0.07
    
    # Normalize weights
    weights = np.array([w_freq, w_dist, w_cent, w_mst, w_tri, w_nn])
    weights = weights / weights.sum()
    
    # Combine features
    features = [edge_frequency, inv_dist_feature, edge_centrality, mst_feature, triangle_feature, nn_feature]
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    for w, f in zip(weights, features):
        # Normalize each feature to [0, 1] range
        f_min, f_max = f.min(), f.max()
        if f_max - f_min > 1e-10:
            f_norm = (f - f_min) / (f_max - f_min)
        else:
            f_norm = f
        heuristic_matrix += w * f_norm
    
    # Boost edges that appear in both MST and high-frequency tours
    overlap_mask = mst_mask & (edge_frequency > np.percentile(edge_frequency, 70))
    heuristic_matrix[overlap_mask] *= 1.4
    
    # Penalize very long edges (unless they're in MST)
    long_edge_mask = (distance_matrix > np.percentile(distance_matrix, 85)) & ~mst_mask
    heuristic_matrix[long_edge_mask] *= 0.7
    
    # ===== 7. INTELLIGENT SPARSIFICATION =====
    
    # Determine edges to keep per node
    min_edges = max(2, int(1.5 * log_n))
    max_edges = min(n-1, int(3.5 * log_n))
    
    sparsity_mask = np.zeros_like(heuristic_matrix, dtype=bool)
    
    for i in range(n):
        scores = heuristic_matrix[i].copy()
        scores[i] = -np.inf
        
        # Adaptive number of edges based on node centrality
        centrality_factor = 0.5 + 0.5 * centrality[i]
        n_edges = int(min_edges + (max_edges - min_edges) * centrality_factor)
        n_edges = min(max_edges, max(min_edges, n_edges))
        
        # Select top edges
        top_indices = np.argsort(scores)[-n_edges:]
        sparsity_mask[i, top_indices] = True
    
    # Ensure MST edges and nearest neighbors are included
    sparsity_mask = sparsity_mask | mst_mask | (nn_feature > 0)
    
    # Ensure symmetry and minimum degree
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Guarantee each node has at least min_edges connections
    for i in range(n):
        connections = np.sum(sparsity_mask[i])
        if connections < min_edges:
            # Add best available edges
            candidates = ~sparsity_mask[i]
            candidates[i] = False
            if np.any(candidates):
                candidate_scores = heuristic_matrix[i, candidates]
                best_idx = np.where(candidates)[0][np.argsort(candidate_scores)[-min(3, sum(candidates)):]]
                for idx in best_idx:
                    sparsity_mask[i, idx] = True
                    sparsity_mask[idx, i] = True
    
    # Apply sparsification
    heuristic_matrix = heuristic_matrix * sparsity_mask
    
    # ===== 8. FINAL REFINEMENT =====
    
    # Ensure symmetry
    heuristic_matrix = 0.5 * (heuristic_matrix + heuristic_matrix.T)
    
    # Zero diagonal
    np.fill_diagonal(heuristic_matrix, 0)
    
    # Soft normalization with outlier clipping
    non_zero = heuristic_matrix[heuristic_matrix > 0]
    if len(non_zero) > 0:
        q25, q75 = np.percentile(non_zero, [25, 75])
        iqr = q75 - q25
        lower_bound = q25 - 1.5 * iqr
        upper_bound = q75 + 1.5 * iqr
        heuristic_matrix = np.clip(heuristic_matrix, max(0, lower_bound), upper_bound)
        
        # Normalize to [0, 1]
        h_min, h_max = heuristic_matrix.min(), heuristic_matrix.max()
        if h_max - h_min > 1e-10:
            heuristic_matrix = (heuristic_matrix - h_min) / (h_max - h_min)
    
    return heuristic_matrix
```
