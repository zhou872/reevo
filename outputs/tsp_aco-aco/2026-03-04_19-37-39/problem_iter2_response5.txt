import numpy as np
from scipy.spatial import distance_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors.
    
    Key improvements over v1:
    1. Dynamic weight adaptation based on problem size
    2. Vectorized computation of triangle factor
    3. Efficient edge sparsification preserving graph connectivity
    4. Focus on TSP-relevant patterns
    
    Returns: Heuristic matrix of same shape as input, with positive values for promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Dynamic parameter adaptation based on problem size
    k_nn = max(2, min(8, n // 10))  # Nearest neighbors to consider
    density_target = min(0.3, max(0.1, 50 / n))  # Target edge density
    
    # 2. Base inverse distance factor (handling diagonal)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0)
    
    # 3. Nearest neighbor factor (vectorized implementation)
    nn_factor = np.zeros_like(distance_matrix)
    for i in range(n):
        # Get k_nn+1 nearest neighbors (including self)
        nearest = np.argpartition(distance_matrix[i], k_nn + 1)[:k_nn + 1]
        nearest = nearest[nearest != i]  # Remove self
        if len(nearest) > 0:
            nn_factor[i, nearest] = 1.0
    # Ensure symmetry
    nn_factor = np.maximum(nn_factor, nn_factor.T)
    
    # 4. Triangle completion factor (vectorized computation)
    triangle_factor = np.zeros_like(distance_matrix)
    if n >= 4:
        # Define "close" threshold as 30th percentile for each node
        percentiles = np.percentile(distance_matrix, 30, axis=1)
        # Create boolean matrix of close connections
        close_matrix = distance_matrix <= percentiles[:, None]
        np.fill_diagonal(close_matrix, False)  # Remove self-connections
        
        # Count common close neighbors for each pair (vectorized)
        # Matrix multiplication counts common neighbors
        common_close = close_matrix @ close_matrix.T
        
        # Normalize by maximum possible common neighbors
        max_possible = n - 2
        if max_possible > 0:
            triangle_factor = common_close / max_possible
    
    # 5. Dynamic weight adaptation based on problem size
    if n < 20:
        weights = {'base': 2.5, 'nn': 1.8, 'triangle': 0.7}
    elif n < 100:
        weights = {'base': 2.0, 'nn': 1.5, 'triangle': 1.0}
    else:
        weights = {'base': 1.5, 'nn': 2.0, 'triangle': 1.2}
    
    # 6. Combine factors with dynamic weights
    combined = (
        weights['base'] * base_heuristic +
        weights['nn'] * nn_factor +
        weights['triangle'] * triangle_factor
    )
    
    # 7. Adaptive sparsification preserving connectivity
    if n > 3:
        # Get all unique edge values (excluding diagonal)
        upper_values = combined[np.triu_indices(n, k=1)]
        
        if len(upper_values) > 0:
            # Calculate threshold to achieve target density
            # We want to keep approximately density_target * (n*(n-1)/2) edges
            target_count = int(density_target * n * (n - 1) / 2)
            target_count = max(k_nn * n // 2, target_count)  # Ensure minimum connectivity
            
            if len(upper_values) > target_count:
                # Find threshold that keeps target_count largest values
                threshold = np.partition(upper_values, -target_count)[-target_count]
                
                # Create mask for promising edges
                promising_mask = combined >= threshold
                
                # Ensure each node has at least k_nn connections
                for i in range(n):
                    row = combined[i]
                    current_connections = np.sum(promising_mask[i])
                    if current_connections < k_nn:
                        # Add top k_nn edges for this node
                        top_indices = np.argsort(row)[-k_nn:]
                        promising_mask[i, top_indices] = True
                        promising_mask[top_indices, i] = True  # Maintain symmetry
                
                # Apply sparsification
                result = np.zeros_like(combined)
                result[promising_mask] = combined[promising_mask]
                
                # Final symmetry enforcement
                result = np.maximum(result, result.T)
                return result
    
    return combined
