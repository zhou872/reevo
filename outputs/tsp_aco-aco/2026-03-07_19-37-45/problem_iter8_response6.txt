```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform
import heapq


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics combining multi-scale regret analysis,
    structural centrality measures, and reinforced local search feedback.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges.
    """
    
    def analyze_instance(dist_mat: np.ndarray) -> dict:
        """Comprehensive instance analysis for parameter adaptation."""
        n = dist_mat.shape[0]
        if n == 0:
            return {'n': 0}
        
        # Basic statistics
        triu_idx = np.triu_indices(n, k=1)
        distances = dist_mat[triu_idx]
        
        stats = {
            'n': n,
            'mean_dist': np.mean(distances),
            'std_dist': np.std(distances),
            'cv_dist': np.std(distances) / (np.mean(distances) + 1e-10),
            'skewness': 0.0
        }
        
        if len(distances) > 2:
            from scipy.stats import skew
            stats['skewness'] = skew(distances)
        
        # Topology analysis
        if n >= 4:
            # Local vs global connectivity
            k_local = min(3, n-1)
            local_dists = []
            for i in range(n):
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                local_dists.append(np.mean(np.partition(dists, k_local)[:k_local]))
            
            stats['local_connectivity'] = np.mean(local_dists)
            stats['global_connectivity'] = stats['mean_dist']
            stats['structure_index'] = stats['local_connectivity'] / (stats['global_connectivity'] + 1e-10)
            
            # Cluster tendency via silhouette approximation
            if n >= 8:
                sample_size = min(10, n // 2)
                sample = np.random.choice(n, sample_size, replace=False)
                intra_inter_ratios = []
                
                for i in sample:
                    # Find nearest neighbor
                    dists = dist_mat[i].copy()
                    dists[i] = np.inf
                    nearest = np.argmin(dists)
                    
                    # Average distance to nearest's neighbors
                    intra_dists = dist_mat[nearest].copy()
                    intra_dists[nearest] = np.inf
                    intra = np.mean(np.partition(intra_dists, k_local)[:k_local])
                    
                    # Average distance to other nodes
                    others = [j for j in range(n) if j not in [i, nearest]]
                    if others:
                        inter = np.mean(dist_mat[i, others])
                        if inter > 0:
                            intra_inter_ratios.append(intra / inter)
                
                if intra_inter_ratios:
                    stats['cluster_tendency'] = np.mean(intra_inter_ratios)
                else:
                    stats['cluster_tendency'] = 0.5
            else:
                stats['cluster_tendency'] = 0.5
        else:
            stats['structure_index'] = 1.0
            stats['cluster_tendency'] = 0.5
        
        return stats
    
    def compute_centrality_scores(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Compute node centrality scores using multiple metrics."""
        n = stats['n']
        if n < 3:
            return np.ones((n, n))
        
        # Closeness centrality: inverse of average distance to all other nodes
        closeness = np.zeros(n)
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            avg_dist = np.mean(dists[dists < np.inf])
            closeness[i] = 1 / (avg_dist + 1e-10)
        
        # Betweenness approximation via random paths
        betweenness = np.zeros(n)
        if n >= 8:
            n_samples = min(50, n * 3)
            for _ in range(n_samples):
                s, t = np.random.choice(n, 2, replace=False)
                
                # Simple 2-step path search
                min_path = dist_mat[s, t]
                intermediate = -1
                
                for k in range(n):
                    if k not in [s, t]:
                        path_len = dist_mat[s, k] + dist_mat[k, t]
                        if path_len < min_path:
                            min_path = path_len
                            intermediate = k
                
                if intermediate >= 0:
                    betweenness[intermediate] += 1
        
        # Degree centrality approximation via MST
        mst_degree = np.zeros(n)
        if n > 2:
            try:
                mst = minimum_spanning_tree(dist_mat).toarray()
                mst_degree = np.sum(mst > 0, axis=1) + np.sum(mst > 0, axis=0)
            except:
                pass
        
        # Combine centrality measures
        centrality = np.zeros(n)
        if np.max(closeness) > 0:
            centrality += closeness / np.max(closeness)
        if np.max(betweenness) > 0:
            centrality += betweenness / np.max(betweenness)
        if np.max(mst_degree) > 0:
            centrality += mst_degree / np.max(mst_degree)
        
        # Normalize and create edge scores
        if np.max(centrality) > 0:
            centrality = centrality / np.max(centrality)
        
        centrality_matrix = np.ones((n, n))
        for i in range(n):
            for j in range(i+1, n):
                # Edge between two central nodes gets higher score
                score = (centrality[i] * centrality[j]) ** 0.5
                centrality_matrix[i, j] = centrality_matrix[j, i] = 1.0 + score
        
        np.fill_diagonal(centrality_matrix, 0)
        return centrality_matrix
    
    def compute_multi_scale_regret(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Compute regret scores at multiple scales (1, 2, and 3 hops)."""
        n = stats['n']
        if n < 4:
            return np.ones((n, n))
        
        regret_scores = np.ones((n, n))
        
        # Precompute sorted neighbors for efficiency
        sorted_neighbors = []
        neighbor_dists = []
        
        k_neighbors = min(10, n-1)
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            indices = np.argsort(dists)[:k_neighbors]
            sorted_neighbors.append(indices)
            neighbor_dists.append(dists[indices])
        
        # Compute regrets for each edge
        for i in range(n):
            for j in range(i+1, n):
                d_ij = dist_mat[i, j]
                
                # 1-hop regret: best alternative from i and j
                alt_i = float('inf')
                for k in sorted_neighbors[i]:
                    if k != j:
                        alt_i = min(alt_i, dist_mat[i, k])
                
                alt_j = float('inf')
                for k in sorted_neighbors[j]:
                    if k != i:
                        alt_j = min(alt_j, dist_mat[j, k])
                
                regret_1hop = (max(0, d_ij - alt_i) + max(0, d_ij - alt_j)) / 2
                
                # 2-hop regret: best i-k-j alternative
                regret_2hop = float('inf')
                for ki in sorted_neighbors[i][:5]:
                    if ki == j:
                        continue
                    for kj in sorted_neighbors[j][:5]:
                        if kj == i or kj == ki:
                            continue
                        path = dist_mat[i, ki] + dist_mat[ki, kj] + dist_mat[kj, j]
                        regret_2hop = min(regret_2hop, max(0, d_ij - path/2))
                
                # 3-hop regret (simplified)
                regret_3hop = float('inf')
                if n >= 8:
                    # Sample a few 3-hop paths
                    samples = min(5, n-2)
                    candidates = np.random.choice([k for k in range(n) if k not in [i, j]], 
                                                  samples, replace=False)
                    for k in candidates:
                        # Find l that connects well
                        dists_k = dist_mat[k].copy()
                        dists_k[k] = np.inf
                        l_candidates = np.argsort(dists_k)[:3]
                        
                        for l in l_candidates:
                            if l in [i, j, k]:
                                continue
                            path = dist_mat[i, k] + dist_mat[k, l] + dist_mat[l, j]
                            regret_3hop = min(regret_3hop, max(0, d_ij - path/3))
                
                # Combine regrets with adaptive weights
                weight_1 = 0.4
                weight_2 = 0.4
                weight_3 = 0.2 if regret_3hop < float('inf') else 0.0
                
                total_regret = 0
                if regret_1hop < float('inf'):
                    total_regret += weight_1 * regret_1hop
                if regret_2hop < float('inf'):
                    total_regret += weight_2 * regret_2hop
                if regret_3hop < float('inf'):
                    total_regret += weight_3 * regret_3hop
                
                # Convert to score (exponential decay)
                score = np.exp(-total_regret / (stats['mean_dist'] + 1e-10))
                regret_scores[i, j] = regret_scores[j, i] = score
        
        np.fill_diagonal(regret_scores, 0)
        return regret_scores
    
    def reinforced_local_search(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Use local search with reinforcement learning principles."""
        n = stats['n']
        if n < 5:
            return np.ones((n, n))
        
        # Initialize Q-values for edges
        Q = np.ones((n, n))
        np.fill_diagonal(Q, 0)
        
        # Learning parameters
        alpha = 0.3  # learning rate
        gamma = 0.7  # discount factor
        episodes = min(20, n * 2)
        
        for episode in range(episodes):
            # Start with random tour
            tour = np.random.permutation(n)
            tour = list(tour) + [tour[0]]  # make circular
            
            # Evaluate and update
            for idx in range(n):
                i, j = tour[idx], tour[idx + 1]
                
                # Find best alternative edge from i (excluding current)
                alternatives = []
                for k in range(n):
                    if k != i and k != j and k != tour[idx-1]:
                        alternatives.append((k, dist_mat[i, k]))
                
                if alternatives:
                    # Sort by distance
                    alternatives.sort(key=lambda x: x[1])
                    best_alt = alternatives[0][0]
                    
                    # Compute reward: negative of improvement
                    current_cost = dist_mat[i, j]
                    alt_cost = dist_mat[i, best_alt]
                    
                    # Simple reward: relative improvement
                    reward = (current_cost - alt_cost) / (current_cost + 1e-10)
                    reward = max(-1, min(1, reward * 2))
                    
                    # Update Q-value using simplified Q-learning
                    future_value = np.max([Q[best_alt, l] for l in range(n) if l != best_alt])
                    Q[i, j] = (1 - alpha) * Q[i, j] + alpha * (reward + gamma * future_value)
                    Q[j, i] = Q[i, j]  # symmetric
        
        # Normalize Q-values
        max_q = np.max(Q)
        if max_q > 0:
            Q = Q / max_q
        
        return Q
    
    def compute_inverse_distance(dist_mat: np.ndarray) -> np.ndarray:
        """Normalized inverse distance scores."""
        n = dist_mat.shape[0]
        inv_dist = 1 / (dist_mat + np.eye(n) * 1e-8)
        np.fill_diagonal(inv_dist, 0)
        
        if np.max(inv_dist) > 0:
            inv_dist = inv_dist / np.max(inv_dist)
        
        return inv_dist
    
    def compute_mst_scores(dist_mat: np.ndarray) -> np.ndarray:
        """Boost edges in Minimum Spanning Tree."""
        n = dist_mat.shape[0]
        mst_scores = np.ones((n, n))
        
        if n > 2:
            try:
                mst = minimum_spanning_tree(dist_mat).toarray()
                mst_edges = (mst + mst.T) > 0
                # Boost based on problem size
                boost = 1.5 + 0.3 * np.log(n + 1)
                mst_scores[mst_edges] = boost
            except Exception:
                pass
        
        np.fill_diagonal(mst_scores, 0)
        return mst_scores
    
    def compute_adaptive_weights(stats: dict) -> dict:
        """Compute adaptive weights based on instance characteristics."""
        n = stats['n']
        
        # Base weights
        weights = {
            'w_inv': 1.0,
            'w_regret': 0.8,
            'w_centrality': 0.6,
            'w_reinforced': 0.5,
            'w_mst': 0.4
        }
        
        if n > 0:
            # Adjust based on instance properties
            cv = stats['cv_dist']
            structure = stats.get('structure_index', 1.0)
            cluster = stats.get('cluster_tendency', 0.5)
            
            # More weight to regret for structured instances
            weights['w_regret'] *= (0.7 + 0.3 * structure)
            
            # More weight to centrality for clustered instances
            weights['w_centrality'] *= (0.6 + 0.4 * cluster)
            
            # More weight to inverse distance for uniform instances
            weights['w_inv'] *= (1.2 - 0.4 * cv)
            
            # Adjust for problem size
            size_factor = min(1.5, 0.8 + 0.02 * n)
            for key in weights:
                weights[key] *= size_factor
        
        # Normalize
        total = sum(weights.values())
        if total > 0:
            for key in weights:
                weights[key] = weights[key] / total * 5
        
        return weights
    
    def combine_factors(factors: dict, weights: dict, candidate_mask: np.ndarray) -> np.ndarray:
        """Combine multiple factors using weighted geometric mean."""
        n = factors['inv_dist'].shape[0]
        H = np.zeros((n, n))
        
        for i in range(n):
            for j in range(i+1, n):
                if candidate_mask[i, j]:
                    product = 1.0
                    weight_sum = 0
                    
                    for factor_name, weight in weights.items():
                        if factor_name in factors and weight > 0:
                            factor_val = factors[factor_name][i, j]
                            if factor_val > 0:
                                product *= factor_val ** weight
                                weight_sum += weight
                    
                    if weight_sum > 0:
                        H[i, j] = H[j, i] = product ** (1 / weight_sum)
        
        return H
    
    def build_candidate_mask(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Build mask of candidate edges to consider."""
        n = stats['n']
        if n <= 2:
            return ~np.eye(n, dtype=bool)
        
        # Adaptive number of nearest neighbors
        base_k = min(15, max(4, int(4 * np.log2(n + 1))))
        if stats.get('structure_index', 1.0) < 0.8:
            base_k = int(base_k * 1.3)
        
        candidate_mask = np.zeros((n, n), dtype=bool)
        
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            
            k = min(base_k, n-1)
            if k < len(dists):
                indices = np.argpartition(dists, k)[:k]
            else:
                indices = np.arange(n)
                indices = indices[indices != i]
            
            candidate_mask[i, indices] = True
        
        # Ensure symmetry and connectivity
        candidate_mask = candidate_mask | candidate_mask.T
        
        # Ensure minimum connectivity
        for i in range(n):
            if np.sum(candidate_mask[i]) < 3:
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                nearest = np.argsort(dists)[:3]
                candidate_mask[i, nearest] = True
                candidate_mask[nearest, i] = True
        
        np.fill_diagonal(candidate_mask, False)
        return candidate_mask
    
    def post_process(H: np.ndarray, dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Final post-processing of heuristic matrix."""
        n = stats['n']
        
        # Ensure symmetry
        H = (H + H.T) / 2
        
        # Apply soft thresholding
        if n > 10:
            non_zero = H[H > 0]
            if len(non_zero) > 0:
                threshold = np.percentile(non_zero, 20)
                H[H < threshold] = 0
        
        # Normalize
        max_val = np.max(H)
        if max_val > 0:
            H = H / max_val
        
        np.fill_diagonal(H, 0)
        return H
    
    # ------------------- Main Execution -------------------
    
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    # Step 1: Analyze instance
    stats = analyze_instance(distance_matrix)
    
    # Step 2: Compute individual factors
    inv_dist = compute_inverse_distance(distance_matrix)
    centrality = compute_centrality_scores(distance_matrix, stats)
    regret = compute_multi_scale_regret(distance_matrix, stats)
    reinforced = reinforced_local_search(distance_matrix, stats)
    mst_scores = compute_mst_scores(distance_matrix)
    
    # Step 3: Build candidate mask
    candidate_mask = build_candidate_mask(distance_matrix, stats)
    
    # Step 4: Compute adaptive weights
    weights = compute_adaptive_weights(stats)
    
    # Step 5: Combine factors
    factors = {
        'inv_dist': inv_dist,
        'centrality': centrality,
        'regret': regret,
        'reinforced': reinforced,
        'mst': mst_scores
    }
    
    H = combine_factors(factors, weights, candidate_mask)
    
    # Step 6: Post-processing
    H = post_process(H, distance_matrix, stats)
    
    return H
```
