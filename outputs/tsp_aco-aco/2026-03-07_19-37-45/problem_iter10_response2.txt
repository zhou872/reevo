import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced deterministic heuristic for TSP edge selection.
    
    Key improvements over v1:
    1. Extended regret analysis using 2-3 hop alternative paths
    2. Integration of graph centrality measures (betweenness and closeness)
    3. Enhanced instance analysis for adaptive parameter tuning
    4. More sophisticated candidate selection with sparsity awareness
    
    Returns a matrix of same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Enhanced instance analysis
    stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Compute heuristic factors
    factors = _compute_factors_v2(distance_matrix, stats)
    
    # Step 3: Compute adaptive weights with centrality awareness
    weights = _compute_weights_v2(stats, factors)
    
    # Step 4: Multi-factor combination with adaptive exponents
    H = _combine_factors_v2(factors, weights, distance_matrix, stats)
    
    # Step 5: Intelligent sparsification and normalization
    H = _post_process_v2(H, distance_matrix, stats)
    
    return H


def _analyze_instance_v2(distance_matrix: np.ndarray) -> dict:
    """Comprehensive instance analysis with advanced metrics."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal for statistics
    mask = ~np.eye(n, dtype=bool)
    distances = distance_matrix[mask]
    
    # Basic statistics
    mean_dist = np.mean(distances)
    std_dist = np.std(distances)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Percentile-based metrics
    q10, q25, q50, q75, q90 = np.percentile(distances, [10, 25, 50, 75, 90])
    
    # Structural metrics
    stats = {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'q10': q10, 'q25': q25, 'q50': q50, 'q75': q75, 'q90': q90,
        'sparsity': _compute_sparsity_v2(distance_matrix),
        'cluster_tendency': _compute_cluster_tendency_v2(distance_matrix),
        'regularity': _compute_regularity(distance_matrix),
        'topology': _analyze_topology(distance_matrix)
    }
    
    return stats


def _compute_sparsity_v2(distance_matrix: np.ndarray) -> float:
    """Enhanced sparsity estimation using multiple metrics."""
    n = distance_matrix.shape[0]
    if n < 4:
        return 0.5
    
    distances = distance_matrix[~np.eye(n, dtype=bool)]
    
    # Method 1: IQR-based
    q25, q75 = np.percentile(distances, [25, 75])
    iqr = q75 - q25
    if iqr == 0:
        sparsity1 = 0.5
    else:
        norm_iqr = iqr / (np.max(distances) - np.min(distances) + 1e-8)
        sparsity1 = 1.0 - norm_iqr
    
    # Method 2: Gap analysis
    sorted_dists = np.sort(distances)
    gaps = sorted_dists[1:] - sorted_dists[:-1]
    avg_gap = np.mean(gaps)
    std_gap = np.std(gaps)
    if avg_gap > 0:
        sparsity2 = std_gap / avg_gap
    else:
        sparsity2 = 0.5
    
    # Combined sparsity score
    sparsity = 0.7 * sparsity1 + 0.3 * sparsity2
    return np.clip(sparsity, 0.1, 0.9)


def _compute_cluster_tendency_v2(distance_matrix: np.ndarray) -> float:
    """Enhanced cluster tendency using multiple methods."""
    n = distance_matrix.shape[0]
    if n < 5:
        return 0.5
    
    # Method 1: Variance ratio of nearest neighbor distances
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    
    overall_var = np.var(distance_matrix[~np.eye(n, dtype=bool)])
    if overall_var > 0:
        tendency1 = np.var(nn_dists) / overall_var
    else:
        tendency1 = 0.5
    
    # Method 2: Silhouette-like metric
    k = min(3, n // 4)
    intra_dists = []
    inter_dists = []
    
    for i in range(n):
        # Intra-cluster: distance to k nearest
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        k_nearest = np.partition(dists_i, k)[:k]
        intra_dists.extend(k_nearest)
        
        # Inter-cluster: distance to random non-neighbors
        if k < n - 1:
            non_neighbors = np.where(dists_i > np.max(k_nearest))[0]
            if len(non_neighbors) > 0:
                inter_dists.append(np.mean(distance_matrix[i, non_neighbors]))
    
    if intra_dists and inter_dists:
        avg_intra = np.mean(intra_dists)
        avg_inter = np.mean(inter_dists)
        if avg_inter > avg_intra:
            tendency2 = (avg_inter - avg_intra) / max(avg_inter, avg_intra)
        else:
            tendency2 = 0.0
    else:
        tendency2 = 0.5
    
    # Combined score
    tendency = 0.6 * tendency1 + 0.4 * tendency2
    return np.clip(tendency, 0.1, 0.9)


def _compute_regularity(distance_matrix: np.ndarray) -> float:
    """Measure how regularly spaced the points are."""
    n = distance_matrix.shape[0]
    if n < 4:
        return 0.5
    
    # Compute MST and analyze edge length distribution
    try:
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = mst[mst > 0]
        
        if len(mst_edges) > 2:
            cv_mst = np.std(mst_edges) / (np.mean(mst_edges) + 1e-8)
            regularity = 1.0 / (1.0 + cv_mst)
        else:
            regularity = 0.5
    except:
        regularity = 0.5
    
    return np.clip(regularity, 0.1, 0.9)


def _analyze_topology(distance_matrix: np.ndarray) -> dict:
    """Analyze graph topology characteristics."""
    n = distance_matrix.shape[0]
    
    # Build k-nearest neighbor graph (k = sqrt(n))
    k = int(np.sqrt(n)) + 1
    k = min(k, n - 1)
    
    nn_graph = np.zeros((n, n), dtype=bool)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest = np.argpartition(dists, k)[:k]
        nn_graph[i, nearest] = True
    
    # Make symmetric
    nn_graph = nn_graph | nn_graph.T
    
    # Compute degree distribution
    degrees = np.sum(nn_graph, axis=1)
    avg_degree = np.mean(degrees)
    cv_degree = np.std(degrees) / (avg_degree + 1e-8)
    
    return {
        'avg_degree': avg_degree,
        'cv_degree': cv_degree,
        'max_degree': np.max(degrees),
        'min_degree': np.min(degrees)
    }


def _compute_factors_v2(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute all heuristic factors including centrality measures."""
    n = stats['n']
    
    factors = {}
    
    # 1. Inverse distance (normalized)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    max_inv = np.max(inv_dist)
    if max_inv > 0:
        inv_dist = inv_dist / max_inv
    factors['inv_dist'] = inv_dist
    
    # 2. Multi-hop regret analysis
    factors['regret'] = _compute_multi_hop_regret(distance_matrix, stats)
    
    # 3. Density-based scores
    factors['density'] = _compute_density_scores_v2(distance_matrix, stats)
    
    # 4. MST-based scores
    if n > 2:
        factors['mst'] = _compute_mst_scores_v2(distance_matrix)
    
    # 5. Centrality measures
    factors['centrality'] = _compute_centrality_scores(distance_matrix, stats)
    
    return factors


def _compute_multi_hop_regret(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """
    Compute regret scores using 2-hop and 3-hop alternative paths.
    Higher scores for edges with poor multi-hop alternatives.
    """
    n = stats['n']
    cv = stats['cv_dist']
    regret = np.ones((n, n))
    
    # Adaptive parameters based on instance characteristics
    # For regular instances, consider more hops
    max_hops = 2 if stats['regularity'] < 0.7 else 3
    max_hops = min(max_hops, int(np.log2(n)))
    
    for i in range(n):
        for j in range(i+1, n):
            d_ij = distance_matrix[i, j]
            
            # Find best alternative paths (2-hop and 3-hop)
            best_alternative = np.inf
            
            # 2-hop alternatives
            for k in range(n):
                if k != i and k != j:
                    alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                    best_alternative = min(best_alternative, alt_cost)
            
            # 3-hop alternatives (if enabled)
            if max_hops >= 3 and n > 10:
                # Sample some 3-hop paths to avoid O(n^3) complexity
                sample_size = min(50, n - 2)
                candidates = np.random.choice([x for x in range(n) if x not in [i, j]], 
                                            size=min(sample_size, n-2), replace=False)
                
                for k in candidates:
                    for l in candidates:
                        if l != k and l != i and l != j:
                            alt_cost = distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j]
                            best_alternative = min(best_alternative, alt_cost)
            
            # Compute regret score
            if d_ij > 0:
                regret_ratio = best_alternative / d_ij
                
                # Adaptive penalty: stronger for uniform instances
                penalty_factor = 1.0 + (1.0 - cv) * 0.8
                
                # Boost score for high regret (poor alternatives)
                regret_score = regret_ratio ** penalty_factor
                
                # Additional boost for very poor alternatives
                if regret_ratio > 1.5:
                    regret_score *= 1.2
                
                regret[i, j] = regret[j, i] = regret_score
    
    # Normalize
    max_regret = np.max(regret)
    if max_regret > 0:
        regret = regret / max_regret
    
    np.fill_diagonal(regret, 0)
    return regret


def _compute_density_scores_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Enhanced density scores with local structure analysis."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Compute local density using adaptive k
    k = min(max(3, n // 15), 10)
    local_density = np.zeros(n)
    local_std = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k)[:k]
        local_density[i] = np.mean(k_nearest)
        local_std[i] = np.std(k_nearest)
    
    # Normalize densities
    if np.std(local_density) > 1e-8:
        density_norm = (local_density - np.min(local_density)) / \
                      (np.max(local_density) - np.min(local_density) + 1e-8)
    else:
        density_norm = np.ones(n) * 0.5
    
    # Normalize local variance
    if np.max(local_std) > 0:
        std_norm = local_std / np.max(local_std)
    else:
        std_norm = np.zeros(n)
    
    # Compute scores with local structure awareness
    for i in range(n):
        for j in range(i+1, n):
            # Base density score (favor sparse connections)
            density_score = 1.0 - (density_norm[i] + density_norm[j]) / 2
            
            # Adjust based on local variance
            # High variance in both ends suggests boundary edge - boost
            if std_norm[i] > 0.7 and std_norm[j] > 0.7:
                density_score *= 1.3
            # Both low variance suggests interior of cluster - penalize
            elif std_norm[i] < 0.3 and std_norm[j] < 0.3:
                density_score *= 0.8
            
            density_scores[i, j] = density_scores[j, i] = max(0.1, density_score)
    
    return density_scores


def _compute_mst_scores_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Enhanced MST scores with edge importance analysis."""
    n = distance_matrix.shape[0]
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            # Compute MST
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Base boost for MST edges
            mst_scores[mst_present] = 1.8
            
            # Additional analysis: identify critical edges
            # Compute node degrees in MST
            mst_degree = np.sum(mst_present, axis=1)
            
            # Boost connections to leaves (degree 1)
            leaves = np.where(mst_degree == 1)[0]
            for leaf in leaves:
                # Find MST neighbor
                neighbors = np.where(mst_present[leaf])[0]
                if len(neighbors) > 0:
                    mst_scores[leaf, neighbors[0]] = 2.2
            
            # Penalize connections between high-degree nodes (redundant)
            hubs = np.where(mst_degree >= 3)[0]
            for i in hubs:
                for j in hubs:
                    if i != j and mst_present[i, j]:
                        mst_scores[i, j] = 1.4
                        
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores


def _compute_centrality_scores(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute edge centrality scores using multiple measures."""
    n = stats['n']
    centrality = np.ones((n, n))
    
    if n < 4:
        np.fill_diagonal(centrality, 0)
        return centrality
    
    # Measure 1: Betweenness approximation using k-shortest paths
    k = min(20, n * (n - 1) // 10)
    
    # Build candidate graph for centrality computation
    k_nn = min(max(3, int(np.log2(n))), n - 1)
    candidate_graph = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest = np.argpartition(dists, k_nn)[:k_nn]
        candidate_graph[i, nearest] = True
    
    # Make symmetric
    candidate_graph = candidate_graph | candidate_graph.T
    
    # Approximate edge betweenness by counting shortest paths through edges
    edge_count = np.zeros((n, n))
    
    # Sample node pairs for betweenness estimation
    if n > 50:
        sample_pairs = min(200, n * 2)
        sources = np.random.choice(n, min(sample_pairs, n), replace=False)
    else:
        sources = range(n)
    
    for src in sources:
        # Dijkstra-like distance computation
        dist = np.full(n, np.inf)
        prev = [None] * n
        dist[src] = 0
        
        # Simple BFS on candidate graph
        visited = np.zeros(n, dtype=bool)
        visited[src] = True
        queue = [src]
        
        while queue:
            current = queue.pop(0)
            
            # Consider neighbors in candidate graph
            neighbors = np.where(candidate_graph[current])[0]
            
            for neighbor in neighbors:
                new_dist = dist[current] + distance_matrix[current, neighbor]
                if new_dist < dist[neighbor]:
                    dist[neighbor] = new_dist
                    prev[neighbor] = current
                
                if not visited[neighbor]:
                    visited[neighbor] = True
                    queue.append(neighbor)
        
        # Count paths through edges
        for target in range(n):
            if target != src and prev[target] is not None:
                # Trace back path
                current = target
                while prev[current] is not None:
                    edge_count[prev[current], current] += 1
                    edge_count[current, prev[current]] += 1
                    current = prev[current]
    
    # Normalize edge counts to get centrality scores
    max_count = np.max(edge_count)
    if max_count > 0:
        centrality = 0.5 + 0.5 * (edge_count / max_count)
    else:
        centrality = np.ones((n, n)) * 0.5
    
    # Adjust based on instance characteristics
    # In clustered instances, penalize inter-cluster edges less
    if stats['cluster_tendency'] > 0.7:
        centrality = np.power(centrality, 0.8)
    
    np.fill_diagonal(centrality, 0)
    return centrality


def _compute_weights_v2(stats: dict, factors: dict) -> dict:
    """Compute adaptive weights with instance-aware optimization."""
    n = stats['n']
    cv = stats['cv_dist']
    sparsity = stats['sparsity']
    cluster_tendency = stats['cluster_tendency']
    regularity = stats['regularity']
    
    # Base weights that adapt to instance structure
    weights = {
        'inv_dist': 1.0,  # Always important
        
        # Regret: more important for uniform/regular instances
        'regret': 0.7 + 0.6 * regularity,
        
        # Density: more important for clustered instances
        'density': 0.4 + 0.8 * cluster_tendency,
        
        # MST: more important for sparse/structured instances
        'mst': 0.5 * sparsity + 0.3 * (1.0 - cv),
        
        # Centrality: more important for complex topologies
        'centrality': 0.3 + 0.5 * (1.0 - regularity)
    }
    
    # Adjust weights based on instance size
    size_factor = min(1.0, np.log10(n) / 2)
    weights['centrality'] *= size_factor
    weights['regret'] *= (1.0 + 0.3 * size_factor)
    
    # Normalize weights to maintain balance
    total = sum(weights.values())
    if total > 0:
        # Target sum of 4.0 for stability
        scale = 4.0 / total
        for key in weights:
            weights[key] *= scale
    
    return weights


def _combine_factors_v2(factors: dict, weights: dict, 
                       distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Combine heuristic factors with adaptive geometric mean."""
    n = stats['n']
    H = np.zeros((n, n))
    
    # Build intelligent candidate mask
    candidate_mask = _build_candidate_mask_v2(distance_matrix, stats)
    
    # Combine factors for candidate edges
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                combined = 1.0
                for factor_name in ['inv_dist', 'regret', 'density', 'mst', 'centrality']:
                    if factor_name in factors:
                        factor_value = factors[factor_name][i, j]
                        weight = weights.get(factor_name, 1.0)
                        combined *= (factor_value ** weight)
                
                H[i, j] = H[j, i] = combined
    
    return H


def _build_candidate_mask_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Build intelligent candidate mask with sparsity awareness."""
    n = stats['n']
    
    # Adaptive candidate selection based on instance characteristics
    base_k = min(15, max(5, int(np.log2(n) * 3)))
    
    # More candidates for sparse instances, fewer for dense
    k = int(base_k * (0.8 + 0.4 * stats['sparsity']))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # Use distance percentiles for adaptive thresholding
    q25 = stats['q25']
    q75 = stats['q75']
    
    for i in range(n):
        # Get distances
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Always include nearest neighbors
        nearest = np.argpartition(dists, min(k, n-1))[:min(k, n-1)]
        candidate_mask[i, nearest] = True
        
        # Include edges that are exceptionally good (below 25th percentile)
        good_edges = dists < q25
        candidate_mask[i] = candidate_mask[i] | good_edges
    
    # Ensure symmetry
    candidate_mask = candidate_mask | candidate_mask.T
    
    # Remove very long edges (above 90th percentile)
    long_edge_threshold = stats['q90'] * 1.2
    candidate_mask = candidate_mask & (distance_matrix <= long_edge_threshold)
    
    # Ensure minimum connectivity
    min_edges = min(4, n - 1)
    for i in range(n):
        if np.sum(candidate_mask[i]) < min_edges:
            # Add more nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            additional = np.argsort(dists)[:min_edges]
            candidate_mask[i, additional] = True
    
    np.fill_diagonal(candidate_mask, False)
    return candidate_mask


def _post_process_v2(H: np.ndarray, distance_matrix: np.ndarray, 
                    stats: dict) -> np.ndarray:
    """Final post-processing with intelligent sparsification."""
    n = stats['n']
    
    # Zero diagonal
    np.fill_diagonal(H, 0)
    
    # Ensure connectivity: each node must have at least 2 connections
    min_connections = min(3, n - 1)
    for i in range(n):
        non_zero = H[i] > 0
        if np.sum(non_zero) < min_connections:
            # Add nearest neighbors not already included
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Sort by distance
            sorted_indices = np.argsort(dists)
            
            added = 0
            for j in sorted_indices:
                if added >= min_connections:
                    break
                if H[i, j] == 0:
                    # Assign reasonable score based on inverse distance
                    base_score = 1 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score * 0.5  # Conservative boost
                    added += 1
    
    # Adaptive sparsification
    keep_per_node = min(8, max(3, n // 4))
    
    # Use soft thresholding with instance-aware parameters
    if n > 15:
        # More aggressive sparsification for larger instances
        percentile = 30 if n > 100 else 25
        global_threshold = np.percentile(H[H > 0], percentile)
        H[H < global_threshold] = 0
    
    # Ensure top connections per node are kept
    H_sparse = np.zeros_like(H)
    for i in range(n):
        row = H[i].copy()
        row[i] = -np.inf
        
        # Keep top connections
        if np.sum(row > 0) > keep_per_node:
            threshold = np.partition(row, -keep_per_node)[-keep_per_node]
            row[row < threshold] = 0
        
        H_sparse[i] = row
    
    # Symmetrize: keep edge if selected in either direction
    H_final = np.maximum(H_sparse, H_sparse.T)
    
    # Final normalization
    max_val = np.max(H_final)
    if max_val > 0:
        H_final = H_final / max_val
    
    return H_final
