import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics combining two-hop regret, partial tour lookahead,
    and adaptive instance analysis with geometric mean combination.
    
    Returns a matrix of same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Analyze instance characteristics
    stats = _analyze_instance(distance_matrix)
    
    # Step 2: Compute base heuristic factors
    base_factors = _compute_base_factors(distance_matrix, stats)
    
    # Step 3: Compute lookahead scores via partial tour simulations
    lookahead_scores = _compute_lookahead_scores(distance_matrix, stats)
    
    # Step 4: Compute adaptive weights based on instance characteristics
    weights = _compute_adaptive_weights(distance_matrix, stats)
    
    # Step 5: Geometric mean combination
    H = np.zeros((n, n))
    candidate_mask = _build_candidate_mask(distance_matrix, stats)
    
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                combined = (
                    (base_factors['inv_dist'][i, j] ** weights['w_inv']) *
                    (base_factors['regret'][i, j] ** weights['w_regret']) *
                    (base_factors['density'][i, j] ** weights['w_density']) *
                    (lookahead_scores[i, j] ** weights['w_lookahead']) *
                    (base_factors['mst'][i, j] ** weights['w_mst'])
                )
                H[i, j] = H[j, i] = combined
    
    # Step 6: Post-processing and sparsification
    H = _post_process(H, distance_matrix, stats)
    
    return H


def _analyze_instance(distance_matrix: np.ndarray) -> dict:
    """Analyze instance characteristics for adaptive parameter tuning."""
    n = distance_matrix.shape[0]
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    # Basic statistics
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Regularity analysis
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    cv_nn = np.std(nn_dists) / (np.mean(nn_dists) + 1e-8)
    regularity = 1.0 / (1.0 + cv_nn)
    
    # Sparsity estimation
    q1 = np.percentile(dist_no_diag, 25)
    q3 = np.percentile(dist_no_diag, 75)
    qcd = (q3 - q1) / (q3 + q1 + 1e-8)
    sparsity = 1.0 - qcd
    
    # Cluster tendency (simplified silhouette)
    cluster_tendency = 0.5  # Default
    if n >= 10:
        try:
            from sklearn.cluster import KMeans
            kmeans = KMeans(n_clusters=min(3, n//4), n_init=1, max_iter=10, random_state=42)
            row_means = np.mean(distance_matrix, axis=1, keepdims=True)
            row_stds = np.std(distance_matrix, axis=1, keepdims=True) + 1e-8
            normalized = (distance_matrix - row_means) / row_stds
            labels = kmeans.fit_predict(normalized)
            
            intra_dist, inter_dist = 0, 0
            for i in range(n):
                same_cluster = labels == labels[i]
                diff_cluster = ~same_cluster
                same_cluster[i] = False
                
                if np.any(same_cluster):
                    intra_dist += np.mean(distance_matrix[i, same_cluster])
                if np.any(diff_cluster):
                    inter_dist += np.mean(distance_matrix[i, diff_cluster])
            
            silhouette = (inter_dist - intra_dist) / (max(intra_dist, inter_dist) + 1e-8)
            cluster_tendency = max(0, min(1, (silhouette + 1) / 2))
        except:
            pass
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'regularity': regularity,
        'sparsity': sparsity,
        'cluster_tendency': cluster_tendency,
        'percentile_25': q1,
        'percentile_75': q3
    }


def _compute_base_factors(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute base heuristic factors including two-hop regret."""
    n = stats['n']
    factors = {}
    
    # 1. Inverse distance
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    factors['inv_dist'] = inv_dist
    
    # 2. Two-hop regret with adaptive threshold
    regret = np.ones((n, n))
    cv = stats['cv_dist']
    penalty_factor = 1.0 + 0.5 * cv  # Stricter for uniform instances
    
    for i in range(n):
        for j in range(i+1, n):
            direct_cost = distance_matrix[i, j]
            
            # Find best two-hop alternative
            mask = np.ones(n, dtype=bool)
            mask[[i, j]] = False
            if np.any(mask):
                two_hop_costs = distance_matrix[i, mask] + distance_matrix[mask, j]
                best_two_hop = np.min(two_hop_costs)
                
                if best_two_hop < direct_cost * penalty_factor:
                    regret_score = best_two_hop / (direct_cost + 1e-8)
                else:
                    regret_score = direct_cost / (best_two_hop + 1e-8)
                
                # Sigmoid transformation
                regret[i, j] = regret[j, i] = 1 / (1 + np.exp(-3 * (regret_score - 0.5)))
    
    np.fill_diagonal(regret, 0)
    factors['regret'] = regret
    
    # 3. Density-based scores
    density_scores = np.ones((n, n))
    k = min(7, max(3, n // 8))
    
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k)[:k]
        local_density[i] = np.mean(k_nearest)
    
    # Normalize with robust scaling
    q1, q3 = np.percentile(local_density, [25, 75])
    iqr = q3 - q1 + 1e-8
    local_density = (local_density - np.median(local_density)) / iqr
    
    for i in range(n):
        for j in range(i+1, n):
            density_score = np.exp(-0.3 * (abs(local_density[i]) + abs(local_density[j])))
            density_scores[i, j] = density_scores[j, i] = density_score
    
    factors['density'] = density_scores
    
    # 4. MST-based scores
    mst_scores = np.ones((n, n))
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            for i in range(n):
                for j in range(i+1, n):
                    if mst_present[i, j]:
                        degree_i = np.sum(mst_present[i])
                        degree_j = np.sum(mst_present[j])
                        if degree_i == 1 or degree_j == 1:
                            mst_scores[i, j] = mst_scores[j, i] = 1.5
                        else:
                            mst_scores[i, j] = mst_scores[j, i] = 2.0
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    factors['mst'] = mst_scores
    
    return factors


def _compute_lookahead_scores(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute lookahead scores via partial tour simulations."""
    n = stats['n']
    lookahead = np.ones((n, n))
    
    if n < 4:
        return lookahead
    
    # Adaptive parameters
    complexity = stats['cv_dist'] * stats['cluster_tendency']
    n_simulations = min(20, max(3, int(8 + 4 * complexity)))
    threshold = stats['mean_dist'] * (1.0 + 0.3 * stats['cv_dist'])
    
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] > threshold:
                continue
                
            scores = []
            for _ in range(n_simulations):
                score = _simulate_partial_tour(distance_matrix, i, j, stats)
                scores.append(score)
            
            if scores:
                scores_sorted = np.sort(scores)
                trim = max(1, n_simulations // 4)
                trimmed_scores = scores_sorted[trim:-trim] if len(scores_sorted) > 2*trim else scores_sorted
                lookahead[i, j] = lookahead[j, i] = np.mean(trimmed_scores) if len(trimmed_scores) > 0 else np.median(scores)
    
    # Normalize
    max_val = np.max(lookahead)
    if max_val > 0:
        lookahead = lookahead / max_val
    
    np.fill_diagonal(lookahead, 0)
    return lookahead


def _simulate_partial_tour(distance_matrix: np.ndarray, start_i: int, start_j: int, stats: dict) -> float:
    """Simulate partial tour extension starting with edge (start_i, start_j)."""
    n = stats['n']
    visited = np.zeros(n, dtype=bool)
    visited[start_i] = visited[start_j] = True
    
    current = start_j
    tour_length = distance_matrix[start_i, start_j]
    
    # Adaptive number of steps
    n_steps = min(8, max(3, n // 3))
    
    for step in range(n_steps):
        unvisited = np.where(~visited)[0]
        if len(unvisited) == 0:
            break
        
        dists_to_current = distance_matrix[current, unvisited]
        
        # Balance exploration and exploitation
        if np.random.random() < 0.7:
            next_idx = np.argmin(dists_to_current)
        else:
            weights = np.exp(-dists_to_current / (stats['mean_dist'] + 1e-8))
            if np.sum(weights) > 0:
                weights = weights / np.sum(weights)
                next_idx = np.random.choice(len(unvisited), p=weights)
            else:
                next_idx = np.argmin(dists_to_current)
        
        next_node = unvisited[next_idx]
        tour_length += dists_to_current[next_idx]
        visited[next_node] = True
        current = next_node
    
    # Calculate score
    nodes_visited = np.sum(visited)
    expected_length = n_steps * stats['mean_dist']
    efficiency = nodes_visited / (n_steps + 1)
    score = (expected_length / (tour_length + 1e-8)) * (0.8 + 0.2 * efficiency)
    
    return min(score, 3.0)


def _compute_adaptive_weights(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute adaptive weights based on instance characteristics."""
    n = stats['n']
    cv = stats['cv_dist']
    sparsity = stats['sparsity']
    cluster_tendency = stats['cluster_tendency']
    regularity = stats['regularity']
    
    weights = {
        'w_inv': 1.2 - 0.2 * regularity,
        'w_regret': 0.4 + 0.3 * (1.0 - sparsity),
        'w_density': 0.3 + 0.4 * cluster_tendency,
        'w_lookahead': 0.2 + 0.3 * cv,
        'w_mst': max(0.1, 0.6 - 0.01 * n - 0.2 * regularity)
    }
    
    # Adjust based on coefficient of variation
    cv_factor = 1.0 + 0.5 * cv
    weights['w_regret'] *= cv_factor
    weights['w_lookahead'] *= cv_factor
    
    # Normalize to target sum
    target_sum = 4.0
    current_sum = sum(weights.values())
    if current_sum > 0:
        scale_factor = target_sum / current_sum
        for key in weights:
            weights[key] *= scale_factor
    
    return weights


def _build_candidate_mask(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Build mask of candidate edges to consider."""
    n = stats['n']
    
    # Adaptive candidate list size
    base_k = min(15, max(4, int(np.log2(n) * 3)))
    k = int(base_k * (1.2 - 0.4 * stats['sparsity'] + 0.3 * stats['regularity']))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < len(dists):
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Add mutual neighbors and MST edges
    mutual_mask = candidate_mask & candidate_mask.T
    
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            candidate_mask = candidate_mask | mst_present | mutual_mask
        except:
            candidate_mask = candidate_mask | mutual_mask
    
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask


def _post_process(H: np.ndarray, distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Post-process heuristic matrix with sparsification and connectivity checks."""
    n = stats['n']
    
    # Ensure minimum connectivity
    min_edges = min(3, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i])[0]
        if len(non_zero) < min_edges:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    base_score = 1 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    # Sparsify based on percentile
    if n > 10:
        threshold = np.percentile(H[H > 0], 30)
        H[H < threshold] = 0
    
    # Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
