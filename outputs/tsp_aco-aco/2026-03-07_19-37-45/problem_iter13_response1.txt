import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with dual-regret decomposition,
    two-step lookahead penalty, and meta-weighted multi-factor combination.
    
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Analyze instance characteristics
    instance_stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Compute enhanced factors
    factors = _compute_enhanced_factors_v2(distance_matrix, instance_stats)
    
    # Step 3: Compute adaptive weights based on instance properties
    weights = _compute_meta_weights_v2(instance_stats)
    
    # Step 4: Build candidate mask for efficiency
    candidate_mask = _build_enhanced_candidate_mask(distance_matrix, instance_stats)
    
    # Step 5: Combine factors with geometric mean and meta-weights
    H = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                combined = (
                    (factors['inv_dist'][i, j] ** weights['w_inv']) *
                    (factors['dual_regret'][i, j] ** weights['w_regret']) *
                    (factors['spectral_density'][i, j] ** weights['w_density']) *
                    (factors['two_step_penalty'][i, j] ** weights['w_lookahead']) *
                    (factors['mst_betweenness'][i, j] ** weights['w_mst']) *
                    (factors['centrality'][i, j] ** weights['w_centrality'])
                )
                H[i, j] = H[j, i] = combined
    
    # Step 6: Post-processing and sparsification
    H = _post_process_v2(H, distance_matrix, instance_stats, candidate_mask)
    
    return H

def _analyze_instance_v2(distance_matrix: np.ndarray) -> dict:
    """Enhanced instance analysis with spectral properties."""
    n = distance_matrix.shape[0]
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    stats = {
        'n': n,
        'mean_dist': np.mean(dist_no_diag),
        'std_dist': np.std(dist_no_diag),
        'cv_dist': np.std(dist_no_diag) / (np.mean(dist_no_diag) + 1e-8),
        'sparsity': _estimate_enhanced_sparsity(distance_matrix),
        'cluster_tendency': _estimate_hopkins_statistic(distance_matrix),
        'regularity': _estimate_regularity(distance_matrix)
    }
    
    return stats

def _estimate_enhanced_sparsity(distance_matrix: np.ndarray) -> float:
    """Estimate instance sparsity using distance distribution analysis."""
    n = distance_matrix.shape[0]
    if n < 4:
        return 0.5
    
    # Remove diagonal and consider only upper triangle
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    
    # Compute skewness of distance distribution
    skew = (np.mean((upper_tri - np.mean(upper_tri))**3) / 
            (np.std(upper_tri)**3 + 1e-8))
    
    # Normalize to [0,1] range
    sparsity = 1 / (1 + np.exp(-skew / 2))
    return sparsity

def _estimate_hopkins_statistic(distance_matrix: np.ndarray) -> float:
    """Compute Hopkins statistic for cluster tendency estimation."""
    n = distance_matrix.shape[0]
    if n < 10:
        return 0.5
    
    m = min(15, n // 3)
    sample_indices = np.random.choice(n, m, replace=False)
    
    sample_nn_dists = []
    random_nn_dists = []
    
    for idx in sample_indices:
        # Distance to nearest neighbor (excluding self)
        dists = distance_matrix[idx].copy()
        dists[idx] = np.inf
        sample_nn_dists.append(np.min(dists))
        
        # Generate random distances
        random_dists = np.random.choice(distance_matrix.flatten(), m)
        random_nn_dists.append(np.min(random_dists))
    
    hopkins = np.sum(sample_nn_dists) / (np.sum(sample_nn_dists) + np.sum(random_nn_dists) + 1e-8)
    return hopkins

def _estimate_regularity(distance_matrix: np.ndarray) -> float:
    """Estimate how regularly distributed the points are."""
    n = distance_matrix.shape[0]
    if n < 4:
        return 0.5
    
    # Compute variance of nearest neighbor distances
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    
    cv_nn = np.std(nn_dists) / (np.mean(nn_dists) + 1e-8)
    regularity = 1 / (1 + cv_nn)
    return regularity

def _compute_enhanced_factors_v2(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute enhanced heuristic factors with dual-regret and spectral density."""
    n = stats['n']
    factors = {}
    
    # 1. Inverse distance (normalized)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    factors['inv_dist'] = inv_dist / np.max(inv_dist) if np.max(inv_dist) > 0 else inv_dist
    
    # 2. Dual-regret decomposition
    factors['dual_regret'] = _compute_dual_regret_scores(distance_matrix, stats)
    
    # 3. Spectral density scoring
    factors['spectral_density'] = _compute_spectral_density_scores(distance_matrix, stats)
    
    # 4. Two-step lookahead penalty
    factors['two_step_penalty'] = _compute_two_step_penalty_scores(distance_matrix, stats)
    
    # 5. MST betweenness scores
    factors['mst_betweenness'] = _compute_mst_betweenness_scores(distance_matrix, stats)
    
    # 6. Centrality scores
    factors['centrality'] = _compute_centrality_scores(distance_matrix, stats)
    
    return factors

def _compute_dual_regret_scores(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute regret scores using dual-regret decomposition (one-hop and two-hop)."""
    n = stats['n']
    regret = np.ones((n, n))
    regularity = stats['regularity']
    
    for i in range(n):
        # Get distances from node i
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        
        # Sort distances for one-hop analysis
        sorted_idx = np.argsort(dists_i)
        if len(sorted_idx) >= 2:
            best_dist = dists_i[sorted_idx[0]]
            second_best = dists_i[sorted_idx[1]]
            
            # One-hop regret (local alternative)
            one_hop_regret = (second_best - best_dist) / (best_dist + 1e-8)
            
            for k, j in enumerate(sorted_idx):
                if j != i:
                    # Compute one-hop regret for this edge
                    edge_one_hop = max(0, dists_i[j] - best_dist) / (best_dist + 1e-8)
                    
                    # Compute two-hop regret (semi-local alternative)
                    if n >= 3:
                        # Find best two-hop path i -> k -> j (k != i,j)
                        mask = np.ones(n, dtype=bool)
                        mask[[i, j]] = False
                        if np.any(mask):
                            two_hop_costs = distance_matrix[i][mask] + distance_matrix[:, j][mask]
                            best_two_hop = np.min(two_hop_costs)
                            edge_two_hop = max(0, dists_i[j] - best_two_hop) / (best_two_hop + 1e-8)
                        else:
                            edge_two_hop = 0
                    else:
                        edge_two_hop = 0
                    
                    # Combine one-hop and two-hop regret with regularity-based weighting
                    w1 = 0.7 + 0.3 * regularity  # More weight to one-hop in regular instances
                    w2 = 1.0 - w1
                    combined_regret = (w1 * edge_one_hop + w2 * edge_two_hop) / (w1 + w2 + 1e-8)
                    
                    # Convert to score (lower regret = higher score)
                    regret[i, j] = 1 / (1 + combined_regret)
    
    # Symmetrize and zero diagonal
    regret = (regret + regret.T) / 2
    np.fill_diagonal(regret, 0)
    return regret

def _compute_spectral_density_scores(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute density scores using spectral clustering principles."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    if n < 3:
        return density_scores
    
    # Create affinity matrix (closer = higher affinity)
    gamma = 1.0 / (2.0 * (stats['mean_dist'] ** 2) + 1e-8)
    affinity = np.exp(-gamma * distance_matrix ** 2)
    np.fill_diagonal(affinity, 0)
    
    # Compute degree matrix
    degree = np.diag(np.sum(affinity, axis=1))
    
    # Normalized Laplacian (simplified density estimation)
    try:
        L = np.eye(n) - np.linalg.inv(degree) @ affinity
        # Use Fiedler vector (second smallest eigenvector) for density estimation
        eigvals, eigvecs = np.linalg.eigh(L)
        fiedler = eigvecs[:, 1]
        
        # Normalize Fiedler vector
        fiedler_norm = (fiedler - np.min(fiedler)) / (np.max(fiedler) - np.min(fiedler) + 1e-8)
        
        # Density score: favor edges connecting nodes with similar density characteristics
        for i in range(n):
            for j in range(i+1, n):
                density_diff = abs(fiedler_norm[i] - fiedler_norm[j])
                density_scores[i, j] = density_scores[j, i] = np.exp(-2.0 * density_diff)
    except:
        # Fallback to k-NN density
        k = min(5, max(2, n // 10))
        local_density = np.zeros(n)
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            k_nearest = np.partition(dists, k)[:k]
            local_density[i] = np.mean(k_nearest)
        
        # Normalize and score
        if np.std(local_density) > 1e-8:
            local_density = (local_density - np.mean(local_density)) / np.std(local_density)
        
        for i in range(n):
            for j in range(i+1, n):
                density_score = np.exp(-0.5 * abs(local_density[i] - local_density[j]))
                density_scores[i, j] = density_scores[j, i] = density_score
    
    return density_scores

def _compute_two_step_penalty_scores(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute penalty scores for edges with good two-hop alternatives."""
    n = stats['n']
    penalty_scores = np.ones((n, n))
    
    if n < 3:
        return penalty_scores
    
    # Adaptive penalty factor based on instance size and structure
    penalty_factor = 2.0 + 0.1 * stats['n']  # Stronger penalty for larger instances
    
    for i in range(n):
        for j in range(i+1, n):
            # Skip if edge is already very long
            if distance_matrix[i, j] > 3 * stats['mean_dist']:
                penalty_scores[i, j] = penalty_scores[j, i] = 0.1
                continue
            
            # Find best two-hop alternative
            mask = np.ones(n, dtype=bool)
            mask[[i, j]] = False
            if np.any(mask):
                two_hop_costs = distance_matrix[i][mask] + distance_matrix[:, j][mask]
                best_two_hop = np.min(two_hop_costs)
                
                # Apply penalty if two-hop is significantly better
                improvement_ratio = (distance_matrix[i, j] - best_two_hop) / (distance_matrix[i, j] + 1e-8)
                
                if improvement_ratio > 0.1:  # At least 10% improvement
                    penalty = np.exp(-penalty_factor * improvement_ratio)
                    penalty_scores[i, j] = penalty_scores[j, i] = max(0.1, penalty)
    
    return penalty_scores

def _compute_mst_betweenness_scores(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute scores based on MST edges with betweenness bonus."""
    n = stats['n']
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            # Compute MST
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Boost edges in MST
            mst_scores[mst_present] = 2.0
            
            # Additional boost for high-degree nodes in MST
            mst_degree = np.sum(mst_present, axis=1)
            if np.max(mst_degree) > 0:
                normalized_degree = mst_degree / np.max(mst_degree)
                
                # Give bonus to edges connecting high-degree nodes
                for i in range(n):
                    for j in range(i+1, n):
                        if mst_present[i, j]:
                            degree_bonus = 0.5 * (normalized_degree[i] + normalized_degree[j])
                            mst_scores[i, j] = mst_scores[j, i] += degree_bonus
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores

def _compute_centrality_scores(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute centrality scores for nodes and edges."""
    n = stats['n']
    centrality_scores = np.ones((n, n))
    
    if n < 3:
        return centrality_scores
    
    # Compute closeness centrality for each node
    closeness = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if np.sum(dists) > 0:
            closeness[i] = (n - 1) / np.sum(dists)
    
    # Normalize closeness
    if np.max(closeness) > 0:
        closeness = closeness / np.max(closeness)
    
    # Edge centrality: geometric mean of endpoint centralities
    for i in range(n):
        for j in range(i+1, n):
            centrality_scores[i, j] = centrality_scores[j, i] = np.sqrt(closeness[i] * closeness[j] + 1e-8)
    
    return centrality_scores

def _compute_meta_weights_v2(stats: dict) -> dict:
    """Compute meta-weights for factor combination based on instance properties."""
    n = stats['n']
    cv = stats['cv_dist']
    sparsity = stats['sparsity']
    cluster_tendency = stats['cluster_tendency']
    regularity = stats['regularity']
    
    # Meta-weights that adapt to multiple instance characteristics
    weights = {
        'w_inv': 1.0,  # Always important
        
        # Regret weight: higher for instances with more variation and regularity
        'w_regret': 0.4 + 0.3 * cv * regularity,
        
        # Density weight: higher for clustered instances
        'w_density': 0.3 + 0.4 * cluster_tendency,
        
        # Lookahead weight: higher for sparse, structured instances
        'w_lookahead': 0.2 + 0.3 * sparsity,
        
        # MST weight: decreases with size but increases with sparsity
        'w_mst': max(0.1, 0.4 - 0.008 * n) * (1 + 0.5 * sparsity),
        
        # Centrality weight: higher for regular instances
        'w_centrality': 0.1 + 0.2 * regularity
    }
    
    # Normalize to maintain scale (sum to 6 for the 6 factors)
    total = sum(weights.values())
    for key in weights:
        weights[key] = weights[key] / total * 6
    
    return weights

def _build_enhanced_candidate_mask(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Build mask of candidate edges using multiple criteria."""
    n = stats['n']
    
    # Base candidate count
    base_k = min(12, max(4, int(np.log2(n) * 4)))
    
    # Adjust based on instance properties
    k = int(base_k * (1 + 0.5 * stats['sparsity']))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # Primary criterion: k-nearest neighbors
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < len(dists):
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Secondary criterion: include edges with low two-step penalty
    two_step_penalty = _compute_two_step_penalty_scores(distance_matrix, stats)
    penalty_threshold = np.percentile(two_step_penalty[~np.eye(n, dtype=bool)], 75)
    
    high_penalty_edges = two_step_penalty >= penalty_threshold
    candidate_mask = candidate_mask | high_penalty_edges
    
    # Make symmetric and zero diagonal
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask

def _post_process_v2(H: np.ndarray, distance_matrix: np.ndarray, 
                     stats: dict, candidate_mask: np.ndarray) -> np.ndarray:
    """Enhanced post-processing with sparsification."""
    n = stats['n']
    
    # Set non-candidate edges to zero (sparsification)
    H[~candidate_mask] = 0
    
    # Ensure minimum connectivity for each node
    min_edges = min(3, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i])[0]
        if len(non_zero) < min_edges:
            # Add nearest neighbors not already in candidates
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    # Assign base score
                    H[i, j] = H[j, i] = 1 / (distance_matrix[i, j] + 1e-8)
    
    # Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Normalize final scores
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
