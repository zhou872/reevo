import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with adaptive multi-hop regret scoring.
    Returns a matrix of the same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Analyze instance characteristics
    stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Build candidate mask for efficiency
    candidate_mask = _build_candidate_mask_v2(distance_matrix, stats)
    
    # Step 3: Compute heuristic scores only for candidate edges
    H = np.zeros((n, n))
    
    for i in range(n):
        for j in range(i + 1, n):
            if candidate_mask[i, j]:
                # Compute combined heuristic score
                score = _compute_edge_score_v2(distance_matrix, i, j, stats)
                H[i, j] = H[j, i] = score
    
    # Step 4: Post-processing
    H = _post_process_v2(H, distance_matrix, stats)
    
    return H

def _analyze_instance_v2(distance_matrix: np.ndarray) -> dict:
    """Analyze instance characteristics for adaptive parameter tuning."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal for statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n - 1)
    
    # Basic statistics
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Regularity measure (inverse of coefficient of variation)
    regularity = 1 / (cv_dist + 1e-8)
    
    # Distance distribution skewness
    skewness = np.mean((dist_no_diag - mean_dist) ** 3) / (std_dist ** 3 + 1e-8)
    
    stats = {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'regularity': min(regularity, 10.0),  # Cap extreme values
        'skewness': abs(skewness),
        'min_dist': np.min(dist_no_diag),
        'max_dist': np.max(dist_no_diag)
    }
    
    return stats

def _build_candidate_mask_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Build mask of candidate edges to consider (for efficiency)."""
    n = stats['n']
    
    # Adaptive candidate list size based on instance characteristics
    base_k = min(15, max(5, int(np.log2(n) * 4)))
    
    # Adjust based on regularity: more regular instances need more candidates
    k = int(base_k * (1 + 0.3 * stats['regularity'] / 5))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k-nearest neighbors
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < len(dists):
            # Get indices of k smallest distances
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Make symmetric
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask

def _compute_edge_score_v2(distance_matrix: np.ndarray, i: int, j: int, stats: dict) -> float:
    """Compute heuristic score for a specific edge using multi-hop regret analysis."""
    n = stats['n']
    d_ij = distance_matrix[i, j]
    
    if d_ij <= 0:
        return 0.0
    
    # Base inverse distance component
    inv_dist = 1.0 / d_ij
    
    # Compute alternative path costs (2-hop and 3-hop)
    alt_costs = []
    
    # 2-hop alternatives: i -> k -> j
    for k in range(n):
        if k != i and k != j:
            alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
            if alt_cost > 0:
                alt_costs.append(alt_cost)
    
    # 3-hop alternatives: i -> k1 -> k2 -> j
    if n >= 5:
        for k1 in range(min(n, 10)):  # Limit for efficiency
            if k1 != i and k1 != j:
                for k2 in range(min(n, 10)):
                    if k2 != i and k2 != j and k2 != k1:
                        alt_cost = distance_matrix[i, k1] + distance_matrix[k1, k2] + distance_matrix[k2, j]
                        if alt_cost > 0:
                            alt_costs.append(alt_cost)
    
    if not alt_costs:
        return inv_dist
    
    # Find best alternative cost
    best_alt = min(alt_costs)
    
    # Compute regret ratio: best alternative / direct edge
    regret_ratio = best_alt / d_ij
    
    # Apply adaptive penalty/bonus based on regret ratio
    if regret_ratio > 1.0:
        # Direct edge is better than best alternative
        # Apply bonus: exponential function with adaptive scaling
        # Higher CV (more uniform) reduces bonus emphasis
        cv_factor = 1.0 / (1.0 + 2.0 * stats['cv_dist'])
        bonus = np.exp(cv_factor * (regret_ratio - 1.0))
        regret_score = bonus
    else:
        # Alternative is better than direct edge
        # Apply penalty: stronger when CV is low (more structured)
        cv_factor = 1.0 + (1.0 - stats['cv_dist'])
        penalty = np.exp(-cv_factor * (1.0 - regret_ratio))
        regret_score = penalty
    
    # Combine with base inverse distance
    # Adjust weight based on instance regularity
    w_inv = 0.7 + 0.3 * (1.0 / (1.0 + stats['regularity'] / 5.0))
    w_regret = 1.0 - w_inv
    
    combined_score = (inv_dist ** w_inv) * (regret_score ** w_regret)
    
    # Additional adjustment based on distance percentile
    # Favor edges that are neither too short nor too long
    dist_percentile = np.sum(distance_matrix <= d_ij) / (n * n)
    if 0.2 <= dist_percentile <= 0.8:
        combined_score *= 1.1
    
    return combined_score

def _post_process_v2(H: np.ndarray, distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Post-process the heuristic matrix."""
    n = stats['n']
    
    # Ensure symmetry and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Apply softmax normalization per row
    for i in range(n):
        row = H[i]
        if np.any(row > 0):
            # Shift to positive values for numerical stability
            row_min = np.min(row[row > 0])
            if row_min <= 0:
                row = row - np.min(row) + 1e-8
            # Softmax
            exp_row = np.exp(row - np.max(row))
            H[i] = exp_row / np.sum(exp_row)
    
    # Scale by overall matrix statistics
    if np.max(H) > 0:
        H = H / np.max(H)
    
    # Ensure at least some minimum value for top candidates
    for i in range(n):
        top_k = min(3, n - 1)
        indices = np.argsort(H[i])[-top_k:]
        for j in indices:
            if H[i, j] < 0.1:
                H[i, j] = H[j, i] = 0.1
    
    return H
