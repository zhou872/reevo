import numpy as np
from typing import Tuple, Optional

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with adaptive parameter tuning,
    three-hop path analysis, and data-driven heuristic combination.
    
    Args:
        distance_matrix: Symmetric matrix of distances between nodes.
        
    Returns:
        Heuristic matrix of same shape indicating edge desirability.
    """
    n = distance_matrix.shape[0]
    if n <= 3:
        # Handle trivial cases
        H = 1 / (distance_matrix + np.eye(n) * 1e-8)
        np.fill_diagonal(H, 0)
        return H
    
    # 1. Adaptive instance analysis for dynamic parameter tuning
    instance_features = analyze_instance(distance_matrix)
    
    # 2. Base components with adaptive parameters
    inv_dist = compute_base_inverse(distance_matrix)
    candidate_mask = build_adaptive_candidate_list(distance_matrix, instance_features)
    regret_scores = compute_enhanced_regret(distance_matrix, candidate_mask, instance_features)
    density_weights = compute_cluster_density(distance_matrix, instance_features)
    mst_bias = compute_mst_bias(distance_matrix)
    path_potentials = compute_three_hop_potentials(distance_matrix, candidate_mask)
    
    # 3. Machine-learning inspired heuristic combination
    H = combine_heuristics_adaptively(
        inv_dist, regret_scores, density_weights, mst_bias, path_potentials,
        candidate_mask, instance_features
    )
    
    # 4. Post-processing
    H = post_process_heuristics(H, distance_matrix, instance_features)
    
    return H


def analyze_instance(dist_mat: np.ndarray) -> dict:
    """Extract features from the distance matrix for adaptive parameter tuning."""
    n = dist_mat.shape[0]
    
    # Remove diagonal and flatten
    flat_dists = dist_mat[~np.eye(n, dtype=bool)].flatten()
    
    # Compute statistical features
    features = {
        'n': n,
        'mean_dist': np.mean(flat_dists),
        'std_dist': np.std(flat_dists),
        'cv_dist': np.std(flat_dists) / (np.mean(flat_dists) + 1e-10),  # Coefficient of variation
        'skewness': calculate_skewness(flat_dists),
        'sparsity_ratio': calculate_sparsity_ratio(dist_mat)
    }
    
    # Normalize features for weight computation
    features['size_factor'] = np.log2(n + 1) / 10
    features['complexity'] = features['cv_dist'] * features['skewness']
    
    return features


def calculate_skewness(data: np.ndarray) -> float:
    """Calculate skewness of distance distribution."""
    if len(data) < 2:
        return 0.0
    mean = np.mean(data)
    std = np.std(data)
    if std < 1e-10:
        return 0.0
    return np.mean(((data - mean) / std) ** 3)


def calculate_sparsity_ratio(dist_mat: np.ndarray) -> float:
    """Calculate ratio of edges significantly longer than average."""
    n = dist_mat.shape[0]
    flat_dists = dist_mat[~np.eye(n, dtype=bool)].flatten()
    median_dist = np.median(flat_dists)
    sparse_edges = np.sum(flat_dists > 2 * median_dist)
    return sparse_edges / len(flat_dists)


def compute_base_inverse(dist_mat: np.ndarray) -> np.ndarray:
    """Compute normalized inverse distances."""
    n = dist_mat.shape[0]
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (dist_mat + np.eye(n) * 1e-8)
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.isnan(inv_dist)] = 0
    
    # Normalize by node
    row_sums = inv_dist.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    inv_dist = inv_dist / row_sums
    
    np.fill_diagonal(inv_dist, 0)
    return inv_dist


def build_adaptive_candidate_list(dist_mat: np.ndarray, features: dict) -> np.ndarray:
    """Build candidate list with adaptive size based on instance features."""
    n = features['n']
    
    # Adaptive candidate size
    base_k = int(np.log2(n + 1)) + 3
    if features['sparsity_ratio'] > 0.3:
        base_k = min(base_k + 2, n - 1)
    
    k = min(max(5, base_k), n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k+1 smallest distances
        indices = np.argpartition(dist_mat[i], k + 1)[:k + 1]
        indices = indices[indices != i][:k]
        candidate_mask[i, indices] = True
    
    # Make symmetric
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask


def compute_enhanced_regret(dist_mat: np.ndarray, candidate_mask: np.ndarray, 
                           features: dict) -> np.ndarray:
    """Compute regret scores considering multiple alternatives."""
    n = features['n']
    regret_scores = np.zeros((n, n))
    
    # Adaptive number of alternatives to consider
    num_alternatives = min(4, n - 1)
    
    for i in range(n):
        if not np.any(candidate_mask[i]):
            continue
            
        # Get distances to candidate neighbors
        candidate_indices = np.where(candidate_mask[i])[0]
        if len(candidate_indices) < 2:
            continue
            
        candidate_dists = dist_mat[i, candidate_indices]
        
        # Sort candidates by distance
        sorted_idx = np.argsort(candidate_dists)
        sorted_candidates = candidate_indices[sorted_idx]
        
        # Compute regret for each candidate
        for rank, j in enumerate(sorted_candidates[:num_alternatives]):
            if rank == 0:
                # Best edge: regret is difference to second best
                if len(sorted_candidates) > 1:
                    second_best = candidate_dists[sorted_idx[1]]
                    regret = max(0, second_best - dist_mat[i, j])
                    regret_scores[i, j] = regret / (dist_mat[i, j] + 1e-8)
            else:
                # Other candidates: weighted regret based on rank
                best_dist = candidate_dists[sorted_idx[0]]
                regret = max(0, best_dist - dist_mat[i, j])
                weight = 1.0 / (rank + 1)  # Lower weight for worse alternatives
                regret_scores[i, j] = weight * regret / (dist_mat[i, j] + 1e-8)
    
    # Make symmetric and normalize
    regret_scores = (regret_scores + regret_scores.T) / 2
    max_regret = np.max(regret_scores)
    if max_regret > 0:
        regret_scores = regret_scores / max_regret
    
    return regret_scores


def compute_cluster_density(dist_mat: np.ndarray, features: dict) -> np.ndarray:
    """Compute density weights with cluster awareness."""
    n = features['n']
    density_weights = np.ones((n, n))
    
    # Adaptive neighborhood size based on instance complexity
    k_density = min(max(3, int(np.log2(n))), n - 1)
    if features['complexity'] > 0.5:
        k_density = min(k_density + 2, n - 1)
    
    # Compute local densities using k-nearest neighbors
    local_densities = np.zeros(n)
    for i in range(n):
        dists = dist_mat[i].copy()
        dists[i] = np.inf
        # Get k_density smallest distances
        nearest_dists = np.partition(dists, k_density)[:k_density]
        local_densities[i] = np.mean(nearest_dists)
    
    # Normalize and apply sigmoid transformation
    if np.std(local_densities) > 1e-8:
        local_densities_norm = (local_densities - np.mean(local_densities)) / np.std(local_densities)
    else:
        local_densities_norm = np.zeros_like(local_densities)
    
    # Compute pairwise density weights
    for i in range(n):
        for j in range(i + 1, n):
            # Combined density measure
            combined_density = (local_densities_norm[i] + local_densities_norm[j]) / 2
            
            # Sigmoid with adaptive steepness based on instance complexity
            steepness = 2.0 + features['complexity']
            density_weights[i, j] = 1 / (1 + np.exp(-steepness * combined_density))
            density_weights[j, i] = density_weights[i, j]
    
    return density_weights


def compute_mst_bias(dist_mat: np.ndarray) -> np.ndarray:
    """Compute MST bias using Prim's algorithm."""
    n = dist_mat.shape[0]
    mst_bias = np.ones((n, n))
    
    if n > 100:  # Use approximate MST for large instances
        return mst_bias
    
    # Prim's algorithm for exact MST
    visited = np.zeros(n, dtype=bool)
    in_mst = np.zeros((n, n), dtype=bool)
    
    # Start from node 0
    visited[0] = True
    
    for _ in range(n - 1):
        min_edge = np.inf
        u, v = -1, -1
        
        # Find minimum edge connecting visited to unvisited
        for i in np.where(visited)[0]:
            for j in np.where(~visited)[0]:
                if dist_mat[i, j] < min_edge:
                    min_edge = dist_mat[i, j]
                    u, v = i, j
        
        if u != -1 and v != -1:
            in_mst[u, v] = True
            in_mst[v, u] = True
            visited[v] = True
    
    # Apply bias to MST edges
    mst_bias[in_mst] = 1.5  # Moderate boost for MST edges
    
    return mst_bias


def compute_three_hop_potentials(dist_mat: np.ndarray, candidate_mask: np.ndarray) -> np.ndarray:
    """Compute three-hop path potentials for each edge."""
    n = dist_mat.shape[0]
    path_potentials = np.zeros((n, n))
    
    # Only compute for candidate edges to save computation
    candidate_indices = np.where(candidate_mask)
    
    for idx in range(len(candidate_indices[0])):
        i, j = candidate_indices[0][idx], candidate_indices[1][idx]
        if i >= j:  # Compute only once per undirected edge
            continue
        
        # Find common neighbors for three-hop analysis
        neighbors_i = np.where(candidate_mask[i])[0]
        neighbors_j = np.where(candidate_mask[j])[0]
        
        # Consider two-hop and three-hop paths through common neighbors
        best_two_hop = np.inf
        best_three_hop = np.inf
        
        # Two-hop paths: i -> k -> j
        common_neighbors = np.intersect1d(neighbors_i, neighbors_j)
        for k in common_neighbors:
            if k != i and k != j:
                two_hop = dist_mat[i, k] + dist_mat[k, j]
                best_two_hop = min(best_two_hop, two_hop)
        
        # Three-hop paths: i -> k -> l -> j
        for k in neighbors_i:
            if k == i or k == j:
                continue
            for l in neighbors_j:
                if l == i or l == j or l == k:
                    continue
                if candidate_mask[k, l]:
                    three_hop = dist_mat[i, k] + dist_mat[k, l] + dist_mat[l, j]
                    best_three_hop = min(best_three_hop, three_hop)
        
        # Compute potential score
        direct_cost = dist_mat[i, j]
        
        if best_two_hop < np.inf:
            two_hop_ratio = best_two_hop / (direct_cost + 1e-8)
        else:
            two_hop_ratio = 2.0  # Default if no two-hop path found
        
        if best_three_hop < np.inf:
            three_hop_ratio = best_three_hop / (direct_cost + 1e-8)
        else:
            three_hop_ratio = 3.0  # Default if no three-hop path found
        
        # Combine ratios (favor edges with poor alternative paths)
        path_potential = (two_hop_ratio + three_hop_ratio) / 2
        path_potentials[i, j] = path_potential
        path_potentials[j, i] = path_potential
    
    # Normalize
    max_potential = np.max(path_potentials)
    if max_potential > 0:
        path_potentials = path_potentials / max_potential
    
    return path_potentials


def combine_heuristics_adaptively(inv_dist: np.ndarray, regret: np.ndarray,
                                 density: np.ndarray, mst: np.ndarray,
                                 paths: np.ndarray, candidate_mask: np.ndarray,
                                 features: dict) -> np.ndarray:
    """Combine heuristics with adaptive weights based on instance features."""
    n = features['n']
    H = np.zeros((n, n))
    
    # Adaptive weight computation based on instance characteristics
    if features['cv_dist'] < 0.3:  # Uniform distances
        alpha, beta, gamma, delta, epsilon = 0.7, 0.4, 0.3, 0.2, 0.4
    elif features['sparsity_ratio'] > 0.4:  # Sparse instance
        alpha, beta, gamma, delta, epsilon = 0.5, 0.6, 0.4, 0.3, 0.5
    else:  # General case
        alpha, beta, gamma, delta, epsilon = 0.6, 0.5, 0.3, 0.2, 0.4
    
    # Adjust weights based on instance size
    size_factor = features['size_factor']
    alpha *= (1 + 0.2 * size_factor)
    epsilon *= (1 + 0.3 * size_factor)  # Emphasize path potentials for larger instances
    
    # Combine heuristics multiplicatively for candidate edges
    for i in range(n):
        for j in range(n):
            if i != j and candidate_mask[i, j]:
                # Multiplicative combination with adaptive weights
                combined = (
                    (inv_dist[i, j] ** alpha) *
                    ((1 + regret[i, j]) ** beta) *
                    (density[i, j] ** gamma) *
                    (mst[i, j] ** delta) *
                    ((1 + paths[i, j]) ** epsilon)
                )
                H[i, j] = combined
    
    return H


def post_process_heuristics(H: np.ndarray, dist_mat: np.ndarray, 
                           features: dict) -> np.ndarray:
    """Post-process heuristic matrix with symmetry and normalization."""
    n = features['n']
    
    # Ensure symmetry
    H = (H + H.T) / 2
    
    # Zero diagonal
    np.fill_diagonal(H, 0)
    
    # Apply adaptive sparsification
    if n > 20:
        # Keep top edges per node
        edges_per_node = max(4, int(np.sqrt(n)))
        H_sparse = np.zeros_like(H)
        
        for i in range(n):
            if np.any(H[i] > 0):
                # Get indices of top edges
                top_indices = np.argsort(H[i])[-edges_per_node:]
                H_sparse[i, top_indices] = H[i, top_indices]
        
        H = H_sparse
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    # Ensure symmetry again after sparsification
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    return H
