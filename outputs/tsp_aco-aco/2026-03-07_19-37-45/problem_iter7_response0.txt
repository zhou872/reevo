import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Dict, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with two-hop regret and adaptive weighting.
    Returns a matrix of the same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Analyze instance characteristics for adaptive parameter tuning
    instance_stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Compute base factors with two-hop regret
    base_factors = _compute_base_factors_v2(distance_matrix, instance_stats)
    
    # Step 3: Perform limited lookahead simulations for critical edges
    lookahead_scores = _compute_lookahead_scores_v2(distance_matrix, instance_stats)
    
    # Step 4: Dynamic weight adjustment based on instance characteristics
    weights = _compute_adaptive_weights_v2(distance_matrix, instance_stats)
    
    # Step 5: Combine all factors with geometric mean
    H = np.zeros((n, n))
    
    # Define candidate edges for efficiency
    candidate_mask = _build_candidate_mask_v2(distance_matrix, instance_stats)
    
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Multiplicative combination with adaptive exponents
                combined = (
                    (base_factors['inv_dist'][i, j] ** weights['w_inv']) *
                    (base_factors['two_hop_regret'][i, j] ** weights['w_regret']) *
                    (base_factors['density'][i, j] ** weights['w_density']) *
                    (lookahead_scores[i, j] ** weights['w_lookahead']) *
                    (base_factors['mst'][i, j] ** weights['w_mst'])
                )
                H[i, j] = H[j, i] = combined
    
    # Step 6: Post-processing and sparsification
    H = _post_process_heuristics_v2(H, distance_matrix, instance_stats)
    
    return H

def _analyze_instance_v2(distance_matrix: np.ndarray) -> Dict:
    """Analyze instance characteristics for adaptive parameter tuning."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal for statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    stats = {
        'n': n,
        'mean_dist': np.mean(dist_no_diag),
        'std_dist': np.std(dist_no_diag),
        'cv_dist': np.std(dist_no_diag) / (np.mean(dist_no_diag) + 1e-8),
        'percentile_25': np.percentile(dist_no_diag, 25),
        'percentile_75': np.percentile(dist_no_diag, 75),
        'sparsity': _estimate_sparsity_v2(distance_matrix),
        'cluster_tendency': _estimate_cluster_tendency_v2(distance_matrix),
        'regularity': _estimate_regularity(distance_matrix)
    }
    
    return stats

def _estimate_sparsity_v2(distance_matrix: np.ndarray) -> float:
    """Estimate how sparse/structured the instance is."""
    n = distance_matrix.shape[0]
    if n <= 10:
        return 0.5
    
    # Use quartile coefficient of dispersion
    flat_dists = distance_matrix[~np.eye(n, dtype=bool)]
    q1 = np.percentile(flat_dists, 25)
    q3 = np.percentile(flat_dists, 75)
    qcd = (q3 - q1) / (q3 + q1 + 1e-8)
    
    return 1.0 - qcd  # Higher for more uniform distributions

def _estimate_cluster_tendency_v2(distance_matrix: np.ndarray) -> float:
    """Estimate whether points form natural clusters."""
    n = distance_matrix.shape[0]
    
    if n < 10:
        return 0.5
    
    # Approximate silhouette score using k-means++ initialization
    from sklearn.cluster import KMeans
    try:
        # Use 2 clusters for simplicity
        kmeans = KMeans(n_clusters=min(5, n//3), n_init=1, max_iter=10, random_state=42)
        
        # Use normalized distances as features
        row_means = np.mean(distance_matrix, axis=1, keepdims=True)
        row_stds = np.std(distance_matrix, axis=1, keepdims=True) + 1e-8
        normalized = (distance_matrix - row_means) / row_stds
        
        labels = kmeans.fit_predict(normalized)
        
        # Calculate average intra-cluster distance
        intra_dist = 0
        inter_dist = 0
        for i in range(n):
            same_cluster = labels == labels[i]
            diff_cluster = ~same_cluster
            same_cluster[i] = False  # Exclude self
            
            if np.any(same_cluster):
                intra_dist += np.mean(distance_matrix[i, same_cluster])
            if np.any(diff_cluster):
                inter_dist += np.mean(distance_matrix[i, diff_cluster])
        
        silhouette = (inter_dist - intra_dist) / (max(intra_dist, inter_dist) + 1e-8)
        return max(0, min(1, (silhouette + 1) / 2))  # Normalize to [0,1]
    except:
        return 0.5

def _estimate_regularity(distance_matrix: np.ndarray) -> float:
    """Estimate regularity of the instance (how uniform vs clustered)."""
    n = distance_matrix.shape[0]
    
    # Coefficient of variation of nearest neighbor distances
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    
    cv_nn = np.std(nn_dists) / (np.mean(nn_dists) + 1e-8)
    regularity = 1.0 / (1.0 + cv_nn)  # Higher for more regular instances
    
    return regularity

def _compute_base_factors_v2(distance_matrix: np.ndarray, stats: Dict) -> Dict:
    """Compute base heuristic factors with two-hop regret."""
    n = stats['n']
    
    factors = {}
    
    # 1. Inverse distance
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    factors['inv_dist'] = inv_dist
    
    # 2. Two-hop regret-based scoring
    factors['two_hop_regret'] = _compute_two_hop_regret(distance_matrix, stats)
    
    # 3. Density-based scoring
    factors['density'] = _compute_density_scores_v2(distance_matrix, stats)
    
    # 4. MST-based scoring
    factors['mst'] = _compute_mst_scores_v2(distance_matrix)
    
    return factors

def _compute_two_hop_regret(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute two-hop regret scores for each edge with variance-adaptive penalty."""
    n = stats['n']
    regret = np.ones((n, n))
    cv = stats['cv_dist']
    
    # Variance-adaptive penalty threshold
    penalty_factor = 1.0 + 0.5 * cv  # Stricter for uniform instances
    
    for i in range(n):
        for j in range(i+1, n):
            if i == j:
                continue
                
            direct_cost = distance_matrix[i, j]
            
            # Find best two-hop alternative i-k-j
            two_hop_costs = []
            for k in range(n):
                if k != i and k != j:
                    two_hop = distance_matrix[i, k] + distance_matrix[k, j]
                    two_hop_costs.append(two_hop)
            
            if two_hop_costs:
                best_two_hop = min(two_hop_costs)
                
                # Compute regret with adaptive threshold
                if best_two_hop < direct_cost * penalty_factor:
                    # There's a significantly better two-hop path
                    regret_score = best_two_hop / (direct_cost + 1e-8)
                else:
                    # Direct edge is good
                    regret_score = direct_cost / (best_two_hop + 1e-8)
                
                # Apply softmax normalization
                regret[i, j] = regret[j, i] = np.exp(regret_score - 1.0)
    
    np.fill_diagonal(regret, 0)
    return regret

def _compute_density_scores_v2(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute density-based scores favoring edges in sparse regions."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Estimate local density using adaptive k-nearest neighbors
    k = min(7, max(3, n // 8))
    
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Average distance to k nearest neighbors
        k_nearest = np.partition(dists, k)[:k]
        local_density[i] = np.mean(k_nearest)
    
    # Normalize densities with robust scaling
    q1, q3 = np.percentile(local_density, [25, 75])
    iqr = q3 - q1 + 1e-8
    local_density = (local_density - np.median(local_density)) / iqr
    
    # Create density scores: favor edges connecting sparse areas
    for i in range(n):
        for j in range(i+1, n):
            # Combined density score (higher for sparse-sparse connections)
            density_score = np.exp(-0.3 * (abs(local_density[i]) + abs(local_density[j])))
            density_scores[i, j] = density_scores[j, i] = density_score
    
    return density_scores

def _compute_mst_scores_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute scores based on Minimum Spanning Tree with enhanced weighting."""
    n = distance_matrix.shape[0]
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Enhanced MST scoring based on edge centrality
            for i in range(n):
                for j in range(i+1, n):
                    if mst_present[i, j]:
                        # Edges in MST get higher scores
                        mst_scores[i, j] = mst_scores[j, i] = 2.0
                        
                        # Check if edge is critical (connects two components with many nodes)
                        # Simplified: check if removing it would create large components
                        degree_i = np.sum(mst_present[i])
                        degree_j = np.sum(mst_present[j])
                        if degree_i == 1 or degree_j == 1:
                            # Leaf edge - slightly less important
                            mst_scores[i, j] = mst_scores[j, i] = 1.5
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores

def _compute_lookahead_scores_v2(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """
    Compute lookahead scores by simulating partial tour extensions
    with adaptive thresholds based on variance.
    """
    n = stats['n']
    lookahead = np.ones((n, n))
    
    if n < 4:  # Too small for meaningful lookahead
        return lookahead
    
    # Adaptive number of simulations based on instance size and structure
    complexity = stats['cv_dist'] * stats['cluster_tendency']
    n_simulations = min(25, max(5, int(10 + 5 * complexity)))
    
    # Variance-adaptive threshold for considering edges
    threshold = stats['mean_dist'] * (1.0 + 0.3 * stats['cv_dist'])
    
    for i in range(n):
        for j in range(i+1, n):
            # Skip if not promising based on adaptive threshold
            if distance_matrix[i, j] > threshold:
                continue
                
            scores = []
            for _ in range(n_simulations):
                # Simulate a partial tour starting with edge (i,j)
                score = _simulate_partial_tour_v2(distance_matrix, i, j, stats)
                scores.append(score)
            
            if scores:
                # Use trimmed mean to reduce outlier effects
                scores_sorted = np.sort(scores)
                trim = max(1, n_simulations // 4)
                trimmed_scores = scores_sorted[trim:-trim]
                lookahead[i, j] = lookahead[j, i] = np.mean(trimmed_scores) if len(trimmed_scores) > 0 else np.median(scores)
    
    # Normalize
    max_val = np.max(lookahead)
    if max_val > 0:
        lookahead = lookahead / max_val
    
    np.fill_diagonal(lookahead, 0)
    return lookahead

def _simulate_partial_tour_v2(distance_matrix: np.ndarray, start_i: int, start_j: int, stats: Dict) -> float:
    """
    Simulate extending a partial tour starting with edge (i,j).
    Returns a score based on quality of the extended partial tour.
    """
    n = stats['n']
    visited = np.zeros(n, dtype=bool)
    visited[start_i] = visited[start_j] = True
    
    current = start_j
    tour_length = distance_matrix[start_i, start_j]
    
    # Adaptive number of steps based on instance size
    n_steps = min(6, max(3, n // 4))
    
    for step in range(n_steps):
        # Find unvisited nodes
        unvisited = np.where(~visited)[0]
        if len(unvisited) == 0:
            break
        
        # Choose next node using a combination of greedy and exploratory behavior
        dists_to_current = distance_matrix[current, unvisited]
        
        # Balance between exploration and exploitation
        if np.random.random() < 0.7:  # 70% greedy
            next_idx = np.argmin(dists_to_current)
        else:  # 30% probabilistic
            weights = np.exp(-dists_to_current / (stats['mean_dist'] + 1e-8))
            weights = weights / (np.sum(weights) + 1e-8)
            next_idx = np.random.choice(len(unvisited), p=weights)
        
        next_node = unvisited[next_idx]
        tour_length += dists_to_current[next_idx]
        visited[next_node] = True
        current = next_node
    
    # Score based on comparison to expected random walk length
    expected_length = n_steps * stats['mean_dist']
    # Add a small bonus for using fewer edges than steps (if we visited more nodes)
    nodes_visited = np.sum(visited)
    efficiency = nodes_visited / (n_steps + 1)
    
    score = (expected_length / (tour_length + 1e-8)) * (0.8 + 0.2 * efficiency)
    
    return min(score, 4.0)  # Cap extreme values

def _compute_adaptive_weights_v2(distance_matrix: np.ndarray, stats: Dict) -> Dict:
    """Dynamically compute weights based on instance characteristics with instance-based weighting."""
    n = stats['n']
    cv = stats['cv_dist']
    sparsity = stats['sparsity']
    cluster_tendency = stats['cluster_tendency']
    regularity = stats['regularity']
    
    # Base weights with instance-based adjustments
    weights = {
        'w_inv': 1.2 - 0.2 * regularity,  # Less important for regular instances
        
        # Two-hop regret weight: higher for instances with clear structure
        'w_regret': 0.4 + 0.3 * (1.0 - sparsity),
        
        # Density weight: higher for clustered instances
        'w_density': 0.3 + 0.4 * cluster_tendency,
        
        # Lookahead weight: higher for complex, non-uniform instances
        'w_lookahead': 0.2 + 0.3 * cv,
        
        # MST weight: decreases with instance size and regularity
        'w_mst': max(0.1, 0.6 - 0.01 * n - 0.2 * regularity)
    }
    
    # Adjust based on coefficient of variation
    cv_factor = 1.0 + 0.5 * cv
    for key in ['w_regret', 'w_lookahead']:
        weights[key] *= cv_factor
    
    # Normalize to maintain overall scale (sum to target of 4.0 as per feature 4)
    target_sum = 4.0
    current_sum = sum(weights.values())
    if current_sum > 0:
        scale_factor = target_sum / current_sum
        for key in weights:
            weights[key] *= scale_factor
    
    return weights

def _build_candidate_mask_v2(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Build mask of candidate edges to consider with adaptive sparsification."""
    n = stats['n']
    
    # Adaptive candidate list size based on instance characteristics
    base_k = min(12, max(4, int(np.log2(n) * 2.5)))
    
    # Adjust based on sparsity and regularity
    k = int(base_k * (1.2 - 0.4 * stats['sparsity'] + 0.3 * stats['regularity']))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # Also consider reciprocal nearest neighbors
    for i in range(n):
        # Get k-nearest neighbors
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < len(dists):
            # Get indices of k smallest distances
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Add edges where nodes are mutually in each other's candidate lists
    mutual_mask = candidate_mask & candidate_mask.T
    
    # Also include edges that are in MST
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            candidate_mask = candidate_mask | mst_present | mutual_mask
        except:
            candidate_mask = candidate_mask | mutual_mask
    
    # Make symmetric
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask

def _post_process_heuristics_v2(H: np.ndarray, distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Post-process the heuristic matrix with sparsification."""
    n = stats['n']
    
    # Ensure minimum connectivity for each node
    min_edges = min(3, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i])[0]
        if len(non_zero) < min_edges:
            # Add nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    # Assign base score
                    base_score = 1 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    # Sparsify: set low values to zero based on percentile
    if n > 10:
        threshold = np.percentile(H[H > 0], 30)  # Keep top 70%
        H[H < threshold] = 0
    
    # Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
