import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge prioritization heuristic with community detection,
    bandit-based lookahead, and spectral graph analysis.
    
    Returns matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Enhanced instance analysis with community detection
    stats, communities = _analyze_instance_with_communities(distance_matrix)
    
    # Step 2: Compute base factors with community awareness
    base_factors = _compute_community_aware_factors(distance_matrix, stats, communities)
    
    # Step 3: Bandit-based lookahead simulations
    lookahead_scores = _compute_bandit_lookahead(distance_matrix, base_factors, stats, communities)
    
    # Step 4: Spectral graph analysis for structure understanding
    spectral_scores = _compute_spectral_scores(distance_matrix, stats, communities)
    
    # Step 5: Adaptive factor fusion with entropy balancing
    weights = _compute_adaptive_fusion_weights(distance_matrix, base_factors, stats, communities)
    
    # Step 6: Multi-factor integration
    H = _integrate_factors_with_communities(distance_matrix, base_factors, 
                                           lookahead_scores, spectral_scores,
                                           weights, stats, communities)
    
    # Step 7: Post-processing with community preservation
    H = _post_process_with_communities(H, distance_matrix, stats, communities)
    
    return H

def _analyze_instance_with_communities(distance_matrix: np.ndarray) -> Tuple[Dict, np.ndarray]:
    """Enhanced instance analysis with community detection and spectral analysis."""
    n = distance_matrix.shape[0]
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    # Basic statistics
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Community detection via hierarchical clustering
    communities = _detect_communities(distance_matrix)
    n_communities = len(np.unique(communities))
    
    # Compute community statistics
    intra_community_dists = []
    inter_community_dists = []
    
    for i in range(n):
        for j in range(i+1, n):
            if communities[i] == communities[j]:
                intra_community_dists.append(distance_matrix[i, j])
            else:
                inter_community_dists.append(distance_matrix[i, j])
    
    intra_mean = np.mean(intra_community_dists) if intra_community_dists else mean_dist
    inter_mean = np.mean(inter_community_dists) if inter_community_dists else mean_dist
    community_ratio = intra_mean / (inter_mean + 1e-8)
    
    # Spectral analysis
    spectral_properties = _compute_spectral_properties(distance_matrix)
    
    stats = {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'n_communities': n_communities,
        'community_ratio': community_ratio,
        'intra_mean': intra_mean,
        'inter_mean': inter_mean,
        'spectral_gap': spectral_properties['spectral_gap'],
        'algebraic_connectivity': spectral_properties['algebraic_connectivity']
    }
    
    return stats, communities

def _detect_communities(distance_matrix: np.ndarray) -> np.ndarray:
    """Detect communities using hierarchical clustering with adaptive parameters."""
    n = distance_matrix.shape[0]
    
    if n <= 10:
        # For small instances, consider all nodes as one community
        return np.zeros(n, dtype=int)
    
    # Convert distance to similarity for clustering
    max_dist = np.max(distance_matrix)
    similarity = max_dist - distance_matrix + 1e-8
    
    # Perform hierarchical clustering
    condensed_dist = squareform(1 / (similarity + 1e-8))
    Z = linkage(condensed_dist, method='average')
    
    # Adaptive threshold based on instance statistics
    median_dist = np.median(distance_matrix[~np.eye(n, dtype=bool)])
    threshold = median_dist * 1.5
    
    # Determine optimal number of clusters
    max_clusters = min(10, n // 5)
    min_clusters = max(2, n // 20)
    
    # Convert threshold to cophenetic distance scale
    cophenetic_threshold = threshold * (n / 10)  # Rough scaling
    
    clusters = fcluster(Z, cophenetic_threshold, criterion='distance')
    
    # Ensure reasonable number of clusters
    unique_clusters = np.unique(clusters)
    if len(unique_clusters) > max_clusters:
        # Force fewer clusters
        clusters = fcluster(Z, max_clusters, criterion='maxclust')
    elif len(unique_clusters) < min_clusters and n > 20:
        clusters = fcluster(Z, min_clusters, criterion='maxclust')
    
    # Renumber clusters from 0
    unique_clusters = np.unique(clusters)
    cluster_map = {old: new for new, old in enumerate(unique_clusters)}
    return np.array([cluster_map[c] for c in clusters])

def _compute_spectral_properties(distance_matrix: np.ndarray) -> Dict:
    """Compute spectral properties of the distance graph."""
    n = distance_matrix.shape[0]
    
    if n < 4:
        return {'spectral_gap': 0.5, 'algebraic_connectivity': 0.5}
    
    # Convert distances to similarities for Laplacian
    sigma = np.median(distance_matrix) * 0.5
    similarity = np.exp(-distance_matrix ** 2 / (2 * sigma ** 2))
    np.fill_diagonal(similarity, 0)
    
    # Compute graph Laplacian
    D = np.diag(np.sum(similarity, axis=1))
    L = D - similarity
    
    # Compute eigenvalues (only a few)
    try:
        eigenvalues = np.linalg.eigvalsh(L)
        eigenvalues = np.sort(eigenvalues)
        
        # Spectral gap: difference between first and second smallest eigenvalues
        algebraic_connectivity = eigenvalues[1] if len(eigenvalues) > 1 else 0
        
        # Estimate spectral gap (difference between two smallest non-zero eigenvalues)
        spectral_gap = eigenvalues[2] - eigenvalues[1] if len(eigenvalues) > 2 else 0.1
    except:
        algebraic_connectivity = 0.1
        spectral_gap = 0.1
    
    return {
        'spectral_gap': float(spectral_gap),
        'algebraic_connectivity': float(algebraic_connectivity)
    }

def _compute_community_aware_factors(distance_matrix: np.ndarray, 
                                   stats: Dict, 
                                   communities: np.ndarray) -> Dict[str, np.ndarray]:
    """Compute base heuristic factors with community awareness."""
    n = stats['n']
    factors = {}
    
    # 1. Inverse distance with community adjustment
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    # Boost intra-community edges, penalize inter-community edges
    community_factor = np.ones((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if communities[i] == communities[j]:
                # Intra-community boost
                boost = 1.0 + 0.3 * min(stats['community_ratio'], 2.0)
                community_factor[i, j] = community_factor[j, i] = boost
            else:
                # Inter-community penalty (but not too harsh)
                penalty = 0.8 + 0.2 * (1 / (1 + abs(stats['community_ratio'] - 1)))
                community_factor[i, j] = community_factor[j, i] = penalty
    
    factors['inv_dist'] = inv_dist * community_factor
    
    # 2. Community-aware regret scores
    factors['regret'] = _compute_community_regret(distance_matrix, stats, communities)
    
    # 3. MST scores with community consideration
    factors['mst'] = _compute_community_mst(distance_matrix, stats, communities)
    
    # 4. Connectivity scores based on community structure
    factors['connectivity'] = _compute_connectivity_scores(distance_matrix, stats, communities)
    
    return factors

def _compute_community_regret(distance_matrix: np.ndarray, 
                            stats: Dict, 
                            communities: np.ndarray) -> np.ndarray:
    """Compute regret scores considering community structure."""
    n = stats['n']
    regret = np.ones((n, n))
    
    # Different k for intra vs inter community
    k_intra = min(3, max(2, n // 10))
    k_inter = min(5, max(2, n // 8))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Separate intra and inter community distances
        intra_mask = (communities == communities[i])
        intra_mask[i] = False
        inter_mask = ~intra_mask
        
        intra_dists = dists[intra_mask]
        inter_dists = dists[inter_mask]
        
        # Compute regret within community
        if len(intra_dists) > 0:
            if len(intra_dists) >= k_intra:
                kth_intra = np.partition(intra_dists, min(k_intra, len(intra_dists)-1))[min(k_intra, len(intra_dists)-1)]
            else:
                kth_intra = np.max(intra_dists) if len(intra_dists) > 0 else stats['mean_dist']
            
            intra_indices = np.where(intra_mask)[0]
            for idx, j in enumerate(intra_indices):
                if intra_dists[idx] > kth_intra:
                    r = (intra_dists[idx] - kth_intra) / (kth_intra + 1e-8)
                    regret[i, j] = 1 / (1 + r)
        
        # Compute regret between communities
        if len(inter_dists) > 0:
            if len(inter_dists) >= k_inter:
                kth_inter = np.partition(inter_dists, min(k_inter, len(inter_dists)-1))[min(k_inter, len(inter_dists)-1)]
            else:
                kth_inter = np.max(inter_dists) if len(inter_dists) > 0 else stats['mean_dist']
            
            inter_indices = np.where(inter_mask)[0]
            for idx, j in enumerate(inter_indices):
                if inter_dists[idx] > kth_inter:
                    r = (inter_dists[idx] - kth_inter) / (kth_inter + 1e-8)
                    # Less penalty for inter-community edges
                    regret[i, j] = 1 / (1 + 0.5 * r)
    
    # Symmetrize
    regret = np.sqrt(regret * regret.T)
    np.fill_diagonal(regret, 0)
    return regret

def _compute_community_mst(distance_matrix: np.ndarray, 
                         stats: Dict, 
                         communities: np.ndarray) -> np.ndarray:
    """Compute MST scores with community awareness."""
    n = stats['n']
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            # Compute global MST
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Boost edges in MST
            base_boost = 1.5
            
            for i in range(n):
                for j in range(i+1, n):
                    if mst_present[i, j]:
                        # Extra boost for intra-community MST edges
                        if communities[i] == communities[j]:
                            boost = base_boost * (1.0 + 0.2 * min(stats['community_ratio'], 2.0))
                        else:
                            # Inter-community MST edges are crucial
                            boost = base_boost * 1.3
                        mst_scores[i, j] = mst_scores[j, i] = boost
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores

def _compute_connectivity_scores(distance_matrix: np.ndarray,
                               stats: Dict,
                               communities: np.ndarray) -> np.ndarray:
    """Compute scores based on node connectivity within and between communities."""
    n = stats['n']
    n_communities = stats['n_communities']
    
    connectivity = np.ones((n, n))
    
    # Compute community centers
    community_centers = {}
    for c in range(n_communities):
        nodes = np.where(communities == c)[0]
        if len(nodes) > 0:
            # Use node with minimum average distance to others in community as center
            submatrix = distance_matrix[nodes][:, nodes]
            center_idx = nodes[np.argmin(np.mean(submatrix, axis=1))]
            community_centers[c] = center_idx
    
    # For each edge, compute connectivity score
    for i in range(n):
        ci = communities[i]
        for j in range(i+1, n):
            cj = communities[j]
            
            if ci == cj:
                # Intra-community: prefer edges that connect to community center
                if ci in community_centers:
                    center = community_centers[ci]
                    if i == center or j == center:
                        connectivity[i, j] = 1.2  # Boost connection to center
                    else:
                        # Check if edge connects two peripheral nodes
                        dist_to_center_i = distance_matrix[i, center]
                        dist_to_center_j = distance_matrix[j, center]
                        avg_dist_to_center = (dist_to_center_i + dist_to_center_j) / 2
                        
                        # Penalize edges between peripheral nodes that are far from center
                        if avg_dist_to_center > stats['intra_mean'] * 1.5:
                            connectivity[i, j] = 0.8
            else:
                # Inter-community: prefer edges that are bridges
                # Check if this is one of the best connections between these communities
                bridge_quality = 1.0
                
                # Find minimum distance between these communities
                nodes_i = np.where(communities == ci)[0]
                nodes_j = np.where(communities == cj)[0]
                
                if len(nodes_i) > 0 and len(nodes_j) > 0:
                    cross_dists = distance_matrix[nodes_i][:, nodes_j]
                    min_cross_dist = np.min(cross_dists)
                    
                    if abs(distance_matrix[i, j] - min_cross_dist) < stats['mean_dist'] * 0.1:
                        # This is among the best bridges
                        bridge_quality = 1.3
                
                connectivity[i, j] = bridge_quality
            
            connectivity[j, i] = connectivity[i, j]
    
    return connectivity

def _compute_bandit_lookahead(distance_matrix: np.ndarray,
                            base_factors: Dict[str, np.ndarray],
                            stats: Dict,
                            communities: np.ndarray) -> np.ndarray:
    """Bandit-based lookahead simulations with adaptive exploration."""
    n = stats['n']
    lookahead = np.ones((n, n))
    
    if n < 4:
        return lookahead
    
    # Adaptive simulation parameters
    n_simulations = min(40, max(8, int(15 * np.log2(n) / (1 + stats['spectral_gap']))))
    max_steps = min(8, max(3, n // 4))
    
    # Select candidate edges for simulation
    candidate_edges = _select_bandit_candidates(distance_matrix, base_factors, stats, communities)
    
    # Create policy heuristic for bandit
    policy_heuristic = _create_bandit_policy(distance_matrix, base_factors, stats, communities)
    
    for i, j in candidate_edges:
        if i >= j:
            continue
            
        scores = []
        for sim in range(n_simulations):
            score = _bandit_simulation(distance_matrix, i, j, policy_heuristic,
                                     stats, communities, max_steps, sim)
            scores.append(score)
        
        if scores:
            # Use winsorized mean to handle outliers
            scores_sorted = np.sort(scores)
            trim = max(1, len(scores) // 5)
            lookahead_score = np.mean(scores_sorted[trim:-trim]) if len(scores_sorted) > 2*trim else np.median(scores)
            lookahead[i, j] = lookahead[j, i] = lookahead_score
    
    # Normalize
    max_val = np.max(lookahead)
    if max_val > 0:
        lookahead = lookahead / max_val
    
    np.fill_diagonal(lookahead, 0)
    return lookahead

def _select_bandit_candidates(distance_matrix: np.ndarray,
                            base_factors: Dict[str, np.ndarray],
                            stats: Dict,
                            communities: np.ndarray) -> List[Tuple[int, int]]:
    """Select candidate edges for bandit simulations."""
    n = stats['n']
    candidates = []
    n_communities = stats['n_communities']
    
    # Base number of candidates per node
    base_k = min(10, max(3, int(3 * np.log2(n))))
    
    # Adjust based on community structure
    if n_communities > 1:
        # More candidates for inter-community edges
        base_k = int(base_k * 1.2)
    
    for i in range(n):
        # Compute quick score for each potential edge
        quick_scores = np.ones(n) * -np.inf
        for j in range(n):
            if i != j:
                quick_scores[j] = (
                    base_factors['inv_dist'][i, j] ** 0.4 *
                    base_factors['regret'][i, j] ** 0.3 *
                    base_factors['connectivity'][i, j] ** 0.3
                )
        
        # Select top candidates
        k = min(base_k, n-1)
        if k < n-1:
            threshold = np.partition(quick_scores, -k)[-k]
            indices = np.where(quick_scores >= threshold)[0]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        for j in indices:
            if i < j:
                candidates.append((i, j))
    
    # Ensure representation of inter-community bridges
    if n_communities > 1:
        bridges = _find_community_bridges(distance_matrix, communities)
        for i, j in bridges:
            if (i, j) not in candidates and i < j:
                candidates.append((i, j))
    
    return list(set(candidates))

def _find_community_bridges(distance_matrix: np.ndarray,
                          communities: np.ndarray) -> List[Tuple[int, int]]:
    """Find important bridges between communities."""
    n = distance_matrix.shape[0]
    n_communities = len(np.unique(communities))
    bridges = []
    
    if n_communities <= 1:
        return bridges
    
    # For each pair of communities, find the best connection
    for c1 in range(n_communities):
        for c2 in range(c1+1, n_communities):
            nodes1 = np.where(communities == c1)[0]
            nodes2 = np.where(communities == c2)[0]
            
            if len(nodes1) == 0 or len(nodes2) == 0:
                continue
            
            # Find minimum distance between communities
            min_dist = np.inf
            best_pair = (nodes1[0], nodes2[0])
            
            for i in nodes1:
                for j in nodes2:
                    if distance_matrix[i, j] < min_dist:
                        min_dist = distance_matrix[i, j]
                        best_pair = (i, j)
            
            bridges.append(best_pair)
            
            # Also add a few additional good connections
            cross_dists = []
            for i in nodes1:
                for j in nodes2:
                    cross_dists.append((distance_matrix[i, j], i, j))
            
            cross_dists.sort()
            # Take up to 3 best connections per community pair
            for _, i, j in cross_dists[:3]:
                if (i, j) != best_pair:
                    bridges.append((i, j))
    
    return bridges

def _create_bandit_policy(distance_matrix: np.ndarray,
                         base_factors: Dict[str, np.ndarray],
                         stats: Dict,
                         communities: np.ndarray) -> np.ndarray:
    """Create heuristic matrix for bandit policy."""
    n = stats['n']
    policy = np.ones((n, n))
    
    for i in range(n):
        for j in range(i+1, n):
            # Combine factors with community-aware weights
            ci, cj = communities[i], communities[j]
            
            if ci == cj:
                # Intra-community: emphasize connectivity and inverse distance
                score = (
                    base_factors['inv_dist'][i, j] ** 0.5 *
                    base_factors['connectivity'][i, j] ** 0.3 *
                    base_factors['regret'][i, j] ** 0.2
                )
            else:
                # Inter-community: emphasize MST and regret
                score = (
                    base_factors['mst'][i, j] ** 0.4 *
                    base_factors['regret'][i, j] ** 0.3 *
                    base_factors['inv_dist'][i, j] ** 0.3
                )
            
            policy[i, j] = policy[j, i] = score
    
    np.fill_diagonal(policy, 0)
    return policy

def _bandit_simulation(distance_matrix: np.ndarray,
                      start_i: int, start_j: int,
                      policy_heuristic: np.ndarray,
                      stats: Dict,
                      communities: np.ndarray,
                      max_steps: int,
                      sim_id: int) -> float:
    """Bandit simulation using UCB-like selection strategy."""
    n = stats['n']
    visited = np.zeros(n, dtype=bool)
    visited[start_i] = visited[start_j] = True
    
    current = start_j
    tour_length = distance_matrix[start_i, start_j]
    
    # Track visit counts for UCB
    visit_counts = np.zeros(n)
    visit_counts[start_i] = 1
    visit_counts[start_j] = 1
    
    # Community completion tracking
    community_visited = {}
    for c in np.unique(communities):
        community_visited[c] = 0
    
    community_visited[communities[start_i]] += 1
    community_visited[communities[start_j]] += 1
    
    for step in range(max_steps):
        unvisited = np.where(~visited)[0]
        if len(unvisited) == 0:
            break
        
        # UCB scores for each unvisited node
        ucb_scores = []
        for v in unvisited:
            # Exploitation: policy heuristic value
            exploit = policy_heuristic[current, v]
            
            # Exploration: encourage trying less-visited nodes
            # Also encourage completing current community
            cv = communities[v]
            community_completion = community_visited[cv] / np.sum(communities == cv)
            
            # Adaptive exploration parameter
            exploration_factor = np.sqrt(2 * np.log(step + 2) / (visit_counts[v] + 1e-8))
            
            # Community-aware bonus
            community_bonus = 1.0
            if cv == communities[current]:
                # Same community: bonus if we haven't completed it
                if community_completion < 0.8:
                    community_bonus = 1.2
            else:
                # Different community: bonus if current community is mostly complete
                current_community = communities[current]
                if community_visited[current_community] / np.sum(communities == current_community) > 0.7:
                    community_bonus = 1.1
            
            ucb_score = exploit * exploration_factor * community_bonus
            ucb_scores.append(ucb_score)
        
        ucb_scores = np.array(ucb_scores)
        
        # Selection: sometimes random for exploration
        if np.random.random() < 0.2:  # 20% random exploration
            next_idx = np.random.randint(len(unvisited))
        else:
            # Softmax selection with temperature
            temperature = max(0.2, 1.0 - 0.1 * step)
            exp_scores = np.exp(ucb_scores / temperature)
            probs = exp_scores / (np.sum(exp_scores) + 1e-8)
            next_idx = np.random.choice(len(unvisited), p=probs)
        
        next_node = unvisited[next_idx]
        tour_length += distance_matrix[current, next_node]
        
        # Update tracking
        visited[next_node] = True
        visit_counts[next_node] += 1
        community_visited[communities[next_node]] += 1
        current = next_node
    
    # Score based on efficiency and community completion
    expected_length = max_steps * stats['mean_dist']
    efficiency = expected_length / (tour_length + 1e-8)
    
    # Bonus for completing communities
    completion_bonus = 1.0
    total_completion = sum(community_visited.values()) / n
    if total_completion > 0.7:
        completion_bonus = 1.0 + 0.3 * total_completion
    
    final_score = efficiency * completion_bonus
    return min(final_score, 3.0)

def _compute_spectral_scores(distance_matrix: np.ndarray,
                           stats: Dict,
                           communities: np.ndarray) -> np.ndarray:
    """Compute scores based on spectral graph properties."""
    n = stats['n']
    spectral_scores = np.ones((n, n))
    
    if n < 4:
        return spectral_scores
    
    # Convert distances to similarities for spectral analysis
    sigma = stats['mean_dist'] * 0.7
    similarity = np.exp(-distance_matrix ** 2 / (2 * sigma ** 2))
    np.fill_diagonal(similarity, 0)
    
    # Compute Fiedler vector (second smallest eigenvector of Laplacian)
    try:
        D = np.diag(np.sum(similarity, axis=1))
        L = D - similarity
        
        # Compute a few eigenvectors
        eigenvalues, eigenvectors = np.linalg.eigh(L)
        idx = np.argsort(eigenvalues)
        
        # Fiedler vector
        if len(eigenvalues) > 1:
            fiedler = eigenvectors[:, idx[1]]
            
            # Score edges based on Fiedler vector values
            # Edges connecting nodes with similar Fiedler values are good
            for i in range(n):
                for j in range(i+1, n):
                    fiedler_diff = abs(fiedler[i] - fiedler[j])
                    # Convert to similarity (small diff = high score)
                    spectral_scores[i, j] = spectral_scores[j, i] = np.exp(-fiedler_diff ** 2 * 10)
    except:
        # Fallback if eigendecomposition fails
        pass
    
    # Adjust based on algebraic connectivity
    ac = stats['algebraic_connectivity']
    if ac > 0.5:
        # Well-connected graph: trust spectral scores more
        spectral_scores = spectral_scores ** 1.2
    else:
        # Poorly connected: reduce reliance on spectral scores
        spectral_scores = spectral_scores ** 0.8
    
    return spectral_scores

def _compute_adaptive_fusion_weights(distance_matrix: np.ndarray,
                                   base_factors: Dict[str, np.ndarray],
                                   stats: Dict,
                                   communities: np.ndarray) -> Dict[str, float]:
    """Compute adaptive weights using entropy balancing and instance characteristics."""
    n = stats['n']
    cv = stats['cv_dist']
    spectral_gap = stats['spectral_gap']
    n_communities = stats['n_communities']
    community_ratio = stats['community_ratio']
    
    # Base weights influenced by multiple factors
    weights = {
        'w_inv': 1.0,
        'w_regret': 0.3 + 0.3 * cv,
        'w_mst': 0.4 + 0.2 * min(spectral_gap * 5, 1.0),
        'w_connectivity': 0.2 + 0.4 * (n_communities / min(n, 10)),
        'w_lookahead': 0.2 + 0.3 * (1 - community_ratio),
        'w_spectral': 0.1 + 0.2 * spectral_gap
    }
    
    # Adjust based on instance size
    size_factor = min(1.0, n / 50)
    for key in weights:
        if key != 'w_inv':
            weights[key] *= (0.8 + 0.4 * size_factor)
    
    # Normalize using entropy balancing
    # Ensure weights sum to a constant while maximizing entropy
    target_sum = 6.0
    current_sum = sum(weights.values())
    if current_sum > 0:
        # Scale to target sum
        for key in weights:
            weights[key] = weights[key] / current_sum * target_sum
    
    return weights

def _integrate_factors_with_communities(distance_matrix: np.ndarray,
                                      base_factors: Dict[str, np.ndarray],
                                      lookahead_scores: np.ndarray,
                                      spectral_scores: np.ndarray,
                                      weights: Dict[str, float],
                                      stats: Dict,
                                      communities: np.ndarray) -> np.ndarray:
    """Integrate all factors with community-aware combination."""
    n = stats['n']
    
    # Build community-aware candidate mask
    candidate_mask = _build_community_candidate_mask(distance_matrix, base_factors, stats, communities)
    
    # Initialize result matrix
    H = np.zeros((n, n))
    
    # Multiplicative integration with community awareness
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Get community-specific adjustment
                ci, cj = communities[i], communities[j]
                community_adjustment = 1.0
                
                if ci == cj:
                    # Intra-community: slight boost
                    community_adjustment = 1.1
                else:
                    # Inter-community: check if this is a good bridge
                    if _is_good_bridge(i, j, distance_matrix, communities, stats):
                        community_adjustment = 1.2
                
                combined = (
                    base_factors['inv_dist'][i, j] ** weights['w_inv'] *
                    base_factors['regret'][i, j] ** weights['w_regret'] *
                    base_factors['mst'][i, j] ** weights['w_mst'] *
                    base_factors['connectivity'][i, j] ** weights['w_connectivity'] *
                    lookahead_scores[i, j] ** weights['w_lookahead'] *
                    spectral_scores[i, j] ** weights['w_spectral'] *
                    community_adjustment
                )
                H[i, j] = H[j, i] = combined
    
    # Ensure connectivity within communities
    H = _ensure_community_connectivity(H, distance_matrix, stats, communities)
    
    return H

def _build_community_candidate_mask(distance_matrix: np.ndarray,
                                  base_factors: Dict[str, np.ndarray],
                                  stats: Dict,
                                  communities: np.ndarray) -> np.ndarray:
    """Build candidate mask considering community structure."""
    n = stats['n']
    n_communities = stats['n_communities']
    
    # Base candidate count
    base_k = min(12, max(4, int(3 * np.log2(n))))
    
    # Adjust for community structure
    if n_communities > 1:
        # More candidates for inter-community connections
        base_k = int(base_k * 1.3)
    
    # Initialize mask
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # Quick scoring for candidate selection
    quick_scores = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            quick_scores[i, j] = quick_scores[j, i] = (
                base_factors['inv_dist'][i, j] ** 0.5 *
                base_factors['mst'][i, j] ** 0.3 *
                base_factors['connectivity'][i, j] ** 0.2
            )
    
    # Select candidates per node
    for i in range(n):
        scores = quick_scores[i].copy()
        scores[i] = -np.inf
        
        k = min(base_k, n-1)
        if k < n-1:
            threshold = np.partition(scores, -k)[-k]
            candidates = np.where(scores >= threshold)[0]
        else:
            candidates = np.arange(n)
            candidates = candidates[candidates != i]
        
        candidate_mask[i, candidates] = True
    
    # Ensure inter-community connectivity
    if n_communities > 1:
        for c1 in range(n_communities):
            for c2 in range(c1+1, n_communities):
                nodes1 = np.where(communities == c1)[0]
                nodes2 = np.where(communities == c2)[0]
                
                if len(nodes1) > 0 and len(nodes2) > 0:
                    # Ensure at least one connection between communities
                    best_i, best_j = nodes1[0], nodes2[0]
                    best_score = -np.inf
                    
                    for i in nodes1:
                        for j in nodes2:
                            if quick_scores[i, j] > best_score:
                                best_score = quick_scores[i, j]
                                best_i, best_j = i, j
                    
                    candidate_mask[best_i, best_j] = True
                    candidate_mask[best_j, best_i] = True
    
    np.fill_diagonal(candidate_mask, False)
    return candidate_mask

def _is_good_bridge(i: int, j: int,
                   distance_matrix: np.ndarray,
                   communities: np.ndarray,
                   stats: Dict) -> bool:
    """Check if edge (i,j) is a good bridge between communities."""
    ci, cj = communities[i], communities[j]
    
    if ci == cj:
        return False
    
    # Get all nodes in the two communities
    nodes_i = np.where(communities == ci)[0]
    nodes_j = np.where(communities == cj)[0]
    
    # Check if this is among the top 3 shortest connections
    cross_dists = []
    for ii in nodes_i:
        for jj in nodes_j:
            cross_dists.append((distance_matrix[ii, jj], ii, jj))
    
    cross_dists.sort()
    
    # Check if (i,j) is in top 3
    for idx in range(min(3, len(cross_dists))):
        if cross_dists[idx][1] == i and cross_dists[idx][2] == j:
            return True
    
    return False

def _ensure_community_connectivity(H: np.ndarray,
                                 distance_matrix: np.ndarray,
                                 stats: Dict,
                                 communities: np.ndarray) -> np.ndarray:
    """Ensure each community has sufficient internal connectivity."""
    n = stats['n']
    n_communities = stats['n_communities']
    
    for c in range(n_communities):
        nodes = np.where(communities == c)[0]
        if len(nodes) <= 1:
            continue
        
        # Ensure each node in community has at least 2 connections within community
        for i in nodes:
            intra_connections = 0
            for j in nodes:
                if i != j and H[i, j] > 0:
                    intra_connections += 1
            
            if intra_connections < min(2, len(nodes)-1):
                # Add connections to nearest neighbors in same community
                dists = distance_matrix[i, nodes].copy()
                dists[nodes == i] = np.inf
                
                needed = min(2, len(nodes)-1) - intra_connections
                nearest = np.argsort(dists)[:needed]
                
                for idx in nearest:
                    j = nodes[idx]
                    if H[i, j] == 0:
                        base_score = 1 / (distance_matrix[i, j] + 1e-8)
                        H[i, j] = H[j, i] = base_score
    
    return H

def _post_process_with_communities(H: np.ndarray,
                                 distance_matrix: np.ndarray,
                                 stats: Dict,
                                 communities: np.ndarray) -> np.ndarray:
    """Final post-processing with community preservation."""
    n = stats['n']
    
    # Ensure symmetry and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Community-aware sparsification
    if n > 15:
        H = _community_sparsify(H, distance_matrix, stats, communities)
    
    # Enhance community bridges
    H = _enhance_community_bridges(H, distance_matrix, stats, communities)
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    # Ensure minimum connectivity
    for i in range(n):
        if np.sum(H[i] > 0) < 2:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:2]
            for j in nearest:
                if H[i, j] == 0:
                    H[i, j] = H[j, i] = 1 / (distance_matrix[i, j] + 1e-8)
    
    return H

def _community_sparsify(H: np.ndarray,
                       distance_matrix: np.ndarray,
                       stats: Dict,
                       communities: np.ndarray) -> np.ndarray:
    """Sparsify heuristic matrix while preserving community structure."""
    n = stats['n']
    n_communities = stats['n_communities']
    
    H_sparse = H.copy()
    
    # Different keep counts for intra vs inter community
    intra_keep = min(6, max(3, int(2 * np.log2(n))))
    inter_keep = min(4, max(2, int(1.5 * np.log2(n))))
    
    for i in range(n):
        ci = communities[i]
        
        # Separate intra and inter community scores
        intra_scores = []
        inter_scores = []
        intra_indices = []
        inter_indices = []
        
        for j in range(n):
            if i != j:
                if communities[j] == ci:
                    intra_scores.append(H[i, j])
                    intra_indices.append(j)
                else:
                    inter_scores.append(H[i, j])
                    inter_indices.append(j)
        
        # Keep top intra-community edges
        if len(intra_scores) > intra_keep:
            intra_scores_arr = np.array(intra_scores)
            intra_threshold = np.partition(intra_scores_arr, -intra_keep)[-intra_keep]
            
            for idx, score in enumerate(intra_scores):
                if score < intra_threshold:
                    j = intra_indices[idx]
                    H_sparse[i, j] = H_sparse[j, i] = 0
        
        # Keep top inter-community edges
        if len(inter_scores) > inter_keep:
            inter_scores_arr = np.array(inter_scores)
            inter_threshold = np.partition(inter_scores_arr, -inter_keep)[-inter_keep]
            
            for idx, score in enumerate(inter_scores):
                if score < inter_threshold:
                    j = inter_indices[idx]
                    H_sparse[i, j] = H_sparse[j, i] = 0
    
    # Ensure MST edges are preserved
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            for i in range(n):
                for j in range(i+1, n):
                    if mst_present[i, j] and H_sparse[i, j] == 0:
                        # Restore with moderate value
                        median_val = np.median(H[H > 0]) if np.any(H > 0) else 0.3
                        H_sparse[i, j] = H_sparse[j, i] = median_val
        except:
            pass
    
    return H_sparse

def _enhance_community_bridges(H: np.ndarray,
                             distance_matrix: np.ndarray,
                             stats: Dict,
                             communities: np.ndarray) -> np.ndarray:
    """Enhance scores for important community bridges."""
    n = stats['n']
    n_communities = stats['n_communities']
    
    if n_communities <= 1:
        return H
    
    H_enhanced = H.copy()
    
    # Find and enhance important bridges
    for c1 in range(n_communities):
        for c2 in range(c1+1, n_communities):
            nodes1 = np.where(communities == c1)[0]
            nodes2 = np.where(communities == c2)[0]
            
            if len(nodes1) == 0 or len(nodes2) == 0:
                continue
            
            # Find top 2 bridges between these communities
            bridge_scores = []
            for i in nodes1:
                for j in nodes2:
                    if H[i, j] > 0:
                        bridge_scores.append((H[i, j], i, j))
            
            bridge_scores.sort(reverse=True)
            
            # Enhance top bridges
            for idx, (score, i, j) in enumerate(bridge_scores[:2]):
                enhancement = 1.0 + 0.2 * (1 - idx * 0.3)  # 20% for best, 14% for second
                H_enhanced[i, j] = H_enhanced[j, i] = score * enhancement
    
    return H_enhanced
