import numpy as np
from typing import Dict, Tuple, List

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristic with graph-theoretic clustering, hybrid regret ensemble,
    and adaptive contextual parameter tuning.
    Returns a heuristic matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Advanced instance characterization with graph metrics
    stats = _analyze_instance_v3(distance_matrix)
    
    # Step 2: Graph-based candidate selection with clustering coefficients
    candidate_mask, graph_properties = _build_graph_candidate_mask_v3(distance_matrix, stats)
    
    # Step 3: Compute hybrid regret ensemble scores
    regret_ensemble = _compute_hybrid_regret_ensemble_v3(
        distance_matrix, stats, candidate_mask, graph_properties
    )
    
    # Step 4: Compute base heuristic components
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    # Step 5: Context-aware edge scoring with adaptive ensemble weighting
    H = np.zeros((n, n))
    ensemble_weights = _compute_ensemble_weights_v3(stats, graph_properties)
    
    for i in range(n):
        for j in range(i + 1, n):
            if candidate_mask[i, j]:
                # Base components
                base_score = inv_dist[i, j] ** ensemble_weights['w_inv']
                
                # Hybrid regret ensemble with dynamic weighting
                regret_components = [
                    regret_ensemble['direct_regret'][i, j] ** ensemble_weights['w_regret'],
                    regret_ensemble['clustering_regret'][i, j] ** ensemble_weights['w_cluster'],
                    regret_ensemble['path_diversity'][i, j] ** ensemble_weights['w_diversity']
                ]
                
                # Weighted geometric mean of regret components
                regret_score = np.prod([
                    comp ** (ensemble_weights['w_regret_group'] / len(regret_components))
                    for comp in regret_components
                ])
                
                # Local clustering synergy bonus
                if graph_properties['local_clustering'][i] > 0.7 and graph_properties['local_clustering'][j] > 0.7:
                    cluster_bonus = 1.0 + ensemble_weights['w_cluster_bonus'] * (
                        graph_properties['local_clustering'][i] + graph_properties['local_clustering'][j]
                    ) / 2
                else:
                    cluster_bonus = 1.0
                
                combined = base_score * regret_score * cluster_bonus
                H[i, j] = H[j, i] = combined
    
    # Step 6: Robust post-processing with connectivity preservation
    H = _post_process_v3(H, distance_matrix, stats, candidate_mask, graph_properties)
    
    return H


def _analyze_instance_v3(distance_matrix: np.ndarray) -> Dict:
    """Advanced instance analysis with geometric and statistical properties."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal for statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n - 1)
    
    # Statistical properties
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Geometric properties
    quartiles = np.percentile(dist_no_diag, [25, 50, 75])
    iqr = quartiles[2] - quartiles[0]
    
    # Skewness and kurtosis
    skewness = np.mean(((dist_no_diag - mean_dist) / (std_dist + 1e-8)) ** 3)
    kurtosis = np.mean(((dist_no_diag - mean_dist) / (std_dist + 1e-8)) ** 4) - 3
    
    stats = {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'q1': quartiles[0],
        'median': quartiles[1],
        'q3': quartiles[2],
        'iqr': iqr,
        'skewness': abs(skewness),
        'kurtosis': kurtosis,
        'sparsity': _compute_sparsity_index(distance_matrix),
        'dimension_estimate': _estimate_intrinsic_dimension(distance_matrix)
    }
    
    return stats


def _compute_sparsity_index(distance_matrix: np.ndarray) -> float:
    """Measure how spread out points are based on distance distribution."""
    n = distance_matrix.shape[0]
    if n < 4:
        return 0.5
    
    # Compare nearest neighbor distances to overall distribution
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    
    global_median = np.median(distance_matrix[~np.eye(n, dtype=bool)])
    nn_median = np.median(nn_dists)
    
    # Ratio indicates sparsity (higher = more sparse)
    sparsity = nn_median / (global_median + 1e-8)
    return np.clip(sparsity, 0.5, 2.0)


def _estimate_intrinsic_dimension(distance_matrix: np.ndarray) -> float:
    """Rough estimate of intrinsic dimension using distance ratios."""
    n = distance_matrix.shape[0]
    if n < 5:
        return 2.0  # Default assumption
    
    # Use ratio of distances to estimate dimensionality
    dim_estimates = []
    for i in range(min(10, n)):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)
        
        if len(sorted_dists) >= 3:
            # Ratio-based dimension estimation
            r1, r2 = sorted_dists[1], sorted_dists[2]
            if r1 > 0 and r2 > 0:
                dim_est = np.log(2) / np.log(r2 / r1)
                dim_estimates.append(dim_est)
    
    if dim_estimates:
        return np.clip(np.median(dim_estimates), 1.0, 10.0)
    return 2.0


def _build_graph_candidate_mask_v3(distance_matrix: np.ndarray, 
                                   stats: Dict) -> Tuple[np.ndarray, Dict]:
    """Build candidate mask using graph-theoretic properties and clustering coefficients."""
    n = stats['n']
    
    # Compute local clustering coefficients
    local_clustering = _compute_local_clustering(distance_matrix, stats)
    
    # Adaptive neighborhood size based on multiple factors
    base_k = min(12, max(4, int(np.log2(n) * 2.5)))
    
    # Adjust based on local clustering and sparsity
    clustering_factor = 1.0 + np.mean(local_clustering) * 0.5
    sparsity_factor = 1.5 - min(1.0, stats['sparsity'] / 2.0)
    
    k_adaptive = int(base_k * clustering_factor * sparsity_factor)
    k_adaptive = min(k_adaptive, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # Build candidate set with clustering-aware selection
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Primary candidates: k-nearest neighbors
        if k_adaptive < len(dists):
            primary_indices = np.argpartition(dists, k_adaptive)[:k_adaptive]
        else:
            primary_indices = np.arange(n)
            primary_indices = primary_indices[primary_indices != i]
        
        # Secondary candidates: high clustering coefficient neighbors
        clustering_threshold = np.percentile(local_clustering, 70)
        if local_clustering[i] > clustering_threshold:
            # Include neighbors of neighbors for clustered regions
            for neighbor in primary_indices:
                if distance_matrix[i, neighbor] < stats['q3']:
                    # Add this neighbor's best connections
                    neighbor_dists = distance_matrix[neighbor].copy()
                    neighbor_dists[neighbor] = np.inf
                    secondary_k = min(3, n - 1)
                    if secondary_k < len(neighbor_dists):
                        secondary_indices = np.argpartition(neighbor_dists, secondary_k)[:secondary_k]
                        for sec in secondary_indices:
                            if sec != i:
                                candidate_mask[i, sec] = True
        
        candidate_mask[i, primary_indices] = True
    
    # Make symmetric and ensure connectivity
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    # Ensure minimum connectivity
    min_edges = min(3, n - 1)
    for i in range(n):
        if np.sum(candidate_mask[i]) < min_edges:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            additional = np.argsort(dists)[:min_edges]
            candidate_mask[i, additional] = True
            candidate_mask[additional, i] = True
    
    graph_properties = {
        'local_clustering': local_clustering,
        'avg_clustering': np.mean(local_clustering),
        'k_used': k_adaptive
    }
    
    return candidate_mask, graph_properties


def _compute_local_clustering(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute local clustering coefficients based on distance thresholds."""
    n = stats['n']
    clustering_coeffs = np.zeros(n)
    
    # Adaptive threshold based on distance distribution
    threshold = stats['q2']  # Median distance
    
    for i in range(n):
        # Find neighbors within threshold
        dists = distance_matrix[i]
        neighbors = np.where((dists > 0) & (dists < threshold))[0]
        
        if len(neighbors) < 2:
            clustering_coeffs[i] = 0.5  # Default for isolated nodes
            continue
        
        # Count triangles among neighbors
        triangle_count = 0
        possible_triangles = len(neighbors) * (len(neighbors) - 1) / 2
        
        for idx_j, j in enumerate(neighbors):
            for k in neighbors[idx_j + 1:]:
                if distance_matrix[j, k] < threshold:
                    triangle_count += 1
        
        clustering_coeffs[i] = triangle_count / (possible_triangles + 1e-8)
    
    return np.clip(clustering_coeffs, 0.0, 1.0)


def _compute_hybrid_regret_ensemble_v3(distance_matrix: np.ndarray, 
                                        stats: Dict, 
                                        candidate_mask: np.ndarray,
                                        graph_properties: Dict) -> Dict:
    """
    Compute multiple regret-based scores and combine into an ensemble.
    """
    n = stats['n']
    
    # Initialize score matrices
    direct_regret = np.ones((n, n))
    clustering_regret = np.ones((n, n))
    path_diversity = np.ones((n, n))
    
    # Adaptive parameters
    max_hops = min(3, max(2, int(3.5 - stats['cv_dist'])))
    clustering_weight = graph_properties['avg_clustering']
    
    for i in range(n):
        for j in range(i + 1, n):
            if not candidate_mask[i, j]:
                direct_regret[i, j] = direct_regret[j, i] = 0
                clustering_regret[i, j] = clustering_regret[j, i] = 0
                path_diversity[i, j] = path_diversity[j, i] = 0
                continue
            
            direct_cost = distance_matrix[i, j]
            
            # 1. Direct regret with multi-hop alternatives
            alt_costs = []
            for k in range(n):
                if k == i or k == j:
                    continue
                
                # 2-hop alternative
                if candidate_mask[i, k] and candidate_mask[k, j]:
                    alt_costs.append(distance_matrix[i, k] + distance_matrix[k, j])
                
                # 3-hop alternative if allowed
                if max_hops >= 3:
                    for l in range(n):
                        if l in (i, j, k):
                            continue
                        if (candidate_mask[i, k] and candidate_mask[k, l] and 
                            candidate_mask[l, j]):
                            alt_costs.append(
                                distance_matrix[i, k] + 
                                distance_matrix[k, l] + 
                                distance_matrix[l, j]
                            )
            
            # Compute direct regret score
            if alt_costs:
                best_alt = min(alt_costs)
                regret_ratio = best_alt / (direct_cost + 1e-8)
                
                # Adaptive scoring based on instance characteristics
                if regret_ratio >= 1.0:
                    score = 1.0 + (regret_ratio - 1.0) * (1.0 - stats['cv_dist'])
                else:
                    penalty = 0.3 + 0.7 * stats['regularity'] if 'regularity' in stats else 0.5
                    score = 1.0 - (1.0 - regret_ratio) * penalty
                
                direct_regret[i, j] = direct_regret[j, i] = np.clip(score, 0.1, 3.0)
            
            # 2. Clustering-aware regret
            if graph_properties['local_clustering'][i] > 0.6 or graph_properties['local_clustering'][j] > 0.6:
                # For clustered nodes, consider intra-cluster alternatives
                cluster_nodes = []
                for k in range(n):
                    if (k != i and k != j and 
                        (distance_matrix[i, k] < stats['q2'] or 
                         distance_matrix[j, k] < stats['q2'])):
                        cluster_nodes.append(k)
                
                if len(cluster_nodes) >= 2:
                    cluster_alts = []
                    for k in cluster_nodes:
                        if candidate_mask[i, k] and candidate_mask[k, j]:
                            cluster_alts.append(distance_matrix[i, k] + distance_matrix[k, j])
                    
                    if cluster_alts:
                        best_cluster_alt = min(cluster_alts)
                        cluster_ratio = best_cluster_alt / (direct_cost + 1e-8)
                        
                        # Bonus for edges that are best within their cluster
                        if cluster_ratio > 1.5:
                            clustering_regret[i, j] = clustering_regret[j, i] = 1.5
                        else:
                            clustering_regret[i, j] = clustering_regret[j, i] = 1.0
            
            # 3. Path diversity score
            # Count number of distinct good alternative paths
            good_paths = 0
            for k in range(n):
                if k == i or k == j:
                    continue
                if (candidate_mask[i, k] and candidate_mask[k, j] and
                    distance_matrix[i, k] + distance_matrix[k, j] < direct_cost * 1.5):
                    good_paths += 1
            
            # Diversity bonus: edges with few alternatives are more unique/important
            diversity_score = 1.0 + (3.0 - min(good_paths, 3)) * 0.2
            path_diversity[i, j] = path_diversity[j, i] = diversity_score
    
    # Symmetrize all matrices
    for matrix in [direct_regret, clustering_regret, path_diversity]:
        matrix = (matrix + matrix.T) / 2
        np.fill_diagonal(matrix, 0)
    
    return {
        'direct_regret': direct_regret,
        'clustering_regret': clustering_regret,
        'path_diversity': path_diversity
    }


def _compute_ensemble_weights_v3(stats: Dict, graph_properties: Dict) -> Dict:
    """Compute adaptive weights for ensemble components based on instance context."""
    n = stats['n']
    
    # Base weights influenced by instance characteristics
    weights = {
        'w_inv': 1.0,  # Inverse distance weight
        
        # Regret group weight adapts to instance regularity
        'w_regret_group': 0.6 + 0.4 * (1.0 - stats['cv_dist']),
        
        # Individual regret component weights
        'w_regret': 0.8,
        'w_cluster': 0.5 + 0.5 * graph_properties['avg_clustering'],
        'w_diversity': 0.7 - 0.3 * min(1.0, stats['sparsity']),
        
        # Clustering synergy bonus
        'w_cluster_bonus': 0.2 * graph_properties['avg_clustering']
    }
    
    # Adjust based on instance size
    size_factor = min(1.0, n / 50.0)
    weights['w_regret_group'] *= 0.8 + 0.2 * size_factor
    
    # Normalize the regret component weights to sum to w_regret_group
    regret_components = ['w_regret', 'w_cluster', 'w_diversity']
    component_sum = sum(weights[key] for key in regret_components)
    if component_sum > 0:
        scale = weights['w_regret_group'] / component_sum
        for key in regret_components:
            weights[key] *= scale
    
    return weights


def _post_process_v3(H: np.ndarray, 
                     distance_matrix: np.ndarray, 
                     stats: Dict, 
                     candidate_mask: np.ndarray,
                     graph_properties: Dict) -> np.ndarray:
    """Final post-processing with connectivity preservation and noise reduction."""
    n = stats['n']
    
    # 1. Ensure minimum node degree
    min_degree = min(3, n - 1)
    for i in range(n):
        non_zero = np.where(H[i] > 0)[0]
        if len(non_zero) < min_degree:
            # Find closest nodes not already connected
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            sorted_idx = np.argsort(dists)
            
            added = 0
            for j in sorted_idx:
                if H[i, j] == 0 and added < (min_degree - len(non_zero)):
                    base_score = 1 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score * 0.5
                    added += 1
    
    # 2. Apply candidate mask with soft threshold
    H = H * candidate_mask
    
    # 3. Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # 4. Adaptive noise filtering
    if n > 10:
        # Use percentile-based threshold
        threshold = np.percentile(H[H > 0], 10) if np.any(H > 0) else 0
        H[H < threshold] = 0
    
    # 5. Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    # 6. Ensure numerical stability
    H = np.clip(H, 0.0, 1.0)
    
    return H
