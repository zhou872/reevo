import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with adaptive mechanisms.
    
    Features:
    1. Adaptive power-law scaling based on distance distribution
    2. Regret-based edge bonus using lookahead regret calculation
    3. Mutual nearest neighbor affinity with dynamic neighborhood size
    4. Dynamic sparsification using statistical edge significance
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    # 1. Adaptive power-law scaling based on distance distribution
    # Calculate coefficient of variation to adjust emphasis on short edges
    flat_distances = distance_matrix[np.triu_indices(n, k=1)]
    if len(flat_distances) > 0:
        cv = np.std(flat_distances) / (np.mean(flat_distances) + 1e-12)
        # Higher CV -> more emphasis on very short edges
        power = 1.0 + 2.0 * min(cv, 1.0)  # Power between 1.0 and 3.0
    else:
        power = 1.5
    
    # Base inverse distance with adaptive power scaling
    inv_dist = 1 / (distance_matrix + 1e-12)
    inv_dist_pow = inv_dist ** power
    
    # 2. Regret-based edge bonus
    # For each edge (i,j), calculate the regret of not taking it
    # Regret = how much worse the alternative path through another node would be
    regret_bonus = np.ones((n, n))
    
    # For efficiency, compute for top candidates only
    k_regret = min(10, n - 1)
    for i in range(n):
        # Find k_regret nearest neighbors to i
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        nearest_i = np.argpartition(dist_i, k_regret)[:k_regret]
        
        for j in nearest_i:
            if i == j:
                continue
                
            # Find best alternative path i->k->j
            best_alternative = float('inf')
            for k in range(n):
                if k == i or k == j:
                    continue
                alternative = distance_matrix[i, k] + distance_matrix[k, j]
                if alternative < best_alternative:
                    best_alternative = alternative
            
            direct_cost = distance_matrix[i, j]
            if best_alternative > direct_cost:
                # Calculate normalized regret bonus
                regret = (best_alternative - direct_cost) / (direct_cost + 1e-12)
                regret_bonus[i, j] = 1.0 + min(regret, 2.0)  # Cap at 3.0
            else:
                regret_bonus[i, j] = 1.0
    
    # Symmetrize regret bonus
    regret_bonus = (regret_bonus + regret_bonus.T) / 2
    
    # 3. Mutual nearest neighbor affinity with dynamic neighborhood
    # Neighborhood size adapts based on graph density
    avg_degree = min(5, max(2, int(0.1 * n)))
    mutual_affinity = np.ones((n, n))
    
    # Build adjacency of top neighbors for each node
    top_neighbors = []
    for i in range(n):
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        neighbors = np.argpartition(dist_i, avg_degree)[:avg_degree]
        top_neighbors.append(set(neighbors))
    
    # Calculate mutual affinity scores
    for i in range(n):
        for j in range(i + 1, n):
            if i in top_neighbors[j] and j in top_neighbors[i]:
                # Mutual nearest neighbors get highest bonus
                mutual_affinity[i, j] = mutual_affinity[j, i] = 2.0
            elif i in top_neighbors[j] or j in top_neighbors[i]:
                # One-way affinity
                mutual_affinity[i, j] = mutual_affinity[j, i] = 1.5
    
    # 4. Dynamic sparsification with statistical significance
    # Calculate edge significance based on z-score of inverse distance
    inv_dist_mean = np.mean(inv_dist_pow[np.triu_indices(n, k=1)])
    inv_dist_std = np.std(inv_dist_pow[np.triu_indices(n, k=1)])
    
    if inv_dist_std > 0:
        # Calculate z-scores and keep edges with z-score > -1 (above average or slightly below)
        z_scores = (inv_dist_pow - inv_dist_mean) / inv_dist_std
        significance_mask = z_scores > -1.0
    else:
        significance_mask = np.ones_like(distance_matrix, dtype=bool)
    
    # Also apply distance-based sparsification
    if len(flat_distances) > 0:
        dist_threshold = np.quantile(flat_distances, 0.8)
        dist_mask = distance_matrix <= dist_threshold
    else:
        dist_mask = np.ones_like(distance_matrix, dtype=bool)
    
    # Combine masks
    combined_mask = significance_mask & dist_mask
    np.fill_diagonal(combined_mask, False)
    
    # 5. Combine all factors with weighted composition
    heuristics_matrix = inv_dist_pow * regret_bonus * mutual_affinity
    
    # Apply sparsification mask
    heuristics_matrix = heuristics_matrix * combined_mask
    
    # 6. Degree-aware normalization
    # Nodes with high average distances should have more selective edges
    node_avg_dist = distance_matrix.mean(axis=1)
    node_selectivity = node_avg_dist.max() / (node_avg_dist + 1e-12)
    degree_matrix = np.outer(node_selectivity, node_selectivity)
    
    heuristics_matrix = heuristics_matrix * degree_matrix
    
    # 7. Symmetrize and normalize
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    max_val = heuristics_matrix.max()
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix

if __name__ == "__main__":
    # Test with a simple 4-city TSP
    np.random.seed(42)
    n_cities = 6
    coords = np.random.rand(n_cities, 2) * 100
    
    # Create distance matrix

    # Compute heuristics
    h_matrix = heuristics_v2(distances)
