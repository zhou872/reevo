import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Dict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced deterministic TSP edge heuristics incorporating graph centrality,
    multi-hop regret analysis, and adaptive factor combination.
    
    Returns a matrix of same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Enhanced instance analysis
    stats = _compute_enhanced_stats(distance_matrix)
    
    # Step 2: Compute comprehensive heuristic factors
    factors = _compute_comprehensive_factors(distance_matrix, stats)
    
    # Step 3: Dynamic weight optimization based on instance structure
    weights = _optimize_factor_weights(stats)
    
    # Step 4: Geometric mean combination with connectivity constraints
    H = _combine_factors_intelligently(factors, weights, distance_matrix, stats)
    
    # Step 5: Post-processing with connectivity preservation
    H = _finalize_heuristic_matrix(H, distance_matrix, stats)
    
    return H


def _compute_enhanced_stats(distance_matrix: np.ndarray) -> Dict:
    """Compute advanced statistics about TSP instance including centrality measures."""
    n = distance_matrix.shape[0]
    mask = ~np.eye(n, dtype=bool)
    dist_values = distance_matrix[mask]
    
    # Basic statistics
    mean_dist = np.mean(dist_values)
    std_dist = np.std(dist_values)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Nearest neighbor analysis
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    nn_mean = np.mean(nn_dists)
    nn_cv = np.std(nn_dists) / (nn_mean + 1e-8)
    
    # Centrality measures
    closeness_centrality = np.zeros(n)
    betweenness_centrality = np.zeros(n)
    
    for i in range(n):
        # Closeness centrality: inverse of average distance to all other nodes
        sum_dist = np.sum(distance_matrix[i]) - distance_matrix[i, i]
        closeness_centrality[i] = (n - 1) / (sum_dist + 1e-8) if n > 1 else 0
        
        # Approximate betweenness centrality (simplified)
        if n > 3:
            for j in range(n):
                if j == i:
                    continue
                for k in range(j + 1, n):
                    if k == i:
                        continue
                    # Count if i lies on shortest path between j and k
                    direct = distance_matrix[j, k]
                    via_i = distance_matrix[j, i] + distance_matrix[i, k]
                    if via_i < direct * 1.01:  # Within 1% tolerance
                        betweenness_centrality[i] += 1
    
    # Normalize betweenness
    if np.max(betweenness_centrality) > 0:
        betweenness_centrality /= np.max(betweenness_centrality)
    
    # Cluster tendency using distance distribution skewness
    from scipy.stats import skew
    skewness = skew(dist_values) if len(dist_values) > 1 else 0
    cluster_tendency = 1 / (1 + np.exp(-0.5 * skewness))  # Sigmoid transform
    
    # Edge distribution uniformity
    uniformity = 1 / (1 + cv_dist)
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'cv_dist': cv_dist,
        'nn_cv': nn_cv,
        'uniformity': uniformity,
        'cluster_tendency': cluster_tendency,
        'closeness_centrality': closeness_centrality,
        'betweenness_centrality': betweenness_centrality
    }


def _compute_comprehensive_factors(distance_matrix: np.ndarray, stats: Dict) -> Dict:
    """Compute all heuristic factors including centrality-based measures."""
    n = stats['n']
    factors = {}
    
    # 1. Base inverse distance
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    factors['inv_dist'] = inv_dist
    
    # 2. Enhanced regret with multi-hop alternatives
    factors['regret'] = _compute_enhanced_regret(distance_matrix, stats)
    
    # 3. Centrality-aware scoring
    factors['centrality'] = _compute_centrality_scores(distance_matrix, stats)
    
    # 4. MST-based scores with degree awareness
    factors['mst'] = _compute_mst_scores_enhanced(distance_matrix, stats)
    
    # 5. Structural consistency (local vs global)
    factors['consistency'] = _compute_consistency_scores(distance_matrix, stats)
    
    return factors


def _compute_enhanced_regret(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute regret scores using adaptive multi-hop analysis."""
    n = distance_matrix.shape[0]
    regret_scores = np.ones((n, n))
    cv = stats['cv_dist']
    uniformity = stats['uniformity']
    
    for i in range(n):
        for j in range(i + 1, n):
            direct = distance_matrix[i, j]
            
            # Two-hop alternatives
            best_two_hop = np.inf
            for k in range(n):
                if k != i and k != j:
                    alt = distance_matrix[i, k] + distance_matrix[k, j]
                    if alt < best_two_hop:
                        best_two_hop = alt
            
            # Three-hop alternatives for structured instances (when n > 10)
            best_three_hop = np.inf
            if n > 10 and uniformity > 0.6:
                for k in range(min(10, n)):  # Sample limited nodes
                    if k == i or k == j:
                        continue
                    for l in range(min(10, n)):
                        if l == i or l == j or l == k:
                            continue
                        alt = distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j]
                        if alt < best_three_hop:
                            best_three_hop = alt
            
            # Take best alternative
            best_alternative = min(best_two_hop, best_three_hop)
            
            if best_alternative < np.inf:
                # Adaptive regret calculation
                regret_ratio = best_alternative / (direct + 1e-8)
                
                # Weight by instance uniformity: more emphasis in regular instances
                uniformity_weight = 1.0 + 0.8 * uniformity
                
                # Transform to score: higher when no good alternative exists
                if regret_ratio > 1.0:
                    score = np.log1p(regret_ratio - 1.0) * uniformity_weight
                else:
                    score = 1.0 / (1.0 + np.exp(5.0 * (regret_ratio - 0.8)))  # Sigmoid
                
                regret_scores[i, j] = regret_scores[j, i] = min(score, 2.5)
    
    np.fill_diagonal(regret_scores, 0)
    return regret_scores


def _compute_centrality_scores(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute scores based on graph centrality measures."""
    n = stats['n']
    centrality_scores = np.ones((n, n))
    
    closeness = stats['closeness_centrality']
    betweenness = stats['betweenness_centrality']
    
    # Normalize centrality measures
    if np.max(closeness) > 0:
        closeness = closeness / np.max(closeness)
    if np.max(betweenness) > 0:
        betweenness = betweenness / np.max(betweenness)
    
    # Combine centrality measures with adaptive weighting
    for i in range(n):
        for j in range(i + 1, n):
            # Favor edges connecting central nodes
            combined_centrality = (closeness[i] * closeness[j]) ** 0.5
            betweenness_bonus = (betweenness[i] + betweenness[j]) / 2
            
            # Edge centrality: favor edges that are part of many shortest paths
            edge_betweenness = 0
            for k in range(n):
                if k == i or k == j:
                    continue
                # Check if (i,j) is on shortest path between i/k or j/k
                if (distance_matrix[i, k] + distance_matrix[k, j] < 
                    distance_matrix[i, j] * 1.01):
                    edge_betweenness += 1
            
            edge_betweenness = min(edge_betweenness / (n - 2), 1.0) if n > 2 else 0
            
            # Final score combination
            centrality_score = (
                0.5 * combined_centrality +
                0.3 * betweenness_bonus +
                0.2 * edge_betweenness
            )
            
            centrality_scores[i, j] = centrality_scores[j, i] = centrality_score
    
    np.fill_diagonal(centrality_scores, 0)
    return centrality_scores


def _compute_mst_scores_enhanced(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute MST scores with degree and structure awareness."""
    n = stats['n']
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            # Compute MST
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Calculate node degrees in MST
            degrees = np.sum(mst_present, axis=1)
            max_degree = np.max(degrees) if len(degrees) > 0 else 0
            
            # Assign scores based on MST inclusion and structure
            for i in range(n):
                for j in range(i + 1, n):
                    if mst_present[i, j]:
                        # Base score for being in MST
                        base_score = 1.5
                        
                        # Bonus for connecting leaf nodes (degree 1)
                        if degrees[i] == 1 or degrees[j] == 1:
                            base_score = 2.0
                        
                        # Penalty for high-degree nodes connecting (to avoid hubs)
                        degree_factor = (max_degree - max(degrees[i], degrees[j])) / max_degree
                        base_score *= (0.8 + 0.2 * degree_factor)
                        
                        mst_scores[i, j] = mst_scores[j, i] = base_score
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores


def _compute_consistency_scores(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute consistency scores measuring local vs global edge properties."""
    n = stats['n']
    consistency_scores = np.ones((n, n))
    
    # Local vs global distance ratios
    mean_dist = stats['mean_dist']
    
    for i in range(n):
        for j in range(i + 1, n):
            direct = distance_matrix[i, j]
            
            # Compare with average distances from i and j
            avg_i = (np.sum(distance_matrix[i]) - distance_matrix[i, i]) / (n - 1) if n > 1 else direct
            avg_j = (np.sum(distance_matrix[j]) - distance_matrix[j, j]) / (n - 1) if n > 1 else direct
            
            # Consistency measure: how typical is this edge length?
            consistency_i = 1.0 / (1.0 + abs(direct - avg_i) / (mean_dist + 1e-8))
            consistency_j = 1.0 / (1.0 + abs(direct - avg_j) / (mean_dist + 1e-8))
            
            # Combined consistency score
            consistency_score = (consistency_i * consistency_j) ** 0.5
            
            consistency_scores[i, j] = consistency_scores[j, i] = consistency_score
    
    np.fill_diagonal(consistency_scores, 0)
    return consistency_scores


def _optimize_factor_weights(stats: Dict) -> Dict:
    """Dynamically optimize factor weights based on instance characteristics."""
    n = stats['n']
    cv = stats['cv_dist']
    uniformity = stats['uniformity']
    cluster_tendency = stats['cluster_tendency']
    nn_cv = stats['nn_cv']
    
    # Base weights adjusted by instance properties
    weights = {
        'inv_dist': 1.0,  # Always important
        
        # Regret weight: higher for uniform instances with low variance
        'regret': 0.7 + 0.5 * uniformity - 0.2 * cv,
        
        # Centrality weight: higher for clustered or structured instances
        'centrality': 0.3 + 0.6 * cluster_tendency + 0.2 * (1 - nn_cv),
        
        # MST weight: decreases with size, increases with structure
        'mst': max(0.2, 0.8 - 0.015 * n) * (1.0 + uniformity * 0.3),
        
        # Consistency weight: higher for irregular instances
        'consistency': 0.4 + 0.4 * (1 - uniformity)
    }
    
    # Ensure non-negative weights
    weights = {k: max(0.1, v) for k, v in weights.items()}
    
    # Normalize to sum to 1 for geometric mean
    total = sum(weights.values())
    weights = {k: v / total for k, v in weights.items()}
    
    return weights


def _combine_factors_intelligently(factors: Dict, weights: Dict, 
                                  distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Combine factors using weighted geometric mean with connectivity constraints."""
    n = stats['n']
    H = np.zeros((n, n))
    
    # Build adaptive candidate mask
    candidate_mask = _build_adaptive_mask(distance_matrix, stats)
    
    for i in range(n):
        for j in range(i + 1, n):
            if candidate_mask[i, j]:
                # Weighted geometric mean of all factors
                combined = 1.0
                for factor_name in factors:
                    factor_value = factors[factor_name][i, j]
                    weight = weights[factor_name]
                    combined *= (factor_value ** weight)
                
                H[i, j] = H[j, i] = combined
    
    return H


def _build_adaptive_mask(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Build mask of candidate edges using adaptive strategies."""
    n = stats['n']
    uniformity = stats['uniformity']
    
    # Base k: more candidates for uniform instances, fewer for structured ones
    base_k = min(20, max(6, int(np.log2(n) * 5)))
    k = int(base_k * (0.8 + 0.4 * uniformity))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # Include k-nearest neighbors for each node
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < len(dists):
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Always include MST edges
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            candidate_mask = candidate_mask | mst_present
        except:
            pass
    
    # Ensure symmetry
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask


def _finalize_heuristic_matrix(H: np.ndarray, distance_matrix: np.ndarray, 
                               stats: Dict) -> np.ndarray:
    """Final processing: sparsification, normalization, and connectivity."""
    n = stats['n']
    
    # Adaptive sparsification based on instance size
    if n > 15:
        # Keep top edges per node
        edges_per_node = min(8, max(3, n // 4))
        for i in range(n):
            row = H[i].copy()
            row[i] = -np.inf
            
            if np.sum(row > 0) > edges_per_node:
                threshold = np.partition(row, -edges_per_node)[-edges_per_node]
                H[i][row < threshold] = 0
    
    # Ensure symmetry
    H = np.maximum(H, H.T)
    
    # Ensure minimum connectivity: each node should have at least 3 connections
    min_edges = min(3, n - 1)
    for i in range(n):
        connected = np.nonzero(H[i])[0]
        if len(connected) < min_edges:
            # Add nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    base_score = 1 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    np.fill_diagonal(H, 0)
    return H
