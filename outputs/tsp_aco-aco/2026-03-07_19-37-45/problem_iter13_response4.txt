import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with dual-regret decomposition,
    multi-hop alternative comparison, and adaptive two-step lookahead penalty.
    Returns a matrix of the same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # ==================== 1. INSTANCE CHARACTERIZATION ====================
    # Remove diagonal for statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    # Basic statistics
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Estimate cluster tendency using Hopkins statistic approximation
    if n >= 10:
        m = min(10, n // 3)
        sample_indices = np.random.choice(n, m, replace=False)
        sample_nn_dists = []
        for idx in sample_indices:
            dists = distance_matrix[idx].copy()
            dists[idx] = np.inf
            sample_nn_dists.append(np.min(dists))
        random_dists = np.random.choice(distance_matrix.flatten(), m)
        hopkins = np.sum(sample_nn_dists) / (np.sum(sample_nn_dists) + np.min(random_dists) + 1e-8)
    else:
        hopkins = 0.5
    
    # Estimate sparsity via distance distribution analysis
    actual_dists = distance_matrix[~np.eye(n, dtype=bool)]
    if len(actual_dists) > 20:
        hist_actual, _ = np.histogram(actual_dists, bins=20, density=True)
        random_dists = np.random.exponential(scale=mean_dist, size=len(actual_dists))
        hist_random, _ = np.histogram(random_dists, bins=20, density=True)
        hist_actual = np.maximum(hist_actual, 1e-10)
        hist_random = np.maximum(hist_random, 1e-10)
        kl_div = np.sum(hist_actual * np.log(hist_actual / hist_random))
        sparsity = 1 / (1 + np.exp(-kl_div / 10))
    else:
        sparsity = 0.5
    
    # ==================== 2. BASE FACTORS COMPUTATION ====================
    # Factor 2.1: Inverse distance (always important)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    # Factor 2.2: DUAL-REGRET DECOMPOSITION
    regret_one_hop = np.ones((n, n))
    regret_two_hop = np.ones((n, n))
    
    for i in range(n):
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        sorted_i = np.argsort(dists_i)
        
        if len(sorted_i) >= 3:
            best_one = dists_i[sorted_i[0]]
            second_best = dists_i[sorted_i[1]]
            
            # One-hop regret: compare to best alternative direct edge from i
            for j in range(n):
                if j != i:
                    reg_val = max(0, dists_i[j] - best_one) / (best_one + 1e-8)
                    regret_one_hop[i, j] = 1 / (1 + reg_val)
            
            # Two-hop regret: compare to best two-hop alternative from i to j
            for j in range(n):
                if j != i:
                    # Find best two-hop path i->k->j
                    two_hop_costs = []
                    for k in range(n):
                        if k != i and k != j:
                            two_hop_costs.append(distance_matrix[i, k] + distance_matrix[k, j])
                    if two_hop_costs:
                        best_two = np.min(two_hop_costs)
                        reg_two = max(0, distance_matrix[i, j] - best_two) / (best_two + 1e-8)
                        regret_two_hop[i, j] = 1 / (1 + reg_two)
    
    # Combine regrets with adaptive weighting based on CV
    w_one_hop = 0.7 - 0.3 * cv_dist  # More one-hop regret for regular instances
    w_two_hop = 0.3 + 0.3 * cv_dist  # More two-hop regret for irregular instances
    regret_combined = (regret_one_hop ** w_one_hop) * (regret_two_hop ** w_two_hop)
    regret_combined = (regret_combined + regret_combined.T) / 2
    np.fill_diagonal(regret_combined, 0)
    
    # Factor 2.3: Density-based scoring (favors edges in sparse regions)
    k_density = min(5, max(2, n // 10))
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k_density)[:k_density]
        local_density[i] = np.mean(k_nearest)
    
    if np.std(local_density) > 1e-8:
        local_density = (local_density - np.mean(local_density)) / np.std(local_density)
    
    density_scores = np.ones((n, n))
    for i in range(n):
        for j in range(i+1, n):
            density_score = np.exp(-0.5 * (local_density[i] + local_density[j]))
            density_scores[i, j] = density_scores[j, i] = density_score
    
    # Factor 2.4: MST-based scoring
    mst_scores = np.ones((n, n))
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            mst_scores[mst_present] = 1.5  # Boost edges in MST
        except:
            pass
    np.fill_diagonal(mst_scores, 0)
    
    # ==================== 3. ADAPTIVE TWO-STEP LOOKAHEAD PENALTY ====================
    lookahead_penalty = np.ones((n, n))
    
    if n >= 4:
        # Adaptive penalty factor based on instance size and regularity
        penalty_factor = 1.2 - 0.1 * np.log2(n) + 0.1 * cv_dist
        
        for i in range(n):
            for j in range(i+1, n):
                if distance_matrix[i, j] > 3 * mean_dist:
                    continue
                
                # Find best two-hop alternative
                best_two_hop = np.inf
                for k in range(n):
                    if k != i and k != j:
                        two_hop = distance_matrix[i, k] + distance_matrix[k, j]
                        if two_hop < best_two_hop:
                            best_two_hop = two_hop
                
                # Apply penalty if two-hop is significantly shorter
                if best_two_hop < distance_matrix[i, j] / penalty_factor:
                    # Quadratic penalty based on the ratio
                    ratio = distance_matrix[i, j] / (best_two_hop + 1e-8)
                    penalty = np.exp(-(ratio - 1) * 2)
                    lookahead_penalty[i, j] = lookahead_penalty[j, i] = penalty
    
    # ==================== 4. CANDIDATE EDGE SELECTION ====================
    # Select top-k nearest neighbors for each node
    k_candidates = min(max(3, int(np.log2(n) * 2)), n - 1)
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if k_candidates < len(dists):
            indices = np.argpartition(dists, k_candidates)[:k_candidates]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        candidate_mask[i, indices] = True
    
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    # ==================== 5. ADAPTIVE WEIGHT COMPUTATION ====================
    # Weights based on instance characteristics
    weights = {
        'w_inv': 1.0,  # Base inverse distance
        'w_regret': 0.8 + 0.2 * cv_dist,  # More regret for varied instances
        'w_density': 0.3 + 0.4 * hopkins,  # More density for clustered instances
        'w_mst': max(0.1, 0.4 - 0.01 * n),  # Less MST for larger instances
        'w_lookahead': 0.2 + 0.3 * sparsity  # More lookahead for sparse instances
    }
    
    # Normalize weights to sum to 5
    total = sum(weights.values())
    for key in weights:
        weights[key] = weights[key] / total * 5
    
    # ==================== 6. COMBINE FACTORS WITH GEOMETRIC MEAN ====================
    H = np.zeros((n, n))
    
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Geometric combination of all factors
                combined = (
                    (inv_dist[i, j] ** weights['w_inv']) *
                    (regret_combined[i, j] ** weights['w_regret']) *
                    (density_scores[i, j] ** weights['w_density']) *
                    (mst_scores[i, j] ** weights['w_mst']) *
                    (lookahead_penalty[i, j] ** weights['w_lookahead'])
                )
                H[i, j] = H[j, i] = combined
    
    # ==================== 7. POST-PROCESSING ====================
    # Ensure minimum connectivity
    min_edges = min(3, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i])[0]
        if len(non_zero) < min_edges:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            for j in nearest:
                if H[i, j] == 0:
                    base_score = 1 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    # Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Sparsify: keep only top values for each node
    for i in range(n):
        row = H[i].copy()
        row[i] = -np.inf
        threshold = np.partition(row, n - k_candidates - 1)[n - k_candidates - 1]
        H[i, H[i] < threshold] = 0
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
