import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Dict, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge prioritization heuristic with policy-guided lookahead
    and graph-structure informed adaptive weighting.
    Returns matrix of same shape with higher values for promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Instance analysis for adaptive parameterization
    instance_stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Compute foundational heuristic components
    base_factors = _compute_base_factors_v2(distance_matrix, instance_stats)
    
    # Step 3: Policy-guided lookahead simulations
    lookahead_scores = _compute_policy_lookahead(distance_matrix, base_factors, instance_stats)
    
    # Step 4: Graph-structure informed weight adaptation
    weights = _compute_graph_aware_weights(distance_matrix, base_factors, instance_stats)
    
    # Step 5: Multi-factor integration with candidate filtering
    H = _integrate_factors_v2(distance_matrix, base_factors, lookahead_scores, weights, instance_stats)
    
    # Step 6: Post-processing with structure preservation
    H = _post_process_v2(H, distance_matrix, instance_stats)
    
    return H


def _analyze_instance_v2(distance_matrix: np.ndarray) -> Dict:
    """Enhanced instance analysis with graph structure metrics."""
    n = distance_matrix.shape[0]
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    # Basic statistics
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Graph structure analysis
    sparsity = _estimate_structured_sparsity(distance_matrix)
    cluster_tendency = _estimate_cluster_tendency_v2(distance_matrix)
    regularity = _estimate_regularity(distance_matrix)
    
    # Distance distribution analysis
    percentile_25 = np.percentile(dist_no_diag, 25)
    percentile_75 = np.percentile(dist_no_diag, 75)
    iqr = percentile_75 - percentile_25
    
    stats = {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'sparsity': sparsity,
        'cluster_tendency': cluster_tendency,
        'regularity': regularity,
        'iqr': iqr,
        'percentile_25': percentile_25,
        'percentile_75': percentile_75
    }
    
    return stats


def _estimate_structured_sparsity(distance_matrix: np.ndarray) -> float:
    """Estimate how structured/sparse the instance is using distance distribution analysis."""
    n = distance_matrix.shape[0]
    if n < 5:
        return 0.5
    
    # Flatten distances without diagonal
    flat_dists = distance_matrix[~np.eye(n, dtype=bool)]
    
    # Compare with multiple reference distributions
    n_samples = min(100, len(flat_dists))
    
    # Exponential reference
    exp_samples = np.random.exponential(scale=np.mean(flat_dists), size=n_samples)
    
    # Uniform reference
    uniform_samples = np.random.uniform(low=np.min(flat_dists), 
                                        high=np.max(flat_dists), 
                                        size=n_samples)
    
    # Compute Wasserstein distances to references
    def wasserstein_1d(a, b):
        a_sorted = np.sort(a[:min(len(a), n_samples)])
        b_sorted = np.sort(b[:min(len(b), n_samples)])
        return np.mean(np.abs(a_sorted - b_sorted))
    
    w_exp = wasserstein_1d(flat_dists, exp_samples)
    w_uniform = wasserstein_1d(flat_dists, uniform_samples)
    
    # Normalize and combine
    max_w = max(w_exp, w_uniform, 1e-8)
    score = (w_exp/max_w + w_uniform/max_w) / 2
    
    # Sigmoid transformation
    return 1 / (1 + np.exp(-5 * (score - 0.5)))


def _estimate_cluster_tendency_v2(distance_matrix: np.ndarray) -> float:
    """Enhanced cluster tendency estimation using multiple metrics."""
    n = distance_matrix.shape[0]
    if n < 8:
        return 0.5
    
    # Method 1: Hopkins statistic approximation
    m = min(15, n // 3)
    sample_indices = np.random.choice(n, m, replace=False)
    
    sample_nn_dists = []
    for idx in sample_indices:
        dists = distance_matrix[idx].copy()
        dists[idx] = np.inf
        sample_nn_dists.append(np.min(dists))
    
    # Generate random points in distance space
    all_dists = distance_matrix[~np.eye(n, dtype=bool)]
    random_dists = np.random.choice(all_dists, m)
    random_nn_dists = []
    for rd in random_dists:
        # Find closest actual distance
        random_nn_dists.append(np.min(np.abs(all_dists - rd)))
    
    hopkins = np.sum(sample_nn_dists) / (np.sum(sample_nn_dists) + np.sum(random_nn_dists) + 1e-8)
    
    # Method 2: Variance of nearest neighbor distances
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    
    nn_cv = np.std(nn_dists) / (np.mean(nn_dists) + 1e-8)
    
    # Combine methods
    cluster_score = 0.7 * hopkins + 0.3 * min(nn_cv, 1.0)
    return min(max(cluster_score, 0.0), 1.0)


def _estimate_regularity(distance_matrix: np.ndarray) -> float:
    """Estimate how regularly distributed the points are."""
    n = distance_matrix.shape[0]
    if n < 4:
        return 0.5
    
    # Compare each point's distance distribution to the global distribution
    ks_scores = []
    for i in range(min(20, n)):
        row_dists = distance_matrix[i].copy()
        row_dists[i] = np.inf
        
        # Skip if too few points
        if len(row_dists) < 5:
            continue
            
        # Approximate Kolmogorov-Smirnov statistic
        global_sorted = np.sort(distance_matrix[~np.eye(n, dtype=bool)])
        local_sorted = np.sort(row_dists)
        
        # Interpolate to same number of points
        n_points = min(len(global_sorted), len(local_sorted))
        global_quantiles = np.linspace(0, 1, n_points)
        local_quantiles = np.linspace(0, 1, n_points)
        
        ks = np.max(np.abs(global_quantiles - local_quantiles))
        ks_scores.append(ks)
    
    if not ks_scores:
        return 0.5
    
    avg_ks = np.mean(ks_scores)
    # Higher regularity = lower KS statistic
    return max(0.0, 1.0 - 2 * avg_ks)


def _compute_base_factors_v2(distance_matrix: np.ndarray, 
                            stats: Dict) -> Dict[str, np.ndarray]:
    """Compute enhanced base heuristic factors."""
    n = stats['n']
    factors = {}
    
    # 1. Inverse distance with adaptive scaling
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    # Normalize by instance characteristics
    if stats['cv_dist'] > 0.5:  # High variance instance
        inv_dist = inv_dist ** 0.8
    factors['inv_dist'] = inv_dist
    
    # 2. Regret scores with two-step consideration
    factors['regret'] = _compute_regret_scores_v2(distance_matrix, stats)
    
    # 3. Density-aware scores
    factors['density'] = _compute_density_scores_v2(distance_matrix, stats)
    
    # 4. MST-based scores with confidence weighting
    factors['mst'] = _compute_mst_scores_v2(distance_matrix, stats)
    
    # 5. Structural scores based on graph properties
    factors['structure'] = _compute_structural_scores(distance_matrix, stats)
    
    return factors


def _compute_regret_scores_v2(distance_matrix: np.ndarray, 
                             stats: Dict) -> np.ndarray:
    """Enhanced regret computation with adaptive parameters."""
    n = stats['n']
    regret = np.ones((n, n))
    
    # Adaptive k based on instance size
    k_regret = min(5, max(2, n // 8))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Sort distances
        sorted_idx = np.argsort(dists)
        if len(sorted_idx) >= k_regret + 1:
            # Use k-th nearest for regret calculation
            base_dists = dists[sorted_idx[:k_regret + 1]]
            best_dist = base_dists[0]
            
            for pos, j in enumerate(sorted_idx):
                if j == i:
                    continue
                    
                # Position-aware regret
                if pos < len(base_dists):
                    # Compute regret relative to k-th best
                    kth_best = base_dists[min(k_regret, len(base_dists)-1)]
                    edge_regret = max(0, dists[j] - best_dist) / (kth_best + 1e-8)
                else:
                    # Far away edges
                    edge_regret = (dists[j] - best_dist) / (best_dist + 1e-8)
                
                # Smooth transformation
                regret[i, j] = 1 / (1 + np.sqrt(edge_regret))
    
    # Symmetrize with geometric mean
    regret_sym = np.sqrt(regret * regret.T)
    np.fill_diagonal(regret_sym, 0)
    return regret_sym


def _compute_density_scores_v2(distance_matrix: np.ndarray, 
                              stats: Dict) -> np.ndarray:
    """Density scores with adaptive neighborhood detection."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Adaptive neighborhood size
    k_density = min(8, max(3, int(np.sqrt(n))))
    
    # Compute local densities using k-nearest neighbors
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if len(dists) >= k_density:
            k_nearest = np.partition(dists, k_density)[:k_density]
            # Use harmonic mean for density (more sensitive to close points)
            local_density[i] = k_density / np.sum(1/(k_nearest + 1e-8))
        else:
            local_density[i] = np.mean(dists)
    
    # Normalize densities
    if np.std(local_density) > 1e-8:
        local_density = (local_density - np.mean(local_density)) / (np.std(local_density) + 1e-8)
    
    # Compute edge scores: favor connections between similarly dense areas
    for i in range(n):
        for j in range(i+1, n):
            density_diff = abs(local_density[i] - local_density[j])
            # Penalize large density differences
            penalty = np.exp(-density_diff ** 2 / 2)
            
            # Bonus for connecting sparse areas
            sparse_bonus = np.exp(-0.5 * (max(local_density[i], 0) + max(local_density[j], 0)))
            
            density_scores[i, j] = density_scores[j, i] = penalty * (0.7 + 0.3 * sparse_bonus)
    
    return density_scores


def _compute_mst_scores_v2(distance_matrix: np.ndarray, 
                          stats: Dict) -> np.ndarray:
    """MST scores with confidence based on instance characteristics."""
    n = stats['n']
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Boost edges in MST, with stronger boost for structured instances
            mst_boost = 1.5 + 0.5 * stats['sparsity']
            mst_scores[mst_present] = mst_boost
            
            # Also give moderate boost to edges that are "near" MST edges
            mst_edges = np.where(mst_present)
            for i, j in zip(*mst_edges):
                if i < j:
                    # Find alternative edges that are slightly longer
                    alt_factor = 1.3
                    threshold = distance_matrix[i, j] * alt_factor
                    
                    # Boost edges that are close substitutes
                    for k in range(n):
                        if k != i and k != j:
                            if distance_matrix[i, k] + distance_matrix[k, j] < threshold:
                                mst_scores[i, k] = max(mst_scores[i, k], 1.2)
                                mst_scores[k, j] = max(mst_scores[k, j], 1.2)
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores


def _compute_structural_scores(distance_matrix: np.ndarray, 
                              stats: Dict) -> np.ndarray:
    """Scores based on graph structural properties."""
    n = stats['n']
    structure_scores = np.ones((n, n))
    
    # Compute node centrality approximation
    centrality = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        centrality[i] = 1 / (np.mean(dists) + 1e-8)
    
    # Normalize centrality
    if np.max(centrality) > 0:
        centrality = centrality / np.max(centrality)
    
    # Edge scores based on centrality and triangle relationships
    for i in range(n):
        for j in range(i+1, n):
            # Centrality penalty: avoid edges between highly central nodes
            centrality_penalty = 1.5 - 0.5 * (centrality[i] + centrality[j])
            
            # Triangle completeness bonus
            triangle_bonus = 1.0
            if n > 10:
                # Check if edge completes many triangles
                common_neighbors = 0
                for k in range(n):
                    if k != i and k != j:
                        if (distance_matrix[i, k] < 1.5 * stats['mean_dist'] and 
                            distance_matrix[j, k] < 1.5 * stats['mean_dist']):
                            common_neighbors += 1
                
                if common_neighbors > 0:
                    triangle_bonus = 1.0 + 0.1 * min(common_neighbors, 5)
            
            structure_scores[i, j] = structure_scores[j, i] = (
                centrality_penalty * triangle_bonus
            )
    
    return structure_scores


def _compute_policy_lookahead(distance_matrix: np.ndarray,
                             base_factors: Dict[str, np.ndarray],
                             stats: Dict) -> np.ndarray:
    """Policy-guided lookahead simulations using base heuristics."""
    n = stats['n']
    lookahead = np.ones((n, n))
    
    if n < 4:
        return lookahead
    
    # Create temporary heuristic for policy guidance
    policy_heuristic = _create_policy_heuristic(distance_matrix, base_factors, stats)
    
    # Adaptive simulation parameters
    n_simulations = min(30, max(5, int(10 * np.log2(n) / (1 + stats['regularity']))))
    n_steps = min(6, max(3, n // 5))
    
    # Pre-compute candidate edges for efficiency
    candidate_edges = _get_candidate_edges_for_lookahead(distance_matrix, stats)
    
    for i, j in candidate_edges:
        if i >= j:
            continue
            
        scores = []
        for _ in range(n_simulations):
            score = _simulate_with_policy(distance_matrix, i, j, 
                                         policy_heuristic, stats, n_steps)
            scores.append(score)
        
        if scores:
            # Use trimmed mean to reduce outlier effects
            scores_sorted = np.sort(scores)
            trim = max(1, len(scores) // 4)
            lookahead_score = np.mean(scores_sorted[trim:-trim]) if len(scores_sorted) > 2*trim else np.median(scores)
            lookahead[i, j] = lookahead[j, i] = lookahead_score
    
    # Normalize
    max_val = np.max(lookahead)
    if max_val > 0:
        lookahead = lookahead / max_val
    
    np.fill_diagonal(lookahead, 0)
    return lookahead


def _create_policy_heuristic(distance_matrix: np.ndarray,
                            base_factors: Dict[str, np.ndarray],
                            stats: Dict) -> np.ndarray:
    """Create heuristic matrix for policy guidance."""
    n = stats['n']
    
    # Combine base factors with fixed weights for policy
    policy_H = np.ones((n, n))
    
    for i in range(n):
        for j in range(i+1, n):
            # Policy weights (empirically tuned)
            combined = (
                base_factors['inv_dist'][i, j] ** 0.35 *
                base_factors['regret'][i, j] ** 0.25 *
                base_factors['density'][i, j] ** 0.15 *
                base_factors['mst'][i, j] ** 0.15 *
                base_factors['structure'][i, j] ** 0.10
            )
            policy_H[i, j] = policy_H[j, i] = combined
    
    np.fill_diagonal(policy_H, 0)
    return policy_H


def _get_candidate_edges_for_lookahead(distance_matrix: np.ndarray,
                                      stats: Dict) -> list:
    """Select candidate edges for lookahead simulations."""
    n = stats['n']
    candidates = []
    
    # Number of candidates per node
    k_candidates = min(12, max(4, int(3 * np.log2(n))))
    
    for i in range(n):
        # Get distances and exclude self
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Select k nearest neighbors
        if k_candidates < n-1:
            indices = np.argpartition(dists, k_candidates)[:k_candidates]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        for j in indices:
            if i < j:
                candidates.append((i, j))
    
    # Add some random connections for exploration
    n_random = min(20, max(5, n))
    for _ in range(n_random):
        i, j = np.random.randint(0, n, 2)
        if i != j and i < j and (i, j) not in candidates:
            candidates.append((i, j))
    
    return candidates


def _simulate_with_policy(distance_matrix: np.ndarray,
                         start_i: int, start_j: int,
                         policy_heuristic: np.ndarray,
                         stats: Dict,
                         n_steps: int) -> float:
    """Simulate partial tour extension using policy heuristic."""
    n = stats['n']
    visited = np.zeros(n, dtype=bool)
    visited[start_i] = visited[start_j] = True
    
    current = start_j
    tour_length = distance_matrix[start_i, start_j]
    
    for step in range(n_steps):
        unvisited = np.where(~visited)[0]
        if len(unvisited) == 0:
            break
        
        # Get policy scores for possible next nodes
        scores = []
        for v in unvisited:
            # Combine policy heuristic with distance
            policy_score = policy_heuristic[current, v]
            dist_score = 1 / (distance_matrix[current, v] + 1e-8)
            
            # Adaptive mixing based on step
            alpha = 0.7 - 0.1 * step  # More distance-focused early on
            combined = (policy_score ** alpha) * (dist_score ** (1 - alpha))
            scores.append(combined)
        
        scores = np.array(scores)
        
        # Softmax selection with temperature
        temperature = max(0.3, 1.0 - 0.1 * step)
        exp_scores = np.exp(scores / temperature)
        if np.sum(exp_scores) > 0:
            probs = exp_scores / np.sum(exp_scores)
            
            # Sometimes use greedy for efficiency
            if np.random.random() < 0.7:
                next_idx = np.argmax(probs)
            else:
                next_idx = np.random.choice(len(unvisited), p=probs)
            
            next_node = unvisited[next_idx]
            tour_length += distance_matrix[current, next_node]
            visited[next_node] = True
            current = next_node
        else:
            break
    
    # Score based on efficiency
    expected_length = n_steps * stats['mean_dist']
    efficiency = expected_length / (tour_length + 1e-8)
    
    # Penalize if visited very few nodes
    visited_count = np.sum(visited)
    if visited_count < 3:
        efficiency *= 0.5
    
    return min(efficiency, 2.5)


def _compute_graph_aware_weights(distance_matrix: np.ndarray,
                                base_factors: Dict[str, np.ndarray],
                                stats: Dict) -> Dict[str, float]:
    """Compute adaptive weights based on graph structure analysis."""
    n = stats['n']
    cv = stats['cv_dist']
    sparsity = stats['sparsity']
    cluster_tendency = stats['cluster_tendency']
    regularity = stats['regularity']
    
    # Base weights influenced by instance characteristics
    weights = {
        'w_inv': 1.0,  # Always important
        
        # Regret weight: higher for instances with clear structure
        'w_regret': 0.4 + 0.3 * sparsity * (1 - regularity),
        
        # Density weight: important for clustered instances
        'w_density': 0.3 + 0.4 * cluster_tendency,
        
        # MST weight: more important for larger, structured instances
        'w_mst': max(0.1, 0.6 - 0.015 * n + 0.2 * sparsity),
        
        # Structure weight: for graph properties
        'w_structure': 0.2 + 0.2 * (1 - regularity),
        
        # Lookahead weight: more valuable for complex instances
        'w_lookahead': 0.2 + 0.3 * min(sparsity + cluster_tendency, 1.0)
    }
    
    # Adjust based on instance size
    size_factor = min(1.0, n / 50)
    for key in weights:
        if key != 'w_inv':
            weights[key] *= (0.8 + 0.4 * size_factor)
    
    # Normalize to maintain balance
    total = sum(weights.values())
    for key in weights:
        weights[key] = weights[key] / total * 6  # Sum to 6
    
    return weights


def _integrate_factors_v2(distance_matrix: np.ndarray,
                         base_factors: Dict[str, np.ndarray],
                         lookahead_scores: np.ndarray,
                         weights: Dict[str, float],
                         stats: Dict) -> np.ndarray:
    """Integrate all factors with graph-informed combination."""
    n = stats['n']
    
    # Build candidate mask
    candidate_mask = _build_candidate_mask_v2(distance_matrix, base_factors, stats)
    
    # Initialize result matrix
    H = np.zeros((n, n))
    
    # Multiplicative integration
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                combined = (
                    base_factors['inv_dist'][i, j] ** weights['w_inv'] *
                    base_factors['regret'][i, j] ** weights['w_regret'] *
                    base_factors['density'][i, j] ** weights['w_density'] *
                    base_factors['mst'][i, j] ** weights['w_mst'] *
                    base_factors['structure'][i, j] ** weights['w_structure'] *
                    lookahead_scores[i, j] ** weights['w_lookahead']
                )
                H[i, j] = H[j, i] = combined
    
    # Ensure all nodes have some connections
    min_edges = min(3, n - 1)
    for i in range(n):
        if np.sum(H[i] > 0) < min_edges:
            # Add nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    base_score = 1 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    return H


def _build_candidate_mask_v2(distance_matrix: np.ndarray,
                            base_factors: Dict[str, np.ndarray],
                            stats: Dict) -> np.ndarray:
    """Build mask of candidate edges considering multiple factors."""
    n = stats['n']
    
    # Adaptive candidate size
    base_k = min(15, max(5, int(4 * np.log2(n))))
    
    # Adjust based on instance characteristics
    if stats['sparsity'] > 0.7:
        base_k = int(base_k * 1.3)  # More candidates for structured instances
    if stats['cluster_tendency'] > 0.7:
        base_k = int(base_k * 1.2)  # More candidates for clustered instances
    
    k = min(base_k, n - 1)
    
    # Initialize mask
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # Use combined scores for candidate selection
    combined_for_selection = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            quick_score = (
                base_factors['inv_dist'][i, j] ** 0.5 *
                base_factors['regret'][i, j] ** 0.3 *
                base_factors['mst'][i, j] ** 0.2
            )
            combined_for_selection[i, j] = combined_for_selection[j, i] = quick_score
    
    # Select top k edges per node
    for i in range(n):
        scores = combined_for_selection[i].copy()
        scores[i] = -np.inf
        
        if k < n-1:
            threshold = np.partition(scores, -k)[-k]
            candidates = np.where(scores >= threshold)[0]
        else:
            candidates = np.arange(n)
            candidates = candidates[candidates != i]
        
        candidate_mask[i, candidates] = True
    
    # Ensure symmetry
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask


def _post_process_v2(H: np.ndarray,
                    distance_matrix: np.ndarray,
                    stats: Dict) -> np.ndarray:
    """Final post-processing with structure preservation."""
    n = stats['n']
    
    # Ensure symmetry and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Apply sparsification for large instances
    if n > 20:
        H = _adaptive_sparsify(H, distance_matrix, stats)
    
    # Boost edges that appear in high-quality triangles
    if n > 10:
        H = _enhance_triangle_edges(H, distance_matrix, stats)
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    # Ensure minimum values for connectivity
    min_nonzero = np.min(H[H > 0]) if np.any(H > 0) else 1.0
    for i in range(n):
        if np.sum(H[i] > 0) < 2:  # Node with too few connections
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:2]
            for j in nearest:
                H[i, j] = H[j, i] = max(H[i, j], min_nonzero * 0.5)
    
    return H


def _adaptive_sparsify(H: np.ndarray,
                      distance_matrix: np.ndarray,
                      stats: Dict) -> np.ndarray:
    """Sparsify heuristic matrix adaptively."""
    n = stats['n']
    
    # Determine keep count per node
    base_keep = min(10, max(4, int(2.5 * np.log2(n))))
    
    # Adjust based on instance characteristics
    if stats['sparsity'] > 0.7:
        base_keep = int(base_keep * 1.2)  # Keep more for structured
    
    H_sparse = H.copy()
    
    for i in range(n):
        row = H[i].copy()
        row[i] = -np.inf
        
        if len(row) > base_keep:
            # Find threshold for top edges
            threshold = np.partition(row, -base_keep)[-base_keep]
            
            # Zero out below threshold
            mask = row < threshold
            H_sparse[i, mask] = 0
            H_sparse[mask, i] = 0
    
    # Ensure MST edges are preserved
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            for i in range(n):
                for j in range(i+1, n):
                    if mst_present[i, j] and H_sparse[i, j] == 0:
                        # Restore with moderate value
                        H_sparse[i, j] = H_sparse[j, i] = np.median(H[H > 0]) if np.any(H > 0) else 0.3
        except:
            pass
    
    return H_sparse


def _enhance_triangle_edges(H: np.ndarray,
                           distance_matrix: np.ndarray,
                           stats: Dict) -> np.ndarray:
    """Enhance scores for edges that complete high-quality triangles."""
    n = stats['n']
    H_enhanced = H.copy()
    
    # Only process for moderate to large instances
    if n < 8:
        return H_enhanced
    
    threshold_dist = stats['percentile_75']
    
    for i in range(n):
        for j in range(i+1, n):
            if H[i, j] == 0:
                continue
                
            # Count high-quality triangles involving this edge
            good_triangles = 0
            for k in range(n):
                if k != i and k != j:
                    # Check if triangle (i,j,k) has all reasonable edges
                    if (distance_matrix[i, k] < threshold_dist and 
                        distance_matrix[j, k] < threshold_dist):
                        
                        # Check if edge completes a good triangle
                        triangle_perimeter = (distance_matrix[i, j] + 
                                             distance_matrix[i, k] + 
                                             distance_matrix[j, k])
                        
                        if triangle_perimeter < 3 * threshold_dist:
                            good_triangles += 1
            
            # Enhance edge score based on triangle count
            if good_triangles >= 2:
                enhancement = 1.0 + 0.05 * min(good_triangles, 10)
                H_enhanced[i, j] = H_enhanced[j, i] = H[i, j] * enhancement
    
    return H_enhanced
