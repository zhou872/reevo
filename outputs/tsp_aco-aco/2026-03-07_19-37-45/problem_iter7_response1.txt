import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with two-hop regret computation
    and adaptive instance-based weighting.
    Returns a matrix of the same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Analyze instance characteristics
    instance_stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Compute base heuristic factors
    base_factors = _compute_base_factors_v2(distance_matrix, instance_stats)
    
    # Step 3: Adaptive weight computation
    weights = _compute_adaptive_weights_v2(instance_stats)
    
    # Step 4: Combine factors with adaptive weighting
    H = np.zeros((n, n))
    
    # Candidate edges for efficiency
    candidate_mask = _build_candidate_mask_v2(distance_matrix, instance_stats)
    
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Multiplicative combination with instance-adaptive exponents
                combined = (
                    (base_factors['inv_dist'][i, j] ** weights['w_inv']) *
                    (base_factors['regret_one_hop'][i, j] ** weights['w_regret_one']) *
                    (base_factors['regret_two_hop'][i, j] ** weights['w_regret_two']) *
                    (base_factors['density'][i, j] ** weights['w_density']) *
                    (base_factors['mst'][i, j] ** weights['w_mst'])
                )
                H[i, j] = H[j, i] = combined
    
    # Step 5: Post-processing and sparsification
    H = _post_process_heuristics_v2(H, distance_matrix, instance_stats)
    
    return H

def _analyze_instance_v2(distance_matrix: np.ndarray) -> dict:
    """Analyze instance characteristics for adaptive parameter tuning."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal for statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    # Calculate coefficient of variation
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Regularity measure (percentile-based)
    p25 = np.percentile(dist_no_diag, 25)
    p75 = np.percentile(dist_no_diag, 75)
    regularity = 1.0 - min(1.0, (p75 - p25) / (mean_dist + 1e-8))
    
    stats = {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'regularity': regularity,
        'p25': p25,
        'p75': p75
    }
    
    return stats

def _compute_base_factors_v2(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute base heuristic factors including two-hop regret."""
    n = stats['n']
    
    factors = {}
    
    # 1. Inverse distance (always useful)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    factors['inv_dist'] = inv_dist
    
    # 2. One-hop regret scoring
    factors['regret_one_hop'] = _compute_one_hop_regret(distance_matrix)
    
    # 3. Two-hop regret scoring (new feature)
    factors['regret_two_hop'] = _compute_two_hop_regret(distance_matrix, stats)
    
    # 4. Density-based scoring
    factors['density'] = _compute_density_scores_v2(distance_matrix, stats)
    
    # 5. MST-based scoring
    factors['mst'] = _compute_mst_scores_v2(distance_matrix)
    
    return factors

def _compute_one_hop_regret(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute one-hop regret scores for each edge."""
    n = distance_matrix.shape[0]
    regret = np.ones((n, n))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if n > 2:
            sorted_dists = np.sort(dists)
            best_dist = sorted_dists[0]
            second_best = sorted_dists[1]
            
            for j in range(n):
                if j != i:
                    edge_regret = max(0, dists[j] - best_dist) / (best_dist + 1e-8)
                    regret[i, j] = 1 / (1 + edge_regret)
    
    # Symmetrize
    regret = (regret + regret.T) / 2
    np.fill_diagonal(regret, 0)
    return regret

def _compute_two_hop_regret(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute two-hop regret scores using variance-adaptive thresholding."""
    n = stats['n']
    cv = stats['cv_dist']
    regret_two = np.ones((n, n))
    
    # Adaptive penalty threshold based on coefficient of variation
    # Higher CV → more variable distances → looser threshold
    penalty_threshold = 0.3 + 0.4 * (1 - np.tanh(cv))
    
    for i in range(n):
        for j in range(i+1, n):
            direct_cost = distance_matrix[i, j]
            two_hop_costs = []
            
            # Consider intermediate nodes k
            for k in range(n):
                if k != i and k != j:
                    two_hop_cost = distance_matrix[i, k] + distance_matrix[k, j]
                    two_hop_costs.append(two_hop_cost)
            
            if two_hop_costs:
                best_two_hop = min(two_hop_costs)
                regret_value = max(0, best_two_hop - direct_cost) / (direct_cost + 1e-8)
                
                # Apply variance-adaptive thresholding
                if regret_value < penalty_threshold:
                    # Good edge - low regret compared to two-hop alternatives
                    regret_two[i, j] = 1.2  # Boost score
                else:
                    # Bad edge - high regret
                    regret_two[i, j] = 0.8  # Penalize score
    
    # Symmetrize
    regret_two = (regret_two + regret_two.T) / 2
    np.fill_diagonal(regret_two, 0)
    return regret_two

def _compute_density_scores_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute density-based scores favoring edges in sparse regions."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Adaptive k based on instance size
    k = min(5, max(2, n // 10))
    
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k)[:k]
        local_density[i] = np.mean(k_nearest)
    
    # Normalize densities
    if np.std(local_density) > 1e-8:
        local_density = (local_density - np.mean(local_density)) / np.std(local_density)
    
    # Create density scores: favor edges connecting sparse areas
    for i in range(n):
        for j in range(i+1, n):
            # Higher score for connecting two sparse nodes
            density_score = np.exp(-0.5 * abs(local_density[i] + local_density[j]))
            density_scores[i, j] = density_scores[j, i] = density_score
    
    return density_scores

def _compute_mst_scores_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute scores based on Minimum Spanning Tree."""
    n = distance_matrix.shape[0]
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            # Boost edges in MST
            mst_scores[mst_present] = 1.5
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores

def _compute_adaptive_weights_v2(stats: dict) -> dict:
    """Dynamically compute weights based on instance characteristics."""
    n = stats['n']
    cv = stats['cv_dist']
    regularity = stats['regularity']
    
    # Base weights adjust based on instance properties
    # Target sum of weights is 4 for balanced responsiveness
    target_sum = 4.0
    
    # Weights determined by instance characteristics
    weights = {
        'w_inv': 1.0,  # Always important
        
        # Regret weights: split between one-hop and two-hop
        'w_regret_one': 0.5 + 0.3 * regularity,  # More regular → more one-hop
        'w_regret_two': 0.5 + 0.3 * (1 - regularity),  # Less regular → more two-hop
        
        # Density weight: higher for more variable instances
        'w_density': 0.3 + 0.2 * cv,
        
        # MST weight: decreases with instance size
        'w_mst': max(0.2, 0.6 - 0.02 * n)
    }
    
    # Normalize to target sum
    current_sum = sum(weights.values())
    if current_sum > 0:
        scale_factor = target_sum / current_sum
        for key in weights:
            weights[key] *= scale_factor
    
    return weights

def _build_candidate_mask_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Build mask of candidate edges to consider (for efficiency)."""
    n = stats['n']
    
    # Adaptive candidate list size based on instance characteristics
    base_k = min(15, max(5, int(np.log2(n) * 3)))
    # Increase for more regular instances
    k = int(base_k * (1 + stats['regularity']))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < len(dists):
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Make symmetric
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask

def _post_process_heuristics_v2(H: np.ndarray, distance_matrix: np.ndarray, 
                               stats: dict) -> np.ndarray:
    """Post-process the heuristic matrix with sparsification."""
    n = stats['n']
    
    # Sparsify by setting low values to zero
    if n > 10:
        threshold = np.percentile(H[H > 0], 30)  # Keep top 70%
        H[H < threshold] = 0
    
    # Ensure minimum connectivity
    min_edges = min(3, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i])[0]
        if len(non_zero) < min_edges:
            # Add nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    # Assign base inverse distance score
                    H[i, j] = H[j, i] = 1 / (distance_matrix[i, j] + 1e-8)
    
    # Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
