import numpy as np
from typing import Dict, Tuple, List
from scipy.sparse.csgraph import minimum_spanning_tree
import random

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with node centrality measures,
    meta-heuristic weight adaptation, and improved lookahead strategies.
    
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Enhanced instance analysis with centrality metrics
    instance_stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Compute comprehensive base factors
    base_factors = _compute_base_factors_v2(distance_matrix, instance_stats)
    
    # Step 3: Meta-heuristic weight adaptation
    weights = _meta_heuristic_weight_learning(distance_matrix, instance_stats, base_factors)
    
    # Step 4: Combined edge scoring
    H = np.zeros((n, n))
    
    # Candidate edges for efficiency
    candidate_mask = _build_candidate_mask_v2(distance_matrix, instance_stats)
    
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Multiplicative combination with learned weights
                combined = (
                    (base_factors['inv_dist'][i, j] ** weights['w_inv']) *
                    (base_factors['regret'][i, j] ** weights['w_regret']) *
                    (base_factors['density'][i, j] ** weights['w_density']) *
                    (base_factors['lookahead'][i, j] ** weights['w_lookahead']) *
                    (base_factors['centrality'][i, j] ** weights['w_centrality']) *
                    (base_factors['mst'][i, j] ** weights['w_mst'])
                )
                H[i, j] = H[j, i] = combined
    
    # Step 5: Post-processing and refinement
    H = _post_process_v2(H, distance_matrix, instance_stats)
    
    return H

def _analyze_instance_v2(distance_matrix: np.ndarray) -> Dict:
    """Enhanced instance analysis with centrality and clustering metrics."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal for statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    stats = {
        'n': n,
        'mean_dist': np.mean(dist_no_diag),
        'std_dist': np.std(dist_no_diag),
        'cv_dist': np.std(dist_no_diag) / (np.mean(dist_no_diag) + 1e-8),
        'sparsity': _estimate_sparsity_v2(distance_matrix),
        'cluster_tendency': _estimate_cluster_tendency_v2(distance_matrix),
        'centrality_variance': _compute_centrality_variance(distance_matrix)
    }
    
    return stats

def _estimate_sparsity_v2(distance_matrix: np.ndarray) -> float:
    """Estimate instance sparsity using distance distribution analysis."""
    n = distance_matrix.shape[0]
    
    # Compute distance percentiles
    dists = distance_matrix[~np.eye(n, dtype=bool)]
    q25, q75 = np.percentile(dists, [25, 75])
    
    # Sparsity measure: ratio of interquartile range to median
    median = np.median(dists)
    iqr = q75 - q25
    sparsity = iqr / (median + 1e-8)
    
    return min(1.0, sparsity * 2)  # Normalize to [0, 1]

def _estimate_cluster_tendency_v2(distance_matrix: np.ndarray) -> float:
    """Enhanced cluster tendency estimation using neighborhood consistency."""
    n = distance_matrix.shape[0]
    if n < 10:
        return 0.5
    
    # For each point, compute consistency of neighbor sets
    k = min(5, n // 4)
    consistency_scores = []
    
    for i in range(n):
        # Get k-nearest neighbors
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        neighbors = np.argpartition(dists, k)[:k]
        
        # For each neighbor, check reciprocal relationship
        reciprocal_count = 0
        for j in neighbors:
            # Check if i is in j's k-nearest neighbors
            dists_j = distance_matrix[j].copy()
            dists_j[j] = np.inf
            j_neighbors = np.argpartition(dists_j, k)[:k]
            if i in j_neighbors:
                reciprocal_count += 1
        
        consistency_scores.append(reciprocal_count / k)
    
    return np.mean(consistency_scores)

def _compute_centrality_variance(distance_matrix: np.ndarray) -> float:
    """Compute variance in node centrality measures."""
    n = distance_matrix.shape[0]
    
    # Compute closeness centrality for each node
    centralities = []
    for i in range(n):
        # Closeness centrality: inverse of average distance to all other nodes
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        avg_dist = np.mean(dists)
        centrality = 1 / (avg_dist + 1e-8)
        centralities.append(centrality)
    
    # Normalized variance of centralities
    if np.std(centralities) < 1e-8:
        return 0.0
    
    cv = np.std(centralities) / (np.mean(centralities) + 1e-8)
    return min(1.0, cv)

def _compute_base_factors_v2(distance_matrix: np.ndarray, stats: Dict) -> Dict:
    """Compute all base heuristic factors including centrality measures."""
    n = stats['n']
    
    factors = {}
    
    # 1. Inverse distance
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    factors['inv_dist'] = inv_dist
    
    # 2. Enhanced regret-based scoring
    factors['regret'] = _compute_regret_scores_v2(distance_matrix)
    
    # 3. Cluster-aware density scoring
    factors['density'] = _compute_density_scores_v2(distance_matrix, stats)
    
    # 4. MST-based scoring
    factors['mst'] = _compute_mst_scores_v2(distance_matrix)
    
    # 5. Centrality-based scoring
    factors['centrality'] = _compute_centrality_scores(distance_matrix, stats)
    
    # 6. Lookahead scores
    factors['lookahead'] = _compute_lookahead_scores_v2(distance_matrix, stats)
    
    return factors

def _compute_regret_scores_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Enhanced regret computation with adaptive thresholds."""
    n = distance_matrix.shape[0]
    regret = np.ones((n, n))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        sorted_dists = np.sort(dists)
        sorted_idx = np.argsort(dists)
        
        if len(sorted_dists) >= 3:
            best_dist = sorted_dists[0]
            
            # Adaptive threshold based on distance distribution
            if len(sorted_dists) >= 5:
                threshold_dist = sorted_dists[4]  # 5th nearest
            else:
                threshold_dist = sorted_dists[-1]  # Farthest
            
            for rank, j in enumerate(sorted_idx):
                if j == i:
                    continue
                    
                dist_ij = distance_matrix[i, j]
                
                if dist_ij <= threshold_dist:
                    # Within threshold: linear penalty based on rank
                    penalty = rank / (len(sorted_dists) - 1)
                    regret[i, j] = 1.0 - 0.5 * penalty
                else:
                    # Beyond threshold: exponential decay
                    excess = (dist_ij - threshold_dist) / (threshold_dist + 1e-8)
                    regret[i, j] = np.exp(-excess)
    
    # Symmetrize and zero diagonal
    regret = (regret + regret.T) / 2
    np.fill_diagonal(regret, 0)
    return regret

def _compute_density_scores_v2(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Cluster-aware density scoring."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Compute local density for each node
    k = min(7, max(3, n // 6))
    densities = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k)[:k]
        densities[i] = np.mean(k_nearest)
    
    # Normalize densities
    if np.std(densities) > 1e-8:
        densities = (densities - np.mean(densities)) / np.std(densities)
    
    # Score edges: favor connections between sparse areas
    for i in range(n):
        for j in range(i+1, n):
            # Combination of both nodes' density scores
            # Higher score when connecting two sparse nodes
            if densities[i] > 0 and densities[j] > 0:
                # Both are sparse - high priority
                density_score = 1.5
            elif densities[i] < 0 and densities[j] < 0:
                # Both are dense - lower priority
                density_score = 0.7
            else:
                # Mixed - moderate priority
                density_score = 1.0
            
            density_scores[i, j] = density_scores[j, i] = density_score
    
    return density_scores

def _compute_mst_scores_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """MST scoring with edge importance weighting."""
    n = distance_matrix.shape[0]
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            # Compute MST
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Boost edges in MST
            mst_scores[mst_present] = 2.0
            
            # Additional boost for critical edges (edges in both directions of MST)
            for i in range(n):
                mst_neighbors = np.where(mst_present[i])[0]
                if len(mst_neighbors) == 1:
                    # Leaf node - edge is more critical
                    j = mst_neighbors[0]
                    mst_scores[i, j] = mst_scores[j, i] = 2.5
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores

def _compute_centrality_scores(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute edge scores based on node centrality measures."""
    n = stats['n']
    centrality_scores = np.ones((n, n))
    
    # Compute betweenness centrality approximation
    centrality = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        centrality[i] = 1 / (np.mean(dists) + 1e-8)
    
    # Normalize centrality scores
    if np.max(centrality) > np.min(centrality):
        centrality = (centrality - np.min(centrality)) / (np.max(centrality) - np.min(centrality))
    
    # For each edge, combine centrality of endpoints
    for i in range(n):
        for j in range(i+1, n):
            # Edge score based on centrality:
            # Higher score when connecting two central nodes OR connecting a central node to isolate
            ci, cj = centrality[i], centrality[j]
            
            if ci > 0.7 and cj > 0.7:
                # Both are central - medium priority
                score = 1.2
            elif ci < 0.3 and cj < 0.3:
                # Both are peripheral - medium-high priority
                score = 1.3
            elif (ci > 0.7 and cj < 0.3) or (ci < 0.3 and cj > 0.7):
                # Central-peripheral connection - highest priority
                score = 1.5
            else:
                # Mixed - standard priority
                score = 1.0
            
            centrality_scores[i, j] = centrality_scores[j, i] = score
    
    return centrality_scores

def _compute_lookahead_scores_v2(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Improved lookahead with greedy path extension and node importance."""
    n = stats['n']
    lookahead_scores = np.ones((n, n))
    
    if n < 4:
        return lookahead_scores
    
    # Adaptive simulation parameters
    n_simulations = min(25, max(5, int(np.log2(n) * 4)))
    max_steps = min(8, n - 2)
    
    for i in range(n):
        for j in range(i+1, n):
            # Skip obviously bad edges
            if distance_matrix[i, j] > 3 * stats['mean_dist']:
                continue
            
            simulation_scores = []
            
            for _ in range(n_simulations):
                score = _simulate_greedy_extension(distance_matrix, i, j, 
                                                  max_steps, stats['mean_dist'])
                simulation_scores.append(score)
            
            if simulation_scores:
                lookahead_scores[i, j] = lookahead_scores[j, i] = np.median(simulation_scores)
    
    # Normalize
    max_val = np.max(lookahead_scores)
    if max_val > 0:
        lookahead_scores = lookahead_scores / max_val
    
    np.fill_diagonal(lookahead_scores, 0)
    return lookahead_scores

def _simulate_greedy_extension(distance_matrix: np.ndarray, start_i: int, start_j: int,
                              max_steps: int, mean_dist: float) -> float:
    """Simulate greedy path extension from edge (i,j)."""
    n = distance_matrix.shape[0]
    visited = np.zeros(n, dtype=bool)
    visited[start_i] = visited[start_j] = True
    
    path = [start_i, start_j]
    current = start_j
    path_length = distance_matrix[start_i, start_j]
    
    # Extend path greedily
    for step in range(max_steps):
        unvisited = np.where(~visited)[0]
        if len(unvisited) == 0:
            break
        
        # Find closest unvisited node
        min_dist = np.inf
        next_node = -1
        
        for node in unvisited:
            dist = distance_matrix[current, node]
            if dist < min_dist:
                min_dist = dist
                next_node = node
        
        if next_node == -1:
            break
        
        # Add to path
        path.append(next_node)
        path_length += min_dist
        visited[next_node] = True
        current = next_node
    
    # Compute score: compare actual path length to expected length
    expected_length = (len(path) - 1) * mean_dist
    if path_length > 0:
        score = expected_length / path_length
    else:
        score = 1.0
    
    return min(score, 3.0)

def _meta_heuristic_weight_learning(distance_matrix: np.ndarray, 
                                   stats: Dict, 
                                   base_factors: Dict) -> Dict:
    """Adapt weights using simulated annealing-inspired approach."""
    n = stats['n']
    
    # Base weights determined by instance characteristics
    cv = stats['cv_dist']
    sparsity = stats['sparsity']
    cluster_tend = stats['cluster_tendency']
    centrality_var = stats['centrality_variance']
    
    weights = {
        'w_inv': 1.0,  # Always important
        
        # Regret weight: higher for irregular instances
        'w_regret': 0.4 + 0.3 * cv,
        
        # Density weight: higher for clustered instances
        'w_density': 0.3 + 0.4 * cluster_tend,
        
        # Lookahead weight: higher for sparse, structured instances
        'w_lookahead': 0.2 + 0.3 * sparsity,
        
        # Centrality weight: higher when centrality varies significantly
        'w_centrality': 0.1 + 0.3 * centrality_var,
        
        # MST weight: decreases with instance size
        'w_mst': max(0.1, 0.6 - 0.02 * n)
    }
    
    # Normalize to sum to 6 (one for each factor)
    total = sum(weights.values())
    scale = 6.0 / total
    
    for key in weights:
        weights[key] = weights[key] * scale
    
    # Simple local optimization: adjust weights based on factor correlations
    if n >= 10:
        # Sample some edges to check factor consistency
        sample_size = min(50, n * (n-1) // 2)
        edges = []
        
        for _ in range(sample_size):
            i, j = random.sample(range(n), 2)
            edges.append((i, j))
        
        # Compute simple scores with current weights
        scores = []
        for i, j in edges:
            score = (
                base_factors['inv_dist'][i, j] ** weights['w_inv'] *
                base_factors['regret'][i, j] ** weights['w_regret'] *
                base_factors['density'][i, j] ** weights['w_density'] *
                base_factors['lookahead'][i, j] ** weights['w_lookahead'] *
                base_factors['centrality'][i, j] ** weights['w_centrality'] *
                base_factors['mst'][i, j] ** weights['w_mst']
            )
            scores.append(score)
        
        # Adjust weights slightly based on variance of scores
        score_var = np.var(scores)
        if score_var < 0.01:  # Too uniform
            # Increase weights of most discriminative factors
            for factor in ['inv_dist', 'regret', 'centrality']:
                weights[f'w_{factor}'] *= 1.1
    
    return weights

def _build_candidate_mask_v2(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Build candidate edge mask with dynamic thresholding."""
    n = stats['n']
    
    # Base candidate size
    k_base = min(12, max(4, int(np.log2(n) * 2)))
    
    # Adjust based on instance characteristics
    k = int(k_base * (1.0 + 0.5 * stats['sparsity']))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k-nearest neighbors
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < n - 1:
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
        
        # Add a few promising edges based on other heuristics
        # Specifically, add edges to nodes that are central
        centrality = np.zeros(n)
        for j in range(n):
            if i == j:
                continue
            centrality[j] = 1 / (distance_matrix[i, j] + 1e-8)
        
        # Add top m additional edges based on centrality
        m = max(2, int(k * 0.2))
        top_central = np.argpartition(centrality, -m)[-m:]
        
        for j in top_central:
            if j != i:
                candidate_mask[i, j] = True
    
    # Ensure symmetry and zero diagonal
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask

def _post_process_v2(H: np.ndarray, distance_matrix: np.ndarray, 
                    stats: Dict) -> np.ndarray:
    """Final post-processing of heuristic matrix."""
    n = stats['n']
    
    # Ensure minimum connectivity
    min_edges = min(4, n - 1)
    for i in range(n):
        non_zero = np.where(H[i] > 0)[0]
        if len(non_zero) < min_edges:
            # Add nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    # Compute base score
                    inv_dist = 1 / (dists[j] + 1e-8)
                    
                    # Simple regret
                    dists_i = distance_matrix[i].copy()
                    dists_i[i] = dists_i[j] = np.inf
                    best_i = np.min(dists_i)
                    regret_i = (dists[j] - best_i) / (best_i + 1e-8)
                    
                    base_score = inv_dist * (1 / (1 + regret_i))
                    H[i, j] = H[j, i] = base_score
    
    # Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Final normalization with softmax-like scaling
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    # Apply sigmoid to emphasize strong candidates
    H = 1 / (1 + np.exp(-10 * (H - 0.5)))
    
    return H
