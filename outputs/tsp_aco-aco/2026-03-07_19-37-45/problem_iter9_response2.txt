import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced deterministic heuristic for TSP edge prioritization using multi-factor
    geometric combination with adaptive instance-aware weighting.
    Returns heuristic matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Analyze instance characteristics
    stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Compute core heuristic factors
    factors = _compute_core_factors_v2(distance_matrix, stats)
    
    # Step 3: Compute adaptive weights based on instance properties
    weights = _compute_adaptive_weights_v2(stats)
    
    # Step 4: Build candidate mask for efficiency
    candidate_mask = _build_candidate_mask_v2(distance_matrix, stats)
    
    # Step 5: Combine factors using weighted geometric mean
    H = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            if candidate_mask[i, j]:
                combined = (
                    factors['inv_dist'][i, j] ** weights['w_inv'] *
                    factors['regret'][i, j] ** weights['w_regret'] *
                    factors['density'][i, j] ** weights['w_density'] *
                    factors['mst'][i, j] ** weights['w_mst']
                )
                H[i, j] = H[j, i] = combined
    
    # Step 6: Post-processing and sparsification
    H = _post_process_heuristics_v2(H, distance_matrix, stats)
    
    return H

def _analyze_instance_v2(distance_matrix: np.ndarray) -> dict:
    """Compute instance statistics for adaptive parameter tuning."""
    n = distance_matrix.shape[0]
    mask = ~np.eye(n, dtype=bool)
    flat_dists = distance_matrix[mask]
    
    # Compute basic statistics
    mean_dist = np.mean(flat_dists)
    std_dist = np.std(flat_dists)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Estimate cluster tendency using Hopkins-like statistic
    cluster_tendency = _estimate_cluster_tendency_v2(distance_matrix)
    
    # Estimate sparsity using entropy of distance distribution
    sparsity = _estimate_sparsity_v2(distance_matrix)
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'cluster_tendency': cluster_tendency,
        'sparsity': sparsity
    }

def _estimate_cluster_tendency_v2(distance_matrix: np.ndarray) -> float:
    """Estimate whether points form natural clusters (0=uniform, 1=clustered)."""
    n = distance_matrix.shape[0]
    if n < 8:
        return 0.5
    
    # Sample points and compute nearest neighbor distances
    m = min(15, n // 4)
    sample_indices = np.random.choice(n, m, replace=False)
    
    sample_dists = []
    uniform_dists = []
    
    for idx in sample_indices:
        # Distance to nearest neighbor (excluding self)
        dists = distance_matrix[idx].copy()
        dists[idx] = np.inf
        sample_dists.append(np.min(dists))
        
        # Generate uniform random distances
        uniform_dists.append(np.random.exponential(scale=np.mean(distance_matrix) / 2))
    
    # Hopkins statistic approximation
    hopkins = np.sum(sample_dists) / (np.sum(sample_dists) + np.sum(uniform_dists) + 1e-8)
    return max(0.1, min(0.9, hopkins))

def _estimate_sparsity_v2(distance_matrix: np.ndarray) -> float:
    """Estimate how structured/sparse the instance is (0=uniform, 1=structured)."""
    n = distance_matrix.shape[0]
    flat_dists = distance_matrix[~np.eye(n, dtype=bool)]
    
    # Compute histogram and entropy
    hist, _ = np.histogram(flat_dists, bins=min(20, n), density=True)
    hist = np.maximum(hist, 1e-10)
    entropy = -np.sum(hist * np.log(hist))
    
    # Normalize entropy to [0,1] range
    max_entropy = np.log(len(hist))
    normalized_entropy = entropy / max_entropy if max_entropy > 0 else 0.5
    
    # Low entropy = more structured/sparse
    return 1 - normalized_entropy

def _compute_core_factors_v2(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute the four core heuristic factors."""
    n = stats['n']
    factors = {}
    
    # 1. Inverse distance (normalized)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    # Normalize to [0,1] range
    max_val = np.max(inv_dist)
    if max_val > 0:
        inv_dist = inv_dist / max_val
    factors['inv_dist'] = inv_dist
    
    # 2. Two-hop regret analysis with variance adaptation
    factors['regret'] = _compute_regret_scores_v2(distance_matrix, stats)
    
    # 3. Density-based scoring
    factors['density'] = _compute_density_scores_v2(distance_matrix, stats)
    
    # 4. MST-based scoring
    factors['mst'] = _compute_mst_scores_v2(distance_matrix)
    
    return factors

def _compute_regret_scores_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """
    Compute regret scores using two-hop alternative analysis.
    Higher score = edge has poor alternatives (more critical).
    """
    n = stats['n']
    cv = stats['cv_dist']
    regret_scores = np.ones((n, n))
    
    # Pre-compute k-nearest neighbors for efficiency
    k = min(8, max(3, n // 10))
    nearest_neighbors = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest = np.argpartition(dists, k)[:k]
        nearest_neighbors.append(nearest)
    
    for i in range(n):
        for j in range(i + 1, n):
            direct_cost = distance_matrix[i, j]
            
            # Find best two-hop alternative through intermediate nodes
            candidates = np.union1d(nearest_neighbors[i], nearest_neighbors[j])
            candidates = candidates[(candidates != i) & (candidates != j)]
            
            if len(candidates) > 0:
                two_hop_costs = distance_matrix[i, candidates] + distance_matrix[candidates, j]
                best_alt = np.min(two_hop_costs)
                
                # Regret ratio: higher means alternative is much worse
                regret_ratio = best_alt / (direct_cost + 1e-8)
                
                # Apply variance-adaptive scaling
                # In uniform instances (low cv), penalize edges with good alternatives more
                regret_score = regret_ratio ** (1 + cv * 2)
            else:
                regret_score = 2.0  # No alternative available
            
            regret_scores[i, j] = regret_scores[j, i] = regret_score
    
    # Normalize
    max_val = np.max(regret_scores)
    if max_val > 0:
        regret_scores = regret_scores / max_val
    
    np.fill_diagonal(regret_scores, 0)
    return regret_scores

def _compute_density_scores_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute density scores favoring edges in sparse regions."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Estimate local density using k-nearest neighbors
    k = min(6, max(3, n // 8))
    local_sparsity = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k)[:k]
        local_sparsity[i] = np.mean(k_nearest)  # Higher = more sparse
    
    # Normalize sparsity scores
    if np.std(local_sparsity) > 1e-8:
        local_sparsity = (local_sparsity - np.mean(local_sparsity)) / np.std(local_sparsity)
    
    # Create scores: favor edges connecting sparse areas
    for i in range(n):
        for j in range(i + 1, n):
            # Combined sparsity (higher for sparse-sparse connections)
            sparsity_score = np.exp(0.5 * (local_sparsity[i] + local_sparsity[j]))
            density_scores[i, j] = density_scores[j, i] = sparsity_score
    
    # Normalize
    max_val = np.max(density_scores)
    if max_val > 0:
        density_scores = density_scores / max_val
    
    np.fill_diagonal(density_scores, 0)
    return density_scores

def _compute_mst_scores_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Boost edges that appear in the Minimum Spanning Tree."""
    n = distance_matrix.shape[0]
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            # Strong boost for MST edges
            mst_scores[mst_present] = 3.0
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores

def _compute_adaptive_weights_v2(stats: dict) -> dict:
    """Dynamically compute factor weights based on instance characteristics."""
    n = stats['n']
    cv = stats['cv_dist']
    sparsity = stats['sparsity']
    cluster_tendency = stats['cluster_tendency']
    
    # Weight computation with instance adaptation
    weights = {
        'w_inv': 1.0,  # Base inverse distance weight
        
        # Regret weight: higher for uniform instances (low cv)
        'w_regret': 0.4 + 0.3 * (1 - cv),
        
        # Density weight: higher for clustered instances
        'w_density': 0.2 + 0.4 * cluster_tendency,
        
        # MST weight: higher for smaller instances, adjusted by sparsity
        'w_mst': max(0.1, 0.6 - 0.02 * n) * (1 + sparsity * 0.5)
    }
    
    # Normalize weights to sum to approximately 4
    total = sum(weights.values())
    for key in weights:
        weights[key] = weights[key] / total * 4
    
    return weights

def _build_candidate_mask_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Build mask of promising candidate edges for efficiency."""
    n = stats['n']
    
    # Adaptive candidate list size based on instance properties
    base_k = min(12, max(4, int(np.log2(n) * 2.5)))
    # More candidates for sparse/structured instances
    k = int(base_k * (1 + stats['sparsity'] * 0.5))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Select k-nearest neighbors
        if k < len(dists):
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Make symmetric
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask

def _post_process_heuristics_v2(H: np.ndarray, distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Post-process heuristic matrix with sparsification and normalization."""
    n = stats['n']
    
    # Ensure minimum connectivity for each node
    min_edges = min(3, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i])[0]
        if len(non_zero) < min_edges:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            for j in nearest:
                if H[i, j] == 0:
                    base_score = 1 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    # Aggressive sparsification: zero out weak edges
    if n > 10:
        threshold = np.percentile(H[H > 0], 30)  # Keep top 70%
        H[H < threshold] = 0
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    np.fill_diagonal(H, 0)
    return H
