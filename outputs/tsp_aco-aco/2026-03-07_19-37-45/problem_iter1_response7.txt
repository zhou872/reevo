import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection.
    Returns a sparse matrix indicating edge promise with the same shape as input.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (handle zeros to avoid division by zero)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.isnan(inv_dist)] = 0
    
    # 2. Nearest neighbor factor (sparse connectivity)
    nn_factor = np.zeros_like(distance_matrix)
    for i in range(n):
        # Find k nearest neighbors (excluding self)
        k = max(3, n // 10)  # adaptive k based on problem size
        indices = np.argsort(distance_matrix[i])[1:k+1]
        nn_factor[i, indices] = 1 / distance_matrix[i, indices]
    
    # 3. Two-hop path density (local cluster potential)
    density_factor = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Find common neighbors that could form triangles
                common = np.where((distance_matrix[i] < np.inf) & 
                                  (distance_matrix[j] < np.inf) & 
                                  (np.arange(n) != i) & (np.arange(n) != j))[0]
                if len(common) > 0:
                    # Average path length through common neighbors
                    path_lengths = distance_matrix[i, common] + distance_matrix[common, j]
                    density_factor[i, j] = 1 / (np.mean(path_lengths) + 1e-10)
    
    # 4. Normalize and combine factors
    def safe_normalize(arr):
        if np.max(arr) > 0:
            return arr / np.max(arr)
        return arr
    
    inv_dist_norm = safe_normalize(inv_dist)
    nn_factor_norm = safe_normalize(nn_factor)
    density_factor_norm = safe_normalize(density_factor)
    
    # Weighted combination
    alpha, beta, gamma = 0.5, 0.3, 0.2  # tuned weights
    combined = (alpha * inv_dist_norm + 
                beta * nn_factor_norm + 
                gamma * density_factor_norm)
    
    # 5. Sparsification: keep only promising edges
    # Strategy 1: Keep top edges per node
    sparse_matrix = np.zeros_like(combined)
    for i in range(n):
        # Keep top m edges for each node
        m = max(4, n // 5)
        indices = np.argsort(combined[i])[-m:]
        sparse_matrix[i, indices] = combined[i, indices]
    
    # Strategy 2: Remove edges that are too weak globally
    threshold = np.percentile(sparse_matrix[sparse_matrix > 0], 30)
    sparse_matrix[sparse_matrix < threshold] = 0
    
    # Strategy 3: Ensure each node has at least 2 connections (degree constraint)
    for i in range(n):
        non_zero = np.where(sparse_matrix[i] > 0)[0]
        if len(non_zero) < 2:
            # Add back the strongest missing edges
            missing = np.argsort(combined[i])[-2:]
            sparse_matrix[i, missing] = combined[i, missing]
    
    # Ensure symmetry for undirected TSP
    sparse_matrix = np.maximum(sparse_matrix, sparse_matrix.T)
    
    # Final cleanup: set diagonal to zero
    np.fill_diagonal(sparse_matrix, 0)
    
    return sparse_matrix
