import numpy as np
from scipy.spatial.distance import squareform, pdist

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors.
    Returns a matrix of the same shape indicating edge desirability.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        # Trivial case for very small instances
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Initialize heuristic matrix
    H = np.zeros((n, n))
    
    # 1. Base inverse distance (avoid division by zero)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Candidate list construction (k-nearest neighbors with adaptive k)
    k = min(max(5, int(np.log2(n))), n - 1)  # Adaptive k based on instance size
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k+1 smallest distances (including self)
        indices = np.argpartition(distance_matrix[i], k+1)[:k+1]
        # Remove self if present
        indices = indices[indices != i][:k]
        candidate_mask[i, indices] = True
    
    # Make candidate mask symmetric (edge is candidate if either endpoint considers it)
    candidate_mask = candidate_mask | candidate_mask.T
    
    # 3. Local regret heuristic: how much worse is the second-best alternative?
    regret_scores = np.zeros((n, n))
    for i in range(n):
        # Get distances from node i to all others
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf  # Ignore self
        
        # Find best and second-best distances for each node
        sorted_indices = np.argsort(dists_i)
        if len(sorted_indices) >= 2:
            best_idx = sorted_indices[0]
            second_best_val = dists_i[sorted_indices[1]]
            
            # Regret for the best edge: difference between second best and actual
            if best_idx < n:
                regret = max(0, second_best_val - dists_i[best_idx])
                regret_scores[i, best_idx] = regret / (dists_i[best_idx] + 1e-8)
    
    # Make regret symmetric
    regret_scores = (regret_scores + regret_scores.T) / 2
    
    # 4. Neighborhood density factor: favor edges in sparse regions
    density_weights = np.ones((n, n))
    
    # Calculate local density around each node
    local_densities = np.zeros(n)
    for i in range(n):
        # Average distance to 3 nearest neighbors as density measure
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest_dists = np.partition(dists, 3)[:3]
        local_densities[i] = np.mean(nearest_dists)
    
    # Normalize densities
    if np.std(local_densities) > 1e-8:
        local_densities = (local_densities - np.mean(local_densities)) / np.std(local_densities)
    
    # Assign weights based on combined density of endpoints
    for i in range(n):
        for j in range(n):
            if i != j:
                # Favor edges connecting sparse regions
                combined_density = local_densities[i] + local_densities[j]
                density_weights[i, j] = 1 / (1 + np.exp(-combined_density))
    
    # 5. Minimum spanning tree bias
    mst_bias = np.ones((n, n))
    
    # Simple approximation: edges that are in the Euclidean MST (if coordinates were available)
    # Since we only have distances, we use a greedy nearest-neighbor chain
    if n <= 100:  # Only for reasonably sized instances
        visited = np.zeros(n, dtype=bool)
        current = 0
        visited[current] = True
        
        for _ in range(n - 1):
            # Find unvisited node with minimum distance to any visited node
            unvisited = ~visited
            if not np.any(unvisited):
                break
                
            # Get minimum distance from visited to unvisited nodes
            min_dist = np.inf
            best_i, best_j = -1, -1
            
            for i in np.where(visited)[0]:
                for j in np.where(unvisited)[0]:
                    if distance_matrix[i, j] < min_dist:
                        min_dist = distance_matrix[i, j]
                        best_i, best_j = i, j
            
            if best_i != -1 and best_j != -1:
                # Boost edges in the approximate MST
                mst_bias[best_i, best_j] = 2.0
                mst_bias[best_j, best_i] = 2.0
                visited[best_j] = True
    
    # 6. Combine all factors with weighted geometric mean
    alpha, beta, gamma, delta = 1.0, 0.3, 0.2, 0.1  # Weight parameters
    
    for i in range(n):
        for j in range(n):
            if i != j and candidate_mask[i, j]:
                # Multiplicative combination of factors
                base = inv_dist[i, j] ** alpha
                regret_factor = (1 + regret_scores[i, j]) ** beta
                density_factor = density_weights[i, j] ** gamma
                mst_factor = mst_bias[i, j] ** delta
                
                H[i, j] = base * regret_factor * density_factor * mst_factor
    
    # 7. Normalize to make values more comparable across instances
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    # Ensure symmetry and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    return H
