```python
import numpy as np
from typing import Dict, Tuple, List
import heapq
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray, local_search_iterations: int = 3) -> np.ndarray:
    """
    Advanced TSP edge heuristics with multi-step regret computation and 
    iterative local search feedback.
    
    Core innovations:
    1. Multi-step Regret with Priority Queue: Computes k-step (k=2,3) regret 
       using efficient dynamic programming with pruning via priority queues
    2. Dynamic Weight Adaptation via Instance Analysis: Uses comprehensive 
       instance characterization to adapt factor weights non-linearly
    3. Iterative Refinement through Local Search Feedback: Incorporates 
       simple 2-opt improvement feedback to adjust edge scores
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances between nodes
    local_search_iterations : int
        Number of local search feedback iterations (default: 3)
        
    Returns:
    --------
    np.ndarray
        Matrix of heuristic scores where higher values indicate more promising edges
    """
    
    def analyze_instance(distance_matrix: np.ndarray) -> Dict:
        """Comprehensive instance analysis for adaptive parameter tuning."""
        n = distance_matrix.shape[0]
        stats = {}
        
        # Remove diagonal
        mask = ~np.eye(n, dtype=bool)
        flat_dists = distance_matrix[mask]
        
        # Basic statistics
        stats['n'] = n
        stats['mean_dist'] = np.mean(flat_dists)
        stats['std_dist'] = np.std(flat_dists)
        stats['cv_dist'] = stats['std_dist'] / (stats['mean_dist'] + 1e-10)
        
        # Nearest neighbor analysis
        nn_distances = []
        second_nn_distances = []
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            sorted_dists = np.sort(dists)
            nn_distances.append(sorted_dists[0])
            second_nn_distances.append(sorted_dists[1])
        
        nn_distances = np.array(nn_distances)
        second_nn_distances = np.array(second_nn_distances)
        
        stats['nn_regularity'] = 1.0 - (np.std(nn_distances) / (np.mean(nn_distances) + 1e-10))
        stats['nn_gap_ratio'] = np.mean(second_nn_distances / (nn_distances + 1e-10))
        
        # Cluster tendency (based on nearest neighbor ratios)
        stats['cluster_tendency'] = np.clip(
            (stats['nn_gap_ratio'] - 1.2) / 0.8, 0.0, 1.0
        )
        
        # Distance distribution skewness
        from scipy.stats import skew
        stats['skewness'] = skew(flat_dists) if len(flat_dists) > 0 else 0
        
        return stats
    
    def compute_multi_step_regret(distance_matrix: np.ndarray, 
                                  stats: Dict) -> np.ndarray:
        """
        Multi-step regret computation using priority queue-based DP.
        
        For each edge (i,j), computes the minimum cost 2-step and 3-step
        alternative paths, then combines them into a regret score.
        """
        n = stats['n']
        regret_matrix = np.ones((n, n))
        
        # Determine optimal step based on instance size
        max_steps = 2 if n <= 50 else 2  # Cap at 2 steps for efficiency
        
        # Precompute sorted neighbors for each node
        sorted_neighbors = []
        neighbor_distances = []
        
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            sorted_idx = np.argsort(dists)
            sorted_neighbors.append(sorted_idx)
            neighbor_distances.append(dists[sorted_idx])
        
        # Compute 2-step regret efficiently
        two_step_regret = np.ones((n, n))
        
        for i in range(n):
            # Get top K neighbors for efficiency
            K = min(15, n - 1)
            top_neighbors_i = sorted_neighbors[i][:K]
            top_dists_i = neighbor_distances[i][:K]
            
            for j in range(i + 1, n):
                if i == j:
                    continue
                
                direct_cost = distance_matrix[i, j]
                
                # 2-step alternatives through intermediate nodes
                best_two_step = np.inf
                
                # Check alternatives through i's neighbors
                for idx_k, k in enumerate(top_neighbors_i):
                    if k == j:
                        continue
                    
                    two_step_cost = top_dists_i[idx_k] + distance_matrix[k, j]
                    if two_step_cost < best_two_step:
                        best_two_step = two_step_cost
                
                # Also check through j's neighbors
                top_neighbors_j = sorted_neighbors[j][:K]
                top_dists_j = neighbor_distances[j][:K]
                
                for idx_k, k in enumerate(top_neighbors_j):
                    if k == i:
                        continue
                    
                    two_step_cost = distance_matrix[i, k] + top_dists_j[idx_k]
                    if two_step_cost < best_two_step:
                        best_two_step = two_step_cost
                
                # Compute regret score
                if best_two_step < np.inf and best_two_step > 0:
                    regret = (best_two_step - direct_cost) / best_two_step
                    # Convert regret to score: positive regret is good
                    score = 1.0 + max(0, regret)
                    # Apply non-linear transformation based on instance characteristics
                    if stats['cluster_tendency'] > 0.7:
                        # More aggressive scoring for clustered instances
                        score = score ** 1.5
                    two_step_regret[i, j] = two_step_regret[j, i] = score
        
        # Compute 3-step regret for larger instances with low regularity
        three_step_regret = np.ones((n, n))
        
        if n <= 100 and stats['nn_regularity'] < 0.7:
            # Use simplified 3-step computation for manageable instances
            for i in range(n):
                for j in range(i + 1, n):
                    if i == j:
                        continue
                    
                    direct_cost = distance_matrix[i, j]
                    best_three_step = np.inf
                    
                    # Sample limited intermediate nodes
                    K_sample = min(5, n - 2)
                    
                    # Get diverse intermediate nodes
                    mid_indices = np.random.choice(
                        [k for k in range(n) if k not in [i, j]],
                        size=min(K_sample, n - 2),
                        replace=False
                    )
                    
                    for k in mid_indices:
                        # Get l candidates from k's neighbors
                        L_sample = min(3, n - 3)
                        l_candidates = sorted_neighbors[k][:L_sample]
                        
                        for l in l_candidates:
                            if l in [i, j, k]:
                                continue
                            
                            three_step_cost = (distance_matrix[i, k] + 
                                               distance_matrix[k, l] + 
                                               distance_matrix[l, j])
                            if three_step_cost < best_three_step:
                                best_three_step = three_step_cost
                    
                    if best_three_step < np.inf and best_three_step > 0:
                        regret = (best_three_step - direct_cost) / best_three_step
                        score = 1.0 + max(0, min(regret, 1.0)) * 0.5  # Reduced weight
                        three_step_regret[i, j] = three_step_regret[j, i] = score
        
        # Combine multi-step regrets
        regret_matrix = two_step_regret
        
        if n <= 100:
            # Blend with 3-step regret
            blend_weight = 0.3 * (1 - stats['nn_regularity'])
            regret_matrix = (1 - blend_weight) * two_step_regret + blend_weight * three_step_regret
        
        np.fill_diagonal(regret_matrix, 0)
        return regret_matrix
    
    def compute_degree_aware_penalty(distance_matrix: np.ndarray, 
                                     inv_dist: np.ndarray,
                                     stats: Dict) -> np.ndarray:
        """
        Sophisticated degree penalization with instance-aware thresholds.
        """
        n = stats['n']
        penalty_matrix = np.ones((n, n))
        
        # Adaptive threshold based on instance characteristics
        if stats['cv_dist'] < 0.3:
            # Uniform distances: use stricter threshold
            percentile = 70
        elif stats['cluster_tendency'] > 0.6:
            # Clustered: use more lenient threshold
            percentile = 60
        else:
            percentile = 65
        
        # Compute node degrees based on promising connections
        node_degrees = np.zeros(n)
        
        for i in range(n):
            inv_dists = inv_dist[i].copy()
            inv_dists[i] = -np.inf
            
            # Dynamic threshold based on instance size
            dynamic_percentile = percentile + 5 * (1 - stats['nn_regularity'])
            threshold = np.percentile(inv_dists[inv_dists > -np.inf], dynamic_percentile)
            
            node_degrees[i] = np.sum(inv_dists >= threshold)
        
        # Normalize degrees
        if np.max(node_degrees) > 0:
            node_degrees = node_degrees / np.max(node_degrees)
        
        # Compute penalty using asymmetric sigmoid
        for i in range(n):
            for j in range(i + 1, n):
                deg_sum = node_degrees[i] + node_degrees[j]
                
                # Instance-adaptive sigmoid parameters
                if stats['cluster_tendency'] > 0.7:
                    # Clustered: less penalty for high-degree nodes
                    k = 3.0
                    x0 = 1.2
                else:
                    k = 4.0
                    x0 = 1.0
                
                penalty = 1.0 / (1.0 + np.exp(k * (deg_sum - x0)))
                penalty_matrix[i, j] = penalty_matrix[j, i] = penalty
        
        return penalty_matrix
    
    def compute_bypass_penalty(distance_matrix: np.ndarray,
                               stats: Dict) -> np.ndarray:
        """
        Computes penalty for edges that can be efficiently bypassed.
        """
        n = stats['n']
        penalty_matrix = np.ones((n, n))
        
        # Use instance-adaptive thresholds
        if stats['nn_regularity'] > 0.8:
            # Regular instances: stricter bypass detection
            strong_threshold = 0.65
            moderate_threshold = 0.85
        else:
            strong_threshold = 0.6
            moderate_threshold = 0.8
        
        # Efficient computation using matrix operations for 2-hop bypass
        for i in range(n):
            # Get distances from i to all other nodes
            dist_i = distance_matrix[i]
            
            for j in range(i + 1, n):
                direct = distance_matrix[i, j]
                
                # Find minimum 2-hop path through any k
                # Use vectorized computation
                two_hop_costs = dist_i + distance_matrix[:, j]
                two_hop_costs[i] = np.inf  # Exclude i as intermediate
                two_hop_costs[j] = np.inf  # Exclude j as intermediate
                
                best_two_hop = np.min(two_hop_costs)
                
                if best_two_hop < np.inf:
                    ratio = best_two_hop / (direct + 1e-10)
                    
                    if ratio < strong_threshold:
                        penalty_matrix[i, j] = penalty_matrix[j, i] = 0.4
                    elif ratio < moderate_threshold:
                        penalty_matrix[i, j] = penalty_matrix[j, i] = 0.7
        
        return penalty_matrix
    
    def compute_adaptive_weights(stats: Dict) -> Dict:
        """Compute factor weights using comprehensive instance analysis."""
        weights = {}
        
        # Base weights adjusted by multiple instance features
        w_inv_base = 1.2
        
        # Adjust for distance distribution
        if stats['cv_dist'] < 0.2:
            w_inv_base *= 0.9  # Less important for uniform distances
        elif stats['cv_dist'] > 0.8:
            w_inv_base *= 1.3  # More important for varied distances
        
        weights['w_inv'] = w_inv_base
        
        # Regret weight based on regularity and cluster tendency
        w_regret_base = 0.8 + 0.4 * (1 - stats['nn_regularity'])
        if stats['cluster_tendency'] > 0.6:
            w_regret_base *= 1.2  # More important for clustered instances
        weights['w_regret'] = w_regret_base
        
        # Degree penalty weight
        w_degree_base = 0.4 + 0.3 * stats['nn_regularity']
        if stats['skewness'] > 0.5:
            w_degree_base *= 1.2  # More important for skewed distributions
        weights['w_degree'] = w_degree_base
        
        # Bypass penalty weight
        w_bypass_base = 0.5 + 0.3 * (1 - stats['cv_dist'])
        weights['w_bypass'] = w_bypass_base
        
        # Normalize to maintain balance
        total = sum(weights.values())
        target_total = 4.0
        scale = target_total / total
        
        for key in weights:
            weights[key] *= scale
        
        return weights
    
    def build_candidate_mask(distance_matrix: np.ndarray,
                            stats: Dict) -> np.ndarray:
        """
        Builds adaptive candidate edge mask with exploration- exploitation balance.
        """
        n = stats['n']
        mask = np.zeros((n, n), dtype=bool)
        
        # Base number of nearest neighbors
        if n <= 20:
            k_base = min(8, n - 1)
        elif n <= 100:
            k_base = min(12, n - 1)
        else:
            k_base = min(15, n - 1)
        
        # Adjust based on instance characteristics
        k = int(k_base * (1.3 - 0.4 * stats['nn_regularity']))
        k = max(3, min(k, n - 1))
        
        # Include k-nearest neighbors for each node
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest_indices = np.argpartition(dists, k)[:k]
            mask[i, nearest_indices] = True
        
        # Add exploration edges based on instance properties
        if stats['cluster_tendency'] < 0.8:  # Only for non-clustered instances
            n_possible = n * (n - 1) // 2
            n_current = np.sum(mask[np.triu_indices(n, 1)])
            
            # Add random long-distance connections for exploration
            exploration_ratio = 0.05 * (1 - stats['cluster_tendency'])
            n_explore = int(exploration_ratio * (n_possible - n_current))
            
            if n_explore > 0:
                # Get all non-masked edges
                triu_indices = np.triu_indices(n, 1)
                non_masked = ~mask[triu_indices]
                
                if np.any(non_masked):
                    candidate_idx = np.where(non_masked)[0]
                    
                    # Weight by distance (favor medium-distance edges for exploration)
                    candidate_dists = distance_matrix[triu_indices[0][candidate_idx],
                                                      triu_indices[1][candidate_idx]]
                    
                    # Avoid the very shortest and longest edges
                    dist_ranks = np.argsort(candidate_dists)
                    mid_start = len(dist_ranks) // 4
                    mid_end = 3 * len(dist_ranks) // 4
                    mid_indices = dist_ranks[mid_start:mid_end]
                    
                    if len(mid_indices) > 0:
                        selected = np.random.choice(
                            mid_indices,
                            size=min(n_explore, len(mid_indices)),
                            replace=False
                        )
                        
                        for idx in selected:
                            i, j = triu_indices[0][candidate_idx[idx]], triu_indices[1][candidate_idx[idx]]
                            mask[i, j] = mask[j, i] = True
        
        np.fill_diagonal(mask, False)
        return mask
    
    def apply_local_search_feedback(H: np.ndarray,
                                   distance_matrix: np.ndarray,
                                   stats: Dict) -> np.ndarray:
        """
        Apply simple local search feedback to refine heuristic scores.
        """
        n = stats['n']
        
        # Only apply for moderately sized instances
        if n > 200 or n < 10:
            return H
        
        H_refined = H.copy()
        
        # Perform simple 2-opt moves on a few random paths
        n_paths = min(10, n // 2)
        
        for _ in range(n_paths):
            # Generate a random path
            path = np.random.permutation(n)
            
            # Try a few 2-opt moves
            for i in range(n - 1):
                for j in range(i + 2, min(i + 6, n)):  # Limited lookahead
                    # Current edges: (path[i], path[i+1]) and (path[j], path[(j+1)%n])
                    # Proposed edges: (path[i], path[j]) and (path[i+1], path[(j+1)%n])
                    
                    current_cost = (distance_matrix[path[i], path[i+1]] +
                                   distance_matrix[path[j], path[(j+1)%n]])
                    proposed_cost = (distance_matrix[path[i], path[j]] +
                                    distance_matrix[path[i+1], path[(j+1)%n]])
                    
                    if proposed_cost < current_cost:
                        # Good swap found: boost the new edges
                        improvement = (current_cost - proposed_cost) / (current_cost + 1e-10)
                        
                        # Apply feedback to heuristic scores
                        boost_factor = 1.0 + 0.5 * improvement
                        
                        # New edges
                        H_refined[path[i], path[j]] *= boost_factor
                        H_refined[path[j], path[i]] *= boost_factor
                        H_refined[path[i+1], path[(j+1)%n]] *= boost_factor
                        H_refined[path[(j+1)%n], path[i+1]] *= boost_factor
        
        return H_refined
    
    def post_process(H: np.ndarray,
                    distance_matrix: np.ndarray,
                    stats: Dict) -> np.ndarray:
        """Final post-processing of the heuristic matrix."""
        n = stats['n']
        
        # Ensure minimum connectivity
        min_connections = min(3, n - 1)
        
        for i in range(n):
            row = H[i].copy()
            row[i] = -np.inf
            non_zero = np.where(row > 0)[0]
            
            if len(non_zero) < min_connections:
                # Add missing connections based on distance
                dists = distance_matrix[i].copy()
                dists[i] = np.inf
                
                # Sort by distance and add missing connections
                sorted_indices = np.argsort(dists)
                added = 0
                
                for j in sorted_indices:
                    if H[i, j] == 0:
                        # Add with distance-based score
                        H[i, j] = H[j, i] = 1.0 / (dists[j] + 1e-8)
                        added += 1
                        if added >= min_connections - len(non_zero):
                            break
        
        # Ensure symmetry
        H = (H + H.T) / 2
        np.fill_diagonal(H, 0)
        
        # Normalize
        max_val = np.max(H)
        if max_val > 0:
            H = H / max_val
        
        # Apply soft threshold
        threshold = 0.01 + 0.04 * np.exp(-n / 25.0)
        H[H < threshold] = 0
        
        return H
    
    # Main execution
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        result = 1 / (distance_matrix + np.eye(n) * 1e-8)
        np.fill_diagonal(result, 0)
        return result
    
    # Step 1: Analyze instance
    stats = analyze_instance(distance_matrix)
    
    # Step 2: Compute base components
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    # Step 3: Compute heuristic factors
    regret = compute_multi_step_regret(distance_matrix, stats)
    degree_penalty = compute_degree_aware_penalty(distance_matrix, inv_dist, stats)
    bypass_penalty = compute_bypass_penalty(distance_matrix, stats)
    
    # Step 4: Build candidate mask
    candidate_mask = build_candidate_mask(distance_matrix, stats)
    
    # Step 5: Compute adaptive weights
    weights = compute_adaptive_weights(stats)
    
    # Step 6: Initialize heuristic matrix
    H = np.zeros((n, n))
    
    # Step 7: Combine factors for candidate edges
    for i in range(n):
        neighbors = np.where(candidate_mask[i])[0]
        neighbors = neighbors[neighbors > i]  # Upper triangle only
        
        for j in neighbors:
            combined = (
                (inv_dist[i, j] ** weights['w_inv']) *
                (regret[i, j] ** weights['w_regret']) *
                (degree_penalty[i, j] ** weights['w_degree']) *
                (bypass_penalty[i, j] ** weights['w_bypass'])
            )
            H[i, j] = H[j, i] = combined
    
    # Step 8: Apply iterative local search feedback
    for iteration in range(local_search_iterations):
        H = apply_local_search_feedback(H, distance_matrix, stats)
    
    # Step 9: Final post-processing
    H = post_process(H, distance_matrix, stats)
    
    return H
```
