import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with adaptive multi-hop regret scoring.
    Returns a matrix of the same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Analyze instance characteristics
    stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Compute candidate mask for sparsification
    candidate_mask = _build_candidate_mask_v2(distance_matrix, stats)
    
    # Step 3: Compute adaptive multi-hop regret scores for candidate edges
    regret_scores = _compute_multi_hop_regret_v2(distance_matrix, stats, candidate_mask)
    
    # Step 4: Compute base inverse distance scores
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    # Step 5: Adaptive combination with variance-based weighting
    H = np.zeros((n, n))
    
    # Get adaptive weights based on instance characteristics
    weights = _compute_adaptive_weights_v2(stats)
    
    for i in range(n):
        for j in range(i + 1, n):
            if candidate_mask[i, j]:
                # Combine factors with adaptive weighting
                base_score = inv_dist[i, j] ** weights['w_inv']
                regret_score = regret_scores[i, j] ** weights['w_regret']
                combined = base_score * regret_score
                H[i, j] = H[j, i] = combined
    
    # Step 6: Post-processing
    H = _post_process_v2(H, distance_matrix, stats, candidate_mask)
    
    return H

def _analyze_instance_v2(distance_matrix: np.ndarray) -> dict:
    """Analyze instance characteristics for adaptive parameter tuning."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal for statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    stats = {
        'n': n,
        'mean_dist': np.mean(dist_no_diag),
        'std_dist': np.std(dist_no_diag),
        'cv_dist': np.std(dist_no_diag) / (np.mean(dist_no_diag) + 1e-8),
        'regularity': _compute_regularity(distance_matrix),
        'density_ratio': _compute_density_ratio(distance_matrix)
    }
    
    return stats

def _compute_regularity(distance_matrix: np.ndarray) -> float:
    """Measure instance regularity (how uniform distances are)."""
    n = distance_matrix.shape[0]
    
    # Compute coefficient of variation for each node's distances to others
    cv_per_node = []
    for i in range(n):
        dists = distance_matrix[i]
        # Remove self-distance
        dists = dists[dists != 0]
        if len(dists) > 1:
            cv = np.std(dists) / (np.mean(dists) + 1e-8)
            cv_per_node.append(cv)
    
    if len(cv_per_node) == 0:
        return 1.0
    
    avg_cv = np.mean(cv_per_node)
    # Convert to regularity measure (low CV = high regularity)
    regularity = 1.0 / (1.0 + avg_cv)
    return np.clip(regularity, 0.1, 1.0)

def _compute_density_ratio(distance_matrix: np.ndarray) -> float:
    """Compute ratio of local vs global density."""
    n = distance_matrix.shape[0]
    if n < 4:
        return 0.5
    
    # Global average distance
    global_avg = np.mean(distance_matrix[~np.eye(n, dtype=bool)])
    
    # Local density (nearest neighbor distances)
    local_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        local_dists.append(np.min(dists))
    
    local_avg = np.mean(local_dists)
    
    # Ratio: <1 means points are clustered, >1 means evenly spaced
    ratio = local_avg / (global_avg + 1e-8)
    return np.clip(ratio, 0.5, 2.0)

def _build_candidate_mask_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Build mask of candidate edges based on adaptive thresholds."""
    n = stats['n']
    
    # Adaptive number of candidates based on instance size and regularity
    base_k = min(10, max(3, int(np.log2(n) * 2)))
    # More candidates for regular instances, fewer for irregular
    k = int(base_k * (1.5 - 0.5 * stats['regularity']))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get distances to all other nodes
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Get indices of k smallest distances
        if k < len(dists):
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Make symmetric and zero diagonal
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask

def _compute_multi_hop_regret_v2(distance_matrix: np.ndarray, stats: dict, 
                                 candidate_mask: np.ndarray) -> np.ndarray:
    """
    Compute adaptive multi-hop regret scores for candidate edges.
    Implements features: Simplified Regret Scoring, Adaptive Multi-Hop Regret,
    and Variance-Adaptive Regret Scoring.
    """
    n = stats['n']
    regret_scores = np.ones((n, n))
    
    # Adaptive parameters based on instance characteristics
    max_hops = min(3, max(2, int(4 - 2 * stats['regularity'])))
    variance_factor = 1.0 / (1.0 + stats['cv_dist'])
    
    for i in range(n):
        for j in range(i + 1, n):
            if not candidate_mask[i, j]:
                regret_scores[i, j] = 0
                continue
                
            direct_cost = distance_matrix[i, j]
            
            # Find best alternative paths
            best_alternative = float('inf')
            
            # 2-hop alternatives: i -> k -> j
            for k in range(n):
                if k == i or k == j:
                    continue
                if candidate_mask[i, k] and candidate_mask[k, j]:
                    alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                    if alt_cost < best_alternative:
                        best_alternative = alt_cost
            
            # 3-hop alternatives if allowed and if 2-hop wasn't good enough
            if max_hops >= 3 and best_alternative / direct_cost < 1.5:
                for k in range(n):
                    if k == i or k == j:
                        continue
                    for l in range(n):
                        if l == i or l == j or l == k:
                            continue
                        if (candidate_mask[i, k] and candidate_mask[k, l] and 
                            candidate_mask[l, j]):
                            alt_cost = (distance_matrix[i, k] + 
                                       distance_matrix[k, l] + 
                                       distance_matrix[l, j])
                            if alt_cost < best_alternative:
                                best_alternative = alt_cost
            
            # Compute regret ratio
            if best_alternative < float('inf'):
                regret_ratio = best_alternative / (direct_cost + 1e-8)
                
                # Apply adaptive penalty/bonus based on ratio
                if regret_ratio >= 1.0:
                    # Alternative is worse - bonus for direct edge
                    score = 1.0 + (regret_ratio - 1.0) * variance_factor
                else:
                    # Alternative is better - penalty for direct edge
                    penalty_factor = 0.5 + 0.5 * stats['regularity']
                    score = 1.0 - (1.0 - regret_ratio) * penalty_factor
                
                # Apply variance-adaptive scaling
                score = score ** variance_factor
                
                regret_scores[i, j] = score
            else:
                # No alternative found - moderate bonus
                regret_scores[i, j] = 1.2
    
    # Symmetrize and ensure non-negative
    regret_scores = (regret_scores + regret_scores.T) / 2
    regret_scores = np.maximum(regret_scores, 0)
    np.fill_diagonal(regret_scores, 0)
    
    return regret_scores

def _compute_adaptive_weights_v2(stats: dict) -> dict:
    """Compute adaptive weights based on instance characteristics."""
    n = stats['n']
    cv = stats['cv_dist']
    regularity = stats['regularity']
    density_ratio = stats['density_ratio']
    
    # Base weights that adapt to instance characteristics
    weights = {
        'w_inv': 1.0,  # Base inverse distance weight
        
        # Regret weight: higher for irregular instances with clear alternatives
        'w_regret': 0.5 + 0.5 * (1.0 - regularity),
    }
    
    # Adjust based on density pattern
    if density_ratio < 0.8:  # Clustered instances
        weights['w_regret'] *= 1.2
    elif density_ratio > 1.2:  # Uniform instances
        weights['w_regret'] *= 0.8
    
    # Normalize to maintain reasonable scale
    total = sum(weights.values())
    for key in weights:
        weights[key] = weights[key] / total * 2.0
    
    return weights

def _post_process_v2(H: np.ndarray, distance_matrix: np.ndarray, 
                     stats: dict, candidate_mask: np.ndarray) -> np.ndarray:
    """Post-process the heuristic matrix."""
    n = stats['n']
    
    # Ensure minimum connectivity for each node
    min_edges = min(3, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i] > 0)[0]
        if len(non_zero) < min_edges:
            # Add nearest neighbors not already in candidates
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    base_score = 1 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    # Apply candidate mask (set non-candidate edges to 0)
    H = H * candidate_mask
    
    # Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Final normalization to [0, 1] range
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
