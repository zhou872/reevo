import numpy as np
from typing import Dict, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with improved mathematical strategies.
    
    This function implements a sophisticated scoring mechanism that prioritizes edges based on:
    1. Two-step lookahead regret estimation
    2. Degree-aware penalization to balance connectivity
    3. Multiplicative combination of factors for non-linear interactions
    4. Adaptive parameter tuning based on instance characteristics
    
    Args:
        distance_matrix: Symmetric distance matrix of shape (n, n)
    
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges
    """
    
    def _analyze_instance(distance_matrix: np.ndarray) -> Dict:
        """Analyze instance characteristics for adaptive parameter tuning."""
        n = distance_matrix.shape[0]
        dist_no_diag = distance_matrix.copy()
        np.fill_diagonal(dist_no_diag, np.inf)
        
        # Basic statistics
        stats = {
            'n': n,
            'mean_dist': np.mean(dist_no_diag[dist_no_diag < np.inf]),
            'std_dist': np.std(dist_no_diag[dist_no_diag < np.inf]),
        }
        
        # Regularity estimation
        nn_dists = []
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nn_dists.append(np.min(dists))
        
        nn_dists = np.array(nn_dists)
        stats['regularity'] = 1.0 - (np.std(nn_dists) / (np.mean(nn_dists) + 1e-10))
        stats['regularity'] = np.clip(stats['regularity'], 0.1, 0.9)
        
        # Variation coefficient
        stats['cv_dist'] = stats['std_dist'] / (stats['mean_dist'] + 1e-10)
        
        return stats
    
    def _compute_two_step_regret(distance_matrix: np.ndarray) -> np.ndarray:
        """
        Two-step lookahead regret estimation.
        
        For each edge (i,j), computes the difference between:
        1. Best alternative two-step path through a third node
        2. Direct cost plus best continuation from j
        """
        n = distance_matrix.shape[0]
        regret = np.ones((n, n))
        
        for i in range(n):
            # Get distances from i (excluding self)
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            
            for j in range(n):
                if i == j:
                    continue
                
                # Direct cost + best continuation from j (excluding i)
                dists_j = distance_matrix[j].copy()
                dists_j[i] = np.inf
                dists_j[j] = np.inf
                best_from_j = np.min(dists_j)
                
                direct_path = distance_matrix[i, j] + best_from_j
                
                # Best alternative two-step path through k (k != i,j)
                best_alternative = np.inf
                for k in range(n):
                    if k == i or k == j:
                        continue
                    alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                    best_alternative = min(best_alternative, alt_cost)
                
                # Regret calculation
                if best_alternative > 0 and best_alternative < np.inf:
                    regret_ij = (best_alternative - direct_path) / best_alternative
                    # Convert to score: higher regret -> higher score
                    regret[i, j] = 1.0 + max(0, regret_ij)
                else:
                    regret[i, j] = 1.0
        
        np.fill_diagonal(regret, 0)
        return regret
    
    def _compute_degree_penalty(distance_matrix: np.ndarray, inv_dist: np.ndarray) -> np.ndarray:
        """
        Degree-aware penalization to balance node connectivity.
        
        Scales edge heuristics inversely with the count of promising connections per node.
        """
        n = distance_matrix.shape[0]
        penalty = np.ones((n, n))
        
        # Count promising connections per node (top 25% of inverse distances)
        for i in range(n):
            inv_dists_i = inv_dist[i].copy()
            inv_dists_i[i] = -np.inf
            threshold_i = np.percentile(inv_dists_i[inv_dists_i > -np.inf], 75)
            degree_i = np.sum(inv_dists_i >= threshold_i)
            
            for j in range(n):
                if i == j:
                    continue
                
                inv_dists_j = inv_dist[j].copy()
                inv_dists_j[j] = -np.inf
                threshold_j = np.percentile(inv_dists_j[inv_dists_j > -np.inf], 75)
                degree_j = np.sum(inv_dists_j >= threshold_j)
                
                # Penalize edges where both nodes have many promising connections
                # Using a sigmoid-like function for smooth penalization
                penalty[i, j] = 1.0 / (1.0 + 0.1 * (degree_i + degree_j))
        
        np.fill_diagonal(penalty, 0)
        return penalty
    
    def _compute_two_hop_penalty(distance_matrix: np.ndarray) -> np.ndarray:
        """
        Two-hop lookahead penalty.
        
        Penalizes direct edges if a two-hop path through an intermediate node
        is significantly shorter, addressing the bypass problem.
        """
        n = distance_matrix.shape[0]
        penalty = np.ones((n, n))
        
        for i in range(n):
            for j in range(i + 1, n):
                direct = distance_matrix[i, j]
                
                # Find best two-hop alternative
                best_two_hop = np.inf
                for k in range(n):
                    if k == i or k == j:
                        continue
                    two_hop = distance_matrix[i, k] + distance_matrix[k, j]
                    best_two_hop = min(best_two_hop, two_hop)
                
                # Apply penalty based on relative improvement
                if best_two_hop < np.inf:
                    if best_two_hop < direct * 0.7:  # Significant improvement
                        penalty[i, j] = penalty[j, i] = 0.3
                    elif best_two_hop < direct * 0.9:  # Moderate improvement
                        penalty[i, j] = penalty[j, i] = 0.7
                
        return penalty
    
    def _build_sparse_candidate_mask(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
        """
        Adaptive sparse computation focusing on promising edges.
        
        Keeps nearest neighbors plus exploration connections based on instance characteristics.
        """
        n = stats['n']
        mask = np.zeros((n, n), dtype=bool)
        
        # Base number of nearest neighbors
        k_base = min(8, max(3, int(np.log2(n) * 2)))
        
        # Adjust based on regularity (more neighbors for irregular instances)
        k = int(k_base * (1.5 - 0.5 * stats['regularity']))
        k = min(k, n - 1)
        
        # Include k-nearest neighbors for each node
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argpartition(dists, k)[:k]
            mask[i, nearest] = True
        
        # Add random connections for exploration
        remaining_edges = n * (n - 1) // 2 - np.sum(mask[np.triu_indices(n, 1)])
        n_random = max(1, int(0.05 * remaining_edges))
        
        if n_random > 0:
            triu_indices = np.triu_indices(n, 1)
            candidate_indices = np.where(~mask[triu_indices])[0]
            
            if len(candidate_indices) > 0:
                selected = np.random.choice(
                    candidate_indices,
                    size=min(n_random, len(candidate_indices)),
                    replace=False
                )
                
                for idx in selected:
                    i, j = triu_indices[0][idx], triu_indices[1][idx]
                    mask[i, j] = mask[j, i] = True
        
        np.fill_diagonal(mask, False)
        return mask
    
    def _compute_adaptive_weights(stats: Dict) -> Dict:
        """Compute adaptive weights based on instance characteristics."""
        cv = stats['cv_dist']
        regularity = stats['regularity']
        
        # Weights adjust based on instance properties
        weights = {
            'w_inv': 1.2,  # Base importance of inverse distance
            
            # Two-step regret more important for irregular instances
            'w_regret': 0.8 + 0.4 * (1 - regularity),
            
            # Degree penalty more important for regular/structured instances
            'w_degree': 0.3 + 0.4 * regularity,
            
            # Two-hop penalty more important when variation is low
            'w_two_hop': 0.4 + 0.3 * (1 - cv)
        }
        
        # Normalize weights to maintain balance
        total = sum(weights.values())
        scale_factor = 4.0 / total  # Target sum of 4 for stability
        for key in weights:
            weights[key] *= scale_factor
        
        return weights
    
    def _post_process(H: np.ndarray, distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
        """Post-process the heuristic matrix."""
        n = stats['n']
        
        # Ensure each node has at least 2 connections
        min_connections = min(3, n - 1)
        for i in range(n):
            row = H[i].copy()
            row[i] = -np.inf
            non_zero = np.where(row > 0)[0]
            
            if len(non_zero) < min_connections:
                # Add nearest neighbors not already included
                dists = distance_matrix[i].copy()
                dists[i] = np.inf
                
                # Sort by distance
                sorted_indices = np.argsort(dists)
                added = 0
                
                for j in sorted_indices:
                    if H[i, j] == 0:
                        H[i, j] = H[j, i] = 1 / (dists[j] + 1e-8)
                        added += 1
                        if added >= min_connections - len(non_zero):
                            break
        
        # Final symmetrization and zero diagonal
        H = (H + H.T) / 2
        np.fill_diagonal(H, 0)
        
        # Normalize to [0, 1] range
        max_val = np.max(H)
        if max_val > 0:
            H = H / max_val
        
        return H
    
    # Main execution
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # 1. Analyze instance characteristics
    stats = _analyze_instance(distance_matrix)
    
    # 2. Compute base inverse distance
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    # 3. Compute heuristic factors
    regret = _compute_two_step_regret(distance_matrix)
    degree_penalty = _compute_degree_penalty(distance_matrix, inv_dist)
    two_hop_penalty = _compute_two_hop_penalty(distance_matrix)
    
    # 4. Build sparse candidate mask
    candidate_mask = _build_sparse_candidate_mask(distance_matrix, stats)
    
    # 5. Compute adaptive weights
    weights = _compute_adaptive_weights(stats)
    
    # 6. Initialize heuristics matrix
    H = np.zeros((n, n))
    
    # 7. Combine factors multiplicatively for candidate edges only
    for i in range(n):
        # Get candidate neighbors (only upper triangle to avoid double counting)
        neighbors = np.where(candidate_mask[i])[0]
        neighbors = neighbors[neighbors > i]
        
        for j in neighbors:
            # Multiplicative combination with adaptive weights
            combined = (
                (inv_dist[i, j] ** weights['w_inv']) *
                (regret[i, j] ** weights['w_regret']) *
                (degree_penalty[i, j] ** weights['w_degree']) *
                (two_hop_penalty[i, j] ** weights['w_two_hop'])
            )
            H[i, j] = H[j, i] = combined
    
    # 8. Post-processing
    H = _post_process(H, distance_matrix, stats)
    
    return H
