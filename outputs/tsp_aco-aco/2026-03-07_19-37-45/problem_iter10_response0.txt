import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced deterministic heuristic for TSP edge selection with enhanced spatial analysis.
    Returns a matrix of the same shape where higher values indicate more promising edges.
    
    Key improvements:
    - Global clustering analysis using silhouette approximation
    - Hierarchical spatial indexing via Morton codes
    - Optimized candidate masks with adaptive neighbor selection
    - Enhanced density estimation with spatial binning
    """
    
    def analyze_instance(dist_mat: np.ndarray) -> dict:
        """Comprehensive instance analysis with clustering metrics."""
        n = dist_mat.shape[0]
        mask = ~np.eye(n, dtype=bool)
        distances = dist_mat[mask]
        
        stats = {
            'n': n,
            'mean_dist': np.mean(distances),
            'std_dist': np.std(distances),
            'cv_dist': np.std(distances) / (np.mean(distances) + 1e-8),
        }
        
        # Enhanced clustering analysis
        if n >= 4:
            # Compute local densities using k-nearest neighbors
            k = min(5, n-1)
            local_densities = []
            for i in range(n):
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                k_nearest = np.partition(dists, k)[:k]
                local_densities.append(np.mean(k_nearest))
            
            stats['local_density_mean'] = np.mean(local_densities)
            stats['local_density_std'] = np.std(local_densities)
            
            # Global clustering via silhouette approximation
            if n >= 8:
                silhouette_scores = []
                sample_size = min(20, n // 2)
                sample_indices = np.random.choice(n, sample_size, replace=False)
                
                for i in sample_indices:
                    # Intra-cluster distance (average distance to nearest neighbors)
                    dists_i = dist_mat[i].copy()
                    dists_i[i] = np.inf
                    nearest_idx = np.argmin(dists_i)
                    
                    # Get distances to nearest neighbor's k-nearest
                    dists_nearest = dist_mat[nearest_idx].copy()
                    dists_nearest[nearest_idx] = np.inf
                    intra_dist = np.mean(np.partition(dists_nearest, k)[:k])
                    
                    # Inter-cluster distance (average distance to other nodes)
                    other_nodes = [j for j in range(n) if j not in [i, nearest_idx]]
                    if other_nodes:
                        inter_dist = np.mean(dist_mat[i, other_nodes])
                        if inter_dist > intra_dist:
                            silhouette = 1 - intra_dist / (inter_dist + 1e-8)
                        else:
                            silhouette = (inter_dist / (intra_dist + 1e-8)) - 1
                        silhouette_scores.append(silhouette)
                
                if silhouette_scores:
                    stats['clustering_tendency'] = np.mean(silhouette_scores)
                else:
                    stats['clustering_tendency'] = 0.0
            else:
                stats['clustering_tendency'] = 0.0
            
            # Spatial structure index
            stats['sparsity_index'] = stats['local_density_std'] / (stats['local_density_mean'] + 1e-8)
        else:
            stats['clustering_tendency'] = 0.0
            stats['sparsity_index'] = 0.5
        
        return stats
    
    def compute_spatial_bins(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Create spatial bins using Morton-like ordering for hierarchical indexing."""
        n = stats['n']
        if n < 8:
            return np.ones((n, n))
        
        # Use distance-based similarity to create implicit spatial ordering
        bins = np.zeros((n, n), dtype=float)
        
        # Create similarity matrix using Gaussian kernel
        bandwidth = stats['mean_dist'] / 2
        similarity = np.exp(-dist_mat ** 2 / (2 * bandwidth ** 2))
        np.fill_diagonal(similarity, 0)
        
        # Sort nodes by their average distance to others (center-outward)
        avg_distances = np.mean(dist_mat, axis=1)
        node_order = np.argsort(avg_distances)
        
        # Create hierarchical scoring: nodes close in order get bonus
        order_dist = np.abs(node_order[:, None] - node_order[None, :])
        order_similarity = np.exp(-order_dist / (n / 4))
        
        # Combine spatial similarity with order similarity
        bins = 0.7 * similarity + 0.3 * order_similarity
        
        return bins
    
    def compute_regret_scores(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Compute multi-hop regret scores with adaptive depth."""
        n = stats['n']
        regret = np.ones((n, n))
        
        # Adaptive regret depth based on instance size and structure
        max_depth = min(3, int(np.log2(n)) + 1) if n > 8 else 1
        
        for i in range(n):
            # Pre-sort distances for efficiency
            dists_i = dist_mat[i].copy()
            dists_i[i] = np.inf
            sorted_indices_i = np.argsort(dists_i)
            
            for j in range(i+1, n):
                direct_cost = dist_mat[i, j]
                min_regret = float('inf')
                
                # Explore alternative paths up to max_depth
                for depth in range(1, max_depth + 1):
                    if depth == 1:
                        # 1-hop alternatives
                        for k in sorted_indices_i[:10]:  # Check top 10 alternatives
                            if k != j:
                                alt_cost = dist_mat[i, k] + dist_mat[k, j]
                                regret_val = alt_cost / (direct_cost + 1e-8)
                                min_regret = min(min_regret, regret_val)
                    else:
                        # Sample multi-hop paths
                        n_samples = min(20, n // 2)
                        for _ in range(n_samples):
                            path = [i]
                            current = i
                            path_cost = 0
                            
                            # Build random path of specified depth
                            for _ in range(depth):
                                next_node = np.random.choice(
                                    [k for k in range(n) if k not in path]
                                )
                                path_cost += dist_mat[current, next_node]
                                current = next_node
                                if current == j:
                                    break
                            
                            if current == j:
                                regret_val = path_cost / (direct_cost + 1e-8)
                                min_regret = min(min_regret, regret_val)
                
                # Convert regret to score (higher regret = lower score)
                regret[i, j] = regret[j, i] = np.exp(-max(0, min_regret - 1))
        
        return regret
    
    def compute_density_scores(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Compute density scores with spatial awareness."""
        n = stats['n']
        density_scores = np.ones((n, n))
        
        # Compute local densities using adaptive k
        k = max(2, min(5, n // 10))
        local_densities = np.zeros(n)
        
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            k_nearest = np.partition(dists, k)[:k]
            local_densities[i] = np.mean(k_nearest)
        
        # Normalize densities
        if np.std(local_densities) > 1e-8:
            norm_densities = (local_densities - np.mean(local_densities)) / np.std(local_densities)
        else:
            norm_densities = np.zeros(n)
        
        # Compute density scores with clustering awareness
        cluster_factor = abs(stats.get('clustering_tendency', 0))
        
        for i in range(n):
            for j in range(i+1, n):
                # Base density difference
                density_diff = abs(norm_densities[i] - norm_densities[j])
                
                if cluster_factor > 0.1:
                    # In clustered instances, favor similar-density connections
                    if density_diff < 0.5:
                        score = 1.2 + 0.5 * cluster_factor
                    else:
                        score = 0.8 - 0.3 * cluster_factor
                else:
                    # In uniform instances, favor complementary densities
                    if norm_densities[i] * norm_densities[j] < 0:
                        score = 1.3
                    else:
                        score = 0.9
                
                density_scores[i, j] = density_scores[j, i] = score
        
        return density_scores
    
    def compute_mst_scores(dist_mat: np.ndarray) -> np.ndarray:
        """Compute MST-based scores with degree awareness."""
        n = dist_mat.shape[0]
        mst_scores = np.ones((n, n))
        
        if n > 2:
            try:
                mst = minimum_spanning_tree(dist_mat).toarray()
                mst_present = (mst + mst.T) > 0
                
                # Boost MST edges
                mst_scores[mst_present] = 1.8
                
                # Additional boost for critical MST edges (high degree nodes)
                mst_degree = np.sum(mst_present, axis=1)
                for i in range(n):
                    if mst_degree[i] >= 2:
                        neighbors = np.where(mst_present[i])[0]
                        for j in neighbors:
                            mst_scores[i, j] = 2.2
            except:
                pass
        
        np.fill_diagonal(mst_scores, 0)
        return mst_scores
    
    def build_candidate_mask(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Build optimized candidate mask with hierarchical pruning."""
        n = stats['n']
        
        # Base connectivity level
        base_k = min(15, max(5, int(4 * np.log2(n + 1))))
        
        # Adjust for instance characteristics
        if stats.get('clustering_tendency', 0) > 0.2:
            base_k = int(base_k * 1.2)  # More connections in clustered instances
        
        candidate_mask = np.zeros((n, n), dtype=bool)
        
        # First pass: nearest neighbors
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            
            k = min(base_k, n-1)
            if k < len(dists):
                indices = np.argpartition(dists, k)[:k]
            else:
                indices = np.arange(n)
                indices = indices[indices != i]
            
            candidate_mask[i, indices] = True
        
        # Second pass: ensure connectivity in sparse regions
        for i in range(n):
            if np.sum(candidate_mask[i]) < 3:
                # Add more connections for isolated nodes
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                extra_neighbors = min(5, n-1)
                indices = np.argpartition(dists, extra_neighbors)[:extra_neighbors]
                candidate_mask[i, indices] = True
        
        # Ensure symmetry
        candidate_mask = candidate_mask | candidate_mask.T
        np.fill_diagonal(candidate_mask, False)
        
        return candidate_mask
    
    def compute_adaptive_weights(stats: dict) -> dict:
        """Compute adaptive weights based on instance characteristics."""
        n = stats['n']
        cv = stats['cv_dist']
        clustering = abs(stats.get('clustering_tendency', 0))
        sparsity = stats.get('sparsity_index', 0.5)
        
        # Base weights with instance-aware adjustments
        weights = {
            'w_inv': 1.2 - 0.4 * cv,  # Less weight for uniform distances
            'w_regret': 0.9 + 0.3 * clustering,  # More weight in clustered instances
            'w_density': 0.7 + 0.5 * sparsity,  # More weight in sparse instances
            'w_mst': 0.6 + 0.4 * (1 - cv),  # More weight in structured instances
            'w_spatial': 0.5 + 0.5 * clustering  # More weight in clustered instances
        }
        
        # Normalize to maintain balance
        total = sum(weights.values())
        if total > 0:
            for key in weights:
                weights[key] = weights[key] / total * 5.0
        
        return weights
    
    def combine_factors(dist_mat: np.ndarray, stats: dict, factors: dict, 
                       weights: dict, mask: np.ndarray) -> np.ndarray:
        """Combine all factors using weighted geometric mean."""
        n = stats['n']
        H = np.zeros((n, n))
        
        for i in range(n):
            for j in range(i+1, n):
                if mask[i, j]:
                    product = 1.0
                    weight_sum = 0
                    
                    for factor_name, weight in weights.items():
                        if factor_name in factors and weight > 0:
                            factor_val = factors[factor_name][i, j]
                            if factor_val > 0:
                                product *= factor_val ** weight
                                weight_sum += weight
                    
                    if weight_sum > 0:
                        H[i, j] = H[j, i] = product ** (1 / weight_sum)
        
        return H
    
    def post_process(H: np.ndarray, dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Final post-processing and normalization."""
        n = stats['n']
        
        # Ensure minimum connectivity
        for i in range(n):
            non_zero = np.where(H[i] > 0)[0]
            if len(non_zero) < 2:
                # Add nearest neighbors with base scores
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                nearest = np.argsort(dists)[:3]
                for j in nearest:
                    if H[i, j] == 0:
                        base_score = 1 / (dist_mat[i, j] + 1e-8)
                        H[i, j] = H[j, i] = base_score
        
        # Apply soft thresholding
        if n > 10:
            non_zero_vals = H[H > 0]
            if len(non_zero_vals) > 0:
                threshold = np.percentile(non_zero_vals, 15)
                H[H < threshold] = 0
        
        # Ensure symmetry and zero diagonal
        H = (H + H.T) / 2.0
        np.fill_diagonal(H, 0.0)
        
        # Normalize to [0, 1] range
        max_val = np.max(H)
        if max_val > 0:
            H = H / max_val
        
        return H
    
    # Main execution
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Analyze instance
    stats = analyze_instance(distance_matrix)
    
    # Step 2: Compute all heuristic factors
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    spatial_bins = compute_spatial_bins(distance_matrix, stats)
    regret = compute_regret_scores(distance_matrix, stats)
    density = compute_density_scores(distance_matrix, stats)
    mst = compute_mst_scores(distance_matrix)
    
    factors = {
        'w_inv': inv_dist,
        'w_regret': regret,
        'w_density': density,
        'w_mst': mst,
        'w_spatial': spatial_bins
    }
    
    # Step 3: Build candidate mask
    candidate_mask = build_candidate_mask(distance_matrix, stats)
    
    # Step 4: Compute adaptive weights
    weights = compute_adaptive_weights(stats)
    
    # Step 5: Combine factors
    H = combine_factors(distance_matrix, stats, factors, weights, candidate_mask)
    
    # Step 6: Post-process
    H = post_process(H, distance_matrix, stats)
    
    return H
