import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection using deterministic multi-factor scoring.
    Returns a matrix of the same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Compute instance statistics
    stats = _compute_instance_stats(distance_matrix)
    
    # Step 2: Compute all heuristic factors
    factors = _compute_all_factors(distance_matrix, stats)
    
    # Step 3: Compute adaptive weights based on instance characteristics
    weights = _compute_adaptive_weights(stats)
    
    # Step 4: Combine factors using weighted geometric mean
    H = _combine_factors(factors, weights, distance_matrix, stats)
    
    # Step 5: Sparsify and normalize
    H = _sparsify_and_normalize(H, distance_matrix, stats)
    
    return H


def _compute_instance_stats(distance_matrix: np.ndarray) -> dict:
    """Compute statistics about the TSP instance."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal for statistics
    mask = ~np.eye(n, dtype=bool)
    dist_values = distance_matrix[mask]
    
    # Basic statistics
    mean_dist = np.mean(dist_values)
    std_dist = np.std(dist_values)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Sparsity estimation: coefficient of variation of nearest neighbor distances
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    sparsity = np.std(nn_dists) / (np.mean(nn_dists) + 1e-8)
    
    # Cluster tendency: ratio of mean nearest neighbor distance to mean distance
    cluster_tendency = np.mean(nn_dists) / (mean_dist + 1e-8)
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'cv_dist': cv_dist,
        'sparsity': sparsity,
        'cluster_tendency': cluster_tendency
    }


def _compute_all_factors(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute all heuristic factors for each edge."""
    n = stats['n']
    factors = {}
    
    # 1. Inverse distance (base factor)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    factors['inv_dist'] = inv_dist
    
    # 2. Variance-adaptive regret ratio
    factors['regret'] = _compute_regret_scores(distance_matrix, stats)
    
    # 3. Cluster-density bonus
    factors['density'] = _compute_density_scores(distance_matrix, stats)
    
    # 4. MST inclusion score
    factors['mst'] = _compute_mst_scores(distance_matrix)
    
    return factors


def _compute_regret_scores(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute regret scores using variance-adaptive penalization."""
    n = distance_matrix.shape[0]
    regret_scores = np.ones((n, n))
    cv = stats['cv_dist']
    
    for i in range(n):
        for j in range(i+1, n):
            direct_cost = distance_matrix[i, j]
            
            # Find best two-hop alternative (i->k->j where k≠i,j)
            best_alternative = np.inf
            for k in range(n):
                if k != i and k != j:
                    alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                    best_alternative = min(best_alternative, alt_cost)
            
            if best_alternative < np.inf:
                # Regret ratio: higher when no good alternative exists
                regret_ratio = best_alternative / (direct_cost + 1e-8)
                
                # Variance-adaptive scaling: emphasize regret more in uniform instances
                # When CV is low (uniform distances), regret is more informative
                regret_importance = 1.0 + (1.0 - min(cv, 1.0))
                
                # Convert to score: higher score when regret ratio is high (no good alternative)
                score = regret_ratio ** regret_importance
                regret_scores[i, j] = regret_scores[j, i] = min(score, 3.0)  # Cap extreme values
    
    np.fill_diagonal(regret_scores, 0)
    return regret_scores


def _compute_density_scores(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute density scores favoring edges in sparse regions."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Compute local densities using k-nearest neighbors
    k = min(5, max(2, n // 10))
    local_densities = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k)[:k]
        local_densities[i] = np.mean(k_nearest)
    
    # Normalize and invert: high value means sparse region
    if np.std(local_densities) > 1e-8:
        local_densities = (local_densities - np.mean(local_densities)) / np.std(local_densities)
    
    # Create scores: favor edges connecting two sparse nodes
    for i in range(n):
        for j in range(i+1, n):
            # Exponential bonus for sparse-sparse connections
            density_bonus = np.exp(-0.5 * (local_densities[i] + local_densities[j]))
            density_scores[i, j] = density_scores[j, i] = density_bonus
    
    np.fill_diagonal(density_scores, 0)
    return density_scores


def _compute_mst_scores(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute scores based on Minimum Spanning Tree inclusion."""
    n = distance_matrix.shape[0]
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Boost edges in MST
            mst_scores[mst_present] = 1.5
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores


def _compute_adaptive_weights(stats: dict) -> dict:
    """Dynamically compute factor weights based on instance characteristics."""
    n = stats['n']
    cv = stats['cv_dist']
    sparsity = stats['sparsity']
    cluster_tendency = stats['cluster_tendency']
    
    # Adaptive weighting based on instance properties
    weights = {
        'inv_dist': 1.0,  # Base weight for inverse distance
        
        # Regret weight: higher for uniform instances (low CV)
        'regret': 0.8 + 0.4 * (1.0 - min(cv, 1.0)),
        
        # Density weight: higher for clustered instances
        'density': 0.5 + 0.5 * cluster_tendency,
        
        # MST weight: decreases with instance size, increases with structure
        'mst': max(0.3, 0.8 - 0.02 * n) * (1.0 + sparsity)
    }
    
    # Normalize weights to sum to 1 for geometric mean
    total = sum(weights.values())
    for key in weights:
        weights[key] = weights[key] / total
    
    return weights


def _combine_factors(factors: dict, weights: dict, 
                    distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Combine factors using weighted geometric mean with adaptive candidate selection."""
    n = stats['n']
    H = np.zeros((n, n))
    
    # Build candidate mask for efficiency
    candidate_mask = _build_candidate_mask(distance_matrix, stats)
    
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Weighted geometric mean of all factors
                combined = 1.0
                for factor_name in factors:
                    factor_value = factors[factor_name][i, j]
                    weight = weights[factor_name]
                    combined *= (factor_value ** weight)
                
                H[i, j] = H[j, i] = combined
    
    return H


def _build_candidate_mask(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Build mask of candidate edges to consider (for efficiency)."""
    n = stats['n']
    
    # Adaptive k: more candidates for sparse/structured instances
    base_k = min(15, max(5, int(np.log2(n) * 4)))
    k = int(base_k * (1.0 + stats['sparsity']))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k-nearest neighbors
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < len(dists):
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Make symmetric
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask


def _sparsify_and_normalize(H: np.ndarray, distance_matrix: np.ndarray, 
                           stats: dict) -> np.ndarray:
    """Sparsify unpromising edges and normalize the heuristic matrix."""
    n = stats['n']
    
    # Zero out unpromising edges (keep only top candidates per node)
    k_keep = min(10, max(3, n // 3))
    for i in range(n):
        row = H[i].copy()
        row[i] = -np.inf  # Ignore diagonal
        
        # Find threshold for top k_keep edges
        if np.sum(row > 0) > k_keep:
            threshold = np.partition(row, -k_keep)[-k_keep]
            H[i][row < threshold] = 0
    
    # Ensure symmetry
    H = (H + H.T) / 2
    
    # Ensure minimum connectivity
    min_edges = min(3, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i])[0]
        if len(non_zero) < min_edges:
            # Add nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    base_score = 1 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    np.fill_diagonal(H, 0)
    return H
