                    else:
                        # Normalized regret: (cost - best) / best
                        regret = (dists[j] - best_dist) / (best_dist + 1e-8)
                    
                    # Invert: high regret becomes low score
                    regret_matrix[i, j] = 1.0 / (1.0 + regret)

            threshold = best_dist * 1.2
            good_edges_count = np.sum(dists <= threshold)
            degree_penalty[i] = 1.0 / (1.0 + 0.1 * good_edges_count)
    
    # Step 2: Two-step lookahead scoring
    lookahead_scores = np.ones((n, n))
    
    for i in range(n):
        for j in range(i+1, n):
            # Skip if edge is too long

                continue

                    if two_step_cost < best_two_step:
                        best_two_step = two_step_cost
            
            # If two-step is significantly better, penalize direct edge
            direct_cost = distance_matrix[i, j]
            if best_two_step < direct_cost * lookahead_factor:
                penalty = 0.3  # Strong penalty
            elif best_two_step < direct_cost * 0.9:
                penalty = 0.7  # Moderate penalty
            else:
                penalty = 1.0  # No penalty
            
            lookahead_scores[i, j] = lookahead_scores[j, i] = penalty

    candidate_mask = np.zeros((n, n), dtype=bool)

        # Get distances from node i
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Select k nearest neighbors
        if k_neighbors < n-1:
            # Get indices of k smallest distances
            indices = np.argpartition(dists, k_neighbors)[:k_neighbors]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]

                    H[i, j] = H[j, i] = inv_dist
    
    # Symmetrize final matrix
    H = (H + H.T) / 2.0
    
    # Normalize to [0, 1] range
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
