import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristic with improved instance analysis and simplified scoring.
    
    Computes heuristic scores for each edge using multiple objective measures:
    1. Inverse distance (normalized)
    2. Regret-based penalty (simplified computation)
    3. Local density similarity
    4. MST membership bonus
    5. Limited lookahead simulation
    
    Higher values indicate more promising edges for inclusion in TSP solutions.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric n×n matrix of pairwise distances between n nodes.
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape with scores in [0, 1].
    """
    
    n = distance_matrix.shape[0]
    if n <= 2:
        # Trivial cases: return uniform scores
        H = np.ones_like(distance_matrix)
        np.fill_diagonal(H, 0)
        return H
    
    # ------------------------- Instance Analysis -------------------------
    
    def analyze_instance(dist_mat: np.ndarray) -> dict:
        """Analyze TSP instance characteristics for adaptive parameter tuning."""
        n = dist_mat.shape[0]
        
        # Remove diagonal for statistics
        mask = ~np.eye(n, dtype=bool)
        flat_dists = dist_mat[mask].reshape(n, n-1)
        
        # Basic statistics
        mean_dist = np.mean(flat_dists)
        std_dist = np.std(flat_dists)
        cv_dist = std_dist / (mean_dist + 1e-8)
        
        # Sparsity estimation via distance distribution divergence
        # Simplified KL divergence approximation
        actual_dists = flat_dists.flatten()
        random_dists = np.random.exponential(scale=mean_dist, 
                                           size=min(500, len(actual_dists)))
        
        # Histogram-based divergence
        bins = np.linspace(0, np.max(actual_dists) * 1.2, 30)
        hist_act, _ = np.histogram(actual_dists, bins=bins, density=True)
        hist_rand, _ = np.histogram(random_dists, bins=bins, density=True)
        
        # Avoid zeros
        hist_act = np.maximum(hist_act, 1e-10)
        hist_rand = np.maximum(hist_rand, 1e-10)
        
        kl_div = np.sum(hist_act * np.log(hist_act / hist_rand))
        sparsity = 1 / (1 + np.exp(-kl_div / 15))  # Sigmoid
        
        # Simplified cluster tendency (Hopkins-like)
        cluster_tendency = 0.5
        if n >= 10:
            m = min(15, n // 3)
            indices = np.random.choice(n, m, replace=False)
            
            sample_nn = []
            for i in indices:
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                sample_nn.append(np.min(dists))
            
            # Compare to random distances from matrix
            random_nn = np.random.choice(flat_dists.flatten(), m)
            
            hopkins = np.sum(sample_nn) / (np.sum(sample_nn) + np.sum(random_nn) + 1e-8)
            cluster_tendency = min(max(hopkins, 0), 1)
        
        return {
            'n': n,
            'mean_dist': mean_dist,
            'std_dist': std_dist,
            'cv_dist': cv_dist,
            'sparsity': sparsity,
            'cluster_tendency': cluster_tendency,
            'k_regret': min(max(2, n // 10), 5)
        }
    
    # ------------------------- Base Scoring Components -------------------------
    
    def compute_inverse_distance(dist_mat: np.ndarray) -> np.ndarray:
        """Normalized inverse distance scores."""
        n = dist_mat.shape[0]
        inv_dist = 1 / (dist_mat + np.eye(n) * 1e-8)
        np.fill_diagonal(inv_dist, 0)
        
        max_val = np.max(inv_dist)
        if max_val > 0:
            inv_dist = inv_dist / max_val
        
        return inv_dist
    
    def compute_regret_scores(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Simplified regret computation focusing on 1-hop alternatives."""
        n = stats['n']
        k = stats['k_regret']
        regret_scores = np.ones((n, n))
        
        for i in range(n):
            # Get distances from node i
            dists_i = dist_mat[i].copy()
            dists_i[i] = np.inf
            
            # Find k+1 nearest to compute regret
            k_nearest = min(k + 1, n - 1)
            nearest_indices = np.argpartition(dists_i, k_nearest)[:k_nearest]
            nearest_dists = dists_i[nearest_indices]
            
            # Sort the nearest distances
            sorted_order = np.argsort(nearest_dists)
            nearest_indices = nearest_indices[sorted_order]
            nearest_dists = nearest_dists[sorted_order]
            
            if len(nearest_dists) >= 2:
                best_dist = nearest_dists[0]
                
                for idx, j in enumerate(nearest_indices):
                    if idx < k and j != i:
                        # Regret: how much worse is this than the best?
                        regret = max(0, dist_mat[i, j] - best_dist) / (best_dist + 1e-8)
                        # Convert to score: higher = less regret
                        regret_scores[i, j] = 1 / (1 + regret)
        
        # Symmetrize
        regret_scores = (regret_scores + regret_scores.T) / 2
        np.fill_diagonal(regret_scores, 0)
        
        return regret_scores
    
    def compute_density_scores(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Score edges based on local density similarity."""
        n = stats['n']
        density_scores = np.ones((n, n))
        
        # Compute local density as average distance to 3 nearest neighbors
        local_density = np.zeros(n)
        k_density = min(3, n - 1)
        
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            
            if k_density > 0:
                k_nearest = np.partition(dists, k_density)[:k_density]
                local_density[i] = np.mean(k_nearest)
            else:
                local_density[i] = np.mean(dists[dists < np.inf])
        
        # Normalize densities
        if np.std(local_density) > 1e-8:
            local_density = (local_density - np.mean(local_density)) / np.std(local_density)
        
        # Score edges: prefer connecting nodes with similar density
        for i in range(n):
            for j in range(i+1, n):
                density_diff = abs(local_density[i] - local_density[j])
                density_scores[i, j] = density_scores[j, i] = np.exp(-0.3 * density_diff)
        
        np.fill_diagonal(density_scores, 0)
        return density_scores
    
    def compute_mst_scores(dist_mat: np.ndarray) -> np.ndarray:
        """Bonus for edges in Minimum Spanning Tree."""
        n = dist_mat.shape[0]
        mst_scores = np.ones((n, n))
        
        if n > 2:
            try:
                mst = minimum_spanning_tree(dist_mat).toarray()
                mst_edges = (mst + mst.T) > 0
                mst_scores[mst_edges] = 1.5  # Fixed bonus
            except:
                pass
        
        np.fill_diagonal(mst_scores, 0)
        return mst_scores
    
    def compute_lookahead_scores(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """
        Limited lookahead simulation for promising edges.
        Uses greedy expansion with occasional random exploration.
        """
        n = stats['n']
        lookahead_scores = np.ones((n, n))
        
        if n < 4:
            return lookahead_scores
        
        # Adaptive simulation parameters
        n_simulations = min(12, max(3, int(3 * np.log(n + 1))))
        max_steps = min(6, n - 2)
        
        # Only simulate for edges not too long
        threshold = 1.5 * stats['mean_dist']
        
        for i in range(n):
            for j in range(i+1, n):
                if dist_mat[i, j] > threshold:
                    continue
                
                scores = []
                for _ in range(n_simulations):
                    # Start with edge (i, j)
                    visited = np.zeros(n, dtype=bool)
                    visited[[i, j]] = True
                    ends = [i, j]
                    path_len = dist_mat[i, j]
                    
                    # Extend path for a few steps
                    for step in range(max_steps):
                        # Choose which end to extend (random)
                        end_idx = np.random.randint(2)
                        current = ends[end_idx]
                        
                        # Find unvisited nodes
                        unvisited = np.where(~visited)[0]
                        if len(unvisited) == 0:
                            break
                        
                        # Compute distances to unvisited nodes
                        dists = dist_mat[current, unvisited]
                        
                        # 80% chance greedy, 20% chance random exploration
                        if np.random.random() < 0.8:
                            # Greedy: choose nearest unvisited
                            next_idx = np.argmin(dists)
                        else:
                            # Random proportional to inverse distance
                            weights = 1 / (dists + 1e-8)
                            weights = weights / np.sum(weights)
                            next_idx = np.random.choice(len(unvisited), p=weights)
                        
                        next_node = unvisited[next_idx]
                        path_len += dists[next_idx]
                        visited[next_node] = True
                        ends[end_idx] = next_node
                    
                    # Score based on path efficiency
                    expected_len = max_steps * stats['mean_dist']
                    score = expected_len / (path_len + 1e-8)
                    scores.append(min(score, 2.0))
                
                if scores:
                    lookahead_scores[i, j] = lookahead_scores[j, i] = np.median(scores)
        
        # Normalize
        max_val = np.max(lookahead_scores)
        if max_val > 0:
            lookahead_scores = lookahead_scores / max_val
        
        np.fill_diagonal(lookahead_scores, 0)
        return lookahead_scores
    
    def build_candidate_mask(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
        """Identify candidate edges to reduce computation."""
        n = stats['n']
        
        # Adaptive neighbor count
        base_k = min(12, max(3, int(np.log2(n + 1) * 4)))
        k = int(base_k * (1 + 0.3 * stats['sparsity']))
        k = min(k, n - 1)
        
        candidate_mask = np.zeros((n, n), dtype=bool)
        
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            
            if k < n - 1:
                indices = np.argpartition(dists, k)[:k]
            else:
                indices = np.arange(n)
                indices = indices[indices != i]
            
            candidate_mask[i, indices] = True
        
        # Symmetrize
        candidate_mask = candidate_mask | candidate_mask.T
        np.fill_diagonal(candidate_mask, False)
        
        return candidate_mask
    
    def compute_adaptive_weights(stats: dict) -> dict:
        """
        Compute adaptive weights for combining scoring components.
        Based on instance characteristics.
        """
        n = stats['n']
        cv = stats['cv_dist']
        sparsity = stats['sparsity']
        cluster = stats['cluster_tendency']
        
        weights = {
            'w_inv': 1.0,  # Base component
            'w_regret': 0.8 + 0.4 * cv,  # More important when distances vary
            'w_density': 0.3 + 0.4 * cluster,  # More important with clusters
            'w_lookahead': 0.2 + 0.3 * sparsity,  # More important for structured
            'w_mst': max(0.1, 0.4 - 0.01 * n)  # Less important for large n
        }
        
        # Normalize to sum to approximately 3 (geometric mean scale)
        total = sum(weights.values())
        scale = 3.0 / total
        
        for key in weights:
            weights[key] = weights[key] * scale
        
        return weights
    
    # ------------------------- Main Execution -------------------------
    
    # Step 1: Analyze instance
    stats = analyze_instance(distance_matrix)
    
    # Step 2: Compute scoring components
    inv_dist = compute_inverse_distance(distance_matrix)
    regret = compute_regret_scores(distance_matrix, stats)
    density = compute_density_scores(distance_matrix, stats)
    mst = compute_mst_scores(distance_matrix)
    lookahead = compute_lookahead_scores(distance_matrix, stats)
    
    # Step 3: Get adaptive weights
    weights = compute_adaptive_weights(stats)
    
    # Step 4: Build candidate mask for efficiency
    candidate_mask = build_candidate_mask(distance_matrix, stats)
    
    # Step 5: Combine components using weighted geometric mean
    H = np.zeros((n, n))
    
    # Pre-compute exponents for efficiency
    exponents = {k: v for k, v in weights.items()}
    
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                combined = (
                    (inv_dist[i, j] ** exponents['w_inv']) *
                    (regret[i, j] ** exponents['w_regret']) *
                    (density[i, j] ** exponents['w_density']) *
                    (lookahead[i, j] ** exponents['w_lookahead']) *
                    (mst[i, j] ** exponents['w_mst'])
                )
                H[i, j] = H[j, i] = combined
            else:
                H[i, j] = H[j, i] = 0
    
    # ------------------------- Post-processing -------------------------
    
    # Ensure minimum connectivity
    min_edges = min(3, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i])[0]
        if len(non_zero) < min_edges:
            # Add nearest neighbors with low scores
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    # Assign small but non-zero score
                    H[i, j] = H[j, i] = 0.1 / (distance_matrix[i, j] + 1e-8)
    
    # Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Final normalization to [0, 1]
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    # Apply mild sparsification for larger instances
    if n > 20:
        threshold = np.percentile(H[H > 0], 20)  # Keep top 80%
        H[H < threshold] = 0
    
    return H
