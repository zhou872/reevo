import numpy as np
import warnings
from typing import Dict, List, Tuple
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic with multi-scale instance analysis, 
    enhanced regret analysis, reinforcement-guided weight optimization,
    and intelligent edge sparsification.
    
    Returns heuristic matrix where higher values indicate more promising edges.
    """
    np.random.seed(42)  # For reproducibility
    warnings.filterwarnings('ignore')
    
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # ==================== ENHANCED INSTANCE ANALYSIS ====================
    def analyze_instance(dist_mat: np.ndarray) -> Dict:
        """Multi-scale instance analysis with clustering detection."""
        n = dist_mat.shape[0]
        dist_no_diag = dist_mat[~np.eye(n, dtype=bool)].reshape(n, n-1)
        dist_vals = dist_no_diag.flatten()
        
        mean_dist = np.mean(dist_vals)
        std_dist = np.std(dist_vals)
        cv_dist = std_dist / (mean_dist + 1e-8)
        percentiles = np.percentile(dist_vals, [10, 25, 50, 75, 90])
        
        # Clustering detection using multiple methods
        clustering_scores = []
        if n >= 8:
            # Method 1: Gap statistic approximation
            intra_cluster_ratios = []
            for i in range(min(20, n)):
                sample_size = min(5, n//4)
                indices = np.random.choice(n, sample_size, replace=False)
                sample_dists = []
                for idx in indices:
                    for jdx in indices:
                        if idx != jdx:
                            sample_dists.append(dist_mat[idx, jdx])
                if sample_dists:
                    intra_mean = np.mean(sample_dists)
                    intra_cluster_ratios.append(intra_mean / (mean_dist + 1e-8))
            
            if intra_cluster_ratios:
                clustering_scores.append(1 - np.mean(intra_cluster_ratios))
            
            # Method 2: Distance distribution skewness
            from scipy.stats import skew
            if len(dist_vals) > 1:
                skewness = abs(skew(dist_vals))
                clustering_scores.append(min(1.0, skewness / 2))
        
        clustering_level = np.mean(clustering_scores) if clustering_scores else 0.0
        
        return {
            'n': n,
            'mean_dist': mean_dist,
            'std_dist': std_dist,
            'cv_dist': cv_dist,
            'percentiles': percentiles,
            'p50': percentiles[2],
            'p90': percentiles[4],
            'clustering_level': clustering_level
        }
    
    # ==================== ADAPTIVE CANDIDATE MASK ====================
    def build_candidate_mask(dist_mat: np.ndarray, stats: Dict) -> np.ndarray:
        """Build candidate edge mask with clustering-aware connectivity."""
        n = stats['n']
        clustering = stats['clustering_level']
        
        # Adaptive k based on instance characteristics
        base_k = min(20, max(6, int(5 * np.log2(n + 1))))
        if clustering > 0.3:  # More candidates for clustered instances
            base_k = int(base_k * (1 + clustering))
        
        candidate_mask = np.zeros((n, n), dtype=bool)
        
        # Build initial k-NN mask
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            k = min(base_k, n-1)
            indices = np.argpartition(dists, k)[:k] if k < n-1 else np.arange(n)[dists < np.inf]
            candidate_mask[i, indices] = True
        
        # Ensure connectivity for clustered instances
        if clustering > 0.4 and n >= 10:
            # Add minimum spanning tree edges
            try:
                mst = minimum_spanning_tree(dist_mat).toarray()
                mst_edges = np.where(mst + mst.T > 0)
                candidate_mask[mst_edges] = True
            except:
                pass
            
            # Add long-distance connections for better connectivity
            if clustering > 0.6:
                for i in range(n):
                    dists = dist_mat[i].copy()
                    dists[i] = np.inf
                    # Add a few farthest connections
                    far_indices = np.argsort(dists)[-min(3, n//10):]
                    candidate_mask[i, far_indices] = True
        
        candidate_mask = candidate_mask | candidate_mask.T
        np.fill_diagonal(candidate_mask, False)
        
        return candidate_mask
    
    # ==================== ENHANCED REGRET ANALYSIS ====================
    def compute_regret(dist_mat: np.ndarray, stats: Dict, 
                      candidate_mask: np.ndarray) -> np.ndarray:
        """
        Multi-hop regret analysis with adaptive depth based on clustering.
        Considers 2-hop to 4-hop alternative paths for clustered instances.
        """
        n = stats['n']
        clustering = stats['clustering_level']
        regret = np.ones((n, n))
        
        # Precompute sorted neighbors for efficiency
        sorted_neighbors = []
        neighbor_indices = []
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            sorted_idx = np.argsort(dists)
            sorted_neighbors.append(sorted_idx)
            neighbor_indices.append(np.where(candidate_mask[i])[0])
        
        # Determine regret depth based on clustering
        max_depth = 2  # Default
        if clustering > 0.5 and n >= 15:
            max_depth = 3
        if clustering > 0.7 and n >= 25:
            max_depth = 4
        
        for i in range(n):
            i_candidates = neighbor_indices[i]
            
            for j in i_candidates:
                if j <= i:
                    continue
                
                direct_cost = dist_mat[i, j]
                best_alternative = np.inf
                
                # 2-hop alternatives
                for k in i_candidates:
                    if k == j or not candidate_mask[k, j]:
                        continue
                    alt_cost = dist_mat[i, k] + dist_mat[k, j]
                    best_alternative = min(best_alternative, alt_cost)
                
                # Consider deeper paths for clustered instances
                if max_depth >= 3 and n >= 10 and clustering > 0.3:
                    # Sample 3-hop paths
                    samples = min(30, max(10, n//3))
                    for _ in range(samples):
                        k = np.random.choice([x for x in range(n) if x not in [i, j]])
                        if not candidate_mask[i, k]:
                            continue
                        
                        # Try through k's nearest neighbors
                        k_neighbors = sorted_neighbors[k][:min(4, len(sorted_neighbors[k]))]
                        for l in k_neighbors:
                            if l in [i, j, k] or not candidate_mask[l, j]:
                                continue
                            alt_cost = dist_mat[i, k] + dist_mat[k, l] + dist_mat[l, j]
                            best_alternative = min(best_alternative, alt_cost)
                
                if max_depth >= 4 and n >= 20 and clustering > 0.6:
                    # Sample 4-hop paths for highly clustered instances
                    for _ in range(min(15, n//4)):
                        path = [i]
                        current = i
                        path_cost = 0
                        
                        for _ in range(3):  # Build 3 intermediate hops
                            available = [x for x in neighbor_indices[current] 
                                       if x not in path and x != j]
                            if not available:
                                break
                            next_node = np.random.choice(available)
                            path_cost += dist_mat[current, next_node]
                            path.append(next_node)
                            current = next_node
                        
                        if len(path) == 4 and candidate_mask[path[-1], j]:
                            path_cost += dist_mat[path[-1], j]
                            best_alternative = min(best_alternative, path_cost)
                
                if best_alternative < np.inf:
                    # Enhanced penalty/reward function
                    ratio = best_alternative / (direct_cost + 1e-8)
                    
                    if ratio < 0.9:  # Alternative is significantly better
                        penalty = 0.5 + 0.5 * np.exp(-8 * (1.0 - ratio))
                    elif ratio < 1.1:  # Comparable
                        penalty = 1.0
                    else:  # Direct edge is better
                        bonus = 1.0 + np.log1p(min(5.0, ratio - 1.0))
                        penalty = min(bonus, 4.0)
                    
                    # Adjust based on clustering
                    if clustering > 0.5:
                        penalty = penalty ** (1 + clustering * 0.5)
                    
                    regret[i, j] = regret[j, i] = penalty
        
        np.fill_diagonal(regret, 0)
        return regret
    
    # ==================== FACTOR COMPUTATION ====================
    def compute_factors(dist_mat: np.ndarray, stats: Dict, 
                       candidate_mask: np.ndarray) -> Dict:
        """Compute all heuristic factors including clustering-aware scoring."""
        n = stats['n']
        clustering = stats['clustering_level']
        factors = {}
        
        # 1. Inverse distance (normalized)
        inv_dist = 1 / (dist_mat + np.eye(n) * 1e-8)
        np.fill_diagonal(inv_dist, 0)
        if np.max(inv_dist) > 0:
            inv_dist = inv_dist / np.max(inv_dist)
        factors['inv_dist'] = inv_dist
        
        # 2. Enhanced regret
        factors['regret'] = compute_regret(dist_mat, stats, candidate_mask)
        
        # 3. MST-based scoring with clustering awareness
        mst_scores = np.ones((n, n))
        if n > 2:
            try:
                mst = minimum_spanning_tree(dist_mat).toarray()
                mst_present = (mst + mst.T) > 0
                
                # Boost MST edges more in clustered instances
                boost_factor = 1.5 + clustering * 0.7
                mst_scores[mst_present] = boost_factor
                
                # Special handling for bridge edges in clustered instances
                if clustering > 0.4:
                    # Identify potential bridge edges (connect different clusters)
                    degree = np.sum(mst_present, axis=1)
                    for i in range(n):
                        if degree[i] == 1:  # Leaf node
                            j = np.where(mst_present[i])[0][0]
                            # If leaf connects to a high-degree node, might be bridge
                            if degree[j] >= 3:
                                mst_scores[i, j] = mst_scores[j, i] = boost_factor * 1.3
            except:
                pass
        np.fill_diagonal(mst_scores, 0)
        factors['mst'] = mst_scores
        
        # 4. Centrality-based scoring with clustering adjustment
        centrality_scores = np.ones((n, n))
        if n >= 3:
            # Compute multiple centrality measures
            closeness = np.zeros(n)
            betweenness = np.zeros(n)
            
            for i in range(n):
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                closeness[i] = 1 / (np.mean(dists) + 1e-8)
                
                # Simple betweenness approximation
                if n <= 50:  # Computationally expensive for large n
                    count = 0
                    for a in range(n):
                        if a == i: continue
                        for b in range(a+1, n):
                            if b == i: continue
                            direct = dist_mat[a, b]
                            via_i = dist_mat[a, i] + dist_mat[i, b]
                            if via_i < direct * 1.1:  # i is on a competitive path
                                count += 1
                    betweenness[i] = count
            
            # Normalize centrality measures
            if np.max(closeness) > np.min(closeness):
                closeness = (closeness - np.min(closeness)) / \
                           (np.max(closeness) - np.min(closeness))
            
            if np.max(betweenness) > 0:
                betweenness = betweenness / np.max(betweenness)
            
            # Combine centrality measures
            combined_centrality = 0.7 * closeness + 0.3 * betweenness
            
            # Generate edge scores based on centrality
            for i in range(n):
                for j in range(i+1, n):
                    if candidate_mask[i, j]:
                        ci, cj = combined_centrality[i], combined_centrality[j]
                        
                        # Adjust scoring based on clustering
                        if clustering > 0.5:
                            # In clustered instances, favor connections between central nodes
                            score = 1.0 + (ci + cj) / 3
                        else:
                            # In uniform instances, balance centrality
                            if ci < 0.3 and cj < 0.3:
                                score = 1.4  # Connect peripheral nodes
                            elif (ci > 0.7 and cj < 0.3) or (ci < 0.3 and cj > 0.7):
                                score = 1.7  # Bridge connections
                            else:
                                score = 1.0 + (ci + cj) / 4
                        
                        centrality_scores[i, j] = centrality_scores[j, i] = score
        
        factors['centrality'] = centrality_scores
        
        # 5. Clustering-aware scoring (new factor)
        cluster_scores = np.ones((n, n))
        if clustering > 0.3 and n >= 10:
            # Estimate local density for each node
            local_density = np.zeros(n)
            for i in range(n):
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                # Average distance to k nearest neighbors
                k = min(5, n-1)
                local_density[i] = 1 / (np.mean(np.partition(dists, k)[:k]) + 1e-8)
            
            if np.max(local_density) > 0:
                local_density = local_density / np.max(local_density)
            
            for i in range(n):
                for j in range(i+1, n):
                    if candidate_mask[i, j]:
                        di, dj = local_density[i], local_density[j]
                        
                        if di > 0.7 and dj > 0.7:
                            # Both in dense regions - likely same cluster
                            score = 1.8 if clustering > 0.6 else 1.4
                        elif di < 0.3 and dj < 0.3:
                            # Both in sparse regions
                            score = 1.3
                        else:
                            # Cross-cluster connection
                            score = 1.6 if clustering > 0.6 else 1.2
                        
                        cluster_scores[i, j] = cluster_scores[j, i] = score
        
        factors['cluster'] = cluster_scores
        
        return factors
    
    # ==================== REINFORCEMENT-GUIDED WEIGHT OPTIMIZATION ====================
    def optimize_weights(dist_mat: np.ndarray, factors: Dict, 
                        candidate_mask: np.ndarray, stats: Dict) -> Dict:
        """
        Advanced weight optimization using guided search with memory.
        Learns weights that correlate with tour quality across multiple samples.
        """
        n = stats['n']
        clustering = stats['clustering_level']
        
        # Base weights from instance analysis
        base_weights = {
            'w_inv': 1.0 - 0.3 * min(1.0, stats['cv_dist']),
            'w_regret': 0.6 + 0.4 * clustering,
            'w_mst': max(0.2, 0.7 - 0.015 * n),
            'w_centrality': 0.4 + 0.3 * clustering,
            'w_cluster': 0.3 * clustering
        }
        
        # Normalize
        total = sum(base_weights.values())
        for key in base_weights:
            base_weights[key] /= total
        
        if n < 8:
            return base_weights
        
        # Generate diverse evaluation tours
        def generate_tours(n_tours: int) -> List[Tuple[np.ndarray, float]]:
            tours = []
            for tour_idx in range(n_tours):
                if tour_idx < n_tours // 3:
                    # Random tour
                    tour = np.random.permutation(n)
                elif tour_idx < 2 * n_tours // 3:
                    # Greedy tour from random start
                    tour = np.zeros(n, dtype=int)
                    visited = np.zeros(n, dtype=bool)
                    current = np.random.randint(0, n)
                    tour[0] = current
                    visited[current] = True
                    
                    for idx in range(1, n):
                        dists = dist_mat[current].copy()
                        dists[visited] = np.inf
                        if np.all(dists == np.inf):
                            unvisited = np.where(~visited)[0]
                            current = np.random.choice(unvisited)
                        else:
                            # Stochastic greedy: pick from top 3
                            top3 = np.argsort(dists)[:min(3, len(dists))]
                            if len(top3) > 1:
                                probs = 1 / (dist_mat[current, top3] + 1e-8)
                                probs = probs / np.sum(probs)
                                current = np.random.choice(top3, p=probs)
                            else:
                                current = top3[0]
                        tour[idx] = current
                        visited[current] = True
                else:
                    # Christofides-inspired (simplified)
                    try:
                        mst = minimum_spanning_tree(dist_mat).toarray()
                        degree = np.sum(mst + mst.T > 0, axis=1)
                        odds = np.where(degree % 2 == 1)[0]
                        if len(odds) >= 2:
                            # Create minimum matching on odd nodes (approximate)
                            tour = list(range(n))
                            np.random.shuffle(tour)
                        else:
                            tour = np.random.permutation(n)
                    except:
                        tour = np.random.permutation(n)
                
                # Calculate tour length
                tour_length = sum(dist_mat[tour[i], tour[(i+1)%n]] for i in range(n))
                tours.append((tour, tour_length))
            
            return tours
        
        # Evaluate weight configuration
        def evaluate_weights(weights: Dict, tours: List[Tuple[np.ndarray, float]]) -> float:
            n = factors['inv_dist'].shape[0]
            H_temp = np.zeros((n, n))
            
            # Precompute combined scores for candidate edges
            for i in range(n):
                for j in range(i+1, n):
                    if candidate_mask[i, j]:
                        combined = (
                            factors['inv_dist'][i, j] ** weights['w_inv'] *
                            factors['regret'][i, j] ** weights['w_regret'] *
                            factors['mst'][i, j] ** weights['w_mst'] *
                            factors['centrality'][i, j] ** weights['w_centrality'] *
                            factors['cluster'][i, j] ** weights['w_cluster']
                        )
                        H_temp[i, j] = H_temp[j, i] = combined
            
            # Score tours: prefer tours with high heuristic scores and short lengths
            scores = []
            for tour, tour_length in tours[:min(15, len(tours))]:
                tour_score = 0
                for k in range(n):
                    i, j = tour[k], tour[(k+1) % n]
                    if H_temp[i, j] > 0:
                        tour_score += H_temp[i, j]
                
                # Combine score and length (higher is better)
                if tour_length > 0:
                    normalized_score = tour_score / (tour_length ** 0.5)
                    scores.append(normalized_score)
            
            return np.mean(scores) if scores else 0.0
        
        # Guided search with memory
        tours = generate_tours(min(30, n * 3))
        
        # Sort tours by quality
        tours.sort(key=lambda x: x[1])
        elite_tours = tours[:min(10, len(tours)//3)]
        
        best_score = -np.inf
        best_weights = base_weights.copy()
        history = []
        
        iterations = min(25, max(10, n//3))
        for iteration in range(iterations):
            # Generate candidate weights
            if iteration == 0:
                current_weights = base_weights.copy()
            else:
                # Perturb from best or random restart
                if np.random.random() < 0.7:
                    # Perturb from best
                    base = best_weights.copy()
                else:
                    # Random restart
                    base = base_weights.copy()
                
                current_weights = {}
                for key, value in base.items():
                    # Adaptive perturbation based on iteration
                    max_perturb = 0.4 * (1 - iteration/iterations)
                    perturbation = 1.0 + np.random.uniform(-max_perturb, max_perturb)
                    current_weights[key] = max(0.05, value * perturbation)
            
            # Renormalize
            total = sum(current_weights.values())
            for key in current_weights:
                current_weights[key] /= total
            
            # Evaluate on elite tours
            score = evaluate_weights(current_weights, elite_tours)
            
            history.append((score, current_weights.copy()))
            
            if score > best_score:
                best_score = score
                best_weights = current_weights.copy()
        
        # Optional: refine by averaging top performers
        if len(history) >= 5:
            history.sort(key=lambda x: x[0], reverse=True)
            top_weights = [w for _, w in history[:3]]
            
            # Average top weights
            avg_weights = {key: np.mean([w[key] for w in top_weights]) 
                          for key in base_weights.keys()}
            
            # Normalize and evaluate
            total = sum(avg_weights.values())
            for key in avg_weights:
                avg_weights[key] /= total
            
            avg_score = evaluate_weights(avg_weights, elite_tours)
            if avg_score > best_score:
                best_weights = avg_weights
        
        return best_weights
    
    # ==================== FACTOR COMBINATION ====================
    def combine_factors(factors: Dict, weights: Dict, 
                       candidate_mask: np.ndarray) -> np.ndarray:
        """Weighted geometric mean combination of factors."""
        n = factors['inv_dist'].shape[0]
        H = np.zeros((n, n))
        
        for i in range(n):
            for j in range(i+1, n):
                if candidate_mask[i, j]:
                    combined = (
                        factors['inv_dist'][i, j] ** weights['w_inv'] *
                        factors['regret'][i, j] ** weights['w_regret'] *
                        factors['mst'][i, j] ** weights['w_mst'] *
                        factors['centrality'][i, j] ** weights['w_centrality'] *
                        factors['cluster'][i, j] ** weights['w_cluster']
                    )
                    H[i, j] = H[j, i] = combined
        
        np.fill_diagonal(H, 0)
        return H
    
    # ==================== INTELLIGENT SPARSIFICATION ====================
    def sparsify_heuristics(H: np.ndarray, dist_mat: np.ndarray,
                           candidate_mask: np.ndarray, stats: Dict) -> np.ndarray:
        """
        Intelligent sparsification with clustering-aware edge selection.
        Uses adaptive thresholding and emphasis functions.
        """
        n = stats['n']
        clustering = stats['clustering_level']
        
        # Ensure minimum connectivity
        min_edges = min(max(4, int(np.log2(n+1)) * 2), n - 1)
        if clustering > 0.5:
            min_edges = int(min_edges * 1.2)
        
        for i in range(n):
            non_zero = np.where(H[i] > 0)[0]
            if len(non_zero) < min_edges:
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                sorted_idx = np.argsort(dists)
                
                added = 0
                for j in sorted_idx:
                    if added >= min_edges - len(non_zero):
                        break
                    if H[i, j] == 0 and j != i:
                        # Add with score based on inverse distance
                        H[i, j] = H[j, i] = 1 / (dists[j] + 1e-8)
                        added += 1
        
        # Adaptive thresholding per node
        for i in range(n):
            row = H[i].copy()
            non_zero = row[row > 0]
            if len(non_zero) > min_edges:
                # Dynamic threshold based on distribution
                if clustering > 0.6:
                    # Keep more edges in clustered instances
                    threshold = np.percentile(non_zero, 40)
                else:
                    threshold = np.percentile(non_zero, 30)
                H[i, row < threshold] = 0
        
        # Symmetrize
        H = (H + H.T) / 2
        
        # Apply emphasis function
        max_val = np.max(H)
        if max_val > 0:
            H = H / max_val
            
            # Different emphasis based on clustering
            if clustering > 0.6:
                # Stronger emphasis for clustered instances
                H = np.tanh(3 * H)
            else:
                # Sigmoid emphasis
                H = 1 / (1 + np.exp(-8 * (H - 0.6)))
        
        # Final cleanup
        np.fill_diagonal(H, 0)
        
        # Ensure at least min_edges per node
        for i in range(n):
            non_zero = np.where(H[i] > 0)[0]
            if len(non_zero) < min_edges:
                # Add nearest neighbors
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                sorted_idx = np.argsort(dists)
                for j in sorted_idx[:min_edges - len(non_zero)]:
                    if H[i, j] == 0 and j != i:
                        H[i, j] = H[j, i] = 0.1  # Small but non-zero
        
        return H
    
    # ==================== MAIN EXECUTION ====================
    
    # Step 1: Enhanced instance analysis
    stats = analyze_instance(distance_matrix)
    
    # Step 2: Adaptive candidate mask
    candidate_mask = build_candidate_mask(distance_matrix, stats)
    
    # Step 3: Compute factors with clustering awareness
    factors = compute_factors(distance_matrix, stats, candidate_mask)
    
    # Step 4: Optimize weights using reinforcement-guided learning
    weights = optimize_weights(distance_matrix, factors, candidate_mask, stats)
    
    # Step 5: Combine factors
    H = combine_factors(factors, weights, candidate_mask)
    
    # Step 6: Intelligent sparsification
    H = sparsify_heuristics(H, distance_matrix, candidate_mask, stats)
    
    return H

            cluster_size = n // clusters
            
            for c in range(clusters):
                center = np.random.rand(2) * 100
                for _ in range(cluster_size if c < clusters-1 else n - (clusters-1)*cluster_size):
                    if np.random.random() < cluster_strength:
                        # Points close to cluster center
                        point = center + np.random.randn(2) * (5 + 10*(1-cluster_strength))
                    else:
                        # Random point
                        point = np.random.rand(2) * 100
                    points.append(point)
            
            points = np.array(points)[:n]  # Ensure exactly n points
        else:
            # Uniform random instance
            points = np.random.rand(n, 2) * 100
        
        dist_matrix = squareform(pdist(points, 'euclidean'))
        
        # Compute heuristics
        H = heuristics_v2(dist_matrix)
        
        # Statistics
        print(f"Shape: {H.shape}")
        print(f"Non-zero entries: {np.count_nonzero(H)}")
        print(f"Sparsity: {(H == 0).sum() / H.size:.1%}")
        print(f"Mean non-zero score: {np.mean(H[H > 0]):.4f}")
        print(f"Std of scores: {np.std(H[H > 0]):.4f}")
        
        # Verify symmetry
        symmetric_diff = np.max(np.abs(H - H.T))
        print(f"Max asymmetry: {symmetric_diff:.6f}")
        
        # Show top edges if instance is small enough
        if n <= 30:
            triu_idx = np.triu_indices_from(H, k=1)
            top_k = min(8, len(triu_idx[0]))
            if top_k > 0:
                values = H[triu_idx]
                top_indices = np.argsort(values)[-top_k:][::-1]
                
                print("\nTop edges (i, j): distance, heuristic score")
                print("-" * 50)
                for idx in top_indices:
                    i, j = triu_idx[0][idx], triu_idx[1][idx]
                    print(f"  ({i:2d}, {j:2d}): dist={dist_matrix[i,j]:6.2f}, score={H[i,j]:.4f}")
