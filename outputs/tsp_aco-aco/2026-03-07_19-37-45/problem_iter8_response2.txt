def _analyze_instance(distance_matrix: np.ndarray) -> dict:
    """Comprehensive instance analysis for adaptive parameter tuning."""
    n = distance_matrix.shape[0]
    
    # Distance statistics
    triu_indices = np.triu_indices(n, k=1)
    flat_dists = distance_matrix[triu_indices]
    
    mean_dist = np.mean(flat_dists)
    std_dist = np.std(flat_dists)
    cv_dist = std_dist / (mean_dist + 1e-12)
    
    # Clustering analysis
    cluster_score = _estimate_clustering_tendency(distance_matrix)
    
    # Dimensionality analysis (estimating if points are in low/high dimension)
    dim_score = _estimate_dimensionality(distance_matrix)
    
    # Symmetry and triangle inequality violation analysis
    symmetry_error = np.max(np.abs(distance_matrix - distance_matrix.T))
    triangle_violations = _check_triangle_inequality(distance_matrix)
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'cluster_score': cluster_score,
        'dim_score': dim_score,
        'symmetry_error': symmetry_error,
        'triangle_violations': triangle_violations,
        'is_euclidean': symmetry_error < 1e-10 and triangle_violations < 0.1
    }

def _estimate_clustering_tendency(distance_matrix: np.ndarray) -> float:
    """Estimate the clustering tendency using Hopkins statistic."""
    n = distance_matrix.shape[0]
    if n < 10:
        return 0.5
    
    # Sample m points
    m = min(20, n // 3)
    sample_indices = np.random.choice(n, m, replace=False)
    
    # For each sample, find nearest neighbor distance
    sample_nn_dists = []
    random_nn_dists = []
    
    for idx in sample_indices:
        dists = distance_matrix[idx].copy()
        dists[idx] = np.inf
        sample_nn_dists.append(np.min(dists))
    
    # Generate random distances for comparison
    for _ in range(m):
        # Sample two random nodes
        i, j = np.random.choice(n, 2, replace=False)
        random_nn_dists.append(distance_matrix[i, j])
    
    hopkins = np.sum(sample_nn_dists) / (np.sum(sample_nn_dists) + np.sum(random_nn_dists) + 1e-12)
    
    return min(max(hopkins, 0.0), 1.0)

def _estimate_dimensionality(distance_matrix: np.ndarray) -> float:
    """Estimate if distances behave like low-dimensional Euclidean space."""
    n = distance_matrix.shape[0]
    if n < 5:
        return 0.5
    
    # Use triangle inequality violations as proxy
    violations = 0
    total_triples = 0
    
    for i in range(n):
        for j in range(i+1, n):
            for k in range(j+1, n):
                total_triples += 1
                if (distance_matrix[i, j] > distance_matrix[i, k] + distance_matrix[k, j] + 1e-6 or
                    distance_matrix[i, k] > distance_matrix[i, j] + distance_matrix[j, k] + 1e-6 or
                    distance_matrix[j, k] > distance_matrix[j, i] + distance_matrix[i, k] + 1e-6):
                    violations += 1
    
    violation_rate = violations / (total_triples + 1e-12)
    dim_score = 1.0 - min(violation_rate, 1.0)
    
    return dim_score

def _check_triangle_inequality(distance_matrix: np.ndarray) -> float:
    """Check triangle inequality violations."""
    n = distance_matrix.shape[0]
    violations = 0
    
    # Sample triplets
    samples = min(1000, n * (n-1) * (n-2) // 6)
    
    for _ in range(samples):
        i, j, k = np.random.choice(n, 3, replace=False)
        if distance_matrix[i, j] > distance_matrix[i, k] + distance_matrix[k, j] + 1e-6:
            violations += 1
    
    return violations / (samples + 1e-12)

def _compute_multi_scale_factors(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute multiple heuristic factors at different scales."""
    n = stats['n']
    eps = 1e-12
    
    factors = {}
    
    # 1. Inverse distance with adaptive exponent
    power = 1.0 + 0.5 * stats['cv_dist'] + 0.3 * (1 - stats['cluster_score'])
    inv_dist = 1 / (distance_matrix + np.eye(n) * eps)
    np.fill_diagonal(inv_dist, 0)
    factors['inv_dist'] = inv_dist ** power
    
    # 2. Adaptive regret with multi-scale analysis
    factors['regret'] = _compute_adaptive_regret(distance_matrix, stats)
    
    # 3. Density-based scoring with cluster awareness
    factors['density'] = _compute_cluster_aware_density(distance_matrix, stats)
    
    # 4. MST-based scoring with confidence weighting
    factors['mst'] = _compute_mst_scores(distance_matrix, stats)
    
    # 5. Convex hull scoring (for Euclidean-like instances)
    factors['convex'] = _compute_convex_hull_scores(distance_matrix, stats)
    
    # 6. Neighborhood preservation score
    factors['neighborhood'] = _compute_neighborhood_preservation(distance_matrix, stats)
    
    return factors

def _compute_adaptive_regret(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute regret scores with adaptive k and lookahead."""
    n = stats['n']
    eps = 1e-12
    regret = np.ones((n, n))
    
    # Adaptive k based on instance characteristics
    base_k = min(10, max(3, int(5 + 2 * stats['cv_dist'] * 5)))
    k = min(base_k, n - 1)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Get k+2 nearest neighbors for deeper analysis
        if k + 2 < len(dists):
            sorted_idx = np.argsort(dists)[:k+2]
        else:
            sorted_idx = np.argsort(dists)
        
        if len(sorted_idx) >= 3:
            best_dist = dists[sorted_idx[0]]
            second_best = dists[sorted_idx[1]]
            
            # Compute regret for each potential edge from i
            for idx, j in enumerate(sorted_idx):
                if j == i:
                    continue
                
                # Base regret (distance penalty relative to best)
                edge_regret = max(0, dists[j] - best_dist) / (best_dist + eps)
                
                # Lookahead regret: consider 2-hop alternatives
                if idx > 1:  # Not the first two choices
                    # Find best 2-hop path through intermediate nodes
                    intermediates = sorted_idx[:min(3, len(sorted_idx))]
                    intermediates = [x for x in intermediates if x != i and x != j]
                    
                    if len(intermediates) > 0:
                        two_hop_lengths = distance_matrix[i, intermediates] + distance_matrix[intermediates, j]
                        best_two_hop = np.min(two_hop_lengths) if len(two_hop_lengths) > 0 else np.inf
                        
                        if best_two_hop < np.inf:
                            two_hop_regret = max(0, dists[j] - best_two_hop) / (best_two_hop + eps)
                            edge_regret = max(edge_regret, two_hop_regret)
                
                # Sigmoid transformation for smooth penalty
                penalty = 1 / (1 + np.exp(2 * edge_regret))
                regret[i, j] = penalty
    
    # Symmetrize
    regret = (regret + regret.T) / 2
    np.fill_diagonal(regret, 0)
    
    return regret

def _compute_cluster_aware_density(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute density scores with cluster awareness."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Estimate local density at multiple scales
    k_small = min(5, max(2, n // 20))
    k_large = min(15, max(5, n // 10))
    
    small_density = np.zeros(n)
    large_density = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Small-scale density
        if len(dists) >= k_small:
            k_nearest = np.partition(dists, k_small)[:k_small]
            small_density[i] = 1 / (np.mean(k_nearest) + 1e-12)
        
        # Large-scale density
        if len(dists) >= k_large:
            k_nearest = np.partition(dists, k_large)[:k_large]
            large_density[i] = 1 / (np.mean(k_nearest) + 1e-12)
    
    # Normalize densities
    if np.std(small_density) > 1e-12:
        small_density = (small_density - np.mean(small_density)) / (np.std(small_density) + 1e-12)
    if np.std(large_density) > 1e-12:
        large_density = (large_density - np.mean(large_density)) / (np.std(large_density) + 1e-12)
    
    # Create density scores favoring different connections based on clustering
    for i in range(n):
        for j in range(i+1, n):
            # Intra-cluster connections (both nodes have high small-scale density)
            intra_cluster = np.exp(-0.5 * (abs(small_density[i]) + abs(small_density[j])))
            
            # Inter-cluster connections (nodes in different density regions)
            density_diff = abs(small_density[i] - small_density[j])
            inter_cluster = np.exp(-0.3 * density_diff)
            
            # Combined score: bias based on cluster tendency
            if stats['cluster_score'] > 0.7:
                # Strong clustering: favor intra-cluster connections
                density_score = 0.7 * intra_cluster + 0.3 * inter_cluster
            elif stats['cluster_score'] < 0.3:
                # Weak clustering: favor inter-cluster connections
                density_score = 0.3 * intra_cluster + 0.7 * inter_cluster
            else:
                # Balanced
                density_score = 0.5 * intra_cluster + 0.5 * inter_cluster
            
            density_scores[i, j] = density_scores[j, i] = density_score
    
    return density_scores

def _compute_mst_scores(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute MST-based scores with confidence weighting."""
    n = stats['n']
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            # Compute MST
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Boost edges in MST, with stronger boost for Euclidean-like instances
            if stats['is_euclidean']:
                boost = 2.5
            else:
                boost = 2.0
            
            mst_scores[mst_present] = boost
            
            # Also give partial credit to edges that are almost in MST
            # (edges that would be added if we expanded MST by one more edge)
            if n > 10:
                # Find edges that are just outside MST
                mst_edges = np.where(mst_present)
                if len(mst_edges[0]) > 0:
                    # For each node, consider edges to its nearest non-MST neighbors
                    for i in range(n):
                        dists = distance_matrix[i].copy()
                        dists[i] = np.inf
                        
                        # Find nearest nodes not connected in MST
                        sorted_idx = np.argsort(dists)
                        for j in sorted_idx[:10]:
                            if j != i and not mst_present[i, j]:
                                # Check if this edge is close to being in MST
                                # (it connects nodes that are relatively close compared to their MST connections)
                                mst_dist_i = np.min(distance_matrix[i][mst_present[i]]) if np.any(mst_present[i]) else np.inf
                                mst_dist_j = np.min(distance_matrix[j][mst_present[j]]) if np.any(mst_present[j]) else np.inf
                                
                                if (distance_matrix[i, j] < 1.5 * min(mst_dist_i, mst_dist_j) + 1e-12):
                                    mst_scores[i, j] = mst_scores[j, i] = 1.5
                                break
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores

def _compute_convex_hull_scores(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute scores based on convex hull properties (for Euclidean-like instances)."""
    n = stats['n']
    convex_scores = np.ones((n, n))
    
    if not stats['is_euclidean'] or n < 4:
        return convex_scores
    
    # Approximate convex hull using distances
    # Nodes with high average distance are likely on convex hull
    avg_dists = np.mean(distance_matrix, axis=1)
    hull_likelihood = (avg_dists - np.min(avg_dists)) / (np.max(avg_dists) - np.min(avg_dists) + 1e-12)
    
    # Boost edges between nodes likely on convex hull
    for i in range(n):
        for j in range(i+1, n):
            hull_score = np.sqrt(hull_likelihood[i] * hull_likelihood[j])
            convex_scores[i, j] = convex_scores[j, i] = 1.0 + 0.5 * hull_score
    
    return convex_scores

def _compute_neighborhood_preservation(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute scores based on neighborhood preservation (local structure)."""
    n = stats['n']
    neighborhood_scores = np.ones((n, n))
    
    if n < 5:
        return neighborhood_scores
    
    # For each node, find its k-nearest neighbors
    k = min(8, max(3, n // 10))
    
    # Precompute neighborhoods
    neighborhoods = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if len(dists) >= k:
            nn_indices = np.argsort(dists)[:k]
        else:
            nn_indices = np.argsort(dists)
        neighborhoods.append(set(nn_indices))
    
    # Score edges based on neighborhood overlap
    for i in range(n):
        for j in range(i+1, n):
            if len(neighborhoods[i]) > 0 and len(neighborhoods[j]) > 0:
                overlap = len(neighborhoods[i].intersection(neighborhoods[j]))
                total = len(neighborhoods[i].union(neighborhoods[j]))
                jaccard = overlap / (total + 1e-12)
                
                # Higher score for edges connecting nodes with overlapping neighborhoods
                neighborhood_scores[i, j] = neighborhood_scores[j, i] = 1.0 + jaccard
    
    return neighborhood_scores

def _compute_advanced_lookahead(distance_matrix: np.ndarray, stats: dict, factors: dict) -> np.ndarray:
    """Compute lookahead scores with 2-opt evaluation and adaptive simulation."""
    n = stats['n']
    lookahead = np.ones((n, n))
    
    if n < 5:
        return lookahead
    
    # Adaptive number of simulations
    n_simulations = min(30, max(5, int(np.log2(n) * 8)))
    
    # Pre-select candidate edges for lookahead (top edges by base score)
    base_score = np.ones((n, n))
    for factor_name, factor_matrix in factors.items():
        if factor_name in ['inv_dist', 'regret', 'density']:
            base_score *= factor_matrix
    
    # Select top edges per node
    candidate_edges = set()
    edges_per_node = min(15, max(5, n // 5))
    
    for i in range(n):
        scores_i = base_score[i].copy()
        scores_i[i] = -np.inf
        
        top_indices = np.argsort(scores_i)[-edges_per_node:]
        for j in top_indices:
            if i < j:
                candidate_edges.add((i, j))
            else:
                candidate_edges.add((j, i))
    
    # Compute lookahead for candidate edges
    for i, j in candidate_edges:
        scores = []
        for _ in range(n_simulations):
            score = _simulate_advanced_lookahead(distance_matrix, i, j, stats, factors)
            scores.append(score)
        
        if scores:
            lookahead[i, j] = lookahead[j, i] = np.median(scores)
    
    return lookahead

def _simulate_advanced_lookahead(distance_matrix: np.ndarray, start_i: int, start_j: int, 
                                 stats: dict, factors: dict) -> float:
    """Simulate partial tour with 2-opt evaluation."""
    n = stats['n']
    visited = np.zeros(n, dtype=bool)
    visited[start_i] = visited[start_j] = True
    
    # Start with the initial edge
    current_path = [start_i, start_j]
    path_length = distance_matrix[start_i, start_j]
    
    # Extend tour using a combination of heuristics
    n_steps = min(8, n - 2)
    
    for step in range(n_steps):
        current_node = current_path[-1]
        unvisited = np.where(~visited)[0]
        
        if len(unvisited) == 0:
            break
        
        # Choose next node using multi-criteria decision
        candidates = []
        for node in unvisited:
            # Compute score based on multiple factors
            dist_score = 1 / (distance_matrix[current_node, node] + 1e-12)
            
            # Regret-aware score (look at alternatives)
            dists_from_current = distance_matrix[current_node, unvisited]
            if len(dists_from_current) >= 2:
                sorted_dists = np.sort(dists_from_current)
                regret = (sorted_dists[1] - distance_matrix[current_node, node]) / (distance_matrix[current_node, node] + 1e-12)
                regret_score = 1 / (1 + np.exp(2 * max(0, regret)))
            else:
                regret_score = 1.0
            
            # Density-aware score
            if current_node < node:
                density_score = factors['density'][current_node, node]
            else:
                density_score = factors['density'][node, current_node]
            
            combined_score = dist_score * regret_score * density_score
            candidates.append((combined_score, node))
        
        # Choose next node with some randomness
        candidates.sort(reverse=True)
        if len(candidates) >= 3:
            # Select from top 3 with probability based on scores
            top_scores = [c[0] for c in candidates[:3]]
            top_nodes = [c[1] for c in candidates[:3]]
            probs = np.array(top_scores) / (np.sum(top_scores) + 1e-12)
            next_node = np.random.choice(top_nodes, p=probs)
        else:
            next_node = candidates[0][1]
        
        # Add to path
        path_length += distance_matrix[current_node, next_node]
        current_path.append(next_node)
        visited[next_node] = True
    
    # Try to improve the partial path with 2-opt moves
    if len(current_path) >= 4:
        improved_length = _apply_2opt_to_partial(current_path, distance_matrix, path_length)
        path_length = min(path_length, improved_length)
    
    # Score is inverse of normalized length
    expected_length = len(current_path) * stats['mean_dist']
    score = expected_length / (path_length + 1e-12)
    
    return min(score, 5.0)  # Cap extreme values

def _apply_2opt_to_partial(path: list, distance_matrix: np.ndarray, current_length: float) -> float:
    """Apply 2-opt improvements to a partial path."""
    n_path = len(path)
    if n_path < 4:
        return current_length
    
    improved = True
    best_length = current_length
    
    while improved:
        improved = False
        for i in range(1, n_path - 2):
            for j in range(i + 1, n_path - 1):
                # Check if 2-opt swap would improve
                old_edges = (distance_matrix[path[i-1], path[i]] + 
                            distance_matrix[path[j], path[j+1]])
                new_edges = (distance_matrix[path[i-1], path[j]] + 
                            distance_matrix[path[i], path[j+1]])
                
                if new_edges < old_edges - 1e-6:
                    # Perform the swap
                    path[i:j+1] = reversed(path[i:j+1])
                    best_length = best_length - old_edges + new_edges
                    improved = True
        
        # Limit iterations
        if improved:
            break
    
    return best_length

def _compute_meta_heuristic_scores(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Learn edge preferences from random solution sampling."""
    n = stats['n']
    meta_scores = np.ones((n, n))
    
    if n < 10:
        return meta_scores
    
    # Sample random tours and collect edge statistics
    n_samples = min(200, max(50, n * 5))
    edge_counts = defaultdict(int)
    edge_quality = defaultdict(float)
    
    for _ in range(n_samples):
        # Generate a random tour
        tour = np.random.permutation(n)
        tour_length = 0
        
        # Calculate tour length and edge qualities
        for idx in range(n):
            i = tour[idx]
            j = tour[(idx + 1) % n]
            
            edge = (min(i, j), max(i, j))
            edge_counts[edge] += 1
            
            # Estimate edge quality based on its position in the tour
            # Edges that appear in short tours are better
            tour_length += distance_matrix[i, j]
        
        # Update edge qualities
        for idx in range(n):
            i = tour[idx]
            j = tour[(idx + 1) % n]
            edge = (min(i, j), max(i, j))
            
            # Quality inversely proportional to tour length
            edge_quality[edge] += 1.0 / (tour_length + 1e-12)
    
    # Convert to score matrix
    for (i, j), count in edge_counts.items():
        if count > 0:
            # Score based on frequency and average quality
            freq_score = count / n_samples
            quality_score = edge_quality[(i, j)] / (count + 1e-12)
            
            # Combine with emphasis on consistency
            meta_scores[i, j] = meta_scores[j, i] = 1.0 + 2.0 * freq_score * quality_score
    
    return meta_scores

def _optimize_weights_adaptively(distance_matrix: np.ndarray, stats: dict, 
                                factors: dict, lookahead_scores: np.ndarray,
                                meta_scores: np.ndarray) -> dict:
    """Optimize factor weights adaptively based on instance characteristics."""
    n = stats['n']
    cv = stats['cv_dist']
    cluster = stats['cluster_score']
    dim = stats['dim_score']
    
    # Base weights adjusted by instance properties
    weights = {
        'inv_dist': 1.2 - 0.3 * cv,  # Less important when distances vary greatly
        'regret': 0.8 + 0.4 * cv,    # More important when distances vary
        'density': 0.6 + 0.8 * cluster,  # More important with clustering
        'mst': 0.7 + 0.3 * dim,      # More important in Euclidean-like spaces
        'convex': 0.3 * dim,         # Only relevant for Euclidean-like
        'neighborhood': 0.5 + 0.3 * (1 - cv),  # More important when distances are uniform
        'lookahead': 0.4 + 0.4 * min(1.0, np.log2(n) / 10),  # Scale with problem size
        'meta': 0.3 + 0.2 * (n / 100)  # More important for larger instances
    }
    
    # Normalize weights to sum to approximately 1 for each edge combination
    total = sum(weights.values())
    normalization = total / 5.0  # Target sum of 5 for multiplicative combination
    
    for key in weights:
        weights[key] = weights[key] / normalization
    
    return weights

def _combine_factors_hierarchically(factors: dict, lookahead_scores: np.ndarray,
                                   meta_scores: np.ndarray, weights: dict) -> np.ndarray:
    """Combine all factors hierarchically with optimized weights."""
    n = factors['inv_dist'].shape[0]
    H = np.ones((n, n))
    
    # Combine base factors
    for i in range(n):
        for j in range(i+1, n):
            combined = 1.0
            
            # Base factors with their weights as exponents
            for factor_name, factor_matrix in factors.items():
                if factor_name in weights:
                    weight = weights[factor_name]
                    combined *= (factor_matrix[i, j] ** weight)
            
            # Lookahead score
            combined *= (lookahead_scores[i, j] ** weights['lookahead'])
            
            # Meta-heuristic score
            combined *= (meta_scores[i, j] ** weights['meta'])
            
            H[i, j] = H[j, i] = combined
    
    np.fill_diagonal(H, 0)
    return H

def _adaptive_sparsification(H: np.ndarray, distance_matrix: np.ndarray, 
                            stats: dict, weights: dict) -> np.ndarray:
    """Intelligent sparsification preserving important connections."""
    n = stats['n']
    sparsified = np.zeros_like(H)
    
    # Determine sparsity level based on instance size and characteristics
    if n < 30:
        edges_per_node = min(8, n - 1)
    elif n < 100:
        edges_per_node = min(12, int(n * 0.15))
    else:
        edges_per_node = min(20, int(n * 0.1))
    
    # Adjust based on clustering tendency (more edges for clustered instances)
    edges_per_node = int(edges_per_node * (1.0 + 0.3 * stats['cluster_score']))
    
    # Keep top edges for each node
    for i in range(n):
        scores_i = H[i].copy()
        scores_i[i] = -np.inf
        
        top_indices = np.argsort(scores_i)[-edges_per_node:]
        sparsified[i, top_indices] = scores_i[top_indices]
    
    # Ensure symmetry and minimum connectivity
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Ensure each node has at least 2 connections
    for i in range(n):
        non_zero = np.where(sparsified[i] > 0)[0]
        if len(non_zero) < 2:
            # Add nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Find nearest nodes not already connected
            sorted_idx = np.argsort(dists)
            added = 0
            for j in sorted_idx:
                if j != i and sparsified[i, j] == 0:
                    sparsified[i, j] = sparsified[j, i] = H[i, j]
                    added += 1
                    if added >= 2 - len(non_zero):
                        break
    
    return sparsified

def _final_normalization(H: np.ndarray, distance_matrix: np.ndarray) -> np.ndarray:
    """Final normalization and cleanup."""
    n = H.shape[0]
    
    # Ensure symmetry
    H = (H + H.T) / 2
    
    # Zero diagonal
    np.fill_diagonal(H, 0)
    
    # Normalize to [0, 1]
    max_val = np.max(H)
    if max_val > 1e-12:
        H = H / max_val
    
    return H
