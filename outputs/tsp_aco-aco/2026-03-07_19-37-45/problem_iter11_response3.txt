import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic for TSP edge selection with adaptive multi-hop regret and candidate masking.
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Compute instance statistics
    stats = _compute_instance_stats(distance_matrix)
    
    # Build candidate mask (only consider promising edges)
    candidate_mask = _build_adaptive_candidate_mask(distance_matrix, stats)
    
    # Initialize heuristic matrix
    H = np.zeros((n, n))
    
    # Compute scores only for candidate edges
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Compute multi-hop regret ratio with adaptive penalty
                regret_score = _compute_multi_hop_regret(distance_matrix, i, j, stats)
                
                # Compute inverse distance with variance adaptation
                inv_dist = 1 / (distance_matrix[i, j] + 1e-8)
                variance_factor = 1 / (1 + stats['cv_dist'])  # Penalize less when variance is high
                distance_score = inv_dist ** variance_factor
                
                # Combine factors with geometric mean
                combined = (regret_score * distance_score) ** 0.5
                H[i, j] = H[j, i] = combined
    
    # Apply sparsity-preserving normalization
    H = _normalize_with_sparsity_preservation(H, candidate_mask, stats)
    
    return H


def _compute_instance_stats(distance_matrix: np.ndarray) -> dict:
    """Compute basic instance statistics."""
    n = distance_matrix.shape[0]
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Compute kurtosis as measure of regularity
    kurtosis = np.mean((dist_no_diag - mean_dist) ** 4) / (std_dist ** 4 + 1e-8)
    regularity = 1 / (1 + np.exp(-0.1 * (kurtosis - 3)))  # Sigmoid centered at normal kurtosis
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'regularity': regularity,
        'dist_no_diag': dist_no_diag
    }


def _build_adaptive_candidate_mask(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Build mask of candidate edges using adaptive k-nearest neighbors."""
    n = stats['n']
    
    # Adaptive k based on instance size and regularity
    base_k = min(15, max(5, int(np.log2(n) * 4)))
    k = int(base_k * (1.5 - 0.5 * stats['regularity']))  # More candidates for irregular instances
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < len(dists):
            # Get indices of k smallest distances
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Ensure symmetry
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask


def _compute_multi_hop_regret(distance_matrix: np.ndarray, i: int, j: int, stats: dict) -> float:
    """Compute adaptive multi-hop regret ratio with penalty/bonus."""
    n = stats['n']
    d_ij = distance_matrix[i, j]
    
    if n < 4:
        return 1.0
    
    # Find 2-hop alternative paths (i -> k -> j)
    alternative_2hop_costs = []
    for k in range(n):
        if k != i and k != j:
            alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
            alternative_2hop_costs.append(alt_cost)
    
    if not alternative_2hop_costs:
        return 1.0
    
    # Get best 2-hop alternative
    best_2hop = min(alternative_2hop_costs)
    
    # Compute regret ratio
    if d_ij > 0:
        regret_ratio = best_2hop / (d_ij + 1e-8)
    else:
        regret_ratio = 1.0
    
    # Apply adaptive penalty/bonus based on ratio
    if regret_ratio > 1.0:  # Direct edge is better than best alternative
        # Bonus for direct edge superiority
        bonus = np.exp(min(1.0, regret_ratio - 1.0))
        return bonus
    else:  # Alternative is better than direct edge
        # Penalty based on variance
        variance_factor = 1 / (1 + 2 * stats['cv_dist'])  # Less penalty when variance is high
        penalty = max(0.1, regret_ratio ** variance_factor)
        return penalty


def _normalize_with_sparsity_preservation(H: np.ndarray, candidate_mask: np.ndarray, 
                                         stats: dict) -> np.ndarray:
    """Normalize heuristic matrix while preserving sparsity pattern."""
    n = stats['n']
    
    # Only normalize non-zero entries
    non_zero_mask = H > 0
    if np.any(non_zero_mask):
        # Get percentiles for robust normalization
        non_zero_vals = H[non_zero_mask]
        p95 = np.percentile(non_zero_vals, 95) if len(non_zero_vals) > 0 else 1.0
        
        if p95 > 0:
            # Soft normalization: scale by 95th percentile, preserving zeros
            H = np.where(non_zero_mask, H / p95, 0)
    
    # Ensure minimum connectivity for each node
    min_edges = min(3, n - 1)
    for i in range(n):
        candidate_indices = np.where(candidate_mask[i])[0]
        if len(candidate_indices) < min_edges:
            # Add more nearest neighbors
            dists = stats['dist_no_diag'][i]
            additional_needed = min_edges - len(candidate_indices)
            if additional_needed > 0:
                all_indices = np.arange(n)
                all_indices = all_indices[all_indices != i]
                sorted_by_dist = all_indices[np.argsort(dists)]
                additional = sorted_by_dist[:additional_needed]
                
                for j in additional:
                    if H[i, j] == 0:
                        # Assign base score
                        base_score = 1 / (dists[np.where(all_indices == j)[0][0]] + 1e-8)
                        H[i, j] = H[j, i] = base_score
    
    # Final symmetry and diagonal zeroing
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    return H
