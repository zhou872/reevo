import numpy as np
import random
from typing import Dict, Tuple, List
from scipy.sparse.csgraph import minimum_spanning_tree
import networkx as nx

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic with multi-hop regret, learned weight optimization,
    and dynamic centrality integration.
    
    Returns a heuristic matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Instance analysis with enhanced metrics
    instance_stats = _analyze_instance_v3(distance_matrix)
    
    # Step 2: Build dynamic candidate mask
    candidate_mask = _build_dynamic_candidate_mask(distance_matrix, instance_stats)
    
    # Step 3: Compute comprehensive factors
    factors = _compute_enhanced_factors(distance_matrix, instance_stats, candidate_mask)
    
    # Step 4: Reinforcement-inspired weight optimization
    weights = _optimize_weights_reinforcement(distance_matrix, factors, instance_stats, candidate_mask)
    
    # Step 5: Combine factors with learned weights
    H = _combine_factors_with_weights(factors, weights, candidate_mask)
    
    # Step 6: Predictive sparsification and refinement
    H = _predictive_sparsification(H, distance_matrix, instance_stats, candidate_mask)
    
    return H


def _analyze_instance_v3(distance_matrix: np.ndarray) -> Dict:
    """Enhanced instance analysis with multi-scale metrics."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    dist_vals = dist_no_diag.flatten()
    
    # Basic statistics
    mean_dist = np.mean(dist_vals)
    std_dist = np.std(dist_vals)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Distance percentiles for distribution analysis
    percentiles = np.percentile(dist_vals, [10, 25, 50, 75, 90])
    
    # Multi-scale neighborhood analysis
    neighborhood_metrics = _analyze_neighborhood_scales(distance_matrix)
    
    # Centrality dispersion
    centrality_vals = _compute_closeness_centralities(distance_matrix)
    centrality_cv = np.std(centrality_vals) / (np.mean(centrality_vals) + 1e-8)
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'percentiles': percentiles,
        'p10': percentiles[0],
        'p50': percentiles[2],
        'p90': percentiles[4],
        'neighborhood_consistency': neighborhood_metrics['consistency'],
        'multi_scale_variance': neighborhood_metrics['variance'],
        'centrality_cv': min(1.0, centrality_cv * 2),
        'distance_skew': _compute_distance_skewness(distance_matrix)
    }


def _analyze_neighborhood_scales(distance_matrix: np.ndarray) -> Dict:
    """Analyze neighborhood relationships at different scales."""
    n = distance_matrix.shape[0]
    
    if n < 5:
        return {'consistency': 0.5, 'variance': 0.5}
    
    scales = [3, 5, min(8, n//2)]
    consistency_scores = []
    variance_scores = []
    
    for k in scales:
        if k >= n:
            continue
            
        neighbor_maps = []
        density_vals = []
        
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            neighbors = np.argsort(dists)[:k]
            neighbor_maps.append(set(neighbors))
            density_vals.append(np.mean(dists[neighbors]))
        
        # Compute neighborhood consistency
        reciprocal_count = 0
        total_pairs = 0
        
        for i in range(n):
            for j in range(i+1, n):
                if j in neighbor_maps[i] and i in neighbor_maps[j]:
                    reciprocal_count += 1
                total_pairs += 1
        
        consistency_scores.append(reciprocal_count / max(1, total_pairs))
        variance_scores.append(np.std(density_vals) / (np.mean(density_vals) + 1e-8))
    
    return {
        'consistency': np.mean(consistency_scores) if consistency_scores else 0.5,
        'variance': np.mean(variance_scores) if variance_scores else 0.5
    }


def _compute_closeness_centralities(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute closeness centrality for all nodes."""
    n = distance_matrix.shape[0]
    centralities = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        avg_dist = np.mean(dists)
        centralities[i] = 1 / (avg_dist + 1e-8)
    
    return centralities


def _compute_distance_skewness(distance_matrix: np.ndarray) -> float:
    """Compute skewness of distance distribution."""
    n = distance_matrix.shape[0]
    dists = distance_matrix[~np.eye(n, dtype=bool)]
    
    if len(dists) < 2:
        return 0.5
    
    # Normalized skewness measure
    mean = np.mean(dists)
    std = np.std(dists)
    if std < 1e-8:
        return 0.5
    
    skewness = np.mean(((dists - mean) / std) ** 3)
    return 1 / (1 + np.exp(-5 * abs(skewness)))


def _build_dynamic_candidate_mask(distance_matrix: np.ndarray, 
                                 stats: Dict) -> np.ndarray:
    """Build adaptive candidate mask based on instance structure."""
    n = stats['n']
    
    # Base candidate count
    base_k = min(15, max(4, int(np.sqrt(n) + 3)))
    
    # Adjust based on instance characteristics
    # More candidates for structured instances
    structure_factor = 0.5 * stats['neighborhood_consistency'] + 0.5 * stats['multi_scale_variance']
    k = int(base_k * (1.0 + structure_factor))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # k-nearest neighbors
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < n - 1:
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
        
        # Add promising edges based on centrality gaps
        centrality = _compute_closeness_centralities(distance_matrix)
        sorted_by_centrality = np.argsort(centrality)[::-1]  # Descending
        
        # Connect to a few high-centrality nodes
        high_central_count = min(3, n // 4)
        for j in sorted_by_centrality[:high_central_count]:
            if j != i and not candidate_mask[i, j]:
                candidate_mask[i, j] = True
    
    # Ensure symmetry
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask


def _compute_enhanced_factors(distance_matrix: np.ndarray,
                             stats: Dict,
                             candidate_mask: np.ndarray) -> Dict:
    """Compute all heuristic factors including multi-hop regret."""
    n = stats['n']
    factors = {}
    
    # 1. Inverse distance (normalized)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    # Normalize by median
    non_zero = inv_dist[inv_dist > 0]
    if len(non_zero) > 0:
        inv_dist = inv_dist / np.median(non_zero)
    factors['inv_dist'] = inv_dist
    
    # 2. Multi-hop regret
    factors['regret'] = _compute_multi_hop_regret(distance_matrix, stats, candidate_mask)
    
    # 3. Dynamic density scoring
    factors['density'] = _compute_dynamic_density(distance_matrix, stats)
    
    # 4. MST-based scoring
    factors['mst'] = _compute_enhanced_mst_score(distance_matrix)
    
    # 5. Adaptive centrality scoring
    factors['centrality'] = _compute_adaptive_centrality(distance_matrix, stats)
    
    return factors


def _compute_multi_hop_regret(distance_matrix: np.ndarray,
                             stats: Dict,
                             candidate_mask: np.ndarray) -> np.ndarray:
    """Compute regret scores considering 2-hop and 3-hop alternatives."""
    n = stats['n']
    regret = np.ones((n, n))
    
    # Adaptive parameters
    max_hops = min(3, max(2, n // 10))
    max_alternatives = min(20, n - 2)
    
    for i in range(n):
        # Get candidate neighbors for this node
        i_candidates = np.where(candidate_mask[i])[0]
        
        for j in i_candidates:
            if j <= i:
                continue
                
            if not candidate_mask[i, j]:
                regret[i, j] = 0.1
                regret[j, i] = 0.1
                continue
            
            direct_cost = distance_matrix[i, j]
            
            # Find best alternative paths
            best_alternative = np.inf
            
            # Consider 2-hop paths through candidate nodes
            for k in i_candidates:
                if k == j:
                    continue
                
                if candidate_mask[k, j]:
                    alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                    best_alternative = min(best_alternative, alt_cost)
            
            # Consider 3-hop paths if n is large enough
            if max_hops >= 3 and n >= 8:
                # Sample a few 3-hop paths
                sample_count = min(max_alternatives, len(i_candidates) * 2)
                sampled_k = np.random.choice(i_candidates, min(sample_count, len(i_candidates)), replace=False)
                
                for k in sampled_k:
                    if k == j:
                        continue
                    
                    # Get candidates from k
                    k_candidates = np.where(candidate_mask[k])[0]
                    for l in k_candidates:
                        if l == i or l == j or l == k:
                            continue
                        
                        if candidate_mask[l, j]:
                            alt_cost = distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j]
                            best_alternative = min(best_alternative, alt_cost)
            
            if best_alternative < np.inf:
                # Compute regret ratio
                ratio = best_alternative / (direct_cost + 1e-8)
                
                # Adaptive penalty function
                if ratio < 1.0:  # Alternative is better
                    penalty = np.exp(-5 * (1.0 - ratio))
                else:  # Direct edge is better
                    bonus = 1.0 + np.log1p(ratio - 1.0)
                    penalty = min(bonus, 3.0)
                
                regret[i, j] = regret[j, i] = penalty
            else:
                regret[i, j] = regret[j, i] = 1.0
    
    np.fill_diagonal(regret, 0)
    return regret


def _compute_dynamic_density(distance_matrix: np.ndarray,
                            stats: Dict) -> np.ndarray:
    """Compute density scores that adapt to instance structure."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Adaptive k based on instance characteristics
    k_base = min(7, max(3, n // 8))
    k = int(k_base * (1.0 + 0.5 * stats['neighborhood_consistency']))
    
    # Compute multi-scale density
    local_density = np.zeros(n)
    regional_density = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Local density (k-nearest)
        if len(dists) >= k:
            k_nearest = np.partition(dists, k)[:k]
            local_density[i] = np.mean(k_nearest)
        else:
            local_density[i] = np.mean(dists)
        
        # Regional density (all within p90 threshold)
        threshold = stats['p90']
        regional_dists = dists[dists <= threshold]
        if len(regional_dists) > 0:
            regional_density[i] = np.mean(regional_dists)
        else:
            regional_density[i] = local_density[i]
    
    # Normalize densities
    if np.std(local_density) > 1e-8:
        local_density = (local_density - np.mean(local_density)) / np.std(local_density)
    if np.std(regional_density) > 1e-8:
        regional_density = (regional_density - np.mean(regional_density)) / np.std(regional_density)
    
    # Combined density score
    combined_density = 0.6 * local_density + 0.4 * regional_density
    
    # Create edge scores
    for i in range(n):
        for j in range(i+1, n):
            # Favor connections between sparse nodes
            if combined_density[i] > 0.5 and combined_density[j] > 0.5:
                # Both are in sparse regions
                score = 1.8
            elif combined_density[i] < -0.5 and combined_density[j] < -0.5:
                # Both are in dense regions
                score = 0.6
            else:
                # Mixed or moderate
                score = 1.0
            
            density_scores[i, j] = density_scores[j, i] = score
    
    return density_scores


def _compute_enhanced_mst_score(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute MST scores with edge criticality analysis."""
    n = distance_matrix.shape[0]
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            # Compute MST
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Basic boost for MST edges
            mst_scores[mst_present] = 1.8
            
            # Analyze edge criticality
            degree = np.sum(mst_present, axis=1)
            
            for i in range(n):
                if degree[i] == 1:  # Leaf node
                    j = np.where(mst_present[i])[0][0]
                    mst_scores[i, j] = mst_scores[j, i] = 2.2
                elif degree[i] >= 3:  # Hub node
                    neighbors = np.where(mst_present[i])[0]
                    for j in neighbors:
                        if degree[j] == 1:  # Connecting to leaf
                            mst_scores[i, j] = mst_scores[j, i] = 2.0
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores


def _compute_adaptive_centrality(distance_matrix: np.ndarray,
                                stats: Dict) -> np.ndarray:
    """Compute centrality scores that adapt to centrality distribution."""
    n = stats['n']
    centrality_scores = np.ones((n, n))
    
    # Compute multiple centrality measures
    closeness = _compute_closeness_centralities(distance_matrix)
    
    # Normalize
    if np.max(closeness) > np.min(closeness):
        closeness = (closeness - np.min(closeness)) / (np.max(closeness) - np.min(closeness) + 1e-8)
    
    # Adaptive thresholds based on centrality distribution
    high_thresh = 0.7 if stats['centrality_cv'] > 0.3 else 0.8
    low_thresh = 0.3 if stats['centrality_cv'] > 0.3 else 0.2
    
    for i in range(n):
        for j in range(i+1, n):
            ci, cj = closeness[i], closeness[j]
            
            # Dynamic scoring based on centrality pattern
            if ci > high_thresh and cj > high_thresh:
                # Both highly central - moderate priority
                score = 1.2 + 0.3 * stats['centrality_cv']
            elif ci < low_thresh and cj < low_thresh:
                # Both peripheral - high priority to connect them
                score = 1.5
            elif (ci > high_thresh and cj < low_thresh) or (ci < low_thresh and cj > high_thresh):
                # Bridge connection - highest priority
                score = 1.8
            elif abs(ci - cj) > 0.4:
                # Significant centrality difference - medium priority
                score = 1.3
            else:
                # Similar moderate centrality
                score = 1.0
            
            centrality_scores[i, j] = centrality_scores[j, i] = score
    
    return centrality_scores


def _optimize_weights_reinforcement(distance_matrix: np.ndarray,
                                   factors: Dict,
                                   stats: Dict,
                                   candidate_mask: np.ndarray) -> Dict:
    """Optimize factor weights using reinforcement-inspired exploration."""
    n = stats['n']
    
    if n < 8:
        # Fallback to instance-based weights for small instances
        return _compute_instance_based_weights(stats)
    
    # Base weights from instance analysis
    base_weights = _compute_instance_based_weights(stats)
    
    # Reinforcement exploration
    weights = _explore_weight_space(distance_matrix, factors, base_weights, 
                                   candidate_mask, stats)
    
    return weights


def _compute_instance_based_weights(stats: Dict) -> Dict:
    """Compute base weights based on instance characteristics."""
    n = stats['n']
    cv = stats['cv_dist']
    consistency = stats['neighborhood_consistency']
    centrality_var = stats['centrality_cv']
    distance_skew = stats['distance_skew']
    
    weights = {
        'w_inv': 1.2 - 0.4 * distance_skew,  # Less important for skewed distributions
        
        # Regret weight: more important for structured instances
        'w_regret': 0.5 + 0.5 * consistency,
        
        # Density weight: adapts to neighborhood consistency
        'w_density': 0.3 + 0.4 * consistency,
        
        # MST weight: decreases with size and increases with structure
        'w_mst': max(0.2, 0.8 - 0.03 * n) * (0.5 + 0.5 * consistency),
        
        # Centrality weight: more important when centrality varies
        'w_centrality': 0.1 + 0.6 * centrality_var
    }
    
    # Normalize
    total = sum(weights.values())
    scale = 5.0 / total  # 5 factors
    
    for key in weights:
        weights[key] = weights[key] * scale
    
    return weights


def _explore_weight_space(distance_matrix: np.ndarray,
                         factors: Dict,
                         base_weights: Dict,
                         candidate_mask: np.ndarray,
                         stats: Dict) -> Dict:
    """Explore weight space using guided random search."""
    n = stats['n']
    
    if n < 10:
        return base_weights
    
    # Number of weight configurations to try
    n_configs = min(20, 5 + n // 10)
    
    best_score = -np.inf
    best_weights = base_weights.copy()
    
    # Generate candidate tours for evaluation
    eval_tours = _generate_evaluation_tours(distance_matrix, n_configs * 2)
    
    for _ in range(n_configs):
        # Perturb base weights
        current_weights = {}
        for key, value in base_weights.items():
            # Random perturbation ±30%
            perturbation = 1.0 + random.uniform(-0.3, 0.3)
            current_weights[key] = max(0.1, value * perturbation)
        
        # Normalize
        total = sum(current_weights.values())
        scale = 5.0 / total
        
        for key in current_weights:
            current_weights[key] = current_weights[key] * scale
        
        # Evaluate weight configuration
        score = _evaluate_weight_config(distance_matrix, factors, current_weights, 
                                       candidate_mask, eval_tours, stats)
        
        if score > best_score:
            best_score = score
            best_weights = current_weights.copy()
    
    return best_weights


def _generate_evaluation_tours(distance_matrix: np.ndarray,
                              n_tours: int) -> List[np.ndarray]:
    """Generate diverse tours for weight evaluation."""
    n = distance_matrix.shape[0]
    tours = []
    
    # Generate random tours
    for _ in range(n_tours // 2):
        tour = np.random.permutation(n)
        tours.append(tour)
    
    # Generate greedy tours
    for _ in range(n_tours // 2):
        tour = _generate_greedy_tour(distance_matrix)
        tours.append(tour)
    
    return tours


def _generate_greedy_tour(distance_matrix: np.ndarray) -> np.ndarray:
    """Generate a tour using nearest neighbor heuristic."""
    n = distance_matrix.shape[0]
    tour = np.zeros(n, dtype=int)
    visited = np.zeros(n, dtype=bool)
    
    # Start from random node
    current = random.randint(0, n-1)
    tour[0] = current
    visited[current] = True
    
    for i in range(1, n):
        # Find nearest unvisited neighbor
        dists = distance_matrix[current].copy()
        dists[visited] = np.inf
        
        if np.all(dists == np.inf):
            # Fallback to random unvisited
            unvisited = np.where(~visited)[0]
            next_node = random.choice(unvisited)
        else:
            next_node = np.argmin(dists)
        
        tour[i] = next_node
        visited[next_node] = True
        current = next_node
    
    return tour


def _evaluate_weight_config(distance_matrix: np.ndarray,
                           factors: Dict,
                           weights: Dict,
                           candidate_mask: np.ndarray,
                           eval_tours: List[np.ndarray],
                           stats: Dict) -> float:
    """Evaluate a weight configuration using tour analysis."""
    n = stats['n']
    
    # Compute heuristic matrix with given weights
    H_temp = _combine_factors_with_weights(factors, weights, candidate_mask)
    
    # Evaluate on sample tours
    scores = []
    
    for tour in eval_tours[:min(10, len(eval_tours))]:
        # Compute tour heuristic score
        tour_score = 0
        for k in range(n):
            i, j = tour[k], tour[(k+1) % n]
            if candidate_mask[i, j]:
                tour_score += H_temp[i, j]
        
        # Normalize by tour length
        tour_length = sum(distance_matrix[tour[k], tour[(k+1) % n]] for k in range(n))
        if tour_length > 0:
            normalized_score = tour_score / tour_length
            scores.append(normalized_score)
    
    if scores:
        return np.mean(scores)
    else:
        return 0.0


def _combine_factors_with_weights(factors: Dict,
                                 weights: Dict,
                                 candidate_mask: np.ndarray) -> np.ndarray:
    """Combine factors using weighted geometric mean."""
    n = factors['inv_dist'].shape[0]
    H = np.zeros((n, n))
    
    # Precompute factor arrays for efficiency
    inv_dist = factors['inv_dist']
    regret = factors['regret']
    density = factors['density']
    mst = factors['mst']
    centrality = factors['centrality']
    
    # Combine only for candidate edges
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                combined = (
                    (inv_dist[i, j] ** weights['w_inv']) *
                    (regret[i, j] ** weights['w_regret']) *
                    (density[i, j] ** weights['w_density']) *
                    (mst[i, j] ** weights['w_mst']) *
                    (centrality[i, j] ** weights['w_centrality'])
                )
                H[i, j] = H[j, i] = combined
    
    np.fill_diagonal(H, 0)
    return H


def _predictive_sparsification(H: np.ndarray,
                              distance_matrix: np.ndarray,
                              stats: Dict,
                              candidate_mask: np.ndarray) -> np.ndarray:
    """Apply predictive sparsification to focus on most promising edges."""
    n = stats['n']
    
    # Ensure minimum connectivity
    min_edges = min(4, n - 1)
    
    for i in range(n):
        # Count non-zero edges
        non_zero = np.where(H[i] > 0)[0]
        
        if len(non_zero) < min_edges:
            # Add nearest neighbors not already in H
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Sort by distance
            sorted_indices = np.argsort(dists)
            
            added = 0
            for j in sorted_indices:
                if added >= min_edges - len(non_zero):
                    break
                
                if H[i, j] == 0 and j != i:
                    # Assign base score
                    base_score = 1 / (dists[j] + 1e-8)
                    avg_score = np.mean(H[H > 0]) if np.any(H > 0) else 1.0
                    H[i, j] = H[j, i] = min(base_score, avg_score)
                    added += 1
    
    # Symmetrize
    H = (H + H.T) / 2
    
    # Predictive sparsification: keep edges that are promising relative to alternatives
    for i in range(n):
        row = H[i].copy()
        if np.sum(row > 0) > min_edges:
            # Keep edges that are in top percentiles
            threshold = np.percentile(row[row > 0], 30)  # Keep top 70%
            H[i, row < threshold] = 0
    
    # Final normalization with adaptive scaling
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    # Apply sigmoid to emphasize strong candidates
    H = 1 / (1 + np.exp(-8 * (H - 0.6)))
    
    np.fill_diagonal(H, 0)
    return H
