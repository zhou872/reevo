```python
import numpy as np
from scipy.spatial.distance import pdist, squareform
from scipy.sparse.csgraph import minimum_spanning_tree
import random
from typing import Dict, List, Tuple
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic with adaptive multi-scale analysis, 
    reinforcement-guided weight optimization, and predictive edge sparsification.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges.
    """
    np.random.seed(42)  # For reproducibility of stochastic components
    warnings.filterwarnings('ignore')
    
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # ==================== INSTANCE ANALYSIS ====================
    def analyze_instance(dist_mat: np.ndarray) -> Dict:
        """Multi-scale instance analysis for adaptive parameter tuning."""
        n = dist_mat.shape[0]
        # Remove diagonal for distance statistics
        dist_no_diag = dist_mat[~np.eye(n, dtype=bool)].reshape(n, n-1)
        dist_vals = dist_no_diag.flatten()
        
        mean_dist = np.mean(dist_vals)
        std_dist = np.std(dist_vals)
        cv_dist = std_dist / (mean_dist + 1e-8)
        percentiles = np.percentile(dist_vals, [10, 25, 50, 75, 90])
        
        # Neighborhood analysis at multiple scales
        if n >= 5:
            scales = [3, min(5, n//2), min(8, n//2)]
            consistency_scores = []
            for k in scales:
                if k >= n:
                    continue
                reciprocal = 0
                total = 0
                for i in range(n):
                    dists = dist_mat[i].copy()
                    dists[i] = np.inf
                    neighbors = set(np.argsort(dists)[:k])
                    for j in range(i+1, n):
                        j_dists = dist_mat[j].copy()
                        j_dists[j] = np.inf
                        j_neighbors = set(np.argsort(j_dists)[:k])
                        if j in neighbors and i in j_neighbors:
                            reciprocal += 1
                        total += 1
                consistency_scores.append(reciprocal / max(1, total))
            neighborhood_consistency = np.mean(consistency_scores) if consistency_scores else 0.5
        else:
            neighborhood_consistency = 0.5
        
        return {
            'n': n,
            'mean_dist': mean_dist,
            'std_dist': std_dist,
            'cv_dist': cv_dist,
            'percentiles': percentiles,
            'neighborhood_consistency': neighborhood_consistency,
            'p50': percentiles[2],
            'p90': percentiles[4]
        }
    
    # ==================== CANDIDATE MASK ====================
    def build_candidate_mask(dist_mat: np.ndarray, stats: Dict) -> np.ndarray:
        """Build adaptive candidate edge mask based on instance structure."""
        n = stats['n']
        base_k = min(15, max(4, int(4 * np.log2(n + 1))))
        if stats['neighborhood_consistency'] > 0.7:
            base_k = int(base_k * 1.2)
        
        candidate_mask = np.zeros((n, n), dtype=bool)
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            k = min(base_k, n-1)
            indices = np.argpartition(dists, k)[:k] if k < n-1 else np.arange(n)[dists < np.inf]
            candidate_mask[i, indices] = True
        
        candidate_mask = candidate_mask | candidate_mask.T
        np.fill_diagonal(candidate_mask, False)
        return candidate_mask
    
    # ==================== REGRET CALCULATION ====================
    def compute_regret(dist_mat: np.ndarray, stats: Dict, candidate_mask: np.ndarray) -> np.ndarray:
        """
        Multi-hop regret analysis with ratio-based penalty/reward.
        Considers 2-hop and sampled 3-hop alternative paths.
        """
        n = stats['n']
        regret = np.ones((n, n))
        
        # Pre-sort neighbors for efficiency
        sorted_neighbors = []
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            sorted_neighbors.append(np.argsort(dists))
        
        for i in range(n):
            i_candidates = np.where(candidate_mask[i])[0]
            for j in i_candidates:
                if j <= i:
                    continue
                
                direct_cost = dist_mat[i, j]
                best_alternative = np.inf
                
                # 2-hop alternatives through candidate nodes
                for k in i_candidates:
                    if k == j or not candidate_mask[k, j]:
                        continue
                    alt_cost = dist_mat[i, k] + dist_mat[k, j]
                    best_alternative = min(best_alternative, alt_cost)
                
                # Sample 3-hop alternatives for larger instances
                if n >= 10 and stats['neighborhood_consistency'] < 0.8:
                    samples = min(20, max(5, n//4))
                    for _ in range(samples):
                        k = np.random.choice([x for x in range(n) if x not in [i, j]])
                        l_candidates = sorted_neighbors[k][:3]
                        for l in l_candidates:
                            if l in [i, j, k] or not candidate_mask[l, j]:
                                continue
                            alt_cost = dist_mat[i, k] + dist_mat[k, l] + dist_mat[l, j]
                            best_alternative = min(best_alternative, alt_cost)
                
                if best_alternative < np.inf:
                    # Ratio-based penalty/reward
                    ratio = best_alternative / (direct_cost + 1e-8)
                    if ratio < 1.0:  # Alternative is better
                        penalty = np.exp(-5 * (1.0 - ratio))
                    else:  # Direct edge is better
                        bonus = 1.0 + np.log1p(ratio - 1.0)
                        penalty = min(bonus, 3.0)
                    regret[i, j] = regret[j, i] = penalty
        
        np.fill_diagonal(regret, 0)
        return regret
    
    # ==================== FACTOR COMPUTATION ====================
    def compute_factors(dist_mat: np.ndarray, stats: Dict, candidate_mask: np.ndarray) -> Dict:
        """Compute all heuristic factors."""
        n = stats['n']
        factors = {}
        
        # 1. Inverse distance (normalized)
        inv_dist = 1 / (dist_mat + np.eye(n) * 1e-8)
        np.fill_diagonal(inv_dist, 0)
        if np.max(inv_dist) > 0:
            inv_dist = inv_dist / np.max(inv_dist)
        factors['inv_dist'] = inv_dist
        
        # 2. Regret
        factors['regret'] = compute_regret(dist_mat, stats, candidate_mask)
        
        # 3. MST-based scoring
        mst_scores = np.ones((n, n))
        if n > 2:
            try:
                mst = minimum_spanning_tree(dist_mat).toarray()
                mst_present = (mst + mst.T) > 0
                mst_scores[mst_present] = 1.8
                
                # Boost critical edges (leaves and hubs)
                degree = np.sum(mst_present, axis=1)
                for i in range(n):
                    if degree[i] == 1:  # Leaf
                        j = np.where(mst_present[i])[0][0]
                        mst_scores[i, j] = mst_scores[j, i] = 2.2
            except:
                pass
        np.fill_diagonal(mst_scores, 0)
        factors['mst'] = mst_scores
        
        # 4. Centrality-based scoring
        centrality_scores = np.ones((n, n))
        if n >= 3:
            closeness = np.zeros(n)
            for i in range(n):
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                closeness[i] = 1 / (np.mean(dists) + 1e-8)
            
            if np.max(closeness) > np.min(closeness):
                closeness = (closeness - np.min(closeness)) / (np.max(closeness) - np.min(closeness))
            
            for i in range(n):
                for j in range(i+1, n):
                    ci, cj = closeness[i], closeness[j]
                    if ci < 0.3 and cj < 0.3:
                        score = 1.5  # Connect peripheral nodes
                    elif (ci > 0.7 and cj < 0.3) or (ci < 0.3 and cj > 0.7):
                        score = 1.8  # Bridge connections
                    else:
                        score = 1.0 + (ci + cj) / 4
                    centrality_scores[i, j] = centrality_scores[j, i] = score
        factors['centrality'] = centrality_scores
        
        return factors
    
    # ==================== REINFORCEMENT-GUIDED WEIGHT LEARNING ====================
    def optimize_weights(dist_mat: np.ndarray, factors: Dict, 
                        candidate_mask: np.ndarray, stats: Dict) -> Dict:
        """
        Reinforcement-inspired weight optimization using guided search on sample tours.
        Learns weights that correlate with tour quality.
        """
        n = stats['n']
        
        # Base weights from instance analysis
        base_weights = {
            'w_inv': 1.2 - 0.4 * min(1.0, stats['cv_dist']),
            'w_regret': 0.5 + 0.5 * stats['neighborhood_consistency'],
            'w_mst': max(0.2, 0.8 - 0.02 * n),
            'w_centrality': 0.3 + 0.5 * stats['neighborhood_consistency']
        }
        
        # Normalize to sum to 1
        total = sum(base_weights.values())
        for key in base_weights:
            base_weights[key] /= total
        
        if n < 8:
            return base_weights
        
        # Generate evaluation tours
        def generate_tours(n_tours: int) -> List[np.ndarray]:
            tours = []
            for _ in range(n_tours):
                if np.random.random() < 0.7:
                    # Random tour
                    tour = np.random.permutation(n)
                else:
                    # Greedy tour
                    tour = np.zeros(n, dtype=int)
                    visited = np.zeros(n, dtype=bool)
                    current = np.random.randint(0, n)
                    tour[0] = current
                    visited[current] = True
                    for idx in range(1, n):
                        dists = dist_mat[current].copy()
                        dists[visited] = np.inf
                        if np.all(dists == np.inf):
                            unvisited = np.where(~visited)[0]
                            current = np.random.choice(unvisited)
                        else:
                            current = np.argmin(dists)
                        tour[idx] = current
                        visited[current] = True
                tours.append(tour)
            return tours
        
        # Evaluate weight configuration
        def evaluate_weights(weights: Dict, tours: List[np.ndarray]) -> float:
            # Combine factors with current weights
            n = factors['inv_dist'].shape[0]
            H_temp = np.zeros((n, n))
            for i in range(n):
                for j in range(i+1, n):
                    if candidate_mask[i, j]:
                        combined = (
                            factors['inv_dist'][i, j] ** weights['w_inv'] *
                            factors['regret'][i, j] ** weights['w_regret'] *
                            factors['mst'][i, j] ** weights['w_mst'] *
                            factors['centrality'][i, j] ** weights['w_centrality']
                        )
                        H_temp[i, j] = H_temp[j, i] = combined
            
            # Score tours
            scores = []
            for tour in tours[:min(10, len(tours))]:
                tour_score = 0
                tour_length = 0
                for k in range(n):
                    i, j = tour[k], tour[(k+1) % n]
                    if H_temp[i, j] > 0:
                        tour_score += H_temp[i, j]
                    tour_length += dist_mat[i, j]
                if tour_length > 0:
                    scores.append(tour_score / tour_length)
            return np.mean(scores) if scores else 0.0
        
        # Guided search
        tours = generate_tours(min(20, n * 2))
        best_score = -np.inf
        best_weights = base_weights.copy()
        
        for iteration in range(min(15, n)):
            # Perturb weights
            current_weights = {}
            for key, value in base_weights.items():
                perturbation = 1.0 + np.random.uniform(-0.3, 0.3)
                current_weights[key] = max(0.05, value * perturbation)
            
            # Renormalize
            total = sum(current_weights.values())
            for key in current_weights:
                current_weights[key] /= total
            
            score = evaluate_weights(current_weights, tours)
            if score > best_score:
                best_score = score
                best_weights = current_weights.copy()
        
        return best_weights
    
    # ==================== FACTOR COMBINATION ====================
    def combine_factors(factors: Dict, weights: Dict, candidate_mask: np.ndarray) -> np.ndarray:
        """Weighted geometric mean combination of factors."""
        n = factors['inv_dist'].shape[0]
        H = np.zeros((n, n))
        
        for i in range(n):
            for j in range(i+1, n):
                if candidate_mask[i, j]:
                    combined = (
                        factors['inv_dist'][i, j] ** weights['w_inv'] *
                        factors['regret'][i, j] ** weights['w_regret'] *
                        factors['mst'][i, j] ** weights['w_mst'] *
                        factors['centrality'][i, j] ** weights['w_centrality']
                    )
                    H[i, j] = H[j, i] = combined
        
        np.fill_diagonal(H, 0)
        return H
    
    # ==================== PREDICTIVE SPARSIFICATION ====================
    def sparsify_heuristics(H: np.ndarray, dist_mat: np.ndarray, 
                           candidate_mask: np.ndarray, stats: Dict) -> np.ndarray:
        """
        Predictive sparsification: focus on most promising edges.
        Uses adaptive thresholding and sigmoid emphasis.
        """
        n = stats['n']
        
        # Ensure minimum connectivity
        min_edges = min(4, n - 1)
        for i in range(n):
            non_zero = np.where(H[i] > 0)[0]
            if len(non_zero) < min_edges:
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                sorted_idx = np.argsort(dists)
                added = 0
                for j in sorted_idx:
                    if added >= min_edges - len(non_zero):
                        break
                    if H[i, j] == 0 and j != i:
                        H[i, j] = H[j, i] = 1 / (dists[j] + 1e-8)
                        added += 1
        
        # Adaptive thresholding
        for i in range(n):
            row = H[i].copy()
            non_zero = row[row > 0]
            if len(non_zero) > min_edges:
                threshold = np.percentile(non_zero, 30)
                H[i, row < threshold] = 0
        
        # Symmetrize
        H = (H + H.T) / 2
        
        # Sigmoid emphasis
        max_val = np.max(H)
        if max_val > 0:
            H = H / max_val
            H = 1 / (1 + np.exp(-8 * (H - 0.6)))
        
        np.fill_diagonal(H, 0)
        return H
    
    # ==================== MAIN EXECUTION ====================
    
    # Step 1: Instance analysis
    stats = analyze_instance(distance_matrix)
    
    # Step 2: Candidate mask
    candidate_mask = build_candidate_mask(distance_matrix, stats)
    
    # Step 3: Compute factors
    factors = compute_factors(distance_matrix, stats, candidate_mask)
    
    # Step 4: Optimize weights via reinforcement-guided learning
    weights = optimize_weights(distance_matrix, factors, candidate_mask, stats)
    
    # Step 5: Combine factors
    H = combine_factors(factors, weights, candidate_mask)
    
    # Step 6: Predictive sparsification
    H = sparsify_heuristics(H, distance_matrix, candidate_mask, stats)
    
    return H

        symmetric_diff = np.max(np.abs(H - H.T))

                for idx in top_indices:
                    i, j = triu_idx[0][idx], triu_idx[1][idx]
                    print(f"  ({i:2d}, {j:2d}): dist={dist_matrix[i,j]:6.2f}, score={H[i,j]:.4f}")
```
