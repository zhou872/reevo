import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Distance reciprocal (closer nodes are more promising)
    2. Sparsification based on nearest neighbors
    3. Minimum Spanning Tree edge bonuses
    4. Degree constraint considerations
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of distances between nodes
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, with higher values
        indicating more promising edges
    """
    n = distance_matrix.shape[0]
    
    # 1. Base reciprocal heuristic (handle division by zero for diagonal)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        np.fill_diagonal(reciprocal, 0)
    
    # 2. Nearest neighbor sparsification
    # Keep only edges to k nearest neighbors for each node
    k = max(2, n // 4)  # Adaptive k based on problem size
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k+1 smallest distances (including self)
        nearest = np.argsort(distance_matrix[i])[:k+1]
        # Remove self from nearest neighbors
        nearest = nearest[nearest != i]
        nn_mask[i, nearest] = True
    
    # Make symmetric (if i is in j's nearest neighbors or vice versa)
    nn_mask = nn_mask | nn_mask.T
    
    # 3. Minimum Spanning Tree edge bonus
    def prim_mst(dist_mat: np.ndarray) -> np.ndarray:
        """Prim's algorithm for MST with adjacency matrix"""
        n_nodes = dist_mat.shape[0]
        mst_mask = np.zeros_like(dist_mat, dtype=bool)
        visited = np.zeros(n_nodes, dtype=bool)
        visited[0] = True
        
        for _ in range(n_nodes - 1):
            # Find minimum edge from visited to unvisited
            min_val = np.inf
            min_edge = (0, 0)
            
            for i in np.where(visited)[0]:
                for j in np.where(~visited)[0]:
                    if dist_mat[i, j] < min_val:
                        min_val = dist_mat[i, j]
                        min_edge = (i, j)
            
            u, v = min_edge
            mst_mask[u, v] = True
            mst_mask[v, u] = True
            visited[v] = True
        
        return mst_mask
    
    mst_mask = prim_mst(distance_matrix)
    
    # 4. Degree-based penalty (prevent nodes from having too many connections)
    # Compute approximate degree importance
    degree_penalty = np.zeros_like(distance_matrix)
    for i in range(n):
        # Penalize edges from nodes that already have many promising connections
        row_heuristic = reciprocal[i] * nn_mask[i].astype(float)
        degree_factor = 1.0 / (1.0 + np.sum(row_heuristic > 0) / k)
        degree_penalty[i] = degree_factor
    
    # 5. Combine all factors
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    # Base: reciprocal values for nearest neighbor edges
    base_values = reciprocal * nn_mask.astype(float)
    
    # Apply MST bonus (double value for MST edges)
    mst_bonus = 2.0 if n > 10 else 3.0  # Higher bonus for larger problems
    base_values[mst_mask] *= mst_bonus
    
    # Apply degree penalty
    heuristic_matrix = base_values * (degree_penalty + degree_penalty.T) / 2
    
    # 6. Additional sparsification: zero out the least promising edges
    # Keep only top m edges per node
    m = max(2, n // 3)
    final_mask = np.zeros_like(heuristic_matrix, dtype=bool)
    
    for i in range(n):
        row = heuristic_matrix[i]
        # Get indices of top m values (excluding diagonal)
        top_indices = np.argsort(row)[-m:]
        top_indices = top_indices[top_indices != i]
        if len(top_indices) > m:
            top_indices = top_indices[-m:]
        final_mask[i, top_indices] = True
    
    # Make symmetric
    final_mask = final_mask | final_mask.T
    
    # Apply final mask
    heuristic_matrix = heuristic_matrix * final_mask.astype(float)
    
    # Normalize to [0, 1] range for stability
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
    
    return heuristic_matrix
