import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with dual-regret decomposition and
    adaptive two-step lookahead penalty. Returns a matrix where higher values
    indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Analyze instance characteristics
    stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Compute base factors
    factors = _compute_base_factors_v2(distance_matrix, stats)
    
    # Step 3: Compute dual-regret components
    regret_scores = _compute_dual_regret_scores(distance_matrix, stats)
    
    # Step 4: Compute two-step lookahead penalty
    lookahead_penalty = _compute_two_step_penalty(distance_matrix, stats)
    
    # Step 5: Dynamic weight computation
    weights = _compute_adaptive_weights_v2(distance_matrix, stats)
    
    # Step 6: Build candidate mask for sparsification
    candidate_mask = _build_candidate_mask_v2(distance_matrix, stats)
    
    # Step 7: Combine factors with geometric mean
    H = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                combined = (
                    (factors['inv_dist'][i, j] ** weights['w_inv']) *
                    (regret_scores['one_hop'][i, j] ** weights['w_regret1']) *
                    (regret_scores['two_hop'][i, j] ** weights['w_regret2']) *
                    (lookahead_penalty[i, j] ** weights['w_lookahead']) *
                    (factors['mst'][i, j] ** weights['w_mst']) *
                    (factors['centrality'][i, j] ** weights['w_central'])
                )
                H[i, j] = H[j, i] = combined
    
    # Step 8: Post-processing
    H = _post_process_heuristics_v2(H, distance_matrix, stats)
    
    return H


def _analyze_instance_v2(distance_matrix: np.ndarray) -> dict:
    """Analyze instance characteristics with additional metrics."""
    n = distance_matrix.shape[0]
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    # Basic statistics
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Estimate regularity via coefficient of variation of nearest neighbor distances
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    regularity = 1.0 / (1.0 + np.std(nn_dists) / (np.mean(nn_dists) + 1e-8))
    
    # Clustering coefficient approximation
    clustering = _estimate_clustering_coefficient(distance_matrix)
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'cv_dist': cv_dist,
        'regularity': regularity,
        'clustering': clustering
    }


def _estimate_clustering_coefficient(distance_matrix: np.ndarray) -> float:
    """Estimate graph clustering coefficient from distance matrix."""
    n = distance_matrix.shape[0]
    if n < 3:
        return 0.5
    
    # Convert to proximity matrix (closer = higher value)
    max_dist = np.max(distance_matrix)
    proximity = 1.0 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(proximity, 0)
    
    # Normalize
    proximity = proximity / np.max(proximity)
    
    # Sample triangles
    n_samples = min(100, n * (n-1) * (n-2) // 6)
    total_clustering = 0.0
    
    for _ in range(n_samples):
        i, j, k = np.random.choice(n, 3, replace=False)
        # Weight by edge strengths
        wij = proximity[i, j]
        wik = proximity[i, k]
        wjk = proximity[j, k]
        
        if wij > 0 and wik > 0 and wjk > 0:
            triangle_strength = (wij * wik * wjk) ** (1/3)
            total_clustering += triangle_strength
    
    return total_clustering / (n_samples + 1e-8)


def _compute_base_factors_v2(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute base heuristic factors including centrality."""
    n = stats['n']
    
    factors = {}
    
    # 1. Inverse distance (normalized)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    factors['inv_dist'] = inv_dist / np.max(inv_dist)
    
    # 2. MST-based scoring
    mst_scores = np.ones((n, n))
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            mst_scores[mst_present] = 2.0
        except:
            pass
    np.fill_diagonal(mst_scores, 0)
    factors['mst'] = mst_scores
    
    # 3. Centrality-based scoring
    centrality = np.zeros((n, n))
    # Degree centrality approximation using nearest neighbors
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Number of nodes within 1.5 * mean nearest neighbor distance
        threshold = 1.5 * np.mean(np.partition(dists, min(3, n-1))[:min(3, n-1)])
        centrality_i = np.sum(dists <= threshold) / (n - 1)
        for j in range(n):
            if i != j:
                # Average centrality of endpoints
                dists_j = distance_matrix[j].copy()
                dists_j[j] = np.inf
                threshold_j = 1.5 * np.mean(np.partition(dists_j, min(3, n-1))[:min(3, n-1)])
                centrality_j = np.sum(dists_j <= threshold_j) / (n - 1)
                centrality[i, j] = (centrality_i + centrality_j) / 2
    
    factors['centrality'] = centrality
    
    return factors


def _compute_dual_regret_scores(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute one-hop and two-hop regret scores separately."""
    n = stats['n']
    one_hop = np.ones((n, n))
    two_hop = np.ones((n, n))
    
    for i in range(n):
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        
        # One-hop regret
        sorted_idx = np.argsort(dists_i)
        if len(sorted_idx) >= 2:
            best_dist = dists_i[sorted_idx[0]]
            for k, j in enumerate(sorted_idx):
                if j != i:
                    edge_regret = max(0, dists_i[j] - best_dist) / (best_dist + 1e-8)
                    one_hop[i, j] = 1.0 / (1.0 + edge_regret)
        
        # Two-hop regret
        for j in range(n):
            if i != j:
                # Find best two-hop alternative through k
                best_two_hop = np.inf
                for k in range(n):
                    if k != i and k != j:
                        two_hop_dist = distance_matrix[i, k] + distance_matrix[k, j]
                        if two_hop_dist < best_two_hop:
                            best_two_hop = two_hop_dist
                
                if best_two_hop < np.inf:
                    direct_dist = distance_matrix[i, j]
                    two_hop_regret = max(0, direct_dist - best_two_hop) / (best_two_hop + 1e-8)
                    two_hop[i, j] = 1.0 / (1.0 + two_hop_regret)
    
    # Symmetrize
    one_hop = (one_hop + one_hop.T) / 2
    two_hop = (two_hop + two_hop.T) / 2
    np.fill_diagonal(one_hop, 0)
    np.fill_diagonal(two_hop, 0)
    
    return {'one_hop': one_hop, 'two_hop': two_hop}


def _compute_two_step_penalty(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute penalty for edges that can be bypassed by significantly shorter two-hop paths."""
    n = stats['n']
    penalty = np.ones((n, n))
    
    if n < 3:
        return penalty
    
    # Adaptive threshold based on instance size and regularity
    threshold_factor = 0.8 + 0.1 * stats['regularity']  # More regular = stricter threshold
    
    for i in range(n):
        for j in range(i+1, n):
            direct = distance_matrix[i, j]
            best_bypass = np.inf
            
            # Find best bypass through k
            for k in range(n):
                if k != i and k != j:
                    bypass = distance_matrix[i, k] + distance_matrix[k, j]
                    if bypass < best_bypass:
                        best_bypass = bypass
            
            if best_bypass < threshold_factor * direct:
                # Apply penalty proportional to the bypass advantage
                penalty_factor = best_bypass / (direct + 1e-8)
                # Scale penalty based on instance characteristics
                scaled_penalty = 0.3 + 0.7 * penalty_factor
                penalty[i, j] = penalty[j, i] = scaled_penalty
    
    np.fill_diagonal(penalty, 0)
    return penalty


def _compute_adaptive_weights_v2(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Dynamically compute weights based on instance characteristics."""
    n = stats['n']
    cv = stats['cv_dist']
    regularity = stats['regularity']
    clustering = stats['clustering']
    
    # Adjust weights based on instance properties
    weights = {
        'w_inv': 1.0,  # Base importance
        
        # One-hop regret weight: higher for irregular instances
        'w_regret1': 0.5 + 0.3 * (1 - regularity),
        
        # Two-hop regret weight: higher for regular instances
        'w_regret2': 0.3 + 0.4 * regularity,
        
        # Lookahead penalty weight: higher for clustered instances
        'w_lookahead': 0.2 + 0.3 * clustering,
        
        # MST weight: decreases with instance size
        'w_mst': max(0.1, 0.4 - 0.005 * n),
        
        # Centrality weight: higher for clustered instances
        'w_central': 0.1 + 0.2 * clustering
    }
    
    # Normalize to maintain overall scale (sum to ~5)
    total = sum(weights.values())
    scale = 5.0 / total
    for key in weights:
        weights[key] *= scale
    
    return weights


def _build_candidate_mask_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Build mask of candidate edges with adaptive sparsification."""
    n = stats['n']
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # Base candidate list size
    base_k = min(15, max(5, int(np.log2(n) * 4)))
    
    # Adjust based on instance characteristics
    # More clustered instances -> fewer candidates
    k = int(base_k * (1.0 - 0.3 * stats['clustering']))
    k = max(3, min(k, n - 1))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < len(dists):
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Make symmetric
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask


def _post_process_heuristics_v2(H: np.ndarray, distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Post-process the heuristic matrix with normalization and connectivity assurance."""
    n = stats['n']
    
    # Ensure minimum connectivity
    min_edges = min(4, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i])[0]
        if len(non_zero) < min_edges:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    base_score = 1 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    # Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Final normalization to [0, 1] range
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
