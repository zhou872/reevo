import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Dict, List, Tuple
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with multi-hop regret analysis and iterative
    refinement through local search feedback.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges.
    """
    warnings.filterwarnings('ignore')
    
    n = distance_matrix.shape[0]
    if n <= 2:
        H = 1 / (distance_matrix + np.eye(n) * 1e-8)
        np.fill_diagonal(H, 0)
        return H
    
    # ==================== INSTANCE ANALYSIS ====================
    def analyze_instance(dist_mat: np.ndarray) -> Dict:
        """Multi-scale instance analysis with embedding approximation."""
        n = dist_mat.shape[0]
        dist_no_diag = dist_mat[~np.eye(n, dtype=bool)].reshape(n, n-1)
        dist_vals = dist_no_diag.flatten()
        
        mean_dist = np.mean(dist_vals)
        std_dist = np.std(dist_vals)
        cv_dist = std_dist / (mean_dist + 1e-8)
        
        # Multi-scale neighborhood consistency
        consistency_scores = []
        if n >= 5:
            for k in [3, min(5, n//2), min(8, n//2)]:
                if k >= n:
                    continue
                reciprocal = 0
                total = 0
                for i in range(n):
                    dists = dist_mat[i].copy()
                    dists[i] = np.inf
                    neighbors = set(np.argsort(dists)[:k])
                    for j in range(i+1, n):
                        j_dists = dist_mat[j].copy()
                        j_dists[j] = np.inf
                        j_neighbors = set(np.argsort(j_dists)[:k])
                        if j in neighbors and i in j_neighbors:
                            reciprocal += 1
                        total += 1
                if total > 0:
                    consistency_scores.append(reciprocal / total)
        
        neighborhood_consistency = np.mean(consistency_scores) if consistency_scores else 0.5
        
        # Approximate graph embedding (2D MDS approximation)
        embedding_quality = 0.0
        if n >= 4:
            try:
                # Simple 2D approximation using pivot points
                pivots = min(5, n)
                pivot_indices = np.random.choice(n, pivots, replace=False)
                D_pivot = dist_mat[pivot_indices][:, pivot_indices]
                
                # Compute approximate coordinates
                from sklearn.manifold import MDS
                mds = MDS(n_components=2, dissimilarity='precomputed', random_state=42)
                coords = mds.fit_transform(D_pivot)
                
                # Check reconstruction quality
                recon_dist = np.sqrt(((coords[:, None] - coords[None, :]) ** 2).sum(axis=2))
                embedding_quality = 1.0 / (1.0 + np.mean(np.abs(recon_dist - D_pivot) / (D_pivot + 1e-8)))
            except:
                pass
        
        return {
            'n': n,
            'mean_dist': mean_dist,
            'std_dist': std_dist,
            'cv_dist': cv_dist,
            'neighborhood_consistency': neighborhood_consistency,
            'embedding_quality': embedding_quality
        }
    
    # ==================== ADVANCED REGRET CALCULATION ====================
    def compute_advanced_regret(dist_mat: np.ndarray, stats: Dict, 
                               candidate_mask: np.ndarray) -> np.ndarray:
        """Multi-hop regret analysis with adaptive sampling and path scoring."""
        n = stats['n']
        regret = np.ones((n, n))
        
        # Precompute k-nearest neighbors for each node
        knn = []
        k_val = min(15, max(5, n//3))
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            knn.append(np.argsort(dists)[:k_val])
        
        for i in range(n):
            i_candidates = np.where(candidate_mask[i])[0]
            for j in i_candidates:
                if j <= i:
                    continue
                
                direct_cost = dist_mat[i, j]
                best_alternative = np.inf
                samples_collected = []
                
                # 2-hop alternatives through candidate nodes
                for k in i_candidates:
                    if k == j or not candidate_mask[k, j]:
                        continue
                    alt_cost = dist_mat[i, k] + dist_mat[k, j]
                    best_alternative = min(best_alternative, alt_cost)
                    samples_collected.append(alt_cost)
                
                # 3-hop alternatives with adaptive sampling
                if n >= 10:
                    samples = min(25, max(8, n//2))
                    for _ in range(samples):
                        # Sample intermediate nodes biased by proximity
                        k = np.random.choice([x for x in range(n) if x not in [i, j]])
                        l_options = knn[k][:min(5, len(knn[k]))]
                        for l in l_options:
                            if l in [i, j, k] or not candidate_mask[l, j]:
                                continue
                            alt_cost = dist_mat[i, k] + dist_mat[k, l] + dist_mat[l, j]
                            best_alternative = min(best_alternative, alt_cost)
                            samples_collected.append(alt_cost)
                
                if len(samples_collected) > 0:
                    # Enhanced regret scoring with distribution awareness
                    ratio = best_alternative / (direct_cost + 1e-8)
                    
                    if ratio < 0.95:  # Much better alternative exists
                        penalty = np.exp(-8 * (1.0 - ratio))
                    elif ratio < 1.05:  # Comparable alternatives
                        penalty = 1.0
                    else:  # Direct edge is best
                        # Consider quality of alternatives
                        alt_std = np.std(samples_collected) if len(samples_collected) > 1 else 0
                        stability_factor = 1.0 / (1.0 + alt_std / (direct_cost + 1e-8))
                        bonus = 1.0 + np.log1p(ratio - 1.0) * stability_factor
                        penalty = min(bonus, 3.5)
                    
                    regret[i, j] = regret[j, i] = penalty
        
        np.fill_diagonal(regret, 0)
        return regret
    
    # ==================== ITERATIVE REFINEMENT WITH LOCAL SEARCH ====================
    def iterative_refinement(initial_H: np.ndarray, dist_mat: np.ndarray,
                            candidate_mask: np.ndarray, stats: Dict) -> np.ndarray:
        """Refine heuristics through local search feedback loops."""
        n = stats['n']
        H = initial_H.copy()
        
        if n < 8:
            return H
        
        # Generate initial tours using heuristic-guided construction
        def generate_tours_heuristic(H_matrix: np.ndarray, n_tours: int) -> List[Tuple[List[int], float]]:
            tours = []
            for _ in range(n_tours):
                # Start from random node
                current = np.random.randint(0, n)
                tour = [current]
                visited = set([current])
                
                while len(visited) < n:
                    # Get unvisited candidates with heuristic scores
                    candidates = [c for c in range(n) if c not in visited]
                    if not candidates:
                        break
                    
                    # Probability proportional to heuristic score
                    scores = H_matrix[current, candidates]
                    scores = np.maximum(scores, 1e-8)
                    probs = scores / scores.sum()
                    
                    # Stochastic selection
                    next_city = np.random.choice(candidates, p=probs)
                    tour.append(next_city)
                    visited.add(next_city)
                    current = next_city
                
                # Calculate tour length
                length = 0
                for k in range(n):
                    i, j = tour[k], tour[(k+1) % n]
                    length += dist_mat[i, j]
                
                tours.append((tour, length))
            return tours
        
        # Perform 2-3 iterations of refinement
        for iteration in range(3):
            # Generate diverse tours
            n_tours = min(30, max(10, n * 2))
            tours = generate_tours_heuristic(H, n_tours)
            
            if len(tours) < 3:
                continue
            
            # Sort by quality
            tours.sort(key=lambda x: x[1])
            best_tours = tours[:min(5, len(tours)//3)]
            worst_tours = tours[-min(3, len(tours)//4):]
            
            # Collect edge statistics
            edge_counts = np.zeros((n, n))
            for tour, length in best_tours:
                for k in range(n):
                    i, j = tour[k], tour[(k+1) % n]
                    edge_counts[i, j] += 1
                    edge_counts[j, i] += 1
            
            # Update heuristic matrix
            learning_rate = 0.3 * (0.8 ** iteration)  # Decreasing learning rate
            
            for i in range(n):
                for j in range(i+1, n):
                    if candidate_mask[i, j]:
                        # Reinforcement from good tours
                        if edge_counts[i, j] > 0:
                            boost = 1.0 + learning_rate * edge_counts[i, j]
                            H[i, j] = H[j, i] = H[i, j] * boost
                        
                        # Mild penalty for edges in bad tours
                        for tour, _ in worst_tours:
                            for k in range(n):
                                ti, tj = tour[k], tour[(k+1) % n]
                                if (ti == i and tj == j) or (ti == j and tj == i):
                                    H[i, j] = H[j, i] = H[i, j] * (1.0 - learning_rate * 0.1)
                                    break
            
            # Renormalize
            max_val = np.max(H)
            if max_val > 0:
                H = H / max_val
        
        return H
    
    # ==================== MAIN PIPELINE ====================
    
    # Step 1: Instance analysis
    stats = analyze_instance(distance_matrix)
    
    # Step 2: Candidate mask (adaptive)
    n = stats['n']
    base_k = min(18, max(5, int(5 * np.log2(n + 1))))
    if stats['neighborhood_consistency'] > 0.7:
        base_k = int(base_k * 1.2)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k = min(base_k, n-1)
        indices = np.argpartition(dists, k)[:k] if k < n-1 else np.arange(n)[dists < np.inf]
        candidate_mask[i, indices] = True
    
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    # Step 3: Compute factors
    factors = {}
    
    # 1. Inverse distance
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    if np.max(inv_dist) > 0:
        inv_dist = inv_dist / np.max(inv_dist)
    factors['inv_dist'] = inv_dist
    
    # 2. Advanced regret
    factors['regret'] = compute_advanced_regret(distance_matrix, stats, candidate_mask)
    
    # 3. MST-based scoring
    mst_scores = np.ones((n, n))
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            mst_scores[mst_present] = 1.8
            
            # Identify critical edges
            degree = np.sum(mst_present, axis=1)
            for i in range(n):
                if degree[i] == 1:  # Leaf node
                    j = np.where(mst_present[i])[0][0]
                    mst_scores[i, j] = mst_scores[j, i] = 2.2
                elif degree[i] >= 3:  # Hub node
                    neighbors = np.where(mst_present[i])[0]
                    for j in neighbors:
                        if degree[j] == 1:  # Connect hub to leaf
                            mst_scores[i, j] = mst_scores[j, i] = 2.0
        except:
            pass
    np.fill_diagonal(mst_scores, 0)
    factors['mst'] = mst_scores
    
    # 4. Centrality scoring
    centrality_scores = np.ones((n, n))
    if n >= 3:
        # Simple closeness centrality approximation
        closeness = np.zeros(n)
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            closeness[i] = 1 / (np.mean(dists) + 1e-8)
        
        if np.max(closeness) > np.min(closeness):
            closeness = (closeness - np.min(closeness)) / (np.max(closeness) - np.min(closeness) + 1e-8)
        
        for i in range(n):
            for j in range(i+1, n):
                ci, cj = closeness[i], closeness[j]
                # Strategic edge scoring
                if ci < 0.2 and cj < 0.2:
                    score = 1.6  # Connect peripheral nodes
                elif (ci > 0.8 and cj < 0.2) or (ci < 0.2 and cj > 0.8):
                    score = 1.9  # Bridge connections
                elif ci > 0.7 and cj > 0.7:
                    score = 1.3  # Connect central nodes moderately
                else:
                    score = 1.0 + (ci + cj) / 3
                centrality_scores[i, j] = centrality_scores[j, i] = score
    factors['centrality'] = centrality_scores
    
    # Step 4: Weight optimization (simplified adaptive)
    weights = {
        'w_inv': 1.0,
        'w_regret': 0.8 + 0.4 * stats['neighborhood_consistency'] - 0.2 * min(1.0, stats['cv_dist']),
        'w_mst': max(0.3, 0.9 - 0.015 * n),
        'w_centrality': 0.4 + 0.4 * stats['embedding_quality']
    }
    
    # Normalize weights
    total = sum(weights.values())
    for key in weights:
        weights[key] /= total
    
    # Step 5: Geometric combination
    H = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                combined = (
                    factors['inv_dist'][i, j] ** weights['w_inv'] *
                    factors['regret'][i, j] ** weights['w_regret'] *
                    factors['mst'][i, j] ** weights['w_mst'] *
                    factors['centrality'][i, j] ** weights['w_centrality']
                )
                H[i, j] = H[j, i] = combined
    
    np.fill_diagonal(H, 0)
    
    # Step 6: Iterative refinement with local search feedback
    H = iterative_refinement(H, distance_matrix, candidate_mask, stats)
    
    # Step 7: Final processing
    # Ensure minimum connectivity
    min_edges = min(4, n - 1)
    for i in range(n):
        non_zero = np.where(H[i] > 0)[0]
        if len(non_zero) < min_edges:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            sorted_idx = np.argsort(dists)
            added = 0
            for j in sorted_idx:
                if added >= min_edges - len(non_zero):
                    break
                if H[i, j] == 0 and j != i:
                    H[i, j] = H[j, i] = 1 / (dists[j] + 1e-8)
                    added += 1
    
    # Symmetrize and normalize
    H = (H + H.T) / 2
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    np.fill_diagonal(H, 0)
    return H
