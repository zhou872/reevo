def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with spectral embeddings and 
    graph centrality measures. Returns a matrix of the same shape where 
    higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Comprehensive instance analysis
    instance_stats = _analyze_instance_enhanced(distance_matrix)
    
    # Step 2: Compute multi-faceted base scores
    base_scores = _compute_multi_facet_scores(distance_matrix, instance_stats)
    
    # Step 3: Advanced lookahead with reinforcement learning principles
    lookahead_scores = _compute_rl_lookahead_scores(distance_matrix, instance_stats)

    )

def _analyze_instance_enhanced(distance_matrix: np.ndarray) -> Dict:
    """Comprehensive instance analysis with spectral properties."""
    n = distance_matrix.shape[0]
    
    # Basic statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Spatial distribution analysis
    spatial_stats = _analyze_spatial_distribution(distance_matrix)
    
    # Graph theoretic properties
    graph_stats = _analyze_graph_properties(distance_matrix)
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'spatial_entropy': spatial_stats['entropy'],
        'clustering_tendency': spatial_stats['clustering'],
        'graph_diameter': graph_stats['diameter'],
        'avg_path_length': graph_stats['avg_path_length'],
        'connectivity': graph_stats['connectivity']
    }

def _analyze_spatial_distribution(distance_matrix: np.ndarray) -> Dict:
    """Analyze spatial distribution of points."""
    n = distance_matrix.shape[0]
    
    # Hopkins statistic for clustering tendency
    m = min(15, n // 4)
    if m > 2:
        sample_indices = np.random.choice(n, m, replace=False)
        sample_nn_dists = []
        random_nn_dists = []
        
        for idx in sample_indices:
            dists = distance_matrix[idx].copy()
            dists[idx] = np.inf
            sample_nn_dists.append(np.min(dists))
            
            # Generate random distances from distribution
            random_dists = np.random.choice(distance_matrix.flatten(), m)
            random_nn_dists.append(np.min(random_dists))
        
        hopkins = np.sum(sample_nn_dists) / (np.sum(sample_nn_dists) + np.sum(random_nn_dists) + 1e-8)
    else:
        hopkins = 0.5
    
    # Spatial entropy using distance quartiles
    if n > 4:
        flat_dists = distance_matrix[~np.eye(n, dtype=bool)]
        quartiles = np.percentile(flat_dists, [25, 50, 75])
        bins = np.digitize(flat_dists, quartiles)
        hist = np.bincount(bins, minlength=4)
        hist = hist / np.sum(hist)
        entropy = -np.sum(hist * np.log(hist + 1e-8))
        entropy = entropy / np.log(4)  # Normalize to [0,1]
    else:
        entropy = 0.5
    
    return {
        'clustering': hopkins,
        'entropy': entropy
    }

def _analyze_graph_properties(distance_matrix: np.ndarray) -> Dict:
    """Analyze graph properties of the distance matrix."""
    n = distance_matrix.shape[0]
    
    if n <= 3:
        return {'diameter': 1.0, 'avg_path_length': 1.0, 'connectivity': 1.0}
    
    # Approximate graph diameter using Floyd-Warshall (sparse approximation)
    k = min(50, n)
    sample_indices = np.random.choice(n, k, replace=False)
    
    max_path = 0
    total_paths = 0
    path_sum = 0
    
    for i in sample_indices:
        # Dijkstra-like approximation from node i
        dists = distance_matrix[i].copy()
        unvisited = set(range(n))
        current = i
        
        for _ in range(min(10, n)):
            unvisited.remove(current)
            if not unvisited:
                break
                
            # Update distances through current node
            for v in list(unvisited)[:min(20, len(unvisited))]:
                alt = dists[current] + distance_matrix[current, v]
                if alt < dists[v]:
                    dists[v] = alt
            
            # Move to closest unvisited
            current = min(unvisited, key=lambda x: dists[x])
        
        # Update statistics
        valid_dists = dists[dists < np.inf]
        if len(valid_dists) > 0:
            max_path = max(max_path, np.max(valid_dists))
            path_sum += np.sum(valid_dists)
            total_paths += len(valid_dists)
    
    diameter = max_path / (np.mean(distance_matrix) + 1e-8)
    avg_path_length = path_sum / (total_paths + 1e-8) / (np.mean(distance_matrix) + 1e-8)
    
    # Connectivity measure (approximate)
    mst = minimum_spanning_tree(distance_matrix)
    mst_density = mst.nnz / (n * (n - 1) / 2)
    
    return {
        'diameter': min(diameter, 5.0),
        'avg_path_length': min(avg_path_length, 3.0),
        'connectivity': 1.0 / (1.0 + np.exp(-10 * (mst_density - 0.1)))
    }

def _compute_multi_facet_scores(distance_matrix: np.ndarray, stats: Dict) -> Dict:
    """Compute multiple complementary base scores."""
    n = stats['n']
    
    scores = {}
    
    # 1. Enhanced inverse distance with adaptive scaling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
        inv_dist[np.diag_indices(n)] = 0
    # Adaptive scaling based on distance distribution
    scale = 1.0 / (stats['mean_dist'] + 1e-8)
    scores['inv_dist'] = np.tanh(inv_dist * scale)  # Non-linear scaling
    
    # 2. Advanced regret scores
    scores['regret'] = _compute_enhanced_regret(distance_matrix, stats)
    
    # 3. Density-adaptive scores
    scores['density'] = _compute_density_adaptive(distance_matrix, stats)
    
    # 4. MST-based scores with edge importance
    scores['mst'] = _compute_mst_importance(distance_matrix, stats)
    
    # 5. Structural scores based on point distribution
    scores['structural'] = _compute_structural_scores(distance_matrix, stats)
    
    return scores

def _compute_enhanced_regret(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute regret scores considering multiple alternatives."""
    n = stats['n']
    regret = np.ones((n, n))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Sort distances
        sorted_idx = np.argsort(dists)
        if len(sorted_idx) >= 3:
            # Compute regret for top-k alternatives
            k = min(5, len(sorted_idx))
            best_dist = dists[sorted_idx[0]]
            alt_dists = dists[sorted_idx[1:k]]
            
            # Multi-alternative regret
            regret_values = (alt_dists - best_dist) / (best_dist + 1e-8)
            avg_regret = np.mean(regret_values)
            
            # Assign scores based on relative position
            for rank, j in enumerate(sorted_idx):
                if j != i:
                    rel_regret = max(0, dists[j] - best_dist) / (best_dist + 1e-8)
                    # Sigmoid transformation for smooth scoring
                    score = 2.0 / (1.0 + np.exp(2.0 * rel_regret))
                    regret[i, j] = score
    
    # Symmetrize and normalize
    regret = (regret + regret.T) / 2
    regret[np.diag_indices(n)] = 0
    
    max_val = np.max(regret)
    if max_val > 0:
        regret = regret / max_val
    
    return regret

def _compute_density_adaptive(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute density scores with adaptive neighborhood."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Adaptive neighborhood size
    k_base = min(7, max(3, int(np.log2(n))))
    k = int(k_base * (1.0 + 0.5 * stats['clustering_tendency']))
    k = min(k, n - 1)
    
    # Compute local density measures
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k)[:k]
        # Use harmonic mean for robustness
        local_density[i] = k / np.sum(1.0 / (k_nearest + 1e-8))
    
    # Normalize and transform
    if np.std(local_density) > 1e-8:
        local_density = (local_density - np.mean(local_density)) / np.std(local_density)
    
    # Score edges: favor connections between sparse regions
    for i in range(n):
        for j in range(i + 1, n):
            # Combined sparse-region preference
            sparse_score = np.exp(-0.3 * (abs(local_density[i]) + abs(local_density[j])))
            
            # Distance-adjusted density
            d_ij = distance_matrix[i, j]
            expected_d = (local_density[i] + local_density[j]) / 2.0
            dist_ratio = d_ij / (expected_d + 1e-8) if expected_d > 0 else 1.0
            
            # Combine factors
            density_scores[i, j] = density_scores[j, i] = sparse_score / (1.0 + 0.5 * abs(dist_ratio - 1.0))
    
    return density_scores

def _compute_mst_importance(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute MST importance with edge criticality."""
    n = stats['n']
    mst_scores = np.ones((n, n))
    
    if n <= 2:
        return mst_scores
    
    try:
        # Compute MST
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_present = (mst + mst.T) > 0
        
        # Basic MST boost
        mst_scores[mst_present] = 1.5
        
        # Additional boost for critical edges (edges whose removal increases MST weight significantly)
        if n >= 10:
            # Sample alternative MSTs for edge importance
            n_samples = min(20, n)
            for _ in range(n_samples):
                # Perturb distances slightly
                perturbation = np.random.normal(1.0, 0.01, distance_matrix.shape)
                perturbed_dist = distance_matrix * perturbation
                perturbed_mst = minimum_spanning_tree(perturbed_dist).toarray()
                perturbed_present = (perturbed_mst + perturbed_mst.T) > 0
                
                # Boost edges that appear in multiple MSTs
                mst_scores[perturbed_present] *= 1.05
        
        # Cap scores
        mst_scores = np.minimum(mst_scores, 2.5)
        
    except Exception:
        pass
    
    mst_scores[np.diag_indices(n)] = 0
    return mst_scores

def _compute_structural_scores(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute scores based on global structural patterns."""
    n = stats['n']
    structural = np.ones((n, n))
    
    if n <= 3:
        return structural
    
    # Distance percentile analysis
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Compute percentiles
        percentiles = np.zeros(n)
        for j in range(n):
            if j != i:
                percentiles[j] = np.mean(dists <= dists[j])
        
        # Score edges based on optimal percentile range
        for j in range(n):
            if j != i:
                p = percentiles[j]
                # Prefer edges in medium percentile range (avoid extremes)
                if 0.2 <= p <= 0.8:
                    score = 1.2
                elif 0.1 <= p <= 0.9:
                    score = 1.0
                else:
                    score = 0.8
                structural[i, j] = score
    
    # Symmetrize
    structural = (structural + structural.T) / 2
    structural[np.diag_indices(n)] = 0
    
    return structural

def _compute_rl_lookahead_scores(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute lookahead scores with reinforcement learning principles."""
    n = stats['n']
    lookahead = np.ones((n, n))
    
    if n < 5:
        return lookahead
    
    # Adaptive parameters
    beam_width = min(15, max(3, int(np.log2(n) * 3)))
    lookahead_depth = min(4, max(2, int(np.log2(n))))
    n_simulations = min(25, max(5, n // 4))
    
    for i in range(n):
        for j in range(i + 1, n):
            # Skip obviously bad edges
            if distance_matrix[i, j] > 3 * stats['mean_dist']:
                lookahead[i, j] = lookahead[j, i] = 0.5
                continue
            
            scores = []
            for _ in range(n_simulations):
                # Beam search simulation
                score = _beam_search_simulation(distance_matrix, i, j, 
                                               beam_width, lookahead_depth, stats)
                scores.append(score)
            
            if scores:
                lookahead[i, j] = lookahead[j, i] = np.percentile(scores, 75)  # Use 75th percentile
    
    # Normalize
    max_val = np.max(lookahead)
    if max_val > 0:
        lookahead = lookahead / max_val
    
    lookahead[np.diag_indices(n)] = 0
    return lookahead

def _beam_search_simulation(distance_matrix: np.ndarray, start_i: int, start_j: int,
                           beam_width: int, depth: int, stats: Dict) -> float:
    """Perform beam search simulation starting with edge (i,j)."""
    n = stats['n']
    
    # Initial partial tour
    beam = [{
        'tour': [start_i, start_j],
        'length': distance_matrix[start_i, start_j],
        'visited': set([start_i, start_j])
    }]
    
    for step in range(depth):
        if len(beam) == 0:
            break
        
        new_beam = []
        for state in beam:
            current = state['tour'][-1]
            unvisited = [v for v in range(n) if v not in state['visited']]
            
            if not unvisited:
                continue
            
            # Evaluate candidate extensions
            candidates = []
            for next_node in unvisited[:min(20, len(unvisited))]:
                new_length = state['length'] + distance_matrix[current, next_node]
                
                # Heuristic estimate for remaining path
                remaining = len(unvisited) - 1
                if remaining > 0:
                    # Estimate based on average distance to nearest neighbors
                    est_remaining = remaining * stats['mean_dist'] * 0.8
                else:
                    est_remaining = distance_matrix[next_node, start_i]  # Return to start
                
                total_estimate = new_length + est_remaining
                candidates.append((next_node, total_estimate))
            
            # Keep top candidates
            candidates.sort(key=lambda x: x[1])
            for next_node, _ in candidates[:beam_width]:
                new_state = {
                    'tour': state['tour'] + [next_node],
                    'length': state['length'] + distance_matrix[current, next_node],
                    'visited': state['visited'].union([next_node])
                }
                new_beam.append(new_state)
        
        # Prune beam
        if new_beam:
            new_beam.sort(key=lambda x: x['length'])
            beam = new_beam[:beam_width]
    
    # Evaluate final beam states
    if not beam:
        return 0.5
    
    # Score based on best tour length in beam
    best_length = min(state['length'] for state in beam)
    expected_length = (depth + 1) * stats['mean_dist']
    score = expected_length / (best_length + 1e-8)
    
    return min(score, 3.0)

def _compute_spectral_scores(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute scores using spectral embedding."""
    n = stats['n']
    spectral = np.ones((n, n))
    
    if n <= 5:
        return spectral
    
    try:
        # Convert distances to similarities
        sigma = np.median(distance_matrix[distance_matrix > 0])
        similarities = np.exp(-distance_matrix ** 2 / (2 * sigma ** 2))
        np.fill_diagonal(similarities, 0)
        
        # Compute normalized Laplacian
        D = np.diag(np.sum(similarities, axis=1))
        D_inv_sqrt = np.diag(1.0 / np.sqrt(np.diag(D) + 1e-8))
        L_norm = np.eye(n) - D_inv_sqrt @ similarities @ D_inv_sqrt
        
        # Compute eigenvectors (smallest eigenvalues)
        k = min(3, n - 2)
        _, eigenvectors = eigsh(L_norm, k=k, which='SM')
        
        # Compute embedding distances
        for i in range(n):
            for j in range(i + 1, n):
                embed_dist = np.linalg.norm(eigenvectors[i] - eigenvectors[j])
                # Convert to similarity score
                spectral[i, j] = spectral[j, i] = np.exp(-embed_dist)
        
        # Normalize
        max_val = np.max(spectral)
        if max_val > 0:
            spectral = spectral / max_val
            
    except Exception:
        # Fallback to distance-based embedding
        try:
            # Use MDS approximation
            from sklearn.manifold import MDS
            mds = MDS(n_components=2, dissimilarity='precomputed', random_state=42)
            embedding = mds.fit_transform(distance_matrix)
            
            for i in range(n):
                for j in range(i + 1, n):
                    embed_dist = np.linalg.norm(embedding[i] - embedding[j])
                    spectral[i, j] = spectral[j, i] = 1.0 / (1.0 + embed_dist)
        except Exception:
            pass
    
    spectral[np.diag_indices(n)] = 0
    return spectral

def _compute_centrality_scores(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute scores using graph centrality measures."""
    n = stats['n']
    centrality = np.ones((n, n))
    
    if n <= 3:
        return centrality
    
    try:
        # Build k-nearest neighbor graph
        k = min(10, max(3, int(np.log2(n) * 2)))
        
        # Edge betweenness approximation
        betweenness = np.zeros((n, n))
        n_samples = min(50, n * (n - 1) // 2)
        
        # Sample source-destination pairs
        for _ in range(n_samples):
            src, dst = np.random.choice(n, 2, replace=False)
            
            # Simple shortest path approximation
            path = _approximate_shortest_path(distance_matrix, src, dst, k)
            
            # Update edge counts
            for i in range(len(path) - 1):
                u, v = path[i], path[i + 1]
                betweenness[u, v] += 1
                betweenness[v, u] += 1
        
        # Convert to scores (edges on more shortest paths get higher scores)
        max_between = np.max(betweenness)
        if max_between > 0:
            centrality = 0.5 + 0.5 * (betweenness / max_between)
        
    except Exception:
        pass
    
    centrality[np.diag_indices(n)] = 0
    return centrality

def _approximate_shortest_path(distance_matrix: np.ndarray, src: int, dst: int, k: int) -> list:
    """Approximate shortest path using greedy approach with k-nearest neighbors."""
    n = distance_matrix.shape[0]
    
    current = src
    visited = set([src])
    path = [src]
    
    while current != dst and len(path) < n:
        # Consider k nearest neighbors of current
        dists = distance_matrix[current].copy()
        for v in visited:
            dists[v] = np.inf
        
        candidates = np.argsort(dists)[:k]
        if len(candidates) == 0:
            break
        
        # Choose candidate minimizing distance to destination
        best = min(candidates, key=lambda x: distance_matrix[x, dst])
        
        path.append(best)
        visited.add(best)
        current = best
    
    return path

def _compute_intelligent_weights(distance_matrix: np.ndarray, stats: Dict) -> Dict:
    """Compute intelligent weights for component fusion."""
    n = stats['n']
    cv = stats['cv_dist']
    spatial_entropy = stats['spatial_entropy']
    clustering = stats['clustering_tendency']
    
    # Base weights with instance-aware adjustments
    weights = {
        'w_inv': 1.0,  # Always important
        
        # Regret weight: higher for structured instances
        'w_regret': 0.4 + 0.3 * (1.0 - spatial_entropy),
        
        # Density weight: higher for clustered instances
        'w_density': 0.3 + 0.4 * clustering,
        
        # MST weight: higher for sparse, tree-like instances
        'w_mst': 0.2 + 0.3 * stats['connectivity'],
        
        # Structural weight: balances other factors
        'w_structural': 0.1 + 0.2 * cv,
        
        # Lookahead weight: higher for complex instances
        'w_lookahead': 0.2 + 0.3 * min(stats['graph_diameter'], 2.0),
        
        # Spectral weight: higher for geometrically regular instances
        'w_spectral': 0.3 + 0.2 * (1.0 - cv),
        
        # Centrality weight: higher for hub-and-spoke patterns
        'w_centrality': 0.2 + 0.2 * stats['avg_path_length']
    }
    
    # Normalize to maintain balance
    total = sum(weights.values())
    scale_factor = 7.0 / total  # Target sum of 7
    for key in weights:
        weights[key] *= scale_factor
    
    return weights

def _fuse_heuristic_components(base_scores: Dict, lookahead_scores: np.ndarray,
                              spectral_scores: np.ndarray, centrality_scores: np.ndarray,
                              weights: Dict, stats: Dict) -> np.ndarray:
    """Intelligently fuse all heuristic components."""
    n = stats['n']
    H = np.ones((n, n))
    
    # Candidate mask for efficiency
    candidate_mask = _build_adaptive_mask(base_scores['inv_dist'], stats)
    
    # Geometric mean fusion with adaptive weights
    for i in range(n):
        for j in range(i + 1, n):
            if candidate_mask[i, j]:
                # Collect all components
                components = {
                    'inv_dist': base_scores['inv_dist'][i, j],
                    'regret': base_scores['regret'][i, j],
                    'density': base_scores['density'][i, j],
                    'mst': base_scores['mst'][i, j],
                    'structural': base_scores['structural'][i, j],
                    'lookahead': lookahead_scores[i, j],
                    'spectral': spectral_scores[i, j],
                    'centrality': centrality_scores[i, j]
                }
                
                # Weighted geometric mean with epsilon to avoid zeros
                log_sum = 0
                weight_sum = 0
                
                for key, weight in weights.items():
                    comp_key = key[2:]  # Remove 'w_' prefix
                    if comp_key in components:
                        value = max(components[comp_key], 1e-8)
                        log_sum += weight * np.log(value)
                        weight_sum += weight
                
                if weight_sum > 0:
                    fused = np.exp(log_sum / weight_sum)
                else:
                    fused = 1.0
                
                H[i, j] = H[j, i] = fused
    
    return H

def _build_adaptive_mask(inv_dist_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Build adaptive candidate mask."""
    n = stats['n']
    
    # Base candidate count
    k_base = min(12, max(5, int(np.log2(n) * 3)))
    
    # Adjust based on instance characteristics
    k = int(k_base * (1.0 + 0.3 * stats['clustering_tendency']))
    k = min(k, n - 1)
    
    mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Use inverse distance scores to select candidates
        scores = inv_dist_matrix[i].copy()
        scores[i] = -np.inf
        
        # Select top-k scores
        if k < n - 1:
            indices = np.argpartition(-scores, k)[:k]
        else:
            indices = np.where(scores > 0)[0]
        
        mask[i, indices] = True
    
    # Make symmetric and remove diagonal
    mask = mask | mask.T
    np.fill_diagonal(mask, False)
    
    return mask

def _advanced_post_process(H: np.ndarray, distance_matrix: np.ndarray, 
                          stats: Dict) -> np.ndarray:
    """Advanced post-processing of heuristic matrix."""
    n = stats['n']
    
    # Ensure symmetry and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Ensure minimum connectivity
    min_edges = min(4, n - 1)
    for i in range(n):
        # Count non-zero entries
        non_zero = np.where(H[i] > 0)[0]
        if len(non_zero) < min_edges:
            # Add nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    base_score = 1.0 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    # Row-wise softmax normalization
    for i in range(n):
        row = H[i]
        if np.any(row > 0):
            # Shift and scale for numerical stability
            row = row - np.min(row) + 1e-8
            exp_row = np.exp(row - np.max(row))
            H[i] = exp_row / np.sum(exp_row)
    
    # Global scaling
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
