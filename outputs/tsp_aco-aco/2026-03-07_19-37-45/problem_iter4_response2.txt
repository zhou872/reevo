import numpy as np
from typing import Dict, Tuple
import heapq

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with dynamic regret thresholds,
    multi-scale lookahead, and topological clustering metrics.
    
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Instance analysis
    stats = analyze_instance_v2(distance_matrix)
    
    # Compute core heuristic factors
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    regret_scores = compute_dynamic_regret(distance_matrix, stats)
    clustering_bonus = compute_topological_clustering(distance_matrix, stats)
    lookahead_penalty = compute_multi_scale_lookahead(distance_matrix, stats)
    
    # Build adaptive candidate mask
    candidate_mask = build_adaptive_candidate_mask(distance_matrix, inv_dist, stats)
    
    # Initialize heuristic matrix
    H = np.zeros((n, n))
    
    # Combine factors for candidate edges
    for i in range(n):
        for j in range(i + 1, n):
            if candidate_mask[i, j]:
                # Multiplicative combination with dynamic weighting
                combined = (
                    (inv_dist[i, j] ** stats['weights']['w_inv']) *
                    (regret_scores[i, j] ** stats['weights']['w_regret']) *
                    (clustering_bonus[i, j] ** stats['weights']['w_cluster']) *
                    (lookahead_penalty[i, j] ** stats['weights']['w_lookahead'])
                )
                H[i, j] = H[j, i] = combined
    
    # Post-processing and connectivity assurance
    H = post_process_v2(H, distance_matrix, stats)
    
    return H


def analyze_instance_v2(distance_matrix: np.ndarray) -> Dict:
    """Comprehensive instance analysis for adaptive parameter tuning."""
    n = distance_matrix.shape[0]
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n - 1)
    
    # Basic statistics
    stats = {
        'n': n,
        'mean_dist': np.mean(dist_no_diag),
        'median_dist': np.median(dist_no_diag),
        'std_dist': np.std(dist_no_diag),
        'cv_dist': np.std(dist_no_diag) / (np.mean(dist_no_diag) + 1e-8),
    }
    
    # Topology metrics
    stats['regularity'] = compute_regularity_metric(distance_matrix)
    stats['clustering'] = compute_clustering_metric(distance_matrix)
    stats['sparsity_factor'] = compute_sparsity_factor(distance_matrix)
    
    # Adaptive weights based on instance characteristics
    stats['weights'] = compute_adaptive_weights_v2(stats)
    
    return stats


def compute_regularity_metric(distance_matrix: np.ndarray) -> float:
    """Measure regularity based on nearest neighbor distribution."""
    n = distance_matrix.shape[0]
    
    # Compute nearest neighbor distances
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    
    nn_dists = np.array(nn_dists)
    
    # Regularity: 1 - coefficient of variation of nearest neighbor distances
    regularity = 1.0 - (np.std(nn_dists) / (np.mean(nn_dists) + 1e-8))
    return np.clip(regularity, 0.1, 0.9)


def compute_clustering_metric(distance_matrix: np.ndarray) -> float:
    """Compute topological clustering coefficient based on triangle inequality violations."""
    n = distance_matrix.shape[0]
    if n < 3:
        return 0.5
    
    violations = 0
    total_triangles = 0
    
    # Sample triangles for efficiency
    max_samples = min(1000, n * (n - 1) * (n - 2) // 6)
    sampled = 0
    
    for i in range(n):
        for j in range(i + 1, n):
            for k in range(j + 1, n):
                if sampled >= max_samples:
                    break
                    
                # Check triangle inequality
                d_ij = distance_matrix[i, j]
                d_ik = distance_matrix[i, k]
                d_jk = distance_matrix[j, k]
                
                violation_score = 0
                if d_ij > d_ik + d_jk * 1.05:  # 5% tolerance
                    violation_score += 1
                if d_ik > d_ij + d_jk * 1.05:
                    violation_score += 1
                if d_jk > d_ij + d_ik * 1.05:
                    violation_score += 1
                
                violations += violation_score
                total_triangles += 3
                sampled += 1
    
    if total_triangles == 0:
        return 0.5
    
    # Lower violations indicate more clustering structure
    clustering = 1.0 - (violations / total_triangles)
    return np.clip(clustering, 0.1, 0.9)


def compute_sparsity_factor(distance_matrix: np.ndarray) -> float:
    """Compute factor indicating how sparse/dense promising connections are."""
    n = distance_matrix.shape[0]
    
    # Count edges within 150% of minimum distance for each node
    densities = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        min_dist = np.min(dists)
        threshold = min_dist * 1.5
        
        # Count how many edges are within threshold
        density = np.sum(dists <= threshold) / (n - 1)
        densities.append(density)
    
    return np.mean(densities)


def compute_dynamic_regret(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute regret scores with dynamic thresholding based on local distance distribution."""
    n = stats['n']
    regret = np.ones((n, n))
    
    # Dynamic threshold based on instance characteristics
    base_threshold = 0.3 + 0.4 * stats['regularity']  # More regular -> higher threshold
    
    for i in range(n):
        # Get distances from node i
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Sort distances
        sorted_dists = np.sort(dists)
        sorted_indices = np.argsort(dists)
        
        # Find best distance
        best_dist = sorted_dists[0]
        
        # Dynamic threshold: consider more alternatives in clustered instances
        k = max(2, min(5, int(2 + 3 * stats['clustering'])))
        threshold_idx = min(k - 1, n - 2)
        threshold_dist = sorted_dists[threshold_idx]
        
        for j in range(n):
            if i == j:
                continue
                
            # Compute adaptive regret
            if distance_matrix[i, j] <= threshold_dist:
                # Within threshold: linear interpolation
                rank = np.where(sorted_indices == j)[0][0]
                regret_val = 1.0 - (rank / (threshold_idx + 1)) * base_threshold
            else:
                # Beyond threshold: exponential decay
                excess_ratio = (distance_matrix[i, j] - threshold_dist) / (threshold_dist + 1e-8)
                regret_val = (1.0 - base_threshold) * np.exp(-excess_ratio)
            
            regret[i, j] = max(0.1, regret_val)
    
    # Symmetrize
    regret = (regret + regret.T) / 2
    np.fill_diagonal(regret, 0)
    return regret


def compute_topological_clustering(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute clustering bonus based on local neighborhood structure."""
    n = stats['n']
    bonus = np.ones((n, n))
    
    # For each node, identify its neighborhood
    neighborhood_size = max(3, min(8, int(np.log2(n) * 1.5)))
    
    for i in range(n):
        # Get nearest neighbors
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest = np.argpartition(dists, neighborhood_size)[:neighborhood_size]
        
        for j in range(n):
            if i == j:
                continue
            
            # Check if j is in i's neighborhood
            if j in nearest:
                # Compute intra-neighborhood connectivity
                shared_neighbors = 0
                for k in nearest:
                    if k == j:
                        continue
                    # Check if j and k are also close
                    if distance_matrix[j, k] <= 1.5 * min(distance_matrix[i, j], distance_matrix[i, k]):
                        shared_neighbors += 1
                
                # Bonus proportional to shared neighbors
                cluster_bonus = 1.0 + (shared_neighbors / (neighborhood_size - 1)) * 0.5
                bonus[i, j] = min(1.5, cluster_bonus)
    
    # Symmetrize
    bonus = (bonus + bonus.T) / 2
    np.fill_diagonal(bonus, 0)
    return bonus


def compute_multi_scale_lookahead(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute lookahead penalty with adaptive depth based on instance size."""
    n = stats['n']
    penalty = np.ones((n, n))
    
    # Determine lookahead depth
    if n <= 50:
        depth = 2  # Two-step lookahead for small instances
    elif n <= 200:
        depth = 3  # Three-step for medium instances
    else:
        depth = 4  # Four-step for large instances (approximated)
    
    if depth == 2:
        # Two-step lookahead (exact)
        for i in range(n):
            for j in range(i + 1, n):
                direct = distance_matrix[i, j]
                
                # Find best two-hop alternative
                best_two_step = np.inf
                for k in range(n):
                    if k == i or k == j:
                        continue
                    two_step = distance_matrix[i, k] + distance_matrix[k, j]
                    best_two_step = min(best_two_step, two_step)
                
                # Apply penalty if two-step is better
                if best_two_step < direct * 0.8:
                    penalty[i, j] = penalty[j, i] = 0.4
                elif best_two_step < direct * 0.9:
                    penalty[i, j] = penalty[j, i] = 0.7
    
    elif depth == 3:
        # Three-step lookahead (approximate with beam search)
        beam_width = max(5, int(np.sqrt(n)))
        
        for i in range(n):
            for j in range(i + 1, n):
                direct = distance_matrix[i, j]
                
                # Beam search for best three-step path
                best_three_step = np.inf
                
                # First step: from i to intermediate nodes
                dists_i = distance_matrix[i].copy()
                dists_i[i] = dists_i[j] = np.inf
                candidates_i = np.argpartition(dists_i, beam_width)[:beam_width]
                
                for k in candidates_i:
                    # Second step: from k to potential l
                    dists_k = distance_matrix[k].copy()
                    dists_k[i] = dists_k[k] = dists_k[j] = np.inf
                    candidates_k = np.argpartition(dists_k, beam_width)[:beam_width]
                    
                    for l in candidates_k:
                        three_step = distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j]
                        best_three_step = min(best_three_step, three_step)
                
                # Apply penalty
                if best_three_step < direct * 0.7:
                    penalty[i, j] = penalty[j, i] = 0.3
                elif best_three_step < direct * 0.85:
                    penalty[i, j] = penalty[j, i] = 0.6
    
    else:
        # Four-step lookahead (highly approximate)
        for i in range(n):
            for j in range(i + 1, n):
                direct = distance_matrix[i, j]
                
                # Use triangle inequality heuristic for multi-step
                # Estimate lower bound using nearest neighbors
                dists_i = distance_matrix[i].copy()
                dists_i[i] = np.inf
                nn_i = np.min(dists_i)
                
                dists_j = distance_matrix[j].copy()
                dists_j[j] = np.inf
                nn_j = np.min(dists_j)
                
                # Estimated multi-step lower bound
                estimated_multi_step = nn_i + stats['mean_dist'] * (depth - 2) + nn_j
                
                if estimated_multi_step < direct * 0.6:
                    penalty[i, j] = penalty[j, i] = 0.2
                elif estimated_multi_step < direct * 0.8:
                    penalty[i, j] = penalty[j, i] = 0.5
    
    np.fill_diagonal(penalty, 0)
    return penalty


def build_adaptive_candidate_mask(distance_matrix: np.ndarray, 
                                  inv_dist: np.ndarray, 
                                  stats: Dict) -> np.ndarray:
    """Build candidate edge mask with adaptive sparsity control."""
    n = stats['n']
    mask = np.zeros((n, n), dtype=bool)
    
    # Base number of neighbors
    k_base = max(3, min(15, int(np.log2(n) * 2.5)))
    
    # Adjust based on sparsity factor
    k = int(k_base * (1.0 + stats['sparsity_factor']))
    k = min(k, n - 1)
    
    # Include k-nearest neighbors for each node
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < n - 1:
            nearest = np.argpartition(dists, k)[:k]
        else:
            nearest = np.arange(n)
            nearest = nearest[nearest != i]
        
        mask[i, nearest] = True
    
    # Add promising edges based on inverse distance ranking
    for i in range(n):
        inv_dists = inv_dist[i].copy()
        inv_dists[i] = -np.inf
        
        # Take top m additional edges beyond nearest neighbors
        m = max(1, int(k * 0.3))
        top_indices = np.argpartition(inv_dists, -m)[-m:]
        
        for j in top_indices:
            if not mask[i, j]:
                mask[i, j] = True
    
    # Ensure symmetry
    mask = mask | mask.T
    np.fill_diagonal(mask, False)
    
    return mask


def compute_adaptive_weights_v2(stats: Dict) -> Dict[str, float]:
    """Compute adaptive weights based on comprehensive instance analysis."""
    cv = stats['cv_dist']
    regularity = stats['regularity']
    clustering = stats['clustering']
    n = stats['n']
    
    # Base weights adjust based on all instance properties
    weights = {
        'w_inv': 1.2 - 0.3 * clustering,  # Less important in clustered instances
        
        # Regret more important for irregular instances
        'w_regret': 0.7 + 0.5 * (1 - regularity),
        
        # Clustering bonus more important in structured instances
        'w_cluster': 0.4 + 0.4 * clustering,
        
        # Lookahead more important when variation is low and instance is larger
        'w_lookahead': 0.3 + 0.3 * (1 - cv) + 0.2 * min(1.0, np.log(n) / 10)
    }
    
    # Normalize to sum to 4
    total = sum(weights.values())
    scale = 4.0 / total
    for key in weights:
        weights[key] *= scale
    
    return weights


def post_process_v2(H: np.ndarray, 
                   distance_matrix: np.ndarray, 
                   stats: Dict) -> np.ndarray:
    """Final post-processing of the heuristic matrix."""
    n = stats['n']
    
    # Ensure minimum connectivity
    min_connections = max(2, min(4, int(np.log2(n))))
    
    for i in range(n):
        row = H[i].copy()
        row[i] = -np.inf
        non_zero = np.where(row > 0)[0]
        
        if len(non_zero) < min_connections:
            # Add nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_connections]
            
            for j in nearest:
                if H[i, j] == 0:
                    # Compute base score
                    inv_dist = 1.0 / (dists[j] + 1e-8)
                    regret = compute_simple_regret(distance_matrix, i, j)
                    H[i, j] = H[j, i] = inv_dist * regret
    
    # Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Normalize to [0, 1] range with softmax-like scaling
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    # Apply sigmoid scaling to emphasize differences
    H = 1 / (1 + np.exp(-8 * (H - 0.5)))
    
    return H


def compute_simple_regret(distance_matrix: np.ndarray, i: int, j: int) -> float:
    """Quick regret computation for post-processing."""
    n = distance_matrix.shape[0]
    
    # Best alternative from i
    dists_i = distance_matrix[i].copy()
    dists_i[i] = dists_i[j] = np.inf
    best_i = np.min(dists_i)
    
    # Best alternative from j
    dists_j = distance_matrix[j].copy()
    dists_j[j] = dists_j[i] = np.inf
    best_j = np.min(dists_j)
    
    # Symmetric regret
    regret_i = (distance_matrix[i, j] - best_i) / (best_i + 1e-8)
    regret_j = (distance_matrix[i, j] - best_j) / (best_j + 1e-8)
    
    regret_score = 1.0 / (1.0 + (regret_i + regret_j) / 2)
    return max(0.1, regret_score)
