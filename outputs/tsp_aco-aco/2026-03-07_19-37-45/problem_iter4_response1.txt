import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with instance-size-aware parameter tuning,
    efficient two-step lookahead, and adaptive factor weighting.
    
    Key improvements:
    1. Instance-size-aware parameter scaling for all neighborhood sizes
    2. Efficient two-step regret using top-k neighbors only
    3. Adaptive weighting of factors based on instance characteristics
    4. Normalized combination with variance-aware scaling
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    # ========== 1. INSTANCE-SIZE-AWARE PARAMETER TUNING ==========
    # Scale all neighborhood parameters based on instance size
    base_k = max(2, min(10, int(np.sqrt(n))))  # Adaptive base neighborhood size
    
    # ========== 2. DYNAMIC POWER SCALING ==========
    flat_distances = distance_matrix[np.triu_indices(n, k=1)]
    if len(flat_distances) > 0:
        distance_std = np.std(flat_distances)
        distance_mean = np.mean(flat_distances)
        concentration = distance_std / (distance_mean + 1e-12)
        # More aggressive power scaling for concentrated instances
        power = 1.5 + 1.2 * np.exp(-concentration * 2)
    else:
        power = 1.5
    
    inv_dist = 1 / (distance_matrix + 1e-12)
    inv_dist_pow = inv_dist ** power
    
    # ========== 3. EFFICIENT TWO-STEP LOOKAHEAD REGRET ==========
    # Precompute k-nearest neighbors for efficiency
    k_regret = min(base_k * 2, n - 1)
    nn_indices = np.zeros((n, k_regret), dtype=int)
    nn_distances = np.zeros((n, k_regret))
    
    for i in range(n):
        distances_i = distance_matrix[i].copy()
        distances_i[i] = np.inf
        indices = np.argpartition(distances_i, k_regret)[:k_regret]
        sorted_idx = indices[np.argsort(distances_i[indices])]
        nn_indices[i] = sorted_idx
        nn_distances[i] = distances_i[sorted_idx]
    
    # Compute two-step regret efficiently using precomputed neighbors
    two_step_regret = np.ones_like(distance_matrix)
    
    for i in range(n):
        for j in range(i+1, n):
            # Find best two-step path i->k->j using precomputed neighbors
            min_two_step = np.inf
            
            # Check intersection of i's and j's neighbors
            i_neighbors = set(nn_indices[i][:k_regret//2])
            j_neighbors = set(nn_indices[j][:k_regret//2])
            candidate_ks = list(i_neighbors.intersection(j_neighbors))
            
            # Also consider direct neighbors of both endpoints
            if len(candidate_ks) == 0:
                candidate_ks = list(set(nn_indices[i][:k_regret//2]) | set(nn_indices[j][:k_regret//2]))
                candidate_ks = [k for k in candidate_ks if k != i and k != j]
            
            # Evaluate candidate two-step paths
            for k in candidate_ks[:min(10, len(candidate_ks))]:  # Limit computation
                two_step_cost = distance_matrix[i, k] + distance_matrix[k, j]
                min_two_step = min(min_two_step, two_step_cost)
            
            if np.isfinite(min_two_step):
                direct_cost = distance_matrix[i, j]
                regret_ratio = (min_two_step - direct_cost) / (direct_cost + 1e-12)
                # Exponential decay: high penalty when two-step is much better
                two_step_regret[i, j] = two_step_regret[j, i] = np.exp(-max(0, -regret_ratio) * 3)
    
    # ========== 4. ADAPTIVE REGRET-BASED PENALTY ==========
    regret_matrix = np.ones_like(distance_matrix)
    k_local = min(base_k, n - 1)
    
    for i in range(n):
        distances_i = distance_matrix[i].copy()
        distances_i[i] = np.inf
        
        if k_local < len(distances_i):
            kth_idx = np.argpartition(distances_i, k_local)[k_local-1]
            kth_best = distances_i[kth_idx]
            
            for j in range(n):
                if i != j:
                    dist_ij = distance_matrix[i, j]
                    regret = max(0, dist_ij - kth_best)
                    regret_norm = regret / (np.min(distances_i) + 1e-12)
                    # Sigmoid penalty: smooth transition
                    regret_matrix[i, j] *= 1.0 / (1.0 + np.exp(regret_norm - 2))
    
    # Symmetrize regret matrix
    regret_matrix = np.sqrt(regret_matrix * regret_matrix.T)
    
    # ========== 5. LOCAL CLUSTER DENSITY ==========
    density_factor = np.ones_like(distance_matrix)
    k_density = min(base_k * 2, n - 1)
    
    for i in range(n):
        distances_i = distance_matrix[i].copy()
        distances_i[i] = np.inf
        nn_indices_i = np.argpartition(distances_i, k_density)[:k_density]
        
        if len(nn_indices_i) > 1:
            # Calculate local density around node i
            local_dists = []
            for a in range(len(nn_indices_i)):
                for b in range(a+1, len(nn_indices_i)):
                    local_dists.append(distance_matrix[nn_indices_i[a], nn_indices_i[b]])
            
            if len(local_dists) > 0:
                avg_local_dist = np.mean(local_dists)
                if avg_local_dist > 0:
                    density_i = 1.0 / avg_local_dist
                    
                    # Boost edges within dense local neighborhoods
                    for j in nn_indices_i:
                        if i != j:
                            # Use soft boosting that saturates at high densities
                            boost = 1.0 + 0.3 * np.tanh(density_i / (distance_mean + 1e-12))
                            density_factor[i, j] *= boost
    
    # Symmetrize density factor
    density_factor = (density_factor + density_factor.T) / 2
    
    # ========== 6. ADAPTIVE FACTOR WEIGHTING ==========
    # Analyze instance characteristics to weight factors
    if len(flat_distances) > 0:
        # Measure distance distribution skewness
        skewness = np.mean((flat_distances - distance_mean) ** 3) / (distance_std ** 3 + 1e-12)
        
        # Weight factors based on instance characteristics
        # For skewed distributions, emphasize distance and regret more
        distance_weight = 1.2 if abs(skewness) > 0.5 else 1.0
        regret_weight = 1.1 if skewness > 0 else 1.0  # Positive skew: long tails
        two_step_weight = 0.9 if n > 15 else 1.0  # Reduce for large instances
        density_weight = 1.0 if distance_std/distance_mean > 0.3 else 0.8  # Reduce for uniform instances
    else:
        distance_weight = regret_weight = two_step_weight = density_weight = 1.0
    
    # ========== 7. COMBINE FACTORS WITH ADAPTIVE WEIGHTS ==========
    # Apply weights and normalize each factor to comparable scales
    factors = []
    weights = []
    
    # Normalize inv_dist_pow
    inv_norm = inv_dist_pow.copy()
    np.fill_diagonal(inv_norm, 0)
    if inv_norm.max() > 0:
        inv_norm = inv_norm / inv_norm.max()
    factors.append(inv_norm)
    weights.append(distance_weight)
    
    # Normalize regret_matrix
    regret_norm = regret_matrix.copy()
    np.fill_diagonal(regret_norm, 0)
    if regret_norm.max() > 0:
        regret_norm = regret_norm / regret_norm.max()
    factors.append(regret_norm)
    weights.append(regret_weight)
    
    # Normalize two_step_regret
    two_step_norm = two_step_regret.copy()
    np.fill_diagonal(two_step_norm, 0)
    if two_step_norm.max() > 0:
        two_step_norm = two_step_norm / two_step_norm.max()
    factors.append(two_step_norm)
    weights.append(two_step_weight)
    
    # Normalize density_factor
    density_norm = density_factor.copy()
    np.fill_diagonal(density_norm, 0)
    if density_norm.max() > 0:
        density_norm = density_norm / density_norm.max()
    factors.append(density_norm)
    weights.append(density_weight)
    
    # Weighted geometric mean (more stable than weighted sum)
    weights = np.array(weights)
    weights = weights / weights.sum()  # Normalize weights
    
    heuristics_matrix = np.ones_like(distance_matrix)
    for factor, weight in zip(factors, weights):
        heuristics_matrix *= factor ** weight
    
    # ========== 8. ADAPTIVE SPARSIFICATION ==========
    # Instance-size-aware sparsification
    k_nn = max(3, min(base_k * 2, n - 1))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        distances_i = distance_matrix[i].copy()
        distances_i[i] = np.inf
        nn_indices = np.argpartition(distances_i, k_nn)[:k_nn]
        nn_mask[i, nn_indices] = True
    
    # Symmetrize mask
    nn_mask = nn_mask | nn_mask.T
    
    # Distance threshold based on adaptive quantile
    if len(flat_distances) > 0:
        # Use tighter threshold for larger instances
        quantile = 0.6 + 0.2 * np.exp(-n/50)  # Ranges from 0.8 (small) to 0.6 (large)
        threshold = np.quantile(flat_distances, quantile)
        dist_mask = distance_matrix <= threshold
        np.fill_diagonal(dist_mask, False)
    else:
        dist_mask = np.ones_like(distance_matrix, dtype=bool)
        np.fill_diagonal(dist_mask, False)
    
    # Apply combined mask
    combined_mask = nn_mask & dist_mask
    heuristics_matrix = heuristics_matrix * combined_mask
    
    # Ensure minimum connectivity
    min_degree = min(3, n - 1)
    for i in range(n):
        non_zero = np.where(heuristics_matrix[i] > 0)[0]
        if len(non_zero) < min_degree:
            # Find strongest missing edges from unsparsified version
            original_score = np.ones_like(distance_matrix[i])
            for factor, weight in zip(factors, weights):
                original_score *= factor[i] ** weight
            original_score[i] = 0
            
            # Select top missing edges
            candidate_indices = np.argsort(original_score)[-min_degree:]
            for j in candidate_indices:
                if heuristics_matrix[i, j] == 0 and i != j:
                    heuristics_matrix[i, j] = original_score[j]
                    heuristics_matrix[j, i] = original_score[j]
    
    # ========== 9. FINAL PROCESSING ==========
    # Symmetrize
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    # Normalize to [0, 1]
    max_val = heuristics_matrix.max()
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
