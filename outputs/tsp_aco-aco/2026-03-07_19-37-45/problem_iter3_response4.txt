import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple advanced strategies.
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Adaptive parameter tuning
    stats = _analyze_instance_v2(distance_matrix)
    
    # Compute core factors with focus on long-term suggestions
    factors = _compute_core_factors_v2(distance_matrix, stats)
    
    # Build sparse candidate mask
    candidate_mask = _build_sparse_candidate_mask_v2(distance_matrix, stats)
    
    # Initialize heuristics matrix
    H = np.zeros((n, n))
    
    # Compute adaptive weights
    weights = _compute_adaptive_weights_v2(stats)
    
    # Combine factors for candidate edges only
    for i in range(n):
        # Get candidate neighbors
        neighbors = np.where(candidate_mask[i])[0]
        neighbors = neighbors[neighbors > i]  # Avoid double counting
        
        for j in neighbors:
            # Multiplicative combination of factors
            combined = (
                (factors['inv_dist'][i, j] ** weights['w_inv']) *
                (factors['two_step_regret'][i, j] ** weights['w_regret']) *
                (factors['degree_penalty'][i, j] ** weights['w_degree']) *
                (factors['two_hop_penalty'][i, j] ** weights['w_two_hop'])
            )
            H[i, j] = H[j, i] = combined
    
    # Post-processing
    H = _post_process_v2(H, distance_matrix, stats)
    
    return H

def _analyze_instance_v2(distance_matrix: np.ndarray) -> dict:
    """Analyze instance for adaptive parameter tuning."""
    n = distance_matrix.shape[0]
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    # Basic statistics
    stats = {
        'n': n,
        'mean_dist': np.mean(dist_no_diag),
        'std_dist': np.std(dist_no_diag),
        'cv_dist': np.std(dist_no_diag) / (np.mean(dist_no_diag) + 1e-8),
    }
    
    # Estimate sparsity/regularity
    stats['regularity'] = _estimate_regularity_v2(distance_matrix)
    
    return stats

def _estimate_regularity_v2(distance_matrix: np.ndarray) -> float:
    """Estimate how regular/structured the instance is."""
    n = distance_matrix.shape[0]
    
    # Compute nearest neighbor distances for all nodes
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    
    nn_dists = np.array(nn_dists)
    
    # Regular instances have similar nearest neighbor distances
    regularity = 1.0 - (np.std(nn_dists) / (np.mean(nn_dists) + 1e-8))
    return np.clip(regularity, 0.1, 0.9)

def _compute_core_factors_v2(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute core heuristic factors focusing on long-term suggestions."""
    n = stats['n']
    
    factors = {}
    
    # 1. Inverse distance (always important)
    factors['inv_dist'] = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(factors['inv_dist'], 0)
    
    # 2. Two-step lookahead regret (Suggestion 3)
    factors['two_step_regret'] = _compute_two_step_regret_v2(distance_matrix)
    
    # 3. Degree-aware penalization (Suggestion 2)
    factors['degree_penalty'] = _compute_degree_penalty_v2(distance_matrix, factors['inv_dist'])
    
    # 4. Two-hop lookahead penalty (Suggestion 4)
    factors['two_hop_penalty'] = _compute_two_hop_penalty_v2(distance_matrix)
    
    return factors

def _compute_two_step_regret_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Two-step lookahead regret estimation (Suggestion 3).
    For edge (i,j): compute difference between minimum alternative two-step path
    and (direct cost + minimum next step cost).
    """
    n = distance_matrix.shape[0]
    regret = np.ones((n, n))
    
    for i in range(n):
        # Prepare distances from i
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        
        # For each possible next node j
        for j in range(n):
            if i == j:
                continue
                
            # Direct cost + best continuation from j (excluding i)
            dists_j = distance_matrix[j].copy()
            dists_j[i] = np.inf
            dists_j[j] = np.inf
            best_from_j = np.min(dists_j)
            
            direct_path = distance_matrix[i, j] + best_from_j
            
            # Best alternative two-step path through k (k != i,j)
            best_alternative = np.inf
            for k in range(n):
                if k == i or k == j:
                    continue
                alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                if alt_cost < best_alternative:
                    best_alternative = alt_cost
            
            # Regret calculation
            if best_alternative > 0:
                regret_ij = (best_alternative - direct_path) / best_alternative
                # Convert to score: higher regret -> higher score (normalized to 0-2)
                regret[i, j] = 1.0 + max(0, regret_ij)
    
    np.fill_diagonal(regret, 0)
    return regret

def _compute_degree_penalty_v2(distance_matrix: np.ndarray, inv_dist: np.ndarray) -> np.ndarray:
    """
    Degree-aware penalization (Suggestion 2).
    Scale edge heuristics inversely with count of promising connections per node.
    """
    n = distance_matrix.shape[0]
    penalty = np.ones((n, n))
    
    # Count promising connections per node (top 25% of inverse distances)
    for i in range(n):
        inv_dists = inv_dist[i].copy()
        inv_dists[i] = -np.inf
        threshold = np.percentile(inv_dists, 75)
        degree_i = np.sum(inv_dists >= threshold)
        
        for j in range(n):
            if i == j:
                continue
                
            inv_dists_j = inv_dist[j].copy()
            inv_dists_j[j] = -np.inf
            threshold_j = np.percentile(inv_dists_j, 75)
            degree_j = np.sum(inv_dists_j >= threshold_j)
            
            # Penalize edges where both nodes have many promising connections
            penalty[i, j] = 1.0 / (1.0 + 0.1 * (degree_i + degree_j))
    
    np.fill_diagonal(penalty, 0)
    return penalty

def _compute_two_hop_penalty_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Two-hop lookahead penalty (Suggestion 4).
    Penalize direct edges if a two-hop path through intermediate node is significantly shorter.
    """
    n = distance_matrix.shape[0]
    penalty = np.ones((n, n))
    
    for i in range(n):
        for j in range(i+1, n):
            direct = distance_matrix[i, j]
            
            # Find best two-hop alternative
            best_two_hop = np.inf
            for k in range(n):
                if k == i or k == j:
                    continue
                two_hop = distance_matrix[i, k] + distance_matrix[k, j]
                if two_hop < best_two_hop:
                    best_two_hop = two_hop
            
            # Apply penalty if two-hop is much better (bypass penalty)
            if best_two_hop < direct * 0.7:  # Suggestion 4 uses 0.7 factor
                penalty[i, j] = penalty[j, i] = 0.3  # Suggestion 4 uses 0.3 factor
            elif best_two_hop < direct * 0.9:
                penalty[i, j] = penalty[j, i] = 0.7
    
    return penalty

def _build_sparse_candidate_mask_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """
    Adaptive sparse computation (Suggestion 5).
    Keep only promising edges: nearest neighbors plus some random connections.
    """
    n = stats['n']
    mask = np.zeros((n, n), dtype=bool)
    
    # Base number of nearest neighbors
    k_base = min(8, max(3, int(np.log2(n) * 2)))
    
    # Adjust based on regularity
    k = int(k_base * (1.5 - 0.5 * stats['regularity']))
    k = min(k, n - 1)
    
    # Include k-nearest neighbors for each node
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest = np.argpartition(dists, k)[:k]
        mask[i, nearest] = True
    
    # Add random connections for exploration (5% of remaining edges)
    remaining_edges = n * (n - 1) // 2 - np.sum(mask[np.triu_indices(n, 1)])
    n_random = max(1, int(0.05 * remaining_edges))
    
    if n_random > 0:
        triu_indices = np.triu_indices(n, 1)
        candidate_indices = np.where(~mask[triu_indices])[0]
        
        if len(candidate_indices) > 0:
            selected = np.random.choice(candidate_indices, 
                                       size=min(n_random, len(candidate_indices)), 
                                       replace=False)
            
            for idx in selected:
                i, j = triu_indices[0][idx], triu_indices[1][idx]
                mask[i, j] = mask[j, i] = True
    
    np.fill_diagonal(mask, False)
    return mask

def _compute_adaptive_weights_v2(stats: dict) -> dict:
    """Compute adaptive weights based on instance characteristics."""
    cv = stats['cv_dist']
    regularity = stats['regularity']
    n = stats['n']
    
    # Weights adjust based on instance properties
    weights = {
        'w_inv': 1.2,  # Base importance
        
        # Two-step regret more important for irregular instances
        'w_regret': 0.8 + 0.4 * (1 - regularity),
        
        # Degree penalty more important for regular/structured instances
        'w_degree': 0.3 + 0.4 * regularity,
        
        # Two-hop penalty more important when variation is low
        'w_two_hop': 0.4 + 0.3 * (1 - cv)
    }
    
    # Normalize weights (softmax-like normalization)
    total = sum(weights.values())
    scale_factor = 4.0 / total  # Target sum of 4
    for key in weights:
        weights[key] *= scale_factor
    
    return weights

def _post_process_v2(H: np.ndarray, distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Post-process the heuristic matrix."""
    n = stats['n']
    
    # Ensure each node has at least 2 connections
    min_connections = min(3, n - 1)
    for i in range(n):
        row = H[i].copy()
        row[i] = -np.inf
        non_zero = np.where(row > 0)[0]
        
        if len(non_zero) < min_connections:
            # Add nearest neighbors not already included
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Sort by distance
            sorted_indices = np.argsort(dists)
            added = 0
            
            for j in sorted_indices:
                if H[i, j] == 0:
                    H[i, j] = H[j, i] = 1 / (dists[j] + 1e-8)
                    added += 1
                    if added >= min_connections - len(non_zero):
                        break
    
    # Final symmetrization and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
