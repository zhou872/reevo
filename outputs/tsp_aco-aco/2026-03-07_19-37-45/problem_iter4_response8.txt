import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.cluster.hierarchy import linkage, fcluster
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection with meta-adaptive learning and
    ant colony simulation lookahead. Dynamically tunes all parameters based on
    instance characteristics using learned meta-rules.
    """
    n = distance_matrix.shape[0]
    if n <= 3:
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
        np.fill_diagonal(inv_dist, 0)
        return inv_dist
    
    # 1. Enhanced instance analysis with meta-learning inspired features
    instance_features = _analyze_instance_v2(distance_matrix)
    
    # 2. Meta-adaptive parameter selection
    params = _meta_adaptive_parameters(instance_features)
    
    # 3. Compute base components with instance-adaptive methods
    base_scores = _compute_adaptive_base_scores(distance_matrix, instance_features, params)
    
    # 4. Ant colony simulation for lookahead (enhanced from v1's simulation)
    ant_scores = _ant_colony_simulation_lookahead(distance_matrix, instance_features, params)
    
    # 5. Local-global edge betweenness analysis
    betweenness_scores = _compute_edge_betweenness(distance_matrix, instance_features, params)
    
    # 6. Multi-strategy ensemble combination
    heuristics_matrix = _ensemble_combination(
        base_scores, ant_scores, betweenness_scores, 
        instance_features, params
    )
    
    # 7. Post-processing with connectivity guarantees
    heuristics_matrix = _post_process_v2(heuristics_matrix, distance_matrix, params)
    
    return heuristics_matrix


def _analyze_instance_v2(distance_matrix: np.ndarray) -> dict:
    """Enhanced instance analysis with hierarchical clustering and geometry detection."""
    n = distance_matrix.shape[0]
    
    # Distance statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Shape analysis via distance distribution
    skewness = np.mean(((dist_no_diag - mean_dist) / (std_dist + 1e-8)) ** 3)
    kurtosis = np.mean(((dist_no_diag - mean_dist) / (std_dist + 1e-8)) ** 4) - 3
    
    # Hierarchical clustering structure
    cluster_structure = _analyze_cluster_structure(distance_matrix)
    
    # Nearest neighbor statistics
    nn_stats = _analyze_nearest_neighbors(distance_matrix)
    
    # Spatial regularity (grid-like vs random)
    regularity = _compute_spatial_regularity(distance_matrix)
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'skewness': skewness,
        'kurtosis': kurtosis,
        'cluster_tendency': cluster_structure['tendency'],
        'n_clusters': cluster_structure['n_clusters'],
        'cluster_compactness': cluster_structure['compactness'],
        'nn_mean_ratio': nn_stats['mean_ratio'],
        'nn_cv': nn_stats['cv'],
        'regularity': regularity,
        'is_euclidean': _check_euclidean(distance_matrix)
    }


def _analyze_cluster_structure(distance_matrix: np.ndarray) -> dict:
    """Analyze clustering tendency using hierarchical clustering."""
    n = distance_matrix.shape[0]
    if n < 10:
        return {'tendency': 0.5, 'n_clusters': 1, 'compactness': 1.0}
    
    # Perform hierarchical clustering
    condensed_dist = distance_matrix[np.triu_indices(n, k=1)]
    Z = linkage(condensed_dist, method='average')
    
    # Cut at different heights to analyze cluster structure
    cuts = np.linspace(np.min(Z[:, 2]), np.max(Z[:, 2]), 10)
    n_clusters_list = []
    for cut in cuts:
        if cut > 0:
            clusters = fcluster(Z, cut, criterion='distance')
            n_clusters_list.append(len(np.unique(clusters)))
    
    if n_clusters_list:
        n_clusters = np.median(n_clusters_list)
        # Compactness: ratio of within-cluster to between-cluster distances
        clusters = fcluster(Z, np.median(cuts), criterion='distance')
        within_dist = []
        between_dist = []
        for i in range(n):
            for j in range(i+1, n):
                if clusters[i] == clusters[j]:
                    within_dist.append(distance_matrix[i, j])
                else:
                    between_dist.append(distance_matrix[i, j])
        
        if within_dist and between_dist:
            compactness = np.mean(within_dist) / (np.mean(between_dist) + 1e-8)
            # Normalize compactness to [0, 1]
            compactness = np.exp(-compactness)
        else:
            compactness = 0.5
    else:
        n_clusters = 1
        compactness = 0.5
    
    # Cluster tendency: how distinct are clusters
    tendency = min(1.0, (n / (n_clusters + 1)) / 20)
    
    return {
        'tendency': tendency,
        'n_clusters': n_clusters,
        'compactness': compactness
    }


def _analyze_nearest_neighbors(distance_matrix: np.ndarray) -> dict:
    """Analyze nearest neighbor distribution characteristics."""
    n = distance_matrix.shape[0]
    nn_dists = []
    nn_ratios = []
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)
        if len(sorted_dists) >= 2:
            nn_dists.append(sorted_dists[0])
            nn_ratios.append(sorted_dists[1] / (sorted_dists[0] + 1e-8))
    
    if nn_dists:
        mean_ratio = np.mean(nn_ratios)
        cv = np.std(nn_dists) / (np.mean(nn_dists) + 1e-8)
    else:
        mean_ratio = 1.0
        cv = 0.0
    
    return {'mean_ratio': mean_ratio, 'cv': cv}


def _compute_spatial_regularity(distance_matrix: np.ndarray) -> float:
    """Measure spatial regularity (grid-like vs random)."""
    n = distance_matrix.shape[0]
    if n < 10:
        return 0.5
    
    # Analyze distance distribution compared to uniform random
    actual_dists = distance_matrix[np.triu_indices(n, k=1)]
    
    # Generate comparison distribution
    mean_dist = np.mean(actual_dists)
    random_dists = np.random.exponential(mean_dist, size=min(1000, len(actual_dists)))
    
    # Compare histograms using Earth Mover's Distance approximation
    hist_actual, _ = np.histogram(actual_dists, bins=20, density=True)
    hist_random, _ = np.histogram(random_dists, bins=20, density=True)
    
    # EMD approximation (1D Wasserstein distance)
    cdf_actual = np.cumsum(hist_actual) / np.sum(hist_actual)
    cdf_random = np.cumsum(hist_random) / np.sum(hist_random)
    
    emd = np.sum(np.abs(cdf_actual - cdf_random))
    # Convert to regularity score (0=random, 1=very regular)
    regularity = np.exp(-emd * 5)
    
    return regularity


def _check_euclidean(distance_matrix: np.ndarray) -> bool:
    """Check if distance matrix satisfies triangle inequality strongly."""
    n = distance_matrix.shape[0]
    if n < 3:
        return True
    
    violations = 0
    for i in range(n):
        for j in range(n):
            if i != j:
                for k in range(n):
                    if k != i and k != j:
                        if distance_matrix[i, j] > distance_matrix[i, k] + distance_matrix[k, j] * 1.01:
                            violations += 1
    
    return violations < n * 0.1


def _meta_adaptive_parameters(features: dict) -> dict:
    """Dynamically select parameters based on instance characteristics using meta-rules."""
    n = features['n']
    cv = features['cv_dist']
    regularity = features['regularity']
    cluster_tendency = features['cluster_tendency']
    nn_mean_ratio = features['nn_mean_ratio']
    
    # Meta-rules for parameter selection
    params = {}
    
    # Lookahead parameters
    if n < 50:
        params['n_ants'] = min(10, max(3, n // 5))
        params['lookahead_steps'] = min(8, max(3, n // 6))
    else:
        params['n_ants'] = min(20, max(5, int(np.log2(n) * 3)))
        params['lookahead_steps'] = min(15, max(5, int(np.log2(n) * 2)))
    
    # Increase simulations for irregular instances
    if cv > 0.5:
        params['n_ants'] = int(params['n_ants'] * 1.5)
    
    # Betweenness computation parameters
    params['betweenness_k'] = min(15, max(3, int(np.sqrt(n))))
    params['betweenness_samples'] = min(100, max(20, n * 2))
    
    # Adaptive weights for ensemble
    params['w_inv'] = 1.0  # Always important
    
    # Regret weight: higher when nearest neighbors are distinct
    params['w_regret'] = 0.5 + 0.3 * min(1.0, nn_mean_ratio - 1.0)
    
    # Cluster weight: higher for clustered instances
    params['w_cluster'] = 0.3 + 0.4 * cluster_tendency
    
    # Betweenness weight: higher for structured/regular instances
    params['w_betweenness'] = 0.2 + 0.3 * regularity
    
    # Ant simulation weight: higher for complex instances
    complexity = cv * (1 + cluster_tendency)
    params['w_ant'] = 0.3 + 0.3 * min(1.0, complexity)
    
    # MST weight: decreases with size but increases with regularity
    params['w_mst'] = max(0.1, 0.4 - 0.005 * n + 0.2 * regularity)
    
    # Normalize weights
    total = sum(params.get(k, 0) for k in ['w_inv', 'w_regret', 'w_cluster', 
                                          'w_betweenness', 'w_ant', 'w_mst'])
    for key in ['w_inv', 'w_regret', 'w_cluster', 'w_betweenness', 'w_ant', 'w_mst']:
        params[key] = params[key] / total * 6  # Sum to 6
    
    # Pheromone parameters for ant colony
    params['alpha'] = 1.0  # Pheromone importance
    params['beta'] = 2.0 + cv  # Heuristic importance
    params['evaporation'] = 0.3 + 0.2 * (1 - regularity)  # Higher for random
    
    return params


def _compute_adaptive_base_scores(distance_matrix: np.ndarray, 
                                  features: dict, params: dict) -> dict:
    """Compute base heuristic scores with adaptive methods."""
    n = features['n']
    
    # 1. Inverse distance (always computed)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Enhanced regret scores
    regret = _compute_enhanced_regret(distance_matrix, features)
    
    # 3. Cluster-aware scores
    cluster_scores = _compute_cluster_scores(distance_matrix, features)
    
    # 4. MST scores
    mst_scores = np.ones((n, n))
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            mst_scores[mst_present] = 2.0
        except:
            pass
    np.fill_diagonal(mst_scores, 0)
    
    return {
        'inv_dist': inv_dist,
        'regret': regret,
        'cluster': cluster_scores,
        'mst': mst_scores
    }


def _compute_enhanced_regret(distance_matrix: np.ndarray, features: dict) -> np.ndarray:
    """Compute regret scores considering multiple alternatives."""
    n = features['n']
    regret = np.ones((n, n))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if len(dists) < 3:
            continue
        
        # Consider top k alternatives
        k = min(5, n - 2)
        sorted_idx = np.argsort(dists)
        best_dist = dists[sorted_idx[0]]
        
        # Weighted regret considering multiple alternatives
        for j in range(n):
            if i == j:
                continue
                
            dist_ij = distance_matrix[i, j]
            
            # Find average of top k alternative distances
            alt_dists = []
            for idx in sorted_idx[:k]:
                if idx != j:
                    alt_dists.append(dists[idx])
            
            if alt_dists:
                # Use geometric mean for robustness
                alt_mean = np.exp(np.mean(np.log(np.array(alt_dists) + 1e-8)))
                regret_val = (alt_mean - dist_ij) / (alt_mean + 1e-8)
                regret[i, j] = 1.0 + max(0, regret_val)  # Boost if good
    
    # Symmetrize
    regret = (regret + regret.T) / 2
    np.fill_diagonal(regret, 0)
    return regret


def _compute_cluster_scores(distance_matrix: np.ndarray, features: dict) -> np.ndarray:
    """Compute scores based on cluster structure."""
    n = features['n']
    cluster_scores = np.ones((n, n))
    
    if features['cluster_tendency'] < 0.3:
        # Not strongly clustered, return uniform scores
        return cluster_scores
    
    # Simple clustering via distance threshold
    threshold = features['mean_dist'] * (1 + features['cv_dist'])
    adj_matrix = distance_matrix <= threshold
    
    # Compute connected components as clusters
    visited = np.zeros(n, dtype=bool)
    clusters = []
    
    for i in range(n):
        if not visited[i]:
            # BFS to find cluster
            stack = [i]
            cluster = []
            while stack:
                node = stack.pop()
                if not visited[node]:
                    visited[node] = True
                    cluster.append(node)
                    neighbors = np.where(adj_matrix[node])[0]
                    for neighbor in neighbors:
                        if not visited[neighbor]:
                            stack.append(neighbor)
            clusters.append(cluster)
    
    # Assign scores: boost intra-cluster edges, penalize inter-cluster
    for i in range(n):
        for j in range(i+1, n):
            # Find clusters
            i_cluster = -1
            j_cluster = -1
            for idx, cluster in enumerate(clusters):
                if i in cluster:
                    i_cluster = idx
                if j in cluster:
                    j_cluster = idx
            
            if i_cluster == j_cluster and i_cluster != -1:
                # Same cluster: boost
                cluster_scores[i, j] = cluster_scores[j, i] = 1.5
            elif i_cluster != j_cluster and i_cluster != -1 and j_cluster != -1:
                # Different clusters: penalize based on distance
                penalty = np.exp(-distance_matrix[i, j] / (features['mean_dist'] * 2))
                cluster_scores[i, j] = cluster_scores[j, i] = 0.5 + 0.5 * penalty
    
    return cluster_scores


def _ant_colony_simulation_lookahead(distance_matrix: np.ndarray, 
                                     features: dict, params: dict) -> np.ndarray:
    """Enhanced lookahead using ant colony simulation with adaptive strategies."""
    n = features['n']
    if n < 4:
        return np.ones((n, n))
    
    n_ants = params['n_ants']
    n_steps = params['lookahead_steps']
    alpha = params['alpha']
    beta = params['beta']
    
    # Initialize pheromone matrix
    pheromone = np.ones((n, n)) * 0.1
    np.fill_diagonal(pheromone, 0)
    
    # Heuristic information (inverse distance)
    heuristic = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(heuristic, 0)
    
    ant_scores = np.zeros((n, n))
    
    # Run ant simulations
    for ant in range(n_ants):
        # Each ant starts from a random node
        start = np.random.randint(n)
        current = start
        visited = np.zeros(n, dtype=bool)
        visited[current] = True
        
        # Keep track of edges used
        edges_used = []
        
        for step in range(n_steps):
            # Find unvisited nodes
            unvisited = np.where(~visited)[0]
            if len(unvisited) == 0:
                break
            
            # Compute probabilities for next node
            probs = np.zeros(len(unvisited))
            for idx, node in enumerate(unvisited):
                tau = pheromone[current, node] ** alpha
                eta = heuristic[current, node] ** beta
                probs[idx] = tau * eta
            
            if np.sum(probs) > 0:
                probs = probs / np.sum(probs)
                next_node = np.random.choice(unvisited, p=probs)
                
                # Record edge
                edges_used.append((current, next_node))
                
                # Move
                current = next_node
                visited[current] = True
            else:
                break
        
        # Update scores based on partial tour quality
        if edges_used:
            # Compute quality of this partial tour
            tour_length = sum(distance_matrix[i, j] for i, j in edges_used)
            expected_length = len(edges_used) * features['mean_dist']
            quality = expected_length / (tour_length + 1e-8)
            
            # Update scores for used edges
            for i, j in edges_used:
                ant_scores[i, j] += quality
                ant_scores[j, i] += quality
        
        # Evaporate pheromone
        pheromone *= (1 - params['evaporation'])
        
        # Deposit pheromone on used edges
        for i, j in edges_used:
            pheromone[i, j] += quality / (distance_matrix[i, j] + 1e-8)
            pheromone[j, i] = pheromone[i, j]
    
    # Normalize ant scores
    max_score = np.max(ant_scores)
    if max_score > 0:
        ant_scores = ant_scores / max_score
    
    np.fill_diagonal(ant_scores, 0)
    return ant_scores


def _compute_edge_betweenness(distance_matrix: np.ndarray, 
                              features: dict, params: dict) -> np.ndarray:
    """Compute approximate edge betweenness in k-nearest neighbor graph."""
    n = features['n']
    k = params['betweenness_k']
    n_samples = params['betweenness_samples']
    
    betweenness = np.zeros((n, n))
    
    if n < 4:
        return betweenness
    
    # Build k-nearest neighbor graph
    adj_list = {i: [] for i in range(n)}
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest = np.argpartition(dists, k)[:k]
        adj_list[i] = nearest.tolist()
    
    # Approximate betweenness by sampling pairs
    for _ in range(n_samples):
        s, t = np.random.choice(n, 2, replace=False)
        
        # Simple BFS to find shortest path in k-NN graph
        queue = [s]
        visited = {s}
        parent = {s: -1}
        
        found = False
        while queue and not found:
            current = queue.pop(0)
            for neighbor in adj_list[current]:
                if neighbor not in visited:
                    visited.add(neighbor)
                    parent[neighbor] = current
                    queue.append(neighbor)
                    
                    if neighbor == t:
                        found = True
                        break
        
        # If path found, update betweenness for edges on path
        if found:
            # Reconstruct path
            path = []
            node = t
            while node != -1:
                path.append(node)
                node = parent[node]
            path.reverse()
            
            # Update betweenness for edges in path
            for i in range(len(path) - 1):
                u, v = path[i], path[i+1]
                betweenness[u, v] += 1
                betweenness[v, u] += 1
    
    # Normalize betweenness scores
    max_between = np.max(betweenness)
    if max_between > 0:
        betweenness = betweenness / max_between
    
    # Boost scores by inverse distance
    betweenness = betweenness * (1 / (distance_matrix + np.eye(n) * 1e-8))
    
    np.fill_diagonal(betweenness, 0)
    return betweenness


def _ensemble_combination(base_scores: dict, ant_scores: np.ndarray,
                          betweenness_scores: np.ndarray, 
                          features: dict, params: dict) -> np.ndarray:
    """Combine all scores using adaptive ensemble weighting."""
    n = features['n']
    H = np.zeros((n, n))
    
    # Build candidate mask for efficiency
    candidate_mask = _build_candidate_mask_v2(base_scores['inv_dist'], features)
    
    # Extract weights
    weights = {k: params.get(k, 1.0) for k in 
              ['w_inv', 'w_regret', 'w_cluster', 'w_mst', 'w_ant', 'w_betweenness']}
    
    # Combine scores for candidate edges
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Use weighted geometric mean
                components = [
                    (base_scores['inv_dist'][i, j] + 1e-8) ** weights['w_inv'],
                    (base_scores['regret'][i, j] + 1e-8) ** weights['w_regret'],
                    (base_scores['cluster'][i, j] + 1e-8) ** weights['w_cluster'],
                    (base_scores['mst'][i, j] + 1e-8) ** weights['w_mst'],
                    (ant_scores[i, j] + 1e-8) ** weights['w_ant'],
                    (betweenness_scores[i, j] + 1e-8) ** weights['w_betweenness']
                ]
                
                # Use geometric mean
                H[i, j] = H[j, i] = np.exp(np.mean(np.log(np.array(components) + 1e-8)))
    
    return H


def _build_candidate_mask_v2(inv_dist_matrix: np.ndarray, features: dict) -> np.ndarray:
    """Build mask of candidate edges considering multiple criteria."""
    n = features['n']
    
    # Base candidate count
    base_k = min(15, max(3, int(np.log2(n) * 4)))
    
    # Adjust based on instance characteristics
    if features['cluster_tendency'] > 0.6:
        # Clustered instance: need more candidates for inter-cluster connections
        k = int(base_k * 1.5)
    elif features['regularity'] > 0.7:
        # Regular instance: can be more selective
        k = int(base_k * 0.8)
    else:
        k = base_k
    
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k-best edges by heuristic score
        scores = inv_dist_matrix[i].copy()
        scores[i] = -np.inf
        
        if k < n - 1:
            indices = np.argpartition(-scores, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Make symmetric
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask


def _post_process_v2(H: np.ndarray, distance_matrix: np.ndarray, 
                     params: dict) -> np.ndarray:
    """Post-process heuristic matrix with connectivity guarantees."""
    n = H.shape[0]
    
    # Ensure minimum connectivity
    min_edges = min(4, max(2, n // 5))
    for i in range(n):
        # Count non-zero entries
        non_zero = np.where(H[i] > 0)[0]
        if len(non_zero) < min_edges:
            # Add nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    base_score = 1 / (distance_matrix[i, j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    # Symmetrize
    H = (H + H.T) / 2
    
    # Apply softmax per row for better relative scoring
    for i in range(n):
        row = H[i]
        if np.sum(row) > 0:
            # Softmax with temperature
            temperature = 0.5  # Makes distribution sharper
            exp_row = np.exp(row / temperature)
            H[i] = exp_row / np.sum(exp_row)
    
    # Zero diagonal
    np.fill_diagonal(H, 0)
    
    return H
