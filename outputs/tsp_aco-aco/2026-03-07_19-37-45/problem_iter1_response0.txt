import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection combining multiple factors.
    
    Features:
    1. Distance-inverse weighting with normalization
    2. Nearest-neighbor bonuses (direct and second-order)
    3. Regional density scaling (clustering-aware)
    4. Geometric sparsification to zero out unpromising edges
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes
    
    Returns:
        Heuristic matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # Avoid division by zero and self-connections
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    safe_dist[safe_dist == 0] = np.finfo(float).eps
    
    # 1. Base inverse distance with normalization
    inv_dist = 1.0 / safe_dist
    inv_dist = inv_dist / np.max(inv_dist)  # Normalize to [0, 1]
    
    # 2. Nearest-neighbor bonuses
    nn_bonus = np.zeros_like(distance_matrix)
    for i in range(n):
        # Find nearest and second nearest neighbors
        sorted_idx = np.argsort(safe_dist[i])
        nearest = sorted_idx[0]
        second_nearest = sorted_idx[1] if n > 1 else nearest
        
        # Direct nearest neighbor bonus
        nn_bonus[i, nearest] += 0.5
        
        # Second-order neighbor bonus (edge between two nodes that share a common nearest neighbor)
        for j in range(n):
            if i != j:
                # If j is the nearest neighbor of i's nearest neighbor
                if nearest < n:
                    j_sorted = np.argsort(safe_dist[j])
                    if nearest in j_sorted[:max(3, n//4)]:  # Top few neighbors
                        nn_bonus[i, j] += 0.2
    
    # 3. Regional density scaling
    density_scale = np.ones_like(distance_matrix)
    
    # Compute local density for each node (inverse of average distance to k nearest neighbors)
    k = max(2, n // 5)  # Adaptive k based on problem size
    densities = np.zeros(n)
    for i in range(n):
        sorted_dists = np.sort(safe_dist[i])[:k]
        densities[i] = k / np.sum(sorted_dists) if np.sum(sorted_dists) > 0 else 0
    
    # Normalize densities
    if np.max(densities) > 0:
        densities = densities / np.max(densities)
    
    # Scale edges between dense nodes
    for i in range(n):
        for j in range(i+1, n):
            density_product = densities[i] * densities[j]
            density_scale[i, j] = 1.0 + 0.5 * density_product
            density_scale[j, i] = density_scale[i, j]
    
    # 4. Combine factors with weights
    weights = {
        'inv_dist': 1.0,
        'nn_bonus': 0.8,
        'density': 0.6
    }
    
    heuristic = (weights['inv_dist'] * inv_dist +
                 weights['nn_bonus'] * nn_bonus) * density_scale
    
    # Normalize to [0, 1] range
    if np.max(heuristic) > 0:
        heuristic = heuristic / np.max(heuristic)
    
    # 5. Geometric sparsification
    # Keep only top connections for each node based on geometric mean threshold
    sparsified = heuristic.copy()
    for i in range(n):
        row = heuristic[i]
        if np.max(row) > 0:
            # Use geometric mean of non-zero values as threshold
            non_zero = row[row > 0]
            if len(non_zero) > 0:
                geo_mean = np.exp(np.mean(np.log(non_zero)))
                threshold = 0.3 * geo_mean  # Keep values above 30% of geometric mean
                sparsified[i, row <= threshold] = 0
    
    # Ensure symmetry (since TSP edges are undirected in distance matrix)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Zero out diagonal (self-connections)
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
