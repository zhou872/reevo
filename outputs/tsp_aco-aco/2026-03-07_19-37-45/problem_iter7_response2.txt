import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with two-hop regret and adaptive instance-based weighting.
    Returns a matrix of the same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Instance analysis
    stats = _analyze_instance_v2(distance_matrix)
    
    # Compute factors
    base_factors = _compute_base_factors_v2(distance_matrix, stats)
    
    # Compute two-hop regret penalty
    regret_2hop = _compute_two_hop_regret(distance_matrix, stats)
    
    # Build candidate mask for efficiency
    candidate_mask = _build_candidate_mask_v2(distance_matrix, stats)
    
    # Get adaptive weights
    weights = _compute_adaptive_weights_v2(stats)
    
    # Combine factors with adaptive weighting
    H = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Multiplicative combination with adaptive exponents
                combined = (
                    (base_factors['inv_dist'][i, j] ** weights['w_inv']) *
                    (base_factors['regret_1hop'][i, j] ** weights['w_regret1']) *
                    (regret_2hop[i, j] ** weights['w_regret2']) *
                    (base_factors['density'][i, j] ** weights['w_density']) *
                    (base_factors['mst'][i, j] ** weights['w_mst'])
                )
                H[i, j] = H[j, i] = combined
    
    # Post-processing
    H = _post_process_heuristics_v2(H, distance_matrix, stats)
    
    return H

def _analyze_instance_v2(distance_matrix: np.ndarray) -> dict:
    """Analyze instance characteristics for adaptive parameter tuning."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal for statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    # Basic statistics
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Percentiles for distribution shape
    p25 = np.percentile(dist_no_diag, 25)
    p75 = np.percentile(dist_no_diag, 75)
    iqr = p75 - p25
    
    # Regularity measure (0=uniform, 1=clustered)
    if iqr > 0:
        regularity = min(1.0, (p75 - p25) / (mean_dist + 1e-8))
    else:
        regularity = 0.5
    
    # Coefficient of variation categories
    if cv_dist < 0.2:
        dist_type = 'uniform'
    elif cv_dist < 0.5:
        dist_type = 'moderate'
    else:
        dist_type = 'variable'
    
    stats = {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'dist_type': dist_type,
        'regularity': regularity,
        'sparsity': _estimate_sparsity_v2(distance_matrix),
        'cluster_tendency': _estimate_cluster_tendency_v2(distance_matrix)
    }
    
    return stats

def _estimate_sparsity_v2(distance_matrix: np.ndarray) -> float:
    """Estimate how sparse/structured the instance is."""
    n = distance_matrix.shape[0]
    if n < 4:
        return 0.5
    
    # Use nearest neighbor ratios
    nn_ratios = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)
        if len(sorted_dists) >= 3:
            nn_ratio = sorted_dists[1] / sorted_dists[0] if sorted_dists[0] > 0 else 1.0
            nn_ratios.append(nn_ratio)
    
    if nn_ratios:
        avg_ratio = np.mean(nn_ratios)
        # Convert to 0-1 scale (higher = more structured)
        sparsity = min(1.0, max(0.0, (avg_ratio - 1.0) / 2.0))
    else:
        sparsity = 0.5
    
    return sparsity

def _estimate_cluster_tendency_v2(distance_matrix: np.ndarray) -> float:
    """Estimate whether points form natural clusters."""
    n = distance_matrix.shape[0]
    
    if n < 10:
        return 0.5
    
    # Simple Hopkins-like statistic
    m = min(10, n // 3)
    sample_indices = np.random.choice(n, m, replace=False)
    
    sample_nn_dists = []
    for idx in sample_indices:
        dists = distance_matrix[idx]
        dists[idx] = np.inf
        sample_nn_dists.append(np.min(dists))
    
    # Generate random distances from the empirical distribution
    flat_dists = distance_matrix[~np.eye(n, dtype=bool)]
    random_dists = np.random.choice(flat_dists, m * (n-1), replace=True)
    random_nn_dists = []
    
    for i in range(m):
        start = i * (n-1)
        end = start + (n-1)
        random_nn_dists.append(np.min(random_dists[start:end]))
    
    hopkins = np.sum(sample_nn_dists) / (np.sum(sample_nn_dists) + np.sum(random_nn_dists) + 1e-8)
    return hopkins

def _compute_base_factors_v2(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute base heuristic factors with improved regret."""
    n = stats['n']
    
    factors = {}
    
    # 1. Inverse distance (normalized)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    # Normalize to [0, 1]
    max_inv = np.max(inv_dist)
    if max_inv > 0:
        inv_dist = inv_dist / max_inv
    factors['inv_dist'] = inv_dist
    
    # 2. One-hop regret
    factors['regret_1hop'] = _compute_one_hop_regret(distance_matrix)
    
    # 3. Density-based scoring
    factors['density'] = _compute_density_scores_v2(distance_matrix, stats)
    
    # 4. MST-based scoring
    factors['mst'] = _compute_mst_scores_v2(distance_matrix)
    
    return factors

def _compute_one_hop_regret(distance_matrix: np.ndarray) -> np.ndarray:

    regret = np.ones((n, n))
    
    for i in range(n):

                    regret[i, j] = 1 / (1 + regret_val)

    regret = (regret + regret.T) / 2
    np.fill_diagonal(regret, 0)
    
    # Normalize
    max_reg = np.max(regret)
    if max_reg > 0:
        regret = regret / max_reg
    
    return regret

def _compute_two_hop_regret(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute two-hop regret with variance-adaptive threshold."""
    n = stats['n']
    regret_2hop = np.ones((n, n))
    
    if n < 4:
        return regret_2hop
    
    # Variance-adaptive penalty threshold
    cv = stats['cv_dist']
    # For uniform instances (low cv), be stricter; for variable, be more lenient
    penalty_factor = 1.0 + 0.5 * cv  # Higher cv -> higher threshold
    
    # Precompute nearest neighbors for efficiency
    k_neighbors = min(10, n-2)
    
    for i in range(n):
        for j in range(i+1, n):
            # Skip if too expensive relative to mean
            if distance_matrix[i, j] > 3 * stats['mean_dist']:
                regret_2hop[i, j] = 0.1
                continue
            
            # Get candidate intermediate nodes
            candidates = []
            for k in range(n):
                if k != i and k != j:
                    candidates.append(k)
            
            if not candidates:
                continue
            
            # Find best two-hop alternative
            two_hop_costs = distance_matrix[i, candidates] + distance_matrix[candidates, j]
            min_two_hop = np.min(two_hop_costs)
            
            direct_cost = distance_matrix[i, j]
            if direct_cost > 0:
                regret = max(0, (min_two_hop - direct_cost) / direct_cost)
                
                # Apply variance-adaptive penalty threshold
                penalty_threshold = 0.3 * penalty_factor  # Base threshold scaled by cv
                if regret > penalty_threshold:
                    # Strong penalty for edges with much cheaper two-hop alternatives
                    regret_2hop[i, j] = max(0.1, 1.0 - (regret / (2 * penalty_threshold)))
                else:
                    # Mild or no penalty
                    regret_2hop[i, j] = 1.0 - (regret / (3 * penalty_threshold))
            else:
                regret_2hop[i, j] = 1.0
    
    # Symmetrize
    regret_2hop = (regret_2hop + regret_2hop.T) / 2
    np.fill_diagonal(regret_2hop, 0)
    
    return regret_2hop

def _compute_density_scores_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute density-based scores favoring edges in sparse regions."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Adaptive k based on instance size
    k = max(2, min(5, n // 10))
    
    # Estimate local density using k-nearest neighbors
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if len(dists) >= k:
            k_nearest = np.partition(dists, k)[:k]
            local_density[i] = np.mean(k_nearest)
        else:
            local_density[i] = np.mean(dists)
    
    # Normalize densities to z-scores
    if np.std(local_density) > 1e-8:
        local_density_z = (local_density - np.mean(local_density)) / np.std(local_density)
    else:
        local_density_z = np.zeros(n)
    
    # Create density scores: favor edges connecting sparse areas
    for i in range(n):
        for j in range(i+1, n):
            # Combined z-score (lower density = more sparse)
            combined_z = -(local_density_z[i] + local_density_z[j]) / 2.0
            # Convert to score in [0, 2] range
            density_scores[i, j] = density_scores[j, i] = np.exp(combined_z * 0.5)
    
    # Normalize
    max_density = np.max(density_scores)
    if max_density > 0:
        density_scores = density_scores / max_density
    
    return density_scores

def _compute_mst_scores_v2(distance_matrix: np.ndarray) -> np.ndarray:

    n = distance_matrix.shape[0]
    mst_scores = np.ones((n, n))
    
    if n > 2:

            mst = minimum_spanning_tree(distance_matrix)

            mst_present = (mst_dense + mst_dense.T) > 0
            
            # Boost edges in MST
            mst_scores[mst_present] = 1.5
            
            # Additional boost for critical edges (high degree in MST approximation)
            # Simple approximation: edges that appear in many MSTs of subgraphs
        except Exception as e:
            # Fallback: use distance-based approximation
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores

def _build_candidate_mask_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Build mask of candidate edges to consider (for efficiency)."""
    n = stats['n']
    
    # Adaptive candidate list size based on instance properties
    base_k = min(15, max(5, int(np.log2(n) * 3)))
    
    # Adjust based on sparsity and cluster tendency
    # More structured instances can use fewer candidates
    k = int(base_k * (1.0 - 0.3 * stats['sparsity'] - 0.2 * stats['cluster_tendency']))
    k = max(3, min(k, n - 1))
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k-nearest neighbors
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < len(dists):
            # Get indices of k smallest distances
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Make symmetric and ensure reciprocity
    candidate_mask = candidate_mask | candidate_mask.T
    
    # Ensure each node has at least min_edges candidates
    min_edges = max(3, min(5, n // 4))
    for i in range(n):
        if np.sum(candidate_mask[i]) < min_edges:
            # Add more nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            additional = min(min_edges, n-1) - np.sum(candidate_mask[i])
            if additional > 0:
                # Get indices sorted by distance
                sorted_indices = np.argsort(dists)
                for idx in sorted_indices:
                    if not candidate_mask[i, idx] and idx != i:
                        candidate_mask[i, idx] = True
                        candidate_mask[idx, i] = True
                        additional -= 1
                        if additional <= 0:
                            break
    
    np.fill_diagonal(candidate_mask, False)
    return candidate_mask

def _compute_adaptive_weights_v2(stats: dict) -> dict:
    """Dynamically compute weights based on instance characteristics."""
    n = stats['n']
    cv = stats['cv_dist']
    regularity = stats['regularity']
    sparsity = stats['sparsity']
    cluster_tendency = stats['cluster_tendency']
    
    # Base weights that sum to approximately 4 for responsiveness
    weights = {
        'w_inv': 1.0,  # Always important
        
        # One-hop regret: more important for regular instances
        'w_regret1': 0.5 + 0.3 * regularity,
        
        # Two-hop regret: more important for sparse/structured instances
        'w_regret2': 0.3 + 0.4 * sparsity,
        
        # Density: more important for clustered instances
        'w_density': 0.2 + 0.5 * cluster_tendency,
        
        # MST: more important for larger instances
        'w_mst': max(0.1, 0.8 - 0.02 * n)
    }
    
    # Adjust based on coefficient of variation
    if cv < 0.2:  # Uniform distances
        weights['w_regret1'] *= 1.2
        weights['w_regret2'] *= 0.8
    elif cv > 0.5:  # Highly variable distances
        weights['w_regret1'] *= 0.8
        weights['w_regret2'] *= 1.2
        weights['w_density'] *= 1.1
    
    # Normalize to maintain overall scale (sum ≈ 4)
    current_sum = sum(weights.values())
    target_sum = 4.0
    if current_sum > 0:
        scale_factor = target_sum / current_sum
        for key in weights:
            weights[key] *= scale_factor
    
    return weights

def _post_process_heuristics_v2(H: np.ndarray, distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Post-process the heuristic matrix."""
    n = stats['n']
    
    # Ensure symmetry and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Apply soft threshold to sparsify: set very low values to 0
    threshold = np.percentile(H[H > 0], 20) if np.any(H > 0) else 0.01
    H[H < threshold] = 0
    
    # Ensure each node has at least min_connections non-zero entries
    min_connections = max(2, min(4, n // 6))
    for i in range(n):
        non_zero = np.where(H[i] > 0)[0]
        if len(non_zero) < min_connections:
            # Add nearest neighbors with basic inverse distance score
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            # Sort by distance
            sorted_indices = np.argsort(dists)
            added = 0
            for j in sorted_indices:
                if j != i and H[i, j] == 0:
                    H[i, j] = H[j, i] = 1 / (distance_matrix[i, j] + 1e-8)
                    added += 1
                    if added >= min_connections - len(non_zero):
                        break
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
