import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Dict, Tuple
import warnings
warnings.filterwarnings('ignore')

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with community-aware scoring and meta-adaptive parameters.
    
    Key innovations:
    1. Community detection via modularity optimization for cluster-aware scoring
    2. Meta-learning inspired parameter adaptation using instance fingerprinting
    3. Three-tier edge viability assessment: local, community, and global
    4. Adaptive graph sparsification with connectivity preservation
    
    Args:
        distance_matrix: Symmetric matrix of distances between nodes.
        
    Returns:
        Heuristic matrix of same shape indicating edge desirability.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    eps = 1e-12
    
    # 1. Instance fingerprinting for meta-adaptive parameters
    instance_stats = compute_instance_fingerprint(distance_matrix)
    
    # 2. Community detection via modularity optimization
    community_labels = detect_communities(distance_matrix, instance_stats)
    
    # 3. Multi-scale distance normalization
    local_scale, community_scale, global_scale = compute_distance_scales(
        distance_matrix, community_labels, instance_stats)
    
    # 4. Enhanced inverse distance with scale adaptation
    norm_dist = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            if community_labels[i] == community_labels[j]:
                scale = local_scale[community_labels[i]]
            else:
                ci, cj = community_labels[i], community_labels[j]
                scale = community_scale.get((ci, cj), global_scale)
            norm_dist[i, j] = norm_dist[j, i] = distance_matrix[i, j] / (scale + eps)
    
    inv_dist = 1 / (norm_dist + eps)
    np.fill_diagonal(inv_dist, 0)
    
    # 5. Community-aware regret computation
    intra_regret, inter_regret = compute_community_regret(
        distance_matrix, community_labels, instance_stats)
    
    # 6. Multi-hop viability assessment
    viability = assess_edge_viability(
        distance_matrix, community_labels, instance_stats)
    
    # 7. Community connectivity bonus
    connectivity_bonus = compute_connectivity_bonus(
        distance_matrix, community_labels, instance_stats)
    
    # 8. Adaptive parameter weighting based on instance fingerprint
    weights = compute_adaptive_weights(instance_stats, community_labels)
    
    # 9. Combined scoring with meta-adaptive weighting
    heuristics = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(i + 1, n):
            # Base components
            base_score = inv_dist[i, j] ** weights['w_inv']
            
            # Regret component
            if community_labels[i] == community_labels[j]:
                regret_score = intra_regret[i, j] ** weights['w_regret_intra']
            else:
                regret_score = inter_regret[i, j] ** weights['w_regret_inter']
            
            # Viability component
            viability_score = viability[i, j] ** weights['w_viability']
            
            # Connectivity bonus component
            conn_score = connectivity_bonus[i, j] ** weights['w_connectivity']
            
            # Combine with geometric mean
            heuristics[i, j] = heuristics[j, i] = (
                base_score * regret_score * viability_score * conn_score
            ) ** 0.25
    
    # 10. Community-aware sparsification
    sparsified = community_aware_sparsification(
        heuristics, distance_matrix, community_labels, instance_stats)
    
    # 11. MST connectivity guarantee with community awareness
    final_heuristics = ensure_connectivity(
        sparsified, distance_matrix, community_labels, instance_stats)
    
    # 12. Final normalization and symmetry
    max_val = np.max(final_heuristics)
    if max_val > eps:
        final_heuristics = final_heuristics / max_val
    
    # Apply sigmoid to emphasize differences
    final_heuristics = 1 / (1 + np.exp(-6 * (final_heuristics - 0.5)))
    
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics


def compute_instance_fingerprint(distance_matrix: np.ndarray) -> Dict:
    """Compute comprehensive instance characteristics for parameter adaptation."""
    n = distance_matrix.shape[0]
    eps = 1e-12
    
    # Remove diagonal for statistics
    mask = ~np.eye(n, dtype=bool)
    flat_dists = distance_matrix[mask]
    
    # Basic statistics
    mean_dist = np.mean(flat_dists)
    std_dist = np.std(flat_dists)
    cv_dist = std_dist / (mean_dist + eps)
    
    # Distance distribution percentiles
    percentiles = np.percentile(flat_dists, [10, 25, 50, 75, 90])
    
    # Spatial regularity metrics
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    
    nn_cv = np.std(nn_dists) / (np.mean(nn_dists) + eps)
    
    # Triangle inequality violations (approximate)
    violations = 0
    samples = min(500, n * (n - 1) // 2)
    for _ in range(samples):
        i, j, k = np.random.choice(n, 3, replace=False)
        if (distance_matrix[i, j] > 
            distance_matrix[i, k] + distance_matrix[k, j] * 1.1):
            violations += 1
    
    triangle_violation = violations / samples
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'percentiles': percentiles,
        'nn_cv': nn_cv,
        'triangle_violation': triangle_violation,
        'log_n': np.log(n + 1),
        'sparsity': np.log(mean_dist / (percentiles[0] + eps))
    }


def detect_communities(distance_matrix: np.ndarray, 
                      stats: Dict) -> np.ndarray:
    """Detect communities using modularity optimization on distance-based graph."""
    n = stats['n']
    
    # Build adjacency matrix with adaptive thresholding
    adj_threshold = stats['percentiles'][2]  # Median distance
    adjacency = np.zeros((n, n))
    
    for i in range(n):
        for j in range(i + 1, n):
            if distance_matrix[i, j] <= adj_threshold:
                adjacency[i, j] = adjacency[j, i] = 1
    
    # Simple label propagation for community detection
    labels = np.arange(n)
    
    # Number of iterations based on instance size
    max_iter = min(20, int(np.log2(n)) * 2)
    
    for _ in range(max_iter):
        changed = False
        order = np.random.permutation(n)
        
        for i in order:
            # Find neighboring labels
            neighbor_labels = labels[adjacency[i] > 0]
            if len(neighbor_labels) == 0:
                continue
            
            # Count label frequencies
            unique_labels, counts = np.unique(neighbor_labels, return_counts=True)
            
            # Choose most frequent neighbor label
            new_label = unique_labels[np.argmax(counts)]
            
            if labels[i] != new_label:
                labels[i] = new_label
                changed = True
        
        if not changed:
            break
    
    # Re-label for compactness
    unique_labels = np.unique(labels)
    relabel_map = {old: new for new, old in enumerate(unique_labels)}
    return np.array([relabel_map[l] for l in labels])


def compute_distance_scales(distance_matrix: np.ndarray,
                           community_labels: np.ndarray,
                           stats: Dict) -> Tuple[Dict, Dict, float]:
    """Compute distance scales at local, community, and global levels."""
    n = stats['n']
    eps = 1e-12
    n_communities = len(np.unique(community_labels))
    
    # Local scale (within communities)
    local_scale = {}
    for c in range(n_communities):
        nodes = np.where(community_labels == c)[0]
        if len(nodes) < 2:
            local_scale[c] = stats['mean_dist']
            continue
        
        # Extract intra-community distances
        intra_dists = []
        for i in range(len(nodes)):
            for j in range(i + 1, len(nodes)):
                intra_dists.append(distance_matrix[nodes[i], nodes[j]])
        
        local_scale[c] = np.median(intra_dists) if intra_dists else stats['mean_dist']
    
    # Community-to-community scale
    community_scale = {}
    for c1 in range(n_communities):
        nodes1 = np.where(community_labels == c1)[0]
        for c2 in range(c1 + 1, n_communities):
            nodes2 = np.where(community_labels == c2)[0]
            
            # Sample inter-community distances
            inter_dists = []
            max_samples = min(20, len(nodes1) * len(nodes2))
            for _ in range(max_samples):
                i = np.random.choice(nodes1)
                j = np.random.choice(nodes2)
                inter_dists.append(distance_matrix[i, j])
            
            community_scale[(c1, c2)] = np.median(inter_dists)
            community_scale[(c2, c1)] = community_scale[(c1, c2)]
    
    # Global scale
    global_scale = stats['percentiles'][3]  # 75th percentile
    
    return local_scale, community_scale, global_scale


def compute_community_regret(distance_matrix: np.ndarray,
                            community_labels: np.ndarray,
                            stats: Dict) -> Tuple[np.ndarray, np.ndarray]:
    """Compute regret scores considering both intra and inter-community alternatives."""
    n = stats['n']
    eps = 1e-12
    intra_regret = np.ones((n, n))
    inter_regret = np.ones((n, n))
    
    # Intra-community regret
    for i in range(n):
        ci = community_labels[i]
        same_comm = np.where(community_labels == ci)[0]
        
        if len(same_comm) < 2:
            continue
            
        # Distances within community
        comm_dists = distance_matrix[i, same_comm]
        comm_dists[same_comm == i] = np.inf
        
        if np.all(comm_dists == np.inf):
            continue
            
        best_intra = np.min(comm_dists[comm_dists > 0])
        
        for j in same_comm:
            if i == j:
                continue
                
            regret_val = 1.0 / (1.0 + (distance_matrix[i, j] - best_intra) / (best_intra + eps))
            intra_regret[i, j] = max(0.2, regret_val)
    
    # Inter-community regret
    for i in range(n):
        ci = community_labels[i]
        other_comms = np.unique(community_labels)
        other_comms = other_comms[other_comms != ci]
        
        # Find best connection to each other community
        best_to_comm = {}
        for cj in other_comms:
            nodes_j = np.where(community_labels == cj)[0]
            if len(nodes_j) == 0:
                continue
            best_to_comm[cj] = np.min(distance_matrix[i, nodes_j])
        
        for j in range(n):
            if i == j or community_labels[j] == ci:
                continue
                
            cj = community_labels[j]
            if cj not in best_to_comm:
                continue
                
            best_inter = best_to_comm[cj]
            regret_val = 1.0 / (1.0 + (distance_matrix[i, j] - best_inter) / (best_inter + eps))
            inter_regret[i, j] = max(0.2, regret_val)
    
    # Symmetrize
    intra_regret = (intra_regret + intra_regret.T) / 2
    inter_regret = (inter_regret + inter_regret.T) / 2
    
    return intra_regret, inter_regret


def assess_edge_viability(distance_matrix: np.ndarray,
                         community_labels: np.ndarray,
                         stats: Dict) -> np.ndarray:
    """Assess edge viability through multi-hop path analysis."""
    n = stats['n']
    viability = np.ones((n, n))
    
    # Adaptive depth based on instance size
    if n <= 50:
        depth = 2
    elif n <= 200:
        depth = 3
    else:
        depth = 4
    
    # For small instances, compute exact multi-hop
    if n <= 100:
        for i in range(n):
            for j in range(i + 1, n):
                direct = distance_matrix[i, j]
                best_multi = np.inf
                
                # Explore k-hop paths
                for k in range(n):
                    if k == i or k == j:
                        continue
                    
                    path_dist = distance_matrix[i, k] + distance_matrix[k, j]
                    if depth >= 3:
                        for l in range(n):
                            if l == i or l == j or l == k:
                                continue
                            path_dist = min(path_dist,
                                          distance_matrix[i, k] + 
                                          distance_matrix[k, l] + 
                                          distance_matrix[l, j])
                    
                    best_multi = min(best_multi, path_dist)
                
                if best_multi < direct * 0.9:
                    viability[i, j] = viability[j, i] = 0.6
                elif best_multi < direct * 0.95:
                    viability[i, j] = viability[j, i] = 0.8
    else:
        # For larger instances, use sampling
        for i in range(n):
            # Sample potential intermediate nodes
            sample_size = min(20, n // 10)
            candidates = np.random.choice([x for x in range(n) if x != i], 
                                         min(sample_size, n-1), replace=False)
            
            for j in range(i + 1, n):
                if j in candidates:
                    continue
                    
                direct = distance_matrix[i, j]
                best_multi = np.inf
                
                # Check paths through candidate nodes
                for k in candidates:
                    if k == j:
                        continue
                    
                    # 2-hop
                    path_2 = distance_matrix[i, k] + distance_matrix[k, j]
                    best_multi = min(best_multi, path_2)
                    
                    # 3-hop (sample another intermediate)
                    if depth >= 3 and len(candidates) > 1:
                        for l in candidates:
                            if l == k or l == j:
                                continue
                            path_3 = (distance_matrix[i, k] + 
                                     distance_matrix[k, l] + 
                                     distance_matrix[l, j])
                            best_multi = min(best_multi, path_3)
                
                if best_multi < np.inf:
                    if best_multi < direct * 0.9:
                        viability[i, j] = viability[j, i] = 0.6
                    elif best_multi < direct * 0.95:
                        viability[i, j] = viability[j, i] = 0.8
    
    return viability


def compute_connectivity_bonus(distance_matrix: np.ndarray,
                              community_labels: np.ndarray,
                              stats: Dict) -> np.ndarray:
    """Compute bonus for edges that maintain or improve connectivity."""
    n = stats['n']
    bonus = np.ones((n, n))
    n_communities = len(np.unique(community_labels))
    
    # Compute MST for connectivity baseline
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # Intra-community connectivity bonus
    for c in range(n_communities):
        nodes = np.where(community_labels == c)[0]
        if len(nodes) <= 1:
            continue
        
        # Bonus for MST edges within community
        for i in nodes:
            for j in nodes:
                if i >= j:
                    continue
                if mst_mask[i, j]:
                    bonus[i, j] = bonus[j, i] = 1.5
    
    # Inter-community connectivity bonus (bridges)
    if n_communities > 1:
        # Find minimum spanning tree between communities
        comm_centers = []
        for c in range(n_communities):
            nodes = np.where(community_labels == c)[0]
            # Use median node as representative
            if len(nodes) > 0:
                comm_centers.append(nodes[len(nodes)//2])
        
        # Build distance matrix between community centers
        comm_dist = np.zeros((n_communities, n_communities))
        for i in range(n_communities):
            for j in range(i + 1, n_communities):
                comm_dist[i, j] = comm_dist[j, i] = distance_matrix[
                    comm_centers[i], comm_centers[j]]
        
        # MST between communities
        comm_mst = minimum_spanning_tree(comm_dist).toarray()
        
        # Bonus for edges that connect communities in the community MST
        for i in range(n_communities):
            for j in range(i + 1, n_communities):
                if comm_mst[i, j] > 0:
                    # Find closest nodes between these communities
                    nodes_i = np.where(community_labels == i)[0]
                    nodes_j = np.where(community_labels == j)[0]
                    
                    if len(nodes_i) == 0 or len(nodes_j) == 0:
                        continue
                    
                    # Find minimum distance edge between communities
                    min_dist = np.inf
                    best_pair = (nodes_i[0], nodes_j[0])
                    
                    for ii in nodes_i[:min(5, len(nodes_i))]:
                        for jj in nodes_j[:min(5, len(nodes_j))]:
                            if distance_matrix[ii, jj] < min_dist:
                                min_dist = distance_matrix[ii, jj]
                                best_pair = (ii, jj)
                    
                    bonus[best_pair] = bonus[best_pair[1], best_pair[0]] = 1.3
    
    return bonus


def compute_adaptive_weights(stats: Dict, 
                            community_labels: np.ndarray) -> Dict:
    """Compute adaptive weights based on instance fingerprint and community structure."""
    n_communities = len(np.unique(community_labels))
    
    # Meta-learning inspired parameter adaptation
    weights = {
        'w_inv': 1.0,
        'w_regret_intra': 0.8 + 0.4 * (1 - stats['triangle_violation']),
        'w_regret_inter': 0.7 + 0.3 * stats['cv_dist'],
        'w_viability': 0.6 + 0.3 * min(1.0, stats['log_n'] / 5),
        'w_connectivity': 0.5 + 0.3 * (n_communities / max(stats['n'] / 20, 5))
    }
    
    # Normalize weights to maintain balance
    total = sum(weights.values())
    target_total = 4.0
    for key in weights:
        weights[key] = weights[key] * target_total / total
    
    return weights


def community_aware_sparsification(heuristics: np.ndarray,
                                  distance_matrix: np.ndarray,
                                  community_labels: np.ndarray,
                                  stats: Dict) -> np.ndarray:
    """Sparsify heuristic matrix while preserving community structure."""
    n = stats['n']
    n_communities = len(np.unique(community_labels))
    sparsified = np.zeros_like(heuristics)
    
    # Adaptive sparsity parameters
    if n <= 50:
        intra_keep = 4
        inter_keep = 3
    elif n <= 200:
        intra_keep = max(3, int(np.log2(n)))
        inter_keep = max(2, int(np.log2(n)) - 1)
    else:
        intra_keep = max(2, int(np.log2(n)) - 1)
        inter_keep = max(1, int(np.log2(n)) - 2)
    
    # Keep best intra-community edges
    for c in range(n_communities):
        nodes = np.where(community_labels == c)[0]
        if len(nodes) < 2:
            continue
        
        # For each node in community, keep top intra_keep edges within community
        for i in nodes:
            # Get heuristic scores to other nodes in same community
            comm_heuristics = heuristics[i, nodes]
            comm_heuristics[nodes == i] = -np.inf
            
            if len(nodes) > intra_keep:
                top_indices = np.argpartition(comm_heuristics, -intra_keep)[-intra_keep:]
                top_nodes = nodes[top_indices]
            else:
                top_nodes = nodes[nodes != i]
            
            for j in top_nodes:
                sparsified[i, j] = heuristics[i, j]
    
    # Keep best inter-community edges
    for i in range(n):
        ci = community_labels[i]
        other_nodes = np.where(community_labels != ci)[0]
        
        if len(other_nodes) == 0:
            continue
        
        # Get heuristic scores to other communities
        inter_heuristics = heuristics[i, other_nodes]
        
        if len(other_nodes) > inter_keep:
            top_indices = np.argpartition(inter_heuristics, -inter_keep)[-inter_keep:]
            top_nodes = other_nodes[top_indices]
        else:
            top_nodes = other_nodes
        
        for j in top_nodes:
            sparsified[i, j] = heuristics[i, j]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified


def ensure_connectivity(heuristics: np.ndarray,
                       distance_matrix: np.ndarray,
                       community_labels: np.ndarray,
                       stats: Dict) -> np.ndarray:
    """Ensure connectivity within and between communities."""
    n = stats['n']
    result = heuristics.copy()
    
    # Check connectivity within each community
    for c in range(len(np.unique(community_labels))):
        nodes = np.where(community_labels == c)[0]
        if len(nodes) < 2:
            continue
        
        # Build adjacency from current heuristics
        adj = (result[nodes][:, nodes] > 0).astype(int)
        
        # Check if community is connected
        visited = np.zeros(len(nodes), dtype=bool)
        stack = [0]
        visited[0] = True
        
        while stack:
            current = stack.pop()
            neighbors = np.where(adj[current] > 0)[0]
            for neighbor in neighbors:
                if not visited[neighbor]:
                    visited[neighbor] = True
                    stack.append(neighbor)
        
        # If not all nodes visited, add missing connections
        if not np.all(visited):
            missing = np.where(~visited)[0]
            for m in missing:
                # Connect to nearest connected node
                connected = np.where(visited)[0]
                if len(connected) > 0:
                    # Find minimum distance to connected nodes
                    min_dist = np.inf
                    best_j = connected[0]
                    
                    for j in connected:
                        dist = distance_matrix[nodes[m], nodes[j]]
                        if dist < min_dist:
                            min_dist = dist
                            best_j = j
                    
                    # Add connection with appropriate heuristic value
                    heuristic_val = 1.0 / (min_dist / stats['mean_dist'] + 1e-12)
                    result[nodes[m], nodes[best_j]] = max(
                        result[nodes[m], nodes[best_j]], heuristic_val)
                    result[nodes[best_j], nodes[m]] = result[nodes[m], nodes[best_j]]
    
    # Ensure inter-community connectivity
    n_communities = len(np.unique(community_labels))
    if n_communities > 1:
        # Build community connectivity graph
        comm_adj = np.zeros((n_communities, n_communities))
        for i in range(n):
            ci = community_labels[i]
            for j in range(i + 1, n):
                cj = community_labels[j]
                if result[i, j] > 0:
                    comm_adj[ci, cj] = comm_adj[cj, ci] = 1
        
        # Check if community graph is connected
        visited = np.zeros(n_communities, dtype=bool)
        stack = [0]
        visited[0] = True
        
        while stack:
            current = stack.pop()
            neighbors = np.where(comm_adj[current] > 0)[0]
            for neighbor in neighbors:
                if not visited[neighbor]:
                    visited[neighbor] = True
                    stack.append(neighbor)
        
        # Add connections between disconnected communities
        if not np.all(visited):
            disconnected = np.where(~visited)[0]
            for dc in disconnected:
                # Find closest connected community
                connected = np.where(visited)[0]
                min_dist = np.inf
                best_pair = (0, 0)
                
                for cc in connected:
                    nodes_dc = np.where(community_labels == dc)[0]
                    nodes_cc = np.where(community_labels == cc)[0]
                    
                    # Find minimum distance between communities
                    for i in nodes_dc[:min(5, len(nodes_dc))]:
                        for j in nodes_cc[:min(5, len(nodes_cc))]:
                            dist = distance_matrix[i, j]
                            if dist < min_dist:
                                min_dist = dist
                                best_pair = (i, j)
                
                # Add connection
                heuristic_val = 1.0 / (min_dist / stats['mean_dist'] + 1e-12)
                result[best_pair] = result[best_pair[1], best_pair[0]] = max(
                    result[best_pair], heuristic_val)
    
    return result
