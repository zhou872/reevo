import numpy as np
from typing import Tuple, List
from dataclasses import dataclass

@dataclass
class Edge:
    u: int
    v: int
    weight: float

class UnionFind:
    """Union-Find data structure for Kruskal's MST algorithm"""
    def __init__(self, n: int):
        self.parent = list(range(n))
        self.rank = [0] * n
    
    def find(self, x: int) -> int:
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]
    
    def union(self, x: int, y: int) -> bool:
        root_x = self.find(x)
        root_y = self.find(y)
        if root_x == root_y:
            return False
        
        if self.rank[root_x] < self.rank[root_y]:
            self.parent[root_x] = root_y
        elif self.rank[root_x] > self.rank[root_y]:
            self.parent[root_y] = root_x
        else:
            self.parent[root_y] = root_x
            self.rank[root_x] += 1
        return True

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection with regret-based scoring,
    efficient MST calculation, and adaptive sparsification.
    
    Features:
    1. Regret-weighted edge scoring (penalizes edges with small regret between nearest neighbors)
    2. Kruskal's MST algorithm with union-find for efficiency
    3. Cluster density adjustment (edges in dense regions get bonus)
    4. Two-stage adaptive sparsification with connectivity preservation
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Base reciprocal with epsilon to avoid division by zero
    eps = 1e-10
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (distance_matrix + eps)
    np.fill_diagonal(reciprocal, 0)
    
    # 2. Regret-based scoring
    # For each node, compute regret as difference between 2nd and 1st nearest distances
    regret_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get distances to all other nodes
        row = distance_matrix[i].copy()
        row[i] = np.inf  # Exclude self
        
        # Find two smallest distances
        smallest_idx = np.argpartition(row, 2)[:2]
        smallest_distances = row[smallest_idx]
        
        if len(smallest_distances) >= 2:
            d1, d2 = np.sort(smallest_distances)[:2]
            regret = max(d2 - d1, eps)  # Regret is difference between 2nd and 1st
            
            # Apply regret weighting: edges with larger regret get higher score
            # Normalize regret by average distance for stability
            avg_dist = np.mean(row[row < np.inf])
            regret_weight = 1.0 + 0.5 * (regret / (avg_dist + eps))
            
            # Apply to the best edge
            best_neighbor = smallest_idx[0]
            regret_matrix[i, best_neighbor] = regret_weight
    
    # Make regret matrix symmetric
    regret_matrix = np.maximum(regret_matrix, regret_matrix.T)
    
    # 3. Efficient MST calculation using Kruskal's algorithm
    def kruskal_mst(dist_mat: np.ndarray) -> np.ndarray:
        """Return MST edge mask using Kruskal's algorithm with union-find"""
        n_nodes = dist_mat.shape[0]
        edges = []
        
        # Create list of all edges (only upper triangular to avoid duplicates)
        for i in range(n_nodes):
            for j in range(i + 1, n_nodes):
                edges.append(Edge(i, j, dist_mat[i, j]))
        
        # Sort edges by weight
        edges.sort(key=lambda e: e.weight)
        
        # Initialize union-find and MST mask
        uf = UnionFind(n_nodes)
        mst_mask = np.zeros_like(dist_mat, dtype=bool)
        
        # Build MST
        for edge in edges:
            if uf.union(edge.u, edge.v):
                mst_mask[edge.u, edge.v] = True
                mst_mask[edge.v, edge.u] = True
                
                # Early stop if we have n-1 edges
                if np.sum(mst_mask) // 2 == n_nodes - 1:
                    break
        
        return mst_mask
    
    mst_mask = kruskal_mst(distance_matrix)
    
    # 4. Cluster density adjustment
    # Identify dense clusters by looking at local neighborhood statistics
    density_adjustment = np.ones_like(distance_matrix)
    
    # For each node, compute average distance to k nearest neighbors
    k_cluster = max(2, min(10, n // 5))  # Adaptive k based on problem size
    avg_local_dist = np.zeros(n)
    
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        k_smallest = np.partition(row, k_cluster)[:k_cluster]
        avg_local_dist[i] = np.mean(k_smallest)
    
    # Normalize local densities
    if np.max(avg_local_dist) > eps:
        normalized_density = avg_local_dist / np.max(avg_local_dist)
    else:
        normalized_density = np.ones(n)
    
    # Adjust scores based on cluster density
    # Edges within dense regions get bonus, edges between sparse and dense get penalty
    for i in range(n):
        for j in range(i + 1, n):
            density_i = normalized_density[i]
            density_j = normalized_density[j]
            
            # Bonus if both nodes are in dense regions
            if density_i < 0.5 and density_j < 0.5:
                density_adjustment[i, j] = 1.2
                density_adjustment[j, i] = 1.2
            # Penalty if connecting very dense to very sparse
            elif (density_i < 0.3 and density_j > 0.7) or (density_i > 0.7 and density_j < 0.3):
                density_adjustment[i, j] = 0.7
                density_adjustment[j, i] = 0.7
    
    # 5. Combine factors with adaptive weights
    base_heuristic = reciprocal.copy()
    
    # Apply regret weighting
    base_heuristic *= (0.7 + 0.3 * regret_matrix)
    
    # Apply MST bonus (higher bonus for larger problems)
    mst_bonus = 2.0 + 0.1 * min(10, n // 10)  # Scale with problem size
    base_heuristic[mst_mask] *= mst_bonus
    
    # Apply density adjustment
    base_heuristic *= density_adjustment
    
    # 6. Two-stage adaptive sparsification
    # Stage 1: Keep top connections per node with adaptive count
    m1 = max(2, min(15, n // 3))  # Initial sparsification
    stage1_mask = np.zeros_like(base_heuristic, dtype=bool)
    
    for i in range(n):
        row = base_heuristic[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get indices of top m1 values
        top_indices = np.argpartition(row, -m1)[-m1:]
        stage1_mask[i, top_indices] = True
    
    # Make symmetric
    stage1_mask = stage1_mask | stage1_mask.T
    
    # Stage 2: Further sparsify but ensure connectivity
    # Keep MST edges and additional promising edges
    m2 = max(2, min(8, n // 5))  # Final target connections per node
    final_heuristic = np.zeros_like(base_heuristic)
    
    for i in range(n):
        # Get candidate edges from stage 1
        candidates = np.where(stage1_mask[i])[0]
        candidate_scores = base_heuristic[i, candidates]
        
        # Always keep MST edges
        mst_edges = np.where(mst_mask[i])[0]
        
        # Sort candidates by score (excluding MST edges we already keep)
        non_mst_mask = ~np.isin(candidates, mst_edges)
        non_mst_candidates = candidates[non_mst_mask]
        non_mst_scores = candidate_scores[non_mst_mask]
        
        # Select top additional edges
        num_additional = max(0, m2 - len(mst_edges))
        if len(non_mst_candidates) > 0:
            top_additional_idx = np.argsort(non_mst_scores)[-num_additional:]
            selected_non_mst = non_mst_candidates[top_additional_idx]
        else:
            selected_non_mst = []
        
        # Combine MST and selected edges
        selected = np.concatenate([mst_edges, selected_non_mst])
        final_heuristic[i, selected] = base_heuristic[i, selected]
    
    # Ensure symmetry
    final_heuristic = np.maximum(final_heuristic, final_heuristic.T)
    
    # 7. Final normalization and zero-out weak edges
    max_val = np.max(final_heuristic)
    if max_val > eps:
        final_heuristic = final_heuristic / max_val
    
    # Zero out very weak edges (below threshold)
    threshold = 0.05 * np.max(final_heuristic) if np.max(final_heuristic) > 0 else 0
    final_heuristic[final_heuristic < threshold] = 0
    
    return final_heuristic
