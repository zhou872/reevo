import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection combining multiple factors.
    
    Features:
    1. Inverse distance (base factor)
    2. Nearest neighbor affinity (edges connecting mutual nearest neighbors get bonus)
    3. Two-hop penalty (edges that bypass good intermediate nodes get penalty)
    4. Sparsification (unpromising edges set to zero)
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (handling zeros to avoid division by zero)
    eps = 1e-10
    inv_dist = 1 / (distance_matrix + eps)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Nearest neighbor affinity
    k = max(2, int(np.ceil(0.1 * n)))  # Consider top 10% as nearest neighbors
    nn_affinity = np.zeros((n, n))
    
    # Find k nearest neighbors for each node
    for i in range(n):
        # Get indices of k+1 smallest distances (including self)
        nearest = np.argpartition(distance_matrix[i], k+1)[:k+1]
        nearest = nearest[nearest != i]  # Remove self
        
        # Check mutual nearest neighbors
        for j in nearest:
            # Check if j is among i's nearest neighbors
            j_nearest = np.argpartition(distance_matrix[j], k+1)[:k+1]
            if i in j_nearest:
                # Mutual nearest neighbor bonus
                nn_affinity[i, j] = 1.5
            else:
                # One-way nearest neighbor bonus
                nn_affinity[i, j] = 1.2
    
    # 3. Two-hop lookahead penalty
    # For edge (i,j), check if there exists a node k such that
    # d(i,k) + d(k,j) < 1.2 * d(i,j) - if yes, penalize direct edge
    penalty_factor = np.ones((n, n))
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            direct_dist = distance_matrix[i, j]
            best_two_hop = float('inf')
            
            # Find best two-hop path through any intermediate node
            for k in range(n):
                if k == i or k == j:
                    continue
                two_hop_dist = distance_matrix[i, k] + distance_matrix[k, j]
                if two_hop_dist < best_two_hop:
                    best_two_hop = two_hop_dist
            
            # Apply penalty if two-hop is significantly better
            if best_two_hop < 0.8 * direct_dist:
                penalty_factor[i, j] = 0.3  # Strong penalty
            elif best_two_hop < 0.95 * direct_dist:
                penalty_factor[i, j] = 0.7  # Moderate penalty
    
    # 4. Combine factors
    heuristics = inv_dist * nn_affinity * penalty_factor
    
    # 5. Sparsification: keep only top promising edges
    # Keep edges in the top 2*n per node (ensuring connectivity)
    sparsified = np.zeros_like(heuristics)
    
    for i in range(n):
        # Get indices of top 2*n edges from node i
        top_indices = np.argpartition(heuristics[i], -min(2*n, n-1))[-min(2*n, n-1):]
        sparsified[i, top_indices] = heuristics[i, top_indices]
    
    # Ensure symmetry (TSP is undirected)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Normalize to similar scale as v1
    if np.max(sparsified) > 0:
        sparsified = sparsified / np.max(sparsified) * np.max(inv_dist)
    
    return sparsified
