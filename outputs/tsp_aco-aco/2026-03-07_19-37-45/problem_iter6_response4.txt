import numpy as np
from scipy.spatial import distance
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.cluster import KMeans
import warnings
warnings.filterwarnings('ignore')

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with multi-scale regret analysis, strategic lookahead,
    adaptive geometric fusion, and cluster-aware sparsification.
    
    Key innovations:
    1. Multi-Scale Regret Penalization: Evaluates alternatives at multiple neighborhood sizes
    2. Strategic Lookahead Simulation: Uses guided Monte Carlo simulations with memory
    3. Adaptive Geometric Fusion: Instance-aware weighting with correlation analysis
    4. Cluster-Aware Sparsification: Hierarchical pruning preserving global connectivity
    """
    
    n = distance_matrix.shape[0]
    
    # Base case handling
    if n <= 2:
        result = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(result, 0)
        return result
    
    # Phase 1: Comprehensive instance analysis
    stats = analyze_instance(distance_matrix)
    
    # Phase 2: Multi-factor computation
    inv_dist = compute_adaptive_inverse(distance_matrix, stats)
    regret = compute_multi_scale_regret(distance_matrix, stats)
    lookahead = compute_strategic_lookahead(distance_matrix, stats)
    
    # Phase 3: Cluster structure analysis
    cluster_bonus = compute_cluster_structure(distance_matrix, stats)
    
    # Phase 4: Intelligent fusion with correlation-aware weights
    base_heuristic = fuse_factors(inv_dist, regret, lookahead, cluster_bonus, stats)
    
    # Phase 5: Hierarchical sparsification
    final_heuristic = hierarchical_sparsify(base_heuristic, distance_matrix, stats)
    
    return final_heuristic


def analyze_instance(dist_mat: np.ndarray) -> dict:
    """Comprehensive instance analysis with structural characterization."""
    n = dist_mat.shape[0]
    mask = ~np.eye(n, dtype=bool)
    flat_dists = dist_mat[mask]
    
    # Basic statistics
    mean_dist = np.mean(flat_dists)
    std_dist = np.std(flat_dists)
    cv_dist = std_dist / (mean_dist + 1e-12)
    
    # Structural analysis
    q25, median, q75 = np.percentile(flat_dists, [25, 50, 75])
    iqr = q75 - q25
    concentration = iqr / (median + 1e-12)
    
    # Distance matrix eigen-analysis for dimensionality
    centered = dist_mat - np.mean(dist_mat, axis=1, keepdims=True)
    cov = centered @ centered.T / n
    eigvals = np.linalg.eigvalsh(cov)
    eigvals = eigvals[eigvals > 1e-10]
    effective_dim = np.sum(eigvals) ** 2 / np.sum(eigvals ** 2) if len(eigvals) > 0 else 1
    
    # Connectivity pattern
    nn_dist = np.sort(dist_mat, axis=1)[:, 1]  # nearest neighbor distances
    nn_uniformity = np.std(nn_dist) / (np.mean(nn_dist) + 1e-12)
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'cv_dist': cv_dist,
        'concentration': concentration,
        'effective_dim': effective_dim,
        'nn_uniformity': nn_uniformity,
        'median_dist': median,
        'q25_dist': q25,
        'q75_dist': q75
    }


def compute_adaptive_inverse(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
    """Adaptive inverse distance with context-aware power scaling."""
    n = stats['n']
    cv = stats['cv_dist']
    
    # Dynamic power based on distance distribution characteristics
    if cv < 0.3:
        # Concentrated distances: emphasize differences more
        power = 2.5 - 1.5 * cv
    elif cv > 0.7:
        # Diverse distances: be less aggressive
        power = 1.2
    else:
        power = 2.0 - 1.0 * cv
    
    inv_dist = 1 / (dist_mat + np.eye(n) * 1e-12) ** power
    np.fill_diagonal(inv_dist, 0)
    
    # Normalize to unit range
    max_val = np.max(inv_dist)
    if max_val > 0:
        inv_dist = inv_dist / max_val
    
    return inv_dist


def compute_multi_scale_regret(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
    """Multi-scale regret analysis evaluating alternatives at different neighborhood sizes."""
    n = stats['n']
    regret = np.ones((n, n))
    
    # Adaptive neighborhood sizes based on instance characteristics
    k_values = [min(2, n-1), min(4, n-1), min(8, n-1)]
    k_values = [k for k in k_values if k > 0]
    
    for i in range(n):
        dists_i = dist_mat[i].copy()
        dists_i[i] = np.inf
        
        # Compute regrets at multiple scales
        scale_regrets = []
        for k in k_values:
            if len(dists_i) >= k:
                kth_best = np.partition(dists_i, k-1)[k-1]
                regret_k = np.zeros(n)
                for j in range(n):
                    if i == j:
                        continue
                    if dist_mat[i, j] <= kth_best:
                        regret_k[j] = 1.5 - (dist_mat[i, j] / (kth_best + 1e-12))
                    else:
                        gap = (dist_mat[i, j] - kth_best) / (kth_best + 1e-12)
                        regret_k[j] = np.exp(-gap * 2)
                scale_regrets.append(regret_k)
        
        # Combine multi-scale regrets with weighted average
        if scale_regrets:
            weights = [1.0, 0.6, 0.3][:len(scale_regrets)]
            total_weight = sum(weights)
            combined = sum(w * r for w, r in zip(weights, scale_regrets)) / total_weight
            regret[i] = np.minimum(combined, 2.0)
    
    # Symmetrize and normalize
    regret = np.sqrt(regret * regret.T)
    np.fill_diagonal(regret, 0)
    
    max_val = np.max(regret)
    if max_val > 0:
        regret = regret / max_val
    
    return regret


def compute_strategic_lookahead(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
    """Strategic lookahead with guided simulations and path memory."""
    n = stats['n']
    
    if n < 4:
        lookahead = np.ones((n, n))
        np.fill_diagonal(lookahead, 0)
        return lookahead
    
    lookahead = np.ones((n, n))
    mean_dist = stats['mean_dist']
    
    # Adaptive simulation parameters
    n_simulations = min(20, max(5, int(np.log2(n) * 5)))
    depth = min(5, n - 2)
    
    # Temperature for softmax selection (adaptive based on concentration)
    temperature = mean_dist * (0.5 + 0.5 * stats['concentration'])
    
    # Cache for previously evaluated paths
    path_cache = {}
    
    for i in range(n):
        for j in range(i+1, n):
            if dist_mat[i, j] > 4 * mean_dist:
                lookahead[i, j] = lookahead[j, i] = 0.05
                continue
            
            cache_key = tuple(sorted([i, j]))
            if cache_key in path_cache:
                lookahead[i, j] = lookahead[j, i] = path_cache[cache_key]
                continue
            
            scores = []
            for _ in range(n_simulations):
                # Guided simulation with softmax selection
                visited = set([i, j])
                current = j
                path_length = dist_mat[i, j]
                
                for step in range(depth):
                    available = [k for k in range(n) if k not in visited]
                    if not available:
                        break
                    
                    # Distance-based probabilities with temperature
                    dists = dist_mat[current, available]
                    probs = np.exp(-dists / temperature)
                    probs = probs / (np.sum(probs) + 1e-12)
                    
                    # Biased random selection
                    if np.random.rand() < 0.7:  # 70% follow probabilities
                        next_idx = np.random.choice(len(available), p=probs)
                    else:  # 30% explore alternatives
                        sorted_idx = np.argsort(dists)
                        if len(sorted_idx) > 3:
                            next_idx = np.random.choice(sorted_idx[:3])
                        else:
                            next_idx = 0
                    
                    next_node = available[next_idx]
                    path_length += dists[next_idx]
                    visited.add(next_node)
                    current = next_node
                
                # Score based on expected vs actual performance
                expected = depth * mean_dist
                score = expected / (path_length + 1e-12)
                scores.append(min(score, 4.0))
            
            median_score = np.median(scores)
            lookahead[i, j] = lookahead[j, i] = median_score
            path_cache[cache_key] = median_score
    
    # Normalize
    max_val = np.max(lookahead)
    if max_val > 0:
        lookahead = lookahead / max_val
    
    np.fill_diagonal(lookahead, 0)
    return lookahead


def compute_cluster_structure(dist_mat: np.ndarray, stats: dict) -> np.ndarray:
    """Analyze cluster structure and assign bonuses/penalties."""
    n = stats['n']
    bonus = np.ones((n, n))
    
    if n < 10:
        return bonus
    
    # Adaptive clustering based on instance size
    n_clusters = max(2, min(6, int(np.sqrt(n))))
    
    try:
        # Perform clustering on distance matrix
        kmeans = KMeans(n_clusters=n_clusters, random_state=42, n_init=10)
        labels = kmeans.fit_predict(dist_mat)
        
        # Compute cluster statistics
        cluster_sizes = np.bincount(labels)
        cluster_density = np.zeros(n_clusters)
        
        for c in range(n_clusters):
            mask = labels == c
            if np.sum(mask) > 1:
                sub_dist = dist_mat[mask][:, mask]
                cluster_density[c] = 1 / (np.mean(sub_dist) + 1e-12)
        
        # Normalize densities
        if np.max(cluster_density) > 0:
            cluster_density = cluster_density / np.max(cluster_density)
        
        # Assign bonuses based on cluster relationships
        for i in range(n):
            for j in range(i+1, n):
                ci, cj = labels[i], labels[j]
                
                if ci == cj:
                    # Intra-cluster: bonus based on cluster density
                    density = cluster_density[ci]
                    if density > 0.7:
                        bonus[i, j] = bonus[j, i] = 1.5
                    elif density > 0.4:
                        bonus[i, j] = bonus[j, i] = 1.2
                else:
                    # Inter-cluster: penalize unless it's a good bridge
                    dist_ij = dist_mat[i, j]
                    
                    # Find best connections between these clusters
                    mask_i = labels == ci
                    mask_j = labels == cj
                    inter_dists = dist_mat[mask_i][:, mask_j]
                    
                    if inter_dists.size > 0:
                        min_inter = np.min(inter_dists)
                        if dist_ij <= 1.2 * min_inter:
                            # Good bridge connection
                            bonus[i, j] = bonus[j, i] = 1.1
                        else:
                            # Poor connection
                            bonus[i, j] = bonus[j, i] = 0.7
    
    except Exception:
        # Fallback if clustering fails
        pass
    
    return bonus


def fuse_factors(inv_dist: np.ndarray, regret: np.ndarray, 
                 lookahead: np.ndarray, cluster_bonus: np.ndarray,
                 stats: dict) -> np.ndarray:
    """Intelligent factor fusion with correlation-aware adaptive weights."""
    n = stats['n']
    
    # Analyze factor correlations to avoid redundancy
    factors = [inv_dist, regret, lookahead, cluster_bonus]
    factor_names = ['inv_dist', 'regret', 'lookahead', 'cluster']
    
    # Base weights with instance adaptation
    base_weights = {
        'inv_dist': 1.2,
        'regret': 0.8 + 0.4 * stats['cv_dist'],
        'lookahead': 0.4 + 0.3 * (1 - stats['concentration']),
        'cluster': 0.3 + 0.2 * (stats['effective_dim'] > 1.5)
    }
    
    # Adjust weights based on factor usefulness
    adj_weights = {}
    for name, factor in zip(factor_names, factors):
        # Measure factor variation (higher variation = more informative)
        flat_factor = factor[~np.eye(n, dtype=bool)]
        variation = np.std(flat_factor) / (np.mean(flat_factor) + 1e-12)
        adj_weights[name] = base_weights[name] * (0.8 + 0.4 * variation)
    
    # Normalize weights
    total = sum(adj_weights.values())
    weights = {name: w/total for name, w in adj_weights.items()}
    
    # Geometric mean fusion
    heuristic = np.ones((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Skip clearly bad edges early
            if (inv_dist[i, j] < 0.1 and regret[i, j] < 0.1 and 
                lookahead[i, j] < 0.1):
                continue
                
            combined = (
                (inv_dist[i, j] ** weights['inv_dist']) *
                (regret[i, j] ** weights['regret']) *
                (lookahead[i, j] ** weights['lookahead']) *
                (cluster_bonus[i, j] ** weights['cluster'])
            )
            heuristic[i, j] = heuristic[j, i] = combined
    
    return heuristic


def hierarchical_sparsify(heuristic: np.ndarray, dist_mat: np.ndarray, 
                         stats: dict) -> np.ndarray:
    """Hierarchical sparsification preserving essential connectivity."""
    n = stats['n']
    result = heuristic.copy()
    
    # Level 1: Nearest neighbor preservation
    k_nn = min(8, max(3, int(np.log2(n) * 3)))
    nn_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        row_vals = result[i].copy()
        row_vals[i] = -np.inf
        if k_nn < n-1:
            threshold = np.partition(row_vals, -k_nn)[-k_nn]
            nn_mask[i] = row_vals >= threshold
        else:
            nn_mask[i] = row_vals > -np.inf
    
    # Level 2: MST preservation
    try:
        mst = minimum_spanning_tree(dist_mat).toarray()
        mst_mask = (mst + mst.T) > 0
    except:
        mst_mask = np.zeros((n, n), dtype=bool)
    
    # Level 3: Distance threshold
    dist_mask = np.ones((n, n), dtype=bool)
    if n > 10:
        threshold = stats['q75_dist'] * 1.5
        dist_mask = dist_mat <= threshold
    
    # Combine masks
    combined_mask = nn_mask | mst_mask | (result > 0.1)
    combined_mask = combined_mask & dist_mask
    combined_mask = combined_mask | combined_mask.T
    
    # Apply mask
    result = result * combined_mask
    
    # Ensure minimum connectivity
    min_conn = min(3, n-1)
    for i in range(n):
        connections = np.sum(result[i] > 0)
        if connections < min_conn:
            candidates = heuristic[i].copy()
            candidates[i] = -np.inf
            candidates[result[i] > 0] = -np.inf
            
            if np.any(candidates > -np.inf):
                needed = min_conn - connections
                best_idx = np.argsort(candidates)[-needed:]
                for j in best_idx:
                    if candidates[j] > -np.inf:
                        result[i, j] = result[j, i] = candidates[j]
    
    # Final normalization and cleanup
    result = (result + result.T) / 2
    np.fill_diagonal(result, 0)
    
    max_val = np.max(result)
    if max_val > 0:
        result = result / max_val
    
    # Adaptive thresholding
    threshold = 0.05 + 0.05 * np.exp(-n/25)
    result[result < threshold] = 0
    
    return result
