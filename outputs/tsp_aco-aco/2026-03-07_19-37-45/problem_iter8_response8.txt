import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
import random

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with multi-hop regret analysis,
    ensemble lookahead strategies, and variance-adaptive weighting.
    
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Comprehensive instance analysis
    instance_stats = _analyze_instance_v3(distance_matrix)
    
    # Step 2: Compute base factors with multi-hop regret
    base_factors = _compute_base_factors_v3(distance_matrix, instance_stats)
    
    # Step 3: Ensemble lookahead with multiple strategies
    lookahead_ensemble = _compute_ensemble_lookahead(distance_matrix, instance_stats)

    weights = _compute_variance_adaptive_weights(distance_matrix, instance_stats, base_factors, lookahead_ensemble)
    
    # Step 5: Build candidate edge mask for efficiency
    candidate_mask = _build_adaptive_candidate_mask(distance_matrix, instance_stats)
    
    # Step 6: Combine factors with learned weights
    H = np.zeros((n, n))
    
    # Precompute factor products for candidate edges
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Geometric mean with adaptive exponents
                combined = (
                    (base_factors['inv_dist'][i, j] ** weights['w_inv']) *
                    (base_factors['multi_hop_regret'][i, j] ** weights['w_regret']) *
                    (base_factors['density'][i, j] ** weights['w_density']) *
                    (lookahead_ensemble[i, j] ** weights['w_lookahead']) *
                    (base_factors['mst'][i, j] ** weights['w_mst']) *
                    (base_factors['connectivity'][i, j] ** weights['w_connectivity'])
                )
                H[i, j] = H[j, i] = combined
    
    # Step 7: Smart post-processing with adaptive sparsification
    H = _post_process_with_sparsification(H, distance_matrix, instance_stats)
    
    return H

def _analyze_instance_v3(distance_matrix: np.ndarray) -> Dict:
    """Comprehensive instance analysis for adaptive parameter tuning."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal for statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    # Basic statistics
    stats = {
        'n': n,
        'mean_dist': np.mean(dist_no_diag),
        'std_dist': np.std(dist_no_diag),
        'cv_dist': np.std(dist_no_diag) / (np.mean(dist_no_diag) + 1e-8),
        'percentile_25': np.percentile(dist_no_diag, 25),
        'percentile_75': np.percentile(dist_no_diag, 75),
    }
    
    # Advanced statistics
    stats['sparsity'] = _estimate_structured_sparsity(distance_matrix)
    stats['cluster_tendency'] = _estimate_cluster_tendency_v3(distance_matrix)
    stats['regularity'] = _estimate_geometric_regularity(distance_matrix)
    stats['distance_skewness'] = _estimate_distance_skewness(distance_matrix)
    
    # Neighborhood statistics
    stats['neighbor_consistency'] = _compute_neighbor_consistency(distance_matrix)
    stats['hub_tendency'] = _estimate_hub_tendency(distance_matrix)
    
    return stats

def _estimate_structured_sparsity(distance_matrix: np.ndarray) -> float:
    """Estimate structured sparsity using distance distribution analysis."""
    n = distance_matrix.shape[0]
    if n <= 10:
        return 0.5
    
    flat_dists = distance_matrix[~np.eye(n, dtype=bool)]
    
    # Compute quartile coefficient of dispersion
    q1 = np.percentile(flat_dists, 25)
    q3 = np.percentile(flat_dists, 75)
    median = np.median(flat_dists)
    
    if median > 0:
        # Lower sparsity for more uniform distributions
        iqr = q3 - q1
        sparsity = iqr / (median + 1e-8)
        return min(1.0, sparsity)
    return 0.5

def _estimate_cluster_tendency_v3(distance_matrix: np.ndarray) -> float:
    """Enhanced cluster tendency using local neighborhood analysis."""
    n = distance_matrix.shape[0]
    if n < 10:
        return 0.5
    
    # Analyze local density variations
    k = min(5, n // 4)
    local_densities = []
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k)[:k]
        local_densities.append(np.mean(k_nearest))
    
    # Compute coefficient of variation of local densities
    if np.std(local_densities) > 0:
        cv_local = np.std(local_densities) / (np.mean(local_densities) + 1e-8)
        return min(1.0, cv_local * 2)
    return 0.5

def _estimate_geometric_regularity(distance_matrix: np.ndarray) -> float:
    """Estimate geometric regularity of node distribution."""
    n = distance_matrix.shape[0]
    
    # Analyze nearest neighbor distance consistency
    nn_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists.append(np.min(dists))
    
    cv_nn = np.std(nn_dists) / (np.mean(nn_dists) + 1e-8)
    regularity = 1.0 / (1.0 + cv_nn * 2)
    return min(1.0, max(0.0, regularity))

def _estimate_distance_skewness(distance_matrix: np.ndarray) -> float:
    """Estimate skewness of distance distribution."""
    n = distance_matrix.shape[0]
    flat_dists = distance_matrix[~np.eye(n, dtype=bool)]
    
    # Compute moment-based skewness
    if len(flat_dists) >= 3 and np.std(flat_dists) > 0:
        skewness = np.mean((flat_dists - np.mean(flat_dists))**3) / (np.std(flat_dists)**3 + 1e-8)
        return min(1.0, max(0.0, abs(skewness) * 0.5))
    return 0.0

def _compute_neighbor_consistency(distance_matrix: np.ndarray) -> float:
    """Compute consistency of neighborhood relationships."""
    n = distance_matrix.shape[0]
    if n < 8:
        return 0.5
    
    k = min(4, n // 3)
    consistency_scores = []
    
    for i in range(n):
        # Get k-nearest neighbors
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        neighbors = np.argpartition(dists, k)[:k]
        
        # Check reciprocal relationships
        reciprocal_count = 0
        for j in neighbors:
            dists_j = distance_matrix[j].copy()
            dists_j[j] = np.inf
            j_neighbors = np.argpartition(dists_j, k)[:k]
            if i in j_neighbors:
                reciprocal_count += 1
        
        consistency_scores.append(reciprocal_count / k)
    
    return np.mean(consistency_scores)

def _estimate_hub_tendency(distance_matrix: np.ndarray) -> float:
    """Estimate tendency for hub nodes to exist."""
    n = distance_matrix.shape[0]
    
    # Compute degree-like centrality
    centralities = []
    threshold = np.median(distance_matrix[~np.eye(n, dtype=bool)])
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Count nodes within median distance
        close_nodes = np.sum(dists <= threshold * 1.5)
        centralities.append(close_nodes / (n - 1))
    
    # Measure variance in centrality
    if np.std(centralities) > 0:
        cv_centrality = np.std(centralities) / (np.mean(centralities) + 1e-8)
        return min(1.0, cv_centrality)
    return 0.0

def _compute_base_factors_v3(distance_matrix: np.ndarray, stats: Dict) -> Dict:
    """Compute all base heuristic factors including multi-hop regret."""
    n = stats['n']
    
    factors = {}
    
    # 1. Inverse distance (normalized)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    factors['inv_dist'] = inv_dist
    
    # 2. Multi-hop regret analysis (2-3 hops)
    factors['multi_hop_regret'] = _compute_multi_hop_regret(distance_matrix, stats)
    
    # 3. Density-based scoring
    factors['density'] = _compute_density_scores_v3(distance_matrix, stats)
    
    # 4. MST-based scoring with edge criticality
    factors['mst'] = _compute_mst_scores_v3(distance_matrix)
    
    # 5. Node connectivity scoring
    factors['connectivity'] = _compute_connectivity_scores(distance_matrix, stats)
    
    return factors

def _compute_multi_hop_regret(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute regret scores using 2-hop and 3-hop alternative paths."""
    n = stats['n']
    regret_scores = np.ones((n, n))
    
    # Adaptive parameters based on instance size
    max_hops = 3 if n <= 50 else 2
    neighbor_limit = min(15, max(5, n // 4))
    
    for i in range(n):
        # Get candidate intermediate nodes (nearest neighbors)
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        candidates_i = np.argpartition(dists_i, neighbor_limit)[:neighbor_limit]
        
        for j in range(i+1, n):
            if i == j:
                continue
                
            direct_cost = distance_matrix[i, j]
            
            # Get candidate intermediate nodes for j
            dists_j = distance_matrix[j].copy()
            dists_j[j] = np.inf
            candidates_j = np.argpartition(dists_j, neighbor_limit)[:neighbor_limit]
            
            # Combine candidate sets
            candidates = list(set(candidates_i).union(set(candidates_j)))
            if i in candidates:
                candidates.remove(i)
            if j in candidates:
                candidates.remove(j)
            
            if not candidates:
                continue
            
            # Find best 2-hop path i-k-j
            best_2hop = np.inf
            for k in candidates:
                cost_2hop = distance_matrix[i, k] + distance_matrix[k, j]
                if cost_2hop < best_2hop:
                    best_2hop = cost_2hop
            
            # Find best 3-hop path i-k1-k2-j (if allowed)
            best_3hop = np.inf
            if max_hops >= 3 and len(candidates) >= 2:
                for idx1, k1 in enumerate(candidates):
                    for k2 in candidates[idx1+1:]:
                        cost_3hop = (distance_matrix[i, k1] + 
                                   distance_matrix[k1, k2] + 
                                   distance_matrix[k2, j])
                        if cost_3hop < best_3hop:
                            best_3hop = cost_3hop
            
            # Best alternative path
            best_alternative = min(best_2hop, best_3hop)
            
            if best_alternative < direct_cost:
                # Good alternative exists - penalize this edge
                regret_ratio = best_alternative / (direct_cost + 1e-8)
                # Transform to score: lower for higher regret
                regret_scores[i, j] = regret_scores[j, i] = np.sqrt(regret_ratio)
            else:
                # No good alternative - edge is valuable
                # Bonus for edges that are significantly better than alternatives
                improvement = direct_cost / (best_alternative + 1e-8)
                regret_scores[i, j] = regret_scores[j, i] = 1.0 + 0.5 * improvement
    
    np.fill_diagonal(regret_scores, 0)
    return regret_scores

def _compute_density_scores_v3(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute density scores with adaptive neighborhood sizing."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Adaptive neighborhood size
    k = min(8, max(3, n // 6))
    
    # Compute local sparsity for each node
    sparsities = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k)[:k]
        sparsities[i] = np.mean(k_nearest)
    
    # Normalize sparsities
    if np.std(sparsities) > 0:
        sparsities = (sparsities - np.mean(sparsities)) / np.std(sparsities)
    
    # Score edges based on combined sparsity
    for i in range(n):
        for j in range(i+1, n):
            combined_sparsity = sparsities[i] + sparsities[j]
            
            # Edges connecting sparse areas are more valuable
            if combined_sparsity > 1.0:
                # Both nodes are sparse - high priority
                score = 1.5
            elif combined_sparsity < -1.0:
                # Both nodes are dense - lower priority
                score = 0.7
            else:
                # Mixed - moderate priority
                score = 1.0
            
            density_scores[i, j] = density_scores[j, i] = score
    
    return density_scores

def _compute_mst_scores_v3(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute MST-based scores with edge criticality analysis."""
    n = distance_matrix.shape[0]
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            # Compute MST
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Basic MST edge boost
            mst_scores[mst_present] = 2.0
            
            # Analyze edge criticality
            for i in range(n):
                mst_neighbors = np.where(mst_present[i])[0]
                
                if len(mst_neighbors) == 1:
                    # Leaf edge - critical for connectivity
                    j = mst_neighbors[0]
                    mst_scores[i, j] = mst_scores[j, i] = 3.0
                elif len(mst_neighbors) == 2:
                    # Bridge edge - also important
                    for j in mst_neighbors:
                        mst_scores[i, j] = mst_scores[j, i] = 2.5
        except Exception:
            # Fallback if MST computation fails
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores

def _compute_connectivity_scores(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute scores based on node connectivity patterns."""
    n = stats['n']
    connectivity_scores = np.ones((n, n))
    
    # Estimate node degrees in candidate solution
    k = min(10, n // 3)
    pseudo_degrees = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Count nodes within adaptive radius
        radius = np.percentile(dists, 25) * 1.5
        pseudo_degrees[i] = np.sum(dists <= radius)
    
    # Normalize pseudo-degrees
    if np.max(pseudo_degrees) > 0:
        pseudo_degrees = pseudo_degrees / np.max(pseudo_degrees)
    
    # Score edges: favor connections that balance degrees
    for i in range(n):
        for j in range(i+1, n):
            deg_i, deg_j = pseudo_degrees[i], pseudo_degrees[j]
            
            if deg_i < 0.3 and deg_j < 0.3:
                # Both nodes are isolated - high priority
                score = 1.8
            elif deg_i > 0.7 and deg_j > 0.7:
                # Both nodes are well-connected - lower priority
                score = 0.8
            elif abs(deg_i - deg_j) > 0.5:
                # Connecting hub with isolated node - high priority
                score = 1.5
            else:
                # Balanced connection
                score = 1.0
            
            connectivity_scores[i, j] = connectivity_scores[j, i] = score
    
    return connectivity_scores

def _compute_ensemble_lookahead(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Compute lookahead scores using ensemble of strategies."""
    n = stats['n']
    
    if n < 4:
        return np.ones((n, n))
    
    # Initialize ensemble scores
    lookahead_scores = np.ones((n, n))
    
    # Strategy 1: Greedy path extension
    scores_greedy = _compute_greedy_lookahead(distance_matrix, stats)
    
    # Strategy 2: Beam search lookahead
    scores_beam = _compute_beam_lookahead(distance_matrix, stats)
    
    # Strategy 3: Regret-based lookahead
    scores_regret = _compute_regret_lookahead(distance_matrix, stats)
    
    # Combine strategies with instance-adaptive weights
    # Weight greedy higher for regular instances, beam higher for complex ones
    weight_greedy = 0.4 + 0.3 * stats['regularity']
    weight_beam = 0.3 + 0.4 * (1 - stats['regularity'])
    weight_regret = 0.3
    
    # Normalize weights
    total_weight = weight_greedy + weight_beam + weight_regret
    weight_greedy /= total_weight
    weight_beam /= total_weight
    weight_regret /= total_weight
    
    # Combine scores
    for i in range(n):
        for j in range(i+1, n):
            if i == j:
                continue
            combined = (
                scores_greedy[i, j] * weight_greedy +
                scores_beam[i, j] * weight_beam +
                scores_regret[i, j] * weight_regret
            )
            lookahead_scores[i, j] = lookahead_scores[j, i] = combined
    
    # Normalize final scores
    max_val = np.max(lookahead_scores)
    if max_val > 0:
        lookahead_scores = lookahead_scores / max_val
    
    np.fill_diagonal(lookahead_scores, 0)
    return lookahead_scores

def _compute_greedy_lookahead(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Greedy path extension lookahead."""
    n = stats['n']
    scores = np.ones((n, n))
    
    n_simulations = min(20, max(5, int(np.sqrt(n) * 2)))
    max_steps = min(6, n - 2)
    
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] > 2.5 * stats['mean_dist']:
                continue
            
            simulation_scores = []
            for _ in range(n_simulations):
                # Simulate greedy extension
                visited = np.zeros(n, dtype=bool)
                visited[i] = visited[j] = True
                current = j
                path_length = distance_matrix[i, j]
                
                for step in range(max_steps):
                    unvisited = np.where(~visited)[0]
                    if len(unvisited) == 0:
                        break
                    
                    # Find closest unvisited node
                    dists = distance_matrix[current, unvisited]
                    next_idx = np.argmin(dists)
                    next_node = unvisited[next_idx]
                    
                    path_length += dists[next_idx]
                    visited[next_node] = True
                    current = next_node
                
                # Score based on efficiency
                expected = max_steps * stats['mean_dist']
                if path_length > 0:
                    score = expected / path_length
                    simulation_scores.append(min(score, 3.0))
            
            if simulation_scores:
                scores[i, j] = scores[j, i] = np.median(simulation_scores)
    
    return scores

def _compute_beam_lookahead(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Beam search lookahead with limited beam width."""
    n = stats['n']
    scores = np.ones((n, n))
    
    if n < 8:
        return scores
    
    beam_width = min(3, n // 4)
    max_depth = min(4, n - 2)
    
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] > 3 * stats['mean_dist']:
                scores[i, j] = scores[j, i] = 0.5
                continue
            
            # Initialize beam with starting edge
            beam = [{
                'path': [i, j],
                'visited': set([i, j]),
                'current': j,
                'length': distance_matrix[i, j]
            }]
            
            for depth in range(max_depth):
                new_beam = []
                
                for state in beam:
                    current = state['current']
                    visited = state['visited']
                    
                    # Get unvisited nodes
                    unvisited = [node for node in range(n) if node not in visited]
                    
                    if not unvisited:
                        continue
                    
                    # Evaluate potential next steps
                    candidates = []
                    for next_node in unvisited:
                        # Simple heuristic: distance + potential
                        dist = distance_matrix[current, next_node]
                        
                        # Estimate remaining tour (simplified)
                        remaining_nodes = n - len(visited) - 1
                        est_remaining = remaining_nodes * stats['mean_dist']
                        
                        total_est = state['length'] + dist + est_remaining
                        candidates.append((next_node, dist, total_est))
                    
                    # Keep best candidates
                    candidates.sort(key=lambda x: x[2])
                    candidates = candidates[:beam_width]
                    
                    for next_node, dist, _ in candidates:
                        new_state = {
                            'path': state['path'] + [next_node],
                            'visited': state['visited'].union([next_node]),
                            'current': next_node,
                            'length': state['length'] + dist
                        }
                        new_beam.append(new_state)
                
                if not new_beam:
                    break
                    
                # Keep best beam
                new_beam.sort(key=lambda x: x['length'])
                beam = new_beam[:beam_width]
            
            # Score based on best found path
            if beam:
                best_length = beam[0]['length']
                expected = (len(beam[0]['path']) - 1) * stats['mean_dist']
                if best_length > 0:
                    score = expected / best_length
                    scores[i, j] = scores[j, i] = min(score, 2.5)
    
    return scores

def _compute_regret_lookahead(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Regret-based lookahead considering future choices."""
    n = stats['n']
    scores = np.ones((n, n))
    
    for i in range(n):
        for j in range(i+1, n):
            if i == j:
                continue
            
            direct_cost = distance_matrix[i, j]
            
            # Consider the next step from both ends
            regrets_i = []
            regrets_j = []
            
            for k in range(n):
                if k != i and k != j:
                    # Regret if we take edge (i,j) instead of (i,k) now
                    regret_i = distance_matrix[i, k] - direct_cost
                    regrets_i.append(regret_i)
                    
                    # Regret if we take edge (i,j) instead of (j,k) now
                    regret_j = distance_matrix[j, k] - direct_cost
                    regrets_j.append(regret_j)
            
            if regrets_i and regrets_j:
                # Average regret (negative values become bonuses)
                avg_regret_i = np.mean(regrets_i)
                avg_regret_j = np.mean(regrets_j)
                total_regret = (avg_regret_i + avg_regret_j) / 2
                
                # Convert to score: lower regret -> higher score
                score = np.exp(-total_regret / (stats['mean_dist'] + 1e-8))
                scores[i, j] = scores[j, i] = min(max(score, 0.2), 2.0)
    
    return scores

def _compute_variance_adaptive_weights(distance_matrix: np.ndarray, stats: Dict, 
                                      base_factors: Dict, lookahead_ensemble: np.ndarray) -> Dict:
    """Compute adaptive weights based on factor variance and instance characteristics."""
    n = stats['n']
    
    # Analyze discriminative power of each factor
    factor_variances = {}
    sample_edges = []
    
    # Sample edges for analysis
    sample_size = min(100, n * (n-1) // 4)
    for _ in range(sample_size):
        i, j = random.sample(range(n), 2)
        sample_edges.append((i, j))
    
    # Compute variance for each factor
    for factor_name, factor_matrix in base_factors.items():
        values = []
        for i, j in sample_edges:
            values.append(factor_matrix[i, j])
        
        if values:
            # Use coefficient of variation to measure discriminative power
            mean_val = np.mean(values)
            std_val = np.std(values)
            if mean_val > 1e-8:
                factor_variances[factor_name] = std_val / mean_val
            else:
                factor_variances[factor_name] = 0.0
        else:
            factor_variances[factor_name] = 0.0
    
    # Also compute variance for lookahead
    lookahead_values = []
    for i, j in sample_edges:
        lookahead_values.append(lookahead_ensemble[i, j])
    
    if lookahead_values and np.mean(lookahead_values) > 1e-8:
        factor_variances['lookahead'] = np.std(lookahead_values) / np.mean(lookahead_values)
    else:
        factor_variances['lookahead'] = 0.0
    
    # Base weights from instance characteristics
    weights = {
        'w_inv': 1.0,  # Always important
        
        # Regret weight: higher when regret analysis is discriminative
        'w_regret': 0.5 + 0.5 * min(1.0, factor_variances.get('multi_hop_regret', 0.5)),
        
        # Density weight: higher for clustered instances
        'w_density': 0.3 + 0.4 * stats['cluster_tendency'],
        
        # Lookahead weight: higher when lookahead is discriminative and instance is complex
        'w_lookahead': 0.2 + 0.3 * factor_variances.get('lookahead', 0.5) + 0.2 * (1 - stats['regularity']),
        
        # MST weight: decreases with instance size
        'w_mst': max(0.1, 0.6 - 0.01 * n),
        
        # Connectivity weight: higher when nodes have varying connectivity
        'w_connectivity': 0.2 + 0.3 * stats['hub_tendency']
    }
    
    # Adjust weights based on factor correlations (avoid redundant factors)
    if n >= 10:
        # Simple redundancy check: if two factors are highly correlated, reduce weight of less discriminative one
        factor_pairs = [('multi_hop_regret', 'inv_dist'), ('density', 'connectivity')]
        
        for factor1, factor2 in factor_pairs:
            if factor1 in factor_variances and factor2 in factor_variances:
                # If both factors are present and highly correlated in the sample
                values1 = []
                values2 = []
                for i, j in sample_edges:
                    values1.append(base_factors[factor1][i, j])
                    values2.append(base_factors[factor2][i, j])
                
                if len(values1) >= 5:
                    correlation = abs(np.corrcoef(values1, values2)[0, 1])
                    if correlation > 0.7:
                        # Reduce weight of less discriminative factor
                        if factor_variances[factor1] < factor_variances[factor2]:
                            weights[f'w_{factor1}'] *= 0.7
                        else:
                            weights[f'w_{factor2}'] *= 0.7
    
    # Normalize to target sum (approximately 4.0)
    target_sum = 4.0
    current_sum = sum(weights.values())
    if current_sum > 0:
        scale = target_sum / current_sum
        for key in weights:
            weights[key] *= scale
    
    return weights

def _build_adaptive_candidate_mask(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Build mask of candidate edges with adaptive sparsification."""
    n = stats['n']
    
    # Base candidate size
    base_k = min(15, max(5, int(np.log2(n) * 3)))
    
    # Adjust based on instance characteristics
    # More candidates for sparse/irregular instances
    k = int(base_k * (1.0 + 0.3 * stats['sparsity'] + 0.2 * (1 - stats['regularity'])))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # Include k-nearest neighbors for each node
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < n - 1:
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Include MST edges
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            candidate_mask = candidate_mask | mst_present
        except Exception:
            pass
    
    # Ensure symmetry and zero diagonal
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask

def _post_process_with_sparsification(H: np.ndarray, distance_matrix: np.ndarray, 
                                     stats: Dict) -> np.ndarray:
    """Post-process heuristic matrix with intelligent sparsification."""
    n = stats['n']
    
    # Step 1: Ensure minimum connectivity
    min_edges = min(4, max(2, n // 10))
    for i in range(n):
        non_zero = np.where(H[i] > 0)[0]
        if len(non_zero) < min_edges:
            # Add nearest neighbors with base scores
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    base_score = 1 / (dists[j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    # Step 2: Adaptive sparsification
    if n > 15:
        # Determine sparsification threshold based on instance characteristics
        # Keep more edges for complex instances
        keep_ratio = 0.3 + 0.3 * stats['regularity']  # 30-60% of edges
        
        # Compute thresholds per node (adaptive to local density)
        for i in range(n):
            row_values = H[i].copy()
            row_values[i] = -np.inf  # Ignore diagonal
            
            if np.max(row_values) > 0:
                # Determine threshold to keep top keep_ratio of candidate edges
                sorted_vals = np.sort(row_values)[::-1]
                n_candidates = len(sorted_vals[sorted_vals > 0])
                
                if n_candidates > min_edges:
                    threshold_idx = min(n_candidates - 1, int(n_candidates * keep_ratio))
                    threshold = sorted_vals[threshold_idx]
                    
                    # Zero out values below threshold
                    mask = (H[i] < threshold) & (H[i] > 0)
                    H[i, mask] = 0
                    H[mask, i] = 0
    
    # Step 3: Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Step 4: Final normalization with sigmoid emphasis
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    # Apply sigmoid to emphasize strong candidates
    H = 1 / (1 + np.exp(-8 * (H - 0.6)))
    
    return H
