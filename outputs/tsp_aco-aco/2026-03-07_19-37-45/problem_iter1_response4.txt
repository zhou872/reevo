import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    Returns a sparse matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    # 1. Base reciprocal distance (preserve original intuition)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.diag_indices(n)] = 0
    
    # 2. Local neighborhood factor (edges within local clusters are prioritized)
    # Compute average distance to k-nearest neighbors for each node
    k = max(2, min(5, n // 4))  # Adaptive k based on problem size
    local_radii = np.zeros(n)
    for i in range(n):
        sorted_dists = np.sort(distance_matrix[i])
        local_radii[i] = np.mean(sorted_dists[1:k+1])  # Exclude self
    
    # Create local neighborhood mask
    local_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        local_mask[i] = distance_matrix[i] <= 1.5 * local_radii[i]
    
    # 3. Minimum Spanning Tree prior (MST edges often appear in optimal TSP tours)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0  # Make symmetric
    
    # 4. Symmetry bonus (edges connecting symmetric regions in distance space)
    # Compute node centrality via average distance to all other nodes
    centrality = np.mean(distance_matrix, axis=1)
    centrality_norm = centrality / np.max(centrality)
    
    # Symmetry score: edges connecting nodes with similar centrality
    sym_matrix = np.abs(centrality_norm[:, None] - centrality_norm[None, :])
    sym_bonus = 1.0 / (1.0 + 10 * sym_matrix)
    
    # 5. Combine factors with adaptive weights
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Weight factors based on problem size
    w_inv = 1.0
    w_local = 0.7 + 0.3 * (n <= 50)  # Higher weight for local connections in smaller problems
    w_mst = 1.2
    w_sym = 0.5
    
    # Apply local sparsification: only compute scores for promising edges
    for i in range(n):
        # Consider local neighbors plus a few random distant connections for exploration
        local_indices = np.where(local_mask[i])[0]
        if len(local_indices) < 3:
            # Ensure minimum connectivity
            all_indices = np.setdiff1d(np.arange(n), [i])
            additional = np.random.choice(all_indices, 
                                        size=min(3, n-1), 
                                        replace=False)
            local_indices = np.union1d(local_indices, additional)
        
        for j in local_indices:
            if j <= i:
                continue  # Process each edge once
                
            # Combined score
            base_score = w_inv * inv_dist[i, j]
            local_score = w_local if local_mask[i, j] else 0.1
            mst_score = w_mst if mst_mask[i, j] else 0.5
            symmetry_score = w_sym * sym_bonus[i, j]
            
            # Additional bonus for edges in dense regions
            density_i = np.sum(local_mask[i]) / n
            density_j = np.sum(local_mask[j]) / n
            density_bonus = 1.0 + 0.5 * (density_i + density_j)
            
            combined = base_score * local_score * mst_score * symmetry_score * density_bonus
            
            heuristics_matrix[i, j] = combined
            heuristics_matrix[j, i] = combined  # Symmetric
    
    # 6. Final sparsification: keep only top edges
    # Determine sparsity level adaptively
    max_edges = max(2 * n, n * (n - 1) // 10)  # At least 2n edges, at most 10% of all edges
    flat_heuristics = heuristics_matrix.flatten()
    threshold = np.sort(flat_heuristics)[-max_edges] if len(flat_heuristics) > max_edges else 0
    
    # Apply threshold
    heuristics_matrix[heuristics_matrix < threshold] = 0
    
    # Ensure at least minimal connectivity for each node
    min_connections = min(3, n - 1)
    for i in range(n):
        node_edges = heuristics_matrix[i]
        non_zero = np.nonzero(node_edges)[0]
        if len(non_zero) < min_connections:
            # Add connections to nearest nodes not already connected
            sorted_indices = np.argsort(distance_matrix[i])
            for idx in sorted_indices:
                if idx != i and node_edges[idx] == 0:
                    heuristics_matrix[i, idx] = inv_dist[i, idx]
                    heuristics_matrix[idx, i] = inv_dist[i, idx]
                    if len(np.nonzero(heuristics_matrix[i])[0]) >= min_connections:
                        break
    
    # Normalize for better numerical stability
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
