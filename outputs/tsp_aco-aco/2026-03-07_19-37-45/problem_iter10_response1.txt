import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import ConvexHull
from typing import Dict, List, Tuple
import random

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristic with geometric and topological features,
    deep predictive simulations, and adaptive clustering.
    
    Returns a heuristic matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Enhanced instance analysis with geometric features
    instance_stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Build intelligent candidate mask using geometric properties
    candidate_mask = _build_geometric_candidate_mask(distance_matrix, instance_stats)
    
    # Step 3: Compute comprehensive factors with geometric bonuses
    factors = _compute_geometric_factors(distance_matrix, instance_stats, candidate_mask)
    
    # Step 4: Deep predictive simulation for critical edges
    simulation_scores = _deep_predictive_simulation(distance_matrix, instance_stats, candidate_mask)
    
    # Step 5: Adaptive weight optimization using instance characteristics
    weights = _optimize_weights_geometric(distance_matrix, factors, simulation_scores, instance_stats)
    
    # Step 6: Combine all factors with geometric mean
    H = _combine_all_factors(factors, simulation_scores, weights, candidate_mask)
    
    # Step 7: Topological refinement and post-processing
    H = _topological_refinement(H, distance_matrix, instance_stats, candidate_mask)
    
    return H


def _analyze_instance_v2(distance_matrix: np.ndarray) -> Dict:
    """
    Enhanced instance analysis with geometric and topological features.
    """
    n = distance_matrix.shape[0]
    
    # Basic distance statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    dist_vals = dist_no_diag.flatten()
    
    mean_dist = np.mean(dist_vals)
    std_dist = np.std(dist_vals)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    percentiles = np.percentile(dist_vals, [10, 25, 50, 75, 90])
    
    # Geometric properties
    geo_features = _compute_geometric_features(distance_matrix)
    
    # Clustering analysis
    cluster_features = _analyze_cluster_structure(distance_matrix)
    
    # Topological properties
    topo_features = _compute_topological_features(distance_matrix)
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'p10': percentiles[0],
        'p50': percentiles[2],
        'p90': percentiles[4],
        'geo_spread': geo_features['spread'],
        'convexity': geo_features['convexity'],
        'cluster_quality': cluster_features['quality'],
        'cluster_count': cluster_features['count'],
        'topo_connectivity': topo_features['connectivity'],
        'centrality_variance': topo_features['centrality_var']
    }


def _compute_geometric_features(distance_matrix: np.ndarray) -> Dict:
    """
    Estimate geometric properties from distance matrix using multidimensional scaling.
    """
    n = distance_matrix.shape[0]
    
    if n < 4:
        return {'spread': 0.5, 'convexity': 0.5}
    
    try:
        # Use MDS to estimate spatial spread
        from sklearn.manifold import MDS
        mds = MDS(n_components=2, dissimilarity='precomputed', random_state=42, max_iter=100)
        coords = mds.fit_transform(distance_matrix)
        
        # Estimate convex hull properties
        hull = ConvexHull(coords)
        hull_area = hull.volume
        point_area = np.var(coords[:, 0]) + np.var(coords[:, 1])
        
        # Spread measure: ratio of hull area to point variance
        spread = min(1.0, hull_area / (point_area * 10 + 1e-8))
        
        # Convexity measure: perimeter^2 / area ratio
        hull_perimeter = hull.area
        convexity = hull_perimeter**2 / (4 * np.pi * hull_area + 1e-8) if hull_area > 0 else 1.0
        convexity = 1 / (1 + convexity)  # Invert: 1 is highly convex, 0 is irregular
        
        return {'spread': spread, 'convexity': convexity}
    except:
        # Fallback for MDS failure
        return {'spread': 0.5, 'convexity': 0.5}


def _analyze_cluster_structure(distance_matrix: np.ndarray) -> Dict:
    """
    Analyze clustering structure using spectral clustering principles.
    """
    n = distance_matrix.shape[0]
    
    if n < 6:
        return {'quality': 0.5, 'count': 1}
    
    try:
        from sklearn.cluster import SpectralClustering
        
        # Estimate number of clusters
        k_max = min(5, n // 3)
        silhouette_scores = []
        
        for k in range(2, k_max + 1):
            try:
                sc = SpectralClustering(n_clusters=k, affinity='precomputed', random_state=42)
                labels = sc.fit_predict(1 / (distance_matrix + np.eye(n) * 1e-8))
                
                # Compute silhouette-like score
                intra_dists = []
                inter_dists = []
                
                for i in range(n):
                    for j in range(i+1, n):
                        if labels[i] == labels[j]:
                            intra_dists.append(distance_matrix[i, j])
                        else:
                            inter_dists.append(distance_matrix[i, j])
                
                if intra_dists and inter_dists:
                    mean_intra = np.mean(intra_dists)
                    mean_inter = np.mean(inter_dists)
                    score = (mean_inter - mean_intra) / max(mean_inter, mean_intra)
                    silhouette_scores.append((k, max(0, score)))
            except:
                continue
        
        if silhouette_scores:
            best_k, best_score = max(silhouette_scores, key=lambda x: x[1])
            return {'quality': best_score, 'count': best_k}
    
    except ImportError:
        # Fallback: use hierarchical clustering
        pass
    
    # Simple fallback
    return {'quality': 0.5, 'count': max(1, n // 5)}


def _compute_topological_features(distance_matrix: np.ndarray) -> Dict:
    """
    Compute topological network features.
    """
    n = distance_matrix.shape[0]
    
    # Centrality measures
    closeness = np.zeros(n)
    betweenness = np.zeros(n)
    
    for i in range(n):
        # Closeness centrality
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        closeness[i] = 1 / np.mean(dists)
        
        # Betweenness approximation
        for j in range(n):
            if i != j:
                shortest = distance_matrix[i, j]
                alternative = np.min(distance_matrix[i] + distance_matrix[:, j])
                if shortest < alternative * 0.99:  # i lies on shortest path
                    betweenness[i] += 1
    
    centrality_var = np.var(closeness + betweenness)
    
    # Graph connectivity approximation
    mst = minimum_spanning_tree(distance_matrix).toarray()
    connectivity = np.sum(mst > 0) / (n * (n-1))
    
    return {
        'connectivity': connectivity,
        'centrality_var': min(1.0, centrality_var * 10)
    }


def _build_geometric_candidate_mask(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """
    Build candidate mask using geometric and topological properties.
    """
    n = stats['n']
    
    # Base candidate count with geometric adjustment
    base_k = min(12, max(3, int(np.sqrt(n) + 2)))
    geo_factor = 0.5 * stats['geo_spread'] + 0.5 * stats['convexity']
    k = int(base_k * (1.0 + geo_factor))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # k-nearest neighbors
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < n - 1:
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Add MST edges for connectivity
    if n > 2:
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = (mst + mst.T) > 0
        candidate_mask = candidate_mask | mst_edges
    
    # Add bridge edges for clusters
    if stats['cluster_quality'] > 0.3:
        bridge_edges = _identify_cluster_bridges(distance_matrix, stats)
        candidate_mask = candidate_mask | bridge_edges
    
    # Ensure symmetry
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask


def _identify_cluster_bridges(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """
    Identify important edges connecting clusters.
    """
    n = stats['n']
    bridge_mask = np.zeros((n, n), dtype=bool)
    
    # Simple heuristic: connect each node to a few farthest nodes
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Add some farthest connections as potential bridges
        far_count = min(3, n // 10)
        if far_count > 0:
            far_indices = np.argsort(dists)[-far_count:]
            bridge_mask[i, far_indices] = True
    
    return bridge_mask


def _compute_geometric_factors(distance_matrix: np.ndarray, stats: Dict, 
                               candidate_mask: np.ndarray) -> Dict:
    """
    Compute comprehensive heuristic factors with geometric bonuses.
    """
    n = stats['n']
    factors = {}
    
    # 1. Inverse distance with geometric normalization
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    if np.any(inv_dist > 0):
        inv_dist = inv_dist / np.percentile(inv_dist[inv_dist > 0], 75)
    factors['inv_dist'] = inv_dist
    
    # 2. Enhanced regret with geometric considerations
    factors['regret'] = _compute_enhanced_regret(distance_matrix, stats, candidate_mask)
    
    # 3. Cluster-aware density scoring
    factors['density'] = _compute_cluster_density(distance_matrix, stats)
    
    # 4. MST with criticality analysis
    factors['mst'] = _compute_critical_mst_score(distance_matrix)
    
    # 5. Convex hull bonus for geometric edges
    factors['convex'] = _compute_convex_hull_bonus(distance_matrix, stats)
    
    return factors


def _compute_enhanced_regret(distance_matrix: np.ndarray, stats: Dict,
                             candidate_mask: np.ndarray) -> np.ndarray:
    """
    Compute regret scores with geometric path alternatives.
    """
    n = stats['n']
    regret = np.ones((n, n))
    
    # Adaptive parameters
    lookahead_depth = min(3, max(2, int(np.log2(n))))
    
    for i in range(n):
        i_candidates = np.where(candidate_mask[i])[0]
        
        for j in i_candidates:
            if j <= i or not candidate_mask[i, j]:
                continue
            
            direct_cost = distance_matrix[i, j]
            
            # Find best alternative paths through candidate network
            best_alternative = _find_best_alternative_path(distance_matrix, i, j, 
                                                          i_candidates, candidate_mask, 
                                                          lookahead_depth)
            
            if best_alternative < np.inf:
                # Adaptive regret ratio
                ratio = best_alternative / (direct_cost + 1e-8)
                
                if ratio < 1.0:  # Alternative is better
                    penalty = np.exp(-3 * (1.0 - ratio))
                else:  # Direct edge is better
                    bonus = 1.0 + np.log1p(ratio - 1.0) * 0.5
                    penalty = min(bonus, 2.0)
                
                regret[i, j] = regret[j, i] = penalty
            else:
                regret[i, j] = regret[j, i] = 1.0
    
    np.fill_diagonal(regret, 0)
    return regret


def _find_best_alternative_path(distance_matrix: np.ndarray, start: int, end: int,
                                candidates: np.ndarray, candidate_mask: np.ndarray,
                                depth: int) -> float:
    """
    Find the best alternative path using limited BFS through candidate network.
    """
    n = distance_matrix.shape[0]
    
    if depth == 1 or len(candidates) < 2:
        return np.inf
    
    best_cost = np.inf
    
    # Try one intermediate node
    for k in candidates:
        if k == end:
            continue
        if candidate_mask[k, end]:
            cost = distance_matrix[start, k] + distance_matrix[k, end]
            best_cost = min(best_cost, cost)
    
    # Try two intermediate nodes if depth > 2
    if depth >= 3:
        for k in candidates:
            if k == end:
                continue
            k_candidates = np.where(candidate_mask[k])[0]
            for l in k_candidates:
                if l == start or l == end:
                    continue
                if candidate_mask[l, end]:
                    cost = distance_matrix[start, k] + distance_matrix[k, l] + distance_matrix[l, end]
                    best_cost = min(best_cost, cost)
    
    return best_cost


def _compute_cluster_density(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """
    Compute density scores that adapt to cluster structure.
    """
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Adaptive density estimation
    k_local = min(5, max(2, n // 8))
    k_global = min(10, max(3, n // 4))
    
    local_density = np.zeros(n)
    global_density = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Local density (nearest neighbors)
        k_nearest = np.partition(dists, k_local)[:k_local]
        local_density[i] = np.mean(k_nearest)
        
        # Global density (regional)
        k_global_nearest = np.partition(dists, k_global)[:k_global]
        global_density[i] = np.mean(k_global_nearest)
    
    # Normalize and combine
    local_norm = (local_density - np.mean(local_density)) / (np.std(local_density) + 1e-8)
    global_norm = (global_density - np.mean(global_density)) / (np.std(global_density) + 1e-8)
    
    combined_density = 0.7 * local_norm + 0.3 * global_norm
    
    # Create edge scores
    for i in range(n):
        for j in range(i+1, n):
            # Favor connections between sparse nodes, penalize dense-dense connections
            if combined_density[i] > 0.5 and combined_density[j] > 0.5:
                # Both sparse regions - high priority
                score = 1.5
            elif combined_density[i] < -0.5 and combined_density[j] < -0.5:
                # Both dense regions - moderate priority
                score = 0.8
            elif abs(combined_density[i] - combined_density[j]) > 1.0:
                # Bridge connection - high priority
                score = 1.3
            else:
                # Similar moderate density
                score = 1.0
            
            density_scores[i, j] = density_scores[j, i] = score
    
    return density_scores


def _compute_critical_mst_score(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Compute MST scores with edge criticality analysis.
    """
    n = distance_matrix.shape[0]
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Basic MST edge boost
            mst_scores[mst_present] = 1.8
            
            # Analyze node degrees in MST
            degree = np.sum(mst_present, axis=1)
            
            # Enhance leaf connections
            for i in range(n):
                if degree[i] == 1:
                    j = np.where(mst_present[i])[0][0]
                    mst_scores[i, j] = mst_scores[j, i] = 2.2
                elif degree[i] == 2:
                    # Chain edges get moderate boost
                    neighbors = np.where(mst_present[i])[0]
                    for j in neighbors:
                        mst_scores[i, j] = mst_scores[j, i] = 1.9
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores


def _compute_convex_hull_bonus(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """
    Estimate convex hull edges and give them bonus scores.
    """
    n = stats['n']
    convex_scores = np.ones((n, n))
    
    if n < 4 or stats['geo_spread'] < 0.3:
        return convex_scores
    
    try:
        # Use MDS to estimate 2D coordinates
        from sklearn.manifold import MDS
        mds = MDS(n_components=2, dissimilarity='precomputed', random_state=42, max_iter=50)
        coords = mds.fit_transform(distance_matrix)
        
        # Find convex hull
        hull = ConvexHull(coords)
        
        # Mark hull edges
        for simplex in hull.simplices:
            i, j = simplex
            convex_scores[i, j] = convex_scores[j, i] = 1.5
        
        # Also boost edges that connect to hull vertices
        for vertex in hull.vertices:
            # Connect to a few nearest non-hull vertices
            dists = distance_matrix[vertex].copy()
            dists[vertex] = np.inf
            nearest = np.argsort(dists)[:min(3, n // 4)]
            for nb in nearest:
                if nb not in hull.vertices:
                    convex_scores[vertex, nb] = convex_scores[nb, vertex] = 1.3
    except:
        # Fallback: boost long edges (likely on hull)
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            far_indices = np.argsort(dists)[-min(3, n // 4):]
            for j in far_indices:
                if distance_matrix[i, j] > stats['p90']:
                    convex_scores[i, j] = convex_scores[j, i] = 1.4
    
    np.fill_diagonal(convex_scores, 0)
    return convex_scores


def _deep_predictive_simulation(distance_matrix: np.ndarray, stats: Dict,
                                candidate_mask: np.ndarray) -> np.ndarray:
    """
    Perform deep predictive simulations for edge evaluation.
    """
    n = stats['n']
    sim_scores = np.ones((n, n))
    
    if n < 4:
        return sim_scores
    
    # Adaptive simulation parameters
    n_simulations = min(30, max(5, n // 2))
    sim_depth = min(6, max(3, int(np.log2(n) * 2)))
    
    # Precompute promising edges for simulation
    promising_edges = []
    for i in range(n):
        row_candidates = np.where(candidate_mask[i])[0]
        row_candidates = row_candidates[row_candidates > i]
        for j in row_candidates:
            if distance_matrix[i, j] < stats['p75']:  # Focus on reasonable edges
                promising_edges.append((i, j))
    
    # Limit number of edges to simulate
    max_edges = min(200, n * (n-1) // 4)
    if len(promising_edges) > max_edges:
        promising_edges = random.sample(promising_edges, max_edges)
    
    # Parallel simulation (simulated with loops)
    for i, j in promising_edges:
        scores = []
        for _ in range(n_simulations):
            score = _simulate_extended_tour(distance_matrix, i, j, sim_depth, stats)
            scores.append(score)
        
        if scores:
            sim_scores[i, j] = sim_scores[j, i] = np.percentile(scores, 75)  # Use upper quartile
    
    # Normalize simulation scores
    max_score = np.max(sim_scores)
    if max_score > 0:
        sim_scores = sim_scores / max_score
    
    return sim_scores


def _simulate_extended_tour(distance_matrix: np.ndarray, start_i: int, start_j: int,
                            depth: int, stats: Dict) -> float:
    """
    Simulate extended tour building with multiple heuristics.
    """
    n = distance_matrix.shape[0]
    
    # Multiple simulation strategies
    strategies = ['greedy', 'regret', 'random']
    strategy = random.choice(strategies)
    
    visited = np.zeros(n, dtype=bool)
    visited[start_i] = visited[start_j] = True
    
    current = start_j
    path_length = distance_matrix[start_i, start_j]
    path_nodes = [start_i, start_j]
    
    steps_taken = 0
    
    while steps_taken < depth and np.sum(visited) < n:
        unvisited = np.where(~visited)[0]
        if len(unvisited) == 0:
            break
        
        if strategy == 'greedy':
            # Greedy selection
            dists = distance_matrix[current, unvisited]
            next_idx = np.argmin(dists)
            next_node = unvisited[next_idx]
            move_cost = dists[next_idx]
            
        elif strategy == 'regret':
            # Regret-based selection
            dists = distance_matrix[current, unvisited]
            sorted_idx = np.argsort(dists)
            if len(sorted_idx) >= 3:
                # Choose based on regret
                best_cost = dists[sorted_idx[0]]
                second_best = dists[sorted_idx[1]]
                regret = second_best - best_cost
                
                # Sometimes choose second best for diversity
                if random.random() < 0.3 and regret > best_cost * 0.1:
                    next_node = unvisited[sorted_idx[1]]
                    move_cost = second_best
                else:
                    next_node = unvisited[sorted_idx[0]]
                    move_cost = best_cost
            else:
                next_node = unvisited[sorted_idx[0]]
                move_cost = dists[sorted_idx[0]]
                
        else:  # random
            # Random with distance bias
            dists = distance_matrix[current, unvisited]
            weights = 1 / (dists + 1e-8)
            weights = weights / np.sum(weights)
            next_idx = np.random.choice(len(unvisited), p=weights)
            next_node = unvisited[next_idx]
            move_cost = dists[next_idx]
        
        path_length += move_cost
        visited[next_node] = True
        path_nodes.append(next_node)
        current = next_node
        steps_taken += 1
    
    # Complete the partial tour back to start
    if len(path_nodes) > 2:
        # Connect last node back to start_i
        final_cost = distance_matrix[current, start_i]
        path_length += final_cost
        
        # Estimate remaining tour completion
        remaining = np.where(~visited)[0]
        if len(remaining) > 0:
            # Estimate MST cost for remaining nodes plus connection
            if len(remaining) > 1:
                sub_dist = distance_matrix[np.ix_(remaining, remaining)]
                try:
                    mst = minimum_spanning_tree(sub_dist)
                    mst_cost = mst.sum()
                    
                    # Connect MST to current partial tour
                    min_connect = np.min(distance_matrix[current, remaining])
                    path_length += mst_cost + min_connect
                except:
                    # Fallback: nearest neighbor completion
                    for _ in range(len(remaining)):
                        dists = distance_matrix[current, remaining]
                        next_idx = np.argmin(dists)
                        path_length += dists[next_idx]
                        current = remaining[next_idx]
                        remaining = np.delete(remaining, next_idx)
    
    # Score is normalized inverse of path length
    expected_length = (len(path_nodes) + max(0, len(remaining))) * stats['mean_dist']
    score = expected_length / (path_length + 1e-8)
    
    return min(score, 4.0)  # Cap extreme values


def _optimize_weights_geometric(distance_matrix: np.ndarray, factors: Dict,
                                sim_scores: np.ndarray, stats: Dict) -> Dict:
    """
    Optimize factor weights using geometric properties.
    """
    n = stats['n']
    
    # Base weights from geometric properties
    base_weights = _compute_geometric_weights(stats)
    
    # Refine with simple grid search for larger instances
    if n >= 10:
        weights = _refine_weights_grid(distance_matrix, factors, sim_scores, base_weights, stats)
    else:
        weights = base_weights
    
    return weights


def _compute_geometric_weights(stats: Dict) -> Dict:
    """
    Compute base weights from geometric and topological properties.
    """
    weights = {
        'w_inv': 1.0,
        'w_regret': 0.8,
        'w_density': 0.5 + 0.3 * stats['cluster_quality'],
        'w_mst': 0.7 - 0.01 * stats['n'],
        'w_convex': 0.3 * stats['geo_spread'],
        'w_sim': 0.4 + 0.3 * stats['topo_connectivity']
    }
    
    # Adjust based on instance size
    size_factor = min(1.0, stats['n'] / 50)
    weights['w_sim'] *= size_factor
    weights['w_mst'] *= (1 - 0.3 * size_factor)
    
    # Normalize to maintain overall scale
    total = sum(weights.values())
    scale = 6.0 / total  # 6 factors
    
    for key in weights:
        weights[key] = weights[key] * scale
    
    return weights


def _refine_weights_grid(distance_matrix: np.ndarray, factors: Dict,
                         sim_scores: np.ndarray, base_weights: Dict,
                         stats: Dict) -> Dict:
    """
    Refine weights using simple grid search.
    """
    n = stats['n']
    
    # Generate candidate weight configurations
    candidates = []
    
    for _ in range(10):  # Limited number of configurations
        candidate = {}
        for key, value in base_weights.items():
            # Perturb each weight by ±20%
            perturbation = 1.0 + random.uniform(-0.2, 0.2)
            candidate[key] = max(0.1, value * perturbation)
        
        # Normalize
        total = sum(candidate.values())
        scale = 6.0 / total
        for key in candidate:
            candidate[key] = candidate[key] * scale
        
        candidates.append(candidate)
    
    # Evaluate candidates
    best_score = -np.inf
    best_weights = base_weights
    
    # Generate test tours for evaluation
    test_tours = _generate_test_tours(distance_matrix, 10)
    
    for candidate in candidates:
        score = _evaluate_weight_candidate(distance_matrix, factors, sim_scores,
                                          candidate, test_tours, stats)
        if score > best_score:
            best_score = score
            best_weights = candidate
    
    return best_weights


def _generate_test_tours(distance_matrix: np.ndarray, n_tours: int) -> List[np.ndarray]:
    """
    Generate diverse test tours for weight evaluation.
    """
    n = distance_matrix.shape[0]
    tours = []
    
    # Random tours
    for _ in range(n_tours // 2):
        tour = np.random.permutation(n)
        tours.append(tour)
    
    # Greedy tours from different starting points
    for start in range(min(5, n_tours // 2)):
        tour = np.zeros(n, dtype=int)
        visited = np.zeros(n, dtype=bool)
        
        current = start
        tour[0] = current
        visited[current] = True
        
        for i in range(1, n):
            dists = distance_matrix[current].copy()
            dists[visited] = np.inf
            
            if np.all(dists == np.inf):
                unvisited = np.where(~visited)[0]
                next_node = random.choice(unvisited)
            else:
                next_node = np.argmin(dists)
            
            tour[i] = next_node
            visited[next_node] = True
            current = next_node
        
        tours.append(tour)
    
    return tours


def _evaluate_weight_candidate(distance_matrix: np.ndarray, factors: Dict,
                               sim_scores: np.ndarray, weights: Dict,
                               test_tours: List[np.ndarray], stats: Dict) -> float:
    """
    Evaluate a weight candidate using test tours.
    """
    n = stats['n']
    
    # Build candidate mask for evaluation
    candidate_mask = _build_simple_candidate_mask(distance_matrix, stats)
    
    # Compute heuristic with given weights
    H_temp = _combine_factors_quick(factors, sim_scores, weights, candidate_mask)
    
    # Evaluate on test tours
    scores = []
    for tour in test_tours[:min(5, len(test_tours))]:
        tour_score = 0
        for k in range(n):
            i, j = tour[k], tour[(k+1) % n]
            if candidate_mask[i, j]:
                tour_score += H_temp[i, j]
        
        # Normalize by tour length
        tour_length = sum(distance_matrix[tour[k], tour[(k+1) % n]] for k in range(n))
        if tour_length > 0:
            scores.append(tour_score / tour_length)
    
    return np.mean(scores) if scores else 0.0


def _build_simple_candidate_mask(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """
    Build a simple candidate mask for weight evaluation.
    """
    n = stats['n']
    k = min(10, n - 1)
    
    mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < n - 1:
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        mask[i, indices] = True
    
    mask = mask | mask.T
    np.fill_diagonal(mask, False)
    return mask


def _combine_factors_quick(factors: Dict, sim_scores: np.ndarray,
                           weights: Dict, candidate_mask: np.ndarray) -> np.ndarray:
    """
    Quick combination of factors for weight evaluation.
    """
    n = factors['inv_dist'].shape[0]
    H = np.zeros((n, n))
    
    inv_dist = factors['inv_dist']
    regret = factors['regret']
    density = factors['density']
    mst = factors['mst']
    convex = factors['convex']
    
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                combined = (
                    (inv_dist[i, j] ** weights['w_inv']) *
                    (regret[i, j] ** weights['w_regret']) *
                    (density[i, j] ** weights['w_density']) *
                    (mst[i, j] ** weights['w_mst']) *
                    (convex[i, j] ** weights['w_convex']) *
                    (sim_scores[i, j] ** weights['w_sim'])
                )
                H[i, j] = H[j, i] = combined
    
    np.fill_diagonal(H, 0)
    return H


def _combine_all_factors(factors: Dict, sim_scores: np.ndarray,
                         weights: Dict, candidate_mask: np.ndarray) -> np.ndarray:
    """
    Combine all factors using weighted geometric mean.
    """
    n = factors['inv_dist'].shape[0]
    H = np.zeros((n, n))
    
    inv_dist = factors['inv_dist']
    regret = factors['regret']
    density = factors['density']
    mst = factors['mst']
    convex = factors['convex']
    
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                combined = (
                    (inv_dist[i, j] ** weights['w_inv']) *
                    (regret[i, j] ** weights['w_regret']) *
                    (density[i, j] ** weights['w_density']) *
                    (mst[i, j] ** weights['w_mst']) *
                    (convex[i, j] ** weights['w_convex']) *
                    (sim_scores[i, j] ** weights['w_sim'])
                )
                H[i, j] = H[j, i] = combined
            else:
                # Non-candidate edges get very low score
                H[i, j] = H[j, i] = 1e-6
    
    np.fill_diagonal(H, 0)
    return H


def _topological_refinement(H: np.ndarray, distance_matrix: np.ndarray,
                            stats: Dict, candidate_mask: np.ndarray) -> np.ndarray:
    """
    Apply topological refinement to the heuristic matrix.
    """
    n = stats['n']
    
    # Ensure minimum connectivity
    min_edges = min(3, n - 1)
    for i in range(n):
        non_zero = np.where(H[i] > 1e-6)[0]
        if len(non_zero) < min_edges:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] <= 1e-6:
                    base_score = 1 / (dists[j] + 1e-8)
                    H[i, j] = H[j, i] = base_score
    
    # Symmetrize
    H = (H + H.T) / 2
    
    # Apply topological smoothing
    if n >= 5:
        H_smooth = np.zeros((n, n))
        for i in range(n):
            neighbors = np.where(candidate_mask[i])[0]
            if len(neighbors) > 0:
                for j in neighbors:
                    if j > i:
                        # Consider common neighbors
                        common = np.intersect1d(
                            np.where(candidate_mask[i])[0],
                            np.where(candidate_mask[j])[0]
                        )
                        if len(common) > 0:
                            # Boost edges that are part of triangles
                            triangle_boost = len(common) / min(5, n)
                            H_smooth[i, j] = H_smooth[j, i] = H[i, j] * (1 + 0.2 * triangle_boost)
                        else:
                            H_smooth[i, j] = H_smooth[j, i] = H[i, j]
        
        # Use smoothing where available
        mask_smooth = H_smooth > 0
        H[mask_smooth] = 0.7 * H[mask_smooth] + 0.3 * H_smooth[mask_smooth]
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    # Apply sigmoid to emphasize strong candidates
    H = 1 / (1 + np.exp(-6 * (H - 0.5)))
    
    np.fill_diagonal(H, 0)
    return H
