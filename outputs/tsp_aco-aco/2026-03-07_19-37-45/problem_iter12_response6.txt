def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic with adaptive multi-hop regret, non-linear spectral embedding,
    meta-weighted ensemble combination, and contextual edge criticality analysis.
    
    Returns a heuristic matrix where higher values indicate more promising edges.
    """

        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Comprehensive instance analysis
    stats = _analyze_instance_v5(distance_matrix)
    
    # Step 2: Compute base factors with improved embeddings
    factors = _compute_factors_v5(distance_matrix, stats)
    
    # Step 3: Contextual edge criticality analysis
    criticality_scores = _compute_contextual_criticality(distance_matrix, stats, factors)
    
    # Step 4: Adaptive multi-hop regret with reinforcement-inspired weighting
    regret_scores = _compute_adaptive_regret_v5(distance_matrix, stats, factors, criticality_scores)

    weights = _compute_meta_weights_v5(stats, factors, regret_scores, criticality_scores)

                           criticality_scores, weights, candidate_mask)
    
    # Step 8: Adaptive post-processing with iterative refinement
    H = _post_process_v5(H, distance_matrix, stats, factors, weights)

def _analyze_instance_v5(distance_matrix: np.ndarray) -> Dict:
    """Comprehensive instance analysis with graph-theoretic and geometric properties."""
    n = distance_matrix.shape[0]
    
    # Basic distance statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Robust percentile analysis
    percentiles = np.percentile(dist_no_diag, [10, 25, 50, 75, 90])
    p10, p25, p50, p75, p90 = percentiles
    iqr = p75 - p25
    
    # Distance distribution skewness and kurtosis
    skewness = (p90 - p50) / (p50 - p10 + 1e-8)
    
    # Nearest neighbor statistics
    nn_ratios = []
    nn_variability = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if len(dists) >= 3:
            sorted_dists = np.sort(dists)
            nn_ratios.append(sorted_dists[1] / (sorted_dists[0] + 1e-8))
            nn_variability.append(sorted_dists[2] / sorted_dists[1] if sorted_dists[1] > 0 else 1.0)
    
    regularity = np.mean(nn_ratios) if nn_ratios else 1.0
    variability = np.mean(nn_variability) if nn_variability else 1.0
    
    # Graph connectivity analysis
    threshold = p50 * 1.2
    adj = (distance_matrix <= threshold).astype(float)
    np.fill_diagonal(adj, 0)
    
    # Clustering coefficient
    clustering_coeffs = []
    for i in range(n):
        neighbors = np.where(adj[i] > 0)[0]
        if len(neighbors) >= 2:
            subgraph = adj[neighbors][:, neighbors]
            triangles = np.sum(subgraph) / 2
            possible = len(neighbors) * (len(neighbors) - 1) / 2
            clustering_coeffs.append(triangles / possible if possible > 0 else 0)
    avg_clustering = np.mean(clustering_coeffs) if clustering_coeffs else 0.0
    
    # Spectral analysis for structure detection
    spectral_properties = _analyze_spectral_properties(distance_matrix, n)
    
    # Geometric spread estimation
    spread_ratio = np.max(dist_no_diag) / (np.min(dist_no_diag[dist_no_diag > 0]) + 1e-8)
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'regularity': regularity,
        'variability': variability,
        'skewness': skewness,
        'clustering': avg_clustering,
        'iqr': iqr,
        'p10': p10,
        'p25': p25,
        'p50': p50,
        'p75': p75,
        'p90': p90,
        'spread_ratio': spread_ratio,
        **spectral_properties
    }

def _analyze_spectral_properties(distance_matrix: np.ndarray, n: int) -> Dict:
    """Analyze spectral properties of the distance matrix."""
    spectral_gap = 0.1
    spectral_clustering = 0.0
    
    if 10 <= n <= 500:
        try:
            # Normalized Laplacian for spectral analysis
            W = np.exp(-distance_matrix / np.median(distance_matrix[distance_matrix > 0]))
            np.fill_diagonal(W, 0)
            D = np.diag(np.sum(W, axis=1))
            D_inv_sqrt = np.diag(1 / np.sqrt(np.diag(D) + 1e-8))
            L_norm = np.eye(n) - D_inv_sqrt @ W @ D_inv_sqrt
            
            # Compute eigenvalues
            eigvals = eigsh(L_norm, k=min(4, n-1), which='SM', return_eigenvectors=False)
            if len(eigvals) >= 2:
                spectral_gap = eigvals[1] - eigvals[0]
                spectral_clustering = 1.0 if eigvals[1] > 0.5 else 0.0
        except:
            pass
    
    return {
        'spectral_gap': spectral_gap,
        'spectral_clustering': spectral_clustering
    }

def _compute_factors_v5(distance_matrix: np.ndarray, stats: Dict) -> Dict:
    """Compute base heuristic factors with improved embeddings."""
    n = stats['n']
    factors = {}
    
    # 1. Inverse distance with adaptive scaling
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    # Scale by instance characteristics
    scaling = 1.0 + 0.5 * stats['clustering'] - 0.3 * stats['regularity']
    inv_dist = np.power(inv_dist, scaling)
    
    max_inv = np.max(inv_dist)
    if max_inv > 0:
        inv_dist = inv_dist / max_inv
    factors['inv_dist'] = inv_dist
    
    # 2. Non-linear spectral embedding with kernel PCA
    factors['spectral_embedding'] = _compute_spectral_embedding_v3(distance_matrix, stats)
    
    # 3. MST-based scoring with contextual importance
    factors['mst_contextual'] = _compute_mst_contextual_scores(distance_matrix, stats)
    
    # 4. Local-global centrality fusion
    factors['centrality_fusion'] = _compute_centrality_fusion(distance_matrix, stats)
    
    return factors

def _compute_spectral_embedding_v3(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Non-linear spectral embedding using RBF kernel and Laplacian Eigenmaps."""
    n = stats['n']
    embedding_scores = np.ones((n, n))
    
    if n < 8:
        return embedding_scores
    
    try:
        # Create similarity matrix with adaptive bandwidth
        bandwidth = stats['p50'] * 0.5
        similarity = np.exp(-distance_matrix ** 2 / (2 * bandwidth ** 2))
        np.fill_diagonal(similarity, 0)
        
        # Degree matrix
        D = np.diag(np.sum(similarity, axis=1))
        
        # Normalized Laplacian for embedding
        D_inv_sqrt = np.diag(1 / np.sqrt(np.diag(D) + 1e-8))
        L_sym = np.eye(n) - D_inv_sqrt @ similarity @ D_inv_sqrt
        
        # Embed in low-dimensional space
        k = min(4, n - 1)
        eigvals, eigvecs = eigsh(L_sym, k=k, which='SM')
        
        if k >= 2:
            # Use first k eigenvectors (excluding constant vector)
            embedding = eigvecs[:, 1:k]
            
            # Compute pairwise distances in embedding space
            for i in range(n):
                for j in range(i+1, n):
                    diff = embedding[i] - embedding[j]
                    dist_emb = np.sqrt(np.sum(diff ** 2))
                    
                    # Convert to similarity score
                    score = np.exp(-dist_emb)
                    embedding_scores[i, j] = embedding_scores[j, i] = score
            
            # Normalize
            max_score = np.max(embedding_scores)
            if max_score > 0:
                embedding_scores = embedding_scores / max_score
    
    except Exception:
        # Fallback to simpler method
        embedding_scores = _compute_simple_spectral_embedding(distance_matrix, stats)
    
    return embedding_scores

def _compute_simple_spectral_embedding(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Simple spectral embedding fallback."""
    n = stats['n']
    scores = np.ones((n, n))
    
    # Use distance matrix directly with Gaussian kernel
    similarity = np.exp(-distance_matrix / stats['p50'])
    np.fill_diagonal(similarity, 0)
    
    # Row-wise normalization
    row_sums = np.sum(similarity, axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    scores = similarity / row_sums
    
    # Symmetrize
    scores = (scores + scores.T) / 2
    np.fill_diagonal(scores, 0)
    
    return scores

def _compute_mst_contextual_scores(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """MST scoring with contextual edge importance."""
    n = stats['n']
    mst_scores = np.ones((n, n))
    
    if n <= 2:
        return mst_scores
    
    try:
        # Compute MST
        mst = minimum_spanning_tree(distance_matrix)
        mst_array = mst.toarray()
        mst_present = (mst_array + mst_array.T) > 0
        
        # Base score for MST edges
        mst_scores[mst_present] = 1.8
        
        # Contextual analysis
        if n >= 10:
            # Compute edge betweenness in MST
            G = nx.Graph()
            for i in range(n):
                for j in range(i+1, n):
                    if mst_present[i, j]:
                        G.add_edge(i, j, weight=distance_matrix[i, j])
            
            # Approximate betweenness centrality
            try:
                betweenness = nx.edge_betweenness_centrality(G, weight='weight', normalized=True)
                for (i, j), val in betweenness.items():
                    # Boost high-betweenness edges (critical connections)
                    mst_scores[i, j] = mst_scores[j, i] = 2.0 + val * 0.8
            except:
                pass
            
            # Degree-based analysis
            node_degrees = np.sum(mst_present, axis=1)
            
            # Boost connections to high-degree nodes (hubs)
            for i in range(n):
                if node_degrees[i] >= 3:
                    neighbors = np.where(mst_present[i])[0]
                    for j in neighbors:
                        if node_degrees[j] >= 2:
                            mst_scores[i, j] = max(mst_scores[i, j], 2.4)
    
    except Exception:
        # Simple MST scoring as fallback
        for i in range(n):
            for j in range(i+1, n):
                mst_scores[i, j] = mst_scores[j, i] = 1.5
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores

def _compute_centrality_fusion(distance_matrix: np.ndarray, stats: Dict) -> np.ndarray:
    """Fusion of local and global centrality measures."""
    n = stats['n']
    centrality_scores = np.ones((n, n))
    
    # Local centrality: inverse distance sum
    local_centrality = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        local_centrality[i] = np.sum(1 / (dists + 1e-8))
    
    # Global centrality: closeness approximation
    global_centrality = np.zeros(n)
    if n > 10:
        # Sample-based closeness approximation
        sample_size = min(10, n)
        sampled_nodes = np.random.choice(range(n), sample_size, replace=False)
        
        for i in range(n):
            total_dist = 0
            count = 0
            for s in sampled_nodes:
                if s != i:
                    # A* search approximation for larger instances
                    if n > 50:
                        # Use 2-hop approximation
                        dists_i = distance_matrix[i].copy()
                        dists_i[i] = np.inf
                        nearest_i = np.argmin(dists_i)
                        approx_dist = distance_matrix[i, nearest_i] + distance_matrix[nearest_i, s]
                        total_dist += min(approx_dist, distance_matrix[i, s])
                    else:
                        total_dist += distance_matrix[i, s]
                    count += 1
            
            if count > 0:
                global_centrality[i] = count / (total_dist + 1e-8)
    
    # Normalize centralities
    if np.max(local_centrality) > 0:
        local_centrality = local_centrality / np.max(local_centrality)
    if np.max(global_centrality) > 0:
        global_centrality = global_centrality / np.max(global_centrality)
    
    # Fuse with adaptive weights
    fusion_weight = 0.3 + 0.4 * stats['clustering']  # More global for clustered instances
    
    for i in range(n):
        for j in range(i+1, n):
            local_score = (local_centrality[i] + local_centrality[j]) / 2
            global_score = (global_centrality[i] + global_centrality[j]) / 2
            
            # Adaptive fusion
            fused_score = (1 - fusion_weight) * local_score + fusion_weight * global_score
            centrality_scores[i, j] = centrality_scores[j, i] = fused_score
    
    return centrality_scores

def _compute_contextual_criticality(distance_matrix: np.ndarray, 
                                   stats: Dict, 
                                   factors: Dict) -> np.ndarray:
    """Contextual edge criticality analysis considering multiple factors."""
    n = stats['n']
    criticality = np.ones((n, n))
    
    if n < 4:
        return criticality
    
    # Create thresholded graph for connectivity analysis
    threshold = stats['p75']
    adj = (distance_matrix <= threshold).astype(float)
    np.fill_diagonal(adj, 0)
    
    # Compute basic betweenness with sampling
    sample_size = min(15, n)
    if sample_size >= 3:
        sampled_nodes = np.random.choice(range(n), sample_size, replace=False)
        
        for s in sampled_nodes:
            for t in sampled_nodes:
                if s >= t:
                    continue
                
                try:
                    # Find shortest path in thresholded graph
                    G = nx.from_numpy_array(adj)
                    try:
                        path = nx.shortest_path(G, s, t)
                        for k in range(len(path)-1):
                            i, j = path[k], path[k+1]
                            criticality[i, j] += 1.0
                            criticality[j, i] += 1.0
                    except:
                        continue
                except:
                    continue
    
    # Incorporate other factors for contextual criticality
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] > 3 * stats['mean_dist']:
                criticality[i, j] *= 0.5
            
            # Boost criticality for edges that are important in multiple factors
            factor_agreement = 0
            if factors['inv_dist'][i, j] > np.percentile(factors['inv_dist'][factors['inv_dist'] > 0], 70):
                factor_agreement += 1
            if factors['spectral_embedding'][i, j] > np.percentile(factors['spectral_embedding'][factors['spectral_embedding'] > 0], 70):
                factor_agreement += 1
            if factors['mst_contextual'][i, j] > 1.5:
                factor_agreement += 1
            
            if factor_agreement >= 2:
                criticality[i, j] *= (1.0 + 0.3 * factor_agreement)
    
    # Normalize
    max_crit = np.max(criticality)
    if max_crit > 0:
        criticality = criticality / max_crit
    
    return criticality

def _compute_adaptive_regret_v5(distance_matrix: np.ndarray, 
                               stats: Dict, 
                               factors: Dict,
                               criticality_scores: np.ndarray) -> np.ndarray:
    """Adaptive multi-hop regret analysis with reinforcement-inspired exploration."""
    n = stats['n']
    regret_scores = np.ones((n, n))
    
    if n < 4:
        return regret_scores
    
    # Determine regret horizon adaptively
    horizon = _determine_regret_horizon(stats)
    
    # Precompute candidate alternatives for each node
    alternatives = _precompute_alternatives(distance_matrix, stats, horizon)
    
    for i in range(n):
        for j in range(i+1, n):
            direct_cost = distance_matrix[i, j]
            
            # Skip very long edges
            if direct_cost > 4 * stats['mean_dist']:
                regret_scores[i, j] = 0.1
                continue
            
            # Explore alternatives
            best_alternative = float('inf')
            
            if horizon >= 1 and i in alternatives and j in alternatives:
                # Check 1-hop alternatives
                for k in alternatives[i]['1hop']:
                    if k != j:
                        alt = distance_matrix[i, k] + distance_matrix[k, j]
                        best_alternative = min(best_alternative, alt)
                
                # Check 2-hop alternatives if horizon >= 2
                if horizon >= 2:
                    for u in alternatives[i]['1hop'][:5]:  # Top 5 from i
                        if u == j:
                            continue
                        for v in alternatives[j]['1hop'][:5]:  # Top 5 to j
                            if v == i or v == u:
                                continue
                            alt = distance_matrix[i, u] + distance_matrix[u, v] + distance_matrix[v, j]
                            best_alternative = min(best_alternative, alt)
            
            # Compute regret score with adaptive transformation
            if direct_cost > 0 and best_alternative < float('inf'):
                regret_ratio = best_alternative / direct_cost
                
                # Sigmoid transformation for smooth scoring
                if regret_ratio > 1.0:
                    score = 1.0 / (1.0 + np.exp(3.0 * (regret_ratio - 1.2)))
                else:
                    score = 1.2  # Bonus for edges better than alternatives
                
                # Adjust based on criticality
                if criticality_scores[i, j] > 0.7:
                    score *= 1.1
                
                regret_scores[i, j] = score
    
    # Symmetrize and normalize
    regret_scores = (regret_scores + regret_scores.T) / 2
    np.fill_diagonal(regret_scores, 0)
    
    max_val = np.max(regret_scores)
    if max_val > 0:
        regret_scores = regret_scores / max_val
    
    return regret_scores

def _determine_regret_horizon(stats: Dict) -> int:
    """Determine optimal regret horizon based on instance characteristics."""
    n = stats['n']
    clustering = stats['clustering']
    regularity = stats['regularity']
    
    if n < 10:
        return 1
    elif n < 30:
        return 2
    else:
        # Use more hops for larger, less clustered instances
        if clustering < 0.3 and regularity > 1.5:
            return 3
        else:
            return 2

    n = stats['n']
    alternatives = {}
    
    k_nn = min(15, n - 1)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Get k nearest neighbors
        if k_nn < len(dists):
            nearest = np.argpartition(dists, k_nn)[:k_nn]
        else:
            nearest = [j for j in range(n) if j != i]
        
        alternatives[i] = {
            '1hop': nearest,
            'dist': dists
        }
    
    return alternatives

def _compute_meta_weights_v5(stats: Dict, 
                            factors: Dict, 
                            regret_scores: np.ndarray,
                            criticality_scores: np.ndarray) -> Dict:
    """Compute adaptive weights using instance meta-features and factor consistency."""
    n = stats['n']
    cv = stats['cv_dist']
    clustering = stats['clustering']
    regularity = stats['regularity']
    spectral_gap = stats['spectral_gap']
    
    # Base weights with instance adaptation
    weights = {
        'w_inv': 0.9 + 0.2 * (1.0 - regularity),  # More weight for irregular instances
        
        # Spectral embedding weight: depends on spectral structure
        'w_spectral': 0.4 + 0.5 * spectral_gap + 0.3 * clustering,
        
        # MST contextual weight: decreases with size, increases with clustering
        'w_mst': max(0.2, 0.7 - 0.01 * n) * (1.0 + 0.5 * clustering),
        
        # Centrality fusion weight
        'w_centrality': 0.3 + 0.4 * clustering - 0.2 * regularity,
        
        # Regret weight: higher for instances with clear alternatives
        'w_regret': 0.5 + 0.3 * regularity - 0.2 * clustering,
        
        # Criticality weight: higher for structured instances
        'w_criticality': 0.2 + 0.6 * spectral_gap
    }
    
    # Adjust based on coefficient of variation
    if cv < 0.15:  # Very uniform
        weights['w_regret'] *= 1.5
        weights['w_spectral'] *= 0.7
    elif cv > 0.7:  # Highly variable
        weights['w_regret'] *= 0.8
        weights['w_criticality'] *= 1.4
        weights['w_centrality'] *= 1.3
    
    # Factor consistency adjustment
    factor_consistency = _measure_factor_consistency(factors, regret_scores, criticality_scores)
    for key in weights:
        weights[key] *= (0.8 + 0.4 * factor_consistency)
    
    # Ensure non-negative and normalize
    for key in weights:
        weights[key] = max(0.1, weights[key])
    
    # Normalize for balanced influence
    total = sum(weights.values())
    scale = 6.0 / total if total > 0 else 1.0
    for key in weights:
        weights[key] *= scale
    
    return weights

def _measure_factor_consistency(factors: Dict, 
                               regret_scores: np.ndarray,
                               criticality_scores: np.ndarray) -> float:
    """Measure consistency between different heuristic factors."""
    n = factors['inv_dist'].shape[0]
    
    # Sample edges to measure consistency
    sample_size = min(100, n * (n-1) // 2)
    edges = []
    
    # Generate random edges
    for _ in range(sample_size):
        i, j = np.random.randint(0, n, 2)
        while i == j:
            i, j = np.random.randint(0, n, 2)
        edges.append((i, j))
    
    # Compute scores for each factor
    factor_scores = []
    for i, j in edges:
        scores = [
            factors['inv_dist'][i, j],
            factors['spectral_embedding'][i, j],
            factors['mst_contextual'][i, j],
            factors['centrality_fusion'][i, j],
            regret_scores[i, j],
            criticality_scores[i, j]
        ]
        factor_scores.append(scores)
    
    # Measure correlation between factors
    if len(factor_scores) > 10:
        factor_scores_array = np.array(factor_scores)
        correlations = np.corrcoef(factor_scores_array.T)
        
        # Average absolute correlation (excluding diagonal)
        mask = ~np.eye(correlations.shape[0], dtype=bool)
        avg_correlation = np.mean(np.abs(correlations[mask]))
        
        # Convert to consistency measure (higher = more consistent)
        consistency = min(1.0, avg_correlation * 1.5)
        return consistency
    
    return 0.5

def _build_candidate_mask_v5(distance_matrix: np.ndarray,
                            stats: Dict,
                            factors: Dict,
                            regret_scores: np.ndarray,
                            criticality_scores: np.ndarray,
                            weights: Dict) -> np.ndarray:
    """Build candidate mask using ensemble agreement and adaptive thresholds."""
    n = stats['n']
    
    # Base candidate size
    base_k = min(15, max(5, int(np.log2(n) * 2.5)))
    
    # Adjust based on instance characteristics
    k = int(base_k * (1.0 + 0.3 * stats['clustering'] - 0.2 * stats['regularity']))
    k = max(3, min(k, n - 1))
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # Compute ensemble scores for initial candidate selection
    ensemble_scores = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            ensemble_scores[i, j] = ensemble_scores[j, i] = (
                factors['inv_dist'][i, j] ** weights['w_inv'] *
                factors['spectral_embedding'][i, j] ** weights['w_spectral'] *
                regret_scores[i, j] ** weights['w_regret']
            )
    
    # Phase 1: Select candidates based on ensemble scores
    for i in range(n):
        # Top k ensemble scores
        scores_i = ensemble_scores[i].copy()
        scores_i[i] = -np.inf
        
        top_k = min(k, n-1)
        if top_k > 0:
            threshold = np.partition(scores_i, -top_k)[-top_k]
            candidate_mask[i, scores_i >= threshold] = True
        
        # Always include top regret edges
        regret_i = regret_scores[i].copy()
        regret_i[i] = -np.inf
        top_regret = np.argpartition(regret_i, -min(3, n-1))[-min(3, n-1):]
        candidate_mask[i, top_regret] = True
        
        # Include high criticality edges
        crit_i = criticality_scores[i].copy()
        crit_i[i] = -np.inf
        top_crit = np.argpartition(crit_i, -min(2, n-1))[-min(2, n-1):]
        candidate_mask[i, top_crit] = True
    
    # Ensure symmetry
    candidate_mask = candidate_mask | candidate_mask.T
    
    # Phase 2: Ensure minimum connectivity
    min_edges = max(3, min(6, n // 3))
    for i in range(n):
        if np.sum(candidate_mask[i]) < min_edges:
            # Add most promising edges based on weighted combination
            combined = np.zeros(n)
            for j in range(n):
                if j != i and not candidate_mask[i, j]:
                    combined[j] = ensemble_scores[i, j]
            
            needed = min_edges - np.sum(candidate_mask[i])
            if needed > 0:
                top_candidates = np.argpartition(combined, -needed)[-needed:]
                for j in top_candidates:
                    if j != i and combined[j] > 0:
                        candidate_mask[i, j] = candidate_mask[j, i] = True
    
    np.fill_diagonal(candidate_mask, False)
    return candidate_mask

def _combine_factors_v5(distance_matrix: np.ndarray,
                       stats: Dict,
                       factors: Dict,
                       regret_scores: np.ndarray,
                       criticality_scores: np.ndarray,
                       weights: Dict,
                       candidate_mask: np.ndarray) -> np.ndarray:
    """Combine all factors using weighted geometric mean with adaptive blending."""
    n = stats['n']
    H = np.zeros((n, n))
    
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Geometric mean of all factors
                combined = (
                    factors['inv_dist'][i, j] ** weights['w_inv'] *
                    factors['spectral_embedding'][i, j] ** weights['w_spectral'] *
                    factors['mst_contextual'][i, j] ** weights['w_mst'] *
                    factors['centrality_fusion'][i, j] ** weights['w_centrality'] *
                    regret_scores[i, j] ** weights['w_regret'] *
                    criticality_scores[i, j] ** weights['w_criticality']
                )
                
                H[i, j] = H[j, i] = combined
    
    return H

def _post_process_v5(H: np.ndarray,
                    distance_matrix: np.ndarray,
                    stats: Dict,
                    factors: Dict,
                    weights: Dict) -> np.ndarray:
    """Adaptive post-processing with iterative refinement."""
    n = stats['n']
    
    # Ensure symmetry and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Adaptive sparsification
    if n > 12:
        # Keep adaptive number of edges per node
        k_keep = max(3, min(10, int(n ** 0.6)))
        H_filtered = np.zeros_like(H)
        
        for i in range(n):
            row = H[i].copy()
            row[i] = -np.inf
            
            non_zero = np.where(row > 0)[0]
            if len(non_zero) > k_keep:
                threshold = np.partition(row, -k_keep)[-k_keep]
                indices = np.where(row >= threshold)[0]
                H_filtered[i, indices] = row[indices]
                H_filtered[indices, i] = row[indices]
            else:
                H_filtered[i, row > 0] = row[row > 0]
                H_filtered[row > 0, i] = row[row > 0]
        
        H = H_filtered
    
    # Ensure minimum connectivity
    min_connections = max(2, min(4, n // 5))
    for i in range(n):
        non_zero = np.where(H[i] > 0)[0]
        if len(non_zero) < min_connections:
            # Score potential edges using simplified regret
            edge_scores = []
            for j in range(n):
                if j != i and H[i, j] == 0:
                    # Quick regret estimate
                    dists_i = distance_matrix[i].copy()
                    dists_i[i] = np.inf
                    if len(dists_i) > 0:
                        best_alt = np.min(dists_i)
                        regret = (distance_matrix[i, j] - best_alt) / (best_alt + 1e-8)
                        
                        # Score based on inverse distance and regret
                        score = (1 / (distance_matrix[i, j] + 1e-8)) * np.exp(-regret)
                        edge_scores.append((j, score))
            
            # Add best edges
            edge_scores.sort(key=lambda x: x[1], reverse=True)
            to_add = min(min_connections - len(non_zero), len(edge_scores))
            for idx in range(to_add):
                j, score = edge_scores[idx]
                H[i, j] = H[j, i] = score
    
    # Final normalization with power transformation
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    # Contextual power transformation
    if stats['regularity'] > 1.2:
        power_exp = 0.8  # More emphasis on top edges for regular instances
    else:
        power_exp = 1.0  # Linear for irregular instances
    
    H = np.power(H, power_exp)
    
    # Final cleanup
    H[H < 1e-8] = 0
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    return H
