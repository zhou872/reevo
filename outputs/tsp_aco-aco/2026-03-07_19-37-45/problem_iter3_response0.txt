import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Optimized heuristic for TSP edge selection combining regret-based prioritization,
    degree-aware penalization, two-hop lookahead, and adaptive sparsity.
    Returns a matrix of same shape with higher values indicating more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    H = np.zeros((n, n))
    
    # 1. Compute adaptive neighborhood size
    k = max(3, min(10, int(np.log2(n) * 2)))
    
    # 2. Regret-based edge prioritization
    regret_matrix = np.zeros((n, n))
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Find best and second-best distances
        sorted_idx = np.argsort(dists)
        if len(sorted_idx) >= 2:
            best_dist = dists[sorted_idx[0]]
            second_best = dists[sorted_idx[1]]
            base_regret = (second_best - best_dist) / (best_dist + 1e-8)
            
            # Assign regret scores to edges
            for idx, j in enumerate(sorted_idx):
                if j != i:
                    edge_regret = max(0, dists[j] - best_dist) / (best_dist + 1e-8)
                    # Higher score for edges with lower regret (normalized to [0,1])
                    regret_matrix[i, j] = 1 / (1 + edge_regret)
    
    # Symmetrize regret matrix
    regret_matrix = (regret_matrix + regret_matrix.T) / 2
    
    # 3. Degree-aware penalization
    degree_penalty = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        best_dist = np.min(dists)
        # Count edges within 20% of best distance
        good_edges = np.sum(dists <= best_dist * 1.2)
        degree_penalty[i] = max(1, good_edges)  # Avoid division by zero
    
    # 4. Two-hop lookahead penalty
    two_hop_penalty = np.ones((n, n))
    mean_dist = np.mean(distance_matrix[~np.eye(n, dtype=bool)])
    
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] > 2 * mean_dist:
                continue
                
            # Find best intermediate node
            min_two_hop = np.inf
            for k in range(n):
                if k != i and k != j:
                    two_hop_cost = distance_matrix[i, k] + distance_matrix[k, j]
                    min_two_hop = min(min_two_hop, two_hop_cost)
            
            if min_two_hop < np.inf:
                # Penalize if two-hop path is significantly shorter
                if min_two_hop < distance_matrix[i, j] * 0.7:
                    two_hop_penalty[i, j] = 0.3
                elif min_two_hop < distance_matrix[i, j] * 0.9:
                    two_hop_penalty[i, j] = 0.7
    
    two_hop_penalty = (two_hop_penalty + two_hop_penalty.T) / 2
    
    # 5. Adaptive candidate selection (sparse computation)
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Select k-nearest neighbors
        if k < n-1:
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
        
        # Add a few random connections for exploration
        if n > 10:
            remaining = [j for j in range(n) if j != i and not candidate_mask[i, j]]
            if len(remaining) > 0:
                n_random = max(1, int(np.log2(n)))
                random_choices = np.random.choice(remaining, 
                                                min(n_random, len(remaining)), 
                                                replace=False)
                candidate_mask[i, random_choices] = True
    
    # Make symmetric
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    # 6. Combine factors for candidate edges
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Base inverse distance
                inv_dist = 1 / (distance_matrix[i, j] + 1e-8)
                
                # Regret component (geometric mean of both directions)
                regret_score = np.sqrt(regret_matrix[i, j] * regret_matrix[j, i])
                
                # Degree penalty component
                deg_penalty = 1 / np.sqrt(degree_penalty[i] * degree_penalty[j])
                
                # Two-hop penalty
                hop_penalty = two_hop_penalty[i, j]
                
                # Combine factors multiplicatively
                combined = inv_dist * regret_score * deg_penalty * hop_penalty
                
                H[i, j] = H[j, i] = combined
    
    # 7. MST boost for critical edges
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Boost MST edges that are already candidates
            boost_mask = candidate_mask & mst_present
            H[boost_mask] *= 2.0
        except:
            pass
    
    # 8. Post-processing
    np.fill_diagonal(H, 0)
    
    # Normalize non-zero entries
    non_zero = H[H > 0]
    if len(non_zero) > 0:
        max_val = np.max(non_zero)
        if max_val > 0:
            H = H / max_val
    
    # Ensure minimum connectivity
    min_edges = min(3, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i])[0]
        if len(non_zero) < min_edges:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            for j in nearest:
                if H[i, j] == 0:
                    H[i, j] = H[j, i] = 1 / (distance_matrix[i, j] + 1e-8)
    
    return H
