import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with multi-hop regret decomposition
    and adaptive sparsification. Returns a matrix of the same shape where higher
    values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Analyze instance characteristics
    stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Compute decomposed factors
    factors = _compute_decomposed_factors(distance_matrix, stats)
    
    # Step 3: Dynamic candidate sparsification
    candidate_mask = _dynamic_candidate_sparsification(distance_matrix, stats, factors)
    
    # Step 4: Compute adaptive weights
    weights = _compute_adaptive_weights_v2(stats, distance_matrix)
    
    # Step 5: Geometric combination with sparsification
    H = np.zeros((n, n))
    
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Multiplicative combination with adaptive exponents
                combined = (
                    (factors['inv_dist'][i, j] ** weights['w_inv']) *
                    (factors['regret_1hop'][i, j] ** weights['w_reg1']) *
                    (factors['regret_2hop'][i, j] ** weights['w_reg2']) *
                    (factors['spectral_density'][i, j] ** weights['w_spectral']) *
                    (factors['two_step_penalty'][i, j] ** weights['w_penalty']) *
                    (factors['mst'][i, j] ** weights['w_mst'])
                )
                H[i, j] = H[j, i] = combined
    
    # Step 6: Post-processing with enhanced sparsification
    H = _post_process_sparse(H, distance_matrix, stats)
    
    return H

def _analyze_instance_v2(distance_matrix: np.ndarray) -> dict:
    """Enhanced instance analysis with spectral properties."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal for statistics
    mask = ~np.eye(n, dtype=bool)
    dist_no_diag = distance_matrix[mask]
    
    # Basic statistics
    mean_dist = np.mean(dist_no_diag)
    std_dist = np.std(dist_no_diag)
    cv_dist = std_dist / (mean_dist + 1e-8)
    
    # Sparsity estimation via distance distribution analysis
    sorted_dists = np.sort(dist_no_diag)
    q25 = np.percentile(sorted_dists, 25)
    q75 = np.percentile(sorted_dists, 75)
    iqr_sparsity = (q75 - q25) / (mean_dist + 1e-8)
    
    # Spectral properties (approximated)
    if n > 10:
        # Use normalized Laplacian for clustering tendency
        W = np.exp(-distance_matrix**2 / (2 * np.median(distance_matrix)**2))
        np.fill_diagonal(W, 0)
        D = np.diag(np.sum(W, axis=1))
        L = np.eye(n) - np.linalg.inv(D) @ W
        eigvals = np.linalg.eigvalsh(L)[:min(3, n)]
        spectral_gap = eigvals[1] if len(eigvals) > 1 else 0
    else:
        spectral_gap = 0
    
    return {
        'n': n,
        'mean_dist': mean_dist,
        'std_dist': std_dist,
        'cv_dist': cv_dist,
        'iqr_sparsity': iqr_sparsity,
        'spectral_gap': spectral_gap,
        'density_range': (np.min(dist_no_diag), np.max(dist_no_diag))
    }

def _compute_decomposed_factors(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute decomposed heuristic factors."""
    n = stats['n']
    factors = {}
    
    # 1. Inverse distance (normalized)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    if np.max(inv_dist) > 0:
        inv_dist = inv_dist / np.max(inv_dist)
    factors['inv_dist'] = inv_dist
    
    # 2. Decomposed regret: 1-hop and 2-hop
    factors['regret_1hop'], factors['regret_2hop'] = _compute_decomposed_regret(distance_matrix, stats)
    
    # 3. Spectral density using adaptive kernel
    factors['spectral_density'] = _compute_spectral_density(distance_matrix, stats)
    
    # 4. Two-step lookahead penalty
    factors['two_step_penalty'] = _compute_two_step_penalty(distance_matrix, stats)
    
    # 5. MST factor
    factors['mst'] = _compute_mst_factor(distance_matrix)
    
    return factors

def _compute_decomposed_regret(distance_matrix: np.ndarray, stats: dict) -> tuple:
    """Compute 1-hop and 2-hop regret scores separately."""
    n = stats['n']
    regret_1hop = np.ones((n, n))
    regret_2hop = np.ones((n, n))
    
    # Precompute sorted distances for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_dists = np.sort(distance_matrix, axis=1)
    
    for i in range(n):
        # 1-hop regret: compare to direct alternatives
        best_1hop = sorted_dists[i, 1]  # Skip self
        second_best = sorted_dists[i, 2] if n > 2 else best_1hop
        
        for j in range(n):
            if i == j:
                continue
                
            # 1-hop regret score
            edge_cost = distance_matrix[i, j]
            regret_1hop_val = max(0, edge_cost - best_1hop) / (best_1hop + 1e-8)
            regret_1hop[i, j] = 1 / (1 + regret_1hop_val)
            
            # 2-hop regret: find best 2-hop alternative (i->k->j)
            if n >= 4:
                best_2hop = float('inf')
                for k in range(n):
                    if k == i or k == j:
                        continue
                    alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                    best_2hop = min(best_2hop, alt_cost)
                
                if best_2hop < float('inf'):
                    regret_2hop_val = max(0, edge_cost - best_2hop) / (best_2hop + 1e-8)
                    # Scale by instance regularity
                    scale = 1 + stats['cv_dist']
                    regret_2hop[i, j] = np.exp(-regret_2hop_val * scale)
    
    np.fill_diagonal(regret_1hop, 0)
    np.fill_diagonal(regret_2hop, 0)
    
    return regret_1hop, regret_2hop

def _compute_spectral_density(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute density scores using spectral clustering principles."""
    n = stats['n']
    
    if n < 4:
        return np.ones((n, n))
    
    # Adaptive kernel bandwidth
    bandwidth = stats['mean_dist'] * (1 + stats['cv_dist'])
    
    # Compute similarity matrix with adaptive kernel
    similarity = np.exp(-distance_matrix**2 / (2 * bandwidth**2))
    np.fill_diagonal(similarity, 0)
    
    # Local density via row sums (normalized)
    local_density = np.sum(similarity, axis=1)
    if np.max(local_density) > 0:
        local_density = local_density / np.max(local_density)
    
    # Create density scores favoring connections in sparse regions
    density_scores = np.ones((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Reward connecting low-density nodes
            density_product = local_density[i] * local_density[j]
            density_scores[i, j] = density_scores[j, i] = np.exp(-density_product)
    
    return density_scores

def _compute_two_step_penalty(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute penalty for edges with good two-step alternatives."""
    n = stats['n']
    penalty_scores = np.ones((n, n))
    
    if n < 4:
        return penalty_scores
    
    # Adaptive threshold based on instance size and regularity
    threshold_factor = 0.7 + 0.3 * (1 - stats['cv_dist'])  # More penalty for regular instances
    
    for i in range(n):
        for j in range(i+1, n):
            # Find best two-step alternative (i->k->j)
            best_alternative = float('inf')
            
            # Sample candidate k's for efficiency
            if n > 20:
                # Use nearest neighbors to i and j
                k_candidates = set()
                for node in [i, j]:
                    neighbors = np.argsort(distance_matrix[node])[:min(10, n//2)]
                    k_candidates.update(neighbors)
                k_candidates = list(k_candidates)
                k_candidates = [k for k in k_candidates if k not in [i, j]]
            else:
                k_candidates = [k for k in range(n) if k not in [i, j]]
            
            for k in k_candidates:
                alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                best_alternative = min(best_alternative, alt_cost)
            
            if best_alternative < float('inf'):
                direct_cost = distance_matrix[i, j]
                ratio = best_alternative / (direct_cost + 1e-8)
                
                if ratio < threshold_factor:
                    # Significant two-step alternative exists - penalize
                    penalty = 1 - (ratio / threshold_factor)
                    penalty_scores[i, j] = penalty_scores[j, i] = max(0.1, 1 - penalty)
                else:
                    # No significant alternative
                    penalty_scores[i, j] = penalty_scores[j, i] = 1.0
    
    return penalty_scores

def _compute_mst_factor(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute MST-based scores with edge centrality."""
    n = distance_matrix.shape[0]
    mst_scores = np.ones((n, n))
    
    if n > 2:
        try:
            # Compute MST
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            
            # Boost edges in MST, with bonus for central edges
            row_sums = np.sum(mst_present, axis=1)
            centrality = row_sums / np.max(row_sums) if np.max(row_sums) > 0 else np.ones(n)
            
            for i in range(n):
                for j in range(i+1, n):
                    if mst_present[i, j]:
                        # Higher score for edges connecting nodes with higher centrality
                        centrality_bonus = 0.5 + 0.5 * (centrality[i] + centrality[j]) / 2
                        mst_scores[i, j] = mst_scores[j, i] = 1.5 * centrality_bonus
        except:
            pass
    
    np.fill_diagonal(mst_scores, 0)
    return mst_scores

def _dynamic_candidate_sparsification(distance_matrix: np.ndarray, stats: dict, factors: dict) -> np.ndarray:
    """Create adaptive candidate mask using multiple criteria."""
    n = stats['n']
    
    # Base candidate count adapts to instance properties
    base_k = min(15, max(3, int(np.log2(n) * 4)))
    # More candidates for irregular/sparse instances
    k = int(base_k * (1 + stats['cv_dist'] + stats['iqr_sparsity']))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # Use combined scores for candidate selection
    combined_scores = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Quick combination of key factors
            score = (
                factors['inv_dist'][i, j] *
                factors['regret_1hop'][i, j] *
                (0.7 + 0.3 * factors['spectral_density'][i, j])
            )
            combined_scores[i, j] = combined_scores[j, i] = score
    
    # Select top-k edges per node
    for i in range(n):
        scores_i = combined_scores[i].copy()
        scores_i[i] = -np.inf  # Exclude self
        
        if k < n:
            threshold = np.partition(scores_i, -k)[-k]
            mask_i = scores_i >= threshold
        else:
            mask_i = scores_i > 0
        
        candidate_mask[i] = mask_i
    
    # Ensure symmetry
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    # Additional filter: remove extremely long edges
    max_allowed = stats['mean_dist'] * (2 + stats['cv_dist'])
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] > max_allowed:
                candidate_mask[i, j] = candidate_mask[j, i] = False
    
    return candidate_mask

def _compute_adaptive_weights_v2(stats: dict, distance_matrix: np.ndarray) -> dict:
    """Compute adaptive weights based on instance properties."""
    n = stats['n']
    cv = stats['cv_dist']
    spectral_gap = stats['spectral_gap']
    
    # Base weights that adapt to instance characteristics
    weights = {
        'w_inv': 1.0,  # Always important
        
        # Regret weights: more emphasis on multi-hop for regular instances
        'w_reg1': 0.4 + 0.2 * cv,  # Higher for irregular
        'w_reg2': 0.3 + 0.2 * (1 - cv),  # Higher for regular
        
        # Spectral density: more important when spectral gap indicates clusters
        'w_spectral': 0.2 + 0.3 * spectral_gap,
        
        # Two-step penalty: more important for larger, regular instances
        'w_penalty': 0.1 + 0.2 * (1 - cv) + 0.1 * min(1, n/50),
        
        # MST: less important for larger instances
        'w_mst': max(0.05, 0.3 - 0.005 * n)
    }
    
    # Normalize to maintain scale
    total = sum(weights.values())
    scale_factor = 5.0 / total if total > 0 else 1.0
    for key in weights:
        weights[key] *= scale_factor
    
    return weights

def _post_process_sparse(H: np.ndarray, distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Post-process with enhanced sparsification and connectivity guarantee."""
    n = stats['n']
    
    # Zero out diagonal
    np.fill_diagonal(H, 0)
    
    # Ensure minimum connectivity: each node must have at least 2 edges
    min_edges = min(2, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i])[0]
        if len(non_zero) < min_edges:
            # Add nearest neighbors not already included
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Sort by distance
            sorted_indices = np.argsort(dists)
            added = 0
            
            for j in sorted_indices:
                if added >= min_edges - len(non_zero):
                    break
                if H[i, j] == 0:
                    # Assign a base score based on inverse distance
                    H[i, j] = H[j, i] = 1 / (distance_matrix[i, j] + 1e-8)
                    added += 1
    
    # Symmetrize
    H = (H + H.T) / 2
    
    # Final sparsification: keep only top edges
    if n > 10:
        for i in range(n):
            row = H[i].copy()
            if np.count_nonzero(row) > 0:
                # Keep top edges based on percentile
                threshold = np.percentile(row[row > 0], 30)  # Keep top 70%
                H[i, row < threshold] = 0
    
    # Re-symmetrize
    H = np.maximum(H, H.T)
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
