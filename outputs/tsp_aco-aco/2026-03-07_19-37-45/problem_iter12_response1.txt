import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import squareform, pdist
from typing import Dict, List, Tuple
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic with Monte Carlo regret sampling,
    multi-armed bandit weight optimization, and adaptive graph-theoretic features.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges.
    """
    np.random.seed(42)  # For reproducibility
    warnings.filterwarnings('ignore')
    
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # ==================== ADVANCED INSTANCE ANALYSIS ====================
    def analyze_instance_advanced(dist_mat: np.ndarray) -> Dict:
        """Multi-scale instance analysis with graph-theoretic metrics."""
        n = dist_mat.shape[0]
        dist_no_diag = dist_mat[~np.eye(n, dtype=bool)].reshape(n, n-1)
        dist_vals = dist_no_diag.flatten()
        
        # Basic statistics
        mean_dist = np.mean(dist_vals)
        std_dist = np.std(dist_vals)
        cv_dist = std_dist / (mean_dist + 1e-8)
        percentiles = np.percentile(dist_vals, [10, 25, 50, 75, 90])
        
        # Graph-theoretic analysis
        if n >= 5:
            # Clustering tendency via nearest neighbor reciprocity
            scales = [min(3, n-1), min(5, n-1), min(8, n-1)]
            reciprocity_scores = []
            density_scores = []
            
            for k in scales:
                if k < 2:
                    continue
                    
                # Compute k-nearest neighbor graph
                knn_graph = np.zeros((n, n), dtype=bool)
                for i in range(n):
                    dists = dist_mat[i].copy()
                    dists[i] = np.inf
                    neighbors = np.argpartition(dists, k-1)[:k]
                    knn_graph[i, neighbors] = True
                
                # Reciprocity (bidirectional connections)
                reciprocal = np.sum(knn_graph & knn_graph.T)
                total_edges = np.sum(knn_graph)
                reciprocity_scores.append(reciprocal / max(total_edges, 1))
                
                # Local density
                for i in range(n):
                    neighbors = np.where(knn_graph[i])[0]
                    if len(neighbors) > 1:
                        subgraph = dist_mat[np.ix_([i] + list(neighbors), [i] + list(neighbors))]
                        subgraph_mean = np.mean(subgraph[~np.eye(len(subgraph), dtype=bool)])
                        density_scores.append(mean_dist / max(subgraph_mean, 1e-8))
            
            reciprocity = np.mean(reciprocity_scores) if reciprocity_scores else 0.5
            density = np.mean(density_scores) if density_scores else 1.0
            
            # Eccentricity analysis
            eccentricities = []
            for i in range(n):
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                eccentricities.append(np.max(dists))
            
            eccentricity_cv = np.std(eccentricities) / (np.mean(eccentricities) + 1e-8)
        else:
            reciprocity = 0.5
            density = 1.0
            eccentricity_cv = 0.5
        
        return {
            'n': n,
            'mean_dist': mean_dist,
            'std_dist': std_dist,
            'cv_dist': cv_dist,
            'percentiles': percentiles,
            'reciprocity': reciprocity,
            'density': density,
            'eccentricity_cv': eccentricity_cv,
            'p50': percentiles[2],
            'p90': percentiles[4]
        }
    
    # ==================== MONTE CARLO REGRET SAMPLING ====================
    def compute_monte_carlo_regret(dist_mat: np.ndarray, stats: Dict, 
                                  candidate_mask: np.ndarray) -> np.ndarray:
        """
        Monte Carlo regret analysis with adaptive sampling.
        Samples random alternative paths of varying lengths.
        """
        n = stats['n']
        regret = np.ones((n, n))
        
        # Adaptive sampling parameters
        if n <= 20:
            n_samples = min(50, n * 3)
            max_path_length = 3
        elif n <= 50:
            n_samples = min(100, n * 2)
            max_path_length = 4
        else:
            n_samples = min(200, n)
            max_path_length = 5
        
        # Precompute nearest neighbors for each node
        neighbor_lists = []
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            k_neighbors = min(10, n-1)
            if k_neighbors > 0:
                neighbors = np.argpartition(dists, k_neighbors-1)[:k_neighbors]
                neighbor_lists.append(neighbors)
            else:
                neighbor_lists.append(np.array([]))
        
        # Monte Carlo regret computation
        for i in range(n):
            i_candidates = np.where(candidate_mask[i])[0]
            for j in i_candidates:
                if j <= i:
                    continue
                
                direct_cost = dist_mat[i, j]
                best_alternative = np.inf
                
                # Sample alternative paths
                for _ in range(n_samples):
                    # Random path length (2 to max_path_length)
                    path_len = np.random.randint(2, max_path_length + 1)
                    
                    # Construct random path
                    current = i
                    path_cost = 0
                    valid_path = True
                    
                    for step in range(path_len):
                        # Choose next node from candidates or neighbors
                        if step == path_len - 1:
                            # Last step must go to j
                            if candidate_mask[current, j] or current in neighbor_lists[j]:
                                path_cost += dist_mat[current, j]
                                current = j
                            else:
                                valid_path = False
                                break
                        else:
                            # Choose random next node (not i or j, and not visited)
                            candidates = []
                            if len(neighbor_lists[current]) > 0:
                                candidates = [nbr for nbr in neighbor_lists[current] 
                                           if nbr != i and nbr != j]
                            
                            if len(candidates) == 0:
                                # Fall back to any candidate
                                candidates = [k for k in range(n) 
                                           if k != i and k != j and k != current]
                            
                            if len(candidates) == 0:
                                valid_path = False
                                break
                            
                            next_node = np.random.choice(candidates)
                            path_cost += dist_mat[current, next_node]
                            current = next_node
                    
                    if valid_path and current == j:
                        best_alternative = min(best_alternative, path_cost)
                
                # Compute regret score
                if best_alternative < np.inf:
                    ratio = best_alternative / (direct_cost + 1e-8)
                    if ratio < 0.9:  # Alternative is significantly better
                        penalty = np.exp(-8 * (1.0 - ratio))
                    elif ratio < 1.1:  # Comparable
                        penalty = 1.0
                    else:  # Direct edge is better
                        bonus = 1.0 + np.log1p(min(ratio - 1.0, 2.0))
                        penalty = min(bonus, 2.5)
                    
                    # Adjust based on instance characteristics
                    if stats['reciprocity'] > 0.7:
                        penalty *= 0.9  # Favor direct edges in clustered instances
                    
                    regret[i, j] = regret[j, i] = penalty
        
        np.fill_diagonal(regret, 0)
        return regret
    
    # ==================== GRAPH-THEORETIC FEATURES ====================
    def compute_graph_features(dist_mat: np.ndarray, stats: Dict) -> Dict[str, np.ndarray]:
        """Compute advanced graph-theoretic features for edge scoring."""
        n = stats['n']
        features = {}
        
        # 1. Betweenness approximation via random paths
        betweenness = np.zeros((n, n))
        if n >= 5:
            n_paths = min(100, n * 5)
            for _ in range(n_paths):
                # Generate random simple path
                path_len = np.random.randint(3, min(8, n))
                nodes = np.random.choice(n, size=min(path_len, n), replace=False)
                
                # Score edges along path
                for idx in range(len(nodes) - 1):
                    i, j = nodes[idx], nodes[idx + 1]
                    betweenness[i, j] += 1
                    betweenness[j, i] += 1
        
        if np.max(betweenness) > 0:
            betweenness = betweenness / np.max(betweenness)
        features['betweenness'] = betweenness
        
        # 2. Clustering coefficient approximation
        clustering = np.ones((n, n))
        if n >= 4:
            for i in range(n):
                # Find neighbors within radius
                radius = np.percentile(dist_mat[i], 25)
                neighbors = [j for j in range(n) if j != i and dist_mat[i, j] <= radius]
                
                if len(neighbors) >= 2:
                    # Compute local density
                    for j in neighbors:
                        if j > i:
                            # Count triangles involving edge (i,j)
                            common_neighbors = 0
                            for k in neighbors:
                                if k != j and dist_mat[j, k] <= radius:
                                    common_neighbors += 1
                            
                            clustering_score = common_neighbors / max(len(neighbors) - 1, 1)
                            clustering[i, j] = clustering[j, i] = 1.0 + clustering_score
        
        features['clustering'] = clustering
        
        # 3. Edge centrality (combining multiple measures)
        edge_centrality = np.ones((n, n))
        
        # Degree centrality
        degree_centrality = np.zeros(n)
        threshold = stats['p50']
        for i in range(n):
            degree_centrality[i] = np.sum(dist_mat[i] <= threshold) - 1  # exclude self
        
        if np.max(degree_centrality) > 0:
            degree_centrality = degree_centrality / np.max(degree_centrality)
        
        # Combine measures
        for i in range(n):
            for j in range(i + 1, n):
                centrality_score = (
                    0.4 * degree_centrality[i] +
                    0.4 * degree_centrality[j] +
                    0.1 * betweenness[i, j] +
                    0.1 * clustering[i, j]
                )
                edge_centrality[i, j] = edge_centrality[j, i] = 1.0 + centrality_score
        
        features['edge_centrality'] = edge_centrality
        
        return features
    
    # ==================== MULTI-ARMED BANDIT WEIGHT OPTIMIZATION ====================
    def optimize_weights_bandit(dist_mat: np.ndarray, factors: Dict, 
                               candidate_mask: np.ndarray, stats: Dict) -> Dict:
        """
        Multi-armed bandit approach for weight optimization.
        Uses Thompson sampling for exploration-exploitation tradeoff.
        """
        n = stats['n']
        
        # Define factor names
        factor_names = ['inv_dist', 'regret', 'mst', 'edge_centrality']
        
        # Initialize bandit parameters
        n_factors = len(factor_names)
        alpha = np.ones(n_factors)  # Success counts
        beta = np.ones(n_factors)   # Failure counts
        n_bandit_pulls = min(30, max(10, n // 3))
        
        def evaluate_weight_config(weights: np.ndarray, tours: List[np.ndarray]) -> float:
            """Evaluate a weight configuration on sample tours."""
            # Normalize weights
            weights_norm = weights / np.sum(weights)
            
            # Combine factors
            n = factors['inv_dist'].shape[0]
            H_temp = np.zeros((n, n))
            
            for i in range(n):
                for j in range(i + 1, n):
                    if candidate_mask[i, j]:
                        combined = 1.0
                        for idx, name in enumerate(factor_names):
                            combined *= factors[name][i, j] ** weights_norm[idx]
                        H_temp[i, j] = H_temp[j, i] = combined
            
            # Score tours
            scores = []
            for tour in tours[:min(15, len(tours))]:
                tour_score = 0
                tour_length = 0
                for k in range(n):
                    i, j = tour[k], tour[(k + 1) % n]
                    if H_temp[i, j] > 0:
                        tour_score += H_temp[i, j]
                    tour_length += dist_mat[i, j]
                if tour_length > 0:
                    scores.append(tour_score / tour_length)
            
            return np.mean(scores) if scores else 0.0
        
        # Generate diverse tours for evaluation
        def generate_evaluation_tours(n_tours: int) -> List[np.ndarray]:
            tours = []
            
            # Strategy 1: Random tours
            for _ in range(n_tours // 3):
                tours.append(np.random.permutation(n))
            
            # Strategy 2: Nearest neighbor tours
            for _ in range(n_tours // 3):
                tour = np.zeros(n, dtype=int)
                visited = np.zeros(n, dtype=bool)
                current = np.random.randint(0, n)
                tour[0] = current
                visited[current] = True
                
                for idx in range(1, n):
                    dists = dist_mat[current].copy()
                    dists[visited] = np.inf
                    if np.all(dists == np.inf):
                        unvisited = np.where(~visited)[0]
                        current = np.random.choice(unvisited)
                    else:
                        # Biased random selection favoring closer nodes
                        valid_dists = dists[~np.isinf(dists)]
                        if len(valid_dists) > 0:
                            probs = 1 / (valid_dists + 1e-8)
                            probs = probs / np.sum(probs)
                            candidates = np.where(~np.isinf(dists))[0]
                            current = np.random.choice(candidates, p=probs)
                        else:
                            unvisited = np.where(~visited)[0]
                            current = np.random.choice(unvisited)
                    
                    tour[idx] = current
                    visited[current] = True
                tours.append(tour)
            
            # Strategy 3: Insertion heuristic tours
            for _ in range(n_tours - len(tours)):
                if n >= 3:
                    # Random insertion
                    tour = list(np.random.choice(n, size=3, replace=False))
                    remaining = [i for i in range(n) if i not in tour]
                    
                    while remaining:
                        # Select node to insert
                        node = np.random.choice(remaining)
                        
                        # Find best insertion position
                        best_cost = np.inf
                        best_pos = 0
                        
                        for pos in range(len(tour) + 1):
                            if pos == 0:
                                cost = (dist_mat[node, tour[0]] + 
                                       dist_mat[node, tour[-1]] - 
                                       dist_mat[tour[0], tour[-1]])
                            elif pos == len(tour):
                                cost = (dist_mat[tour[-1], node] + 
                                       dist_mat[node, tour[0]] - 
                                       dist_mat[tour[-1], tour[0]])
                            else:
                                cost = (dist_mat[tour[pos-1], node] + 
                                       dist_mat[node, tour[pos]] - 
                                       dist_mat[tour[pos-1], tour[pos]])
                            
                            if cost < best_cost:
                                best_cost = cost
                                best_pos = pos
                        
                        tour.insert(best_pos, node)
                        remaining.remove(node)
                    
                    tours.append(np.array(tour))
            
            return tours
        
        # Generate evaluation tours
        eval_tours = generate_evaluation_tours(min(30, n * 2))
        
        # Bandit optimization loop
        best_score = -np.inf
        best_weights = None
        
        for pull in range(n_bandit_pulls):
            # Thompson sampling: sample from Beta distributions
            sampled_weights = np.zeros(n_factors)
            for i in range(n_factors):
                sampled_weights[i] = np.random.beta(alpha[i], beta[i])
            
            # Evaluate
            score = evaluate_weight_config(sampled_weights, eval_tours)
            
            # Update bandit parameters
            success_prob = score / (1.0 + score)  # Convert to probability
            if np.random.random() < success_prob:
                alpha += sampled_weights
            else:
                beta += (1 - sampled_weights)
            
            # Track best
            if score > best_score:
                best_score = score
                best_weights = sampled_weights.copy()
        
        # Convert to dictionary format
        if best_weights is not None:
            weights_norm = best_weights / np.sum(best_weights)
            weight_dict = {
                'w_inv': weights_norm[0],
                'w_regret': weights_norm[1],
                'w_mst': weights_norm[2],
                'w_edge_centrality': weights_norm[3]
            }
        else:
            # Fallback weights
            weight_dict = {
                'w_inv': 0.25,
                'w_regret': 0.25,
                'w_mst': 0.25,
                'w_edge_centrality': 0.25
            }
        
        return weight_dict
    
    # ==================== MAIN EXECUTION ====================
    
    # Step 1: Advanced instance analysis
    stats = analyze_instance_advanced(distance_matrix)
    
    # Step 2: Build candidate mask (similar to v1 but more adaptive)
    def build_candidate_mask(dist_mat: np.ndarray, stats: Dict) -> np.ndarray:
        n = stats['n']
        base_k = min(15, max(4, int(4 * np.log2(n + 1))))
        
        # Adjust based on instance characteristics
        if stats['reciprocity'] > 0.7:  # More clustered
            base_k = int(base_k * 1.3)
        if stats['density'] > 1.2:  # Denser
            base_k = int(base_k * 0.8)
        
        candidate_mask = np.zeros((n, n), dtype=bool)
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            k = min(base_k, n - 1)
            indices = np.argpartition(dists, k)[:k] if k < n - 1 else np.arange(n)[dists < np.inf]
            candidate_mask[i, indices] = True
        
        candidate_mask = candidate_mask | candidate_mask.T
        
        # Ensure minimum connectivity
        min_edges = max(2, min(4, n // 8))
        for i in range(n):
            if np.sum(candidate_mask[i]) < min_edges:
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                sorted_idx = np.argsort(dists)
                needed = min_edges - np.sum(candidate_mask[i])
                for j in sorted_idx[:needed]:
                    if j != i:
                        candidate_mask[i, j] = candidate_mask[j, i] = True
        
        np.fill_diagonal(candidate_mask, False)
        return candidate_mask
    
    candidate_mask = build_candidate_mask(distance_matrix, stats)
    
    # Step 3: Compute factors
    factors = {}
    
    # 1. Inverse distance
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    if np.max(inv_dist) > 0:
        inv_dist = inv_dist / np.max(inv_dist)
    factors['inv_dist'] = inv_dist
    
    # 2. Monte Carlo regret
    factors['regret'] = compute_monte_carlo_regret(distance_matrix, stats, candidate_mask)
    
    # 3. MST-based scoring
    mst_scores = np.ones((n, n))
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            mst_scores[mst_present] = 1.8
            
            # Extra boost for critical MST edges
            degree = np.sum(mst_present, axis=1)
            for i in range(n):
                if degree[i] == 1:  # Leaf node
                    j = np.where(mst_present[i])[0][0]
                    mst_scores[i, j] = mst_scores[j, i] = 2.2
                elif degree[i] >= 3:  # Hub node
                    for j in np.where(mst_present[i])[0]:
                        mst_scores[i, j] = mst_scores[j, i] = 2.0
        except:
            pass
    np.fill_diagonal(mst_scores, 0)
    factors['mst'] = mst_scores
    
    # 4. Graph-theoretic features
    graph_features = compute_graph_features(distance_matrix, stats)
    factors['edge_centrality'] = graph_features['edge_centrality']
    
    # Step 4: Multi-armed bandit weight optimization
    weights = optimize_weights_bandit(distance_matrix, factors, candidate_mask, stats)
    
    # Step 5: Combine factors with optimized weights
    H = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            if candidate_mask[i, j]:
                combined = (
                    factors['inv_dist'][i, j] ** weights['w_inv'] *
                    factors['regret'][i, j] ** weights['w_regret'] *
                    factors['mst'][i, j] ** weights['w_mst'] *
                    factors['edge_centrality'][i, j] ** weights['w_edge_centrality']
                )
                H[i, j] = H[j, i] = combined
    
    np.fill_diagonal(H, 0)
    
    # Step 6: Adaptive sparsification with sigmoid emphasis
    def adaptive_sparsify(H: np.ndarray, dist_mat: np.ndarray, 
                         candidate_mask: np.ndarray, stats: Dict) -> np.ndarray:
        n = stats['n']
        
        # Ensure minimum connectivity
        min_edges = max(2, min(4, n // 6))
        for i in range(n):
            non_zero = np.where(H[i] > 0)[0]
            if len(non_zero) < min_edges:
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                sorted_idx = np.argsort(dists)
                added = 0
                for j in sorted_idx:
                    if added >= min_edges - len(non_zero):
                        break
                    if H[i, j] == 0 and j != i:
                        H[i, j] = H[j, i] = 1 / (dists[j] + 1e-8)
                        added += 1
        
        # Adaptive thresholding per node
        for i in range(n):
            row = H[i].copy()
            non_zero = row[row > 0]
            if len(non_zero) > min_edges:
                # Dynamic threshold based on distribution
                q25 = np.percentile(non_zero, 25)
                q75 = np.percentile(non_zero, 75)
                threshold = q25 + 0.3 * (q75 - q25)
                H[i, row < threshold] = 0
        
        # Symmetrize
        H = (H + H.T) / 2
        
        # Sigmoid emphasis with adaptive steepness
        max_val = np.max(H)
        if max_val > 0:
            H = H / max_val
            # Steeper sigmoid for more uniform instances
            steepness = 6 + 4 * (1 - stats['reciprocity'])
            H = 1 / (1 + np.exp(-steepness * (H - 0.5)))
        
        np.fill_diagonal(H, 0)
        return H
    
    H = adaptive_sparsify(H, distance_matrix, candidate_mask, stats)
    
    return H
