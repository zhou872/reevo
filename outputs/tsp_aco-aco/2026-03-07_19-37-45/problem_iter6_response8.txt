import numpy as np
from scipy.spatial.distance import pdist, squareform
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with dynamic local adaptation and hierarchical scoring.
    
    Key improvements:
    1. Dynamic k-selection for regret computation based on local node density
    2. Multiplicative hierarchical combination of distance, regret, and density factors
    3. Instance-aware parameter adaptation using distance concentration analysis
    4. Efficient degree-based sparsification with guaranteed connectivity
    
    Args:
        distance_matrix: Symmetric matrix of distances between nodes.
        
    Returns:
        Heuristic matrix of same shape indicating edge desirability.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    eps = 1e-12
    
    # ====================
    # 1. Instance Analysis
    # ====================
    triu_indices = np.triu_indices(n, k=1)
    flat_dists = distance_matrix[triu_indices]
    
    if len(flat_dists) == 0:
        return np.zeros_like(distance_matrix)
    
    dist_mean = np.mean(flat_dists)
    dist_std = np.std(flat_dists)
    concentration = dist_std / (dist_mean + eps)
    
    # ====================
    # 2. Adaptive Parameter Setup
    # ====================
    
    # Base parameters adjusted by problem size and concentration
    if n < 20:
        base_k = 4
        min_neighbors = 3
        power_min, power_max = 1.2, 2.0
    elif n < 100:
        base_k = max(4, n // 20)
        min_neighbors = max(3, n // 10)
        power_min, power_max = 1.3, 2.2
    else:
        base_k = max(5, n // 25)
        min_neighbors = max(4, n // 15)
        power_min, power_max = 1.4, 2.4
    
    # Concentration-adjusted parameters
    power = power_min + (power_max - power_min) * np.tanh(concentration)
    keep_factor = 1.5 + 0.5 * concentration
    
    # ====================
    # 3. Base Distance Factor
    # ====================
    
    # Inverse distance with adaptive power scaling
    inv_dist = 1 / (distance_matrix + eps)
    np.fill_diagonal(inv_dist, 0)
    inv_dist_pow = inv_dist ** power
    
    # ====================
    # 4. Local Density Analysis with Dynamic k
    # ====================
    
    # Pre-sort distances for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_dists = np.take_along_axis(distance_matrix, sorted_indices, axis=1)
    
    # Compute adaptive k for each node based on local density
    local_k = np.zeros(n, dtype=int)
    local_density = np.zeros(n)
    
    for i in range(n):
        # Skip self
        valid_dists = sorted_dists[i, 1:]
        if len(valid_dists) == 0:
            local_k[i] = 0
            local_density[i] = 0
            continue
            
        # Find elbow point in distance progression
        diffs = np.diff(valid_dists[:min(10, len(valid_dists))])
        if len(diffs) > 1:
            # Simple elbow detection: significant increase in gap
            threshold = 1.5 * np.mean(diffs[:max(1, len(diffs)//2)])
            elbow = np.where(diffs > threshold)[0]
            if len(elbow) > 0:
                adaptive_k = elbow[0] + 1
            else:
                adaptive_k = min(5, len(valid_dists))
        else:
            adaptive_k = min(3, len(valid_dists))
        
        # Adjust based on concentration
        adjusted_k = min(max(adaptive_k, min_neighbors), len(valid_dists))
        if concentration > 0.5:  # More variable distances
            adjusted_k = min(adjusted_k + 1, len(valid_dists))
        
        local_k[i] = adjusted_k
        if adjusted_k > 0:
            local_density[i] = 1 / (np.mean(valid_dists[:adjusted_k]) + eps)
    
    # Normalize densities
    max_density = np.max(local_density)
    if max_density > eps:
        local_density = local_density / max_density
    
    # ====================
    # 5. Two-hop Regret with Dynamic k-selection
    # ====================
    
    regret_matrix = np.ones((n, n))
    
    # Precompute k-nearest neighbors for each node
    max_k = np.max(local_k) if np.any(local_k > 0) else min_neighbors
    knn_indices = np.zeros((n, max_k + 2), dtype=int)
    knn_dists = np.full((n, max_k + 2), np.inf)
    
    for i in range(n):
        k_i = min(local_k[i] + 2, n - 1) if local_k[i] > 0 else min_neighbors + 2
        k_i = min(k_i, n - 1)
        
        # Get k_i nearest neighbors
        if k_i > 0:
            indices = sorted_indices[i, 1:k_i+1]
            dists = sorted_dists[i, 1:k_i+1]
            knn_indices[i, :len(indices)] = indices
            knn_dists[i, :len(dists)] = dists
    
    # Compute regret for each edge
    for i in range(n):
        if local_k[i] == 0:
            continue
            
        # Get valid neighbors for this node
        valid_mask = knn_indices[i] != 0
        neighbors_i = knn_indices[i, valid_mask]
        if len(neighbors_i) == 0:
            continue
            
        # For each neighbor, compute two-hop regret
        for j in neighbors_i:
            if i >= j:
                continue
                
            dist_ij = distance_matrix[i, j]
            
            # Find best alternative two-hop path i->k->j
            best_alternative = np.inf
            
            # Consider intermediate nodes from both neighborhoods
            candidates_i = knn_indices[i, :min(local_k[i] + 1, n-1)]
            candidates_j = knn_indices[j, :min(local_k[j] + 1, n-1)]
            
            # Combine and deduplicate candidates
            all_candidates = np.unique(np.concatenate([candidates_i, candidates_j]))
            all_candidates = all_candidates[all_candidates != i]
            all_candidates = all_candidates[all_candidates != j]
            
            if len(all_candidates) > 0:
                # Vectorized computation for efficiency
                paths_i = distance_matrix[i, all_candidates]
                paths_j = distance_matrix[all_candidates, j]
                two_hop_lengths = paths_i + paths_j
                
                if len(two_hop_lengths) > 0:
                    best_alternative = np.min(two_hop_lengths)
            
            # Compute regret score
            if best_alternative < np.inf:
                # Normalized regret: negative if direct edge is better
                regret = (best_alternative - dist_ij) / (dist_ij + eps)
                
                # Sigmoid transformation for smooth penalty
                # Higher regret -> lower score (penalize edges with good alternatives)
                penalty = 1 / (1 + np.exp(3 * regret))
                
                regret_matrix[i, j] = penalty
                regret_matrix[j, i] = penalty
    
    # ====================
    # 6. Hierarchical Edge Scoring
    # ====================
    
    # Density factor: boost edges between nodes in dense regions
    density_factor = np.ones((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if local_density[i] > eps and local_density[j] > eps:
                # Geometric mean for combined density
                combined_density = np.sqrt(local_density[i] * local_density[j])
                # Non-linear boost for high-density connections
                density_factor[i, j] = 1.0 + 0.4 * combined_density ** 1.5
                density_factor[j, i] = density_factor[i, j]
    
    # Degree weight: nodes with higher average distance get more weight
    node_avg_dist = np.mean(distance_matrix, axis=1)
    max_avg = np.max(node_avg_dist)
    if max_avg > eps:
        node_weights = max_avg / (node_avg_dist + eps)
        # Normalize weights
        node_weights = node_weights / np.max(node_weights)
        degree_matrix = np.outer(node_weights, node_weights)
    else:
        degree_matrix = np.ones((n, n))
    
    # ====================
    # 7. Multiplicative Combination
    # ====================
    
    # Hierarchical combination: distance * regret * density * degree
    heuristics = inv_dist_pow * regret_matrix * density_factor * degree_matrix
    
    # ====================
    # 8. Adaptive Sparsification
    # ====================
    
    # Dynamic sparsification based on local connectivity
    sparsified = np.zeros_like(heuristics)
    
    for i in range(n):
        # Determine how many edges to keep for this node
        if local_k[i] > 0:
            keep_count = min(int(local_k[i] * keep_factor), n // 2)
            keep_count = max(keep_count, min_neighbors)
        else:
            keep_count = min_neighbors
        
        keep_count = min(keep_count, n - 1)
        
        # Get top edges for this node
        if keep_count > 0:
            top_indices = np.argsort(heuristics[i])[-keep_count:]
            sparsified[i, top_indices] = heuristics[i, top_indices]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # ====================
    # 9. Connectivity Repair
    # ====================
    
    # Ensure minimum degree of 2 for all nodes
    for i in range(n):
        non_zero = np.where(sparsified[i] > eps)[0]
        if len(non_zero) < 2:
            # Use combined score to find best additional edges
            combined = inv_dist_pow[i] * regret_matrix[i] * density_factor[i] * degree_matrix[i]
            combined[i] = 0
            
            # Sort by combined score and add missing edges
            sorted_idx = np.argsort(combined)[::-1]
            added = 0
            for j in sorted_idx:
                if j != i and sparsified[i, j] == 0:
                    sparsified[i, j] = combined[j]
                    sparsified[j, i] = combined[j]
                    added += 1
                    if added >= 2 - len(non_zero):
                        break
    
    # ====================
    # 10. Final Normalization
    # ====================
    
    # Ensure symmetry after repairs
    sparsified = (sparsified + sparsified.T) / 2
    
    # Normalize to [0, 1]
    max_val = np.max(sparsified)
    if max_val > eps:
        sparsified = sparsified / max_val
    
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
