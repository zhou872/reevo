import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge prioritization heuristic with two-step lookahead,
    degree-aware penalization, and adaptive sparsity.
    Returns matrix of same shape with higher values for promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Core metrics computation
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    
    regret_scores = _compute_regret_scores_v2(distance_matrix)
    degree_penalty = _compute_degree_penalty(distance_matrix)
    lookahead_scores = _compute_two_step_lookahead(distance_matrix)
    
    # Adaptive candidate selection
    candidate_mask = _adaptive_candidate_mask(distance_matrix)
    
    # Combine factors with geometric mean
    H = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                combined = (
                    inv_dist[i, j] ** 0.4 *
                    regret_scores[i, j] ** 0.25 *
                    degree_penalty[i, j] ** 0.2 *
                    lookahead_scores[i, j] ** 0.15
                )
                H[i, j] = H[j, i] = combined
    
    # Sparsify: keep only top connections per node
    H = _sparsify_heuristics(H, distance_matrix)
    
    # Final normalization
    np.fill_diagonal(H, 0)
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H


def _compute_regret_scores_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Enhanced regret scoring with two-step consideration."""
    n = distance_matrix.shape[0]
    regret = np.ones((n, n))
    
    for i in range(n):
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        
        # Sort distances from i
        sorted_i = np.argsort(dists_i)
        if len(sorted_i) >= 2:
            best_i = dists_i[sorted_i[0]]
            second_best_i = dists_i[sorted_i[1]]
            base_regret_i = (second_best_i - best_i) / (best_i + 1e-8)
            
            for j in range(n):
                if j != i:
                    # Direct regret
                    direct_regret = max(0, dists_i[j] - best_i) / (best_i + 1e-8)
                    
                    # Two-step regret estimate: best alternative path through another node
                    min_two_step = np.inf
                    for k in range(n):
                        if k != i and k != j:
                            two_step_cost = dists_i[k] + distance_matrix[k, j]
                            min_two_step = min(min_two_step, two_step_cost)
                    
                    if min_two_step < np.inf:
                        two_step_regret = max(0, min_two_step - dists_i[j]) / (dists_i[j] + 1e-8)
                    else:
                        two_step_regret = 0
                    
                    # Combined regret score (inverted so low regret = high score)
                    total_regret = 0.7 * direct_regret + 0.3 * two_step_regret
                    regret[i, j] = 1 / (1 + total_regret)
    
    # Symmetrize
    regret = (regret + regret.T) / 2
    np.fill_diagonal(regret, 0)
    return regret


def _compute_degree_penalty(distance_matrix: np.ndarray) -> np.ndarray:
    """Penalize edges from nodes with many good connections."""
    n = distance_matrix.shape[0]
    penalty = np.ones((n, n))
    
    # For each node, count promising connections
    good_connections = np.zeros(n)
    k = min(5, n - 1)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        threshold = np.partition(dists, k)[k-1]
        good_connections[i] = np.sum(dists <= threshold * 1.2)
    
    # Normalize connection counts
    if np.std(good_connections) > 0:
        good_connections = (good_connections - np.mean(good_connections)) / np.std(good_connections)
    
    # Apply penalty: edges from nodes with many good connections get reduced scores
    for i in range(n):
        for j in range(i+1, n):
            # Penalty factor: inverse of geometric mean of connection counts
            penalty_factor = 1 / (1 + 0.3 * (abs(good_connections[i]) + abs(good_connections[j])))
            penalty[i, j] = penalty[j, i] = penalty_factor
    
    return penalty


def _compute_two_step_lookahead(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute two-step lookahead scores with penalty for bypass opportunities."""
    n = distance_matrix.shape[0]
    scores = np.ones((n, n))
    
    if n < 3:
        return scores
    
    for i in range(n):
        for j in range(i+1, n):
            direct_cost = distance_matrix[i, j]
            
            # Find best two-hop alternative
            min_two_hop = np.inf
            for k in range(n):
                if k != i and k != j:
                    two_hop_cost = distance_matrix[i, k] + distance_matrix[k, j]
                    min_two_hop = min(min_two_hop, two_hop_cost)
            
            if min_two_hop < np.inf:
                # If two-hop is significantly better, penalize direct edge
                if min_two_hop < direct_cost * 0.8:
                    # Strong penalty if two-hop is 20%+ better
                    scores[i, j] = 0.3
                elif min_two_hop < direct_cost * 0.95:
                    # Moderate penalty
                    scores[i, j] = 0.7
                else:
                    # No penalty, maybe slight bonus if direct is clearly better
                    scores[i, j] = 1.1 if direct_cost < min_two_hop * 0.9 else 1.0
            else:
                scores[i, j] = 1.0
    
    np.fill_diagonal(scores, 0)
    return scores


def _adaptive_candidate_mask(distance_matrix: np.ndarray) -> np.ndarray:
    """Create mask of candidate edges based on local neighborhoods."""
    n = distance_matrix.shape[0]
    mask = np.zeros((n, n), dtype=bool)
    
    # Adaptive neighborhood size
    base_k = min(10, max(3, int(np.log2(n) * 2)))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Keep k nearest neighbors
        k = base_k
        if n > 20:
            # Add some random connections for exploration
            k += int(np.log2(n))
        
        k = min(k, n - 1)
        
        if k < len(dists):
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        mask[i, indices] = True
        
        # Add a few random connections (5% chance per non-neighbor)
        if n > 10:
            for j in range(n):
                if j != i and not mask[i, j] and np.random.random() < 0.05:
                    mask[i, j] = True
    
    # Make symmetric
    mask = mask | mask.T
    np.fill_diagonal(mask, False)
    
    return mask


def _sparsify_heuristics(H: np.ndarray, distance_matrix: np.ndarray) -> np.ndarray:
    """Keep only top edges per node and sparsify further."""
    n = H.shape[0]
    
    # Determine keep count per node
    keep_per_node = min(8, max(3, int(np.log2(n) * 1.5)))
    
    # Zero out non-top edges for each node
    for i in range(n):
        row = H[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Find threshold for top edges
        if len(row) > keep_per_node:
            threshold = np.partition(row, -keep_per_node)[-keep_per_node]
        else:
            threshold = 0
        
        # Zero out below threshold
        for j in range(n):
            if j != i and H[i, j] < threshold:
                H[i, j] = 0
    
    # Ensure symmetry
    H = np.maximum(H, H.T)
    
    # Add MST edges to ensure connectivity
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_present = (mst + mst.T) > 0
            # Boost MST edges that were zeroed out
            for i in range(n):
                for j in range(i+1, n):
                    if mst_present[i, j] and H[i, j] == 0:
                        H[i, j] = H[j, i] = np.mean(H[H > 0]) if np.any(H > 0) else 0.5
        except:
            pass
    
    return H
