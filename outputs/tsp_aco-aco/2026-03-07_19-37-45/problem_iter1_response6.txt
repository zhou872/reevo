import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics combining multiple factors to identify promising edges.
    
    Strategies:
    1. **Neighbor Affinity Scoring**: For each node, identify promising connections
       based on both direct distance and two-hop indirect connections
    2. **Sparsification**: Set unpromising edges to zero using dynamic thresholding
    3. **Connectivity Density**: Reward edges that connect dense clusters
    4. **Two-Hop Lookahead**: Consider potential connections beyond immediate neighbors
    
    Returns a matrix of the same shape with heuristic scores (higher = more promising)
    """
    n = distance_matrix.shape[0]
    
    # Avoid division by zero and self-connections
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Factor 1: Two-hop path potential
    # For each edge (i,j), consider the best possible continuation through a third node k
    two_hop_potential = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Find the best intermediate node k (not i or j)
                mask = np.ones(n, dtype=bool)
                mask[[i, j]] = False
                if np.sum(mask) > 0:
                    # Best two-hop path: i -> k -> j
                    path_lengths = distance_matrix[i, mask] + distance_matrix[mask, j]
                    two_hop_potential[i, j] = np.min(path_lengths) if len(path_lengths) > 0 else 0
    
    # Normalize the two-hop potential
    max_two_hop = np.max(two_hop_potential)
    if max_two_hop > 0:
        two_hop_score = 1 / (1 + two_hop_potential / max_two_hop)
    else:
        two_hop_score = np.ones((n, n))
    
    # Factor 2: Node connectivity density
    # Nodes with many short connections are in dense regions
    k = min(5, n - 1)  # Number of nearest neighbors to consider
    node_density = np.zeros(n)
    
    for i in range(n):
        # Find k nearest neighbors (excluding self)
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        if k > 0:
            nearest_dists = np.partition(distances, k)[:k]
            node_density[i] = 1 / (np.mean(nearest_dists) + 1e-10)
    
    # Density bonus for edges connecting dense regions
    density_bonus = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                density_bonus[i, j] = (node_density[i] + node_density[j]) / 2
    
    # Factor 3: Edge centrality - reward edges that are part of many short triangles
    triangle_score = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Look for nodes k that form triangles with (i,j)
                mask = np.ones(n, dtype=bool)
                mask[[i, j]] = False
                if np.sum(mask) > 0:
                    # Quality of triangles through k
                    triangle_quality = 1 / (distance_matrix[i, mask] + distance_matrix[mask, j] + 1e-10)
                    triangle_score[i, j] = np.sum(triangle_quality) / len(triangle_quality) if len(triangle_quality) > 0 else 0
    
    # Combine factors with weights
    combined_scores = (
        0.4 * inv_dist +           # Direct distance (40%)
        0.25 * two_hop_score +     # Two-hop potential (25%)
        0.2 * density_bonus +      # Density bonus (20%)
        0.15 * triangle_score      # Triangle quality (15%)
    )
    
    # Sparsification: Set unpromising edges to zero
    # Use dynamic thresholding based on percentiles
    flat_scores = combined_scores.flatten()
    
    # Remove diagonal and zero elements for threshold calculation
    mask_idx = np.arange(n * n).reshape(n, n)
    diag_mask = np.eye(n, dtype=bool).flatten()
    valid_scores = flat_scores[~diag_mask & (flat_scores > 0)]
    
    if len(valid_scores) > 0:
        # Keep top 30% of edges (can be adjusted)
        threshold = np.percentile(valid_scores, 70)
        combined_scores[combined_scores < threshold] = 0
    
    # Ensure symmetry (assuming symmetric TSP)
    combined_scores = (combined_scores + combined_scores.T) / 2
    
    # Normalize to reasonable range
    max_score = np.max(combined_scores)
    if max_score > 0:
        combined_scores = combined_scores / max_score
    
    return combined_scores
