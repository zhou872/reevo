import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection.
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Basic inverse distance (preserving symmetry)
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    np.fill_diagonal(inv_dist, 0)
    mask = distance_matrix > 0
    inv_dist[mask] = 1.0 / distance_matrix[mask]
    
    # 1. Two-hop connectivity enhancement
    two_hop_bonus = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] == 0:
                continue
            # Find the closest third node that connects i and j
            min_sum = float('inf')
            for k in range(n):
                if k == i or k == j:
                    continue
                if distance_matrix[i, k] > 0 and distance_matrix[k, j] > 0:
                    sum_dist = distance_matrix[i, k] + distance_matrix[k, j]
                    if sum_dist < min_sum:
                        min_sum = sum_dist
            if min_sum < float('inf'):
                # Bonus if direct edge is better than best two-hop path
                two_hop_bonus[i, j] = two_hop_bonus[j, i] = 1.2 ** (min_sum / distance_matrix[i, j])
    
    # 2. Spatial clustering density factor
    density_factor = np.zeros_like(distance_matrix)
    for i in range(n):
        # Find average distance to k nearest neighbors
        k = min(5, n-1)
        distances = np.sort(distance_matrix[i])
        avg_near_dist = np.mean(distances[1:k+1])  # exclude self
        
        for j in range(n):
            if i == j:
                continue
            if distance_matrix[i, j] == 0:
                continue
            # Penalize edges connecting dense clusters to sparse areas
            density_j = np.mean(np.sort(distance_matrix[j])[1:k+1])
            density_factor[i, j] = 0.5 * (avg_near_dist + density_j) / distance_matrix[i, j]
    
    # 3. Minimum spanning tree similarity
    mst_factor = np.zeros_like(distance_matrix)
    # Prim's algorithm for MST
    in_mst = np.zeros(n, dtype=bool)
    mst_edges = []
    
    in_mst[0] = True
    for _ in range(n-1):
        min_edge = None
        min_weight = float('inf')
        for i in range(n):
            if not in_mst[i]:
                continue
            for j in range(n):
                if in_mst[j] or distance_matrix[i, j] == 0:
                    continue
                if distance_matrix[i, j] < min_weight:
                    min_weight = distance_matrix[i, j]
                    min_edge = (i, j)
        if min_edge:
            u, v = min_edge
            in_mst[v] = True
            mst_edges.append((u, v))
    
    # Bonus for edges in or similar to MST edges
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] == 0:
                continue
            # Check if edge is in MST
            in_mst_edge = any((i == u and j == v) or (i == v and j == u) 
                            for u, v in mst_edges)
            if in_mst_edge:
                mst_factor[i, j] = mst_factor[j, i] = 2.0
            else:
                # Check similarity to MST edges
                min_mst_ratio = float('inf')
                for u, v in mst_edges:
                    if distance_matrix[u, v] > 0:
                        ratio = distance_matrix[i, j] / distance_matrix[u, v]
                        min_mst_ratio = min(min_mst_ratio, ratio)
                if min_mst_ratio < float('inf'):
                    mst_factor[i, j] = mst_factor[j, i] = 1.0 / min_mst_ratio
    
    # 4. Combine factors with weights
    weights = [1.0, 0.8, 1.2, 0.9]  # Weights for each factor
    heuristics = (weights[0] * inv_dist + 
                  weights[1] * two_hop_bonus + 
                  weights[2] * density_factor + 
                  weights[3] * mst_factor)
    
    # 5. Asymmetry-aware pruning
    # Keep only promising edges for each node
    threshold_factor = 0.6  # Keep top 60% of edges for each node
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -float('inf')  # Exclude self
        if np.all(row == -float('inf')):
            continue
        
        # Find threshold for this node
        non_zero = row[row > 0]
        if len(non_zero) > 0:
            threshold = np.percentile(non_zero, (1 - threshold_factor) * 100)
        else:
            threshold = 0
        
        # Apply threshold
        for j in range(n):
            if i != j and heuristics[i, j] < threshold:
                heuristics[i, j] = 0
    
    # Ensure symmetry
    heuristics = np.maximum(heuristics, heuristics.T)
    
    return heuristics
