import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with regret-based penalization
    and efficient lookahead simulation.
    Returns a matrix of the same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-8)
    
    # Step 1: Adaptive parameter setup
    instance_stats = _analyze_instance_v2(distance_matrix)
    
    # Step 2: Compute core factors
    base_factors = _compute_base_factors_v2(distance_matrix, instance_stats)
    
    # Step 3: Sparse candidate selection
    candidate_mask = _build_candidate_mask_v2(distance_matrix, instance_stats)
    
    # Step 4: Lookahead simulation for promising edges
    lookahead_scores = _compute_lookahead_scores_v2(distance_matrix, instance_stats, candidate_mask)
    
    # Step 5: Combine with adaptive geometric weighting
    H = np.zeros((n, n))
    weights = _compute_adaptive_weights_v2(instance_stats)
    
    # Combine factors only for candidate edges
    i_idx, j_idx = np.nonzero(np.triu(candidate_mask, 1))
    for i, j in zip(i_idx, j_idx):
        combined = (
            (base_factors['inv_dist'][i, j] ** weights['w_inv']) *
            (base_factors['regret'][i, j] ** weights['w_regret']) *
            (base_factors['density'][i, j] ** weights['w_density']) *
            (lookahead_scores[i, j] ** weights['w_lookahead'])
        )
        H[i, j] = H[j, i] = combined
    
    # Step 6: Post-processing
    H = _post_process_heuristics_v2(H, distance_matrix, instance_stats)
    
    return H

def _analyze_instance_v2(distance_matrix: np.ndarray) -> dict:
    """Analyze instance characteristics for adaptive parameter tuning."""
    n = distance_matrix.shape[0]
    
    # Remove diagonal for statistics
    dist_no_diag = distance_matrix[~np.eye(n, dtype=bool)].reshape(n, n-1)
    
    stats = {
        'n': n,
        'mean_dist': np.mean(dist_no_diag),
        'std_dist': np.std(dist_no_diag),
        'cv_dist': np.std(dist_no_diag) / (np.mean(dist_no_diag) + 1e-8),
        'min_dist': np.min(dist_no_diag),
        'max_dist': np.max(dist_no_diag),
        'density_factor': _estimate_density_factor(dist_no_diag)
    }
    
    return stats

def _estimate_density_factor(dist_no_diag: np.ndarray) -> float:
    """Estimate density factor using quartile analysis."""
    q1 = np.percentile(dist_no_diag, 25)
    q3 = np.percentile(dist_no_diag, 75)
    iqr = q3 - q1
    
    if iqr < 1e-8:
        return 0.5
    
    # Measure of spread relative to median
    median = np.median(dist_no_diag)
    spread_factor = iqr / median
    
    # Transform to [0, 1] range
    return min(1.0, max(0.0, spread_factor))

def _compute_base_factors_v2(distance_matrix: np.ndarray, stats: dict) -> dict:
    """Compute base heuristic factors with improved regret calculation."""
    n = stats['n']
    
    factors = {}
    
    # 1. Inverse distance (standard)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)
    np.fill_diagonal(inv_dist, 0)
    factors['inv_dist'] = inv_dist
    
    # 2. Enhanced regret-based scoring using k-th best comparison
    factors['regret'] = _compute_regret_scores_v2(distance_matrix, stats)
    
    # 3. Density-based scoring
    factors['density'] = _compute_density_scores_v2(distance_matrix, stats)
    
    return factors

def _compute_regret_scores_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """
    Compute regret scores using k-th best alternative comparison.
    Higher scores for edges that are uniquely good choices.
    """
    n = stats['n']
    regret_scores = np.ones((n, n))
    
    # Adaptive k based on instance size
    k = min(5, max(3, n // 4))
    
    for i in range(n):
        # Get distances from node i to all others
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Find k-th best distance (k-th smallest, excluding self)
        if len(dists) > k:
            kth_best_dist = np.partition(dists, k)[k]
        else:
            kth_best_dist = np.max(dists)
        
        # Compute regret-based score for each edge from i
        for j in range(n):
            if j != i:
                # Regret: how much worse is this edge compared to k-th best
                if dists[j] <= kth_best_dist:
                    # Edge is among the k best - boost it
                    regret_factor = 1.0 + (kth_best_dist - dists[j]) / (kth_best_dist + 1e-8)
                else:
                    # Edge is worse than k-th best - penalize it
                    penalty = (dists[j] - kth_best_dist) / (stats['max_dist'] - kth_best_dist + 1e-8)
                    regret_factor = np.exp(-2.0 * penalty)
                
                regret_scores[i, j] *= regret_factor
    
    # Symmetrize using geometric mean
    regret_scores = np.sqrt(regret_scores * regret_scores.T)
    np.fill_diagonal(regret_scores, 0)
    
    # Normalize
    max_val = np.max(regret_scores)
    if max_val > 0:
        regret_scores = regret_scores / max_val
    
    return regret_scores

def _compute_density_scores_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute density scores favoring edges in sparse regions."""
    n = stats['n']
    density_scores = np.ones((n, n))
    
    # Estimate local density using nearest neighbor distances
    local_density = np.zeros(n)
    k = min(3, n - 1)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k)[:k]
        local_density[i] = np.mean(k_nearest)
    
    # Normalize densities
    if np.std(local_density) > 1e-8:
        local_density = (local_density - np.mean(local_density)) / np.std(local_density)
    
    # Create density scores: favor edges connecting sparse areas
    for i in range(n):
        for j in range(i+1, n):
            # Higher score when both endpoints are in sparse regions
            sparsity_factor = np.exp(-0.3 * (abs(local_density[i]) + abs(local_density[j])))
            density_scores[i, j] = density_scores[j, i] = sparsity_factor
    
    return density_scores

def _build_candidate_mask_v2(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Build mask of candidate edges to consider (for efficiency)."""
    n = stats['n']
    
    # Adaptive candidate list size
    base_k = min(15, max(5, int(np.sqrt(n) * 2)))
    
    # Increase candidate size for dense instances
    k = int(base_k * (1 + stats['density_factor']))
    k = min(k, n - 1)
    
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k-nearest neighbors
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < len(dists):
            # Get indices of k smallest distances
            indices = np.argpartition(dists, k)[:k]
        else:
            indices = np.arange(n)
            indices = indices[indices != i]
        
        candidate_mask[i, indices] = True
    
    # Make symmetric and zero diagonal
    candidate_mask = candidate_mask | candidate_mask.T
    np.fill_diagonal(candidate_mask, False)
    
    return candidate_mask

def _compute_lookahead_scores_v2(distance_matrix: np.ndarray, stats: dict, 
                                candidate_mask: np.ndarray) -> np.ndarray:
    """
    Compute lookahead scores by simulating partial tour extensions.
    Only performed for candidate edges.
    """
    n = stats['n']
    lookahead = np.ones((n, n))
    
    if n < 4:  # Too small for meaningful lookahead
        return lookahead
    
    # Adaptive number of simulations
    n_simulations = min(15, max(3, int(np.log2(n) * 4)))
    
    # Only simulate for candidate edges
    i_idx, j_idx = np.nonzero(np.triu(candidate_mask, 1))
    
    for i, j in zip(i_idx, j_idx):
        # Skip if edge is too long
        if distance_matrix[i, j] > 3 * stats['mean_dist']:
            continue
        
        scores = []
        for _ in range(n_simulations):
            # Simulate a partial tour starting with edge (i,j)
            score = _simulate_partial_tour_v2(distance_matrix, i, j, stats)
            scores.append(score)
        
        if scores:
            lookahead[i, j] = lookahead[j, i] = np.mean(scores)
    
    # Normalize
    max_val = np.max(lookahead)
    if max_val > 0:
        lookahead = lookahead / max_val
    
    np.fill_diagonal(lookahead, 0)
    return lookahead

def _simulate_partial_tour_v2(distance_matrix: np.ndarray, start_i: int, 
                             start_j: int, stats: dict) -> float:
    """
    Simulate extending a partial tour starting with edge (i,j).
    Returns a score based on quality of the extended partial tour.
    """
    n = stats['n']
    visited = np.zeros(n, dtype=bool)
    visited[start_i] = visited[start_j] = True
    
    current = start_j
    tour_length = distance_matrix[start_i, start_j]
    
    # Extend tour for adaptive number of steps
    n_steps = min(8, n - 2)
    
    for step in range(n_steps):
        # Find unvisited nodes
        unvisited = np.where(~visited)[0]
        if len(unvisited) == 0:
            break
        
        # Choose next node using softmin selection
        dists_to_current = distance_matrix[current, unvisited]
        
        # Temperature based on instance statistics
        temperature = stats['mean_dist'] * 0.5
        
        # Softmin selection
        weights = np.exp(-dists_to_current / temperature)
        if np.sum(weights) > 0:
            weights = weights / np.sum(weights)
            next_idx = np.random.choice(len(unvisited), p=weights)
            next_node = unvisited[next_idx]
            
            tour_length += dists_to_current[next_idx]
            visited[next_node] = True
            current = next_node
        else:
            break
    
    # Score is inverse of normalized tour length
    expected_length = n_steps * stats['mean_dist']
    if tour_length < 1e-8:
        return 1.0
    
    score = expected_length / tour_length
    return min(score, 2.0)  # Cap extreme values

def _compute_adaptive_weights_v2(stats: dict) -> dict:
    """Dynamically compute weights based on instance characteristics."""
    n = stats['n']
    cv = stats['cv_dist']
    density_factor = stats['density_factor']
    
    # Weights adjust based on instance properties
    weights = {
        'w_inv': 1.0,  # Base inverse distance
        
        # Higher regret weight for instances with more variation
        'w_regret': 0.4 + 0.3 * min(1.0, cv),
        
        # Density weight depends on density factor
        'w_density': 0.2 + 0.2 * density_factor,
        
        # Lookahead weight depends on size and variation
        'w_lookahead': 0.3 + 0.2 * min(1.0, np.log(n) / 10)
    }
    
    # Normalize to maintain overall scale
    total = sum(weights.values())
    for key in weights:
        weights[key] = weights[key] / total * 4  # Sum to 4
    
    return weights

def _post_process_heuristics_v2(H: np.ndarray, distance_matrix: np.ndarray, 
                               stats: dict) -> np.ndarray:
    """Post-process the heuristic matrix."""
    n = stats['n']
    
    # Ensure minimum connectivity
    min_edges = min(3, n - 1)
    for i in range(n):
        non_zero = np.nonzero(H[i])[0]
        if len(non_zero) < min_edges:
            # Add nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:min_edges]
            
            for j in nearest:
                if H[i, j] == 0:
                    # Assign base score
                    H[i, j] = H[j, i] = 1 / (distance_matrix[i, j] + 1e-8)
    
    # Symmetrize and zero diagonal
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # Sparsify: set low values to zero
    if n > 10:
        threshold = np.percentile(H[H > 0], 20)  # Keep top 80%
        H[H < threshold] = 0
    
    # Final normalization
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
