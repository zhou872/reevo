import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with dynamic instance adaptation and hybrid weighting.
    
    Key improvements over v1:
    1. Variance-adaptive penalty thresholds based on instance statistics
    2. Enhanced caching for efficient multi-hop path computation
    3. Hybrid weighting that balances exploration vs exploitation
    4. Dynamic neighbor thresholds using statistical clustering
    
    Args:
        distance_matrix: Symmetric matrix of distances between nodes.
        
    Returns:
        Heuristic matrix of same shape indicating edge desirability.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    eps = 1e-12
    
    # 1. Instance analysis for adaptive parameter tuning
    flat_dists = distance_matrix[~np.eye(n, dtype=bool)]
    mean_dist = np.mean(flat_dists)
    std_dist = np.std(flat_dists)
    cv_dist = std_dist / (mean_dist + eps)  # Coefficient of variation
    
    # 2. Base inverse distance with normalization
    inv_dist = 1 / (distance_matrix + eps)
    np.fill_diagonal(inv_dist, 0)
    
    # 3. Dynamic neighbor thresholding based on instance variance
    # Compute adaptive base_k using instance statistics
    if cv_dist < 0.2:  # Uniform distances
        base_k = max(3, n // 20)
    elif cv_dist > 0.8:  # Highly variable distances
        base_k = max(5, n // 10)
    else:  # Moderate variation
        base_k = max(4, n // 15)
    
    # 4. Compute local density with adaptive radius
    local_density = np.zeros(n)
    neighbor_counts = np.zeros(n, dtype=int)
    
    for i in range(n):
        sorted_dists = np.sort(distance_matrix[i])
        # Use dynamic number of neighbors for density calculation
        k_density = min(base_k + 2, n - 1)
        local_density[i] = np.mean(sorted_dists[1:k_density + 1])
        
        # Adaptive threshold based on instance variance
        threshold_factor = 1.5 + 0.5 * cv_dist
        neighbor_counts[i] = np.sum(distance_matrix[i] <= threshold_factor * local_density[i]) - 1
    
    # 5. Enhanced multi-hop lookahead with better caching
    sorted_neighbors = np.argsort(distance_matrix, axis=1)
    
    # Caches for multi-hop distances
    two_hop_cache = {}
    three_hop_cache = {}
    
    def get_two_hop_dist(i: int, j: int) -> float:
        """Compute minimum 2-hop distance with caching."""
        key = (min(i, j), max(i, j))
        if key in two_hop_cache:
            return two_hop_cache[key]
        
        min_dist = float('inf')
        # Dynamic lookahead based on local density
        k_lookahead = min(max(3, neighbor_counts[i] // 2), n - 2)
        
        for inter in sorted_neighbors[i, 1:k_lookahead + 1]:
            if inter == j:
                continue
            two_hop = distance_matrix[i, inter] + distance_matrix[inter, j]
            min_dist = min(min_dist, two_hop)
        
        two_hop_cache[key] = min_dist
        return min_dist
    
    def get_three_hop_dist(i: int, j: int) -> float:
        """Compute approximate 3-hop distance with caching."""
        key = (min(i, j), max(i, j))
        if key in three_hop_cache:
            return three_hop_cache[key]
        
        min_dist = float('inf')
        # Limit search space for efficiency
        k_limit = min(4, n - 3)
        
        for k in sorted_neighbors[i, 1:k_limit + 1]:
            if k == j:
                continue
            for l in sorted_neighbors[j, 1:k_limit + 1]:
                if l == i or l == k:
                    continue
                # Check if k and l are connected or close
                if distance_matrix[k, l] < 1.5 * (local_density[k] + local_density[l]) / 2:
                    three_hop = distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j]
                    min_dist = min(min_dist, three_hop)
        
        three_hop_cache[key] = min_dist if min_dist < float('inf') else 3 * distance_matrix[i, j]
        return three_hop_cache[key]
    
    # 6. Variance-adaptive penalty computation
    penalty = np.ones((n, n))
    
    # Dynamic penalty thresholds based on instance variance
    if cv_dist < 0.3:
        two_hop_threshold = 0.85  # Stricter for uniform instances
        three_hop_threshold = 0.80
    elif cv_dist > 0.7:
        two_hop_threshold = 0.90  # Looser for variable instances
        three_hop_threshold = 0.85
    else:
        two_hop_threshold = 0.88
        three_hop_threshold = 0.83
    
    for i in range(n):
        for idx_j, j in enumerate(sorted_neighbors[i, 1:]):
            # Consider dynamic number of neighbors
            max_neighbors = min(neighbor_counts[i] + 3, n - 1)
            if idx_j >= max_neighbors:
                break
            
            if i >= j:  # Compute only once per undirected edge
                continue
                
            direct = distance_matrix[i, j]
            
            # 2-hop penalty with variance-adaptive threshold
            two_hop_min = get_two_hop_dist(i, j)
            if two_hop_min < two_hop_threshold * direct:
                penalty[i, j] = 0.3  # Strong penalty
                penalty[j, i] = 0.3
            elif two_hop_min < 0.95 * direct:
                penalty_val = 0.5 + 0.2 * (two_hop_min / direct - two_hop_threshold) / (0.95 - two_hop_threshold)
                penalty[i, j] = min(penalty_val, 0.8)
                penalty[j, i] = penalty[i, j]
            
            # 3-hop penalty only if not already heavily penalized
            if penalty[i, j] > 0.6:
                three_hop_min = get_three_hop_dist(i, j)
                if three_hop_min < three_hop_threshold * direct:
                    penalty[i, j] = min(penalty[i, j], 0.2)
                    penalty[j, i] = penalty[i, j]
                elif three_hop_min < 0.92 * direct:
                    penalty_val = 0.4 + 0.2 * (three_hop_min / direct - three_hop_threshold) / (0.92 - three_hop_threshold)
                    penalty[i, j] = min(penalty[i, j], penalty_val)
                    penalty[j, i] = penalty[i, j]
    
    np.fill_diagonal(penalty, 0)
    
    # 7. Enhanced regret-based scoring
    regret_matrix = np.zeros((n, n))
    
    for i in range(n):
        # Get distances to all other nodes
        distances = distance_matrix[i].copy()
        distances[i] = float('inf')
        
        for j in range(n):
            if i == j:
                continue
                
            dist_ij = distance_matrix[i, j]
            # Find best alternative excluding j
            alt_dists = distances.copy()
            alt_dists[j] = float('inf')
            best_alt = np.min(alt_dists)
            
            if best_alt < float('inf'):
                # Enhanced regret calculation with opportunity awareness
                regret = (best_alt - dist_ij) / (best_alt + eps)
                
                # Adjust regret based on node degree opportunity
                opportunity_factor = 1.0
                if neighbor_counts[i] > base_k + 2:  # Dense region
                    opportunity_factor = 1.2  # Higher regret for missing critical edges in dense areas
                
                regret_matrix[i, j] = max(0, regret) * opportunity_factor
    
    # Symmetrize and normalize regret
    regret_matrix = (regret_matrix + regret_matrix.T) / 2
    max_regret = np.max(regret_matrix)
    if max_regret > eps:
        regret_matrix = regret_matrix / max_regret
    
    # 8. MST prior
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 9. Hybrid weighting: balance exploration vs exploitation
    # Parameter α controls exploration-exploitation tradeoff
    if cv_dist < 0.3:
        alpha = 0.7  # More exploitation for uniform instances
    elif cv_dist > 0.7:
        alpha = 0.4  # More exploration for variable instances
    else:
        alpha = 0.55  # Balanced
    
    # Base heuristics with exploration-exploitation balance
    heuristics = inv_dist * penalty
    
    # Apply hybrid weighting to regret boost
    regret_boost = 1.0 + (2.0 * alpha) * regret_matrix + (1.0 - alpha) * np.sqrt(regret_matrix + eps)
    heuristics *= regret_boost
    
    # MST bonus with adaptive strength
    mst_bonus = 1.0 + 0.5 * alpha  # Stronger bonus for exploitation-focused instances
    heuristics[mst_mask] *= mst_bonus
    
    # 10. Adaptive sparsification with connectivity guarantee
    sparsified = np.zeros_like(heuristics)
    
    for i in range(n):
        # Dynamic keep count based on local density and instance variance
        base_keep = max(3, neighbor_counts[i] // 2)
        variance_factor = 1.0 + 0.5 * cv_dist
        keep_count = min(int(base_keep * variance_factor), n // 2)
        
        # Get top edges
        top_indices = np.argpartition(heuristics[i], -keep_count)[-keep_count:]
        sparsified[i, top_indices] = heuristics[i, top_indices]
    
    # Ensure symmetric connectivity
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Connectivity repair for isolated nodes
    for i in range(n):
        if np.count_nonzero(sparsified[i]) < 2:
            # Connect to nearest neighbors with bonus for MST edges
            sorted_idx = np.argsort(distance_matrix[i])
            added = 0
            for j in sorted_idx[1:]:  # Skip self
                if sparsified[i, j] == 0:
                    base_value = inv_dist[i, j]
                    # Extra bonus for MST edges in connectivity repair
                    if mst_mask[i, j]:
                        base_value *= 1.5
                    sparsified[i, j] = base_value
                    sparsified[j, i] = base_value
                    added += 1
                    if added >= 2:
                        break
    
    # Final normalization
    max_val = np.max(sparsified)
    if max_val > eps:
        sparsified = sparsified / max_val
    
    np.fill_diagonal(sparsified, 0)
    return sparsified
