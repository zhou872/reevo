import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    
    Features:
    1. Sparsification based on nearest neighbor distances
    2. Power-law scaling of inverse distances
    3. Degree-aware normalization
    4. Thresholding based on distance quantiles
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    # 1. Basic inverse distance with power scaling (emphasizes shorter edges)
    inv_dist = 1 / (distance_matrix + 1e-12)  # Avoid division by zero
    power = 1.5  # Stronger emphasis on very short edges
    inv_dist_pow = inv_dist ** power
    
    # 2. Nearest neighbor analysis for sparsification
    # For each node, identify promising edges to its k nearest neighbors
    k = min(5, n - 1)
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        distances_i = distance_matrix[i].copy()
        distances_i[i] = np.inf  # Ignore self
        nn_indices = np.argpartition(distances_i, k)[:k]
        nn_mask[i, nn_indices] = True
    
    # Symmetrize the mask: edge is promising if either endpoint considers it
    nn_mask = nn_mask | nn_mask.T
    
    # 3. Distance threshold based on quantiles
    # Only consider edges below a certain distance percentile
    flat_distances = distance_matrix[np.triu_indices(n, k=1)]
    if len(flat_distances) > 0:
        threshold = np.quantile(flat_distances, 0.75)  # Keep edges in bottom 75%
        dist_threshold_mask = distance_matrix <= threshold
        # Keep diagonal False for self-connections
        np.fill_diagonal(dist_threshold_mask, False)
    else:
        dist_threshold_mask = np.ones_like(distance_matrix, dtype=bool)
        np.fill_diagonal(dist_threshold_mask, False)
    
    # 4. Degree-aware normalization
    # Nodes with higher average distances should be more selective
    node_avg_dist = distance_matrix.mean(axis=1)
    node_weights = node_avg_dist.max() / (node_avg_dist + 1e-12)
    
    # Create degree-aware matrix
    degree_matrix = np.outer(node_weights, node_weights)
    
    # 5. Combine all factors
    heuristics_matrix = inv_dist_pow * degree_matrix
    
    # 6. Apply sparsification masks
    combined_mask = nn_mask & dist_threshold_mask
    heuristics_matrix = heuristics_matrix * combined_mask
    
    # 7. Symmetrize the heuristics matrix
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    # 8. Normalize for numerical stability
    max_val = heuristics_matrix.max()
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    # 9. Ensure diagonal is zero (no self-loops)
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix


# Example usage and validation
if __name__ == "__main__":
    # Test with a simple 4-city TSP
    np.random.seed(42)
    n_cities = 6
    coords = np.random.rand(n_cities, 2) * 100
    
    # Create distance matrix
    distances = squareform(pdist(coords, 'euclidean'))
    
    # Compute heuristics
    h_matrix = heuristics_v2(distances)
    
    print("Distance matrix (first 4x4):")
    print(distances[:4, :4])
    print("\nHeuristics matrix (first 4x4):")
    print(h_matrix[:4, :4])
    print(f"\nSparsity: {(h_matrix == 0).sum() / h_matrix.size:.1%}")
    print(f"Non-zero entries: {(h_matrix != 0).sum()}")
