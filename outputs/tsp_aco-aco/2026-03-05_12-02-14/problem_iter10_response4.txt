def compute_instance_features(distance_matrix: np.ndarray, n: int) -> Dict:
    """Compute comprehensive instance features for meta-learning."""
    # Extract upper triangular distances
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    # Basic statistics
    stats = {
        'mean': np.mean(distances),
        'std': np.std(distances),
        'cv': np.std(distances) / (np.mean(distances) + 1e-10),
        'skewness': (np.mean((distances - np.mean(distances))**3) / 
                    (np.std(distances)**3 + 1e-10)),
        'kurtosis': (np.mean((distances - np.mean(distances))**4) / 
                    (np.std(distances)**4 + 1e-10)) - 3
    }
    
    # Geometric features
    stats['euclidean_score'] = estimate_euclidean_score(distance_matrix, n)
    
    # Graph theoretical features
    if n <= 1000:
        mst = csgraph.minimum_spanning_tree(distance_matrix)
        mst_density = (mst.nnz / 2) / (n * (n - 1) / 2)
        stats['mst_density'] = mst_density
    
    # Distance distribution percentiles
    percentiles = np.percentile(distances, [10, 25, 50, 75, 90])
    for i, p in enumerate([10, 25, 50, 75, 90]):
        stats[f'p{p}'] = percentiles[i]
    
    # Sparsity indicator
    stats['sparsity'] = np.sum(distances > stats['mean'] * 2) / len(distances)
    
    return stats

def compute_mst_with_criticality(distance_matrix: np.ndarray, n: int) -> Tuple[np.ndarray, np.ndarray]:
    """Compute MST with edge criticality scores."""
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    
    # Initialize criticality matrix
    criticality = np.zeros((n, n))
    
    # For each edge in MST, compute how many shortest paths would be affected
    mst_edges = np.where(mst_array > 0)
    
    # Simple approximation: edges connecting high-degree nodes are more critical
    if n <= 100:
        # Compute betweenness centrality approximation for small graphs
        for i, j in zip(*mst_edges):
            if i != j:
                # Criticality inversely proportional to alternative paths
                min_alt = float('inf')
                for k in range(n):
                    if k != i and k != j:
                        alt_path = distance_matrix[i, k] + distance_matrix[k, j]
                        min_alt = min(min_alt, alt_path)
                
                if min_alt > 0:
                    criticality[i, j] = criticality[j, i] = min_alt / distance_matrix[i, j]
    else:
        # For large graphs, use simpler metric
        degrees = (mst_array > 0).sum(axis=1)
        for i, j in zip(*mst_edges):
            if i != j:
                criticality[i, j] = criticality[j, i] = (degrees[i] + degrees[j]) / 2
    
    return mst_array, criticality


def compute_enhanced_matching(distance_matrix: np.ndarray, mst_array: np.ndarray, n: int) -> np.ndarray:
    """Compute enhanced perfect matching using blossom algorithm approximation."""
    matching_importance = np.zeros((n, n))
    
    # Find odd-degree vertices in MST
    degrees = (mst_array > 0).sum(axis=1)
    odd_vertices = np.where(degrees % 2 == 1)[0]
    
    if len(odd_vertices) == 0:
        return matching_importance
    
    # Greedy minimum-weight matching with lookahead
    matched = set()
    remaining = list(odd_vertices)
    
    while len(remaining) >= 2:
        best_pair = None
        best_score = float('inf')
        
        # Look for the best pair considering both direct distance and neighborhood
        for i_idx, i in enumerate(remaining):
            for j in remaining[i_idx+1:]:
                if i in matched or j in matched:
                    continue
                
                # Score considers distance and connectivity to other odd vertices
                dist = distance_matrix[i, j]
                
                # Penalty for isolating other vertices
                penalty = 0
                for k in remaining:
                    if k != i and k != j and k not in matched:
                        if distance_matrix[i, k] < dist * 1.5 or distance_matrix[j, k] < dist * 1.5:
                            penalty += 0.1
                
                score = dist * (1 + penalty)
                if score < best_score:
                    best_score = score
                    best_pair = (i, j)
        
        if best_pair:
            i, j = best_pair
            matching_importance[i, j] = matching_importance[j, i] = 1.0 / (distance_matrix[i, j] + 1e-10)
            matched.update([i, j])
            remaining.remove(i)
            remaining.remove(j)
        else:
            break
    
    return matching_importance


def generate_initial_tours(distance_matrix: np.ndarray, n: int, params: Dict) -> List[List[int]]:
    """Generate initial tours using multiple heuristics."""
    tours = []
    
    # 1. Nearest neighbor tours from different starting points
    num_nn = min(10, max(3, n // 20))
    starts = np.random.choice(n, num_nn, replace=False)
    
    for start in starts:
        tour = [start]
        visited = set([start])
        
        while len(visited) < n:
            current = tour[-1]
            mask = np.ones(n, dtype=bool)
            mask[list(visited)] = False
            
            if not np.any(mask):
                break
            
            distances = distance_matrix[current].copy()
            distances[~mask] = np.inf
            
            # Add small noise to break ties and encourage exploration
            noise = np.random.rand(n) * 1e-6 * np.std(distance_matrix)
            distances += noise
            
            next_node = np.argmin(distances)
            tour.append(next_node)
            visited.add(next_node)
        
        tours.append(tour)
    
    # 2. Christofides algorithm approximation
    if n <= 1000:
        try:
            # Simplified Christofides
            mst = csgraph.minimum_spanning_tree(distance_matrix)
            mst_array = mst.toarray()
            
            degrees = (mst_array > 0).sum(axis=1)
            odd_vertices = np.where(degrees % 2 == 1)[0]
            
            if len(odd_vertices) > 0:
                # Greedy matching on odd vertices
                matched = set()
                christofides_tour = []
                
                # Create Eulerian tour (simplified)
                visited = [False] * n
                stack = [0]
                
                while stack:
                    node = stack[-1]
                    if not visited[node]:
                        visited[node] = True
                        christofides_tour.append(node)
                    
                    # Find unvisited neighbor in MST
                    neighbors = np.where(mst_array[node] > 0)[0]
                    unvisited_neighbors = [n for n in neighbors if not visited[n]]
                    
                    if unvisited_neighbors:
                        next_node = min(unvisited_neighbors, 
                                      key=lambda x: distance_matrix[node, x])
                        stack.append(next_node)
                    else:
                        stack.pop()
                
                tours.append(christofides_tour)
        except:
            pass
    
    return tours


def apply_lin_kernighan(distance_matrix: np.ndarray, initial_tours: List[List[int]], 
                       depth: int, iterations: int) -> Tuple[List[List[int]], Dict]:
    """Apply Lin-Kernighan heuristic to improve tours."""
    best_tours = []
    edge_frequency = defaultdict(int)
    
    for tour in initial_tours:
        improved_tour = tour.copy()
        
        for _ in range(iterations):
            improved_tour = lk_move(improved_tour, distance_matrix, depth)
        
        best_tours.append(improved_tour)
        
        # Record edges from improved tour
        n = len(improved_tour)
        for i in range(n):
            u, v = improved_tour[i], improved_tour[(i+1)%n]
            edge_frequency[(min(u, v), max(u, v))] += 1
    
    return best_tours, edge_frequency


def lk_move(tour: List[int], distance_matrix: np.ndarray, depth: int) -> List[int]:
    """Perform one Lin-Kernighan move with given depth."""
    n = len(tour)
    best_tour = tour.copy()
    best_gain = 0
    
    # Try starting from each position
    for start in range(n):
        current_tour = tour.copy()
        
        for d in range(1, depth + 1):
            # Try to break d edges and reconnect differently
            gain, new_tour = try_k_opt(current_tour, distance_matrix, d, start)
            
            if gain > best_gain:
                best_gain = gain
                best_tour = new_tour.copy()
                break  # Take the first improving move
    
    return best_tour if best_gain > 0 else tour


def try_k_opt(tour: List[int], distance_matrix: np.ndarray, 
             k: int, start_pos: int) -> Tuple[float, List[int]]:
    """Try a k-opt move starting from given position."""
    n = len(tour)
    
    if k > 4:  # Limit complexity
        k = 4
    
    # Generate candidate edge breaks
    breaks = []
    for i in range(k):
        pos = (start_pos + i) % n
        breaks.append(pos)
    
    # Calculate current cost of broken edges
    current_cost = 0
    for pos in breaks:
        u, v = tour[pos], tour[(pos+1)%n]
        current_cost += distance_matrix[u, v]
    
    # Try different reconnections (simplified)
    # For k=2, this is standard 2-opt
    if k == 2:
        i, j = breaks[0], breaks[1]
        new_tour = tour.copy()
        
        # Reverse segment between i+1 and j
        if i < j:
            new_tour[i+1:j+1] = new_tour[i+1:j+1][::-1]
        else:
            segment = new_tour[i+1:] + new_tour[:j+1]
            segment.reverse()
            new_tour[i+1:] = segment[:n-i-1]
            new_tour[:j+1] = segment[n-i-1:]
        
        # Calculate new cost
        new_cost = 0
        for pos in breaks:
            u, v = new_tour[pos], new_tour[(pos+1)%n]
            new_cost += distance_matrix[u, v]
        
        gain = current_cost - new_cost
        return gain, new_tour
    
    # For higher k, return original tour (simplification)
    return 0, tour


def update_q_memory(q_memory: np.ndarray, edge_visit_count: np.ndarray, 
                   tours: List[List[int]], edges: Dict, 
                   distance_matrix: np.ndarray, 
                   learning_rate: float, discount_factor: float):
    """Update Q-learning memory based on tour quality."""
    if not tours:
        return
    
    # Calculate tour qualities
    tour_qualities = []
    for tour in tours:
        n = len(tour)
        length = sum(distance_matrix[tour[i], tour[(i+1)%n]] for i in range(n))
        tour_qualities.append(1.0 / (length + 1e-10))
    
    # Normalize qualities
    max_quality = max(tour_qualities) if tour_qualities else 1.0
    if max_quality > 0:
        tour_qualities = [q / max_quality for q in tour_qualities]
    
    # Update Q-values
    for tour_idx, tour in enumerate(tours):
        quality = tour_qualities[tour_idx]
        n = len(tour)
        
        for i in range(n):
            u, v = tour[i], tour[(i+1)%n]
            u, v = min(u, v), max(u, v)
            
            # Increment visit count
            edge_visit_count[u, v] += 1
            
            # Q-learning update
            current_q = q_memory[u, v]
            reward = quality
            
            # Edge frequency bonus
            freq_bonus = edges.get((u, v), 0) / len(tours)
            
            # Learning rate decay based on visit count
            alpha = learning_rate / (1 + 0.1 * edge_visit_count[u, v])
            
            # Update Q-value
            new_q = (1 - alpha) * current_q + alpha * (reward + discount_factor * freq_bonus)
            q_memory[u, v] = q_memory[v, u] = new_q


def extract_node_features(distance_matrix: np.ndarray, mst_array: np.ndarray, n: int) -> np.ndarray:
    """Extract features for each node for neural prediction."""
    features = []
    
    for i in range(n):
        # Distance-based features
        distances = distance_matrix[i]
        dist_features = [
            np.mean(distances),
            np.std(distances),
            np.min(distances[distances > 0]) if np.any(distances > 0) else 0,
            np.percentile(distances[distances > 0], 25) if np.any(distances > 0) else 0,
            np.percentile(distances[distances > 0], 75) if np.any(distances > 0) else 0
        ]
        
        # MST-based features
        mst_neighbors = np.where(mst_array[i] > 0)[0]
        mst_features = [
            len(mst_neighbors),
            np.mean(distance_matrix[i, mst_neighbors]) if len(mst_neighbors) > 0 else 0,
            np.sum(mst_array[i]) if len(mst_neighbors) > 0 else 0
        ]
        
        # Centrality approximation
        closeness = 1 / (np.sum(distances) + 1e-10)
        
        # Combine features
        node_features = dist_features + mst_features + [closeness]
        features.append(node_features)
    
    return np.array(features)

def multi_scale_clustering(distance_matrix: np.ndarray, n: int, params: Dict) -> List[np.ndarray]:
    """Perform multi-scale clustering for hierarchical structure discovery."""
    clusterings = []
    
    if n < 10:
        return clusterings
    
    # Convert to similarity matrix for clustering
    max_dist = np.max(distance_matrix)
    similarity = max_dist - distance_matrix
    
    # Try different epsilon values for DBSCAN
    eps_values = []
    
    # Use distance percentiles for epsilon values
    triu_distances = distance_matrix[np.triu_indices(n, k=1)]
    for percentile in [25, 50, 75]:
        eps = np.percentile(triu_distances, percentile)
        eps_values.append(eps)
    
    # Remove duplicates
    eps_values = list(set(eps_values))
    
    for eps in eps_values[:2]:  # Limit to 2 scales
        if eps <= 0:
            continue
            
        try:
            db = DBSCAN(eps=eps, min_samples=max(2, n//20), metric='precomputed')
            labels = db.fit_predict(distance_matrix)
            clusterings.append(labels)
        except:
            continue
    
    return clusterings


def compute_structural_importance(distance_matrix: np.ndarray, 
                                cluster_hierarchy: List[np.ndarray], n: int) -> np.ndarray:
    """Compute structural importance based on clustering hierarchy."""
    importance = np.ones((n, n))
    
    if not cluster_hierarchy:
        return importance
    
    for labels in cluster_hierarchy:
        if len(set(labels)) <= 1:
            continue
        
        for i in range(n):
            for j in range(i+1, n):
                if labels[i] == labels[j] and labels[i] != -1:
                    # Same cluster: boost importance
                    cluster_size = np.sum(labels == labels[i])
                    boost = 1.0 + 0.3 * min(1.0, cluster_size / n)
                    importance[i, j] *= boost
                elif labels[i] != labels[j]:
                    # Different clusters: penalize based on distance ratio
                    avg_intra_dist = compute_avg_intra_cluster_distance(
                        distance_matrix, labels, i, j
                    )
                    if avg_intra_dist > 0:
                        penalty = distance_matrix[i, j] / avg_intra_dist
                        importance[i, j] *= max(0.3, 1.0 / (1.0 + 0.1 * penalty))
    
    return importance


def compute_avg_intra_cluster_distance(distance_matrix: np.ndarray, 
                                      labels: np.ndarray, i: int, j: int) -> float:
    """Compute average intra-cluster distance for clusters containing i and j."""
    clusters = []
    
    # Find clusters for i and j
    if labels[i] != -1:
        cluster_i = np.where(labels == labels[i])[0]
        if len(cluster_i) > 1:
            # Compute average distance within cluster i
            intra_dists = []
            for a in cluster_i:
                for b in cluster_i:
                    if a < b:
                        intra_dists.append(distance_matrix[a, b])
            if intra_dists:
                clusters.append(np.mean(intra_dists))
    
    if labels[j] != -1 and labels[j] != labels[i]:
        cluster_j = np.where(labels == labels[j])[0]
        if len(cluster_j) > 1:
            # Compute average distance within cluster j
            intra_dists = []
            for a in cluster_j:
                for b in cluster_j:
                    if a < b:
                        intra_dists.append(distance_matrix[a, b])
            if intra_dists:
                clusters.append(np.mean(intra_dists))
    
    return np.mean(clusters) if clusters else 1.0


def normalize_matrix(matrix: np.ndarray) -> np.ndarray:
    """Normalize matrix to [0, 1] range."""
    if np.max(matrix) - np.min(matrix) == 0:
        return np.ones_like(matrix) / matrix.size
    
    normalized = (matrix - np.min(matrix)) / (np.max(matrix) - np.min(matrix) + 1e-10)
    return normalized


def learn_component_weights(components: Dict, instance_features: Dict, n: int) -> Dict:
    """Learn optimal weights for component combination based on instance features."""
    base_weights = {
        'mst': 0.2,
        'matching': 0.15,
        'rl': 0.25,
        'neural': 0.25,
        'structural': 0.15
    }
    
    # Adjust based on instance size
    if n < 50:
        base_weights['rl'] *= 1.2
        base_weights['neural'] *= 0.8
    elif n > 500:
        base_weights['mst'] *= 1.2
        base_weights['structural'] *= 0.8
    
    # Adjust based on Euclidean score
    euclidean_score = instance_features.get('euclidean_score', 0.5)
    if euclidean_score > 0.8:
        base_weights['mst'] *= 1.3
        base_weights['matching'] *= 1.2
        base_weights['neural'] *= 0.7
    elif euclidean_score < 0.3:
        base_weights['neural'] *= 1.3
        base_weights['structural'] *= 1.2
        base_weights['mst'] *= 0.7
    
    # Adjust based on distance variation
    cv = instance_features.get('cv', 1.0)
    if cv > 1.5:
        base_weights['neural'] *= 1.2
        base_weights['rl'] *= 1.1
    elif cv < 0.5:
        base_weights['mst'] *= 1.2
        base_weights['matching'] *= 1.1
    
    # Normalize weights
    total = sum(base_weights.values())
    return {k: v / total for k, v in base_weights.items()}


def combine_components(components: Dict, weights: Dict) -> np.ndarray:
    """Combine components with learned weights."""
    result = np.zeros_like(list(components.values())[0])
    
    for name, weight in weights.items():
        if name in components:
            result += weight * components[name]
    
    return result


def identify_critical_edges(mst_array: np.ndarray, lk_edges: Dict, 
                           q_memory: np.ndarray, neural_predictions: np.ndarray,
                           n: int, params: Dict) -> set:
    """Identify critical edges from multiple sources."""
    critical_edges = set()
    
    # 1. MST edges
    mst_edges = np.where(mst_array > 0)
    for i, j in zip(*mst_edges):
        if i != j:
            critical_edges.add((min(i, j), max(i, j)))
    
    # 2. Top LK edges
    if lk_edges:
        sorted_lk_edges = sorted(lk_edges.items(), key=lambda x: x[1], reverse=True)
        top_lk_count = min(len(sorted_lk_edges), max(10, n // 10))
        for (u, v), _ in sorted_lk_edges[:top_lk_count]:
            critical_edges.add((u, v))
    
    # 3. Top Q-memory edges
    q_values = []
    for i in range(n):
        for j in range(i+1, n):
            if q_memory[i, j] > 0:
                q_values.append((q_memory[i, j], i, j))
    
    if q_values:
        q_values.sort(reverse=True)
        top_q_count = min(len(q_values), max(5, n // 20))
        for _, i, j in q_values[:top_q_count]:
            critical_edges.add((i, j))
    
    # 4. Top neural prediction edges
    neural_values = []
    for i in range(n):
        for j in range(i+1, n):
            if neural_predictions[i, j] > 0:
                neural_values.append((neural_predictions[i, j], i, j))
    
    if neural_values:
        neural_values.sort(reverse=True)
        top_neural_count = min(len(neural_values), max(5, n // 20))
        for _, i, j in neural_values[:top_neural_count]:
            critical_edges.add((i, j))
    
    return critical_edges


def hierarchical_sparsification(matrix: np.ndarray, distance_matrix: np.ndarray,
                              critical_edges: set, instance_features: Dict,
                              n: int, params: Dict) -> np.ndarray:
    """Apply hierarchical sparsification with structural preservation."""
    sparsified = np.zeros_like(matrix)
    
    # Determine edges per node based on instance characteristics
    if n < 50:
        edges_per_node = max(3, n - 1)
    else:
        base_edges = int(2.5 * np.sqrt(n))
        cv = instance_features.get('cv', 1.0)
        # Adjust based on distance variation
        edges_per_node = int(base_edges * (1.0 + 0.3 * min(cv, 2.0)))
        edges_per_node = min(edges_per_node, n - 1)
    
    # Phase 1: Preserve critical edges
    for u, v in critical_edges:
        if u != v:
            sparsified[u, v] = sparsified[v, u] = matrix[u, v]
    
    # Phase 2: Add edges based on hierarchical importance
    for i in range(n):
        # Count already preserved edges for this node
        preserved_count = sum(1 for j in range(n) 
                            if (i, j) in critical_edges or (j, i) in critical_edges)
        
        if preserved_count >= edges_per_node:
            continue
        
        # Find candidate edges not already preserved
        candidates = []
        for j in range(n):
            if i != j and (i, j) not in critical_edges and (j, i) not in critical_edges:
                candidates.append((j, matrix[i, j]))
        
        if not candidates:
            continue
        
        # Sort candidates by importance
        candidates.sort(key=lambda x: x[1], reverse=True)
        
        # Select additional edges with exponential decay
        additional_needed = edges_per_node - preserved_count
        selected = candidates[:additional_needed]
        
        for rank, (j, value) in enumerate(selected):
            decay = np.exp(-rank / 2.0)  # Exponential decay
            sparsified[i, j] = max(sparsified[i, j], value * decay)
            sparsified[j, i] = max(sparsified[j, i], value * decay)
    
    # Ensure symmetry and zero diagonal
    sparsified = (sparsified + sparsified.T) / 2
    np.fill_diagonal(sparsified, 0)
    
    return sparsified


def final_refinement(matrix: np.ndarray, distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Apply final refinement and normalization."""
    result = matrix.copy()
    
    # Ensure non-negativity
    result = np.maximum(result, 0)
    
    # Add reciprocal distance safety net for zero rows
    row_sums = result.sum(axis=1)
    zero_rows = row_sums == 0
    
    if np.any(zero_rows):
        with np.errstate(divide='ignore', invalid='ignore'):
            fallback = 1 / distance_matrix
            fallback[np.isinf(fallback)] = 0
            fallback = np.nan_to_num(fallback, nan=0)
        
        np.fill_diagonal(fallback, 0)
        result[zero_rows] = fallback[zero_rows]
    
    # Row normalization for stochastic sampling
    row_sums = result.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    result = result / row_sums
    
    # Final symmetry enforcement
    result = (result + result.T) / 2
    np.fill_diagonal(result, 0)
    
    return result


def estimate_euclidean_score(distance_matrix: np.ndarray, n: int) -> float:
    """Estimate how Euclidean the distance matrix is."""
    if n < 3:
        return 1.0
    
    sample_size = min(100, n)
    indices = np.random.choice(n, sample_size, replace=False)
    
    violations = 0
    total_checks = 0
    
    # Sample triplets for triangle inequality check
    for i in range(sample_size):
        for j in range(i + 1, sample_size):
            for k in range(j + 1, sample_size):
                a, b, c = indices[i], indices[j], indices[k]
                d_ab = distance_matrix[a, b]
                d_ac = distance_matrix[a, c]
                d_bc = distance_matrix[b, c]
                
                # Check triangle inequality
                violations += int(d_ab > d_ac + d_bc + 1e-6)
                violations += int(d_ac > d_ab + d_bc + 1e-6)
                violations += int(d_bc > d_ab + d_ac + 1e-6)
                total_checks += 3
    
    return 1.0 - violations / max(total_checks, 1)
