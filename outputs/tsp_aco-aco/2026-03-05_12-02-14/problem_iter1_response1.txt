import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Compute edge selection heuristics for TSP by combining multiple factors
    and sparsifying unpromising edges.
    """
    n = distance_matrix.shape[0]
    
    # Avoid division by zero for self-distances
    safe_dist = np.where(distance_matrix > 0, distance_matrix, np.inf)
    
    # 1. Base inverse-distance component
    inv_dist = 1.0 / safe_dist
    
    # 2. Local topology component: relative closeness to nearest neighbors
    # For each node, find distances to its k nearest neighbors
    k = max(2, int(np.sqrt(n)))  # Adaptive k based on problem size
    nn_distances = np.partition(safe_dist, k, axis=1)[:, :k]
    local_scale = np.mean(nn_distances, axis=1, keepdims=True)
    
    # Normalize distances by local scale
    scaled_dist = safe_dist / (local_scale + local_scale.T + 1e-10)
    local_factor = np.exp(-scaled_dist ** 2)
    
    # 3. Connectivity gradient: favor edges that connect to different clusters
    # Approximate by difference in nearest neighbor distances
    min_dist_i = np.min(safe_dist, axis=1, keepdims=True)
    min_dist_j = np.min(safe_dist, axis=0, keepdims=True)
    gradient_factor = np.exp(-np.abs(min_dist_i - min_dist_j.T) / 
                           (np.mean(safe_dist) + 1e-10))
    
    # 4. Combine factors with adaptive weights
    heuristics = (
        0.5 * inv_dist + 
        0.3 * local_factor + 
        0.2 * gradient_factor
    )
    
    # 5. Sparsification: keep only promising edges
    # Threshold based on node-specific quantiles
    thresholds = np.percentile(heuristics, 70, axis=1, keepdims=True)
    mask = heuristics > (0.5 * thresholds + 0.5 * thresholds.T)
    
    # Additional sparsification: keep at least 2 edges per node
    for i in range(n):
        row = heuristics[i] * mask[i]
        if np.sum(row > 0) < 2:
            # Keep top-2 edges for this node
            top2_idx = np.argsort(row)[-2:]
            mask[i, top2_idx] = True
    
    # Apply mask and zero diagonal
    heuristics = heuristics * mask
    np.fill_diagonal(heuristics, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics /= max_val
    
    return heuristics
