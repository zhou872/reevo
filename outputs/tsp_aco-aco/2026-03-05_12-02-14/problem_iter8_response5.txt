import numpy as np
from scipy import sparse
from scipy.sparse import csgraph
import heapq
from typing import Dict, Tuple, List

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with multi-metric feedback tracking,
    population-guided edge sampling, and large-instance optimizations.
    
    Key innovations:
    1. Multi-metric feedback system tracking stagnation, diversity, and convergence rate
    2. Population-level information via fast constructive heuristics
    3. Approximate algorithms for large instances (sampled MST, fast centrality)
    4. Adaptive noise scaling based on convergence metrics
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        result = np.zeros_like(distance_matrix)
        if n == 2:
            result[0, 1] = result[1, 0] = 1.0
        return result
    
    # ==================== MULTI-METRIC FEEDBACK SYSTEM ====================
    # Initialize or update multi-metric tracking
    if not hasattr(heuristics_v2, 'metric_history'):
        heuristics_v2.metric_history = {}
    
    # Create instance signature for history tracking
    instance_key = f"{n}_{distance_matrix.mean():.2f}_{distance_matrix.std():.2f}"
    
    # Initialize or retrieve metrics for this instance type
    if instance_key not in heuristics_v2.metric_history:
        heuristics_v2.metric_history[instance_key] = {
            'stagnation_counter': 0,
            'diversity_tracker': 0.5,  # 0-1, higher = more diverse recent solutions
            'convergence_rate': 0.0,   # Rate of improvement
            'best_quality': float('inf'),
            'adaptive_exploration': 0.1
        }
    
    metrics = heuristics_v2.metric_history[instance_key]
    
    # ==================== INSTANCE CHARACTERIZATION ====================
    # Compute distance statistics
    mask = ~np.eye(n, dtype=bool)
    distances = distance_matrix[mask]
    mean_dist = np.mean(distances)
    std_dist = np.std(distances)
    cv_dist = std_dist / (mean_dist + 1e-10)
    
    # Update exploration based on distance variability and diversity
    exploration_base = 0.05 + 0.1 * min(cv_dist, 2.0)
    metrics['adaptive_exploration'] = exploration_base * (1 + 0.5 * metrics['diversity_tracker'])
    
    # ==================== POPULATION-GUIDED SAMPLING ====================
    # Generate diverse initial solutions using fast constructive heuristics
    def fast_constructive_heuristic(start_node: int) -> List[int]:
        """Fast nearest neighbor heuristic starting from given node."""
        unvisited = set(range(n))
        unvisited.remove(start_node)
        tour = [start_node]
        current = start_node
        
        while unvisited:
            # Get k nearest unvisited nodes
            row = distance_matrix[current]
            candidates = sorted([(row[i], i) for i in unvisited])[:min(5, len(unvisited))]
            # Stochastic selection: closer nodes have higher probability
            if len(candidates) > 1:
                probs = [1/(d+1e-10) for d, _ in candidates]
                probs = np.array(probs) / sum(probs)
                next_node = np.random.choice([i for _, i in candidates], p=probs)
            else:
                next_node = candidates[0][1]
            tour.append(next_node)
            unvisited.remove(next_node)
            current = next_node
        
        return tour
    
    # Generate diverse population
    population_size = min(20, max(5, n // 10))
    population = []
    
    for i in range(population_size):
        start = i % n
        tour = fast_constructive_heuristic(start)
        # Compute tour length (excluding return to start for efficiency)
        length = sum(distance_matrix[tour[i], tour[(i+1)%n]] for i in range(len(tour)))
        population.append((length, tour))
    
    # Extract edge frequencies and qualities from population
    edge_weights = np.zeros_like(distance_matrix)
    tour_lengths = [length for length, _ in population]
    min_length = min(tour_lengths)
    
    for length, tour in population:
        weight = min_length / (length + 1e-10)  # Better tours get higher weight
        for i in range(len(tour)):
            a, b = tour[i], tour[(i+1)%n]
            edge_weights[a, b] += weight
            edge_weights[b, a] += weight
    
    # Normalize edge weights
    if edge_weights.max() > 0:
        edge_weights = edge_weights / edge_weights.max()
    
    # Update diversity metric based on population variation
    length_std = np.std(tour_lengths) / (np.mean(tour_lengths) + 1e-10)
    metrics['diversity_tracker'] = 0.7 * metrics['diversity_tracker'] + 0.3 * min(length_std, 1.0)
    
    # ==================== APPROXIMATE BASE HEURISTICS ====================
    # 1. Reciprocal distance with adaptive transformation
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / np.maximum(distance_matrix, 1e-10)
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    np.fill_diagonal(reciprocal, 0)
    
    # Adaptive power transformation
    if cv_dist > 1.0:
        reciprocal = reciprocal ** (1 + 0.3 * min(cv_dist - 1.0, 1.0))
    elif cv_dist < 0.3:
        reciprocal = reciprocal ** 0.7
    
    # 2. Fast approximate centrality (sampled PageRank for large n)
    centrality = np.ones(n) / n
    
    if n > 100:
        # For large instances, use sampled neighbors for centrality
        for _ in range(3):  # Few iterations for efficiency
            new_centrality = np.zeros(n)
            for i in range(n):
                # Sample neighbors based on similarity
                sample_size = min(50, n-1)
                if sample_size < n-1:
                    # Weighted sampling favoring closer nodes
                    row = distance_matrix[i].copy()
                    row[i] = np.inf
                    probs = 1 / (row + 1e-10)
                    probs = probs / probs.sum()
                    neighbors = np.random.choice(n, size=sample_size, replace=False, p=probs)
                else:
                    neighbors = [j for j in range(n) if j != i]
                
                # Update centrality contribution
                for j in neighbors:
                    sim = 1 / (1 + distance_matrix[i, j])
                    new_centrality[j] += 0.85 * centrality[i] * sim
            new_centrality += 0.15 / n
            centrality = new_centrality / new_centrality.sum()
    else:
        # Exact for small instances
        sigma = np.median(distance_matrix[distance_matrix > 0])
        gaussian_sim = np.exp(-distance_matrix**2 / (2 * sigma**2))
        np.fill_diagonal(gaussian_sim, 0)
        for _ in range(5):
            centrality = 0.85 * gaussian_sim.T @ (centrality / (gaussian_sim.sum(axis=1) + 1e-10))
            centrality += 0.15 / n
            centrality = centrality / centrality.sum()
    
    centrality = centrality / np.max(centrality) if np.max(centrality) > 0 else centrality
    centrality_matrix = np.outer(centrality, centrality) * reciprocal
    
    # 3. Approximate MST component with sampling for large instances
    mst_component = np.zeros_like(distance_matrix)
    
    if n > 300:
        # For very large instances, use sampled MST
        sample_size = min(100, n)
        sampled_nodes = np.random.choice(n, size=sample_size, replace=False)
        submatrix = distance_matrix[sampled_nodes][:, sampled_nodes]
        
        try:
            mst = csgraph.minimum_spanning_tree(submatrix)
            mst_dense = mst.toarray()
            mst_mask = (mst_dense > 0)
            
            # Map back to original nodes with interpolation
            for idx_i, i in enumerate(sampled_nodes):
                for idx_j, j in enumerate(sampled_nodes):
                    if mst_mask[idx_i, idx_j]:
                        # Boost edges between sampled MST nodes
                        boost = 1 + centrality[i] * centrality[j]
                        mst_component[i, j] = reciprocal[i, j] * boost
        except Exception:
            pass  # Fall back to no MST component
    else:
        # Exact MST for manageable sizes
        try:
            mst = csgraph.minimum_spanning_tree(distance_matrix)
            mst_mask = (mst > 0).astype(float)
            mst_degree = mst_mask.sum(axis=1)
            
            for i in range(n):
                for j in range(i+1, n):
                    if mst_mask[i, j]:
                        score = (1 + centrality[i]) * (1 + centrality[j]) * reciprocal[i, j]
                        mst_component[i, j] = mst_component[j, i] = score
        except Exception:
            pass
    
    if mst_component.max() > 0:
        mst_component = mst_component / mst_component.max()
    
    # ==================== ADAPTIVE WEIGHTING WITH FEEDBACK ====================
    # Dynamic weights based on instance characteristics and metrics
    size_factor = min(1.0, np.log1p(n) / np.log1p(100))
    
    # Base weights
    if cv_dist > 1.0:
        w_rec, w_cen, w_mst, w_pop = 0.25, 0.25, 0.25, 0.25
    elif cv_dist < 0.3:
        w_rec, w_cen, w_mst, w_pop = 0.4, 0.2, 0.2, 0.2
    else:
        w_rec = 0.3 - 0.1 * size_factor
        w_cen = 0.25 + 0.05 * size_factor
        w_mst = 0.25 + 0.05 * size_factor
        w_pop = 0.2 + 0.05 * (1 - metrics['diversity_tracker'])
    
    # Adjust weights based on stagnation and convergence
    if metrics['stagnation_counter'] > 5:
        # During stagnation, increase exploration components
        w_pop += 0.15
        w_rec -= 0.1
        w_mst -= 0.05
    
    # Ensure weights sum to 1
    total = w_rec + w_cen + w_mst + w_pop
    w_rec, w_cen, w_mst, w_pop = w_rec/total, w_cen/total, w_mst/total, w_pop/total
    
    # Combine components
    combined = (w_rec * reciprocal + 
                w_cen * centrality_matrix + 
                w_mst * mst_component + 
                w_pop * edge_weights)
    
    # ==================== ADAPTIVE SPARSIFICATION ====================
    sparsified = np.zeros_like(combined)
    
    # Determine sparsity level based on instance size and metrics
    if n < 50:
        base_k = max(3, n // 3)
    elif n < 200:
        base_k = max(5, int(np.sqrt(n)))
    else:
        base_k = max(10, int(np.log2(n) * 3))
    
    # Adjust based on convergence state
    if metrics['stagnation_counter'] > 3:
        base_k = min(base_k + 2, n-1)  # Keep more edges during stagnation
    if metrics['diversity_tracker'] < 0.3:
        base_k = min(base_k + 1, n-1)  # Keep more edges when diversity is low
    
    for i in range(n):
        # Adaptive k per node based on centrality
        node_k = int(base_k * (0.8 + 0.4 * centrality[i]))
        node_k = max(2, min(n-1, node_k))
        
        row = combined[i].copy()
        row[i] = -np.inf
        
        if node_k < n-1:
            # Keep top-k edges with some stochastic variation
            noise = np.random.randn(n) * 0.1 * metrics['adaptive_exploration']
            noisy_row = row * (1 + noise)
            sorted_indices = np.argsort(noisy_row)[::-1]
            keep_indices = sorted_indices[:node_k]
        else:
            keep_indices = [j for j in range(n) if j != i]
        
        # Apply exponential decay to kept edges
        sorted_keep = sorted(keep_indices, key=lambda j: row[j], reverse=True)
        for rank, j in enumerate(sorted_keep):
            decay = np.exp(-rank / (1.5 + 0.5 * metrics['adaptive_exploration']))
            
            # Add adaptive noise
            noise_level = 0.05 + 0.1 * metrics['adaptive_exploration']
            if metrics['stagnation_counter'] > 5:
                noise_level *= 1.5
            
            perturbation = 1 + noise_level * np.random.randn()
            perturbation = np.clip(perturbation, 0.7, 1.3)
            
            sparsified[i, j] = combined[i, j] * decay * perturbation
    
    # ==================== SYMMETRIZATION AND NORMALIZATION ====================
    symmetric = 0.6 * sparsified + 0.4 * sparsified.T
    
    # Add strategic noise for exploration
    noise_scale = 0.02 * metrics['adaptive_exploration']
    if metrics['stagnation_counter'] > 2:
        noise_scale *= (1 + 0.5 * min(metrics['stagnation_counter'] / 10, 1.0))
    
    noise_matrix = np.random.randn(n, n) * noise_scale
    np.fill_diagonal(noise_matrix, 0)
    symmetric = symmetric + noise_matrix
    symmetric = np.maximum(symmetric, 0)
    
    # Row normalization for stochastic sampling
    row_sums = symmetric.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    symmetric = symmetric / row_sums
    
    np.fill_diagonal(symmetric, 0)
    
    # ==================== METRIC UPDATES ====================
    # Update stagnation counter based on population quality
    current_best = min(tour_lengths)
    if current_best < metrics['best_quality']:
        improvement = (metrics['best_quality'] - current_best) / metrics['best_quality']
        metrics['convergence_rate'] = 0.9 * metrics['convergence_rate'] + 0.1 * improvement
        metrics['best_quality'] = current_best
        metrics['stagnation_counter'] = max(0, metrics['stagnation_counter'] - 1)
    else:
        metrics['stagnation_counter'] += 1
    
    # Update exploration parameter
    if metrics['stagnation_counter'] > 10:
        metrics['adaptive_exploration'] = min(0.3, metrics['adaptive_exploration'] * 1.2)
    elif metrics['convergence_rate'] > 0.1:
        metrics['adaptive_exploration'] = max(0.05, metrics['adaptive_exploration'] * 0.9)
    
    return symmetric
