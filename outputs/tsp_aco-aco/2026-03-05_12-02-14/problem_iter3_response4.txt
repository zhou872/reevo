import numpy as np
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection with improved sparsification
    and stochastic edge ranking. Returns matrix of edge promise indicators.
    """
    n = distance_matrix.shape[0]
    
    # Base reciprocal distance with robust handling
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / np.maximum(distance_matrix, 1e-10)
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    np.fill_diagonal(reciprocal, 0)
    
    # Improved node centrality using direct connectivity
    centrality = np.zeros(n)
    for i in range(n):
        # Use inverse distances for centrality (closer nodes contribute more)
        row = distance_matrix[i].copy()
        row[i] = np.inf  # Exclude self
        centrality[i] = np.sum(1 / (1 + row))
    
    # Normalize centrality
    if np.max(centrality) > 0:
        centrality = centrality / np.max(centrality)
    
    # Edge promise based on centrality product with distance weighting
    centrality_matrix = np.outer(centrality, centrality)
    centrality_matrix = centrality_matrix * reciprocal  # Weight by inverse distance
    
    # MST indicator with probabilistic sampling
    try:
        mst = csgraph.minimum_spanning_tree(distance_matrix)
        mst_indicator = (mst > 0).astype(float)
        # Boost MST edges that connect to high-centrality nodes
        mst_boost = np.zeros_like(mst_indicator)
        for i in range(n):
            mst_neighbors = np.where(mst_indicator[i] > 0)[0]
            if len(mst_neighbors) > 0:
                boost_factor = centrality[i] * centrality[mst_neighbors].mean()
                mst_boost[i, mst_neighbors] = boost_factor
        mst_indicator = mst_indicator * (1 + mst_boost)
    except:
        mst_indicator = np.zeros_like(distance_matrix)
    
    # Combined heuristic with adaptive weighting
    # Weight factors based on problem size
    size_factor = min(1.0, n / 50.0)  # Normalize by size
    w1 = 0.4 + 0.2 * size_factor  # Reciprocal distance weight increases with size
    w2 = 0.3 - 0.1 * size_factor  # Centrality weight decreases with size
    w3 = 0.3 - 0.1 * size_factor  # MST weight decreases with size
    
    combined = (w1 * reciprocal + 
                w2 * centrality_matrix + 
                w3 * mst_indicator)
    
    # Adaptive sparsification with stochastic ranking
    sparsified = np.zeros_like(combined)
    
    # Determine number of edges to keep per node adaptively
    base_k = max(3, min(10, n // 4))
    # Add stochastic variation: ±1 edge
    k_vals = np.random.randint(max(2, base_k-1), min(n, base_k+2), size=n)
    
    for i in range(n):
        k = min(k_vals[i], n-1)  # Can't keep more than n-1 edges per node
        
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get indices of top edges
        if k < n-1:
            # Use argsort for more stable ranking
            sorted_indices = np.argsort(row)[::-1]
            keep_indices = sorted_indices[:k]
        else:
            keep_indices = np.arange(n)
            keep_indices = keep_indices[keep_indices != i]
        
        # Apply stochastic decay based on rank
        sorted_keep = keep_indices[np.argsort(row[keep_indices])[::-1]]
        for rank, j in enumerate(sorted_keep):
            # Exponential decay with random perturbation
            decay = np.exp(-rank / (1.5 + 0.5 * np.random.random()))
            # Add small noise to break ties
            noise = 1 + 0.05 * np.random.randn()
            sparsified[i, j] = combined[i, j] * decay * max(0.5, noise)
    
    # Ensure symmetry with probabilistic averaging
    symmetric = (sparsified + sparsified.T) / 2
    # Add small random component to avoid exact symmetry
    noise_matrix = np.random.randn(n, n) * 0.01
    np.fill_diagonal(noise_matrix, 0)
    symmetric = symmetric + noise_matrix
    symmetric = np.maximum(symmetric, 0)  # Ensure non-negative
    
    # Final cleanup
    np.fill_diagonal(symmetric, 0)
    
    return symmetric
