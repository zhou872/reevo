import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    Returns a sparse matrix with promising edges having positive values.
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return 1 / (distance_matrix + 1e-10)
    
    # 1. Base inverse distance component (preserve structure)
    inv_dist = 1 / (distance_matrix + 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree component (structural importance)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_heuristic = (mst + mst.T) / (distance_matrix + 1e-10)
    
    # 3. Nearest neighbor clustering component (local connectivity)
    nn_heuristic = np.zeros_like(distance_matrix)
    k = max(2, n // 10)  # Adaptive neighborhood size
    
    for i in range(n):
        # Find k nearest neighbors (excluding self)
        indices = np.argsort(distance_matrix[i])[1:k+1]
        # Assign weights based on rank (closer = higher weight)
        for rank, j in enumerate(indices):
            nn_heuristic[i, j] = (k - rank) / k
    
    # 4. Combine components with adaptive weights
    heuristics = (
        0.4 * inv_dist +      # Distance importance
        0.3 * mst_heuristic + # Structural importance
        0.3 * nn_heuristic    # Local connectivity
    )
    
    # 5. Sparsification: keep only promising edges
    # Threshold based on percentile of non-zero values
    flat_heuristics = heuristics[heuristics > 0]
    if len(flat_heuristics) > 0:
        threshold = np.percentile(flat_heuristics, 70)  # Keep top 30%
        heuristics[heuristics < threshold] = 0
    
    # Ensure symmetry for undirected TSP
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # Normalize rows to create probability distributions
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1  # Avoid division by zero
    heuristics = heuristics / row_sums
    
    return heuristics
