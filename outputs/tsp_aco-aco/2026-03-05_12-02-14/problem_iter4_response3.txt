import numpy as np
from scipy.sparse import csgraph
from collections import defaultdict
import heapq

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic matrix for TSP edge selection with adaptive feature weighting
    and efficient centrality computations.
    
    Features:
    1. Reciprocal distance (inverse edge weight)
    2. Fast node centrality approximation using random walks
    3. Minimum spanning tree indicator
    4. Edge-betweenness centrality (approximated for speed)
    5. Adaptive feature weighting based on empirical effectiveness
    
    Returns: Heuristic matrix of same shape as input
    """
    n = distance_matrix.shape[0]
    
    # 1. Reciprocal distance (base heuristic)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # Normalize reciprocal
    if np.max(reciprocal) > 0:
        reciprocal = reciprocal / np.max(reciprocal)
    
    # 2. Fast node centrality using random walk approximation
    centrality = fast_node_centrality(distance_matrix)
    
    # Create centrality matrix from node centralities
    centrality_matrix = np.outer(centrality, centrality)
    
    # 3. Minimum spanning tree indicator
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_indicator = (mst > 0).astype(float)
    
    # 4. Edge-betweenness centrality (approximated)
    edge_betweenness = approximate_edge_betweenness(distance_matrix, sample_size=min(50, n))
    
    # Normalize edge betweenness
    if np.max(edge_betweenness) > 0:
        edge_betweenness = edge_betweenness / np.max(edge_betweenness)
    
    # 5. Adaptive feature weighting
    features = {
        'reciprocal': reciprocal,
        'centrality': centrality_matrix,
        'mst': mst_indicator,
        'betweenness': edge_betweenness
    }
    
    # Weights determined by feature discriminative power
    weights = adaptive_feature_weights(features, distance_matrix)
    
    # 6. Combine features with adaptive weights
    combined = (weights['reciprocal'] * reciprocal +
                weights['centrality'] * centrality_matrix +
                weights['mst'] * mst_indicator +
                weights['betweenness'] * edge_betweenness)
    
    # 7. Intelligent sparsification
    sparsified = intelligent_sparsification(combined, n)
    
    # Ensure symmetry and zero diagonal
    sparsified = (sparsified + sparsified.T) / 2
    np.fill_diagonal(sparsified, 0)
    
    return sparsified

def fast_node_centrality(distance_matrix: np.ndarray, num_walks: int = 1000) -> np.ndarray:
    """
    Approximate node centrality using random walks with teleportation.
    Faster than all-pairs shortest paths for large graphs.
    """
    n = distance_matrix.shape[0]
    
    # Convert distances to transition probabilities (closer nodes more likely)
    with np.errstate(divide='ignore', invalid='ignore'):
        affinity = 1 / (1 + distance_matrix)
        np.fill_diagonal(affinity, 0)
    
    # Normalize rows to get transition probabilities
    row_sums = affinity.sum(axis=1, keepdims=True)
    transition = affinity / (row_sums + 1e-10)
    
    # Random walk with teleportation (like PageRank)
    alpha = 0.15  # teleportation probability
    visits = np.zeros(n)
    
    for _ in range(num_walks):
        current = np.random.randint(n)
        walk_length = np.random.randint(5, 20)
        
        for _ in range(walk_length):
            visits[current] += 1
            
            # Teleport with probability alpha
            if np.random.random() < alpha:
                current = np.random.randint(n)
            else:
                # Follow transition probabilities
                current = np.random.choice(n, p=transition[current])
    
    # Normalize centrality scores
    if np.max(visits) > 0:
        visits = visits / np.max(visits)
    
    return visits

def approximate_edge_betweenness(distance_matrix: np.ndarray, sample_size: int = 50) -> np.ndarray:
    """
    Approximate edge-betweenness centrality by sampling source nodes.
    Much faster than full computation for large graphs.
    """
    n = distance_matrix.shape[0]
    edge_counts = np.zeros((n, n))
    
    # Sample source nodes
    sources = np.random.choice(n, size=min(sample_size, n), replace=False)
    
    for s in sources:
        # Dijkstra from source s
        dist = np.full(n, np.inf)
        dist[s] = 0
        prev = [set() for _ in range(n)]
        sigma = np.zeros(n)
        sigma[s] = 1
        
        # Priority queue for Dijkstra
        pq = [(0, s)]
        
        while pq:
            d, u = heapq.heappop(pq)
            if d > dist[u]:
                continue
                
            for v in range(n):
                if v == u:
                    continue
                    
                alt = d + distance_matrix[u, v]
                if alt < dist[v] - 1e-10:  # Strictly better
                    dist[v] = alt
                    prev[v] = {u}
                    sigma[v] = sigma[u]
                    heapq.heappush(pq, (alt, v))
                elif abs(alt - dist[v]) < 1e-10:  # Equal distance
                    prev[v].add(u)
                    sigma[v] += sigma[u]
        
        # Accumulate edge betweenness for this source
        delta = np.zeros(n)
        nodes_by_dist = np.argsort(dist)
        
        for w in reversed(nodes_by_dist):
            if w == s:
                continue
                
            for v in prev[w]:
                # Contribution of paths through edge (v, w)
                contribution = sigma[v] / sigma[w] * (1 + delta[w])
                edge_counts[v, w] += contribution
                edge_counts[w, v] += contribution
                
            if prev[w]:
                delta[w] += 1
    
    # Average over sampled sources
    if len(sources) > 0:
        edge_counts = edge_counts / len(sources)
    
    return edge_counts

def adaptive_feature_weights(features: dict, distance_matrix: np.ndarray) -> dict:
    """
    Determine feature weights adaptively based on their empirical effectiveness
    in predicting good edges for TSP.
    """
    n = distance_matrix.shape[0]
    
    # Initialize weights
    weights = {
        'reciprocal': 0.25,
        'centrality': 0.25,
        'mst': 0.25,
        'betweenness': 0.25
    }
    
    # Adjust based on feature characteristics
    
    # 1. Boost reciprocal for graphs with clear nearest-neighbor structure
    nn_scores = []
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        nearest = np.min(row)
        second_nearest = np.partition(row, 1)[1]
        nn_scores.append(second_nearest / nearest if nearest > 0 else 1)
    
    nn_structure = np.mean(nn_scores)
    if nn_structure > 2.0:  # Clear nearest-neighbor structure
        weights['reciprocal'] *= 1.5
    
    # 2. Boost MST for sparse-like graphs
    mst_density = np.sum(features['mst']) / (n * (n - 1))
    if mst_density < 0.1:  # Very sparse MST
        weights['mst'] *= 2.0
    
    # 3. Boost betweenness for graphs with bottleneck structure
    # Estimate bottleneck by checking if removing high-betweenness edges
    # significantly increases average path length
    high_betweenness_mask = features['betweenness'] > np.percentile(features['betweenness'], 90)
    if np.sum(high_betweenness_mask) > 0:
        weights['betweenness'] *= 1.2
    
    # Normalize weights to sum to 1
    total = sum(weights.values())
    weights = {k: v/total for k, v in weights.items()}
    
    return weights

def intelligent_sparsification(combined: np.ndarray, n: int) -> np.ndarray:
    """
    Sparsify the heuristic matrix intelligently, keeping promising edges
    and applying smooth decay to less promising ones.
    """
    sparsified = np.zeros_like(combined)
    
    # Adaptive sparsity: keep more edges for smaller graphs
    base_k = max(3, min(15, int(np.sqrt(n))))
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf
        
        # Determine k for this node based on row variance
        row_variance = np.var(row[row > -np.inf])
        k = base_k + int(row_variance * 5)  # More edges for high-variance rows
        
        k = min(k, n - 1)
        
        # Get top k edges
        if k < n:
            # Get indices of top k values
            top_k_idx = np.argpartition(row, -k)[-k:]
        else:
            top_k_idx = np.arange(n)
            top_k_idx = top_k_idx[top_k_idx != i]
        
        # Sort top edges
        sorted_idx = top_k_idx[np.argsort(row[top_k_idx])[::-1]]
        
        # Apply smooth sigmoid decay based on rank
        for rank, j in enumerate(sorted_idx):
            # Sigmoid decay: slower initial drop, then faster
            decay = 1 / (1 + np.exp((rank - k/2) / (k/10)))
            sparsified[i, j] = combined[i, j] * decay
    
    return sparsified
