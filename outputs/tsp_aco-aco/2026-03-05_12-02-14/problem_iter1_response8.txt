import numpy as np
from scipy.sparse import csgraph
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Compute edge prior indicators for TSP by combining multiple heuristics:
    1. Minimum Spanning Tree (MST) inclusion
    2. Nearest neighbor relationships
    3. Edge betweenness in MST
    4. Sparsification of unpromising edges
    """
    n = distance_matrix.shape[0]
    
    # Prevent division by zero and self-loops
    np.fill_diagonal(distance_matrix, np.inf)
    
    # 1. Base heuristic: inverse distance (like v1)
    inv_dist = 1.0 / distance_matrix
    
    # 2. MST-based heuristic
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)  # Undirected MST edges
    
    # 3. Nearest neighbor heuristic
    k = max(2, n // 10)  # Adaptive k based on problem size
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        neighbors = np.argsort(distance_matrix[i])[:k]
        nn_mask[i, neighbors] = True
        nn_mask[neighbors, i] = True  # Symmetric
    
    # 4. Edge betweenness in MST (approximate)
    betweenness = np.zeros_like(distance_matrix)
    
    # Simple betweenness approximation: edges connecting different MST subtrees
    # Count how many node pairs have this edge on their MST path
    for i in range(n):
        for j in range(i + 1, n):
            if mst_mask[i, j]:
                # This edge is in MST, so it separates the tree into two components
                # All pairs (u in comp1, v in comp2) have this edge on their MST path
                comp1_size = np.sum(mst_mask[i] > 0)  # Approximate component size
                comp2_size = n - comp1_size
                betweenness[i, j] = betweenness[j, i] = comp1_size * comp2_size
    
    # Normalize betweenness
    if betweenness.max() > 0:
        betweenness = betweenness / betweenness.max()
    
    # 5. Combine heuristics with weights
    mst_weight = 3.0
    nn_weight = 2.0
    betweenness_weight = 1.5
    
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base inverse distance
    heuristics_matrix = inv_dist.copy()
    
    # Boost MST edges
    heuristics_matrix[mst_mask] *= mst_weight
    
    # Boost nearest neighbor edges
    heuristics_matrix[nn_mask] *= nn_weight
    
    # Boost high-betweenness edges
    heuristics_matrix += betweenness_weight * betweenness * inv_dist
    
    # 6. Sparsification: set unpromising edges to zero
    # Keep top edges per node and MST edges
    
    # Keep all MST edges
    sparsity_mask = mst_mask.copy()
    
    # For each node, keep top k promising edges
    top_k = max(3, n // 5)
    for i in range(n):
        # Get indices of top promising edges for this node
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Exclude self
        top_indices = np.argsort(row)[-top_k:]
        sparsity_mask[i, top_indices] = True
        sparsity_mask[top_indices, i] = True  # Symmetric
    
    # Apply sparsification
    heuristics_matrix[~sparsity_mask] = 0
    
    # Ensure symmetry
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    
    # Reset diagonal to zero
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
