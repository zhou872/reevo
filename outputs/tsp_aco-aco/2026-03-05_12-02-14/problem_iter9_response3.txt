import numpy as np
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Heuristic matrix for TSP edge selection using simplified, stable components.
    
    Features:
    1. Simplified composition with fixed weights: 60% reciprocal distance, 
       30% centrality measure, 10% MST indicator
    2. Adaptive sparsity control based on problem size with exponential decay
    3. No random sampling or iterative refinement for stability
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape indicating edge promise (higher = better)
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic[np.isinf(heuristic)] = 0
            heuristic = np.nan_to_num(heuristic, nan=0)
        np.fill_diagonal(heuristic, 0)
        return heuristic
    
    # ========== Phase 1: Compute Heuristic Components ==========
    
    # Component 1: Reciprocal distance (60% weight)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal = np.nan_to_num(reciprocal, nan=0)
    np.fill_diagonal(reciprocal, 0)
    
    # Normalize reciprocal
    if reciprocal.max() > 0:
        reciprocal = reciprocal / reciprocal.max()
    
    # Component 2: Centrality matrix (30% weight)
    # Use closeness centrality: nodes with smaller average distance to others are more central
    centrality = np.zeros(n)
    for i in range(n):
        # Average distance from node i to all other nodes
        centrality[i] = 1 / (np.mean([distance_matrix[i, j] for j in range(n) if j != i]) + 1e-10)
    
    # Create centrality matrix where edge (i,j) gets average centrality of its endpoints
    centrality_matrix = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                centrality_matrix[i, j] = (centrality[i] + centrality[j]) / 2
    
    # Normalize centrality matrix
    if centrality_matrix.max() > 0:
        centrality_matrix = centrality_matrix / centrality_matrix.max()
    
    # Component 3: MST indicator (10% weight)
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    mst_indicator = (mst_array > 0).astype(float)
    
    # ========== Phase 2: Combine Components with Fixed Weights ==========
    
    # Fixed weights from long-term suggestions
    weights = {'reciprocal': 0.6, 'centrality': 0.3, 'mst': 0.1}
    
    heuristic_matrix = (
        weights['reciprocal'] * reciprocal +
        weights['centrality'] * centrality_matrix +
        weights['mst'] * mst_indicator
    )
    
    np.fill_diagonal(heuristic_matrix, 0)
    
    # ========== Phase 3: Adaptive Sparsification ==========
    
    # Determine sparsity level based on problem size
    if n < 50:
        edges_per_node = n - 1  # Keep all edges for small problems
    elif n < 200:
        edges_per_node = int(2 * np.sqrt(n))
    else:
        edges_per_node = int(1.5 * np.sqrt(n))
    
    # Always preserve MST edges
    mst_edges = set()
    for i in range(n):
        for j in range(i + 1, n):
            if mst_indicator[i, j] > 0:
                mst_edges.add((i, j))
    
    # Create sparsified matrix
    sparsified = np.zeros((n, n))
    
    # First, preserve MST edges
    for i, j in mst_edges:
        sparsified[i, j] = sparsified[j, i] = heuristic_matrix[i, j]
    
    # For each node, select top additional edges with exponential decay
    for i in range(n):
        # Get row values, excluding self and already preserved MST edges
        row_values = []
        for j in range(n):
            if i != j and (i, j) not in mst_edges and (j, i) not in mst_edges:
                row_values.append((j, heuristic_matrix[i, j]))
        
        if not row_values:
            continue
            
        # Sort by heuristic value (descending)
        row_values.sort(key=lambda x: x[1], reverse=True)
        
        # Calculate how many additional edges we can add
        current_edges = sum(1 for j in range(n) if sparsified[i, j] > 0)
        max_additional = max(0, edges_per_node - current_edges)
        
        # Select top edges with exponential decay
        for rank, (j, value) in enumerate(row_values[:max_additional]):
            decay = np.exp(-rank / 3.0)  # Exponential decay factor
            sparsified[i, j] = value * decay
    
    # Ensure symmetry (take max of both directions)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Final normalization
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
