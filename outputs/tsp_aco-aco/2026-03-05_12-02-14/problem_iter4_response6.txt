import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic for TSP edge selection with size-adaptive components
    and controlled stochastic sparsification.
    
    Features:
    1. Inverse distance as base heuristic
    2. MST connectivity enhancement
    3. Adaptive sparsification with random edge retention
    4. Simple size-based weight adjustment
    5. Symmetric, non-negative output for stochastic sampling
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        # For trivial cases, use inverse distance directly
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / np.maximum(distance_matrix, 1e-10)
        np.fill_diagonal(result, 0)
        return result / result.sum(axis=1, keepdims=True).clip(min=1e-10)
    
    # ===== 1. BASE COMPONENTS =====
    # 1.1 Inverse distance (basic proximity)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / np.maximum(distance_matrix, 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # 1.2 MST connectivity
    try:
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = mst > 0
        
        # Boost MST edges with size-aware factor
        mst_strength = np.ones_like(distance_matrix)
        boost_factor = np.clip(1.0 + np.log(n) / 10, 1.0, 1.5)
        mst_strength[mst_mask] = boost_factor
    except Exception:
        # Fallback if MST fails
        mst_strength = np.ones_like(distance_matrix)
    
    # ===== 2. COMBINE WITH SIZE-ADAPTIVE WEIGHTS =====
    # Weights adjust based on problem size
    size_factor = min(1.0, np.log(n + 1) / np.log(100))
    w_inv = 0.6 - 0.2 * size_factor  # Less weight for inverse distance as size increases
    w_mst = 0.4 + 0.2 * size_factor  # More weight for MST as size increases
    
    combined = w_inv * inv_dist * mst_strength
    # Ensure symmetry by averaging
    combined = (combined + combined.T) / 2
    
    # ===== 3. STOCHASTIC SPARSIFICATION =====
    sparsified = np.zeros_like(combined)
    
    # Determine edges to keep per node adaptively
    base_k = max(3, min(n // 3, 15))  # Adaptive base: 3 to 15 edges
    k_vals = np.random.randint(max(2, base_k - 2), min(n, base_k + 3), size=n)
    
    for i in range(n):
        k = min(k_vals[i], n - 1)
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Stochastic ranking: sample edges proportional to their value
        row_nonzero = np.where(row > 0)[0]
        if len(row_nonzero) == 0:
            # Fallback: keep closest neighbors
            sorted_indices = np.argsort(distance_matrix[i])[1:k+1]
            sparsified[i, sorted_indices] = 1.0
            continue
            
        row_values = row[row_nonzero]
        row_values = np.maximum(row_values, 0)  # Ensure non-negative
        
        # Normalize to probabilities
        probs = row_values / row_values.sum()
        
        # Sample k edges with probability proportional to their value
        # Add small uniform noise to ensure exploration
        noise = 0.05 * np.random.random(probs.shape)
        probs = (1 - 0.05) * probs + 0.05 * (1 / len(probs))
        
        # Use weighted sampling without replacement approximation
        # We'll take top k edges after adding noise to break ties
        scores = row_values * (1 + 0.1 * np.random.randn(*row_values.shape))
        sorted_idx = np.argsort(scores)[::-1]
        keep_indices = row_nonzero[sorted_idx[:k]]
        
        # Apply values with mild decay based on rank
        for rank, j in enumerate(keep_indices):
            decay = np.exp(-rank / 3.0)  # Exponential decay
            sparsified[i, j] = combined[i, j] * decay
    
    # ===== 4. FINAL PROCESSING =====
    # Ensure symmetry
    result = (sparsified + sparsified.T) / 2
    
    # Add small random component for exploration (scaled by size)
    exploration_noise = 0.02 * np.log(n + 1) * np.random.randn(n, n)
    exploration_noise = np.abs(exploration_noise)  # Keep non-negative
    np.fill_diagonal(exploration_noise, 0)
    result = result + exploration_noise
    
    # Ensure non-negativity
    result = np.maximum(result, 0)
    np.fill_diagonal(result, 0)
    
    # Normalize rows for stochastic sampling
    row_sums = result.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    result = result / row_sums
    
    return result
