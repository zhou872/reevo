import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection that combines multiple factors
    and sparsifies unpromising edges to zero.
    
    Strategy:
    1. Combine inverse distance with node connectivity information
    2. Use mutual nearest neighbor relationships to identify promising edges
    3. Sparsify by setting edges to zero that aren't in top connections
    4. Incorporate local density information for adaptive thresholding
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1.0 / (distance_matrix + np.eye(n) * 1e-12)
    
    # 1. Basic inverse distance component (like v1)
    inv_dist = 1.0 / (distance_matrix + np.eye(n) * 1e-12)
    
    # 2. Node centrality/connectivity component
    # Nodes with many nearby neighbors might be in dense clusters
    node_connectivity = np.zeros(n)
    for i in range(n):
        # Consider average distance to k nearest neighbors
        k = max(2, n // 10)  # Adaptive k based on problem size
        nearest_dists = np.partition(distance_matrix[i], k)[:k+1]
        node_connectivity[i] = 1.0 / (np.mean(nearest_dists[nearest_dists > 0]) + 1e-12)
    
    # 3. Mutual nearest neighbor indicator
    # An edge is more promising if nodes are in each other's nearest neighbors
    nn_rank = np.argsort(distance_matrix, axis=1)
    mutual_nn = np.zeros((n, n))
    top_m = max(3, n // 8)  # Consider top m neighbors
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Check if j is in i's top neighbors and i is in j's top neighbors
            if (j in nn_rank[i, :top_m] and i in nn_rank[j, :top_m]):
                mutual_nn[i, j] = 1.0
            elif (j in nn_rank[i, :top_m*2] or i in nn_rank[j, :top_m*2]):
                mutual_nn[i, j] = 0.5  # Partial credit for one-way
    
    # 4. Density-aware adaptive thresholding
    # In dense regions, be more selective; in sparse regions, consider more edges
    local_density = np.zeros(n)
    for i in range(n):
        median_dist = np.median(distance_matrix[i, distance_matrix[i] > 0])
        local_density[i] = 1.0 / (median_dist + 1e-12)
    
    # 5. Combine all factors with adaptive weights
    heuristics = np.zeros((n, n))
    
    for i in range(n):
        for j in range(i+1, n):  # Use symmetry for efficiency
            if distance_matrix[i, j] <= 0:
                continue
                
            # Base score from inverse distance
            base_score = inv_dist[i, j]
            
            # Boost if both nodes have good connectivity
            connectivity_factor = (node_connectivity[i] + node_connectivity[j]) / 2.0
            
            # Mutual nearest neighbor factor
            nn_factor = 1.0 + mutual_nn[i, j]
            
            # Density adjustment: in dense areas, favor shorter edges more strongly
            density_factor = 1.0 + (local_density[i] + local_density[j]) / np.max(local_density)
            
            # Combined score
            combined_score = base_score * connectivity_factor * nn_factor * density_factor
            
            # Adaptive threshold: keep only edges that are sufficiently promising
            # Threshold based on node's local characteristics
            threshold_i = np.percentile(distance_matrix[i], 30) / distance_matrix[i, j]
            threshold_j = np.percentile(distance_matrix[j], 30) / distance_matrix[i, j]
            
            if combined_score > max(threshold_i, threshold_j) * 0.5:
                heuristics[i, j] = combined_score
                heuristics[j, i] = combined_score  # Symmetric
    
    # 6. Ensure connectivity: each node should have at least 2 promising connections
    for i in range(n):
        non_zero = np.count_nonzero(heuristics[i])
        if non_zero < 2:
            # Add the 2 nearest neighbors if they're missing
            nearest = np.argsort(distance_matrix[i])[1:3]  # Skip self
            for j in nearest:
                if heuristics[i, j] == 0:
                    heuristics[i, j] = inv_dist[i, j] * 0.5  # Reduced weight
                    heuristics[j, i] = inv_dist[i, j] * 0.5
    
    # 7. Final normalization for numerical stability
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
