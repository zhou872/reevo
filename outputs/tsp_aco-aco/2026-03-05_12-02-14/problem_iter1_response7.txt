import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP that combines multiple factors to assess
    edge promise and sparsifies the matrix by zeroing unpromising edges.
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.zeros_like(distance_matrix)
    
    # Avoid division by zero by setting diagonal to inf and handling zeros
    d = distance_matrix.copy()
    np.fill_diagonal(d, np.inf)
    d[d == 0] = 1e-12
    
    # 1. Basic inverse distance heuristic (as in v1)
    inverse_dist = 1.0 / d
    
    # 2. Nearest-neighbor factor: edges to k nearest neighbors get bonus
    k = max(2, int(np.sqrt(n)))
    nn_bonus = np.zeros_like(d)
    for i in range(n):
        # Get indices of k+1 smallest distances (first is self due to inf diagonal)
        nearest = np.argsort(d[i])[1:k+1]
        nn_bonus[i, nearest] = 1.0
    
    # Symmetrize nearest-neighbor bonus
    nn_bonus = np.maximum(nn_bonus, nn_bonus.T)
    
    # 3. Local density factor: favor edges within dense clusters
    # Compute average distance to m nearest neighbors for each node
    m = max(2, n // 5)
    density_scores = np.zeros(n)
    for i in range(n):
        nearest_dists = np.sort(d[i])[1:m+1]
        density_scores[i] = 1.0 / (np.mean(nearest_dists) + 1e-12)
    
    # Edge density score is geometric mean of endpoint densities
    density_factor = np.zeros_like(d)
    for i in range(n):
        for j in range(i+1, n):
            if d[i, j] < np.inf:
                geo_mean = np.sqrt(density_scores[i] * density_scores[j])
                density_factor[i, j] = density_factor[j, i] = geo_mean
    
    # Normalize density factor
    if density_factor.max() > 0:
        density_factor = density_factor / density_factor.max()
    
    # 4. Betweenness approximation: edges connecting different neighborhoods
    betweenness = np.zeros_like(d)
    # Use 2-opt style local connections: connect to nodes that are not too close
    for i in range(n):
        # Median distance as threshold
        median_dist = np.median(d[i, d[i] < np.inf])
        for j in range(i+1, n):
            if d[i, j] < np.inf:
                # Check if j is neither too close nor too far
                if 0.5 * median_dist < d[i, j] < 2.0 * median_dist:
                    betweenness[i, j] = betweenness[j, i] = 1.0
    
    # Combine factors with weighted sum
    w1, w2, w3, w4 = 1.0, 0.5, 0.3, 0.2  # Weights for different factors
    combined = (w1 * inverse_dist + 
                w2 * nn_bonus * inverse_dist + 
                w3 * density_factor + 
                w4 * betweenness)
    
    # Sparsification: keep only top edges per node
    sparsified = np.zeros_like(combined)
    keep_per_node = max(3, int(np.log2(n)) * 2)
    
    for i in range(n):
        # Get indices of top 'keep_per_node' edges for this node
        top_indices = np.argsort(combined[i])[-keep_per_node:]
        sparsified[i, top_indices] = combined[i, top_indices]
    
    # Ensure symmetry (take maximum of both directions)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Zero out diagonal
    np.fill_diagonal(sparsified, 0.0)
    
    return sparsified
