import numpy as np
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection combining multiple factors:
    1. Reciprocal distance (closer nodes are more promising)
    2. Node closeness centrality (well-connected nodes indicate promising edges)
    3. Adaptive sparsification with stochastic decay to focus on promising edges
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape indicating edge promise
    """
    n = distance_matrix.shape[0]
    
    # 1. Base reciprocal distance (handle zeros to avoid division by zero)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # 2. Node closeness centrality using distance matrix row sums
    # Nodes with smaller total distance to others are more central
    row_sums = np.sum(distance_matrix, axis=1)
    
    # Avoid division by zero for zero-sum rows (shouldn't happen in TSP)
    row_sums_safe = row_sums.copy()
    row_sums_safe[row_sums_safe == 0] = 1
    
    # Calculate closeness centrality: inverse of average distance to other nodes
    closeness_centrality = (n - 1) / row_sums_safe
    
    # Normalize to [0, 1]
    if np.max(closeness_centrality) > 0:
        closeness_centrality = closeness_centrality / np.max(closeness_centrality)
    
    # Create centrality matrix: edge promise based on centrality of its endpoints
    # Using outer product: edges between two central nodes are more promising
    centrality_matrix = np.outer(closeness_centrality, closeness_centrality)
    
    # 3. Combine factors with adaptive weighting
    # Weight factors based on graph size to adapt to different scenarios
    # For larger graphs, rely more on sparsification and centrality
    if n > 50:
        w1, w2 = 0.4, 0.6  # Less weight on reciprocal for large graphs
    else:
        w1, w2 = 0.6, 0.4  # More weight on reciprocal for small graphs
    
    combined = (w1 * reciprocal + w2 * centrality_matrix)
    
    # 4. Enhanced adaptive sparsification with stochastic decay
    sparsified = np.zeros_like(combined)
    
    # Adaptive k: keep more edges for smaller graphs, fewer for larger
    k = max(2, min(15, n // 3))  # Keep 2-15 edges per node
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self-edge
        
        # Get indices of top k edges for this node
        if k < n:
            # Use argpartition for efficiency with large n
            top_k_idx = np.argpartition(row, -k)[-k:]
        else:
            top_k_idx = np.arange(n)
            top_k_idx = top_k_idx[top_k_idx != i]
        
        # Get the values for top edges
        top_values = row[top_k_idx]
        
        # Sort top values in descending order
        sorted_indices = np.argsort(top_values)[::-1]
        sorted_top_k_idx = top_k_idx[sorted_indices]
        
        # Apply stochastic decay: reduce weight based on rank
        for rank, j in enumerate(sorted_top_k_idx):
            # Exponential decay based on rank: exp(-rank/τ)
            # Using τ=2 as in suggestion, but adapt based on k
            tau = max(1.5, k / 3)  # Adaptive tau based on k
            decay = np.exp(-rank / tau)
            sparsified[i, j] = combined[i, j] * decay
    
    # 5. Symmetrize the matrix (for symmetric TSP)
    # Take maximum of (i,j) and (j,i) to preserve the stronger signal
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 6. Ensure diagonal is zero (no self-edges)
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
