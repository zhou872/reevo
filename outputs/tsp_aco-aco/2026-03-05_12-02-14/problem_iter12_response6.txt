import numpy as np
from scipy import sparse, spatial, stats
from scipy.sparse import csgraph
import warnings
from typing import Dict, Tuple, List, Optional

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristic with ensemble learning and adaptive online tuning.
    
    Key innovations:
    1. Multi-strategy ensemble with performance-based weighting
    2. Online parameter refinement via solution feedback
    3. Enhanced feature engineering with topological analysis
    4. Adaptive connectivity-preserving sparsification
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Normalized heuristic matrix for stochastic edge sampling
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / np.maximum(distance_matrix, 1e-10)
        np.fill_diagonal(result, 0)
        row_sums = result.sum(axis=1, keepdims=True)
        row_sums[row_sums == 0] = 1
        return result / row_sums
    
    # ===== 1. GRAPH CHARACTERIZATION =====
    graph_metrics = compute_enhanced_metrics(distance_matrix, n)
    graph_type = classify_graph_type(graph_metrics)
    
    # ===== 2. INITIALIZE ENSEMBLE STRATEGIES =====
    strategies = initialize_ensemble_strategies(graph_type, graph_metrics, n)
    
    # ===== 3. ONLINE LEARNING LOOP =====
    best_weights = adaptive_ensemble_tuning(
        strategies, distance_matrix, graph_metrics, n
    )
    
    # ===== 4. COMPUTE ENSEMBLE HEURISTIC =====
    ensemble_heuristic = compute_ensemble_heuristic(
        strategies, best_weights, distance_matrix, n
    )
    
    # ===== 5. ADAPTIVE SPARSIFICATION =====
    sparsified = adaptive_sparsification(
        ensemble_heuristic, distance_matrix, graph_metrics, n
    )
    
    # ===== 6. FINAL REFINEMENT =====
    result = final_refinement(sparsified, distance_matrix, graph_metrics, n)
    
    return result


def compute_enhanced_metrics(distance_matrix: np.ndarray, n: int) -> Dict:
    """Compute comprehensive graph metrics with topological analysis."""
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    metrics = {
        'n': n,
        'dist_mean': np.mean(distances),
        'dist_std': np.std(distances),
        'dist_cv': np.std(distances) / (np.mean(distances) + 1e-10),
        'dist_min': np.min(distances),
        'dist_max': np.max(distances),
    }
    
    # Add distribution metrics
    if len(distances) > 2:
        metrics['dist_skew'] = stats.skew(distances)
        metrics['dist_kurtosis'] = stats.kurtosis(distances)
    else:
        metrics['dist_skew'] = 0
        metrics['dist_kurtosis'] = 0
    
    # Euclidean characteristics
    metrics['euclidean_score'] = estimate_euclidean_score(distance_matrix, n)
    
    # Topological metrics
    metrics['topology'] = analyze_topology(distance_matrix, n)
    
    # MST analysis
    try:
        mst = csgraph.minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = np.where(mst > 0)
        mst_lengths = mst[mst_edges]
        metrics['mst_mean'] = np.mean(mst_lengths)
        metrics['mst_std'] = np.std(mst_lengths)
        metrics['mst_ratio'] = np.sum(mst_lengths) / np.sum(distances)
    except:
        metrics['mst_mean'] = metrics['dist_mean']
        metrics['mst_std'] = metrics['dist_std']
        metrics['mst_ratio'] = 0.5
    
    return metrics


def estimate_euclidean_score(distance_matrix: np.ndarray, n: int) -> float:
    """Estimate Euclidean properties using triangle inequality."""
    if n < 3:
        return 1.0
    
    sample_size = min(50, n)
    indices = np.random.choice(n, sample_size, replace=False)
    
    violations = 0
    total_checks = 0
    
    for i in range(sample_size):
        for j in range(i + 1, sample_size):
            for k in range(j + 1, sample_size):
                a, b, c = indices[i], indices[j], indices[k]
                d_ab = distance_matrix[a, b]
                d_ac = distance_matrix[a, c]
                d_bc = distance_matrix[b, c]
                
                # Check triangle inequality with tolerance
                violations += int(d_ab > d_ac + d_bc + 1e-6 * d_ab)
                violations += int(d_ac > d_ab + d_bc + 1e-6 * d_ac)
                violations += int(d_bc > d_ab + d_ac + 1e-6 * d_bc)
                total_checks += 3
    
    return 1.0 - violations / max(total_checks, 1)


def analyze_topology(distance_matrix: np.ndarray, n: int) -> Dict:
    """Analyze graph topology and structure."""
    # Distance distribution analysis
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    q1, median, q3 = np.percentile(distances, [25, 50, 75])
    iqr = q3 - q1
    
    # Clustering tendency
    cluster_scores = []
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        sorted_dists = np.sort(row)
        if sorted_dists[1] > 0:
            cluster_scores.append(sorted_dists[0] / sorted_dists[1])
    
    cluster_score = 1.0 - np.mean(cluster_scores) if cluster_scores else 0.5
    
    # Neighborhood analysis
    avg_neighbors = []
    for i in range(n):
        threshold = np.percentile(distances, 25)
        neighbors = np.sum(distance_matrix[i] <= threshold) - 1
        avg_neighbors.append(neighbors)
    
    return {
        'q1': q1, 'median': median, 'q3': q3, 'iqr': iqr,
        'cluster_score': np.clip(cluster_score, 0, 1),
        'avg_neighbors': np.mean(avg_neighbors) / (n - 1),
        'density': len(distances) / (n * (n - 1) / 2)
    }


def classify_graph_type(metrics: Dict) -> str:
    """Classify graph type for adaptive strategy selection."""
    euclidean = metrics['euclidean_score']
    cv = metrics['dist_cv']
    cluster = metrics['topology']['cluster_score']
    
    if euclidean > 0.9:
        if cluster > 0.7:
            return "Euclidean_Clustered"
        elif cv < 0.3:
            return "Euclidean_Uniform"
        else:
            return "Euclidean_Varied"
    elif euclidean > 0.6:
        return "Semi_Euclidean"
    else:
        if cv > 1.2:
            return "NonEuclidean_Sparse"
        elif cluster > 0.6:
            return "NonEuclidean_Clustered"
        else:
            return "NonEuclidean_General"


def initialize_ensemble_strategies(
    graph_type: str, metrics: Dict, n: int
) -> List[Dict]:
    """Initialize multiple heuristic strategies for ensemble."""
    strategies = []
    
    # Strategy 1: Distance-focused
    strategies.append({
        'name': 'distance_focused',
        'weights': {'inv_dist': 0.6, 'mst': 0.3, 'centrality': 0.1},
        'sparsity_factor': min(0.3, 3/n),
        'temperature': 0.3
    })
    
    # Strategy 2: Topology-focused
    strategies.append({
        'name': 'topology_focused',
        'weights': {'inv_dist': 0.3, 'mst': 0.4, 'centrality': 0.3},
        'sparsity_factor': min(0.5, 5/n),
        'temperature': 0.5
    })
    
    # Strategy 3: Balanced
    strategies.append({
        'name': 'balanced',
        'weights': {'inv_dist': 0.4, 'mst': 0.3, 'centrality': 0.3},
        'sparsity_factor': min(0.4, 4/n),
        'temperature': 0.4
    })
    
    # Adjust based on graph type
    type_adjustments = {
        "Euclidean_Clustered": [
            {'weights': {'inv_dist': 0.3, 'mst': 0.5, 'centrality': 0.2}},
            {'weights': {'inv_dist': 0.4, 'mst': 0.4, 'centrality': 0.2}},
            {'weights': {'inv_dist': 0.3, 'mst': 0.5, 'centrality': 0.2}}
        ],
        "Euclidean_Uniform": [
            {'weights': {'inv_dist': 0.7, 'mst': 0.2, 'centrality': 0.1}},
            {'weights': {'inv_dist': 0.5, 'mst': 0.3, 'centrality': 0.2}},
            {'weights': {'inv_dist': 0.6, 'mst': 0.2, 'centrality': 0.2}}
        ],
        "NonEuclidean_Sparse": [
            {'weights': {'inv_dist': 0.2, 'mst': 0.3, 'centrality': 0.5}},
            {'weights': {'inv_dist': 0.3, 'mst': 0.3, 'centrality': 0.4}},
            {'weights': {'inv_dist': 0.2, 'mst': 0.4, 'centrality': 0.4}}
        ]
    }
    
    if graph_type in type_adjustments:
        adjustments = type_adjustments[graph_type]
        for i, adj in enumerate(adjustments):
            if i < len(strategies):
                strategies[i].update(adj)
    
    # Add exploration strategy
    strategies.append({
        'name': 'exploratory',
        'weights': {'inv_dist': 0.25, 'mst': 0.25, 'centrality': 0.5},
        'sparsity_factor': min(0.6, 6/n),
        'temperature': 0.8
    })
    
    return strategies


def compute_strategy_heuristic(
    strategy: Dict, distance_matrix: np.ndarray, n: int
) -> np.ndarray:
    """Compute heuristic matrix for a specific strategy."""
    # Compute base components
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / np.maximum(distance_matrix, np.percentile(distance_matrix, 10))
    np.fill_diagonal(inv_dist, 0)
    
    # MST component
    try:
        mst = csgraph.minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0).astype(float)
    except:
        mst_mask = np.zeros((n, n))
    
    # Centrality component
    centrality = compute_centrality(distance_matrix, n)
    
    # Combine with strategy weights
    weights = strategy['weights']
    total = sum(weights.values())
    normalized_weights = {k: v/total for k, v in weights.items()}
    
    heuristic = (
        normalized_weights['inv_dist'] * inv_dist +
        normalized_weights['mst'] * mst_mask +
        normalized_weights['centrality'] * centrality
    )
    
    # Apply temperature-based smoothing
    if strategy['temperature'] > 0:
        exp_heuristic = np.exp(heuristic / strategy['temperature'])
        row_sums = exp_heuristic.sum(axis=1, keepdims=True)
        row_sums[row_sums == 0] = 1
        heuristic = exp_heuristic / row_sums
    
    return heuristic


def compute_centrality(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute combined centrality measure."""
    # Closeness centrality
    row_sums = distance_matrix.sum(axis=1)
    closeness = 1 / (row_sums + 1e-10)
    
    # Degree centrality (based on proximity)
    threshold = np.percentile(distance_matrix, 25)
    degree = np.sum(distance_matrix <= threshold, axis=1) - 1
    
    # Harmonic centrality
    with np.errstate(divide='ignore', invalid='ignore'):
        harmonic = np.sum(1 / np.maximum(distance_matrix, 1e-10), axis=1)
    harmonic = np.nan_to_num(harmonic, nan=0)
    
    # Combine
    closeness_norm = closeness / (closeness.max() + 1e-10)
    degree_norm = degree / (degree.max() + 1e-10)
    harmonic_norm = harmonic / (harmonic.max() + 1e-10)
    
    combined = 0.4 * closeness_norm + 0.3 * degree_norm + 0.3 * harmonic_norm
    
    # Create centrality matrix
    centrality_matrix = np.outer(combined, combined)
    np.fill_diagonal(centrality_matrix, 0)
    
    return centrality_matrix


def adaptive_ensemble_tuning(
    strategies: List[Dict],
    distance_matrix: np.ndarray,
    metrics: Dict,
    n: int
) -> np.ndarray:
    """Tune ensemble weights using online performance feedback."""
    # Initialize weights
    strategy_weights = np.ones(len(strategies)) / len(strategies)
    
    # Number of tuning iterations
    n_iterations = min(100, n * 2)
    
    # Performance tracking
    performances = []
    
    for iteration in range(n_iterations):
        # Generate heuristic matrices for each strategy
        heuristics = []
        for strategy in strategies:
            heuristic = compute_strategy_heuristic(strategy, distance_matrix, n)
            heuristics.append(heuristic)
        
        # Combine with current weights
        combined = np.zeros_like(heuristics[0])
        for i, heuristic in enumerate(heuristics):
            combined += strategy_weights[i] * heuristic
        
        # Sample tours using combined heuristic
        tour_quality = evaluate_heuristic(combined, distance_matrix, n)
        
        # Update performance tracking
        performances.append(tour_quality)
        
        # Update strategy weights based on performance
        if iteration > 10:  # Warm-up period
            # Compute strategy contributions
            contributions = []
            for heuristic in heuristics:
                # Evaluate each strategy individually
                quality = evaluate_heuristic(heuristic, distance_matrix, n)
                contributions.append(quality)
            
            # Softmax update
            contributions = np.array(contributions)
            exp_contrib = np.exp((contributions - contributions.max()) / 0.1)
            new_weights = exp_contrib / exp_contrib.sum()
            
            # Momentum update
            strategy_weights = 0.7 * strategy_weights + 0.3 * new_weights
        
        # Add exploration noise
        if iteration < n_iterations // 2:
            noise = np.random.randn(len(strategies)) * 0.1
            strategy_weights += noise
            strategy_weights = np.maximum(strategy_weights, 0)
            strategy_weights /= strategy_weights.sum()
    
    return strategy_weights


def evaluate_heuristic(
    heuristic: np.ndarray,
    distance_matrix: np.ndarray,
    n: int,
    n_samples: int = 20
) -> float:
    """Evaluate heuristic quality by sampling tours."""
    if n_samples == 0:
        return 1.0
    
    tour_lengths = []
    
    for _ in range(min(n_samples, 50)):
        # Sample tour using heuristic as transition probabilities
        tour = sample_tour(heuristic, distance_matrix, n)
        
        # Compute tour length
        length = 0
        for i in range(n):
            j = (i + 1) % n
            length += distance_matrix[tour[i], tour[j]]
        
        tour_lengths.append(length)
    
    # Convert to quality score (lower length = higher quality)
    if len(tour_lengths) == 0:
        return 0.5
    
    avg_length = np.mean(tour_lengths)
    min_length = np.min(tour_lengths)
    
    # Quality score between 0 and 1
    if avg_length > min_length:
        quality = min_length / avg_length
    else:
        quality = 1.0
    
    return quality


def sample_tour(
    heuristic: np.ndarray,
    distance_matrix: np.ndarray,
    n: int
) -> np.ndarray:
    """Sample a tour using heuristic-guided stochastic selection."""
    # Start from random node
    current = np.random.randint(n)
    tour = [current]
    visited = set([current])
    
    while len(visited) < n:
        # Get unvisited nodes
        unvisited = [j for j in range(n) if j not in visited]
        
        if not unvisited:
            break
        
        # Get heuristic scores for unvisited nodes
        scores = heuristic[current, unvisited]
        
        # Add distance-based component for exploration
        distances = distance_matrix[current, unvisited]
        distance_scores = 1 / (distances + 1e-10)
        
        # Combined score
        combined_scores = 0.7 * scores + 0.3 * distance_scores
        
        # Normalize to probabilities
        if combined_scores.sum() > 0:
            probs = combined_scores / combined_scores.sum()
        else:
            probs = np.ones_like(combined_scores) / len(combined_scores)
        
        # Sample next node
        next_node = np.random.choice(unvisited, p=probs)
        
        tour.append(next_node)
        visited.add(next_node)
        current = next_node
    
    return np.array(tour)


def compute_ensemble_heuristic(
    strategies: List[Dict],
    weights: np.ndarray,
    distance_matrix: np.ndarray,
    n: int
) -> np.ndarray:
    """Compute final ensemble heuristic."""
    ensemble = np.zeros((n, n))
    
    for i, strategy in enumerate(strategies):
        heuristic = compute_strategy_heuristic(strategy, distance_matrix, n)
        ensemble += weights[i] * heuristic
    
    # Ensure symmetry and zero diagonal
    ensemble = (ensemble + ensemble.T) / 2
    np.fill_diagonal(ensemble, 0)
    
    return ensemble


def adaptive_sparsification(
    heuristic: np.ndarray,
    distance_matrix: np.ndarray,
    metrics: Dict,
    n: int
) -> np.ndarray:
    """Adaptive sparsification preserving connectivity."""
    # Determine sparsity level based on graph size and metrics
    if n < 50:
        edges_per_node = n - 1
    elif n < 200:
        edges_per_node = int(3 * np.sqrt(n))
    else:
        edges_per_node = int(2 * np.sqrt(n))
    
    # Adjust based on graph characteristics
    if metrics['dist_cv'] > 1.0:
        edges_per_node = int(edges_per_node * 1.2)
    
    if metrics['topology']['cluster_score'] > 0.6:
        edges_per_node = int(edges_per_node * 1.1)
    
    sparsified = np.zeros_like(heuristic)
    
    # Ensure MST edges are preserved
    try:
        mst = csgraph.minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = np.where(mst > 0)
        for i, j in zip(*mst_edges):
            sparsified[i, j] = heuristic[i, j]
    except:
        pass
    
    # Select top edges for each node
    for i in range(n):
        # Get candidate edges (excluding self and already preserved)
        candidates = []
        for j in range(n):
            if i != j and sparsified[i, j] == 0:
                candidates.append((j, heuristic[i, j]))
        
        if not candidates:
            continue
        
        # Sort by heuristic value
        candidates.sort(key=lambda x: x[1], reverse=True)
        
        # Select edges with rank-based decay
        n_select = min(edges_per_node, len(candidates))
        for rank, (j, value) in enumerate(candidates[:n_select]):
            decay = np.exp(-rank / 5.0)  # Exponential decay
            sparsified[i, j] = value * decay
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Ensure connectivity
    sparsified = ensure_connectivity(sparsified, distance_matrix, n)
    
    return sparsified


def ensure_connectivity(
    matrix: np.ndarray,
    distance_matrix: np.ndarray,
    n: int
) -> np.ndarray:
    """Ensure the matrix represents a connected graph."""
    result = matrix.copy()
    
    # Check connectivity
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        node = stack.pop()
        neighbors = np.where(result[node] > 0)[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    if not np.all(visited):
        # Add edges from MST to connect components
        try:
            mst = csgraph.minimum_spanning_tree(distance_matrix).toarray()
            for i in range(n):
                for j in range(n):
                    if i != j and mst[i, j] > 0 and result[i, j] == 0:
                        result[i, j] = result[j, i] = np.mean(result[result > 0])
        except:
            # Fallback: connect to nearest neighbors
            for i in range(n):
                if np.sum(result[i] > 0) == 0:
                    distances = distance_matrix[i].copy()
                    distances[i] = np.inf
                    nearest = np.argmin(distances)
                    result[i, nearest] = result[nearest, i] = 1.0
    
    return result


def final_refinement(
    matrix: np.ndarray,
    distance_matrix: np.ndarray,
    metrics: Dict,
    n: int
) -> np.ndarray:
    """Final refinement and normalization."""
    result = matrix.copy()
    
    # Add small random exploration
    if n > 10:
        noise_scale = 0.05 * (1 - metrics['euclidean_score'])
        noise = np.random.randn(n, n) * noise_scale
        noise = (noise + noise.T) / 2
        np.fill_diagonal(noise, 0)
        result = np.maximum(result + noise, 0)
    
    # Row normalization
    row_sums = result.sum(axis=1, keepdims=True)
    zero_rows = row_sums.flatten() == 0
    
    if np.any(zero_rows):
        # Fallback for isolated nodes
        with np.errstate(divide='ignore', invalid='ignore'):
            fallback = 1 / np.maximum(distance_matrix[zero_rows], metrics['dist_mean'] * 0.01)
        np.fill_diagonal(fallback, 0)
        result[zero_rows] = fallback
        row_sums[zero_rows] = fallback.sum(axis=1, keepdims=True)
    
    row_sums[row_sums == 0] = 1
    result = result / row_sums
    
    return result
