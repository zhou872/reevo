import numpy as np
from scipy import sparse

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic matrix for TSP edge selection with multiple graph theory
    principles and adaptive, instance-aware parameter tuning.
    
    The heuristic combines:
    1. Reciprocal distance (inverse distance) - favors short edges
    2. Closeness centrality - prioritizes edges between central nodes
    3. Minimum spanning tree indicator - edges likely in optimal solutions
    4. Nearest neighbor heuristic - edges to nearby nodes
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between nodes
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, with higher values
        indicating more promising edges (normalized to [0, 1])
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic[np.isinf(heuristic)] = 0
            heuristic[np.isnan(heuristic)] = 0
        np.fill_diagonal(heuristic, 0)
        if heuristic.max() > 0:
            heuristic /= heuristic.max()
        return heuristic
    
    # ========== COMPONENT 1: Reciprocal Distance ==========
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # ========== COMPONENT 2: Closeness Centrality ==========
    # Compute row sums (total distance from each node to all others)
    row_sums = np.sum(distance_matrix, axis=1, where=~np.eye(n, dtype=bool))
    # Avoid division by zero for isolated nodes
    row_sums[row_sums == 0] = 1
    closeness = 1 / row_sums
    
    # Normalize closeness to [0, 1]
    if closeness.max() > 0:
        closeness /= closeness.max()
    
    # Create centrality matrix via outer product
    centrality_matrix = np.outer(closeness, closeness)
    
    # ========== COMPONENT 3: Minimum Spanning Tree ==========
    # Edges in MST are often part of good TSP solutions
    mst = sparse.csgraph.minimum_spanning_tree(distance_matrix)
    mst_indicator = (mst > 0).astype(float)
    
    # ========== COMPONENT 4: Nearest Neighbor Heuristic ==========
    # For each node, boost edges to its k nearest neighbors
    nn_heuristic = np.zeros_like(distance_matrix)
    k = min(5, n - 1)  # Consider up to 5 nearest neighbors
    
    for i in range(n):
        # Get distances from node i to all others
        distances = distance_matrix[i].copy()
        distances[i] = np.inf  # Ignore self
        
        # Find indices of k nearest neighbors
        if k < n - 1:
            nn_indices = np.argpartition(distances, k)[:k]
        else:
            nn_indices = np.arange(n)
            nn_indices = nn_indices[nn_indices != i]
        
        # Apply exponential decay based on rank
        if len(nn_indices) > 0:
            # Sort by distance
            sorted_values = distances[nn_indices]
            sort_order = np.argsort(sorted_values)
            sorted_idx = nn_indices[sort_order]
            
            for rank, j in enumerate(sorted_idx):
                # Stronger boost for closer neighbors
                decay = np.exp(-rank / 2.0)
                nn_heuristic[i, j] = decay
    
    # Symmetrize the nearest neighbor heuristic
    nn_heuristic = np.maximum(nn_heuristic, nn_heuristic.T)
    
    # ========== ADAPTIVE WEIGHTING ==========
    # Analyze problem instance characteristics
    flat_dists = distance_matrix[~np.eye(n, dtype=bool)]
    dist_mean = np.mean(flat_dists)
    dist_std = np.std(flat_dists)
    dist_cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    
    # Determine weights based on problem characteristics
    if n < 20:
        # Small problems: emphasize reciprocal distance
        w_recip = 0.5
        w_cent = 0.2
        w_mst = 0.2
        w_nn = 0.1
    elif n < 100:
        # Medium problems: balanced approach
        w_recip = 0.4
        w_cent = 0.3
        w_mst = 0.2
        w_nn = 0.1
    else:
        # Large problems: emphasize centrality and MST
        w_recip = 0.3
        w_cent = 0.35
        w_mst = 0.25
        w_nn = 0.1
    
    # Adjust weights based on distance variation
    if dist_cv > 1.0:  # Highly variable distances
        # Give more weight to reciprocal distance
        w_recip *= 1.2
        w_cent *= 0.9
        w_mst *= 0.9
        w_nn *= 1.0
    
    # Normalize weights to sum to 1
    total = w_recip + w_cent + w_mst + w_nn
    w_recip /= total
    w_cent /= total
    w_mst /= total
    w_nn /= total
    
    # Combine all components
    combined = (w_recip * reciprocal +
                w_cent * centrality_matrix +
                w_mst * mst_indicator +
                w_nn * nn_heuristic)
    
    # ========== ADAPTIVE SPARSIFICATION ==========
    sparsified = np.zeros_like(combined)
    
    # Determine number of edges to keep per node
    # Keep more edges for smaller graphs, fewer for larger ones
    if n < 20:
        k_keep = min(10, n - 1)
    elif n < 50:
        k_keep = min(8, n - 1)
    elif n < 100:
        k_keep = min(6, n - 1)
    else:
        k_keep = min(5, n - 1)
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        if k_keep < n - 1:
            # Get indices of top k_keep edges
            top_indices = np.argpartition(row, -k_keep)[-k_keep:]
        else:
            top_indices = np.arange(n)
            top_indices = top_indices[top_indices != i]
        
        if len(top_indices) > 0:
            # Sort in descending order
            sorted_values = row[top_indices]
            sort_order = np.argsort(sorted_values)[::-1]
            sorted_idx = top_indices[sort_order]
            
            # Apply exponential decay based on rank
            for rank, j in enumerate(sorted_idx):
                decay = np.exp(-rank / 2.5)  # Smoother decay for more edges
                sparsified[i, j] = combined[i, j] * decay
    
    # ========== POST-PROCESSING ==========
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Zero out diagonal
    np.fill_diagonal(sparsified, 0)
    
    # Normalize to [0, 1] range
    max_val = sparsified.max()
    if max_val > 0:
        sparsified /= max_val
    
    return sparsified
