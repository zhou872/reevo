import numpy as np
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Simplified heuristic matrix for TSP edge selection with fixed component weights
    and adaptive sparsity control.
    
    Components:
    1. Reciprocal distances (60% weight)
    2. Node centrality matrix (30% weight)
    3. MST edge indicator (10% weight)
    
    Followed by adaptive sparsification preserving important edges.
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape indicating edge promise (higher = better)
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic[np.isinf(heuristic)] = 0
            heuristic = np.nan_to_num(heuristic, nan=0)
        np.fill_diagonal(heuristic, 0)
        return heuristic
    
    # ========== Component 1: Reciprocal Distances ==========
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal = np.nan_to_num(reciprocal, nan=0)
    np.fill_diagonal(reciprocal, 0)
    
    # Normalize reciprocal component
    if np.max(reciprocal) > 0:
        reciprocal_norm = reciprocal / np.max(reciprocal)
    else:
        reciprocal_norm = reciprocal
    
    # ========== Component 2: Node Centrality Matrix ==========
    # Compute node centrality as inverse of average distance to other nodes
    avg_distances = np.sum(distance_matrix, axis=1) / (n - 1)
    with np.errstate(divide='ignore', invalid='ignore'):
        node_centrality = 1 / avg_distances
        node_centrality[np.isinf(node_centrality)] = 0
        node_centrality = np.nan_to_num(node_centrality, nan=0)
    
    # Create centrality matrix (geometric mean of node centralities)
    centrality_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            if node_centrality[i] > 0 and node_centrality[j] > 0:
                centrality_matrix[i, j] = centrality_matrix[j, i] = np.sqrt(
                    node_centrality[i] * node_centrality[j]
                )
    
    # Normalize centrality component
    if np.max(centrality_matrix) > 0:
        centrality_norm = centrality_matrix / np.max(centrality_matrix)
    else:
        centrality_norm = centrality_matrix
    
    # ========== Component 3: MST Edge Indicator ==========
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    mst_indicator = (mst_array > 0).astype(float)
    
    # Normalize MST component
    if np.max(mst_indicator) > 0:
        mst_norm = mst_indicator / np.max(mst_indicator)
    else:
        mst_norm = mst_indicator
    
    # ========== Combine Components with Fixed Weights ==========
    combined = (0.6 * reciprocal_norm + 
                0.3 * centrality_norm + 
                0.1 * mst_norm)
    
    # Ensure symmetry
    combined = np.maximum(combined, combined.T)
    np.fill_diagonal(combined, 0)
    
    # ========== Adaptive Sparsity Control ==========
    # Determine edges to keep per node based on problem size
    if n < 50:
        edges_per_node = n - 1  # Keep all edges for small problems
    elif n < 200:
        edges_per_node = int(2 * np.sqrt(n))
    else:
        edges_per_node = int(1.5 * np.sqrt(n))
    
    # Always preserve MST edges
    preserved_edges = set()
    mst_edges = np.where(mst_array > 0)
    for i, j in zip(*mst_edges):
        if i != j:
            preserved_edges.add((i, j))
    
    # Create sparsified matrix
    sparsified = np.zeros_like(combined)
    
    # First, preserve MST edges
    for i, j in preserved_edges:
        sparsified[i, j] = combined[i, j]
    
    # For each node, select top additional edges with exponential decay
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get candidate edges (excluding already preserved)
        candidates = []
        for j in range(n):
            if i != j and (i, j) not in preserved_edges and (j, i) not in preserved_edges:
                candidates.append((j, row[j]))
        
        if not candidates:
            continue
            
        # Sort candidates by value
        candidates.sort(key=lambda x: x[1], reverse=True)
        
        # Select top edges with exponential decay
        max_additional = max(0, edges_per_node - 
                           sum(1 for j in range(n) if (i, j) in preserved_edges))
        selected = candidates[:max_additional]
        
        for rank, (j, value) in enumerate(selected):
            decay = np.exp(-rank / 3.0)  # Exponential decay
            sparsified[i, j] = value * decay
    
    # Ensure symmetry and final processing
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    # Final normalization
    max_val = sparsified.max()
    if max_val > 0:
        sparsified /= max_val
    
    return sparsified
