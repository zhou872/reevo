import numpy as np
from scipy import sparse
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection using harmonic centrality,
    minimum spanning tree, random tour sampling, and adaptive weight tuning.
    
    Args:
        distance_matrix: Square numpy array of pairwise distances
        
    Returns:
        Heuristic matrix of same shape indicating edge promise (higher = better)
    """
    n = distance_matrix.shape[0]
    
    # 1. Base reciprocal distance (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    np.fill_diagonal(reciprocal, 0)
    if np.max(reciprocal) > 0:
        reciprocal /= np.max(reciprocal)
    
    # 2. Harmonic centrality (efficient graph centrality measure)
    # Use 1/(1+distance) to avoid division by zero while emphasizing closeness
    dist_nonzero = np.where(np.eye(n), np.inf, distance_matrix)
    harmonic_centrality = np.sum(1 / (1 + dist_nonzero), axis=1)
    if np.max(harmonic_centrality) > 0:
        harmonic_centrality /= np.max(harmonic_centrality)
    centrality_matrix = np.outer(harmonic_centrality, harmonic_centrality)
    
    # 3. Minimum spanning tree indicator
    dist_sparse = sparse.csr_matrix(distance_matrix)
    mst = csgraph.minimum_spanning_tree(dist_sparse)
    mst_indicator = (mst.toarray() > 0).astype(float)
    
    # 4. Random tour sampling (local search feedback approximation)
    # Adaptive sampling based on instance size
    if n < 100:
        num_tours = 200
    elif n < 500:
        num_tours = 100
    else:
        num_tours = 50
    top_k = max(1, int(0.2 * num_tours))
    
    tour_based = np.zeros((n, n))
    tour_lengths = []
    tours = []
    
    # Generate random tours
    for _ in range(num_tours):
        tour = np.random.permutation(n)
        length = distance_matrix[tour, np.roll(tour, -1)].sum()
        tour_lengths.append(length)
        tours.append(tour)
    
    # Select best tours and weight edges by inverse tour length
    top_indices = np.argsort(tour_lengths)[:top_k]
    total_inv_length = 0
    
    for idx in top_indices:
        inv_len = 1 / tour_lengths[idx]
        total_inv_length += inv_len
        tour = tours[idx]
        
        # Add edges from this tour (both directions)
        for i in range(n):
            a = tour[i]
            b = tour[(i + 1) % n]
            tour_based[a, b] += inv_len
            tour_based[b, a] += inv_len
    
    # Normalize tour-based heuristic
    if total_inv_length > 0:
        tour_based /= total_inv_length
    if np.max(tour_based) > 0:
        tour_based /= np.max(tour_based)
    
    # 5. Adaptive weight tuning based on instance features
    # Compute coefficient of variation of distances
    triu_indices = np.triu_indices(n, 1)
    distances_upper = distance_matrix[triu_indices]
    mean_dist = np.mean(distances_upper)
    std_dist = np.std(distances_upper)
    cv = std_dist / (mean_dist + 1e-12)
    
    # Base weights: [reciprocal, centrality, MST, tour_based]
    base_weights = np.array([0.5, 0.2, 0.15, 0.15])
    
    # Adjust weights based on instance characteristics
    if n > 200:  # Larger instances benefit more from global structures
        base_weights[2] *= 1.2  # MST weight
        base_weights[3] *= 1.2  # Tour-based weight
    
    if cv > 1.0:  # High distance variability favors reciprocal distances
        base_weights[0] *= 1.5  # Reciprocal weight
    
    # Normalize to sum to 1
    weights = base_weights / np.sum(base_weights)
    
    # 6. Combine all components with tuned weights
    combined = (weights[0] * reciprocal +
                weights[1] * centrality_matrix +
                weights[2] * mst_indicator +
                weights[3] * tour_based)
    
    # 7. Adaptive sparsification with exponential decay
    sparsified = np.zeros_like(combined)
    
    # Adaptive k: keep more edges for larger graphs
    k = max(3, min(15, int(np.sqrt(n))))
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get top k edges for node i
        if k < len(row) - 1:
            top_k_idx = np.argpartition(row, -k)[-k:]
        else:
            top_k_idx = np.arange(n)
            top_k_idx = top_k_idx[top_k_idx != i]
        
        # Sort by value and apply exponential decay
        top_k_values = row[top_k_idx]
        sorted_indices = top_k_idx[np.argsort(top_k_values)[::-1]]
        
        for rank, j in enumerate(sorted_indices):
            decay = np.exp(-rank / 2.0)
            sparsified[i, j] = combined[i, j] * decay
    
    # Ensure symmetry and clean up
    symmetric_sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(symmetric_sparsified, 0)
    symmetric_sparsified[symmetric_sparsified < 1e-10] = 0
    
    return symmetric_sparsified
