import numpy as np
from scipy import sparse

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection featuring:
    1. Adaptive fusion of multiple centrality measures (closeness, shortest-path)
    2. Stochastic edge sparsification with CV-dependent adaptive ranges
    3. Improved exploration-exploitation balance via entropy-guided probabilities
    4. Graph topology-aware parameter adaptation
    5. Efficient computation with fallbacks for different graph types
    
    Returns: Non-negative matrix of same shape as input, normalized by row for stochastic sampling
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / np.maximum(distance_matrix, 1e-10)
        np.fill_diagonal(result, 0)
        row_sums = result.sum(axis=1, keepdims=True)
        row_sums[row_sums == 0] = 1
        return result / row_sums
    
    # ===== 1. BASE COMPONENTS =====
    # 1.1 Inverse distance (proximity heuristic)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / np.maximum(distance_matrix, 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # 1.2 MST connectivity (sparse computation for efficiency)
    mst_mask = np.zeros((n, n), dtype=bool)
    try:
        csr_dist = sparse.csr_matrix(distance_matrix)
        mst = sparse.csgraph.minimum_spanning_tree(csr_dist)
        mst_mask = mst.toarray() > 0
    except Exception:
        pass  # Keep zero mask if MST fails
    
    # ===== 2. GRAPH TOPOLOGY ANALYSIS =====
    # 2.1 Euclidean-ness measure
    euclidean_score = 0.0
    try:
        sample_indices = np.random.choice(n, min(20, n), replace=False)
        violations = 0
        total_checks = 0
        
        for i in sample_indices:
            for j in sample_indices:
                if i == j: continue
                for k in sample_indices:
                    if k == i or k == j: continue
                    if distance_matrix[i, j] > distance_matrix[i, k] + distance_matrix[k, j] + 1e-6:
                        violations += 1
                    total_checks += 1
        
        euclidean_score = 1.0 - (violations / max(total_checks, 1))
    except:
        euclidean_score = 0.5
    
    # 2.2 Density and structure analysis
    triu_indices = np.triu_indices(n, k=1)
    if len(triu_indices[0]) > 0:
        distances = distance_matrix[triu_indices]
        avg_dist = np.median(distances)
        std_dist = np.std(distances)
        cv_dist = std_dist / (avg_dist + 1e-10) if avg_dist > 0 else 1.0
        density_factor = np.clip(cv_dist, 0.5, 2.0)
    else:
        density_factor = 1.0
    
    # ===== 3. ENHANCED CENTRALITY COMBINATION =====
    # 3.1 Closeness centrality from row sums
    row_sums = distance_matrix.sum(axis=1)
    closeness = 1 / (row_sums + 1e-10)
    closeness = closeness / (closeness.max() + 1e-10)  # Normalize
    
    # 3.2 Shortest-path centrality via Floyd-Warshall (for moderate n)
    shortest_path_centrality = np.zeros(n)
    if n <= 500:  # Limit for computational feasibility
        try:
            # Compute all-pairs shortest paths
            dist_copy = distance_matrix.copy()
            np.fill_diagonal(dist_copy, 0)
            
            for k in range(n):
                dist_i_k = dist_copy[:, k:k+1]
                dist_k_j = dist_copy[k:k+1, :]
                alternative = dist_i_k + dist_k_j
                dist_copy = np.minimum(dist_copy, alternative)
            
            # Compute centrality based on shortest paths
            sp_row_sums = dist_copy.sum(axis=1)
            sp_centrality = 1 / (sp_row_sums + 1e-10)
            shortest_path_centrality = sp_centrality / (sp_centrality.max() + 1e-10)
        except:
            shortest_path_centrality = closeness  # Fallback
    else:
        shortest_path_centrality = closeness  # Fallback for large graphs
    
    # 3.3 Geometric mean combination
    centrality_combined = np.sqrt(closeness * shortest_path_centrality + 1e-10)
    
    # 3.4 MST degree centrality
    mst_degree_centrality = np.zeros(n)
    if np.any(mst_mask):
        mst_degree = mst_mask.sum(axis=1)
        if mst_degree.max() > 0:
            mst_degree_centrality = mst_degree / mst_degree.max()
    
    # 3.5 Final centrality fusion
    centrality_final = 0.5 * centrality_combined + 0.3 * closeness + 0.2 * mst_degree_centrality
    centrality_final = centrality_final / (centrality_final.max() + 1e-10)
    
    # 3.6 Create centrality matrix using outer product with geometric mean
    centrality_matrix = np.outer(centrality_final, centrality_final)
    
    # ===== 4. ADAPTIVE COMPONENT FUSION =====
    # Dynamic weights based on graph characteristics
    w_inv = 0.4 + 0.2 * euclidean_score  # More weight on inverse distance for Euclidean graphs
    w_mst = 0.3 * density_factor  # Adjust based on graph density
    w_cent = 0.3 * (1.0 - 0.5 * euclidean_score)  # More weight for non-Euclidean
    
    # Normalize
    total_weight = w_inv + w_mst + w_cent + 1e-10
    w_inv /= total_weight
    w_mst /= total_weight
    w_cent /= total_weight
    
    # Boost for MST edges in sparse graphs
    mst_boost = 1.0 + 0.8 * (1.0 / (density_factor + 0.5))
    
    # Combine components
    combined = (w_inv * inv_dist + 
                w_mst * mst_boost * mst_mask.astype(float) + 
                w_cent * centrality_matrix)
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    np.fill_diagonal(combined, 0)
    
    # ===== 5. STOCHASTIC EDGE SPARSIFICATION =====
    sparsified = np.zeros_like(combined)
    
    # Base k using sqrt(n) scaling
    base_k = max(3, min(20, int(np.sqrt(n))))
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf
        
        # Adaptive k based on coefficient of variation
        positive_mask = row > 0
        positive_values = row[positive_mask]
        
        if len(positive_values) == 0:
            # Fallback: use closest neighbors
            sorted_idx = np.argsort(distance_matrix[i])[1:base_k+1]
            sparsified[i, sorted_idx] = 1.0
            continue
        
        # Calculate coefficient of variation
        mean_val = positive_values.mean()
        std_val = positive_values.std()
        cv = std_val / (mean_val + 1e-10)
        
        # Adaptive k: more edges for high CV (diverse options)
        k = base_k + int(3 * cv)  # CV-dependent range
        k = min(max(k, 3), min(n-1, len(positive_values)))
        
        # Get candidate indices
        candidates = np.where(positive_mask)[0]
        candidate_values = positive_values
        
        # Exploration-exploitation balance via entropy
        if len(candidate_values) > 1:
            # Normalize for entropy calculation
            norm_values = candidate_values / (candidate_values.sum() + 1e-10)
            entropy = -np.sum(norm_values * np.log(norm_values + 1e-10))
            max_entropy = np.log(len(candidate_values))
            normalized_entropy = entropy / (max_entropy + 1e-10)
            
            # Temperature: higher for more uniform distributions (more exploration)
            temperature = 0.5 + 1.5 * normalized_entropy
        else:
            temperature = 1.0
        
        # Softmax with adaptive temperature
        exp_values = np.exp(candidate_values / (temperature + 1e-10))
        probs = exp_values / (exp_values.sum() + 1e-10)
        
        # Stochastic selection with controlled noise
        noise_scale = 0.1 * (1.0 - normalized_entropy if 'normalized_entropy' in locals() else 0.5)
        noise = noise_scale * np.random.randn(len(candidates))
        noisy_scores = candidate_values * (1.0 + noise)
        
        # Select top k with exponential decay
        top_k_idx = np.argsort(noisy_scores)[-k:]
        selected = candidates[top_k_idx]
        selected_values = candidate_values[top_k_idx]
        
        # Apply exponential decay to selected edges
        sorted_indices = np.argsort(selected_values)[::-1]
        for rank, idx in enumerate(sorted_indices):
            j = selected[idx]
            decay = np.exp(-rank / (1.0 + cv))  # CV-dependent decay rate
            sparsified[i, j] = selected_values[idx] * decay
    
    # ===== 6. FINAL ENHANCEMENTS =====
    # Ensure symmetry
    result = (sparsified + sparsified.T) / 2
    
    # Add adaptive exploration noise
    exploration_strength = 0.03 * (1.0 + cv_dist if 'cv_dist' in locals() else 1.0)
    noise = exploration_strength * np.random.rand(n, n)
    np.fill_diagonal(noise, 0)
    noise = np.abs(noise)  # Keep non-negative
    result = result + noise * result.mean()  # Scale noise by average value
    
    # Ensure non-negativity and zero diagonal
    result = np.maximum(result, 0)
    np.fill_diagonal(result, 0)
    
    # Row normalization for stochastic sampling
    row_sums = result.sum(axis=1, keepdims=True)
    zero_rows = row_sums.flatten() == 0
    
    if np.any(zero_rows):
        # For rows with no connections, use inverse distance as fallback
        fallback_weights = inv_dist[zero_rows]
        row_sums[zero_rows] = fallback_weights.sum(axis=1, keepdims=True)
        result[zero_rows] = fallback_weights
    
    # Normalize with stability
    row_sums = row_sums.copy()
    row_sums[row_sums == 0] = 1
    result = result / row_sums
    
    # Final symmetry enforcement
    result = (result + result.T) / 2
    
    return result
