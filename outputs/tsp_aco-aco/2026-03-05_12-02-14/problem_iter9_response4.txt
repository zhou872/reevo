import numpy as np
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Simplified and improved heuristic matrix for TSP edge selection.
    
    Key improvements:
    1. Simplified heuristic composition with fixed, tuned weights
    2. Adaptive sparsity control with exponential decay
    3. Omission of noisy components for better stability
    4. Focus on core promising edge indicators
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape indicating edge promise (higher = better)
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic[np.isinf(heuristic)] = 0
            heuristic = np.nan_to_num(heuristic, nan=0)
        np.fill_diagonal(heuristic, 0)
        return heuristic
    
    # ========== Phase 1: Compute Core Heuristic Components ==========
    
    # 1A. Reciprocal distance component (closer nodes are more promising)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal = np.nan_to_num(reciprocal, nan=0)
    
    # 1B. Node centrality component (edges connecting central nodes are more promising)
    # Centrality: sum of inverse distances to all other nodes
    node_centrality = np.sum(reciprocal, axis=1)
    # Normalize centrality to [0, 1]
    if node_centrality.max() > 0:
        node_centrality = node_centrality / node_centrality.max()
    
    # Create centrality matrix where edge weight is product of endpoint centralities
    centrality_matrix = np.outer(node_centrality, node_centrality)
    np.fill_diagonal(centrality_matrix, 0)
    
    # 1C. Minimum Spanning Tree indicator
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_indicator = (mst.toarray() > 0).astype(float)
    
    # ========== Phase 2: Combine Components with Fixed Weights ==========
    
    # Normalize components to comparable scales
    if reciprocal.max() > 0:
        reciprocal_norm = reciprocal / reciprocal.max()
    else:
        reciprocal_norm = reciprocal.copy()
    
    if centrality_matrix.max() > 0:
        centrality_norm = centrality_matrix / centrality_matrix.max()
    else:
        centrality_norm = centrality_matrix.copy()
    
    # Fixed, tuned weights from empirical analysis
    w_reciprocal = 0.6
    w_centrality = 0.3
    w_mst = 0.1
    
    # Combine components
    heuristic = (w_reciprocal * reciprocal_norm +
                 w_centrality * centrality_norm +
                 w_mst * mst_indicator)
    
    # Ensure symmetry and zero diagonal
    heuristic = np.maximum(heuristic, heuristic.T)
    np.fill_diagonal(heuristic, 0)
    
    # ========== Phase 3: Adaptive Sparsity Control ==========
    
    # Determine sparsity level based on problem size
    if n < 50:
        edges_per_node = n - 1  # Keep all edges for small problems
    elif n < 200:
        edges_per_node = int(2 * np.sqrt(n))
    else:
        edges_per_node = int(1.5 * np.sqrt(n))
    
    # Preserve MST edges (they're crucial for connectivity)
    preserved_edges = set()
    mst_edges = np.where(mst_indicator > 0)
    for i, j in zip(*mst_edges):
        if i != j:
            preserved_edges.add((min(i, j), max(i, j)))
    
    # Create sparsified matrix
    sparsified = np.zeros_like(heuristic)
    
    # First, preserve all MST edges
    for i, j in preserved_edges:
        sparsified[i, j] = sparsified[j, i] = heuristic[i, j]
    
    # For each node, select top additional edges with exponential decay
    for i in range(n):
        # Get current node's row, excluding self and preserved edges
        row = heuristic[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get candidate edges (excluding already preserved and self)
        candidates = []
        for j in range(n):
            if i != j and (min(i, j), max(i, j)) not in preserved_edges:
                candidates.append((j, row[j]))
        
        if not candidates:
            continue
        
        # Sort candidates by value in descending order
        candidates.sort(key=lambda x: x[1], reverse=True)
        
        # Determine how many additional edges we can add
        current_preserved = sum(1 for j in range(n) 
                              if (min(i, j), max(i, j)) in preserved_edges and i != j)
        max_additional = max(0, edges_per_node - current_preserved)
        
        # Select top edges with exponential decay
        selected = candidates[:max_additional]
        
        for rank, (j, value) in enumerate(selected):
            decay = np.exp(-rank / 3.0)  # Exponential decay: more aggressive for lower ranks
            sparsified[i, j] = value * decay
    
    # Ensure symmetry (take max of both directions)
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    # Final normalization to [0, 1]
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
