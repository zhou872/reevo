import numpy as np
from scipy import sparse

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection with improved strategies:
    1. Closeness centrality (well-connected nodes indicate promising edges)
    2. Adaptive sparsification with stochastic decay
    3. Edge prioritization based on normalized importance scores
    """
    n = distance_matrix.shape[0]
    
    # 1. Closeness centrality from distance matrix
    # Compute row sums (excluding self-distance of 0)
    row_sums = np.sum(distance_matrix, axis=1)
    
    # Avoid division by zero for small graphs or pathological cases
    row_sums = np.where(row_sums > 0, row_sums, 1)
    
    # Closeness centrality: inverse of average distance to other nodes
    # Normalize to [0,1] range
    closeness = 1 / row_sums
    closeness = closeness / np.max(closeness) if np.max(closeness) > 0 else closeness
    
    # 2. Create edge importance matrix via outer product of centrality scores
    # Edges connecting central nodes are more promising
    centrality_matrix = np.outer(closeness, closeness)
    
    # 3. Inverse distance component (closer nodes are more promising)
    # Handle zeros on diagonal and potential zeros elsewhere
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.isnan(inv_dist)] = 0
    
    # 4. Combine factors with adaptive weights
    # Weight factors based on their normalized contributions
    dist_norm = inv_dist / np.max(inv_dist) if np.max(inv_dist) > 0 else inv_dist
    cent_norm = centrality_matrix / np.max(centrality_matrix) if np.max(centrality_matrix) > 0 else centrality_matrix
    
    # Dynamic weighting: give more importance to centrality for larger graphs
    alpha = min(0.7, 0.3 + 0.1 * np.log10(n))  # 0.3-0.7 range
    beta = 1 - alpha
    
    combined = alpha * cent_norm + beta * dist_norm
    
    # 5. Adaptive sparsification with stochastic decay
    sparsified = np.zeros_like(combined)
    
    # Determine number of edges to keep per node adaptively
    # Keep more edges for smaller graphs, fewer for larger
    base_k = max(2, int(np.sqrt(n)))  # Between 2 and sqrt(n)
    k = min(base_k + 2, n - 1)  # Cap at n-1
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        if k < n:
            # Get top k indices without full sort
            top_k_idx = np.argpartition(row, -k)[-k:]
        else:
            top_k_idx = np.arange(n)
            top_k_idx = top_k_idx[top_k_idx != i]
        
        # Sort only the top k edges by importance
        top_k_values = row[top_k_idx]
        sorted_indices = top_k_idx[np.argsort(top_k_values)[::-1]]
        
        # Apply stochastic exponential decay based on rank
        for rank, j in enumerate(sorted_indices):
            # More aggressive decay for lower-ranked edges
            decay = np.exp(-rank / (k / 3))  # Adjust decay rate based on k
            sparsified[i, j] = combined[i, j] * decay
    
    # 6. Symmetrize the matrix (TSP is symmetric)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 7. Final normalization and diagonal zeroing
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
