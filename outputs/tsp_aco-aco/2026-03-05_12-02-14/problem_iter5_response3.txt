import numpy as np
from scipy import sparse

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection combining multiple factors:
    1. Reciprocal distance heuristic (closer nodes are more promising)
    2. Closeness centrality integration (well-connected nodes indicate promising edges)
    3. Adaptive sparsification with stochastic decay (keep most promising edges)
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of distances between nodes
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input with higher values indicating
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    
    # 1. Base reciprocal distance heuristic
    # Avoid division by zero for diagonal elements
    with np.errstate(divide='ignore', invalid='ignore'):
        # Use inverse distance as base heuristic
        # Add small epsilon to avoid division by zero
        epsilon = 1e-10
        reciprocal = 1 / (distance_matrix + epsilon)
        # Set diagonal to zero
        np.fill_diagonal(reciprocal, 0)
    
    # 2. Closeness centrality integration
    # Calculate node centrality as inverse of row sums (excluding diagonal)
    row_sums = np.sum(distance_matrix, axis=1) - np.diag(distance_matrix)
    closeness_centrality = 1 / (row_sums + epsilon)
    
    # Normalize centrality to [0, 1] range
    if np.max(closeness_centrality) > np.min(closeness_centrality):
        closeness_centrality = (closeness_centrality - np.min(closeness_centrality)) / \
                              (np.max(closeness_centrality) - np.min(closeness_centrality))
    
    # Create centrality matrix via outer product
    # Edges between central nodes are more promising
    centrality_matrix = np.outer(closeness_centrality, closeness_centrality)
    
    # 3. Combine factors with adaptive weighting
    # Weight factors based on problem size and structure
    if n < 20:
        w1, w2 = 0.7, 0.3  # More weight to reciprocal for small graphs
    else:
        w1, w2 = 0.5, 0.5  # Balanced weights for larger graphs
    
    # Combine heuristic factors
    combined = w1 * reciprocal + w2 * centrality_matrix
    
    # 4. Adaptive sparsification with stochastic decay
    sparsified = np.zeros_like(combined)
    
    # Determine sparsity level adaptively based on graph size
    # Keep more edges for smaller graphs to maintain connectivity
    k = max(2, min(8, n // 3))  # Keep 2-8 edges per node
    
    for i in range(n):
        # Get row values excluding self
        row = combined[i].copy()
        row[i] = -np.inf  # Mark diagonal as excluded
        
        # Get indices of top k edges
        if k < n - 1:  # -1 because we exclude diagonal
            # Get indices of top k values
            top_k_idx = np.argpartition(row, -k)[-k:]
        else:
            # If k is too large, take all except diagonal
            top_k_idx = np.arange(n)
            top_k_idx = top_k_idx[top_k_idx != i]
        
        # Sort top indices by their values in descending order
        sorted_idx = top_k_idx[np.argsort(row[top_k_idx])[::-1]]
        
        # Apply exponential decay based on rank
        for rank, j in enumerate(sorted_idx):
            # Exponential decay: higher rank (0-based) gets more decay
            decay = np.exp(-rank / 2.0)  # Using /2.0 as in the suggestion
            sparsified[i, j] = combined[i, j] * decay
    
    # Ensure symmetry (since TSP is symmetric)
    sparsified = (sparsified + sparsified.T) / 2
    
    # Final cleanup: ensure diagonal is zero
    np.fill_diagonal(sparsified, 0)
    
    # Normalize to [0, 1] range for consistency
    if np.max(sparsified) > 0:
        sparsified = sparsified / np.max(sparsified)
    
    return sparsified
