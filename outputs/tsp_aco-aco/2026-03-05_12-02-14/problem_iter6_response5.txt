import numpy as np
from scipy import sparse, spatial, stats
from scipy.sparse import csgraph
import numba

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection with local search feedback
    and improved structural analysis.
    
    Key improvements over v1:
    1. Local search feedback: Uses 2-opt improvement on sampled tours to refine edge preferences
    2. Adaptive centrality fusion: Dynamically weights centrality measures based on graph structure
    3. Distance distribution analysis: Uses statistical features to adjust heuristic weights
    4. Efficient sparse representation: Optimizes memory usage for large instances
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape indicating edge promise (higher = better)
    """
    n = distance_matrix.shape[0]
    
    # 1. Base reciprocal distance (fundamental attraction: closer nodes are better)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = np.zeros_like(distance_matrix, dtype=np.float64)
        mask = distance_matrix > 0
        reciprocal[mask] = 1.0 / distance_matrix[mask]
        np.fill_diagonal(reciprocal, 0)
    
    # Normalize to [0, 1] range
    recip_max = reciprocal.max()
    if recip_max > 0:
        reciprocal /= recip_max
    
    # 2. Enhanced centrality fusion with adaptive weighting
    def compute_centrality_measures():
        """Compute and fuse multiple centrality measures adaptively."""
        # Closeness centrality: inverse of average distance
        row_sums = np.zeros(n, dtype=np.float64)
        for i in range(n):
            valid_distances = distance_matrix[i][distance_matrix[i] > 0]
            if len(valid_distances) > 0:
                row_sums[i] = np.mean(valid_distances)
        closeness = np.zeros(n, dtype=np.float64)
        valid_mask = row_sums > 0
        closeness[valid_mask] = 1.0 / row_sums[valid_mask]
        
        # Harmonic centrality: average of inverse distances (more robust for disconnected graphs)
        harmonic = np.zeros(n, dtype=np.float64)
        for i in range(n):
            valid_mask = (distance_matrix[i] > 0) & (~np.isinf(distance_matrix[i]))
            if np.any(valid_mask):
                harmonic[i] = np.mean(1.0 / distance_matrix[i, valid_mask])
        
        # Betweenness approximation using triangle inequality violation
        triangle_violations = np.zeros(n, dtype=np.float64)
        for i in range(n):
            violations = 0
            for j in range(n):
                if i == j: continue
                for k in range(n):
                    if k == i or k == j: continue
                    if distance_matrix[i, j] + distance_matrix[j, k] < distance_matrix[i, k]:
                        violations += 1
            triangle_violations[i] = violations
        
        # Normalize each measure
        measures = []
        for measure in [closeness, harmonic, triangle_violations]:
            if np.ptp(measure) > 0:
                normalized = (measure - measure.min()) / np.ptp(measure)
                measures.append(normalized)
            else:
                measures.append(np.zeros_like(measure))
        
        # Adaptive weighting based on measure consistency
        correlations = np.zeros(3, dtype=np.float64)
        for i in range(3):
            for j in range(i+1, 3):
                corr = np.corrcoef(measures[i], measures[j])[0, 1]
                if not np.isnan(corr):
                    correlations[i] += abs(corr)
                    correlations[j] += abs(corr)
        
        if correlations.sum() > 0:
            weights = correlations / correlations.sum()
        else:
            weights = np.array([0.4, 0.4, 0.2])
        
        # Fuse measures
        fused = np.zeros(n, dtype=np.float64)
        for i in range(3):
            fused += weights[i] * measures[i]
        
        # Final normalization
        if np.ptp(fused) > 0:
            fused = (fused - fused.min()) / np.ptp(fused)
        
        return fused
    
    centrality = compute_centrality_measures()
    centrality_matrix = np.outer(centrality, centrality)
    
    # 3. Minimum spanning tree with edge importance weighting
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    
    # Weight MST edges by their relative importance in the tree
    mst_weighted = np.zeros_like(distance_matrix, dtype=np.float64)
    non_zero_mask = mst_array > 0
    mst_weighted[non_zero_mask] = 1.0 / mst_array[non_zero_mask]
    
    # Normalize MST weights
    if mst_weighted.max() > 0:
        mst_weighted /= mst_weighted.max()
    
    # 4. Local search feedback with stochastic tour sampling
    @numba.njit(nogil=True, parallel=True)
    def sample_and_improve_tours(num_tours, top_k, distance_matrix):
        """Sample random tours, apply 2-opt improvements, and collect edge frequencies."""
        n = distance_matrix.shape[0]
        edge_counts = np.zeros((n, n), dtype=np.float64)
        tour_lengths = np.zeros(num_tours, dtype=np.float64)
        
        for t in numba.prange(num_tours):
            # Generate random tour
            tour = np.arange(n)
            np.random.shuffle(tour)
            
            # Simple 2-opt improvement (limited iterations for speed)
            improved = True
            for _ in range(min(10, n)):
                if not improved:
                    break
                improved = False
                for i in range(n):
                    for j in range(i+2, n):
                        # Check if swap improves
                        a1, a2 = tour[i], tour[(i+1)%n]
                        b1, b2 = tour[j], tour[(j+1)%n]
                        old_dist = (distance_matrix[a1, a2] + distance_matrix[b1, b2])
                        new_dist = (distance_matrix[a1, b1] + distance_matrix[a2, b2])
                        if new_dist < old_dist:
                            # Reverse segment between i+1 and j
                            tour[i+1:j+1] = tour[i+1:j+1][::-1]
                            improved = True
            
            # Calculate tour length
            length = 0.0
            for i in range(n):
                length += distance_matrix[tour[i], tour[(i+1)%n]]
            tour_lengths[t] = length
            
            # Add edges to counts (weighted by inverse length)
            weight = 1.0 / (length + 1e-10)
            for i in range(n):
                a, b = tour[i], tour[(i+1)%n]
                edge_counts[a, b] += weight
                edge_counts[b, a] += weight
        
        return edge_counts, tour_lengths
    
    # Adaptive sampling based on instance size
    if n < 100:
        num_tours = 200
    elif n < 500:
        num_tours = 100
    else:
        num_tours = 50
    
    top_k = max(1, int(0.2 * num_tours))
    local_feedback, lengths = sample_and_improve_tours(num_tours, top_k, distance_matrix)
    
    # Normalize local feedback
    if local_feedback.max() > 0:
        local_feedback /= local_feedback.max()
    
    # 5. Adaptive weight tuning based on distance distribution analysis
    # Extract upper triangle distances (excluding diagonal)
    triu_indices = np.triu_indices(n, 1)
    distances_flat = distance_matrix[triu_indices]
    
    # Compute distribution statistics
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    dist_skew = stats.skew(distances_flat) if len(distances_flat) > 2 else 0
    
    # Base weights for components
    weights = np.array([0.35, 0.30, 0.20, 0.15], dtype=np.float64)  # [reciprocal, centrality, MST, local_feedback]
    
    # Adjust weights based on instance characteristics
    if dist_std / (dist_mean + 1e-10) > 1.0:  # High variance
        weights[0] *= 1.2  # Emphasize reciprocal distances
        weights[3] *= 0.8  # Reduce reliance on local feedback (less reliable in varied instances)
    
    if dist_skew > 1.0:  # Right-skewed distribution (many short distances, few long)
        weights[1] *= 1.1  # Centrality more important
        weights[2] *= 0.9  # MST less reliable
    
    if n > 200:  # Large instances
        weights[3] *= 1.2  # Local feedback more valuable
        weights[0] *= 0.9  # Reciprocal less discriminative
    
    # Normalize weights to sum to 1
    weights /= weights.sum()
    
    # 6. Combine all components
    combined = (weights[0] * reciprocal +
                weights[1] * centrality_matrix +
                weights[2] * mst_weighted +
                weights[3] * local_feedback)
    
    # 7. Intelligent sparsification with structural preservation
    sparsified = np.zeros_like(combined, dtype=np.float64)
    
    # Determine sparsity level adaptively
    if n < 50:
        k_per_node = min(10, n-1)
    elif n < 200:
        k_per_node = int(np.ceil(np.log2(n) * 2))
    else:
        k_per_node = int(np.ceil(np.sqrt(n)))
    
    # Preserve MST edges regardless of ranking
    mst_edges = np.where(mst_array > 0)
    for i, j in zip(*mst_edges):
        if i != j:
            sparsified[i, j] = combined[i, j]
    
    # For each node, select top k edges with exponential decay
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get top indices
        if k_per_node < n-1:
            # Get top k_per_node + some extra for safety
            top_indices = np.argpartition(row, -(k_per_node + 5))[-(k_per_node + 5):]
        else:
            top_indices = np.arange(n)
            top_indices = top_indices[top_indices != i]
        
        # Sort and apply decay
        top_values = row[top_indices]
        sorted_indices = top_indices[np.argsort(top_values)[::-1][:k_per_node]]
        
        for rank, j in enumerate(sorted_indices):
            # Exponential decay based on rank
            decay = np.exp(-rank / 2.0)
            # Boost if also in MST
            if mst_array[i, j] > 0:
                decay *= 1.5
            sparsified[i, j] = combined[i, j] * decay
    
    # Ensure symmetry and clean up
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    # Final normalization
    max_val = sparsified.max()
    if max_val > 0:
        sparsified /= max_val
    
    # Clip very small values to zero for sparsity
    sparsified[sparsified < 1e-10] = 0
    
    return sparsified
