import numpy as np
from scipy.sparse import csgraph
from typing import Tuple, List

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic matrix for TSP edge selection with adaptive learning.
    
    Features:
    1. Dynamically tuned factor weights based on solution quality estimation
    2. Incorporates 2-opt and nearest neighbor heuristics
    3. Statistical edge promise from sampling diverse candidate solutions
    4. Hierarchical sparsification with adaptive thresholds
    """
    n = distance_matrix.shape[0]
    
    # Base reciprocal distance with smoothing
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1.0 / (distance_matrix + np.eye(n) * 1e-10)
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # 1. Dynamic weight adaptation based on matrix statistics
    weights = compute_adaptive_weights(distance_matrix)
    
    # 2. MST-based heuristic with edge criticality
    mst_heuristic = compute_mst_heuristic(distance_matrix)
    
    # 3. Local search heuristic (2-opt neighborhood potential)
    local_search_heuristic = compute_local_search_heuristic(distance_matrix)
    
    # 4. Statistical sampling of promising edges
    statistical_heuristic = compute_statistical_heuristic(distance_matrix)
    
    # 5. Combine with dynamic weights
    combined = (weights[0] * reciprocal + 
                weights[1] * mst_heuristic + 
                weights[2] * local_search_heuristic + 
                weights[3] * statistical_heuristic)
    
    # 6. Hierarchical sparsification
    sparsified = hierarchical_sparsification(combined, distance_matrix)
    
    # 7. Ensure symmetry and clean diagonal
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    # 8. Normalize for probabilistic interpretation
    row_sums = sparsified.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    sparsified = sparsified / row_sums
    
    return sparsified

def compute_adaptive_weights(distance_matrix: np.ndarray) -> np.ndarray:
    """Dynamically adjust heuristic weights based on distance matrix properties."""
    n = distance_matrix.shape[0]
    
    # Analyze distance distribution
    distances_flat = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    dist_cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    
    # Weight adaptation rules
    if dist_cv < 0.2:  # Uniform distances
        # Emphasize MST and local search more
        weights = np.array([0.2, 0.4, 0.3, 0.1])
    elif dist_cv > 0.8:  # Highly varied distances
        # Emphasize reciprocal distance more
        weights = np.array([0.5, 0.2, 0.2, 0.1])
    else:  # Moderate variation
        weights = np.array([0.3, 0.3, 0.25, 0.15])
    
    # Adjust based on problem size
    size_factor = min(1.0, 50 / n)  # Favor statistical methods for larger problems
    weights[3] *= (1 + size_factor)
    weights = weights / weights.sum()  # Renormalize
    
    return weights

def compute_mst_heuristic(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute MST-based heuristic with edge criticality analysis."""
    n = distance_matrix.shape[0]
    
    # Get MST
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_indicator = (mst > 0).astype(float)
    
    # Edge criticality: how many alternative MSTs include this edge?
    criticality = np.zeros((n, n))
    
    for i in range(n):
        for j in range(i + 1, n):
            if mst_indicator[i, j] > 0:
                # Temporarily remove edge (i,j)
                temp_dist = distance_matrix.copy()
                temp_dist[i, j] = temp_dist[j, i] = np.inf
                
                # Compute new MST
                new_mst = csgraph.minimum_spanning_tree(temp_dist)
                new_length = new_mst.sum()
                
                # Criticality inversely proportional to length increase
                original_length = mst.sum()
                length_increase = max(0, new_length - original_length)
                criticality[i, j] = criticality[j, i] = 1.0 / (1.0 + length_increase)
    
    return mst_indicator * (1.0 + 0.5 * criticality)

def compute_local_search_heuristic(distance_matrix: np.ndarray) -> np.ndarray:
    """Estimate edge promise based on 2-opt and nearest neighbor heuristics."""
    n = distance_matrix.shape[0]
    heuristic = np.zeros((n, n))
    
    # Nearest neighbor analysis
    for i in range(n):
        # Find k nearest neighbors
        k = min(10, n // 2)
        nearest = np.argsort(distance_matrix[i])[1:k+1]
        
        # Check 2-opt potential for each neighbor
        for j in nearest:
            if i == j:
                continue
                
            # Estimate 2-opt gain if edge (i,j) is included
            gain_sum = 0
            count = 0
            
            # Sample potential 2-opt swaps
            for _ in range(min(20, n)):
                k = np.random.randint(n)
                l = np.random.randint(n)
                if len(set([i, j, k, l])) == 4:
                    current_cost = (distance_matrix[i, j] + distance_matrix[k, l])
                    swapped_cost = (distance_matrix[i, k] + distance_matrix[j, l])
                    gain = max(0, current_cost - swapped_cost)
                    gain_sum += gain
                    count += 1
            
            if count > 0:
                avg_gain = gain_sum / count
                heuristic[i, j] = 1.0 / (1.0 + np.exp(-avg_gain))
    
    return heuristic

def compute_statistical_heuristic(distance_matrix: np.ndarray) -> np.ndarray:
    """Estimate edge promise by sampling diverse candidate solutions."""
    n = distance_matrix.shape[0]
    edge_counts = np.zeros((n, n))
    n_samples = min(100, 10 * n)
    
    # Generate diverse candidate solutions
    for _ in range(n_samples):
        # Mix construction methods
        if np.random.random() < 0.5:
            tour = nearest_neighbor_tour(distance_matrix)
        else:
            tour = random_insertion_tour(distance_matrix)
        
        # Apply 2-opt local search
        improved_tour = two_opt(tour, distance_matrix, max_iterations=50)
        
        # Count edges in final tour
        for i in range(n):
            j = improved_tour[(i + 1) % n]
            edge_counts[i, j] += 1
            edge_counts[j, i] += 1
    
    # Convert counts to probabilities with Laplace smoothing
    edge_probs = (edge_counts + 1) / (n_samples + n)
    return edge_probs

def nearest_neighbor_tour(distance_matrix: np.ndarray) -> List[int]:
    """Construct tour using nearest neighbor heuristic."""
    n = distance_matrix.shape[0]
    unvisited = set(range(n))
    tour = []
    
    # Start from random city
    current = np.random.randint(n)
    tour.append(current)
    unvisited.remove(current)
    
    while unvisited:
        # Find nearest unvisited neighbor
        nearest = min(unvisited, key=lambda x: distance_matrix[current, x])
        tour.append(nearest)
        unvisited.remove(nearest)
        current = nearest
    
    return tour

def random_insertion_tour(distance_matrix: np.ndarray) -> List[int]:
    """Construct tour using random insertion heuristic."""
    n = distance_matrix.shape[0]
    
    # Start with triangle of 3 random cities
    start_indices = np.random.choice(n, 3, replace=False)
    tour = list(start_indices)
    
    remaining = [i for i in range(n) if i not in tour]
    np.random.shuffle(remaining)
    
    for city in remaining:
        # Find best insertion position
        best_pos = 0
        best_increase = np.inf
        
        for i in range(len(tour)):
            j = (i + 1) % len(tour)
            increase = (distance_matrix[tour[i], city] + 
                       distance_matrix[city, tour[j]] - 
                       distance_matrix[tour[i], tour[j]])
            
            if increase < best_increase:
                best_increase = increase
                best_pos = i + 1
        
        tour.insert(best_pos, city)
    
    return tour

def two_opt(tour: List[int], distance_matrix: np.ndarray, max_iterations: int = 100) -> List[int]:
    """Apply 2-opt local search to improve tour."""
    n = len(tour)
    improved = True
    iterations = 0
    
    while improved and iterations < max_iterations:
        improved = False
        iterations += 1
        
        for i in range(n):
            for j in range(i + 2, n):
                # Calculate gain from swapping edges
                a, b = tour[i], tour[(i + 1) % n]
                c, d = tour[j], tour[(j + 1) % n]
                
                current = distance_matrix[a, b] + distance_matrix[c, d]
                proposed = distance_matrix[a, c] + distance_matrix[b, d]
                
                if proposed < current:
                    # Reverse segment between i+1 and j
                    tour[(i + 1) % n:(j + 1) % n] = reversed(tour[(i + 1) % n:(j + 1) % n])
                    improved = True
                    break
            
            if improved:
                break
    
    return tour

def hierarchical_sparsification(matrix: np.ndarray, distance_matrix: np.ndarray) -> np.ndarray:
    """Apply multi-level sparsification with adaptive thresholds."""
    n = matrix.shape[0]
    sparsified = np.zeros_like(matrix)
    
    # Level 1: Keep MST edges
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_mask = (mst > 0)
    
    # Level 2: Keep top-k edges per node (adaptive k)
    base_k = max(3, min(10, int(np.sqrt(n))))
    
    # Adjust k based on node degree in MST
    for i in range(n):
        # Node-specific k based on centrality
        mst_degree = mst_mask[i].sum() + mst_mask[:, i].sum()
        k_i = base_k + int(mst_degree / 2)
        
        # Get top k_i edges
        row = matrix[i].copy()
        row[i] = -np.inf
        
        if k_i < n - 1:
            top_indices = np.argpartition(row, -k_i)[-k_i:]
        else:
            top_indices = [j for j in range(n) if j != i]
        
        # Apply rank-based decay
        sorted_indices = top_indices[np.argsort(row[top_indices])[::-1]]
        for rank, j in enumerate(sorted_indices):
            decay = np.exp(-rank / 3.0)  # Slower decay than v1
            sparsified[i, j] = matrix[i, j] * decay
    
    # Level 3: Reinforce MST edges
    sparsified[mst_mask] = np.maximum(sparsified[mst_mask], matrix[mst_mask])
    
    return sparsified
