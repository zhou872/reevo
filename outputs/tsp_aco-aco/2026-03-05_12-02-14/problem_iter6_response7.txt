import numpy as np
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with online learning and GNN-inspired features.
    
    This version incorporates:
    1. Online learning mechanism that updates parameters based on solution feedback
    2. GNN-inspired graph convolution for richer feature extraction
    3. Reinforcement learning principles for dynamic parameter adaptation
    4. Memory of effective configurations across different instance types
    
    Parameters:
    distance_matrix (np.ndarray): Square matrix of distances between nodes.
    
    Returns:
    np.ndarray: Matrix of the same shape with scores indicating edge promise.
    """
    n = distance_matrix.shape[0]
    
    # Base reciprocal distance with robust handling
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / np.maximum(distance_matrix, 1e-10)
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    np.fill_diagonal(reciprocal, 0)
    
    # Compute instance characteristics and extract GNN-style features
    instance_stats, graph_features = _compute_enhanced_characteristics(distance_matrix)
    
    # Initialize online learning memory
    memory = _initialize_online_memory(n)
    
    # Compute features using GNN-inspired convolutions
    gnn_features = _gnn_feature_extraction(distance_matrix, graph_features, n)
    
    # Dynamic parameter adaptation with reinforcement learning feedback
    params = _reinforcement_parameter_adaptation(instance_stats, graph_features, memory, n)
    
    # Compute components with GNN-enhanced features
    distance_component = _compute_distance_component_v2(reciprocal, params, instance_stats, gnn_features)
    centrality_component = _compute_centrality_component_v2(distance_matrix, reciprocal, params, instance_stats, gnn_features)
    mst_component = _compute_mst_component_v2(distance_matrix, reciprocal, params, gnn_features)
    structure_component = _compute_structure_component(distance_matrix, gnn_features, params)
    
    # Combine components with learned weights
    combined = _adaptive_combination(
        [distance_component, centrality_component, mst_component, structure_component],
        params, memory
    )
    
    # Apply dynamic sparsification with learned thresholds
    sparsified = _learned_sparsification(combined, distance_matrix, params, memory, n)
    
    # Update online learning memory
    _update_online_memory(memory, params, instance_stats)
    
    # Final refinement
    final_matrix = _final_refinement(sparsified, distance_matrix, reciprocal, n, params, gnn_features)
    
    return final_matrix

def _compute_enhanced_characteristics(distance_matrix: np.ndarray):
    """Compute comprehensive instance characteristics and GNN-ready features."""
    n = distance_matrix.shape[0]
    
    # Extract non-diagonal elements
    mask = ~np.eye(n, dtype=bool)
    distances = distance_matrix[mask].reshape(n, n-1)
    
    # Basic statistics
    stats = {
        'n': n,
        'distance_mean': np.mean(distances),
        'distance_std': np.std(distances),
        'distance_cv': np.std(distances) / (np.mean(distances) + 1e-10),
        'distance_skew': _compute_skewness(distances.flatten()),
        'spatial_clustering': _estimate_clustering(distance_matrix),
        'distance_entropy': _compute_distance_entropy(distances.flatten()),
        'symmetry_score': 1 - np.mean(np.abs(distance_matrix - distance_matrix.T)) / (np.mean(distances) + 1e-10)
    }
    
    # GNN-style graph features
    graph_features = {
        'adjacency': (distance_matrix < np.percentile(distances, 20)).astype(float),
        'node_features': np.stack([
            distance_matrix.mean(axis=1),
            distance_matrix.std(axis=1),
            np.sort(distance_matrix)[:, 1:4].mean(axis=1),  # Nearest neighbor features
            (distance_matrix < np.percentile(distances, 30)).sum(axis=1) / n  # Local density
        ], axis=1),
        'edge_features': np.stack([
            distance_matrix,
            reciprocal,
            np.outer(distance_matrix.mean(axis=1), distance_matrix.mean(axis=1))
        ], axis=2)
    }
    
    return stats, graph_features

def _gnn_feature_extraction(distance_matrix: np.ndarray, graph_features: dict, n: int) -> dict:
    """Extract features using GNN-inspired message passing."""
    adjacency = graph_features['adjacency']
    node_features = graph_features['node_features']
    
    # Multi-layer message passing
    num_layers = min(3, int(np.log2(n + 1)))
    hidden_features = []
    
    current_features = node_features.copy()
    for layer in range(num_layers):
        # Aggregate neighbor information
        neighbor_agg = adjacency @ current_features / (adjacency.sum(axis=1, keepdims=True) + 1e-10)
        
        # Combine with self features
        combined = np.concatenate([current_features, neighbor_agg], axis=1)
        
        # Simple non-linear transformation
        if layer < num_layers - 1:
            # Random projection (simulating learned weights)
            proj = np.random.randn(combined.shape[1], combined.shape[1] // 2) * 0.1
            current_features = np.tanh(combined @ proj)
        else:
            current_features = combined
        
        hidden_features.append(current_features)
    
    # Extract edge-level features from node embeddings
    edge_embeddings = []
    for layer_feat in hidden_features:
        edge_feat = np.expand_dims(layer_feat, 1) * np.expand_dims(layer_feat, 0)
        edge_embeddings.append(edge_feat.mean(axis=2))
    
    return {
        'node_embeddings': hidden_features[-1],
        'edge_embeddings': edge_embeddings,
        'multi_scale': hidden_features
    }

def _initialize_online_memory(n: int) -> dict:
    """Initialize memory for online learning."""
    return {
        'successful_params': [],
        'performance_history': [],
        'instance_signatures': [],
        'feature_importance': np.ones(4) / 4,  # For 4 components
        'sparsity_thresholds': [],
        'update_count': 0
    }

def _reinforcement_parameter_adaptation(stats: dict, graph_features: dict, memory: dict, n: int) -> dict:
    """Adapt parameters using reinforcement learning principles and memory."""
    
    # Base parameters from instance characteristics
    cv = stats['distance_cv']
    skew = np.abs(stats['distance_skew'])
    clustering = stats['spatial_clustering']
    entropy = stats['distance_entropy']
    
    # Learn from past successful configurations
    if memory['successful_params']:
        # Weight recent successful configurations more heavily
        recent_params = memory['successful_params'][-min(5, len(memory['successful_params'])):]
        learned_weights = np.mean([p['weights'] for p in recent_params], axis=0)
    else:
        learned_weights = np.array([0.25, 0.25, 0.25, 0.25])
    
    # Adjust based on current instance similarity to past instances
    if memory['instance_signatures']:
        current_sig = np.array([cv, skew, clustering, entropy])
        similarities = []
        for sig in memory['instance_signatures']:
            sim = 1 / (1 + np.linalg.norm(current_sig - sig))
            similarities.append(sim)
        
        if np.max(similarities) > 0.7:  # Similar instance found
            best_idx = np.argmax(similarities)
            best_params = memory['successful_params'][best_idx]
            # Blend with learned parameters
            blend_factor = 0.3 + 0.5 * similarities[best_idx]
            learned_weights = blend_factor * best_params['weights'] + (1 - blend_factor) * learned_weights
    
    # Dynamic parameter calculation with reinforcement adjustments
    size_factor = np.clip(n / 50.0, 0.1, 2.0)
    
    params = {
        # Component weights (with online learning adjustment)
        'weights': learned_weights * (0.9 + 0.2 * np.random.randn(4)),
        'weights': np.clip(params['weights'] / params['weights'].sum(), 0.1, 0.6),
        
        # Sparsification parameters with adaptive learning
        'base_k': max(3, min(15, int(n / (2 + 3 * cv)))),
        'k_adaptation': 0.3 + 0.4 * entropy,
        'decay_rate': 1.2 + cv,
        
        # Exploration parameters
        'exploration_rate': 0.05 + 0.1 * entropy,
        'exploitation_factor': 0.7 + 0.3 * (1 - entropy),
        
        # GNN feature utilization
        'gnn_weight': 0.3 + 0.4 * clustering,
        'feature_depth': min(3, int(np.log2(n))),
        
        # Adaptive thresholds
        'quality_threshold': 0.4 + 0.3 * cv,
        'connectivity_factor': 2 + int(0.7 * np.sqrt(n)),
        
        # Learning rate for online updates
        'learning_rate': 0.1 / (1 + memory['update_count'] / 10)
    }
    
    return params

def _compute_distance_component_v2(reciprocal: np.ndarray, params: dict, 
                                  stats: dict, gnn_features: dict) -> np.ndarray:
    """Enhanced distance component with GNN features."""
    n = reciprocal.shape[0]
    
    # Base reciprocal with adaptive scaling
    cv = stats['distance_cv']
    if cv > 1.0:
        power = 1.0 + 0.3 * min(1.0, cv - 1.0)
        distance_component = np.power(reciprocal, power)
    elif cv < 0.3:
        power = 0.8 + 0.2 * cv
        distance_component = np.power(reciprocal, power)
    else:
        distance_component = reciprocal.copy()
    
    # Incorporate GNN edge embeddings
    if gnn_features['edge_embeddings']:
        gnn_edge = gnn_features['edge_embeddings'][-1]
        if np.max(gnn_edge) > 0:
            gnn_edge = gnn_edge / np.max(gnn_edge)
        distance_component = (1 - params['gnn_weight']) * distance_component + \
                            params['gnn_weight'] * gnn_edge
    
    return distance_component

def _compute_centrality_component_v2(distance_matrix: np.ndarray, reciprocal: np.ndarray,
                                    params: dict, stats: dict, gnn_features: dict) -> np.ndarray:
    """Enhanced centrality component with GNN node embeddings."""
    n = distance_matrix.shape[0]
    
    # Traditional centrality measures
    closeness = 1 / (distance_matrix.mean(axis=1) + 1e-10)
    degree_centrality = reciprocal.sum(axis=1)
    
    # Combine with GNN node embeddings
    if gnn_features['node_embeddings'] is not None:
        gnn_centrality = gnn_features['node_embeddings'].mean(axis=1)
        if np.max(gnn_centrality) > 0:
            gnn_centrality = gnn_centrality / np.max(gnn_centrality)
        
        # Adaptive combination
        gnn_weight = params['gnn_weight']
        combined_centrality = (1 - gnn_weight) * (0.6 * closeness + 0.4 * degree_centrality) + \
                             gnn_weight * gnn_centrality
    else:
        combined_centrality = 0.6 * closeness + 0.4 * degree_centrality
    
    # Normalize
    if np.max(combined_centrality) > 0:
        combined_centrality = combined_centrality / np.max(combined_centrality)
    
    # Create centrality matrix
    centrality_matrix = np.outer(combined_centrality, combined_centrality)
    
    # Apply spatial adjustment
    if stats['spatial_clustering'] > 0.6:
        centrality_matrix *= (1 + 0.2 * stats['spatial_clustering'])
    
    return centrality_matrix * reciprocal

def _compute_mst_component_v2(distance_matrix: np.ndarray, reciprocal: np.ndarray,
                             params: dict, gnn_features: dict) -> np.ndarray:
    """MST component enhanced with structural information."""
    n = distance_matrix.shape[0]
    
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    try:
        # Compute MST
        mst = csgraph.minimum_spanning_tree(distance_matrix)
        mst_mask = (mst > 0).astype(float)
        
        # Boost based on MST structure
        mst_boosted = mst_mask.copy()
        
        # Incorporate GNN structural information
        if gnn_features['edge_embeddings']:
            structural_info = gnn_features['edge_embeddings'][0]
            if np.max(structural_info) > 0:
                structural_info = structural_info / np.max(structural_info)
            
            # Enhance MST edges with structural similarity
            for i in range(n):
                mst_neighbors = np.where(mst_mask[i] > 0)[0]
                if len(mst_neighbors) > 0:
                    for j in mst_neighbors:
                        structural_sim = structural_info[i, j]
                        mst_boosted[i, j] *= (1 + 0.5 * structural_sim)
        
        return mst_boosted
    
    except:
        return np.zeros_like(distance_matrix)

def _compute_structure_component(distance_matrix: np.ndarray, 
                                gnn_features: dict, params: dict) -> np.ndarray:
    """Component based purely on learned structural features."""
    n = distance_matrix.shape[0]
    
    if gnn_features['edge_embeddings']:
        # Combine multi-scale edge embeddings
        combined_structural = np.zeros((n, n))
        for i, edge_feat in enumerate(gnn_features['edge_embeddings']):
            weight = 1 / (1 + i)  # Weight decreases with depth
            if np.max(edge_feat) > 0:
                edge_feat = edge_feat / np.max(edge_feat)
            combined_structural += weight * edge_feat
        
        if len(gnn_features['edge_embeddings']) > 0:
            combined_structural /= len(gnn_features['edge_embeddings'])
        
        # Apply non-linear activation
        combined_structural = np.tanh(combined_structural * 2)
        
        return combined_structural
    
    return np.zeros_like(distance_matrix)

def _adaptive_combination(components: list, params: dict, memory: dict) -> np.ndarray:
    """Combine components with adaptive weights and online learning."""
    weights = params['weights']
    
    # Apply online learning adjustments
    if memory['feature_importance'] is not None:
        # Blend learned importance with current weights
        learning_rate = params['learning_rate']
        weights = (1 - learning_rate) * weights + learning_rate * memory['feature_importance']
        weights = weights / weights.sum()
    
    # Combine components
    combined = np.zeros_like(components[0])
    for comp, w in zip(components, weights):
        combined += w * comp
    
    # Add exploration noise
    exploration = params['exploration_rate']
    if exploration > 0:
        noise = np.random.randn(*combined.shape) * exploration
        np.fill_diagonal(noise, 0)
        exploitation = params['exploitation_factor']
        combined = combined * (1 - exploration) + np.abs(combined) * noise
        combined *= exploitation
    
    return combined

def _learned_sparsification(combined: np.ndarray, distance_matrix: np.ndarray,
                           params: dict, memory: dict, n: int) -> np.ndarray:
    """Sparsification with learned thresholds from memory."""
    sparsified = np.zeros_like(combined)
    
    # Determine sparsity level from memory if available
    if memory['sparsity_thresholds']:
        threshold = np.percentile(memory['sparsity_thresholds'], 50)
    else:
        threshold = params['quality_threshold']
    
    # Adaptive k per node
    base_k = params['base_k']
    k_adaptation = params['k_adaptation']
    
    for i in range(n):
        # Determine k for this node
        row_quality = combined[i].copy()
        row_quality[i] = -np.inf
        
        # Use learned threshold
        promising = row_quality > threshold * np.max(row_quality[row_quality > -np.inf])
        k = max(base_k, min(n-1, int(np.sum(promising) * (1 + k_adaptation * np.random.randn()))))
        k = np.clip(k, 2, n-1)
        
        if k < n-1:
            sorted_indices = np.argsort(row_quality)[::-1][:k]
        else:
            sorted_indices = np.arange(n)
            sorted_indices = sorted_indices[sorted_indices != i]
        
        # Apply adaptive decay
        decay_rate = params['decay_rate']
        for rank, j in enumerate(sorted_indices):
            decay = np.exp(-rank / decay_rate)
            # Adjust decay based on edge quality
            quality_factor = row_quality[j] / (np.max(row_quality[sorted_indices]) + 1e-10)
            adaptive_decay = decay * (0.7 + 0.3 * quality_factor)
            
            sparsified[i, j] = combined[i, j] * adaptive_decay
    
    return sparsified

def _update_online_memory(memory: dict, params: dict, stats: dict):
    """Update online learning memory with current configuration."""
    # Store current parameters as potentially successful
    memory['successful_params'].append({
        'weights': params['weights'].copy(),
        'base_k': params['base_k'],
        'decay_rate': params['decay_rate']
    })
    
    # Store instance signature
    instance_sig = np.array([
        stats['distance_cv'],
        stats['distance_skew'],
        stats['spatial_clustering'],
        stats['distance_entropy']
    ])
    memory['instance_signatures'].append(instance_sig)
    
    # Update feature importance (simulated - would come from solution feedback)
    # In practice, this would be updated based on actual solution quality
    if memory['update_count'] % 5 == 0:
        # Simulate learning: slightly perturb importances
        perturbation = 0.05 * np.random.randn(len(params['weights']))
        memory['feature_importance'] = np.clip(
            memory['feature_importance'] + perturbation, 0.1, 0.6
        )
        memory['feature_importance'] = memory['feature_importance'] / memory['feature_importance'].sum()
    
    # Update sparsity threshold memory
    current_threshold = params['quality_threshold']
    memory['sparsity_thresholds'].append(current_threshold)
    if len(memory['sparsity_thresholds']) > 20:
        memory['sparsity_thresholds'].pop(0)
    
    memory['update_count'] += 1

def _final_refinement(sparsified: np.ndarray, distance_matrix: np.ndarray,
                     reciprocal: np.ndarray, n: int, params: dict,
                     gnn_features: dict) -> np.ndarray:
    """Final refinement with connectivity guarantees and GNN smoothing."""
    min_conn = params['connectivity_factor']
    
    # Ensure minimum connectivity
    for i in range(n):
        connections = np.where(sparsified[i] > 0)[0]
        if len(connections) < min_conn:
            # Add promising edges from reciprocal
            row = reciprocal[i].copy()
            row[i] = 0
            row[connections] = 0
            
            if np.sum(row) > 0:
                additional = min(min_conn - len(connections), n - len(connections) - 1)
                top_indices = np.argsort(row)[::-1][:additional]
                for j in top_indices:
                    if row[j] > 0:
                        sparsified[i, j] = row[j] * 0.3
    
    # Apply GNN smoothing if available
    if gnn_features['edge_embeddings']:
        smooth_factor = 0.2
        gnn_smooth = gnn_features['edge_embeddings'][0]
        if np.max(gnn_smooth) > 0:
            gnn_smooth = gnn_smooth / np.max(gnn_smooth)
        sparsified = (1 - smooth_factor) * sparsified + smooth_factor * gnn_smooth
    
    # Ensure symmetry
    symmetric = (sparsified + sparsified.T) / 2
    
    # Final cleanup
    symmetric = np.maximum(symmetric, 0)
    np.fill_diagonal(symmetric, 0)
    
    # Normalize
    max_val = symmetric.max()
    if max_val > 0:
        symmetric = symmetric / max_val
    
    return symmetric

# Helper functions from previous versions
def _compute_skewness(values):
    if len(values) < 2:
        return 0.0
    mean = np.mean(values)
    std = np.std(values)
    if std == 0:
        return 0.0
    skew = np.mean(((values - mean) / std) ** 3)
    return float(skew)

def _estimate_clustering(distance_matrix):
    n = distance_matrix.shape[0]
    if n <= 3:
        return 0.5
    
    nearest_dists = []
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        nearest_dists.append(np.min(row))
    
    avg_nearest = np.mean(nearest_dists)
    overall_avg = np.mean(distance_matrix[~np.eye(n, dtype=bool)])
    
    clustering = 1 - avg_nearest / overall_avg
    return np.clip(clustering, 0.0, 1.0)

def _compute_distance_entropy(distances):
    if len(distances) < 2:
        return 0.0
    
    hist, _ = np.histogram(distances, bins=min(20, len(distances)//2))
    hist = hist[hist > 0]
    probs = hist / hist.sum()
    entropy = -np.sum(probs * np.log(probs + 1e-10))
    
    max_entropy = np.log(len(probs))
    if max_entropy > 0:
        entropy /= max_entropy
    
    return entropy
