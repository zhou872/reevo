import numpy as np
from typing import Tuple, List, Dict
import random

def heuristics_v2(distance_matrix: np.ndarray, 
                  exploration_factor: float = 1.0) -> np.ndarray:
    """
    Improved heuristic prior computation for TSP with adaptive mechanisms.
    
    Key improvements:
    1. Dynamic weight adjustment based on solution quality feedback
    2. Local search feedback loop to refine edge quality estimates
    3. Adaptive sparsification that responds to problem structure
    4. Reinforcement mechanism for promising edge patterns
    
    Parameters:
    distance_matrix (np.ndarray): Square matrix of distances between nodes
    exploration_factor (float): Controls exploration vs exploitation (0.5-2.0)
    
    Returns:
    np.ndarray: Matrix of heuristic scores indicating edge promise
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # Setup: avoid self-loops and division by zero
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. BASE HEURISTICS with exploration factor adjustment
    base_heuristic = 1.0 / safe_dist  # Inverse distance
    
    # 2. MST-BASED HEURISTICS with adaptive computation
    mst_edges = compute_enhanced_mst(safe_dist, n)
    mst_mask = np.zeros_like(safe_dist, dtype=bool)
    for i, j in mst_edges:
        mst_mask[i, j] = mst_mask[j, i] = True
    
    # 3. LOCAL SEARCH FEEDBACK: generate and analyze sample tours
    sample_tours = generate_sample_tours(safe_dist, n, min(50, max(10, n*2)))
    edge_frequencies = compute_edge_frequencies(sample_tours, n)
    
    # 4. STRUCTURAL ANALYSIS: edge betweenness and centrality
    edge_betweenness = compute_edge_betweenness(mst_edges, n)
    node_centrality = compute_node_centrality(mst_edges, edge_frequencies, n)
    
    # 5. DYNAMIC WEIGHT ADJUSTMENT based on problem characteristics
    weights = compute_adaptive_weights(n, exploration_factor, sample_tours, safe_dist)
    
    # 6. INTEGRATE HEURISTICS with reinforcement from sample tours
    heuristic_matrix = integrate_heuristics(
        base_heuristic, mst_mask, edge_frequencies, 
        edge_betweenness, node_centrality, weights
    )
    
    # 7. ADAPTIVE SPARSIFICATION with quality preservation
    heuristic_matrix = adaptive_sparsification(
        heuristic_matrix, mst_mask, n, 
        min_density=0.1 + 0.05 * exploration_factor
    )
    
    # 8. SYMMETRY ENFORCEMENT and final cleanup
    heuristic_matrix = np.maximum(heuristic_matrix, heuristic_matrix.T)
    np.fill_diagonal(heuristic_matrix, 0)
    
    return heuristic_matrix

def compute_enhanced_mst(dist_matrix: np.ndarray, n: int) -> List[Tuple[int, int]]:
    """Compute Minimum Spanning Tree with edge quality considerations."""
    mst_edges = []
    visited = set([0])
    available = set(range(1, n))
    
    while available:
        min_edge = None
        min_weight = np.inf
        
        # Consider both distance and potential edge quality
        for i in visited:
            for j in available:
                weight = dist_matrix[i, j]
                # Adjust weight by local density factor
                local_density = np.mean(sorted(dist_matrix[i])[1:4]) if n > 4 else 1
                adjusted_weight = weight / (local_density + 1e-8)
                
                if adjusted_weight < min_weight:
                    min_weight = adjusted_weight
                    min_edge = (i, j)
        
        if min_edge:
            i, j = min_edge
            mst_edges.append((i, j))
            visited.add(j)
            available.remove(j)
    
    return mst_edges

def generate_sample_tours(dist_matrix: np.ndarray, n: int, 
                          num_samples: int) -> List[List[int]]:
    """Generate diverse sample tours using multiple heuristics."""
    tours = []
    
    # Strategy 1: Nearest Neighbor from random starts
    for _ in range(num_samples // 2):
        start = random.randint(0, n-1)
        tour = nearest_neighbor_tour(dist_matrix, start)
        tours.append(tour)
    
    # Strategy 2: Insertion heuristics
    for _ in range(num_samples // 4):
        tour = insertion_heuristic(dist_matrix)
        tours.append(tour)
    
    # Strategy 3: Random permutations with local improvement
    for _ in range(num_samples - len(tours)):
        tour = list(range(n))
        random.shuffle(tour)
        # Quick 2-opt improvement
        tour = local_improvement(tour, dist_matrix, iterations=10)
        tours.append(tour)
    
    return tours

def nearest_neighbor_tour(dist_matrix: np.ndarray, start: int) -> List[int]:
    """Construct tour using nearest neighbor heuristic."""
    n = dist_matrix.shape[0]
    unvisited = set(range(n))
    tour = [start]
    unvisited.remove(start)
    
    current = start
    while unvisited:
        nearest = min(unvisited, key=lambda x: dist_matrix[current, x])
        tour.append(nearest)
        unvisited.remove(nearest)
        current = nearest
    
    return tour

def insertion_heuristic(dist_matrix: np.ndarray) -> List[int]:
    """Construct tour using farthest insertion heuristic."""
    n = dist_matrix.shape[0]
    
    # Start with two farthest apart cities
    max_dist = -1
    start_pair = (0, 1)
    for i in range(n):
        for j in range(i+1, n):
            if dist_matrix[i, j] > max_dist:
                max_dist = dist_matrix[i, j]
                start_pair = (i, j)
    
    tour = [start_pair[0], start_pair[1]]
    remaining = set(range(n)) - set(tour)
    
    while remaining:
        # Find city farthest from current tour
        farthest_city = max(remaining, 
                           key=lambda x: min(dist_matrix[x, y] for y in tour))
        
        # Find best insertion position
        best_pos = 0
        best_increase = np.inf
        
        for i in range(len(tour)):
            j = (i + 1) % len(tour)
            increase = (dist_matrix[tour[i], farthest_city] + 
                       dist_matrix[farthest_city, tour[j]] - 
                       dist_matrix[tour[i], tour[j]])
            if increase < best_increase:
                best_increase = increase
                best_pos = i + 1
        
        tour.insert(best_pos, farthest_city)
        remaining.remove(farthest_city)
    
    return tour

def local_improvement(tour: List[int], dist_matrix: np.ndarray, 
                      iterations: int) -> List[int]:
    """Apply quick 2-opt local improvement."""
    n = len(tour)
    improved_tour = tour.copy()
    
    for _ in range(iterations):
        improved = False
        for i in range(n):
            for j in range(i+2, n):
                # Check if 2-opt swap would improve
                a1, a2 = improved_tour[i], improved_tour[(i+1)%n]
                b1, b2 = improved_tour[j], improved_tour[(j+1)%n]
                
                current = dist_matrix[a1, a2] + dist_matrix[b1, b2]
                new = dist_matrix[a1, b1] + dist_matrix[a2, b2]
                
                if new < current:
                    # Reverse segment between i+1 and j
                    improved_tour[i+1:j+1] = reversed(improved_tour[i+1:j+1])
                    improved = True
        
        if not improved:
            break
    
    return improved_tour

def compute_edge_frequencies(tours: List[List[int]], n: int) -> np.ndarray:
    """Compute frequency and quality of edges in sample tours."""
    frequencies = np.zeros((n, n))
    
    for tour in tours:
        # Compute tour length for weighting
        tour_length = sum(dist_matrix[tour[i], tour[(i+1)%n]] 
                         for i in range(len(tour)))
        weight = 1.0 / (tour_length + 1e-8)
        
        # Add weighted frequency
        for i in range(len(tour)):
            j = (i + 1) % len(tour)
            frequencies[tour[i], tour[j]] += weight
            frequencies[tour[j], tour[i]] += weight
    
    # Normalize
    if frequencies.max() > 0:
        frequencies = frequencies / frequencies.max()
    
    return frequencies

def compute_edge_betweenness(mst_edges: List[Tuple[int, int]], n: int) -> np.ndarray:
    """Compute approximate edge betweenness centrality in MST."""
    betweenness = np.zeros((n, n))
    
    # Build adjacency from MST
    adj = {i: [] for i in range(n)}
    for i, j in mst_edges:
        adj[i].append(j)
        adj[j].append(i)
    
    # For each node pair, find unique path in MST and increment edge counts
    for source in range(n):
        # BFS to find paths from source
        parent = {source: -1}
        queue = [source]
        
        while queue:
            current = queue.pop(0)
            for neighbor in adj[current]:
                if neighbor not in parent:
                    parent[neighbor] = current
                    queue.append(neighbor)
        
        # Backtrack paths and increment betweenness
        for target in range(n):
            if target == source:
                continue
            
            # Trace path back to source
            node = target
            while parent[node] != -1:
                p = parent[node]
                betweenness[min(node, p), max(node, p)] += 1
                betweenness[max(node, p), min(node, p)] += 1
                node = p
    
    # Normalize
    if betweenness.max() > 0:
        betweenness = betweenness / betweenness.max()
    
    return betweenness

def compute_node_centrality(mst_edges: List[Tuple[int, int]], 
                           edge_frequencies: np.ndarray, 
                           n: int) -> np.ndarray:
    """Compute node centrality combining MST degree and edge frequencies."""
    centrality = np.zeros(n)
    
    # MST degree component
    mst_degree = np.zeros(n)
    for i, j in mst_edges:
        mst_degree[i] += 1
        mst_degree[j] += 1
    
    # Edge frequency component (sum of incident edge frequencies)
    freq_sum = edge_frequencies.sum(axis=1)
    
    # Combine components
    if mst_degree.max() > 0:
        mst_component = mst_degree / mst_degree.max()
    else:
        mst_component = np.zeros(n)
    
    if freq_sum.max() > 0:
        freq_component = freq_sum / freq_sum.max()
    else:
        freq_component = np.zeros(n)
    
    centrality = 0.6 * mst_component + 0.4 * freq_component
    
    return centrality

def compute_adaptive_weights(n: int, exploration_factor: float,
                            sample_tours: List[List[int]],
                            dist_matrix: np.ndarray) -> Dict[str, float]:
    """Compute dynamic weights based on problem characteristics."""
    # Analyze solution quality variance
    tour_lengths = []
    for tour in sample_tours:
        length = sum(dist_matrix[tour[i], tour[(i+1)%len(tour)]] 
                    for i in range(len(tour)))
        tour_lengths.append(length)
    
    if tour_lengths:
        quality_variance = np.std(tour_lengths) / (np.mean(tour_lengths) + 1e-8)
    else:
        quality_variance = 0.5
    
    # Adjust weights based on variance and exploration factor
    base_mst = 2.0 + exploration_factor
    base_freq = 1.5 + quality_variance * 2
    base_between = 0.8 + exploration_factor * 0.5
    base_centrality = 0.5 + quality_variance
    
    # Normalize for problem size
    size_factor = min(1.0, 100.0 / n) if n > 0 else 1.0
    
    return {
        'mst': base_mst * size_factor,
        'frequency': base_freq,
        'betweenness': base_between,
        'centrality': base_centrality
    }

def integrate_heuristics(base_heuristic: np.ndarray,
                        mst_mask: np.ndarray,
                        edge_frequencies: np.ndarray,
                        edge_betweenness: np.ndarray,
                        node_centrality: np.ndarray,
                        weights: Dict[str, float]) -> np.ndarray:
    """Integrate multiple heuristic components with reinforcement."""
    n = base_heuristic.shape[0]
    heuristic_matrix = base_heuristic.copy()
    
    # Apply MST boost with reinforcement
    mst_boost = 1.0 + weights['mst']
    heuristic_matrix[mst_mask] *= mst_boost
    
    # Add frequency-based reinforcement
    heuristic_matrix += weights['frequency'] * edge_frequencies
    
    # Add betweenness-weighted contribution
    heuristic_matrix += weights['betweenness'] * edge_betweenness * base_heuristic
    
    # Add node centrality contribution
    centrality_matrix = np.zeros_like(base_heuristic)
    for i in range(n):
        for j in range(n):
            if i != j:
                avg_centrality = (node_centrality[i] + node_centrality[j]) / 2
                centrality_matrix[i, j] = avg_centrality
    
    heuristic_matrix += weights['centrality'] * centrality_matrix * base_heuristic
    
    # Reinforcement from consistently promising edges
    promising_mask = (heuristic_matrix > np.percentile(heuristic_matrix[heuristic_matrix > 0], 75))
    heuristic_matrix[promising_mask] *= 1.2
    
    return heuristic_matrix

def adaptive_sparsification(heuristic_matrix: np.ndarray,
                          mst_mask: np.ndarray,
                          n: int,
                          min_density: float) -> np.ndarray:
    """Sparsify heuristic matrix while preserving essential structure."""
    result = heuristic_matrix.copy()
    
    # Always preserve MST edges
    preserved_mask = mst_mask.copy()
    
    # Determine target number of edges per node
    base_edges = max(3, int(np.sqrt(n)))
    target_edges = min(base_edges + 2, n - 1)
    
    # For each node, keep top edges
    for i in range(n):
        row = result[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Keep top edges based on score
        top_indices = np.argsort(row)[-target_edges:]
        node_mask = np.zeros(n, dtype=bool)
        node_mask[top_indices] = True
        preserved_mask[i] = preserved_mask[i] | node_mask
    
    # Apply sparsification
    result[~preserved_mask] = 0
    
    # Ensure minimum density
    current_density = np.sum(result > 0) / (n * n - n)
    if current_density < min_density:
        # Add more edges to meet minimum density
        needed_edges = int((min_density - current_density) * (n * n - n))
        if needed_edges > 0:
            # Add edges with highest scores not already included
            flat_indices = np.argsort(heuristic_matrix.flatten())[::-1]
            added = 0
            
            for idx in flat_indices:
                i, j = divmod(idx, n)
                if i != j and result[i, j] == 0:
                    result[i, j] = heuristic_matrix[i, j]
                    added += 1
                    if added >= needed_edges:
                        break
    
    return result

# For backward compatibility and direct testing
def heuristics(distance_matrix: np.ndarray) -> np.ndarray:
    """Wrapper for backward compatibility."""
    return heuristics_v2(distance_matrix)
