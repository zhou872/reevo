import numpy as np
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection with improved features:
    1. Rank-based edge sparsification with adaptive k selection
    2. Node connectivity centrality using normalized path frequencies
    3. Distance reciprocal with exponential transformation
    4. Combined weighting with symmetry enforcement
    """
    n = distance_matrix.shape[0]
    
    # 1. Distance reciprocal with exponential transformation
    # This emphasizes nearby nodes more strongly while preventing extreme values
    with np.errstate(divide='ignore', invalid='ignore'):
        # Add small epsilon to avoid division by zero and transform
        reciprocal = 1 / (distance_matrix + 1e-10)
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # Apply exponential transformation to emphasize nearby nodes
    reciprocal = np.exp(-distance_matrix / np.median(distance_matrix[distance_matrix > 0]))
    
    # 2. Enhanced node connectivity centrality
    centrality = np.zeros(n)
    
    # Compute centrality using average inverse path length
    for i in range(n):
        # Get shortest paths from node i to all others
        paths = csgraph.shortest_path(distance_matrix, directed=False, indices=i)
        # Avoid division by zero and infinity
        paths = np.where(np.isinf(paths), 0, paths)
        paths = np.where(paths == 0, 1, paths)  # Prevent division by zero
        centrality[i] = np.sum(1 / paths) - 1  # Subtract self-connection
    
    # Normalize centrality using min-max scaling
    if np.max(centrality) - np.min(centrality) > 0:
        centrality = (centrality - np.min(centrality)) / (np.max(centrality) - np.min(centrality))
    
    # Create centrality matrix as geometric mean of endpoint centralities
    # This emphasizes edges between two central nodes
    centrality_matrix = np.sqrt(np.outer(centrality, centrality))
    
    # 3. Minimum spanning tree indicator with weighted edges
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_indicator = (mst > 0).astype(float)
    
    # 4. Combined heuristic with adaptive weights
    # Weights are normalized based on relative importance
    w1, w2, w3 = 0.4, 0.35, 0.25  # Adjusted weights for better balance
    combined = (w1 * reciprocal + 
                w2 * centrality_matrix + 
                w3 * mst_indicator)
    
    # 5. Enhanced adaptive sparsification with rank-based decay
    sparsified = np.zeros_like(combined)
    
    # Adaptive k selection: more edges for smaller graphs, fewer for larger
    k = max(3, min(10, n // 3))  # Adjusted bounds for better connectivity
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self-connection
        
        # Get indices of top k edges for this node
        if k < n - 1:  # -1 to exclude self
            # Use argpartition for efficiency: get indices of top k values
            top_k_idx = np.argpartition(row, -k)[-k:]
        else:
            top_k_idx = np.arange(n)
            top_k_idx = top_k_idx[top_k_idx != i]
        
        # Sort top indices by their values (descending)
        top_k_values = row[top_k_idx]
        sorted_indices = top_k_idx[np.argsort(top_k_values)[::-1]]
        
        # Apply exponential decay based on rank
        for rank, j in enumerate(sorted_indices):
            # Exponential decay: higher rank (worse) gets more penalty
            decay = np.exp(-rank / 1.5)  # Adjusted decay factor
            sparsified[i, j] = combined[i, j] * decay
    
    # 6. Ensure symmetry with geometric mean for consistency
    sparsified = np.sqrt(sparsified * sparsified.T)
    
    # Final cleanup: zero out diagonal
    np.fill_diagonal(sparsified, 0)
    
    # 7. Final normalization to [0, 1] range for consistency
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
