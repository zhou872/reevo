import numpy as np
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection with improved factors and adaptive sparsification.
    
    Combines:
    1. Normalized reciprocal distance
    2. Node centrality based on path frequencies
    3. Continuous MST edge importance
    4. Adaptive sparsification with exponential decay
    
    Parameters
    ----------
    distance_matrix : np.ndarray
        Symmetric distance matrix between nodes.
    
    Returns
    -------
    np.ndarray
        Heuristic matrix of same shape, with higher values indicating more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Reciprocal distance (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # Normalize to [0,1]
    max_reciprocal = np.max(reciprocal)
    if max_reciprocal > 0:
        reciprocal = reciprocal / max_reciprocal
    
    # 2. Node connectivity centrality (using path frequency)
    centrality = np.zeros(n)
    for i in range(n):
        paths = csgraph.shortest_path(distance_matrix, directed=False, indices=i)
        centrality[i] = np.sum(1 / (1 + paths))  # Path frequency measure
    
    # Normalize centrality
    if np.max(centrality) > 0:
        centrality = centrality / np.max(centrality)
    
    # Create centrality matrix from endpoint centralities
    centrality_matrix = np.outer(centrality, centrality)
    
    # 3. Continuous MST importance (symmetric)
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_sym = np.maximum(mst, mst.T)  # Ensure symmetry
    mst_continuous = np.zeros_like(distance_matrix)
    mst_edges = mst_sym > 0
    mst_continuous[mst_edges] = 1 / (1 + mst_sym[mst_edges])  # Continuous importance
    
    # Normalize MST values
    max_mst = np.max(mst_continuous)
    if max_mst > 0:
        mst_continuous = mst_continuous / max_mst
    
    # 4. Multi-factor fusion with tuned weights
    w1, w2, w3 = 0.4, 0.3, 0.3  # Tuned weights
    combined = (w1 * reciprocal + 
                w2 * centrality_matrix + 
                w3 * mst_continuous)
    
    # 5. Adaptive sparsification with stochastic decay
    sparsified = np.zeros_like(combined)
    
    # Adaptive k: 3-10 or n//4, ensuring at least 1 neighbor
    k = max(1, min(10, n // 4))
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get top k indices
        if k < n:
            top_k_idx = np.argpartition(row, -k)[-k:]
        else:
            top_k_idx = np.arange(n)
            top_k_idx = top_k_idx[top_k_idx != i]
        
        # Apply exponential decay based on rank
        sorted_idx = top_k_idx[np.argsort(row[top_k_idx])[::-1]]
        for rank, j in enumerate(sorted_idx):
            decay = np.exp(-rank / 2)  # Exponential decay
            sparsified[i, j] = combined[i, j] * decay
    
    # Ensure symmetry and zero diagonal
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
