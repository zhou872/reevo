import numpy as np
from scipy.spatial import distance_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors:
    1. Reciprocal of distance (base measure)
    2. Local clustering coefficients to identify community structure
    3. Mutual rank-based filtering to focus on locally optimal connections
    4. Sparsification by zeroing out unpromising edges
    """
    n = distance_matrix.shape[0]
    
    # Base heuristic: reciprocal of distance
    with np.errstate(divide='ignore', invalid='ignore'):
        heuristic = 1 / distance_matrix
    np.fill_diagonal(heuristic, 0)  # Zero out diagonal
    
    # Compute local clustering coefficients
    cluster_scores = np.zeros(n)
    for i in range(n):
        # Find k nearest neighbors (k = sqrt(n) for balance)
        k = max(2, int(np.sqrt(n)))
        neighbor_indices = np.argsort(distance_matrix[i])[1:k+1]
        
        # Count connections among neighbors
        connections = 0
        for a_idx, a in enumerate(neighbor_indices):
            for b in neighbor_indices[a_idx+1:]:
                if distance_matrix[a, b] < np.median(distance_matrix[a]):
                    connections += 1
        
        # Local clustering coefficient
        if k > 1:
            cluster_scores[i] = 2 * connections / (k * (k - 1))
    
    # Mutual prominence score combining distance and clustering
    mutual_prominence = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Base reciprocal distance
            base = heuristic[i, j]
            
            # Clustering synergy: nodes in similar clusters
            cluster_synergy = 1 + 0.5 * (cluster_scores[i] + cluster_scores[j])
            
            # Rank-based filtering
            rank_i = np.sum(distance_matrix[i] < distance_matrix[i, j]) / n
            rank_j = np.sum(distance_matrix[j] < distance_matrix[j, i]) / n
            
            # Mutual prominence: both nodes rank each other highly
            if rank_i < 0.3 and rank_j < 0.3:  # Both in top 30% for each other
                rank_factor = 2.0
            elif rank_i < 0.5 or rank_j < 0.5:  # At least one in top 50%
                rank_factor = 1.5
            else:
                rank_factor = 0.8
            
            # Combine factors
            mutual_prominence[i, j] = base * cluster_synergy * rank_factor
            mutual_prominence[j, i] = mutual_prominence[i, j]
    
    # Sparsification: keep only promising edges
    # Keep edges where at least one endpoint considers it in top m connections
    m = max(3, int(np.log2(n)))  # Logarithmic sparsity
    
    sparse_heuristic = np.zeros_like(mutual_prominence)
    
    for i in range(n):
        # Get top m edges for node i
        row = mutual_prominence[i].copy()
        row[i] = 0  # Ignore diagonal
        top_m_indices = np.argsort(row)[-m:]
        
        # Keep mutual connections (if j also considers i important)
        for j in top_m_indices:
            # Check if i is in top 2m for j (more permissive)
            j_row = mutual_prominence[j].copy()
            j_row[j] = 0
            j_top_2m = np.argsort(j_row)[-2*m:]
            
            if i in j_top_2m:
                sparse_heuristic[i, j] = mutual_prominence[i, j]
    
    # Normalize to make values comparable
    max_val = np.max(sparse_heuristic)
    if max_val > 0:
        sparse_heuristic /= max_val
    
    return sparse_heuristic
