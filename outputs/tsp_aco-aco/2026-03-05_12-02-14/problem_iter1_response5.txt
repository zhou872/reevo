import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining:
    1. Reciprocal distance weighting
    2. Local clustering importance
    3. Sparsification via nearest neighbors
    """
    n = distance_matrix.shape[0]
    
    # 1. Base reciprocal weights (avoid division by zero)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1.0 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.eye(n, dtype=bool)] = 0  # Zero diagonal
    
    # 2. Local clustering coefficient weighting
    # Nodes that are well-connected to each other's neighbors are more promising
    cluster_weight = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i == j: continue
            # Find common neighbors between i and j
            neighbors_i = set(np.argsort(distance_matrix[i])[1:4])  # Top 3 nearest
            neighbors_j = set(np.argsort(distance_matrix[j])[1:4])
            common = len(neighbors_i & neighbors_j)
            cluster_weight[i, j] = 1.0 + 0.5 * common  # Boost edges within clusters
    
    # 3. Node centrality adjustment (closer to "center" nodes are more important)
    avg_distances = np.mean(distance_matrix, axis=1)
    centrality = 1.0 / (1.0 + avg_distances / np.max(avg_distances))
    centrality_matrix = np.outer(centrality, centrality)  # Edge importance factor
    
    # 4. Combine factors
    heuristics_matrix = reciprocal * cluster_weight * centrality_matrix
    
    # 5. Sparsification: Keep only promising edges
    # For each node, keep connections to nearest neighbors and reciprocal top choices
    k = max(3, int(np.sqrt(n)))  # Adaptive sparsity factor
    
    # Keep k-nearest neighbors for each node
    nearest_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    for i in range(n):
        # Get indices of k+1 smallest distances (skip self at index 0)
        nearest_indices = np.argsort(distance_matrix[i])[1:k+1]
        nearest_mask[i, nearest_indices] = True
    
    # Also keep edges with highest heuristic values (reciprocal top choices)
    heuristic_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    for i in range(n):
        # Get indices of k highest heuristic values
        top_indices = np.argsort(heuristics_matrix[i])[-k:]
        heuristic_mask[i, top_indices] = True
    
    # Combined mask: keep if either nearest neighbor OR high heuristic value
    final_mask = nearest_mask | heuristic_mask
    
    # 6. Apply sparsification
    heuristics_matrix = heuristics_matrix * final_mask
    
    # 7. Ensure symmetry (since TSP is undirected)
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    
    # 8. Normalize row-wise for probabilistic interpretation
    row_sums = heuristics_matrix.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1  # Avoid division by zero
    heuristics_matrix = heuristics_matrix / row_sums
    
    return heuristics_matrix
