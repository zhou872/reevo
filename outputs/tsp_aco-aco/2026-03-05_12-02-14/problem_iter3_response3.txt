import numpy as np
from scipy import sparse

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection using harmonic centrality
    and adaptive sparsification with exponential decay.
    
    Args:
        distance_matrix: Square numpy array of pairwise distances
        
    Returns:
        Heuristic matrix of same shape indicating edge promise
    """
    n = distance_matrix.shape[0]
    
    # 1. Base reciprocal distance with safe handling
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # 2. Harmonic centrality (more efficient than shortest path centrality)
    # Nodes that are close to many other nodes are more central
    # Using harmonic centrality: sum of 1/distance (avoiding zero diagonal)
    dist_nonzero = np.where(np.eye(n), np.inf, distance_matrix)
    harmonic_centrality = np.sum(1 / (1 + dist_nonzero), axis=1)
    
    # Normalize centrality
    if np.max(harmonic_centrality) > 0:
        harmonic_centrality = harmonic_centrality / np.max(harmonic_centrality)
    
    # Create centrality matrix: edge promise based on centrality of endpoints
    centrality_matrix = np.outer(harmonic_centrality, harmonic_centrality)
    
    # 3. Minimum spanning tree indicator (using efficient sparse representation)
    # Convert to sparse for efficient MST computation
    dist_sparse = sparse.csr_matrix(distance_matrix)
    mst = sparse.csgraph.minimum_spanning_tree(dist_sparse)
    mst_indicator = (mst.toarray() > 0).astype(float)
    
    # 4. Combine factors with balanced weights
    # Weight each factor based on graph characteristics
    w1, w2, w3 = 0.6, 0.3, 0.1  # Adjusted weights
    
    # Enhance reciprocal with centrality: closer and more central edges are better
    combined = (w1 * reciprocal * (1 + 0.5 * centrality_matrix) + 
                w2 * centrality_matrix + 
                w3 * mst_indicator)
    
    # 5. Adaptive sparsification with exponential decay
    sparsified = np.zeros_like(combined)
    
    # Adaptive k: keep more edges for smaller graphs, fewer for larger
    k = max(3, min(10, max(n // 6, 3)))  # More conservative bound
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get indices of top k edges (handle small n)
        if k < len(row) - 1:
            # Use argpartition for efficiency
            top_k_idx = np.argpartition(row, -k)[-k:]
        else:
            top_k_idx = np.arange(n)
            top_k_idx = top_k_idx[top_k_idx != i]
        
        # Sort top k indices by value
        top_k_values = row[top_k_idx]
        sorted_indices = top_k_idx[np.argsort(top_k_values)[::-1]]
        
        # Apply exponential decay based on rank
        for rank, j in enumerate(sorted_indices):
            decay = np.exp(-rank / 2.0)  # Exponential decay factor
            sparsified[i, j] = combined[i, j] * decay
    
    # Ensure symmetry for undirected TSP
    symmetric_sparsified = np.maximum(sparsified, sparsified.T)
    
    # Final cleanup: zero out diagonal and small values
    np.fill_diagonal(symmetric_sparsified, 0)
    
    # Remove very small values (numerical stability)
    symmetric_sparsified[symmetric_sparsified < 1e-10] = 0
    
    return symmetric_sparsified
