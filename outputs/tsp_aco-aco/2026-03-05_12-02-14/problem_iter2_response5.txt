import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with adaptive instance-aware tuning
    and feedback-enhanced refinement.
    
    Features:
    1. Instance-aware weight adaptation based on problem characteristics
    2. Minimum Spanning Tree for structural importance
    3. Adaptive nearest neighbor clustering
    4. Feedback loop that refines heuristics based on initial results
    5. Intelligent sparsification preserving exploration-exploitation balance
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return 1 / (distance_matrix + 1e-10)
    
    # ===== 1. INSTANCE CHARACTERISTICS ANALYSIS =====
    instance_stats = analyze_instance(distance_matrix, n)
    
    # ===== 2. ADAPTIVE COMPONENT WEIGHTS =====
    weights = compute_adaptive_weights(instance_stats)
    
    # ===== 3. BASE HEURISTIC COMPONENTS =====
    # 3.1 Inverse distance (basic proximity)
    inv_dist = 1 / (distance_matrix + 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # 3.2 MST structural importance
    mst_heuristic = compute_mst_component(distance_matrix, n, instance_stats)
    
    # 3.3 Adaptive nearest neighbor clustering
    nn_heuristic = compute_nn_component(distance_matrix, n, instance_stats)
    
    # ===== 4. INITIAL COMBINATION =====
    heuristics = (
        weights["inv_dist"] * inv_dist +
        weights["mst"] * mst_heuristic +
        weights["nn"] * nn_heuristic
    )
    
    # ===== 5. FEEDBACK-ENHANCED REFINEMENT =====
    heuristics = apply_feedback_refinement(heuristics, distance_matrix, instance_stats, weights)
    
    # ===== 6. INTELLIGENT SPARSIFICATION =====
    heuristics = intelligent_sparsification(heuristics, distance_matrix, n, instance_stats)
    
    # ===== 7. BALANCE EXPLORATION-EXPLOITATION =====
    heuristics = balance_exploration_exploitation(heuristics, instance_stats)
    
    # Ensure symmetry for undirected TSP
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # Normalize for stochastic sampling
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    return heuristics


def analyze_instance(distance_matrix: np.ndarray, n: int) -> dict:
    """Analyze problem instance characteristics for adaptive tuning."""
    # Compute distance statistics (excluding diagonal)
    mask = ~np.eye(n, dtype=bool)
    distances = distance_matrix[mask]
    
    stats = {
        "n": n,
        "mean_dist": np.mean(distances),
        "std_dist": np.std(distances),
        "cv_dist": np.std(distances) / (np.mean(distances) + 1e-10),  # Coefficient of variation
        "min_dist": np.min(distances),
        "max_dist": np.max(distances),
        "sparsity": np.sum(distance_matrix <= np.percentile(distances, 10)) / (n * n),
    }
    
    # Add density-based classification
    avg_neighbor_dist = np.mean([np.partition(distance_matrix[i], 2)[2] for i in range(n)])
    stats["density_ratio"] = avg_neighbor_dist / stats["mean_dist"]
    
    return stats


def compute_adaptive_weights(stats: dict) -> dict:
    """Compute component weights based on instance characteristics."""
    n = stats["n"]
    cv = stats["cv_dist"]
    density_ratio = stats["density_ratio"]
    
    # Base weights
    base_weights = {"inv_dist": 0.4, "mst": 0.3, "nn": 0.3}
    
    # Adjust based on coefficient of variation
    if cv > 1.0:  # Highly variable distances
        # Emphasize MST and NN for structural guidance
        base_weights["inv_dist"] *= 0.7
        base_weights["mst"] *= 1.2
        base_weights["nn"] *= 1.1
    elif cv < 0.3:  # Uniform distances
        # Emphasize NN for local structure
        base_weights["inv_dist"] *= 0.8
        base_weights["nn"] *= 1.3
    
    # Adjust based on problem size
    if n > 100:
        # Larger problems benefit more from structural guidance
        scale = np.log10(n) / 2
        base_weights["mst"] *= min(1.2, 1 + 0.1 * scale)
        base_weights["nn"] *= min(1.1, 1 + 0.05 * scale)
    
    # Adjust based on density pattern
    if density_ratio < 0.5:  # Dense clusters
        base_weights["nn"] *= 1.3
    elif density_ratio > 2.0:  # Sparse, uniform
        base_weights["inv_dist"] *= 1.2
    
    # Normalize to sum to 1
    total = sum(base_weights.values())
    return {k: v / total for k, v in base_weights.items()}


def compute_mst_component(distance_matrix: np.ndarray, n: int, stats: dict) -> np.ndarray:
    """Compute MST-based heuristic with instance-aware adjustments."""
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_heuristic = np.zeros_like(distance_matrix)
    
    # MST edges get base score
    mst_mask = mst > 0
    mst_heuristic[mst_mask] = 1 / (distance_matrix[mst_mask] + 1e-10)
    
    # Boost important MST edges (those connecting large subtrees)
    if n > 10:
        # Simple approximation: edges connecting to high-degree nodes in MST
        mst_degrees = (mst > 0).sum(axis=1)
        high_degree_nodes = np.where(mst_degrees > np.percentile(mst_degrees, 75))[0]
        
        for i in high_degree_nodes:
            neighbors = np.where(mst[i] > 0)[0]
            for j in neighbors:
                if mst_degrees[j] > 1:  # Connection between important nodes
                    mst_heuristic[i, j] *= 1.5
    
    return mst_heuristic


def compute_nn_component(distance_matrix: np.ndarray, n: int, stats: dict) -> np.ndarray:
    """Compute adaptive nearest neighbor heuristic."""
    nn_heuristic = np.zeros_like(distance_matrix)
    
    # Adaptive neighborhood size based on instance characteristics
    if stats["cv_dist"] > 1.0:  # Variable distances
        k = max(2, n // 15)  # Smaller neighborhood for variable instances
    elif stats["density_ratio"] < 0.5:  # Dense clusters
        k = max(3, n // 8)   # Larger neighborhood for clusters
    else:
        k = max(2, n // 10)  # Default
    
    for i in range(n):
        # Find k nearest neighbors (excluding self)
        indices = np.argsort(distance_matrix[i])[1:k+1]
        
        # Adaptive weighting based on distance distribution
        if stats["cv_dist"] < 0.5:  # Uniform distances
            # Use rank-based weights
            for rank, j in enumerate(indices):
                nn_heuristic[i, j] = (k - rank) / k
        else:
            # Use distance-proportional weights
            for j in indices:
                nn_heuristic[i, j] = 1 / (distance_matrix[i, j] + 1e-10)
    
    # Normalize each row
    row_sums = nn_heuristic.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    nn_heuristic = nn_heuristic / row_sums
    
    return nn_heuristic


def apply_feedback_refinement(heuristics: np.ndarray, distance_matrix: np.ndarray, 
                             stats: dict, weights: dict) -> np.ndarray:
    """Apply feedback loop to refine heuristics based on initial results."""
    n = stats["n"]
    refined = heuristics.copy()
    
    # Identify potentially over/under-valued edges
    row_max = refined.max(axis=1, keepdims=True)
    row_min = refined.min(axis=1, keepdims=True)
    row_range = row_max - row_min
    
    # Normalize to [0, 1] range per row
    normalized = (refined - row_min) / (row_range + 1e-10)
    
    # Boost edges that are both locally and globally promising
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] == 0:
                continue
                
            # Local consistency: check if edge is promising from both endpoints
            local_score = (normalized[i, j] + normalized[j, i]) / 2
            
            # Global perspective: compare to MST-like connectivity
            if heuristics[i, j] > 0:
                # Find alternative paths
                alt_path_score = 0
                count = 0
                for k in range(n):
                    if k != i and k != j:
                        if heuristics[i, k] > 0 and heuristics[k, j] > 0:
                            alt_path_score += distance_matrix[i, k] + distance_matrix[k, j]
                            count += 1
                
                if count > 0:
                    alt_path_score /= count
                    # Boost edge if it's significantly better than alternatives
                    if distance_matrix[i, j] < 0.8 * alt_path_score:
                        refined[i, j] *= 1.2
                        refined[j, i] *= 1.2
    
    return refined


def intelligent_sparsification(heuristics: np.ndarray, distance_matrix: np.ndarray, 
                              n: int, stats: dict) -> np.ndarray:
    """Intelligent sparsification preserving connectivity and diversity."""
    result = heuristics.copy()
    
    # Adaptive threshold based on instance characteristics
    if stats["cv_dist"] > 1.0:  # Variable distances
        percentile = 60  # Keep more edges for exploration
    elif stats["density_ratio"] < 0.5:  # Dense clusters
        percentile = 75  # Be more selective in dense areas
    else:
        percentile = 70  # Default
    
    # Global thresholding
    flat_heuristics = result[result > 0]
    if len(flat_heuristics) > 0:
        threshold = np.percentile(flat_heuristics, percentile)
        result[result < threshold] = 0
    
    # Ensure minimum connectivity per node
    for i in range(n):
        non_zero = np.count_nonzero(result[i])
        min_edges = max(2, int(np.log2(n)))  # Adaptive minimum
        
        if non_zero < min_edges:
            # Add promising missing edges
            candidates = np.argsort(distance_matrix[i])[1:min_edges*2]
            added = 0
            
            for j in candidates:
                if result[i, j] == 0 and added < (min_edges - non_zero):
                    result[i, j] = 1 / (distance_matrix[i, j] + 1e-10) * 0.5
                    result[j, i] = 1 / (distance_matrix[j, i] + 1e-10) * 0.5
                    added += 1
    
    return result


def balance_exploration_exploitation(heuristics: np.ndarray, stats: dict) -> np.ndarray:
    """Balance exploration and exploitation based on instance characteristics."""
    result = heuristics.copy()
    n = stats["n"]
    
    # Add small exploration bonus to diverse edges
    exploration_bonus = 0.1 * np.log(n) / n
    
    for i in range(n):
        row = result[i]
        non_zero_mask = row > 0
        
        if np.sum(non_zero_mask) > 1:
            # Current distribution
            probs = row[non_zero_mask] / row[non_zero_mask].sum()
            entropy = -np.sum(probs * np.log(probs + 1e-10))
            max_entropy = np.log(len(probs))
            
            # If distribution is too peaked (low entropy), encourage exploration
            if entropy < 0.5 * max_entropy:
                # Add small bonus to all non-zero edges
                result[i, non_zero_mask] *= (1 + exploration_bonus)
            
            # Ensure minimum probability for exploration
            min_prob = 0.01 / n
            row_sum = result[i].sum()
            if row_sum > 0:
                min_value = min_prob * row_sum
                result[i, non_zero_mask] = np.maximum(result[i, non_zero_mask], min_value)
    
    return result
