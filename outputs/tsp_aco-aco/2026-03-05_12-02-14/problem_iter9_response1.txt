import numpy as np
from scipy.sparse import csgraph
from scipy.spatial.distance import pdist, squareform
import warnings
warnings.filterwarnings('ignore')

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic matrix for TSP edge selection with simplified composition,
    adaptive sparsity control, and enhanced stability.
    
    Key improvements over v1:
    1. Simplified heuristic composition with fixed tuned weights
    2. Adaptive sparsity control based on problem size
    3. Removal of noisy components for stability
    4. Structural preservation of MST edges
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape indicating edge promise (higher = better)
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic[np.isinf(heuristic)] = 0
            heuristic = np.nan_to_num(heuristic, nan=0)
        np.fill_diagonal(heuristic, 0)
        return heuristic
    
    # ========== Phase 1: Simplified Heuristic Components ==========
    
    # 1. Reciprocal distance baseline (0.6 weight)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal = np.nan_to_num(reciprocal, nan=0)
    
    # Normalize reciprocal component
    if np.max(reciprocal) > 0:
        reciprocal = reciprocal / np.max(reciprocal)
    
    # 2. Centrality-based matrix (0.3 weight)
    # Compute centrality as inverse of average distance to other nodes
    with np.errstate(divide='ignore', invalid='ignore'):
        row_sums = np.sum(distance_matrix, axis=1)
        centrality = 1 / (row_sums + 1e-10)
        centrality[np.isinf(centrality)] = 0
        centrality = np.nan_to_num(centrality, nan=0)
    
    # Create centrality matrix by combining node centralities
    centrality = centrality / np.max(centrality) if np.max(centrality) > 0 else centrality
    centrality_matrix = np.outer(centrality, centrality)
    
    # Normalize centrality matrix
    if np.max(centrality_matrix) > 0:
        centrality_matrix = centrality_matrix / np.max(centrality_matrix)
    
    # 3. MST indicator (0.1 weight)
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    mst_indicator = (mst_array > 0).astype(float)
    
    # Optional: Boost MST edges that are particularly short
    mst_distances = distance_matrix * mst_indicator
    mst_boost = np.zeros_like(distance_matrix)
    mst_edges = np.where(mst_indicator > 0)
    if len(mst_edges[0]) > 0:
        for i, j in zip(*mst_edges):
            if i != j:
                # Edges that are short relative to average distance get higher boost
                avg_dist = np.mean(distance_matrix[i])
                rel_length = distance_matrix[i, j] / (avg_dist + 1e-10)
                mst_boost[i, j] = 1.0 / (1.0 + rel_length)
    
    # Combine MST indicator with boost
    mst_component = mst_indicator + 0.5 * mst_boost
    
    # Normalize MST component
    if np.max(mst_component) > 0:
        mst_component = mst_component / np.max(mst_component)
    
    # ========== Phase 2: Combine with Fixed Weights ==========
    
    # Fixed weights from simplified heuristic composition
    weights = [0.6, 0.3, 0.1]  # reciprocal, centrality, MST
    
    # Combine components
    heuristic_matrix = (
        weights[0] * reciprocal +
        weights[1] * centrality_matrix +
        weights[2] * mst_component
    )
    
    # Ensure symmetry and zero diagonal
    heuristic_matrix = (heuristic_matrix + heuristic_matrix.T) / 2
    np.fill_diagonal(heuristic_matrix, 0)
    
    # ========== Phase 3: Adaptive Sparsity Control ==========
    
    # Determine sparsity level based on problem size
    if n < 50:
        # Small problems: keep more edges
        edges_per_node = min(n - 1, max(5, int(0.4 * n)))
    elif n < 200:
        # Medium problems: moderate sparsity
        edges_per_node = int(2 * np.sqrt(n))
    else:
        # Large problems: high sparsity
        edges_per_node = int(1.5 * np.sqrt(n))
    
    # Preserve MST edges - these are structurally important
    preserved_edges = set()
    mst_edges = set(zip(*np.where(mst_indicator > 0)))
    preserved_edges.update(mst_edges)
    
    # Create sparsified matrix
    sparsified = np.zeros_like(heuristic_matrix)
    
    # First, preserve important MST edges
    for i, j in preserved_edges:
        if i != j:
            sparsified[i, j] = heuristic_matrix[i, j]
    
    # For each node, select top additional edges with exponential decay
    for i in range(n):
        row = heuristic_matrix[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get candidate edges (excluding already preserved)
        candidates = []
        for j in range(n):
            if i != j and (i, j) not in preserved_edges and (j, i) not in preserved_edges:
                candidates.append((j, row[j]))
        
        if not candidates:
            continue
            
        # Sort candidates by value in descending order
        candidates.sort(key=lambda x: x[1], reverse=True)
        
        # How many more edges can we add for this node?
        current_edges = sum(1 for j in range(n) if (i, j) in preserved_edges)
        max_additional = max(0, edges_per_node - current_edges)
        
        # Select top edges with exponential decay
        selected = candidates[:max_additional]
        for rank, (j, value) in enumerate(selected):
            # Exponential decay: higher rank = stronger decay
            decay = np.exp(-rank / 3.0)
            sparsified[i, j] = value * decay
    
    # Ensure symmetry and final processing
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    # Final normalization
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
