import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic matrix for TSP edge selection with adaptive, data-driven adjustments.
    
    Features:
    1. Reciprocal distance (prioritize short edges)
    2. Node clustering awareness (identify natural clusters in the problem)
    3. Adaptive smoothing and sparsification based on problem characteristics
    4. Symmetric, robust handling of edge cases
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between nodes
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, with higher values
        indicating more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            reciprocal = 1 / distance_matrix
            reciprocal[np.isinf(reciprocal)] = 0
            reciprocal[np.isnan(reciprocal)] = 0
        np.fill_diagonal(reciprocal, 0)
        return reciprocal
    
    # 1. Reciprocal distance with safe handling
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # 2. Adaptive node clustering awareness
    # Compute median distance for each node to estimate local density
    median_distances = np.zeros(n)
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        median_distances[i] = np.median(row[row < np.inf])
    
    # Identify natural clusters using simple k-means initialization heuristic
    # Sort nodes by median distance and sample cluster centers
    sorted_indices = np.argsort(median_distances)
    k_clusters = max(2, n // 10)  # Adaptive number of clusters
    cluster_centers = sorted_indices[::max(1, n // k_clusters)][:k_clusters]
    
    # Assign each node to nearest cluster center
    cluster_assignments = np.argmin(distance_matrix[:, cluster_centers], axis=1)
    
    # Intra-cluster bonus: edges within same cluster get a boost
    intra_cluster_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        same_cluster = (cluster_assignments == cluster_assignments[i])
        intra_cluster_mask[i, same_cluster] = True
    
    # Inter-cluster penalty: edges between very distant clusters get reduced weight
    inter_cluster_penalty = np.ones((n, n))
    if k_clusters > 1:
        # Compute average inter-cluster distances
        cluster_distances = np.zeros((k_clusters, k_clusters))
        for c1 in range(k_clusters):
            for c2 in range(k_clusters):
                if c1 != c2:
                    mask = (cluster_assignments == c1)[:, None] & (cluster_assignments == c2)
                    if mask.any():
                        cluster_distances[c1, c2] = np.median(distance_matrix[mask])
        
        # Apply penalty based on cluster distance ratio
        max_cluster_dist = cluster_distances.max()
        if max_cluster_dist > 0:
            for i in range(n):
                for j in range(i + 1, n):
                    c1, c2 = cluster_assignments[i], cluster_assignments[j]
                    if c1 != c2:
                        penalty = 1.0 - 0.3 * (cluster_distances[c1, c2] / max_cluster_dist)
                        inter_cluster_penalty[i, j] = inter_cluster_penalty[j, i] = max(0.5, penalty)
    
    # 3. Combine factors with adaptive weights
    # Weight tuning based on problem size and distance distribution
    flat_dists = distance_matrix[np.triu_indices(n, k=1)]
    dist_cv = np.std(flat_dists) / (np.mean(flat_dists) + 1e-10)  # Coefficient of variation
    
    # More weight to reciprocal when distances are uniform, less when highly variable
    w_reciprocal = 0.7 if dist_cv < 0.5 else 0.5
    w_intra_cluster = 0.3 if n > 20 else 0.2  # More important for larger problems
    
    # Combine components
    combined = w_reciprocal * reciprocal
    combined[intra_cluster_mask] *= (1.0 + w_intra_cluster)
    combined *= inter_cluster_penalty
    
    # 4. Adaptive smoothing and sparsification
    # Determine sparsity level based on problem size
    base_top_k = max(3, min(10, int(np.sqrt(n))))
    
    sparsified = np.zeros_like(combined)
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Keep top edges with smooth transition
        if base_top_k < n - 1:
            # Get indices of top base_top_k edges
            top_k_idx = np.argpartition(row, -base_top_k)[-base_top_k:]
            # Sort them for smooth decay
            sorted_indices = top_k_idx[np.argsort(row[top_k_idx])[::-1]]
            
            # Apply smooth exponential decay
            for rank, j in enumerate(sorted_indices):
                # More gradual decay for larger problems
                decay_rate = 2.0 if n < 50 else 3.0
                decay = np.exp(-rank / decay_rate)
                sparsified[i, j] = combined[i, j] * decay
        else:
            # For very small problems, keep all edges with mild decay
            for j in range(n):
                if i != j:
                    rank = np.sum(combined[i] > combined[i, j])
                    decay = np.exp(-rank / 3.0)
                    sparsified[i, j] = combined[i, j] * decay
    
    # 5. Ensure symmetry and clean up
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    # Normalize to [0, 1] range if not all zeros
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
