import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Compute heuristic prior indicators for each edge in TSP.
    
    Parameters:
    distance_matrix (np.ndarray): Square matrix of distances between nodes.
    
    Returns:
    np.ndarray: Matrix of the same shape with scores indicating edge promise.
                Unpromising edges are set to zero.
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # Step 1: Inverse distance with zero diagonal
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    np.divide(1.0, distance_matrix, out=inv_dist, where=distance_matrix != 0)
    np.fill_diagonal(inv_dist, 0.0)  # Exclude self-edges
    
    # Step 2: Minimum Spanning Tree (Prim's algorithm)
    mst_mask = np.zeros_like(distance_matrix, dtype=bool)
    if n > 1:
        visited = np.zeros(n, dtype=bool)
        visited[0] = True
        for _ in range(n - 1):
            # Find minimum edge from visited to unvisited
            min_edge = (None, None)
            min_dist = np.inf
            for i in np.where(visited)[0]:
                for j in np.where(~visited)[0]:
                    if distance_matrix[i, j] < min_dist:
                        min_dist = distance_matrix[i, j]
                        min_edge = (i, j)
            if min_edge[0] is not None:
                i, j = min_edge
                mst_mask[i, j] = True
                mst_mask[j, i] = True  # Undirected graph
                visited[j] = True
    
    # Step 3: Node centrality via k-nearest neighbors
    k = max(1, min(5, n - 1))  # Choose k adaptively
    degree = np.zeros(n, dtype=int)
    for i in range(n):
        # Get indices of k smallest distances excluding self
        row = distance_matrix[i].copy()
        row[i] = np.inf  # Exclude diagonal
        if n > 1:
            nn_indices = np.argpartition(row, k)[:k]
            for j in nn_indices:
                degree[j] += 1  # j is a nearest neighbor of i
    
    # Normalize degree to [0,1] for weighting
    if degree.max() > 0:
        degree_norm = degree / degree.max()
    else:
        degree_norm = np.zeros(n)
    
    # Step 4: Combine factors
    alpha = 0.5  # Boost for MST edges
    beta = 0.2   # Weight for node centrality
    scores = inv_dist.copy()
    
    # Apply MST boost
    scores[mst_mask] *= (1.0 + alpha)
    
    # Apply node centrality: for edge (i,j), use average centrality
    for i in range(n):
        for j in range(n):
            if i != j:
                centrality_factor = 1.0 + beta * (degree_norm[i] + degree_norm[j]) / 2.0
                scores[i, j] *= centrality_factor
    
    # Step 5: Sparsify by thresholding
    # Use median of off-diagonal scores as threshold
    off_diag_scores = scores[~np.eye(n, dtype=bool)].flatten()
    if len(off_diag_scores) > 0:
        threshold = np.median(off_diag_scores)
        scores[scores < threshold] = 0.0
    
    return scores
