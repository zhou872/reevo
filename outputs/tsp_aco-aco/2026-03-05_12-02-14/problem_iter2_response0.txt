import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining graph theory concepts
    with efficient vectorized operations and adaptive sparsity.
    
    Strategy:
    1. Use Minimum Spanning Tree (MST) as a backbone for promising edges
    2. Augment with nearest neighbor and mutual connectivity information
    3. Vectorized operations for O(n²) complexity instead of O(n³)
    4. Adaptive sparsity based on problem size and structure
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) / (distance_matrix + np.eye(n) * 1e-12)
    
    # 1. Compute Minimum Spanning Tree (Prim's algorithm - vectorized)
    mst_mask = _compute_mst_mask(distance_matrix)
    
    # 2. Base heuristic: inverse distance with diagonal handling
    inv_dist = 1.0 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(inv_dist, 0)
    
    # 3. Compute nearest neighbor statistics (vectorized)
    k_nn = max(2, min(10, n // 5))  # Adaptive number of nearest neighbors
    nn_mask = _compute_nearest_neighbor_mask(distance_matrix, k_nn)
    
    # 4. Mutual nearest neighbor indicator (vectorized)
    nn_rank = np.argsort(distance_matrix, axis=1)
    top_m = max(3, n // 6)
    
    # Create binary masks for top neighbors
    top_mask = np.zeros_like(distance_matrix, dtype=bool)
    rows = np.arange(n)[:, None]
    top_mask[rows, nn_rank[:, :top_m]] = True
    
    # Mutual nearest neighbor: both i in j's top and j in i's top
    mutual_nn = top_mask & top_mask.T
    
    # One-way nearest neighbor: at least one direction
    one_way_nn = top_mask | top_mask.T
    
    # 5. Compute centrality measures (vectorized)
    # Local density: inverse of median distance to neighbors
    with np.errstate(divide='ignore'):
        # Compute row-wise medians efficiently
        sorted_dists = np.partition(distance_matrix, n//2, axis=1)
        median_dists = sorted_dists[:, n//2]
        local_density = 1.0 / (median_dists + 1e-12)
        local_density = local_density / np.max(local_density)
    
    # 6. Combine all factors (vectorized)
    # Start with inverse distance
    heuristics = inv_dist.copy()
    
    # Boost MST edges significantly
    mst_boost = 2.0 + local_density[:, None] + local_density[None, :]
    heuristics[mst_mask] *= mst_boost[mst_mask]
    
    # Boost mutual nearest neighbors
    mm_factor = 1.5 + 0.5 * (local_density[:, None] + local_density[None, :])
    heuristics[mutual_nn] *= mm_factor[mutual_nn]
    
    # Moderate boost for one-way nearest neighbors
    heuristics[one_way_nn & ~mutual_nn] *= 1.2
    
    # Penalize very long edges (beyond 3x median distance)
    long_edge_mask = distance_matrix > (3 * median_dists[:, None])
    heuristics[long_edge_mask] *= 0.1
    
    # 7. Adaptive sparsification
    sparsity_factor = max(0.2, min(0.5, 10/n))  # More sparse for larger n
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Ignore diagonal
        threshold = np.percentile(row[row > 0], 100 * (1 - sparsity_factor))
        row[row < threshold] = 0
        heuristics[i] = row
    
    # Ensure symmetry
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 8. Ensure minimum connectivity (each node connects to at least 2 others)
    for i in range(n):
        non_zero = np.count_nonzero(heuristics[i])
        if non_zero < 2:
            # Add k nearest neighbors that aren't already included
            nearest = np.argsort(distance_matrix[i])[1:max(3, k_nn+1)]
            for j in nearest:
                if heuristics[i, j] == 0:
                    boost = 1.0 + local_density[i] + local_density[j]
                    heuristics[i, j] = inv_dist[i, j] * boost * 0.8
                    heuristics[j, i] = heuristics[i, j]
    
    # 9. Normalize for probabilistic interpretation
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristics = heuristics / row_sums
    
    # Final symmetry enforcement
    heuristics = (heuristics + heuristics.T) / 2
    
    return heuristics

def _compute_mst_mask(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute Minimum Spanning Tree mask using Prim's algorithm."""
    n = distance_matrix.shape[0]
    mst_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    if n == 0:
        return mst_mask
    
    # Initialize
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    
    # While not all nodes are visited
    while not np.all(visited):
        # Find minimum edge between visited and unvisited nodes
        visited_indices = np.where(visited)[0]
        unvisited_indices = np.where(~visited)[0]
        
        if len(unvisited_indices) == 0:
            break
        
        # Create a submatrix of distances between visited and unvisited
        # This is the bottleneck but still O(n²) not O(n³)
        min_dist = np.inf
        min_i, min_j = 0, 0
        
        for i in visited_indices:
            for j in unvisited_indices:
                if distance_matrix[i, j] < min_dist:
                    min_dist = distance_matrix[i, j]
                    min_i, min_j = i, j
        
        # Add edge to MST
        mst_mask[min_i, min_j] = True
        mst_mask[min_j, min_i] = True
        visited[min_j] = True
    
    return mst_mask

def _compute_nearest_neighbor_mask(distance_matrix: np.ndarray, k: int) -> np.ndarray:
    """Compute mask for k-nearest neighbors of each node."""
    n = distance_matrix.shape[0]
    mask = np.zeros_like(distance_matrix, dtype=bool)
    
    if k >= n:
        mask = np.ones_like(distance_matrix, dtype=bool)
        np.fill_diagonal(mask, False)
        return mask
    
    # Get indices of k+1 nearest neighbors (including self)
    nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:k+1]
    
    # Set mask for nearest neighbors
    rows = np.arange(n)[:, None]
    mask[rows, nearest_indices] = True
    
    return mask
