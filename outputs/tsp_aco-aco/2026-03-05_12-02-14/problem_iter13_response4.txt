import numpy as np
from scipy import sparse
import warnings
warnings.filterwarnings('ignore')


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge sampling with entropy-based adaptive sparsification.
    
    Features:
    1. Entropy-based adaptive edge selection per node
    2. Vectorized exponential decay filtering
    3. MST connectivity preservation
    4. Multi-component edge scoring
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Row-normalized heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic = np.nan_to_num(heuristic, nan=0, posinf=0, neginf=0)
        np.fill_diagonal(heuristic, 0)
        return heuristic
    
    # ========== BASIC COMPONENTS ==========
    # 1. Reciprocal distances with scaling
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / np.maximum(distance_matrix, 1e-10)
        reciprocal = np.nan_to_num(reciprocal, nan=0, posinf=0, neginf=0)
    
    # Adaptive scaling based on distance variance
    triu_distances = distance_matrix[np.triu_indices(n, k=1)]
    dist_std = np.std(triu_distances) if len(triu_distances) > 1 else 1.0
    scale_factor = 0.5 + 0.5 * np.tanh(dist_std / np.mean(triu_distances) if np.mean(triu_distances) > 0 else 0)
    reciprocal = reciprocal ** scale_factor
    
    # 2. Minimum Spanning Tree
    try:
        mst = sparse.csgraph.minimum_spanning_tree(distance_matrix)
        mst_mask = mst.toarray() > 0
    except:
        # Fallback: nearest neighbors
        mst_mask = np.zeros_like(distance_matrix, dtype=bool)
        for i in range(n):
            row = distance_matrix[i].copy()
            row[i] = np.inf
            j = np.argmin(row)
            mst_mask[i, j] = mst_mask[j, i] = True
    
    # 3. Local structure: nearest neighbors
    nearest_k = min(5, n - 1)
    nearest_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        nearest_idx = np.argpartition(row, nearest_k)[:nearest_k]
        nearest_mask[i, nearest_idx] = True
    nearest_mask = nearest_mask | nearest_mask.T  # Ensure symmetry
    
    # ========== ENTROPY-BASED ADAPTIVE SPARSIFICATION ==========
    def row_entropy(row: np.ndarray) -> float:
        """Calculate normalized entropy of a row (excluding diagonal)."""
        # Remove diagonal and negative values
        mask = (row > 0) & (~np.eye(len(row), dtype=bool)[0])
        if not np.any(mask):
            return 0.0
        
        values = row[mask]
        probs = values / np.sum(values)
        
        # Calculate entropy
        entropy = -np.sum(probs * np.log(probs + 1e-10))
        
        # Normalize by maximum possible entropy
        max_entropy = np.log(len(values))
        return entropy / max_entropy if max_entropy > 0 else 0.0
    
    # Calculate entropy for each row
    row_entropies = np.array([row_entropy(reciprocal[i]) for i in range(n)])
    
    # Adaptive number of edges per node based on entropy
    base_k = int(np.sqrt(n))
    min_k = max(2, base_k // 2)
    max_k = min(n - 1, base_k * 2)
    
    # Higher entropy = more uniform distribution = keep more edges
    k_per_node = np.round(min_k + (max_k - min_k) * row_entropies).astype(int)
    
    # ========== VECTORIZED EDGE SELECTION WITH EXPONENTIAL DECAY ==========
    def select_edges_with_decay(row: np.ndarray, k: int, mask: np.ndarray = None) -> np.ndarray:
        """
        Select top k edges with exponential decay on rank.
        If mask is provided, edges outside mask are set to -inf.
        """
        row = row.copy()
        np.fill_diagonal(row, -np.inf)
        
        if mask is not None:
            row[~mask] = -np.inf
        
        if k >= len(row) - 1:  # Keep all edges (excluding diagonal)
            valid_mask = row > -np.inf
            result = np.zeros_like(row)
            result[valid_mask] = row[valid_mask]
            return result
        
        # Find top k indices
        top_k_idx = np.argpartition(-row, k)[:k]
        top_k_values = row[top_k_idx]
        
        # Sort descending
        sorted_idx = top_k_idx[np.argsort(-top_k_values)]
        sorted_values = row[sorted_idx]
        
        # Apply exponential decay
        ranks = np.arange(len(sorted_values))
        decay = np.exp(-ranks / 2.0)  # Half-life of 2 ranks
        decayed_values = sorted_values * decay
        
        # Create result array
        result = np.zeros_like(row)
        result[sorted_idx] = decayed_values
        
        return result
    
    # Preserve MST edges (critical for connectivity)
    preserved_mask = mst_mask
    
    # Combine edge scores from multiple components
    edge_scores = np.zeros_like(distance_matrix)
    
    # Component 1: Reciprocal distances (weighted by MST presence)
    recip_component = reciprocal.copy()
    recip_component[mst_mask] *= 1.5  # Boost MST edges
    
    # Component 2: Local structure
    local_component = np.zeros_like(distance_matrix)
    local_component[nearest_mask] = reciprocal[nearest_mask]
    
    # Component 3: Global structure (inverse distance rank)
    rank_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        ranks = np.argsort(row)
        # Convert rank to score: higher score for smaller ranks (shorter distances)
        rank_scores = 1.0 / (ranks + 1)
        rank_matrix[i] = rank_scores
    rank_matrix = (rank_matrix + rank_matrix.T) / 2
    
    # Combine components with adaptive weights
    # More weight to local structure for high entropy nodes
    local_weight = 0.3 + 0.4 * row_entropies[:, np.newaxis]
    global_weight = 0.5 - 0.3 * row_entropies[:, np.newaxis]
    mst_weight = 0.2 * np.ones((n, 1))
    
    combined_scores = (local_weight * local_component + 
                      global_weight * rank_matrix + 
                      mst_weight * recip_component)
    
    # Ensure symmetry
    combined_scores = (combined_scores + combined_scores.T) / 2
    np.fill_diagonal(combined_scores, 0)
    
    # ========== ADAPTIVE SPARSIFICATION ==========
    heuristic = np.zeros_like(distance_matrix)
    
    # Process each node
    for i in range(n):
        k = k_per_node[i]
        
        # Always include MST edges for this node
        mst_edges = np.where(preserved_mask[i])[0]
        if len(mst_edges) > 0:
            heuristic[i, mst_edges] = combined_scores[i, mst_edges]
        
        # Calculate how many additional edges we need
        current_edges = np.count_nonzero(heuristic[i])
        additional_needed = max(0, k - current_edges)
        
        if additional_needed > 0:
            # Mask out already selected edges
            selection_mask = np.ones(n, dtype=bool)
            selection_mask[i] = False  # Exclude diagonal
            selection_mask[np.where(heuristic[i] > 0)[0]] = False  # Exclude already selected
            
            # Select additional edges with decay
            additional_edges = select_edges_with_decay(
                combined_scores[i], 
                additional_needed,
                selection_mask
            )
            
            # Add to heuristic
            heuristic[i] += additional_edges
    
    # Ensure symmetry
    heuristic = np.maximum(heuristic, heuristic.T)
    np.fill_diagonal(heuristic, 0)
    
    # ========== NORMALIZATION ==========
    # Row normalization with regularization
    row_sums = heuristic.sum(axis=1, keepdims=True)
    zero_rows = row_sums.flatten() == 0
    
    if np.any(zero_rows):
        # Use reciprocal as fallback for isolated rows
        heuristic[zero_rows] = reciprocal[zero_rows]
        row_sums = heuristic.sum(axis=1, keepdims=True)
    
    # Add small epsilon to avoid division by zero
    row_sums = row_sums + 1e-10
    heuristic = heuristic / row_sums
    
    # Final symmetry and diagonal cleanup
    heuristic = (heuristic + heuristic.T) / 2
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
