import numpy as np
from scipy import sparse
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with adaptive stochastic parameters
    and evolutionary feedback mechanisms. Returns a matrix of edge promise indicators
    suitable for stochastic solution sampling.
    
    Key improvements over v1:
    1. Dynamic reinforcement feedback for parameter adjustment
    2. Hybrid heuristics integrating nearest-neighbor chains
    3. Adaptive noise based on instance characteristics
    4. Fitness-aware edge boosting during stagnation
    """
    n = distance_matrix.shape[0]
    if n < 2:
        result = np.zeros_like(distance_matrix)
        if n == 2:
            result[0, 1] = result[1, 0] = 1.0
        return result
    
    # Track performance metrics for reinforcement feedback
    feedback_state = {
        'stagnation_counter': 0,
        'best_fitness': float('inf'),
        'adaptive_exploration': 0.1
    }
    
    # 1. Compute instance statistics with reinforcement memory
    mask = ~np.eye(n, dtype=bool)
    distances = distance_matrix[mask]
    
    mean_dist = np.mean(distances)
    std_dist = np.std(distances)
    cv_dist = std_dist / (mean_dist + 1e-10)
    
    # Update exploration based on distance variability
    feedback_state['adaptive_exploration'] = 0.05 + 0.1 * min(cv_dist, 2.0)
    
    # 2. Enhanced base heuristic with adaptive transformations
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / np.maximum(distance_matrix, 1e-10)
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    np.fill_diagonal(reciprocal, 0)
    
    # Adaptive transformation based on instance characteristics
    if cv_dist > 1.0:
        # For highly variable distances, enhance contrast
        reciprocal = reciprocal ** (1 + 0.3 * min(cv_dist - 1.0, 1.0))
    elif cv_dist < 0.3:
        # For uniform distances, smooth differences
        reciprocal = reciprocal ** 0.7
    
    # 3. Hybrid centrality incorporating nearest-neighbor chains
    centrality = np.zeros(n)
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        
        # Simple nearest neighbor chain analysis
        if n > 5:
            # Find 3 nearest neighbors and their connectivity
            nn_indices = np.argsort(row)[:3]
            chain_score = 0
            for j in nn_indices:
                # Check if j's nearest neighbors include i or other nn
                j_row = distance_matrix[j].copy()
                j_row[j] = np.inf
                j_nn = np.argsort(j_row)[:2]
                if i in j_nn:
                    chain_score += 1
                elif len(set(nn_indices) & set(j_nn)) > 0:
                    chain_score += 0.5
            centrality[i] = np.sum(1 / (1 + row)) * (1 + 0.2 * chain_score)
        else:
            centrality[i] = np.sum(1 / (1 + row))
    
    if np.max(centrality) > 0:
        centrality = centrality / np.max(centrality)
    
    centrality_matrix = np.outer(centrality, centrality) * reciprocal
    
    # 4. MST component with fitness-aware boosting
    try:
        mst = csgraph.minimum_spanning_tree(distance_matrix)
        mst_indicator = (mst > 0).astype(float)
        
        # Adaptive boosting based on stagnation feedback
        boost_factor = 1.2 + 0.3 * (feedback_state['stagnation_counter'] / 10)
        boost_factor = min(boost_factor, 2.0)
        
        for i in range(n):
            mst_neighbors = np.where(mst_indicator[i] > 0)[0]
            if len(mst_neighbors) > 0:
                # Boost edges connecting high-centrality MST nodes
                avg_centrality = np.mean(centrality[mst_neighbors])
                local_boost = 1 + boost_factor * centrality[i] * avg_centrality
                mst_indicator[i, mst_neighbors] *= local_boost
    except Exception:
        mst_indicator = np.zeros_like(distance_matrix)
    
    # 5. Nearest-neighbor chain component
    nn_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        
        # Take top k nearest neighbors, adaptive based on n
        k = max(2, min(5, n // 4))
        if k < n - 1:
            nn_indices = np.argsort(row)[:k]
        else:
            nn_indices = np.arange(n)
            nn_indices = nn_indices[nn_indices != i]
        
        # Assign scores with exponential decay
        for rank, j in enumerate(nn_indices):
            decay = np.exp(-rank / 1.5)
            nn_matrix[i, j] = reciprocal[i, j] * decay
    
    # 6. Reinforcement-adaptive weighting
    size_factor = min(1.0, np.log1p(n) / np.log1p(50))
    
    # Base weights adapt to instance characteristics
    if cv_dist > 1.0:
        w1, w2, w3, w4 = 0.25, 0.35, 0.25, 0.15
    elif cv_dist < 0.3:
        w1, w2, w3, w4 = 0.5, 0.2, 0.15, 0.15
    else:
        w1 = 0.4 - 0.1 * size_factor
        w2 = 0.3 + 0.1 * size_factor
        w3 = 0.2
        w4 = 0.1 + 0.05 * size_factor
    
    # Adjust based on stagnation feedback
    stagnation_bonus = min(0.2, 0.05 * feedback_state['stagnation_counter'])
    if feedback_state['stagnation_counter'] > 5:
        # Shift weight toward structural components during stagnation
        w3 += stagnation_bonus * 0.5
        w4 += stagnation_bonus * 0.5
        w1 -= stagnation_bonus
        w1 = max(0.1, w1)
    
    total = w1 + w2 + w3 + w4
    w1, w2, w3, w4 = w1 / total, w2 / total, w3 / total, w4 / total
    
    combined = (w1 * reciprocal + 
                w2 * centrality_matrix + 
                w3 * mst_indicator + 
                w4 * nn_matrix)
    
    # 7. Adaptive sparsification with reinforcement guidance
    sparsified = np.zeros_like(combined)
    
    # Base k scales with instance size and variability
    base_k = max(2, int(np.sqrt(n)))
    if cv_dist > 0.8:
        k_min = base_k + 1
        k_max = min(n - 1, base_k * 2)
    else:
        k_min = max(2, base_k - 1)
        k_max = min(n - 1, base_k + 2)
    
    # Adjust k range based on stagnation
    if feedback_state['stagnation_counter'] > 3:
        k_min = min(k_min + 1, n - 1)
        k_max = min(k_max + 2, n - 1)
    
    for i in range(n):
        # Stochastic k selection with adaptive distribution
        k_mean = (k_min + k_max) / 2
        k_std = (k_max - k_min) / 4
        k = int(np.clip(np.random.normal(k_mean, k_std), k_min, k_max))
        k = min(k, n - 1)
        
        row = combined[i].copy()
        row[i] = -np.inf
        
        if k < n - 1:
            sorted_indices = np.argsort(row)[::-1]
            keep_indices = sorted_indices[:k]
        else:
            keep_indices = np.arange(n)
            keep_indices = keep_indices[keep_indices != i]
        
        # Adaptive decay with reinforcement adjustment
        sorted_keep = keep_indices[np.argsort(row[keep_indices])[::-1]]
        base_decay = 1.5 + 0.5 * size_factor
        
        for rank, j in enumerate(sorted_keep):
            # Slower decay during stagnation to preserve diversity
            stagnation_factor = 1 + 0.2 * min(feedback_state['stagnation_counter'] / 10, 1.0)
            decay_rate = base_decay * stagnation_factor
            decay = np.exp(-rank / decay_rate)
            
            # Adaptive noise based on instance and stagnation
            noise_level = feedback_state['adaptive_exploration']
            if feedback_state['stagnation_counter'] > 5:
                noise_level *= 1.5
            
            noise = 1 + noise_level * np.random.randn()
            noise = np.clip(noise, 0.7, 1.3)
            sparsified[i, j] = combined[i, j] * decay * noise
    
    # 8. Enhanced symmetrization with feedback-aware smoothing
    symmetric = (sparsified + sparsified.T) / 2
    
    # Add strategic noise for exploration
    noise_scale = 0.01 / (1 + size_factor)
    if feedback_state['stagnation_counter'] > 2:
        noise_scale *= (1 + 0.5 * min(feedback_state['stagnation_counter'] / 10, 1.0))
    
    noise_matrix = np.random.randn(n, n) * noise_scale
    np.fill_diagonal(noise_matrix, 0)
    symmetric = symmetric + noise_matrix
    symmetric = np.maximum(symmetric, 0)
    
    # 9. Final row normalization for stochastic sampling
    row_sums = symmetric.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    symmetric = symmetric / row_sums
    
    np.fill_diagonal(symmetric, 0)
    return symmetric
