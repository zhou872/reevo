import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse import csr_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with MST, Christofides-based tour generation,
    2-opt local search, and adaptive hybrid scoring for improved edge quality.
    
    Parameters:
    distance_matrix (np.ndarray): Square symmetric matrix of distances between nodes.
    
    Returns:
    np.ndarray: Matrix of the same shape with scores indicating edge promise.
                Higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        inv_dist = np.zeros_like(distance_matrix, dtype=float)
        np.divide(1.0, distance_matrix, out=inv_dist, where=distance_matrix != 0)
        np.fill_diagonal(inv_dist, 0.0)
        return inv_dist
    
    # Step 1: Base heuristic - inverse distance
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    np.divide(1.0, distance_matrix, out=inv_dist, where=distance_matrix != 0)
    np.fill_diagonal(inv_dist, 0.0)
    
    # Step 2: Compute Minimum Spanning Tree (MST)
    mst_csr = minimum_spanning_tree(csr_matrix(distance_matrix))
    mst_mask = mst_csr.toarray() != 0
    mst_mask = mst_mask | mst_mask.T  # Make symmetric
    
    # Step 3: Adaptive parameter tuning based on instance size
    alpha = 0.5 + 0.3 * np.exp(-n / 50.0)  # MST boost factor
    beta = 0.2 + 0.2 * np.exp(-n / 100.0)  # Centrality weight
    gamma1 = 0.3 * np.exp(-n / 80.0)  # Initial tour boost
    gamma2 = 0.5 * np.exp(-n / 80.0)  # Improved tour boost
    k = max(2, min(8, n // 20))  # For k-NN centrality
    
    # Step 4: Generate initial tour using Christofides approximation or Nearest Neighbor
    def christofides_tour(dist_mat):
        n_nodes = dist_mat.shape[0]
        # Compute MST
        mst = minimum_spanning_tree(csr_matrix(dist_mat)).toarray()
        mst_bool = mst != 0
        mst_bool = mst_bool | mst_bool.T
        
        # Find odd-degree vertices
        degree = mst_bool.sum(axis=1)
        odd_vertices = np.where(degree % 2 == 1)[0]
        
        # Greedy matching on odd vertices
        matched = np.zeros(n_nodes, dtype=bool)
        matching_edges = []
        odd_list = list(odd_vertices)
        
        # Create and sort pairs by distance
        pairs = []
        for idx_i, i in enumerate(odd_list):
            for j in odd_list[idx_i+1:]:
                pairs.append((dist_mat[i, j], i, j))
        pairs.sort()
        
        for d, i, j in pairs:
            if not matched[i] and not matched[j]:
                matched[i] = matched[j] = True
                matching_edges.append((i, j))
        
        # Build adjacency list for Eulerian circuit (multigraph with MST and matching edges)
        adj_list = {i: [] for i in range(n_nodes)}
        for i in range(n_nodes):
            for j in range(i+1, n_nodes):
                if mst_bool[i, j]:
                    adj_list[i].append(j)
                    adj_list[j].append(i)
        for i, j in matching_edges:
            adj_list[i].append(j)
            adj_list[j].append(i)
        
        # Hierholzer's algorithm for Eulerian circuit
        circuit = []
        stack = [0]
        while stack:
            v = stack[-1]
            if adj_list[v]:
                u = adj_list[v].pop()
                stack.append(u)
            else:
                circuit.append(stack.pop())
        circuit.reverse()
        
        # Shortcut to Hamiltonian cycle
        visited = set()
        tour = []
        for node in circuit:
            if node not in visited:
                visited.add(node)
                tour.append(node)
        return tour
    
    def nearest_neighbor_tour(dist_mat):
        n_nodes = dist_mat.shape[0]
        start = np.random.randint(0, n_nodes)
        visited = set([start])
        tour = [start]
        current = start
        while len(visited) < n_nodes:
            row = dist_mat[current]
            # Find nearest unvisited node
            min_dist = np.inf
            next_node = -1
            for j in range(n_nodes):
                if j not in visited and row[j] < min_dist:
                    min_dist = row[j]
                    next_node = j
            if next_node == -1:
                break
            visited.add(next_node)
            tour.append(next_node)
            current = next_node
        return tour
    
    # Choose tour generation method based on instance size
    if n <= 200:
        initial_tour = christofides_tour(distance_matrix)
    else:
        initial_tour = nearest_neighbor_tour(distance_matrix)
    
    # Step 5: Improve tour with 2-opt local search (limited for efficiency)
    def two_opt_improve(tour, dist_mat, max_iter=5):
        n_nodes = len(tour)
        improved_tour = tour[:]
        improved = True
        iter_count = 0
        while improved and iter_count < max_iter:
            improved = False
            for i in range(n_nodes):
                for j in range(i+2, n_nodes):
                    # Indices modulo n for cycle
                    a = improved_tour[i]
                    b = improved_tour[(i+1) % n_nodes]
                    c = improved_tour[j]
                    d = improved_tour[(j+1) % n_nodes]
                    
                    current_len = dist_mat[a, b] + dist_mat[c, d]
                    new_len = dist_mat[a, c] + dist_mat[b, d]
                    
                    if new_len < current_len:
                        # Reverse segment between i+1 and j
                        start_rev = (i+1) % n_nodes
                        end_rev = j % n_nodes
                        if start_rev <= end_rev:
                            improved_tour[start_rev:end_rev+1] = reversed(improved_tour[start_rev:end_rev+1])
                        else:
                            # Handle wrap-around case
                            segment = improved_tour[start_rev:] + improved_tour[:end_rev+1]
                            segment.reverse()
                            improved_tour[start_rev:] = segment[:n_nodes-start_rev]
                            improved_tour[:end_rev+1] = segment[n_nodes-start_rev:]
                        improved = True
                        break
                if improved:
                    break
            iter_count += 1
        return improved_tour
    
    # Apply 2-opt only for manageable instance sizes
    if n <= 1000:
        improved_tour = two_opt_improve(initial_tour, distance_matrix, max_iter=3)
    else:
        improved_tour = initial_tour  # Skip 2-opt for very large instances
    
    # Step 6: Create masks for edges in tours
    def tour_to_mask(tour, n_nodes):
        mask = np.zeros((n_nodes, n_nodes), dtype=bool)
        for idx in range(len(tour)):
            i = tour[idx]
            j = tour[(idx + 1) % len(tour)]
            mask[i, j] = mask[j, i] = True
        return mask
    
    initial_tour_mask = tour_to_mask(initial_tour, n)
    improved_tour_mask = tour_to_mask(improved_tour, n)
    
    # Step 7: Compute node centrality using MST degree and k-NN
    centrality = np.zeros(n, dtype=float)
    
    # MST degree component
    if mst_mask.any():
        mst_degree = mst_mask.sum(axis=1)
        if mst_degree.max() > 0:
            centrality += mst_degree / mst_degree.max()
    
    # k-nearest neighbor component
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        if n > k:
            nn_indices = np.argpartition(row, k)[:k]
            avg_nn_dist = row[nn_indices].mean()
            centrality[i] += 1.0 / (avg_nn_dist + 1e-12)
    
    # Normalize centrality
    if centrality.max() > 0:
        centrality = centrality / centrality.max()
    
    # Step 8: Combine scores with adaptive weights
    scores = inv_dist.copy()
    
    # Apply MST boost
    scores[mst_mask] *= (1.0 + alpha)
    
    # Apply tour boosts
    scores[initial_tour_mask] *= (1.0 + gamma1)
    scores[improved_tour_mask] *= (1.0 + gamma2)
    
    # Apply centrality weighting
    for i in range(n):
        for j in range(i+1, n):
            if i != j and distance_matrix[i, j] > 0:
                centrality_factor = 1.0 + beta * (centrality[i] + centrality[j]) / 2.0
                scores[i, j] *= centrality_factor
                scores[j, i] = scores[i, j]
    
    # Step 9: Adaptive sparsification with percentile threshold
    if n > 10:
        off_diag_scores = scores[~np.eye(n, dtype=bool)]
        if len(off_diag_scores) > 0:
            percentile = 40 + 20 * np.exp(-n / 50.0)  # 40-60th percentile based on n
            threshold = np.percentile(off_diag_scores, percentile)
            scores[scores < threshold] = 0.0
    
    # Step 10: Ensure minimum connectivity (each node has at least 2 edges)
    for i in range(n):
        non_zero_count = np.count_nonzero(scores[i])
        if non_zero_count < 2:
            row = distance_matrix[i].copy()
            row[i] = np.inf
            non_zero_mask = scores[i] > 0
            row[non_zero_mask] = np.inf
            nearest_indices = np.argsort(row)[:2]
            for j in nearest_indices:
                if row[j] < np.inf:
                    scores[i, j] = inv_dist[i, j] * 0.3
                    scores[j, i] = scores[i, j]
    
    # Step 11: Final normalization
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    return scores
