import numpy as np
from scipy.sparse import csgraph
from sklearn.cluster import KMeans
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic matrix for TSP edge selection with adaptive tuning and clustering.
    
    Key innovations:
    1. Adaptive weight tuning based on problem metrics (size, distance distribution)
    2. Hierarchical clustering heuristics for edge grouping
    3. Solution sampling via random and greedy tours for edge frequency analysis
    4. Dynamic sparsification with clustering-aware edge preservation
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape indicating edge promise (higher = better)
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic[np.isinf(heuristic)] = 0
            heuristic = np.nan_to_num(heuristic, nan=0)
        np.fill_diagonal(heuristic, 0)
        return heuristic
    
    # ========== Phase 1: Problem Analysis for Adaptive Tuning ==========
    
    # Analyze distance distribution
    non_diag = distance_matrix[~np.eye(n, dtype=bool)]
    dist_mean = np.mean(non_diag)
    dist_std = np.std(non_diag)
    dist_cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    
    # Determine problem characteristics
    is_sparse = dist_cv > 1.0  # High CV indicates sparse/irregular distribution
    size_factor = n / 100.0  # Normalized size factor
    
    # ========== Phase 2: Compute Core Components ==========
    
    # 1. Reciprocal distance component (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal = np.nan_to_num(reciprocal, nan=0)
    np.fill_diagonal(reciprocal, 0)
    if reciprocal.max() > 0:
        reciprocal = reciprocal / reciprocal.max()
    
    # 2. Enhanced centrality using PageRank-like algorithm
    # Create transition matrix based on inverse distances
    transition = reciprocal.copy()
    np.fill_diagonal(transition, 0)
    row_sums = transition.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    transition = transition / row_sums
    
    # Power iteration for centrality (simplified PageRank)
    centrality = np.ones(n) / n
    for _ in range(20):
        centrality = 0.85 * transition.T @ centrality + 0.15 * np.ones(n) / n
    
    # Create centrality matrix (geometric mean for edges)
    centrality_matrix = np.sqrt(np.outer(centrality, centrality))
    np.fill_diagonal(centrality_matrix, 0)
    if centrality_matrix.max() > 0:
        centrality_matrix = centrality_matrix / centrality_matrix.max()
    
    # 3. Minimum Spanning Tree indicator
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_indicator = (mst.toarray() > 0).astype(float)
    
    # 4. Clustering-based heuristic (if n >= 4)
    cluster_component = np.zeros((n, n))
    if n >= 4:
        # Use KMeans to cluster nodes (coordinates approximated by MDS)
        # Simple approximation: use distance matrix as feature vector
        try:
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                # Number of clusters based on problem size
                n_clusters = min(10, max(2, int(np.sqrt(n))))
                
                # Use distance to all other nodes as features
                features = distance_matrix.copy()
                
                kmeans = KMeans(n_clusters=n_clusters, n_init=5, random_state=42)
                labels = kmeans.fit_predict(features)
                
                # Boost edges within same cluster
                for i in range(n):
                    for j in range(i + 1, n):
                        if labels[i] == labels[j]:
                            # Boost inversely proportional to cluster size
                            cluster_size = np.sum(labels == labels[i])
                            boost = 1.0 / (1.0 + np.log1p(cluster_size))
                            cluster_component[i, j] = cluster_component[j, i] = boost
                
                # Normalize cluster component
                if cluster_component.max() > 0:
                    cluster_component = cluster_component / cluster_component.max()
        except:
            # Fallback to simple distance-based clustering
            pass
    
    # 5. Solution sampling for edge frequency analysis
    frequency_component = np.zeros((n, n))
    
    # Generate random and greedy tours
    n_samples = min(100, n * 5)  # Adaptive sampling size
    np.random.seed(42)  # For reproducibility
    
    for _ in range(n_samples):
        # Random tour
        tour = np.random.permutation(n)
        for k in range(n):
            i, j = tour[k], tour[(k + 1) % n]
            frequency_component[i, j] += 1
            frequency_component[j, i] += 1
        
        # Greedy tour starting from random node
        if n > 10:
            start = np.random.randint(n)
            visited = set([start])
            current = start
            
            while len(visited) < n:
                # Find nearest unvisited neighbor
                unvisited = [j for j in range(n) if j not in visited]
                if not unvisited:
                    break
                nearest = min(unvisited, key=lambda j: distance_matrix[current, j])
                frequency_component[current, nearest] += 2  # Higher weight for greedy edges
                frequency_component[nearest, current] += 2
                visited.add(nearest)
                current = nearest
            
            # Close the tour
            frequency_component[current, start] += 2
            frequency_component[start, current] += 2
    
    # Normalize frequency component
    if frequency_component.max() > 0:
        frequency_component = frequency_component / frequency_component.max()
    
    # ========== Phase 3: Adaptive Weight Combination ==========
    
    # Dynamic weight adjustment based on problem characteristics
    if is_sparse:
        # For sparse/irregular problems, emphasize clustering and sampling
        base_weights = {
            'reciprocal': 0.3,
            'centrality': 0.2,
            'mst': 0.1,
            'cluster': 0.25,
            'frequency': 0.15
        }
    else:
        # For regular problems, balanced approach
        base_weights = {
            'reciprocal': 0.35,
            'centrality': 0.25,
            'mst': 0.1,
            'cluster': 0.15,
            'frequency': 0.15
        }
    
    # Adjust weights based on problem size
    size_adjustment = min(1.0, size_factor)
    base_weights['cluster'] *= (1.0 + 0.5 * size_adjustment)
    base_weights['frequency'] *= (1.0 + 0.3 * size_adjustment)
    
    # Normalize weights to sum to 1
    total_weight = sum(base_weights.values())
    weights = {k: v / total_weight for k, v in base_weights.items()}
    
    # Combine components
    heuristic = (
        weights['reciprocal'] * reciprocal +
        weights['centrality'] * centrality_matrix +
        weights['mst'] * mst_indicator +
        weights['cluster'] * cluster_component +
        weights['frequency'] * frequency_component
    )
    
    # Ensure symmetry
    heuristic = (heuristic + heuristic.T) / 2
    np.fill_diagonal(heuristic, 0)
    
    # ========== Phase 4: Dynamic Sparsification ==========
    
    # Determine edges to preserve based on clustering
    edges_to_preserve = set()
    
    # Always preserve MST edges
    mst_edges = np.where(mst_indicator > 0)
    for i, j in zip(*mst_edges):
        if i != j:
            edges_to_preserve.add((min(i, j), max(i, j)))
    
    # Preserve top intra-cluster edges if clustering was successful
    if n >= 4 and cluster_component.max() > 0:
        cluster_threshold = np.percentile(cluster_component[cluster_component > 0], 80)
        cluster_edges = np.where(cluster_component > cluster_threshold)
        for i, j in zip(*cluster_edges):
            if i != j:
                edges_to_preserve.add((min(i, j), max(i, j)))
    
    # Preserve top frequency edges
    if frequency_component.max() > 0:
        freq_threshold = np.percentile(frequency_component[frequency_component > 0], 75)
        freq_edges = np.where(frequency_component > freq_threshold)
        for i, j in zip(*freq_edges):
            if i != j:
                edges_to_preserve.add((min(i, j), max(i, j)))
    
    # Determine target edges per node based on problem size
    if n < 50:
        edges_per_node = n - 1
    elif n < 200:
        edges_per_node = int(3 * np.sqrt(n))
    else:
        edges_per_node = int(2 * np.sqrt(n))
    
    # Create sparsified matrix
    sparsified = np.zeros_like(heuristic)
    
    # First, preserve all essential edges
    for i, j in edges_to_preserve:
        sparsified[i, j] = sparsified[j, i] = heuristic[i, j]
    
    # For each node, select additional promising edges
    for i in range(n):
        # Count already preserved edges for this node
        current_preserved = sum(1 for j in range(n) 
                              if (min(i, j), max(i, j)) in edges_to_preserve and i != j)
        
        if current_preserved >= edges_per_node:
            continue
        
        # Get candidate edges for this node, sorted by heuristic value
        candidates = [(j, heuristic[i, j]) for j in range(n) 
                     if i != j and (min(i, j), max(i, j)) not in edges_to_preserve]
        
        if not candidates:
            continue
        
        candidates.sort(key=lambda x: x[1], reverse=True)
        
        # Select additional edges with rank-based decay
        max_additional = edges_per_node - current_preserved
        selected = candidates[:max_additional]
        
        for rank, (j, value) in enumerate(selected):
            # Exponential decay with slower decay for high-quality edges
            if rank < 3:
                decay = 1.0  # Full weight for top 3 edges
            else:
                decay = np.exp(-(rank - 2) / 5.0)
            
            sparsified[i, j] = value * decay
    
    # Ensure symmetry and zero diagonal
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    # Final normalization to [0, 1]
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
