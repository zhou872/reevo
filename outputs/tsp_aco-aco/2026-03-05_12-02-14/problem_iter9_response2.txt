import numpy as np
from scipy.sparse import csgraph
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Simplified and stabilized heuristic matrix for TSP edge selection.
    Uses fixed component weights with adaptive sparsity control.
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape indicating edge promise (higher = better)
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic[np.isinf(heuristic)] = 0
            heuristic = np.nan_to_num(heuristic, nan=0)
        np.fill_diagonal(heuristic, 0)
        return heuristic
    
    # ========== Phase 1: Core Components ==========
    
    # 1. Reciprocal distance baseline (inversely proportional to distance)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal = np.nan_to_num(reciprocal, nan=0)
    np.fill_diagonal(reciprocal, 0)
    
    # Normalize reciprocal component
    if reciprocal.max() > 0:
        reciprocal = reciprocal / reciprocal.max()
    
    # 2. MST edge indicator (binary mask for MST edges)
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    mst_indicator = (mst_array > 0).astype(float)
    
    # Ensure symmetry of MST indicator (MST is undirected)
    mst_indicator = np.maximum(mst_indicator, mst_indicator.T)
    
    # 3. Centrality matrix (sum of reciprocal distances per node, then combine)
    # Node centrality = sum of reciprocal distances to all other nodes
    node_centrality = reciprocal.sum(axis=1, keepdims=True)
    
    # Edge centrality = product of node centralities (captures "hub" connections)
    centrality_matrix = node_centrality @ node_centrality.T
    
    # Normalize centrality matrix
    if centrality_matrix.max() > 0:
        centrality_matrix = centrality_matrix / centrality_matrix.max()
    
    # ========== Phase 2: Weighted Combination ==========
    
    # Fixed weights based on empirical tuning
    # Weights: 0.6 for reciprocal, 0.3 for centrality, 0.1 for MST
    W_RECIPROCAL = 0.6
    W_CENTRALITY = 0.3
    W_MST = 0.1
    
    # Combine components with fixed weights
    combined = (W_RECIPROCAL * reciprocal + 
                W_CENTRALITY * centrality_matrix + 
                W_MST * mst_indicator)
    
    # Ensure symmetry (components are symmetric but numerical errors may occur)
    combined = (combined + combined.T) / 2
    np.fill_diagonal(combined, 0)
    
    # ========== Phase 3: Adaptive Sparsity Control ==========
    
    # Determine target edges per node based on problem size
    if n < 50:
        edges_per_node = n - 1  # Keep all edges for small problems
    elif n < 200:
        edges_per_node = int(2 * np.sqrt(n))
    else:
        edges_per_node = int(1.5 * np.sqrt(n))
    
    # Always preserve MST edges
    mst_edges = set()
    mst_edges.update(zip(*np.where(mst_indicator > 0)))
    mst_edges.update([(j, i) for i, j in mst_edges])  # Add symmetric counterparts
    
    # Create sparsified matrix
    sparsified = np.zeros_like(combined)
    
    # First, preserve all MST edges with full weight
    for i, j in mst_edges:
        if i != j:
            sparsified[i, j] = combined[i, j]
    
    # For each node, select additional top edges with exponential decay
    for i in range(n):
        # Get all edges from node i, excluding self and MST edges
        candidates = []
        for j in range(n):
            if i != j and (i, j) not in mst_edges:
                candidates.append((j, combined[i, j]))
        
        if not candidates:
            continue
        
        # Sort candidates by heuristic value (descending)
        candidates.sort(key=lambda x: x[1], reverse=True)
        
        # Calculate how many additional edges we can add for this node
        current_mst_edges = sum(1 for j in range(n) if (i, j) in mst_edges and i != j)
        max_additional = max(0, edges_per_node - current_mst_edges)
        
        # Take top candidates with exponential decay
        selected = candidates[:max_additional]
        for rank, (j, value) in enumerate(selected):
            decay = np.exp(-rank / 3.0)  # Exponential decay factor
            sparsified[i, j] = value * decay
    
    # Ensure symmetry and final processing
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    # Final normalization
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
