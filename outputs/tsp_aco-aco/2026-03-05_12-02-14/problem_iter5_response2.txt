import numpy as np
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection with improved centrality measures
    and adaptive sparsification based on long-term suggestions.
    
    Features:
    1. Reciprocal distance: Closer nodes are more promising
    2. Centrality products: Combine closeness and shortest-path centrality
    3. Adaptive sparsification: Keep top k edges per node
    4. Stochastic decay: Apply exponential decay based on edge rank
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape indicating edge promise
    """
    n = distance_matrix.shape[0]
    
    # 1. Base reciprocal distance (closer nodes are more promising)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # 2. Enhanced centrality measures
    
    # 2a. Closeness centrality: inverse of row sums (as suggested)
    # This measures how central a node is based on its average distance to others
    row_sums = np.sum(distance_matrix, axis=1, where=distance_matrix > 0)
    closeness_centrality = np.zeros(n)
    valid_mask = row_sums > 0
    closeness_centrality[valid_mask] = 1 / row_sums[valid_mask]
    
    # Normalize to [0, 1] range
    if np.any(closeness_centrality > 0):
        closeness_centrality = (closeness_centrality - np.min(closeness_centrality)) / \
                              (np.max(closeness_centrality) - np.min(closeness_centrality) + 1e-10)
    
    # 2b. Shortest path centrality using Floyd-Warshall (as suggested)
    # This measures how central a node is in the network of shortest paths
    all_pairs_shortest = csgraph.shortest_path(distance_matrix, directed=False)
    
    # Avoid division by zero and handle disconnected nodes
    all_pairs_shortest = np.where(all_pairs_shortest > 0, all_pairs_shortest, np.inf)
    
    # Compute centrality as sum of inverse distances
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_shortest = 1 / all_pairs_shortest
        inv_shortest[np.isinf(inv_shortest)] = 0
        inv_shortest[np.isnan(inv_shortest)] = 0
    
    # Sum over all target nodes for each source node
    shortest_path_centrality = np.sum(inv_shortest, axis=1)
    
    # Normalize to [0, 1] range
    if np.any(shortest_path_centrality > 0):
        shortest_path_centrality = (shortest_path_centrality - np.min(shortest_path_centrality)) / \
                                  (np.max(shortest_path_centrality) - np.min(shortest_path_centrality) + 1e-10)
    
    # 2c. Combine centrality measures via outer product (as suggested)
    # Use the geometric mean to combine both centrality measures
    combined_centrality = np.sqrt(closeness_centrality * shortest_path_centrality)
    centrality_matrix = np.outer(combined_centrality, combined_centrality)
    
    # 3. Minimum spanning tree indicator
    # Edges in MST are often part of good TSP solutions
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_indicator = (mst > 0).astype(float)
    
    # 4. Combine factors with tuned weights
    w1, w2, w3 = 0.6, 0.3, 0.1  # Slightly adjusted weights
    combined = (w1 * reciprocal + 
                w2 * centrality_matrix + 
                w3 * mst_indicator)
    
    # 5. Adaptive sparsification with stochastic decay
    sparsified = np.zeros_like(combined)
    
    # Adaptive k: keep more edges for smaller graphs
    # Base k on log of n to balance sparsity and connectivity
    base_k = int(np.ceil(np.log2(n + 1)))
    k = max(2, min(base_k, n // 2))  # Keep between 2 and n/2 edges
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get indices of top k edges
        if k < n:
            top_k_idx = np.argpartition(row, -k)[-k:]
        else:
            top_k_idx = np.arange(n)
            top_k_idx = top_k_idx[top_k_idx != i]
        
        # Apply stochastic decay based on rank
        sorted_idx = top_k_idx[np.argsort(row[top_k_idx])[::-1]]
        for rank, j in enumerate(sorted_idx):
            # Exponential decay with steeper falloff for lower ranks
            decay = np.exp(-rank / 1.5)  # Adjusted decay rate
            sparsified[i, j] = combined[i, j] * decay
    
    # Ensure symmetry for undirected TSP
    sparsified = (sparsified + sparsified.T) / 2
    
    # Final cleanup: zero out diagonal
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
