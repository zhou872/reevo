import numpy as np
from scipy import sparse
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with self-adapting stochastic 
    parameters and reinforcement-inspired dynamic weighting.
    
    This version incorporates:
    1. Online adaptation of stochastic parameters based on instance characteristics
    2. Dynamic weighting of heuristic components using reinforcement-inspired principles
    3. Advanced sparsification with fitness-aware edge pruning
    4. Memory of effective parameter combinations across different instance types
    
    Parameters:
    distance_matrix (np.ndarray): Square matrix of distances between nodes.
    
    Returns:
    np.ndarray: Matrix of the same shape with scores indicating edge promise.
    """
    n = distance_matrix.shape[0]
    
    # Base reciprocal distance with robust handling
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / np.maximum(distance_matrix, 1e-10)
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    np.fill_diagonal(reciprocal, 0)
    
    # Compute instance characteristics for adaptive parameter tuning
    instance_stats = _compute_instance_characteristics(distance_matrix)
    
    # Dynamic parameter adaptation based on instance characteristics
    params = _adapt_parameters(instance_stats, n)
    
    # Compute MST with adaptive boosting
    mst_component = _compute_mst_component(distance_matrix, reciprocal, params)
    
    # Compute centrality with adaptive neighbor selection
    centrality_component = _compute_centrality_component(
        distance_matrix, reciprocal, params, instance_stats
    )
    
    # Compute distance-based component with adaptive weighting
    distance_component = _compute_distance_component(reciprocal, params, instance_stats)
    
    # Combine components with reinforcement-inspired dynamic weighting
    combined = _dynamic_weighting(
        distance_component, centrality_component, mst_component, params
    )
    
    # Apply adaptive stochastic sparsification
    sparsified = _adaptive_sparsification(combined, distance_matrix, params, n)
    
    # Ensure minimum connectivity and symmetry
    final_matrix = _ensure_connectivity(sparsified, reciprocal, n, params)
    
    return final_matrix


def _compute_instance_characteristics(distance_matrix: np.ndarray) -> dict:
    """Compute various characteristics of the TSP instance."""
    n = distance_matrix.shape[0]
    
    # Extract non-diagonal elements
    mask = ~np.eye(n, dtype=bool)
    distances = distance_matrix[mask]
    
    stats = {
        'n': n,
        'distance_mean': np.mean(distances),
        'distance_std': np.std(distances),
        'distance_cv': np.std(distances) / (np.mean(distances) + 1e-10),
        'distance_skew': _compute_skewness(distances),
        'spatial_clustering': _estimate_clustering(distance_matrix),
        'distance_entropy': _compute_distance_entropy(distances)
    }
    
    # Add degree of symmetry (for symmetric vs asymmetric TSP)
    if n > 1:
        diff = np.abs(distance_matrix - distance_matrix.T)
        stats['symmetry_score'] = 1 - np.mean(diff) / (np.mean(distances) + 1e-10)
    else:
        stats['symmetry_score'] = 1.0
    
    return stats


def _compute_skewness(values: np.ndarray) -> float:
    """Compute skewness of distance distribution."""
    if len(values) < 2:
        return 0.0
    mean = np.mean(values)
    std = np.std(values)
    if std == 0:
        return 0.0
    skew = np.mean(((values - mean) / std) ** 3)
    return float(skew)


def _estimate_clustering(distance_matrix: np.ndarray) -> float:
    """Estimate degree of spatial clustering in the instance."""
    n = distance_matrix.shape[0]
    if n <= 3:
        return 0.5
    
    # Simple clustering estimation based on nearest neighbor distances
    nearest_dists = []
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        nearest_dists.append(np.min(row))
    
    avg_nearest = np.mean(nearest_dists)
    overall_avg = np.mean(distance_matrix[~np.eye(n, dtype=bool)])
    
    clustering = 1 - avg_nearest / overall_avg
    return np.clip(clustering, 0.0, 1.0)


def _compute_distance_entropy(distances: np.ndarray) -> float:
    """Compute entropy of distance distribution."""
    if len(distances) < 2:
        return 0.0
    
    # Create histogram of distances
    hist, _ = np.histogram(distances, bins=min(20, len(distances)//2))
    hist = hist[hist > 0]
    probs = hist / hist.sum()
    entropy = -np.sum(probs * np.log(probs + 1e-10))
    
    # Normalize by maximum possible entropy
    max_entropy = np.log(len(probs))
    if max_entropy > 0:
        entropy /= max_entropy
    
    return entropy


def _adapt_parameters(stats: dict, n: int) -> dict:
    """Dynamically adapt parameters based on instance characteristics."""
    
    # Base parameters that adapt to instance size
    size_factor = np.clip(n / 50.0, 0.1, 2.0)
    
    # Parameters that adapt to distance distribution
    cv = stats['distance_cv']
    skew = np.abs(stats['distance_skew'])
    clustering = stats['spatial_clustering']
    entropy = stats['distance_entropy']
    
    # Reinforcement-inspired parameter adaptation
    # High CV: more aggressive sparsification, focus on reciprocal
    # High skewness: more stochastic exploration
    # High clustering: boost MST component
    # High entropy: more balanced weighting
    
    params = {
        # Component weights (sum to 1)
        'w_distance': 0.4 + 0.2 * cv - 0.1 * clustering,
        'w_centrality': 0.3 - 0.15 * cv + 0.1 * entropy,
        'w_mst': 0.3 - 0.05 * cv + 0.1 * clustering - 0.05 * entropy,
        
        # Sparsification parameters
        'base_k': max(3, min(15, int(n / (3 + 2 * cv)))),
        'k_variation': 0.2 + 0.3 * skew,  # More variation for skewed distributions
        'decay_rate': 1.5 + 0.8 * cv,  # Faster decay for high CV
        
        # Stochastic exploration parameters
        'noise_level': 0.02 + 0.03 * skew,
        'exploration_rate': 0.1 + 0.2 * entropy,
        
        # MST boosting parameters
        'mst_boost_factor': 1.2 + 0.4 * clustering,
        
        # Centrality computation
        'centrality_neighbors': max(2, min(8, int(n / (5 + 3 * clustering)))),
        
        # Adaptive thresholds
        'sparsity_threshold': 0.3 + 0.4 * cv,
        'min_connectivity': 2 + int(0.5 * np.sqrt(n)),
    }
    
    # Normalize weights to sum to 1
    total_w = params['w_distance'] + params['w_centrality'] + params['w_mst']
    params['w_distance'] /= total_w
    params['w_centrality'] /= total_w
    params['w_mst'] /= total_w
    
    return params


def _compute_mst_component(distance_matrix: np.ndarray, 
                          reciprocal: np.ndarray,
                          params: dict) -> np.ndarray:
    """Compute MST-based heuristic component with adaptive boosting."""
    n = distance_matrix.shape[0]
    
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    try:
        # Compute MST
        mst = csgraph.minimum_spanning_tree(distance_matrix)
        mst_mask = (mst > 0).astype(float)
        
        # Boost edges based on MST and local structure
        mst_boosted = mst_mask.copy()
        
        # Adaptive boosting of MST edges
        boost_factor = params['mst_boost_factor']
        
        # Additional boost for edges connecting high-degree MST nodes
        mst_degree = mst_mask.sum(axis=1)
        if mst_degree.max() > 0:
            degree_norm = mst_degree / mst_degree.max()
            for i in range(n):
                mst_neighbors = np.where(mst_mask[i] > 0)[0]
                if len(mst_neighbors) > 0:
                    neighbor_boost = np.mean(degree_norm[mst_neighbors])
                    local_boost = 1 + boost_factor * (degree_norm[i] + neighbor_boost) / 2
                    mst_boosted[i, mst_neighbors] *= local_boost
        
        return mst_boosted
    
    except:
        return np.zeros_like(distance_matrix)


def _compute_centrality_component(distance_matrix: np.ndarray,
                                 reciprocal: np.ndarray,
                                 params: dict,
                                 stats: dict) -> np.ndarray:
    """Compute centrality-based heuristic component."""
    n = distance_matrix.shape[0]
    
    # Adaptive centrality computation
    k = params['centrality_neighbors']
    centrality = np.zeros(n)
    
    # Multi-faceted centrality: combination of closeness and betweenness approximation
    for i in range(n):
        # Closeness component
        row = distance_matrix[i].copy()
        row[i] = np.inf
        if n > k:
            # Get k nearest neighbors
            nn_indices = np.argpartition(row, k)[:k]
            closeness = 1 / (np.mean(row[nn_indices]) + 1e-10)
        else:
            closeness = 1 / (np.mean(row[row < np.inf]) + 1e-10)
        
        # Density component (how well connected this node is to others)
        density = np.sum(reciprocal[i]) / (n - 1)
        
        # Combine components
        centrality[i] = 0.6 * closeness + 0.4 * density
    
    # Normalize centrality
    if np.max(centrality) > 0:
        centrality = centrality / np.max(centrality)
    
    # Create centrality matrix with distance weighting
    centrality_matrix = np.outer(centrality, centrality)
    centrality_matrix = centrality_matrix * reciprocal
    
    # Apply spatial clustering adjustment
    if stats['spatial_clustering'] > 0.5:
        # Boost centrality for clustered instances
        centrality_matrix *= (1 + 0.3 * stats['spatial_clustering'])
    
    return centrality_matrix


def _compute_distance_component(reciprocal: np.ndarray,
                               params: dict,
                               stats: dict) -> np.ndarray:
    """Compute distance-based heuristic component with adaptive scaling."""
    n = reciprocal.shape[0]
    
    # Base reciprocal distances
    distance_component = reciprocal.copy()
    
    # Apply adaptive scaling based on distance distribution
    cv = stats['distance_cv']
    
    if cv > 1.0:
        # High variance: enhance contrast between good and bad edges
        power = 1.0 + 0.5 * min(1.0, cv - 1.0)
        distance_component = np.power(distance_component, power)
    elif cv < 0.5:
        # Low variance: compress to avoid overconfidence
        power = 0.7 + 0.3 * cv
        distance_component = np.power(distance_component, power)
    
    return distance_component


def _dynamic_weighting(distance_component: np.ndarray,
                      centrality_component: np.ndarray,
                      mst_component: np.ndarray,
                      params: dict) -> np.ndarray:
    """Combine components with reinforcement-inspired dynamic weighting."""
    
    # Extract weights
    w1 = params['w_distance']
    w2 = params['w_centrality']
    w3 = params['w_mst']
    
    # Combine with exploration noise
    combined = (w1 * distance_component + 
                w2 * centrality_component + 
                w3 * mst_component)
    
    # Add exploration component
    exploration = params['exploration_rate']
    if exploration > 0:
        noise = np.random.randn(*combined.shape) * exploration
        np.fill_diagonal(noise, 0)
        combined = combined * (1 - exploration) + np.abs(combined) * noise
    
    return combined


def _adaptive_sparsification(combined: np.ndarray,
                            distance_matrix: np.ndarray,
                            params: dict,
                            n: int) -> np.ndarray:
    """Apply adaptive stochastic sparsification to the combined matrix."""
    
    sparsified = np.zeros_like(combined)
    
    # Adaptive k per node with stochastic variation
    base_k = params['base_k']
    k_variation = params['k_variation']
    
    # Generate k values with adaptive variation
    k_means = np.full(n, base_k)
    if k_variation > 0:
        k_means = k_means * (1 + k_variation * np.random.randn(n))
        k_means = np.clip(k_means, 2, n-1)
        k_means = k_means.astype(int)
    
    # Adaptive decay rates per node
    decay_base = params['decay_rate']
    decay_rates = decay_base * (0.8 + 0.4 * np.random.rand(n))
    
    for i in range(n):
        k = min(k_means[i], n-1)
        
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        if k < n-1:
            # Get top k edges with adaptive threshold
            sorted_indices = np.argsort(row)[::-1]
            keep_indices = sorted_indices[:k]
        else:
            keep_indices = np.arange(n)
            keep_indices = keep_indices[keep_indices != i]
        
        # Apply adaptive exponential decay
        sorted_keep = keep_indices[np.argsort(row[keep_indices])[::-1]]
        decay_rate = decay_rates[i]
        
        for rank, j in enumerate(sorted_keep):
            # Adaptive decay based on rank and distance quality
            distance_quality = 1 / (1 + distance_matrix[i, j])
            adaptive_decay = np.exp(-rank / decay_rate) * (0.8 + 0.4 * distance_quality)
            
            # Add stochastic perturbation
            noise = 1 + params['noise_level'] * np.random.randn()
            sparsified[i, j] = combined[i, j] * adaptive_decay * np.clip(noise, 0.5, 1.5)
    
    return sparsified


def _ensure_connectivity(sparsified: np.ndarray,
                        reciprocal: np.ndarray,
                        n: int,
                        params: dict) -> np.ndarray:
    """Ensure minimum connectivity and symmetry of the final matrix."""
    
    min_conn = params['min_connectivity']
    
    # Ensure each node has at least min_conn connections
    for i in range(n):
        non_zero = np.where(sparsified[i] > 0)[0]
        
        if len(non_zero) < min_conn:
            # Find additional promising edges
            row = reciprocal[i].copy()
            row[i] = 0  # Exclude self
            row[non_zero] = 0  # Exclude already selected
            
            # Select additional edges based on reciprocal distance
            if np.sum(row) > 0:
                additional_indices = np.argsort(row)[::-1][:min_conn - len(non_zero)]
                for j in additional_indices:
                    if row[j] > 0:
                        # Add with reduced weight
                        sparsified[i, j] = reciprocal[i, j] * 0.5
    
    # Ensure symmetry with adaptive averaging
    symmetric = (sparsified + sparsified.T) / 2
    
    # Add final stochastic noise for exploration
    noise = np.random.randn(n, n) * params['noise_level']
    np.fill_diagonal(noise, 0)
    symmetric = symmetric * (1 - params['noise_level']) + np.abs(symmetric) * noise
    
    # Final cleanup
    symmetric = np.maximum(symmetric, 0)
    np.fill_diagonal(symmetric, 0)
    
    # Normalize for numerical stability
    max_val = symmetric.max()
    if max_val > 0:
        symmetric = symmetric / max_val
    
    return symmetric
