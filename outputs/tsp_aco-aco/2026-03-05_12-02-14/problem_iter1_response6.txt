import numpy as np
import scipy.sparse.csgraph as csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection combining multiple factors:
    1. Reciprocal distance (closer nodes are more promising)
    2. Node connectivity centrality (well-connected nodes indicate promising edges)
    3. Sparsification by setting low-potential edges to zero
    """
    n = distance_matrix.shape[0]
    
    # 1. Base reciprocal distance (handle zeros to avoid division by zero)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # 2. Node connectivity centrality
    # Compute shortest path centrality using Floyd-Warshall
    # Nodes that are central in shortest paths are more important
    centrality = np.zeros(n)
    for i in range(n):
        # Count how often node i appears in shortest paths between other nodes
        paths = csgraph.shortest_path(distance_matrix, directed=False, indices=i)
        centrality[i] = np.sum(1 / (1 + paths))  # Transform to centrality measure
    
    # Normalize centrality
    if np.max(centrality) > 0:
        centrality = centrality / np.max(centrality)
    
    # Create centrality matrix: edge promise based on centrality of its endpoints
    centrality_matrix = np.outer(centrality, centrality)
    
    # 3. Minimum spanning tree indicator
    # Edges in MST are often part of good TSP solutions
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_indicator = (mst > 0).astype(float)
    
    # 4. Combine factors
    # Weight each factor - these weights can be tuned
    w1, w2, w3 = 0.5, 0.3, 0.2
    combined = (w1 * reciprocal + 
                w2 * centrality_matrix + 
                w3 * mst_indicator)
    
    # 5. Adaptive sparsification
    # Keep top edges for each node, zero out the rest
    sparsified = np.zeros_like(combined)
    
    # Determine sparsity level adaptively based on matrix size
    # Keep more edges for smaller graphs, fewer for larger
    k = max(3, min(10, n // 4))  # Keep 3-10 nearest neighbors
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get indices of top k edges
        if k < n:
            top_k_idx = np.argpartition(row, -k)[-k:]
        else:
            top_k_idx = np.arange(n)
            top_k_idx = top_k_idx[top_k_idx != i]
        
        # Apply stochastic decay: even top edges get reduced weight
        # based on their rank
        sorted_idx = top_k_idx[np.argsort(row[top_k_idx])[::-1]]
        for rank, j in enumerate(sorted_idx):
            # Exponential decay based on rank
            decay = np.exp(-rank / 2)
            sparsified[i, j] = combined[i, j] * decay
    
    # Ensure symmetry (since TSP is typically symmetric)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Final cleanup: zero out diagonal
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
