class ReinforcementTuner:
    """Reinforcement learning agent for online parameter tuning."""
    
    def __init__(self, n_components: int = 5):
        self.n_components = n_components
        self.q_table = defaultdict(lambda: np.ones(n_components) / n_components)
        self.state_memory = deque(maxlen=100)
        self.reward_memory = deque(maxlen=100)
        self.learning_rate = 0.1
        self.discount_factor = 0.9
        self.exploration_rate = 0.2
        
    def get_state_signature(self, instance_stats: Dict) -> str:
        """Create compact state representation."""
        features = []
        features.append(f"n_{instance_stats.get('n', 0)}")
        features.append(f"cv_{instance_stats.get('cv', 1.0):.2f}")
        features.append(f"dens_{instance_stats.get('density', 0.5):.2f}")
        return "_".join(features)
    
    def select_action(self, state: str, current_weights: np.ndarray) -> np.ndarray:
        """Select component weights using epsilon-greedy strategy."""
        if random.random() < self.exploration_rate:
            # Exploration: random perturbation
            noise = np.random.randn(self.n_components) * 0.1
            new_weights = current_weights + noise
            new_weights = np.maximum(new_weights, 0)
            return new_weights / new_weights.sum()
        else:
            # Exploitation: use Q-table
            if state in self.q_table:
                return self.q_table[state]
            return current_weights
    
    def update_policy(self, state: str, action: np.ndarray, reward: float):
        """Update Q-values using TD learning."""
        if state in self.q_table:
            old_value = self.q_table[state]
            best_next = np.max(old_value)
            td_target = reward + self.discount_factor * best_next
            td_error = td_target - np.mean(old_value)
            self.q_table[state] = old_value + self.learning_rate * td_error * action
            self.q_table[state] = np.maximum(self.q_table[state], 0)
            self.q_table[state] /= self.q_table[state].sum()


class EnhancedkOptSolver:
    """Enhanced k-opt local search with LKH-inspired moves."""
    
    def __init__(self, max_k: int = 5):
        self.max_k = max_k
        self.gain_criteria = 0.001  # Minimum relative improvement
        self.candidate_set_size = 50
        
    def create_candidate_sets(self, dist_matrix: np.ndarray, alpha: float = 0.5) -> List[List[int]]:
        """Create candidate sets for each node using neighbor lists."""
        n = dist_matrix.shape[0]
        candidates = []
        
        for i in range(n):
            # Compute distances to all other nodes
            distances = dist_matrix[i].copy()
            distances[i] = np.inf
            
            # Sort by distance
            sorted_indices = np.argsort(distances)
            
            # Select candidate set: mix of closest and random promising nodes
            n_close = min(self.candidate_set_size // 2, n - 1)
            close_candidates = sorted_indices[:n_close]
            
            # Add random candidates based on reciprocal distance
            if n_close < self.candidate_set_size:
                remaining = sorted_indices[n_close:]
                if len(remaining) > 0:
                    recip_dist = 1.0 / (distances[remaining] + 1e-10)
                    recip_dist = recip_dist / recip_dist.sum()
                    n_random = min(len(remaining), self.candidate_set_size - n_close)
                    random_candidates = np.random.choice(
                        remaining, size=n_random, p=recip_dist, replace=False
                    )
                    all_candidates = np.concatenate([close_candidates, random_candidates])
                else:
                    all_candidates = close_candidates
            else:
                all_candidates = close_candidates
            
            candidates.append(all_candidates.tolist())
        
        return candidates
    
    def sequential_k_opt(self, tour: List[int], dist_matrix: np.ndarray,
                        candidate_sets: List[List[int]], max_iter: int = 100) -> List[int]:
        """Perform sequential k-opt moves with backtracking."""
        n = len(tour)
        if n <= 3:
            return tour
        
        # Create position mapping
        pos = {node: idx for idx, node in enumerate(tour)}
        current_tour = tour.copy()
        best_length = self.tour_length(current_tour, dist_matrix)
        improved = True
        iterations = 0
        
        while improved and iterations < max_iter:
            improved = False
            
            for i in range(n):
                if improved:
                    break
                    
                t1 = current_tour[i]
                t2 = current_tour[(i + 1) % n]
                
                # Try candidate edges for breaking
                for j_candidate in candidate_sets[t1]:
                    j = pos[j_candidate]
                    if j == i or (j + 1) % n == i:
                        continue
                    
                    t3 = current_tour[j]
                    t4 = current_tour[(j + 1) % n]
                    
                    # Try 2-opt first (most common improvement)
                    gain = dist_matrix[t1, t2] + dist_matrix[t3, t4] - \
                           dist_matrix[t1, t3] - dist_matrix[t2, t4]
                    
                    if gain > self.gain_criteria * dist_matrix[t1, t2]:
                        # Apply 2-opt move
                        new_tour = self.apply_2opt(current_tour, i, j)
                        new_length = self.tour_length(new_tour, dist_matrix)
                        
                        if new_length < best_length - 1e-10:
                            current_tour = new_tour
                            best_length = new_length
                            improved = True
                            # Update position mapping
                            pos = {node: idx for idx, node in enumerate(current_tour)}
                            break
                    
                    # Try 3-opt if 2-opt didn't work
                    for k_candidate in candidate_sets[t2]:
                        k = pos[k_candidate]
                        if k == i or k == j or (k + 1) % n == i or (k + 1) % n == j:
                            continue
                        
                        t5 = current_tour[k]
                        t6 = current_tour[(k + 1) % n]
                        
                        # Evaluate different 3-opt configurations
                        gains = []
                        
                        # Configuration 1
                        gain1 = (dist_matrix[t1, t2] + dist_matrix[t3, t4] + dist_matrix[t5, t6] -
                                dist_matrix[t1, t3] - dist_matrix[t2, t5] - dist_matrix[t4, t6])
                        gains.append((gain1, 1))
                        
                        # Configuration 2
                        gain2 = (dist_matrix[t1, t2] + dist_matrix[t3, t4] + dist_matrix[t5, t6] -
                                dist_matrix[t1, t5] - dist_matrix[t2, t4] - dist_matrix[t3, t6])
                        gains.append((gain2, 2))
                        
                        # Configuration 3
                        gain3 = (dist_matrix[t1, t2] + dist_matrix[t3, t4] + dist_matrix[t5, t6] -
                                dist_matrix[t1, t5] - dist_matrix[t2, t6] - dist_matrix[t3, t4])
                        gains.append((gain3, 3))
                        
                        # Find best configuration
                        best_gain, best_config = max(gains, key=lambda x: x[0])
                        
                        if best_gain > self.gain_criteria * dist_matrix[t1, t2]:
                            new_tour = self.apply_kopt(current_tour, i, j, k, best_config)
                            new_length = self.tour_length(new_tour, dist_matrix)
                            
                            if new_length < best_length - 1e-10:
                                current_tour = new_tour
                                best_length = new_length
                                improved = True
                                pos = {node: idx for idx, node in enumerate(current_tour)}
                                break
                    
                    if improved:
                        break
            
            iterations += 1
        
        return current_tour
    
    def apply_2opt(self, tour: List[int], i: int, j: int) -> List[int]:
        """Apply 2-opt move."""
        n = len(tour)
        if i < j:
            return tour[:i+1] + tour[i+1:j+1][::-1] + tour[j+1:]
        else:
            segment = tour[i+1:] + tour[:j+1]
            segment = segment[::-1]
            new_tour = segment[-j-1:] + tour[j+1:i+1] + segment[:-j-1]
            return new_tour
    
    def apply_kopt(self, tour: List[int], i: int, j: int, k: int, config: int) -> List[int]:
        """Apply k-opt move for different configurations."""
        n = len(tour)
        # Simplified implementation - in practice would implement all configurations
        return self.apply_2opt(tour, i, j)
    
    def tour_length(self, tour: List[int], dist_matrix: np.ndarray) -> float:
        """Calculate tour length."""
        return sum(dist_matrix[tour[i], tour[(i + 1) % len(tour)]] for i in range(len(tour)))


class HierarchicalMemory:
    """Hierarchical memory management with LRU and frequency-based caching."""
    
    def __init__(self, max_size: int = 100, hierarchy_levels: int = 3):
        self.max_size = max_size
        self.hierarchy_levels = hierarchy_levels
        self.cache = {}
        self.access_freq = defaultdict(int)
        self.access_time = {}
        self.time_counter = 0
        
    def get_signature(self, dist_matrix: np.ndarray) -> str:
        """Create hierarchical signature for instance."""
        n = dist_matrix.shape[0]
        
        # Level 1: Basic statistics
        triu_dists = dist_matrix[np.triu_indices(n, 1)]
        stats1 = (n, np.mean(triu_dists) if len(triu_dists) > 0 else 0,
                 np.std(triu_dists) if len(triu_dists) > 0 else 0)
        
        # Level 2: Distance distribution percentiles
        if len(triu_dists) > 0:
            percentiles = np.percentile(triu_dists, [10, 25, 50, 75, 90])
            stats2 = tuple(percentiles.tolist())
        else:
            stats2 = (0, 0, 0, 0, 0)
        
        # Level 3: Geometric signature (for medium/large instances)
        if n <= 100:
            # Use all nodes for small instances
            sample_indices = np.arange(n)
        else:
            # Stratified sampling for larger instances
            sample_size = min(100, n)
            strata = min(10, n)
            samples_per_stratum = sample_size // strata
            sample_indices = []
            
            for stratum in range(strata):
                start = stratum * (n // strata)
                end = start + (n // strata) if stratum < strata - 1 else n
                stratum_indices = np.arange(start, end)
                if len(stratum_indices) > samples_per_stratum:
                    selected = np.random.choice(stratum_indices, samples_per_stratum, replace=False)
                else:
                    selected = stratum_indices
                sample_indices.extend(selected)
        
        # Compute pairwise distances for sampled nodes
        sample_matrix = dist_matrix[np.ix_(sample_indices, sample_indices)]
        triu_sample = sample_matrix[np.triu_indices(len(sample_indices), 1)]
        if len(triu_sample) > 0:
            stats3 = (np.mean(triu_sample), np.std(triu_sample), 
                     np.median(triu_sample), len(triu_sample))
        else:
            stats3 = (0, 0, 0, 0)
        
        # Combine all levels
        combined = stats1 + stats2 + stats3
        signature = hashlib.md5(str(combined).encode()).hexdigest()[:20]
        
        return signature
    
    def get(self, signature: str) -> Optional[Dict]:
        """Retrieve from cache with frequency bonus."""
        self.time_counter += 1
        
        if signature in self.cache:
            # Update access info
            self.access_freq[signature] += 1
            self.access_time[signature] = self.time_counter
            
            # Return cached value with frequency-based confidence
            cached = self.cache[signature]
            freq = self.access_freq[signature]
            confidence = 1.0 - np.exp(-freq / 10.0)
            cached['confidence'] = confidence
            
            return cached
        
        return None
    
    def put(self, signature: str, value: Dict):
        """Store in cache with eviction policy."""
        self.time_counter += 1
        
        if signature not in self.cache:
            # Evict if needed (LRU + frequency considered)
            if len(self.cache) >= self.max_size:
                # Score = recency + 0.5 * (1/frequency)
                scores = {}
                for sig in self.cache:
                    recency = self.time_counter - self.access_time.get(sig, 0)
                    freq = self.access_freq.get(sig, 1)
                    scores[sig] = recency + 0.5 * (1.0 / freq)
                
                # Remove worst scoring item
                evict_sig = max(scores.items(), key=lambda x: x[1])[0]
                del self.cache[evict_sig]
                del self.access_freq[evict_sig]
                del self.access_time[evict_sig]
        
        # Store new entry
        self.cache[signature] = value
        self.access_freq[signature] = 1
        self.access_time[signature] = self.time_counter

        base_weights = np.array([0.30, 0.20, 0.25, 0.15, 0.10])
    elif n < 1000:
        base_weights = np.array([0.25, 0.15, 0.30, 0.15, 0.15])
    else:
        base_weights = np.array([0.20, 0.10, 0.35, 0.15, 0.20])
    
    # Adjust based on distance characteristics
    if dist_cv > 1.0:
        base_weights[0] *= 1.1
        base_weights[2] *= 0.9
    
    if density < 0.1:
        base_weights[3] *= 1.2
    
    # Normalize
    base_weights /= base_weights.sum()
    
    # RL-based weight adjustment
    if cached_weights is not None:
        # Blend with cached weights
        blend_factor = min(0.5, confidence * 0.6)
        weights = (1 - blend_factor) * base_weights + blend_factor * cached_weights

        weights = rl_tuner.select_action(state_sig, base_weights)

        reciprocal,
        centrality_matrix,

                if matching_component[i, j] > 0:
                    boost *= 1.3
                if local_search_component[i, j] > 0.5:
                    boost *= 1.2
                
                sparsified[i, j] = combined[i, j] * decay * boost
    
    # ==================== SYMMETRIZATION AND FINAL PROCESSING ====================
    # Ensure symmetry with weighted averaging
    symmetric = 0.7 * sparsified + 0.3 * sparsified.T
    
    # Add exploration noise (reduced for large instances)
    noise_level = 0.02 if n < 1000 else 0.01
    if n > 10:
        noise = rng.randn(n, n) * noise_level
        noise = np.triu(noise) + np.triu(noise, 1).T
        np.fill_diagonal(noise, 0)
        symmetric = symmetric * (1.0 + noise)
    
    # Final cleanup
    symmetric = np.maximum(symmetric, 0)
    np.fill_diagonal(symmetric, 0)
    
    # Normalize
    max_val = symmetric.max()
    if max_val > 0:
        symmetric /= max_val
    
    # ==================== REINFORCEMENT LEARNING UPDATE ====================
    # Evaluate solution quality (simplified proxy)
    if n <= 1000:
        # Quick evaluation on small sample
        sample_tour = list(range(n))
        improved_tour = kopt_solver.sequential_k_opt(sample_tour, distance_matrix, candidate_sets, max_iter=10)
        tour_length = kopt_solver.tour_length(improved_tour, distance_matrix)
        
        # Reward based on tour quality (lower is better)
        baseline = np.mean(triu_dists) * n if len(triu_dists) > 0 else 1.0
        reward = baseline / max(tour_length, 1e-10)
        
        # Update RL policy
        rl_tuner.update_policy(state_sig, weights, reward)
    
    # ==================== MEMORY UPDATE ====================
    # Store in memory
    memory.put(instance_signature, {
        'weights': weights,
        'heuristic': symmetric,
        'stats': instance_stats
    })
    
    return symmetric
