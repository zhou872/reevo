import numpy as np
from scipy import sparse

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection with adaptive sparsification
    and improved centrality measures.
    """
    n = distance_matrix.shape[0]
    
    # 1. Reciprocal distance (closer nodes are more promising)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # 2. Closeness centrality (inverse of row sums)
    # Nodes with smaller total distance to others are more central
    row_sums = np.sum(distance_matrix, axis=1, keepdims=True)
    # Avoid division by zero
    row_sums[row_sums == 0] = 1
    closeness_centrality = 1 / row_sums
    closeness_centrality = closeness_centrality.flatten()
    
    # Normalize to [0, 1]
    if np.max(closeness_centrality) > 0:
        closeness_centrality = closeness_centrality / np.max(closeness_centrality)
    
    # Create centrality matrix via outer product
    # Edges connecting two central nodes are more promising
    centrality_matrix = np.outer(closeness_centrality, closeness_centrality)
    
    # 3. Minimum spanning tree indicator
    # Edges in MST are often part of good TSP solutions
    mst = sparse.csgraph.minimum_spanning_tree(distance_matrix)
    mst_indicator = (mst > 0).astype(float)
    
    # 4. Combine factors with adaptive weights
    # Weight based on matrix size (larger graphs need more sparsification)
    if n < 20:
        w1, w2, w3 = 0.6, 0.3, 0.1  # More weight to distance for small graphs
    elif n < 50:
        w1, w2, w3 = 0.5, 0.35, 0.15
    else:
        w1, w2, w3 = 0.4, 0.4, 0.2  # More weight to centrality for large graphs
    
    combined = (w1 * reciprocal + 
                w2 * centrality_matrix + 
                w3 * mst_indicator)
    
    # 5. Adaptive sparsification with stochastic decay
    sparsified = np.zeros_like(combined)
    
    # Adaptive k: keep more edges for smaller graphs
    k = max(2, min(15, int(np.sqrt(n))))  # Adaptive based on sqrt(n)
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get indices of top k edges
        if k < n - 1:
            top_k_idx = np.argpartition(row, -k)[-k:]
        else:
            top_k_idx = np.arange(n)
            top_k_idx = top_k_idx[top_k_idx != i]
        
        # Apply exponential decay based on rank
        if len(top_k_idx) > 0:
            # Sort in descending order
            sorted_values = row[top_k_idx]
            sort_order = np.argsort(sorted_values)[::-1]
            sorted_idx = top_k_idx[sort_order]
            
            for rank, j in enumerate(sorted_idx):
                # Exponential decay: top edge gets weight 1, others decay exponentially
                decay = np.exp(-rank / 2)  # Smooth decay
                sparsified[i, j] = combined[i, j] * decay
    
    # 6. Ensure symmetry (since TSP is typically symmetric)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 7. Final cleanup: zero out diagonal
    np.fill_diagonal(sparsified, 0)
    
    # 8. Normalize to [0, 1] range for consistency
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
