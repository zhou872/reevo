import numpy as np
from scipy.sparse import csgraph
from scipy.stats import skew, kurtosis

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristic with reinforcement learning-inspired parameter tuning
    and enhanced graph metric-based adaptation for broader problem domains.
    
    Key innovations:
    1. Graph topology classification for domain-specific adaptation
    2. Online parameter tuning via simulated bandit learning
    3. Multi-criteria edge evaluation with dynamic feature importance
    4. Intelligent sparsification with connectivity preservation
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Normalized heuristic matrix for stochastic edge sampling
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / np.maximum(distance_matrix, 1e-10)
        np.fill_diagonal(result, 0)
        row_sums = result.sum(axis=1, keepdims=True)
        row_sums[row_sums == 0] = 1
        return result / row_sums
    
    # ===== 1. ADVANCED GRAPH CHARACTERIZATION =====
    graph_metrics = compute_graph_metrics(distance_matrix, n)
    
    # Classify graph type for domain-specific adaptation
    graph_type = classify_graph_type(graph_metrics, n)
    
    # ===== 2. ONLINE PARAMETER TUNING VIA SIMULATED BANDIT =====
    tuning_params = adaptive_parameter_tuning(graph_metrics, graph_type, n)
    
    # ===== 3. MULTI-CRITERIA EDGE EVALUATION =====
    edge_features = compute_edge_features(distance_matrix, graph_metrics, n)
    
    # ===== 4. DYNAMIC FEATURE WEIGHTING =====
    feature_weights = compute_feature_weights(edge_features, graph_metrics, tuning_params)
    
    # ===== 5. FUSED EDGE SCORES =====
    edge_scores = compute_fused_scores(edge_features, feature_weights, tuning_params)
    
    # ===== 6. INTELLIGENT SPARSIFICATION =====
    sparsified = intelligent_sparsification(
        edge_scores, distance_matrix, graph_metrics, tuning_params, n
    )
    
    # ===== 7. FINAL ENHANCEMENTS =====
    result = final_enhancement(sparsified, distance_matrix, graph_metrics, tuning_params)
    
    return result

def compute_graph_metrics(distance_matrix: np.ndarray, n: int) -> dict:
    """Compute comprehensive graph metrics for domain adaptation."""
    # Extract upper triangle distances
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    # Basic statistics
    metrics = {
        'n': n,
        'dist_mean': np.mean(distances),
        'dist_std': np.std(distances),
        'dist_cv': np.std(distances) / (np.mean(distances) + 1e-10),
        'dist_skew': skew(distances) if len(distances) > 2 else 0,
        'dist_kurtosis': kurtosis(distances) if len(distances) > 2 else 0,
        'dist_min': np.min(distances),
        'dist_max': np.max(distances)
    }
    
    # Euclidean characteristics
    metrics['euclidean_score'] = estimate_euclidean_score(distance_matrix, n)
    
    # Graph structure metrics
    metrics['sparsity_pattern'] = analyze_sparsity_pattern(distance_matrix, n)
    metrics['cluster_score'] = estimate_cluster_tendency(distance_matrix, n)
    
    # Compute MST metrics
    try:
        mst = csgraph.minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = np.where(mst > 0)
        mst_lengths = mst[mst_edges]
        metrics['mst_length_mean'] = np.mean(mst_lengths)
        metrics['mst_length_std'] = np.std(mst_lengths)
        metrics['mst_edge_count'] = len(mst_lengths)
    except Exception:
        metrics['mst_length_mean'] = metrics['dist_mean']
        metrics['mst_length_std'] = metrics['dist_std']
        metrics['mst_edge_count'] = n - 1
    
    return metrics

def estimate_euclidean_score(distance_matrix: np.ndarray, n: int) -> float:
    """Enhanced Euclidean estimation with sampling and triangle inequality checks."""
    if n < 3:
        return 1.0
    
    sample_size = min(100, n)
    indices = np.random.choice(n, sample_size, replace=False)
    
    violations = 0
    total_checks = 0
    
    # Vectorized triangle inequality check
    for i in range(sample_size):
        for j in range(i + 1, sample_size):
            for k in range(j + 1, sample_size):
                a, b, c = indices[i], indices[j], indices[k]
                d_ab = distance_matrix[a, b]
                d_ac = distance_matrix[a, c]
                d_bc = distance_matrix[b, c]
                
                # Check all permutations with tolerance
                violations += int(d_ab > d_ac + d_bc + 1e-6 * d_ab)
                violations += int(d_ac > d_ab + d_bc + 1e-6 * d_ac)
                violations += int(d_bc > d_ab + d_ac + 1e-6 * d_bc)
                total_checks += 3
    
    return 1.0 - violations / max(total_checks, 1)

def analyze_sparsity_pattern(distance_matrix: np.ndarray, n: int) -> dict:
    """Analyze distance distribution patterns."""
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    # Quartiles
    q1, median, q3 = np.percentile(distances, [25, 50, 75])
    iqr = q3 - q1
    
    # Proportion of distances in different ranges
    prop_near = np.sum(distances < q1) / len(distances)
    prop_mid = np.sum((distances >= q1) & (distances <= q3)) / len(distances)
    prop_far = np.sum(distances > q3) / len(distances)
    
    return {
        'q1': q1, 'median': median, 'q3': q3, 'iqr': iqr,
        'prop_near': prop_near, 'prop_mid': prop_mid, 'prop_far': prop_far
    }

def estimate_cluster_tendency(distance_matrix: np.ndarray, n: int) -> float:
    """Estimate clustering tendency using nearest neighbor ratios."""
    if n < 4:
        return 0.5
    
    # For each node, compute ratio of 1st to 2nd nearest neighbor
    ratios = []
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf  # Exclude self
        sorted_dists = np.sort(row)
        if sorted_dists[1] > 0:  # Avoid division by zero
            ratio = sorted_dists[0] / sorted_dists[1]
            ratios.append(ratio)
    
    if ratios:
        avg_ratio = np.mean(ratios)
        # Convert to score between 0 and 1
        # Lower ratio indicates clearer clustering
        cluster_score = 1.0 - np.clip(avg_ratio, 0, 1)
    else:
        cluster_score = 0.5
    
    return cluster_score

def classify_graph_type(metrics: dict, n: int) -> str:
    """Classify graph type for domain-specific adaptation."""
    euclidean_score = metrics['euclidean_score']
    dist_cv = metrics['dist_cv']
    cluster_score = metrics['cluster_score']
    
    if euclidean_score > 0.85:
        if cluster_score > 0.7:
            return "Euclidean_Clustered"
        elif dist_cv < 0.3:
            return "Euclidean_Uniform"
        else:
            return "Euclidean_Varied"
    elif euclidean_score > 0.5:
        return "Semi_Euclidean"
    else:
        if dist_cv > 1.0:
            return "NonEuclidean_HighVar"
        elif cluster_score > 0.6:
            return "NonEuclidean_Clustered"
        else:
            return "NonEuclidean_General"

def adaptive_parameter_tuning(metrics: dict, graph_type: str, n: int) -> dict:
    """Reinforcement learning-inspired parameter tuning."""
    # Base parameters
    params = {
        # Component weights
        'w_inv_dist': 0.4,
        'w_mst': 0.3,
        'w_cent': 0.3,
        
        # Sparsification parameters
        'base_k': max(3, min(20, int(np.sqrt(n)))),
        'decay_rate': 1.5,
        'temperature': 0.5,
        
        # Feature importance parameters
        'feat_dist': 0.4,
        'feat_cluster': 0.3,
        'feat_mst': 0.3,
        
        # Exploration-exploitation
        'exploration_rate': 0.1,
        'noise_scale': 0.05
    }
    
    # Graph type specific adjustments
    type_adjustments = {
        "Euclidean_Clustered": {'w_mst': 0.4, 'w_cent': 0.2, 'feat_cluster': 0.4},
        "Euclidean_Uniform": {'w_inv_dist': 0.5, 'w_mst': 0.3, 'feat_dist': 0.5},
        "Euclidean_Varied": {'w_cent': 0.4, 'w_inv_dist': 0.3, 'temperature': 0.7},
        "Semi_Euclidean": {'w_mst': 0.25, 'w_cent': 0.35, 'exploration_rate': 0.15},
        "NonEuclidean_HighVar": {'w_cent': 0.5, 'w_inv_dist': 0.2, 'noise_scale': 0.1},
        "NonEuclidean_Clustered": {'w_cent': 0.4, 'feat_cluster': 0.4, 'base_k': params['base_k'] + 2},
        "NonEuclidean_General": {'w_inv_dist': 0.35, 'w_mst': 0.25, 'temperature': 0.6}
    }
    
    # Apply type adjustments
    if graph_type in type_adjustments:
        params.update(type_adjustments[graph_type])
    
    # Metric-based fine-tuning
    # Adjust based on distance statistics
    dist_cv = metrics['dist_cv']
    if dist_cv > 1.0:
        params['w_cent'] += 0.1
        params['w_inv_dist'] -= 0.05
        params['noise_scale'] *= 1.2
    elif dist_cv < 0.3:
        params['w_mst'] += 0.1
        params['exploration_rate'] *= 0.8
    
    # Adjust based on Euclidean score
    euclidean_score = metrics['euclidean_score']
    params['w_mst'] += 0.1 * euclidean_score
    params['w_cent'] -= 0.05 * euclidean_score
    
    # Adjust based on graph size
    size_factor = n / 1000
    params['base_k'] *= (1.0 + 0.3 * np.clip(size_factor, 0, 1))
    params['exploration_rate'] *= (1.0 - 0.2 * np.clip(size_factor, 0, 1))
    
    # Normalize weights
    total_w = params['w_inv_dist'] + params['w_mst'] + params['w_cent']
    params['w_inv_dist'] /= total_w
    params['w_mst'] /= total_w
    params['w_cent'] /= total_w
    
    total_feat = params['feat_dist'] + params['feat_cluster'] + params['feat_mst']
    params['feat_dist'] /= total_feat
    params['feat_cluster'] /= total_feat
    params['feat_mst'] /= total_feat
    
    return params

def compute_edge_features(distance_matrix: np.ndarray, metrics: dict, n: int) -> dict:
    """Compute multiple edge features for comprehensive evaluation."""
    features = {}
    
    # 1. Inverse distance feature
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / np.maximum(distance_matrix, metrics['dist_mean'] * 0.01)
    np.fill_diagonal(inv_dist, 0)
    features['inv_dist'] = inv_dist
    
    # 2. MST feature
    try:
        mst = csgraph.minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0).astype(float)
    except Exception:
        mst_mask = np.zeros((n, n))
    features['mst'] = mst_mask
    
    # 3. Centrality feature
    features['centrality'] = compute_enhanced_centrality(distance_matrix, metrics, n)
    
    # 4. Clustering feature
    features['clustering'] = compute_clustering_feature(distance_matrix, metrics, n)
    
    # 5. Neighborhood feature
    features['neighborhood'] = compute_neighborhood_feature(distance_matrix, n)
    
    return features

def compute_enhanced_centrality(distance_matrix: np.ndarray, metrics: dict, n: int) -> np.ndarray:
    """Compute enhanced centrality measures."""
    # Multiple centrality measures
    closeness_centrality = 1 / (distance_matrix.sum(axis=1) + 1e-10)
    
    # Degree centrality (using thresholded distances)
    threshold = metrics['dist_mean'] * 1.5
    degree_centrality = np.sum(distance_matrix <= threshold, axis=1) / (n - 1)
    
    # Harmonic centrality (works for disconnected graphs)
    with np.errstate(divide='ignore', invalid='ignore'):
        harmonic_centrality = np.sum(1 / np.maximum(distance_matrix, 1e-10), axis=1)
    harmonic_centrality = np.nan_to_num(harmonic_centrality, nan=0)
    
    # Combine based on graph characteristics
    euclidean_score = metrics['euclidean_score']
    dist_cv = metrics['dist_cv']
    
    if euclidean_score > 0.8:
        # Euclidean graphs: emphasize harmonic centrality
        weights = [0.2, 0.3, 0.5]  # closeness, degree, harmonic
    elif dist_cv > 1.0:
        # High variance: emphasize degree centrality
        weights = [0.3, 0.5, 0.2]
    else:
        weights = [0.4, 0.3, 0.3]
    
    combined_centrality = (
        weights[0] * closeness_centrality / (np.max(closeness_centrality) + 1e-10) +
        weights[1] * degree_centrality +
        weights[2] * harmonic_centrality / (np.max(harmonic_centrality) + 1e-10)
    )
    
    # Create centrality matrix
    centrality_matrix = np.outer(combined_centrality, combined_centrality)
    
    return centrality_matrix

def compute_clustering_feature(distance_matrix: np.ndarray, metrics: dict, n: int) -> np.ndarray:
    """Compute clustering tendency feature."""
    cluster_feature = np.zeros((n, n))
    
    # For each node pair, compute shared neighborhood similarity
    threshold = metrics['dist_mean']
    
    for i in range(n):
        # Find neighbors within threshold
        neighbors_i = distance_matrix[i] <= threshold
        for j in range(i + 1, n):
            neighbors_j = distance_matrix[j] <= threshold
            
            # Jaccard similarity of neighborhoods
            intersection = np.sum(neighbors_i & neighbors_j)
            union = np.sum(neighbors_i | neighbors_j)
            
            if union > 0:
                similarity = intersection / union
                # Adjust by distance (closer nodes in same cluster get higher score)
                dist_factor = np.exp(-distance_matrix[i, j] / threshold)
                cluster_feature[i, j] = cluster_feature[j, i] = similarity * dist_factor
    
    return cluster_feature

def compute_neighborhood_feature(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute neighborhood density feature."""
    neighborhood = np.zeros((n, n))
    
    # For each node, compute local density
    k = min(10, n - 1)
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argsort(distances)[:k]
        
        # Compute average distance to neighbors
        avg_nn_dist = np.mean(distances[nn_indices])
        
        # Assign feature value based on being in each other's neighborhood
        for j in range(n):
            if i != j:
                # Check if j is in i's neighborhood and vice versa
                in_i_neighborhood = distance_matrix[i, j] <= avg_nn_dist * 1.5
                # Symmetric feature
                if in_i_neighborhood:
                    density = 1.0 / (1.0 + distance_matrix[i, j] / avg_nn_dist)
                    neighborhood[i, j] = max(neighborhood[i, j], density)
    
    # Ensure symmetry
    neighborhood = np.maximum(neighborhood, neighborhood.T)
    
    return neighborhood

def compute_feature_weights(features: dict, metrics: dict, params: dict) -> dict:
    """Compute dynamic feature weights based on graph characteristics."""
    n = features['inv_dist'].shape[0]
    
    # Initialize with base weights
    weights = {
        'inv_dist': params['feat_dist'],
        'clustering': params['feat_cluster'],
        'mst': params['feat_mst'],
        'centrality': 0.0,  # Will be computed
        'neighborhood': 0.0  # Will be computed
    }
    
    # Adjust based on feature quality metrics
    
    # 1. Centrality feature quality
    centrality_var = np.var(features['centrality'])
    centrality_range = np.max(features['centrality']) - np.min(features['centrality'])
    if centrality_range > 0:
        centrality_quality = centrality_var / centrality_range
        weights['centrality'] = 0.2 * centrality_quality
    
    # 2. Neighborhood feature quality
    neighborhood_nonzero = np.sum(features['neighborhood'] > 0) / (n * n)
    weights['neighborhood'] = 0.15 * neighborhood_nonzero
    
    # 3. Adjust based on graph type
    if metrics['cluster_score'] > 0.6:
        weights['clustering'] *= 1.3
        weights['neighborhood'] *= 1.2
    
    if metrics['euclidean_score'] < 0.5:
        weights['centrality'] *= 1.4
        weights['clustering'] *= 0.8
    
    # Normalize weights
    total_weight = sum(weights.values())
    if total_weight > 0:
        for key in weights:
            weights[key] /= total_weight
    
    return weights

def compute_fused_scores(features: dict, weights: dict, params: dict) -> np.ndarray:
    """Compute fused edge scores with dynamic weighting."""
    n = features['inv_dist'].shape[0]
    
    # Initialize with weighted sum
    fused = np.zeros((n, n))
    
    for feature_name, weight in weights.items():
        if feature_name in features and weight > 0:
            feature_data = features[feature_name]
            # Normalize feature
            feature_max = np.max(feature_data)
            if feature_max > 0:
                feature_norm = feature_data / feature_max
            else:
                feature_norm = feature_data
            
            fused += weight * feature_norm
    
    # Apply component weights
    fused = (
        params['w_inv_dist'] * features['inv_dist'] +
        params['w_mst'] * features['mst'] * 2.0 +
        params['w_cent'] * fused
    )
    
    # Ensure symmetry and zero diagonal
    fused = (fused + fused.T) / 2
    np.fill_diagonal(fused, 0)
    
    return fused

def intelligent_sparsification(scores: np.ndarray, distance_matrix: np.ndarray,
                              metrics: dict, params: dict, n: int) -> np.ndarray:
    """Intelligent sparsification with connectivity preservation."""
    sparsified = np.zeros((n, n))
    
    # Determine k for each node adaptively
    base_k = int(params['base_k'])
    decay_rate = params['decay_rate']
    temperature = params['temperature']
    
    # Ensure MST edges are preserved
    try:
        mst = csgraph.minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = np.where(mst > 0)
        for i, j in zip(*mst_edges):
            if i != j:
                sparsified[i, j] = scores[i, j]
    except Exception:
        pass
    
    # Adaptive k per node based on local structure
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        # Calculate local statistics
        positive_mask = row_scores > 0
        if not positive_mask.any():
            # Fallback to distance-based selection
            k_fallback = min(base_k, n - 1)
            sorted_idx = np.argsort(distance_matrix[i])[1:k_fallback + 1]
            sparsified[i, sorted_idx] = 1.0
            continue
        
        positive_indices = np.where(positive_mask)[0]
        positive_scores = row_scores[positive_indices]
        
        # Determine adaptive k for this node
        row_mean = np.mean(positive_scores)
        row_std = np.std(positive_scores)
        row_cv = row_std / (row_mean + 1e-10)
        
        # Adjust k based on score distribution
        k_adj = base_k * (1.0 + 0.5 * np.clip(row_cv - 0.5, -0.5, 0.5))
        k_adj = min(max(int(k_adj), 2), n - 1, len(positive_indices))
        
        # Apply temperature-based stochastic selection
        if temperature > 0 and len(positive_scores) > 1:
            # Softmax sampling
            scaled_scores = positive_scores / temperature
            exp_scores = np.exp(scaled_scores - np.max(scaled_scores))
            probs = exp_scores / np.sum(exp_scores)
            
            # Sample with exploration
            if params['exploration_rate'] > 0 and np.random.random() < params['exploration_rate']:
                # Exploration: sample randomly
                selected_indices = np.random.choice(
                    positive_indices,
                    size=min(k_adj, len(positive_indices)),
                    replace=False
                )
            else:
                # Exploitation: sample according to probabilities
                selected_indices = np.random.choice(
                    positive_indices,
                    size=min(k_adj, len(positive_indices)),
                    p=probs,
                    replace=False
                )
        else:
            # Deterministic selection
            top_k = min(k_adj, len(positive_indices))
            top_idx = np.argsort(positive_scores)[-top_k:]
            selected_indices = positive_indices[top_idx]
        
        # Apply exponential decay
        selected_scores = row_scores[selected_indices]
        sorted_order = np.argsort(selected_scores)[::-1]
        ranks = np.arange(len(sorted_order))
        decay_factors = np.exp(-ranks / decay_rate)
        
        for rank_idx, orig_idx in enumerate(sorted_order):
            j = selected_indices[orig_idx]
            sparsified[i, j] = selected_scores[orig_idx] * decay_factors[rank_idx]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified

def final_enhancement(sparsified: np.ndarray, distance_matrix: np.ndarray,
                      metrics: dict, params: dict) -> np.ndarray:
    """Final enhancements including connectivity and exploration."""
    n = sparsified.shape[0]
    result = sparsified.copy()
    
    # 1. Ensure connectivity
    result = ensure_connectivity(result, distance_matrix, n)
    
    # 2. Add adaptive exploration noise
    if params['noise_scale'] > 0:
        noise = params['noise_scale'] * np.abs(np.random.randn(n, n))
        np.fill_diagonal(noise, 0)
        result = np.maximum(result + noise, 0)
    
    # 3. Row normalization with fallback
    row_sums = result.sum(axis=1, keepdims=True)
    zero_rows = row_sums.flatten() == 0
    
    if np.any(zero_rows):
        # Fallback to inverse distance
        with np.errstate(divide='ignore', invalid='ignore'):
            fallback = 1 / np.maximum(distance_matrix[zero_rows], 1e-10)
        np.fill_diagonal(fallback, 0)
        result[zero_rows] = fallback
        row_sums[zero_rows] = fallback.sum(axis=1, keepdims=True)
    
    row_sums[row_sums == 0] = 1
    result = result / row_sums
    
    return result

def ensure_connectivity(matrix: np.ndarray, distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Ensure the sparsified matrix maintains graph connectivity."""
    result = matrix.copy()
    
    # Check connectivity via BFS
    def is_connected(adj_matrix):
        visited = np.zeros(n, dtype=bool)
        stack = [0]
        visited[0] = True
        
        while stack:
            node = stack.pop()
            neighbors = np.where(adj_matrix[node] > 0)[0]
            for neighbor in neighbors:
                if not visited[neighbor]:
                    visited[neighbor] = True
                    stack.append(neighbor)
        
        return np.all(visited)
    
    if not is_connected(result):
        # Add missing edges from MST to ensure connectivity
        try:
            mst = csgraph.minimum_spanning_tree(distance_matrix).toarray()
            mst_edges = np.where(mst > 0)
            for i, j in zip(*mst_edges):
                if i != j and result[i, j] == 0:
                    # Add with appropriate weight
                    result[i, j] = result[j, i] = np.mean(result[result > 0]) if np.any(result > 0) else 1.0
        except Exception:
            # Fallback: connect to nearest neighbors
            for i in range(n):
                if np.sum(result[i] > 0) == 0:
                    # Find closest node
                    distances = distance_matrix[i].copy()
                    distances[i] = np.inf
                    closest = np.argmin(distances)
                    result[i, closest] = result[closest, i] = 1.0
    
    return result
