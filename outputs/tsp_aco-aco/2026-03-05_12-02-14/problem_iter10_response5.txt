import numpy as np
from scipy import sparse
from scipy.sparse import csgraph
from collections import defaultdict, deque
import warnings
from sklearn.cluster import DBSCAN
import random

warnings.filterwarnings('ignore')

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic matrix for TSP edge selection with 3-opt refinement,
    dynamic parameter tuning, and edge quality learning.
    
    Key innovations:
    1. 3-opt local search: More powerful tour improvement than 2-opt
    2. Dynamic parameter tuning: Adjusts weights based on problem characteristics
    3. Edge quality learning: Statistical learning with multiple features
    4. Christofides algorithm components: MST, perfect matching on odd vertices
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape indicating edge promise (higher = better)
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic[np.isinf(heuristic)] = 0
            heuristic = np.nan_to_num(heuristic, nan=0)
        np.fill_diagonal(heuristic, 0)
        return heuristic
    
    # ========== Phase 1: Christofides-Inspired Components ==========
    
    # 1A. Minimum Spanning Tree (MST) with importance weighting
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    
    # Compute MST edge importance based on tree structure
    mst_importance = np.zeros_like(distance_matrix)
    mst_edges = np.where(mst_array > 0)
    for i, j in zip(*mst_edges):
        if i != j:
            # Weight by inverse distance and node degree in MST
            degree_i = (mst_array[i] > 0).sum() + (mst_array[:, i] > 0).sum()
            degree_j = (mst_array[j] > 0).sum() + (mst_array[:, j] > 0).sum()
            mst_importance[i, j] = (1.0 / distance_matrix[i, j]) * (1 + 0.1 * (degree_i + degree_j))
    
    # 1B. Perfect Matching on odd-degree vertices
    degrees = (mst_array > 0).sum(axis=1)
    odd_vertices = np.where(degrees % 2 == 1)[0]
    
    matching_boost = np.zeros_like(distance_matrix)
    if len(odd_vertices) > 0:
        # Improved greedy matching with lookahead
        sorted_pairs = []
        for idx, i in enumerate(odd_vertices):
            for j in odd_vertices[idx+1:]:
                # Consider both distance and MST connectivity
                mst_weight = 1.0 if mst_array[i, j] > 0 or mst_array[j, i] > 0 else 0.5
                score = mst_weight / distance_matrix[i, j]
                sorted_pairs.append((score, distance_matrix[i, j], i, j))
        
        sorted_pairs.sort(reverse=True)  # Sort by score descending
        matched = set()
        
        for score, dist, i, j in sorted_pairs:
            if i not in matched and j not in matched:
                matching_boost[i, j] = matching_boost[j, i] = 1.0 / dist
                matched.update([i, j])
    
    # ========== Phase 2: 3-opt Tour Refinement with Memory ==========
    
    edge_memory = defaultdict(float)
    recent_tours = deque(maxlen=10)  # Keep more tours for better statistics
    
    def apply_3opt(tour, max_iterations=50):
        """Apply 3-opt local search to improve a tour."""
        n_nodes = len(tour)
        best_tour = tour.copy()
        best_length = sum(distance_matrix[tour[i], tour[(i+1)%n_nodes]] 
                         for i in range(n_nodes))
        
        improved = True
        iterations = 0
        
        while improved and iterations < max_iterations:
            improved = False
            
            # Randomize the order of i, j, k to explore different moves
            indices = list(range(n_nodes))
            random.shuffle(indices)
            
            for i_idx in indices:
                i = i_idx
                for j in range((i+2)%n_nodes, (i+n_nodes-2)%n_nodes, 1):
                    j = j % n_nodes
                    for k in range((j+2)%n_nodes, (i+n_nodes-1)%n_nodes, 1):
                        k = k % n_nodes
                        
                        if i == j or j == k or i == k:
                            continue
                        
                        # Get the six nodes involved in the 3-opt move
                        a, b = tour[i], tour[(i+1)%n_nodes]
                        c, d = tour[j], tour[(j+1)%n_nodes]
                        e, f = tour[k], tour[(k+1)%n_nodes]
                        
                        # Current edges
                        current = (distance_matrix[a, b] + 
                                  distance_matrix[c, d] + 
                                  distance_matrix[e, f])
                        
                        # Try all 3-opt moves (4 possibilities excluding the original)
                        # Move 1: (a,c), (b,d), (e,f)
                        new1 = (distance_matrix[a, c] + 
                               distance_matrix[b, d] + 
                               distance_matrix[e, f])
                        
                        # Move 2: (a,b), (c,e), (d,f)
                        new2 = (distance_matrix[a, b] + 
                               distance_matrix[c, e] + 
                               distance_matrix[d, f])
                        
                        # Move 3: (a,d), (e,b), (c,f)
                        new3 = (distance_matrix[a, d] + 
                               distance_matrix[e, b] + 
                               distance_matrix[c, f])
                        
                        # Move 4: (a,e), (d,b), (c,f)
                        new4 = (distance_matrix[a, e] + 
                               distance_matrix[d, b] + 
                               distance_matrix[c, f])
                        
                        # Find best improvement
                        best_new = min(new1, new2, new3, new4)
                        
                        if best_new < current - 1e-10:  # Small tolerance
                            # Apply the best move
                            new_tour = tour.copy()
                            if best_new == new1:
                                # Reverse segment b..c
                                if (i+1)%n_nodes < j:
                                    new_tour[(i+1)%n_nodes:j+1] = new_tour[(i+1)%n_nodes:j+1][::-1]
                                else:
                                    segment = new_tour[(i+1)%n_nodes:] + new_tour[:j+1]
                                    segment = segment[::-1]
                                    new_tour[(i+1)%n_nodes:] = segment[:n_nodes-(i+1)%n_nodes]
                                    new_tour[:j+1] = segment[n_nodes-(i+1)%n_nodes:]
                            elif best_new == new2:
                                # Reverse segment d..e
                                if (j+1)%n_nodes < k:
                                    new_tour[(j+1)%n_nodes:k+1] = new_tour[(j+1)%n_nodes:k+1][::-1]
                                else:
                                    segment = new_tour[(j+1)%n_nodes:] + new_tour[:k+1]
                                    segment = segment[::-1]
                                    new_tour[(j+1)%n_nodes:] = segment[:n_nodes-(j+1)%n_nodes]
                                    new_tour[:k+1] = segment[n_nodes-(j+1)%n_nodes:]
                            elif best_new == new3:
                                # Complex reversal: reverse b..c and d..e
                                # This is equivalent to a 4-opt move, skip for simplicity
                                continue
                            elif best_new == new4:
                                # Reverse segments b..c and d..e
                                # Skip complex moves for stability
                                continue
                            
                            tour = new_tour
                            improved = True
                            break
                    if improved:
                        break
                if improved:
                    break
            
            iterations += 1
        
        # Fallback to 2-opt if 3-opt didn't improve much
        if iterations == 0:
            # Simple 2-opt
            for i in range(n_nodes):
                for j in range(i+2, n_nodes):
                    a, b = tour[i], tour[(i+1)%n_nodes]
                    c, d = tour[j], tour[(j+1)%n_nodes]
                    
                    current = distance_matrix[a, b] + distance_matrix[c, d]
                    new = distance_matrix[a, c] + distance_matrix[b, d]
                    
                    if new < current - 1e-10:
                        tour[i+1:j+1] = tour[i+1:j+1][::-1]
                        improved = True
                        break
                if improved:
                    break
        
        return tour
    
    # Generate diverse initial tours
    initial_tours = []
    
    # 1. Nearest neighbor tours from different starts
    starts = np.random.choice(n, min(15, n), replace=False)
    for start in starts:
        tour = [start]
        visited = set([start])
        
        while len(visited) < n:
            current = tour[-1]
            mask = np.ones(n, dtype=bool)
            mask[list(visited)] = False
            if not np.any(mask):
                break
            distances = distance_matrix[current].copy()
            distances[~mask] = np.inf
            distances += np.random.rand(n) * 1e-6 * np.max(distance_matrix)
            next_node = np.argmin(distances)
            tour.append(next_node)
            visited.add(next_node)
        
        initial_tours.append(tour)
    
    # 2. Random insertion tours
    for _ in range(min(5, n//2)):
        nodes = list(range(n))
        random.shuffle(nodes)
        initial_tours.append(nodes)
    
    # Improve tours with 3-opt and build edge memory
    for tour in initial_tours:
        improved_tour = apply_3opt(tour.copy())
        recent_tours.append(improved_tour)
        
        # Calculate tour quality weight
        tour_length = sum(distance_matrix[improved_tour[i], 
                         improved_tour[(i+1)%n]] for i in range(n))
        weight = 1.0 / (tour_length + 1e-10)
        
        # Learn edge features from this tour
        for i in range(n):
            u, v = improved_tour[i], improved_tour[(i+1)%n]
            edge_key = (min(u, v), max(u, v))
            
            # Enhanced edge memory with feature learning
            current_weight = edge_memory.get(edge_key, 0)
            
            # Feature 1: Distance-based importance
            dist_feature = 1.0 / distance_matrix[u, v]
            
            # Feature 2: Position in tour (edges near start/end are often important)
            position_feature = 1.0 - min(i, n-i) / (n/2)
            
            # Feature 3: Node degree in tour
            degree_u = sum(1 for j in range(n) if improved_tour[j] == u or 
                          improved_tour[(j+1)%n] == u)
            degree_v = sum(1 for j in range(n) if improved_tour[j] == v or 
                          improved_tour[(j+1)%n] == v)
            degree_feature = (degree_u + degree_v) / 4.0
            
            # Combine features
            feature_score = (0.5 * dist_feature + 
                           0.3 * position_feature + 
                           0.2 * degree_feature)
            
            edge_memory[edge_key] = current_weight + weight * feature_score
    
    # ========== Phase 3: Dynamic Parameter Tuning ==========
    
    # Analyze problem characteristics
    triu_distances = distance_matrix[np.triu_indices(n, k=1)]
    dist_stats = {
        'mean': np.mean(triu_distances),
        'std': np.std(triu_distances),
        'cv': np.std(triu_distances) / (np.mean(triu_distances) + 1e-10),
        'q25': np.percentile(triu_distances, 25),
        'q75': np.percentile(triu_distances, 75)
    }
    
    # Clustering analysis
    epsilon = dist_stats['q25']  # Adaptive epsilon based on distances
    db = DBSCAN(eps=epsilon, min_samples=max(3, n//20), metric='precomputed')
    cluster_labels = db.fit_predict(distance_matrix)
    
    n_clusters = len(set(cluster_labels)) - (1 if -1 in cluster_labels else 0)
    
    cluster_boost = np.ones_like(distance_matrix)
    if n_clusters > 1:
        for i in range(n):
            for j in range(i+1, n):
                if cluster_labels[i] == cluster_labels[j] and cluster_labels[i] != -1:
                    # Intra-cluster edges
                    cluster_size = np.sum(cluster_labels == cluster_labels[i])
                    density = cluster_size / n
                    boost = 1.0 + 0.7 * min(1.0, density)
                    cluster_boost[i, j] = cluster_boost[j, i] = boost
                elif cluster_labels[i] != cluster_labels[j]:
                    # Inter-cluster edges
                    penalty = 0.3 + 0.7 * (distance_matrix[i, j] / 
                                          np.max(triu_distances))
                    cluster_boost[i, j] = cluster_boost[j, i] = max(0.1, penalty)
    
    # Dynamic weight adjustment based on problem characteristics
    if dist_stats['cv'] < 0.2:
        # Very uniform distances: emphasize structure
        weights = {'mst': 0.35, 'matching': 0.3, 'memory': 0.25, 'cluster': 0.1}
    elif dist_stats['cv'] > 0.8:
        # Highly variable distances: emphasize memory and distances
        weights = {'memory': 0.45, 'cluster': 0.25, 'mst': 0.2, 'matching': 0.1}
    elif n_clusters > max(3, n//10):
        # Clear cluster structure
        weights = {'cluster': 0.4, 'mst': 0.25, 'memory': 0.2, 'matching': 0.15}
    else:
        # Balanced case
        weights = {'memory': 0.35, 'mst': 0.25, 'matching': 0.2, 'cluster': 0.2}
    
    # Adjust based on problem size
    if n > 200:
        weights['memory'] = min(0.5, weights['memory'] * 1.2)
        weights['matching'] = max(0.05, weights['matching'] * 0.8)
    elif n < 50:
        weights['mst'] = min(0.5, weights['mst'] * 1.3)
        weights['matching'] = min(0.4, weights['matching'] * 1.2)
    
    # Normalize weights
    total = sum(weights.values())
    weights = {k: v/total for k, v in weights.items()}
    
    # ========== Phase 4: Combine Components with Learning ==========
    
    # Convert edge memory to matrix
    memory_matrix = np.zeros_like(distance_matrix)
    for (u, v), weight in edge_memory.items():
        memory_matrix[u, v] = memory_matrix[v, u] = weight
    
    # Normalize all components
    components = {'mst': mst_importance, 'matching': matching_boost, 
                  'memory': memory_matrix, 'cluster': cluster_boost}
    
    for key, mat in components.items():
        max_val = mat.max()
        if max_val > 0:
            components[key] = mat / max_val
    
    # Combine with dynamic weights
    combined = sum(weights[key] * components[key] for key in weights)
    
    # Add reciprocal distance as adaptive safety net
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal = np.nan_to_num(reciprocal, nan=0)
    
    if np.max(reciprocal) > 0:
        reciprocal /= np.max(reciprocal)
    
    # Adaptive safety weight based on solution quality variance
    if len(edge_memory) > 0:
        memory_vals = list(edge_memory.values())
        mem_cv = np.std(memory_vals) / (np.mean(memory_vals) + 1e-10)
        safety_weight = max(0.1, min(0.3, 0.3 * mem_cv))
    else:
        safety_weight = 0.2
    
    combined = (1 - safety_weight) * combined + safety_weight * reciprocal
    
    # ========== Phase 5: Intelligent Sparsification ==========
    
    # Determine sparsity based on problem characteristics
    if n < 50:
        edges_per_node = n - 1
    elif n < 200:
        base = 2.5 * np.sqrt(n)
        if dist_stats['cv'] > 0.5:
            base *= 1.2  # More edges for variable distances
        edges_per_node = int(base)
    else:
        base = 1.8 * np.sqrt(n)
        if n_clusters > 3:
            base *= 0.8  # Fewer edges for clustered problems
        edges_per_node = int(base)
    
    # Identify critical edges to preserve
    preserved_edges = set()
    
    # 1. MST edges
    mst_edge_set = set(zip(*np.where(mst_array > 0)))
    preserved_edges.update(mst_edge_set)
    
    # 2. High-quality memory edges
    if edge_memory:
        sorted_memory = sorted(edge_memory.items(), key=lambda x: x[1], reverse=True)
        top_k = min(len(sorted_memory), max(15, n))
        for (u, v), _ in sorted_memory[:top_k]:
            preserved_edges.add((u, v))
            preserved_edges.add((v, u))
    
    # 3. High centrality edges (connect high-degree nodes in MST)
    mst_degrees = (mst_array > 0).sum(axis=1) + (mst_array.T > 0).sum(axis=1)
    high_degree_nodes = np.where(mst_degrees > np.median(mst_degrees) + 1)[0]
    for i in high_degree_nodes:
        for j in high_degree_nodes:
            if i != j and distance_matrix[i, j] < np.percentile(triu_distances, 33):
                preserved_edges.add((i, j))
    
    # Create sparsified matrix
    sparsified = np.zeros_like(combined)
    
    # Preserve critical edges with full weight
    for i, j in preserved_edges:
        if i != j:
            sparsified[i, j] = combined[i, j]
    
    # For each node, add additional promising edges
    for i in range(n):
        # Count already preserved edges for this node
        preserved_count = sum(1 for j in range(n) if (i, j) in preserved_edges and i != j)
        remaining_slots = max(0, edges_per_node - preserved_count)
        
        if remaining_slots == 0:
            continue
        
        # Get candidate edges, sorted by heuristic value
        candidates = []
        for j in range(n):
            if i != j and (i, j) not in preserved_edges and combined[i, j] > 0:
                candidates.append((j, combined[i, j]))
        
        if not candidates:
            continue
        
        candidates.sort(key=lambda x: x[1], reverse=True)
        
        # Select top candidates with adaptive decay
        selected = candidates[:min(remaining_slots * 2, len(candidates))]
        for rank, (j, value) in enumerate(selected):
            if rank < remaining_slots:
                # Exponential decay adjusted by node degree
                degree_factor = 1.0 - 0.1 * min(3, preserved_count / max(1, edges_per_node/2))
                decay = np.exp(-rank / (2.0 * degree_factor))
                sparsified[i, j] = value * decay
    
    # Ensure symmetry and final processing
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    # Final normalization with clipping to avoid extreme values
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    # Clip very small values to zero for sparsity
    sparsified[sparsified < 1e-6] = 0
    
    return sparsified
