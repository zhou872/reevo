def _analyze_instance(distance_matrix: np.ndarray, n: int) -> dict:
    """Analyze instance characteristics for adaptive algorithm selection."""
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    features = {}
    
    # Basic statistics
    features['size'] = n
    features['mean_dist'] = np.mean(distances)
    features['std_dist'] = np.std(distances)
    features['cv_dist'] = features['std_dist'] / (features['mean_dist'] + 1e-10)
    
    # Distribution shape
    features['skewness'] = np.mean((distances - features['mean_dist'])**3) / (features['std_dist']**3 + 1e-10)
    features['kurtosis'] = np.mean((distances - features['mean_dist'])**4) / (features['std_dist']**4 + 1e-10) - 3
    
    # Euclidean properties
    features['euclidean_score'] = _estimate_euclidean_score(distance_matrix, n)
    
    # Graph properties
    if n <= 1000:
        mst = sparse.csgraph.minimum_spanning_tree(distance_matrix)
        mst_density = (mst.nnz / 2) / (n * (n - 1) / 2)
        features['mst_density'] = mst_density
    
    # Percentiles
    percentiles = np.percentile(distances, [10, 25, 50, 75, 90])
    for i, p in enumerate([10, 25, 50, 75, 90]):
        features[f'p{p}'] = percentiles[i]
    
    # Clusterability indicator
    features['cluster_score'] = _estimate_clusterability(distance_matrix, n)
    
    return features


def _select_heuristics(features: dict, n: int) -> dict:
    """
    Select and weight construction heuristics using multi-armed bandit.
    Dynamically adapts to instance characteristics.
    """
    # Initialize bandit if not exists
    if not hasattr(_select_heuristics, 'bandits'):
        _select_heuristics.bandits = {}
    
    # Create instance signature for transfer learning
    sig = f"{features['size']}_{features['cv_dist']:.2f}_{features['euclidean_score']:.2f}"
    
    if sig not in _select_heuristics.bandits:
        # Initialize new bandit
        heuristics = ['nearest_neighbor', 'farthest_insertion', 'cheapest_insertion',
                     'random_insertion', 'savings', 'nearest_insertion']
        bandit = {
            'arms': {h: {'count': 1, 'value': 0.5, 'weight': 1.0/len(heuristics)} for h in heuristics},
            'total_count': len(heuristics)
        }
        _select_heuristics.bandits[sig] = bandit
    else:
        bandit = _select_heuristics.bandits[sig]
    
    # Adjust weights based on instance characteristics
    weights = {}
    
    # Base weights from bandit
    for h, arm in bandit['arms'].items():
        weights[h] = arm['weight']
    
    # Instance-specific adjustments
    if features['euclidean_score'] > 0.8:
        # Euclidean instances favor insertion heuristics
        weights['farthest_insertion'] *= 1.3
        weights['cheapest_insertion'] *= 1.2
        weights['nearest_neighbor'] *= 0.8
    
    if features['cv_dist'] > 1.5:
        # High variation favors diverse strategies
        weights['random_insertion'] *= 1.4
        weights['savings'] *= 1.2
    
    if n > 500:
        # Large instances favor simpler heuristics
        weights['nearest_neighbor'] *= 1.3
        weights['nearest_insertion'] *= 1.2
        weights['cheapest_insertion'] *= 0.7
    
    # Normalize
    total = sum(weights.values())
    if total > 0:
        weights = {k: v/total for k, v in weights.items()}
    
    return weights


def _neural_edge_scoring(distance_matrix: np.ndarray, n: int, features: dict) -> np.ndarray:
    """
    Multi-layer neural scoring of edges with online adaptation.
    Uses handcrafted features and adaptable weights.
    """
    scores = np.zeros((n, n))
    
    # Initialize neural weights if not exists
    if not hasattr(_neural_edge_scoring, 'weights'):
        # Two-layer neural network weights
        _neural_edge_scoring.weights = {
            'W1': np.random.randn(12, 8) * 0.1,  # Input to hidden
            'b1': np.zeros(8),
            'W2': np.random.randn(8, 4) * 0.1,   # Hidden to hidden
            'b2': np.zeros(4),
            'W3': np.random.randn(4, 1) * 0.1,   # Hidden to output
            'b3': np.zeros(1)
        }
    
    # Extract node features
    node_features = _extract_node_features(distance_matrix, n)
    
    # Score edges
    for i in range(n):
        for j in range(i+1, n):
            # Feature vector for edge (i, j)
            edge_feat = _create_edge_features(
                i, j, distance_matrix, node_features, features
            )
            
            # Neural forward pass
            W1, b1, W2, b2, W3, b3 = [
                _neural_edge_scoring.weights[k] for k in ['W1', 'b1', 'W2', 'b2', 'W3', 'b3']
            ]
            
            # Layer 1
            h1 = np.tanh(np.dot(edge_feat, W1) + b1)
            
            # Layer 2
            h2 = np.tanh(np.dot(h1, W2) + b2)
            
            # Output layer
            score = np.dot(h2, W3) + b3
            score = 1.0 / (1.0 + np.exp(-score[0]))  # Sigmoid
            
            scores[i, j] = scores[j, i] = score
    
    return scores


def _extract_node_features(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Extract features for each node."""
    features = np.zeros((n, 5))
    
    for i in range(n):
        distances = distance_matrix[i]
        nonzero = distances[distances > 0]
        
        if len(nonzero) > 0:
            features[i, 0] = np.mean(nonzero)  # Mean distance
            features[i, 1] = np.std(nonzero)   # Std distance
            features[i, 2] = np.min(nonzero)   # Min distance
            features[i, 3] = np.median(nonzero) # Median distance
            features[i, 4] = np.sum(1.0/(nonzero + 1e-10)) / len(nonzero)  # Harmonic mean
        else:
            features[i] = 1.0
    
    # Normalize
    for col in range(5):
        if np.max(features[:, col]) - np.min(features[:, col]) > 0:
            features[:, col] = (features[:, col] - np.min(features[:, col])) / \
                              (np.max(features[:, col]) - np.min(features[:, col]))
    
    return features


def _create_edge_features(i: int, j: int, dist_matrix: np.ndarray, 
                         node_features: np.ndarray, instance_features: dict) -> np.ndarray:
    """Create feature vector for edge (i, j)."""
    d = dist_matrix[i, j]
    
    # Basic features
    features = [
        d,
        1.0 / (d + 1e-10),
        node_features[i, 0],  # Mean distance from i
        node_features[j, 0],  # Mean distance from j
        node_features[i, 4],  # Harmonic centrality of i
        node_features[j, 4],  # Harmonic centrality of j
    ]
    
    # Interaction features
    features.append(np.abs(node_features[i, 0] - node_features[j, 0]))
    features.append(node_features[i, 0] + node_features[j, 0])
    
    # Instance features
    features.append(instance_features['cv_dist'])
    features.append(instance_features['euclidean_score'])
    features.append(instance_features.get('mst_density', 0.5))
    features.append(instance_features['cluster_score'])
    
    return np.array(features)


def _ensemble_sampling(distance_matrix: np.ndarray, n: int, 
                      heuristic_weights: dict) -> tuple:
    """
    Generate ensemble of solutions using multiple heuristics.
    Returns solutions and edge quality scores.
    """
    edge_counts = np.zeros((n, n))
    edge_qualities = np.zeros((n, n))
    solutions = []
    
    # Number of samples based on instance size
    num_samples = min(50, max(10, n))
    
    for _ in range(num_samples):
        # Select heuristic according to weights
        heuristic = np.random.choice(
            list(heuristic_weights.keys()),
            p=list(heuristic_weights.values())
        )
        
        # Generate solution
        tour = _apply_heuristic(heuristic, distance_matrix, n)
        
        if tour is not None:
            # Evaluate solution
            length = _tour_length(distance_matrix, tour)
            quality = 1.0 / (length + 1e-10)
            solutions.append((tour, quality))
            
            # Update edge counts with quality weighting
            for k in range(n):
                a, b = tour[k], tour[(k+1)%n]
                edge_counts[a, b] += 1
                edge_counts[b, a] += 1
                edge_qualities[a, b] += quality
                edge_qualities[b, a] += quality
    
    # Normalize edge qualities
    if edge_counts.max() > 0:
        edge_qualities = edge_qualities / (edge_counts + 1e-10)
    
    return solutions, edge_qualities


def _apply_heuristic(heuristic: str, distance_matrix: np.ndarray, n: int) -> list:
    """Apply specific construction heuristic."""
    if n == 0:
        return None
    
    if heuristic == 'nearest_neighbor':
        return _nearest_neighbor(distance_matrix, np.random.randint(n))
    elif heuristic == 'farthest_insertion':
        return _farthest_insertion(distance_matrix, np.random.randint(n))
    elif heuristic == 'cheapest_insertion':
        return _cheapest_insertion(distance_matrix, np.random.randint(n))
    elif heuristic == 'random_insertion':
        return _random_insertion(distance_matrix, np.random.randint(n))
    elif heuristic == 'savings':
        return _savings_heuristic(distance_matrix)
    elif heuristic == 'nearest_insertion':
        return _nearest_insertion(distance_matrix, np.random.randint(n))
    else:
        return _nearest_neighbor(distance_matrix, np.random.randint(n))


def _nearest_neighbor(distance_matrix: np.ndarray, start: int) -> list:
    """Nearest neighbor heuristic."""
    n = distance_matrix.shape[0]
    tour = [start]
    visited = set([start])
    
    for _ in range(n-1):
        current = tour[-1]
        
        # Find nearest unvisited neighbor
        best_dist = float('inf')
        best_node = -1
        
        for node in range(n):
            if node not in visited:
                dist = distance_matrix[current, node]
                if dist < best_dist:
                    best_dist = dist
                    best_node = node
        
        if best_node == -1:
            break
        
        tour.append(best_node)
        visited.add(best_node)
    
    return tour


def _farthest_insertion(distance_matrix: np.ndarray, start: int) -> list:
    """Farthest insertion heuristic."""
    n = distance_matrix.shape[0]
    tour = [start]
    unvisited = set(range(n))
    unvisited.remove(start)
    
    # Add farthest node
    farthest = max(unvisited, key=lambda x: distance_matrix[start, x])
    tour.append(farthest)
    unvisited.remove(farthest)
    
    while unvisited:
        # Find farthest node from tour
        farthest_node = max(unvisited, 
                          key=lambda x: min(distance_matrix[x, t] for t in tour))
        
        # Find best insertion position
        best_pos = -1
        best_increase = float('inf')
        
        for i in range(len(tour)):
            a, b = tour[i], tour[(i+1)%len(tour)]
            increase = (distance_matrix[a, farthest_node] + 
                       distance_matrix[farthest_node, b] - 
                       distance_matrix[a, b])
            
            if increase < best_increase:
                best_increase = increase
                best_pos = i + 1
        
        tour.insert(best_pos, farthest_node)
        unvisited.remove(farthest_node)
    
    return tour


def _cheapest_insertion(distance_matrix: np.ndarray, start: int) -> list:
    """Cheapest insertion heuristic."""
    n = distance_matrix.shape[0]
    tour = [start]
    unvisited = set(range(n))
    unvisited.remove(start)
    
    # Add nearest neighbor
    nearest = min(unvisited, key=lambda x: distance_matrix[start, x])
    tour.append(nearest)
    unvisited.remove(nearest)
    
    while unvisited:
        best_node = -1
        best_pos = -1
        best_increase = float('inf')
        
        for node in unvisited:
            for i in range(len(tour)):
                a, b = tour[i], tour[(i+1)%len(tour)]
                increase = (distance_matrix[a, node] + 
                           distance_matrix[node, b] - 
                           distance_matrix[a, b])
                
                if increase < best_increase:
                    best_increase = increase
                    best_node = node
                    best_pos = i + 1
        
        if best_node == -1:
            break
        
        tour.insert(best_pos, best_node)
        unvisited.remove(best_node)
    
    return tour


def _random_insertion(distance_matrix: np.ndarray, start: int) -> list:
    """Random insertion heuristic."""
    n = distance_matrix.shape[0]
    tour = [start]
    nodes = list(range(n))
    nodes.remove(start)
    random.shuffle(nodes)
    
    for node in nodes:
        # Find best insertion position
        best_pos = -1
        best_increase = float('inf')
        
        for i in range(len(tour)):
            a, b = tour[i], tour[(i+1)%len(tour)]
            increase = (distance_matrix[a, node] + 
                       distance_matrix[node, b] - 
                       distance_matrix[a, b])
            
            if increase < best_increase:
                best_increase = increase
                best_pos = i + 1
        
        tour.insert(best_pos, node)
    
    return tour


def _savings_heuristic(distance_matrix: np.ndarray) -> list:
    """Savings heuristic (Clarke-Wright)."""
    n = distance_matrix.shape[0]
    if n < 3:
        return list(range(n))
    
    # Initialize routes
    routes = [[i] for i in range(n)]
    
    # Compute savings
    savings = []
    for i in range(n):
        for j in range(i+1, n):
            if i != 0 and j != 0:
                s = distance_matrix[i, 0] + distance_matrix[0, j] - distance_matrix[i, j]
                savings.append((s, i, j))
    
    # Sort savings in descending order
    savings.sort(reverse=True, key=lambda x: x[0])
    
    # Merge routes
    for s, i, j in savings:
        # Find routes containing i and j
        route_i = next((r for r in routes if i in r), None)
        route_j = next((r for r in routes if j in r), None)
        
        if route_i is not None and route_j is not None and route_i != route_j:
            # Check if i is at end of route_i and j is at start of route_j
            if route_i[-1] == i and route_j[0] == j:
                routes.remove(route_i)
                routes.remove(route_j)
                routes.append(route_i + route_j)
            # Check if j is at end of route_j and i is at start of route_i
            elif route_j[-1] == j and route_i[0] == i:
                routes.remove(route_i)
                routes.remove(route_j)
                routes.append(route_j + route_i)
    
    # The route containing 0 is the main tour
    main_route = next(r for r in routes if 0 in r)
    # Rotate to start with 0
    start_idx = main_route.index(0)
    return main_route[start_idx:] + main_route[:start_idx]


def _nearest_insertion(distance_matrix: np.ndarray, start: int) -> list:
    """Nearest insertion heuristic."""
    n = distance_matrix.shape[0]
    tour = [start]
    unvisited = set(range(n))
    unvisited.remove(start)
    
    while unvisited:
        # Find nearest unvisited node to any node in tour
        best_dist = float('inf')
        best_node = -1
        best_tour_node = -1
        
        for node in unvisited:
            for tour_node in tour:
                dist = distance_matrix[tour_node, node]
                if dist < best_dist:
                    best_dist = dist
                    best_node = node
                    best_tour_node = tour_node
        
        if best_node == -1:
            break
        
        # Find best insertion position
        best_pos = -1
        best_increase = float('inf')
        
        for i in range(len(tour)):
            a, b = tour[i], tour[(i+1)%len(tour)]
            increase = (distance_matrix[a, best_node] + 
                       distance_matrix[best_node, b] - 
                       distance_matrix[a, b])
            
            if increase < best_increase:
                best_increase = increase
                best_pos = i + 1
        
        tour.insert(best_pos, best_node)
        unvisited.remove(best_node)
    
    return tour


def _tour_length(distance_matrix: np.ndarray, tour: list) -> float:
    """Calculate tour length."""
    n = len(tour)
    length = 0.0
    for i in range(n):
        length += distance_matrix[tour[i], tour[(i+1)%n]]
    return length


def _analyze_structure(distance_matrix: np.ndarray, n: int, 
                      solutions: list) -> dict:
    """Analyze hierarchical structure of the graph."""
    structure = {}
    
    # Compute MST
    if n <= 1000:
        mst = sparse.csgraph.minimum_spanning_tree(distance_matrix)
        mst_edges = set()
        mst_array = mst.toarray()
        for i in range(n):
            for j in range(i+1, n):
                if mst_array[i, j] > 0:
                    mst_edges.add((min(i, j), max(i, j)))
        structure['mst_edges'] = mst_edges
    
    # Identify frequently used edges in good solutions
    good_edges = set()
    if solutions:
        # Take top 30% of solutions
        solutions.sort(key=lambda x: x[1], reverse=True)
        top_solutions = solutions[:max(1, len(solutions) // 3)]
        
        for tour, quality in top_solutions:
            for i in range(len(tour)):
                a, b = tour[i], tour[(i+1)%n]
                good_edges.add((min(a, b), max(a, b)))
        structure['good_edges'] = good_edges
    
    # Estimate cluster structure
    structure['clusters'] = _estimate_clusters(distance_matrix, n)
    
    return structure


def _estimate_clusters(distance_matrix: np.ndarray, n: int) -> list:
    """Estimate cluster structure using fast approximation."""
    if n <= 10:
        return [list(range(n))]
    
    # Simple clustering based on nearest neighbors
    clusters = []
    visited = [False] * n
    
    for i in range(n):
        if not visited[i]:
            cluster = [i]
            visited[i] = True
            
            # Add nearest neighbors
            for _ in range(min(5, n//10)):
                # Find closest unvisited node to any node in cluster
                best_dist = float('inf')
                best_node = -1
                
                for node in range(n):
                    if not visited[node]:
                        for c_node in cluster:
                            dist = distance_matrix[c_node, node]
                            if dist < best_dist:
                                best_dist = dist
                                best_node = node
                
                if best_node != -1:
                    cluster.append(best_node)
                    visited[best_node] = True
            
            clusters.append(cluster)
    
    # Merge small clusters
    merged_clusters = []
    for cluster in clusters:
        if len(cluster) <= 2 and merged_clusters:
            # Merge with closest cluster
            best_dist = float('inf')
            best_idx = -1
            
            for idx, other_cluster in enumerate(merged_clusters):
                # Compute min distance between clusters
                min_dist = min(distance_matrix[i, j] 
                             for i in cluster for j in other_cluster)
                if min_dist < best_dist:
                    best_dist = min_dist
                    best_idx = idx
            
            if best_idx != -1:
                merged_clusters[best_idx].extend(cluster)
            else:
                merged_clusters.append(cluster)
        else:
            merged_clusters.append(cluster)
    
    return merged_clusters


@numba.jit(nopython=True)
def _enhanced_reciprocal(distance_matrix: np.ndarray, n: int, 
                        features: dict) -> np.ndarray:
    """Compute enhanced reciprocal distances with adaptive scaling."""
    result = np.zeros((n, n))
    median_dist = np.median(distance_matrix[distance_matrix > 0])
    
    for i in range(n):
        for j in range(n):
            if i != j:
                d = distance_matrix[i, j]
                
                # Adaptive transformation
                if d < median_dist:
                    # Close distances: strong preference
                    result[i, j] = 1.0 / (d + 1e-10)
                else:
                    # Far distances: softer preference
                    scale = features['cv_dist']
                    result[i, j] = np.exp(-d / (median_dist * (1 + scale)))
    
    # Row normalization
    for i in range(n):
        row_sum = np.sum(result[i])
        if row_sum > 0:
            result[i] = result[i] / row_sum
    
    return result


def _compute_structural_component(distance_matrix: np.ndarray, 
                                 structure: dict, n: int) -> np.ndarray:
    """Compute structural component based on hierarchical analysis."""
    component = np.zeros((n, n))
    
    # Boost MST edges
    if 'mst_edges' in structure:
        for i, j in structure['mst_edges']:
            component[i, j] = component[j, i] = 2.0
    
    # Boost edges within clusters
    if 'clusters' in structure:
        for cluster in structure['clusters']:
            if len(cluster) > 1:
                for i in cluster:
                    for j in cluster:
                        if i != j:
                            # Stronger boost for closer nodes within cluster
                            d = distance_matrix[i, j]
                            median_intra = np.median([distance_matrix[a, b] 
                                                    for a in cluster for b in cluster 
                                                    if a != b])
                            boost = 1.0 + np.exp(-d / (median_intra + 1e-10))
                            component[i, j] = max(component[i, j], boost)
    
    # Boost edges from good solutions
    if 'good_edges' in structure:
        for i, j in structure['good_edges']:
            component[i, j] = component[j, i] = max(component[i, j], 1.5)
    
    # Normalize
    if component.max() > 0:
        component = component / component.max()
    
    return component


def _compute_local_connectivity(distance_matrix: np.ndarray, n: int, 
                               solutions: list) -> np.ndarray:
    """Compute local connectivity component."""
    connectivity = np.zeros((n, n))
    
    if not solutions:
        return connectivity
    
    # For each node, identify its most common neighbors in good solutions
    for i in range(n):
        neighbor_counts = np.zeros(n)
        
        for tour, quality in solutions:
            # Find position of i in tour
            try:
                pos = tour.index(i)
                # Get neighbors in tour
                prev = tour[(pos-1)%len(tour)]
                next_ = tour[(pos+1)%len(tour)]
                
                neighbor_counts[prev] += quality
                neighbor_counts[next_] += quality
            except ValueError:
                continue
        
        # Normalize and assign to connectivity matrix
        if neighbor_counts.max() > 0:
            neighbor_counts = neighbor_counts / neighbor_counts.max()
            connectivity[i] = neighbor_counts
    
    # Symmetrize
    connectivity = (connectivity + connectivity.T) / 2
    np.fill_diagonal(connectivity, 0)
    
    return connectivity


def _learn_fusion_weights(components: dict, features: dict, 
                         solutions: list, distance_matrix: np.ndarray) -> dict:
    """Learn optimal fusion weights for components."""
    # Base weights
    weights = {
        'distance': 0.2,
        'structural': 0.2,
        'sampling': 0.25,
        'neural': 0.2,
        'local': 0.15
    }
    
    # Adjust based on instance characteristics
    if features['euclidean_score'] > 0.8:
        weights['structural'] *= 1.3
        weights['distance'] *= 1.1
        weights['neural'] *= 0.8
    
    if features['cv_dist'] > 1.5:
        weights['neural'] *= 1.3
        weights['sampling'] *= 1.1
        weights['distance'] *= 0.8
    
    if features['size'] > 500:
        weights['sampling'] *= 1.2
        weights['local'] *= 0.8
        weights['neural'] *= 0.9
    
    # Adjust based on solution quality consistency
    if solutions:
        qualities = [q for _, q in solutions]
        if np.std(qualities) / (np.mean(qualities) + 1e-10) < 0.3:
            # Consistent solutions: trust sampling more
            weights['sampling'] *= 1.2
            weights['local'] *= 1.1
        else:
            # Inconsistent solutions: trust structure more
            weights['structural'] *= 1.2
            weights['distance'] *= 1.1
    
    # Normalize
    total = sum(weights.values())
    return {k: v/total for k, v in weights.items()}


def _fuse_components(components: dict, weights: dict, n: int) -> np.ndarray:
    """Fuse components using learned weights."""
    result = np.zeros((n, n))
    
    for name, component in components.items():
        if name in weights:
            # Normalize component
            if component.max() > 0:
                component = component / component.max()
            result += weights[name] * component
    
    # Ensure non-negativity
    result = np.maximum(result, 0)
    
    return result


def _hierarchical_sparsification(heuristic_matrix: np.ndarray,
                                distance_matrix: np.ndarray,
                                structure: dict, n: int,
                                features: dict) -> np.ndarray:
    """
    Hierarchical sparsification preserving global structure.
    Maintains connectivity and important edges at multiple scales.
    """
    sparsified = np.zeros((n, n))
    
    # Determine target density
    if n < 50:
        target_density = 0.8
    elif n < 200:
        target_density = 0.4
    elif n < 1000:
        target_density = 0.2
    else:
        target_density = 0.1
    
    # Adjust based on instance characteristics
    if features['cluster_score'] > 0.7:
        # Clustered instance: can be sparser
        target_density *= 0.8
    if features['cv_dist'] > 1.5:
        # High variation: need more edges
        target_density *= 1.2
    
    # Phase 1: Preserve critical edges
    critical_edges = set()
    
    # MST edges
    if 'mst_edges' in structure:
        critical_edges.update(structure['mst_edges'])
    
    # Good solution edges
    if 'good_edges' in structure:
        critical_edges.update(structure['good_edges'])
    
    # Preserve critical edges
    for i, j in critical_edges:
        if i < n and j < n:  # Bounds check
            sparsified[i, j] = sparsified[j, i] = heuristic_matrix[i, j]
    
    # Phase 2: Add edges hierarchically
    edges_per_node = max(2, int(target_density * n))
    
    for i in range(n):
        # Get candidate edges for this node
        candidates = []
        for j in range(n):
            if i != j and (i, j) not in critical_edges and (j, i) not in critical_edges:
                candidates.append((j, heuristic_matrix[i, j]))
        
        if not candidates:
            continue
        
        # Sort by heuristic value
        candidates.sort(key=lambda x: x[1], reverse=True)
        
        # Select edges with hierarchical consideration
        selected = 0
        for rank, (j, value) in enumerate(candidates):
            if selected >= edges_per_node:
                break
            
            # Hierarchical selection probability
            if rank == 0:
                prob = 1.0
            else:
                # Exponential decay with rank
                decay = np.exp(-rank / 2.0)
                # Adjust by distance
                dist_factor = np.exp(-distance_matrix[i, j] / 
                                   np.median(distance_matrix))
                prob = decay * dist_factor
            
            if random.random() < prob:
                sparsified[i, j] = max(sparsified[i, j], value)
                sparsified[j, i] = max(sparsified[j, i], value)
                selected += 1
    
    # Ensure connectivity
    _ensure_connectivity(sparsified, distance_matrix, n)
    
    return sparsified


def _ensure_connectivity(matrix: np.ndarray, distance_matrix: np.ndarray, n: int):
    """Ensure the sparsified graph remains connected."""
    # Find connected components
    visited = [False] * n
    components = []
    
    for i in range(n):
        if not visited[i]:
            # BFS to find component
            component = []
            queue = [i]
            visited[i] = True
            
            while queue:
                node = queue.pop(0)
                component.append(node)
                
                for neighbor in range(n):
                    if matrix[node, neighbor] > 0 and not visited[neighbor]:
                        visited[neighbor] = True
                        queue.append(neighbor)
            
            components.append(component)
    
    # If disconnected, add minimal connecting edges
    if len(components) > 1:
        # Connect components with shortest edges
        for i in range(len(components)-1):
            comp1 = components[i]
            comp2 = components[i+1]
            
            # Find shortest edge between components
            best_dist = float('inf')
            best_pair = (comp1[0], comp2[0])
            
            for a in comp1:
                for b in comp2:
                    dist = distance_matrix[a, b]
                    if dist < best_dist:
                        best_dist = dist
                        best_pair = (a, b)
            
            # Add this edge
            a, b = best_pair
            matrix[a, b] = matrix[b, a] = 1.0 / (best_dist + 1e-10)


def _refine_heuristics(heuristic_matrix: np.ndarray,
                      distance_matrix: np.ndarray,
                      n: int, features: dict) -> np.ndarray:
    """Final refinement of heuristic matrix."""
    result = heuristic_matrix.copy()
    
    # Ensure non-negativity
    result = np.maximum(result, 0)
    
    # Add reciprocal fallback for zero rows
    row_sums = result.sum(axis=1)
    zero_rows = row_sums == 0
    
    if np.any(zero_rows):
        with np.errstate(divide='ignore', invalid='ignore'):
            fallback = 1 / distance_matrix
            fallback[np.isinf(fallback)] = 0
            fallback = np.nan_to_num(fallback, nan=0)
        
        np.fill_diagonal(fallback, 0)
        result[zero_rows] = fallback[zero_rows]
    
    # Apply smoothing based on instance characteristics
    if n > 100:
        smoothing_factor = min(0.3, 0.1 * features['cv_dist'])
        if smoothing_factor > 0:
            # Create smoothing kernel
            kernel = np.zeros((n, n))
            median_dist = np.median(distance_matrix[distance_matrix > 0])
            
            for i in range(n):
                for j in range(n):
                    if i != j:
                        kernel[i, j] = np.exp(-(distance_matrix[i, j] / 
                                               (median_dist * 2)) ** 2)
            
            # Normalize kernel
            row_sums_kernel = kernel.sum(axis=1, keepdims=True)
            row_sums_kernel[row_sums_kernel == 0] = 1
            kernel = kernel / row_sums_kernel
            
            # Apply smoothing
            result = (1 - smoothing_factor) * result + smoothing_factor * kernel
    
    # Row normalization for probability interpretation
    row_sums = result.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    result = result / row_sums
    
    # Final symmetry
    result = (result + result.T) / 2
    np.fill_diagonal(result, 0)
    
    return result

def _estimate_euclidean_score(distance_matrix: np.ndarray, n: int) -> float:
    """Estimate how Euclidean the distance matrix is."""
    if n < 3:
        return 1.0
    
    # Sample triangle inequalities
    sample_size = min(100, n)
    indices = np.random.choice(n, sample_size, replace=False)
    
    violations = 0
    total_checks = 0
    
    for i in range(sample_size):
        for j in range(i+1, sample_size):
            for k in range(j+1, sample_size):
                a, b, c = indices[i], indices[j], indices[k]
                d_ab = distance_matrix[a, b]
                d_ac = distance_matrix[a, c]
                d_bc = distance_matrix[b, c]
                
                # Check triangle inequality
                violations += int(d_ab > d_ac + d_bc + 1e-6)
                violations += int(d_ac > d_ab + d_bc + 1e-6)
                violations += int(d_bc > d_ab + d_ac + 1e-6)
                total_checks += 3
    
    return 1.0 - violations / max(total_checks, 1)


def _estimate_clusterability(distance_matrix: np.ndarray, n: int) -> float:
    """Estimate how clustered the instance is."""
    if n < 10:
        return 0.5
    
    # Compute ratio of intra-cluster to inter-cluster distances
    # Using simple k-means approximation with k=3
    try:
        # Initialize random centroids
        centroids = np.random.choice(n, 3, replace=False)
        
        # Simple assignment
        assignments = np.zeros(n, dtype=int)
        for i in range(n):
            best_dist = float('inf')
            best_centroid = 0
            for c_idx, centroid in enumerate(centroids):
                dist = distance_matrix[i, centroid]
                if dist < best_dist:
                    best_dist = dist
                    best_centroid = c_idx
            assignments[i] = best_centroid
        
        # Compute intra and inter cluster distances
        intra_dists = []
        inter_dists = []
        
        for i in range(n):
            for j in range(i+1, n):
                if assignments[i] == assignments[j]:
                    intra_dists.append(distance_matrix[i, j])
                else:
                    inter_dists.append(distance_matrix[i, j])
        
        if intra_dists and inter_dists:
            mean_intra = np.mean(intra_dists)
            mean_inter = np.mean(inter_dists)
            return min(1.0, mean_inter / (mean_intra + 1e-10))
    except:
        pass
    
    return 0.5
