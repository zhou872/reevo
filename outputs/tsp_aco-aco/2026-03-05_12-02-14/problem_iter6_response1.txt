            neighbor_avg[i] = 0

            neighbor_avg[i] = 1 / np.min(row[row < np.inf])

        min_val = np.min(neighbor_avg[neighbor_avg > 0])
        max_val = np.max(neighbor_avg)

    base_k = int(np.ceil(np.log2(n + 1) * (1.0 + min(dist_cv, 2.0))))
    k = max(2, min(base_k, n // 2))

    decay_rate = 1.0 + np.log10(n + 1) / 2.0

        if k < n:
            # Get top k edges with efficient numpy operations
            top_k_idx = np.argpartition(row, -k)[-k:]
            top_k_values = row[top_k_idx]
            sorted_idx = top_k_idx[np.argsort(top_k_values)[::-1]]

            all_idx = np.arange(n)
            all_idx = all_idx[all_idx != i]
            sorted_idx = all_idx[np.argsort(row[all_idx])[::-1]]

            decay = np.exp(-rank / decay_rate)
            sparsified[i, j] = combined[i, j] * decay
    
    # 7. Symmetrize and final adjustments
    sparsified = (sparsified + sparsified.T) / 2
    np.fill_diagonal(sparsified, 0)
    
    # Ensure numerical stability and proper scaling
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
