import numpy as np
import scipy.sparse as sparse
from scipy.sparse import csgraph
from scipy.spatial.distance import squareform, pdist

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge sampling with:
    1. Dynamic weight adjustment based on search progress simulation
    2. Local search enhancement via fast 2-opt gain approximation
    3. Multi-resolution centrality analysis
    4. Scalable sparsification with hierarchical edge selection
    
    Returns: Row-normalized matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / np.maximum(distance_matrix, 1e-10)
            heuristic[np.isinf(heuristic)] = 0
            heuristic[np.isnan(heuristic)] = 0
        np.fill_diagonal(heuristic, 0)
        row_sums = heuristic.sum(axis=1, keepdims=True)
        row_sums[row_sums == 0] = 1
        return heuristic / row_sums
    
    # ===== 1. CORE HEURISTIC COMPONENTS WITH ENHANCEMENTS =====
    
    # 1.1 Reciprocal distance with power-law scaling
    with np.errstate(divide='ignore', invalid='ignore'):
        # Use power-law scaling for better discrimination
        reciprocal = 1 / np.maximum(distance_matrix, 1e-10) ** 0.8
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # 1.2 Multi-resolution centrality analysis
    centrality_components = []
    
    # Local centrality (neighborhood-based)
    k_local = min(10, max(3, n // 20))
    local_centrality = np.zeros(n)
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        # Get k_local nearest neighbors
        if k_local < n - 1:
            nn_idx = np.argpartition(distances, k_local)[:k_local]
        else:
            nn_idx = np.arange(n)
            nn_idx = nn_idx[nn_idx != i]
        local_centrality[i] = np.sum(1 / np.maximum(distances[nn_idx], 1e-10))
    
    # Global centrality (all nodes)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / np.maximum(distance_matrix, 1e-10)
        inv_dist[np.eye(n, dtype=bool)] = 0
        global_centrality = np.sum(inv_dist, axis=1)
    
    # Combine local and global centrality with adaptive weights
    centrality_ratio = min(1.0, k_local / (n - 1))
    node_centrality = (centrality_ratio * local_centrality + 
                      (1 - centrality_ratio) * global_centrality)
    
    # Normalize
    if node_centrality.max() > 0:
        node_centrality = node_centrality / node_centrality.max()
    
    # Create centrality matrix with adaptive interaction
    centrality_matrix = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Use geometric mean with bias toward stronger centralities
            ci, cj = node_centrality[i], node_centrality[j]
            centrality_matrix[i, j] = centrality_matrix[j, i] = np.sqrt(ci * cj)
    
    # 1.3 Enhanced MST component with local search integration
    mst_component = np.zeros((n, n))
    
    try:
        # Get minimum spanning tree
        mst = csgraph.minimum_spanning_tree(distance_matrix)
        mst_dense = mst.toarray()
        mst_mask = mst_dense > 0
        
        # Identify critical edges in MST (bridge edges)
        degrees = mst_mask.sum(axis=1)
        
        # Boost edges based on their role in MST
        for i in range(n):
            for j in range(i+1, n):
                if mst_mask[i, j]:
                    # Base MST edge weight
                    mst_component[i, j] = 1.0
                    
                    # Additional boost for edges connecting low-degree vertices
                    if degrees[i] == 1 or degrees[j] == 1:
                        mst_component[i, j] *= 1.3
                    
                    # Boost for edges that are in many alternative MSTs
                    # Approximate by edge contraction cost
                    alt_cost = distance_matrix[i, j] * 1.1
                    if alt_cost < np.percentile(distance_matrix.flatten(), 10):
                        mst_component[i, j] *= 1.2
        
        mst_component = (mst_component + mst_component.T) / 2
        
        # Normalize
        max_val = mst_component.max()
        if max_val > 0:
            mst_component = mst_component / max_val
            
    except Exception:
        mst_component = np.ones((n, n)) * 0.5
        np.fill_diagonal(mst_component, 0)
    
    # 1.4 Fast 2-opt gain approximation
    kopt_component = np.zeros((n, n))
    
    # For each edge (i,j), estimate potential 2-opt gain
    for i in range(n):
        for j in range(i+1, n):
            current_edge = distance_matrix[i, j]
            
            # Find best alternative crossing edge (k,l) that would replace (i,j)
            # Simplified: look at edges incident to i and j's neighbors
            best_gain = 0
            
            # Check neighbors of i
            i_neighbors = np.argsort(distance_matrix[i])[:min(5, n)]
            j_neighbors = np.argsort(distance_matrix[j])[:min(5, n)]
            
            for k in i_neighbors:
                if k == i or k == j:
                    continue
                for l in j_neighbors:
                    if l == i or l == j or l == k:
                        continue
                    
                    # 2-opt gain: (i,j) + (k,l) -> (i,k) + (j,l)
                    gain = current_edge + distance_matrix[k, l] - distance_matrix[i, k] - distance_matrix[j, l]
                    if gain > best_gain:
                        best_gain = gain
            
            # Normalize gain by average edge length
            avg_edge = np.mean(distance_matrix[i])
            if avg_edge > 0:
                kopt_component[i, j] = kopt_component[j, i] = min(1.0, best_gain / (avg_edge * 2))
    
    # ===== 2. DYNAMIC WEIGHT ADJUSTMENT =====
    
    # Analyze instance characteristics
    flat_dists = distance_matrix[~np.eye(n, dtype=bool)]
    dist_mean = np.mean(flat_dists)
    dist_std = np.std(flat_dists)
    dist_cv = dist_std / max(dist_mean, 1e-10)
    
    # Simulated search progress adjustment
    # Early phase: explore diverse edges
    # Middle phase: focus on promising edges
    # Late phase: intensify on best candidates
    
    # We'll simulate early phase weights (can be adjusted based on actual search progress)
    search_phase = 0.3  # 0=early, 1=late
    
    # Base weights by problem size
    if n < 20:
        base_weights = {
            'recip': 0.4,
            'cent': 0.2,
            'mst': 0.3,
            'kopt': 0.1
        }
    elif n < 100:
        base_weights = {
            'recip': 0.3,
            'cent': 0.3,
            'mst': 0.3,
            'kopt': 0.1
        }
    else:
        base_weights = {
            'recip': 0.2,
            'cent': 0.4,
            'mst': 0.3,
            'kopt': 0.1
        }
    
    # Adjust based on search phase
    if search_phase < 0.33:  # Early phase
        base_weights['cent'] *= 0.8  # Less emphasis on global structure
        base_weights['recip'] *= 1.2  # More emphasis on local information
    elif search_phase > 0.66:  # Late phase
        base_weights['kopt'] *= 1.5  # More emphasis on local improvements
        base_weights['mst'] *= 1.2  # Emphasize structural edges
    
    # Adjust based on distance variability
    if dist_cv > 1.0:
        base_weights['recip'] *= 1.3
        base_weights['cent'] *= 0.7
    elif dist_cv < 0.3:
        base_weights['cent'] *= 1.3
        base_weights['recip'] *= 0.7
    
    # Normalize weights
    total = sum(base_weights.values())
    for key in base_weights:
        base_weights[key] /= total
    
    # Combine components
    combined = (base_weights['recip'] * reciprocal +
                base_weights['cent'] * centrality_matrix +
                base_weights['mst'] * mst_component +
                base_weights['kopt'] * kopt_component)
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    np.fill_diagonal(combined, 0)
    
    # ===== 3. SCALABLE SPARSIFICATION =====
    
    # Hierarchical edge selection
    sparsified = np.zeros_like(combined)
    
    # Determine edge retention strategy based on problem size
    if n < 50:
        # Keep all edges for small problems
        sparsified = combined.copy()
    else:
        # Two-level hierarchy: global promising edges + local promising edges
        global_threshold = np.percentile(combined[combined > 0], 70)
        local_thresholds = np.zeros(n)
        
        for i in range(n):
            row = combined[i].copy()
            row[i] = -np.inf
            local_thresholds[i] = np.percentile(row[row > -np.inf], 60)
        
        # Select edges
        for i in range(n):
            for j in range(i+1, n):
                value = combined[i, j]
                
                # Keep if above global threshold or local threshold for either endpoint
                if (value > global_threshold or 
                    value > local_thresholds[i] or 
                    value > local_thresholds[j]):
                    
                    # Apply decay based on rank within node's edges
                    row_i = combined[i].copy()
                    row_i[i] = -np.inf
                    rank_i = np.sum(row_i > value)
                    
                    row_j = combined[j].copy()
                    row_j[j] = -np.inf
                    rank_j = np.sum(row_j > value)
                    
                    avg_rank = (rank_i + rank_j) / 2
                    decay = np.exp(-avg_rank / 5.0)
                    
                    sparsified[i, j] = sparsified[j, i] = value * decay
    
    # ===== 4. POST-PROCESSING =====
    
    # Ensure symmetry and non-negativity
    result = np.maximum(sparsified, sparsified.T)
    result = np.maximum(result, 0)
    np.fill_diagonal(result, 0)
    
    # Add small deterministic exploration factor
    # Based on edge uniqueness (distance rank)
    exploration = np.zeros((n, n))
    for i in range(n):
        # Get distance-based rank for each edge
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        ranks = np.argsort(np.argsort(distances))
        
        for j in range(n):
            if i != j:
                # Edges with medium distance get more exploration
                rank_ratio = ranks[j] / (n - 1)
                exploration[i, j] = 0.02 * (1 - np.abs(rank_ratio - 0.5))
    
    exploration = (exploration + exploration.T) / 2
    np.fill_diagonal(exploration, 0)
    result = result + exploration
    
    # Row normalization with robust handling
    row_sums = result.sum(axis=1, keepdims=True)
    zero_mask = row_sums.flatten() == 0
    
    if np.any(zero_mask):
        # Use fallback for isolated nodes
        fallback = reciprocal[zero_mask]
        fallback_sum = fallback.sum(axis=1, keepdims=True)
        fallback_sum[fallback_sum == 0] = 1
        result[zero_mask] = fallback / fallback_sum
        row_sums[zero_mask] = 1
    
    row_sums[row_sums == 0] = 1
    result = result / row_sums
    
    return result
