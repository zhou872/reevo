import numpy as np
from scipy import sparse

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection with improved features:
    1. Reciprocal distance (inverse of edge weights)
    2. Node closeness centrality (nodes closer to all others are more central)
    3. Adaptive sparsification with stochastic exponential decay
    """
    n = distance_matrix.shape[0]
    
    # 1. Reciprocal distance (handle zeros/division by zero)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # 2. Node closeness centrality (using row sums)
    # Nodes with smaller total distance to others are more central
    row_sums = np.sum(distance_matrix, axis=1)
    # Avoid division by zero for isolated nodes (shouldn't happen in connected TSP)
    closeness = 1 / np.maximum(row_sums, 1e-10)
    
    # Normalize centrality to [0, 1]
    if np.max(closeness) > 0:
        closeness = closeness / np.max(closeness)
    
    # Edge promise based on centrality product of endpoints
    centrality_matrix = np.outer(closeness, closeness)
    
    # 3. Combine factors with tuned weights
    w1, w2 = 0.6, 0.4  # weights for reciprocal and centrality
    combined = w1 * reciprocal + w2 * centrality_matrix
    
    # 4. Adaptive sparsification with stochastic decay
    sparsified = np.zeros_like(combined)
    
    # Determine sparsity level adaptively
    k = max(3, min(10, n // 4))
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self-loop
        
        if k < n:
            # Get indices of top k edges for this node
            top_k_idx = np.argpartition(row, -k)[-k:]
        else:
            top_k_idx = np.arange(n)
            top_k_idx = top_k_idx[top_k_idx != i]
        
        # Sort top edges by their promise value
        sorted_indices = top_k_idx[np.argsort(row[top_k_idx])[::-1]]
        
        # Apply exponential decay based on rank
        for rank, j in enumerate(sorted_indices):
            decay = np.exp(-rank / 2.0)  # Exponential decay
            sparsified[i, j] = combined[i, j] * decay
    
    # Ensure symmetry for undirected TSP
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Zero out diagonal
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
