import numpy as np
from collections import defaultdict, deque
from scipy.sparse import csgraph
import warnings

warnings.filterwarnings('ignore')

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection with iterative refinement,
    adaptive sparsification, and vectorized operations.
    
    Core innovations:
    1. Iterative refinement with memory: 2-opt improved tours accumulate edge importance
    2. Adaptive sparsity: Exponential decay on edge ranks with MST preservation
    3. Vectorized operations: Efficient row-wise processing for scalability
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape indicating edge promise (higher = better)
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic[np.isinf(heuristic)] = 0
            heuristic = np.nan_to_num(heuristic, nan=0)
        np.fill_diagonal(heuristic, 0)
        return heuristic
    
    # ========== Phase 1: Generate Initial Tours ==========
    
    # Generate diverse starting points
    starts = np.random.choice(n, min(10, n), replace=False)
    nn_tours = []
    
    for start in starts:
        tour = [start]
        visited = np.zeros(n, dtype=bool)
        visited[start] = True
        
        current = start
        for _ in range(n-1):
            # Find nearest unvisited neighbor with tie-breaking
            distances = distance_matrix[current].copy()
            distances[visited] = np.inf
            distances += np.random.rand(n) * 1e-8
            next_node = np.argmin(distances)
            tour.append(next_node)
            visited[next_node] = True
            current = next_node
        
        nn_tours.append(tour)
    
    # ========== Phase 2: Iterative Refinement with Memory ==========
    
    edge_memory = defaultdict(float)
    recent_tours = deque(maxlen=5)
    
    def two_opt_improvement(tour, max_iterations=50):
        """Apply 2-opt local search with early stopping."""
        n_nodes = len(tour)
        improved = True
        iterations = 0
        
        while improved and iterations < max_iterations:
            improved = False
            for i in range(n_nodes):
                for j in range(i+2, n_nodes):
                    # Calculate gain
                    a1, a2 = tour[i], tour[(i+1)%n_nodes]
                    b1, b2 = tour[j], tour[(j+1)%n_nodes]
                    
                    gain = (distance_matrix[a1, a2] + 
                           distance_matrix[b1, b2]) - \
                          (distance_matrix[a1, b1] + 
                           distance_matrix[a2, b2])
                    
                    if gain > 1e-6:
                        # Reverse segment between i+1 and j
                        tour[i+1:j+1] = tour[i+1:j+1][::-1]
                        improved = True
                        break
                if improved:
                    break
            iterations += 1
        return tour
    
    # Improve tours and accumulate edge knowledge
    for tour in nn_tours:
        if len(tour) == n:  # Ensure complete tour
            improved_tour = two_opt_improvement(tour.copy())
            recent_tours.append(improved_tour)
            
            # Calculate tour length
            tour_length = sum(distance_matrix[improved_tour[i], 
                             improved_tour[(i+1)%n]] for i in range(n))
            weight = 1.0 / (tour_length + 1e-10)
            
            # Add edges to memory
            for i in range(n):
                u, v = improved_tour[i], improved_tour[(i+1)%n]
                edge_key = (min(u, v), max(u, v))
                edge_memory[edge_key] += weight
    
    # ========== Phase 3: Compute Structural Components ==========
    
    # 3A. Minimum Spanning Tree edges
    mst = csgraph.minimum_spanning_tree(distance_matrix)
    mst_mask = mst.toarray() > 0
    
    # 3B. Perfect matching on odd-degree vertices (approximated)
    degrees = mst_mask.sum(axis=1)
    odd_vertices = np.where(degrees % 2 == 1)[0]
    matching_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    if len(odd_vertices) >= 2:
        # Greedy minimum matching
        remaining = set(odd_vertices)
        sorted_pairs = []
        
        for idx, i in enumerate(odd_vertices):
            for j in odd_vertices[idx+1:]:
                sorted_pairs.append((distance_matrix[i, j], i, j))
        
        sorted_pairs.sort()
        matched = set()
        
        for _, i, j in sorted_pairs:
            if i in remaining and j in remaining:
                matching_mask[i, j] = matching_mask[j, i] = True
                matched.update([i, j])
                remaining.difference_update([i, j])
            if len(remaining) < 2:
                break
    
    # ========== Phase 4: Construct Raw Heuristic Matrix ==========
    
    # 4A. Memory component
    memory_matrix = np.zeros_like(distance_matrix)
    for (u, v), weight in edge_memory.items():
        memory_matrix[u, v] = memory_matrix[v, u] = weight
    
    # 4B. MST component
    mst_matrix = np.zeros_like(distance_matrix)
    mst_matrix[mst_mask] = 1.0 / (distance_matrix[mst_mask] + 1e-10)
    
    # 4C. Matching component
    matching_matrix = np.zeros_like(distance_matrix)
    matching_matrix[matching_mask] = 1.0 / (distance_matrix[matching_mask] + 1e-10)
    
    # 4D. Reciprocal distance baseline
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal = np.nan_to_num(reciprocal, nan=0)
    
    # Normalize components
    components = [mst_matrix, matching_matrix, memory_matrix, reciprocal]
    normalized = []
    for comp in components:
        if comp.max() > 0:
            normalized.append(comp / comp.max())
        else:
            normalized.append(comp.copy())
    
    # Dynamic weighting based on instance characteristics
    triu_distances = distance_matrix[np.triu_indices(n, k=1)]
    dist_cv = np.std(triu_distances) / (np.mean(triu_distances) + 1e-10)
    
    if dist_cv < 0.3:
        weights = [0.4, 0.3, 0.2, 0.1]  # Emphasize MST and matching
    elif dist_cv > 0.7:
        weights = [0.2, 0.1, 0.4, 0.3]  # Emphasize memory and reciprocal
    else:
        weights = [0.3, 0.25, 0.25, 0.2]  # Balanced
    
    # Combine components
    raw_heuristic = np.zeros_like(distance_matrix)
    for comp, w in zip(normalized, weights):
        raw_heuristic += w * comp
    
    # Ensure symmetry and zero diagonal
    raw_heuristic = (raw_heuristic + raw_heuristic.T) / 2
    np.fill_diagonal(raw_heuristic, 0)
    
    # ========== Phase 5: Adaptive Sparsification ==========
    
    # Determine adaptive k (edges per node)
    if n < 50:
        k = n - 1
    elif n < 200:
        k = int(2 * np.sqrt(n))
    else:
        k = int(1.5 * np.sqrt(n))
    
    # Identify preserved edges (MST and top memory edges)
    preserved = set()
    
    # MST edges
    mst_edges = np.column_stack(np.where(mst_mask))
    for i, j in mst_edges:
        if i != j:
            preserved.add((min(i, j), max(i, j)))
    
    # Top memory edges
    if edge_memory:
        sorted_edges = sorted(edge_memory.items(), key=lambda x: x[1], reverse=True)
        top_count = min(len(sorted_edges), max(10, n // 2))
        for (u, v), _ in sorted_edges[:top_count]:
            preserved.add((u, v))
    
    # Initialize sparsified matrix
    sparsified = np.zeros_like(raw_heuristic)
    
    # Preserve important edges at full strength
    for u, v in preserved:
        sparsified[u, v] = sparsified[v, u] = raw_heuristic[u, v]
    
    # Vectorized row-wise processing with exponential decay
    for i in range(n):
        # Get row values and indices
        row_values = raw_heuristic[i].copy()
        row_values[i] = -np.inf  # Exclude diagonal
        
        # Create mask for non-preserved edges
        non_preserved_mask = np.ones(n, dtype=bool)
        for j in range(n):
            if (min(i, j), max(i, j)) in preserved:
                non_preserved_mask[j] = False
        
        if not np.any(non_preserved_mask):
            continue
            
        # Get indices and values of non-preserved edges
        candidate_indices = np.where(non_preserved_mask)[0]
        candidate_values = row_values[candidate_indices]
        
        if len(candidate_values) == 0:
            continue
        
        # Sort candidates by value (descending)
        sort_order = np.argsort(-candidate_values)
        sorted_indices = candidate_indices[sort_order]
        sorted_values = candidate_values[sort_order]
        
        # Determine how many edges we need to select
        current_preserved = sum(1 for j in range(n) if (min(i, j), max(i, j)) in preserved)
        needed = max(0, k - current_preserved)
        
        # Apply exponential decay to selected edges
        selected_indices = sorted_indices[:needed]
        ranks = np.arange(len(selected_indices))
        decay_factors = np.exp(-ranks / 3.0)  # Exponential decay
        
        for idx, j in enumerate(selected_indices):
            sparsified[i, j] = sorted_values[idx] * decay_factors[idx]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    # Final normalization
    max_val = sparsified.max()
    if max_val > 0:
        sparsified /= max_val
    
    return sparsified
