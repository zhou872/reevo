import numpy as np
from scipy import sparse

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic for TSP edge sampling with:
    1. Simplified but effective heuristic combination (Reciprocal, Centrality, MST, Nearest-Neighbor)
    2. Adaptive weighting based on problem size and instance characteristics
    3. Smart sparsification with instance-aware edge selection
    4. Enhanced Christofides-inspired MST component
    
    Returns: Row-normalized matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic[np.isinf(heuristic)] = 0
            heuristic[np.isnan(heuristic)] = 0
        np.fill_diagonal(heuristic, 0)
        row_sums = heuristic.sum(axis=1, keepdims=True)
        row_sums[row_sums == 0] = 1
        return heuristic / row_sums
    
    # ===== 1. CORE HEURISTIC COMPONENTS =====
    
    # 1.1 Reciprocal distance (inverse distance)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # 1.2 Enhanced closeness centrality with harmonic mean
    # Use harmonic centrality which works better with distances
    harmonic_centrality = np.zeros(n)
    for i in range(n):
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        # Harmonic centrality: sum of 1/distance to all other nodes
        harmonic_centrality[i] = np.sum(1 / np.maximum(distance_matrix[i, mask], 1e-10))
    
    # Normalize centrality
    if harmonic_centrality.max() > 0:
        harmonic_centrality = harmonic_centrality / harmonic_centrality.max()
    
    # Create centrality matrix using geometric mean of centralities
    centrality_matrix = np.sqrt(np.outer(harmonic_centrality, harmonic_centrality))
    
    # 1.3 Enhanced MST component inspired by Christofides algorithm
    mst_component = np.zeros((n, n))
    try:
        # Get minimum spanning tree
        mst = sparse.csgraph.minimum_spanning_tree(distance_matrix)
        mst_dense = mst.toarray()
        mst_mask = mst_dense > 0
        
        # Identify odd-degree vertices in MST (for Christofides-like enhancement)
        degrees = mst_mask.sum(axis=1)
        odd_vertices = np.where(degrees % 2 == 1)[0]
        
        # Boost edges between odd-degree vertices (potential matching edges)
        if len(odd_vertices) > 0:
            odd_pairs = np.ix_(odd_vertices, odd_vertices)
            mst_component[odd_pairs] += 0.5
        
        # Boost MST edges themselves
        mst_component[mst_mask] += 1.0
        
        # Normalize MST component
        if mst_component.max() > 0:
            mst_component = mst_component / mst_component.max()
            
    except Exception:
        # Fallback to simple MST indicator if calculation fails
        mst_component = np.eye(n) * 0.5
        np.fill_diagonal(mst_component, 0)
    
    # 1.4 Nearest neighbor heuristic with adaptive k
    nn_heuristic = np.zeros((n, n))
    # Adaptive k based on problem size
    k_nn = min(max(3, int(np.sqrt(n))), 10)
    
    for i in range(n):
        # Get distances from node i, excluding self
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Find k_nn nearest neighbors
        if k_nn < n - 1:
            nn_indices = np.argpartition(distances, k_nn)[:k_nn]
        else:
            nn_indices = np.arange(n)
            nn_indices = nn_indices[nn_indices != i]
        
        if len(nn_indices) > 0:
            # Sort by distance and apply exponential decay
            sorted_indices = nn_indices[np.argsort(distances[nn_indices])]
            for rank, j in enumerate(sorted_indices):
                decay = np.exp(-rank / 2.0)
                nn_heuristic[i, j] = decay
    
    # Symmetrize nearest neighbor heuristic
    nn_heuristic = np.maximum(nn_heuristic, nn_heuristic.T)
    
    # ===== 2. ADAPTIVE WEIGHTING =====
    
    # Analyze instance characteristics
    flat_dists = distance_matrix[~np.eye(n, dtype=bool)]
    dist_mean = np.mean(flat_dists)
    dist_std = np.std(flat_dists)
    dist_cv = dist_std / max(dist_mean, 1e-10)
    
    # Base weights based on problem size
    if n < 20:
        # Small instances: emphasize local structure
        w_recip = 0.4
        w_cent = 0.2
        w_mst = 0.3
        w_nn = 0.1
    elif n < 100:
        # Medium instances: balanced approach
        w_recip = 0.3
        w_cent = 0.3
        w_mst = 0.3
        w_nn = 0.1
    else:
        # Large instances: emphasize global structure
        w_recip = 0.2
        w_cent = 0.4
        w_mst = 0.3
        w_nn = 0.1
    
    # Adjust weights based on distance variability
    if dist_cv > 1.0:
        # High variability: emphasize reciprocal distance more
        w_recip *= 1.3
        w_cent *= 0.8
    elif dist_cv < 0.3:
        # Low variability: emphasize centrality more
        w_cent *= 1.2
        w_recip *= 0.9
    
    # Normalize weights
    total = w_recip + w_cent + w_mst + w_nn
    w_recip /= total
    w_cent /= total
    w_mst /= total
    w_nn /= total
    
    # Combine components
    combined = (w_recip * reciprocal +
                w_cent * centrality_matrix +
                w_mst * mst_component +
                w_nn * nn_heuristic)
    
    # Ensure symmetry and zero diagonal
    combined = (combined + combined.T) / 2
    np.fill_diagonal(combined, 0)
    
    # ===== 3. ADAPTIVE SPARSIFICATION =====
    
    sparsified = np.zeros_like(combined)
    
    # Determine number of edges to keep per node adaptively
    # Keep more edges for smaller graphs, fewer for larger ones
    if n < 30:
        k_keep = min(10, n - 1)
    elif n < 100:
        k_keep = min(8, n - 1)
    elif n < 500:
        k_keep = min(6, n - 1)
    else:
        k_keep = min(5, n - 1)
    
    # Adjust based on distance variability
    if dist_cv > 1.5:
        k_keep = min(k_keep + 2, n - 1)  # Keep more edges for highly variable instances
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        
        if k_keep < n - 1:
            # Get indices of top k_keep edges
            top_indices = np.argpartition(row, -k_keep)[-k_keep:]
        else:
            top_indices = np.arange(n)
            top_indices = top_indices[top_indices != i]
        
        if len(top_indices) > 0:
            # Sort in descending order
            sorted_values = row[top_indices]
            sort_order = np.argsort(sorted_values)[::-1]
            sorted_idx = top_indices[sort_order]
            
            # Apply smoother exponential decay
            for rank, j in enumerate(sorted_idx):
                decay = np.exp(-rank / 3.0)  # Slower decay for better coverage
                sparsified[i, j] = combined[i, j] * decay
    
    # ===== 4. POST-PROCESSING =====
    
    # Ensure symmetry
    result = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(result, 0)
    
    # Add small exploration factor for diversity
    # Scale exploration with problem size and distance variability
    exploration_factor = 0.01 * (1 + min(1.0, dist_cv))
    exploration_noise = exploration_factor * np.random.rand(n, n)
    exploration_noise = (exploration_noise + exploration_noise.T) / 2
    np.fill_diagonal(exploration_noise, 0)
    
    result = result + exploration_noise
    result = np.maximum(result, 0)
    
    # Row normalization with fallback
    row_sums = result.sum(axis=1, keepdims=True)
    zero_rows = row_sums.flatten() == 0
    
    if np.any(zero_rows):
        # Fallback to simple reciprocal for isolated rows
        fallback = reciprocal[zero_rows]
        row_sums[zero_rows] = fallback.sum(axis=1, keepdims=True)
        result[zero_rows] = fallback
    
    row_sums[row_sums == 0] = 1
    result = result / row_sums
    
    return result
