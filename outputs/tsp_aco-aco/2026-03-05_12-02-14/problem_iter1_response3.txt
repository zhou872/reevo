import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Compute heuristic indicators for TSP edges combining multiple factors:
    1. Reciprocal of distance (closer nodes are more promising)
    2. Minimum spanning tree edges (structural importance)
    3. Sparsification based on nearest neighbors
    """
    n = distance_matrix.shape[0]
    
    # 1. Base heuristic: reciprocal of distance
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # 2. Minimum spanning tree heuristic
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_binary = (mst > 0).astype(float)
    mst_heuristic = reciprocal * mst_binary * 2.0  # Boost MST edges
    
    # 3. Nearest neighbor sparsification
    # For each node, keep connections to k nearest neighbors
    k = max(2, min(5, n // 4))  # Adaptive k based on problem size
    
    # Create mask for nearest neighbors
    neighbor_mask = np.zeros_like(distance_matrix)
    for i in range(n):
        # Get indices of k+1 smallest distances (including self)
        indices = np.argsort(distance_matrix[i])[:k+1]
        # Remove self from neighbors
        indices = indices[indices != i]
        neighbor_mask[i, indices] = 1
    
    # Make symmetric (if i is in j's neighbors or vice versa)
    neighbor_mask = np.maximum(neighbor_mask, neighbor_mask.T)
    
    # 4. Combine all factors
    heuristic_matrix = reciprocal.copy()
    
    # Boost MST edges
    heuristic_matrix += mst_heuristic
    
    # Apply sparsification: zero out non-promising edges
    heuristic_matrix *= neighbor_mask
    
    # 5. Additional smoothing: consider 2-hop connections
    # Nodes that are connected through a common neighbor
    two_hop = neighbor_mask @ neighbor_mask
    two_hop = np.minimum(two_hop, 1)  # Clamp to 1
    two_hop[np.arange(n), np.arange(n)] = 0  # Remove self-loops
    
    # Add small weight for 2-hop connections
    heuristic_matrix += 0.3 * reciprocal * two_hop
    
    # 6. Normalize by node degree (popular nodes get discounted)
    row_sums = heuristic_matrix.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1  # Avoid division by zero
    heuristic_matrix = heuristic_matrix / row_sums
    
    # 7. Ensure symmetry for undirected TSP
    heuristic_matrix = (heuristic_matrix + heuristic_matrix.T) / 2
    
    # Set diagonal to zero
    np.fill_diagonal(heuristic_matrix, 0)
    
    return heuristic_matrix
