def compute_advanced_graph_metrics(distance_matrix: np.ndarray, n: int) -> Dict:
    """Compute comprehensive graph metrics with spectral analysis."""
    # Extract distances
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    # Basic statistics
    metrics = {
        'n': n,
        'dist_mean': np.mean(distances),
        'dist_std': np.std(distances),
        'dist_cv': np.std(distances) / (np.mean(distances) + 1e-10),
        'dist_skew': stats.skew(distances) if len(distances) > 2 else 0,
        'dist_kurtosis': stats.kurtosis(distances) if len(distances) > 2 else 0,
    }
    
    # Euclidean characteristics
    metrics['euclidean_score'] = spectral_euclidean_analysis(distance_matrix, n)
    
    # Graph structure metrics
    metrics['spectral_gap'] = compute_spectral_gap(distance_matrix, n)
    metrics['clustering_coeff'] = estimate_global_clustering(distance_matrix, n)
    
    # Minimum Spanning Tree analysis
    try:
        mst = sparse.csgraph.minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = np.where(mst > 0)
        mst_lengths = mst[mst_edges]
        metrics['mst_length'] = np.sum(mst_lengths)
        metrics['mst_edge_variance'] = np.var(mst_lengths)
        
        # MST topology analysis
        degrees = (mst > 0).sum(axis=1)
        metrics['mst_max_degree'] = np.max(degrees)
        metrics['mst_degree_entropy'] = stats.entropy(np.bincount(degrees.astype(int)) + 1)
    except Exception:
        metrics['mst_length'] = metrics['dist_mean'] * (n - 1)
        metrics['mst_edge_variance'] = metrics['dist_std'] ** 2
    
    # Distance distribution analysis
    metrics['distance_entropy'] = compute_distance_entropy(distances)
    metrics['distance_clusters'] = estimate_distance_clusters(distances)
    
    return metrics

def spectral_euclidean_analysis(distance_matrix: np.ndarray, n: int) -> float:
    """Enhanced Euclidean analysis using spectral methods."""
    if n < 4:
        return 1.0
    
    # Check triangle inequality violations
    violations = 0
    total_checks = 0
    
    # Sample-based checking with vectorization
    sample_size = min(200, n)
    indices = np.random.choice(n, sample_size, replace=False)
    
    # Vectorized triangle inequality check
    for i in range(sample_size):
        for j in range(i + 1, sample_size):
            for k in range(j + 1, sample_size):
                a, b, c = indices[i], indices[j], indices[k]
                d_ab = distance_matrix[a, b]
                d_ac = distance_matrix[a, c]
                d_bc = distance_matrix[b, c]
                
                # Check with tolerance
                violations += int(d_ab > d_ac + d_bc + 1e-6 * d_ab)
                violations += int(d_ac > d_ab + d_bc + 1e-6 * d_ac)
                violations += int(d_bc > d_ab + d_ac + 1e-6 * d_bc)
                total_checks += 3
    
    triangle_score = 1.0 - violations / max(total_checks, 1)
    
    # Additional check: rank of distance matrix
    if n <= 100:  # Eigenvalue computation for moderate sizes
        try:
            # Center the distance matrix
            J = np.eye(n) - np.ones((n, n)) / n
            B = -0.5 * J @ (distance_matrix ** 2) @ J
            
            # Check positive semi-definiteness (for Euclidean)
            eigenvalues = np.linalg.eigvalsh(B)
            positive_eigvals = np.sum(eigenvalues > -1e-10)
            rank_score = positive_eigvals / n
        except:
            rank_score = 0.5
    else:
        rank_score = 0.5
    
    return 0.7 * triangle_score + 0.3 * rank_score

def compute_spectral_gap(distance_matrix: np.ndarray, n: int) -> float:
    """Compute spectral gap of normalized Laplacian."""
    if n < 3:
        return 0.0
    
    try:
        # Convert to similarity matrix
        max_dist = np.max(distance_matrix)
        similarity = 1 - distance_matrix / (max_dist + 1e-10)
        np.fill_diagonal(similarity, 0)
        
        # Compute normalized Laplacian
        D = np.diag(np.sum(similarity, axis=1))
        D_inv_sqrt = np.diag(1.0 / np.sqrt(np.diag(D) + 1e-10))
        L = np.eye(n) - D_inv_sqrt @ similarity @ D_inv_sqrt
        
        # Compute eigenvalues
        eigenvalues = np.linalg.eigvalsh(L)
        eigenvalues = np.sort(eigenvalues)
        
        # Spectral gap is the second smallest eigenvalue
        spectral_gap = eigenvalues[1] if len(eigenvalues) > 1 else 0
        
        return float(spectral_gap)
    except Exception:
        return 0.0

def estimate_global_clustering(distance_matrix: np.ndarray, n: int) -> float:
    """Estimate global clustering coefficient."""
    if n < 3:
        return 0.5
    
    # Threshold for neighborhood definition
    threshold = np.median(distance_matrix[np.triu_indices(n, k=1)])
    
    # Build adjacency matrix
    adjacency = (distance_matrix <= threshold).astype(float)
    np.fill_diagonal(adjacency, 0)
    
    # Compute triangles and triplets
    A_sq = adjacency @ adjacency
    triangles = np.trace(A_sq @ adjacency) / 6
    triplets = np.sum(A_sq) / 2
    
    if triplets > 0:
        clustering = 3 * triangles / triplets
    else:
        clustering = 0.0
    
    return float(clustering)

def compute_distance_entropy(distances: np.ndarray) -> float:
    """Compute entropy of distance distribution."""
    if len(distances) < 2:
        return 0.0
    
    # Discretize distances into bins
    hist, _ = np.histogram(distances, bins=min(20, len(distances)))
    hist = hist[hist > 0]
    probs = hist / hist.sum()
    
    return float(-np.sum(probs * np.log(probs)))

def estimate_distance_clusters(distances: np.ndarray) -> int:
    """Estimate number of distance clusters using gap statistic."""
    if len(distances) < 10:
        return 1
    
    # Use simple histogram-based approach
    hist, bin_edges = np.histogram(distances, bins='auto')
    
    # Find valleys in histogram
    from scipy.signal import find_peaks
    peaks, _ = find_peaks(hist)
    valleys, _ = find_peaks(-hist)
    
    return min(max(len(peaks), 1), 5)

def contextual_bandit_optimization(metrics: Dict, config: Dict, n: int) -> Dict:
    """Contextual bandit with Thompson sampling for parameter optimization."""
    # Define parameter arms
    arms = {
        # Component weights
        'w_structural': np.random.beta(2, 2),
        'w_spectral': np.random.beta(1, 3) if config['use_spectral'] else 0,
        'w_geometric': np.random.beta(3, 1) if config['use_geometric'] else 0,
        'w_topological': np.random.beta(2, 2) if config['use_topological'] else 0,
        
        # Exploration parameters
        'exploration_rate': np.random.beta(1, 4),
        'temperature': np.random.gamma(2, 0.5),
        
        # Sparsification parameters
        'sparsity_factor': np.random.beta(3, 3),
        'decay_rate': np.random.gamma(2, 0.3),
    }
    
    # Adjust based on context
    context_features = [
        metrics['euclidean_score'],
        metrics['dist_cv'],
        metrics['clustering_coeff'],
        n / 1000.0,
    ]
    
    # Thompson sampling simulation
    for _ in range(config['exploration_budget']):
        # Sample from posterior (simplified)
        sampled_weights = {}
        for key, value in arms.items():
            if 'w_' in key:
                # Beta posterior approximation
                alpha = 1 + int(value * 10)
                beta = 1 + int((1 - value) * 10)
                sampled_weights[key] = np.random.beta(alpha, beta)
        
        # Evaluate (simulated reward based on metrics)
        reward = evaluate_parameter_set(sampled_weights, context_features, metrics)
        
        # Update (simplified)
        if reward > 0.5:
            for key in sampled_weights:
                arms[key] = 0.7 * arms[key] + 0.3 * sampled_weights[key]
    
    # Normalize weights
    weight_keys = [k for k in arms if k.startswith('w_')]
    total_weight = sum(arms[k] for k in weight_keys)
    if total_weight > 0:
        for k in weight_keys:
            arms[k] /= total_weight
    
    return arms

def extract_structural_features(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Extract structural features based on MST and centrality."""
    # MST component
    try:
        mst = sparse.csgraph.minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0).astype(float)
    except Exception:
        mst_mask = np.zeros((n, n))
    
    # Centrality measures
    closeness = 1 / (distance_matrix.sum(axis=1) + 1e-10)
    betweenness = estimate_betweenness_centrality(distance_matrix, n)
    
    # Combine
    centrality = (closeness + betweenness) / 2
    structural = mst_mask * np.outer(centrality, centrality)
    
    return structural

def estimate_betweenness_centrality(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Estimate betweenness centrality using sampling."""
    if n < 3:
        return np.zeros(n)
    
    betweenness = np.zeros(n)
    samples = min(100, n * (n - 1) // 2)
    
    for _ in range(samples):
        i, j = np.random.choice(n, 2, replace=False)
        if distance_matrix[i, j] < np.inf:
            # Simple approximation: nodes on shortest path get credit
            path = approximate_shortest_path(distance_matrix, i, j)
            for node in path[1:-1]:
                betweenness[node] += 1
    
    return betweenness / (samples + 1e-10)

def approximate_shortest_path(dist_matrix: np.ndarray, start: int, end: int) -> List[int]:
    """Approximate shortest path using greedy approach."""
    n = dist_matrix.shape[0]
    current = start
    path = [start]
    visited = set([start])
    
    while current != end and len(path) < n:
        # Find unvisited neighbor with smallest distance to end
        unvisited = [j for j in range(n) if j not in visited]
        if not unvisited:
            break
        
        # Greedy choice: minimize distance to end
        next_node = min(unvisited, key=lambda j: dist_matrix[current, j] + dist_matrix[j, end])
        path.append(next_node)
        visited.add(next_node)
        current = next_node
    
    return path

def extract_spectral_features(distance_matrix: np.ndarray, n: int, params: Dict) -> np.ndarray:
    """Extract spectral embedding features."""
    try:
        # Convert to similarity
        max_dist = np.max(distance_matrix)
        similarity = np.exp(-distance_matrix / (max_dist / 10 + 1e-10))
        np.fill_diagonal(similarity, 0)
        
        # Compute Laplacian
        D = np.diag(np.sum(similarity, axis=1))
        L = D - similarity
        
        # Compute eigenvectors
        k = min(params.get('attention_dim', 8), n - 1)
        eigenvalues, eigenvectors = sparse.linalg.eigsh(L, k=k, which='SM')
        
        # Create spectral features: similarity in embedding space
        spectral = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                # Cosine similarity between node embeddings
                emb_i = eigenvectors[i, :]
                emb_j = eigenvectors[j, :]
                similarity_ij = np.dot(emb_i, emb_j) / (np.linalg.norm(emb_i) * np.linalg.norm(emb_j) + 1e-10)
                spectral[i, j] = spectral[j, i] = (1 + similarity_ij) / 2
        
        return spectral
    except Exception:
        return np.ones((n, n)) * 0.5

def extract_geometric_features(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Extract geometric features for Euclidean-like instances."""
    # Try to embed in low-dimensional space
    try:
        # Classical multidimensional scaling
        J = np.eye(n) - np.ones((n, n)) / n
        B = -0.5 * J @ (distance_matrix ** 2) @ J
        
        # Use top 3 dimensions
        eigenvalues, eigenvectors = np.linalg.eigh(B)
        idx = np.argsort(eigenvalues)[::-1][:3]
        embedding = eigenvectors[:, idx] @ np.diag(np.sqrt(np.maximum(eigenvalues[idx], 0)))
        
        # Geometric feature: inverse distance in embedded space
        geometric = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                dist_emb = np.linalg.norm(embedding[i] - embedding[j])
                geometric[i, j] = geometric[j, i] = 1.0 / (1.0 + dist_emb)
        
        return geometric
    except Exception:
        # Fallback: use original distances with transformation
        with np.errstate(divide='ignore', invalid='ignore'):
            geometric = 1 / (distance_matrix + 1e-10)
        np.fill_diagonal(geometric, 0)
        return geometric / (geometric.max() + 1e-10)

def extract_topological_features(distance_matrix: np.ndarray, metrics: Dict, n: int) -> np.ndarray:
    """Extract topological features based on clustering and connectivity."""
    # Local clustering coefficient
    clustering = np.zeros(n)
    threshold = np.median(distance_matrix[np.triu_indices(n, k=1)])
    
    for i in range(n):
        # Find neighbors
        neighbors = np.where(distance_matrix[i] <= threshold)[0]
        neighbors = neighbors[neighbors != i]
        
        if len(neighbors) >= 2:
            # Compute local clustering
            subgraph = distance_matrix[np.ix_(neighbors, neighbors)]
            triangles = 0
            triplets = 0
            
            for a in range(len(neighbors)):
                for b in range(a + 1, len(neighbors)):
                    if subgraph[a, b] <= threshold:
                        triplets += 1
                        for c in range(b + 1, len(neighbors)):
                            if subgraph[a, c] <= threshold and subgraph[b, c] <= threshold:
                                triangles += 1
            
            if triplets > 0:
                clustering[i] = triangles / triplets
    
    # Create topological feature matrix
    topological = np.outer(clustering, clustering)
    
    # Adjust by distance
    with np.errstate(divide='ignore', invalid='ignore'):
        dist_factor = 1 / (distance_matrix + 1e-10)
    np.fill_diagonal(dist_factor, 0)
    
    topological = topological * dist_factor
    return topological / (topological.max() + 1e-10)

def extract_local_features(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Extract local neighborhood features."""
    local = np.zeros((n, n))
    k = min(10, n - 1)
    
    for i in range(n):
        # Get k nearest neighbors
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nn_indices = np.argsort(distances)[:k]
        
        # Create local affinity
        for j in nn_indices:
            # Symmetric local affinity
            affinity = 1.0 / (1.0 + distance_matrix[i, j] / np.mean(distances[nn_indices]))
            local[i, j] = max(local[i, j], affinity)
            local[j, i] = max(local[j, i], affinity)
    
    return local

def attention_fusion(features: Dict, metrics: Dict, params: Dict, n: int) -> np.ndarray:
    """Fuse features using attention mechanism."""
    if params['fusion_type'] == 'adaptive':
        return adaptive_attention_fusion(features, metrics, params, n)
    elif params['fusion_type'] == 'weighted':
        return weighted_fusion(features, params)
    else:  # attention
        return learned_attention_fusion(features, params, n)

def adaptive_attention_fusion(features: Dict, metrics: Dict, params: Dict, n: int) -> np.ndarray:
    """Adaptive attention-based fusion."""
    # Extract feature matrices
    feature_list = []
    weights = []
    
    for key in ['structural', 'spectral', 'geometric', 'topological', 'local']:
        if key in features:
            feature_list.append(features[key])
            # Base weight from bandit parameters
            weight_key = f'w_{key}' if key != 'local' else 'w_structural'
            weights.append(params.get(weight_key, 0.2))
    
    if not feature_list:
        # Fallback to inverse distance
        with np.errstate(divide='ignore', invalid='ignore'):
            fused = 1 / np.maximum(features.get('structural', 
                       np.ones((n, n)) * metrics['dist_mean']), 1e-10)
        np.fill_diagonal(fused, 0)
        return fused
    
    # Normalize weights
    weights = np.array(weights)
    weights = weights / weights.sum()
    
    # Adaptive attention: adjust weights based on feature quality
    for i, feature in enumerate(feature_list):
        # Measure feature quality (variance)
        feature_quality = np.var(feature[feature > 0]) if np.any(feature > 0) else 0
        weights[i] *= (1.0 + 0.5 * feature_quality)
    
    weights = weights / weights.sum()
    
    # Weighted combination
    fused = np.zeros_like(feature_list[0])
    for feature, weight in zip(feature_list, weights):
        # Normalize feature
        feature_max = feature.max()
        if feature_max > 0:
            feature_norm = feature / feature_max
        else:
            feature_norm = feature
        fused += weight * feature_norm
    
    return fused

def weighted_fusion(features: Dict, params: Dict) -> np.ndarray:
    """Simple weighted fusion."""
    n = next(iter(features.values())).shape[0]
    fused = np.zeros((n, n))
    
    for key in features:
        weight_key = f'w_{key}' if key != 'local' else 'w_structural'
        weight = params.get(weight_key, 0.2)
        
        feature = features[key]
        feature_max = feature.max()
        if feature_max > 0:
            feature_norm = feature / feature_max
        else:
            feature_norm = feature
        
        fused += weight * feature_norm
    
    return fused

def learned_attention_fusion(features: Dict, params: Dict, n: int) -> np.ndarray:
    """Learned attention fusion (simplified)."""
    # Simplified attention: compute importance scores
    importance_scores = {}
    
    for key, feature in features.items():
        # Compute multiple quality metrics
        if np.all(feature == 0):
            importance_scores[key] = 0
            continue
        
        # 1. Variance (discriminative power)
        variance = np.var(feature[feature > 0]) if np.any(feature > 0) else 0
        
        # 2. Entropy (information content)
        flat_feature = feature.flatten()
        hist, _ = np.histogram(flat_feature[flat_feature > 0], bins=20)
        hist = hist[hist > 0]
        if len(hist) > 1:
            probs = hist / hist.sum()
            entropy = -np.sum(probs * np.log(probs))
        else:
            entropy = 0
        
        # 3. Sparsity (focus)
        sparsity = np.sum(feature > 0) / (n * n)
        
        importance_scores[key] = 0.4 * variance + 0.4 * entropy + 0.2 * sparsity
    
    # Normalize importance scores
    total_importance = sum(importance_scores.values())
    if total_importance > 0:
        for key in importance_scores:
            importance_scores[key] /= total_importance
    
    # Fuse with importance weights
    fused = np.zeros((n, n))
    for key, importance in importance_scores.items():
        feature = features[key]
        feature_max = feature.max()
        if feature_max > 0:
            feature_norm = feature / feature_max
        else:
            feature_norm = feature
        
        fused += importance * feature_norm
    
    return fused

def spectral_aware_sparsification(scores: np.ndarray, distance_matrix: np.ndarray,
                                 metrics: Dict, params: Dict, n: int) -> np.ndarray:
    """Spectral-aware sparsification preserving connectivity."""
    sparsified = np.zeros((n, n))
    
    # Determine sparsity level
    preserve_count = int(params['preserve_ratio'] * n * (n - 1) / 2)
    k_per_node = max(2, int(np.sqrt(preserve_count * 2 / n)))
    
    # Ensure MST edges are preserved
    try:
        mst = sparse.csgraph.minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = np.where(mst > 0)
        for i, j in zip(*mst_edges):
            if i != j:
                sparsified[i, j] = scores[i, j]
    except Exception:
        pass
    
    # Spectral guidance: preserve edges in same spectral cluster
    if params['use_spectral'] and metrics['spectral_gap'] > 0.01:
        try:
            # Quick spectral clustering
            similarity = np.exp(-distance_matrix / metrics['dist_mean'])
            np.fill_diagonal(similarity, 0)
            
            D = np.diag(np.sum(similarity, axis=1))
            L = D - similarity
            
            # Get Fiedler vector (second smallest eigenvector)
            eigenvalues, eigenvectors = sparse.linalg.eigsh(L, k=2, which='SM')
            fiedler = eigenvectors[:, 1]
            
            # Partition nodes
            median = np.median(fiedler)
            cluster1 = np.where(fiedler < median)[0]
            cluster2 = np.where(fiedler >= median)[0]
            
            # Preserve intra-cluster edges with high scores
            for cluster in [cluster1, cluster2]:
                if len(cluster) > 1:
                    sub_scores = scores[np.ix_(cluster, cluster)]
                    np.fill_diagonal(sub_scores, -np.inf)
                    
                    # Preserve top intra-cluster edges
                    preserve_intra = min(len(cluster) * 2, k_per_node * len(cluster) // 2)
                    flat_indices = np.argsort(sub_scores.flatten())[-preserve_intra:]
                    rows, cols = np.unravel_index(flat_indices, sub_scores.shape)
                    
                    for r, c in zip(rows, cols):
                        i, j = cluster[r], cluster[c]
                        if i != j:
                            sparsified[i, j] = max(sparsified[i, j], scores[i, j])
        except Exception:
            pass
    
    # Node-wise sparsification for remaining edges
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        # Count already preserved edges
        already_preserved = np.sum(sparsified[i] > 0)
        to_select = max(0, k_per_node - already_preserved)
        
        if to_select > 0:
            # Get candidate edges (not already preserved)
            candidates = np.where((sparsified[i] == 0) & (np.arange(n) != i))[0]
            if len(candidates) == 0:
                continue
            
            candidate_scores = row_scores[candidates]
            
            # Temperature-based sampling
            temperature = params['temperature']
            if temperature > 0 and len(candidates) > 1:
                scaled_scores = candidate_scores / temperature
                exp_scores = np.exp(scaled_scores - np.max(scaled_scores))
                probs = exp_scores / np.sum(exp_scores)
                
                # Sample with exploration
                if np.random.random() < params['exploration_rate']:
                    selected = np.random.choice(candidates, 
                                               size=min(to_select, len(candidates)),
                                               replace=False)
                else:
                    # Weighted sampling
                    selected = np.random.choice(candidates,
                                               size=min(to_select, len(candidates)),
                                               p=probs,
                                               replace=False)
            else:
                # Deterministic selection
                top_k = min(to_select, len(candidates))
                top_idx = np.argsort(candidate_scores)[-top_k:]
                selected = candidates[top_idx]
            
            # Apply decay
            selected_scores = row_scores[selected]
            sorted_idx = np.argsort(selected_scores)[::-1]
            ranks = np.arange(len(sorted_idx))
            decay = np.exp(-ranks / params['decay_rate'])
            
            for rank, idx in enumerate(sorted_idx):
                j = selected[idx]
                sparsified[i, j] = selected_scores[idx] * decay[rank]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Final connectivity check
    sparsified = ensure_spectral_connectivity(sparsified, distance_matrix, n)
    
    return sparsified

def ensure_spectral_connectivity(matrix: np.ndarray, distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Ensure the sparsified matrix maintains connectivity."""
    result = matrix.copy()
    
    # Quick connectivity check using BFS
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        node = stack.pop()
        neighbors = np.where(result[node] > 0)[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    if not np.all(visited):
        # Add missing edges from MST
        try:
            mst = sparse.csgraph.minimum_spanning_tree(distance_matrix).toarray()
            mst_edges = np.where(mst > 0)
            for i, j in zip(*mst_edges):
                if i != j and result[i, j] == 0:
                    # Add with appropriate weight
                    avg_weight = np.mean(result[result > 0]) if np.any(result > 0) else 1.0
                    result[i, j] = result[j, i] = avg_weight
        except Exception:
            # Fallback: connect to nearest neighbors
            unvisited = np.where(~visited)[0]
            for node in unvisited:
                distances = distance_matrix[node].copy()
                distances[node] = np.inf
                nearest = np.argmin(distances)
                result[node, nearest] = result[nearest, node] = 1.0
    
    return result

def meta_enhanced_refinement(matrix: np.ndarray, distance_matrix: np.ndarray,
                            metrics: Dict, params: Dict) -> np.ndarray:
    """Final refinement with meta-learning enhancements."""
    n = matrix.shape[0]
    result = matrix.copy()
    
    # 1. Add exploration noise
    if params['exploration_rate'] > 0:
        noise_scale = 0.1 * params['exploration_rate']
        noise = noise_scale * np.abs(np.random.randn(n, n))
        np.fill_diagonal(noise, 0)
        result = np.maximum(result + noise, 0)
    
    # 2. Local smoothing based on graph type
    if metrics['clustering_coeff'] > 0.4:
        # For clustered graphs, enhance intra-cluster edges
        for i in range(n):
            row = result[i].copy()
            # Compute local average
            threshold = np.percentile(row[row > 0], 75) if np.any(row > 0) else 0
            local_mask = row > threshold
            if np.any(local_mask):
                local_avg = np.mean(row[local_mask])
                result[i, local_mask] = 0.7 * row[local_mask] + 0.3 * local_avg
    
    # 3. Ensure symmetry
    result = (result + result.T) / 2
    
    # 4. Row normalization with fallback
    row_sums = result.sum(axis=1, keepdims=True)
    zero_rows = row_sums.flatten() == 0
    
    if np.any(zero_rows):
        # Fallback to inverse distance
        with np.errstate(divide='ignore', invalid='ignore'):
            fallback = 1 / np.maximum(distance_matrix[zero_rows], 1e-10)
        np.fill_diagonal(fallback, 0)
        result[zero_rows] = fallback
        row_sums[zero_rows] = fallback.sum(axis=1, keepdims=True)
    
    row_sums[row_sums == 0] = 1
    result = result / row_sums
    
    return result
