import numpy as np
from scipy import sparse
from scipy.sparse import csgraph
import warnings
from collections import defaultdict, deque

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    warnings.filterwarnings('ignore')
    
    n = distance_matrix.shape[0]
    
    # Trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic = np.nan_to_num(heuristic, nan=0, posinf=0, neginf=0)
        np.fill_diagonal(heuristic, 0)
        return heuristic
    
    # ========== BASE HEURISTIC COMPUTATION ==========
    # Reciprocal distances with adaptive scaling
    triu_distances = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(triu_distances)
    dist_std = np.std(triu_distances)
    dist_cv = dist_std / max(dist_mean, 1e-10)
    
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / np.maximum(distance_matrix, 1e-10)
        reciprocal = np.nan_to_num(reciprocal, nan=0, posinf=0, neginf=0)
    
    # Scale based on distance variability
    scale_exp = 0.5 if dist_cv < 0.3 else 0.8 if dist_cv > 0.7 else 0.65
    reciprocal = reciprocal ** scale_exp
    
    # Minimum Spanning Tree component
    try:
        mst = csgraph.minimum_spanning_tree(distance_matrix)
        mst_mask = mst.toarray() > 0
    except:
        mst_mask = np.zeros_like(distance_matrix, dtype=bool)
        for i in range(n):
            row = distance_matrix[i].copy()
            row[i] = np.inf
            j = np.argmin(row)
            mst_mask[i, j] = mst_mask[j, i] = True
    
    # Boost MST edges
    mst_boosted = reciprocal.copy()
    mst_boosted[mst_mask] *= 1.5
    
    # ========== ENTROPY-BASED SPARSITY ADJUSTMENT ==========
    def row_entropy(row):
        """Compute normalized entropy of a row (excluding diagonal)."""
        row = row.copy()
        np.fill_diagonal(row, 0)
        row_sum = row.sum()
        if row_sum <= 0:
            return 0.0
        
        prob = row / row_sum
        # Avoid log(0) and ignore zeros
        mask = prob > 0
        if not np.any(mask):
            return 0.0
        
        entropy = -np.sum(prob[mask] * np.log(prob[mask]))
        # Normalize by maximum entropy (log(n-1))
        max_entropy = np.log(max(n-1, 1))
        return entropy / max_entropy if max_entropy > 0 else 0.0
    
    # Compute entropy for each node
    row_entropies = np.array([row_entropy(mst_boosted[i]) for i in range(n)])
    
    # Base sparsity parameter
    if n < 50:
        base_k = n - 1
    elif dist_cv > 0.8:
        base_k = int(3 * np.sqrt(n))
    else:
        base_k = int(2 * np.sqrt(n))
    
    # Adjust k per node based on entropy
    k_per_node = np.zeros(n, dtype=int)
    for i in range(n):
        # High entropy -> keep more edges (more uncertainty)
        # Low entropy -> keep fewer edges (clear favorites)
        adjusted = base_k * (1.0 + row_entropies[i])
        k_per_node[i] = min(n-1, max(1, int(adjusted)))
    
    # ========== ADAPTIVE SPARSIFICATION ==========
    heuristic = mst_boosted.copy()
    
    # Preserve MST edges (critical for connectivity)
    preserved_edges = set()
    mst_edges = np.column_stack(np.where(mst_mask))
    for i, j in mst_edges:
        if i != j:
            preserved_edges.add((min(i, j), max(i, j)))
    
    # Vectorized edge selection with exponential decay
    for i in range(n):
        # Get current preserved count for this node
        current_preserved = sum(1 for j in range(n) 
                               if (min(i, j), max(i, j)) in preserved_edges)
        
        # Skip if already have enough edges
        if current_preserved >= k_per_node[i]:
            continue
        
        # Candidate edges (not preserved, not diagonal)
        candidates = []
        for j in range(n):
            if i == j:
                continue
            if (min(i, j), max(i, j)) not in preserved_edges:
                candidates.append((j, heuristic[i, j]))
        
        if not candidates:
            continue
        
        # Sort by heuristic value descending
        candidates.sort(key=lambda x: x[1], reverse=True)
        
        # Select top edges with exponential decay
        needed = k_per_node[i] - current_preserved
        selected = min(needed, len(candidates))
        
        for rank, (j, val) in enumerate(candidates[:selected]):
            decay = np.exp(-rank / 2.0)  # Exponential decay
            heuristic[i, j] = val * decay
            preserved_edges.add((min(i, j), max(i, j)))
        # Zero out non-selected candidates
        for j, _ in candidates[selected:]:
            heuristic[i, j] = 0
    
    # Ensure symmetry
    heuristic = np.maximum(heuristic, heuristic.T)
    
    # ========== EXPLORATION FACTOR ==========
    # Add small exploration bonus to moderate-distance edges
    exploration = np.zeros_like(heuristic)
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        med_dist = np.median(distances[distances < np.inf])
        if med_dist > 0:
            # Gaussian around median distance
            exploration[i] = 0.1 * np.exp(-(distances - med_dist)**2 / (2*(med_dist*0.5)**2))
    
    exploration = (exploration + exploration.T) / 2
    np.fill_diagonal(exploration, 0)
    
    heuristic = heuristic + exploration
    
    # ========== ROW NORMALIZATION ==========
    row_sums = heuristic.sum(axis=1, keepdims=True)
    zero_rows = row_sums.flatten() == 0
    
    if np.any(zero_rows):
        # Fallback to reciprocal distances for isolated rows
        heuristic[zero_rows] = reciprocal[zero_rows]
        row_sums = heuristic.sum(axis=1, keepdims=True)
    
    row_sums[row_sums == 0] = 1
    heuristic = heuristic / row_sums
    
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
