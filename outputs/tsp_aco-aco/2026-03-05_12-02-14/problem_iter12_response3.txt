def _analyze_instance(distance_matrix: np.ndarray, n: int) -> dict:
    """Analyze instance characteristics for adaptive algorithm selection."""
    features = {}
    
    # Extract upper triangular distances
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    # Basic statistics
    features['size'] = n
    features['mean_dist'] = np.mean(distances)
    features['std_dist'] = np.std(distances)
    features['cv_dist'] = features['std_dist'] / (features['mean_dist'] + 1e-10)
    features['skewness'] = np.mean((distances - features['mean_dist'])**3) / (features['std_dist']**3 + 1e-10)
    
    # Geometric properties
    features['euclidean_score'] = _estimate_euclidean_score(distance_matrix, n)
    
    # Cluster structure
    features['cluster_tendency'] = _estimate_cluster_tendency(distance_matrix, n)
    
    # Distance distribution features
    percentiles = np.percentile(distances, [25, 50, 75, 90])
    features['q1'] = percentiles[0]
    features['median'] = percentiles[1]
    features['q3'] = percentiles[2]
    features['p90'] = percentiles[3]
    
    # Symmetry check
    features['symmetry_violation'] = np.max(np.abs(distance_matrix - distance_matrix.T)) / (features['mean_dist'] + 1e-10)
    
    return features


def _estimate_euclidean_score(distance_matrix: np.ndarray, n: int) -> float:
    """Estimate how well distances satisfy Euclidean properties."""
    if n < 4:
        return 1.0
    
    # Sample random triangles to check triangle inequality
    samples = min(100, n * (n-1) * (n-2) // 6)
    violations = 0
    
    for _ in range(samples):
        i, j, k = random.sample(range(n), 3)
        d_ij = distance_matrix[i, j]
        d_ik = distance_matrix[i, k]
        d_jk = distance_matrix[j, k]
        
        # Check triangle inequality
        if d_ij > d_ik + d_jk + 1e-6:
            violations += 1
        if d_ik > d_ij + d_jk + 1e-6:
            violations += 1
        if d_jk > d_ij + d_ik + 1e-6:
            violations += 1
    
    return 1.0 - violations / (samples * 3)


def _estimate_cluster_tendency(distance_matrix: np.ndarray, n: int) -> float:
    """Estimate tendency of nodes to form clusters."""
    if n < 10:
        return 0.5
    
    # Simple heuristic: ratio of short distances to long distances
    median_dist = np.median(distance_matrix[np.triu_indices(n, k=1)])
    short_ratio = np.sum(distance_matrix < median_dist * 0.5) / (n * n - n)
    long_ratio = np.sum(distance_matrix > median_dist * 1.5) / (n * n - n)
    
    # High cluster tendency: many short distances, few long distances
    return short_ratio * (1 - long_ratio)

def _diverse_construction(distance_matrix: np.ndarray, n: int, params: dict) -> list:
    """Generate diverse initial tours using multiple construction heuristics."""
    tours = []
    num_methods = params['construction_diversity']
    
    # 1. Nearest Neighbor from multiple starting points
    for start in random.sample(range(n), min(5, n)):
        tour = _nearest_neighbor(distance_matrix, start)
        tours.append(tour)
    
    # 2. Farthest Insertion
    if num_methods >= 2:
        for start in random.sample(range(n), min(3, n)):
            tour = _farthest_insertion(distance_matrix, start)
            tours.append(tour)
    
    # 3. Cheapest Insertion
    if num_methods >= 3:
        for start in random.sample(range(n), min(3, n)):
            tour = _cheapest_insertion(distance_matrix, start)
            tours.append(tour)
    
    # 4. Random Insertion with Greedy Bias
    if num_methods >= 4:
        for _ in range(min(3, n)):
            tour = _random_insertion(distance_matrix)
            tours.append(tour)
    
    # 5. Christofides Approximation (for medium-sized instances)
    if num_methods >= 5 and n <= 200:
        tour = _christofides_approx(distance_matrix)
        if tour:
            tours.append(tour)
    
    return tours


def _nearest_neighbor(distance_matrix: np.ndarray, start: int) -> list:
    """Nearest neighbor construction heuristic."""
    n = distance_matrix.shape[0]
    unvisited = set(range(n))
    unvisited.remove(start)
    tour = [start]
    current = start
    
    while unvisited:
        # Find nearest unvisited neighbor
        min_dist = float('inf')
        next_node = -1
        for node in unvisited:
            if distance_matrix[current, node] < min_dist:
                min_dist = distance_matrix[current, node]
                next_node = node
        
        if next_node == -1:
            break
            
        tour.append(next_node)
        unvisited.remove(next_node)
        current = next_node
    
    return tour


def _farthest_insertion(distance_matrix: np.ndarray, start: int) -> list:
    """Farthest insertion construction heuristic."""
    n = distance_matrix.shape[0]
    tour = [start]
    unvisited = set(range(n))
    unvisited.remove(start)
    
    # Start with farthest node from start
    farthest = max(unvisited, key=lambda x: distance_matrix[start, x])
    tour.append(farthest)
    unvisited.remove(farthest)
    
    while unvisited:
        # Find farthest node from current tour
        farthest_node = -1
        max_dist = -1
        for node in unvisited:
            min_dist_to_tour = min(distance_matrix[node, t] for t in tour)
            if min_dist_to_tour > max_dist:
                max_dist = min_dist_to_tour
                farthest_node = node
        
        if farthest_node == -1:
            break
        
        # Find best position to insert
        best_pos = -1
        best_increase = float('inf')
        for i in range(len(tour)):
            a, b = tour[i], tour[(i+1)%len(tour)]
            increase = (distance_matrix[a, farthest_node] + 
                       distance_matrix[farthest_node, b] - 
                       distance_matrix[a, b])
            if increase < best_increase:
                best_increase = increase
                best_pos = i + 1
        
        tour.insert(best_pos, farthest_node)
        unvisited.remove(farthest_node)
    
    return tour


def _cheapest_insertion(distance_matrix: np.ndarray, start: int) -> list:
    """Cheapest insertion construction heuristic."""
    n = distance_matrix.shape[0]
    tour = [start]
    unvisited = set(range(n))
    unvisited.remove(start)
    
    # Start with nearest neighbor
    nearest = min(unvisited, key=lambda x: distance_matrix[start, x])
    tour.append(nearest)
    unvisited.remove(nearest)
    
    while unvisited:
        best_node = -1
        best_pos = -1
        best_increase = float('inf')
        
        for node in unvisited:
            for i in range(len(tour)):
                a, b = tour[i], tour[(i+1)%len(tour)]
                increase = (distance_matrix[a, node] + 
                           distance_matrix[node, b] - 
                           distance_matrix[a, b])
                if increase < best_increase:
                    best_increase = increase
                    best_node = node
                    best_pos = i + 1
        
        if best_node == -1:
            break
        
        tour.insert(best_pos, best_node)
        unvisited.remove(best_node)
    
    return tour


def _random_insertion(distance_matrix: np.ndarray) -> list:
    """Random insertion with greedy bias."""
    n = distance_matrix.shape[0]
    nodes = list(range(n))
    random.shuffle(nodes)
    
    tour = [nodes[0]]
    for node in nodes[1:]:
        # Find best position to insert
        best_pos = -1
        best_increase = float('inf')
        for i in range(len(tour)):
            a, b = tour[i], tour[(i+1)%len(tour)]
            increase = (distance_matrix[a, node] + 
                       distance_matrix[node, b] - 
                       distance_matrix[a, b])
            if increase < best_increase:
                best_increase = increase
                best_pos = i + 1
        
        if best_pos == -1:
            tour.append(node)
        else:
            tour.insert(best_pos, node)
    
    return tour


def _christofides_approx(distance_matrix: np.ndarray) -> list:
    """Approximate Christofides algorithm."""
    n = distance_matrix.shape[0]
    
    try:
        # Minimum Spanning Tree
        mst = sparse.csgraph.minimum_spanning_tree(distance_matrix)
        mst_array = mst.toarray()
        
        # Find odd-degree vertices
        degrees = (mst_array > 0).sum(axis=1)
        odd_vertices = np.where(degrees % 2 == 1)[0].tolist()
        
        # Greedy matching on odd vertices
        matched = set()
        matching_edges = []
        
        while len(odd_vertices) >= 2:
            best_pair = None
            best_dist = float('inf')
            
            for i in range(len(odd_vertices)):
                for j in range(i+1, len(odd_vertices)):
                    u, v = odd_vertices[i], odd_vertices[j]
                    if u in matched or v in matched:
                        continue
                    dist = distance_matrix[u, v]
                    if dist < best_dist:
                        best_dist = dist
                        best_pair = (u, v)
            
            if best_pair:
                u, v = best_pair
                matching_edges.append((u, v))
                matched.update([u, v])
                odd_vertices.remove(u)
                odd_vertices.remove(v)
            else:
                break
        
        # Create Eulerian tour (simplified)
        # Combine MST and matching edges
        visited = [False] * n
        tour = []
        stack = [0]
        
        while stack:
            node = stack[-1]
            if not visited[node]:
                visited[node] = True
                tour.append(node)
            
            # Find unvisited neighbors
            neighbors = []
            # MST neighbors
            mst_neighbors = np.where(mst_array[node] > 0)[0]
            neighbors.extend([n for n in mst_neighbors if not visited[n]])
            # Matching neighbors
            for u, v in matching_edges:
                if u == node and not visited[v]:
                    neighbors.append(v)
                elif v == node and not visited[u]:
                    neighbors.append(u)
            
            if neighbors:
                next_node = min(neighbors, key=lambda x: distance_matrix[node, x])
                stack.append(next_node)
            else:
                stack.pop()
        
        return tour
        
    except:
        # Fallback to nearest neighbor
        return _nearest_neighbor(distance_matrix, 0)


def _local_search_portfolio(distance_matrix: np.ndarray, tours: list, params: dict) -> list:
    """Apply portfolio of local search heuristics to improve tours."""
    improved_tours = []
    intensity = params['local_search_intensity']
    
    for tour in tours:
        current = tour.copy()
        
        # 2-opt (always applied)
        current = _two_opt(current, distance_matrix, iterations=3)
        
        # 3-opt for higher intensity
        if intensity >= 2:
            current = _three_opt(current, distance_matrix, iterations=2)
        
        # Lin-Kernighan approximation for highest intensity
        if intensity >= 3 and len(current) <= 100:
            current = _lin_kernighan_approx(current, distance_matrix, steps=5)
        
        improved_tours.append(current)
    
    return improved_tours


def _two_opt(tour: list, distance_matrix: np.ndarray, iterations: int) -> list:
    """Apply 2-opt local search."""
    n = len(tour)
    best_tour = tour.copy()
    best_length = _tour_length(distance_matrix, best_tour)
    
    for _ in range(iterations):
        improved = False
        
        for i in range(n):
            for j in range(i + 2, n - 1):
                # Try reversing segment between i+1 and j
                new_tour = best_tour.copy()
                new_tour[i+1:j+1] = new_tour[i+1:j+1][::-1]
                
                new_length = _tour_length(distance_matrix, new_tour)
                if new_length < best_length:
                    best_tour = new_tour
                    best_length = new_length
                    improved = True
                    break
            
            if improved:
                break
        
        if not improved:
            break
    
    return best_tour


def _three_opt(tour: list, distance_matrix: np.ndarray, iterations: int) -> list:
    """Apply simplified 3-opt local search."""
    n = len(tour)
    best_tour = tour.copy()
    best_length = _tour_length(distance_matrix, best_tour)
    
    for _ in range(iterations):
        improved = False
        
        for i in range(n):
            for j in range(i + 2, n - 2):
                for k in range(j + 2, n - 1):
                    # Current edges
                    a, b = best_tour[i], best_tour[i+1]
                    c, d = best_tour[j], best_tour[j+1]
                    e, f = best_tour[k], best_tour[(k+1)%n]
                    
                    current = (distance_matrix[a, b] + 
                              distance_matrix[c, d] + 
                              distance_matrix[e, f])
                    
                    # Try different reconnections
                    # Option 1: 2-opt on two segments
                    new1 = (distance_matrix[a, c] + 
                           distance_matrix[b, d] + 
                           distance_matrix[e, f])
                    
                    # Option 2: 3-opt reconnection
                    new2 = (distance_matrix[a, d] + 
                           distance_matrix[e, b] + 
                           distance_matrix[c, f])
                    
                    if new1 < current:
                        # Apply 2-opt-like reversal
                        new_tour = best_tour.copy()
                        new_tour[i+1:j+1] = new_tour[i+1:j+1][::-1]
                        best_tour = new_tour
                        best_length = _tour_length(distance_matrix, best_tour)
                        improved = True
                        break
                    elif new2 < current:
                        # Apply 3-opt reconnection
                        new_tour = best_tour.copy()
                        segment1 = new_tour[i+1:j+1]
                        segment2 = new_tour[j+1:k+1]
                        new_tour[i+1:k+1] = segment2 + segment1
                        best_tour = new_tour
                        best_length = _tour_length(distance_matrix, best_tour)
                        improved = True
                        break
                
                if improved:
                    break
            
            if improved:
                break
        
        if not improved:
            break
    
    return best_tour


def _lin_kernighan_approx(tour: list, distance_matrix: np.ndarray, steps: int) -> list:
    """Approximate Lin-Kernighan heuristic."""
    n = len(tour)
    current_tour = tour.copy()
    
    for _ in range(steps):
        best_gain = 0
        best_move = None
        
        # Try k-opt moves up to 4-opt
        for k in range(2, 5):
            for start in random.sample(range(n), min(10, n)):
                gain, new_tour = _try_k_opt(current_tour, distance_matrix, k, start)
                if gain > best_gain:
                    best_gain = gain
                    best_move = new_tour
        
        if best_move is not None:
            current_tour = best_move
    
    return current_tour


def _try_k_opt(tour: list, distance_matrix: np.ndarray, k: int, start: int) -> tuple:
    """Try a k-opt move starting from given position."""
    n = len(tour)
    if k > 4 or n < k + 2:
        return 0, tour
    
    # Simple implementation for 2-opt and 3-opt
    if k == 2:
        i = start
        j = (start + n // 3) % n
        if abs(i - j) < 2:
            j = (j + n // 3) % n
        
        # Calculate gain
        a, b = tour[i], tour[(i+1)%n]
        c, d = tour[j], tour[(j+1)%n]
        current = distance_matrix[a, b] + distance_matrix[c, d]
        new = distance_matrix[a, c] + distance_matrix[b, d]
        gain = current - new
        
        if gain > 0:
            new_tour = tour.copy()
            # Reverse segment between i+1 and j
            if i < j:
                new_tour[i+1:j+1] = new_tour[i+1:j+1][::-1]
            else:
                segment = new_tour[i+1:] + new_tour[:j+1]
                segment.reverse()
                new_tour[i+1:] = segment[:n-i-1]
                new_tour[:j+1] = segment[n-i-1:]
            
            return gain, new_tour
    
    return 0, tour


def _tour_length(distance_matrix: np.ndarray, tour: list) -> float:
    """Calculate tour length."""
    n = len(tour)
    length = 0.0
    for i in range(n):
        length += distance_matrix[tour[i], tour[(i+1)%n]]
    return length


def _extract_edge_statistics(distance_matrix: np.ndarray, tours: list) -> tuple:
    """Extract edge frequencies and solution qualities from tours."""
    n = distance_matrix.shape[0]
    edge_frequencies = np.zeros((n, n))
    solution_qualities = []
    
    if not tours:
        return edge_frequencies, solution_qualities
    
    # Calculate tour qualities
    for tour in tours:
        length = _tour_length(distance_matrix, tour)
        solution_qualities.append(length)
        
        # Weight edges by tour quality (better tours get higher weight)
        weight = 1.0 / (1.0 + length / len(tour))
        
        # Record edges
        for i in range(len(tour)):
            a, b = tour[i], tour[(i+1)%len(tour)]
            edge_frequencies[a, b] += weight
            edge_frequencies[b, a] += weight
    
    # Normalize edge frequencies
    if edge_frequencies.max() > 0:
        edge_frequencies = edge_frequencies / edge_frequencies.max()
    
    return edge_frequencies, solution_qualities


def _geometric_component(distance_matrix: np.ndarray, n: int, 
                         instance_features: dict) -> np.ndarray:
    """Compute geometric component based on distances and clustering."""
    component = np.zeros((n, n))
    median_dist = instance_features['median']
    
    for i in range(n):
        for j in range(n):
            if i != j:
                d = distance_matrix[i, j]
                
                # Adaptive transformation based on distance characteristics
                if d < median_dist * 0.7:
                    # Short distances: strong inverse relationship
                    component[i, j] = 1.0 / (d + 1e-10)
                elif d < median_dist * 1.5:
                    # Medium distances: moderate value
                    component[i, j] = 0.5 / (d + 1e-10)
                else:
                    # Long distances: soft penalty
                    component[i, j] = np.exp(-d / (median_dist * 3))
    
    # Adjust for cluster tendency
    cluster_tendency = instance_features.get('cluster_tendency', 0.5)
    if cluster_tendency > 0.7:
        # Boost intra-cluster connections
        for i in range(n):
            for j in range(n):
                if i != j and distance_matrix[i, j] < median_dist * 0.8:
                    component[i, j] *= (1.0 + cluster_tendency)
    
    # Row normalization
    row_sums = component.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    component = component / row_sums
    
    return component


def _structural_component(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute structural component based on MST and centrality."""
    component = np.zeros((n, n))
    
    try:
        # Minimum Spanning Tree
        mst = sparse.csgraph.minimum_spanning_tree(distance_matrix)
        mst_array = mst.toarray()
        
        # MST edges get high weight
        mst_edges = np.where(mst_array > 0)
        for i, j in zip(*mst_edges):
            if i != j:
                component[i, j] = 1.0
        
        # Add centrality information
        centrality = np.zeros(n)
        for i in range(n):
            # Harmonic centrality
            inv_sum = 0
            for j in range(n):
                if i != j and distance_matrix[i, j] > 0:
                    inv_sum += 1.0 / distance_matrix[i, j]
            centrality[i] = inv_sum / (n - 1)
        
        # Normalize centrality
        if centrality.max() > 0:
            centrality = centrality / centrality.max()
        
        # Add centrality-based connections
        for i in range(n):
            for j in range(n):
                if i != j:
                    component[i, j] += 0.3 * centrality[i] * centrality[j]
    
    except:
        # Fallback: use inverse distances
        for i in range(n):
            for j in range(n):
                if i != j:
                    component[i, j] = 1.0 / (distance_matrix[i, j] + 1e-10)
    
    # Row normalization
    row_sums = component.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    component = component / row_sums
    
    return component


def _empirical_component(edge_frequencies: np.ndarray, solution_qualities: list, 
                         n: int) -> np.ndarray:
    """Compute empirical component from sampled tours."""
    if len(solution_qualities) == 0:
        return np.ones((n, n)) / n
    
    # Create component from edge frequencies
    component = edge_frequencies.copy()
    
    # Adjust based on solution quality variance
    if len(solution_qualities) > 1:
        quality_std = np.std(solution_qualities)
        quality_mean = np.mean(solution_qualities)
        if quality_mean > 0:
            # Higher variance suggests more exploration needed
            exploration_boost = min(2.0, 1.0 + quality_std / quality_mean)
            component *= exploration_boost
    
    # Ensure all nodes have some connections
    row_sums = component.sum(axis=1)
    for i in range(n):
        if row_sums[i] == 0:
            # Add small uniform probability to all edges from this node
            component[i] = 1.0 / (n - 1)
            component[i, i] = 0
    
    # Row normalization
    row_sums = component.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    component = component / row_sums
    
    return component


def _adaptive_weighting(instance_features: dict, n: int, 
                        components: dict) -> dict:
    """Adaptively select weights for component combination."""
    # Base weights
    weights = {
        'geometric': 0.25,
        'structural': 0.25,
        'neural': 0.30,
        'empirical': 0.20,
    }
    
    # Adjust based on instance size
    if n < 30:
        weights['neural'] = 0.20
        weights['empirical'] = 0.30
        weights['geometric'] = 0.25
        weights['structural'] = 0.25
    elif n > 200:
        weights['neural'] = 0.35
        weights['empirical'] = 0.15
        weights['geometric'] = 0.30
        weights['structural'] = 0.20
    
    # Adjust based on Euclidean score
    euclidean = instance_features.get('euclidean_score', 0.5)
    if euclidean > 0.8:
        weights['geometric'] *= 1.3
        weights['structural'] *= 1.2
    elif euclidean < 0.3:
        weights['neural'] *= 1.3
        weights['empirical'] *= 1.2
    
    # Adjust based on cluster tendency
    cluster = instance_features.get('cluster_tendency', 0.5)
    if cluster > 0.7:
        weights['structural'] *= 1.4
        weights['neural'] *= 0.8
    
    # Normalize weights
    total = sum(weights.values())
    weights = {k: v / total for k, v in weights.items()}
    
    return weights


def _combine_components(components: dict, weights: dict) -> np.ndarray:
    """Combine components with given weights."""
    result = np.zeros_like(list(components.values())[0])
    
    for name, component in components.items():
        if name in weights:
            result += weights[name] * component
    
    return result


def _instance_specific_refinement(matrix: np.ndarray, distance_matrix: np.ndarray,
                                 n: int, instance_features: dict, 
                                 params: dict) -> np.ndarray:
    """Apply instance-specific refinement to the combined matrix."""
    result = matrix.copy()
    
    # Ensure non-negativity
    result = np.maximum(result, 0)
    
    # Apply sparsification
    sparsity_factor = params.get('sparsity_factor', 0.3)
    if sparsity_factor < 1.0:
        result = _intelligent_sparsification(result, distance_matrix, n, sparsity_factor)
    
    # Apply distance-based smoothing
    result = _distance_smoothing(result, distance_matrix, n, instance_features)
    
    # Ensure symmetry and proper normalization
    result = (result + result.T) / 2
    np.fill_diagonal(result, 0)
    
    # Final row normalization
    row_sums = result.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    result = result / row_sums
    
    return result


def _intelligent_sparsification(matrix: np.ndarray, distance_matrix: np.ndarray,
                               n: int, sparsity_factor: float) -> np.ndarray:
    """Intelligently sparsify the matrix while preserving important connections."""
    if sparsity_factor >= 1.0 or n < 10:
        return matrix
    
    # Determine number of edges to keep per node
    edges_per_node = max(2, int((n - 1) * (1 - sparsity_factor)))
    
    sparsified = np.zeros((n, n))
    
    for i in range(n):
        # Get row and mask out diagonal
        row = matrix[i].copy()
        row[i] = -np.inf
        
        # Find top edges
        if edges_per_node < n - 1:
            # Use threshold based on percentile
            threshold = np.percentile(row, 100 * (1 - edges_per_node/(n-1)))
            indices = np.where(row >= threshold)[0]
        else:
            indices = [j for j in range(n) if j != i]
        
        # Keep these edges with distance-based decay
        for j in indices:
            if i != j:
                # Apply distance-based decay factor
                dist = distance_matrix[i, j]
                median_dist = np.median(distance_matrix[np.triu_indices(n, k=1)])
                decay = np.exp(-dist / (median_dist * 2))
                sparsified[i, j] = matrix[i, j] * decay
    
    return sparsified


def _distance_smoothing(matrix: np.ndarray, distance_matrix: np.ndarray,
                       n: int, instance_features: dict) -> np.ndarray:
    """Apply distance-based smoothing to the matrix."""
    smoothed = matrix.copy()
    
    # Create distance-based kernel
    median_dist = instance_features['median']
    kernel = np.zeros((n, n))
    
    for i in range(n):
        for j in range(n):
            if i != j:
                # Gaussian kernel based on normalized distance
                normalized_dist = distance_matrix[i, j] / median_dist
                kernel[i, j] = np.exp(-0.5 * (normalized_dist ** 2))
    
    # Blend original matrix with kernel
    # Use more smoothing for non-Euclidean instances
    euclidean = instance_features.get('euclidean_score', 0.5)
    blend_factor = 0.7 + 0.3 * (1 - euclidean)  # More smoothing for non-Euclidean
    
    smoothed = blend_factor * smoothed + (1 - blend_factor) * kernel
    
    return smoothed
