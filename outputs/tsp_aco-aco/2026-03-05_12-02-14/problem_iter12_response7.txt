import numpy as np
from scipy.sparse import csgraph
from collections import defaultdict, deque
import warnings
warnings.filterwarnings('ignore')

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Next-generation heuristic for TSP edge sampling with:
    1. Online learning from search history via solution memory
    2. Adaptive sparsity control with runtime feedback
    3. Vectorized 2-opt local search with early stopping
    4. Memory-guided exploration balancing exploitation
    
    Args:
        distance_matrix: Symmetric n x n matrix of distances between nodes
        
    Returns:
        Row-normalized heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Trivial cases
    if n <= 2:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / distance_matrix
            heuristic = np.nan_to_num(heuristic, nan=0, posinf=0, neginf=0)
        np.fill_diagonal(heuristic, 0)
        return heuristic
    
    # ========== ADAPTIVE PARAMETERS ==========
    # Adjust based on problem size and characteristics
    triu_distances = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(triu_distances)
    dist_std = np.std(triu_distances)
    dist_cv = dist_std / max(dist_mean, 1e-10)
    
    # Number of tours to generate adaptively
    n_tours = min(20, max(5, int(2 * np.sqrt(n))))
    # Sparsity factor adapts to problem characteristics
    if n < 50:
        k_edges = n - 1
    elif dist_cv > 0.8:  # High variance: keep more edges
        k_edges = int(3 * np.sqrt(n))
    else:
        k_edges = int(2 * np.sqrt(n))
    
    # ========== ONLINE LEARNING: TOUR GENERATION & IMPROVEMENT ==========
    edge_memory = defaultdict(float)
    tour_memory = deque(maxlen=10)  # Keep recent high-quality tours
    
    def vectorized_2opt_improvement(tour, max_iterations=100):
        """Fast vectorized 2-opt implementation with early stopping."""
        n_nodes = len(tour)
        improved = True
        iterations = 0
        tour = np.array(tour)
        
        while improved and iterations < max_iterations:
            improved = False
            
            # Create all possible 2-opt move indices
            i_indices, j_indices = [], []
            for i in range(n_nodes):
                for j in range(i + 2, n_nodes):
                    i_indices.append(i)
                    j_indices.append(j)
            
            if not i_indices:
                break
                
            # Vectorized gain calculation
            i_arr = np.array(i_indices)
            j_arr = np.array(j_indices)
            
            # Get edge indices in circular tour
            a1_idx = i_arr
            a2_idx = (i_arr + 1) % n_nodes
            b1_idx = j_arr
            b2_idx = (j_arr + 1) % n_nodes
            
            # Get node indices
            a1 = tour[a1_idx]
            a2 = tour[a2_idx]
            b1 = tour[b1_idx]
            b2 = tour[b2_idx]
            
            # Calculate gains vectorized
            gains = (distance_matrix[a1, a2] + distance_matrix[b1, b2]) - \
                    (distance_matrix[a1, b1] + distance_matrix[a2, b2])
            
            # Find best gain
            best_idx = np.argmax(gains)
            best_gain = gains[best_idx]
            
            if best_gain > 1e-6:
                i_opt = i_arr[best_idx]
                j_opt = j_arr[best_idx]
                
                # Apply 2-opt move
                if i_opt < j_opt:
                    tour[i_opt+1:j_opt+1] = tour[i_opt+1:j_opt+1][::-1]
                else:
                    segment = np.concatenate([tour[i_opt+1:], tour[:j_opt+1]])
                    segment = segment[::-1]
                    tour[i_opt+1:] = segment[:n_nodes-i_opt-1]
                    tour[:j_opt+1] = segment[n_nodes-i_opt-1:]
                
                improved = True
                iterations += 1
            else:
                break
        
        return tour.tolist()
    
    # Generate diverse initial tours
    for _ in range(n_tours):
        # Mix of strategies: nearest neighbor and random walks
        if np.random.random() < 0.7:
            # Nearest neighbor from random start
            start = np.random.randint(n)
            tour = [start]
            visited = np.zeros(n, dtype=bool)
            visited[start] = True
            
            current = start
            for _ in range(n - 1):
                mask = ~visited
                if not np.any(mask):
                    break
                    
                distances = distance_matrix[current].copy()
                distances[~mask] = np.inf
                # Add small noise for tie-breaking
                distances[mask] += np.random.rand(np.sum(mask)) * 1e-8
                next_node = np.argmin(distances)
                tour.append(next_node)
                visited[next_node] = True
                current = next_node
        else:
            # Random tour with local optimization bias
            tour = np.random.permutation(n).tolist()
        
        # Improve with vectorized 2-opt
        if len(set(tour)) == n:  # Valid tour
            improved_tour = vectorized_2opt_improvement(tour)
            tour_memory.append(improved_tour)
    
    # ========== MEMORY UPDATE WITH ADAPTIVE WEIGHTS ==========
    # Update edge memory from recent tours with decayed weights
    memory_decay = 0.95  # Favor recent tours
    for tour_idx, tour in enumerate(tour_memory):
        if len(tour) != n:
            continue
            
        # Calculate tour length
        tour_length = sum(distance_matrix[tour[i], tour[(i+1)%n]] for i in range(n))
        # Weight decays for older tours in memory
        weight = (1.0 / (tour_length + 1e-10)) * (memory_decay ** tour_idx)
        
        # Update edge memory
        for i in range(n):
            u, v = tour[i], tour[(i+1)%n]
            key = (min(u, v), max(u, v))
            edge_memory[key] += weight
    
    # ========== STRUCTURAL COMPONENTS ==========
    # 1. Minimum Spanning Tree
    try:
        mst = csgraph.minimum_spanning_tree(distance_matrix)
        mst_mask = mst.toarray() > 0
    except:
        mst_mask = np.zeros_like(distance_matrix, dtype=bool)
        # Fallback: connect nearest neighbors
        for i in range(n):
            row = distance_matrix[i].copy()
            row[i] = np.inf
            j = np.argmin(row)
            mst_mask[i, j] = mst_mask[j, i] = True
    
    # 2. Reciprocal distances with adaptive scaling
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / np.maximum(distance_matrix, 1e-10)
        reciprocal = np.nan_to_num(reciprocal, nan=0, posinf=0, neginf=0)
    
    # Apply scaling based on distance distribution
    scale_exp = 0.5 if dist_cv < 0.3 else 0.8 if dist_cv > 0.7 else 0.65
    reciprocal = reciprocal ** scale_exp
    
    # ========== COMBINE COMPONENTS WITH ADAPTIVE WEIGHTS ==========
    # Memory component
    memory_matrix = np.zeros_like(distance_matrix)
    for (u, v), weight in edge_memory.items():
        memory_matrix[u, v] = memory_matrix[v, u] = weight
    
    # MST component
    mst_matrix = np.zeros_like(distance_matrix)
    mst_matrix[mst_mask] = reciprocal[mst_mask] * 1.5  # Boost MST edges
    
    # Normalize components
    components = [memory_matrix, mst_matrix, reciprocal]
    normalized = []
    for comp in components:
        max_val = comp.max()
        if max_val > 0:
            normalized.append(comp / max_val)
        else:
            normalized.append(np.zeros_like(comp))
    
    # Adaptive weights based on search progress (simulated)
    # More weight to memory as we have more tours
    memory_weight = min(0.5, 0.2 + 0.3 * len(tour_memory) / n_tours)
    mst_weight = 0.3
    recip_weight = 1.0 - memory_weight - mst_weight
    
    # Combine
    raw_heuristic = (memory_weight * normalized[0] + 
                     mst_weight * normalized[1] + 
                     recip_weight * normalized[2])
    
    # Ensure symmetry and zero diagonal
    raw_heuristic = (raw_heuristic + raw_heuristic.T) / 2
    np.fill_diagonal(raw_heuristic, 0)
    
    # ========== ADAPTIVE SPARSIFICATION WITH RUNTIME FEEDBACK ==========
    sparsified = np.zeros_like(raw_heuristic)
    
    # Preserve critical edges
    preserved = set()
    
    # MST edges
    mst_edges = np.column_stack(np.where(mst_mask))
    for i, j in mst_edges:
        if i != j:
            preserved.add((min(i, j), max(i, j)))
    
    # Top memory edges
    if edge_memory:
        sorted_edges = sorted(edge_memory.items(), key=lambda x: x[1], reverse=True)
        top_k = min(len(sorted_edges), max(5, n // 4))
        for (u, v), _ in sorted_edges[:top_k]:
            preserved.add((u, v))
    
    # Fill preserved edges
    for u, v in preserved:
        sparsified[u, v] = sparsified[v, u] = raw_heuristic[u, v]
    
    # Vectorized selection of additional edges per node
    for i in range(n):
        # Current preserved edges for this node
        current_preserved = sum(1 for j in range(n) if (min(i, j), max(i, j)) in preserved)
        
        if current_preserved >= k_edges:
            continue
            
        # Get candidate edges (not preserved)
        row = raw_heuristic[i].copy()
        row[i] = -np.inf
        candidate_mask = np.ones(n, dtype=bool)
        for j in range(n):
            if (min(i, j), max(i, j)) in preserved:
                candidate_mask[j] = False
                row[j] = -np.inf
        
        if not np.any(candidate_mask):
            continue
            
        # Select top additional edges with exponential decay
        needed = k_edges - current_preserved
        indices = np.where(candidate_mask)[0]
        values = row[indices]
        
        if len(values) == 0:
            continue
            
        # Sort descending
        sorted_idx = np.argsort(-values)
        selected_indices = indices[sorted_idx[:needed]]
        selected_values = values[sorted_idx[:needed]]
        
        # Apply exponential decay
        ranks = np.arange(len(selected_values))
        decay = np.exp(-ranks / 2.0)
        sparsified[i, selected_indices] = selected_values * decay
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    # ========== FINAL PROCESSING ==========
    # Add small exploration factor
    exploration = np.zeros_like(sparsified)
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        # Prefer edges with moderate distance (balance exploration)
        med_dist = np.median(distances[distances < np.inf])
        if med_dist > 0:
            exploration[i] = 0.1 * np.exp(-(distances - med_dist)**2 / (2*(med_dist*0.5)**2))
    
    exploration = (exploration + exploration.T) / 2
    np.fill_diagonal(exploration, 0)
    
    result = sparsified + exploration
    
    # Row normalization with fallback
    row_sums = result.sum(axis=1, keepdims=True)
    zero_mask = row_sums.flatten() == 0
    
    if np.any(zero_mask):
        # Use reciprocal as fallback for isolated rows
        result[zero_mask] = reciprocal[zero_mask]
        row_sums = result.sum(axis=1, keepdims=True)
    
    row_sums[row_sums == 0] = 1
    result = result / row_sums
    
    return result
