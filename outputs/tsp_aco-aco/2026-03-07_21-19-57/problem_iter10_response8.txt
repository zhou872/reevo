import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple, Dict
import heapq

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristic with online reinforcement learning, adaptive horizon lookahead,
    and neural-inspired spatial feature extraction.
    
    Core innovations:
    1. Online Reinforcement Learning Weight Adaptation: Iteratively refines component weights
       during heuristic computation using simulated performance feedback
    2. Adaptive Horizon Lookahead with Pruning: Variable-depth regret evaluation with
       intelligent branch pruning for computational efficiency
    3. Graph Neural Network Inspired Spatial Features: Learns node embeddings through
       message passing on constructed graph structures
    4. Complexity-Aware Adaptive Strategy: Dynamically adjusts computational depth
       based on problem size and structure
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # ==================== 1. PRELIMINARY ANALYSIS ====================
    # Compute global statistics
    triu_indices = np.triu_indices(n, k=1)
    global_dists = distance_matrix[triu_indices]
    global_mean = np.mean(global_dists)
    global_std = np.std(global_dists)
    global_cv = global_std / (global_mean + 1e-8)
    
    # Size categorization
    if n <= 50:
        size_category = 'small'
    elif n <= 500:
        size_category = 'medium'
    else:
        size_category = 'large'
    
    # ==================== 2. ONLINE REINFORCEMENT WEIGHT LEARNING ====================
    def evaluate_weight_profile(weights: Dict[str, float], 
                              components: Dict[str, np.ndarray]) -> float:
        """Evaluate a weight profile using simulated greedy tour construction"""
        # Combine components with current weights
        heuristic = combine_components(components, weights)
        
        # Run multiple greedy constructions with different starting points
        num_starts = min(10, n // 10 + 3)
        tour_lengths = []
        
        for start in np.random.choice(n, num_starts, replace=False):
            visited = {start}
            current = start
            tour_length = 0
            
            while len(visited) < n:
                # Get unvisited nodes and their heuristic scores
                unvisited = [j for j in range(n) if j not in visited]
                scores = heuristic[current, unvisited]
                
                # Choose next node based on heuristic (with some exploration)
                if np.random.random() < 0.8:  # 80% exploitation
                    next_node = unvisited[np.argmax(scores)]
                else:  # 20% exploration
                    probs = np.exp(scores) / np.sum(np.exp(scores))
                    next_node = np.random.choice(unvisited, p=probs)
                
                tour_length += distance_matrix[current, next_node]
                visited.add(next_node)
                current = next_node
            
            # Return to start
            tour_length += distance_matrix[current, start]
            tour_lengths.append(tour_length)
        
        return np.mean(tour_lengths)
    
    # ==================== 3. EFFICIENT COMPONENT COMPUTATION ====================
    components = {}
    
    # 3.1 Precompute sorted distances for efficient ranking
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_distances = np.take_along_axis(distance_matrix, sorted_indices, axis=1)
    
    # Remove self-distances
    self_mask = np.ones((n, n), dtype=bool)
    np.fill_diagonal(self_mask, False)
    valid_sorted_indices = sorted_indices[self_mask].reshape(n, n-1)
    valid_sorted_dists = sorted_distances[self_mask].reshape(n, n-1)
    
    # 3.2 Inverse distance with adaptive regularization
    reg_factor = global_mean * 0.1
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + reg_factor)
    np.fill_diagonal(inv_dist, 0)
    components['inv_dist'] = inv_dist
    
    # 3.3 Rank-based incentive
    rank_matrix = np.ones((n, n))
    for i in range(n):
        ranks = np.zeros(n, dtype=int)
        ranks[valid_sorted_indices[i]] = np.arange(n-1)
        rank_bonus = 1.0 + 0.8 * np.exp(-ranks / (n/10))
        degree_estimate = np.sum(distance_matrix[i] < global_mean * 2) / n
        degree_bonus = 1.0 + 0.3 * degree_estimate
        rank_matrix[i] = rank_bonus * degree_bonus
    components['rank'] = rank_matrix
    
    # 3.4 Graph Neural Network Inspired Spatial Features
    def compute_gnn_features(k_neighbors: int = 5, iterations: int = 3) -> np.ndarray:
        """Compute node embeddings through message passing"""
        # Build k-nearest neighbor graph
        knn_adj = np.zeros((n, n), dtype=bool)
        for i in range(n):
            neighbors = valid_sorted_indices[i, :k_neighbors]
            knn_adj[i, neighbors] = True
        knn_adj = knn_adj | knn_adj.T  # Make undirected
        
        # Initialize node embeddings with normalized distances
        embeddings = np.zeros((n, 2))
        for i in range(n):
            # Use normalized distances to nearest neighbors as features
            nearest_dists = valid_sorted_dists[i, :k_neighbors]
            if len(nearest_dists) > 0:
                embeddings[i, 0] = np.mean(nearest_dists) / global_mean
                embeddings[i, 1] = np.std(nearest_dists) / (global_mean + 1e-8)
        
        # Message passing iterations
        for _ in range(iterations):
            new_embeddings = embeddings.copy()
            for i in range(n):
                neighbors = np.where(knn_adj[i])[0]
                if len(neighbors) > 0:
                    # Aggregate neighbor embeddings
                    neighbor_embeds = embeddings[neighbors]
                    # Weight by inverse distance
                    weights = 1.0 / (distance_matrix[i, neighbors] + 1e-8)
                    weights = weights / np.sum(weights)
                    aggregated = np.sum(neighbor_embeds * weights[:, None], axis=0)
                    # Update with residual connection
                    new_embeddings[i] = 0.7 * embeddings[i] + 0.3 * aggregated
            embeddings = new_embeddings
        
        # Compute similarity matrix from embeddings
        gnn_sim = np.zeros((n, n))
        for i in range(n):
            for j in range(i+1, n):
                # Cosine similarity between embeddings
                dot = np.dot(embeddings[i], embeddings[j])
                norm_i = np.linalg.norm(embeddings[i])
                norm_j = np.linalg.norm(embeddings[j])
                if norm_i > 0 and norm_j > 0:
                    sim = dot / (norm_i * norm_j)
                    gnn_sim[i, j] = gnn_sim[j, i] = 1.0 + 0.5 * sim
        
        return gnn_sim
    
    gnn_features = compute_gnn_features(
        k_neighbors=min(10, n-1) if size_category != 'large' else min(5, n-1),
        iterations=3 if size_category != 'large' else 2
    )
    components['gnn'] = gnn_features
    
    # 3.5 ADAPTIVE HORIZON LOOKAHEAD REGRET
    def adaptive_lookahead_regret(max_depth: int = 3) -> np.ndarray:
        """Compute regret scores with adaptive horizon and pruning"""
        lookahead_matrix = np.ones((n, n))
        
        # Determine candidate count based on problem size
        if size_category == 'small':
            candidate_count = min(12, n-2)
            prune_threshold = 0.3
        elif size_category == 'medium':
            candidate_count = min(8, n-2)
            prune_threshold = 0.5
        else:  # large
            candidate_count = min(5, n-2)
            prune_threshold = 0.7
        
        for i in range(n):
            # Get candidate intermediate nodes
            i_candidates = valid_sorted_indices[i, :candidate_count]
            
            for j in range(n):
                if i == j:
                    continue
                
                direct_cost = distance_matrix[i, j]
                best_alternative = float('inf')
                
                # Depth-limited search with pruning
                def dfs_search(current: int, path: Tuple, current_cost: float, 
                             depth: int, best_so_far: float) -> float:
                    if depth == 0 or current_cost >= best_so_far * prune_threshold:
                        return best_so_far
                    
                    if current == j and len(path) > 1:  # Reached target
                        return min(best_so_far, current_cost)
                    
                    # Get next candidates (excluding visited and current)
                    candidates = [k for k in i_candidates 
                                if k not in path and k != current]
                    
                    for k in candidates:
                        new_cost = current_cost + distance_matrix[current, k]
                        if new_cost < best_so_far * prune_threshold:
                            best_so_far = dfs_search(k, path + (k,), new_cost, 
                                                   depth-1, best_so_far)
                    
                    return best_so_far
                
                # Search for alternative paths
                for start_depth in range(2, max_depth + 1):
                    alt_cost = dfs_search(i, (i,), 0, start_depth, float('inf'))
                    if alt_cost < best_alternative:
                        best_alternative = alt_cost
                
                if best_alternative < float('inf'):
                    regret_ratio = best_alternative / (direct_cost + 1e-8)
                    if regret_ratio > 1.0:
                        lookahead_score = 1.0 + np.tanh(regret_ratio - 1.0)
                    else:
                        lookahead_score = 0.8 + 0.2 * regret_ratio
                    lookahead_matrix[i, j] = lookahead_score
        
        # Symmetrize
        lookahead_matrix = (lookahead_matrix + lookahead_matrix.T) / 2
        return lookahead_matrix
    
    lookahead_depth = 3 if size_category != 'large' else 2
    components['lookahead'] = adaptive_lookahead_regret(lookahead_depth)
    
    # 3.6 Density matrix (simplified for efficiency)
    k_neighbors = min(5, n-1) if size_category != 'large' else min(3, n-1)
    local_density = np.zeros(n)
    for i in range(n):
        if n > k_neighbors:
            kth_dist = np.partition(distance_matrix[i], k_neighbors)[k_neighbors]
            nearest_dists = distance_matrix[i, distance_matrix[i] <= kth_dist]
        else:
            nearest_dists = distance_matrix[i, distance_matrix[i] < np.inf]
        
        if len(nearest_dists) > 0:
            local_density[i] = 1.0 / (np.median(nearest_dists) + 1e-8)
    
    if np.max(local_density) > np.min(local_density):
        local_density = (local_density - np.min(local_density)) / \
                       (np.max(local_density) - np.min(local_density))
    
    density_diff = np.abs(local_density[:, None] - local_density[None, :])
    density_matrix = 1.0 + 0.5 * np.exp(-2.0 * density_diff)
    np.fill_diagonal(density_matrix, 1.0)
    components['density'] = density_matrix
    
    # 3.7 Minimum Spanning Tree boost
    try:
        mst = minimum_spanning_tree(distance_matrix)
        mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
        mst_boost = np.where(mst_mask, 1.3, 1.0)
    except:
        mst_boost = np.ones((n, n))
    components['mst'] = mst_boost
    
    # ==================== 4. ONLINE WEIGHT LEARNING ====================
    def combine_components(components: Dict[str, np.ndarray], 
                         weights: Dict[str, float]) -> np.ndarray:
        """Combine components with given weights"""
        result = np.ones((n, n))
        for key, weight in weights.items():
            if key in components and weight > 0:
                result *= np.power(components[key], weight)
        np.fill_diagonal(result, 0)
        return result
    
    # Initial weight profiles
    weight_profiles = [
        {'inv_dist': 1.2, 'rank': 0.8, 'density': 0.5, 
         'lookahead': 0.3, 'gnn': 0.4, 'mst': 1.0},  # Distance-focused
        {'inv_dist': 0.8, 'rank': 1.0, 'density': 1.2, 
         'lookahead': 0.6, 'gnn': 0.8, 'mst': 1.1},  # Structure-focused
        {'inv_dist': 1.0, 'rank': 0.9, 'density': 0.8, 
         'lookahead': 0.5, 'gnn': 0.6, 'mst': 1.0},  # Balanced
    ]
    
    # Evaluate profiles and select best
    best_score = float('inf')
    best_weights = weight_profiles[2]  # Default to balanced
    
    # Use subset for large problems to save computation
    eval_nodes = n
    if size_category == 'large':
        eval_nodes = min(100, n)
        subset_indices = np.random.choice(n, eval_nodes, replace=False)
        # Create subset components
        subset_components = {}
        for key, comp in components.items():
            subset_components[key] = comp[np.ix_(subset_indices, subset_indices)]
        subset_distance = distance_matrix[np.ix_(subset_indices, subset_indices)]
        
        # Temporarily replace for evaluation
        original_components = components.copy()
        original_distance = distance_matrix.copy()
        components = subset_components
        distance_matrix = subset_distance
        n = eval_nodes
    else:
        subset_indices = np.arange(n)
    
    for weights in weight_profiles:
        score = evaluate_weight_profile(weights, components)
        if score < best_score:
            best_score = score
            best_weights = weights
    
    # Restore original components if using subset
    if size_category == 'large':
        components = original_components
        distance_matrix = original_distance
        n = distance_matrix.shape[0]
    
    # ==================== 5. FINAL HEURISTIC COMPUTATION ====================
    # Combine with learned weights
    heuristic = combine_components(components, best_weights)
    
    # Ensure symmetry
    heuristic = (heuristic + heuristic.T) / 2
    
    # ==================== 6. ADAPTIVE SPARSIFICATION ====================
    # Determine edges to keep based on problem size
    if size_category == 'small':
        edges_per_node = min(8, n-1)
    elif size_category == 'medium':
        edges_per_node = min(6, max(3, int(n/15)))
    else:
        edges_per_node = min(4, max(2, int(n/50)))
    
    keep_mask = np.zeros((n, n), dtype=bool)
    
    # Always keep MST edges
    if 'mst_mask' in locals():
        keep_mask = keep_mask | mst_mask
    
    # Keep top edges per node
    for i in range(n):
        scores = heuristic[i].copy()
        scores[i] = -np.inf
        
        if edges_per_node < n-1:
            threshold_idx = np.argpartition(scores, -edges_per_node)[-edges_per_node]
            threshold = scores[threshold_idx]
            node_mask = scores >= threshold
        else:
            node_mask = scores > -np.inf
        
        keep_mask[i] = keep_mask[i] | node_mask
        keep_mask[:, i] = keep_mask[:, i] | node_mask
    
    # Ensure minimum connectivity
    for i in range(n):
        connected = np.where(keep_mask[i])[0]
        connected = connected[connected != i]
        
        if len(connected) < 2:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            dists[keep_mask[i]] = np.inf
            
            if np.any(dists < np.inf):
                for _ in range(2 - len(connected)):
                    if np.all(dists == np.inf):
                        break
                    nearest = np.argmin(dists)
                    keep_mask[i, nearest] = True
                    keep_mask[nearest, i] = True
                    dists[nearest] = np.inf
    
    # Apply sparsification
    heuristic = heuristic * keep_mask
    
    # ==================== 7. FINAL NORMALIZATION ====================
    # Row-wise softmax normalization
    for i in range(n):
        row = heuristic[i]
        if np.any(row > 0):
            exp_row = np.exp(row - np.max(row))
            heuristic[i] = exp_row / (np.sum(exp_row) + 1e-8)
    
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
