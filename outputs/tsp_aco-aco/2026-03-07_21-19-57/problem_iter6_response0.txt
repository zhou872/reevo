import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge prioritization with dynamic evaluation
    and adaptive topological filtering.
    
    Key innovations:
    1. Three-step anticipatory path scoring: Extends lookahead horizon with
       dynamic programming approximation for more accurate future cost estimation
    2. Dynamic edge value propagation: Recursively updates edge values based on
       neighborhood connectivity patterns using message-passing principles
    3. Hierarchical density clustering: Multi-scale density estimation with
       adaptive bandwidth selection for better cluster boundary detection
    4. Adaptive topological sparsification: Graph-theoretic filtering that
       preserves essential connectivity while removing redundant edges
    
    Args:
        distance_matrix: Square symmetric matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Initialize heuristic matrix
    heuristic = np.zeros((n, n))
    np.fill_diagonal(heuristic, 0)
    
    # 1. THREE-STEP ANTICIPATORY PATH SCORING
    lookahead_matrix = np.ones((n, n))
    median_dist = np.median(distance_matrix[distance_matrix > 0])
    
    # Pre-compute sorted neighbors for efficiency
    sorted_neighbors = np.argsort(distance_matrix, axis=1)
    neighbor_distances = np.sort(distance_matrix, axis=1)
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            # Get top candidates for two-step lookahead
            k_candidates = 5
            remaining_nodes = [k for k in sorted_neighbors[j] 
                             if k != i and k != j][:k_candidates]
            
            if not remaining_nodes:
                lookahead_matrix[i, j] = 1.0
                continue
            
            # Three-step cost approximation using dynamic programming principle
            three_step_costs = []
            for k in remaining_nodes:
                # Get candidates for third step (excluding already visited)
                l_candidates = [l for l in sorted_neighbors[k] 
                              if l not in [i, j, k]][:3]
                
                if l_candidates:
                    # Estimate third step cost (minimum expected continuation)
                    third_step_dists = distance_matrix[k, l_candidates]
                    expected_continuation = 0.7 * np.min(third_step_dists) + \
                                           0.3 * np.median(third_step_dists)
                    
                    three_step_cost = distance_matrix[i, j] + \
                                     distance_matrix[j, k] + \
                                     expected_continuation
                    three_step_costs.append(three_step_cost)
                else:
                    three_step_costs.append(distance_matrix[i, j] + distance_matrix[j, k])
            
            if three_step_costs:
                # Weighted combination of best and median three-step paths
                min_cost = np.min(three_step_costs)
                med_cost = np.median(three_step_costs)
                combined_cost = 0.6 * min_cost + 0.4 * med_cost
                
                # Normalize and apply diminishing returns
                if combined_cost > 0:
                    base_score = median_dist / (combined_cost + 1e-8)
                    # Apply sigmoid for stability
                    lookahead_matrix[i, j] = 2.0 / (1.0 + np.exp(-0.5 * base_score))
    
    # 2. DYNAMIC EDGE VALUE PROPAGATION
    # Initial edge values based on reciprocal distance
    with np.errstate(divide='ignore', invalid='ignore'):
        base_values = 1.0 / (distance_matrix + 0.05 * median_dist)
        np.fill_diagonal(base_values, 0)
    
    # Iterative value propagation (2 iterations)
    propagated_values = base_values.copy()
    for _ in range(2):
        new_values = np.zeros((n, n))
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                # Value from i's perspective
                i_neighbors = [k for k in range(n) if k != i and k != j]
                if i_neighbors:
                    # Consider alternative paths through neighbors
                    alt_paths = []
                    for k in i_neighbors:
                        # Path i->k->j (if exists)
                        if propagated_values[k, j] > 0:
                            path_value = propagated_values[i, k] * 0.7 + \
                                       propagated_values[k, j] * 0.3
                            alt_paths.append(path_value)
                    
                    # Current edge value enhanced by best alternative connection
                    if alt_paths:
                        best_alt = np.max(alt_paths)
                        i_perspective = 0.8 * propagated_values[i, j] + \
                                      0.2 * best_alt
                    else:
                        i_perspective = propagated_values[i, j]
                else:
                    i_perspective = propagated_values[i, j]
                
                new_values[i, j] = i_perspective
        
        # Symmetrize and update
        propagated_values = (new_values + new_values.T) / 2
    
    # 3. HIERARCHICAL DENSITY CLUSTERING
    density_matrix = np.ones((n, n))
    
    # Multi-scale density estimation
    density_levels = []
    scale_factors = [0.1, 0.2, 0.3]  # Different neighborhood scales
    
    for scale in scale_factors:
        k = max(2, int(scale * n))
        level_density = np.zeros(n)
        
        for i in range(n):
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            nearest_dists = np.partition(distances, k)[:k]
            nearest_dists = nearest_dists[nearest_dists < np.inf]
            
            if len(nearest_dists) > 0:
                # Gaussian kernel density estimation
                bandwidth = np.std(nearest_dists) if len(nearest_dists) > 1 else 1.0
                if bandwidth > 0:
                    level_density[i] = np.sum(np.exp(-nearest_dists**2 / (2 * bandwidth**2)))
        
        if np.max(level_density) > 0:
            level_density = level_density / np.max(level_density)
            density_levels.append(level_density)
    
    # Combine density levels
    if density_levels:
        combined_density = np.mean(density_levels, axis=0)
        
        # Compute density-aware similarity
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                density_diff = abs(combined_density[i] - combined_density[j])
                # Adaptive similarity based on density levels
                if len(density_levels) > 1:
                    consistency = 1.0 - np.std([level[i] - level[j] 
                                               for level in density_levels])
                else:
                    consistency = 1.0
                
                # Combined similarity measure
                similarity = np.exp(-3.0 * density_diff) * consistency
                density_matrix[i, j] = 1.0 + 0.6 * similarity
    
    # 4. GRAPH-STRUCTURAL COMPONENT
    # Minimum Spanning Tree with edge importance
    mst = minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    mst_mask = (mst_array > 0) | (mst_array.T > 0)
    
    # Compute edge betweenness in MST
    mst_importance = np.ones((n, n))
    if np.any(mst_mask):
        # Build MST graph for betweenness approximation
        for i in range(n):
            for j in range(i+1, n):
                if mst_mask[i, j]:
                    # Simple betweenness approximation: product of subtree sizes
                    # Temporarily remove edge and count reachable nodes
                    visited_i = set([i])
                    stack = [i]
                    while stack:
                        node = stack.pop()
                        neighbors = [k for k in range(n) 
                                   if (mst_mask[node, k] or mst_mask[k, node]) 
                                   and k != j and k not in visited_i]
                        visited_i.update(neighbors)
                        stack.extend(neighbors)
                    
                    visited_j = set([j])
                    stack = [j]
                    while stack:
                        node = stack.pop()
                        neighbors = [k for k in range(n) 
                                   if (mst_mask[node, k] or mst_mask[k, node]) 
                                   and k != i and k not in visited_j]
                        visited_j.update(neighbors)
                        stack.extend(neighbors)
                    
                    size_i = len(visited_i)
                    size_j = len(visited_j)
                    
                    # Importance proportional to balanced cut
                    balance = 4.0 * size_i * size_j / (n * n)
                    mst_importance[i, j] = mst_importance[j, i] = 1.0 + 0.8 * balance
    
    # 5. COMBINE ALL COMPONENTS WITH ADAPTIVE WEIGHTING
    # Dynamic weighting based on graph size
    size_factor = min(1.0, n / 50.0)  # Normalize by reference size
    
    component_weights = {
        'lookahead': 0.8 + 0.2 * size_factor,     # More important for larger graphs
        'propagation': 1.2 - 0.3 * size_factor,   # Less important for larger graphs
        'density': 0.7,
        'mst': 0.9
    }
    
    # Apply weights with softmax-style normalization
    weighted_components = [
        lookahead_matrix ** component_weights['lookahead'],
        propagated_values ** component_weights['propagation'],
        density_matrix ** component_weights['density'],
        mst_importance ** component_weights['mst']
    ]
    
    # Geometric mean of weighted components
    heuristic = np.ones((n, n))
    for component in weighted_components:
        heuristic *= component
    
    heuristic = np.power(heuristic, 1.0 / len(weighted_components))
    np.fill_diagonal(heuristic, 0)
    
    # 6. ADAPTIVE TOPOLOGICAL SPARSIFICATION
    # Keep MST edges as backbone
    final_mask = mst_mask.copy()
    
    # Add edges based on hierarchical thresholding
    for i in range(n):
        # Sort neighbor heuristics
        neighbor_vals = heuristic[i, :].copy()
        neighbor_vals[i] = -np.inf
        
        # Adaptive threshold based on distribution
        valid_vals = neighbor_vals[neighbor_vals > 0]
        if len(valid_vals) == 0:
            continue
        
        # Multi-level thresholding
        thresholds = [
            np.percentile(valid_vals, 50),  # Median
            np.percentile(valid_vals, 70),  # Top 30%
            np.percentile(valid_vals, 85)   # Top 15%
        ]
        
        # Determine appropriate threshold based on node degree in MST
        mst_degree = np.sum(mst_mask[i, :])
        if mst_degree <= 1:
            threshold = thresholds[0]  # Be more inclusive for isolated nodes
        elif mst_degree <= 3:
            threshold = thresholds[1]
        else:
            threshold = thresholds[2]
        
        # Apply threshold
        high_value_mask = neighbor_vals >= threshold
        final_mask[i, high_value_mask] = True
        final_mask[high_value_mask, i] = True
    
    # Ensure minimum connectivity
    for i in range(n):
        connected = np.where(final_mask[i, :])[0]
        connected = connected[connected != i]
        
        if len(connected) < 2:
            # Add nearest neighbors by distance
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            distances[connected] = np.inf
            nearest = np.argsort(distances)[:2]
            
            for nb in nearest:
                if distances[nb] < np.inf:
                    final_mask[i, nb] = True
                    final_mask[nb, i] = True
    
    # Apply sparsification mask
    heuristic = heuristic * final_mask
    
    # 7. FINAL ENHANCEMENTS
    # Smooth with local averaging
    smoothed = heuristic.copy()
    for i in range(n):
        neighbors = np.where(final_mask[i, :])[0]
        if len(neighbors) > 0:
            # Weighted average with immediate neighbors
            neighbor_weights = 1.0 / (distance_matrix[i, neighbors] + 1e-8)
            neighbor_weights = neighbor_weights / np.sum(neighbor_weights)
            
            for j in neighbors:
                if i != j:
                    # Blend with neighbor's perspective
                    neighbor_avg = np.average(heuristic[neighbors, j], weights=neighbor_weights)
                    smoothed[i, j] = 0.7 * heuristic[i, j] + 0.3 * neighbor_avg
    
    heuristic = smoothed
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    return heuristic
