import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with hybrid rank-regret
    scoring, cluster-aware density weighting, and adaptive multi-criteria sparsification.
    
    Core improvements:
    1. Hybrid Rank-Regret Scoring: Combines direct rank bonuses with normalized
       regret ratios to prioritize edges with few good alternatives
    2. Cluster-Aware Density Weighting: Uses local cluster density to favor
       intra-cluster connections while penalizing cross-cluster edges
    3. Multi-Criteria Edge Sparsification: Intelligently prunes unpromising edges
       using rank, regret, density, and proximity thresholds
    4. Adaptive Proximity Thresholding: Dynamic distance thresholds based on
       local neighborhood statistics
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Initialize heuristic matrix
    heuristic = np.zeros((n, n))
    np.fill_diagonal(heuristic, 0)
    
    # 1. HYBRID RANK-REGRET SCORING
    rank_regret_matrix = np.ones((n, n))
    k_nearest = max(3, int(0.1 * n))
    
    for i in range(n):
        # Get distances from node i
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Sort distances and get indices
        sorted_indices = np.argsort(distances)
        sorted_distances = distances[sorted_indices]
        
        # Calculate rank-based scores
        best_dist = sorted_distances[0]
        for rank_idx, j in enumerate(sorted_indices):
            if i == j:
                continue
                
            current_dist = distances[j]
            
            # Rank-based component (simplified)
            rank_component = 1.0
            if rank_idx == 0:  # Best edge
                rank_component = 1.5
            elif rank_idx < k_nearest:  # Top k edges
                rank_component = 1.0 + 0.5 * (1.0 - rank_idx / k_nearest)
            
            # Regret-based component
            if rank_idx == 0:  # Best edge
                # Regret is the difference to second best
                if len(sorted_distances) > 1:
                    regret = (sorted_distances[1] - current_dist) / (current_dist + 1e-8)
                else:
                    regret = 0.5
            else:
                # Regret is negative for non-best edges
                regret = (best_dist - current_dist) / (best_dist + 1e-8)
            
            # Combine rank and regret components
            regret_weight = max(0.1, 1.0 - abs(regret))
            rank_regret_matrix[i, j] = rank_component * (0.7 + 0.3 * regret_weight)
    
    # 2. CLUSTER-AWARE DENSITY WEIGHTING
    density_matrix = np.ones((n, n))
    
    # Compute local cluster density for each node
    local_density = np.zeros(n)
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Use k nearest neighbors for density estimation
        if len(distances) > k_nearest:
            kth_smallest = np.partition(distances, k_nearest)[k_nearest]
            nearest_mask = distances <= kth_smallest
            nearest_dists = distances[nearest_mask]
        else:
            nearest_dists = distances[distances < np.inf]
        
        if len(nearest_dists) > 0:
            # Inverse of mean distance to k-nearest neighbors
            local_density[i] = 1.0 / (np.mean(nearest_dists) + 1e-8)
        else:
            local_density[i] = 1.0
    
    # Normalize densities
    if np.max(local_density) > 0:
        local_density = local_density / np.max(local_density)
    
    # Apply cluster-aware weighting
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            # Calculate density similarity
            density_sim = 1.0 - abs(local_density[i] - local_density[j])
            
            # Boost intra-cluster edges, penalize cross-cluster
            if density_sim > 0.7:  # Similar densities (likely same cluster)
                density_matrix[i, j] = 1.0 + 0.6 * density_sim
            elif density_sim < 0.3:  # Very different densities
                density_matrix[i, j] = 0.4 + 0.6 * density_sim
            else:  # Moderate similarity
                density_matrix[i, j] = 0.8 + 0.4 * density_sim
    
    # 3. MINIMUM SPANNING TREE STRUCTURAL PRIORITY
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    mst_boost = np.where(mst_mask, 1.4, 1.0)  # 40% boost for MST edges
    
    # 4. ADAPTIVE PROXIMITY THRESHOLDING
    proximity_matrix = np.ones((n, n))
    
    # Compute adaptive thresholds per node
    node_thresholds = np.zeros(n)
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Use median of k nearest as threshold base
        if len(distances) > k_nearest:
            k_smallest = np.partition(distances, min(k_nearest, len(distances)-1))
            k_smallest = k_smallest[:min(k_nearest, len(distances)-1)]
            node_thresholds[i] = np.median(k_smallest[k_smallest < np.inf])
        else:
            valid_dists = distances[distances < np.inf]
            node_thresholds[i] = np.median(valid_dists) if len(valid_dists) > 0 else 1.0
    
    # Apply adaptive proximity scoring
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            dist = distance_matrix[i, j]
            # Use harmonic mean of node thresholds for edge threshold
            threshold = 1.8 * (2 * node_thresholds[i] * node_thresholds[j]) / \
                       (node_thresholds[i] + node_thresholds[j] + 1e-8)
            
            if dist <= threshold:
                # Within adaptive threshold
                proximity_score = 1.0 + 0.4 * (1.0 - dist / (threshold + 1e-8))
                proximity_matrix[i, j] = proximity_score
            else:
                # Beyond threshold, apply distance-based penalty
                proximity_matrix[i, j] = 0.6 * (threshold / (dist + 1e-8))
    
    # COMBINE ALL COMPONENTS
    # Base component: inverse distance with smoothing
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + 0.1 * np.median(distance_matrix[distance_matrix > 0]))
        inv_dist[np.diag_indices(n)] = 0
    
    # Combine with adaptive weights based on matrix properties
    mean_dist = np.mean(distance_matrix[distance_matrix > 0])
    cv_dist = np.std(distance_matrix[distance_matrix > 0]) / (mean_dist + 1e-8)
    
    # Adjust component weights based on distance variability
    if cv_dist > 1.0:  # High variability
        weights = [0.4, 0.3, 0.2, 0.1]  # More weight to rank-regret
    else:  # Low variability
        weights = [0.2, 0.3, 0.3, 0.2]  # Balanced weights
    
    # Weighted combination
    heuristic = (
        weights[0] * inv_dist * rank_regret_matrix +
        weights[1] * inv_dist * density_matrix +
        weights[2] * inv_dist * proximity_matrix +
        weights[3] * inv_dist * mst_boost
    )
    
    # Ensure symmetry
    heuristic = (heuristic + heuristic.T) / 2
    
    # 5. MULTI-CRITERIA EDGE SPARSIFICATION
    sparsity_mask = np.zeros((n, n), dtype=bool)
    
    # Always include MST edges
    sparsity_mask = sparsity_mask | mst_mask
    
    # Determine promising edges per node using multiple criteria
    for i in range(n):
        # Criterion 1: Top edges by combined heuristic score
        node_scores = heuristic[i, :].copy()
        node_scores[i] = -np.inf
        
        # Adaptive number of top edges based on n
        top_count = max(3, int(0.2 * np.sqrt(n)))
        top_indices = np.argpartition(node_scores, -top_count)[-top_count:]
        sparsity_mask[i, top_indices] = True
        
        # Criterion 2: Good rank-regret edges
        node_regret = rank_regret_matrix[i, :].copy()
        node_regret[i] = -np.inf
        top_regret = np.argpartition(node_regret, -2)[-2:]
        sparsity_mask[i, top_regret] = True
        
        # Criterion 3: Intra-cluster edges (similar density)
        density_diffs = np.abs(local_density[i] - local_density)
        density_diffs[i] = np.inf
        similar_density = np.where(density_diffs < 0.2)[0]
        if len(similar_density) > 0:
            sparsity_mask[i, similar_density] = True
        
        # Criterion 4: Proximity edges within adaptive threshold
        within_threshold = distance_matrix[i, :] <= (2.0 * node_thresholds[i])
        within_threshold[i] = False
        sparsity_mask[i, within_threshold] = True
    
    # Ensure connectivity: each node connects to at least 2 others
    for i in range(n):
        connected = np.where(sparsity_mask[i, :])[0]
        connected = connected[connected != i]
        if len(connected) < 2:
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            # Add closest nodes not already connected
            remaining = np.where(~sparsity_mask[i, :])[0]
            remaining = remaining[remaining != i]
            if len(remaining) > 0:
                closest_remaining = remaining[np.argsort(distances[remaining])[:2]]
                sparsity_mask[i, closest_remaining] = True
                sparsity_mask[closest_remaining, i] = True
    
    # Symmetrize the mask
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsification
    heuristic = heuristic * sparsity_mask
    
    # Final normalization
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
