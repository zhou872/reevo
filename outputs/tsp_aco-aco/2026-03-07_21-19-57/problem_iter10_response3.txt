import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with adaptive multiplicative
    weights and lookahead edge evaluation.
    
    Key innovations:
    1. Adaptive Multiplicative Scoring: Component weights dynamically adjust based
       on problem characteristics (density, distribution skew)
    2. Two-step Lookahead Edge Evaluation: Estimates edge impact on future tour
       construction by simulating partial tour extension
    3. Density-aware Parameter Tuning: All decay/penalty parameters adapt to
       local and global node distribution patterns
    4. Regret-weighted Structural Importance: Combines MST importance with
       rank-based regret for more nuanced edge prioritization
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Compute global statistics for adaptive parameter tuning
    np.fill_diagonal(distance_matrix, np.inf)
    flat_dists = distance_matrix[distance_matrix < np.inf]
    dist_skew = 0.0
    if len(flat_dists) > 0:
        dist_mean = np.mean(flat_dists)
        dist_std = np.std(flat_dists)
        if dist_std > 0:
            dist_skew = np.mean(((flat_dists - dist_mean) / dist_std) ** 3)
    np.fill_diagonal(distance_matrix, 0)
    
    # 1. ADAPTIVE MULTIPLICATIVE COMPONENT WEIGHTS
    # Weights adjust based on problem size and distance distribution
    if dist_skew > 1.0:
        # Skewed distribution: emphasize rank and MST components more
        w_inv = 0.2
        w_rank = 0.35
        w_density = 0.2
        w_mst = 0.25
    elif dist_skew < -1.0:
        # Negatively skewed: emphasize proximity and density
        w_inv = 0.35
        w_rank = 0.2
        w_density = 0.25
        w_mst = 0.2
    else:
        # Normal distribution: balanced weights
        w_inv = 0.25
        w_rank = 0.3
        w_density = 0.25
        w_mst = 0.2
    
    # Scale weights by problem size
    size_factor = min(1.0, 100 / n)
    w_inv *= size_factor
    w_rank *= size_factor
    w_density *= size_factor
    w_mst *= size_factor
    
    # 2. BASE INVERSE DISTANCE WITH ADAPTIVE REGULARIZATION
    with np.errstate(divide='ignore', invalid='ignore'):
        mean_dist = np.mean(distance_matrix[distance_matrix > 0])
        regularization = 0.05 * mean_dist + 0.01 * mean_dist * abs(dist_skew)
        inv_dist = 1.0 / (distance_matrix + regularization)
        inv_dist[np.diag_indices(n)] = 0
    
    # 3. ENHANCED RANK-BASED REGRET WITH ADAPTIVE DECAY
    rank_matrix = np.ones((n, n))
    
    # Adaptive k for rank consideration based on problem size and skew
    base_k = max(3, int(0.1 * n))
    adaptive_k = base_k * (1 + 0.5 * min(1.0, abs(dist_skew)))
    
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        if len(distances[distances < np.inf]) == 0:
            continue
        
        # Sort distances and get ranks
        sorted_indices = np.argsort(distances)
        sorted_distances = distances[sorted_indices]
        
        best_dist = sorted_distances[0]
        if best_dist <= 0:
            continue
        
        # Adaptive decay parameter based on local distance distribution
        local_skew = 0.0
        if len(sorted_distances) > 1:
            local_mean = np.mean(sorted_distances[:min(10, len(sorted_distances))])
            local_std = np.std(sorted_distances[:min(10, len(sorted_distances))])
            if local_std > 0:
                local_skew = np.mean(((sorted_distances[:min(10, len(sorted_distances))] - local_mean) / local_std) ** 3)
        
        decay_rate = 1.5 + 1.0 * min(1.0, abs(local_skew))
        
        for rank, j in enumerate(sorted_indices):
            if i == j:
                continue
            
            if rank == 0:
                # Best edge: strongest bonus
                rank_matrix[i, j] = 1.8 + 0.2 * min(1.0, local_skew)
            elif rank == 1:
                # Second best: moderate bonus
                rank_matrix[i, j] = 1.5 + 0.1 * min(1.0, local_skew)
            else:
                # Exponential decay based on relative distance
                dist_ratio = distances[j] / best_dist
                rank_matrix[i, j] = 1.0 + 0.8 * np.exp(-decay_rate * (dist_ratio - 1.0))
    
    # 4. TWO-STEP LOOKAHEAD EDGE EVALUATION
    lookahead_matrix = np.ones((n, n))
    
    # Compute MST for structural guidance
    try:
        mst = minimum_spanning_tree(distance_matrix)
        mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
        mst_weight = np.sum(mst.data) if len(mst.data) > 0 else 0
    except:
        mst_mask = np.zeros((n, n), dtype=bool)
        mst_weight = 0
    
    # For each edge, estimate impact on future tour construction
    for i in range(n):
        for j in range(i+1, n):
            if i == j:
                continue
            
            # Base: edge is in MST
            if mst_mask[i, j]:
                mst_importance = 1.6
            else:
                mst_importance = 1.0
            
            # Lookahead component: estimate impact on potential tour completion
            # by considering alternative connections at both endpoints
            lookahead_score = 1.0
            
            # For each endpoint, consider the cost of alternative connections
            for node in [i, j]:
                # Get distances from this node
                distances_from_node = distance_matrix[node, :].copy()
                distances_from_node[node] = np.inf
                distances_from_node[j if node == i else i] = np.inf  # Exclude current edge
                
                if len(distances_from_node[distances_from_node < np.inf]) >= 2:
                    # Find two best alternative edges
                    sorted_dists = np.sort(distances_from_node[distances_from_node < np.inf])
                    if len(sorted_dists) >= 2:
                        alt_cost = sorted_dists[0] + sorted_dists[1]
                        current_cost = distance_matrix[i, j]
                        
                        # If alternatives are much worse, this edge is more important
                        if alt_cost > 0:
                            cost_ratio = alt_cost / (2 * current_cost + 1e-8)
                            # Boost importance if alternatives are significantly worse
                            if cost_ratio > 1.2:
                                lookahead_score *= min(1.3, 1.0 + 0.1 * (cost_ratio - 1.2))
            
            lookahead_matrix[i, j] = mst_importance * lookahead_score
            lookahead_matrix[j, i] = lookahead_matrix[i, j]
    
    # 5. DENSITY-AWARE CLUSTERING WITH ADAPTIVE SIMILARITY
    density_matrix = np.ones((n, n))
    
    # Compute local density with adaptive neighborhood size
    k_density = max(3, int(0.15 * n * (1 + 0.3 * min(1.0, abs(dist_skew)))))
    local_density = np.zeros(n)
    
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        if len(distances[distances < np.inf]) >= k_density:
            kth_smallest = np.partition(distances, k_density)[k_density]
            nearest_dists = distances[distances <= kth_smallest]
        else:
            nearest_dists = distances[distances < np.inf]
        
        if len(nearest_dists) > 0:
            local_density[i] = 1.0 / (np.mean(nearest_dists) + 1e-8)
        else:
            local_density[i] = 1.0
    
    # Normalize densities with robust scaling
    density_q75 = np.percentile(local_density, 75)
    density_q25 = np.percentile(local_density, 25)
    if density_q75 > density_q25:
        local_density_norm = (local_density - density_q25) / (density_q75 - density_q25 + 1e-8)
        local_density_norm = np.clip(local_density_norm, 0, 1)
    else:
        local_density_norm = local_density / (np.max(local_density) + 1e-8)
    
    # Adaptive similarity decay based on global density variation
    density_std = np.std(local_density_norm)
    density_decay = 3.0 + 2.0 * min(1.0, density_std)
    
    # Apply density similarity with adaptive decay
    for i in range(n):
        for j in range(i+1, n):
            if i == j:
                continue
            
            density_diff = abs(local_density_norm[i] - local_density_norm[j])
            similarity = np.exp(-density_decay * density_diff)
            
            # Stronger bonus for edges within similar density regions
            if density_diff < 0.2:
                density_bonus = 1.0 + 0.7 * similarity
            else:
                density_bonus = 1.0 + 0.4 * similarity
            
            density_matrix[i, j] = density_bonus
            density_matrix[j, i] = density_bonus
    
    # 6. COMBINE WITH ADAPTIVE MULTIPLICATIVE WEIGHTS
    # Apply component weights as exponents (multiplicative combination)
    heuristic = (
        np.power(inv_dist, w_inv) *
        np.power(rank_matrix, w_rank) *
        np.power(density_matrix, w_density) *
        np.power(lookahead_matrix, w_mst)
    )
    
    # Ensure symmetry
    heuristic = (heuristic + heuristic.T) / 2.0
    
    # 7. INTELLIGENT ADAPTIVE SPARSIFICATION
    sparsity_mask = np.zeros((n, n), dtype=bool)
    
    # Always preserve MST edges if available
    if np.any(mst_mask):
        sparsity_mask = sparsity_mask | mst_mask
    
    # Compute adaptive thresholds for each node
    threshold_factor = 1.5 + 0.5 * min(1.0, abs(dist_skew))
    adaptive_thresholds = np.zeros(n)
    
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        if len(distances[distances < np.inf]) > 0:
            # Use median distance adjusted by local density
            median_dist = np.median(distances[distances < np.inf])
            density_factor = 0.8 + 0.4 * local_density_norm[i]
            adaptive_thresholds[i] = threshold_factor * density_factor * median_dist
        else:
            adaptive_thresholds[i] = np.inf
    
    # Multi-criteria edge preservation
    for i in range(n):
        # Criterion A: Top edges by combined heuristic (adaptive count)
        node_scores = heuristic[i, :].copy()
        node_scores[i] = -np.inf
        
        preserve_count = max(3, int(0.15 * n * (1 + 0.2 * local_density_norm[i])))
        if preserve_count < len(node_scores):
            threshold = np.partition(node_scores, -preserve_count)[-preserve_count]
            sparsity_mask[i, node_scores >= threshold] = True
        else:
            sparsity_mask[i, :] = True
        
        # Criterion B: Spatial proximity within adaptive threshold
        within_threshold = distance_matrix[i, :] <= adaptive_thresholds[i]
        within_threshold[i] = False
        sparsity_mask[i, within_threshold] = True
        
        # Criterion C: High-regret edges (top by rank matrix)
        node_ranks = rank_matrix[i, :].copy()
        node_ranks[i] = -np.inf
        top_rank_count = max(2, int(0.08 * n))
        if top_rank_count < len(node_ranks):
            rank_threshold = np.partition(node_ranks, -top_rank_count)[-top_rank_count]
            sparsity_mask[i, node_ranks >= rank_threshold] = True
    
    # Ensure symmetry
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Ensure minimum connectivity
    min_connections = 3 if n > 10 else 2
    for i in range(n):
        connected = np.where(sparsity_mask[i, :])[0]
        connected = connected[connected != i]
        if len(connected) < min_connections:
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            
            needed = min_connections - len(connected)
            available = [j for j in range(n) if j != i and not sparsity_mask[i, j]]
            if available:
                available_dists = [(j, distances[j]) for j in available]
                available_dists.sort(key=lambda x: x[1])
                for j, _ in available_dists[:needed]:
                    sparsity_mask[i, j] = True
                    sparsity_mask[j, i] = True
    
    # Apply sparsification
    heuristic = heuristic * sparsity_mask
    np.fill_diagonal(heuristic, 0)
    
    # 8. FINAL NORMALIZATION WITH STABILITY
    # Softmax-like normalization per row for stability
    for i in range(n):
        row = heuristic[i, :]
        if np.any(row > 0):
            row_max = np.max(row)
            if row_max > 0:
                # Shift and softmax
                exp_row = np.exp(row - row_max)
                row_sum = np.sum(exp_row)
                if row_sum > 0:
                    heuristic[i, :] = exp_row / row_sum
    
    # Ensure all values are positive and symmetric
    heuristic = np.maximum(heuristic, 0)
    heuristic = (heuristic + heuristic.T) / 2.0
    
    return heuristic
