import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with regret-based penalization and adaptive weighting.
    
    Core improvements:
    1. Regret-based penalization: Downweights edges with few viable alternatives
    2. Adaptive feature weighting based on instance statistics
    3. Two-phase sparsification: aggressive for large instances, conservative for small
    4. Harmonic centrality measure for density-aware scoring
    
    Returns:
        np.ndarray: Heuristic matrix with higher values for more promising edges
    """
    n = distance_matrix.shape[0]
    
    # === 1. BASE COMPONENTS (similar to v1 but refined) ===
    
    # Normalized inverse distance (stable computation)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-10)
        np.fill_diagonal(inv_dist, 0)
    
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # Minimum Spanning Tree indicator (symmetric)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # Harmonic centrality density measure (more robust than k-NN density)
    # Computes harmonic mean of distances to all other nodes
    harm_centrality = np.zeros(n)
    for i in range(n):
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        if n > 1:
            # Harmonic centrality: 1/(mean of reciprocal distances)
            valid_dists = distance_matrix[i, mask]
            harm_centrality[i] = (n-1) / np.sum(1/(valid_dists + 1e-10))
    
    if harm_centrality.max() > 0:
        harm_centrality = harm_centrality / harm_centrality.max()
    
    # Edge density score: product of endpoint centralities (prioritizes connections between central nodes)
    density_matrix = np.outer(harm_centrality, harm_centrality)
    np.fill_diagonal(density_matrix, 0)
    
    # === 2. REGRET-BASED PENALIZATION ===
    
    # For each node, compute regret: difference between k-th best and best edge
    # Higher regret -> fewer alternatives -> penalize non-best edges more heavily
    regret_penalty = np.ones((n, n))
    k_regret = min(3, n-1)  # Look ahead k alternatives
    
    for i in range(n):
        if n > 1:
            # Get distances to all other nodes
            dists = distance_matrix[i].copy()
            dists[i] = np.inf  # Ignore self
            
            # Sort distances
            sorted_indices = np.argsort(dists)
            sorted_dists = dists[sorted_indices]
            
            if len(sorted_dists) > k_regret:
                # Regret: how much worse is this edge compared to best k alternatives?
                best_k_dist = sorted_dists[k_regret]
                for j in range(n):
                    if j != i:
                        # Penalty increases with how much worse this edge is than best k alternatives
                        regret_ratio = distance_matrix[i, j] / (best_k_dist + 1e-10)
                        # Sigmoid penalty: near 1 for edges close to best-k threshold, lower for better edges
                        regret_penalty[i, j] = 1.0 / (1.0 + np.exp(-5 * (regret_ratio - 1.2)))
    
    # Symmetrize regret penalty
    regret_penalty = np.minimum(regret_penalty, regret_penalty.T)
    
    # === 3. ADAPTIVE WEIGHT TUNING ===
    
    # Analyze instance characteristics
    if n > 2:
        # Measure distance distribution skewness
        triu_dists = distance_matrix[np.triu_indices(n, k=1)]
        dist_skew = np.mean(triu_dists) / (np.median(triu_dists) + 1e-10)
        
        # Measure sparsity of near neighbors
        mean_nearest = np.mean(np.partition(distance_matrix, 2, axis=1)[:, 1])
        mean_second_nearest = np.mean(np.partition(distance_matrix, 3, axis=1)[:, 2])
        neighbor_sparsity = mean_second_nearest / (mean_nearest + 1e-10)
    else:
        dist_skew = 1.0
        neighbor_sparsity = 1.0
    
    # Adaptive weights based on instance characteristics
    # More weight to MST when distances are uniform (low skew)
    # More weight to density when nodes are clustered (high neighbor_sparsity)
    # More weight to regret when there are clear best edges (high neighbor_sparsity)
    
    base_weight = 0.4
    mst_weight = 0.25 + 0.1 * (1.0 / dist_skew)  # Higher for uniform distances
    density_weight = 0.15 + 0.1 * min(neighbor_sparsity - 1.0, 1.0)  # Higher for clustered instances
    regret_weight = 0.2 + 0.1 * min(neighbor_sparsity - 1.0, 0.5)  # Higher when clear best edges exist
    
    # Normalize weights
    total_weight = base_weight + mst_weight + density_weight + regret_weight
    base_weight /= total_weight
    mst_weight /= total_weight
    density_weight /= total_weight
    regret_weight /= total_weight
    
    # === 4. COMBINE COMPONENTS ===
    
    # MST bonus as multiplicative factor
    mst_bonus = np.where(mst_mask, 1.6, 1.0)  # Slightly higher than v1
    
    heuristics = (
        base_weight * inv_dist +
        mst_weight * inv_dist * mst_bonus +
        density_weight * density_matrix -
        regret_weight * regret_penalty * inv_dist  # Apply regret penalty multiplicatively
    )
    
    # Ensure non-negativity
    heuristics = np.maximum(heuristics, 0)
    
    # === 5. ADAPTIVE SPARSIFICATION ===
    
    # Two-phase sparsification: aggressive for large instances, conservative for small
    promising_mask = np.zeros((n, n), dtype=bool)
    
    # Phase 1: Always include MST edges
    promising_mask |= mst_mask
    
    # Phase 2: Include nearest neighbors based on instance size
    if n <= 10:
        # Small instance: keep more edges
        k_sparse = min(4, n-1)
    elif n <= 50:
        # Medium instance: moderate sparsification
        k_sparse = min(3, n-1)
    else:
        # Large instance: aggressive sparsification
        k_sparse = min(2, n-1)
    
    # Include edges to k_sparse nearest neighbors for each node
    for i in range(n):
        if n > 1:
            # Get indices of k_sparse smallest distances (excluding self)
            nearest_indices = np.argpartition(distance_matrix[i], k_sparse+1)[:k_sparse+1]
            nearest_indices = nearest_indices[nearest_indices != i]
            promising_mask[i, nearest_indices] = True
            promising_mask[nearest_indices, i] = True
    
    # Phase 3: For large instances, also include edges with very high heuristics
    if n > 50:
        threshold = np.percentile(heuristics[heuristics > 0], 90) if np.any(heuristics > 0) else 0
        promising_mask |= (heuristics > threshold)
    
    # Apply sparsification
    heuristics = heuristics * promising_mask
    
    # === 6. FINAL NORMALIZATION ===
    
    # Column-wise normalization to emphasize node-specific preferences
    col_max = heuristics.max(axis=0, keepdims=True)
    col_max[col_max == 0] = 1  # Avoid division by zero
    heuristics = heuristics / col_max
    
    # Global normalization to [0, 1]
    if heuristics.max() > 0:
        heuristics = heuristics / heuristics.max()
    
    # Zero out diagonal
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
