import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection combining multiple factors.
    
    Features:
    1. Normalized inverse distance as base weight
    2. Minimum spanning tree edges get priority boost
    3. Sparsification: keep only promising edges (top neighbors + MST)
    4. Local cluster density bonus for edges connecting dense regions
    """
    n = distance_matrix.shape[0]
    
    # 1. Base weight: normalized inverse distance
    # Add small epsilon to avoid division by zero for diagonal
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-10)
        np.fill_diagonal(inv_dist, 0)
    
    # Normalize to [0, 1]
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0  # Make symmetric
    
    # MST edges get 50% bonus
    mst_bonus = np.where(mst_mask, 1.5, 1.0)
    
    # 3. Local cluster density bonus
    # For each node, find average distance to its k nearest neighbors
    k = min(5, n-1)
    densities = np.zeros(n)
    for i in range(n):
        # Get k smallest distances (excluding self)
        nearest_dists = np.partition(distance_matrix[i], k+1)[:k+1]
        nearest_dists = nearest_dists[nearest_dists > 0]  # Remove zero (self)
        densities[i] = 1.0 / (np.mean(nearest_dists) + 1e-10)
    
    # Normalize densities
    if densities.max() > 0:
        densities = densities / densities.max()
    
    # Edge density bonus: harmonic mean of endpoint densities
    density_matrix = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if densities[i] > 0 and densities[j] > 0:
                # Harmonic mean emphasizes low-density connections
                density_matrix[i, j] = density_matrix[j, i] = 2 * densities[i] * densities[j] / (densities[i] + densities[j] + 1e-10)
    
    # 4. Combine factors with weights
    # Weights: [inverse_dist, mst_bonus, density_bonus]
    weights = [0.6, 0.3, 0.1]
    
    heuristics = (
        weights[0] * inv_dist +
        weights[1] * inv_dist * mst_bonus +
        weights[2] * density_matrix
    )
    
    # 5. Sparsification: keep only promising edges
    # Keep: MST edges + edges to top-k nearest neighbors for each node
    k_sparse = min(3, n-1)
    promising_mask = np.zeros((n, n), dtype=bool)
    
    # Include all MST edges
    promising_mask |= mst_mask
    
    # For each node, include edges to k_sparse nearest neighbors
    for i in range(n):
        # Get indices of k_sparse smallest distances (excluding self)
        nearest_indices = np.argpartition(distance_matrix[i], k_sparse+1)[:k_sparse+1]
        nearest_indices = nearest_indices[nearest_indices != i]  # Remove self
        promising_mask[i, nearest_indices] = True
        promising_mask[nearest_indices, i] = True  # Symmetric
    
    # Apply sparsification: zero out unpromising edges
    heuristics = heuristics * promising_mask
    
    # Normalize final heuristics to [0, 1]
    if heuristics.max() > 0:
        heuristics = heuristics / heuristics.max()
    
    return heuristics
