import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.cluster.hierarchy import fcluster, linkage
from scipy.spatial.distance import squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic with adaptive multi-step lookahead,
    normalized regret minimization, clustering-enhanced density,
    and dynamic weight adjustment.
    
    Args:
        distance_matrix: Square matrix of pairwise distances.
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Base inverse distance with safe handling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.diag_indices(n)] = 0
    
    # 1. ADAPTIVE MULTI-STEP LOOKAHEAD
    lookahead_matrix = np.zeros((n, n))
    
    # Determine lookahead depth adaptively based on graph sparsity
    avg_dist = np.mean(distance_matrix[np.triu_indices(n, 1)])
    min_dist = np.min(distance_matrix[np.triu_indices(n, 1)])
    sparsity_ratio = avg_dist / (min_dist + 1e-8)
    lookahead_depth = min(3, max(2, int(np.log2(1 + sparsity_ratio))))
    
    # Precompute sorted distances and indices for efficient lookahead
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_dists = np.take_along_axis(distance_matrix, sorted_indices, axis=1)
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            total_lookahead = distance_matrix[i, j]
            current_node = j
            visited = {i, j}
            
            # Perform greedy lookahead for determined depth
            for step in range(lookahead_depth - 1):
                if len(visited) >= n:
                    break
                    
                # Find best continuation from current node
                for k in sorted_indices[current_node]:
                    if k not in visited:
                        total_lookahead += distance_matrix[current_node, k]
                        current_node = k
                        visited.add(k)
                        break
            
            lookahead_matrix[i, j] = 1.0 / (total_lookahead + 1e-8)
    
    # Normalize lookahead scores
    max_lookahead = np.max(lookahead_matrix)
    if max_lookahead > 0:
        lookahead_matrix = lookahead_matrix / max_lookahead
    
    # 2. NORMALIZED REGRET MINIMIZATION
    regret_matrix = np.ones((n, n))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Compute statistics for normalization
        mean_dist = np.mean(dists[dists < np.inf])
        std_dist = np.std(dists[dists < np.inf])
        if std_dist < 1e-8:
            std_dist = 1e-8
        
        # Find best distance
        best_dist = np.min(dists)
        
        for j in range(n):
            if i == j:
                continue
                
            # Normalized regret: how many standard deviations worse than best
            regret_norm = (distance_matrix[i, j] - best_dist) / std_dist
            # Transform to priority (higher for smaller regret)
            regret_priority = np.exp(-regret_norm * 0.5)
            regret_matrix[i, j] += regret_priority
    
    # Make symmetric and normalize
    regret_matrix = (regret_matrix + regret_matrix.T) / 2
    max_regret = np.max(regret_matrix)
    if max_regret > 0:
        regret_matrix = regret_matrix / max_regret
    
    # 3. CLUSTERING-ENHANCED DENSITY SCORING
    density_matrix = np.ones((n, n))
    
    if n > 4:
        # Perform hierarchical clustering
        condensed_dist = squareform(distance_matrix)
        Z = linkage(condensed_dist, method='average')
        
        # Determine optimal number of clusters adaptively
        n_clusters = max(2, min(8, int(np.sqrt(n))))
        cluster_labels = fcluster(Z, n_clusters, criterion='maxclust')
        
        # Compute intra-cluster and inter-cluster bonuses
        for i in range(n):
            for j in range(i + 1, n):
                if cluster_labels[i] == cluster_labels[j]:
                    # Intra-cluster: reward based on cluster compactness
                    cluster_mask = cluster_labels == cluster_labels[i]
                    cluster_dists = distance_matrix[cluster_mask][:, cluster_mask]
                    cluster_compactness = 1.0 / (np.mean(cluster_dists) + 1e-8)
                    bonus = 1.0 + 0.5 * np.tanh(cluster_compactness / (avg_dist + 1e-8))
                else:
                    # Inter-cluster: reward based on distance between cluster centers
                    ci_mask = cluster_labels == cluster_labels[i]
                    cj_mask = cluster_labels == cluster_labels[j]
                    ci_center = np.mean(distance_matrix[ci_mask], axis=0)
                    cj_center = np.mean(distance_matrix[cj_mask], axis=0)
                    cluster_dist = np.linalg.norm(ci_center - cj_center)
                    bonus = 1.0 + 0.3 * np.exp(-cluster_dist / (2 * avg_dist + 1e-8))
                
                density_matrix[i, j] = density_matrix[j, i] = bonus
    
    # 4. MST WITH ENHANCED CENTRALITY
    mst = minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    mst_mask = (mst_array > 0) | (mst_array.T > 0)
    
    # Compute MST edge importance based on degree and weight
    mst_importance = np.ones((n, n))
    mst_degrees = np.sum(mst_mask, axis=1)
    
    for i in range(n):
        for j in range(i + 1, n):
            if mst_mask[i, j]:
                # Importance based on node degrees and edge weight
                degree_factor = (mst_degrees[i] + mst_degrees[j]) / (2 * np.max(mst_degrees) + 1e-8)
                weight_factor = 1.0 / (1.0 + distance_matrix[i, j] / avg_dist)
                mst_importance[i, j] = mst_importance[j, i] = 1.0 + 0.4 * degree_factor + 0.3 * weight_factor
    
    # 5. DYNAMIC WEIGHT ADJUSTMENT
    # Analyze graph properties for weight tuning
    graph_variability = np.std(distance_matrix[np.triu_indices(n, 1)]) / (avg_dist + 1e-8)
    graph_uniformity = 1.0 / (1.0 + graph_variability)
    
    # Adjust weights based on graph properties
    if graph_uniformity > 0.7:
        # Uniform graphs benefit more from lookahead
        weights = [0.35, 0.25, 0.20, 0.20]  # lookahead, regret, density, mst
    elif n < 20:
        # Small graphs benefit from balanced approach
        weights = [0.30, 0.30, 0.20, 0.20]
    else:
        # Large, non-uniform graphs benefit from clustering and regret
        weights = [0.25, 0.35, 0.25, 0.15]
    
    # Combine components
    combined = inv_dist * (
        weights[0] * lookahead_matrix +
        weights[1] * regret_matrix +
        weights[2] * density_matrix +
        weights[3] * mst_importance
    )
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    # 6. INTELLIGENT SPARSIFICATION WITH CONNECTIVITY GUARANTEE
    sparsified = combined.copy()
    
    # Always keep MST edges
    sparsified[~mst_mask] *= 0.6
    
    # Adaptive threshold per node
    for i in range(n):
        row_vals = sparsified[i].copy()
        row_vals[i] = -np.inf
        if np.sum(row_vals > 0) > 2:
            threshold = np.percentile(row_vals[row_vals > 0], 60)
            sparsified[i][row_vals < threshold] = 0
    
    # Ensure each node connects to at least 2 others
    for i in range(n):
        connections = np.sum(sparsified[i] > 0)
        if connections < 2:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:2]
            sparsified[i, nearest] = combined[i, nearest]
            sparsified[nearest, i] = combined[nearest, i]
    
    # Final normalization
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
