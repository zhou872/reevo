import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with regret-based penalization
    and lookahead spatial scoring.
    
    Features:
    1. Base inverse distance weighting (short edges preferred)
    2. MST edge bonus (structural importance)
    3. Regret-based penalization (edges with few good alternatives get boost)
    4. Two-step lookahead spatial scoring (approximates future path costs)
    5. Adaptive sparsification (focuses computational effort)
    
    Returns: Heuristic matrix of same shape as input, higher values = more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance with robust handling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree bonus (additive component)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = (mst + mst.T > 0).astype(float) * 2.0
    
    # 3. Regret-based penalization
    regret_factor = np.ones((n, n))
    for i in range(n):
        # Get distances from node i to all others (excluding self)
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Find best and second-best distances (alternative edges)
        sorted_indices = np.argsort(dists)
        best_dist = dists[sorted_indices[0]]
        second_best_dist = dists[sorted_indices[1]]
        
        # Compute regret for each edge: how much worse than the best alternative
        for j in range(n):
            if i != j:
                # Regret is the difference between this edge and the best alternative
                # Normalized by the gap between best and second-best
                regret = (distance_matrix[i, j] - best_dist) / (second_best_dist - best_dist + 1e-12)
                # Transform: edges with high regret (few good alternatives) get boost
                regret_factor[i, j] = np.exp(-0.5 * regret)
    
    # Symmetrize regret factor
    regret_factor = np.minimum(regret_factor, regret_factor.T)
    
    # 4. Two-step lookahead spatial scoring
    lookahead_score = np.zeros((n, n))
    
    # Precompute average distances from each node to all others
    avg_dists = np.sum(distance_matrix, axis=1) / (n - 1)
    
    for i in range(n):
        for j in range(n):
            if i != j:
                # Estimate future cost from j after taking edge (i,j)
                # Consider average distance from j to remaining nodes
                remaining_cost = (avg_dists[j] * (n - 1) - distance_matrix[j, i]) / (n - 2) if n > 2 else avg_dists[j]
                
                # Also consider spatial clustering: if i and j are in dense regions
                # Compute local density around both endpoints
                k_local = max(2, n // 15)
                nearest_i = np.sort(distance_matrix[i])[1:k_local+1]
                nearest_j = np.sort(distance_matrix[j])[1:k_local+1]
                density_i = 1 / (np.mean(nearest_i) + 1e-12)
                density_j = 1 / (np.mean(nearest_j) + 1e-12)
                density_factor = (density_i + density_j) / 2
                
                # Combine into lookahead score (lower future cost = higher score)
                lookahead_score[i, j] = 1 / (1 + 0.3 * remaining_cost) * (0.7 + 0.3 * density_factor / np.max([density_factor, 1e-12]))
    
    # Normalize lookahead scores to [0, 1]
    max_lookahead = np.max(lookahead_score)
    if max_lookahead > 0:
        lookahead_score = lookahead_score / max_lookahead
    
    # 5. Combine all factors with hybrid multiplicative-additive approach
    # Base combination
    combined = inv_dist * (1.0 + 0.4 * mst_bonus) * regret_factor
    
    # Add lookahead component (weighted addition to maintain scale)
    combined = combined * (0.7 + 0.3 * lookahead_score)
    
    # 6. Adaptive sparsification
    # Use k-nearest neighbors with adaptive k
    k_sparse = max(3, int(0.12 * n))
    sparsity_mask = np.zeros((n, n))
    
    for i in range(n):
        # Get k+1 nearest (including self)
        nearest = np.argsort(distance_matrix[i])[:k_sparse+1]
        sparsity_mask[i, nearest] = 1
        # Keep reciprocal edges
        for j in nearest:
            if j != i:
                sparsity_mask[j, i] = 1
    
    np.fill_diagonal(sparsity_mask, 0)
    
    # Apply initial sparsification
    combined = combined * sparsity_mask
    
    # Additional sparsification: keep only top edges per node
    # Adaptive threshold based on node degree
    for i in range(n):
        row = combined[i]
        non_zero = row[row > 0]
        if len(non_zero) > 0:
            # Keep more edges for nodes in sparse regions
            avg_dist = np.mean(distance_matrix[i, distance_matrix[i] > 0])
            percentile = 65 if avg_dist > np.median(distance_matrix) else 70
            threshold = np.percentile(non_zero, percentile)
            row[row < threshold] = 0
        combined[i] = row
    
    # Ensure symmetry
    combined = np.maximum(combined, combined.T)
    
    # 7. Final normalization to probability distribution per row
    row_sums = combined.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    combined = combined / row_sums
    
    # Apply softmax transformation for better probability interpretation
    # Temperature parameter adapts to problem size
    temperature = 0.1 * np.sqrt(n)
    combined = np.exp(combined / temperature)
    np.fill_diagonal(combined, 0)
    
    row_sums = combined.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    combined = combined / row_sums
    
    return combined
