import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.stats import skew, kurtosis

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic with variance-calibrated adaptive weighting,
    geometric-harmonic mean fusion, and enhanced two-step lookahead regret.
    
    Core innovations:
    1. Variance-Adaptive Component Balancing: Dynamically adjusts heuristic component
       weights based on comprehensive instance statistics including higher moments
       (skewness, kurtosis) for more nuanced instance characterization
    2. Two-Step Lookahead with Rank-Based Regret: Evaluates opportunity costs at both
       edge endpoints with adaptive search depth, prioritizing edges with fewer good
       alternatives
    3. Multi-Criteria Sparsification with Connectivity Guarantee: Maintains solution
       feasibility while focusing computational effort on most promising edges
    4. Geometric-Harmonic Mean Fusion: Combines components using a weighted blend of
       geometric and harmonic means to balance influence and prevent dominance
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate more
        promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # =====================================================================
    # 1. COMPREHENSIVE INSTANCE CHARACTERIZATION
    # =====================================================================
    # Extract non-diagonal distances for analysis
    non_diag_mask = ~np.eye(n, dtype=bool)
    global_dists = distance_matrix[non_diag_mask]
    
    # Basic statistics
    global_mean = np.mean(global_dists)
    global_std = np.std(global_dists)
    global_cv = global_std / (global_mean + 1e-8)
    
    # Higher moments for nuanced characterization
    if len(global_dists) > 3:
        global_skew = skew(global_dists)
        global_kurt = kurtosis(global_dists)
    else:
        global_skew = 0.0
        global_kurt = 0.0
    
    # Distance distribution shape analysis
    q25, q50, q75 = np.percentile(global_dists, [25, 50, 75])
    iqr = q75 - q25
    
    # Sparsity and outlier analysis
    long_edge_ratio = np.sum(global_dists > q75 + 1.5 * iqr) / len(global_dists)
    dense_edge_ratio = np.sum(global_dists < q25 - 0.5 * iqr) / len(global_dists)
    
    # Problem size factor (logarithmic scaling)
    size_factor = np.log10(n + 1) / 2.0
    
    # =====================================================================
    # 2. ADAPTIVE PARAMETERS FROM INSTANCE FEATURES
    # =====================================================================
    # Variance factor: higher for more variable instances
    variance_factor = min(2.0, global_cv * 1.5)
    
    # Skewness-aware adjustment: positive skew means long tail of large distances
    skew_factor = max(0.5, 1.0 - abs(global_skew) / 3.0)
    
    # Kurtosis-aware adjustment: high kurtosis means peaked or heavy-tailed
    kurt_factor = 1.0 + min(0.5, abs(global_kurt) / 10.0)
    
    # Sparsity factor: higher for instances with many long edges
    sparsity_factor = 0.5 + 0.5 * long_edge_ratio
    
    # Density factor: higher for instances with many very short edges
    density_factor = 0.5 + 0.5 * dense_edge_ratio
    
    # Complexity factor: combines multiple features
    complexity_factor = 0.3 * variance_factor + 0.2 * sparsity_factor + \
                        0.2 * density_factor + 0.15 * skew_factor + 0.15 * kurt_factor
    
    # =====================================================================
    # 3. ENHANCED TWO-STEP LOOKAHEAD REGRET
    # =====================================================================
    # Adaptive search depth based on instance complexity and size
    base_depth = max(3, min(8, int(np.sqrt(n))))
    lookahead_depth = base_depth + int(3 * complexity_factor)
    
    # Pre-sort distances for efficient lookahead
    sorted_indices = np.zeros((n, n-1), dtype=int)
    sorted_dists = np.zeros((n, n-1))
    
    for i in range(n):
        dists = distance_matrix[i, :].copy()
        dists[i] = np.inf
        sorted_idx = np.argsort(dists)
        sorted_indices[i] = sorted_idx[:n-1]
        sorted_dists[i] = dists[sorted_idx[:n-1]]
    
    # Enhanced regret computation with adaptive depth
    regret_matrix = np.ones((n, n))
    
    for i in range(n):
        best_dist = sorted_dists[i, 0]
        
        for rank, j in enumerate(sorted_indices[i]):
            if i == j:
                continue
                
            current_dist = distance_matrix[i, j]
            
            # Base rank-based score
            base_score = best_dist / (current_dist + 1e-8) if best_dist > 0 else 1.0
            
            # Enhanced two-step regret with adaptive depth
            # Consider alternatives at both endpoints
            alt_score_i = 1.0
            alt_score_j = 1.0
            
            # Alternative at i (excluding j)
            alt_count_i = 0
            for idx in range(1, min(lookahead_depth + 1, n-1)):  # Skip best (0)
                if sorted_indices[i, idx] == j:
                    continue
                alt_dist = sorted_dists[i, idx]
                alt_score_i = min(alt_score_i, alt_dist / (current_dist + 1e-8))
                alt_count_i += 1
                if alt_count_i >= lookahead_depth:
                    break
            
            # Alternative at j (excluding i)
            alt_count_j = 0
            j_best_idx = 0
            while j_best_idx < n-1 and sorted_indices[j, j_best_idx] == i:
                j_best_idx += 1
            
            for idx in range(j_best_idx, min(j_best_idx + lookahead_depth, n-1)):
                if sorted_indices[j, idx] == i:
                    continue
                alt_dist = sorted_dists[j, idx]
                alt_score_j = min(alt_score_j, alt_dist / (current_dist + 1e-8))
                alt_count_j += 1
                if alt_count_j >= lookahead_depth:
                    break
            
            # Combined regret factor with weights based on instance characteristics
            regret_weight_i = 0.4 + 0.3 * density_factor  # More weight on i in dense instances
            regret_weight_j = 0.4 + 0.3 * sparsity_factor  # More weight on j in sparse instances
            
            combined_regret = (regret_weight_i * alt_score_i + 
                             regret_weight_j * alt_score_j) / (regret_weight_i + regret_weight_j)
            
            # Rank penalty with adaptive decay
            rank_penalty = rank / (n - 2) if n > 2 else 0
            penalty_decay = 0.5 + 0.3 * complexity_factor  # Slower decay for complex instances
            adjusted_penalty = rank_penalty * penalty_decay
            
            regret_matrix[i, j] = base_score * combined_regret * (1.0 - adjusted_penalty)
    
    # =====================================================================
    # 4. ADAPTIVE LOCAL STRUCTURE ANALYSIS
    # =====================================================================
    # Compute adaptive neighborhood size based on instance characteristics
    base_k = max(3, int(0.1 * n))
    k = min(base_k + int(5 * variance_factor), n - 1)
    
    # Enhanced local features with multiple distance metrics
    local_features = np.zeros((n, 3))  # [mean_dist, min_dist, concentration]
    
    for i in range(n):
        dists = distance_matrix[i, :].copy()
        dists[i] = np.inf
        
        if n > k:
            # Get k-nearest distances
            kth_dist = np.partition(dists, k)[k]
            neighbor_mask = dists <= kth_dist
            neighbor_dists = dists[neighbor_mask]
        else:
            neighbor_dists = dists[dists < np.inf]
        
        if len(neighbor_dists) > 0:
            local_features[i, 0] = np.mean(neighbor_dists)
            local_features[i, 1] = np.min(neighbor_dists)
            # Concentration: ratio of mean to min (higher = more spread out)
            local_features[i, 2] = local_features[i, 0] / (local_features[i, 1] + 1e-8)
    
    # Normalize features with instance-adaptive scaling
    for feat_idx in range(3):
        feat = local_features[:, feat_idx]
        if np.max(feat) > np.min(feat):
            # Scale based on feature importance
            scale = 1.0 + 0.5 * variance_factor if feat_idx == 2 else 1.0
            local_features[:, feat_idx] = scale * (feat - np.min(feat)) / (np.max(feat) - np.min(feat))
    
    # Feature similarity matrix with adaptive weighting
    structure_matrix = np.ones((n, n))
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            # Weighted feature similarity
            weights = [0.3, 0.4, 0.3]  # Weights for [mean, min, concentration]
            weighted_sim = 0.0
            
            for feat_idx in range(3):
                feat_diff = abs(local_features[i, feat_idx] - local_features[j, feat_idx])
                feat_sim = 1.0 / (1.0 + feat_diff)
                weighted_sim += weights[feat_idx] * feat_sim
            
            # Distance-based adjustment
            norm_dist = distance_matrix[i, j] / (global_mean + 1e-8)
            dist_factor = np.exp(-norm_dist)
            
            structure_matrix[i, j] = weighted_sim * (0.3 + 0.7 * dist_factor)
    
    # =====================================================================
    # 5. MINIMUM SPANNING TREE WITH ENHANCED CENTRALITY
    # =====================================================================
    try:
        mst = minimum_spanning_tree(distance_matrix)
        mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
        
        # Compute MST edge betweenness with size adaptation
        mst_adj = [np.where(mst_mask[i])[0] for i in range(n)]
        
        # Simplified centrality: edges that appear in many shortest paths
        # For trees, edge centrality = size(left_component) * size(right_component)
        centrality_matrix = np.ones((n, n))
        
        for i in range(n):
            for j in mst_adj[i]:
                if i < j:
                    # Simple approximation: boost MST edges more in sparse instances
                    boost = 1.2 + 0.3 * sparsity_factor
                    centrality_matrix[i, j] = centrality_matrix[j, i] = boost
    except:
        mst_mask = np.zeros((n, n), dtype=bool)
        centrality_matrix = np.ones((n, n))
    
    # =====================================================================
    # 6. VARIANCE-ADAPTIVE COMPONENT BALANCING
    # =====================================================================
    # Dynamic weights based on comprehensive instance analysis
    # Component base importance
    base_weights = {
        'regret': 0.35,  # Core regret component
        'structure': 0.30,  # Local structure
        'centrality': 0.20,  # MST-based centrality
        'distance': 0.15  # Inverse distance
    }
    
    # Adjustment factors from instance features
    adjustments = {
        'regret': 1.0 + 0.4 * complexity_factor,  # More important in complex instances
        'structure': 1.0 + 0.3 * density_factor,  # More important in dense instances
        'centrality': 1.0 + 0.2 * sparsity_factor,  # More important in sparse instances
        'distance': 1.0 - 0.2 * variance_factor  # Less important in high-variance instances
    }
    
    # Apply adjustments
    final_weights = {}
    for key in base_weights:
        final_weights[key] = base_weights[key] * adjustments[key]
    
    # Normalize to sum to 1
    weight_sum = sum(final_weights.values())
    for key in final_weights:
        final_weights[key] /= weight_sum
    
    # =====================================================================
    # 7. GEOMETRIC-HARMONIC MEAN FUSION
    # =====================================================================
    # Compute inverse distance component
    reg_coef = 0.01 * global_mean * (1.0 + variance_factor)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + reg_coef)
        inv_dist[np.diag_indices(n)] = 0
    
    # Ensure positivity
    regret_pos = np.maximum(regret_matrix, 1e-8)
    structure_pos = np.maximum(structure_matrix, 1e-8)
    centrality_pos = np.maximum(centrality_matrix, 1e-8)
    inv_dist_pos = np.maximum(inv_dist, 1e-8)
    
    # Weighted geometric mean
    geometric_mean = (
        np.power(regret_pos, final_weights['regret']) *
        np.power(structure_pos, final_weights['structure']) *
        np.power(centrality_pos, final_weights['centrality']) *
        np.power(inv_dist_pos, final_weights['distance'])
    )
    
    # Weighted harmonic mean (for components where low values are bad)
    harmonic_mean = (
        final_weights['regret'] + final_weights['structure'] + 
        final_weights['centrality'] + final_weights['distance']
    ) / (
        final_weights['regret'] / regret_pos + 
        final_weights['structure'] / structure_pos + 
        final_weights['centrality'] / centrality_pos + 
        final_weights['distance'] / inv_dist_pos + 1e-8
    )
    
    # Adaptive blending: use more harmonic mean for instances with high variance
    blend_factor = 0.3 + 0.4 * variance_factor  # More harmonic for high variance
    combined = (1.0 - blend_factor) * geometric_mean + blend_factor * harmonic_mean
    
    # Ensure symmetry and zero diagonal
    combined = (combined + combined.T) / 2.0
    np.fill_diagonal(combined, 0)
    
    # =====================================================================
    # 8. MULTI-CRITERIA SPARSIFICATION
    # =====================================================================
    # Adaptive edge retention based on instance characteristics
    if n <= 15:
        min_edges = max(3, n // 2)
        max_edges = n - 1
    else:
        base_min = max(4, int(0.2 * n))
        base_max = max(8, int(0.3 * n))
        
        # Adjust based on complexity
        min_edges = int(base_min * (1.0 + 0.3 * complexity_factor))
        max_edges = int(base_max * (1.0 + 0.3 * complexity_factor))
    
    keep_mask = np.zeros((n, n), dtype=bool)
    
    # Always keep MST edges
    keep_mask = keep_mask | mst_mask
    
    # Multi-criteria edge selection
    for i in range(n):
        # Collect scores from multiple criteria
        criteria_scores = []
        
        # Criterion 1: Combined heuristic score
        criteria_scores.append(combined[i, :])
        
        # Criterion 2: Regret-based score
        criteria_scores.append(regret_matrix[i, :])
        
        # Criterion 3: Distance-based score
        dist_scores = 1.0 / (distance_matrix[i, :] + 0.1 * global_mean)
        dist_scores[i] = 0
        criteria_scores.append(dist_scores)
        
        # Combine criteria with instance-adaptive weights
        crit_weights = [0.5, 0.3, 0.2]  # Base weights
        if variance_factor > 1.0:
            crit_weights = [0.4, 0.4, 0.2]  # More weight to regret in high variance
        
        combined_scores = np.zeros(n)
        for idx, (score, weight) in enumerate(zip(criteria_scores, crit_weights)):
            if np.max(score) > np.min(score):
                normalized = (score - np.min(score)) / (np.max(score) - np.min(score))
                combined_scores += weight * normalized
        
        combined_scores[i] = -np.inf  # Exclude self
        
        # Select edges
        target = min(max_edges, n-1)
        top_indices = np.argpartition(combined_scores, -target)[-target:]
        keep_mask[i, top_indices] = True
    
    # Ensure symmetry
    keep_mask = keep_mask | keep_mask.T
    
    # Ensure minimum connectivity
    for i in range(n):
        connections = np.sum(keep_mask[i, :])
        if connections < min_edges:
            # Add nearest neighbors
            dists = distance_matrix[i, :].copy()
            dists[i] = np.inf
            candidates = np.argsort(dists)
            
            added = 0
            for j in candidates:
                if j != i and not keep_mask[i, j]:
                    keep_mask[i, j] = keep_mask[j, i] = True
                    added += 1
                    if connections + added >= min_edges:
                        break
    
    # Ensure global connectivity using BFS
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        current = stack.pop()
        neighbors = np.where(keep_mask[current, :])[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    # Connect any disconnected components
    if not np.all(visited):
        # Find all components
        components = []
        comp_labels = -np.ones(n, dtype=int)
        comp_id = 0
        
        for i in range(n):
            if comp_labels[i] == -1:
                comp = []
                stack = [i]
                comp_labels[i] = comp_id
                
                while stack:
                    node = stack.pop()
                    comp.append(node)
                    
                    neighbors = np.where(keep_mask[node, :])[0]
                    for neighbor in neighbors:
                        if comp_labels[neighbor] == -1:
                            comp_labels[neighbor] = comp_id
                            stack.append(neighbor)
                
                components.append(comp)
                comp_id += 1
        
        # Connect components with shortest edges weighted by heuristic score
        for i in range(len(components) - 1):
            comp_a = components[i]
            comp_b = components[i + 1]
            
            best_score = -np.inf
            best_pair = (0, 0)
            
            for a in comp_a:
                for b in comp_b:
                    edge_score = combined[a, b] / (distance_matrix[a, b] + 1e-8)
                    if edge_score > best_score:
                        best_score = edge_score
                        best_pair = (a, b)
            
            keep_mask[best_pair] = keep_mask[best_pair[::-1]] = True
    
    # Apply sparsification
    heuristic = combined * keep_mask
    
    # =====================================================================
    # 9. FINAL ADAPTIVE NORMALIZATION
    # =====================================================================
    # Softmax normalization per node with temperature adaptation
    for i in range(n):
        row = heuristic[i, :]
        mask = row > 0
        
        if np.sum(mask) > 0:
            # Adaptive temperature based on row statistics
            row_vals = row[mask]
            row_mean = np.mean(row_vals)
            row_std = np.std(row_vals)
            
            # Higher temperature for more uniform rows, lower for peaked rows
            temperature = 1.0 + 0.5 * (row_std / (row_mean + 1e-8))
            
            # Temperature-scaled softmax
            scaled_vals = row_vals / temperature
            scaled_vals = scaled_vals - np.max(scaled_vals)  # For numerical stability
            exp_vals = np.exp(scaled_vals)
            heuristic[i, mask] = exp_vals / np.sum(exp_vals)
    
    # Final symmetry and zero diagonal
    heuristic = (heuristic + heuristic.T) / 2.0
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
