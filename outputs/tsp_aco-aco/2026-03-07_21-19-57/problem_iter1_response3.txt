import numpy as np
from scipy.spatial import distance_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection probability.
    Combines multiple factors and sparsifies the matrix.
    """
    n = distance_matrix.shape[0]
    
    # 1. Basic inverse distance (baseline heuristic)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        np.fill_diagonal(inv_dist, 0)
    
    # 2. Nearest-neighbor sparsification: keep only k nearest neighbors for each node
    k = max(2, int(np.sqrt(n)))  # Adaptive k based on problem size
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k+1 nearest neighbors (including self)
        indices = np.argpartition(distance_matrix[i], k+1)[:k+1]
        # Remove self from neighbors
        indices = indices[indices != i]
        nn_mask[i, indices] = True
    
    # 3. Betweenness centrality approximation: edges connecting dense clusters
    centrality = np.zeros(n)
    for i in range(n):
        # Count how many shortest paths would use node i as intermediate
        # Simplified: nodes that are close to many other nodes get higher centrality
        close_nodes = np.sum(distance_matrix[i] < np.percentile(distance_matrix[i], 30))
        centrality[i] = close_nodes
    
    # 4. Create edge scores combining multiple factors
    edge_scores = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(i+1, n):
            if not (nn_mask[i, j] or nn_mask[j, i]):
                continue
                
            # Factor 1: Inverse distance (normalized)
            dist_factor = inv_dist[i, j]
            
            # Factor 2: Node centrality product (edges between central nodes are better)
            centrality_factor = (centrality[i] * centrality[j]) / (n ** 2)
            
            # Factor 3: Rank-based normalization
            # How much better is this edge compared to alternatives from both ends
            rank_i = np.sum(distance_matrix[i] < distance_matrix[i, j]) / (n - 1)
            rank_j = np.sum(distance_matrix[j] < distance_matrix[j, i]) / (n - 1)
            rank_factor = 1 - (rank_i + rank_j) / 2
            
            # Factor 4: Triangle inequality penalty
            # Penalize edges that violate triangle inequality severely
            min_via_k = np.min(distance_matrix[i] + distance_matrix[:, j])
            triangle_factor = distance_matrix[i, j] / (min_via_k + 1e-10)
            triangle_factor = np.clip(2 - triangle_factor, 0, 1)
            
            # Combine factors with weights
            score = (dist_factor * 0.4 + 
                    centrality_factor * 0.2 + 
                    rank_factor * 0.3 + 
                    triangle_factor * 0.1)
            
            edge_scores[i, j] = edge_scores[j, i] = score
    
    # 5. Two-stage filtering
    # First stage: Keep only edges where score > threshold
    threshold = np.percentile(edge_scores[edge_scores > 0], 40)
    edge_scores[edge_scores <= threshold] = 0
    
    # Second stage: Ensure each node has at least 2 connections
    for i in range(n):
        non_zero = np.sum(edge_scores[i] > 0)
        if non_zero < 2:
            # Add back the best missing connections
            mask = edge_scores[i] == 0
            mask[i] = False
            if np.any(mask):
                best_idx = np.argmax(inv_dist[i][mask])
                valid_indices = np.where(mask)[0]
                edge_scores[i, valid_indices[best_idx]] = inv_dist[i, valid_indices[best_idx]]
                edge_scores[valid_indices[best_idx], i] = inv_dist[i, valid_indices[best_idx]]
    
    # 6. Final normalization
    row_sums = edge_scores.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1  # Avoid division by zero
    edge_scores = edge_scores / row_sums
    
    return edge_scores
