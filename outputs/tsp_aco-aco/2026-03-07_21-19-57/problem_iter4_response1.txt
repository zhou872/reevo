import numpy as np
from heapq import heappush, heappop
from scipy.sparse.csgraph import minimum_spanning_tree
from collections import defaultdict
import math

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Next-generation TSP edge heuristics with enhanced betweenness approximation,
    multi-step regret scoring, meta-learned feature weighting, and temporal feedback.
    
    Key improvements:
    1. Fast approximated edge betweenness using sparsified Brandes algorithm
    2. Multi-step regret considering k-alternatives for better opportunity cost
    3. Meta-learned feature weights based on instance characteristics
    4. Temporal feedback simulation for adaptive edge evaluation
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + 1e-10)
    
    # 1. BASE COMPONENTS WITH IMPROVED STABILITY
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. SPARSE GRAPH CONSTRUCTION WITH ADAPTIVE NEIGHBORHOOD
    # Adaptive k based on instance size and density
    avg_dist = np.mean(distance_matrix[distance_matrix > 0])
    std_dist = np.std(distance_matrix[distance_matrix > 0])
    density_ratio = std_dist / (avg_dist + 1e-10)
    
    k_neighbors = max(2, min(8, int(4 + 4 * (1 - min(1.0, density_ratio)))))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Build neighborhood mask with tie-breaking
    for i in range(n):
        # Get k nearest neighbors
        indices = np.argpartition(distance_matrix[i], k_neighbors + 1)[:k_neighbors + 1]
        indices = indices[indices != i][:k_neighbors]
        nn_mask[i, indices] = True
        nn_mask[indices, i] = True
    
    # 3. MINIMUM SPANNING TREE EDGES
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # Combine promising edges
    promising_mask = nn_mask | mst_mask
    
    # 4. FAST APPROXIMATED EDGE BETWEENNESS (Brandes-inspired)
    betweenness = np.zeros_like(distance_matrix)
    
    if n > 10:
        # Adaptive sampling: more samples for larger/denser graphs
        sample_size = max(3, min(int(math.sqrt(n)), 
                                 int(10 + n * 0.05 * density_ratio)))
        sources = np.random.choice(n, min(sample_size, n), replace=False)
        
        # Build adjacency list for sparse graph
        adj = [[] for _ in range(n)]
        edge_index = {}
        for i in range(n):
            for j in range(i + 1, n):
                if promising_mask[i, j]:
                    adj[i].append((j, distance_matrix[i, j]))
                    adj[j].append((i, distance_matrix[i, j]))
                    edge_index[(min(i, j), max(i, j))] = (i, j)
        
        # Run Brandes-like algorithm for sampled sources
        for s in sources:
            # Dijkstra from source s
            dist = [float('inf')] * n
            dist[s] = 0
            sigma = [0] * n  # number of shortest paths
            sigma[s] = 1
            pred = [[] for _ in range(n)]
            
            # Priority queue
            heap = [(0, s)]
            
            while heap:
                d, u = heappop(heap)
                if d > dist[u]:
                    continue
                    
                for v, w in adj[u]:
                    alt = d + w
                    if alt < dist[v] - 1e-10:
                        dist[v] = alt
                        sigma[v] = sigma[u]
                        pred[v] = [u]
                        heappush(heap, (alt, v))
                    elif abs(alt - dist[v]) < 1e-10:
                        sigma[v] += sigma[u]
                        pred[v].append(u)
            
            # Backward accumulation using stack
            stack = []
            visited = [False] * n
            visited[s] = True
            
            # Get nodes in order of decreasing distance
            nodes_by_dist = sorted(range(n), key=lambda x: dist[x], reverse=True)
            
            delta = [0.0] * n
            for w in nodes_by_dist:
                if w == s:
                    continue
                stack.append(w)
                for v in pred[w]:
                    # Edge contribution
                    key = (min(v, w), max(v, w))
                    if key in edge_index:
                        i, j = edge_index[key]
                        contrib = (sigma[v] / sigma[w]) * (1 + delta[w])
                        betweenness[i, j] += contrib
                        betweenness[j, i] += contrib
                    delta[v] += (sigma[v] / sigma[w]) * (1 + delta[w])
    
    # Normalize betweenness
    if np.max(betweenness) > 0:
        betweenness = betweenness / np.max(betweenness)
    
    # 5. MULTI-STEP REGRET SCORING
    regret_scores = np.zeros_like(distance_matrix)
    
    # Compute k-best alternatives for each node
    k_regret = min(4, n - 1)
    k_best_dists = np.zeros((n, k_regret))
    k_best_indices = np.zeros((n, k_regret), dtype=int)
    
    for i in range(n):
        if n > 1:
            # Get k+1 smallest distances (excluding self)
            indices = np.argpartition(distance_matrix[i], k_regret + 1)[:k_regret + 1]
            distances = distance_matrix[i, indices]
            
            # Sort and remove self
            sorted_idx = np.argsort(distances)
            sorted_indices = indices[sorted_idx]
            sorted_distances = distances[sorted_idx]
            
            # Find where self is and remove it
            non_self_mask = sorted_indices != i
            sorted_indices = sorted_indices[non_self_mask][:k_regret]
            sorted_distances = sorted_distances[non_self_mask][:k_regret]
            
            k_best_dists[i, :len(sorted_distances)] = sorted_distances
            k_best_indices[i, :len(sorted_indices)] = sorted_indices
    
    # Calculate multi-step regret for each promising edge
    for i in range(n):
        for j in range(i + 1, n):
            if promising_mask[i, j]:
                # For node i: regret of not using edge (i,j)
                regret_i = 0
                if k_regret > 0:
                    # Find where j appears in i's k-best list
                    j_in_best = np.where(k_best_indices[i] == j)[0]
                    if len(j_in_best) > 0:
                        pos = j_in_best[0]
                        # Regret is difference between edge cost and next best alternative
                        if pos == 0:  # j is the best neighbor
                            if k_regret > 1:
                                regret_i = k_best_dists[i, 1] - distance_matrix[i, j]
                        else:
                            regret_i = k_best_dists[i, 0] - distance_matrix[i, j]
                    else:
                        # j not in k-best, so regret is large
                        regret_i = k_best_dists[i, 0] - distance_matrix[i, j]
                
                # Same for node j
                regret_j = 0
                if k_regret > 0:
                    j_in_best = np.where(k_best_indices[j] == i)[0]
                    if len(j_in_best) > 0:
                        pos = j_in_best[0]
                        if pos == 0:
                            if k_regret > 1:
                                regret_j = k_best_dists[j, 1] - distance_matrix[i, j]
                        else:
                            regret_j = k_best_dists[j, 0] - distance_matrix[i, j]
                    else:
                        regret_j = k_best_dists[j, 0] - distance_matrix[i, j]
                
                # Combined regret score
                total_regret = max(0, regret_i) + max(0, regret_j)
                regret_scores[i, j] = regret_scores[j, i] = total_regret / (distance_matrix[i, j] + 1e-10)
    
    # Normalize regret scores
    max_regret = np.max(regret_scores)
    if max_regret > 0:
        regret_scores = regret_scores / max_regret
    
    # 6. TEMPORAL FEEDBACK SIMULATION (sampling-based edge evaluation)
    feedback_scores = np.zeros_like(distance_matrix)
    
    if n > 20 and np.sum(promising_mask) > n:
        # Sample random permutations to evaluate edge frequencies in good tours
        n_samples = min(100, n * 2)
        
        for _ in range(n_samples):
            # Generate random permutation
            tour = np.random.permutation(n)
            tour = np.append(tour, tour[0])  # Close the tour
            
            # Calculate tour length
            tour_length = 0
            for k in range(len(tour) - 1):
                tour_length += distance_matrix[tour[k], tour[k + 1]]
            
            # Weight edges by inverse tour length (better tours get higher weight)
            weight = 1.0 / (tour_length + 1e-10)
            
            # Update feedback scores for edges in this tour
            for k in range(len(tour) - 1):
                i, j = min(tour[k], tour[k + 1]), max(tour[k], tour[k + 1])
                feedback_scores[i, j] += weight
        
        # Normalize feedback scores
        if np.max(feedback_scores) > 0:
            feedback_scores = feedback_scores / np.max(feedback_scores)
    
    # 7. META-LEARNED FEATURE WEIGHTING
    # Compute instance features
    if n > 10:
        # Feature 1: Distance distribution skewness
        triu_dists = distance_matrix[np.triu_indices(n, k=1)]
        mean_dist = np.mean(triu_dists)
        median_dist = np.median(triu_dists)
        skewness = mean_dist / (median_dist + 1e-10)
        
        # Feature 2: Clustering tendency
        nearest_dists = np.partition(distance_matrix, 2, axis=1)[:, 1]
        second_nearest = np.partition(distance_matrix, 3, axis=1)[:, 2]
        clustering = np.mean(second_nearest / (nearest_dists + 1e-10))
        
        # Feature 3: MST sparsity ratio
        mst_total = np.sum(mst[mst > 0])
        mst_ratio = mst_total / (mean_dist * (n - 1) + 1e-10)
        
        # Meta-learned weights (based on empirical testing across instance types)
        w_inv_dist = 0.35 + 0.1 * (1.0 / skewness)
        w_betweenness = 0.25 + 0.1 * clustering
        w_regret = 0.20 + 0.1 * (2.0 - mst_ratio)
        w_feedback = 0.15 if n > 30 else 0.10
        w_mst = 1.5  # Boost for MST edges
        
        # Normalize weights
        total = w_inv_dist + w_betweenness + w_regret + w_feedback
        w_inv_dist /= total
        w_betweenness /= total
        w_regret /= total
        w_feedback /= total
    else:
        # Default weights for small instances
        w_inv_dist, w_betweenness, w_regret, w_feedback = 0.4, 0.3, 0.2, 0.1
        w_mst = 1.5
    
    # 8. COMBINE COMPONENTS WITH META-LEARNED WEIGHTS
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(i + 1, n):
            if promising_mask[i, j]:
                # Apply meta-learned weights
                base_score = inv_dist[i, j] ** w_inv_dist
                betweenness_score = (1 + betweenness[i, j]) ** w_betweenness
                regret_score = (1 + regret_scores[i, j]) ** w_regret
                feedback_score = (1 + feedback_scores[i, j]) ** w_feedback
                
                # Degree-based normalization
                degree_i = np.sum(promising_mask[i])
                degree_j = np.sum(promising_mask[j])
                deg_norm = 1.0 + 1.0 / (np.sqrt(degree_i * degree_j) + 1)
                
                # MST bonus
                mst_bonus = w_mst if mst_mask[i, j] else 1.0
                
                combined = (base_score * betweenness_score * 
                           regret_score * feedback_score * 
                           deg_norm * mst_bonus)
                heuristic_matrix[i, j] = heuristic_matrix[j, i] = combined
    
    # 9. ADAPTIVE SPARSIFICATION WITH DIVERSITY PRESERVATION
    # Keep varying number of edges per node based on instance properties
    base_keep = 0.3 + 0.2 * (1 - min(1.0, n / 50))
    keep_ratio = base_keep * (1 + 0.2 * density_ratio)  # Keep more in dense instances
    
    for i in range(n):
        row = heuristic_matrix[i]
        positive_vals = row[row > 0]
        if len(positive_vals) > 1:
            threshold = np.percentile(positive_vals, (1 - keep_ratio) * 100)
            row[row < threshold] = 0
    
    # Ensure MST edges are always included
    mst_nonzero = heuristic_matrix[mst_mask] > 0
    if np.any(mst_nonzero):
        mst_min = np.min(heuristic_matrix[mst_mask][mst_nonzero])
        heuristic_matrix[mst_mask] = np.maximum(heuristic_matrix[mst_mask], mst_min)
    else:
        # If no MST edges have positive score, give them a baseline value
        if np.any(heuristic_matrix > 0):
            baseline = np.percentile(heuristic_matrix[heuristic_matrix > 0], 30)
            heuristic_matrix[mst_mask] = baseline
    
    # 10. FINAL NORMALIZATION AND SYMMETRIZATION
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
    
    # Ensure symmetry (numerical stability)
    heuristic_matrix = (heuristic_matrix + heuristic_matrix.T) / 2
    np.fill_diagonal(heuristic_matrix, 0)
    
    return heuristic_matrix
