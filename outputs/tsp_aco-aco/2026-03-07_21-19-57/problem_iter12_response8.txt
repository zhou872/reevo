import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic with multi-step path evaluation, 
    graph structural importance, and instance-adaptive parameter optimization.
    
    Key innovations:
    1. Multi-step Path Potential: Evaluates edge value through 3-step forward paths
       and backward connectivity to identify promising connections
    2. Density-Aware Neighborhood Scoring: Adjusts local influence based on 
       cluster density and uniformity
    3. Graph Structural Importance: Computes edge criticality through multiple
       greedy tours and minimum spanning tree analysis
    4. Instance-Adaptive Parameter Tuning: Dynamically adjusts algorithm parameters
       based on instance characteristics (size, variance, uniformity)
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # =====================================================================
    # 1. INSTANCE ANALYSIS FOR ADAPTIVE PARAMETER TUNING
    # =====================================================================
    
    # Compute instance statistics
    mask = ~np.eye(n, dtype=bool)
    dist_values = distance_matrix[mask]
    dist_mean = np.mean(dist_values)
    dist_std = np.std(dist_values)
    dist_cv = dist_std / (dist_mean + 1e-8)
    
    # Analyze distance distribution uniformity
    nearest_ratios = []
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        sorted_row = np.sort(row[row < np.inf])
        if len(sorted_row) >= 2:
            nearest_ratios.append(sorted_row[1] / sorted_row[0])
    uniformity = 1.0 / (1.0 + np.std(nearest_ratios)) if nearest_ratios else 1.0
    
    # Adaptive parameters based on instance characteristics
    size_factor = np.log10(n + 1)
    
    # Multi-step lookahead depth: deeper for uniform instances, shallower for varied
    lookahead_depth = min(3, max(2, int(2.5 * uniformity)))
    
    # Neighborhood size for density computation
    k_neighbors = max(3, min(15, int(0.2 * n * (1.0 + 0.5 * uniformity))))
    
    # =====================================================================
    # 2. MULTI-STEP PATH POTENTIAL EVALUATION
    # =====================================================================
    
    # Precompute sorted distances and indices
    sorted_indices = np.zeros((n, n-1), dtype=int)
    sorted_dists = np.zeros((n, n-1))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        idx = np.argsort(dists)
        sorted_indices[i] = idx[:n-1]
        sorted_dists[i] = dists[idx[:n-1]]
    
    # Path potential scoring
    path_potential = np.ones((n, n))
    
    for i in range(n):
        for j in range(i+1, n):
            if i == j:
                continue
                
            edge_dist = distance_matrix[i, j]
            
            # Forward path potential (i -> j -> k -> l)
            forward_score = 0.0
            
            # Get top candidates from j (excluding i)
            from_j_indices = sorted_indices[j]
            from_j_dists = sorted_dists[j]
            
            # Find position of i in j's sorted list
            i_in_j = np.where(from_j_indices == i)[0]
            if len(i_in_j) > 0:
                start_idx = 1
            else:
                start_idx = 0
            
            # Consider top lookahead_depth continuations from j
            for step in range(min(lookahead_depth, len(from_j_indices) - start_idx)):
                k_idx = start_idx + step
                if k_idx >= len(from_j_indices):
                    break
                    
                k = from_j_indices[k_idx]
                if k == i:
                    continue
                    
                # Get continuation from k (excluding i, j)
                from_k_indices = sorted_indices[k]
                from_k_dists = sorted_dists[k]
                
                # Find positions of i and j in k's list
                i_pos = np.where(from_k_indices == i)[0]
                j_pos = np.where(from_k_indices == j)[0]
                
                # Find first valid continuation
                for l_idx, l in enumerate(from_k_indices):
                    if l not in [i, j, k]:
                        path_length = edge_dist + from_j_dists[k_idx] + from_k_dists[l_idx]
                        
                        # Normalize by number of steps
                        normalized_length = path_length / 3.0
                        
                        # Compare with direct edge
                        if edge_dist > 0:
                            ratio = edge_dist / normalized_length
                            # Score higher for paths where edge is part of efficient multi-step route
                            forward_score += np.exp(-0.5 * (ratio - 1.0) ** 2)
                        break
            
            # Backward connectivity (edges that connect to i's other good connections)
            backward_score = 0.0
            
            # Get i's top neighbors (excluding j)
            from_i_indices = sorted_indices[i]
            from_i_dists = sorted_dists[i]
            
            # Find position of j in i's sorted list
            j_in_i = np.where(from_i_indices == j)[0]
            if len(j_in_i) > 0:
                i_start_idx = 1
            else:
                i_start_idx = 0
            
            # Score based on how well j connects to i's other good neighbors
            for m in range(min(3, len(from_i_indices) - i_start_idx)):
                m_idx = i_start_idx + m
                if m_idx >= len(from_i_indices):
                    break
                    
                neighbor = from_i_indices[m_idx]
                if neighbor == j:
                    continue
                    
                # Check if j connects well to this neighbor
                neighbor_to_j = distance_matrix[neighbor, j]
                if neighbor_to_j > 0:
                    # Good if j is close to i's other good neighbors
                    avg_dist = (from_i_dists[m_idx] + neighbor_to_j) / 2.0
                    if edge_dist > 0:
                        backward_score += edge_dist / avg_dist
            
            # Combine forward and backward scores
            if forward_score > 0 or backward_score > 0:
                total_steps = min(lookahead_depth, len(from_j_indices) - start_idx)
                if total_steps > 0:
                    path_potential[i, j] = 1.0 + 0.4 * (forward_score / total_steps)
                
                if backward_score > 0:
                    path_potential[i, j] += 0.3 * backward_score / min(3, len(from_i_indices) - i_start_idx)
            
            # Symmetrize
            path_potential[j, i] = path_potential[i, j]
    
    # =====================================================================
    # 3. DENSITY-AWARE NEIGHBORHOOD SCORING
    # =====================================================================
    
    # Compute adaptive local density
    local_density = np.zeros(n)
    density_variance = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Get k_neighbors nearest distances
        if len(dists[dists < np.inf]) >= k_neighbors:
            kth_dist = np.partition(dists, k_neighbors)[k_neighbors]
            neighbor_mask = dists <= kth_dist
            neighbor_dists = dists[neighbor_mask]
        else:
            neighbor_dists = dists[dists < np.inf]
        
        if len(neighbor_dists) > 0:
            local_density[i] = 1.0 / (np.median(neighbor_dists) + 1e-8)
            density_variance[i] = np.std(neighbor_dists) / (np.mean(neighbor_dists) + 1e-8)
    
    # Normalize densities
    if np.max(local_density) > np.min(local_density):
        local_density = (local_density - np.min(local_density)) / (np.max(local_density) - np.min(local_density))
    
    # Density similarity with adaptive scaling
    density_similarity = np.ones((n, n))
    
    for i in range(n):
        for j in range(i+1, n):
            # Exponential similarity based on density difference
            density_diff = abs(local_density[i] - local_density[j])
            
            # Adjust decay based on local variance
            avg_variance = (density_variance[i] + density_variance[j]) / 2.0
            decay_factor = 2.0 + 3.0 * avg_variance
            
            similarity = np.exp(-decay_factor * density_diff)
            
            # Boost based on individual densities (higher for dense regions)
            density_boost = 0.3 + 0.4 * (local_density[i] + local_density[j])
            
            density_similarity[i, j] = 1.0 + density_boost * similarity
            density_similarity[j, i] = density_similarity[i, j]
    
    # =====================================================================
    # 4. GRAPH STRUCTURAL IMPORTANCE
    # =====================================================================
    
    # Edge frequency from multiple greedy tours
    edge_frequency = np.zeros((n, n))
    n_tours = min(40, max(10, n // 3))
    
    for tour_idx in range(n_tours):
        # Vary starting nodes and tie-breaking strategies
        start_node = (tour_idx * 17) % n
        
        # Alternate between nearest neighbor and best-regret strategies
        if tour_idx % 3 == 0:
            # Standard nearest neighbor
            visited = np.zeros(n, dtype=bool)
            current = start_node
            visited[current] = True
            
            for _ in range(n - 1):
                unvisited = ~visited
                unvisited[current] = False
                
                if not np.any(unvisited):
                    break
                
                distances = distance_matrix[current].copy()
                distances[visited] = np.inf
                next_node = np.argmin(distances)
                
                edge_frequency[current, next_node] += 1
                edge_frequency[next_node, current] += 1
                
                current = next_node
                visited[current] = True
            
            # Return to start
            edge_frequency[current, start_node] += 1
            edge_frequency[start_node, current] += 1
        
        else:
            # Regret-based selection
            visited = np.zeros(n, dtype=bool)
            current = start_node
            visited[current] = True
            
            for _ in range(n - 1):
                unvisited = np.where(~visited)[0]
                if len(unvisited) == 0:
                    break
                
                # For each unvisited node, compute regret
                regrets = []
                for node in unvisited:
                    dist_to_current = distance_matrix[current, node]
                    
                    # Find best alternative to this node
                    alt_dists = distance_matrix[node].copy()
                    alt_dists[node] = np.inf
                    alt_dists[visited] = np.inf
                    
                    if np.any(alt_dists < np.inf):
                        best_alt = np.min(alt_dists[alt_dists < np.inf])
                        regret = dist_to_current - best_alt
                    else:
                        regret = dist_to_current
                    
                    regrets.append((node, regret, dist_to_current))
                
                # Select node with highest regret (most unique connection)
                regrets.sort(key=lambda x: (x[1], x[2]))
                next_node = regrets[0][0] if regrets else current
                
                edge_frequency[current, next_node] += 1
                edge_frequency[next_node, current] += 1
                
                current = next_node
                visited[current] = True
    
    # Convert frequencies to importance scores
    if np.max(edge_frequency) > 0:
        structural_importance = 1.0 + 0.6 * edge_frequency / np.max(edge_frequency)
    else:
        structural_importance = np.ones((n, n))
    
    # Minimum spanning tree reinforcement
    try:
        mst = minimum_spanning_tree(distance_matrix)
        mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
        mst_boost = np.where(mst_mask, 1.3 + 0.1 * uniformity, 1.0)
    except:
        mst_boost = np.ones((n, n))
    
    # =====================================================================
    # 5. CORE HEURISTIC COMPONENTS
    # =====================================================================
    
    # Inverse distance with instance-adaptive regularization
    reg_coef = 0.01 * dist_mean * (1.0 + 0.5 * dist_cv - 0.3 * uniformity)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + reg_coef)
        np.fill_diagonal(inv_dist, 0)
    
    # Rank-based regret with opportunity awareness
    rank_score = np.ones((n, n))
    
    for i in range(n):
        if len(sorted_dists[i]) == 0:
            continue
            
        best_dist = sorted_dists[i, 0]
        
        for rank, j in enumerate(sorted_indices[i]):
            if i == j:
                continue
                
            current_dist = distance_matrix[i, j]
            
            if best_dist > 0 and current_dist > 0:
                # Base ratio
                base_ratio = best_dist / current_dist
                
                # Opportunity cost considering alternatives
                # Find best alternative to j from i
                alt_i_idx = 0
                while alt_i_idx < n-1 and sorted_indices[i, alt_i_idx] == j:
                    alt_i_idx += 1
                alt_i_dist = sorted_dists[i, alt_i_idx] if alt_i_idx < n-1 else current_dist
                
                # Find best alternative to i from j
                alt_j_idx = 0
                while alt_j_idx < n-1 and sorted_indices[j, alt_j_idx] == i:
                    alt_j_idx += 1
                alt_j_dist = sorted_dists[j, alt_j_idx] if alt_j_idx < n-1 else current_dist
                
                # Combined opportunity penalty
                opp_penalty_i = max(0, (current_dist - alt_i_dist) / (current_dist + 1e-8))
                opp_penalty_j = max(0, (current_dist - alt_j_dist) / (current_dist + 1e-8))
                opp_penalty = (opp_penalty_i + opp_penalty_j) / 2.0
                
                # Rank penalty (adapts to instance uniformity)
                rank_decay = 0.6 if uniformity > 0.7 else 0.8
                rank_effect = 1.0 - rank_decay * (rank / max(1, n-2))
                
                rank_score[i, j] = base_ratio * (1.0 - 0.3 * opp_penalty) * rank_effect
            else:
                rank_score[i, j] = 1.0
    
    # =====================================================================
    # 6. ADAPTIVE COMBINATION AND SPARSIFICATION
    # =====================================================================
    
    # Dynamic component weights based on instance characteristics
    # Higher weight to structural components for uniform instances
    w_inv_dist = 0.18
    w_rank = 0.22
    w_path = 0.25 * (1.0 + 0.4 * uniformity)
    w_density = 0.15 * (1.0 + 0.3 * (1.0 - dist_cv))
    w_structural = 0.20 * (1.0 + 0.3 * dist_cv)
    
    # Normalize weights
    total_weight = w_inv_dist + w_rank + w_path + w_density + w_structural
    w_inv_dist /= total_weight
    w_rank /= total_weight
    w_path /= total_weight
    w_density /= total_weight
    w_structural /= total_weight
    
    # Ensure positive values
    inv_dist_pos = np.maximum(inv_dist, 1e-8)
    rank_pos = np.maximum(rank_score, 1e-8)
    path_pos = np.maximum(path_potential, 1e-8)
    density_pos = np.maximum(density_similarity, 1e-8)
    structural_pos = np.maximum(structural_importance, 1e-8)
    
    # Weighted geometric combination
    combined = (
        np.power(inv_dist_pos, w_inv_dist) *
        np.power(rank_pos, w_rank) *
        np.power(path_pos, w_path) *
        np.power(density_pos, w_density) *
        np.power(structural_pos, w_structural) *
        mst_boost
    )
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2.0
    np.fill_diagonal(combined, 0)
    
    # Adaptive sparsification
    keep_mask = np.zeros((n, n), dtype=bool)
    
    # Always include MST edges
    if 'mst_mask' in locals():
        keep_mask |= mst_mask
    
    # Node-specific edge selection
    for i in range(n):
        # Determine number of edges to keep based on instance characteristics
        if n <= 20:
            target_edges = max(3, int(0.4 * n))
        elif n <= 100:
            base_edges = 0.3 * n
            adjust = 0.1 * n * (dist_cv - 0.5)
            target_edges = max(5, int(base_edges + adjust))
        else:
            target_edges = max(8, int(0.2 * n * (1.0 + 0.3 * uniformity)))
        
        target_edges = min(target_edges, n - 1)
        
        # Select top edges by combined score
        scores = combined[i].copy()
        scores[i] = -np.inf
        
        if target_edges > 0:
            threshold_idx = n - target_edges - 1
            if threshold_idx >= 0:
                threshold = np.partition(scores, threshold_idx)[threshold_idx]
                keep_mask[i, scores >= threshold] = True
                keep_mask[scores >= threshold, i] = True
    
    # Ensure minimum connectivity
    for i in range(n):
        connected = np.where(keep_mask[i])[0]
        connected = connected[connected != i]
        
        if len(connected) < 2:
            # Add nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            sorted_idx = np.argsort(dists)
            added = 0
            for j in sorted_idx:
                if j != i and not keep_mask[i, j]:
                    keep_mask[i, j] = True
                    keep_mask[j, i] = True
                    added += 1
                    if len(connected) + added >= 2:
                        break
    
    # Ensure graph connectivity via BFS
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        current = stack.pop()
        neighbors = np.where(keep_mask[current])[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    if not np.all(visited):
        # Connect disconnected components with shortest edges
        while not np.all(visited):
            unvisited = np.where(~visited)[0]
            visited_nodes = np.where(visited)[0]
            
            # Find shortest edge between visited and unvisited
            min_dist = np.inf
            best_pair = None
            
            for u in unvisited:
                for v in visited_nodes:
                    if distance_matrix[u, v] < min_dist:
                        min_dist = distance_matrix[u, v]
                        best_pair = (u, v)
            
            if best_pair:
                u, v = best_pair
                keep_mask[u, v] = True
                keep_mask[v, u] = True
                visited[u] = True
    
    # Apply sparsification
    heuristic = combined * keep_mask
    
    # Final normalization per node
    for i in range(n):
        row = heuristic[i]
        mask = row > 0
        
        if np.sum(mask) > 0:
            row_masked = row[mask]
            # Softmax normalization
            row_masked = row_masked - np.max(row_masked)
            exp_row = np.exp(row_masked)
            heuristic[i, mask] = exp_row / np.sum(exp_row)
    
    # Final symmetry and zero diagonal
    heuristic = (heuristic + heuristic.T) / 2.0
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
