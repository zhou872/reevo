import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Next-generation TSP edge heuristics with adaptive multi-step lookahead,
    global feasibility-aware regret, and dynamic heuristic fusion.
    
    Core innovations:
    1. Adaptive multi-step lookahead planning with instance-size-dependent depth
    2. Global tour feasibility regret that considers edge impact on complete tours
    3. Graph neural-inspired edge importance scoring via message passing
    4. Dynamic weighted fusion of multiple heuristic strategies
    
    Args:
        distance_matrix: Square symmetric matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Initialize heuristic matrix
    heuristic = np.zeros((n, n))
    np.fill_diagonal(heuristic, 0)
    
    # 1. ADAPTIVE MULTI-STEP LOOKAHEAD PLANNING
    # Determine lookahead depth based on instance complexity
    if n < 20:
        lookahead_depth = 3
    elif n < 100:
        lookahead_depth = 2
    else:
        lookahead_depth = 1  # For very large instances
    
    lookahead_score = np.ones((n, n))
    
    # Precompute nearest neighbor information for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_distances = np.take_along_axis(distance_matrix, sorted_indices, axis=1)
    
    # Compute lookahead scores using iterative deepening
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            # Start with edge i->j
            current_cost = distance_matrix[i, j]
            visited = {i, j}
            current_node = j
            
            # Perform k-step lookahead
            total_lookahead_cost = current_cost
            for step in range(lookahead_depth - 1):
                # Find best unvisited neighbor from current node
                for neighbor_idx in sorted_indices[current_node]:
                    if neighbor_idx not in visited:
                        total_lookahead_cost += distance_matrix[current_node, neighbor_idx]
                        visited.add(neighbor_idx)
                        current_node = neighbor_idx
                        break
                else:
                    # No unvisited neighbors, return to start
                    total_lookahead_cost += distance_matrix[current_node, i]
                    break
            
            # Normalize lookahead cost by average edge distance
            avg_edge_dist = np.mean(distance_matrix[np.triu_indices(n, 1)])
            lookahead_score[i, j] = 1.0 / (1.0 + total_lookahead_cost / (avg_edge_dist * lookahead_depth))
    
    # 2. GLOBAL FEASIBILITY-AWARE REGRET
    # Compute regret considering impact on complete tour feasibility
    global_regret = np.ones((n, n))
    
    # Estimate optimal tour length using Christofides heuristic approximation
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = mst.toarray()
    mst_weight = np.sum(mst_edges)
    
    # Compute degree of each node in MST (for connectivity analysis)
    mst_degree = np.sum(mst_edges > 0, axis=1)
    
    # Enhanced regret calculation considering:
    # a) Connectivity importance (bridging isolated components)
    # b) Alternative path availability
    # c) Impact on tour structural properties
    
    for i in range(n):
        # Find best and second-best edges from node i
        dist_i = distance_matrix[i, :].copy()
        dist_i[i] = np.inf
        sorted_i = np.argsort(dist_i)
        best_j = sorted_i[0]
        second_best_j = sorted_i[1]
        
        for j in range(n):
            if i == j:
                continue
                
            # Base regret: difference from optimal alternative
            base_regret = distance_matrix[i, j] - distance_matrix[i, best_j]
            
            # Connectivity penalty: penalize edges that connect already well-connected nodes
            connect_penalty = (mst_degree[i] * mst_degree[j]) / (n * n)
            
            # Global impact: estimate impact on complete tour length
            # Approximate by considering MST with and without edge
            alt_mst_cost = mst_weight
            if distance_matrix[i, j] < np.percentile(distance_matrix, 25):
                # This edge might be in optimal tour - estimate impact positively
                global_impact = np.exp(-distance_matrix[i, j] / (mst_weight / n))
            else:
                global_impact = np.exp(-2 * distance_matrix[i, j] / (mst_weight / n))
            
            # Combine regret components
            regret_score = 1.0 + global_impact - 0.5 * connect_penalty
            
            # Normalize and bound
            global_regret[i, j] = np.clip(regret_score, 0.5, 2.0)
    
    # 3. GRAPH NEURAL INSPIRED EDGE IMPORTANCE
    # Use message passing to capture structural relationships
    gnn_score = np.ones((n, n))
    
    # Initialize node features with local properties
    node_features = np.zeros((n, 3))
    for i in range(n):
        # Feature 1: Average distance to nearest 10% of nodes
        k_local = max(3, n // 10)
        local_dists = np.partition(distance_matrix[i, :], k_local)[:k_local]
        node_features[i, 0] = np.mean(local_dists)
        
        # Feature 2: Distance variance (spread)
        node_features[i, 1] = np.var(distance_matrix[i, :])
        
        # Feature 3: MST connectivity
        node_features[i, 2] = mst_degree[i]
    
    # Normalize features
    for feat_idx in range(3):
        feat_min = np.min(node_features[:, feat_idx])
        feat_max = np.max(node_features[:, feat_idx])
        if feat_max > feat_min:
            node_features[:, feat_idx] = (node_features[:, feat_idx] - feat_min) / (feat_max - feat_min)
    
    # Two-step message passing to compute edge importance
    for _ in range(2):  # Message passing iterations
        new_features = node_features.copy()
        
        for i in range(n):
            # Aggregate information from neighbors (weighted by inverse distance)
            neighbor_mask = distance_matrix[i, :] < np.percentile(distance_matrix, 30)
            neighbor_mask[i] = False
            
            if np.any(neighbor_mask):
                neighbor_weights = 1.0 / (distance_matrix[i, neighbor_mask] + 1e-8)
                neighbor_weights = neighbor_weights / np.sum(neighbor_weights)
                
                # Weighted average of neighbor features
                for feat_idx in range(3):
                    weighted_avg = np.sum(neighbor_weights * node_features[neighbor_mask, feat_idx])
                    new_features[i, feat_idx] = 0.7 * node_features[i, feat_idx] + 0.3 * weighted_avg
        
        node_features = new_features
    
    # Compute edge scores from node features
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            # Edge score based on feature compatibility and distance
            feature_similarity = 1.0 - np.mean(np.abs(node_features[i, :] - node_features[j, :]))
            distance_factor = np.exp(-distance_matrix[i, j] / np.mean(node_features[:, 0]))
            
            gnn_score[i, j] = 0.5 + 0.5 * feature_similarity * distance_factor
    
    # 4. DYNAMIC WEIGHTED FUSION OF HEURISTICS
    # Base component: inverse distance (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.diag_indices(n)] = 0
    
    # Normalize inverse distance
    if np.max(inv_dist) > 0:
        inv_dist = inv_dist / np.max(inv_dist)
    
    # Compute instance characteristics for dynamic weighting
    instance_sparsity = np.std(distance_matrix) / np.mean(distance_matrix)
    instance_size_factor = np.log10(n + 1)
    
    # Dynamic weight allocation based on instance properties
    if instance_sparsity > 0.5:  # Highly variable distances
        weights = [0.3, 0.4, 0.2, 0.1]  # [inv_dist, lookahead, global_regret, gnn]
    elif n < 30:  # Small instances
        weights = [0.2, 0.3, 0.3, 0.2]
    elif n < 100:  # Medium instances
        weights = [0.25, 0.35, 0.25, 0.15]
    else:  # Large instances
        weights = [0.3, 0.4, 0.2, 0.1]
    
    # Combine heuristics with dynamic weights
    heuristic = (
        weights[0] * inv_dist +
        weights[1] * lookahead_score +
        weights[2] * global_regret +
        weights[3] * gnn_score
    )
    
    # 5. ENHANCED SPARSIFICATION WITH ADAPTIVE CONNECTIVITY
    # Always include MST edges
    mst_mask = (mst_edges > 0) | (mst_edges.T > 0)
    
    # Include promising edges based on combined score
    # Adaptive number of edges per node
    edges_per_node = max(3, min(10, n // 5))
    
    final_mask = mst_mask.copy()
    
    # Add top edges by heuristic score for each node
    for i in range(n):
        node_scores = heuristic[i, :].copy()
        node_scores[i] = -np.inf
        
        # Get top edges
        top_indices = np.argpartition(node_scores, -edges_per_node)[-edges_per_node:]
        final_mask[i, top_indices] = True
        final_mask[top_indices, i] = True
    
    # Ensure minimum connectivity: run BFS to check connectivity
    visited = np.zeros(n, dtype=bool)
    queue = [0]
    visited[0] = True
    
    while queue:
        current = queue.pop()
        neighbors = np.where(final_mask[current, :])[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                queue.append(neighbor)
    
    # If graph is not fully connected, add necessary edges
    if not np.all(visited):
        # Find connected components and add minimum connections
        components = []
        unvisited = np.where(~visited)[0]
        
        while len(unvisited) > 0:
            start = unvisited[0]
            comp = []
            stack = [start]
            local_visited = np.zeros(n, dtype=bool)
            local_visited[start] = True
            
            while stack:
                node = stack.pop()
                comp.append(node)
                neighbors = np.where(final_mask[node, :])[0]
                for neighbor in neighbors:
                    if not local_visited[neighbor]:
                        local_visited[neighbor] = True
                        stack.append(neighbor)
            
            components.append(comp)
            unvisited = [u for u in unvisited if not local_visited[u]]
        
        # Connect components with shortest edges
        for i in range(len(components) - 1):
            comp1 = components[i]
            comp2 = components[i + 1]
            
            # Find shortest edge between components
            min_dist = np.inf
            min_pair = (0, 0)
            
            for node1 in comp1:
                for node2 in comp2:
                    if distance_matrix[node1, node2] < min_dist:
                        min_dist = distance_matrix[node1, node2]
                        min_pair = (node1, node2)
            
            # Add connecting edge
            final_mask[min_pair[0], min_pair[1]] = True
            final_mask[min_pair[1], min_pair[0]] = True
    
    # Apply final mask
    heuristic = heuristic * final_mask
    
    # Normalize to [0, 1] range
    h_min, h_max = np.min(heuristic[final_mask]), np.max(heuristic[final_mask])
    if h_max > h_min:
        heuristic[final_mask] = (heuristic[final_mask] - h_min) / (h_max - h_min)
    
    # Apply softmax scaling for better differentiation
    heuristic = np.exp(heuristic) / (np.exp(heuristic).sum(axis=1, keepdims=True) + 1e-8)
    
    return heuristic
