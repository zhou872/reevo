import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.cluster import KMeans
from scipy.spatial.distance import cdist

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with global cluster analysis,
    weighted multi-heuristic fusion, and regret-based lookahead mechanisms.
    
    Core innovations:
    1. Cluster-Aware Density Estimation: Uses K-means clustering to identify natural
       clusters and compute intra/inter-cluster density bonuses
    2. Weighted Multi-Heuristic Fusion: Linear combination of multiple heuristic
       components with learnable weights for fine-grained control
    3. Two-Step Regret Lookahead: Anticipates future edge costs by evaluating
       second-order choices from each potential connection
    4. Adaptive Threshold Sparsification: Dynamically adjusts sparsity levels
       based on graph size and cluster structure
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Initialize component matrices
    comp_inverse = np.zeros((n, n))
    comp_rank = np.zeros((n, n))
    comp_cluster = np.ones((n, n))
    comp_regret = np.ones((n, n))
    comp_proximity = np.ones((n, n))
    
    np.fill_diagonal(comp_inverse, 0)
    np.fill_diagonal(comp_rank, 0)
    np.fill_diagonal(comp_cluster, 0)
    np.fill_diagonal(comp_regret, 0)
    np.fill_diagonal(comp_proximity, 0)
    
    # 1. INVERSE DISTANCE COMPONENT (with adaptive regularization)
    mean_dist = np.mean(distance_matrix[distance_matrix > 0])
    with np.errstate(divide='ignore', invalid='ignore'):
        comp_inverse = 1.0 / (distance_matrix + 0.01 * mean_dist)
        comp_inverse[np.isinf(comp_inverse)] = 0
    
    # 2. RANK-BASED REGRET INCENTIVE (enhanced with exponential decay)
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        sorted_indices = np.argsort(distances)
        sorted_distances = distances[sorted_indices]
        
        if len(sorted_distances) > 0 and sorted_distances[0] < np.inf:
            d_best = sorted_distances[0]
            for rank, j in enumerate(sorted_indices):
                if i == j:
                    continue
                d_current = distances[j]
                # Exponential decay based on relative rank difference
                rank_factor = np.exp(-rank / max(1, 0.3 * n))
                dist_ratio = d_best / (d_current + 1e-8)
                comp_rank[i, j] = 1.0 + rank_factor * dist_ratio
    
    # 3. GLOBAL CLUSTER-AWARE DENSITY ESTIMATION
    # Determine optimal number of clusters (max of 8 or n/10)
    n_clusters = min(max(2, int(n / 10)), 8, n)
    
    if n_clusters > 1 and n > n_clusters:
        try:
            # Use coordinates derived from distance matrix via MDS approximation
            # Simple approximation: use first two nodes as reference dimensions
            coords = np.zeros((n, 2))
            coords[:, 0] = distance_matrix[:, 0]
            coords[:, 1] = distance_matrix[:, min(1, n-1)]
            
            kmeans = KMeans(n_clusters=n_clusters, n_init=10, random_state=42)
            cluster_labels = kmeans.fit_predict(coords)
            
            # Compute intra-cluster and inter-cluster bonuses
            for i in range(n):
                for j in range(n):
                    if i == j:
                        continue
                    
                    if cluster_labels[i] == cluster_labels[j]:
                        # Intra-cluster bonus: higher for compact clusters
                        cluster_mask = cluster_labels == cluster_labels[i]
                        intra_dists = distance_matrix[i, cluster_mask]
                        if len(intra_dists) > 1:
                            cluster_density = 1.0 / (np.mean(intra_dists) + 1e-8)
                            comp_cluster[i, j] = 1.0 + 0.5 * np.tanh(cluster_density / mean_dist)
                    else:
                        # Inter-cluster penalty with distance-based modulation
                        cluster_i_mask = cluster_labels == cluster_labels[i]
                        cluster_j_mask = cluster_labels == cluster_labels[j]
                        
                        # Average distance between clusters
                        inter_dists = distance_matrix[cluster_i_mask, :][:, cluster_j_mask]
                        if inter_dists.size > 0:
                            avg_inter_dist = np.mean(inter_dists)
                            penalty = 0.8 + 0.2 * (distance_matrix[i, j] / (avg_inter_dist + 1e-8))
                            comp_cluster[i, j] = max(0.5, min(1.0, penalty))
        except:
            # Fallback to local density if clustering fails
            for i in range(n):
                distances = distance_matrix[i, :].copy()
                distances[i] = np.inf
                k = min(5, n-1)
                if k > 0:
                    nearest = np.partition(distances, k)[:k+1]
                    local_density = 1.0 / (np.mean(nearest[nearest < np.inf]) + 1e-8)
                    for j in range(n):
                        if i != j:
                            comp_cluster[i, j] = 1.0 + 0.3 * np.exp(-abs(local_density - 1.0/distance_matrix[i,j]))
    
    # 4. TWO-STEP REGRET LOOKAHEAD MECHANISM
    for i in range(n):
        # First-order: best edge from i
        distances_i = distance_matrix[i, :].copy()
        distances_i[i] = np.inf
        if np.all(distances_i == np.inf):
            continue
            
        best_from_i = np.min(distances_i[distances_i < np.inf])
        
        for j in range(n):
            if i == j:
                continue
                
            # Cost of choosing edge (i,j)
            cost_ij = distance_matrix[i, j]
            
            # Lookahead: what's the best we can do from j (excluding i)?
            distances_j = distance_matrix[j, :].copy()
            distances_j[i] = np.inf  # Can't go back to i immediately
            distances_j[j] = np.inf
            
            if np.any(distances_j < np.inf):
                best_from_j = np.min(distances_j[distances_j < np.inf])
                # Regret: how much worse is (i,j) compared to optimal two-step?
                optimal_two_step = best_from_i + best_from_j
                actual_two_step = cost_ij + best_from_j
                
                if optimal_two_step > 0:
                    regret_ratio = optimal_two_step / (actual_two_step + 1e-8)
                    comp_regret[i, j] = np.clip(regret_ratio, 0.7, 1.3)
    
    # 5. ADAPTIVE SPATIAL PROXIMITY THRESHOLDING
    # Compute dynamic thresholds based on local neighborhood statistics
    avg_nn_dist = np.zeros(n)
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        k_nn = min(5, n-1)
        if k_nn > 0:
            nn_dists = np.partition(distances, k_nn)[:k_nn+1]
            nn_dists = nn_dists[nn_dists < np.inf]
            if len(nn_dists) > 0:
                avg_nn_dist[i] = np.mean(nn_dists)
            else:
                avg_nn_dist[i] = mean_dist
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            dist = distance_matrix[i, j]
            # Adaptive threshold based on both nodes' neighborhoods
            threshold = 1.8 * min(avg_nn_dist[i], avg_nn_dist[j]) + 0.2 * mean_dist
            
            if dist <= threshold:
                # Bonus decreases linearly with distance
                proximity_factor = 1.0 + 0.4 * (1.0 - dist / threshold)
                comp_proximity[i, j] = proximity_factor
            else:
                # Penalty increases with distance beyond threshold
                penalty = 0.6 * (threshold / (dist + 1e-8))
                comp_proximity[i, j] = max(0.3, penalty)
    
    # 6. MINIMUM SPANNING TREE STRUCTURAL PRIORITY
    try:
        mst = minimum_spanning_tree(distance_matrix)
        mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    except:
        mst_mask = np.zeros((n, n), dtype=bool)
    
    # 7. WEIGHTED MULTI-HEURISTIC FUSION
    # Component weights (sum to 1.0)
    weights = {
        'inverse': 0.25,
        'rank': 0.25,
        'cluster': 0.20,
        'regret': 0.15,
        'proximity': 0.15
    }
    
    # Normalize each component to [0,1] range
    components = {
        'inverse': comp_inverse,
        'rank': comp_rank,
        'cluster': comp_cluster,
        'regret': comp_regret,
        'proximity': comp_proximity
    }
    
    normalized = {}
    for key, matrix in components.items():
        max_val = np.max(matrix)
        min_val = np.min(matrix[matrix > 0]) if np.any(matrix > 0) else 0
        if max_val > min_val:
            normalized[key] = (matrix - min_val) / (max_val - min_val + 1e-8)
        else:
            normalized[key] = matrix
    
    # Linear combination with weights
    heuristic = np.zeros((n, n))
    for key, weight in weights.items():
        heuristic += weight * normalized[key]
    
    # Apply MST boost (additive bonus)
    mst_boost = 0.35  # 35% bonus for MST edges
    heuristic = heuristic * (1.0 + mst_boost * mst_mask)
    
    # Ensure symmetry
    heuristic = (heuristic + heuristic.T) / 2.0
    
    # 8. ADAPTIVE THRESHOLD SPARSIFICATION
    # Determine sparsity level based on graph size and cluster structure
    if n <= 15:
        keep_fraction = 0.4
    elif n <= 50:
        keep_fraction = 0.3
    else:
        keep_fraction = 0.2
    
    # Always keep MST edges
    final_mask = mst_mask.copy()
    
    # Keep top edges by heuristic value
    n_keep = max(2, int(keep_fraction * n * (n-1) / 2))
    if n_keep > 0:
        # Flatten upper triangle (excluding diagonal)
        triu_indices = np.triu_indices(n, k=1)
        heuristic_vals = heuristic[triu_indices]
        
        # Get threshold for top n_keep edges
        if len(heuristic_vals) > n_keep:
            threshold = np.partition(heuristic_vals, -n_keep)[-n_keep]
            top_mask = heuristic >= threshold
            
            # Also include any edge that's in top 3 for any node
            for i in range(n):
                node_heuristics = heuristic[i, :].copy()
                node_heuristics[i] = -np.inf
                top_3 = np.argpartition(node_heuristics, -3)[-3:]
                top_mask[i, top_3] = True
                top_mask[top_3, i] = True
            
            final_mask = final_mask | top_mask
    
    # Ensure minimum connectivity: each node connects to at least 2 others
    for i in range(n):
        connected = np.where(final_mask[i, :])[0]
        connected = connected[connected != i]
        if len(connected) < 2:
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            # Add nearest neighbors
            nearest = np.argsort(distances)[:2]
            for j in nearest:
                if j != i:
                    final_mask[i, j] = True
                    final_mask[j, i] = True
    
    # Apply sparsification mask
    heuristic = heuristic * final_mask
    
    # Final normalization
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
