import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with improved
    bidirectional regret minimization and spectral clustering density analysis.
    
    Core improvements over v1:
    1. Bidirectional Rank-Regret Minimization: Considers opportunity costs
       from both endpoints using rank-normalized regret scores
    2. Spectral Clustering Density Bonus: Uses spectral gap analysis to
       identify natural clusters and boost intra-cluster edges
    3. Dynamic Proximity Thresholding: Adaptive thresholds based on local
       connectivity patterns and cluster membership
    4. Multi-criteria Sparsification: Strategic zeroing of unpromising edges
       while preserving structural diversity
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Initialize heuristic matrix
    heuristic = np.zeros((n, n))
    np.fill_diagonal(heuristic, 0)
    
    # 1. BIDIRECTIONAL RANK-REGRET MINIMIZATION
    regret_matrix = np.ones((n, n))
    
    for i in range(n):
        # Get distances from node i to all others
        distances_i = distance_matrix[i, :].copy()
        distances_i[i] = np.inf
        
        # Sort and rank for node i
        sorted_i = np.argsort(distances_i)
        ranks_i = np.argsort(sorted_i)  # Converts to rank positions
        best_dist_i = distances_i[sorted_i[0]] if n > 1 else 1.0
        
        for j in range(n):
            if i == j:
                continue
                
            # Get distances from node j to all others
            distances_j = distance_matrix[j, :].copy()
            distances_j[j] = np.inf
            
            # Sort and rank for node j
            sorted_j = np.argsort(distances_j)
            ranks_j = np.argsort(sorted_j)
            best_dist_j = distances_j[sorted_j[0]] if n > 1 else 1.0
            
            # Current edge distance
            dist_ij = distance_matrix[i, j]
            
            # Regret from i's perspective (normalized)
            rank_i = ranks_i[j]
            regret_i = (dist_ij - best_dist_i) / (best_dist_i + 1e-8)
            regret_i_norm = 1.0 / (1.0 + regret_i * (1.0 + rank_i/(n-1)))
            
            # Regret from j's perspective (normalized)
            rank_j = ranks_j[i]
            regret_j = (dist_ij - best_dist_j) / (best_dist_j + 1e-8)
            regret_j_norm = 1.0 / (1.0 + regret_j * (1.0 + rank_j/(n-1)))
            
            # Combined bidirectional regret score
            regret_matrix[i, j] = 0.4 * regret_i_norm + 0.4 * regret_j_norm + 0.2
            
    # 2. SPECTRAL CLUSTERING DENSITY BONUS
    density_matrix = np.ones((n, n))
    
    # Compute local density using k-nearest neighbors
    k = max(3, int(0.12 * n))
    local_density = np.zeros(n)
    avg_nn_dist = np.zeros(n)
    
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        if n > k:
            # Get k nearest distances
            kth_smallest = np.partition(distances, k)[k]
            nearest_mask = distances <= kth_smallest
            nearest_dists = distances[nearest_mask]
        else:
            nearest_dists = distances[distances < np.inf]
        
        if len(nearest_dists) > 0:
            avg_nn_dist[i] = np.mean(nearest_dists)
            local_density[i] = 1.0 / (avg_nn_dist[i] + 1e-8)
        else:
            avg_nn_dist[i] = 1.0
            local_density[i] = 1.0
    
    # Normalize densities
    if np.max(local_density) > 0:
        local_density = local_density / np.max(local_density)
    
    # Identify clusters using spectral gap analysis
    # Compute normalized Laplacian
    degree = np.sum(distance_matrix, axis=1)
    degree_matrix = np.diag(degree)
    laplacian = degree_matrix - distance_matrix
    
    # Use Fiedler vector (second smallest eigenvector) for partitioning
    if n >= 3:
        try:
            eigvals, eigvecs = np.linalg.eigh(laplacian)
            # Sort eigenvalues and eigenvectors
            idx = np.argsort(eigvals)
            eigvals = eigvals[idx]
            eigvecs = eigvecs[:, idx]
            
            # Fiedler vector (second eigenvector)
            fiedler = eigvecs[:, 1]
            
            # Create soft cluster assignments based on Fiedler vector
            cluster_assign = np.zeros(n)
            median_fiedler = np.median(fiedler)
            for i in range(n):
                # Soft assignment: value between -1 and 1
                cluster_assign[i] = np.tanh(5.0 * (fiedler[i] - median_fiedler))
        except:
            # Fallback to random assignment if eigendecomposition fails
            cluster_assign = np.random.randn(n)
    else:
        cluster_assign = np.zeros(n)
    
    # Apply spectral clustering density bonus
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            # Base density similarity
            density_diff = abs(local_density[i] - local_density[j])
            density_sim = np.exp(-2.0 * density_diff)
            
            # Cluster coherence bonus
            cluster_sim = 1.0 - 0.5 * abs(cluster_assign[i] - cluster_assign[j])
            
            # Intra-cluster bonus for similar densities
            if density_diff < 0.2 and cluster_sim > 0.7:
                intra_cluster_bonus = 1.0 + 0.5 * cluster_sim * (1.0 - density_diff/0.2)
            else:
                intra_cluster_bonus = 1.0
            
            density_matrix[i, j] = 0.7 * density_sim + 0.3 * cluster_sim * intra_cluster_bonus
    
    # 3. MINIMUM SPANNING TREE STRUCTURAL PRIORITY
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    mst_boost = np.where(mst_mask, 1.4, 1.0)  # 40% boost for MST edges
    
    # 4. DYNAMIC PROXIMITY THRESHOLDING
    proximity_matrix = np.ones((n, n))
    
    # Compute adaptive thresholds based on local statistics and clusters
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            dist = distance_matrix[i, j]
            
            # Base threshold from average nearest neighbor distances
            base_threshold = 1.4 * min(avg_nn_dist[i], avg_nn_dist[j])
            
            # Adjust threshold based on cluster similarity
            cluster_adjust = 1.0 + 0.3 * (1.0 - abs(cluster_assign[i] - cluster_assign[j]))
            dynamic_threshold = base_threshold * cluster_adjust
            
            if dist <= dynamic_threshold:
                # Bonus for edges within dynamic threshold
                proximity_ratio = 1.0 - dist / (dynamic_threshold + 1e-8)
                proximity_matrix[i, j] = 1.0 + 0.4 * proximity_ratio * (1.0 + 0.2 * cluster_adjust)
            else:
                # Penalty for edges beyond threshold, but less severe for inter-cluster edges
                inter_cluster_penalty = 0.6 if abs(cluster_assign[i] - cluster_assign[j]) > 1.0 else 0.8
                proximity_matrix[i, j] = inter_cluster_penalty * (dynamic_threshold / (dist + 1e-8))
    
    # COMBINE ALL COMPONENTS
    # Base component: inverse distance with adaptive scaling
    mean_dist = np.mean(distance_matrix[distance_matrix > 0])
    inv_dist = 1.0 / (distance_matrix + 0.1 * mean_dist)
    inv_dist[np.diag_indices(n)] = 0
    
    # Multiplicative combination with balanced weights
    heuristic = inv_dist * regret_matrix * density_matrix * proximity_matrix * mst_boost
    
    # Ensure symmetry
    heuristic = (heuristic + heuristic.T) / 2.0
    
    # 5. MULTI-CRITERIA SPARSIFICATION
    final_mask = np.zeros((n, n), dtype=bool)
    
    # Always include MST edges for connectivity
    final_mask = final_mask | mst_mask
    
    # Compute edge importance scores for each criterion
    regret_scores = regret_matrix.copy()
    np.fill_diagonal(regret_scores, 0)
    
    density_scores = density_matrix.copy()
    np.fill_diagonal(density_scores, 0)
    
    proximity_scores = proximity_matrix.copy()
    np.fill_diagonal(proximity_scores, 0)
    
    # For each node, select edges that excel in multiple criteria
    for i in range(n):
        # Criterion 1: Top edges by combined heuristic score
        node_scores = heuristic[i, :].copy()
        node_scores[i] = -np.inf
        
        # Adaptive number of top edges based on problem size
        top_k = max(3, int(0.18 * n))
        top_indices = np.argpartition(node_scores, -top_k)[-top_k:]
        final_mask[i, top_indices] = True
        
        # Criterion 2: Low-regret edges (top 4 by regret score)
        node_regrets = regret_scores[i, :].copy()
        node_regrets[i] = -np.inf
        low_regret_indices = np.argpartition(node_regrets, -4)[-4:]
        final_mask[i, low_regret_indices] = True
        
        # Criterion 3: High density similarity edges
        node_density = density_scores[i, :].copy()
        node_density[i] = -np.inf
        density_threshold = np.percentile(node_density[node_density > -np.inf], 75)
        high_density_indices = np.where(node_density >= density_threshold)[0]
        final_mask[i, high_density_indices] = True
        
        # Criterion 4: Critical proximity edges
        # Include edges that are within 1.2x of nearest neighbor distance
        nn_threshold = 1.2 * avg_nn_dist[i]
        proximity_mask = (distance_matrix[i, :] <= nn_threshold) & (np.arange(n) != i)
        final_mask[i, proximity_mask] = True
    
    # Ensure minimum connectivity: each node must have at least 3 connections
    for i in range(n):
        connected = np.where(final_mask[i, :])[0]
        connected = connected[connected != i]
        if len(connected) < 3:
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            # Add closest nodes to reach minimum
            needed = 3 - len(connected)
            if needed > 0:
                # Find closest nodes not already connected
                all_nodes = np.arange(n)
                mask = ~final_mask[i, :]
                mask[i] = False
                available = all_nodes[mask]
                if len(available) > 0:
                    available_dists = distances[available]
                    closest_idx = np.argsort(available_dists)[:min(needed, len(available))]
                    closest_nodes = available[closest_idx]
                    final_mask[i, closest_nodes] = True
                    final_mask[closest_nodes, i] = True
    
    # Apply symmetric mask
    final_mask = final_mask | final_mask.T
    
    # Apply sparsification
    heuristic = heuristic * final_mask
    
    # Normalize to [0, 1] range with softmax-like scaling
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    # Apply final smoothing: boost edges that appear promising across multiple metrics
    for i in range(n):
        for j in range(n):
            if i != j and heuristic[i, j] > 0:
                # Count how many criteria this edge satisfied
                criteria_count = 0
                if regret_scores[i, j] > np.percentile(regret_scores[regret_scores > 0], 70):
                    criteria_count += 1
                if density_scores[i, j] > np.percentile(density_scores[density_scores > 0], 70):
                    criteria_count += 1
                if proximity_scores[i, j] > np.percentile(proximity_scores[proximity_scores > 0], 70):
                    criteria_count += 1
                
                if criteria_count >= 2:
                    heuristic[i, j] = min(1.0, heuristic[i, j] * (1.0 + 0.15 * criteria_count))
    
    return heuristic
