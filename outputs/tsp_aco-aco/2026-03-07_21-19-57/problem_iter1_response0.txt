import numpy as np
from scipy.spatial import KDTree
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection that combines multiple factors
    and returns a sparsified matrix with promising edge indicators.
    """
    n = distance_matrix.shape[0]
    np.fill_diagonal(distance_matrix, np.inf)  # Remove self-loops
    
    # 1. Base inverse distance scoring
    inv_dist = 1 / distance_matrix
    
    # 2. Local density weighting (nodes in dense regions get higher scores)
    avg_dist_to_others = distance_matrix.mean(axis=1)
    density_weights = 1 / (avg_dist_to_others[:, None] + avg_dist_to_others[None, :])
    density_weights /= density_weights.max()
    
    # 3. Geometric locality via KD-tree
    coords = np.random.randn(n, 2)  # For geometric locality simulation
    kdtree = KDTree(coords)
    # Get k-nearest neighbors for each point
    k = min(5, n-1)
    knn_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        distances, indices = kdtree.query(coords[i], k=k+1)
        knn_mask[i, indices[1:]] = True  # Exclude self
    locality_bonus = np.where(knn_mask | knn_mask.T, 1.5, 1.0)
    
    # 4. Minimum Spanning Tree conservation
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = np.where(mst_mask, 2.0, 1.0)
    
    # 5. Edge connectivity importance (edges connecting to low-degree nodes)
    degree_estimate = (distance_matrix < distance_matrix.mean()).sum(axis=1)
    connectivity = degree_estimate[:, None] + degree_estimate[None, :]
    connectivity_norm = connectivity / connectivity.max()
    connectivity_bonus = 1.0 + 0.5 * (1 - connectivity_norm)
    
    # Combine all factors
    combined = (inv_dist * density_weights * 
                locality_bonus * mst_bonus * connectivity_bonus)
    
    # 6. Adaptive sparsification
    # Keep top edges per node and globally
    sparsified = np.zeros_like(combined)
    
    # Keep top k edges per node
    k_sparse = min(3, n-1)
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf
        top_indices = np.argpartition(row, -k_sparse)[-k_sparse:]
        sparsified[i, top_indices] = row[top_indices]
    
    # Keep globally strong edges (above 75th percentile)
    flat_vals = sparsified[sparsified > 0]
    if len(flat_vals) > 0:
        threshold = np.percentile(flat_vals, 75)
        sparsified[sparsified < threshold] = 0
    
    # Symmetrize (for undirected TSP)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    np.fill_diagonal(distance_matrix, 0)  # Restore diagonal
    return sparsified
