import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import cdist

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection probability.
    
    Features:
    1. Inverse distance weighting (base preference for short edges)
    2. Minimum spanning tree edges get bonus (MST often contains TSP edges)
    3. Sparsification based on k-nearest neighbors (only keep promising edges)
    4. Cluster density bonus (edges within dense regions get bonus)
    
    Returns: Heuristic matrix of same shape as input, higher values = more promising
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (handle zero distances)
    with np.errstate(divide='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-10)  # Add tiny to avoid div by zero
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = (mst + mst.T > 0).astype(float) * 2.0  # Symmetrize and boost MST edges
    
    # 3. K-nearest neighbor sparsification mask
    k = max(3, n // 10)  # Adaptive k: at least 3, at most 10% of nodes
    mask = np.zeros((n, n))
    for i in range(n):
        # Get k+1 nearest (including self)
        nearest = np.argsort(distance_matrix[i])[:k+1]
        mask[i, nearest] = 1
        mask[nearest, i] = 1  # Symmetrize
    
    np.fill_diagonal(mask, 0)
    
    # 4. Local cluster density bonus
    # Estimate local density using distance to k-nearest neighbors
    density_bonus = np.zeros((n, n))
    for i in range(n):
        # Get distances to k-nearest neighbors
        nearest_dists = np.sort(distance_matrix[i])[1:k+1]
        avg_nearest_dist = np.mean(nearest_dists) if len(nearest_dists) > 0 else 1
        # For edges to neighbors within 2x average nearest distance
        close_mask = distance_matrix[i] < 2 * avg_nearest_dist
        density_bonus[i, close_mask] = 1.0 / (1 + avg_nearest_dist)
    
    density_bonus = np.maximum(density_bonus, density_bonus.T)
    
    # 5. Combine all factors
    heuristics = inv_dist * (1.0 + 0.5 * mst_bonus + 0.3 * density_bonus)
    heuristics = heuristics * mask  # Apply sparsification
    
    # 6. Additional sparsification: keep only top 30% of non-zero values per row
    for i in range(n):
        row = heuristics[i]
        non_zero = row[row > 0]
        if len(non_zero) > 0:
            threshold = np.percentile(non_zero, 70)  # Keep top 30%
            row[row < threshold] = 0
        heuristics[i] = row
    
    # 7. Normalize rows to sum to 1 for probability interpretation
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1  # Avoid division by zero
    heuristics = heuristics / row_sums
    
    return heuristics
