import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with improved two-step
    lookahead and rank-based regret minimization.
    
    Core innovations:
    1. Two-step lookahead planning: Anticipates next move for more informed scoring
    2. Enhanced MST priority: Dynamic MST weighting based on edge centrality
    3. Rank-based regret: Considers multiple alternatives via rank normalization
    4. Adaptive density similarity: Variable neighborhood size for cluster detection
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Base reciprocal distance with safe handling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.diag_indices(n)] = 0
    
    # Adaptive neighborhood parameter
    k_neighbors = max(3, min(15, int(0.2 * n)))
    
    # 1. TWO-STEP LOOKAHEAD PLANNING
    lookahead_matrix = np.zeros((n, n))
    
    # Precompute nearest neighbors for each node
    nearest_dists = np.zeros((n, n))
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_indices = np.argsort(dists)
        nearest_dists[i] = dists[sorted_indices[np.arange(n)]]
    
    for i in range(n):
        for j in range(i + 1, n):
            # Two-step cost: i->j + j->k (where k is j's nearest excluding i)
            dist_ij = distance_matrix[i, j]
            
            # Find j's nearest neighbor excluding i
            j_dists = distance_matrix[j].copy()
            j_dists[i] = np.inf
            j_dists[j] = np.inf
            nearest_k = np.argmin(j_dists)
            dist_jk = j_dists[nearest_k]
            
            # Lookahead heuristic: prefer edges that lead to good next moves
            lookahead_score = 1.0 / (dist_ij + 0.5 * dist_jk + 1e-8)
            lookahead_matrix[i, j] = lookahead_matrix[j, i] = lookahead_score
    
    # Normalize lookahead component
    max_lookahead = np.max(lookahead_matrix)
    if max_lookahead > 0:
        lookahead_matrix = lookahead_matrix / max_lookahead
    
    # 2. ENHANCED MST PRIORITY WITH CENTRALITY WEIGHTING
    mst = minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    mst_mask = (mst_array > 0) | (mst_array.T > 0)
    
    # Compute edge degree in MST for dynamic weighting
    mst_degree = np.zeros(n)
    for i in range(n):
        mst_degree[i] = np.sum(mst_mask[i])
    
    mst_bonus = np.ones((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            if mst_mask[i, j]:
                # Higher bonus for edges connecting high-degree MST nodes
                degree_bonus = 1.0 + 0.3 * (mst_degree[i] + mst_degree[j]) / max(1, np.max(mst_degree))
                mst_bonus[i, j] = mst_bonus[j, i] = degree_bonus
    
    # 3. RANK-BASED REGRET MINIMIZATION
    regret_matrix = np.ones((n, n))
    
    for i in range(n):
        # Get distances and sort
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_indices = np.argsort(dists)
        sorted_dists = dists[sorted_indices]
        
        # Calculate rank-based regret for top k alternatives
        for rank, j in enumerate(sorted_indices[:k_neighbors]):
            if rank == 0:
                # Best option: maximum priority
                regret = 1.5
            else:
                # Regret = how much worse is this than the best option?
                regret = 1.0 / (1.0 + (sorted_dists[rank] - sorted_dists[0]) / (sorted_dists[0] + 1e-8))
            
            regret_matrix[i, j] += regret
            regret_matrix[j, i] += regret
    
    # Normalize regret component
    max_regret = np.max(regret_matrix)
    if max_regret > 0:
        regret_matrix = regret_matrix / max_regret
    
    # 4. ADAPTIVE DENSITY SIMILARITY BONUS
    density_bonus = np.ones((n, n))
    
    if n > 5:
        # Compute local density for each node with adaptive neighborhood
        local_density = np.zeros(n)
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest_dists = np.sort(dists)[:k_neighbors]
            local_density[i] = 1.0 / (np.mean(nearest_dists) + 1e-8)
        
        # Normalize densities to [0, 1]
        max_density = np.max(local_density)
        if max_density > 0:
            local_density = local_density / max_density
        
        # Bonus for edges connecting nodes with similar densities
        for i in range(n):
            for j in range(i + 1, n):
                density_diff = abs(local_density[i] - local_density[j])
                similarity = np.exp(-density_diff * 2.0)
                density_bonus[i, j] = density_bonus[j, i] = 1.0 + 0.6 * similarity
    
    # COMBINE ALL FACTORS with balanced weighting
    combined = inv_dist * (0.4 * lookahead_matrix + 0.3 * regret_matrix + 
                          0.2 * mst_bonus + 0.1 * density_bonus)
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    # INTELLIGENT SPARSIFICATION
    sparsified = combined.copy()
    
    # Strategy 1: Keep MST edges at full strength
    sparsified[~mst_mask] *= 0.7  # Reduce non-MST edges
    
    # Strategy 2: Keep top connections per node with adaptive threshold
    for i in range(n):
        node_scores = sparsified[i].copy()
        node_scores[i] = -np.inf
        threshold = np.percentile(node_scores[node_scores > 0], 65)  # Keep top 35%
        sparsified[i][node_scores < threshold] = 0
    
    # Strategy 3: Ensure connectivity - each node must connect to at least 2 others
    for i in range(n):
        connections = np.sum(sparsified[i] > 0)
        if connections < 2:
            # Add nearest neighbors by base distance
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:2]
            sparsified[i, nearest] = combined[i, nearest]
            sparsified[nearest, i] = combined[nearest, i]
    
    # FINAL NORMALIZATION
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    # Zero out diagonal
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
