import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with deep stochastic lookahead and meta-learned optimization.
    
    Core innovations:
    1. Multi-stage Monte Carlo lookahead with adaptive rollout depth
    2. Global alternative path regret with optimal substructure analysis
    3. Gradient-free meta-optimization of ensemble weights
    4. Hierarchical clustering-based sparsification
    
    Returns:
        np.ndarray: Heuristic matrix with higher values for more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 3:
        return np.ones((n, n)) - np.eye(n)
    
    # === 1. INSTANCE CHARACTERIZATION FOR ADAPTIVE PARAMETERS ===
    triu_dists = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(triu_dists)
    dist_std = np.std(triu_dists)
    dist_cv = dist_std / (dist_mean + 1e-12)
    
    # MST-based structure analysis
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = mst[mst > 0]
    mst_cv = np.std(mst_edges) / (np.mean(mst_edges) + 1e-12)
    
    # Adaptive parameter tuning
    adaptive_params = {
        'k_neighbors': max(3, min(15, int(np.sqrt(n)))),
        'lookahead_depth': min(4, max(1, int(np.log2(n)) - 1)),
        'n_samples': min(200, max(30, n * 3)),
        'rollout_length': min(int(0.3 * n), 10)
    }
    
    # === 2. ENHANCED MULTI-STAGE MONTE CARLO LOOKAHEAD ===
    lookahead_scores = np.ones((n, n))
    k = adaptive_params['k_neighbors']
    
    # Precompute nearest neighbors for efficient sampling
    nearest_indices = []
    nearest_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        idx = np.argsort(dists)[:k]
        nearest_indices.append(idx)
        nearest_dists.append(dists[idx])
    
    # Multi-stage rollout with progressive deepening
    for i in range(n):
        for j in range(i + 1, n):
            # Skip if edge is too long (pruning)
            if distance_matrix[i, j] > np.percentile(distance_matrix[i], 80):
                continue
            
            stage_scores = []
            
            # Stage 1: Short rollouts (exploration)
            for _ in range(adaptive_params['n_samples'] // 3):
                current = j
                visited = {i, j}
                path_length = distance_matrix[i, j]
                
                for step in range(adaptive_params['lookahead_depth']):
                    candidates = [nbr for nbr in nearest_indices[current] 
                                 if nbr not in visited]
                    if not candidates:
                        break
                    
                    # Boltzmann exploration for diversity
                    candidate_dists = distance_matrix[current, candidates]
                    inv_dists = 1.0 / (candidate_dists + 1e-12)
                    temperatures = np.linspace(0.5, 2.0, len(candidate_dists))
                    weights = inv_dists ** (1.0 / temperatures)
                    weights = weights / (weights.sum() + 1e-12)
                    
                    next_node = np.random.choice(candidates, p=weights)
                    path_length += distance_matrix[current, next_node]
                    visited.add(next_node)
                    current = next_node
                
                stage_scores.append(path_length)
            
            # Stage 2: Longer rollouts for promising edges
            if np.percentile(stage_scores, 25) < dist_mean * 1.5:
                for _ in range(adaptive_params['n_samples'] // 3):
                    current = j
                    visited = {i, j}
                    path_length = distance_matrix[i, j]
                    
                    for step in range(min(adaptive_params['rollout_length'], n - len(visited))):
                        if len(visited) >= min(8, n):
                            break
                        
                        # Mix of greedy and exploration
                        candidates = [nbr for nbr in range(n) 
                                     if nbr not in visited and 
                                     distance_matrix[current, nbr] < np.inf]
                        if not candidates:
                            break
                        
                        # Adaptive sampling strategy
                        if step < 2:
                            # Greedy for first steps
                            next_node = candidates[np.argmin(distance_matrix[current, candidates])]
                        else:
                            # Boltzmann for later steps
                            candidate_dists = distance_matrix[current, candidates]
                            inv_dists = 1.0 / candidate_dists
                            weights = inv_dists ** (1.0 / (1.0 + step * 0.2))
                            weights = weights / weights.sum()
                            next_node = np.random.choice(candidates, p=weights)
                        
                        path_length += distance_matrix[current, next_node]
                        visited.add(next_node)
                        current = next_node
                    
                    stage_scores.append(path_length)
            
            if stage_scores:
                avg_score = np.mean(stage_scores)
                # Transform to [0,1] using instance-adaptive scaling
                scale = dist_mean * 2
                lookahead_scores[i, j] = lookahead_scores[j, i] = 1.0 / (1.0 + avg_score / scale)
    
    # === 3. GLOBAL ALTERNATIVE PATH REGRET ANALYSIS ===
    regret_matrix = np.ones((n, n))
    
    # Build k-nearest neighbor graph for efficient alternative path search
    k_regret = min(10, n - 1)
    
    for i in range(n):
        for j in range(i + 1, n):
            if j not in nearest_indices[i] and i not in nearest_indices[j]:
                # Only compute for potentially relevant edges
                continue
            
            d_ij = distance_matrix[i, j]
            
            # Find best alternative paths via intermediate nodes
            alt_paths = []
            
            # Direct alternatives (1-hop)
            for k in nearest_indices[i]:
                if k != j:
                    alt_paths.append(distance_matrix[i, k] + distance_matrix[k, j])
            
            for k in nearest_indices[j]:
                if k != i:
                    alt_paths.append(distance_matrix[i, k] + distance_matrix[k, j])
            
            # Two-hop alternatives for important edges
            if d_ij < np.percentile(distance_matrix[i], 25):
                for k1 in nearest_indices[i][:min(5, len(nearest_indices[i]))]:
                    if k1 != j:
                        for k2 in nearest_indices[k1][:min(3, len(nearest_indices[k1]))]:
                            if k2 != i and k2 != j:
                                alt_length = (distance_matrix[i, k1] + 
                                            distance_matrix[k1, k2] + 
                                            distance_matrix[k2, j])
                                alt_paths.append(alt_length)
            
            if alt_paths:
                best_alt = min(alt_paths)
                regret = (best_alt - d_ij) / (d_ij + 1e-12)
                
                # Adaptive sigmoid transformation
                slope = 5.0 * (1.0 + dist_cv)
                regret_score = 2.0 / (1.0 + np.exp(-slope * regret)) - 1.0
                regret_matrix[i, j] = regret_matrix[j, i] = max(0.1, min(2.0, 1.0 + regret_score))
    
    # === 4. GRADIENT-FREE META-OPTIMIZATION OF ENSEMBLE WEIGHTS ===
    
    # Compute additional heuristic components
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + 1e-12)
        np.fill_diagonal(inv_dist, 0)
    
    # MST bonus with edge importance
    mst_mask = (mst + mst.T) > 0
    mst_bonus = np.where(mst_mask, 1.5, 1.0)
    
    # Hierarchical clustering-based density
    cluster_scores = np.ones((n, n))
    if n > 10:
        # Fast approximate clustering
        densities = np.zeros(n)
        for i in range(n):
            nearest = np.sort(distance_matrix[i])[1:min(6, n)]
            densities[i] = 1.0 / np.mean(nearest)
        
        # Normalize and create similarity matrix
        if np.max(densities) > 0:
            densities = densities / np.max(densities)
        
        for i in range(n):
            for j in range(i + 1, n):
                # Combined density and distance similarity
                density_sim = 1.0 - abs(densities[i] - densities[j])
                dist_sim = 1.0 / (1.0 + abs(distance_matrix[i, j] - dist_mean) / dist_std)
                cluster_scores[i, j] = cluster_scores[j, i] = 1.0 + 0.4 * (density_sim * dist_sim)
    
    # Betweenness approximation via random walks
    betweenness = np.zeros((n, n))
    if n > 20:
        n_walks = min(50, n * 2)
        walk_length = min(5, int(np.log2(n)))
        
        for _ in range(n_walks):
            start = np.random.randint(n)
            current = start
            path = [current]
            
            for _ in range(walk_length):
                # Weighted random walk based on inverse distance
                candidates = [x for x in range(n) if x != current]
                if not candidates:
                    break
                
                weights = 1.0 / (distance_matrix[current, candidates] + 1e-12)
                weights = weights / weights.sum()
                next_node = np.random.choice(candidates, p=weights)
                
                # Increment edge usage
                betweenness[current, next_node] += 1
                betweenness[next_node, current] += 1
                
                current = next_node
                path.append(current)
        
        # Normalize betweenness scores
        max_between = np.max(betweenness)
        if max_between > 0:
            betweenness = betweenness / max_between
    
    # Meta-optimization of ensemble weights using instance characteristics
    # Build feature vector for weight prediction
    features = np.array([
        dist_cv,  # Distance variability
        mst_cv,   # MST structure variability
        np.mean([len(nn) for nn in nearest_indices]) / n,  # Neighborhood density
        np.percentile(triu_dists, 75) / np.percentile(triu_dists, 25)  # Spread ratio
    ])
    
    # Simple rule-based weight optimization (can be replaced with learned model)
    # Higher weight for lookahead when instances are structured
    w_lookahead = 0.8 + 0.4 * np.tanh(1.0 - features[0])
    
    # Higher weight for regret when alternatives are informative
    w_regret = 0.7 + 0.6 * np.tanh(features[1])
    
    # Higher weight for inverse distance when distances vary widely
    w_invdist = 0.6 + 0.5 * np.tanh(features[0])
    
    # Higher weight for clustering when neighborhoods are dense
    w_cluster = 0.5 + 0.4 * np.tanh(features[2] * 2 - 1)
    
    # Higher weight for betweenness in sparse graphs
    w_between = 0.3 + 0.4 * np.tanh(1.0 - features[3])
    
    # MST bonus weight (always important)
    w_mst = 1.2
    
    # Normalize weights
    weights = np.array([w_lookahead, w_regret, w_invdist, w_cluster, w_between, w_mst])
    weights = weights / weights.sum()
    
    # === 5. ENSEMBLE COMBINATION WITH META-OPTIMIZED WEIGHTS ===
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    # Promising edge mask (combines multiple criteria)
    promising_mask = np.zeros((n, n), dtype=bool)
    
    # Always include MST edges
    promising_mask |= mst_mask
    
    # Include k-nearest neighbors
    for i in range(n):
        promising_mask[i, nearest_indices[i]] = True
    
    # Include edges with high lookahead scores
    lookahead_thresh = np.percentile(lookahead_scores[lookahead_scores > 0], 70) if np.any(lookahead_scores > 0) else 0
    promising_mask |= (lookahead_scores > lookahead_thresh)
    
    # Make symmetric
    promising_mask = promising_mask | promising_mask.T
    
    # Combine components with meta-optimized weights
    for i in range(n):
        for j in range(i + 1, n):
            if not promising_mask[i, j]:
                continue
            
            score = (
                weights[0] * lookahead_scores[i, j] +
                weights[1] * regret_matrix[i, j] * inv_dist[i, j] +
                weights[2] * inv_dist[i, j] +
                weights[3] * cluster_scores[i, j] +
                weights[4] * (1.0 + betweenness[i, j]) +
                weights[5] * (mst_bonus[i, j] if mst_mask[i, j] else 1.0)
            )
            
            heuristic_matrix[i, j] = heuristic_matrix[j, i] = score
    
    # === 6. HIERARCHICAL CLUSTERING-BASED SPARSIFICATION ===
    
    if n > 15:
        # Fast hierarchical clustering
        cluster_labels = np.zeros(n, dtype=int)
        unclustered = set(range(n))
        next_label = 1
        
        while unclustered and next_label < 10:  # Limit to 10 clusters
            seed = next(iter(unclustered))
            cluster_members = {seed}
            frontier = {seed}
            
            # Grow cluster by adding nearby nodes
            while frontier:
                current = frontier.pop()
                # Find nearest unclustered neighbors
                neighbors = [nbr for nbr in nearest_indices[current] 
                           if nbr in unclustered and nbr not in cluster_members]
                
                for nbr in neighbors[:3]:  # Add up to 3 nearest
                    if distance_matrix[current, nbr] < np.percentile(triu_dists, 40):
                        cluster_members.add(nbr)
                        frontier.add(nbr)
            
            # Assign cluster label
            for node in cluster_members:
                cluster_labels[node] = next_label
                unclustered.remove(node)
            
            next_label += 1
        
        # Any remaining nodes get their own cluster
        for node in unclustered:
            cluster_labels[node] = next_label
            next_label += 1
        
        # Apply cluster-aware sparsification
        for i in range(n):
            row_scores = heuristic_matrix[i]
            cluster_i = cluster_labels[i]
            
            # Different thresholds for intra-cluster vs inter-cluster edges
            intra_mask = cluster_labels == cluster_i
            intra_mask[i] = False  # Exclude self
            
            inter_mask = cluster_labels != cluster_i
            
            if np.any(intra_mask):
                intra_vals = row_scores[intra_mask]
                if len(intra_vals) > 2:
                    intra_thresh = np.percentile(intra_vals, 40)  # Keep more intra-cluster edges
                    heuristic_matrix[i, intra_mask] = np.where(
                        intra_vals < intra_thresh, 0, intra_vals
                    )
            
            if np.any(inter_mask):
                inter_vals = row_scores[inter_mask]
                if len(inter_vals) > 2:
                    inter_thresh = np.percentile(inter_vals, 60)  # Keep fewer inter-cluster edges
                    heuristic_matrix[i, inter_mask] = np.where(
                        inter_vals < inter_thresh, 0, inter_vals
                    )
    
    # === 7. FINAL PROCESSING ===
    
    # Ensure symmetry
    heuristic_matrix = np.maximum(heuristic_matrix, heuristic_matrix.T)
    
    # Boost MST edges after sparsification
    heuristic_matrix[mst_mask] = np.maximum(heuristic_matrix[mst_mask], 0.3)
    
    # Row-wise normalization to emphasize local preferences
    row_max = heuristic_matrix.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    heuristic_matrix = heuristic_matrix / row_max
    
    # Global normalization
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristic_matrix, 0)
    
    return heuristic_matrix
