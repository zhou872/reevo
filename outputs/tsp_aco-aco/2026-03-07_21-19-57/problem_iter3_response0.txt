import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection combining multiple
    algorithmic strategies with improved computational efficiency.
    
    Core features:
    1. Two-step Lookahead Edge Scoring: Estimates future connection costs
    2. Rank-Based Regret Minimization: Prioritizes edges with poor alternatives
    3. Local Density Similarity Bonus: Promotes intra-cluster connections
    4. Minimum Spanning Tree Priority: Leverages structural importance
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Base reciprocal distance with safe handling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.diag_indices(n)] = 0
    
    # Adaptive neighborhood parameter
    k = max(3, int(0.15 * n))
    
    # 1. TWO-STEP LOOKAHEAD EDGE SCORING
    lookahead_matrix = np.ones((n, n))
    
    # Precompute second-best distances for each node
    second_best_dist = np.zeros(n)
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        sorted_dists = np.sort(distances)
        second_best_dist[i] = sorted_dists[1] if n > 2 else 0
    
    for i in range(n):
        for j in range(i + 1, n):
            # Two-step cost: current edge + best continuation from j
            continuation_cost = distance_matrix[i, j] + second_best_dist[j]
            lookahead_matrix[i, j] = lookahead_matrix[j, i] = 1.0 / (1.0 + continuation_cost)
    
    # 2. RANK-BASED REGRET MINIMIZATION
    regret_matrix = np.ones((n, n))
    
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Find rank of each edge
        sorted_indices = np.argsort(distances)
        sorted_distances = distances[sorted_indices]
        
        # Compute regret based on rank difference
        for rank, j in enumerate(sorted_indices):
            if rank < len(sorted_indices) - 1:
                # Regret is normalized difference to next alternative
                regret = (sorted_distances[rank + 1] - sorted_distances[rank]) / (sorted_distances[rank] + 1e-8)
                regret_matrix[i, j] += 0.5 * regret  # Weighted contribution
    
    # Normalize regret component
    max_regret = np.max(regret_matrix)
    if max_regret > 0:
        regret_matrix = regret_matrix / max_regret
    
    # 3. LOCAL DENSITY SIMILARITY BONUS
    density_bonus = np.ones((n, n))
    
    # Compute local density for each node
    local_density = np.zeros(n)
    for i in range(n):
        # Use k nearest neighbors for density estimation
        nearest_dists = np.partition(distance_matrix[i, :], kth=min(k, n-1) + 1)[1:k+2]
        local_density[i] = 1.0 / (np.mean(nearest_dists) + 1e-8)
    
    # Normalize densities
    max_density = np.max(local_density)
    if max_density > 0:
        local_density = local_density / max_density
    
    # Apply similarity bonus
    for i in range(n):
        for j in range(i + 1, n):
            density_similarity = 1.0 - abs(local_density[i] - local_density[j])
            density_bonus[i, j] = density_bonus[j, i] = 1.0 + 0.3 * density_similarity
    
    # 4. MINIMUM SPANNING TREE PRIORITY
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    mst_priority = np.where(mst_mask, 1.5, 1.0)  # 50% boost for MST edges
    
    # COMBINE ALL FACTORS with weighted contributions
    combined = (
        inv_dist * 0.4 +           # Base distance importance
        lookahead_matrix * 0.3 +   # Future planning
        regret_matrix * 0.2 +      # Opportunity cost
        density_bonus * 0.1        # Cluster coherence
    ) * mst_priority  # MST acts as multiplier
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    # INTELLIGENT SPARSIFICATION
    sparsified = combined.copy()
    
    # Strategy 1: Keep top edges per node (adaptive threshold)
    edges_per_node = max(3, int(0.2 * n))  # Keep 20% of edges per node, minimum 3
    for i in range(n):
        row = combined[i, :].copy()
        row[i] = -np.inf  # Exclude self
        threshold = np.partition(row, -edges_per_node)[-edges_per_node]
        sparsified[i, :] = np.where(combined[i, :] >= threshold, combined[i, :], 0)
    
    # Strategy 2: Always include MST edges
    sparsified[mst_mask] = combined[mst_mask]
    
    # Strategy 3: Ensure minimum connectivity
    min_connections = 2
    for i in range(n):
        if np.sum(sparsified[i, :] > 0) < min_connections:
            # Add nearest neighbors not already included
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            distances[sparsified[i, :] > 0] = np.inf  # Already included
            additional = np.argsort(distances)[:min_connections]
            sparsified[i, additional] = combined[i, additional]
            sparsified[additional, i] = combined[additional, i]
    
    # NORMALIZATION for consistency
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
