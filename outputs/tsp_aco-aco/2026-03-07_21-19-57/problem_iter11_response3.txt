import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with improved
    regret-based prioritization and two-step lookahead scoring.
    
    Core improvements over v1:
    1. Regret-Based Edge Prioritization: Enhanced regret calculation considering
       opportunity costs at both endpoints with second-best alternatives
    2. Two-Step Lookahead Spatial Scoring: Multi-stage spatial analysis that
       considers both immediate and secondary connection opportunities
    3. Adaptive Density Similarity Bonus: Exponential bonus based on local density
       similarity to encourage intra-cluster connections
    4. Minimum Spanning Tree Structural Priority: MST-based structural preservation
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Initialize heuristic matrix
    heuristic = np.zeros((n, n))
    np.fill_diagonal(heuristic, 0)
    
    # 1. REGRET-BASED EDGE PRIORITIZATION (Enhanced from v1)
    regret_matrix = np.ones((n, n))
    
    for i in range(n):
        # Get distances from node i to all others
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Find best and second best distances for regret calculation
        sorted_indices = np.argsort(distances)
        sorted_distances = distances[sorted_indices]
        
        if len(sorted_distances) >= 2:
            best_dist = sorted_distances[0]
            second_best_dist = sorted_distances[1]
            
            # Calculate regret for each edge from node i's perspective
            for j in range(n):
                if i == j:
                    continue
                    
                dist_ij = distance_matrix[i, j]
                if dist_ij == best_dist:
                    # Best edge gets highest regret-based bonus
                    regret_i = (second_best_dist - best_dist) / (best_dist + 1e-8)
                    regret_matrix[i, j] = 1.0 + min(1.0, max(0.1, regret_i))
                else:
                    # Other edges get penalty based on how much worse they are
                    regret_penalty = max(0, (dist_ij - best_dist) / (best_dist + 1e-8))
                    regret_matrix[i, j] = 1.0 / (1.0 + regret_penalty)
    
    # Make regret matrix symmetric by taking maximum of both directions
    regret_symmetric = np.maximum(regret_matrix, regret_matrix.T)
    
    # 2. TWO-STEP LOOKAHEAD SPATIAL SCORING
    lookahead_matrix = np.ones((n, n))
    
    # Compute local connection quality metrics
    k = max(3, int(0.1 * n))
    local_avg_dist = np.zeros(n)
    
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        if len(distances) > k:
            k_smallest = np.partition(distances, min(k, len(distances)-1))[:min(k, len(distances)-1)]
            local_avg_dist[i] = np.mean(k_smallest[k_smallest < np.inf])
        else:
            valid_dists = distances[distances < np.inf]
            local_avg_dist[i] = np.mean(valid_dists) if len(valid_dists) > 0 else 1.0
    
    # Apply two-step lookahead scoring
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            dist_ij = distance_matrix[i, j]
            
            # Step 1: Immediate spatial quality
            threshold_i = 1.5 * local_avg_dist[i]
            threshold_j = 1.5 * local_avg_dist[j]
            
            # Step 2: Lookahead to next-best connections
            # For node i, find best alternative to j
            distances_i = distance_matrix[i, :].copy()
            distances_i[i] = distances_i[j] = np.inf
            best_alt_i = np.min(distances_i) if np.any(distances_i < np.inf) else dist_ij
            
            # For node j, find best alternative to i
            distances_j = distance_matrix[j, :].copy()
            distances_j[j] = distances_j[i] = np.inf
            best_alt_j = np.min(distances_j) if np.any(distances_j < np.inf) else dist_ij
            
            # Calculate lookahead score
            alt_ratio_i = best_alt_i / (dist_ij + 1e-8)
            alt_ratio_j = best_alt_j / (dist_ij + 1e-8)
            
            # Combined score favors edges with poor alternatives
            lookahead_score = (1.0 / (alt_ratio_i + alt_ratio_j + 1e-8))
            
            # Apply spatial proximity bonus/penalty
            if dist_ij <= min(threshold_i, threshold_j):
                proximity_bonus = 1.0 + 0.4 * (1.0 - dist_ij / (min(threshold_i, threshold_j) + 1e-8))
            else:
                proximity_bonus = 0.7 * (min(threshold_i, threshold_j) / (dist_ij + 1e-8))
            
            lookahead_matrix[i, j] = lookahead_score * proximity_bonus
    
    # 3. ADAPTIVE DENSITY SIMILARITY BONUS
    density_matrix = np.ones((n, n))
    
    # Compute local density using adaptive k-nearest neighbors
    local_density = np.zeros(n)
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Adaptive k based on problem size
        adaptive_k = min(max(3, int(0.15 * n)), n-1)
        if len(distances) > adaptive_k:
            kth_smallest = np.partition(distances, adaptive_k)[adaptive_k]
            nearest_mask = distances <= kth_smallest
            nearest_dists = distances[nearest_mask]
        else:
            nearest_dists = distances[distances < np.inf]
        
        if len(nearest_dists) > 0:
            # Use robust scaling for density
            local_density[i] = 1.0 / (np.median(nearest_dists) + 1e-8)
        else:
            local_density[i] = 1.0
    
    # Normalize densities
    if np.max(local_density) > 0:
        local_density = local_density / np.max(local_density)
    
    # Apply exponential similarity bonus with stronger intra-cluster preference
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            density_diff = abs(local_density[i] - local_density[j])
            # Stronger bonus for similar densities (intra-cluster)
            density_matrix[i, j] = 1.0 + 0.6 * np.exp(-3.0 * density_diff)
    
    # 4. MINIMUM SPANNING TREE STRUCTURAL PRIORITY
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    mst_boost = np.where(mst_mask, 1.4, 1.0)  # 40% boost for MST edges
    
    # COMBINE ALL COMPONENTS with weighted emphasis
    # Base component: inverse distance with stronger regularization
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + 0.1 * np.mean(distance_matrix[distance_matrix > 0]))
        inv_dist[np.diag_indices(n)] = 0
    
    # Combined heuristic with emphasis on regret and lookahead
    heuristic = inv_dist * regret_symmetric * lookahead_matrix * density_matrix * mst_boost
    
    # Ensure symmetry
    heuristic = np.maximum(heuristic, heuristic.T)
    
    # 5. INTELLIGENT SPARSIFICATION WITH REGRET-AWARE THRESHOLDING
    final_mask = np.zeros((n, n), dtype=bool)
    
    # Always include MST edges
    final_mask = final_mask | mst_mask
    
    # Compute node-specific thresholds
    avg_nn_dist = np.zeros(n)
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        k_neighbors = min(max(3, int(0.1 * n)), n-1)
        if len(distances) > k_neighbors:
            k_smallest = np.partition(distances, k_neighbors)[:k_neighbors]
            avg_nn_dist[i] = np.mean(k_smallest[k_smallest < np.inf])
        else:
            valid_dists = distances[distances < np.inf]
            avg_nn_dist[i] = np.mean(valid_dists) if len(valid_dists) > 0 else 1.0
    
    # Multi-criteria edge selection
    for i in range(n):
        # Criterion 1: High regret edges (top 2 by regret matrix)
        node_regrets = regret_symmetric[i, :].copy()
        node_regrets[i] = -np.inf
        top_regret_indices = np.argpartition(node_regrets, -2)[-2:]
        final_mask[i, top_regret_indices] = True
        final_mask[top_regret_indices, i] = True
        
        # Criterion 2: Top combined heuristic edges
        node_heuristics = heuristic[i, :].copy()
        node_heuristics[i] = -np.inf
        top_k = max(3, int(0.2 * n))  # Top 20% or at least 3
        top_indices = np.argpartition(node_heuristics, -top_k)[-top_k:]
        final_mask[i, top_indices] = True
        final_mask[top_indices, i] = True
        
        # Criterion 3: Spatial proximity with dynamic threshold
        dynamic_threshold = 1.8 * avg_nn_dist[i]
        within_threshold = distance_matrix[i, :] <= dynamic_threshold
        within_threshold[i] = False
        final_mask[i, within_threshold] = True
        final_mask[within_threshold, i] = True
    
    # Ensure minimum connectivity of 2 for each node
    for i in range(n):
        connected = np.where(final_mask[i, :])[0]
        connected = connected[connected != i]
        if len(connected) < 2:
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            # Add the closest nodes not already connected
            candidates = np.argsort(distances)
            added = 0
            for j in candidates:
                if j != i and not final_mask[i, j] and added < 2:
                    final_mask[i, j] = True
                    final_mask[j, i] = True
                    added += 1
                if added >= 2:
                    break
    
    # Apply sparsification mask
    heuristic = heuristic * final_mask
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    return heuristic
