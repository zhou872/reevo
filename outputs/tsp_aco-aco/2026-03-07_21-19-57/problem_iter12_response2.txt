import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with adaptive component fusion and global
    structure awareness. Key innovations:
    
    1. Regret-aware adaptive weighting: Dynamically weights heuristic components
       based on instance characteristics using a meta-heuristic approach
    2. Kernel density clustering: Uses Gaussian kernel density estimation for
       more robust cluster detection
    3. Multi-step path opportunity cost: Efficient approximate multi-step
       lookahead using beam search principles
    4. Tour structure penalty: Penalizes edges that disrupt natural geometric
       patterns in optimal TSP tours
    
    Args:
        distance_matrix: Square symmetric matrix of distances between nodes
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # ============================================
    # 1. INSTANCE CHARACTERIZATION FOR ADAPTIVE WEIGHTS
    # ============================================
    # Compute instance statistics
    flat_dists = distance_matrix[np.triu_indices(n, 1)]
    mean_dist = np.mean(flat_dists)
    std_dist = np.std(flat_dists)
    cv_dist = std_dist / (mean_dist + 1e-8)  # Coefficient of variation
    
    # Spatial distribution analysis
    k_local = max(3, n // 20)
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k_local)[:k_local]
        local_density[i] = np.mean(k_nearest)
    
    density_cv = np.std(local_density) / (np.mean(local_density) + 1e-8)
    
    # ============================================
    # 2. KERNEL DENSITY CLUSTERING COMPONENT
    # ============================================
    # Use kernel density estimation for clustering
    density_matrix = np.ones((n, n))
    
    # Bandwidth selection based on instance characteristics
    bandwidth = np.median(local_density) * 0.5
    
    # Compute kernel densities using Gaussian kernel
    for i in range(n):
        for j in range(i+1, n):
            # Gaussian kernel similarity based on local neighborhood overlap
            neighbors_i = np.argsort(distance_matrix[i])[1:k_local+1]
            neighbors_j = np.argsort(distance_matrix[j])[1:k_local+1]
            
            # Jaccard similarity of neighborhoods
            overlap = len(set(neighbors_i) & set(neighbors_j))
            jaccard = overlap / len(set(neighbors_i) | set(neighbors_j))
            
            # Gaussian kernel based on distance
            kernel_val = np.exp(-distance_matrix[i, j]**2 / (2 * bandwidth**2))
            
            # Combined clustering score
            density_score = 0.7 * kernel_val + 0.3 * jaccard
            density_matrix[i, j] = 1.0 + 0.5 * density_score
            density_matrix[j, i] = density_matrix[i, j]
    
    # ============================================
    # 3. MULTI-STEP PATH OPPORTUNITY COST
    # ============================================
    opp_matrix = np.ones((n, n))
    
    # Adaptive lookahead depth based on instance size
    lookahead_depth = min(4, max(2, int(np.log2(n)) - 2))
    
    # Pre-sort distances for efficiency
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_dists = np.sort(distance_matrix, axis=1)
    
    for i in range(n):
        # Get baseline alternative costs
        base_alts = sorted_dists[i, 1:min(5, n)]
        base_alt_cost = np.mean(base_alts)
        
        for j in range(n):
            if i == j:
                continue
                
            dist_ij = distance_matrix[i, j]
            opportunity_cost = 0.0
            
            # Beam search for alternative paths
            beam_width = min(3, n-1)
            beam_candidates = sorted_indices[i, 1:beam_width+1]
            beam_candidates = [c for c in beam_candidates if c != j]
            
            if beam_candidates:
                for alt in beam_candidates:
                    alt_path_cost = distance_matrix[i, alt]
                    
                    # Estimate continuation from alternative node
                    remaining = [k for k in range(n) if k not in [i, j, alt]]
                    if remaining:
                        # Use minimum spanning tree approximation for remaining nodes
                        sub_dist = distance_matrix[alt][remaining]
                        alt_path_cost += np.mean(np.partition(sub_dist, min(2, len(sub_dist)-1))[:min(2, len(sub_dist))])
                    
                    # Multi-step regret with discount factor
                    regret = max(0, dist_ij - alt_path_cost)
                    opportunity_cost += regret / (dist_ij + 1e-8)
                
                opportunity_cost /= len(beam_candidates)
            
            # Transform to heuristic score (higher for lower opportunity cost)
            opp_matrix[i, j] = np.exp(-opportunity_cost * (1.0 + 0.5 * cv_dist))
    
    # Symmetrize
    opp_matrix = (opp_matrix + opp_matrix.T) / 2
    
    # ============================================
    # 4. TOUR STRUCTURE PENALTY
    # ============================================
    # Penalize edges that break natural geometric patterns
    structure_matrix = np.ones((n, n))
    
    # Minimum spanning tree for backbone
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Compute angular consistency for potential edges
    if n >= 3:
        # Approximate convex hull for boundary detection
        from scipy.spatial import ConvexHull
        try:
            # Use 2D MDS embedding for geometric structure
            from sklearn.manifold import MDS
            embedding = MDS(n_components=2, dissimilarity='precomputed', 
                           random_state=42, normalized_stress='auto')
            coords = embedding.fit_transform(distance_matrix)
            
            # Compute angular relationships
            for i in range(n):
                for j in range(i+1, n):
                    if mst_mask[i, j]:
                        structure_matrix[i, j] = 1.2
                        structure_matrix[j, i] = 1.2
                    else:
                        # Penalize edges that cross many other edges in the embedding
                        vec_ij = coords[j] - coords[i]
                        norm_ij = np.linalg.norm(vec_ij)
                        
                        if norm_ij > 0:
                            # Check angular alignment with nearest neighbors
                            neighbors_i = sorted_indices[i, 1:min(4, n)]
                            alignment_score = 0.0
                            count = 0
                            
                            for k in neighbors_i:
                                if k != j:
                                    vec_ik = coords[k] - coords[i]
                                    norm_ik = np.linalg.norm(vec_ik)
                                    if norm_ik > 0:
                                        cos_angle = np.dot(vec_ij, vec_ik) / (norm_ij * norm_ik)
                                        alignment_score += (cos_angle + 1) / 2  # Map to [0,1]
                                        count += 1
                            
                            if count > 0:
                                alignment_score /= count
                                structure_matrix[i, j] = 0.8 + 0.4 * alignment_score
                                structure_matrix[j, i] = structure_matrix[i, j]
        except:
            # Fallback: simple distance-based structure
            for i in range(n):
                for j in range(i+1, n):
                    if mst_mask[i, j]:
                        structure_matrix[i, j] = 1.2
                    else:
                        # Inverse distance fallback
                        structure_matrix[i, j] = 1.0 / (1.0 + 0.1 * distance_matrix[i, j])
                    structure_matrix[j, i] = structure_matrix[i, j]
    
    # ============================================
    # 5. ADAPTIVE COMPONENT FUSION
    # ============================================
    # Base component: inverse distance
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + 0.1 * mean_dist)
    np.fill_diagonal(inv_dist, 0)
    
    # Normalize components to similar scales
    def safe_normalize(mat):
        mask = ~np.eye(n, dtype=bool)
        valid_vals = mat[mask]
        if len(valid_vals) > 0 and np.max(valid_vals) > np.min(valid_vals):
            mat_norm = mat.copy()
            mat_norm[mask] = (valid_vals - np.min(valid_vals)) / (np.max(valid_vals) - np.min(valid_vals))
            np.fill_diagonal(mat_norm, 0)
            return mat_norm
        return mat
    
    density_norm = safe_normalize(density_matrix)
    opp_norm = safe_normalize(opp_matrix)
    structure_norm = safe_normalize(structure_matrix)
    inv_dist_norm = safe_normalize(inv_dist)
    
    # Dynamic weight adaptation based on instance characteristics
    if n < 30:
        # Small instances: emphasize distance and opportunity cost
        weights = [0.15, 0.30, 0.25, 0.30]  # density, opp, structure, distance
    elif cv_dist > 0.8:
        # High variation: emphasize clustering and structure
        weights = [0.35, 0.20, 0.30, 0.15]
    elif density_cv > 0.6:
        # Clustered instances: emphasize density and opportunity
        weights = [0.40, 0.30, 0.15, 0.15]
    else:
        # Balanced weights
        weights = [0.25, 0.35, 0.20, 0.20]
    
    # Weighted combination
    heuristic = (weights[0] * density_norm + 
                 weights[1] * opp_norm + 
                 weights[2] * structure_norm + 
                 weights[3] * inv_dist_norm)
    
    # ============================================
    # 6. CONNECTIVITY-AWARE SPARSIFICATION
    # ============================================
    sparsity_mask = np.zeros((n, n), dtype=bool)
    
    # Always include MST edges
    sparsity_mask = sparsity_mask | mst_mask
    
    # Adaptive per-node edge selection
    avg_nn_dist = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        avg_nn_dist[i] = np.mean(np.partition(dists, min(3, n-1))[:min(3, n-1)])
    
    for i in range(n):
        # Select top heuristic edges
        scores = heuristic[i].copy()
        scores[i] = -np.inf
        
        # Dynamic number of edges based on local density
        n_keep = max(3, int(np.log2(n) * (1.5 if cv_dist > 0.7 else 1.0)))
        n_keep = min(n_keep, n-1)
        
        if n_keep < n-1:
            threshold = np.partition(scores, -n_keep)[-n_keep]
            top_mask = scores >= threshold
        else:
            top_mask = np.ones(n, dtype=bool)
            top_mask[i] = False
        
        # Include nearest neighbors
        nearest = np.argsort(distance_matrix[i])[1:min(3, n)]
        for nb in nearest:
            top_mask[nb] = True
        
        sparsity_mask[i] = sparsity_mask[i] | top_mask
        sparsity_mask[:, i] = sparsity_mask[:, i] | top_mask
    
    # Ensure minimum connectivity
    for i in range(n):
        connected = np.where(sparsity_mask[i])[0]
        connected = connected[connected != i]
        if len(connected) < 2:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            additional = np.argsort(dists)[:2]
            for nb in additional:
                sparsity_mask[i, nb] = True
                sparsity_mask[nb, i] = True
    
    # Apply sparsification mask
    heuristic = heuristic * sparsity_mask
    
    # Row-wise softmax for probabilistic interpretation
    for i in range(n):
        row = heuristic[i]
        mask = row > 0
        if np.sum(mask) > 1:
            # Softmax with temperature based on row variance
            row_vals = row[mask]
            temperature = max(0.5, np.std(row_vals) / (np.mean(row_vals) + 1e-8))
            exp_vals = np.exp(row_vals / temperature)
            row[mask] = exp_vals / np.sum(exp_vals)
    
    # Final symmetry and cleanup
    heuristic = (heuristic + heuristic.T) / 2
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
