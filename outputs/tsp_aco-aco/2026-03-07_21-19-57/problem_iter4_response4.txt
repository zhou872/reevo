import numpy as np
from heapq import heappop, heappush
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple, List

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge selection heuristics with improved strategies:
    1. Multi-source sampled edge betweenness using Dijkstra on k-NN graph
    2. K-order regret scoring considering multiple alternative edges
    3. Exponential factor amplification to emphasize strong signals
    4. Connectivity-preserving sparsification with degree constraints
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + 1e-10)
    
    # 1. BASE SIMILARITY AND INITIALIZATION
    with np.errstate(divide='ignore', invalid='ignore'):
        base_heuristic = 1 / (distance_matrix + 1e-10)
    np.fill_diagonal(base_heuristic, 0)
    
    # Adaptive parameters
    k_neighbors = max(3, min(12, n // 4))  # For k-NN graph
    k_regret = max(2, min(5, n // 10))     # For k-order regret
    
    # 2. K-NEAREST NEIGHBOR GRAPH CONSTRUCTION
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    nn_indices = np.zeros((n, k_neighbors), dtype=int)
    nn_distances = np.zeros((n, k_neighbors))
    
    for i in range(n):
        # Get k nearest neighbors
        indices = np.argsort(distance_matrix[i])[1:k_neighbors+1]
        distances = distance_matrix[i, indices]
        nn_mask[i, indices] = True
        nn_mask[indices, i] = True
        nn_indices[i] = indices
        nn_distances[i] = distances
    
    # 3. MULTI-SOURCE BETWEENNESS APPROXIMATION
    def fast_betweenness_sampling(dist_mat: np.ndarray, mask: np.ndarray, 
                                  n_samples: int = None) -> np.ndarray:
        """Approximate edge betweenness using multi-source Dijkstra sampling."""
        n_nodes = dist_mat.shape[0]
        if n_samples is None:
            n_samples = min(15, max(5, n_nodes // 3))
        
        betweenness = np.zeros_like(dist_mat)
        sources = np.random.choice(n_nodes, min(n_samples, n_nodes), replace=False)
        
        for s in sources:
            # Dijkstra from source s
            dist = np.full(n_nodes, np.inf)
            dist[s] = 0
            prev = [set() for _ in range(n_nodes)]
            visited = np.zeros(n_nodes, dtype=bool)
            heap = [(0, s)]
            
            # Forward pass
            while heap:
                d_u, u = heappop(heap)
                if visited[u]:
                    continue
                visited[u] = True
                
                # Only consider masked edges (k-NN graph)
                neighbors = np.where(mask[u])[0]
                for v in neighbors:
                    alt = d_u + dist_mat[u, v]
                    if alt < dist[v] - 1e-10:
                        dist[v] = alt
                        prev[v] = {u}
                        heappush(heap, (alt, v))
                    elif abs(alt - dist[v]) < 1e-10:
                        prev[v].add(u)
            
            # Backward accumulation using BFS from farthest nodes
            sigma = np.ones(n_nodes)
            delta = np.zeros(n_nodes)
            
            # Process nodes in reverse distance order
            node_order = np.argsort(dist)[::-1]
            for v in node_order:
                if v == s:
                    continue
                # Edge contribution = sigma[v] / sigma[u]
                for u in prev[v]:
                    edge_val = sigma[v] / (len(prev[v]) + 1e-10)
                    betweenness[min(u, v), max(u, v)] += edge_val
                    delta[u] += edge_val
                sigma[u] += delta[u] if u in prev[v] else 0
        
        # Symmetrize and normalize
        betweenness = betweenness + betweenness.T
        max_val = np.max(betweenness)
        if max_val > 0:
            betweenness = betweenness / max_val
        return betweenness
    
    betweenness = fast_betweenness_sampling(distance_matrix, nn_mask)
    
    # 4. K-ORDER REGRET SCORING
    def compute_k_order_regret(dist_mat: np.ndarray, k_nn: int = 3) -> np.ndarray:
        """Compute regret scores considering k alternative edges."""
        n_nodes = dist_mat.shape[0]
        regret_scores = np.zeros_like(dist_mat)
        
        # Precompute sorted distances for each node (excluding self)
        sorted_indices = np.argsort(dist_mat, axis=1)[:, 1:k_nn+2]
        sorted_dists = np.take_along_axis(dist_mat, sorted_indices, axis=1)
        
        for i in range(n_nodes):
            for j in range(i+1, n_nodes):
                if not nn_mask[i, j]:
                    continue
                    
                # Find rank of edge (i,j) in both nodes' sorted lists
                rank_i = np.where(sorted_indices[i] == j)[0]
                rank_j = np.where(sorted_indices[j] == i)[0]
                
                # Compute regret for each node based on k alternatives
                regret_i = 0
                if len(rank_i) > 0:
                    idx = rank_i[0]
                    # Weighted sum of differences with k alternatives
                    weights = np.exp(-np.arange(len(sorted_dists[i])) / 2)
                    alternatives = sorted_dists[i][:k_nn]
                    regret_i = np.sum(weights[:len(alternatives)] * 
                                     (alternatives - dist_mat[i, j])) / (dist_mat[i, j] + 1e-10)
                
                regret_j = 0
                if len(rank_j) > 0:
                    idx = rank_j[0]
                    weights = np.exp(-np.arange(len(sorted_dists[j])) / 2)
                    alternatives = sorted_dists[j][:k_nn]
                    regret_j = np.sum(weights[:len(alternatives)] * 
                                     (alternatives - dist_mat[i, j])) / (dist_mat[i, j] + 1e-10)
                
                regret_scores[i, j] = regret_scores[j, i] = max(0, (regret_i + regret_j) / 2)
        
        # Normalize
        max_val = np.max(regret_scores)
        if max_val > 0:
            regret_scores = regret_scores / max_val
        return regret_scores
    
    regret_scores = compute_k_order_regret(distance_matrix, k_regret)
    
    # 5. MINIMUM SPANNING TREE BONUS
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = (mst + mst.T) > 0
    
    # 6. DYNAMIC WEIGHTING WITH GRAPH PROPERTIES
    def compute_graph_properties(dist_mat: np.ndarray, nn_dist: np.ndarray) -> Tuple[float, float]:
        """Compute clusteriness and sparsity metrics."""
        # Clusteriness: variance ratio of nearest neighbor distances
        if nn_dist.size > 0:
            clusteriness = np.std(nn_dist) / (np.mean(nn_dist) + 1e-10)
        else:
            clusteriness = 0.5
        
        # Sparsity: ratio of MST weight to complete graph weight
        mst_weight = np.sum(mst[mst > 0])
        avg_complete_weight = np.mean(dist_mat[np.triu_indices_from(dist_mat, k=1)])
        sparsity = mst_weight / (avg_complete_weight * (n-1) + 1e-10)
        
        return clusteriness, sparsity
    
    clusteriness, sparsity = compute_graph_properties(distance_matrix, nn_distances)
    
    # Dynamic weights based on properties
    w_base = 1.0
    w_betweenness = 0.8 + 0.4 * clusteriness
    w_regret = 1.2 - 0.4 * sparsity
    w_mst = 1.5  # Constant bonus for MST edges
    
    # 7. EXPONENTIAL FACTOR COMBINATION
    heuristic_matrix = np.zeros_like(distance_matrix)
    promising_mask = nn_mask | mst_bonus
    
    for i in range(n):
        for j in range(i+1, n):
            if promising_mask[i, j]:
                # Apply exponential amplification to strong signals
                base_score = np.exp(w_base * base_heuristic[i, j])
                betweenness_score = np.exp(w_betweenness * betweenness[i, j])
                regret_score = np.exp(w_regret * regret_scores[i, j])
                mst_factor = w_mst if mst_bonus[i, j] else 1.0
                
                # Degree-based normalization (favor balanced nodes)
                degree_i = np.sum(nn_mask[i])
                degree_j = np.sum(nn_mask[j])
                degree_factor = 2.0 / (1 + np.abs(degree_i - degree_j) / max(degree_i, degree_j))
                
                # Combined score with non-linear interaction
                combined = (base_score * betweenness_score * regret_score * 
                           mst_factor * degree_factor)
                heuristic_matrix[i, j] = heuristic_matrix[j, i] = combined
    
    # 8. CONNECTIVITY-PRESERVING SPARSIFICATION
    def ensure_connectivity(heuristic_mat: np.ndarray, dist_mat: np.ndarray, 
                           min_edges_per_node: int = 2) -> np.ndarray:
        """Ensure graph remains connected with minimum edges per node."""
        n_nodes = heuristic_mat.shape[0]
        result = heuristic_mat.copy()
        
        # First, ensure each node has at least min_edges_per_node
        for i in range(n_nodes):
            connected = np.where(result[i] > 0)[0]
            if len(connected) < min_edges_per_node:
                # Find closest nodes not yet connected
                dist_row = dist_mat[i].copy()
                dist_row[i] = np.inf
                dist_row[result[i] > 0] = np.inf
                
                # Add nearest edges to reach min_edges_per_node
                needed = min_edges_per_node - len(connected)
                if needed > 0:
                    candidates = np.argsort(dist_row)[:needed]
                    for j in candidates:
                        if j != i:
                            # Assign moderate heuristic value
                            result[i, j] = result[j, i] = np.median(result[result > 0])
        
        # Second, ensure connectivity via BFS
        visited = np.zeros(n_nodes, dtype=bool)
        stack = [0]
        visited[0] = True
        
        while stack:
            node = stack.pop()
            neighbors = np.where(result[node] > 0)[0]
            for neighbor in neighbors:
                if not visited[neighbor]:
                    visited[neighbor] = True
                    stack.append(neighbor)
        
        # If disconnected, add edges between components
        if not np.all(visited):
            components = []
            unvisited = np.where(~visited)[0]
            while len(unvisited) > 0:
                start = unvisited[0]
                component = []
                stack = [start]
                visited_comp = np.zeros(n_nodes, dtype=bool)
                
                while stack:
                    node = stack.pop()
                    if not visited_comp[node]:
                        visited_comp[node] = True
                        component.append(node)
                        neighbors = np.where(result[node] > 0)[0]
                        for neighbor in neighbors:
                            if not visited_comp[neighbor]:
                                stack.append(neighbor)
                
                components.append(component)
                unvisited = np.where(~visited_comp)[0]
            
            # Connect components with nearest edges
            for comp1 in components:
                for comp2 in components:
                    if comp1 is comp2:
                        continue
                    # Find closest nodes between components
                    min_dist = np.inf
                    best_pair = (0, 0)
                    for i in comp1:
                        for j in comp2:
                            if distance_matrix[i, j] < min_dist:
                                min_dist = distance_matrix[i, j]
                                best_pair = (i, j)
                    
                    i, j = best_pair
                    result[i, j] = result[j, i] = np.median(result[result > 0])
        
        return result
    
    # Adaptive sparsification threshold
    keep_ratio = 0.25 + 0.15 * (1 - min(1, n/50))
    if np.any(heuristic_matrix > 0):
        global_threshold = np.percentile(heuristic_matrix[heuristic_matrix > 0], 
                                        (1 - keep_ratio) * 100)
        heuristic_matrix[heuristic_matrix < global_threshold] = 0
    
    # Ensure connectivity and minimum degree
    heuristic_matrix = ensure_connectivity(heuristic_matrix, distance_matrix)
    
    # 9. FINAL NORMALIZATION
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
    
    np.fill_diagonal(heuristic_matrix, 0)
    return heuristic_matrix
