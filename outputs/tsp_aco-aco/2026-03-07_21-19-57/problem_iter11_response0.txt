import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with advanced regret
    mechanisms, adaptive density clustering, and intelligent sparsification.
    
    Core improvements over v1:
    1. Regret-Based Cost Prioritization: Compute normalized regret as (second_best - best)/best
       for each node, directly targeting edges with poor alternatives
    2. Two-step Lookahead Spatial Scoring: Evaluate spatial alternatives using rank-based
       opportunity costs at both endpoints
    3. Enhanced Density Clustering: Use adaptive k-NN distances with robust scaling and
       exponential decay for intra-cluster connections
    4. Dynamic Multi-Criteria Sparsification: Combine spatial, regret, and density criteria
       with minimum connectivity guarantees
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Initialize heuristic matrix
    heuristic = np.zeros((n, n))
    np.fill_diagonal(heuristic, 0)
    
    # 1. REGRET-BASED COST PRIORITIZATION
    regret_matrix = np.ones((n, n))
    
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Find two nearest neighbors
        sorted_indices = np.argsort(distances)
        nearest_dist = distances[sorted_indices[0]]
        second_nearest_dist = distances[sorted_indices[1]] if len(sorted_indices) > 1 else nearest_dist
        
        # Compute normalized regret for this node
        if nearest_dist > 0:
            node_regret = (second_nearest_dist - nearest_dist) / nearest_dist
        else:
            node_regret = 0
        
        # Apply regret bonus to nearest edge
        j_nearest = sorted_indices[0]
        regret_matrix[i, j_nearest] = 1.0 + 0.6 * node_regret
        
        # Apply reduced bonus to other edges based on distance ratio
        for j in range(n):
            if i == j:
                continue
            if j != j_nearest and distances[j] > 0:
                dist_ratio = nearest_dist / distances[j]
                regret_matrix[i, j] = 1.0 + 0.3 * node_regret * dist_ratio
    
    # 2. TWO-STEP LOOKAHEAD SPATIAL SCORING
    spatial_matrix = np.ones((n, n))
    
    # Pre-compute alternative rankings for each node
    alt_ranks = []
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        sorted_indices = np.argsort(distances)
        # Get ranks (0-based) for all nodes
        ranks = np.zeros(n)
        for rank, idx in enumerate(sorted_indices):
            ranks[idx] = rank
        alt_ranks.append(ranks)
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            # Opportunity cost at node i: rank of alternative edges
            alt_rank_i = alt_ranks[i][j]
            opp_cost_i = alt_rank_i / (n - 1)  # Normalize by max rank
            
            # Opportunity cost at node j
            alt_rank_j = alt_ranks[j][i]
            opp_cost_j = alt_rank_j / (n - 1)
            
            # Combine opportunity costs (lower is better for edge)
            combined_opp_cost = (opp_cost_i + opp_cost_j) / 2
            
            # Apply penalty based on opportunity cost
            spatial_matrix[i, j] = 1.0 - 0.4 * combined_opp_cost
    
    # 3. ADAPTIVE LOCAL DENSITY CLUSTERING
    density_matrix = np.ones((n, n))
    
    # Compute adaptive k for each node (10% of n, min 3)
    k_base = max(3, int(0.1 * n))
    local_density = np.zeros(n)
    
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Use k nearest neighbors (adaptive)
        k = min(k_base, n - 1)
        if k > 0:
            # Get k smallest distances
            kth_idx = min(k, len(distances) - 1)
            kth_smallest = np.partition(distances, kth_idx)[kth_idx]
            nearest_mask = distances <= kth_smallest
            nearest_dists = distances[nearest_mask]
            
            if len(nearest_dists) > 0:
                # Robust scaling: use median instead of mean
                median_dist = np.median(nearest_dists)
                if median_dist > 0:
                    local_density[i] = 1.0 / median_dist
                else:
                    local_density[i] = 1.0
            else:
                local_density[i] = 1.0
    
    # Normalize densities using robust scaling (median absolute deviation)
    if np.any(local_density > 0):
        median_density = np.median(local_density[local_density > 0])
        mad_density = np.median(np.abs(local_density - median_density))
        if mad_density > 0:
            local_density = (local_density - median_density) / mad_density
            # Scale to [0, 1] range
            local_density = (local_density - np.min(local_density)) / (np.max(local_density) - np.min(local_density) + 1e-8)
    
    # Apply exponential decay bonus for similar densities
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            density_diff = abs(local_density[i] - local_density[j])
            # Exponential decay with stronger bonus for similar densities
            density_matrix[i, j] = 1.0 + 0.5 * np.exp(-3.0 * density_diff)
    
    # 4. SPATIAL PROXIMITY THRESHOLDING
    proximity_matrix = np.ones((n, n))
    
    # Compute dynamic thresholds for each node
    thresholds = np.zeros(n)
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Use average of k nearest neighbors
        k = min(k_base, n - 1)
        if k > 0:
            k_smallest = np.partition(distances, k)[:k]
            thresholds[i] = 1.5 * np.mean(k_smallest)
        else:
            thresholds[i] = np.mean(distances[distances < np.inf])
    
    # Apply threshold-based bonuses/penalties
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            dist = distance_matrix[i, j]
            threshold = min(thresholds[i], thresholds[j])
            
            if dist <= threshold:
                # Bonus inversely proportional to distance within threshold
                proximity_matrix[i, j] = 1.0 + 0.4 * (1.0 - dist / (threshold + 1e-8))
            else:
                # Penalty for exceeding threshold
                excess_ratio = dist / threshold
                proximity_matrix[i, j] = 0.6 * (1.0 / excess_ratio)
    
    # 5. MINIMUM SPANNING TREE BOOST
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    mst_boost = np.where(mst_mask, 1.4, 1.0)  # 40% boost for MST edges
    
    # COMBINE ALL COMPONENTS
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + 1e-8 * np.mean(distance_matrix))
        inv_dist[np.diag_indices(n)] = 0
    
    heuristic = inv_dist * regret_matrix * spatial_matrix * density_matrix * proximity_matrix * mst_boost
    
    # Ensure symmetry
    heuristic = np.maximum(heuristic, heuristic.T)
    
    # 6. DYNAMIC MULTI-CRITERIA SPARSIFICATION
    final_mask = np.zeros((n, n), dtype=bool)
    
    # Always include MST edges
    final_mask = final_mask | mst_mask
    
    # Compute composite scores for each edge
    composite_scores = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Combine normalized components
            reg_norm = (regret_matrix[i, j] - 1.0) / 0.6 if 0.6 > 0 else 0
            spa_norm = 1.0 - spatial_matrix[i, j]  # Higher is better
            den_norm = (density_matrix[i, j] - 1.0) / 0.5 if 0.5 > 0 else 0
            prox_norm = (proximity_matrix[i, j] - 1.0) / 0.4 if proximity_matrix[i, j] > 1.0 else (proximity_matrix[i, j] - 0.6) / 0.4
            
            composite_scores[i, j] = 0.3 * reg_norm + 0.25 * spa_norm + 0.2 * den_norm + 0.25 * prox_norm
    
    # Symmetrize composite scores
    composite_scores = (composite_scores + composite_scores.T) / 2
    
    # Select edges based on multiple criteria
    for i in range(n):
        # Criterion 1: Top edges by composite score
        comp_scores = composite_scores[i, :].copy()
        comp_scores[i] = -np.inf
        
        top_k_comp = max(2, int(0.2 * n))
        top_comp_indices = np.argpartition(comp_scores, -top_k_comp)[-top_k_comp:]
        final_mask[i, top_comp_indices] = True
        
        # Criterion 2: Edges with high regret bonus
        regret_scores = regret_matrix[i, :].copy()
        regret_scores[i] = -np.inf
        
        top_regret = max(2, int(0.15 * n))
        top_regret_indices = np.argpartition(regret_scores, -top_regret)[-top_regret:]
        final_mask[i, top_regret_indices] = True
        
        # Criterion 3: Edges within spatial threshold
        within_threshold = distance_matrix[i, :] <= thresholds[i]
        within_threshold[i] = False
        final_mask[i, within_threshold] = True
    
    # Ensure symmetry
    final_mask = final_mask | final_mask.T
    
    # Ensure minimum connectivity of 2 for each node
    for i in range(n):
        connected = np.where(final_mask[i, :])[0]
        connected = connected[connected != i]
        if len(connected) < 2:
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            # Add closest nodes not already connected
            sorted_idx = np.argsort(distances)
            added = 0
            for j in sorted_idx:
                if j != i and not final_mask[i, j]:
                    final_mask[i, j] = True
                    final_mask[j, i] = True
                    added += 1
                    if added >= 2 - len(connected):
                        break
    
    # Apply sparsification mask
    heuristic = heuristic * final_mask
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    return heuristic
