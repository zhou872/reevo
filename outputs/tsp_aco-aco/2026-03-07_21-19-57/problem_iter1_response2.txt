import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors:
    1. Distance inverse weighting (base similarity measure)
    2. Nearest neighbor clustering (sparsification)
    3. Edge betweenness approximation (centrality measure)
    4. Minimum spanning tree alignment (structural bonus)
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + 1e-10)
    
    # 1. Base similarity measure (inverse distance)
    with np.errstate(divide='ignore', invalid='ignore'):
        base_heuristic = 1 / (distance_matrix + 1e-10)
    np.fill_diagonal(base_heuristic, 0)
    
    # 2. Nearest neighbor clustering for sparsification
    k_neighbors = max(2, min(5, n // 4))  # Adaptive k
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        indices = np.argsort(distance_matrix[i])[1:k_neighbors+1]
        nn_mask[i, indices] = True
        nn_mask[indices, i] = True  # Symmetric
    
    # 3. Edge betweenness approximation
    betweenness = np.zeros_like(distance_matrix)
    
    # Compute shortest paths using Floyd-Warshall approximation
    # We use a simplified version focusing on direct and 2-hop paths
    for i in range(n):
        for j in range(i+1, n):
            # Count how many shortest paths go through this edge
            path_count = 0
            for k in range(n):
                if k == i or k == j:
                    continue
                # Check if (i,j) is part of shortest path between i/k or j/k
                if (distance_matrix[i][k] + distance_matrix[k][j] < 
                    distance_matrix[i][j] * 1.1):  # Within 10% tolerance
                    path_count += 1
            betweenness[i][j] = betweenness[j][i] = path_count / (n - 2)
    
    # 4. Minimum spanning tree bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = (mst + mst.T) > 0  # Convert to symmetric boolean
    
    # Combine all factors
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    # Only consider promising edges (nearest neighbors or MST edges)
    promising_mask = nn_mask | mst_bonus
    
    for i in range(n):
        for j in range(i+1, n):
            if promising_mask[i][j]:
                # Weighted combination of factors
                distance_factor = base_heuristic[i][j]
                betweenness_factor = 1 + betweenness[i][j]
                mst_factor = 2.0 if mst_bonus[i][j] else 1.0
                
                # Degree normalization (favor edges connecting low-degree nodes)
                degree_i = np.sum(nn_mask[i])
                degree_j = np.sum(nn_mask[j])
                degree_norm = 1 + 1 / (degree_i + degree_j + 1)
                
                heuristic = (distance_factor * betweenness_factor * 
                           mst_factor * degree_norm)
                
                heuristic_matrix[i][j] = heuristic_matrix[j][i] = heuristic
    
    # Normalize by maximum value
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix /= max_val
    
    # Apply soft thresholding (keep top 30% of values per row)
    for i in range(n):
        row = heuristic_matrix[i]
        threshold = np.percentile(row[row > 0], 70) if np.any(row > 0) else 0
        row[row < threshold] = 0
    
    return heuristic_matrix
