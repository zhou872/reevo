import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    
    Features:
    1. Inverse distance weighting (base heuristic)
    2. Sparsification by masking low-probability edges
    3. Two-step lookahead considering neighbor's neighbors
    4. Bonus for edges that connect spatially isolated nodes
    5. Local connectivity pattern analysis
    """
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance heuristic (avoid division by zero)
    np.fill_diagonal(distance_matrix, np.inf)
    base_heuristic = 1.0 / (distance_matrix + 1e-10)
    np.fill_diagonal(distance_matrix, 0)
    
    # 2. Spatial clustering analysis for isolation bonus
    # Calculate average distance of each node to all others
    node_isolation = np.mean(distance_matrix, axis=1)
    # Normalize and invert (more isolated nodes get higher bonus)
    isolation_factor = (node_isolation.max() - node_isolation) / (node_isolation.max() - node_isolation.min() + 1e-10)
    
    # 3. Two-step lookahead scoring
    # For each edge (i,j), consider best continuation from j
    lookahead_scores = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Find best third node k (not i or j) to continue from j
                mask = np.ones(n, dtype=bool)
                mask[[i, j]] = False
                if np.any(mask):
                    best_continuation = np.min(distance_matrix[j, mask])
                    lookahead_scores[i, j] = 1.0 / (distance_matrix[i, j] + best_continuation + 1e-10)
    
    # 4. Local connectivity pattern - edges that complete triangles get bonus
    triangle_bonus = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Find common neighbors between i and j
                mask_i = distance_matrix[i] < np.percentile(distance_matrix[i], 25)
                mask_j = distance_matrix[j] < np.percentile(distance_matrix[j], 25)
                common_neighbors = np.logical_and(mask_i, mask_j)
                common_neighbors[[i, j]] = False
                if np.any(common_neighbors):
                    triangle_bonus[i, j] = np.sum(common_neighbors) / n
    
    # 5. Combine all factors with weighted geometric mean
    # Weights can be tuned based on problem characteristics
    w_base = 0.4
    w_lookahead = 0.3
    w_isolation = 0.2
    w_triangle = 0.1
    
    for i in range(n):
        for j in range(n):
            if i != j:
                # Isolation bonus is symmetric - average of both nodes' isolation
                iso_bonus = (isolation_factor[i] + isolation_factor[j]) / 2
                
                # Combine factors
                combined = (
                    base_heuristic[i, j] ** w_base *
                    lookahead_scores[i, j] ** w_lookahead *
                    iso_bonus ** w_isolation *
                    (1 + triangle_bonus[i, j]) ** w_triangle
                )
                heuristics[i, j] = combined
    
    # 6. Sparsify by keeping only top connections for each node
    k = max(3, int(np.sqrt(n)))  # Adaptive sparsity level
    for i in range(n):
        row = heuristics[i]
        # Keep only top k + edges with heuristic > 75th percentile
        threshold = max(np.partition(row, -k)[-k], np.percentile(row, 75))
        mask = row < threshold
        heuristics[i, mask] = 0
    
    # 7. Normalize to make values more interpretable
    non_zero_vals = heuristics[heuristics > 0]
    if len(non_zero_vals) > 0:
        heuristics[heuristics > 0] = non_zero_vals / non_zero_vals.max()
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
