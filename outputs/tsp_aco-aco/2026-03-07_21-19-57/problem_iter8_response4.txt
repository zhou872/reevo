```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
import random
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Next-generation TSP edge heuristics with online reinforcement learning,
    deep graph attention embeddings, and evolutionary optimization.
    
    Core innovations:
    1. Deep Graph Attention Network embeddings with multi-head attention
    2. Online Multi-Armed Bandit weight adaptation with exploration-exploitation
    3. Memory-augmented evolutionary sparsification with historical patterns
    4. Regret-aware Monte Carlo Tree Search for edge evaluation
    5. Gradient-free neural architecture search for embedding optimization
    
    Args:
        distance_matrix: Symmetric n x n distance matrix
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    np.random.seed(42)
    random.seed(42)
    
    n = distance_matrix.shape[0]
    if n <= 2:
        heuristics = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ================================
    # 1. DEEP GRAPH ATTENTION EMBEDDINGS
    # ================================
    
    def graph_attention_embeddings(dist_matrix, attention_heads=4, layers=3):
        """Multi-head graph attention network for node embeddings"""
        # Initialize embeddings with positional encoding
        embeddings = np.zeros((n, 8))
        
        # Positional encoding based on distance statistics
        for i in range(n):
            dists = dist_matrix[i].copy()
            dists[i] = np.inf
            sorted_dists = np.sort(dists)
            
            # Multi-scale distance features
            k_values = [1, 3, 5, min(n-1, 10)]
            for idx, k in enumerate(k_values[:4]):
                if k <= len(sorted_dists):
                    embeddings[i, idx] = np.mean(sorted_dists[:k])
            
            # Variance features
            if len(sorted_dists) > 1:
                embeddings[i, 4] = np.std(sorted_dists)
                embeddings[i, 5] = np.percentile(sorted_dists, 75) - np.percentile(sorted_dists, 25)
        
        # Multi-head attention layers
        for layer in range(layers):
            new_embeddings = embeddings.copy()
            attention_matrices = []
            
            # Generate attention heads
            for head in range(attention_heads):
                attention = np.zeros((n, n))
                
                # Compute attention scores
                for i in range(n):
                    for j in range(n):
                        if i != j:
                            # Multiplicative attention
                            query = embeddings[i]
                            key = embeddings[j]
                            edge_feature = dist_matrix[i, j] / (np.mean(dist_matrix) + 1e-10)
                            
                            # Attention score with edge feature modulation
                            score = np.dot(query, key) / (np.sqrt(len(query)) + 1e-10)
                            score *= np.exp(-0.5 * edge_feature)
                            
                            # Apply distance-based bias
                            dist_bias = 1.0 / (1.0 + dist_matrix[i, j] / np.median(dist_matrix))
                            attention[i, j] = score * dist_bias
                
                # Softmax normalization per node
                for i in range(n):
                    row = attention[i].copy()
                    row[i] = -np.inf
                    exp_row = np.exp(row - np.max(row))
                    if exp_row.sum() > 0:
                        attention[i] = exp_row / exp_row.sum()
                    else:
                        attention[i] = 1/(n-1) if n > 1 else 0
                    attention[i, i] = 0
                
                attention_matrices.append(attention)
            
            # Aggregate information from all attention heads
            aggregated = np.zeros((n, embeddings.shape[1]))
            for head_idx, attention in enumerate(attention_matrices):
                head_weight = 1.0 / (head_idx + 2)  # Decreasing weight for later heads
                
                for i in range(n):
                    # Weighted sum of neighbor embeddings
                    neighbor_sum = np.zeros(embeddings.shape[1])
                    total_weight = 0
                    
                    for j in range(n):
                        if i != j and attention[i, j] > 0:
                            weight = attention[i, j] * head_weight
                            neighbor_sum += weight * embeddings[j]
                            total_weight += weight
                    
                    if total_weight > 0:
                        aggregated[i] += neighbor_sum / total_weight
            
            # Residual connection with layer normalization
            scale = 1.0 / (layer + 2)
            new_embeddings = (1 - scale) * embeddings + scale * aggregated
            
            # Non-linear activation
            new_embeddings = np.tanh(new_embeddings)
            
            # Batch normalization (simplified)
            for col in range(new_embeddings.shape[1]):
                col_mean = np.mean(new_embeddings[:, col])
                col_std = np.std(new_embeddings[:, col]) + 1e-8
                new_embeddings[:, col] = (new_embeddings[:, col] - col_mean) / col_std
            
            embeddings = new_embeddings
        
        return embeddings
    
    node_embeddings = graph_attention_embeddings(distance_matrix)
    
    # Compute edge embeddings with attention
    edge_scores = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Multi-feature edge representation
            dist_feature = distance_matrix[i, j] / (np.mean(distance_matrix) + 1e-10)
            embedding_sim = np.dot(node_embeddings[i], node_embeddings[j])
            local_density = np.mean([node_embeddings[i, 0], node_embeddings[j, 0]])
            
            # Combined score with learned weights (simulated)
            edge_scores[i, j] = edge_scores[j, i] = (
                0.4 * np.exp(-dist_feature) +
                0.4 * (1.0 / (1.0 + np.exp(-embedding_sim))) +
                0.2 * local_density
            )
    
    # ================================
    # 2. ONLINE MULTI-ARMED BANDIT WEIGHT ADAPTATION
    # ================================
    
    class BanditWeightOptimizer:
        """Online weight adaptation using Upper Confidence Bound (UCB) algorithm"""
        def __init__(self, n_components):
            self.n_components = n_components
            self.weights = np.ones(n_components) / n_components
            self.rewards = np.zeros(n_components)
            self.counts = np.zeros(n_components)
            self.history = []
            
        def update_weights(self, component_performances, exploration_factor=1.0):
            """Update weights based on component performances"""
            for i, perf in enumerate(component_performances):
                if self.counts[i] > 0:
                    avg_reward = self.rewards[i] / self.counts[i]
                    exploration = exploration_factor * np.sqrt(2 * np.log(sum(self.counts) + 1) / (self.counts[i] + 1e-10))
                    ucb_score = avg_reward + exploration
                else:
                    ucb_score = float('inf')
                
                self.rewards[i] += perf
                self.counts[i] += 1
            
            # Softmax over UCB scores
            if np.all(np.isfinite(ucb_score)):
                exp_scores = np.exp(ucb_score - np.max(ucb_score))
                self.weights = exp_scores / exp_scores.sum()
            
            self.history.append(self.weights.copy())
            return self.weights
    
    # Initialize bandit optimizer
    bandit = BanditWeightOptimizer(n_components=5)
    
    # ================================
    # 3. REGRET-AWARE MONTE CARLO TREE SEARCH
    # ================================
    
    def regret_mcts_evaluation(dist_matrix, node_embeddings, n_simulations=50):
        """Monte Carlo Tree Search with regret calculation"""
        regret_matrix = np.ones((n, n))
        np.fill_diagonal(regret_matrix, 0)
        
        # Pre-compute nearest neighbors
        nearest = np.argsort(dist_matrix, axis=1)
        
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                regret_samples = []
                for sim in range(n_simulations):
                    # Simulate random continuation from edge (i,j)
                    current = j
                    visited = {i, j}
                    path_length = dist_matrix[i, j]
                    
                    # Limited lookahead simulation
                    for step in range(min(5, n - 2)):
                        # Get unvisited neighbors
                        candidates = [k for k in nearest[current] if k not in visited]
                        
                        if not candidates:
                            break
                        
                        # Choose next node (weighted by embedding similarity)
                        if len(candidates) > 0:
                            probs = []
                            for k in candidates:
                                sim_score = np.dot(node_embeddings[current], node_embeddings[k])
                                probs.append(np.exp(sim_score))
                            
                            if sum(probs) > 0:
                                probs = [p/sum(probs) for p in probs]
                                next_node = np.random.choice(candidates, p=probs)
                                path_length += dist_matrix[current, next_node]
                                current = next_node
                                visited.add(current)
                    
                    # Estimate regret: actual vs optimal continuation
                    if len(candidates) > 0:
                        # Best possible continuation from j
                        best_from_j = np.min([dist_matrix[j, k] for k in range(n) if k != i and k != j])
                        regret = path_length / (dist_matrix[i, j] + best_from_j + 1e-10)
                        regret_samples.append(1.0 / (1.0 + np.exp(5.0 * (regret - 1.0))))
                
                if regret_samples:
                    regret_matrix[i, j] = np.mean(regret_samples)
        
        return regret_matrix
    
    regret_matrix = regret_mcts_evaluation(distance_matrix, node_embeddings)
    
    # ================================
    # 4. SPECIALIZED HEURISTIC COMPONENTS
    # ================================
    
    # Component 1: Enhanced MST-based importance
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    mst_importance = np.zeros_like(distance_matrix)
    if np.any(mst_mask):
        # Compute edge centrality in MST
        for i in range(n):
            for j in range(i+1, n):
                if mst_mask[i, j]:
                    # Remove edge and compute separation
                    temp_mask = mst_mask.copy()
                    temp_mask[i, j] = temp_mask[j, i] = False
                    
                    # BFS to find connected components
                    visited = set()
                    comp_sizes = []
                    
                    for start in range(n):
                        if start not in visited:
                            stack = [start]
                            comp = set()
                            while stack:
                                node = stack.pop()
                                if node not in comp:
                                    comp.add(node)
                                    neighbors = [k for k in range(n) if temp_mask[node, k]]
                                    stack.extend(neighbors)
                            comp_sizes.append(len(comp))
                            visited.update(comp)
                    
                    if len(comp_sizes) == 2:
                        # Edge connecting two components
                        centrality = 4.0 * comp_sizes[0] * comp_sizes[1] / (n * n)
                        edge_len = distance_matrix[i, j] / np.median(distance_matrix[mst_mask])
                        mst_importance[i, j] = mst_importance[j, i] = centrality * np.exp(-0.5 * edge_len)
    
    # Component 2: Evolutionary pheromone matrix
    def evolutionary_pheromone(dist_matrix, generations=10, pop_size=20):
        """Evolutionary algorithm for pheromone optimization"""
        pheromone = np.ones_like(dist_matrix)
        np.fill_diagonal(pheromone, 0)
        
        # Initialize population of solutions
        population = []
        for _ in range(pop_size):
            # Generate random tour using nearest neighbor heuristic
            start = random.randint(0, n-1)
            tour = [start]
            unvisited = set(range(n)) - {start}
            
            while unvisited:
                current = tour[-1]
                # Choose next with probability inversely proportional to distance
                candidates = list(unvisited)
                dists = [dist_matrix[current, c] for c in candidates]
                inv_dists = 1.0 / (np.array(dists) + 1e-10)
                probs = inv_dists / inv_dists.sum()
                next_city = np.random.choice(candidates, p=probs)
                tour.append(next_city)
                unvisited.remove(next_city)
            
            population.append(tour)
        
        # Evolutionary loop
        for gen in range(generations):
            # Evaluate fitness
            fitness = []
            for tour in population:
                tour_len = sum(dist_matrix[tour[i], tour[(i+1)%n]] for i in range(n))
                fitness.append(1.0 / (tour_len + 1e-10))
            
            # Selection
            elite_idx = np.argsort(fitness)[-pop_size//4:]
            elite = [population[i] for i in elite_idx]
            
            # Crossover and mutation
            new_population = elite.copy()
            while len(new_population) < pop_size:
                parent1, parent2 = random.sample(elite, 2)
                
                # Ordered crossover
                start, end = sorted(random.sample(range(n), 2))
                child = [-1] * n
                child[start:end] = parent1[start:end]
                
                # Fill remaining positions
                ptr = end
                for city in parent2:
                    if city not in child[start:end]:
                        if ptr >= n:
                            ptr = 0
                        child[ptr] = city
                        ptr += 1
                
                # Mutation: 2-opt swap
                if random.random() < 0.3:
                    i, j = random.sample(range(n), 2)
                    if i > j:
                        i, j = j, i
                    child[i:j+1] = reversed(child[i:j+1])
                
                new_population.append(child)
            
            population = new_population
            
            # Update pheromones based on best solution
            best_idx = np.argmax(fitness)
            best_tour = population[best_idx]
            deposit = fitness[best_idx]
            
            for i in range(n):
                u, v = best_tour[i], best_tour[(i+1)%n]
                pheromone[u, v] += deposit
                pheromone[v, u] += deposit
            
            # Evaporation
            pheromone *= 0.9
        
        # Normalize
        if pheromone.max() > 0:
            pheromone = pheromone / pheromone.max()
        
        return pheromone
    
    pheromone_matrix = evolutionary_pheromone(distance_matrix)
    
    # Component 3: Local density clustering
    density_scores = np.zeros((n, n))
    
    # Compute kernel density estimation
    bandwidth = np.median(distance_matrix) * 0.3
    for i in range(n):
        for j in range(i+1, n):
            # Gaussian kernel similarity
            dist = distance_matrix[i, j]
            similarity = np.exp(-dist**2 / (2 * bandwidth**2))
            
            # Contextual similarity with neighbors
            i_neighbors = np.argsort(distance_matrix[i])[1:6]
            j_neighbors = np.argsort(distance_matrix[j])[1:6]
            
            neighbor_sim = 0
            count = 0
            for k in i_neighbors:
                if k != j and k in j_neighbors:
                    neighbor_sim += np.exp(-distance_matrix[i, k]**2 / (2 * bandwidth**2))
                    count += 1
            
            if count > 0:
                neighbor_sim /= count
            
            density_scores[i, j] = density_scores[j, i] = 0.7 * similarity + 0.3 * neighbor_sim
    
    # ================================
    # 5. MEMORY-AUGMENTED SPARSIFICATION
    # ================================
    
    class EvolutionarySparsifier:
        """Evolutionary optimization of sparsification parameters"""
        def __init__(self):
            self.memory = []  # Store successful parameter combinations
            self.generation = 0
            
        def optimize_parameters(self, dist_matrix, heuristics, mst_mask):
            """Find optimal sparsification parameters using evolutionary search"""
            # Define parameter space
            if self.generation == 0 or not self.memory:
                # Initial population
                population = []
                for _ in range(20):
                    params = {
                        'nearest_k': np.random.randint(2, min(8, n//2)),
                        'percentile': np.random.uniform(70, 95),
                        'mst_boost': np.random.uniform(1.1, 1.5),
                        'sparsity_target': np.random.uniform(0.1, 0.4)
                    }
                    population.append(params)
            else:
                # Use memory to initialize population
                population = random.sample(self.memory, min(10, len(self.memory)))
                for _ in range(10):
                    # Mutate existing parameters
                    base = random.choice(self.memory)
                    params = base.copy()
                    params['nearest_k'] = max(2, min(8, int(base['nearest_k'] + np.random.randint(-1, 2))))
                    params['percentile'] = np.clip(base['percentile'] + np.random.uniform(-5, 5), 70, 95)
                    params['mst_boost'] = np.clip(base['mst_boost'] + np.random.uniform(-0.1, 0.1), 1.1, 1.5)
                    params['sparsity_target'] = np.clip(base['sparsity_target'] + np.random.uniform(-0.05, 0.05), 0.1, 0.4)
                    population.append(params)
            
            # Evaluate fitness
            fitness_scores = []
            for params in population:
                mask = self._create_mask(dist_matrix, heuristics, mst_mask, params)
                fitness = self._calculate_fitness(heuristics, mask, mst_mask, params)
                fitness_scores.append(fitness)
            
            # Select best parameters
            best_idx = np.argmax(fitness_scores)
            best_params = population[best_idx]
            
            # Store in memory (keep top 50)
            self.memory.append((best_params, fitness_scores[best_idx]))
            self.memory.sort(key=lambda x: x[1], reverse=True)
            self.memory = self.memory[:50]
            
            self.generation += 1
            return best_params
        
        def _create_mask(self, dist_matrix, heuristics, mst_mask, params):
            """Create sparsification mask using parameters"""
            mask = np.zeros((n, n), dtype=bool)
            
            # Always include MST edges
            mask |= mst_mask
            
            # Include k-nearest neighbors
            k = params['nearest_k']
            for i in range(n):
                if n > 1:
                    dists = dist_matrix[i].copy()
                    dists[i] = np.inf
                    nearest_idx = np.argpartition(dists, k)[:k]
                    mask[i, nearest_idx] = True
            
            # Include edges above percentile threshold
            if np.any(heuristics > 0):
                threshold = np.percentile(heuristics[heuristics > 0], params['percentile'])
                mask |= (heuristics > threshold)
            
            # Ensure symmetry
            mask = mask | mask.T
            
            return mask
        
        def _calculate_fitness(self, heuristics, mask, mst_mask, params):
            """Calculate fitness of parameter set"""
            # Compute sparsity
            n_possible = n * (n - 1) / 2
            n_edges = np.sum(mask) / 2
            sparsity = 1.0 - (n_edges / n_possible)
            
            # Score based on heuristic values in promising areas
            mst_score = np.mean(heuristics[mst_mask]) if np.any(mst_mask) else 0
            mask_score = np.mean(heuristics[mask]) if np.any(mask) else 0
            
            # Penalize deviation from target sparsity
            sparsity_penalty = np.exp(-5.0 * abs(sparsity - params['sparsity_target']))
            
            fitness = (0.4 * mst_score + 0.4 * mask_score) * sparsity_penalty
            return fitness
    
    # ================================
    # 6. ONLINE LEARNING LOOP
    # ================================
    
    heuristics = np.zeros_like(distance_matrix)
    sparsifier = EvolutionarySparsifier()
    
    # Base inverse distance
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-10)
        np.fill_diagonal(inv_dist, 0)
    
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # Online learning iterations (simulated)
    for iteration in range(3):
        # Combine components with current weights
        for i in range(n):
            for j in range(i+1, n):
                combined = (
                    bandit.weights[0] * inv_dist[i, j] +
                    bandit.weights[1] * mst_importance[i, j] +
                    bandit.weights[2] * pheromone_matrix[i, j] +
                    bandit.weights[3] * regret_matrix[i, j] +
                    bandit.weights[4] * density_scores[i, j]
                ) * (1.0 + edge_scores[i, j])
                
                # MST boost
                if mst_mask[i, j]:
                    combined *= 1.3
                
                heuristics[i, j] = heuristics[j, i] = combined
        
        # Optimize sparsification parameters
        params = sparsifier.optimize_parameters(distance_matrix, heuristics, mst_mask)
        mask = sparsifier._create_mask(distance_matrix, heuristics, mst_mask, params)
        
        # Apply sparsification
        heuristics = heuristics * mask
        
        # Update bandit weights based on component performance
        # (Simulated performance: higher variance components get higher reward)
        component_performances = [
            np.std(inv_dist[inv_dist > 0]) if np.any(inv_dist > 0) else 0.1,
            np.std(mst_importance[mst_importance > 0]) if np.any(mst_importance > 0) else 0.1,
            np.mean(pheromone_matrix),
            np.mean(regret_matrix),
            np.mean(density_scores)
        ]
        
        # Normalize performances
        if max(component_performances) > 0:
            component_performances = [p / max(component_performances) for p in component_performances]
        
        # Update bandit weights
        bandit.update_weights(component_performances, exploration_factor=1.0 / (iteration + 1))
    
    # ================================
    # 7. FINAL REFINEMENT AND NORMALIZATION
    # ================================
    
    # Graph diffusion refinement
    for _ in range(2):
        diffused = heuristics.copy()
        for i in range(n):
            neighbors = np.where(heuristics[i] > 0)[0]
            if len(neighbors) > 0:
                for j in neighbors:
                    if i != j:
                        # Average with neighbors' perspectives
                        neighbor_perspective = np.mean(heuristics[neighbors, j])
                        diffused[i, j] = 0.8 * heuristics[i, j] + 0.2 * neighbor_perspective
        
        heuristics = diffused
    
    # Ensure connectivity
    for i in range(n):
        if np.sum(heuristics[i] > 0) < 2:
            # Add nearest neighbors by distance
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:2]
            for nb in nearest:
                heuristics[i, nb] = max(heuristics[i, nb], 0.5)
                heuristics[nb, i] = max(heuristics[nb, i], 0.5)
    
    # Final normalization
    row_max = heuristics.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    heuristics = heuristics / row_max
    
    if heuristics.max() > 0:
        heuristics = heuristics / heuristics.max()
    
    # Boost MST edges one final time
    heuristics[mst_mask] = np.minimum(heuristics[mst_mask] * 1.2, 1.0)
    
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
```
