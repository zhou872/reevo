import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with rank-based
    regret minimization and two-step lookahead planning.
    
    Core improvements:
    1. Rank-based regret minimization: Compares edge distance to alternatives
       via rank normalization, prioritizing edges with fewer good options
    2. Two-step lookahead planning: Anticipates future path costs by considering
       the next connection after each edge
    3. Local density similarity: Rewards edges connecting nodes with similar
       local densities to promote intra-cluster connections
    4. MST structural boost: Gives priority to edges that form the minimum
       spanning tree backbone
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Initialize heuristic matrix
    heuristic = np.zeros((n, n))
    np.fill_diagonal(heuristic, 0)
    
    # 1. RANK-BASED REGRET MINIMIZATION
    # For each node, rank all edges and compute regret scores
    regret_matrix = np.ones((n, n))
    
    for i in range(n):
        # Get distances from node i to all others (excluding self)
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Sort distances and get ranks
        sorted_indices = np.argsort(distances)
        sorted_distances = distances[sorted_indices]
        
        # Normalize ranks to [0, 1] range
        ranks = np.arange(len(sorted_distances)) / len(sorted_distances)
        
        # Map ranks back to original indices
        rank_dict = dict(zip(sorted_indices, ranks))
        
        # Calculate regret for each edge from node i
        # Regret is high when edge is much better than alternatives
        for j in range(n):
            if i == j:
                continue
                
            # Find alternative edges (excluding j)
            alt_distances = distances.copy()
            alt_distances[j] = np.inf
            
            # Find best alternative distance
            best_alt = np.min(alt_distances)
            
            if best_alt > 0 and distance_matrix[i, j] > 0:
                # Regret ratio: how much worse are alternatives?
                regret_ratio = (best_alt - distance_matrix[i, j]) / distance_matrix[i, j]
                # Combine with rank information
                rank_score = 1.0 - rank_dict[j]  # Higher for better ranks
                regret_matrix[i, j] = 1.0 + np.clip(regret_ratio, 0, 2) * rank_score
    
    # 2. TWO-STEP LOOKAHEAD PLANNING
    lookahead_matrix = np.ones((n, n))
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            # Consider edge i->j and then the best next step from j
            # Find best connection from j (excluding i and j)
            remaining_nodes = [k for k in range(n) if k != i and k != j]
            if remaining_nodes:
                best_next = np.min([distance_matrix[j, k] for k in remaining_nodes])
                # Heuristic: reciprocal of total two-step cost
                total_cost = distance_matrix[i, j] + best_next
                if total_cost > 0:
                    lookahead_matrix[i, j] = 1.0 / (1.0 + total_cost / np.mean(distance_matrix))
                else:
                    lookahead_matrix[i, j] = 1.0
            else:
                lookahead_matrix[i, j] = 1.0
    
    # 3. LOCAL DENSITY SIMILARITY BONUS
    density_matrix = np.ones((n, n))
    
    # Compute local density for each node
    k = max(3, int(0.1 * n))  # Neighborhood size
    local_density = np.zeros(n)
    
    for i in range(n):
        # Get distances to k-nearest neighbors
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        nearest_dists = np.partition(distances, k)[:k+1]
        nearest_dists = nearest_dists[nearest_dists < np.inf]
        
        if len(nearest_dists) > 0:
            # Density is inverse of average distance to neighbors
            local_density[i] = 1.0 / (np.mean(nearest_dists) + 1e-8)
        else:
            local_density[i] = 1.0
    
    # Normalize densities
    if np.max(local_density) > 0:
        local_density = local_density / np.max(local_density)
    
    # Assign bonuses based on density similarity
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Similarity measure: 1 - absolute difference
            density_sim = 1.0 - abs(local_density[i] - local_density[j])
            # Bonus stronger for similar densities
            density_matrix[i, j] = 1.0 + 0.3 * density_sim
    
    # 4. MINIMUM SPANNING TREE BOOST
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    mst_boost = np.where(mst_mask, 1.5, 1.0)  # 50% boost for MST edges
    
    # COMBINE ALL COMPONENTS
    # Base component: inverse distance
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.diag_indices(n)] = 0
    
    # Combine all factors with multiplicative weighting
    heuristic = inv_dist * regret_matrix * lookahead_matrix * density_matrix * mst_boost
    
    # Ensure symmetry
    heuristic = np.maximum(heuristic, heuristic.T)
    
    # 5. INTELLIGENT SPARSIFICATION
    # Strategy: Keep promising edges based on multiple criteria
    
    # Keep MST edges (structural backbone)
    final_mask = mst_mask.copy()
    
    # For each node, keep top connections based on combined heuristic
    top_k = max(3, int(0.2 * n))  # Keep top 20% or at least 3
    for i in range(n):
        node_heuristics = heuristic[i, :].copy()
        node_heuristics[i] = -np.inf  # Exclude self
        
        # Get indices of top k edges
        top_indices = np.argpartition(node_heuristics, -top_k)[-top_k:]
        final_mask[i, top_indices] = True
        final_mask[top_indices, i] = True
    
    # Ensure minimum connectivity: each node should have at least 2 connections
    for i in range(n):
        connected = np.where(final_mask[i, :])[0]
        connected = connected[connected != i]
        if len(connected) < 2:
            # Add nearest neighbors
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            nearest = np.argsort(distances)[:2]
            final_mask[i, nearest] = True
            final_mask[nearest, i] = True
    
    # Apply sparsification mask
    heuristic = heuristic * final_mask
    
    # Normalize to [0, 1] range for stability
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    return heuristic
