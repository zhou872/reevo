import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection with adaptive sparsification and regret-based penalty.
    
    Features:
    1. Minimum spanning tree edges receive strong priority boost
    2. Harmonic mean cluster density to balance intra/inter-cluster connections
    3. Regret-based penalty for edges significantly longer than optimal alternatives
    4. Adaptive sparsification that becomes more aggressive with instance size
    
    Args:
        distance_matrix: Square symmetric matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    
    # Avoid division issues
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base inverse distance score (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / safe_dist
        inv_dist[np.isinf(inv_dist)] = 0
    
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # 2. Minimum Spanning Tree bonus (structural importance)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0  # Symmetric mask for MST edges
    
    # MST edges get strong bonus (factor 2.0)
    mst_bonus = np.where(mst_mask, 2.0, 1.0)
    
    # 3. Harmonic mean cluster density (balanced intra/inter-cluster connections)
    k_density = min(5, n-1)  # Number of nearest neighbors for density estimation
    densities = np.zeros(n)
    
    for i in range(n):
        # Average distance to k_density nearest neighbors
        nearest_dists = np.partition(safe_dist[i], k_density)[:k_density]
        densities[i] = 1.0 / (np.mean(nearest_dists) + 1e-10)
    
    if densities.max() > 0:
        densities = densities / densities.max()
    
    # Edge density score: harmonic mean of endpoint densities
    density_score = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if densities[i] > 0 and densities[j] > 0:
                # Harmonic mean emphasizes balanced connections
                h_mean = 2 * densities[i] * densities[j] / (densities[i] + densities[j] + 1e-10)
                density_score[i, j] = density_score[j, i] = h_mean
    
    # 4. Regret-based penalty (penalize edges much worse than alternatives)
    # Compute minimum distance from each node (excluding self)
    min_dists = np.min(safe_dist, axis=1)
    
    # Regret for edge (i,j): how much worse than the best alternative from each endpoint
    regret_matrix = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            regret_i = safe_dist[i, j] - min_dists[i]
            regret_j = safe_dist[j, i] - min_dists[j]
            max_regret = max(regret_i, regret_j)
            
            # Convert regret to penalty factor using exponential decay
            # Scale parameter adapts to distance distribution
            scale = np.mean(min_dists) + 1e-10
            regret_matrix[i, j] = regret_matrix[j, i] = np.exp(-max_regret / scale)
    
    # 5. Combine factors with instance-adaptive weights
    # Weights adapt based on instance size
    if n <= 20:
        # Small instances: balanced approach
        weights = [0.5, 0.3, 0.1, 0.1]  # [inv_dist, mst, density, regret]
    elif n <= 100:
        # Medium instances: more emphasis on structure
        weights = [0.4, 0.4, 0.1, 0.1]
    else:
        # Large instances: aggressive sparsification and structure
        weights = [0.3, 0.5, 0.1, 0.1]
    
    # Combined heuristic score
    heuristic = (
        weights[0] * inv_dist +
        weights[1] * inv_dist * mst_bonus +
        weights[2] * density_score +
        weights[3] * inv_dist * regret_matrix
    )
    
    # 6. Adaptive sparsification (more aggressive for larger instances)
    # Base neighborhood size adapts to instance size
    k_neighbors = max(2, min(5, n // 10))
    
    promising_mask = np.zeros((n, n), dtype=bool)
    
    # Always include MST edges
    promising_mask |= mst_mask
    
    # Include k_neighbors nearest neighbors for each node (by distance)
    for i in range(n):
        nearest_indices = np.argsort(safe_dist[i])[:k_neighbors + 1]  # +1 for self
        nearest_indices = nearest_indices[nearest_indices != i]
        promising_mask[i, nearest_indices] = True
        promising_mask[nearest_indices, i] = True
    
    # Further sparsification: keep only top edges by combined score within promising set
    if n > 50:  # Apply additional sparsification for larger instances
        edges_per_node = max(3, n // 20)
        for i in range(n):
            # Get indices of promising edges from this node
            node_edges = np.where(promising_mask[i])[0]
            if len(node_edges) > edges_per_node:
                # Keep only top edges_per_node by heuristic score
                scores = heuristic[i, node_edges]
                top_indices = np.argsort(scores)[-edges_per_node:]
                # Reset mask for this node
                promising_mask[i, :] = False
                promising_mask[i, node_edges[top_indices]] = True
                # Keep symmetry
                for j in node_edges[top_indices]:
                    promising_mask[j, i] = True
    
    # Apply final sparsification mask
    heuristic = heuristic * promising_mask
    
    # Normalize to [0, 1]
    max_val = heuristic.max()
    if max_val > 0:
        heuristic = heuristic / max_val
    
    return heuristic
