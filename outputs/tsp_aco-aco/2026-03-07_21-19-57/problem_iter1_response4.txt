import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    Returns a matrix of the same shape indicating edge promise scores.
    """
    n = distance_matrix.shape[0]
    
    # Base heuristic: inverse distance (closer nodes are more promising)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.diag_indices(n)] = 0
    
    # Factor 1: Nearest neighbor connectivity boost
    k = max(3, int(0.15 * n))  # Adaptive k-nearest neighbors
    nn_boost = np.zeros((n, n))
    for i in range(n):
        nearest_indices = np.argsort(distance_matrix[i])[1:k+1]  # Skip self
        nn_boost[i, nearest_indices] = 1.5
    
    # Factor 2: Edge betweenness approximation via MST
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    mst_bonus = np.where(mst_mask, 2.0, 1.0)
    
    # Factor 3: Local cluster density bonus
    cluster_bonus = np.ones((n, n))
    if n > 10:
        # Compute local density for each node (inverse of avg distance to k nearest)
        local_density = np.zeros(n)
        for i in range(n):
            nearest_dists = np.sort(distance_matrix[i])[1:k+1]
            local_density[i] = 1.0 / (np.mean(nearest_dists) + 1e-8)
        
        # Normalize densities
        local_density = local_density / (np.max(local_density) + 1e-8)
        
        # Bonus for edges connecting nodes with similar local densities
        for i in range(n):
            for j in range(i+1, n):
                density_sim = 1.0 - abs(local_density[i] - local_density[j])
                cluster_bonus[i, j] = cluster_bonus[j, i] = 1.0 + 0.5 * density_sim
    
    # Combine all factors
    combined = inv_dist * nn_boost * mst_bonus * cluster_bonus
    
    # Symmetrize the matrix
    combined = np.maximum(combined, combined.T)
    
    # Sparsification: keep only promising edges
    # Strategy: Keep top 30% of edges per node and edges in MST
    threshold_per_node = np.percentile(combined, 70, axis=1, keepdims=True)
    mask_top_edges = combined >= threshold_per_node
    
    # Final mask: keep top edges OR MST edges
    final_mask = mask_top_edges | mst_mask
    sparsified = combined * final_mask
    
    # Normalize to [0, 1] range for stability
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
