import numpy as np
from scipy import spatial

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection.
    
    Combines multiple factors to score each edge:
    1. Reciprocal distance (closer edges are better)
    2. Connectivity (edges connecting well-connected nodes are better)
    3. Sparsification (unpromising edges are set to zero)
    4. Clustering awareness (edges between clusters get bonus)
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    
    # Avoid division by zero and self-loops
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base reciprocal score (like v1)
    recip_score = 1 / safe_dist
    
    # 2. Node connectivity score
    # Nodes with many short connections are more "central"
    node_connectivity = np.zeros(n)
    for i in range(n):
        # Average of reciprocal distances from node i
        node_connectivity[i] = np.mean(1 / safe_dist[i, :])
    
    # Normalize connectivity scores
    if np.max(node_connectivity) > 0:
        node_connectivity = node_connectivity / np.max(node_connectivity)
    
    # 3. Combined edge score
    edge_score = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Base reciprocal score
            base = recip_score[i, j]
            
            # Connectivity bonus: edges between well-connected nodes
            conn_bonus = (node_connectivity[i] + node_connectivity[j]) / 2
            
            # Distance clustering: check if edge connects different clusters
            # by comparing nearest neighbors
            k = min(5, n//2)  # Number of nearest neighbors to consider
            nn_i = np.argsort(safe_dist[i, :])[:k]
            nn_j = np.argsort(safe_dist[j, :])[:k]
            
            # Jaccard similarity of nearest neighbors
            intersection = len(set(nn_i) & set(nn_j))
            union = len(set(nn_i) | set(nn_j))
            jaccard = intersection / union if union > 0 else 0
            
            # Clustering penalty: edges within same cluster get penalty,
            # edges between clusters get bonus
            cluster_penalty = 1.5 - jaccard  # 1.0 when jaccard=0.5
            
            # Combine factors
            edge_score[i, j] = edge_score[j, i] = base * conn_bonus * cluster_penalty
    
    # 4. Sparsification: keep only promising edges
    # Keep edges that are in top k nearest neighbors for at least one endpoint
    k_sparse = min(8, n//3)  # Sparsification parameter
    
    # Create sparsification mask
    mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get indices of k_sparse nearest neighbors (excluding self)
        nearest = np.argsort(safe_dist[i, :])[1:k_sparse+1]
        mask[i, nearest] = True
    
    # Symmetrize the mask (if i is in j's nearest neighbors or vice versa)
    mask = mask | mask.T
    
    # Apply sparsification
    heuristic_matrix = edge_score * mask
    
    # 5. Final normalization (optional, keeps values in reasonable range)
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
    
    return heuristic_matrix
