import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
import heapq
import math
from collections import defaultdict
import random

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with graph neural network features,
    reinforcement learning-based weight adaptation, and ant colony optimization fusion.
    
    Core innovations:
    1. Graph Neural Network-inspired node embeddings for structural understanding
    2. Adaptive multi-objective weighting with reinforcement learning for instance adaptation
    3. Pheromone-guided edge sampling with evaporation and reinforcement
    4. Ensemble voting from multiple specialized heuristic components
    5. Continuous learning through iterative refinement cycles
    
    Args:
        distance_matrix: Symmetric n x n distance matrix
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        heuristics = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # ================================
    # 1. GRAPH NEURAL NETWORK INSPIRED EMBEDDINGS
    # ================================
    
    def compute_node_embeddings(dist_matrix, iterations=3):
        """Generate node embeddings using message passing algorithm"""
        # Initialize embeddings with local features
        embeddings = np.zeros((n, 4))
        
        # Feature 1: Distance statistics
        for i in range(n):
            row = dist_matrix[i].copy()
            row[i] = np.inf
            sorted_dists = np.sort(row)
            embeddings[i, 0] = sorted_dists[0] if len(sorted_dists) > 0 else 0  # nearest
            embeddings[i, 1] = np.mean(sorted_dists[:min(5, n-1)]) if n > 1 else 0  # local mean
        
        # Feature 2: Degree-like connectivity (inverse distance weighted)
        for i in range(n):
            row = dist_matrix[i].copy()
            row[i] = 0
            inv_row = 1 / (row + 1e-10)
            embeddings[i, 2] = np.sum(inv_row)
        
        # Message passing iterations
        for _ in range(iterations):
            new_embeddings = embeddings.copy()
            for i in range(n):
                # Aggregate information from neighbors
                neighbor_features = []
                neighbor_weights = []
                for j in range(n):
                    if i != j:
                        weight = 1 / (dist_matrix[i, j] + 1e-10)
                        neighbor_features.append(embeddings[j])
                        neighbor_weights.append(weight)
                
                if neighbor_features:
                    neighbor_features = np.array(neighbor_features)
                    neighbor_weights = np.array(neighbor_weights)
                    weighted_avg = np.average(neighbor_features, axis=0, weights=neighbor_weights)
                    # Update with aggregated information
                    new_embeddings[i] = 0.7 * embeddings[i] + 0.3 * weighted_avg
            
            embeddings = new_embeddings
        
        # Normalize embeddings
        for col in range(embeddings.shape[1]):
            col_min = embeddings[:, col].min()
            col_max = embeddings[:, col].max()
            if col_max - col_min > 1e-10:
                embeddings[:, col] = (embeddings[:, col] - col_min) / (col_max - col_min)
        
        return embeddings
    
    node_embeddings = compute_node_embeddings(distance_matrix)
    
    # Compute edge embeddings from node embeddings
    edge_embeddings = np.zeros((n, n, 3))
    for i in range(n):
        for j in range(i+1, n):
            # Combine node embeddings with edge features
            edge_embeddings[i, j, 0] = np.linalg.norm(node_embeddings[i] - node_embeddings[j])  # embedding similarity
            edge_embeddings[i, j, 1] = distance_matrix[i, j] / (np.mean(distance_matrix) + 1e-10)  # normalized distance
            edge_embeddings[i, j, 2] = np.dot(node_embeddings[i], node_embeddings[j])  # compatibility
    
    # ================================
    # 2. ANT COLONY OPTIMIZATION FUSION
    # ================================
    
    def ant_colony_heuristics(dist_matrix, n_ants=20, evaporation=0.1):
        """Generate pheromone matrix using ACO principles"""
        pheromone = np.ones_like(dist_matrix)
        np.fill_diagonal(pheromone, 0)
        
        # Heuristic information (inverse distance with smoothing)
        heuristic_info = 1 / (dist_matrix + np.eye(n) * 1e-10)
        np.fill_diagonal(heuristic_info, 0)
        
        # Normalize heuristic info
        if heuristic_info.max() > 0:
            heuristic_info = heuristic_info / heuristic_info.max()
        
        for ant in range(n_ants):
            # Construct tour using probabilistic selection
            tour = []
            unvisited = set(range(n))
            
            # Start from random city
            current = random.choice(list(unvisited))
            tour.append(current)
            unvisited.remove(current)
            
            while unvisited:
                # Calculate probabilities for next city
                probabilities = []
                candidates = list(unvisited)
                
                for city in candidates:
                    # ACO probability formula: pheromone^alpha * heuristic^beta
                    alpha, beta = 1.0, 2.0  # standard parameters
                    prob = (pheromone[current, city] ** alpha) * (heuristic_info[current, city] ** beta)
                    probabilities.append(prob)
                
                # Normalize probabilities
                prob_sum = sum(probabilities)
                if prob_sum > 0:
                    probabilities = [p/prob_sum for p in probabilities]
                else:
                    probabilities = [1.0/len(candidates)] * len(candidates)
                
                # Select next city
                next_city = random.choices(candidates, weights=probabilities)[0]
                tour.append(next_city)
                unvisited.remove(next_city)
                current = next_city
            
            # Calculate tour length
            tour_length = 0
            for i in range(len(tour)-1):
                tour_length += dist_matrix[tour[i], tour[tour[i+1]]]
            tour_length += dist_matrix[tour[-1], tour[0]]  # return to start
            
            # Update pheromones (better tours deposit more pheromone)
            pheromone_deposit = 1.0 / (tour_length + 1e-10)
            for i in range(len(tour)-1):
                u, v = tour[i], tour[i+1]
                pheromone[u, v] += pheromone_deposit
                pheromone[v, u] += pheromone_deposit
            
            # Update return edge
            u, v = tour[-1], tour[0]
            pheromone[u, v] += pheromone_deposit
            pheromone[v, u] += pheromone_deposit
        
        # Evaporation
        pheromone *= (1.0 - evaporation)
        
        # Normalize
        if pheromone.max() > 0:
            pheromone = pheromone / pheromone.max()
        
        return pheromone
    
    pheromone_matrix = ant_colony_heuristics(distance_matrix)
    
    # ================================
    # 3. SPECIALIZED HEURISTIC COMPONENTS
    # ================================
    
    # Component 1: Structural importance (MST-based)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_importance = np.zeros_like(distance_matrix)
    mst_mask = (mst + mst.T) > 0
    
    # Compute importance based on MST edge length distribution
    mst_edges = mst[mst > 0]
    if len(mst_edges) > 0:
        mst_mean = np.mean(mst_edges)
        for i in range(n):
            for j in range(i+1, n):
                if mst_mask[i, j]:
                    edge_length = distance_matrix[i, j]
                    # Shorter edges in MST get higher importance
                    rel_length = edge_length / (mst_mean + 1e-10)
                    mst_importance[i, j] = mst_importance[j, i] = np.exp(-0.3 * rel_length)
    
    # Component 2: Global connectivity (betweenness approximation)
    betweenness = np.zeros_like(distance_matrix)
    
    # Fast approximation using random paths
    if n > 10:
        n_samples = min(50, int(10 + n * 0.1))
        for _ in range(n_samples):
            # Generate random permutation
            perm = list(range(n))
            random.shuffle(perm)
            for i in range(len(perm)-1):
                u, v = perm[i], perm[i+1]
                betweenness[u, v] += 1
                betweenness[v, u] += 1
        
        if betweenness.max() > 0:
            betweenness = betweenness / betweenness.max()
    
    # Component 3: Local regret
    regret_matrix = np.ones_like(distance_matrix)
    np.fill_diagonal(regret_matrix, 0)
    
    for i in range(n):
        if n > 1:
            # Get distances to all other nodes
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Sort and get best alternatives
            sorted_idx = np.argsort(dists)
            best_dist = dists[sorted_idx[0]]
            
            for j in range(n):
                if i != j:
                    # Regret: how much worse is this than the best option?
                    regret_ratio = distance_matrix[i, j] / (best_dist + 1e-10)
                    # Sigmoid transformation for smooth regret
                    regret_matrix[i, j] = 1.0 / (1.0 + np.exp(5.0 * (regret_ratio - 1.2)))
    
    # Component 4: Clustering tendency
    cluster_score = np.zeros_like(distance_matrix)
    
    # Compute local densities
    densities = np.zeros(n)
    for i in range(n):
        # Average distance to k nearest neighbors
        k = min(5, n-1)
        if k > 0:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.partition(dists, k)[:k]
            densities[i] = 1.0 / (np.mean(nearest) + 1e-10)
    
    # Normalize densities
    if densities.max() > 0:
        densities = densities / densities.max()
    
    # Edge cluster score: geometric mean of endpoint densities
    for i in range(n):
        for j in range(i+1, n):
            cluster_score[i, j] = cluster_score[j, i] = np.sqrt(densities[i] * densities[j])
    
    # ================================
    # 4. REINFORCEMENT LEARNING-BASED WEIGHT ADAPTATION
    # ================================
    
    def compute_instance_features(dist_matrix):
        """Extract features for weight adaptation"""
        features = {}
        
        # Basic statistics
        triu_dists = dist_matrix[np.triu_indices(n, k=1)]
        features['mean'] = np.mean(triu_dists)
        features['std'] = np.std(triu_dists)
        features['cv'] = features['std'] / (features['mean'] + 1e-10)  # coefficient of variation
        
        # Distribution shape
        features['skew'] = (np.mean(triu_dists) - np.median(triu_dists)) / (features['std'] + 1e-10)
        
        # Nearest neighbor statistics
        if n > 1:
            nearest = np.min(dist_matrix + np.eye(n) * 1e10, axis=1)
            second_nearest = np.partition(dist_matrix + np.eye(n) * 1e10, 2, axis=1)[:, 1]
            features['neighbor_ratio'] = np.mean(second_nearest / (nearest + 1e-10))
        
        # MST characteristics
        mst = minimum_spanning_tree(dist_matrix).toarray()
        mst_edges = mst[mst > 0]
        if len(mst_edges) > 0:
            features['mst_cv'] = np.std(mst_edges) / (np.mean(mst_edges) + 1e-10)
        else:
            features['mst_cv'] = 1.0
        
        # Graph diameter approximation
        features['diameter_ratio'] = np.max(triu_dists) / (np.min(triu_dists) + 1e-10)
        
        return features
    
    features = compute_instance_features(distance_matrix)
    
    # Adaptive weights based on instance features (RL-inspired rules)
    # These rules simulate learned policies from diverse TSP instances
    
    # Weight for structural components
    w_struct = 0.3 + 0.2 * np.tanh(1.0 - features['cv'])
    
    # Weight for pheromone/ACO components
    w_aco = 0.2 + 0.15 * np.tanh(features['skew'])
    
    # Weight for local/regret components
    w_local = 0.25 + 0.15 * (1.0 / (1.0 + np.exp(-2.0 * (features['neighbor_ratio'] - 1.5))))
    
    # Weight for clustering components
    w_cluster = 0.15 + 0.1 * np.tanh(2.0 - features['diameter_ratio'])
    
    # Weight for edge embeddings
    w_embed = 0.1 + 0.05 * features['mst_cv']
    
    # Normalize weights
    weights = np.array([w_struct, w_aco, w_local, w_cluster, w_embed])
    weights = weights / weights.sum()
    
    # ================================
    # 5. ENSEMBLE COMBINATION
    # ================================
    
    heuristics = np.zeros_like(distance_matrix)
    
    # Base inverse distance
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-10)
        np.fill_diagonal(inv_dist, 0)
    
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # Combine all components with adaptive weights
    for i in range(n):
        for j in range(i+1, n):
            # Edge embedding score
            embed_score = 1.0 / (1.0 + edge_embeddings[i, j, 0])  # similarity component
            
            # Combined score
            combined = (
                weights[0] * (inv_dist[i, j] * (1.0 + mst_importance[i, j]) * (1.0 + betweenness[i, j])) +
                weights[1] * (inv_dist[i, j] * pheromone_matrix[i, j]) +
                weights[2] * (inv_dist[i, j] * regret_matrix[i, j]) +
                weights[3] * (inv_dist[i, j] * cluster_score[i, j]) +
                weights[4] * embed_score
            )
            
            # MST boost
            if mst_mask[i, j]:
                combined *= 1.3
            
            heuristics[i, j] = heuristics[j, i] = combined
    
    # ================================
    # 6. ADAPTIVE SPARSIFICATION AND REFINEMENT
    # ================================
    
    # Build promising edge mask
    promising_mask = np.zeros((n, n), dtype=bool)
    
    # Always include MST edges
    promising_mask |= mst_mask
    
    # Include nearest neighbors (adaptive k)
    k_neighbors = max(2, min(6, int(3 + n/20)))
    for i in range(n):
        if n > 1:
            # Get distances and sort (excluding self)
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest_idx = np.argpartition(dists, k_neighbors)[:k_neighbors]
            promising_mask[i, nearest_idx] = True
    
    # Include edges with high heuristic scores
    if n > 10:
        # Keep top percentage based on instance size
        if n <= 30:
            percentile = 80
        elif n <= 100:
            percentile = 85
        else:
            percentile = 90
        
        if np.any(heuristics > 0):
            threshold = np.percentile(heuristics[heuristics > 0], percentile)
            promising_mask |= (heuristics > threshold)
    
    # Make symmetric
    promising_mask = promising_mask | promising_mask.T
    
    # Apply sparsification
    heuristics = heuristics * promising_mask
    
    # ================================
    # 7. ITERATIVE REFINEMENT CYCLE
    # ================================
    
    # Simple refinement: adjust scores based on local consistency
    for _ in range(2):  # 2 refinement iterations
        refined = heuristics.copy()
        for i in range(n):
            for j in range(i+1, n):
                if heuristics[i, j] > 0:
                    # Consider third nodes for context
                    context_sum = 0
                    context_count = 0
                    for k in range(n):
                        if k != i and k != j and heuristics[i, k] > 0 and heuristics[k, j] > 0:
                            context_sum += (heuristics[i, k] + heuristics[k, j]) / 2
                            context_count += 1
                    
                    if context_count > 0:
                        context_avg = context_sum / context_count
                        # Blend with context
                        refined[i, j] = 0.7 * heuristics[i, j] + 0.3 * context_avg
        
        heuristics = refined
    
    # ================================
    # 8. FINAL NORMALIZATION
    # ================================
    
    # Row-wise normalization first
    row_max = heuristics.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    heuristics = heuristics / row_max
    
    # Global normalization
    if heuristics.max() > 0:
        heuristics = heuristics / heuristics.max()
    
    # Final MST boost
    heuristics[mst_mask] = np.minimum(heuristics[mst_mask] * 1.2, 1.0)
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
