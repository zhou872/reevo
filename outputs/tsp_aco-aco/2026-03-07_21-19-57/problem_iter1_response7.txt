import numpy as np
from scipy.spatial import cKDTree
from scipy.cluster.hierarchy import linkage, fcluster
from scipy.spatial.distance import squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Reciprocal distance weighting (baseline)
    2. Node density bonus (prioritize edges in sparse regions)
    3. Cluster-based penalty (penalize edges crossing distant clusters)
    4. Sparsification (zero out low-promise edges)
    """
    n = distance_matrix.shape[0]
    np.fill_diagonal(distance_matrix, np.inf)  # Ignore self-loops
    
    # 1. Reciprocal distance (baseline heuristic)
    reciprocal = 1 / distance_matrix
    reciprocal[np.isinf(reciprocal)] = 0
    
    # 2. Node density bonus
    density_bonus = np.zeros_like(distance_matrix)
    kdtree = cKDTree(np.random.rand(n, 2))  # Using random coordinates for demonstration
    # In practice, use actual coordinates or multidimensional scaling of distance matrix
    # Here we use distance matrix directly to estimate density
    avg_dist = np.mean(distance_matrix, axis=1)
    density = 1 / (avg_dist + 1e-10)
    density_norm = density / np.max(density)
    
    for i in range(n):
        for j in range(i+1, n):
            # Bonus for connecting sparse nodes
            if density_norm[i] < 0.5 and density_norm[j] < 0.5:
                density_bonus[i, j] = density_bonus[j, i] = 2.0
            elif density_norm[i] < 0.7 or density_norm[j] < 0.7:
                density_bonus[i, j] = density_bonus[j, i] = 1.5
            else:
                density_bonus[i, j] = density_bonus[j, i] = 1.0
    
    # 3. Cluster-based penalty using hierarchical clustering
    cluster_penalty = np.ones_like(distance_matrix)
    if n > 10:  # Only cluster if enough nodes
        condensed_dist = squareform(distance_matrix)
        Z = linkage(condensed_dist, method='average')
        clusters = fcluster(Z, t=0.3 * n, criterion='maxclust')
        
        for i in range(n):
            for j in range(i+1, n):
                if clusters[i] != clusters[j]:
                    # Penalize edges between different clusters
                    penalty = 1.0 / (1.0 + np.abs(distance_matrix[i, j] / np.mean(distance_matrix)))
                    cluster_penalty[i, j] = cluster_penalty[j, i] = penalty
    
    # 4. Combine factors
    heuristics_matrix = reciprocal * density_bonus * cluster_penalty
    
    # 5. Sparsification - zero out low-promise edges
    # Keep only top 30% of edges for each node
    threshold = np.percentile(heuristics_matrix[heuristics_matrix > 0], 70)
    heuristics_matrix[heuristics_matrix < threshold] = 0
    
    # Ensure symmetry
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    np.fill_diagonal(heuristics_matrix, 0)  # Zero diagonal
    return heuristics_matrix
