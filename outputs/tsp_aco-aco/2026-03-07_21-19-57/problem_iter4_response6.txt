import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with adaptive regret
    thresholds and instance-aware dynamic weighting.
    
    Core innovations:
    1. Adaptive regret thresholds based on global tour length estimation
    2. Instance-aware dynamic weighting using reinforcement learning principles
    3. Two-step lookahead spatial averaging for future connectivity assessment
    4. Cluster density similarity for intra-region connections
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Base reciprocal distance with safe handling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.diag_indices(n)] = 0
    
    # Normalize base reciprocal distances
    max_inv = np.max(inv_dist)
    if max_inv > 0:
        inv_dist = inv_dist / max_inv
    
    # 1. ADAPTIVE REGRET THRESHOLDING with global tour estimation
    def estimate_global_tour() -> float:
        """Estimate global tour length using multiple heuristics"""
        # Method 1: MST-based lower bound (2 * MST)
        mst = minimum_spanning_tree(distance_matrix)
        mst_length = mst.sum()
        
        # Method 2: Nearest neighbor heuristic
        nn_tour = 0
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nn_tour += np.min(dists)
        
        # Method 3: Average of distances (scaled by n)
        avg_dist = np.mean(distance_matrix[np.triu_indices(n, 1)])
        avg_estimate = avg_dist * n
        
        # Combine estimates with weights based on instance size
        if n < 50:
            # For smaller instances, trust MST more
            return 0.5 * (2 * mst_length) + 0.3 * nn_tour + 0.2 * avg_estimate
        else:
            # For larger instances, balance all methods
            return 0.4 * (2 * mst_length) + 0.4 * nn_tour + 0.2 * avg_estimate
    
    estimated_tour = estimate_global_tour()
    avg_edge_estimate = estimated_tour / n if n > 0 else 0
    
    # Compute adaptive regret with threshold based on global estimate
    regret_matrix = np.ones((n, n))
    regret_threshold = avg_edge_estimate * 0.3  # 30% of average edge as threshold
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Get three best alternatives
        sorted_indices = np.argsort(distances)
        best_idx = sorted_indices[0]
        second_best_idx = sorted_indices[1]
        third_best_idx = sorted_indices[2] if n > 2 else second_best_idx
        
        best_dist = distances[best_idx]
        second_best_dist = distances[second_best_idx]
        
        # Calculate regret for the best edge
        if best_dist > 0:
            # Adaptive regret: penalize more when alternatives are significantly worse
            regret = (second_best_dist - best_dist) / (best_dist + 1e-8)
            
            # Apply threshold: only significant regrets are amplified
            if regret > regret_threshold / (best_dist + 1e-8):
                regret_boost = 1.0 + min(2.0, regret * 0.5)
            else:
                regret_boost = 1.0 + regret * 0.2
            
            regret_matrix[i, best_idx] *= regret_boost
            regret_matrix[best_idx, i] *= regret_boost
        
        # Also consider second-best with lower priority
        if second_best_dist > 0 and n > 2:
            third_best_dist = distances[third_best_idx]
            regret_second = (third_best_dist - second_best_dist) / (second_best_dist + 1e-8)
            if regret_second > regret_threshold / (second_best_dist + 1e-8):
                regret_matrix[i, second_best_idx] *= 1.0 + min(1.5, regret_second * 0.3)
                regret_matrix[second_best_idx, i] *= 1.0 + min(1.5, regret_second * 0.3)
    
    # Normalize regret component
    max_regret = np.max(regret_matrix)
    if max_regret > 0:
        regret_matrix = regret_matrix / max_regret
    
    # 2. INSTANCE-AWARE DYNAMIC WEIGHTING using reinforcement learning principles
    def analyze_instance_features() -> Tuple[float, float, float]:
        """Extract features from instance to determine heuristic weights"""
        # Feature 1: Distance distribution skewness
        upper_tri = distance_matrix[np.triu_indices(n, 1)]
        mean_dist = np.mean(upper_tri)
        std_dist = np.std(upper_tri)
        cv_dist = std_dist / (mean_dist + 1e-8)  # Coefficient of variation
        
        # Feature 2: Cluster tendency (based on MST analysis)
        mst = minimum_spanning_tree(distance_matrix)
        mst_edges = mst.toarray()
        mst_weights = mst_edges[mst_edges > 0]
        mst_cv = np.std(mst_weights) / (np.mean(mst_weights) + 1e-8)
        
        # Feature 3: Dimensionality (approximated by distance distribution)
        # Calculate nearest neighbor statistics
        nearest_dists = np.zeros(n)
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest_dists[i] = np.min(dists)
        
        nn_ratio = np.mean(nearest_dists) / mean_dist
        
        return cv_dist, mst_cv, nn_ratio
    
    cv_dist, mst_cv, nn_ratio = analyze_instance_features()
    
    # Dynamic weights based on instance features (reinforcement learning inspired)
    # These weights evolve based on what works for different instance types
    
    # Regret weight: higher for instances with high variance (more opportunity cost)
    w_regret = 0.3 + 0.2 * min(1.0, cv_dist)
    
    # Lookahead weight: higher for clustered instances
    w_lookahead = 0.25 + 0.15 * min(1.0, mst_cv)
    
    # MST weight: base importance plus adjustment for uniformity
    w_mst = 0.25 - 0.1 * min(1.0, cv_dist)
    
    # Cluster weight: higher for clearly clustered instances
    w_cluster = 0.2 + 0.1 * min(1.0, mst_cv * 2)
    
    # Normalize weights to sum to 1
    total = w_regret + w_lookahead + w_mst + w_cluster
    w_regret, w_lookahead, w_mst, w_cluster = [w/total for w in [w_regret, w_lookahead, w_mst, w_cluster]]
    
    # 3. TWO-STEP LOOKAHEAD SPATIAL AVERAGING
    lookahead_matrix = np.ones((n, n))
    
    # Adaptive neighborhood size based on instance size and density
    k_neighbors = max(3, min(10, int(0.1 * n) + 2))
    
    # Precompute k-nearest neighbors for efficiency
    nearest_masks = np.zeros((n, n), dtype=bool)
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        nearest_indices = np.argsort(distances)[:k_neighbors]
        nearest_masks[i, nearest_indices] = True
    
    for i in range(n):
        for j in range(i + 1, n):
            # Calculate average distance from i to its other k-nearest neighbors (excluding j)
            i_neighbors = np.where(nearest_masks[i, :])[0]
            i_neighbors = i_neighbors[i_neighbors != j]
            if len(i_neighbors) > 0:
                avg_i = np.mean(distance_matrix[i, i_neighbors])
            else:
                avg_i = distance_matrix[i, j]  # fallback
            
            # Calculate average distance from j to its other k-nearest neighbors (excluding i)
            j_neighbors = np.where(nearest_masks[j, :])[0]
            j_neighbors = j_neighbors[j_neighbors != i]
            if len(j_neighbors) > 0:
                avg_j = np.mean(distance_matrix[j, j_neighbors])
            else:
                avg_j = distance_matrix[i, j]  # fallback
            
            # Two-step lookahead: consider both immediate and future connectivity
            lookahead_score = 1.0 / (1.0 + (distance_matrix[i, j] + 0.3 * (avg_i + avg_j)) / (avg_edge_estimate + 1e-8))
            lookahead_matrix[i, j] = lookahead_matrix[j, i] = lookahead_score
    
    # Normalize lookahead component
    max_lookahead = np.max(lookahead_matrix)
    if max_lookahead > 0:
        lookahead_matrix = lookahead_matrix / max_lookahead
    
    # 4. MST INCLUSION with adaptive importance
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # MST bonus scaled by instance features
    mst_bonus_base = 1.5 + 0.5 * min(1.0, cv_dist)  # Higher bonus for more variable instances
    mst_bonus = np.where(mst_mask, mst_bonus_base, 1.0)
    
    # 5. CLUSTER DENSITY SIMILARITY with adaptive parameters
    cluster_bonus = np.ones((n, n))
    
    if n > 5:
        # Adaptive neighborhood for density calculation
        k_density = max(3, min(8, int(0.08 * n) + 2))
        
        # Compute local density for each node
        local_density = np.zeros(n)
        for i in range(n):
            nearest_dists = np.sort(distance_matrix[i])[1:k_density + 1]
            local_density[i] = 1.0 / (np.mean(nearest_dists) + 1e-8)
        
        # Normalize densities
        max_density = np.max(local_density)
        if max_density > 0:
            local_density = local_density / max_density
        
        # Adaptive similarity threshold based on instance characteristics
        similarity_threshold = 0.3 + 0.2 * min(1.0, mst_cv)
        
        # Bonus for edges connecting nodes with similar local densities
        for i in range(n):
            for j in range(i + 1, n):
                density_sim = 1.0 - abs(local_density[i] - local_density[j])
                if density_sim > similarity_threshold:
                    # Stronger bonus for very similar densities in clustered instances
                    bonus_strength = 1.0 + 0.8 * density_sim * min(1.0, mst_cv * 2)
                else:
                    bonus_strength = 1.0 + 0.3 * density_sim
                cluster_bonus[i, j] = cluster_bonus[j, i] = bonus_strength
    
    # COMBINE ALL FACTORS with instance-aware dynamic weighting
    combined = (inv_dist * 
                (w_regret * regret_matrix + 
                 w_lookahead * lookahead_matrix + 
                 w_mst * mst_bonus + 
                 w_cluster * cluster_bonus))
    
    # Symmetrize
    combined = (combined + combined.T) / 2
    
    # ADAPTIVE SPARSIFICATION based on instance characteristics
    sparsified = combined.copy()
    
    # Strategy 1: Keep MST edges with full strength
    sparsified[~mst_mask] *= 0.6  # Reduce non-MST edges more aggressively for larger n
    
    # Strategy 2: Adaptive threshold per node based on instance size
    for i in range(n):
        node_scores = sparsified[i].copy()
        node_scores[i] = -np.inf
        
        # More aggressive sparsification for larger instances
        if n > 50:
            percentile = 75  # Keep top 25%
        elif n > 20:
            percentile = 70  # Keep top 30%
        else:
            percentile = 60  # Keep top 40% for small instances
        
        threshold = np.percentile(node_scores[node_scores > 0], percentile)
        sparsified[i][node_scores < threshold] = 0
    
    # Strategy 3: Ensure minimum connectivity
    min_connections = max(2, int(0.05 * n))  # Adaptive minimum connections
    for i in range(n):
        connections = np.sum(sparsified[i] > 0)
        if connections < min_connections:
            # Add nearest neighbors by original distance
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            additional_needed = min_connections - connections
            nearest = np.argsort(dists)[:additional_needed]
            sparsified[i, nearest] = combined[i, nearest]
            sparsified[nearest, i] = combined[nearest, i]
    
    # FINAL NORMALIZATION
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    # Zero out diagonal
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
