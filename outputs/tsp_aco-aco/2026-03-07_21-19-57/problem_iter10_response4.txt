import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristic with dynamic weight adjustment, spectral clustering for
    improved density estimation, and extended multi-step regret lookahead.
    
    Core innovations:
    1. Dynamic weight adjustment based on progressive sampling statistics
    2. Spectral clustering for robust density estimation and structural analysis
    3. Extended multi-step regret evaluation for complex opportunity cost assessment
    4. Geometric mean fusion with entropy-based normalization
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # =====================================================================
    # 1. PROGRESSIVE SAMPLING STATISTICS (Dynamic adaptation)
    # =====================================================================
    
    # Global statistics for initial calibration
    non_diag_mask = ~np.eye(n, dtype=bool)
    global_dists = distance_matrix[non_diag_mask]
    global_mean = np.mean(global_dists)
    global_std = np.std(global_dists)
    global_cv = global_std / (global_mean + 1e-8)
    
    # Size factor for parameter scaling
    size_factor = np.log10(n + 1) / 2.0
    
    # =====================================================================
    # 2. SPECTRAL CLUSTERING FOR IMPROVED DENSITY ESTIMATION
    # =====================================================================
    
    # Compute affinity matrix using Gaussian kernel
    sigma = np.median(global_dists) / 3.0
    affinity = np.exp(-distance_matrix**2 / (2 * sigma**2))
    np.fill_diagonal(affinity, 0)
    
    # Degree matrix and normalized Laplacian
    degree = np.diag(np.sum(affinity, axis=1))
    degree_inv_sqrt = np.diag(1.0 / np.sqrt(np.diag(degree) + 1e-8))
    laplacian = np.eye(n) - degree_inv_sqrt @ affinity @ degree_inv_sqrt
    
    # Compute eigenvectors for spectral embedding
    k_clusters = max(2, min(8, n // 4))  # Adaptive cluster count
    if n > 10:
        # Use top k eigenvectors for embedding
        try:
            eigvals, eigvecs = np.linalg.eigh(laplacian)
            # Skip first eigenvector (constant)
            embedding = eigvecs[:, 1:k_clusters+1]
            
            # Normalize rows for spherical clustering
            row_norms = np.linalg.norm(embedding, axis=1, keepdims=True)
            embedding = embedding / (row_norms + 1e-8)
            
            # Spectral density: average similarity in embedding space
            spectral_similarity = embedding @ embedding.T
            spectral_density = np.mean(spectral_similarity, axis=1)
            
            # Cluster assignment via simple k-means on embedding
            from sklearn.cluster import KMeans
            kmeans = KMeans(n_clusters=k_clusters, n_init=10, random_state=42)
            cluster_labels = kmeans.fit_predict(embedding)
            
            # Inter-cluster and intra-cluster statistics
            cluster_stats = {}
            for c in range(k_clusters):
                mask = cluster_labels == c
                if np.sum(mask) > 0:
                    # Intra-cluster distances
                    intra_dists = distance_matrix[mask][:, mask]
                    intra_mean = np.mean(intra_dists[intra_dists > 0]) if np.sum(intra_dists > 0) > 0 else global_mean
                    cluster_stats[c] = {
                        'intra_mean': intra_mean,
                        'density': 1.0 / (intra_mean + 1e-8),
                        'size': np.sum(mask)
                    }
            
            # Enhanced spectral density with cluster weighting
            spectral_density_weighted = np.zeros(n)
            for i in range(n):
                c = cluster_labels[i]
                # Combine local spectral similarity with cluster density
                cluster_weight = cluster_stats[c]['density'] if c in cluster_stats else 1.0
                spectral_density_weighted[i] = spectral_density[i] * cluster_weight
            
            # Normalize spectral density
            if np.max(spectral_density_weighted) > 0:
                spectral_density_weighted = spectral_density_weighted / np.max(spectral_density_weighted)
                
        except:
            # Fallback to traditional density estimation
            spectral_density_weighted = np.ones(n)
            cluster_labels = np.zeros(n, dtype=int)
    else:
        spectral_density_weighted = np.ones(n)
        cluster_labels = np.zeros(n, dtype=int)
    
    # =====================================================================
    # 3. MULTI-STEP REGRET LOOKAHEAD
    # =====================================================================
    
    # Pre-sort distances for all nodes
    sorted_indices = np.zeros((n, n-1), dtype=int)
    sorted_dists = np.zeros((n, n-1))
    
    for i in range(n):
        dists = distance_matrix[i, :].copy()
        dists[i] = np.inf
        sorted_idx = np.argsort(dists)
        sorted_indices[i] = sorted_idx[:n-1]
        sorted_dists[i] = dists[sorted_idx[:n-1]]
    
    # Multi-step regret evaluation
    regret_matrix = np.ones((n, n))
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            dist_ij = distance_matrix[i, j]
            if dist_ij == 0:
                continue
            
            # Step 1: Direct alternatives (two-step as before)
            # Alternative at i
            alt_i_idx = 0
            while alt_i_idx < n-1 and sorted_indices[i, alt_i_idx] == j:
                alt_i_idx += 1
            alt_i = sorted_indices[i, alt_i_idx] if alt_i_idx < n-1 else j
            alt_i_dist = sorted_dists[i, alt_i_idx] if alt_i_idx < n-1 else dist_ij
            
            # Alternative at j
            alt_j_idx = 0
            while alt_j_idx < n-1 and sorted_indices[j, alt_j_idx] == i:
                alt_j_idx += 1
            alt_j = sorted_indices[j, alt_j_idx] if alt_j_idx < n-1 else i
            alt_j_dist = sorted_dists[j, alt_j_idx] if alt_j_idx < n-1 else dist_ij
            
            # Step 2: Extended lookahead (three-step)
            # From alternative at i, find its best alternative (excluding i and j)
            if alt_i != j and alt_i < n:
                alt_i_distances = distance_matrix[alt_i, :].copy()
                alt_i_distances[alt_i] = np.inf
                alt_i_distances[i] = np.inf
                alt_i_distances[j] = np.inf
                
                if np.any(alt_i_distances < np.inf):
                    alt_i_best = np.argmin(alt_i_distances)
                    alt_i_best_dist = alt_i_distances[alt_i_best]
                    # Extended regret: cost of not taking i->j vs i->alt_i->alt_i_best
                    extended_regret_i = (alt_i_dist + alt_i_best_dist - dist_ij) / (dist_ij + 1e-8)
                else:
                    extended_regret_i = 0
            else:
                extended_regret_i = 0
            
            # Similar extended lookahead from alternative at j
            if alt_j != i and alt_j < n:
                alt_j_distances = distance_matrix[alt_j, :].copy()
                alt_j_distances[alt_j] = np.inf
                alt_j_distances[j] = np.inf
                alt_j_distances[i] = np.inf
                
                if np.any(alt_j_distances < np.inf):
                    alt_j_best = np.argmin(alt_j_distances)
                    alt_j_best_dist = alt_j_distances[alt_j_best]
                    extended_regret_j = (alt_j_dist + alt_j_best_dist - dist_ij) / (dist_ij + 1e-8)
                else:
                    extended_regret_j = 0
            else:
                extended_regret_j = 0
            
            # Combine regret components
            direct_regret = (alt_i_dist + alt_j_dist - 2 * dist_ij) / (2 * dist_ij + 1e-8)
            extended_regret = (extended_regret_i + extended_regret_j) / 2
            
            # Weighted regret with emphasis on direct alternatives
            total_regret = 0.7 * direct_regret + 0.3 * extended_regret
            
            # Convert regret to scoring (higher regret -> lower score)
            regret_score = np.exp(-total_regret)
            regret_matrix[i, j] = max(0.1, min(2.0, regret_score))
    
    # =====================================================================
    # 4. DYNAMIC COMPONENT WEIGHT ADJUSTMENT
    # =====================================================================
    
    # Initial component weights based on instance characteristics
    if global_cv < 0.3:  # Uniform instances
        weights = {'inv_dist': 0.15, 'regret': 0.35, 'spectral': 0.30, 'proximity': 0.20}
    elif global_cv < 0.7:  # Moderate variability
        weights = {'inv_dist': 0.20, 'regret': 0.30, 'spectral': 0.25, 'proximity': 0.25}
    else:  # High variability
        weights = {'inv_dist': 0.25, 'regret': 0.25, 'spectral': 0.30, 'proximity': 0.20}
    
    # Adjust weights based on spectral analysis
    if 'cluster_labels' in locals():
        # Measure cluster separation
        intra_inter_ratio = 0
        if len(np.unique(cluster_labels)) > 1:
            intra_dists = []
            inter_dists = []
            for i in range(n):
                for j in range(i+1, n):
                    if cluster_labels[i] == cluster_labels[j]:
                        intra_dists.append(distance_matrix[i, j])
                    else:
                        inter_dists.append(distance_matrix[i, j])
            
            if intra_dists and inter_dists:
                intra_mean = np.mean(intra_dists)
                inter_mean = np.mean(inter_dists)
                intra_inter_ratio = intra_mean / (inter_mean + 1e-8)
        
        # Adjust weights based on cluster structure
        if intra_inter_ratio < 0.5:  # Well-separated clusters
            weights['spectral'] *= 1.5
            weights['inv_dist'] *= 0.7
        elif intra_inter_ratio > 2.0:  # Poor separation
            weights['regret'] *= 1.3
            weights['spectral'] *= 0.7
    
    # Normalize weights
    total_weight = sum(weights.values())
    for key in weights:
        weights[key] /= total_weight
    
    # =====================================================================
    # 5. COMPONENT COMPUTATION
    # =====================================================================
    
    # 5.1 Inverse distance with adaptive regularization
    reg_coef = 0.02 * global_mean * (1.0 + global_cv)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + reg_coef)
        inv_dist[np.diag_indices(n)] = 0
    
    # 5.2 Spectral-based similarity
    spectral_matrix = np.ones((n, n))
    if 'spectral_density_weighted' in locals():
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                # Combine spectral density and cluster membership
                density_sim = 1.0 - abs(spectral_density_weighted[i] - spectral_density_weighted[j])
                cluster_sim = 1.0 if cluster_labels[i] == cluster_labels[j] else 0.3
                
                spectral_matrix[i, j] = 1.0 + 0.8 * (0.6 * density_sim + 0.4 * cluster_sim)
    
    # 5.3 Proximity with adaptive thresholds
    proximity_matrix = np.ones((n, n))
    
    # Local reference distances (using percentiles for robustness)
    local_refs = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i, :].copy()
        dists[i] = np.inf
        if n > 1:
            local_refs[i] = np.percentile(dists[dists < np.inf], 30)
        else:
            local_refs[i] = global_mean
    
    # Adaptive proximity scoring
    threshold_base = 1.5 + 0.5 * global_cv
    for i in range(n):
        threshold_i = threshold_base * local_refs[i]
        
        for j in range(n):
            if i == j:
                continue
            
            dist = distance_matrix[i, j]
            threshold_j = threshold_base * local_refs[j]
            combined_threshold = (threshold_i + threshold_j) / 2
            
            if dist <= combined_threshold:
                proximity_ratio = 1.0 - dist / (combined_threshold + 1e-8)
                proximity_matrix[i, j] = 1.0 + 0.6 * proximity_ratio
            else:
                excess_ratio = dist / combined_threshold
                proximity_matrix[i, j] = np.exp(1.2 - excess_ratio)
    
    # 5.4 Minimum spanning tree boost
    try:
        mst = minimum_spanning_tree(distance_matrix)
        mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
        mst_boost = np.where(mst_mask, 1.25 - 0.1 * size_factor, 1.0)
    except:
        mst_boost = np.ones((n, n))
    
    # =====================================================================
    # 6. GEOMETRIC MEAN FUSION WITH ENTROPY-BASED NORMALIZATION
    # =====================================================================
    
    # Ensure positive values
    inv_dist_pos = np.maximum(inv_dist, 1e-8)
    regret_pos = np.maximum(regret_matrix, 1e-8)
    spectral_pos = np.maximum(spectral_matrix, 1e-8)
    proximity_pos = np.maximum(proximity_matrix, 1e-8)
    
    # Weighted geometric mean
    combined = (
        np.power(inv_dist_pos, weights['inv_dist']) *
        np.power(regret_pos, weights['regret']) *
        np.power(spectral_pos, weights['spectral']) *
        np.power(proximity_pos, weights['proximity']) *
        mst_boost
    )
    
    # Ensure symmetry and zero diagonal
    combined = (combined + combined.T) / 2.0
    np.fill_diagonal(combined, 0)
    
    # =====================================================================
    # 7. ENTROPY-BASED NORMALIZATION AND SPARSIFICATION
    # =====================================================================
    
    # Adaptive sparsification based on instance size
    if n <= 20:
        min_edges = max(3, n // 2)
        max_edges = n - 1
    elif n <= 100:
        min_edges = max(4, int(0.3 * n))
        max_edges = max(8, int(0.4 * n))
    else:
        min_edges = max(5, int(0.25 * n))
        max_edges = max(10, int(0.35 * n))
    
    # Create selection mask
    keep_mask = np.zeros((n, n), dtype=bool)
    
    # Always include MST edges
    if 'mst_mask' in locals():
        keep_mask = keep_mask | mst_mask
    
    # Multi-criteria edge selection with entropy weighting
    for i in range(n):
        # Calculate entropy-based weights for criteria
        criteria_scores = np.column_stack([
            inv_dist_pos[i, :],
            regret_pos[i, :],
            spectral_pos[i, :],
            proximity_pos[i, :]
        ])
        
        # Normalize criteria scores
        criteria_norm = criteria_scores / (np.sum(criteria_scores, axis=0, keepdims=True) + 1e-8)
        
        # Calculate entropy for each criterion
        entropy = -np.sum(criteria_norm * np.log(criteria_norm + 1e-8), axis=0)
        max_entropy = np.log(n)
        entropy_weights = 1.0 - entropy / max_entropy  # Higher weight for more informative criteria
        
        # Normalize entropy weights
        entropy_weights = entropy_weights / (np.sum(entropy_weights) + 1e-8)
        
        # Weighted combination of criteria
        weighted_scores = np.sum(criteria_scores * entropy_weights, axis=1)
        weighted_scores[i] = -np.inf
        
        # Select top edges
        target = min(max_edges, n-1)
        top_indices = np.argpartition(weighted_scores, -target)[-target:]
        keep_mask[i, top_indices] = True
        keep_mask[top_indices, i] = True
    
    # Ensure minimum connectivity
    for i in range(n):
        connected = np.where(keep_mask[i, :])[0]
        connected = connected[connected != i]
        
        if len(connected) < min_edges:
            # Add nearest promising edges
            candidate_scores = combined[i, :].copy()
            candidate_scores[i] = -np.inf
            candidate_scores[keep_mask[i, :]] = -np.inf  # Exclude already selected
            
            if np.any(candidate_scores > 0):
                add_count = min_edges - len(connected)
                top_candidates = np.argpartition(candidate_scores, -add_count)[-add_count:]
                keep_mask[i, top_candidates] = True
                keep_mask[top_candidates, i] = True
    
    # Ensure graph connectivity
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        current = stack.pop()
        neighbors = np.where(keep_mask[current, :])[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    # Connect any disconnected components
    if not np.all(visited):
        # Find all components
        components = []
        remaining = np.where(~visited)[0]
        
        while len(remaining) > 0:
            comp = []
            stack = [remaining[0]]
            
            while stack:
                node = stack.pop()
                if not visited[node]:
                    visited[node] = True
                    comp.append(node)
                    
                    # Get neighbors in the full graph (distance-based)
                    dists = distance_matrix[node, :]
                    closest = np.argsort(dists)[:min(3, n)]
                    for neighbor in closest:
                        if neighbor != node and not visited[neighbor]:
                            stack.append(neighbor)
            
            components.append(comp)
            remaining = np.where(~visited)[0]
        
        # Connect components with shortest edges
        for i in range(len(components) - 1):
            comp_a = components[i]
            comp_b = components[i + 1]
            
            min_dist = np.inf
            best_pair = (0, 0)
            
            for a in comp_a:
                for b in comp_b:
                    dist = distance_matrix[a, b]
                    if dist < min_dist:
                        min_dist = dist
                        best_pair = (a, b)
            
            keep_mask[best_pair] = True
            keep_mask[best_pair[::-1]] = True
    
    # Apply sparsification mask
    heuristic = combined * keep_mask
    
    # =====================================================================
    # 8. FINAL NORMALIZATION WITH TEMPERATURE SCALING
    # =====================================================================
    
    # Temperature scaling based on instance characteristics
    temperature = 0.5 + 0.5 * global_cv  # Higher temperature for more variable instances
    
    for i in range(n):
        row = heuristic[i, :]
        mask = row > 0
        
        if np.sum(mask) > 0:
            # Apply temperature scaling
            scaled_row = row[mask] / temperature
            
            # Softmax normalization
            exp_row = np.exp(scaled_row - np.max(scaled_row))
            heuristic[i, mask] = exp_row / np.sum(exp_row)
    
    # Final symmetry and zero diagonal
    heuristic = (heuristic + heuristic.T) / 2.0
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
