import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristic with dynamic parameter optimization, 
    path-based global structure analysis, and ensemble fusion.
    
    Core innovations:
    1. Dynamic parameter optimization: Uses instance statistics to automatically 
       adjust algorithm parameters for optimal performance
    2. Path-based global structure scoring: Evaluates edges in context of 
       potential paths rather than isolated connections
    3. Edge betweenness centrality: Computes edge importance from multiple 
       greedy tours to identify critical connections
    4. Adaptive ensemble fusion: Dynamically weights different heuristic 
       components based on their predictive value
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate more
        promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # =====================================================================
    # 1. INSTANCE ANALYSIS FOR DYNAMIC PARAMETER OPTIMIZATION
    # =====================================================================
    
    # Compute comprehensive instance statistics
    non_diag_mask = ~np.eye(n, dtype=bool)
    global_dists = distance_matrix[non_diag_mask]
    global_mean = np.mean(global_dists)
    global_std = np.std(global_dists)
    global_cv = global_std / (global_mean + 1e-8)
    
    # Shape statistics for parameter adaptation
    edge_ratios = []
    for i in range(n):
        dists = distance_matrix[i, :].copy()
        dists[i] = np.inf
        if len(dists) >= 3:
            sorted_dists = np.sort(dists[dists < np.inf])
            if len(sorted_dists) >= 2:
                edge_ratios.append(sorted_dists[1] / sorted_dists[0])
    
    uniformity = np.std(edge_ratios) if edge_ratios else 1.0
    
    # Dynamic parameter settings based on instance characteristics
    size_factor = np.log10(n + 1) / 2.0
    
    # Adaptive neighborhood size
    base_k = max(3, n // 8)
    k = min(base_k + int(7 * global_cv), n - 1)
    
    # Sparsity control
    if n <= 20:
        base_sparsity = 0.7 - 0.1 * size_factor
    elif n <= 100:
        base_sparsity = 0.5 - 0.08 * size_factor
    else:
        base_sparsity = 0.3 - 0.05 * size_factor
    
    sparsity_factor = base_sparsity * (1.0 + 0.5 * global_cv - 0.3 * uniformity)
    
    # =====================================================================
    # 2. PATH-BASED GLOBAL STRUCTURE ANALYSIS
    # =====================================================================
    
    # 2.1 Potential path evaluation (2-step lookahead)
    path_scores = np.ones((n, n))
    
    # Precompute sorted distances and indices for all nodes
    sorted_indices = np.zeros((n, n-1), dtype=int)
    sorted_dists = np.zeros((n, n-1))
    
    for i in range(n):
        dists = distance_matrix[i, :].copy()
        dists[i] = np.inf
        sorted_idx = np.argsort(dists)
        sorted_indices[i] = sorted_idx[:n-1]
        sorted_dists[i] = dists[sorted_idx[:n-1]]
    
    # Evaluate edges in context of potential 2-node paths
    for i in range(n):
        for rank_i, j in enumerate(sorted_indices[i]):
            if i == j:
                continue
                
            edge_dist = distance_matrix[i, j]
            
            # Look at potential continuations from both ends
            # From node j perspective
            if rank_i < n - 2:
                next_from_j = sorted_dists[j, 0]  # Best from j (excluding i if it's the best)
                # Adjust if i is the best from j
                if sorted_indices[j, 0] == i and n > 2:
                    next_from_j = sorted_dists[j, 1]
                
                # From node i perspective (excluding j)
                if sorted_indices[i, 0] == j and n > 2:
                    next_from_i = sorted_dists[i, 1]
                else:
                    next_from_i = sorted_dists[i, 0]
                
                # Path quality score: how good is this edge in a 2-step context
                path_length = edge_dist + min(next_from_i, next_from_j)
                reference_length = sorted_dists[i, 0] + sorted_dists[j, 0]
                
                if reference_length > 0:
                    path_ratio = reference_length / (path_length + 1e-8)
                    # Boost edges that form good potential paths
                    path_scores[i, j] = 1.0 + 0.8 * np.tanh(path_ratio - 1.0)
                else:
                    path_scores[i, j] = 1.0
            else:
                path_scores[i, j] = 1.0
    
    # 2.2 Local cluster identification
    cluster_scores = np.ones((n, n))
    
    # Adaptive local density estimation
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i, :].copy()
        dists[i] = np.inf
        
        if n > k:
            kth_dist = np.partition(dists, k)[k]
            neighbor_mask = dists <= kth_dist
            neighbor_dists = dists[neighbor_mask]
        else:
            neighbor_dists = dists[dists < np.inf]
        
        if len(neighbor_dists) > 0:
            # Use median for robustness
            local_density[i] = 1.0 / (np.median(neighbor_dists) + 1e-8)
        else:
            local_density[i] = 1.0
    
    # Normalize densities
    if np.max(local_density) > np.min(local_density):
        local_density = (local_density - np.min(local_density)) / (np.max(local_density) - np.min(local_density))
    
    # Compute cluster connectivity scores
    cluster_threshold = np.percentile(local_density, 70) if n > 5 else 0.5
    
    for i in range(n):
        for j in range(i+1, n):
            # Both nodes in high-density regions
            if local_density[i] > cluster_threshold and local_density[j] > cluster_threshold:
                # Bonus for intra-cluster connections
                similarity = 1.0 - abs(local_density[i] - local_density[j])
                cluster_scores[i, j] = 1.0 + 0.6 * similarity
                cluster_scores[j, i] = cluster_scores[i, j]
            # One node in high-density, other not
            elif local_density[i] > cluster_threshold or local_density[j] > cluster_threshold:
                # Moderate bonus for boundary connections
                cluster_scores[i, j] = 1.0 + 0.3
                cluster_scores[j, i] = cluster_scores[i, j]
    
    # =====================================================================
    # 3. EDGE BETWEENNESS FROM GREEDY TOURS
    # =====================================================================
    
    # Generate multiple greedy tours for edge frequency analysis
    tour_count = min(50, max(10, n // 2))
    edge_frequency = np.zeros((n, n))
    
    for tour_idx in range(tour_count):
        # Start from different nodes
        start_node = (tour_idx * 13) % n  # Simple pseudo-random starting point
        
        visited = np.zeros(n, dtype=bool)
        current = start_node
        visited[current] = True
        
        # Build greedy tour
        for step in range(n - 1):
            # Get unvisited neighbors
            unvisited = ~visited
            unvisited[current] = False  # Exclude current
            
            if not np.any(unvisited):
                break
            
            # Find nearest unvisited neighbor
            distances = distance_matrix[current, :].copy()
            distances[visited] = np.inf
            next_node = np.argmin(distances)
            
            # Record edge
            edge_frequency[current, next_node] += 1
            edge_frequency[next_node, current] += 1
            
            current = next_node
            visited[current] = True
        
        # Return to start
        edge_frequency[current, start_node] += 1
        edge_frequency[start_node, current] += 1
    
    # Convert frequencies to scores (edges that appear in many tours are important)
    if np.max(edge_frequency) > 0:
        frequency_scores = 1.0 + 0.8 * edge_frequency / np.max(edge_frequency)
    else:
        frequency_scores = np.ones((n, n))
    
    # =====================================================================
    # 4. CORE HEURISTIC COMPONENTS
    # =====================================================================
    
    # 4.1 Inverse distance with adaptive regularization
    reg_coef = 0.02 * global_mean * (1.0 + global_cv)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + reg_coef)
        inv_dist[np.diag_indices(n)] = 0
    
    # 4.2 Rank-based regret with opportunity cost
    rank_matrix = np.ones((n, n))
    
    for i in range(n):
        best_dist = sorted_dists[i, 0]
        
        for rank, j in enumerate(sorted_indices[i]):
            if i == j:
                continue
                
            current_dist = distance_matrix[i, j]
            
            if best_dist > 0:
                # Base score
                base_score = best_dist / current_dist
                
                # Opportunity cost: what we lose by choosing this edge
                # Consider alternatives at both ends
                alt_i_idx = 0
                while alt_i_idx < n-1 and sorted_indices[i, alt_i_idx] == j:
                    alt_i_idx += 1
                alt_i_dist = sorted_dists[i, alt_i_idx] if alt_i_idx < n-1 else current_dist
                
                alt_j_idx = 0
                while alt_j_idx < n-1 and sorted_indices[j, alt_j_idx] == i:
                    alt_j_idx += 1
                alt_j_dist = sorted_dists[j, alt_j_idx] if alt_j_idx < n-1 else current_dist
                
                # Combined opportunity cost
                opp_cost_i = (alt_i_dist - current_dist) / (alt_i_dist + 1e-8)
                opp_cost_j = (alt_j_dist - current_dist) / (alt_j_dist + 1e-8)
                combined_opp_cost = max(0, (opp_cost_i + opp_cost_j) / 2.0)
                
                # Rank penalty (decays slower for uniform instances)
                rank_decay = 0.5 if uniformity < 0.3 else 0.7
                rank_penalty = rank / (n - 2) if n > 2 else 0
                
                rank_matrix[i, j] = base_score * (1.0 + combined_opp_cost) * (1.0 - rank_decay * rank_penalty)
            else:
                rank_matrix[i, j] = 1.0
    
    # 4.3 Minimum spanning tree priority
    try:
        mst = minimum_spanning_tree(distance_matrix)
        mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
        mst_boost = np.where(mst_mask, 1.4 - 0.15 * size_factor, 1.0)
    except:
        mst_boost = np.ones((n, n))
    
    # =====================================================================
    # 5. ADAPTIVE ENSEMBLE FUSION
    # =====================================================================
    
    # Component weights based on instance characteristics
    # More weight to path analysis for structured instances
    # More weight to frequency analysis for complex instances
    
    # Base weights
    w_inv_dist = 0.20
    w_rank = 0.25
    w_path = 0.25 * (1.0 + 0.3 * (1.0 - uniformity))  # Higher for structured
    w_cluster = 0.10
    w_frequency = 0.20 * (1.0 + 0.4 * global_cv)  # Higher for complex
    
    # Normalize
    w_sum = w_inv_dist + w_rank + w_path + w_cluster + w_frequency
    w_inv_dist /= w_sum
    w_rank /= w_sum
    w_path /= w_sum
    w_cluster /= w_sum
    w_frequency /= w_sum
    
    # Ensure positive values
    inv_dist_pos = np.maximum(inv_dist, 1e-8)
    rank_pos = np.maximum(rank_matrix, 1e-8)
    path_pos = np.maximum(path_scores, 1e-8)
    cluster_pos = np.maximum(cluster_scores, 1e-8)
    frequency_pos = np.maximum(frequency_scores, 1e-8)
    
    # Weighted geometric mean fusion
    combined = (
        np.power(inv_dist_pos, w_inv_dist) *
        np.power(rank_pos, w_rank) *
        np.power(path_pos, w_path) *
        np.power(cluster_pos, w_cluster) *
        np.power(frequency_pos, w_frequency) *
        mst_boost
    )
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2.0
    np.fill_diagonal(combined, 0)
    
    # =====================================================================
    # 6. INTELLIGENT SPARSIFICATION
    # =====================================================================
    
    # Determine edges to keep
    keep_mask = np.zeros((n, n), dtype=bool)
    
    # Always include MST edges
    if 'mst_mask' in locals():
        keep_mask = keep_mask | mst_mask
    
    # Adaptive edge selection per node
    for i in range(n):
        # Calculate target number of edges for this node
        if n <= 15:
            target_edges = max(3, n // 2)
        elif n <= 50:
            target_edges = max(5, int(0.4 * n * (1.0 - sparsity_factor)))
        else:
            target_edges = max(7, int(0.3 * n * (1.0 - sparsity_factor)))
        
        target_edges = min(target_edges, n - 1)
        
        # Get combined scores for this node
        scores = combined[i, :].copy()
        scores[i] = -np.inf
        
        # Select top edges
        if target_edges > 0:
            # Use argpartition for efficiency
            threshold_idx = n - target_edges - 1
            if threshold_idx >= 0:
                threshold = np.partition(scores, threshold_idx)[threshold_idx]
                keep_mask[i, scores >= threshold] = True
                keep_mask[scores >= threshold, i] = True
    
    # Ensure minimum connectivity of 2 edges per node
    for i in range(n):
        connected = np.where(keep_mask[i, :])[0]
        connected = connected[connected != i]
        
        if len(connected) < 2:
            # Add nearest neighbors
            dists = distance_matrix[i, :].copy()
            dists[i] = np.inf
            
            sorted_idx = np.argsort(dists)
            added = 0
            for j in sorted_idx:
                if j != i and not keep_mask[i, j]:
                    keep_mask[i, j] = True
                    keep_mask[j, i] = True
                    added += 1
                    if len(connected) + added >= 2:
                        break
    
    # Ensure graph connectivity
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        current = stack.pop()
        neighbors = np.where(keep_mask[current, :])[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    if not np.all(visited):
        # Connect disconnected components with shortest edges
        unvisited = np.where(~visited)[0]
        
        for node in unvisited:
            # Find closest visited node
            dists = distance_matrix[node, visited]
            min_dist_idx = np.argmin(dists)
            visited_nodes = np.where(visited)[0]
            closest = visited_nodes[min_dist_idx]
            
            keep_mask[node, closest] = True
            keep_mask[closest, node] = True
            visited[node] = True
    
    # Apply sparsification mask
    heuristic = combined * keep_mask
    
    # =====================================================================
    # 7. FINAL REFINEMENT AND NORMALIZATION
    # =====================================================================
    
    # Softmax normalization per node
    for i in range(n):
        row = heuristic[i, :]
        mask = row > 0
        
        if np.sum(mask) > 0:
            # Shift for numerical stability
            row_masked = row[mask]
            row_masked = row_masked - np.max(row_masked)
            
            # Apply softmax
            exp_row = np.exp(row_masked)
            heuristic[i, mask] = exp_row / np.sum(exp_row)
    
    # Final symmetry enforcement
    heuristic = (heuristic + heuristic.T) / 2.0
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
