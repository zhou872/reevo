import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges combining multiple factors.
    
    Factors considered:
    1. Base inverse distance (shorter edges preferred)
    2. Mutual nearest neighbor bonus (edges connecting mutually close nodes)
    3. Minimum Spanning Tree bonus (edges likely in optimal structure)
    4. Sparsification via distance thresholding (eliminate very long edges)
    
    Returns:
        np.ndarray: Heuristic matrix of same shape as input, with higher values
                   indicating more promising edges for TSP solution.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance with epsilon to avoid division by zero
    eps = 1e-12
    base_score = 1 / (distance_matrix + eps)
    np.fill_diagonal(base_score, 0)  # Exclude self-loops
    
    # 2. Mutual nearest neighbor bonus
    # For each node, identify k nearest neighbors
    k = max(2, min(5, n // 10))  # Adaptive k based on problem size
    nn_bonus = np.ones_like(distance_matrix)
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        nearest_indices = np.argsort(distance_matrix[i])[1:k+1]
        nn_bonus[i, nearest_indices] = 2.0  # Bonus for outgoing nearest neighbors
    
    # Mutual bonus: if edge is in both directions' nearest neighbors
    mutual_mask = (nn_bonus == 2.0) & (nn_bonus.T == 2.0)
    nn_bonus[mutual_mask] = 3.0  # Higher bonus for mutual nearest neighbors
    
    # 3. Minimum Spanning Tree bonus
    try:
        # Get minimum spanning tree from distance matrix
        mst = minimum_spanning_tree(distance_matrix)
        mst_mask = mst.toarray() > 0
        # Bonus for edges that appear in MST
        mst_bonus = np.ones_like(distance_matrix)
        mst_bonus[mst_mask] = 1.5
    except:
        # Fallback if MST computation fails
        mst_bonus = np.ones_like(distance_matrix)
    
    # 4. Combine all factors
    combined_score = base_score * nn_bonus * mst_bonus
    
    # 5. Sparsification: set long edges to zero
    if n > 2:  # Only if we have enough points
        # Calculate distance threshold (75th percentile of unique distances)
        upper_triangle = distance_matrix[np.triu_indices(n, k=1)]
        if len(upper_triangle) > 0:
            threshold = np.percentile(upper_triangle, 75)
            # Zero out edges longer than threshold
            combined_score[distance_matrix > threshold] = 0
    
    # Normalize to make scores comparable across different problem instances
    max_score = combined_score.max()
    if max_score > 0:
        combined_score = combined_score / max_score
    
    return combined_score
