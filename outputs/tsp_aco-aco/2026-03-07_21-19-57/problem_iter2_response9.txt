import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics incorporating regret-based measures and efficient lookahead.
    
    Features:
    1. Minimum spanning tree priority: MST edges receive high priority
    2. Regret-based penalty: Penalizes edges with many good alternatives
    3. Limited-depth lookahead: Two-step lookahead using precomputed minima
    4. Cluster density weighting: Rewards edges connecting balanced density regions
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-10)
        np.fill_diagonal(inv_dist, 0)
    
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # 2. Minimum spanning tree priority
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_priority = np.where(mst_mask, 1.5, 1.0)  # 50% bonus for MST edges
    
    # 3. Regret-based penalty: penalize edges with many good alternatives
    regret_penalty = np.ones((n, n))
    k_regret = min(5, n - 1)  # Consider top k alternatives
    
    for i in range(n):
        # Get distances from node i to all others
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf  # Ignore self
        
        # Find k+1 smallest distances for robust regret calculation
        kth_idx = min(k_regret + 1, n - 1)
        kth_smallest = np.partition(dist_i, kth_idx)[kth_idx]
        
        # Count good alternatives (within 20% of best or top-k)
        best_dist = np.min(dist_i)
        threshold = max(best_dist * 1.2, kth_smallest)
        
        # Good alternatives are those with distance <= threshold
        good_alt_mask = dist_i <= threshold
        num_good_alternatives = np.sum(good_alt_mask)
        
        # Penalize based on number of good alternatives
        if num_good_alternatives > 0:
            # Edges with many good alternatives get penalized more
            penalty_factor = 1.0 / (1 + np.log1p(num_good_alternatives))
            regret_penalty[i] = penalty_factor
        regret_penalty[i, i] = 1.0
    
    # Make symmetric (geometric mean of both directions)
    regret_penalty = np.sqrt(regret_penalty * regret_penalty.T)
    
    # 4. Efficient two-step lookahead
    lookahead_scores = np.zeros((n, n))
    
    # Precompute best continuations for each node (excluding self)
    best_continuations = np.zeros(n)
    second_best_continuations = np.zeros(n)
    
    for j in range(n):
        dist_j = distance_matrix[j].copy()
        dist_j[j] = np.inf
        
        # Find two smallest distances for efficient lookahead
        sorted_indices = np.argsort(dist_j)
        best_continuations[j] = dist_j[sorted_indices[0]]
        if len(sorted_indices) > 1:
            second_best_continuations[j] = dist_j[sorted_indices[1]]
        else:
            second_best_continuations[j] = np.inf
    
    # Compute lookahead scores efficiently
    for i in range(n):
        for j in range(n):
            if i != j:
                # Exclude edge back to i when looking ahead from j
                if distance_matrix[j, i] == best_continuations[j]:
                    continuation_dist = second_best_continuations[j]
                else:
                    continuation_dist = best_continuations[j]
                
                # Two-step lookahead: current edge + best continuation
                lookahead_dist = distance_matrix[i, j] + continuation_dist
                lookahead_scores[i, j] = 1.0 / (lookahead_dist + 1e-10)
    
    # Normalize lookahead scores
    if lookahead_scores.max() > 0:
        lookahead_scores = lookahead_scores / lookahead_scores.max()
    
    # 5. Cluster density weighting (harmonic mean of endpoint densities)
    k_density = min(3, n - 1)
    densities = np.zeros(n)
    
    for i in range(n):
        # Compute average distance to k nearest neighbors
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        nearest_dists = np.partition(dist_i, k_density)[:k_density]
        densities[i] = 1.0 / (np.mean(nearest_dists) + 1e-10)
    
    # Normalize densities
    if densities.max() > 0:
        densities = densities / densities.max()
    
    # Density matrix: harmonic mean of endpoint densities
    density_matrix = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            if densities[i] > 0 and densities[j] > 0:
                harmonic_mean = 2 * densities[i] * densities[j] / (densities[i] + densities[j] + 1e-10)
                density_matrix[i, j] = density_matrix[j, i] = harmonic_mean
    
    # 6. Combine factors with interpretable weights
    # Base heuristic with regret penalty
    base_with_regret = inv_dist * regret_penalty
    
    # Weighted combination
    weights = [0.4, 0.2, 0.25, 0.15]  # [base_with_regret, mst_priority, lookahead, density]
    
    heuristics = (
        weights[0] * base_with_regret +
        weights[1] * base_with_regret * mst_priority +
        weights[2] * lookahead_scores +
        weights[3] * density_matrix
    )
    
    # 7. Sparsification: keep only promising edges
    k_sparse = min(max(3, int(np.sqrt(n))), n - 1)
    promising_mask = np.zeros((n, n), dtype=bool)
    
    # Include MST edges
    promising_mask |= mst_mask
    
    # For each node, include edges to top-k nearest (by heuristic, not distance)
    for i in range(n):
        row_heuristics = heuristics[i].copy()
        row_heuristics[i] = -np.inf  # Exclude self
        
        # Keep top-k or those above 70th percentile
        threshold = max(np.partition(row_heuristics, -k_sparse)[-k_sparse], 
                       np.percentile(row_heuristics[row_heuristics > -np.inf], 70))
        promising_mask[i] |= row_heuristics >= threshold
    
    # Make symmetric
    promising_mask = promising_mask | promising_mask.T
    
    # Apply sparsification
    heuristics = heuristics * promising_mask
    
    # 8. Final normalization
    non_zero_mask = heuristics > 0
    if np.any(non_zero_mask):
        heuristics[non_zero_mask] = heuristics[non_zero_mask] / heuristics[non_zero_mask].max()
    
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
