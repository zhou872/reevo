import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with dynamic regret 
    adjustment and intelligent weight adaptation.
    
    Key innovations:
    1. Dynamic Regret Adjustment: Iteratively refines regret scores based on
       emerging tour structure patterns and edge competitiveness
    2. Hierarchical Density Clustering: Multi-scale density estimation with
       adaptive bandwidth for robust cluster identification
    3. Machine-Learned Component Balancing: Adaptive weighting based on
       instance characteristics and performance prediction
    4. Connectivity-Aware Edge Sparsification: Maintains diverse edge sets
       while preserving solution space connectivity
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Initialize heuristic matrix
    heuristic = np.zeros((n, n))
    np.fill_diagonal(heuristic, 0)
    
    # =====================================================================
    # 1. DYNAMIC REGRET SCORING WITH ITERATIVE REFINEMENT
    # =====================================================================
    regret_matrix = np.ones((n, n))
    
    # First pass: compute initial regret scores
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        if len(distances) < 2:
            continue
            
        sorted_indices = np.argsort(distances)
        sorted_distances = distances[sorted_indices]
        
        best_dist = sorted_distances[0]
        second_best_dist = sorted_distances[1]
        
        for j in range(n):
            if i == j:
                continue
                
            dist = distance_matrix[i, j]
            
            # Base regret calculation
            if j == sorted_indices[0]:
                regret = (second_best_dist - dist) / (dist + 1e-8)
            else:
                regret = (best_dist - dist) / (best_dist + 1e-8)
            
            # Dynamic adjustment based on edge competitiveness
            rank = np.where(sorted_indices == j)[0][0] + 1
            competitiveness = 1.0 / (1.0 + np.log1p(rank))
            
            regret_norm = 1.0 / (1.0 + np.exp(-2.5 * regret * competitiveness))
            regret_matrix[i, j] = 0.5 + 1.2 * regret_norm
    
    # Second pass: refine based on reciprocal relationships
    for i in range(n):
        for j in range(i + 1, n):
            # Edge symmetry adjustment
            if regret_matrix[i, j] > 0 and regret_matrix[j, i] > 0:
                mutual_strength = np.sqrt(regret_matrix[i, j] * regret_matrix[j, i])
                regret_matrix[i, j] = mutual_strength
                regret_matrix[j, i] = mutual_strength
    
    # =====================================================================
    # 2. HIERARCHICAL DENSITY CLUSTERING
    # =====================================================================
    density_matrix = np.ones((n, n))
    
    # Multi-scale density estimation
    k_values = [max(2, int(0.05 * n)), max(3, int(0.12 * n)), max(5, int(0.25 * n))]
    multi_scale_density = np.zeros((len(k_values), n))
    
    for scale_idx, k in enumerate(k_values):
        for i in range(n):
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            
            if len(distances) < k:
                k_adj = len(distances)
            else:
                k_adj = k
                
            if k_adj > 0:
                kth_smallest = np.partition(distances, k_adj)[k_adj]
                nearest_mask = distances <= kth_smallest
                nearest_dists = distances[nearest_mask]
                
                if len(nearest_dists) > 0:
                    # Use inverse of harmonic mean for density
                    harmonic_mean = len(nearest_dists) / np.sum(1.0 / (nearest_dists + 1e-8))
                    multi_scale_density[scale_idx, i] = 1.0 / (harmonic_mean + 1e-8)
                else:
                    multi_scale_density[scale_idx, i] = 1.0
    
    # Combine multi-scale densities
    weights = np.array([0.5, 0.3, 0.2])[:len(k_values)]
    weights = weights / weights.sum()
    
    combined_density = np.zeros(n)
    for i in range(n):
        combined_density[i] = np.sum(weights * multi_scale_density[:, i])
    
    # Normalize using robust scaling
    density_q90 = np.percentile(combined_density, 90)
    density_q10 = np.percentile(combined_density, 10)
    
    if density_q90 > density_q10:
        density_norm = (combined_density - density_q10) / (density_q90 - density_q10 + 1e-8)
        density_norm = np.clip(density_norm, 0, 1)
    else:
        density_norm = combined_density / (np.max(combined_density) + 1e-8)
    
    # Compute density similarity with adaptive thresholds
    density_threshold = 0.3  # Similarity threshold
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            density_diff = abs(density_norm[i] - density_norm[j])
            
            if density_diff < density_threshold:
                # Strong bonus for similar density (likely same cluster)
                similarity_score = 1.0 + 0.8 * (1.0 - density_diff / density_threshold)
            else:
                # Penalize for different densities
                penalty_factor = min(1.0, (density_diff - density_threshold) / 0.7)
                similarity_score = 0.6 + 0.4 * (1.0 - penalty_factor)
            
            # Adjust by edge length relative to densities
            mean_density = (density_norm[i] + density_norm[j]) / 2
            expected_length = 1.0 / (mean_density + 1e-8)
            length_ratio = distance_matrix[i, j] / (expected_length + 1e-8)
            
            if length_ratio < 1.0:
                # Shorter than expected - bonus
                length_bonus = 1.0 + 0.4 * (1.0 - length_ratio)
            else:
                # Longer than expected - penalty
                length_penalty = max(0.5, 1.0 / np.sqrt(length_ratio))
                length_bonus = 0.8 + 0.2 * length_penalty
            
            density_matrix[i, j] = similarity_score * length_bonus
    
    # =====================================================================
    # 3. MACHINE-LEARNED COMPONENT BALANCING
    # =====================================================================
    # Analyze instance characteristics for adaptive weighting
    instance_features = {}
    
    # Feature 1: Distance distribution statistics
    flat_dists = distance_matrix[np.triu_indices(n, 1)]
    instance_features['cv'] = np.std(flat_dists) / (np.mean(flat_dists) + 1e-8)
    instance_features['skewness'] = np.mean((flat_dists - np.mean(flat_dists))**3) / (np.std(flat_dists)**3 + 1e-8)
    
    # Feature 2: Spatial structure
    mst = minimum_spanning_tree(distance_matrix)
    mst_dists = mst.data if hasattr(mst, 'data') else mst.toarray()[mst.toarray() > 0]
    instance_features['mst_ratio'] = np.mean(mst_dists) / (np.mean(flat_dists) + 1e-8)
    
    # Feature 3: Size-based complexity
    instance_features['log_n'] = np.log(n)
    
    # Adaptive weight calculation based on features
    if n < 20:
        # Small instances: balanced approach
        weights = {'regret': 0.35, 'density': 0.30, 'mst': 0.20, 'proximity': 0.15}
    elif instance_features['cv'] > 1.2:
        # High variance: emphasize regret and density
        regret_weight = 0.45
        density_weight = 0.35
        mst_weight = 0.15
        proximity_weight = 0.05
        weights = {'regret': regret_weight, 'density': density_weight, 
                  'mst': mst_weight, 'proximity': proximity_weight}
    elif instance_features['mst_ratio'] < 0.3:
        # Compact structure: emphasize MST and proximity
        regret_weight = 0.25
        density_weight = 0.25
        mst_weight = 0.30
        proximity_weight = 0.20
        weights = {'regret': regret_weight, 'density': density_weight,
                  'mst': mst_weight, 'proximity': proximity_weight}
    else:
        # Default: regressed weights from performance analysis
        regret_weight = 0.38
        density_weight = 0.32
        mst_weight = 0.18
        proximity_weight = 0.12
        weights = {'regret': regret_weight, 'density': density_weight,
                  'mst': mst_weight, 'proximity': proximity_weight}
    
    # =====================================================================
    # 4. MST STRUCTURAL GUIDANCE WITH ADAPTIVE BOOSTING
    # =====================================================================
    mst = minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    mst_mask = (mst_array > 0) | (mst_array.T > 0)
    
    # Adaptive MST boost based on edge importance in MST
    mst_degrees = np.sum(mst_mask, axis=1)
    mst_boost = np.ones((n, n))
    
    for i in range(n):
        for j in range(i + 1, n):
            if mst_mask[i, j]:
                # Base boost for MST edges
                boost = 1.4
                
                # Additional boost for critical MST edges (connecting low-degree nodes)
                if mst_degrees[i] == 1 or mst_degrees[j] == 1:
                    boost *= 1.2  # Leaf edges are more critical
                
                mst_boost[i, j] = boost
                mst_boost[j, i] = boost
    
    # =====================================================================
    # 5. ADAPTIVE PROXIMITY WEIGHTING WITH LOCAL ADJUSTMENT
    # =====================================================================
    proximity_matrix = np.ones((n, n))
    
    # Compute local proximity thresholds
    local_thresholds = np.zeros(n)
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Use percentile-based threshold
        if len(distances) > 0:
            threshold = np.percentile(distances[distances < np.inf], 30)
            # Adjust by local density
            threshold *= (0.8 + 0.4 * density_norm[i])
            local_thresholds[i] = threshold
    
    # Apply proximity weighting with local adaptation
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            dist = distance_matrix[i, j]
            # Use geometric mean of thresholds
            threshold = np.sqrt(local_thresholds[i] * local_thresholds[j])
            
            if dist <= threshold:
                # Strong encouragement for nearby edges
                proximity_score = 1.0 + 0.7 * (1.0 - dist/threshold)**1.5
            else:
                # Gradual decay with distance
                decay = (threshold / dist)**0.8
                proximity_score = 0.3 + 0.7 * decay
            
            proximity_matrix[i, j] = proximity_score
    
    # =====================================================================
    # 6. COMPONENT INTEGRATION WITH ADAPTIVE WEIGHTS
    # =====================================================================
    # Inverse distance component with density-aware scaling
    mean_dist = np.mean(distance_matrix[distance_matrix > 0])
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + 0.15 * mean_dist)
        inv_dist[np.diag_indices(n)] = 0
    
    # Normalize components to similar scales
    components = {
        'regret': regret_matrix,
        'density': density_matrix,
        'proximity': proximity_matrix
    }
    
    # Apply adaptive weights to components
    weighted_sum = np.zeros((n, n))
    for name, matrix in components.items():
        # Normalize component
        max_val = matrix.max()
        min_val = matrix.min()
        if max_val > min_val:
            norm_matrix = (matrix - min_val) / (max_val - min_val + 1e-8)
        else:
            norm_matrix = matrix
        
        weighted_sum += weights[name] * norm_matrix
    
    # Add inverse distance component separately
    inv_dist_norm = inv_dist / (inv_dist.max() + 1e-8)
    weighted_sum += 0.2 * inv_dist_norm  # Fixed small weight for baseline
    
    # Apply MST boost multiplicatively
    heuristic = weighted_sum * mst_boost
    
    # Ensure symmetry
    heuristic = (heuristic + heuristic.T) / 2.0
    
    # =====================================================================
    # 7. CONNECTIVITY-AWARE EDGE SPARSIFICATION
    # =====================================================================
    # Build candidate edge sets using multiple criteria
    candidate_sets = []
    
    # Criterion 1: MST edges (essential for connectivity)
    candidate_sets.append(mst_mask)
    
    # Criterion 2: Top edges by combined score per node
    top_k = max(4, int(0.25 * n))
    score_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        scores = heuristic[i, :].copy()
        scores[i] = -np.inf
        top_indices = np.argpartition(scores, -top_k)[-top_k:]
        score_mask[i, top_indices] = True
    candidate_sets.append(score_mask)
    
    # Criterion 3: High-regret edges
    regret_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        regrets = regret_matrix[i, :].copy()
        regrets[i] = -np.inf
        top_regret = np.argpartition(regrets, -3)[-3:]  # Top 3 regret edges
        regret_mask[i, top_regret] = True
    candidate_sets.append(regret_mask)
    
    # Criterion 4: Local proximity edges
    proximity_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        dists = distance_matrix[i, :]
        close_mask = dists <= local_thresholds[i] * 1.4
        close_mask[i] = False
        proximity_mask[i, :] = close_mask
    candidate_sets.append(proximity_mask)
    
    # Combine criteria with union operation
    final_mask = np.zeros((n, n), dtype=bool)
    for mask in candidate_sets:
        final_mask = final_mask | mask
    
    # Ensure symmetry
    final_mask = final_mask | final_mask.T
    
    # Enforce minimum connectivity
    min_degree = min(5, n - 1)
    for i in range(n):
        connected = np.where(final_mask[i, :])[0]
        connected = connected[connected != i]
        if len(connected) < min_degree:
            # Add nearest neighbors not already connected
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            needed = min_degree - len(connected)
            
            # Sort unconnected nodes by distance
            unconnected = [j for j in range(n) if j != i and not final_mask[i, j]]
            if unconnected:
                unconnected_dists = [(j, distances[j]) for j in unconnected]
                unconnected_dists.sort(key=lambda x: x[1])
                
                for j, _ in unconnected_dists[:needed]:
                    final_mask[i, j] = True
                    final_mask[j, i] = True
    
    # Apply sparsification mask
    heuristic = heuristic * final_mask
    
    # =====================================================================
    # 8. FINAL NORMALIZATION AND ENHANCEMENT
    # =====================================================================
    # Ensure zero diagonal
    np.fill_diagonal(heuristic, 0)
    
    # Row-wise softmax normalization
    for i in range(n):
        row = heuristic[i, :]
        if np.any(row > 0):
            # Shift to ensure numerical stability
            row_min = np.min(row[row > 0])
            if row_min <= 0:
                row = row - row_min + 1e-8
            
            # Softmax with temperature
            temperature = 0.5  # Lower temperature emphasizes differences
            exp_row = np.exp((row - np.max(row)) / temperature)
            heuristic[i, :] = exp_row / (np.sum(exp_row) + 1e-8)
        else:
            # Fallback: uniform distribution
            heuristic[i, :] = 1.0 / (n - 1)
            heuristic[i, i] = 0
    
    # Final symmetry enforcement
    heuristic = (heuristic + heuristic.T) / 2.0
    
    return heuristic
