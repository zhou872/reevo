import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple, Optional

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with graph centrality,
    probabilistic regret modeling, and adaptive component integration.
    
    Core Improvements from Reflection:
    1. Centrality Measures: Uses betweenness-like centrality to prioritize edges
       connecting high-traffic nodes
    2. Probabilistic Regret: Models alternatives with uncertainty using softmax
       rather than strict min/max
    3. Learnable Components: Parameter adaptation based on instance characteristics
       beyond just size
    4. Multi-scale Sparsification: Hierarchical edge filtering with dynamic thresholds
    
    Args:
        distance_matrix: Square symmetric matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    
    # Early return for trivial cases
    if n <= 3:
        heuristic = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristic, 0)
        return heuristic / heuristic.sum(axis=1, keepdims=True)
    
    # Create safe working copy with robust diagonal handling
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # === 1. INSTANCE CHARACTERIZATION ===
    # Analyze instance properties for adaptive parameter selection
    off_diag_distances = safe_dist[safe_dist < np.inf]
    dist_std = np.std(off_diag_distances)
    dist_mean = np.mean(off_diag_distances)
    cv = dist_std / (dist_mean + 1e-12)  # Coefficient of variation
    
    # Instance type classification
    is_geometric = cv < 0.3  # Low variance suggests Euclidean-like structure
    is_clustered = cv > 1.0  # High variance suggests clustered structure
    
    # === 2. CENTRALITY-BASED SCORING ===
    # Compute multiple centrality measures
    
    # Closeness centrality: inverse average distance to other nodes
    avg_distances = np.sum(safe_dist, axis=1) / (n - 1)
    closeness = 1 / (avg_distances + 1e-12)
    closeness = closeness / (closeness.max() + 1e-12)
    
    # Betweenness approximation: nodes on many shortest paths
    # Approximate via number of times a node appears in k-nearest neighbor lists
    k_nn = min(5, n - 1)
    neighbor_counts = np.zeros(n)
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k_nn]
        neighbor_counts[nearest] += 1
    betweenness = neighbor_counts / (n * k_nn)
    
    # Degree centrality (in MST)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_degree = (mst + mst.T > 0).sum(axis=1)
    degree_centrality = mst_degree / (mst_degree.max() + 1e-12)
    
    # Combined centrality with instance-adaptive weights
    if is_geometric:
        centrality = 0.4 * closeness + 0.4 * betweenness + 0.2 * degree_centrality
    else:
        centrality = 0.3 * closeness + 0.5 * betweenness + 0.2 * degree_centrality
    centrality = centrality / (centrality.max() + 1e-12)
    
    # Edge centrality score: geometric mean of endpoint centralities
    centrality_score = np.sqrt(
        np.outer(centrality, centrality) + 1e-12
    )
    np.fill_diagonal(centrality_score, 0)
    
    # === 3. PROBABILISTIC REGRET MODELING ===
    # Model edge alternatives with uncertainty using softmax probabilities
    
    temperature = 0.1 * dist_mean  # Adaptive temperature based on scale
    regret_scores = np.zeros((n, n))
    
    # Precompute softmax probabilities for each node's outgoing edges
    prob_matrices = []
    for i in range(n):
        dists = safe_dist[i].copy()
        probs = np.exp(-dists / temperature)
        probs[i] = 0
        probs = probs / (probs.sum() + 1e-12)
        prob_matrices.append(probs)
    
    # For each edge (i,j), compute regret as expected opportunity cost
    for i in range(n):
        for j in range(n):
            if i != j:
                # Probability-weighted regret
                edge_cost = safe_dist[i, j]
                
                # Alternative edges from i (excluding j)
                alt_probs_i = prob_matrices[i].copy()
                alt_probs_i[j] = 0
                alt_probs_i = alt_probs_i / (alt_probs_i.sum() + 1e-12)
                alt_costs_i = safe_dist[i].copy()
                
                # Expected value of best alternative from i
                exp_alt_i = np.sum(alt_probs_i * alt_costs_i)
                
                # Similar for node j
                alt_probs_j = prob_matrices[j].copy()
                alt_probs_j[i] = 0
                alt_probs_j = alt_probs_j / (alt_probs_j.sum() + 1e-12)
                alt_costs_j = safe_dist[j].copy()
                exp_alt_j = np.sum(alt_probs_j * alt_costs_j)
                
                # Regret is average expected opportunity cost
                regret = 0.5 * ((edge_cost - exp_alt_i) + (edge_cost - exp_alt_j))
                regret_scores[i, j] = np.exp(-regret / (dist_mean + 1e-12))
    
    np.fill_diagonal(regret_scores, 0)
    
    # === 4. ENHANCED DENSITY MODELING ===
    # Multi-scale density estimation
    k_density = min(max(3, n // 20), 10)
    densities = np.zeros(n)
    
    for i in range(n):
        # Local density: average of k-nearest distances
        nearest_dists = np.partition(safe_dist[i], k_density)[:k_density]
        local_density = 1 / (np.mean(nearest_dists) + 1e-12)
        
        # Global density: inverse of distance to median point
        median_dist = np.median(safe_dist[i])
        global_density = 1 / (median_dist + 1e-12)
        
        # Combined density
        densities[i] = 0.7 * local_density + 0.3 * global_density
    
    densities = densities / (densities.max() + 1e-12)
    
    # Edge density: harmonic mean emphasizes balanced connections
    density_score = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if densities[i] > 0 and densities[j] > 0:
                h_mean = 2 * densities[i] * densities[j] / (densities[i] + densities[j] + 1e-12)
                density_score[i, j] = density_score[j, i] = h_mean
    
    # === 5. MST STRUCTURAL IMPORTANCE ===
    mst_mask = (mst + mst.T) > 0
    mst_bonus = np.where(mst_mask, 2.0, 1.0)
    
    # === 6. ADAPTIVE COMPONENT INTEGRATION ===
    # Base inverse distance (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / safe_dist
        inv_dist[np.isinf(inv_dist)] = 0
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # Dynamic weight selection based on instance characteristics
    if n <= 20:
        # Small instances: balanced with emphasis on structure
        if is_clustered:
            weights = np.array([0.25, 0.35, 0.20, 0.15, 0.05])  # [inv_dist, mst, centrality, density, regret]
        else:
            weights = np.array([0.30, 0.40, 0.15, 0.10, 0.05])
    elif n <= 100:
        # Medium instances: more structure and centrality
        if is_clustered:
            weights = np.array([0.20, 0.35, 0.25, 0.15, 0.05])
        else:
            weights = np.array([0.25, 0.40, 0.20, 0.10, 0.05])
    else:
        # Large instances: strong structure, sparsification-friendly
        if is_clustered:
            weights = np.array([0.15, 0.45, 0.25, 0.10, 0.05])
        else:
            weights = np.array([0.20, 0.50, 0.20, 0.05, 0.05])
    
    # Combine components
    heuristic = (
        weights[0] * inv_dist +
        weights[1] * inv_dist * mst_bonus +
        weights[2] * centrality_score +
        weights[3] * density_score +
        weights[4] * inv_dist * regret_scores
    )
    
    # === 7. MULTI-SCALE SPARSIFICATION ===
    # Phase 1: Always include MST edges
    promising_mask = mst_mask.copy()
    
    # Phase 2: Include k-nearest neighbors with adaptive k
    base_k = max(2, min(5, n // 15))
    if is_clustered:
        base_k = int(base_k * 1.5)  # More neighbors for clustered instances
    
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:base_k + 1]
        nearest = nearest[nearest != i]
        promising_mask[i, nearest] = True
        promising_mask[nearest, i] = True
    
    # Phase 3: Top heuristic edges per node (adaptive threshold)
    edges_per_node = max(3, min(15, n // 10))
    if n > 30:
        for i in range(n):
            # Get candidate edges from promising_mask
            candidates = np.where(promising_mask[i])[0]
            if len(candidates) > edges_per_node:
                scores = heuristic[i, candidates]
                # Keep top edges_per_node, but ensure we keep MST edges
                mst_candidates = candidates[mst_mask[i, candidates]]
                non_mst_candidates = candidates[~mst_mask[i, candidates]]
                
                if len(mst_candidates) >= edges_per_node:
                    # Keep only MST edges if we have enough
                    promising_mask[i, :] = False
                    promising_mask[i, mst_candidates] = True
                else:
                    # Keep all MST edges plus top non-MST edges
                    remaining_slots = edges_per_node - len(mst_candidates)
                    non_mst_scores = heuristic[i, non_mst_candidates]
                    top_non_mst = np.argsort(non_mst_scores)[-remaining_slots:]
                    top_non_mst_indices = non_mst_candidates[top_non_mst]
                    
                    # Update mask
                    promising_mask[i, :] = False
                    promising_mask[i, mst_candidates] = True
                    promising_mask[i, top_non_mst_indices] = True
                
                # Ensure symmetry
                for j in np.where(promising_mask[i])[0]:
                    promising_mask[j, i] = True
    
    # Phase 4: Minimum connectivity check
    # Ensure each node has at least 2 connections
    for i in range(n):
        connections = np.where(promising_mask[i])[0]
        if len(connections) < 2:
            # Add nearest neighbors to reach minimum connectivity
            additional = np.argsort(safe_dist[i])[1:3]  # Next two closest
            promising_mask[i, additional] = True
            promising_mask[additional, i] = True
    
    # Apply final mask
    heuristic = heuristic * promising_mask
    
    # === 8. NORMALIZATION ===
    # Row-wise normalization to probability distribution
    np.fill_diagonal(heuristic, 0)
    row_sums = heuristic.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristic = heuristic / row_sums
    
    # Softmax with adaptive temperature
    temperature = 0.2 + 0.1 * cv  # Higher temperature for more varied instances
    heuristic = np.exp(heuristic / temperature)
    np.fill_diagonal(heuristic, 0)
    
    # Final normalization
    row_sums = heuristic.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    heuristic = heuristic / row_sums
    
    return heuristic
