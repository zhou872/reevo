import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with three-step lookahead,
    Christofides candidate sets, and adaptive instance-specific weighting.
    
    Key innovations:
    1. Three-step lookahead with pruning: Evaluates deeper sequences while managing complexity
    2. Christofides-inspired candidate sets: Uses MST and matching edges as high-priority candidates
    3. Adaptive instance-specific weighting: Adjusts component weights based on instance characteristics
    4. Global tour structure bonus: Rewards edges that contribute to balanced, non-self-intersecting tours
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Adaptive parameters based on instance size
    k_neighbors = max(3, min(20, int(0.25 * n)))
    candidate_set_size = min(10, n - 1)
    
    # 1. BASE INVERSE DISTANCE (with safe handling)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.diag_indices(n)] = 0
    
    if np.max(inv_dist) > 0:
        inv_dist = inv_dist / np.max(inv_dist)
    
    # 2. CHRISTOFIDES-INSPIRED CANDIDATE SETS
    candidate_bonus = np.ones((n, n))
    
    # Minimum spanning tree
    mst = minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    mst_mask = (mst_array > 0) | (mst_array.T > 0)
    
    # Identify odd-degree vertices in MST
    mst_degrees = np.sum(mst_mask, axis=1)
    odd_vertices = np.where(mst_degrees % 2 == 1)[0]
    
    # For each vertex, create candidate set: k-nearest neighbors
    candidate_sets = {}
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest_indices = np.argsort(dists)[:candidate_set_size]
        candidate_sets[i] = set(nearest_indices)
    
    # Boost edges that are in both MST and candidate sets
    for i in range(n):
        for j in range(i + 1, n):
            if mst_mask[i, j]:
                # Base MST bonus
                bonus = 1.4
                # Additional bonus if both endpoints have odd degree (relevant for perfect matching)
                if i in odd_vertices and j in odd_vertices:
                    bonus *= 1.2
                # Extra bonus if edge is in candidate sets both ways
                if j in candidate_sets[i] and i in candidate_sets[j]:
                    bonus *= 1.3
                candidate_bonus[i, j] = candidate_bonus[j, i] = bonus
    
    # 3. THREE-STEP LOOKAHEAD WITH PRUNING
    lookahead_scores = np.zeros((n, n))
    
    # Precompute top m continuations for each node for efficient three-step evaluation
    m_continuations = min(5, n - 1)
    top_continuations = np.zeros((n, m_continuations, 2), dtype=int)  # (node, continuation_rank, [next_node, next_next_node])
    
    for i in range(n):
        # Get distances and sort for current node
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        
        # For each possible next node j
        for j in range(n):
            if i == j:
                continue
                
            # Pruning: only evaluate three-step for promising edges
            if inv_dist[i, j] < 0.3 * np.max(inv_dist[i]):
                lookahead_scores[i, j] = 1.0 / (distance_matrix[i, j] + 1e-8)
                continue
            
            # Three-step evaluation: i -> j -> k -> l
            best_three_step = float('inf')
            
            # Get top m continuations from j (excluding i)
            dists_j = distance_matrix[j].copy()
            dists_j[i] = np.inf
            dists_j[j] = np.inf
            
            # Sort potential k nodes by distance from j
            sorted_k = np.argsort(dists_j)[:m_continuations]
            
            for k in sorted_k:
                if k == i or k == j:
                    continue
                    
                # Get top continuation from k (excluding i and j)
                dists_k = distance_matrix[k].copy()
                dists_k[i] = np.inf
                dists_k[j] = np.inf
                dists_k[k] = np.inf
                
                # Find best l from k
                l = np.argmin(dists_k)
                
                three_step_cost = distance_matrix[i, j] + distance_matrix[j, k] + distance_matrix[k, l]
                best_three_step = min(best_three_step, three_step_cost)
            
            lookahead_scores[i, j] = 1.0 / (best_three_step + 1e-8)
    
    # Normalize lookahead scores
    max_lookahead = np.max(lookahead_scores)
    if max_lookahead > 0:
        lookahead_scores = lookahead_scores / max_lookahead
    
    # 4. GLOBAL TOUR STRUCTURE BONUS
    structure_bonus = np.ones((n, n))
    
    # Compute convex hull approximation for geometric instances
    if n >= 4:
        # Estimate "exterior-ness" of nodes
        exterior_scores = np.zeros(n)
        
        # For each node, compute average distance to all other nodes
        for i in range(n):
            exterior_scores[i] = np.mean(distance_matrix[i])
        
        # Normalize
        max_ext = np.max(exterior_scores)
        if max_ext > 0:
            exterior_scores = exterior_scores / max_ext
        
        # Bonus for edges connecting exterior nodes (helps form convex hull)
        for i in range(n):
            for j in range(i + 1, n):
                exterior_factor = (exterior_scores[i] + exterior_scores[j]) / 2
                structure_bonus[i, j] = structure_bonus[j, i] = 1.0 + 0.4 * exterior_factor
    
    # 5. ADAPTIVE INSTANCE-SPECIFIC WEIGHTING
    # Analyze instance characteristics
    if n > 10:
        # Compute distance statistics
        flat_dists = distance_matrix[np.triu_indices(n, 1)]
        dist_mean = np.mean(flat_dists)
        dist_std = np.std(flat_dists)
        dist_cv = dist_std / (dist_mean + 1e-8)
        
        # Compute clustering tendency
        nearest_dists = np.zeros(n)
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest_dists[i] = np.min(dists)
        
        clustering_ratio = np.std(nearest_dists) / (np.mean(nearest_dists) + 1e-8)
        
        # Adjust weights based on instance characteristics
        if dist_cv < 0.2:  # Uniform distances
            weights = [0.3, 0.4, 0.2, 0.1]  # [lookahead, candidate, inv_dist, structure]
        elif clustering_ratio > 1.0:  # Clustered instance
            weights = [0.2, 0.5, 0.2, 0.1]
        else:  # Mixed or random instance
            weights = [0.35, 0.35, 0.2, 0.1]
    else:
        weights = [0.3, 0.4, 0.2, 0.1]
    
    # 6. COMBINE COMPONENTS WITH ADAPTIVE WEIGHTS
    combined = (weights[0] * lookahead_scores +
                weights[1] * inv_dist * candidate_bonus +
                weights[2] * inv_dist +
                weights[3] * structure_bonus)
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    # 7. ENHANCED SPARSIFICATION WITH CONNECTIVITY GUARANTEE
    sparsified = combined.copy()
    
    # Keep all MST edges
    sparsified *= mst_mask + 0.3  # MST edges at full strength, others reduced
    
    # Keep candidate set edges
    candidate_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        for j in candidate_sets[i]:
            if i != j:
                candidate_mask[i, j] = True
    candidate_mask = candidate_mask | candidate_mask.T
    
    sparsified[candidate_mask] = combined[candidate_mask]
    
    # Ensure each node connects to at least 3 other nodes
    for i in range(n):
        non_zero = np.where(sparsified[i] > 0)[0]
        non_zero = non_zero[non_zero != i]
        
        if len(non_zero) < 3:
            # Add nearest neighbors by original distance
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:3]
            
            for j in nearest:
                if j != i and sparsified[i, j] == 0:
                    sparsified[i, j] = combined[i, j]
                    sparsified[j, i] = combined[j, i]
    
    # 8. FINAL NORMALIZATION AND CLEANUP
    # Zero out very weak edges (below 10th percentile)
    if np.sum(sparsified > 0) > 0:
        threshold = np.percentile(sparsified[sparsified > 0], 10)
        sparsified[sparsified < threshold] = 0
    
    # Normalize to [0, 1]
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    # Zero diagonal
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
