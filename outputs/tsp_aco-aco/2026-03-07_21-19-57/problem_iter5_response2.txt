import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection focusing on rank-based 
    regret incentives, adaptive density similarity, and intelligent sparsification.
    
    Core improvements over v1:
    1. Rank-based regret incentive: Directly rewards edges based on their rank in 
       sorted distances, giving highest priority to optimal choices
    2. Adaptive density similarity bonus: Uses exponential decay to reward edges 
       connecting nodes with similar local densities, promoting natural cluster formation
    3. Sparse connectivity preservation: Intelligent sparsification that maintains 
       minimum spanning tree backbone while allowing limited promising alternatives
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones((n, n)) - np.eye(n)
    
    # Initialize heuristic matrix with zeros
    heuristic = np.zeros((n, n))
    
    # 1. BASE INVERSE DISTANCE COMPONENT
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.diag_indices(n)] = 0
    
    # 2. RANK-BASED REGRET INCENTIVE (simplified and more direct)
    rank_matrix = np.ones((n, n))
    
    for i in range(n):
        # Get distances from node i to all others
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Sort distances and get ranks (0 = best, n-2 = worst)
        sorted_indices = np.argsort(distances)
        sorted_distances = distances[sorted_indices]
        
        # Best distance from this node
        d_best = sorted_distances[0] if len(sorted_distances) > 0 else 1
        
        # Assign rank-based incentives
        for rank, j in enumerate(sorted_indices):
            if i == j:
                continue
                
            if rank == 0:
                # Best edge gets fixed bonus (1.5 as suggested)
                rank_matrix[i, j] = 1.5
            else:
                # Other edges scaled by distance ratio
                d_rank = sorted_distances[rank]
                if d_best > 0:
                    # Scale factor: 1/(1 + (d_rank - d_best)/d_best) = d_best/d_rank
                    rank_matrix[i, j] = d_best / d_rank
                else:
                    rank_matrix[i, j] = 1.0
    
    # 3. ADAPTIVE DENSITY SIMILARITY BONUS
    density_bonus = np.ones((n, n))
    
    # Compute local density for each node (inverse of average distance to k-nearest neighbors)
    k = max(3, min(10, int(0.1 * n)))  # Adaptive neighborhood size
    local_density = np.zeros(n)
    
    for i in range(n):
        distances = distance_matrix[i, :].copy()
        distances[i] = np.inf
        
        # Find k-nearest neighbors
        if n > k + 1:
            # Partition to get k smallest distances
            kth_dist = np.partition(distances, k)[k]
            nearest_mask = distances <= kth_dist
            nearest_distances = distances[nearest_mask]
        else:
            nearest_distances = distances[distances < np.inf]
        
        if len(nearest_distances) > 0:
            avg_nearest_dist = np.mean(nearest_distances)
            local_density[i] = 1.0 / (avg_nearest_dist + 1e-8)
        else:
            local_density[i] = 1.0
    
    # Normalize densities to [0, 1] range for stability
    if np.max(local_density) > np.min(local_density):
        local_density = (local_density - np.min(local_density)) / (np.max(local_density) - np.min(local_density) + 1e-8)
    
    # Apply exponential similarity bonus
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            density_diff = abs(local_density[i] - local_density[j])
            # Exponential decay bonus: exp(-2 * |density_i - density_j|)
            density_bonus[i, j] = 1.0 + np.exp(-2.0 * density_diff)
    
    # 4. MINIMUM SPANNING TREE STRUCTURAL SUPPORT
    # Keep MST edges as they form the structural backbone
    try:
        mst = minimum_spanning_tree(distance_matrix)
        mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    except:
        # Fallback if MST computation fails
        mst_mask = np.ones((n, n), dtype=bool)
        np.fill_diagonal(mst_mask, False)
    
    mst_support = np.where(mst_mask, 1.2, 1.0)  # 20% boost for MST edges
    
    # 5. COMBINE ALL COMPONENTS
    heuristic = inv_dist * rank_matrix * density_bonus * mst_support
    
    # Ensure symmetry (average both directions)
    heuristic = (heuristic + heuristic.T) / 2.0
    
    # 6. INTELLIGENT SPARSIFICATION
    # Keep only promising edges while ensuring connectivity
    
    # Always keep MST edges
    final_mask = mst_mask.copy()
    
    # For each node, keep a limited number of additional promising edges
    # Keep more edges for smaller graphs, fewer for larger graphs
    if n <= 10:
        extra_edges = 3
    elif n <= 50:
        extra_edges = max(2, int(0.15 * n))
    else:
        extra_edges = max(2, int(0.1 * n))
    
    for i in range(n):
        node_heuristics = heuristic[i, :].copy()
        node_heuristics[i] = -np.inf  # Exclude self
        
        # Get indices of top extra_edges + MST edges
        # First ensure we have MST edges
        mst_edges = np.where(mst_mask[i, :])[0]
        mst_edges = mst_edges[mst_edges != i]
        
        # Then add top additional edges
        non_mst_mask = ~mst_mask[i, :]
        non_mst_mask[i] = False  # Exclude self
        non_mst_indices = np.where(non_mst_mask)[0]
        
        if len(non_mst_indices) > 0:
            non_mst_heuristics = node_heuristics[non_mst_indices]
            if len(non_mst_heuristics) > extra_edges:
                # Get top extra_edges from non-MST edges
                top_non_mst_idx = np.argpartition(non_mst_heuristics, -extra_edges)[-extra_edges:]
                top_non_mst = non_mst_indices[top_non_mst_idx]
            else:
                top_non_mst = non_mst_indices
            
            # Combine MST and top non-MST edges
            keep_indices = np.concatenate([mst_edges, top_non_mst])
        else:
            keep_indices = mst_edges
        
        final_mask[i, keep_indices] = True
        final_mask[keep_indices, i] = True
    
    # Ensure minimum connectivity: each node must connect to at least 2 others
    for i in range(n):
        connected = np.where(final_mask[i, :])[0]
        connected = connected[connected != i]
        if len(connected) < 2:
            # Add nearest neighbors to meet minimum connectivity
            distances = distance_matrix[i, :].copy()
            distances[i] = np.inf
            # Find 2 - len(connected) nearest not already connected
            candidates = np.argsort(distances)
            added = 0
            for j in candidates:
                if j != i and not final_mask[i, j]:
                    final_mask[i, j] = True
                    final_mask[j, i] = True
                    added += 1
                    if added >= 2 - len(connected):
                        break
    
    # Apply sparsification mask
    heuristic = heuristic * final_mask
    
    # Normalize to [0, 1] range for stability
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    # Set diagonal to zero
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
