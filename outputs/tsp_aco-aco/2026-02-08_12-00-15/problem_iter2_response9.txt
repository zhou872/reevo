import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP focusing on:
    1. MST edges as essential backbone
    2. Nearest neighbor connectivity
    3. Per-node sparsification for computational efficiency
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n) with zeros on diagonal
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, with higher values indicating
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    
    # 1. Baseline: inverse distance (closer nodes are more promising)
    inv_dist = 1 / np.where(distance_matrix > 0, distance_matrix, np.inf)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree edges get highest priority
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst + mst.T) > 0  # Symmetrize
    
    # 3. Nearest neighbor connections
    k = max(2, min(n // 4, 10))  # Adaptive k based on problem size
    nn_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nearest_indices = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, nearest_indices] = True
    
    # Symmetrize: if i is in j's nearest neighbors or vice versa
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Combine factors with priorities
    # MST edges get highest weight, then nearest neighbors, then other edges
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # First, assign baseline inverse distance to all edges
    heuristics = inv_dist.copy()
    
    # Boost MST edges significantly
    mst_boost = 3.0
    heuristics[mst_edges] *= mst_boost
    
    # Give moderate boost to nearest neighbor edges (if not already MST)
    nn_boost = 2.0
    nn_only = nn_mask & ~mst_edges
    heuristics[nn_only] *= nn_boost
    
    # 5. Per-node sparsification: keep only most promising edges for each node
    edges_per_node = max(2, min(n // 2, 15))  # Adaptive edge count
    
    for i in range(n):
        # Get the top 'edges_per_node' edges for this node
        row_values = heuristics[i].copy()
        row_values[i] = -np.inf  # Exclude self
        
        # Get indices of top edges
        if edges_per_node < n - 1:
            top_indices = np.argsort(row_values)[-edges_per_node:]
            # Create mask for this node's top edges
            row_mask = np.zeros(n, dtype=bool)
            row_mask[top_indices] = True
            
            # Keep only top edges for this node
            heuristics[i] *= row_mask
    
    # Ensure symmetry by keeping edge if it's in top for either node
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 6. Final processing
    # Remove numerical noise
    heuristics[heuristics < 1e-10] = 0
    
    # Normalize for stability (optional, but helpful for some applications)
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
