import numpy as np
from scipy.spatial.distance import pdist, squareform
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP that combines multiple factors to assess edge promise.
    Returns a matrix of the same shape as input with higher values for more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base heuristic: reciprocal of distance (like v1)
    with np.errstate(divide='ignore', invalid='ignore'):
        base_heuristic = np.where(distance_matrix > 0, 1 / distance_matrix, 0)
    
    # 2. Nearest neighbor factor: edges to/from nearest neighbors get bonus
    # For each node, find k nearest neighbors
    k = min(5, n // 2)  # adaptive k based on problem size
    nearest_neighbor_factor = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        nearest_indices = np.argsort(distance_matrix[i])[1:k+1]
        nearest_neighbor_factor[i, nearest_indices] = 1.0
    
    # Symmetrize the nearest neighbor factor
    nearest_neighbor_factor = (nearest_neighbor_factor + nearest_neighbor_factor.T) / 2
    
    # 3. Minimum Spanning Tree (MST) factor: edges in MST are promising
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_factor = (mst + mst.T) > 0  # Convert to symmetric boolean
    
    # 4. Triangle inequality factor: edges that complete shorter triangles
    triangle_factor = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            # Check if direct edge (i,j) is shorter than going through any intermediate node k
            min_detour = np.min(distance_matrix[i] + distance_matrix[:, j] - distance_matrix[i, j])
            # Normalize to [0, 1] range, negative values mean direct edge is very competitive
            triangle_factor[i, j] = triangle_factor[j, i] = max(0, 1 - min_detour / distance_matrix[i, j])
    
    # 5. Degree preference: edges from nodes with few good alternatives
    degree_factor = np.zeros_like(distance_matrix)
    for i in range(n):
        # Calculate how much worse alternatives are compared to best edge
        sorted_dists = np.sort(distance_matrix[i])
        if len(sorted_dists) > 1:
            ratio = sorted_dists[1] / sorted_dists[2] if len(sorted_dists) > 2 else 1.0
            # If second best is much worse than best, best edge gets bonus
            for j in range(n):
                if i != j and distance_matrix[i, j] == sorted_dists[1]:
                    degree_factor[i, j] = 1 / ratio
    
    # 6. Local clustering factor: edges within dense clusters
    # Simple clustering: nodes with mutually short distances
    cluster_factor = np.zeros_like(distance_matrix)
    avg_dist = np.mean(distance_matrix)
    
    for i in range(n):
        for j in range(i+1, n):
            # Count how many other nodes are close to both i and j
            close_to_both = np.sum((distance_matrix[i] < avg_dist) & 
                                  (distance_matrix[j] < avg_dist))
            cluster_factor[i, j] = cluster_factor[j, i] = close_to_both / n
    
    # Combine all factors with weights
    combined = (
        2.0 * base_heuristic +                    # Base reciprocal distance
        1.5 * nearest_neighbor_factor +           # Nearest neighbor bonus
        2.0 * mst_factor +                        # MST edges
        1.0 * triangle_factor +                   # Triangle inequality
        0.5 * degree_factor +                     # Degree preference
        0.5 * cluster_factor                      # Local clustering
    )
    
    # Sparsification: keep only top edges for each node
    sparsified = np.zeros_like(combined)
    keep_per_node = max(3, n // 4)  # Keep more edges for smaller instances
    
    for i in range(n):
        # Get indices of top edges for node i
        top_indices = np.argsort(combined[i])[-keep_per_node:]
        sparsified[i, top_indices] = combined[i, top_indices]
        
        # Also keep symmetric counterpart
        for j in top_indices:
            sparsified[j, i] = combined[j, i]
    
    # Ensure symmetry (in case of numerical issues)
    sparsified = (sparsified + sparsified.T) / 2
    
    # Normalize to [0, 1] range for consistency
    if np.max(sparsified) > 0:
        sparsified = sparsified / np.max(sparsified)
    
    return sparsified
