import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristics combining multiple factors multiplicatively,
    with vectorized triangle inequality penalties and logarithmic boosting.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    inv_dist = 1.0 / safe_dist
    np.fill_diagonal(inv_dist, 0.0)
    
    # 2. Minimum Spanning Tree factor
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_factor = np.where(mst_mask, 2.0, 1.0)
    
    # 3. Nearest neighbor factor (adaptive k)
    k = max(2, min(8, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[1:k+1]  # exclude self
        nn_mask[i, nearest] = True
    nn_factor = np.where(nn_mask | nn_mask.T, 1.5, 1.0)
    
    # 4. Triangle inequality penalty (vectorized with sampling)
    penalty = np.ones_like(distance_matrix)
    if n > 3:
        # Sample intermediate nodes based on graph size
        sample_size = min(n, max(20, n // 5))
        k_samples = np.random.choice(n, size=sample_size, replace=False)
        
        # Vectorized computation: for each k, check violations
        for k in k_samples:
            with np.errstate(invalid='ignore'):
                detour_sum = safe_dist[:, k:k+1] + safe_dist[k:k+1, :]
            # Violation when direct path is significantly longer than detour
            violations = safe_dist > detour_sum * 1.15
            penalty[violations] *= 0.85
    
    # 5. Multiplicative combination
    heuristics = inv_dist * mst_factor * nn_factor * penalty
    np.fill_diagonal(heuristics, 0.0)
    
    # 6. Adaptive sparsification
    edges_per_node = max(3, min(15, int(np.sqrt(n))))
    keep_mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf
        # Keep top edges per node
        top_idx = np.argpartition(row, -edges_per_node)[-edges_per_node:]
        keep_mask[i, top_idx] = True
        # Always keep MST edges
        keep_mask[i, mst_mask[i]] = True
    
    keep_mask = keep_mask | keep_mask.T
    heuristics = heuristics * keep_mask
    
    # 7. Logarithmic boosting of top edges
    if np.sum(heuristics > 0) > 0:
        # Identify top percentile edges globally
        non_zero = heuristics[heuristics > 0]
        if len(non_zero) > 0:
            threshold = np.percentile(non_zero, 85)
            top_mask = heuristics >= threshold
            # Apply logarithmic boost to top edges
            boost_factor = np.log1p(heuristics[top_mask])
            heuristics[top_mask] *= (1.0 + boost_factor / 4.0)
    
    # 8. Final processing
    heuristics = (heuristics + heuristics.T) / 2.0  # Ensure symmetry
    np.fill_diagonal(heuristics, 0.0)
    
    # Normalize to [0, 1] range
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
