```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    TSP edge selection heuristics focusing on Minimum Spanning Tree and 
    nearest neighbors with adaptive sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Compute base heuristic: inverse distance
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree mask
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbor mask with adaptive k
    k = max(2, min(int(np.log2(n)) + 2, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Vectorized nearest neighbors
    sorted_indices = np.argsort(safe_dist, axis=1)
    for i in range(n):
        nearest = sorted_indices[i, 1:k+1]  # Exclude self
        nn_mask[i, nearest] = True
    
    # Symmetrize
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Combine with bonuses
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= 2.0  # Strong MST bonus
    heuristics[nn_mask] *= 1.5   # Moderate NN bonus
    
    # 5. Adaptive sparsification per node
    keep_mask = mst_mask | nn_mask  # Always keep MST and NN edges
    
    for i in range(n):
        # Get candidate edges not already kept
        candidates = np.ones(n, dtype=bool)
        candidates[i] = False
        candidates[keep_mask[i]] = False
        
        if np.any(candidates):
            # Keep top edges based on heuristic value
            row_vals = heuristics[i, candidates]
            if len(row_vals) > 0:
                # Adaptive threshold based on problem size
                keep_count = max(1, int(np.log2(n)))
                threshold = np.sort(row_vals)[-keep_count]
                top_indices = np.where(candidates)[0][heuristics[i, candidates] >= threshold]
                keep_mask[i, top_indices] = True
    
    # Symmetrize and apply mask
    keep_mask = keep_mask | keep_mask.T
    heuristics = heuristics * keep_mask
    
    # 6. Normalize and finalize
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
```
