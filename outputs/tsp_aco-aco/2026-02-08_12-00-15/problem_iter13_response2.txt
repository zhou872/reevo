import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristics for TSP edge selection.
    Combines inverse distance, MST inclusion, nearest neighbors with overlap bonuses,
    adaptive logarithmic sparsification, and triangle sampling for connectivity.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 2:
        return np.ones_like(distance_matrix) if n == 2 else np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 3.0  # Strong emphasis on MST edges
    
    # 3. Nearest neighbor bonus with logarithmic scaling
    k = max(2, int(np.log2(n)) + 1)  # Logarithmic scaling of neighbors
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]  # +1 to skip self (inf)
        nearest = nearest[nearest != i]
        nn_mask[i, nearest[:min(k, len(nearest))]] = True
    nn_mask = nn_mask | nn_mask.T  # Symmetrize
    nn_bonus = 2.0
    
    # 4. Overlap bonus: edges that are both in MST and nearest neighbors
    overlap_mask = mst_mask & nn_mask
    overlap_bonus = 4.0  # Strong bonus for overlap
    
    # 5. Triangle sampling bonus for connectivity
    # For each edge (i,j), check if there exists k such that both (i,k) and (j,k) are promising
    triangle_bonus = np.ones_like(base_heuristic)
    if n >= 3:
        # Use base heuristic to identify promising edges for triangle completion
        base_promising = base_heuristic > np.percentile(base_heuristic[base_heuristic > 0], 50)
        
        for i in range(n):
            for j in range(i+1, n):
                # Find common neighbors with promising connections
                common_neighbors = np.where(base_promising[i] & base_promising[j])[0]
                common_neighbors = common_neighbors[(common_neighbors != i) & (common_neighbors != j)]
                
                if len(common_neighbors) > 0:
                    # Bonus proportional to number of connecting triangles
                    triangle_bonus[i, j] = triangle_bonus[j, i] = 1.0 + 0.1 * min(len(common_neighbors), 5)
    
    # 6. Combine all factors multiplicatively
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    heuristics[overlap_mask] *= overlap_bonus
    heuristics *= triangle_bonus
    
    # 7. Adaptive logarithmic sparsification
    # Calculate adaptive percentile: higher for smaller graphs, lower for larger ones
    adaptive_percentile = 100 - 20 * np.log10(max(n, 10))  # Logarithmic scaling
    adaptive_percentile = max(30, min(80, adaptive_percentile))  # Clamp to reasonable range
    
    # Keep edges that are in MST, nearest neighbors, or above adaptive percentile
    combined_mask = mst_mask | nn_mask
    
    for i in range(n):
        row_values = heuristics[i]
        valid_values = row_values[row_values > 0]
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, adaptive_percentile)
            combined_mask[i] = combined_mask[i] | (row_values >= threshold)
    
    # Apply the combined mask
    heuristics = heuristics * combined_mask
    
    # 8. Symmetrize and normalize
    heuristics = np.maximum(heuristics, heuristics.T)  # Ensure symmetry
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
