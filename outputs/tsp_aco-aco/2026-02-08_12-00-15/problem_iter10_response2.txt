import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristics for TSP edge selection focusing on core principles:
    Minimum Spanning Tree, nearest neighbors, and consistent sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.0
    
    # 3. Nearest neighbor bonus with adaptive k
    k = max(2, min(5, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i]
        nn_mask[i, nearest[:min(k, len(nearest))]] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.5
    
    # 4. Combine core heuristics
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    
    # 5. Consistent sparsification
    # Keep edges that are in MST, nearest neighbors, or have high base value
    combined_mask = mst_mask | nn_mask
    
    # Global percentile threshold for additional promising edges
    if n > 3:
        # Extract upper triangular values (excluding diagonal)
        upper_tri = heuristics[np.triu_indices(n, k=1)]
        if len(upper_tri) > 0:
            threshold = np.percentile(upper_tri, 75)
            above_threshold = heuristics >= threshold
            combined_mask = combined_mask | above_threshold
    
    # Apply the combined mask and ensure symmetry
    heuristics = heuristics * combined_mask
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 6. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
