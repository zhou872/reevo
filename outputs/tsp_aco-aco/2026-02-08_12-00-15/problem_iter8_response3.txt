```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Refined TSP heuristics with strong MST emphasis, adaptive edge selection,
    and robust scaling for stochastic solution sampling.
    
    Key features:
    1. Very strong MST edge boosting (factor 8.0)
    2. Adaptive logarithmic scaling for MST edges based on problem size
    3. Progressive edge retention: keep more edges for smaller problems
    4. Simple symmetric finalization using maximum operator
    5. Clean, streamlined implementation
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n) with zeros on diagonal
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix with same shape as input, where higher values
        indicate more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Baseline: inverse distance
    safe_dist = np.where(distance_matrix > 0, distance_matrix, np.inf)
    heuristics = 1.0 / safe_dist
    np.fill_diagonal(heuristics, 0)
    
    # 2. Strong MST emphasis with logarithmic scaling
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst + mst.T) > 0
    
    # Boost factor grows logarithmically with problem size
    mst_boost = 8.0 * (1 + np.log1p(n) / 15)
    heuristics[mst_edges] *= mst_boost
    
    # 3. Adaptive edge retention based on problem size
    # More edges for smaller instances, fewer for larger ones
    if n < 20:
        edges_per_node = min(n - 1, 8)
    elif n < 50:
        edges_per_node = min(n - 1, 6)
    elif n < 100:
        edges_per_node = min(n - 1, 5)
    else:
        edges_per_node = min(n - 1, 4)
    
    # 4. Keep only top edges for each node
    mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Exclude self
        
        if edges_per_node < n - 1:
            # Efficient selection of top edges
            top_indices = np.argpartition(-row, edges_per_node)[:edges_per_node]
            mask[i, top_indices] = True
        else:
            # Keep all edges (for very small instances)
            mask[i, :] = True
            mask[i, i] = False
    
    # 5. Symmetrize mask and apply
    mask = mask | mask.T
    heuristics = heuristics * mask
    
    # 6. Final symmetrization and normalization
    heuristics = np.maximum(heuristics, heuristics.T)
    
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
```
