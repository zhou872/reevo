import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge selection heuristics with triangle inequality penalties,
    adaptive sparsification, and logarithmic scaling of top edges.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create safe distance matrix with diagonal set to infinity
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base heuristic: inverse distance (handle zeros)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree mask
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbor mask with adaptive sparsification
    # Adaptive k based on problem size (logarithmic scaling)
    k = max(2, min(int(np.log2(n)) + 3, n // 3))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Vectorized nearest neighbor computation
    sorted_indices = np.argsort(safe_dist, axis=1)
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nearest = sorted_indices[i, :k+1]
        nearest = nearest[nearest != i]
        nn_mask[i, nearest[:min(k, len(nearest))]] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Triangle inequality penalty
    # Vectorized check for triangle inequality violations
    tri_mask = np.ones_like(distance_matrix, dtype=float)
    
    if n > 2:
        # For each pair (i,j), check if there exists k such that
        # dist[i,j] > dist[i,k] + dist[k,j] (triangle inequality violation)
        for k in range(n):
            # Compute dist[i,k] + dist[k,j] for all i,j
            detour = distance_matrix[:, k:k+1] + distance_matrix[k:k+1, :]
            # Mark violations where direct path is longer than detour
            violations = distance_matrix > detour
            # Apply penalty to violating edges
            tri_mask[violations] *= 0.5  # 50% penalty for violations
    
    # 5. Combine MST and nearest neighbor masks (multiplication as per reflection)
    mask_multiplier = np.ones_like(distance_matrix)
    mask_multiplier[mst_mask] *= 2.0  # MST edges get 2x boost
    mask_multiplier[nn_mask] *= 1.5   # NN edges get 1.5x boost
    
    # 6. Initial combination
    heuristics = base_heuristic * mask_multiplier * tri_mask
    
    # 7. Adaptive sparsification per node
    # Keep edges based on node-specific thresholds
    keep_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get non-zero, non-self heuristic values for this node
        row_vals = heuristics[i].copy()
        row_vals[i] = 0
        non_zero = row_vals[row_vals > 0]
        
        if len(non_zero) > 0:
            # Adaptive threshold based on problem size
            # For smaller problems, keep more edges
            keep_fraction = min(0.3 + 0.1 * np.log(n), 0.7)
            threshold = np.percentile(non_zero, 100 * (1 - keep_fraction))
            
            # Always keep MST and nearest neighbor edges
            keep_mask[i] = (row_vals >= threshold) | mst_mask[i] | nn_mask[i]
    
    # Symmetrize the keep mask
    keep_mask = keep_mask | keep_mask.T
    heuristics = heuristics * keep_mask
    
    # 8. Logarithmic scaling for top edges
    # Apply logarithmic boost to top quartile of edges
    all_vals = heuristics[heuristics > 0]
    if len(all_vals) > 0:
        top_threshold = np.percentile(all_vals, 75)
        top_mask = heuristics >= top_threshold
        # Apply logarithmic boost: log(1 + value) amplifies high values less than linear
        boost_factor = np.log1p(heuristics[top_mask])
        heuristics[top_mask] *= (1.0 + 0.5 * boost_factor / np.max(boost_factor))
    
    # 9. Ensure symmetry (due to potential floating point differences)
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 10. Normalize to [0, 1] range
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Final sparsification: zero out very small values
    heuristics[heuristics < 1e-6] = 0
    
    return heuristics
