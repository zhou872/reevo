```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Simple and effective TSP edge selection heuristics.
    Combines inverse distance with MST and nearest neighbors,
    then applies clear threshold-based sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    heuristics = 1.0 / safe_dist
    np.fill_diagonal(heuristics, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    heuristics[mst_mask] *= 2.0  # Double value for MST edges
    
    # 3. Nearest neighbor bonus
    k = min(5, n // 2)  # Adaptive k, max 5
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k+1 smallest distances (including self at inf)
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i]  # Remove self
        if len(nearest) > 0:
            nn_mask[i, nearest[:k]] = True
    
    nn_mask = nn_mask | nn_mask.T  # Symmetrize
    heuristics[nn_mask] *= 1.5  # Boost nearest neighbor edges
    
    # 4. Sparsification with clear threshold
    # Keep edges above the median of non-zero values
    non_zero_vals = heuristics[heuristics > 0]
    if len(non_zero_vals) > 0:
        threshold = np.percentile(non_zero_vals, 50)  # Median threshold
        keep_mask = heuristics >= threshold
        keep_mask = keep_mask | mst_mask | nn_mask  # Always keep MST and NN edges
        heuristics = heuristics * keep_mask
    
    # 5. Normalize to [0, 1] range
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
```
