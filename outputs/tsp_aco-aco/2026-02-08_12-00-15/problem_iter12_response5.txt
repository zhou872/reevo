import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge selection heuristics with simplified adaptive boosts.
    Prioritizes MST edges and nearest neighbors with clean, effective normalization.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (handle zero/self distances)
    safe_dist = distance_matrix.astype(float).copy()
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree - essential connectivity
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.5  # MST edges get highest priority
    
    # 3. Adaptive nearest neighbors
    # Scale k with problem size: log2(n) ensures logarithmic growth
    k = max(2, int(np.ceil(np.log2(n))))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k+1 smallest distances (including inf diagonal)
        nearest_indices = np.argsort(safe_dist[i])[:k+1]
        # Remove self and keep up to k neighbors
        nearest_indices = nearest_indices[nearest_indices != i][:k]
        nn_mask[i, nearest_indices] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.8  # NN edges get moderate priority
    
    # 4. Combine bonuses multiplicatively
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    
    # 5. Simple sparsification: keep only top edges per node
    # Include MST, NN, and top 30% of remaining edges per node
    combined_mask = mst_mask | nn_mask
    
    for i in range(n):
        row_values = heuristics[i].copy()
        # Exclude already masked edges and self from consideration
        candidate_mask = ~combined_mask[i]
        candidate_mask[i] = False
        
        if np.any(candidate_mask):
            candidate_values = row_values[candidate_mask]
            if len(candidate_values) > 0:
                threshold = np.percentile(candidate_values, 70)  # Top 30%
                high_value_mask = np.zeros(n, dtype=bool)
                high_value_mask[candidate_mask] = candidate_values >= threshold
                combined_mask[i] = combined_mask[i] | high_value_mask
    
    # 6. Apply symmetric mask
    combined_mask = combined_mask | combined_mask.T
    heuristics = heuristics * combined_mask
    
    # 7. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    np.fill_diagonal(heuristics, 0.0)
    return heuristics
