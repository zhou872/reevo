import numpy as np
from scipy.sparse.csgraph import minimum_ spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristics focusing on MST, nearest neighbors, 
    adaptive sparsification, and normalization.
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create working matrix with diagonal set to infinity
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base heuristic: reciprocal of distance
    with np.errstate(divide='ignore', invalid='ignore'):
        base_heuristic = 1.0 / safe_dist
    base_heuristic[np.isinf(base_heuristic)] = 0
    base_heuristic[np.isnan(base_heuristic)] = 0
    
    # 2. MST boost - edges in minimum spanning tree get strong bonus
    if n > 1:
        mst = minimum_spanning_tree(safe_dist).toarray()
        mst_mask = (mst + mst.T) > 0
        mst_boost = 2.0  # Strong boost for MST edges
    else:
        mst_mask = np.zeros_like(base_heuristic, dtype=bool)
        mst_boost = 1.0
    
    # 3. Nearest neighbors boost - focus on local connections
    k_nn = max(2, int(np.log2(n + 1)))  # Adaptive based on problem size
    nn_mask = np.zeros_like(safe_dist, dtype=bool)
    
    for i in range(n):
        # Get k_nn nearest neighbors (excluding self)
        nearest = np.argsort(safe_dist[i])[:k_nn + 1]
        nearest = nearest[nearest != i]  # Remove self
        if len(nearest) > k_nn:
            nearest = nearest[:k_nn]
        nn_mask[i, nearest] = True
    
    # Make symmetric
    nn_mask = nn_mask | nn_mask.T
    nn_boost = 1.5
    
    # 4. Apply MST and NN boosts multiplicatively
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_boost
    heuristics[nn_mask] *= nn_boost
    
    # 5. Adaptive sparsification - keep only the most promising edges
    # Per-node adaptive thresholding
    threshold_mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        row = heuristics[i]
        # Keep top-k edges per node, k adapts with problem size
        k_keep = max(3, int(np.sqrt(n)))
        
        if np.sum(row > 0) > k_keep:
            # Get indices of top k_keep edges
            top_indices = np.argsort(row)[-k_keep:]
            threshold_mask[i, top_indices] = True
        else:
            # Keep all positive edges
            threshold_mask[i] = row > 0
    
    # Make symmetric (if i considers j, j should consider i)
    threshold_mask = threshold_mask | threshold_mask.T
    
    # 6. Degree balancing - ensure each node has at least 2 promising edges
    for i in range(n):
        row_mask = threshold_mask[i]
        if np.sum(row_mask) < 2:
            # Add the next best edges if node has too few connections
            row = heuristics[i]
            row[i] = -np.inf  # Exclude self
            top_indices = np.argsort(row)[-2:]  # Get top 2
            threshold_mask[i, top_indices] = True
            # Symmetrize
            for j in top_indices:
                threshold_mask[j, i] = True
    
    # Apply sparsification mask
    heuristics = heuristics * threshold_mask
    
    # 7. Final normalization to [0, 1] range
    non_zero = heuristics[heuristics > 0]
    if len(non_zero) > 0:
        max_val = np.max(non_zero)
        if max_val > 0:
            heuristics = heuristics / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
