import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristics with triangle inequality penalties,
    adaptive sparsification, logarithmic edge boosting, and
    efficient vectorized operations.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with safe handling
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree mask (vectorized)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbor mask with adaptive k
    k = max(2, min(8, int(np.sqrt(n))))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Vectorized nearest neighbor computation
    sorted_indices = np.argsort(safe_dist, axis=1)
    for i in range(n):
        valid_indices = sorted_indices[i][sorted_indices[i] != i][:k]
        nn_mask[i, valid_indices] = True
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Multiply MST and nearest neighbor masks
    combined_mask = mst_mask & nn_mask
    
    # 5. Triangle inequality violation penalty
    penalty = np.ones_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            # Find the minimum detour distance through any third node
            mask = np.ones(n, dtype=bool)
            mask[[i, j]] = False
            if np.any(mask):
                min_detour = np.min(safe_dist[i, mask] + safe_dist[mask, j])
                if min_detour < safe_dist[i, j]:
                    # Penalize violation proportionally
                    penalty[i, j] = penalty[j, i] = 0.5 + 0.5 * (min_detour / safe_dist[i, j])
    
    # 6. Apply masks and penalties
    heuristics = base_heuristic.copy()
    heuristics[combined_mask] *= 2.5  # Strong bonus for edges in both masks
    heuristics[mst_mask & ~combined_mask] *= 2.0  # Bonus for MST only
    heuristics[nn_mask & ~combined_mask] *= 1.5  # Bonus for NN only
    heuristics *= penalty  # Apply triangle inequality penalties
    
    # 7. Adaptive sparsification based on problem size
    keep_factor = min(0.4, max(0.15, 20/n))
    sparsified = np.zeros_like(heuristics)
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Exclude diagonal
        n_keep = max(2, int(keep_factor * n))
        if n_keep < len(row):
            threshold = np.partition(row, -n_keep)[-n_keep]
            mask = row >= threshold
            sparsified[i, mask] = row[mask]
    
    # Ensure symmetry after sparsification
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 8. Logarithmic boosting of top edges per node
    for i in range(n):
        non_zero = sparsified[i].nonzero()[0]
        if len(non_zero) > 1:
            values = sparsified[i, non_zero]
            sorted_idx = np.argsort(values)[::-1]
            
            # Apply logarithmic boost: top edges get highest boost
            ranks = np.arange(len(sorted_idx)) + 1
            boost_factors = 1.0 + np.log1p(len(sorted_idx) - ranks) / np.log1p(len(sorted_idx))
            
            # Reorder boost factors to match original order
            boost_reordered = np.zeros_like(values)
            boost_reordered[sorted_idx] = boost_factors
            sparsified[i, non_zero] *= boost_reordered
    
    # 9. Final symmetry enforcement and normalization
    heuristics_final = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(heuristics_final, 0.0)
    
    # Normalize only if there are non-zero values
    max_val = heuristics_final.max()
    if max_val > 0:
        heuristics_final = heuristics_final / max_val
    
    return heuristics_final
