import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge selection heuristics with size-adaptive sparsification,
    logarithmic neighbor scaling, MST-neighbor overlap boosting, and triangle
    connectivity enhancement for stochastic sampling approaches.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with safe handling
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree computation
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Logarithmic neighbor scaling with adaptive k
    # k scales logarithmically with problem size, with bounds
    base_k = max(2, int(np.log2(n)))
    k = min(base_k, 8)  # Upper bound for computational efficiency
    
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get distances excluding self
        distances = safe_dist[i]
        sorted_indices = np.argsort(distances)[:k*2]  # Consider 2k nearest
        
        # Logarithmic weighting: closer neighbors get higher weights
        weights = 1.0 / np.log2(np.arange(2, len(sorted_indices) + 2))
        
        # Weighted selection of k neighbors
        if k < len(sorted_indices):
            # Use weighted random selection for diversity
            probs = weights / weights.sum()
            selected = np.random.choice(sorted_indices, size=k, 
                                        replace=False, p=probs)
        else:
            selected = sorted_indices
        
        nn_mask[i, selected] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    
    # 4. MST-neighbor overlap boosting
    overlap_mask = mst_mask & nn_mask
    # Stronger boost for smaller problems, milder for larger
    overlap_bonus = 2.5 + (50 / n)  # Ranges from ~3.5 for n=20 to 2.5 for n=100
    
    # 5. Enhanced triangle connectivity sampling
    triangle_bonus = np.zeros_like(distance_matrix)
    
    if n >= 3:
        # Adaptive sampling: more samples for smaller problems
        sample_size = min(n, max(15, int(n**0.7)))
        sample_indices = np.random.choice(n, sample_size, replace=False)
        
        for i in sample_indices:
            # Get promising neighbors (top 30% by base heuristic)
            neighbor_heuristics = base_heuristic[i]
            threshold = np.percentile(neighbor_heuristics[neighbor_heuristics > 0], 70)
            promising_neighbors = np.where((neighbor_heuristics >= threshold) & 
                                          (np.arange(n) != i))[0]
            
            # Check triangles among promising neighbors
            for j_idx, j in enumerate(promising_neighbors):
                for k in promising_neighbors[j_idx + 1:]:
                    # Boost edge (j,k) if both (i,j) and (i,k) are promising
                    # Boost strength depends on geometric mean of edge qualities
                    edge_quality = np.sqrt(base_heuristic[i, j] * base_heuristic[i, k])
                    boost = 0.7 * edge_quality
                    triangle_bonus[j, k] += boost
                    triangle_bonus[k, j] += boost
    
    # 6. Combine all heuristic factors
    heuristics = base_heuristic.copy()
    
    # MST edges get size-dependent bonus
    mst_bonus = 2.0 + (30 / n)  # From 2.3 for n=100 to 3.5 for n=20
    heuristics[mst_mask] *= mst_bonus
    
    # Nearest neighbors get logarithmic scaling bonus
    nn_bonus = 1.4 + 0.25 * np.log2(n)
    heuristics[nn_mask] *= nn_bonus
    
    # Apply overlap and triangle bonuses
    heuristics[overlap_mask] *= overlap_bonus
    heuristics += triangle_bonus
    
    # 7. Size-adaptive sparsification
    # Determine sparsity parameters based on problem size
    if n <= 15:
        keep_percentile = 65  # Keep more edges for small problems
        min_edges = min(4, n-1)
    elif n <= 40:
        keep_percentile = 75
        min_edges = min(5, n-1)
    elif n <= 80:
        keep_percentile = 82
        min_edges = min(6, n-1)
    else:
        keep_percentile = 88
        min_edges = min(7, n-1)
    
    # Create sparsity mask
    sparsity_mask = np.zeros_like(heuristics, dtype=bool)
    
    # Always keep MST edges for connectivity
    sparsity_mask |= mst_mask
    
    # For each node, keep top edges based on adaptive threshold
    for i in range(n):
        row_heuristics = heuristics[i]
        # Consider only non-MST edges for additional selection
        non_mst_mask = ~mst_mask[i] & (np.arange(n) != i)
        non_mst_values = row_heuristics[non_mst_mask]
        
        if len(non_mst_values) > 0:
            # Calculate threshold based on percentile
            threshold = np.percentile(non_mst_values, keep_percentile)
            
            # Select edges above threshold
            above_threshold = non_mst_values >= threshold
            non_mst_indices = np.where(non_mst_mask)[0]
            selected_indices = non_mst_indices[above_threshold]
            
            # Ensure minimum number of edges per node
            current_edges = np.sum(sparsity_mask[i])
            if current_edges + len(selected_indices) < min_edges:
                # Add additional top edges to reach minimum
                needed = min_edges - current_edges
                if needed > 0:
                    # Get top edges not already selected
                    remaining_mask = non_mst_mask & ~sparsity_mask[i]
                    remaining_values = row_heuristics[remaining_mask]
                    if len(remaining_values) > 0:
                        remaining_indices = np.where(remaining_mask)[0]
                        top_additional = np.argsort(remaining_values)[-needed:]
                        selected_indices = np.concatenate([
                            selected_indices, 
                            remaining_indices[top_additional]
                        ])
            
            sparsity_mask[i, selected_indices] = True
    
    # Ensure symmetry of sparsity mask
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsity mask to heuristics
    heuristics = heuristics * sparsity_mask
    
    # 8. Enhanced normalization with size-dependent scaling
    max_val = heuristics.max()
    if max_val > 0:
        # Scale factor adapts with problem size
        scale_factor = 3.0 + (n / 50)  # From 3.2 for n=10 to 5.0 for n=100
        scaled = heuristics / max_val
        # Use power scaling to amplify differences between promising edges
        heuristics = np.power(scaled, 1.2)
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0.0)
    
    return heuristics
