import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge selection heuristics with:
    1. MST and nearest neighbor prioritization with overlap bonus
    2. Adaptive logarithmic percentile sparsification
    3. Triangle sampling for connectivity
    4. Improved efficiency and simplicity
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances
    
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with regularization
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    with np.errstate(divide='ignore'):
        base_heuristic = 1.0 / safe_dist
    base_heuristic[np.isinf(base_heuristic)] = 0
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree inclusion
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_weight = 2.5  # MST edges are highly promising
    
    # 3. Nearest neighbors with adaptive k
    k = max(2, int(np.log2(n + 1)))  # Logarithmic scaling
    k = min(k, n - 1)
    
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i][:k]
        nn_mask[i, nearest] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    nn_weight = 1.8
    
    # 4. Overlap bonus for edges in both MST and nearest neighbors
    overlap_mask = mst_mask & nn_mask
    overlap_weight = 1.3  # Additional boost for overlap
    
    # 5. Triangle sampling: favor edges that form good triangles
    triangle_bonus = np.ones_like(distance_matrix)
    if n >= 3:
        # For each edge, check if it forms good triangles with other nodes
        for i in range(n):
            for j in range(i + 1, n):
                # Find the best third node that minimizes triangle perimeter
                if safe_dist[i, j] < np.inf:
                    min_perimeter = np.inf
                    for k in range(n):
                        if k != i and k != j:
                            perimeter = safe_dist[i, k] + safe_dist[k, j]
                            if perimeter < min_perimeter:
                                min_perimeter = perimeter
                    
                    # Bonus if direct edge is comparable to best triangle path
                    if min_perimeter > 0:
                        ratio = safe_dist[i, j] / min_perimeter
                        triangle_bonus[i, j] = triangle_bonus[j, i] = 1.5 * np.exp(-ratio)
    
    # 6. Combine all factors
    heuristics = base_heuristic.copy()
    heuristics *= triangle_bonus  # Apply triangle bonus to all edges
    
    # Apply MST, NN, and overlap weights multiplicatively
    heuristics[mst_mask] *= mst_weight
    heuristics[nn_mask] *= nn_weight
    heuristics[overlap_mask] *= overlap_weight
    
    # 7. Adaptive logarithmic percentile sparsification
    # Scale percentile logarithmically with problem size
    base_percentile = 85
    scale_factor = 10 / np.log2(n + 2)  # Logarithmic scaling
    adaptive_percentile = max(60, min(base_percentile + scale_factor, 95))
    
    sparsity_mask = np.zeros_like(heuristics, dtype=bool)
    
    # Keep edges that are in MST, nearest neighbors, or have high heuristic values
    for i in range(n):
        row_values = heuristics[i]
        if np.any(row_values > 0):
            threshold = np.percentile(row_values[row_values > 0], adaptive_percentile)
            sparsity_mask[i] = (row_values >= threshold)
    
    # Always keep MST edges and overlap edges
    sparsity_mask = sparsity_mask | mst_mask | overlap_mask
    
    # Apply sparsification
    heuristics = heuristics * sparsity_mask
    
    # 8. Ensure connectivity: if graph becomes disconnected, add some high-value edges
    # This simple approach ensures at least one edge per node
    for i in range(n):
        if np.sum(heuristics[i] > 0) == 0:
            # Add the best available edge for this node
            best_j = np.argmax(base_heuristic[i])
            heuristics[i, best_j] = base_heuristic[i, best_j]
            heuristics[best_j, i] = base_heuristic[best_j, i]
    
    # 9. Symmetrize and normalize
    heuristics = np.maximum(heuristics, heuristics.T)
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
