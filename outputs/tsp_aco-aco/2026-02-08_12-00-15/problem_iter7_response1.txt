import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with multiplicative combination,
    triangle violation penalties, adaptive sparsification, and logarithmic boosting.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree inclusion (multiplicative bonus)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_factor = np.ones_like(base_heuristic)
    mst_factor[mst_mask] = 2.0  # MST edges get 2x boost
    
    # 3. Nearest neighbor factor (adaptive k)
    k = max(2, min(6, int(np.sqrt(n))))  # Scale with sqrt(n)
    nn_factor = np.ones_like(base_heuristic)
    
    for i in range(n):
        # Get k+1 nearest (skip self with inf)
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i][:k]
        nn_factor[i, nearest] = 1.5  # Nearest neighbors get 1.5x boost
    
    # Symmetrize nearest neighbor factor
    nn_factor = np.maximum(nn_factor, nn_factor.T)
    
    # 4. Triangle violation penalty
    # Penalize edges that violate triangle inequality significantly
    triangle_penalty = np.ones_like(base_heuristic)
    
    if n >= 3:
        for i in range(n):
            for j in range(i+1, n):
                # Find minimum alternative path through any third node
                min_alt = np.inf
                for k in range(n):
                    if k != i and k != j:
                        alt_path = safe_dist[i, k] + safe_dist[k, j]
                        min_alt = min(min_alt, alt_path)
                
                # Apply penalty if direct edge is much longer than best alternative
                if min_alt < np.inf and safe_dist[i, j] > 1.5 * min_alt:
                    penalty = min_alt / safe_dist[i, j]
                    triangle_penalty[i, j] = triangle_penalty[j, i] = penalty
    
    # 5. Combine all factors multiplicatively
    heuristics = base_heuristic * mst_factor * nn_factor * triangle_penalty
    
    # 6. Logarithmic boosting for top edges
    # Apply log1p to boost top edges while compressing range
    heuristics = np.log1p(heuristics * 10)  # Scale before log to preserve differences
    
    # 7. Adaptive sparsification based on problem size
    # Keep more edges for smaller problems, fewer for larger ones
    if n <= 20:
        keep_percentile = 70
    elif n <= 50:
        keep_percentile = 80
    elif n <= 100:
        keep_percentile = 85
    else:
        keep_percentile = 90
    
    # Create sparsification mask
    sparsity_mask = np.zeros_like(heuristics, dtype=bool)
    
    # Always keep MST edges
    sparsity_mask[mst_mask] = True
    
    # For each node, keep edges above percentile threshold
    for i in range(n):
        row_vals = heuristics[i]
        # Exclude already masked MST edges and diagonal
        valid_mask = ~mst_mask[i] & (np.arange(n) != i)
        valid_vals = row_vals[valid_mask]
        
        if len(valid_vals) > 0:
            threshold = np.percentile(valid_vals, keep_percentile)
            sparsity_mask[i] = sparsity_mask[i] | ((row_vals >= threshold) & valid_mask)
    
    # Symmetrize the mask
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsification
    heuristics = heuristics * sparsity_mask
    
    # 8. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Ensure symmetry (numerical errors might cause slight asymmetry)
    heuristics = (heuristics + heuristics.T) / 2
    
    return heuristics
