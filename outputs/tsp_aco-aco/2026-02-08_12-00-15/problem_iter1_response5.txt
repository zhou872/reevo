import numpy as np
from scipy.spatial.distance import squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP that combines multiple factors to determine edge promise.
    Returns a matrix of the same shape with higher values for more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Initialize heuristic matrix with zeros
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # 1. Inverse distance (shorter edges are generally better)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.isnan(inv_dist)] = 0
    
    # 2. Nearest neighbor factor
    # For each node, give higher weight to edges connecting to its k nearest neighbors
    k_nearest = min(5, n-1)
    nn_factor = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get indices of k+1 nearest neighbors (including self)
        nearest_indices = np.argsort(distance_matrix[i])[:k_nearest+1]
        # Remove self if present
        nearest_indices = nearest_indices[nearest_indices != i]
        
        # Assign higher weights to nearest neighbors
        for j in nearest_indices[:k_nearest]:
            # Weight decays with neighbor rank
            rank = np.where(nearest_indices == j)[0][0] + 1
            nn_factor[i, j] = 1.0 / rank
    
    # 3. Minimum spanning tree factor
    # Edges likely to be in MST are more promising
    mst_factor = np.zeros_like(distance_matrix)
    
    try:
        # Use Prim's algorithm to find MST edges
        from scipy.sparse.csgraph import minimum_spanning_tree
        
        mst = minimum_spanning_tree(distance_matrix)
        mst_dense = mst.toarray()
        mst_factor = (mst_dense > 0).astype(float) * 1.5
        
    except ImportError:
        # Fallback: approximate MST using simpler approach
        # Connect each node to its nearest neighbor not already in the tree
        visited = np.zeros(n, dtype=bool)
        visited[0] = True
        
        for _ in range(n-1):
            min_dist = np.inf
            u, v = -1, -1
            
            # Find minimum distance between visited and unvisited nodes
            for i in np.where(visited)[0]:
                for j in np.where(~visited)[0]:
                    if distance_matrix[i, j] < min_dist:
                        min_dist = distance_matrix[i, j]
                        u, v = i, j
            
            if u >= 0 and v >= 0:
                mst_factor[u, v] = 1.5
                mst_factor[v, u] = 1.5
                visited[v] = True
    
    # 4. Angle-based factor for geometric TSP
    # If coordinates were available, we could use angle information
    # Since we only have distances, we approximate using triangle inequality
    angle_factor = np.ones_like(distance_matrix)
    
    if n >= 3:
        # For each edge (i,j), check if it forms small angles with other edges
        for i in range(n):
            for j in range(i+1, n):
                # Look for a third node k
                benefits = []
                for k in range(n):
                    if k != i and k != j:
                        # Using triangle inequality to estimate "straightness"
                        a = distance_matrix[i, j]
                        b = distance_matrix[i, k]
                        c = distance_matrix[j, k]
                        
                        # Law of cosines approximation
                        if a > 0 and b > 0 and c > 0:
                            cos_angle = (b**2 + c**2 - a**2) / (2 * b * c)
                            cos_angle = np.clip(cos_angle, -1, 1)
                            
                            # Prefer edges that are part of "straight" paths
                            if abs(cos_angle) < 0.5:  # Angle between 60 and 120 degrees
                                benefits.append(1.2)
                
                if benefits:
                    angle_factor[i, j] = angle_factor[j, i] = np.mean(benefits)
    
    # 5. Degree-based sparsification
    # Nodes that are too connected might indicate poor choices
    degree_penalty = np.ones_like(distance_matrix)
    
    for i in range(n):
        # Count how many edges from node i are relatively short
        avg_dist = np.mean(distance_matrix[i, distance_matrix[i] > 0])
        short_edges = np.sum(distance_matrix[i] < 0.7 * avg_dist)
        
        if short_edges > n/2:  # Node has many short connections
            # Penalize the longest edges from this node
            for j in range(n):
                if i != j and distance_matrix[i, j] > 1.5 * avg_dist:
                    degree_penalty[i, j] = 0.3
    
    # Combine all factors
    heuristics = inv_dist * nn_factor * mst_factor * angle_factor * degree_penalty
    
    # Normalize to make values more interpretable
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Zero out diagonal (no self-loops)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
