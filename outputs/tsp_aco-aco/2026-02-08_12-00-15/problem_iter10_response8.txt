```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge selection heuristics with clear MST and nearest neighbor bonuses,
    adaptive sparsification, and normalization.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: reciprocal of distance
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree edges with bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.5  # Clear bonus for MST edges
    
    # 3. Adaptive nearest neighbors with bonus
    k = max(3, min(10, n // 5))  # Adaptive k: 3-10, scaled with n
    nn_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        row = safe_dist[i]
        nearest_indices = np.argpartition(row, k)[:k]
        nn_mask[i, nearest_indices] = True
    nn_mask = nn_mask | nn_mask.T  # Symmetrize
    nn_bonus = 1.5  # Clear bonus for nearest neighbors
    
    # 4. Combine heuristics with bonuses
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    
    # 5. Simple adaptive sparsification
    # Keep top edges per node, adaptive to problem size
    keep_per_node = max(2, min(5, n // 4))
    sparsified = np.zeros_like(heuristics)
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Exclude diagonal
        top_indices = np.argpartition(-row, keep_per_node)[:keep_per_node]
        sparsified[i, top_indices] = row[top_indices]
    
    # 6. Ensure symmetry and zero diagonal
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0.0)
    
    # 7. Normalize to [0, 1]
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
```
