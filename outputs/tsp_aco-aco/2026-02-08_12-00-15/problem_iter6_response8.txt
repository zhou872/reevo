import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge heuristics focusing on MST, nearest neighbors,
    triangle inequality violations, and boosting top edges.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
        
    Returns:
        Heuristic matrix with higher values indicating more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    inv_dist = 1.0 / safe_dist
    np.fill_diagonal(inv_dist, 0.0)
    
    # Row-normalize inverse distance
    row_sums = inv_dist.sum(axis=1, keepdims=True)
    inv_dist_norm = np.zeros_like(inv_dist)
    valid_rows = row_sums.flatten() > 0
    inv_dist_norm[valid_rows] = inv_dist[valid_rows] / row_sums[valid_rows]
    
    # 2. Minimum Spanning Tree
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbors with adaptive k
    # Use more neighbors for small instances, fewer for large
    base_k = max(2, min(8, int(np.sqrt(n))))
    # Adjust based on distance distribution
    median_dist = np.median(safe_dist[safe_dist < np.inf])
    if median_dist > 0:
        # If distances are large, be more selective
        k = max(2, int(base_k * (1.0 / np.log1p(median_dist))))
    else:
        k = base_k
    
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i][:k]
        nn_mask[i, nearest] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Triangle inequality penalty (vectorized)
    violation_penalty = np.ones_like(distance_matrix)
    if n >= 3:
        # Check triangles using vectorized approach
        # For each intermediate node k, check violations
        for k in range(n):
            # Compute triangle inequality sum: d(i,k) + d(k,j)
            triangle_sum = safe_dist[:, k:k+1] + safe_dist[k:k+1, :]
            
            # Find violations where direct path is significantly longer
            # Use adaptive tolerance: smaller for shorter edges
            with np.errstate(invalid='ignore'):
                relative_diff = (safe_dist - triangle_sum) / (safe_dist + 1e-10)
            
            # Penalize more severe violations more strongly
            violation_mask = (safe_dist > triangle_sum * 1.05) & (safe_dist < np.inf)
            if np.any(violation_mask):
                # Penalty decays exponentially with violation severity
                penalty_factor = np.exp(-relative_diff * 2)
                violation_penalty[violation_mask] *= penalty_factor[violation_mask]
    
    # 5. Combine MST and nearest neighbor bonuses
    heuristics = inv_dist_norm.copy()
    
    # Bonus values (tuned)
    mst_bonus = 2.2  # Slightly higher for MST edges
    nn_bonus = 1.6   # Moderate bonus for NN edges
    both_bonus = 3.5  # Strong bonus for edges in both
    
    # Apply bonuses
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    # Reset and apply combined bonus for edges in both
    edges_both = mst_mask & nn_mask
    heuristics[edges_both] = inv_dist_norm[edges_both] * both_bonus
    
    # Apply triangle violation penalty
    heuristics *= violation_penalty
    
    # 6. Adaptive sparsification
    # Keep more edges for smaller instances
    if n <= 15:
        edges_per_node = n - 1  # Keep all edges
    elif n <= 50:
        edges_per_node = max(5, n // 2)
    elif n <= 100:
        edges_per_node = max(8, int(np.sqrt(n) * 2))
    else:
        edges_per_node = max(10, int(np.log2(n) * 4))
    
    # Always keep MST edges, then top heuristic edges
    sparsity_mask = mst_mask.copy()
    for i in range(n):
        row_vals = heuristics[i].copy()
        row_vals[i] = -np.inf
        
        # Count how many MST edges we already have for this node
        mst_count = np.sum(mst_mask[i])
        
        # Determine how many more edges to add
        edges_to_add = max(0, edges_per_node - mst_count)
        
        if edges_to_add > 0:
            # Get top edges not already in MST
            non_mst_mask = ~mst_mask[i]
            non_mst_mask[i] = False  # Exclude diagonal
            
            if np.any(non_mst_mask):
                non_mst_indices = np.where(non_mst_mask)[0]
                non_mst_vals = row_vals[non_mst_indices]
                
                if len(non_mst_vals) > 0:
                    # Get top edges_to_add values
                    if edges_to_add < len(non_mst_vals):
                        top_k = np.argsort(non_mst_vals)[-edges_to_add:]
                        top_indices = non_mst_indices[top_k]
                        sparsity_mask[i, top_indices] = True
                    else:
                        sparsity_mask[i, non_mst_indices] = True
    
    # Symmetrize the sparsity mask
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply sparsity mask
    heuristics = heuristics * sparsity_mask
    
    # 7. Boost top edges per node with adaptive strength
    for i in range(n):
        row_vals = heuristics[i].copy()
        row_vals[i] = 0
        
        # Get non-zero edges
        non_zero_mask = row_vals > 0
        if np.sum(non_zero_mask) > 1:
            non_zero_indices = np.where(non_zero_mask)[0]
            values = row_vals[non_zero_indices]
            
            # Sort and assign ranks
            sorted_idx = np.argsort(values)
            ranks = np.arange(len(values)) + 1
            
            # Adaptive boosting: stronger for top-ranked edges
            # Use exponential decay based on rank
            boost_factors = 1.0 + np.exp(-ranks[::-1] / (len(values) / 3))
            
            # Apply boost while maintaining symmetry
            for idx, factor in zip(sorted_idx, boost_factors):
                j = non_zero_indices[idx]
                if heuristics[i, j] > 0:  # Double-check
                    heuristics[i, j] *= factor
                    heuristics[j, i] *= factor
    
    # 8. Final processing
    # Ensure symmetry (handle floating-point errors)
    heuristics = (heuristics + heuristics.T) / 2
    
    # Normalize to [0, 1] range
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Apply soft threshold: zero out very small values
    threshold = 0.01 * heuristics[heuristics > 0].mean() if np.any(heuristics > 0) else 0
    heuristics[heuristics < threshold] = 0
    
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
