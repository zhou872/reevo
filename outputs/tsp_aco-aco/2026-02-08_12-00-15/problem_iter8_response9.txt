import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge selection heuristics using percentile thresholds
    with MST and neighbor bonuses.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Compute Minimum Spanning Tree
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Compute distance percentiles for each node
    # Create a mask to exclude diagonal (inf) values
    valid_mask = ~np.isinf(safe_dist)
    
    # Compute percentile thresholds for each node
    # Use lower percentiles for shorter distances (more promising)
    short_percentile = 25  # Consider distances below 25th percentile as "short"
    neighbor_percentile = 40  # Consider distances below 40th percentile as "neighbors"
    
    short_thresholds = np.zeros(n)
    neighbor_thresholds = np.zeros(n)
    
    for i in range(n):
        valid_distances = safe_dist[i][valid_mask[i]]
        if len(valid_distances) > 0:
            short_thresholds[i] = np.percentile(valid_distances, short_percentile)
            neighbor_thresholds[i] = np.percentile(valid_distances, neighbor_percentile)
        else:
            short_thresholds[i] = np.inf
            neighbor_thresholds[i] = np.inf
    
    # 4. Create masks based on percentile thresholds
    # Short edges mask (more restrictive)
    short_mask = np.zeros_like(distance_matrix, dtype=bool)
    # Neighbor edges mask (less restrictive)
    neighbor_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Short edges: distances below short_percentile threshold
        short_mask[i] = (safe_dist[i] <= short_thresholds[i]) & valid_mask[i]
        
        # Neighbor edges: distances below neighbor_percentile threshold
        # but not already in short_mask (to avoid double counting in bonus)
        neighbor_candidates = (safe_dist[i] <= neighbor_thresholds[i]) & valid_mask[i]
        neighbor_mask[i] = neighbor_candidates & ~short_mask[i]
    
    # Symmetrize masks
    short_mask = short_mask | short_mask.T
    neighbor_mask = neighbor_mask | neighbor_mask.T
    
    # 5. Apply bonuses based on percentile thresholds
    heuristics = base_heuristic.copy()
    
    # MST edges get highest bonus, scaled by how short they are
    for i in range(n):
        for j in range(i+1, n):
            if mst_mask[i, j]:
                # Bonus inversely proportional to distance percentile
                dist_percentile_i = np.sum(safe_dist[i] <= safe_dist[i, j]) / (n-1) * 100
                dist_percentile_j = np.sum(safe_dist[j] <= safe_dist[j, i]) / (n-1) * 100
                avg_percentile = (dist_percentile_i + dist_percentile_j) / 2
                
                # Higher bonus for edges with lower percentiles (shorter distances)
                mst_bonus = 1.0 + (100 - avg_percentile) / 50  # Ranges from 1.0 to 3.0
                heuristics[i, j] *= mst_bonus
                heuristics[j, i] *= mst_bonus
    
    # Short edges (below 25th percentile) get moderate bonus
    short_bonus = 1.8
    heuristics[short_mask] *= short_bonus
    
    # Neighbor edges (below 40th percentile) get smaller bonus
    neighbor_bonus = 1.3
    heuristics[neighbor_mask] *= neighbor_bonus
    
    # 6. Sparsification using percentile thresholds
    # Keep edges that are in MST, or have heuristic value above percentile threshold
    sparsified = np.zeros_like(heuristics)
    
    for i in range(n):
        row_heuristics = heuristics[i].copy()
        # Always keep MST edges
        mst_indices = np.where(mst_mask[i])[0]
        sparsified[i, mst_indices] = row_heuristics[mst_indices]
        
        # For non-MST edges, keep those above 70th percentile
        non_mst_mask = ~mst_mask[i]
        non_mst_mask[i] = False  # Exclude self
        
        if np.sum(non_mst_mask) > 0:
            non_mst_heuristics = row_heuristics[non_mst_mask]
            if len(non_mst_heuristics) > 0:
                threshold = np.percentile(non_mst_heuristics, 70)
                keep_indices = np.where(non_mst_mask & (row_heuristics >= threshold))[0]
                sparsified[i, keep_indices] = row_heuristics[keep_indices]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0.0)
    
    # 7. Normalize for consistency
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
