import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristics with vectorized triangle inequality checks,
    logarithmic neighbor scaling, and median-based edge boosting.
    
    Features:
    1. Vectorized triangle inequality penalty computation
    2. Logarithmic scaling for nearest neighbor connections
    3. Median-based adaptive edge boosting
    4. Problem-size adaptive sparsification
    5. MST and distance-based heuristics
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between cities
        
    Returns:
    --------
    np.ndarray
        Heuristic scores for each edge (same shape as input)
    """
    n = distance_matrix.shape[0]
    
    # Create a copy with diagonal set to infinity to avoid self-connections
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # 1. Baseline: inverse distance
    heuristics = 1 / dist
    
    # 2. MST bonus
    mst = minimum_spanning_tree(dist).toarray()
    mst_mask = (mst + mst.T) > 0
    heuristics[mst_mask] *= 2.0
    
    # 3. Nearest neighbor connections with logarithmic scaling
    k = max(2, int(np.log2(n)) + 2)  # Logarithmic scaling
    nn_mask = np.zeros_like(dist, dtype=bool)
    for i in range(n):
        nearest = np.argsort(dist[i])[:k]
        nn_mask[i, nearest] = True
    nn_mask = nn_mask | nn_mask.T
    heuristics[nn_mask] *= 1.5
    
    # 4. Vectorized triangle inequality penalty
    # For each pair (i,j), find minimum distance via any third node k
    # Using broadcasting for efficient computation
    if n > 3:
        # Reshape for broadcasting: dist[i,k] + dist[k,j] for all k
        via_third = dist[:, :, np.newaxis] + dist[np.newaxis, :, :]  # Shape: (n, n, n)
        
        # Set diagonal elements to infinity to avoid using i or j as intermediate
        i_indices = np.arange(n)
        via_third[i_indices, i_indices, :] = np.inf
        via_third[i_indices, :, i_indices] = np.inf
        
        # Find minimum via third node for each pair (i,j)
        min_via = np.min(via_third, axis=2)
        
        # Apply penalty when a shorter path exists via third node
        penalty_mask = min_via < dist
        penalty_values = np.ones_like(dist)
        penalty_values[penalty_mask] = 0.8
        heuristics *= penalty_values
    
    # 5. Adaptive sparsification based on problem size
    if n > 30:
        # For larger problems, keep fewer edges
        top_k = max(3, int(np.sqrt(n)))
        threshold_mask = np.zeros_like(heuristics, dtype=bool)
        for i in range(n):
            top_indices = np.argsort(heuristics[i])[-top_k:]
            threshold_mask[i, top_indices] = True
    else:
        # For smaller problems, use percentile-based threshold
        sparsity = 0.3 + 0.5 * (n / 100)  # Adaptive sparsity
        threshold_mask = np.zeros_like(heuristics, dtype=bool)
        for i in range(n):
            row = heuristics[i].copy()
            row[i] = -np.inf  # Exclude diagonal
            if np.any(row > 0):
                threshold = np.percentile(row[row > 0], 100 * (1 - sparsity))
                threshold_mask[i] = row >= threshold
    
    # Ensure symmetry
    threshold_mask = threshold_mask | threshold_mask.T
    heuristics = heuristics * threshold_mask
    
    # 6. Median-based edge boosting
    for i in range(n):
        row = heuristics[i].copy()
        non_zero = row[row > 0]
        if len(non_zero) > 1:
            median_val = np.median(non_zero)
            # Boost edges significantly above median
            boost_mask = row > 1.5 * median_val
            heuristics[i, boost_mask] *= 1.3
    
    # 7. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Set diagonal to 0
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
