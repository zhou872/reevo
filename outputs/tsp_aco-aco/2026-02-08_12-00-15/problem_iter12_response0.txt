import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP heuristics with linear parameter scaling and percentile-based sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (closer nodes are more promising)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree connectivity (essential for TSP)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbor factor with linear scaling
    # Scale k linearly from 2 for small n to ~10% of n for large n
    k = max(2, min(int(0.1 * n), 10))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i][:k]
        nn_mask[i, nearest] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Apply bonuses with linear scaling factors
    heuristics = base_heuristic.copy()
    
    # Scale MST bonus from 2.5 for small graphs to 2.0 for large graphs
    mst_bonus = 2.5 - 0.5 * min(1.0, n / 50)
    heuristics[mst_mask] *= mst_bonus
    
    # Scale NN bonus from 1.75 for small graphs to 1.25 for large graphs  
    nn_bonus = 1.75 - 0.5 * min(1.0, n / 50)
    heuristics[nn_mask] *= nn_bonus
    
    # 5. Simplified percentile-based sparsification
    # Use adaptive percentile threshold: higher for larger graphs
    percentile = 75 + 15 * min(1.0, (n - 10) / 40)  # 75% for n=10, up to 90% for nˇÝ50
    
    # Get non-zero heuristic values for percentile calculation
    non_zero_vals = heuristics[heuristics > 0]
    if len(non_zero_vals) > 0:
        threshold = np.percentile(non_zero_vals, percentile)
    else:
        threshold = 0
    
    # Keep edges above threshold OR in MST/NN masks
    keep_mask = (heuristics >= threshold) | mst_mask | nn_mask
    heuristics = heuristics * keep_mask
    
    # 6. Ensure symmetry and final normalization
    heuristics = np.maximum(heuristics, heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # Normalize non-zero values to [0, 1]
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
