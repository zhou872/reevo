```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristics focusing on MST backbone with per-node sparsification.
    
    Key improvements:
    1. Strong MST edge boosting as solution backbone
    2. Adaptive per-node edge selection based on problem size
    3. Guaranteed symmetry through max operator
    4. Clean normalization for stable sampling
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n) with zeros on diagonal
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, with higher values indicating
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    
    # 1. Baseline: inverse distance (closer nodes are more promising)
    inv_dist = 1 / np.where(distance_matrix > 0, distance_matrix, np.inf)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Strong MST emphasis as solution backbone
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst + mst.T) > 0  # Symmetrize
    
    # Strong MST boost - these form the essential solution structure
    heuristics = inv_dist.copy()
    mst_boost = 5.0  # Increased from 3.0 for stronger MST emphasis
    heuristics[mst_edges] *= mst_boost
    
    # 3. Adaptive per-node edge retention
    # Keep more edges for smaller problems, fewer for larger ones
    if n <= 20:
        edges_per_node = min(n - 1, 8)
    elif n <= 50:
        edges_per_node = min(n - 1, 6)
    elif n <= 100:
        edges_per_node = min(n - 1, 4)
    else:
        edges_per_node = min(n - 1, 3)
    
    # Keep only top edges for each node
    for i in range(n):
        row_values = heuristics[i].copy()
        row_values[i] = -np.inf  # Exclude self
        
        if edges_per_node < n - 1:
            # Get indices of top edges
            top_indices = np.argpartition(row_values, -edges_per_node)[-edges_per_node:]
            # Create mask for this node's top edges
            row_mask = np.zeros(n, dtype=bool)
            row_mask[top_indices] = True
            # Keep only top edges for this node
            heuristics[i] *= row_mask
    
    # 4. Ensure symmetry by keeping edge if selected by either endpoint
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 5. Final normalization for stability
    heuristics[heuristics < 1e-10] = 0
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Ensure diagonal remains zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
```
