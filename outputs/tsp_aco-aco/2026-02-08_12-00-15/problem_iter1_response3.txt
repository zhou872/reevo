import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP that combines multiple factors to determine edge promise
    and sparsifies the matrix by setting unpromising edges to zero.
    """
    n = distance_matrix.shape[0]
    np.fill_diagonal(distance_matrix, np.inf)  # Avoid self-loops
    
    # 1. Base heuristic: inverse of distance
    inv_distance = 1 / distance_matrix
    
    # 2. Local clustering coefficient: favor edges between nodes with common neighbors
    clustering_factor = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                common_neighbors = np.sum((distance_matrix[i] < np.inf) & 
                                         (distance_matrix[j] < np.inf) & 
                                         (np.arange(n) != i) & (np.arange(n) != j))
                clustering_factor[i, j] = common_neighbors / (n - 2)
    
    # 3. Relative distance ranking: favor edges that are among the k-nearest neighbors
    k = max(3, int(np.log2(n)))  # Adaptive k based on problem size
    rank_factor = np.zeros((n, n))
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])
        for rank, j in enumerate(sorted_indices[:k]):
            if i != j:
                rank_factor[i, j] = (k - rank) / k
    
    # 4. Path saving potential: edges that connect to nodes with few promising connections
    node_promise = np.sum(inv_distance, axis=1) - np.diag(inv_distance)
    path_saving = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                path_saving[i, j] = 1 / (1 + min(node_promise[i], node_promise[j]))
    
    # Combine factors with weights
    heuristics_matrix = (
        0.5 * inv_distance +
        0.2 * clustering_factor +
        0.2 * rank_factor +
        0.1 * path_saving
    )
    
    # Sparsification: keep only top edges per node
    sparsified = np.zeros((n, n))
    edges_per_node = max(2, int(np.sqrt(n)))  # Adaptive sparsity
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Exclude diagonal
        top_indices = np.argpartition(row, -edges_per_node)[-edges_per_node:]
        sparsified[i, top_indices] = heuristics_matrix[i, top_indices]
    
    # Make symmetric (keep edge if promising in either direction)
    symmetric_heuristics = np.maximum(sparsified, sparsified.T)
    
    # Normalize to [0, 1] range for consistency
    max_val = np.max(symmetric_heuristics)
    if max_val > 0:
        symmetric_heuristics /= max_val
    
    np.fill_diagonal(symmetric_heuristics, 0)
    np.fill_diagonal(distance_matrix, 0)  # Restore original diagonal
    
    return symmetric_heuristics
