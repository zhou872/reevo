import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge selection heuristics with simplified computations,
    random connectivity sampling, and adaptive per-node sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree edges get strong bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.5
    
    # 3. Simplified nearest neighbor scaling
    k = max(2, min(10, int(np.log2(n)) + 1))
    nn_heuristic = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get k nearest neighbors
        sorted_indices = np.argsort(safe_dist[i])
        sorted_indices = sorted_indices[sorted_indices != i][:k]
        
        # Assign weights based on rank
        for rank, j in enumerate(sorted_indices):
            nn_heuristic[i, j] = 1.0 / (rank + 2)  # 1/2, 1/3, 1/4, ...
    
    # Symmetrize and create bonus
    nn_heuristic = np.maximum(nn_heuristic, nn_heuristic.T)
    nn_bonus = 1.0 + 0.8 * nn_heuristic  # Scale between 1.0 and ~1.4
    
    # 4. Random connectivity sampling
    connectivity_bonus = np.ones_like(distance_matrix)
    
    if n >= 3:
        # Sample random edges proportional to their base heuristic
        num_samples = min(200, n * 5)
        edge_probs = base_heuristic.flatten()
        edge_probs = edge_probs / edge_probs.sum()
        
        # Convert flat indices to coordinates
        flat_indices = np.random.choice(
            len(edge_probs), 
            size=num_samples, 
            p=edge_probs,
            replace=True
        )
        
        # Boost sampled edges
        i_indices, j_indices = np.unravel_index(flat_indices, distance_matrix.shape)
        connectivity_bonus[i_indices, j_indices] += 0.3
        
        # Ensure symmetry
        connectivity_bonus = (connectivity_bonus + connectivity_bonus.T) / 2
    
    # 5. Combine factors
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics *= nn_bonus
    heuristics *= connectivity_bonus
    
    # 6. Adaptive per-node sparsification
    for i in range(n):
        row_values = heuristics[i]
        valid_values = row_values[row_values > 0]
        
        if len(valid_values) > 0:
            # Keep fewer edges for larger instances
            percentile = max(50, 95 - n // 2)
            threshold = np.percentile(valid_values, percentile)
            
            # Always keep MST edges and top 2 nearest neighbors
            top_nn_indices = np.argsort(nn_heuristic[i])[-2:]
            keep_mask = (
                (row_values >= threshold) | 
                mst_mask[i] | 
                np.isin(np.arange(n), top_nn_indices)
            )
            
            heuristics[i, ~keep_mask] = 0.0
    
    # 7. Final symmetrization and normalization
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0.0)
    
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
