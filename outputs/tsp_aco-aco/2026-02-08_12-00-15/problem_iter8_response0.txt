```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge selection heuristics using MST, adaptive nearest neighbors,
    and simple sparsification. Returns a matrix where higher values indicate
    more promising edges for inclusion in TSP solutions.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create safe distance matrix with diagonal set to infinity
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # Base heuristic: reciprocal of distance
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # Minimum Spanning Tree mask
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # Adaptive nearest neighbors mask
    k = max(2, n // 10)  # Adaptive k based on instance size
    nn_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        row = safe_dist[i].copy()
        # Get k nearest neighbors (excluding self, which is already inf)
        nearest_indices = np.argpartition(row, k)[:k]
        nn_mask[i, nearest_indices] = True
    nn_mask = nn_mask | nn_mask.T  # Symmetrize
    
    # Combine heuristics with bonuses
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= 2.0  # Strong bonus for MST edges
    heuristics[nn_mask] *= 1.5   # Moderate bonus for nearest neighbors
    
    # Simple sparsification: keep only top edges per node
    sparsified = np.zeros_like(heuristics)
    keep_count = max(2, min(5, n // 4))  # Adaptive number of edges to keep
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Exclude self
        top_indices = np.argpartition(-row, keep_count)[:keep_count]
        sparsified[i, top_indices] = row[top_indices]
    
    # Ensure symmetry and zero diagonal
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0.0)
    
    # Normalize to [0, 1]
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
```
