import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge selection heuristics with focused MST and NN components,
    overlap boosting, rank-based enhancement, and simplified sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    inv_dist = 1.0 / safe_dist
    np.fill_diagonal(inv_dist, 0.0)
    
    # 2. Minimum Spanning Tree edges
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest Neighbor edges (adaptive k)
    k = max(3, min(15, n // 3))
    nn_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[1:k+1]
        nn_mask[i, nearest] = True
    
    # Symmetrize NN mask
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Combine MST and NN with boosted overlaps
    heuristics = inv_dist.copy()
    
    # Apply individual bonuses
    heuristics[mst_mask] *= 2.5      # Strong MST bonus
    heuristics[nn_mask] *= 1.5       # Moderate NN bonus
    
    # Extra boost for edges in both MST and NN
    overlap_mask = mst_mask & nn_mask
    heuristics[overlap_mask] *= 2.0
    
    # 5. Adaptive sparsification
    edges_per_node = max(3, min(20, n // 2))
    sparse_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        row_vals = heuristics[i].copy()
        row_vals[i] = -np.inf
        top_indices = np.argsort(row_vals)[-edges_per_node:]
        sparse_mask[i, top_indices] = True
    
    # Symmetrize and apply sparsification
    sparse_mask = sparse_mask | sparse_mask.T
    heuristics = heuristics * sparse_mask
    
    # 6. Rank-based boosting per node
    for i in range(n):
        row_nonzero = heuristics[i] > 0
        nonzero_indices = np.where(row_nonzero)[0]
        
        if len(nonzero_indices) > 1:
            # Get values and sort descending (higher values get higher boost)
            values = heuristics[i, nonzero_indices]
            sorted_order = np.argsort(values)[::-1]
            
            # Apply rank-based boost (top ranked gets highest multiplier)
            for rank, idx in enumerate(sorted_order):
                boost = 1.0 + (0.5 * (len(sorted_order) - rank - 1)) / (len(sorted_order) - 1)
                j = nonzero_indices[idx]
                heuristics[i, j] *= boost
    
    # 7. Final processing
    heuristics = (heuristics + heuristics.T) / 2.0  # Ensure symmetry
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    np.fill_diagonal(heuristics, 0.0)
    
    return heuristics
