```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP heuristics focusing on three core components:
    1. Inverse distance (primary measure)
    2. Minimum Spanning Tree edges
    3. Nearest neighbor connections
    
    The function returns normalized scores where higher values indicate
    edges that are more likely to be in optimal TSP solutions.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix with same shape as input, values in [0, 1].
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree factor
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_factor = 2.0  # Strong boost for MST edges
    
    # 3. Nearest neighbor factor (adaptive k)
    k = max(2, min(5, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k+1 nearest to skip self (inf), then exclude self
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i][:k]
        nn_mask[i, nearest] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    nn_factor = 1.5  # Moderate boost for nearest neighbors
    
    # 4. Combine factors multiplicatively
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_factor
    heuristics[nn_mask] *= nn_factor
    
    # 5. Simple sparsification: keep only promising edges
    # Keep edges that are in MST, nearest neighbors, or top 20% of base scores
    combined_mask = mst_mask | nn_mask
    
    # Add top-scoring edges from base heuristic
    top_fraction = 0.20
    for i in range(n):
        row = base_heuristic[i]
        valid_values = row[row > 0]
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, 100 * (1 - top_fraction))
            combined_mask[i] = combined_mask[i] | (row >= threshold)
    
    # Ensure symmetry and apply mask
    combined_mask = combined_mask | combined_mask.T
    heuristics = heuristics * combined_mask
    
    # 6. Normalize to [0, 1]
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
```
