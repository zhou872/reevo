import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection.
    Incorporates MST, nearest neighbors, triangle inequality violations,
    adaptive sparsification, and logarithmic boosting.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Convert to float for safety in calculations
    D = distance_matrix.astype(float, copy=True)
    np.fill_diagonal(D, np.inf)  # Exclude diagonal from calculations
    
    # 1. Base inverse distance with logarithmic scaling
    base_heuristic = 1.0 / D
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. MST inclusion factor (binary mask)
    mst = minimum_spanning_tree(D).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbors factor
    # Adaptive k based on problem size
    k = max(2, min(8, int(np.sqrt(n))))
    nn_mask = np.zeros_like(D, dtype=bool)
    for i in range(n):
        # Get k+1 nearest neighbors (skip self at index 0)
        nearest_indices = np.argsort(D[i])[1:k+1]
        nn_mask[i, nearest_indices] = True
    nn_mask = nn_mask | nn_mask.T  # Ensure symmetry
    
    # 4. Triangle inequality violation penalty
    # For each edge (i,j), check if there exists k such that
    # D[i,k] + D[k,j] < D[i,j] * violation_threshold
    violation_penalty = np.ones_like(D)
    if n >= 3:
        violation_threshold = 0.9  # Allow 10% violation before penalizing
        for i in range(n):
            for j in range(i+1, n):
                # Skip if either is infinite
                if D[i,j] == np.inf:
                    continue
                
                # Find minimum two-hop path through any k
                min_two_hop = np.min(D[i,:] + D[:,j])
                
                # Apply penalty if violation exceeds threshold
                if min_two_hop < D[i,j] * violation_threshold:
                    # Quadratic penalty based on severity of violation
                    violation_severity = (D[i,j] - min_two_hop) / D[i,j]
                    penalty = max(0.1, 1.0 - violation_severity**2)
                    violation_penalty[i,j] = penalty
                    violation_penalty[j,i] = penalty
    
    # 5. Combine factors multiplicatively
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= 2.5  # Strong boost for MST edges
    heuristics[nn_mask] *= 1.8   # Moderate boost for nearest neighbors
    heuristics *= violation_penalty  # Apply triangle violation penalties
    
    # Ensure symmetry (numerical precision might cause slight asymmetry)
    heuristics = (heuristics + heuristics.T) / 2
    
    # 6. Adaptive sparsification
    # Keep edges that are in MST, nearest neighbors, or have high heuristic value
    combined_mask = mst_mask | nn_mask
    
    # Add top edges based on percentile thresholds
    if n > 10:
        # For larger problems, be more selective
        top_percentile = max(50, 100 - n/2)  # Adaptive percentile
    else:
        top_percentile = 80
    
    for i in range(n):
        row = heuristics[i]
        # Consider only non-zero, non-diagonal values
        valid_mask = (row > 0) & (~np.isinf(row))
        valid_values = row[valid_mask]
        
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, top_percentile)
            combined_mask[i] = combined_mask[i] | (row >= threshold)
    
    # Apply sparsification
    heuristics = heuristics * combined_mask
    
    # 7. Logarithmic boosting of top edges
    # Apply log transform to emphasize relative differences among promising edges
    non_zero_mask = heuristics > 0
    if np.any(non_zero_mask):
        # Scale to avoid log(1) = 0 issues
        max_val = heuristics.max()
        scaled = heuristics[non_zero_mask] / max_val
        # Apply log transform with offset to preserve ordering
        heuristics[non_zero_mask] = np.log1p(scaled * 9)  # log1p(9x) gives range [0, log10(10)] ＞ [0, 2.3]
    
    # 8. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0.0)
    
    return heuristics
