import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP heuristics focusing on reciprocal distances, MST edges,
    and triangle inequality violations with simple sparsification.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric matrix of distances between nodes
        (diagonal should be zero, but function handles it)
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, with higher values indicating
        more promising edges for TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create safe distance matrix (avoid division by zero)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # Base heuristic: reciprocal of distance (shorter edges = higher value)
    base_heuristic = 1.0 / safe_dist
    
    # Factor 1: MST mask (edges likely to be in optimal solution)
    mst_mask = _compute_mst_mask(safe_dist)
    
    # Factor 2: Nearest neighbor mask (local connectivity)
    # Use log scaling for adaptive k selection
    k = max(2, int(np.log2(n))) if n > 2 else 2
    k = min(k, n - 1)
    nn_mask = _compute_nearest_neighbor_mask(safe_dist, k)
    
    # Factor 3: Triangle inequality penalty (penalize detourable edges)
    triangle_penalty = _compute_triangle_penalty_v2(safe_dist)
    
    # Combine factors
    heuristics = base_heuristic * triangle_penalty
    
    # Apply sparsification: keep only edges in MST or NN masks
    mask = np.logical_or(mst_mask, nn_mask)
    heuristics = np.where(mask, heuristics, 0)
    
    # Normalize for consistent scale
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics

def _compute_mst_mask(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute minimum spanning tree mask using Prim's algorithm."""
    n = distance_matrix.shape[0]
    mask = np.zeros_like(distance_matrix, dtype=bool)
    
    if n <= 1:
        return mask
    
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    
    for _ in range(n - 1):
        # Find minimum edge from visited to unvisited
        min_val = np.inf
        min_i, min_j = -1, -1
        
        # Vectorized search for minimum edge
        visited_indices = np.where(visited)[0]
        unvisited_indices = np.where(~visited)[0]
        
        if len(visited_indices) > 0 and len(unvisited_indices) > 0:
            # Create grid of visited ˇÁ unvisited indices
            vis_grid, unvis_grid = np.meshgrid(visited_indices, unvisited_indices)
            distances = distance_matrix[vis_grid, unvis_grid]
            
            if distances.size > 0:
                min_idx = np.argmin(distances)
                min_val = distances.flat[min_idx]
                min_i = vis_grid.flat[min_idx]
                min_j = unvis_grid.flat[min_idx]
        
        if min_i >= 0:
            mask[min_i, min_j] = True
            mask[min_j, min_i] = True
            visited[min_j] = True
    
    return mask

def _compute_nearest_neighbor_mask(distance_matrix: np.ndarray, k: int) -> np.ndarray:
    """Compute k-nearest neighbor mask for each node."""
    n = distance_matrix.shape[0]
    mask = np.zeros_like(distance_matrix, dtype=bool)
    
    if n <= 1 or k <= 0:
        return mask
    
    k = min(k, n - 1)  # Cannot have more neighbors than n-1
    
    for i in range(n):
        # Get indices of k+1 smallest distances (including self at inf)
        indices = np.argpartition(distance_matrix[i], k+1)[:k+1]
        
        # Filter out self and take top k
        indices = indices[indices != i]
        if len(indices) > k:
            indices = indices[:k]
        
        mask[i, indices] = True
    
    # Make symmetric (if i is neighbor of j, j is neighbor of i)
    mask = np.logical_or(mask, mask.T)
    return mask

def _compute_triangle_penalty_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Compute penalty based on triangle inequality violations.
    Edges that are significantly longer than alternative paths get penalized.
    Uses vectorized computation for efficiency.
    """
    n = distance_matrix.shape[0]
    penalty = np.ones_like(distance_matrix)
    
    if n < 3:
        return penalty
    
    epsilon = 1e-10
    
    # Pre-compute for efficiency
    # For each pair (i,j), find best alternative path through any k
    for i in range(n):
        # Get distances from i to all nodes
        dist_i = distance_matrix[i, :]
        
        # For each j ˇŮ i, compute min_{kˇŮi,j} (dist[i,k] + dist[k,j])
        for j in range(n):
            if i == j:
                continue
            
            # Create mask for intermediate nodes (k ˇŮ i, j)
            mask = np.ones(n, dtype=bool)
            mask[i] = mask[j] = False
            
            if np.any(mask):
                # Compute alternative paths through valid k
                alt_paths = dist_i[mask] + distance_matrix[mask, j]
                min_alt = np.min(alt_paths) if alt_paths.size > 0 else np.inf
                
                # Apply penalty based on how much longer direct edge is
                if min_alt < np.inf:
                    ratio = distance_matrix[i, j] / (min_alt + epsilon)
                    
                    if ratio > 2.0:
                        penalty[i, j] = 0.0  # Completely discard very bad edges
                    elif ratio > 1.5:
                        penalty[i, j] = 0.2  # Strong penalty
                    elif ratio > 1.2:
                        penalty[i, j] = 0.5  # Moderate penalty
                    # else: keep penalty at 1.0
    
    return penalty
