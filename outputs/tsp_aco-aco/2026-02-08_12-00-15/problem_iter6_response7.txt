import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with improved sparsification.
    
    Combines inverse distance, Minimum Spanning Tree inclusion, and nearest
    neighbor information with adaptive percentile-based sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (handle zero/infinite distances)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus (efficient implementation)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.0  # MST edges are highly promising
    
    # 3. Nearest neighbor bonus with adaptive scaling
    # Use logarithm for better scaling with problem size
    k = max(2, min(5, int(np.log2(n + 1))))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Use argpartition for efficiency on large problems
        row = safe_dist[i]
        k_neighbors = min(k, n - 1)
        if k_neighbors > 0:
            nearest_indices = np.argpartition(row, k_neighbors)[:k_neighbors + 1]
            nearest_indices = nearest_indices[nearest_indices != i][:k_neighbors]
            nn_mask[i, nearest_indices] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.5
    
    # 4. Combine all factors with multiplicative bonuses
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    
    # 5. Adaptive percentile-based sparsification
    # Determine sparsity level based on problem size
    if n < 10:
        percentile = 60  # Keep more edges for small problems
    elif n < 50:
        percentile = 70
    elif n < 100:
        percentile = 75
    else:
        percentile = 80  # Keep fewer edges for large problems
    
    # Create mask using adaptive percentile threshold
    combined_mask = mst_mask | nn_mask
    
    # Add edges above percentile threshold for each node
    for i in range(n):
        row_heuristics = heuristics[i]
        # Consider only non-zero, non-self entries
        valid_mask = (row_heuristics > 0) & (~np.isclose(row_heuristics, 0))
        valid_entries = row_heuristics[valid_mask]
        
        if len(valid_entries) > 0:
            # Use percentile-based threshold
            threshold = np.percentile(valid_entries, percentile)
            combined_mask[i] = combined_mask[i] | (row_heuristics >= threshold)
    
    # Apply the final mask
    heuristics = heuristics * combined_mask
    
    # 6. Edge connectivity enhancement
    # Ensure each node has at least 2 connections
    for i in range(n):
        if np.sum(combined_mask[i]) < 2:
            # Find top 2 edges excluding current connections
            row_heuristics = heuristics[i].copy()
            row_heuristics[combined_mask[i]] = -np.inf
            top_indices = np.argsort(-row_heuristics)[:2]
            heuristics[i, top_indices] = row_heuristics[top_indices]
            combined_mask[i, top_indices] = True
    
    # 7. Symmetrize and normalize
    heuristics = np.maximum(heuristics, heuristics.T)
    
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    np.fill_diagonal(heuristics, 0.0)
    
    return heuristics
