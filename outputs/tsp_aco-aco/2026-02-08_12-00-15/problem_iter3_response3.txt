import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP with triangle inequality penalties and adaptive sparsification.
    Features:
    1. Inverse distance baseline
    2. Minimum Spanning Tree inclusion bonus
    3. Nearest neighbor connectivity
    4. Triangle inequality penalties for long edges
    5. Adaptive sparsification based on problem size
    6. Boosting of top edges through non-linear scaling
    """
    n = distance_matrix.shape[0]
    
    # 1. Baseline inverse distance with epsilon to avoid division by zero
    epsilon = 1e-10
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree (MST) inclusion with symmetrical bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.0
    
    # 3. Adaptive nearest neighbor connectivity
    # Use logarithmic scaling for k to handle different problem sizes
    k = max(2, int(np.log2(n + 1)) * 2)
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        nearest = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, nearest] = True
    
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.5
    
    # 4. Triangle inequality penalty for edges much longer than alternatives
    triangle_penalty = np.ones_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            # Find if there's a significantly shorter two-hop path
            if n > 2:  # Need at least 3 nodes for triangle inequality
                min_two_hop = np.min(distance_matrix[i, :] + distance_matrix[:, j])
                # Penalize if direct edge is significantly longer than best two-hop
                if distance_matrix[i, j] > min_two_hop * 1.5:
                    penalty_factor = min_two_hop / (distance_matrix[i, j] + epsilon)
                    triangle_penalty[i, j] = triangle_penalty[j, i] = penalty_factor
    
    # 5. Combine all factors with multiplicative bonuses and penalties
    heuristics = inv_dist.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    heuristics *= triangle_penalty  # Apply triangle inequality penalty
    
    # 6. Adaptive sparsification based on problem size
    # Keep more edges for smaller problems, fewer for larger ones
    if n <= 20:
        sparsity_factor = 0.5  # Keep top 50%
    elif n <= 100:
        sparsity_factor = 0.3  # Keep top 30%
    else:
        sparsity_factor = 0.15  # Keep top 15% for large problems
    
    threshold_mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        row_values = heuristics[i]
        non_zero_mask = row_values > 0
        if non_zero_mask.sum() > 0:
            threshold = np.percentile(row_values[non_zero_mask], 
                                    100 * (1 - sparsity_factor))
            threshold_mask[i] = row_values >= threshold
    
    # Apply sparsification mask
    heuristics = heuristics * threshold_mask
    
    # 7. Boost top edges through non-linear scaling (square root to emphasize differences)
    heuristics = np.sqrt(heuristics + epsilon)
    
    # 8. Row-wise normalization for stability
    row_sums = heuristics.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1  # Avoid division by zero
    heuristics = heuristics / row_sums
    
    return heuristics
