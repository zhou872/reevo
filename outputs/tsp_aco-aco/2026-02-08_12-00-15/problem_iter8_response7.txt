import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Optimized TSP heuristics with vectorized operations, adaptive parameters,
    and efficient sparsification for edge quality indicators.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # Base heuristic: inverse distance with safe handling
    safe_dist = distance_matrix.astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    inv_dist = 1.0 / safe_dist
    np.fill_diagonal(inv_dist, 0.0)
    
    # Minimum Spanning Tree factor (vectorized)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_factor = np.where(mst_mask, 2.0, 1.0)
    
    # Nearest neighbor factor (adaptive k, fully vectorized)
    k = min(max(3, int(np.sqrt(n))), n - 1)
    if k > 0:
        # Get indices of k nearest neighbors for each node
        nearest_indices = np.argpartition(safe_dist, range(1, k + 1), axis=1)[:, 1:k + 1]
        nn_mask = np.zeros_like(distance_matrix, dtype=bool)
        np.put_along_axis(nn_mask, nearest_indices, True, axis=1)
        nn_factor = np.where(nn_mask | nn_mask.T, 1.5, 1.0)
    else:
        nn_factor = np.ones_like(distance_matrix)
    
    # Triangle inequality penalty (vectorized with sampling)
    penalty = np.ones_like(distance_matrix)
    if n > 3:
        # Adaptive sampling: more samples for larger problems
        sample_size = min(n, max(15, int(n * 0.2)))
        if sample_size < n:
            k_samples = np.random.choice(n, size=sample_size, replace=False)
        else:
            k_samples = np.arange(n)
        
        # Vectorized computation using broadcasting
        for k_idx in k_samples:
            detour_sum = safe_dist[:, k_idx][:, np.newaxis] + safe_dist[k_idx, :][np.newaxis, :]
            violations = safe_dist > detour_sum * 1.2  # Slightly stricter threshold
            penalty[violations] *= 0.9  # Milder penalty
    
    # Multiplicative combination of factors
    heuristics = inv_dist * mst_factor * nn_factor * penalty
    np.fill_diagonal(heuristics, 0.0)
    
    # Adaptive sparsification (top edges per node)
    edges_per_node = min(max(4, int(np.log2(n) * 2)), n - 1)
    keep_mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf
        top_idx = np.argpartition(row, -edges_per_node)[-edges_per_node:]
        keep_mask[i, top_idx] = True
    
    # Ensure MST edges are always kept and symmetrize
    keep_mask = (keep_mask | mst_mask)
    keep_mask = keep_mask | keep_mask.T
    heuristics = heuristics * keep_mask
    
    # Logarithmic boosting of top edges (selective)
    non_zero = heuristics[heuristics > 0]
    if len(non_zero) > 0:
        # Boost top 20% of non-zero edges
        threshold = np.percentile(non_zero, 80)
        top_mask = heuristics >= threshold
        # Apply normalized logarithmic boost
        boost_factor = np.log1p(heuristics[top_mask] * 5)
        heuristics[top_mask] *= (1.0 + boost_factor / (boost_factor.max() + 1e-8))
    
    # Final processing
    heuristics = (heuristics + heuristics.T) / 2.0  # Ensure symmetry
    np.fill_diagonal(heuristics, 0.0)
    
    # Normalize to [0, 1] range
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
