```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP combining multiple factors to evaluate edge promise.
    
    This heuristic considers:
    1. Distance reciprocal (closer nodes are better)
    2. Minimum spanning tree membership (edges in MST are promising)
    3. Local connectivity patterns (edges connecting locally dense regions)
    4. Degree constraints (penalizes edges that would force high-degree nodes)
    
    Unpromising edges are sparsified by setting them to zero.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) / (distance_matrix + np.eye(n))
    
    # 1. Base heuristic: reciprocal of distance (avoid division by zero)
    base_heuristic = 1 / (distance_matrix + np.eye(n))
    
    # 2. Minimum Spanning Tree factor
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_factor = (mst > 0).astype(float) * 2.0  # Boost for MST edges
    
    # 3. Local density factor - favor edges between nodes with many close neighbors
    # Calculate average distance to k-nearest neighbors for each node
    k = min(5, n - 1)
    neighbor_mask = np.ones_like(distance_matrix) - np.eye(n)
    sorted_dists = np.sort(distance_matrix * neighbor_mask + np.inf * np.eye(n), axis=1)
    avg_knn = np.mean(sorted_dists[:, :k], axis=1)
    
    # Nodes with small avg_knn are in dense regions
    density_weights = 1 / (avg_knn + 1e-10)
    density_factor = np.outer(density_weights, density_weights)
    
    # 4. Degree penalty - discourage edges that would lead to high-degree patterns
    # Approximate degree by considering edges that are relatively short
    degree_penalty = np.ones((n, n))
    threshold = np.percentile(distance_matrix[np.triu_indices(n, 1)], 25)
    
    for i in range(n):
        # Count potential edges for each node (edges below threshold)
        potential_edges = (distance_matrix[i] < threshold).sum() - 1  # Exclude self
        penalty = 1.0 / (1 + potential_edges / (n / 4))  # Penalty decreases with more potential edges
        degree_penalty[i, :] *= penalty
        degree_penalty[:, i] *= penalty
    
    # 5. Combine factors with weights
    combined = (base_heuristic * 0.4 + 
                mst_factor * 0.3 + 
                density_factor * 0.2 + 
                degree_penalty * 0.1)
    
    # Normalize by row to maintain scale
    row_sums = combined.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1  # Avoid division by zero
    normalized = combined / row_sums
    
    # 6. Sparsify: keep only top edges for each node
    sparsified = np.zeros_like(normalized)
    keep_per_node = max(3, n // 4)  # Keep more edges for smaller graphs
    
    for i in range(n):
        row = normalized[i].copy()
        row[i] = -np.inf  # Exclude self
        top_indices = np.argpartition(-row, keep_per_node)[:keep_per_node]
        sparsified[i, top_indices] = row[top_indices]
    
    # Make symmetric (take maximum of both directions)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Ensure diagonal is zero
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
```
This enhanced heuristic combines multiple factors to evaluate edge promise: 1) distance reciprocity, 2) membership in the minimum spanning tree, 3) local density patterns, and 4) degree constraints to prevent overly-connected nodes. It then sparsifies the matrix by keeping only the most promising edges for each node (top k based on the combined score), ensuring computational efficiency while focusing on high-potential connections.
