import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with adaptive sparsification.
    Combines MST, logarithmic nearest neighbors, triangle connectivity,
    and overlapping edge bonuses.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with safe handling
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.5  # Increased bonus for MST edges
    
    # 3. Logarithmically scaled nearest neighbors
    # Logarithmic scaling: k = max(2, ceil(log2(n)))
    k = max(2, int(np.ceil(np.log2(n))))
    k = min(k, n - 1)  # Ensure k doesn't exceed n-1
    
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]  # +1 to skip self (inf)
        nearest = nearest[nearest != i][:k]  # Take up to k neighbors
        nn_mask[i, nearest] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.8  # Slightly increased bonus
    
    # 4. Overlap bonus: edges that are in both MST and nearest neighbors
    overlap_mask = mst_mask & nn_mask
    overlap_bonus = 3.0  # Strong bonus for overlapping edges
    
    # 5. Triangle connectivity bonus
    # Boost edges that complete triangles with other promising edges
    triangle_bonus = np.ones_like(base_heuristic)
    
    # Create adjacency matrix from promising edges
    # Start with MST and NN edges as promising
    promising_mask = mst_mask | nn_mask
    
    # For each edge, check how many triangles it forms with other promising edges
    for i in range(n):
        for j in range(i+1, n):
            if promising_mask[i, j]:
                # Find common neighbors that connect to both i and j via promising edges
                common_neighbors = np.where(promising_mask[i] & promising_mask[j])[0]
                common_neighbors = common_neighbors[(common_neighbors != i) & (common_neighbors != j)]
                
                if len(common_neighbors) > 0:
                    # Bonus proportional to number of triangles
                    triangle_bonus[i, j] = triangle_bonus[j, i] = 1.0 + 0.2 * len(common_neighbors)
    
    # 6. Combine all factors
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    heuristics[overlap_mask] *= overlap_bonus
    heuristics *= triangle_bonus  # Apply triangle connectivity bonus
    
    # 7. Adaptive percentile sparsification
    # Keep edges that are highly promising based on adaptive thresholds
    # First, always keep MST and overlapping edges
    keep_mask = mst_mask | overlap_mask
    
    # Add nearest neighbor edges
    keep_mask = keep_mask | nn_mask
    
    # Adaptive percentile: keep top edges per node
    # Use lower percentile for larger graphs (sparser), higher for smaller
    adaptive_percentile = max(50, 100 - n//10)  # Ranges from 90% for n=10 to 50% for n=500
    adaptive_percentile = min(90, max(30, adaptive_percentile))  # Clamp between 30 and 90
    
    for i in range(n):
        row_values = heuristics[i]
        # Consider only non-zero, non-diagonal values
        valid_mask = (row_values > 0) & (~np.eye(n, dtype=bool)[i])
        valid_values = row_values[valid_mask]
        
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, adaptive_percentile)
            keep_mask[i] = keep_mask[i] | ((row_values >= threshold) & valid_mask)
    
    # 8. Ensure symmetry and apply mask
    keep_mask = keep_mask | keep_mask.T  # Ensure symmetric
    heuristics = heuristics * keep_mask
    
    # 9. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
