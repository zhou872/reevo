import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristics with logarithmic sparsification, triangle sampling,
    and improved overlap bonuses for MST and nearest neighbors.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with safe handling
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus (overlap-aware)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    # Dynamic MST bonus based on problem size
    mst_bonus = 2.0 + 0.5 * np.log(n) / np.log(10)
    
    # 3. Nearest neighbor bonus with logarithmic scaling
    # Adaptive k: scale with problem size but saturate
    k = int(np.clip(np.log2(n) * 2, 2, 8))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i]
        nn_mask[i, nearest[:min(k, len(nearest))]] = True
    
    # Symmetrize and add overlap bonus
    nn_mask = nn_mask | nn_mask.T
    # Boost mutual nearest neighbors
    mutual_mask = nn_mask & nn_mask.T
    nn_bonus = 1.5
    mutual_bonus = 2.0
    
    # 4. Triangle sampling for connectivity
    # Reward edges that complete short triangles
    triangle_bonus = np.ones_like(base_heuristic)
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] == 0:
                continue
            # Find best third node that minimizes triangle perimeter
            other_nodes = [k for k in range(n) if k != i and k != j]
            if other_nodes:
                perimeters = [
                    distance_matrix[i, j] + distance_matrix[i, k] + distance_matrix[j, k]
                    for k in other_nodes
                ]
                min_perimeter = min(perimeters)
                # Bonus inversely proportional to normalized triangle perimeter
                triangle_bonus[i, j] = 1.0 + 2.0 / (1.0 + min_perimeter / (3 * distance_matrix[i, j]))
                triangle_bonus[j, i] = triangle_bonus[i, j]
    
    # 5. Combine all factors
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    heuristics[mutual_mask] *= mutual_bonus
    heuristics *= triangle_bonus
    
    # 6. Logarithmic adaptive sparsification
    # Keep fewer edges for larger problems
    log_sparsity = np.clip(100 - 15 * np.log10(n), 40, 90)
    combined_mask = mst_mask | nn_mask
    
    # Add edges with high heuristic scores (logarithmic percentile)
    for i in range(n):
        row_values = heuristics[i]
        valid_values = row_values[row_values > 0]
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, log_sparsity)
            combined_mask[i] = combined_mask[i] | (row_values >= threshold)
    
    # Apply the combined mask
    heuristics = heuristics * combined_mask
    
    # 7. Smooth normalization with softmax-inspired scaling
    max_val = heuristics.max()
    if max_val > 0:
        # Soft scaling: preserve relative differences
        heuristics = np.tanh(heuristics / (max_val / 2))
    
    return heuristics
