```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic for TSP using weighted additive combination of multiple factors.
    Returns a matrix with higher values for more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (shorter edges are better)
    with np.errstate(divide='ignore', invalid='ignore'):
        base_heuristic = np.where(distance_matrix > 0, 1 / distance_matrix, 0)
    
    # 2. Nearest neighbor factor
    k_nearest = min(5, n - 1)
    nn_factor = np.zeros_like(distance_matrix)
    
    for i in range(n):
        nearest_indices = np.argsort(distance_matrix[i])[1:k_nearest+1]  # Exclude self
        nn_factor[i, nearest_indices] = 1.0
    
    # 3. Minimum spanning tree factor
    mst_factor = np.zeros_like(distance_matrix)
    try:
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_factor = ((mst + mst.T) > 0).astype(float)
    except:
        pass  # Fallback to zero if MST computation fails
    
    # 4. Triangle inequality factor
    triangle_factor = np.ones_like(distance_matrix)
    if n >= 3:
        for i in range(n):
            for j in range(i+1, n):
                if distance_matrix[i, j] > 0:
                    # Find best detour through intermediate node k
                    min_detour = np.min(distance_matrix[i] + distance_matrix[:, j] 
                                        - distance_matrix[i, j] + 1e-10)
                    # Direct edge is more promising if it's much shorter than detours
                    triangle_factor[i, j] = triangle_factor[j, i] = max(
                        0, 1 - min_detour / distance_matrix[i, j]
                    )
    
    # 5. Degree concentration factor
    degree_factor = np.ones_like(distance_matrix)
    for i in range(n):
        # Count how many edges are within 1.5x of the minimum distance
        min_dist = np.min(distance_matrix[i, distance_matrix[i] > 0])
        close_count = np.sum(distance_matrix[i] <= 1.5 * min_dist) - 1  # Exclude self
        
        if close_count > 0:
            # Penalize edges that are not among the closest
            for j in range(n):
                if i != j and distance_matrix[i, j] > 1.5 * min_dist:
                    degree_factor[i, j] = 0.5
    
    # 6. Weighted additive combination
    weights = {
        'base': 2.0,
        'nn': 1.5,
        'mst': 2.0,
        'triangle': 1.0,
        'degree': 0.8
    }
    
    combined = (
        weights['base'] * base_heuristic +
        weights['nn'] * nn_factor +
        weights['mst'] * mst_factor +
        weights['triangle'] * triangle_factor +
        weights['degree'] * degree_factor
    )
    
    # 7. Edge sparsification - keep top k edges per node
    sparsified = np.zeros_like(combined)
    keep_per_node = min(max(3, n // 4), n - 1)  # Adaptive sparsity
    
    for i in range(n):
        top_indices = np.argsort(combined[i])[-keep_per_node:]
        sparsified[i, top_indices] = combined[i, top_indices]
    
    # Ensure symmetry
    sparsified = (sparsified + sparsified.T) / 2
    
    # 8. Normalize and zero diagonal
    np.fill_diagonal(sparsified, 0)
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
```
