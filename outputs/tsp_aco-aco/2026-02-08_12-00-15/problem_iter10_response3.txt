import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge selection heuristics with MST priority and threshold-based sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Compute Minimum Spanning Tree (MST)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Calculate distance percentiles for each node
    # Compute relative position of each edge in sorted distances for each node
    percentile_matrix = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        valid_distances = safe_dist[i]
        sorted_indices = np.argsort(valid_distances)
        rank_values = np.argsort(sorted_indices)  # Convert to rank positions
        
        # Normalize to percentile (0-100) where lower values = shorter distances
        percentile_matrix[i] = (rank_values / (n - 1)) * 100
    
    # 4. Apply percentile-scaled bonuses with MST prioritization
    heuristics = base_heuristic.copy()
    
    # MST edges get highest priority with scaling based on percentile
    for i in range(n):
        for j in range(i + 1, n):
            if mst_mask[i, j]:
                # Use harmonic mean of percentiles for symmetric scaling
                # Lower percentile = shorter edge = higher bonus
                avg_percentile = np.sqrt(percentile_matrix[i, j] * percentile_matrix[j, i])
                mst_bonus = 2.5 * (1.0 - avg_percentile / 200.0)  # Bonus between 1.25-2.5
                heuristics[i, j] *= mst_bonus
                heuristics[j, i] *= mst_bonus
    
    # Non-MST edges get moderate bonuses based on percentile thresholds
    short_mask = (~mst_mask) & (percentile_matrix < 25) & (percentile_matrix.T < 25)
    medium_mask = (~mst_mask) & (~short_mask) & (percentile_matrix < 40) & (percentile_matrix.T < 40)
    
    heuristics[short_mask] *= 1.8
    heuristics[medium_mask] *= 1.3
    
    # 5. Threshold-based sparsification
    sparsified = np.zeros_like(heuristics)
    
    # Always keep MST edges
    sparsified[mst_mask] = heuristics[mst_mask]
    
    # For each node, select top non-MST edges based on adaptive thresholds
    for i in range(n):
        # Get non-MST edges for this node
        non_mst_indices = np.where(~mst_mask[i] & (np.arange(n) != i))[0]
        
        if len(non_mst_indices) > 0:
            non_mst_values = heuristics[i, non_mst_indices]
            
            # Calculate dynamic threshold based on distribution
            if len(non_mst_values) >= 5:
                # Use top 30% of non-MST edges, but at least 2
                threshold = np.percentile(non_mst_values, 70)
                keep_mask = non_mst_values >= threshold
                min_keep = max(2, len(non_mst_indices) // 4)
                
                # Ensure minimum number of edges
                if np.sum(keep_mask) < min_keep:
                    top_indices = np.argsort(non_mst_values)[-min_keep:]
                    keep_mask = np.zeros_like(non_mst_values, dtype=bool)
                    keep_mask[top_indices] = True
            else:
                # For very small sets, keep top 2 or all if less than 2
                keep_mask = np.ones_like(non_mst_values, dtype=bool)
            
            keep_indices = non_mst_indices[keep_mask]
            sparsified[i, keep_indices] = heuristics[i, keep_indices]
    
    # 6. Ensure symmetry and connectivity
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Verify each node has at least 2 edges (needed for TSP cycle)
    for i in range(n):
        connected = np.where(sparsified[i] > 0)[0]
        if len(connected) < 2:
            # Find nearest nodes not already connected
            sorted_dists = np.argsort(safe_dist[i])
            for j in sorted_dists:
                if j != i and sparsified[i, j] == 0:
                    sparsified[i, j] = base_heuristic[i, j]
                    sparsified[j, i] = base_heuristic[j, i]
                    if len(np.where(sparsified[i] > 0)[0]) >= 2:
                        break
    
    # 7. Final normalization
    np.fill_diagonal(sparsified, 0.0)
    max_val = sparsified.max()
    if max_val > 0:
        # Use softmax-style normalization: exp(heuristic/scale) for relative differences
        scale = max_val / 5.0
        sparsified = np.exp(sparsified / scale - 1.0)
        sparsified = sparsified / sparsified.max()
    
    return sparsified
