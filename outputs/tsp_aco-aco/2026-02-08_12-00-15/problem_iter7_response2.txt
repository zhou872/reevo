import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection.
    Combines MST, nearest neighbors, triangle inequality violations,
    with adaptive sparsification and logarithmic boosting.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.0
    
    # 3. Nearest neighbor bonus - vectorized
    k = max(2, min(5, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Get k nearest neighbors for each node
    sorted_indices = np.argsort(safe_dist, axis=1)
    for i in range(1, k + 1):
        nn_mask[np.arange(n), sorted_indices[:, i]] = True
    
    nn_mask = nn_mask | nn_mask.T  # Symmetrize
    nn_bonus = 1.5
    
    # 4. Triangle inequality penalty
    triangle_penalty = np.ones_like(distance_matrix)
    if n >= 3:  # Triangle inequality only meaningful for n >= 3
        for i in range(n):
            for j in range(i + 1, n):
                if safe_dist[i, j] == np.inf:
                    continue
                    
                # Find minimum violation over all intermediate nodes k
                min_violation = np.inf
                for k in range(n):
                    if k == i or k == j:
                        continue
                    if safe_dist[i, k] == np.inf or safe_dist[k, j] == np.inf:
                        continue
                    
                    # Triangle inequality: d(i,j) should be <= d(i,k) + d(k,j)
                    violation = safe_dist[i, k] + safe_dist[k, j] - safe_dist[i, j]
                    if violation < 0:  # Triangle inequality violation
                        min_violation = min(min_violation, abs(violation))
                
                if min_violation < np.inf:
                    # Penalize more severe violations more heavily
                    penalty = 1.0 / (1.0 + 0.1 * min_violation)
                    triangle_penalty[i, j] = triangle_penalty[j, i] = penalty
    
    # 5. Combine all factors multiplicatively
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    heuristics *= triangle_penalty
    
    # 6. Adaptive sparsification based on problem size
    if n > 10:  # Only sparsify for larger problems
        # Keep top edges based on adaptive threshold
        keep_per_node = max(3, min(10, n // 5))  # Adaptive number of edges to keep per node
        
        for i in range(n):
            row = heuristics[i]
            # Sort indices excluding self and already zero entries
            valid_indices = np.where((row > 0) & (np.arange(n) != i))[0]
            if len(valid_indices) > keep_per_node:
                # Get threshold for top keep_per_node edges
                sorted_values = np.sort(row[valid_indices])[::-1]
                threshold = sorted_values[keep_per_node - 1]
                # Zero out edges below threshold
                mask = (row < threshold) & (np.arange(n) != i)
                heuristics[i, mask] = 0
    
    # 7. Logarithmic boosting of top edges
    non_zero_values = heuristics[heuristics > 0]
    if len(non_zero_values) > 0:
        # Sort and apply logarithmic boost
        sorted_vals = np.sort(non_zero_values)
        ranks = np.arange(1, len(sorted_vals) + 1)
        
        # Logarithmic boosting: log(rank + 1) / log(max_rank + 1)
        boost_factors = np.log(ranks + 1) / np.log(len(sorted_vals) + 1)
        
        # Map boosting factors back to original matrix
        boost_matrix = np.zeros_like(heuristics)
        for i in range(n):
            row = heuristics[i]
            non_zero_mask = row > 0
            if np.any(non_zero_mask):
                row_vals = row[non_zero_mask]
                sorted_indices = np.argsort(row_vals)
                # Apply boost based on rank within the row
                row_ranks = np.argsort(sorted_indices) + 1
                row_boost = np.log(row_ranks + 1) / np.log(len(row_vals) + 1)
                boost_matrix[i, non_zero_mask] = row_boost
        
        # Apply boost multiplicatively
        heuristics *= (1.0 + boost_matrix * 0.5)  # 0.5 controls boost strength
    
    # 8. Ensure symmetry
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 9. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
