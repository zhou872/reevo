import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Simplified TSP edge selection heuristics with adaptive parameters
    and selective edge boosting.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    # Create safe distance matrix
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base heuristic: inverse distance
    inv_dist = 1.0 / safe_dist
    np.fill_diagonal(inv_dist, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    inv_dist[mst_mask] *= 3.0  # Strong MST bonus
    
    # 3. Adaptive nearest neighbor mask
    # Adjust k based on instance size
    k = max(3, min(15, n // 4 + 2))
    nn_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nearest = np.argsort(safe_dist[i])[1:k+1]
        nn_mask[i, nearest] = True
    
    # Symmetrize and apply bonus
    nn_mask = nn_mask | nn_mask.T
    inv_dist[nn_mask] *= 1.8  # Moderate NN bonus
    
    # 4. Selective edge boosting
    # Boost edges that are both in MST and among nearest neighbors
    boost_mask = mst_mask & nn_mask
    inv_dist[boost_mask] *= 2.0  # Extra boost for overlap
    
    # 5. Adaptive sparsification
    # Keep fewer edges for larger instances
    keep_ratio = min(0.5, max(0.2, 10.0 / n))
    edges_per_node = max(3, int(keep_ratio * n))
    
    # Create sparsity mask
    sparse_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        # Get top edges for this node
        row_vals = inv_dist[i].copy()
        row_vals[i] = -np.inf
        top_indices = np.argsort(row_vals)[-edges_per_node:]
        sparse_mask[i, top_indices] = True
    
    # Symmetrize and apply mask
    sparse_mask = sparse_mask | sparse_mask.T
    heuristics = inv_dist * sparse_mask
    
    # 6. Rank-based boosting for top edges
    for i in range(n):
        non_zero_mask = heuristics[i] > 0
        if np.sum(non_zero_mask) > 1:
            # Get non-zero values and their indices
            values = heuristics[i, non_zero_mask]
            indices = np.where(non_zero_mask)[0]
            
            # Sort and apply rank-based boost
            sorted_idx = np.argsort(values)
            for rank, idx in enumerate(sorted_idx):
                # Linear boost: top edge gets 2x, bottom gets 1x
                boost = 1.0 + (rank / (len(values) - 1))
                j = indices[idx]
                heuristics[i, j] *= boost
    
    # 7. Final symmetrization and normalization
    heuristics = (heuristics + heuristics.T) / 2.0
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0.0)
    
    return heuristics
