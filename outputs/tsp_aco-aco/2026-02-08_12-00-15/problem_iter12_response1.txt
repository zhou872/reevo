import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristics for TSP edge selection with robust indexing and 
    efficient mask handling. Combines inverse distance, MST inclusion, 
    and adaptive nearest neighbors with optimized sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with safe handling
    safe_dist = distance_matrix.astype(float, copy=True)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus (using undirected MST)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.0
    
    # 3. Nearest neighbor bonus with robust indexing
    k = max(2, min(5, n // 4, n - 1))  # Ensure k < n
    
    # Vectorized approach for nearest neighbors
    if n > k:
        # Get indices of k+1 smallest distances for each row
        sorted_indices = np.argsort(safe_dist, axis=1)[:, :k+1]
        
        # Create mask for k nearest (excluding self)
        nn_mask = np.zeros((n, n), dtype=bool)
        row_indices = np.arange(n).reshape(-1, 1)
        
        # Mark k nearest neighbors for each node
        for offset in range(1, k+1):
            if offset < sorted_indices.shape[1]:
                nn_mask[row_indices, sorted_indices[:, offset]] = True
        
        # Symmetrize efficiently
        nn_mask = nn_mask | nn_mask.T
    else:
        nn_mask = np.ones((n, n), dtype=bool) - np.eye(n, dtype=bool)
    
    nn_bonus = 1.5
    
    # 4. Combine factors efficiently
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    
    # 5. Intelligent sparsification
    # Start with MST and NN edges
    combined_mask = mst_mask | nn_mask
    
    # Add edges with high heuristic values per node
    for i in range(n):
        row_values = heuristics[i]
        
        # Find threshold for top edges, excluding already selected ones
        unselected_mask = ~combined_mask[i]
        if np.any(unselected_mask):
            unselected_values = row_values[unselected_mask]
            if len(unselected_values) > 0 and np.max(unselected_values) > 0:
                # Keep top 25% of unselected edges
                threshold = np.percentile(unselected_values[unselected_values > 0], 75)
                high_value_mask = (row_values >= threshold) & unselected_mask
                combined_mask[i, high_value_mask] = True
                combined_mask[high_value_mask, i] = True
    
    # Apply combined mask efficiently
    heuristics = heuristics * combined_mask
    
    # 6. Normalize with stability check
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
