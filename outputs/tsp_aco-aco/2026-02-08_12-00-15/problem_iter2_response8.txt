import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP combining multiple sophisticated metrics
    with efficient computation and intelligent sparsification.
    """
    n = distance_matrix.shape[0]
    
    # Handle small cases
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with smoothing
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.where(distance_matrix > 0, 1.0 / distance_matrix, 0)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Local clustering coefficient (vectorized)
    # Create adjacency matrix based on k-nearest neighbors
    k_cluster = max(3, int(np.log2(n)))
    adjacency = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        indices = np.argpartition(distance_matrix[i], k_cluster + 1)[:k_cluster + 1]
        indices = indices[indices != i][:k_cluster]
        adjacency[i, indices] = True
    
    # Compute common neighbors efficiently
    common_neighbors = adjacency @ adjacency.T
    clustering = common_neighbors / (n - 2)
    np.fill_diagonal(clustering, 0)
    
    # 3. Distance ranking with exponential decay
    k_rank = max(3, int(np.sqrt(n)))
    rank_factor = np.zeros((n, n))
    
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])
        valid_indices = sorted_indices[sorted_indices != i][:k_rank]
        ranks = np.arange(1, len(valid_indices) + 1)
        # Exponential decay: closer neighbors get much higher scores
        scores = np.exp(-ranks / (k_rank / 2))
        rank_factor[i, valid_indices] = scores / np.max(scores) if len(scores) > 0 else 0
    
    # 4. Minimum spanning tree influence
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_influence = np.zeros((n, n))
    mst_edges = np.where(mst > 0)
    for i, j in zip(*mst_edges):
        mst_influence[i, j] = 1.0
        mst_influence[j, i] = 1.0
    
    # 5. Node centrality using graph density
    row_sums = np.sum(distance_matrix, axis=1)
    centrality = 1.0 / (row_sums + 1e-10)
    centrality_matrix = np.outer(centrality, centrality)
    np.fill_diagonal(centrality_matrix, 0)
    
    # 6. Combine factors with optimized weights
    heuristic_matrix = (
        0.45 * inv_dist +
        0.20 * clustering +
        0.15 * rank_factor +
        0.10 * mst_influence +
        0.10 * centrality_matrix
    )
    
    # 7. Adaptive sparsification with hysteresis
    edges_per_node = max(3, int(np.sqrt(n) * 1.5))
    sparsified = np.zeros((n, n))
    
    for i in range(n):
        row = heuristic_matrix[i].copy()
        row[i] = -np.inf
        
        # Keep top edges plus any MST edges
        top_indices = np.argpartition(row, -edges_per_node)[-edges_per_node:]
        mst_indices = np.where(mst_influence[i] > 0)[0]
        keep_indices = np.union1d(top_indices, mst_indices)
        
        sparsified[i, keep_indices] = heuristic_matrix[i, keep_indices]
    
    # 8. Symmetrization with consistency check
    symmetric_heuristic = np.maximum(sparsified, sparsified.T)
    
    # Ensure symmetry for MST edges
    for i, j in zip(*mst_edges):
        if symmetric_heuristic[i, j] == 0:
            symmetric_heuristic[i, j] = symmetric_heuristic[j, i] = 0.5 * (
                heuristic_matrix[i, j] + heuristic_matrix[j, i]
            )
    
    # 9. Final normalization
    max_val = np.max(symmetric_heuristic)
    if max_val > 0:
        symmetric_heuristic /= max_val
    
    np.fill_diagonal(symmetric_heuristic, 0)
    return symmetric_heuristic
