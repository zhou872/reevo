import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristics with triangle inequality penalties and adaptive sparsification.
    
    Features:
    1. Multiplicative combination of MST, nearest neighbor, and triangle inequality factors
    2. Adaptive sparsification based on node degree and problem size
    3. Triangle inequality violation penalties
    4. Dynamic parameter adjustment for problem scale
    """
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix)
    
    if n <= 2:
        # Trivial cases
        heuristics = 1 / (distance_matrix + np.eye(n) * np.inf)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # =========== 1. BASELINE: Inverse distance with symmetry check ===========
    inv_dist = 1 / (distance_matrix + np.eye(n) * np.inf)
    np.fill_diagonal(inv_dist, 0)
    
    # =========== 2. MINIMUM SPANNING TREE (MST) INDICATOR ===========
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_strength = np.where(mst_mask, 2.5, 1.0)  # Strong boost for MST edges
    
    # =========== 3. ADAPTIVE NEAREST NEIGHBOR MASKS ===========
    # Dynamic k based on problem size
    k_small = max(2, min(5, n // 4))
    k_large = max(3, min(10, n // 3))
    
    nn_small = np.zeros_like(distance_matrix, dtype=bool)
    nn_large = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        sorted_idx = np.argsort(distance_matrix[i])
        # Very close neighbors (strong signal)
        nn_small[i, sorted_idx[1:k_small+1]] = True
        # Extended neighborhood (weaker signal)
        nn_large[i, sorted_idx[1:k_large+1]] = True
    
    # Symmetrize
    nn_small = nn_small | nn_small.T
    nn_large = nn_large | nn_large.T
    
    nn_strength = np.ones_like(distance_matrix)
    nn_strength[nn_large] = 1.3  # Weak boost for extended neighborhood
    nn_strength[nn_small] = 1.8  # Stronger boost for closest neighbors
    
    # =========== 4. TRIANGLE INEQUALITY PENALTY ===========
    tri_penalty = np.ones_like(distance_matrix)
    if n >= 3:
        # Check for triangle inequality violations
        for i in range(n):
            for j in range(i+1, n):
                # Find minimum detour through third node
                mask = np.ones(n, dtype=bool)
                mask[[i, j]] = False
                if np.any(mask):
                    min_detour = np.min(distance_matrix[i, mask] + distance_matrix[mask, j])
                    direct = distance_matrix[i, j]
                    
                    if min_detour > 0:
                        violation_ratio = direct / min_detour
                        # Penalize edges that are significantly longer than best detour
                        if violation_ratio > 1.15:  # 15% longer than best detour
                            penalty = np.exp(-3.0 * (violation_ratio - 1.15))
                            tri_penalty[i, j] = tri_penalty[j, i] = penalty
    
    # =========== 5. COMBINE ALL FACTORS MULTIPLICATIVELY ===========
    heuristics = inv_dist * mst_strength * nn_strength * tri_penalty
    np.fill_diagonal(heuristics, 0)
    
    # =========== 6. ADAPTIVE SPARSIFICATION ===========
    # Dynamic sparsity based on problem size
    base_sparsity = 0.4  # Keep 40% of edges
    size_factor = min(0.9, 0.4 + 0.1 * np.log10(n))  # More edges for small problems
    final_sparsity = base_sparsity * size_factor
    
    threshold_mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        row_vals = heuristics[i]
        non_zero = row_vals > 0
        
        if np.sum(non_zero) > 1:
            # Keep top edges, ensure at least 2 connections per node
            keep_count = max(2, int(np.sum(non_zero) * final_sparsity))
            keep_count = min(keep_count, n-1)
            
            if keep_count < len(row_vals):
                # Get threshold for top keep_count edges
                sorted_vals = np.sort(row_vals[non_zero])[::-1]
                if len(sorted_vals) > keep_count:
                    threshold = sorted_vals[keep_count-1]
                    threshold_mask[i] = row_vals >= threshold
                else:
                    threshold_mask[i] = non_zero
            else:
                threshold_mask[i] = non_zero
    
    # Ensure symmetry in sparsification
    threshold_mask = threshold_mask | threshold_mask.T
    
    # Apply sparsification
    heuristics = heuristics * threshold_mask
    
    # =========== 7. FINAL ENHANCEMENTS ===========
    # Boost very top edges
    if np.any(heuristics > 0):
        # Find top 5% of non-zero values and boost them
        non_zero_vals = heuristics[heuristics > 0]
        if len(non_zero_vals) > 0:
            top_threshold = np.percentile(non_zero_vals, 95)
            top_mask = heuristics >= top_threshold
            heuristics[top_mask] *= 1.2  # Additional 20% boost
    
    # Normalize for stability
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
