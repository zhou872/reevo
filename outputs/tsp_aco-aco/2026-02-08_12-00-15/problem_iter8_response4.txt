import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge selection heuristics with per-node adaptive sparsification
    and better integration of MST and nearest neighbor information.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with safe handling
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus (highly promising edges)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.5  # Increased from 2.0 to emphasize MST importance
    
    # 3. Nearest neighbor bonus with adaptive k
    # Use logarithmic scaling: more neighbors for larger problems
    k = max(2, min(8, int(np.log2(n)) + 2))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]  # +1 to skip self (inf)
        nearest = nearest[nearest != i]
        nn_mask[i, nearest[:min(k, len(nearest))]] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.75  # Increased from 1.5
    
    # 4. Combine all factors with careful handling
    heuristics = base_heuristic.copy()
    
    # Apply MST bonus (only to edges in MST)
    heuristics[mst_mask] *= mst_bonus
    
    # Apply NN bonus to edges in NN mask but not in MST
    # (to avoid excessive boosting of edges already in MST)
    nn_not_mst = nn_mask & ~mst_mask
    heuristics[nn_not_mst] *= nn_bonus
    
    # 5. Enhanced per-node sparsification
    # For each node, keep top percentile of edges based on combined heuristics
    top_percentile = 70  # Keep more edges than before for better connectivity
    node_masks = []
    
    for i in range(n):
        row_values = heuristics[i].copy()
        # Temporarily mask the diagonal to avoid considering it
        row_values[i] = -np.inf
        
        # Calculate threshold based on actual values in this row
        valid_values = row_values[row_values > 0]
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, top_percentile)
            node_mask = row_values >= threshold
        else:
            node_mask = np.zeros(n, dtype=bool)
        
        # Always include MST and nearest neighbor edges for this node
        node_mask = node_mask | mst_mask[i] | nn_mask[i]
        node_masks.append(node_mask)
    
    # Combine node masks into a symmetric mask
    combined_mask = np.zeros_like(heuristics, dtype=bool)
    for i in range(n):
        combined_mask[i] = node_masks[i]
    
    # Ensure symmetry: if edge (i,j) is kept, also keep (j,i)
    combined_mask = combined_mask | combined_mask.T
    
    # Apply the combined mask
    heuristics = heuristics * combined_mask
    
    # 6. Additional global sparsification for very dense matrices
    # Keep only the strongest connections overall
    if n > 10:
        # Calculate threshold as median of non-zero values
        non_zero_vals = heuristics[heuristics > 0]
        if len(non_zero_vals) > 0:
            global_threshold = np.percentile(non_zero_vals, 40)  # More aggressive
            heuristics[heuristics < global_threshold] = 0
    
    # 7. Final normalization for consistency
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # 8. Ensure symmetry of final heuristic matrix
    heuristics = np.maximum(heuristics, heuristics.T)
    
    return heuristics
