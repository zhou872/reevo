import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection.
    Combines MST, nearest neighbors with logarithmic scaling,
    overlapping edge boosting, and triangle sampling for connectivity.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.0
    
    # 3. Logarithmically scaled nearest neighbor bonus
    # Adaptive k based on problem size
    k = max(2, min(10, int(np.log2(n)) + 1))
    nn_heuristic = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get sorted distances (excluding self)
        sorted_indices = np.argsort(safe_dist[i])
        sorted_indices = sorted_indices[sorted_indices != i][:k]
        
        # Assign logarithmic weights (higher for closer neighbors)
        weights = 1.0 / np.log2(np.arange(2, k + 2))  # +2 for offset in log
        weights = weights[:len(sorted_indices)]
        
        nn_heuristic[i, sorted_indices] = weights
    
    # Symmetrize nearest neighbor heuristic
    nn_heuristic = np.maximum(nn_heuristic, nn_heuristic.T)
    nn_bonus = 1.0 + nn_heuristic  # Scale between 1 and ~2
    
    # 4. Overlap bonus for edges in both MST and NN
    overlap_mask = mst_mask & (nn_heuristic > 0)
    overlap_bonus = 1.3
    
    # 5. Triangle sampling for connectivity
    triangle_bonus = np.zeros_like(distance_matrix)
    
    if n >= 3:
        # Sample random triangles to boost connectivity
        num_triangles = min(100, n * (n-1) // 6)
        
        for _ in range(num_triangles):
            # Random triangle vertices
            i, j, k = np.random.choice(n, 3, replace=False)
            
            # Boost all edges of promising triangles
            triangle_edges = [(i, j), (j, k), (k, i)]
            
            for a, b in triangle_edges:
                # Only boost if edges are reasonably promising
                if base_heuristic[a, b] > np.percentile(base_heuristic, 50):
                    triangle_bonus[a, b] += 0.2
        
        # Normalize triangle bonus and apply symmetrically
        triangle_bonus = (triangle_bonus + triangle_bonus.T) / 2
        triangle_boost = 1.0 + np.tanh(triangle_bonus)  # Limit to [1, ~2]
    
    else:
        triangle_boost = 1.0
    
    # 6. Combine all factors
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics *= nn_bonus
    heuristics[overlap_mask] *= overlap_bonus
    heuristics *= triangle_boost
    
    # 7. Adaptive percentile sparsification
    # Keep top edges per node based on adaptive threshold
    for i in range(n):
        row_values = heuristics[i]
        valid_values = row_values[row_values > 0]
        
        if len(valid_values) > 0:
            # Adaptive percentile: keep more edges for smaller n
            percentile = max(60, 95 - n // 5)
            threshold = np.percentile(valid_values, percentile)
            
            # Also always keep MST and strong NN edges
            keep_mask = (row_values >= threshold) | mst_mask[i] | (nn_heuristic[i] > 0.5)
            heuristics[i, ~keep_mask] = 0.0
    
    # Ensure symmetry
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 8. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
