import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristics for TSP that combines multiple factors additively with weights
    and sparsifies by keeping only top promising edges per node.
    
    Args:
        distance_matrix: Symmetric matrix of distances between nodes.
        
    Returns:
        Heuristic matrix of same shape, with higher values indicating more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Temporary infinite diagonal to avoid self-loops in computations
    np.fill_diagonal(distance_matrix, np.inf)
    
    # Factor 1: Inverse distance (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.isnan(inv_dist)] = 0
    
    # Normalize inverse distance
    inv_dist_max = np.max(inv_dist)
    if inv_dist_max > 0:
        inv_dist = inv_dist / inv_dist_max
    
    # Factor 2: MST participation (binary indicator)
    mst_factor = np.zeros((n, n))
    try:
        mst = minimum_spanning_tree(distance_matrix)
        mst_dense = mst.toarray()
        mst_factor = (mst_dense > 0).astype(float)
    except:
        # Fallback implementation
        visited = np.zeros(n, dtype=bool)
        visited[0] = True
        for _ in range(n-1):
            min_val = np.inf
            u, v = -1, -1
            for i in range(n):
                if visited[i]:
                    for j in range(n):
                        if not visited[j] and distance_matrix[i, j] < min_val:
                            min_val = distance_matrix[i, j]
                            u, v = i, j
            if u >= 0 and v >= 0:
                mst_factor[u, v] = mst_factor[v, u] = 1
                visited[v] = True
    
    # Factor 3: Nearest neighbor rank
    k = max(3, int(np.log2(n)))  # Adaptive k
    rank_factor = np.zeros((n, n))
    for i in range(n):
        # Get indices of k+1 smallest distances (including self/inf)
        nearest_indices = np.argpartition(distance_matrix[i], k+1)[:k+1]
        nearest_indices = nearest_indices[nearest_indices != i][:k]
        
        # Assign weights based on rank
        for rank, j in enumerate(nearest_indices):
            rank_factor[i, j] = (k - rank) / k
    
    # Factor 4: Node centrality adjustment
    centrality = np.zeros(n)
    for i in range(n):
        # Average distance to other nodes as centrality measure
        centrality[i] = np.mean(distance_matrix[i, distance_matrix[i] < np.inf])
    
    # Normalize centrality
    if np.max(centrality) > np.min(centrality):
        centrality = (centrality - np.min(centrality)) / (np.max(centrality) - np.min(centrality))
    
    centrality_factor = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Favor connecting central nodes to peripheral ones
                centrality_diff = abs(centrality[i] - centrality[j])
                centrality_factor[i, j] = centrality_diff
    
    # Normalize centrality factor
    cf_max = np.max(centrality_factor)
    if cf_max > 0:
        centrality_factor = centrality_factor / cf_max
    
    # Combine factors with weights
    weights = [0.4, 0.25, 0.25, 0.1]  # Sum to 1.0
    heuristic_matrix = (
        weights[0] * inv_dist +
        weights[1] * mst_factor +
        weights[2] * rank_factor +
        weights[3] * centrality_factor
    )
    
    # Sparsification: keep only top edges per node
    edges_per_node = max(3, int(np.sqrt(n)))  # Adaptive sparsity
    sparsified = np.zeros((n, n))
    
    for i in range(n):
        row = heuristic_matrix[i].copy()
        row[i] = -np.inf  # Exclude self
        # Get indices of top edges_per_node values
        top_indices = np.argpartition(row, -edges_per_node)[-edges_per_node:]
        sparsified[i, top_indices] = heuristic_matrix[i, top_indices]
    
    # Symmetrize: keep edge if promising in either direction
    symmetric_heuristics = np.maximum(sparsified, sparsified.T)
    
    # Final normalization
    max_val = np.max(symmetric_heuristics)
    if max_val > 0:
        symmetric_heuristics = symmetric_heuristics / max_val
    
    # Restore original diagonal and set heuristic diagonal to 0
    np.fill_diagonal(distance_matrix, 0)
    np.fill_diagonal(symmetric_heuristics, 0)
    
    return symmetric_heuristics
