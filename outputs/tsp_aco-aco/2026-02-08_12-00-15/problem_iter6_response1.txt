import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    TSP edge selection heuristics focusing on MST and nearest neighbors.
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create safe distance matrix with diagonal set to infinity
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base heuristic: reciprocal of distance
    base_heuristic = 1.0 / safe_dist
    
    # 2. Minimum Spanning Tree inclusion
    mst_mask = compute_mst_mask(safe_dist)
    
    # 3. Nearest neighbors (adaptive k)
    k = max(2, n // 10)
    nn_mask = compute_nn_mask(safe_dist, k)
    
    # 4. Combine heuristics with bonuses
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= 2.0  # Strong bonus for MST edges
    heuristics[nn_mask] *= 1.5   # Moderate bonus for nearest neighbors
    
    # 5. Simple sparsification
    # Keep only edges that are promising
    sparsified = np.zeros_like(heuristics)
    
    # For each node, keep top edges
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Keep top edges (adaptive number)
        keep_count = max(2, min(5, n // 4))
        top_indices = np.argpartition(-row, keep_count)[:keep_count]
        sparsified[i, top_indices] = row[top_indices]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0.0)
    
    # 6. Normalize
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified


def compute_mst_mask(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Compute Minimum Spanning Tree mask using Prim's algorithm.
    """
    n = distance_matrix.shape[0]
    mst_mask = np.zeros((n, n), dtype=bool)
    
    if n <= 1:
        return mst_mask
    
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    
    for _ in range(n - 1):
        # Find minimum edge connecting visited to unvisited nodes
        min_dist = np.inf
        min_i, min_j = -1, -1
        
        visited_idx = np.where(visited)[0]
        unvisited_idx = np.where(~visited)[0]
        
        # Vectorized search for efficiency
        for i in visited_idx:
            row = distance_matrix[i]
            min_in_row = np.min(row[unvisited_idx])
            if min_in_row < min_dist:
                j = unvisited_idx[np.argmin(row[unvisited_idx])]
                min_dist = min_in_row
                min_i, min_j = i, j
        
        if min_i != -1:
            mst_mask[min_i, min_j] = True
            mst_mask[min_j, min_i] = True
            visited[min_j] = True
    
    return mst_mask


def compute_nn_mask(distance_matrix: np.ndarray, k: int) -> np.ndarray:
    """
    Compute k-nearest neighbors mask.
    """
    n = distance_matrix.shape[0]
    nn_mask = np.zeros((n, n), dtype=bool)
    
    if n <= 1:
        return nn_mask
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        row = distance_matrix[i].copy()
        row[i] = np.inf  # Exclude self
        nearest_indices = np.argpartition(row, k)[:k]
        nn_mask[i, nearest_indices] = True
    
    # Make symmetric (if i is near j, j is near i)
    nn_mask = nn_mask | nn_mask.T
    
    return nn_mask
