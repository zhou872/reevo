import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP heuristics with adaptive sparsification, logarithmic 
    neighbor scaling, triangle connectivity sampling, and overlap boosting.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus with boosted weight
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.5  # Increased from 2.0 to emphasize MST importance
    
    # 3. Nearest neighbor bonus with logarithmic scaling
    # Logarithmic scaling of k: log2(n) neighbors, min 2, max ~log2(100)=6.6
    k = max(2, min(6, int(np.log2(n))))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Create neighbor list with logarithmic distance weighting
    for i in range(n):
        # Get sorted distances (excluding self)
        sorted_idx = np.argsort(safe_dist[i])
        sorted_idx = sorted_idx[sorted_idx != i]
        
        # Logarithmic weighting: closer neighbors get higher bonuses
        neighbor_indices = sorted_idx[:min(k * 2, len(sorted_idx))]
        weights = 1.0 / np.log2(np.arange(2, len(neighbor_indices) + 2))
        
        # Select top k neighbors with weighted selection
        top_k = neighbor_indices[:k]
        nn_mask[i, top_k] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.5 + 0.3 * np.log2(n)  # Scale bonus logarithmically with n
    
    # 4. Boost edges that appear in both MST and NN (highly promising)
    overlap_mask = mst_mask & nn_mask
    overlap_bonus = 3.0  # Strong bonus for overlapping edges
    
    # 5. Triangle connectivity sampling
    triangle_bonus = np.zeros_like(distance_matrix)
    if n >= 3:
        # Sample nodes for triangle checking (sqrt(n) samples for efficiency)
        sample_size = min(n, max(10, int(np.sqrt(n))))
        sample_indices = np.random.choice(n, sample_size, replace=False)
        
        for i in sample_indices:
            # Find promising neighbors of i
            i_neighbors = np.where((base_heuristic[i] > 0) & 
                                 (np.arange(n) != i))[0]
            
            if len(i_neighbors) >= 2:
                # Check for triangles with high heuristic edges
                for j_idx, j in enumerate(i_neighbors):
                    for k in i_neighbors[j_idx+1:]:
                        # If both edges (i,j) and (i,k) are promising,
                        # boost the edge between j and k for connectivity
                        if base_heuristic[i,j] > 0 and base_heuristic[i,k] > 0:
                            triangle_bonus[j,k] += 0.5
                            triangle_bonus[k,j] += 0.5
    
    # 6. Combine all factors
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    heuristics[overlap_mask] *= overlap_bonus
    heuristics += triangle_bonus
    
    # 7. Adaptive percentile sparsification
    # Dynamic percentile based on problem size: 
    # Smaller problems keep more edges, larger problems keep fewer
    if n <= 10:
        percentile = 60  # Keep more edges for small problems
    elif n <= 50:
        percentile = 70
    else:
        percentile = 80  # Keep fewer edges for large problems
    
    # Create sparsification mask
    sparsity_mask = np.zeros_like(heuristics, dtype=bool)
    
    # Keep MST edges (essential for connectivity)
    sparsity_mask |= mst_mask
    
    # For each node, keep top percentile edges
    for i in range(n):
        row_values = heuristics[i]
        valid_values = row_values[row_values > 0]
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, percentile)
            sparsity_mask[i] = sparsity_mask[i] | (row_values >= threshold)
    
    # Ensure symmetry in sparsity mask
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Apply the sparsity mask
    heuristics = heuristics * sparsity_mask
    
    # 8. Soft normalization: preserve relative differences but bound extremes
    max_val = heuristics.max()
    if max_val > 0:
        # Softmax-style normalization: exp(heuristics/max_val) - 1
        # This amplifies differences between promising edges
        normalized = np.exp(heuristics / max_val) - 1
        heuristics = normalized / normalized.max()
    
    return heuristics
