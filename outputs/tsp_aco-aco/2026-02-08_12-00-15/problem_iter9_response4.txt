import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection.
    Combines multiple factors with adaptive sparsification,
    logarithmic boosting of overlaps, and triangle sampling.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix where higher values indicate more
        promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic from inverse distance
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree inclusion
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Adaptive nearest neighbors
    k = max(2, min(6, n // 3))  # Slightly more neighbors
    nn_rank = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        sorted_indices = np.argsort(safe_dist[i])
        # Assign scores based on rank (closer = higher score)
        ranks = np.zeros(n)
        ranks[sorted_indices[1:k+1]] = np.arange(k, 0, -1) / k  # Normalized
        nn_rank[i] = ranks
    
    # 4. Create overlap matrix (counting heuristic agreements)
    overlap_count = np.zeros_like(base_heuristic)
    
    # Top edges from base heuristic (per node)
    top_base_edges = np.zeros_like(base_heuristic, dtype=bool)
    for i in range(n):
        row = base_heuristic[i]
        threshold = np.percentile(row[row > 0], 70)  # Top 30%
        top_base_edges[i] = row >= threshold
    
    # Count overlaps
    overlap_count += mst_mask.astype(float)
    overlap_count += (nn_rank > 0).astype(float)
    overlap_count += top_base_edges.astype(float)
    
    # 5. Logarithmic boost for overlaps (more overlaps = stronger boost)
    boost_factor = 1.0 + np.log1p(overlap_count)  # log(1+x) to handle zero overlaps
    
    # 6. Combine factors with adaptive weights
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= 2.0  # MST gets fixed strong boost
    heuristics += nn_rank * 0.3  # Add nearest neighbor contribution
    heuristics *= boost_factor   # Apply overlap boost
    
    # 7. Sparsification with adaptive thresholds
    final_mask = np.zeros_like(heuristics, dtype=bool)
    
    # Keep edges that appear in multiple heuristics
    multi_heuristic_edges = overlap_count >= 2
    final_mask |= multi_heuristic_edges
    
    # Keep top edges per node based on percentile
    for i in range(n):
        row = heuristics[i]
        if np.any(row > 0):
            threshold = np.percentile(row[row > 0], 80)  # Top 20%
            final_mask[i] |= row >= threshold
    
    # 8. Triangle inequality sampling (check promising triangles)
    sample_size = min(1000, n * 10)
    sampled = np.random.choice(n, size=(sample_size, 3), replace=True)
    
    for i, j, k in sampled:
        if i == j or i == k or j == k:
            continue
            
        # Check triangle inequality for promising edges
        if final_mask[i, j] and final_mask[i, k] and final_mask[j, k]:
            # If edge violates triangle inequality, reduce its score
            if distance_matrix[i, j] > distance_matrix[i, k] + distance_matrix[k, j]:
                heuristics[i, j] *= 0.7
                heuristics[j, i] *= 0.7
    
    # Apply final mask
    heuristics = heuristics * final_mask
    
    # 9. Ensure symmetry (take max of both directions)
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 10. Normalize
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
