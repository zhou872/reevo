import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP combining multiple factors:
    1. Inverse distance (closer nodes are more promising)
    2. Sparsification based on relative distance thresholds
    3. Local connectivity patterns
    4. Node centrality measures
    """
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
    
    # Avoid division by zero for diagonal elements
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
    
    # Factor 1: Inverse distance as baseline
    heuristics_matrix += inv_dist
    
    # Factor 2: Local clustering coefficient
    # Nodes that are well-connected to each other's neighbors are promising
    k_nearest = min(5, n // 2)  # Number of nearest neighbors to consider
    nearest_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        nearest_indices = np.argsort(distance_matrix[i])[1:k_nearest+1]
        nearest_mask[i, nearest_indices] = True
    
    # Symmetrize the mask
    nearest_mask = nearest_mask | nearest_mask.T
    
    # Factor 3: Path saving potential
    # Prefer edges that connect nodes with few good alternatives
    node_alternatives = np.sum(nearest_mask, axis=1)
    path_saving = np.outer(node_alternatives, node_alternatives)
    path_saving = 1 / (1 + path_saving)  # Lower alternatives = higher score
    
    # Factor 4: Edge betweenness approximation
    # Edges that connect different clusters are often important
    cluster_score = np.zeros_like(distance_matrix)
    if n > 10:
        # Simple cluster approximation using farthest insertion
        cluster_centers = [0]  # Start with first node
        for _ in range(min(3, n // 3)):
            # Find node farthest from all current centers
            distances_to_centers = np.min(distance_matrix[cluster_centers], axis=0)
            new_center = np.argmax(distances_to_centers)
            cluster_centers.append(new_center)
        
        # Assign each node to nearest center
        assignments = np.argmin(distance_matrix[cluster_centers], axis=0)
        
        # Score edges higher if they connect different clusters
        for i in range(n):
            for j in range(i+1, n):
                if assignments[i] != assignments[j]:
                    cluster_score[i, j] = cluster_score[j, i] = 1.0
    
    # Combine all factors with weights
    combined = (
        0.4 * heuristics_matrix +  # Inverse distance
        0.3 * nearest_mask.astype(float) * inv_dist +  # Local connectivity
        0.2 * path_saving +  # Path saving potential
        0.1 * cluster_score  # Between-cluster connections
    )
    
    # Sparsification: Keep only top promising edges
    threshold_percentile = 75  # Keep top 25% of edges
    threshold = np.percentile(combined[combined > 0], threshold_percentile)
    combined[combined < threshold] = 0
    
    # Ensure symmetry
    combined = np.maximum(combined, combined.T)
    
    # Normalize to make scores comparable across instances
    if np.sum(combined) > 0:
        combined = combined / np.max(combined)
    
    return combined
