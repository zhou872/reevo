import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge selection heuristics with adaptive sparsification.
    Combines multiple factors: MST inclusion, nearest neighbors,
    triangle inequality violations, and logarithmic boosting.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create safe distance matrix (no zeros on diagonal)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base heuristic: inverse distance
    inv_dist = 1.0 / safe_dist
    np.fill_diagonal(inv_dist, 0.0)
    
    # 2. Minimum Spanning Tree mask
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbor mask with adaptive k
    k = max(2, min(10, n // 6 + 3))
    nn_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i]
        nn_mask[i, nearest[:min(k, len(nearest))]] = True
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Triangle inequality violation penalty
    tri_penalty = np.ones((n, n))
    # Vectorized triangle inequality check
    for k in range(n):
        # Compute indirect distances through node k
        indirect = safe_dist[:, k:k+1] + safe_dist[k:k+1, :]
        # Find violations where direct > indirect (ignore diagonal)
        violations = (safe_dist > indirect) & (~np.eye(n, dtype=bool))
        # Penalize violating edges
        tri_penalty[violations] *= 0.3  # Strong penalty for violations
    
    # 5. Combine factors multiplicatively
    heuristics = inv_dist.copy()
    heuristics[mst_mask] *= 2.0  # MST edges are highly promising
    heuristics[nn_mask] *= 1.5   # Nearest neighbors are promising
    heuristics *= tri_penalty    # Apply triangle violation penalty
    
    # 6. Adaptive sparsification per node
    sparsity_factor = min(0.7, max(0.3, 1.0 - np.log(n) / 10))
    keep_per_node = max(3, int(sparsity_factor * n / 2))
    
    # Create sparsification mask
    sparse_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        # Get top edges for this node
        row_vals = heuristics[i].copy()
        row_vals[i] = -np.inf  # Ignore self
        top_indices = np.argsort(row_vals)[-keep_per_node:]
        sparse_mask[i, top_indices] = True
    
    # Symmetrize the mask
    sparse_mask = sparse_mask | sparse_mask.T
    
    # Apply sparsification
    heuristics = heuristics * sparse_mask
    
    # 7. Logarithmic boosting of top edges
    if np.count_nonzero(heuristics) > 0:
        non_zero = heuristics[heuristics > 0]
        if len(non_zero) > 0:
            threshold = np.percentile(non_zero, 80)
            boost_mask = heuristics >= threshold
            # Apply logarithmic boosting: log(1 + value/epsilon)
            epsilon = 1e-10
            boost_values = np.log1p(heuristics[boost_mask] / epsilon)
            heuristics[boost_mask] *= (1.0 + 0.5 * boost_values / np.max(boost_values))
    
    # 8. Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2.0
    
    # 9. Normalize
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
