```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    TSP edge heuristics with adaptive per-node sparsification and emphasis on
    Minimum Spanning Tree and nearest neighbor edges. Returns normalized scores
    where higher values indicate edges more likely to be in optimal TSP solutions.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix with same shape as input, values in [0, 1].
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Create safe distance matrix (diagonal = infinity)
    safe_dist = distance_matrix.astype(float).copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 2. Base heuristic: inverse distance
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 3. Minimum Spanning Tree edges (strong preference)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 4. Adaptive nearest neighbor edges
    # k adapts to problem size: more neighbors for larger problems
    k = min(max(3, n // 8), 10)
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[1:k+1]  # k nearest (skip self)
        nn_mask[i, nearest] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    
    # 5. Combine heuristics with edge-specific bonuses
    heuristics = base_heuristic.copy()
    
    # Apply bonuses: MST edges get highest, NN edges get moderate
    heuristics[mst_mask] *= 2.5
    heuristics[nn_mask] *= 1.8
    
    # 6. Adaptive per-node sparsification
    # Keep different number of edges per node based on problem size
    sparsified = np.zeros_like(heuristics)
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Determine how many edges to keep per node
        # Base: 3-8 edges depending on problem size
        keep_count = min(max(3, n // 6), 8)
        
        # Keep top edges for this node
        top_indices = np.argpartition(-row, keep_count)[:keep_count]
        sparsified[i, top_indices] = row[top_indices]
        
        # Always include MST edges even if not in top-k
        mst_neighbors = np.where(mst_mask[i])[0]
        sparsified[i, mst_neighbors] = np.maximum(
            sparsified[i, mst_neighbors], 
            heuristics[i, mst_neighbors]
        )
    
    # 7. Ensure symmetry (if edge is good for either node, keep it)
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0.0)
    
    # 8. Normalize to [0, 1] range
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
```
