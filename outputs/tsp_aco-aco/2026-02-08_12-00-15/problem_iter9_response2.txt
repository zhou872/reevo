import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristics with adaptive MST-nearest neighbor combination,
    logarithmic rank boosting, percentile-based sparsification, and 
    triangle inequality considerations.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with safe handling
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. MST bonus with adaptive weighting
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # Adaptive MST bonus based on problem size
    mst_bonus = 1.0 + np.log1p(n) / 10.0  # Logarithmic scaling
    
    # 3. Nearest neighbors with adaptive k
    # k scales logarithmically with n, with bounds
    k = max(2, min(8, int(np.ceil(np.log2(n)) + 2)))
    nn_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k+1 to skip self
        nearest_indices = np.argsort(safe_dist[i])[:k+1]
        nearest_indices = nearest_indices[nearest_indices != i][:k]
        nn_mask[i, nearest_indices] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    
    # Adaptive NN bonus based on rank position
    nn_bonus_matrix = np.ones_like(base_heuristic)
    for i in range(n):
        row_distances = safe_dist[i].copy()
        row_distances[i] = np.inf
        sorted_indices = np.argsort(row_distances)
        
        # Logarithmic rank-based bonus: higher bonus for closer neighbors
        for rank, j in enumerate(sorted_indices[:k], 1):
            bonus = 1.0 + 0.5 * np.log1p(k - rank + 1)
            nn_bonus_matrix[i, j] *= bonus
    
    # 4. Overlap bonus for edges that appear in both MST and NN
    overlap_mask = mst_mask & nn_mask
    overlap_bonus = 1.0 + np.log1p(n) / 20.0  # Smaller bonus than individual factors
    
    # 5. Triangle inequality consideration
    # Boost edges that form "good triangles" with other promising edges
    triangle_bonus = np.ones_like(base_heuristic)
    
    # Sample a subset of nodes for triangle checks (for efficiency)
    sample_size = min(10, n)
    sample_indices = np.random.choice(n, size=sample_size, replace=False)
    
    for i in range(n):
        for k_sample in sample_indices:
            if i == k_sample:
                continue
            
            # Check if triangle inequality gives competitive alternative
            for j in range(i+1, n):
                if j == k_sample:
                    continue
                
                # If direct edge is not much worse than two-hop path
                direct = safe_dist[i, j]
                indirect = safe_dist[i, k_sample] + safe_dist[k_sample, j]
                
                if direct <= indirect * 1.2:  # Within 20% of indirect path
                    # Boost edges that are part of competitive triangles
                    triangle_bonus[i, j] *= 1.05
                    triangle_bonus[j, i] *= 1.05
    
    # 6. Combine all factors
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics *= nn_bonus_matrix  # Apply rank-based NN bonuses
    heuristics[overlap_mask] *= overlap_bonus
    heuristics *= triangle_bonus
    
    # 7. Adaptive percentile-based sparsification
    # Use tighter threshold for larger problems
    base_percentile = 75
    adaptive_percentile = base_percentile - min(20, n // 10)
    adaptive_percentile = max(50, adaptive_percentile)
    
    combined_mask = mst_mask | nn_mask
    
    # Keep top edges per node based on adaptive percentile
    for i in range(n):
        row_values = heuristics[i].copy()
        valid_values = row_values[row_values > 0]
        
        if len(valid_values) > 1:
            threshold = np.percentile(valid_values, adaptive_percentile)
            row_mask = row_values >= threshold
            combined_mask[i] = combined_mask[i] | row_mask
            combined_mask[:, i] = combined_mask[:, i] | row_mask  # Ensure symmetry
    
    # Also keep edges that are very strong (top 5%)
    strong_threshold = np.percentile(heuristics[heuristics > 0], 95) if np.any(heuristics > 0) else 0
    strong_mask = heuristics >= strong_threshold
    combined_mask = combined_mask | strong_mask
    
    # 8. Ensure symmetry and apply sparsification
    combined_mask = np.maximum(combined_mask, combined_mask.T)
    heuristics = heuristics * combined_mask
    
    # 9. Final normalization with logarithmic scaling
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # 10. Apply final non-linear transformation to emphasize differences
    heuristics = np.sqrt(heuristics)  # Square root to compress range
    
    return heuristics
