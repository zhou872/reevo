import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristics combining MST, nearest neighbors, and triangle inequality.
    Features adaptive sparsification, edge boosting, and problem-size-dependent parameters.
    """
    n = distance_matrix.shape[0]
    np.fill_diagonal(distance_matrix, np.inf)  # Remove self-connections
    
    # 1. Baseline: inverse distance with stability
    inv_dist = 1 / distance_matrix
    
    # 2. MST inclusion bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.0
    
    # 3. Adaptive nearest neighbor connectivity
    k = max(2, int(np.log2(n)) + 2)  # Log-scale adaptation
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nearest = np.argsort(distance_matrix[i])[:k]
        nn_mask[i, nearest] = True
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.5
    
    # 4. Triangle inequality penalties
    tri_penalty = np.ones_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            # Find best intermediate node k for triangle inequality
            min_via = np.min(distance_matrix[i] + distance_matrix[:, j])
            if min_via < distance_matrix[i, j]:
                # Penalize if direct edge violates triangle inequality
                tri_penalty[i, j] = tri_penalty[j, i] = 0.8
    
    # 5. Combine factors multiplicatively
    heuristics = inv_dist.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    heuristics *= tri_penalty
    
    # 6. Adaptive sparsification with problem-size-dependent threshold
    if n > 20:
        # For larger problems, be more aggressive
        top_k_per_node = max(3, n // 5)
        threshold_mask = np.zeros_like(heuristics, dtype=bool)
        for i in range(n):
            top_indices = np.argsort(heuristics[i])[-top_k_per_node:]
            threshold_mask[i, top_indices] = True
    else:
        # For smaller problems, keep more connections
        sparsity_factor = 0.4
        threshold_mask = np.zeros_like(heuristics, dtype=bool)
        for i in range(n):
            row_values = heuristics[i]
            if np.count_nonzero(row_values) > 0:
                threshold = np.percentile(row_values[row_values > 0], 
                                        100 * (1 - sparsity_factor))
                threshold_mask[i] = row_values >= threshold
    
    heuristics = heuristics * threshold_mask
    
    # 7. Edge boosting for top connections
    if n > 10:
        for i in range(n):
            row = heuristics[i]
            non_zero = row[row > 0]
            if len(non_zero) > 0:
                median_val = np.median(non_zero)
                # Boost edges significantly above median
                boost_mask = row > 2 * median_val
                heuristics[i, boost_mask] *= 1.2
    
    # 8. Final normalization and cleanup
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    np.fill_diagonal(heuristics, 0)
    return heuristics
