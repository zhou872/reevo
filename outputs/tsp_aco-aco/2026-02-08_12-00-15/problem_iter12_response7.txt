import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Optimized heuristics for TSP edge selection focusing on MST and nearest neighbors.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus (core component)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.0
    
    # 3. Nearest neighbor bonus with adaptive k (core component)
    # Scale k with log(n) to maintain balance between connectivity and sparsity
    k = max(2, int(np.log2(n)) + 1)
    k = min(k, n - 1)  # Ensure k doesn't exceed n-1
    
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]  # +1 to account for self
        nearest = nearest[nearest != i][:k]  # Remove self and take k
        nn_mask[i, nearest] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.5
    
    # 4. Combine core heuristics
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    
    # 5. Consistent sparsification
    # Keep only MST edges, nearest neighbor edges, and edges with strong base heuristic
    combined_mask = mst_mask | nn_mask
    
    # Add edges with base heuristic in top quartile (excluding diagonal)
    if n > 2:
        base_values = base_heuristic[np.triu_indices(n, k=1)]
        threshold = np.percentile(base_values, 75)
        high_base_mask = base_heuristic >= threshold
        combined_mask = combined_mask | high_base_mask
    
    # Apply sparsification mask
    heuristics = heuristics * combined_mask
    
    # 6. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
