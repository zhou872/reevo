import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge selection heuristics with logarithmic NN scaling,
    percentile sparsification, triangle sampling, and edge overlap boosting.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.0  # MST edges are highly promising
    
    # 3. Logarithmically scaled nearest neighbor bonus
    # Adaptive k: log-scale with problem size
    k = max(2, int(np.log2(n)) + 2)
    nn_heuristic = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        # Get distances to all other nodes and sort
        distances = safe_dist[i].copy()
        sorted_indices = np.argsort(distances)
        
        # Assign logarithmic bonuses (higher for closer nodes)
        for rank, j in enumerate(sorted_indices[:k+1], 1):
            if j == i or rank > k:
                continue
            # Logarithmic decay: 1/log(rank+1) gives higher weight to closer nodes
            nn_heuristic[i, j] = 1.0 / np.log2(rank + 1)
    
    # Symmetrize nearest neighbor heuristic
    nn_heuristic = np.maximum(nn_heuristic, nn_heuristic.T)
    nn_bonus_scale = 1.8  # Slightly higher than v1
    
    # 4. Triangle connectivity sampling
    triangle_boost = np.zeros_like(distance_matrix, dtype=float)
    if n >= 3:
        # Sample random triangles to boost edges that complete promising triangles
        num_samples = min(100, n * (n-1) // 2)
        for _ in range(num_samples):
            i, j = np.random.choice(n, 2, replace=False)
            # Find third node that forms promising triangle with i and j
            k_candidates = [k for k in range(n) if k not in (i, j)]
            if k_candidates:
                k = np.random.choice(k_candidates)
                # Boost if both edges (i,k) and (j,k) are promising
                if base_heuristic[i, k] > 0 and base_heuristic[j, k] > 0:
                    triangle_boost[i, j] += 0.3
                    triangle_boost[j, i] += 0.3
        
        # Normalize triangle boost
        if triangle_boost.max() > 0:
            triangle_boost = triangle_boost / triangle_boost.max()
    
    # 5. Overlap boosting: edges that appear in multiple promising categories
    overlap_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Boost edges that are both in MST and top nearest neighbors
    strong_nn_mask = nn_heuristic > np.percentile(nn_heuristic[nn_heuristic > 0], 70)
    overlap_mask = mst_mask & strong_nn_mask
    
    # 6. Combine all factors with adaptive weights
    heuristics = base_heuristic.copy()
    
    # Apply MST bonus
    heuristics[mst_mask] *= mst_bonus
    
    # Add nearest neighbor contribution (additive to preserve inverse distance scaling)
    heuristics += nn_heuristic * nn_bonus_scale
    
    # Apply triangle connectivity boost
    heuristics += triangle_boost * 0.5
    
    # Boost overlapping edges
    heuristics[overlap_mask] *= 1.5
    
    # 7. Adaptive percentile sparsification
    # Keep edges that are in MST, strong NN, or high combined score
    combined_mask = np.zeros_like(heuristics, dtype=bool)
    
    # Always keep MST edges
    combined_mask |= mst_mask
    
    # Keep top nearest neighbors (based on NN heuristic, not combined score)
    nn_threshold = np.percentile(nn_heuristic[nn_heuristic > 0], 60)
    combined_mask |= (nn_heuristic >= nn_threshold)
    
    # Adaptive percentile for combined heuristics
    # Use stricter sparsification for larger problems
    adaptive_percentile = max(70, 85 - min(15, n // 20))
    
    # Keep top edges per node based on combined heuristics
    for i in range(n):
        row_values = heuristics[i]
        valid_values = row_values[row_values > 0]
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, adaptive_percentile)
            combined_mask[i] = combined_mask[i] | (row_values >= threshold)
    
    # Apply the combined mask
    heuristics = heuristics * combined_mask
    
    # 8. Ensure symmetry (average with transpose)
    heuristics = (heuristics + heuristics.T) / 2
    
    # 9. Normalize for consistency
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
