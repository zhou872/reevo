import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic for TSP edge selection that combines multiple factors
    for stochastic solution sampling. Returns a matrix where higher values
    indicate more promising edges to include in solutions.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create safe distance matrix with diagonal set to infinity
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base heuristic: reciprocal of distance
    base_heuristic = 1.0 / safe_dist
    
    # 2. Triangle inequality penalty
    # Penalize edges that are much longer than alternative paths
    tri_penalty = np.ones_like(safe_dist)
    
    # Efficient computation using broadcasting
    for i in range(n):
        # Compute alternative paths through intermediate nodes k
        # dist(i,k) + dist(k,j) for all k
        alt_paths = safe_dist[i:i+1, :] + safe_dist
        np.fill_diagonal(alt_paths, np.inf)  # Ignore k=i or k=j cases
        
        min_alt_path = np.min(alt_paths, axis=1)
        direct_dist = safe_dist[i, :]
        
        # Apply penalty based on violation ratio
        with np.errstate(invalid='ignore', divide='ignore'):
            ratio = direct_dist / min_alt_path
        
        # Strong penalty for significant violations
        mask = ratio > 1.5
        tri_penalty[i, mask] = 0.2
        # Moderate penalty for smaller violations
        mask = (ratio > 1.2) & (ratio <= 1.5)
        tri_penalty[i, mask] = 0.5
        
        # Small penalty for slight violations
        mask = (ratio > 1.05) & (ratio <= 1.2)
        tri_penalty[i, mask] = 0.8
    
    np.fill_diagonal(tri_penalty, 0)
    
    # 3. Spatial clustering bonus
    # Use DBSCAN-like approach to identify natural clusters
    cluster_bonus = np.ones_like(safe_dist)
    
    if n >= 4:
        # Estimate density-based clusters
        # Compute average distance to k nearest neighbors for each point
        k = max(2, min(5, n // 4))
        knn_dists = np.zeros(n)
        
        for i in range(n):
            # Get k+1 smallest distances (including self/infinity)
            sorted_dists = np.sort(safe_dist[i])
            knn_dists[i] = np.mean(sorted_dists[1:k+1])  # Exclude self/infinity
        
        # Use median of knn distances as density threshold
        eps = np.median(knn_dists) * 1.5
        
        # Identify core points (high density)
        is_core = knn_dists <= eps
        core_indices = np.where(is_core)[0]
        border_indices = np.where(~is_core)[0]
        
        if len(core_indices) > 0 and len(border_indices) > 0:
            # Bonus for edges within same cluster type
            for i in range(n):
                for j in range(i+1, n):
                    if is_core[i] and is_core[j]:
                        # Both core points in same dense region
                        cluster_bonus[i, j] = cluster_bonus[j, i] = 1.5
                    elif not is_core[i] and not is_core[j]:
                        # Both border points
                        cluster_bonus[i, j] = cluster_bonus[j, i] = 1.2
                    else:
                        # Mixed: core-border connection
                        cluster_bonus[i, j] = cluster_bonus[j, i] = 0.8
    
    # 4. Degree control - limit promising edges per node
    # Compute initial scores
    initial_scores = base_heuristic * tri_penalty * cluster_bonus
    
    # Adaptive degree limit based on problem size
    max_degree = max(3, min(8, n // 2))
    degree_mask = np.ones_like(initial_scores)
    
    for i in range(n):
        row_scores = initial_scores[i].copy()
        row_scores[i] = -np.inf  # Ignore self
        
        # Get top max_degree edges
        if np.sum(row_scores > 0) > max_degree:
            # Use argpartition for efficient top-k selection
            top_indices = np.argpartition(-row_scores, max_degree)[:max_degree]
            threshold = np.min(row_scores[top_indices])
            
            # Penalize edges below threshold
            degree_mask[i, row_scores < threshold] = 0.1
            # Slightly boost edges above threshold
            degree_mask[i, row_scores >= threshold] = 1.2
    
    # 5. Combine all factors with degree control
    combined = base_heuristic * tri_penalty * cluster_bonus * degree_mask
    
    # 6. Structural masks: Nearest Neighbors and Minimum Spanning Tree
    # Nearest neighbors mask
    k_nn = max(2, min(5, n // 4))
    nn_mask = np.zeros_like(safe_dist, dtype=bool)
    
    for i in range(n):
        # Get k+1 smallest distances
        nearest = np.argpartition(safe_dist[i], k_nn+1)[:k_nn+1]
        nearest = nearest[nearest != i]  # Remove self
        if len(nearest) > k_nn:
            nearest = nearest[:k_nn]
        nn_mask[i, nearest] = True
    
    # Make symmetric
    nn_mask = nn_mask | nn_mask.T
    
    # Minimum Spanning Tree mask
    mst_mask = np.zeros_like(safe_dist, dtype=bool)
    if n > 1:
        # Use Prim's algorithm
        visited = np.zeros(n, dtype=bool)
        visited[0] = True
        
        for _ in range(n - 1):
            # Vectorized min edge search
            visited_idx = np.where(visited)[0]
            unvisited_idx = np.where(~visited)[0]
            
            if len(visited_idx) == 0 or len(unvisited_idx) == 0:
                break
            
            # Create submatrix of distances between visited and unvisited
            sub_dist = safe_dist[visited_idx][:, unvisited_idx]
            min_val = np.min(sub_dist)
            
            # Find the edge
            min_pos = np.where(sub_dist == min_val)
            if len(min_pos[0]) > 0:
                i_idx = visited_idx[min_pos[0][0]]
                j_idx = unvisited_idx[min_pos[1][0]]
                mst_mask[i_idx, j_idx] = True
                mst_mask[j_idx, i_idx] = True
                visited[j_idx] = True
    
    # 7. Combine masks and apply sparsification
    # Always include NN and MST edges
    final_mask = nn_mask | mst_mask
    
    # Also include top edges based on combined score
    if n > 2:
        # Flatten scores excluding diagonal
        mask_nodiag = ~np.eye(n, dtype=bool)
        flat_scores = combined[mask_nodiag]
        
        if len(flat_scores) > 0 and np.max(flat_scores) > 0:
            # Include edges above 70th percentile of positive scores
            pos_scores = flat_scores[flat_scores > 0]
            if len(pos_scores) > 0:
                threshold = np.percentile(pos_scores, 70)
                score_mask = combined > threshold
                final_mask = final_mask | score_mask
    
    # Apply final mask
    result = combined * final_mask
    
    # 8. Normalization to base heuristic scale
    # Compute average scale of base heuristic (excluding diagonal/inf)
    base_nonzero = base_heuristic[~np.eye(n, dtype=bool) & (base_heuristic > 0)]
    if len(base_nonzero) > 0:
        base_mean = np.mean(base_nonzero)
    else:
        base_mean = 1.0
    
    # Compute average scale of result (excluding zeros)
    result_nonzero = result[result > 0]
    if len(result_nonzero) > 0:
        result_mean = np.mean(result_nonzero)
        if result_mean > 0:
            # Scale result to have similar magnitude as base heuristic
            result = result * (base_mean / result_mean)
    
    # Ensure diagonal is zero
    np.fill_diagonal(result, 0)
    
    return result
