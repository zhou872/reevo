import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP heuristics with triangle sampling and logarithmic sparsification.
    
    Enhances v1 by:
    1. Adding triangle sampling for connectivity insights
    2. Logarithmic adaptive percentile for sparsification
    3. Overlap bonuses for edges in both MST and nearest neighbors
    4. More nuanced edge scoring
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Safe distance matrix for computations
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base heuristic: inverse distance
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbor bonus
    k = max(2, min(5, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i]
        nn_mask[i, nearest[:min(k, len(nearest))]] = True
    
    nn_mask = nn_mask | nn_mask.T  # Symmetrize
    
    # 4. Overlap bonus (edges in both MST and nearest neighbors)
    overlap_mask = mst_mask & nn_mask
    
    # 5. Triangle sampling for connectivity insights
    triangle_bonus = np.ones_like(distance_matrix)
    
    if n >= 3:
        # Sample triangles for connectivity information
        sample_size = min(20, n)  # Adaptive sampling
        
        for i in range(n):
            # Get nearest neighbors for triangle sampling
            neighbors = np.argsort(safe_dist[i])[1:sample_size+1]
            
            for j in range(i+1, n):
                if base_heuristic[i, j] > 0:  # Only consider plausible edges
                    # Find common connections
                    common = np.intersect1d(neighbors, 
                                           np.argsort(safe_dist[j])[1:sample_size+1])
                    
                    if len(common) > 0:
                        # Score based on triangle efficiency
                        triangle_scores = []
                        for k in common[:min(5, len(common))]:  # Limit to top 5
                            if k != i and k != j:
                                # Triangle inequality efficiency
                                direct = safe_dist[i, j]
                                indirect = safe_dist[i, k] + safe_dist[k, j]
                                if indirect > 0:
                                    efficiency = direct / indirect
                                    triangle_scores.append(max(0.1, efficiency))
                        
                        if triangle_scores:
                            # Use median for robustness
                            triangle_bonus[i, j] = triangle_bonus[j, i] = np.median(triangle_scores)
    
    # 6. Apply bonuses with overlap priority
    heuristics = base_heuristic.copy()
    
    # Base bonuses
    heuristics[mst_mask] *= 1.8  # MST edges
    heuristics[nn_mask] *= 1.4   # Nearest neighbors
    heuristics[overlap_mask] *= 2.2  # Overlap bonus
    
    # Triangle bonus (multiplicative)
    heuristics *= triangle_bonus
    
    # 7. Logarithmic adaptive sparsification
    # Keep fewer edges for larger graphs
    log_factor = max(0.1, 1.0 - 0.15 * np.log10(max(2, n)))
    keep_ratio = 0.25 * log_factor  # Adaptive percentage
    
    combined_mask = mst_mask | nn_mask | overlap_mask
    
    # Adaptive percentile based on graph size
    adaptive_percentile = 100 * (1.0 - keep_ratio)
    
    for i in range(n):
        row_values = heuristics[i]
        valid_values = row_values[row_values > 0]
        
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, adaptive_percentile)
            # Keep top edges plus MST/NN/overlap edges
            keep_mask = (row_values >= threshold) | combined_mask[i]
            
            # Apply sparsification
            heuristics[i, ~keep_mask] = 0
    
    # Ensure symmetry
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 8. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
