import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP heuristics with efficient computation and adaptive sparsification.
    
    Features:
    1. Multiplicative combination of MST, nearest neighbor, and inverse distance factors
    2. Efficient triangle inequality violation detection using vectorized operations
    3. Adaptive sparsification based on problem size
    4. Non-linear scaling to emphasize promising edges
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between cities
        
    Returns:
    --------
    np.ndarray
        Heuristic scores for each edge (same shape as input)
    """
    n = distance_matrix.shape[0]
    
    # 1. Baseline: inverse distance (avoid division by zero)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.where(distance_matrix > 0, 1.0 / distance_matrix, 0)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree factor
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_factor = np.ones_like(distance_matrix)
    mst_edges = np.where(mst + mst.T > 0)
    mst_factor[mst_edges] = 2.0  # Boost MST edges
    
    # 3. Nearest neighbor connectivity (adaptive k)
    k_nn = max(2, min(10, n // 5))
    nn_factor = np.ones_like(distance_matrix)
    
    # Vectorized nearest neighbor identification
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        neighbors = np.argsort(distance_matrix[i])[1:k_nn+1]
        nn_factor[i, neighbors] = 1.5
    
    # 4. Efficient triangle inequality penalty
    tri_penalty = np.ones_like(distance_matrix)
    
    if n > 3:  # Only compute for non-trivial cases
        epsilon = 0.15  # Tolerance for triangle inequality
        
        # Precompute sorted neighbors for each node
        sorted_indices = np.argsort(distance_matrix, axis=1)
        neighbor_counts = min(20, n // 2)  # Adaptive candidate count
        
        for i in range(n):
            # Get candidate nodes that are neighbors of i or j
            i_neighbors = sorted_indices[i, 1:neighbor_counts+1]
            
            for j in range(i+1, n):
                if distance_matrix[i, j] == 0:
                    continue
                    
                j_neighbors = sorted_indices[j, 1:neighbor_counts+1]
                candidates = np.union1d(i_neighbors, j_neighbors)
                candidates = candidates[candidates != i]
                candidates = candidates[candidates != j]
                
                if len(candidates) > 0:
                    # Vectorized detour computation
                    detours = distance_matrix[i, candidates] + distance_matrix[candidates, j]
                    min_detour = np.min(detours)
                    
                    if min_detour < distance_matrix[i, j] * (1 - epsilon):
                        penalty = 0.7 + 0.3 * (min_detour / distance_matrix[i, j])
                        tri_penalty[i, j] = tri_penalty[j, i] = penalty
    
    # 5. Multiplicative combination of factors
    heuristics = inv_dist * mst_factor * nn_factor * tri_penalty
    
    # 6. Adaptive sparsification
    # Determine edges to keep based on problem size
    if n <= 20:
        edges_per_node = max(3, n // 2)
    elif n <= 50:
        edges_per_node = max(5, n // 3)
    elif n <= 100:
        edges_per_node = max(7, n // 4)
    else:
        edges_per_node = max(10, int(np.sqrt(n)))
    
    # Create sparse mask
    sparse_mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Ensure self is not selected
        
        # Always keep MST edges
        mst_indices = np.where(mst_factor[i] > 1.0)[0]
        
        # Keep top edges_per_node edges (including MST edges)
        if edges_per_node < n - 1:  # Not all edges
            threshold = np.partition(row, -edges_per_node)[-edges_per_node]
            keep_indices = np.where(row >= threshold)[0]
        else:
            keep_indices = np.arange(n)
        
        # Combine MST edges and top edges
        keep_indices = np.union1d(keep_indices, mst_indices)
        sparse_mask[i, keep_indices] = True
    
    # Ensure symmetry
    sparse_mask = sparse_mask | sparse_mask.T
    
    # Apply sparsification
    heuristics = heuristics * sparse_mask
    
    # 7. Non-linear scaling to emphasize top edges
    if np.max(heuristics) > 0:
        normalized = heuristics / np.max(heuristics)
        # Square root scaling preserves relative ordering while boosting high values
        heuristics = np.sqrt(normalized)
    
    # 8. Final normalization
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
