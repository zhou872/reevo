import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP heuristics with essential components and adaptive sparsity.
    
    Core components:
    1. Inverse distance (closer nodes are more promising)
    2. Minimum Spanning Tree edges (essential for connectivity)
    3. Nearest neighbor edges (local optimal structure)
    
    Uses sqrt(n) for adaptive sparsity and always includes MST and NN edges.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
        
    Returns:
        Heuristic matrix with same shape as input, values in [0, 1],
        where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 1:
        return np.zeros_like(distance_matrix)
    if n == 2:
        return np.ones_like(distance_matrix) - np.eye(2)
    
    # 1. Base heuristic: inverse distance
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    heuristics = 1.0 / safe_dist
    np.fill_diagonal(heuristics, 0.0)
    
    # 2. Minimum Spanning Tree edges (essential connectivity)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbor edges (local optimal structure)
    k = max(2, min(5, int(np.sqrt(n))))  # Adaptive k based on sqrt(n)
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i][:k]  # Exclude self
        nn_mask[i, nearest] = True
    nn_mask = nn_mask | nn_mask.T  # Symmetrize
    
    # 4. Boost essential edges
    heuristics[mst_mask] *= 2.0    # Strong boost for MST edges
    heuristics[nn_mask] *= 1.5     # Moderate boost for NN edges
    
    # 5. Adaptive sparsity: keep sqrt(n) edges per node, always include MST & NN
    keep_per_node = max(3, int(np.sqrt(n)))
    
    # Build keep mask
    keep_mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        # Always include MST and NN edges for this node
        keep_for_i = mst_mask[i] | nn_mask[i]
        
        # If we need more edges, add top heuristic values
        if np.sum(keep_for_i) < keep_per_node:
            row_heuristics = heuristics[i].copy()
            row_heuristics[i] = -np.inf  # Exclude diagonal
            row_heuristics[keep_for_i] = -np.inf  # Exclude already kept
            
            # Find additional edges to reach keep_per_node
            additional_needed = keep_per_node - np.sum(keep_for_i)
            if additional_needed > 0:
                top_indices = np.argsort(row_heuristics)[-additional_needed:]
                keep_for_i[top_indices] = True
        
        keep_mask[i] = keep_for_i
    
    # Symmetrize: keep edge if either endpoint wants it
    keep_mask = keep_mask | keep_mask.T
    
    # 6. Apply sparsity and normalize
    heuristics = heuristics * keep_mask
    np.fill_diagonal(heuristics, 0.0)
    
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
