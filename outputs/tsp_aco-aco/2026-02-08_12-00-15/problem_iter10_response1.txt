import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge selection heuristics with adaptive sparsification.
    Produces a matrix where higher values indicate edges more promising
    for inclusion in TSP solutions.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix with same shape, values normalized to [0, 1].
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with safe diagonal
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree edges (essential connectivity)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbors with logarithmic scaling
    # k = 2 + ceil(log2(n)) - balances locality and connectivity
    k = max(2, int(np.ceil(np.log2(n))) + 1)
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k+1 smallest distances (including inf diagonal)
        nearest = np.argpartition(safe_dist[i], k+1)[:k+1]
        nearest = nearest[nearest != i][:k]  # Remove self, keep k
        nn_mask[i, nearest] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Combine factors with separate bonuses
    heuristics = base_heuristic.copy()
    
    # Apply MST bonus (essential edges get highest boost)
    mst_bonus = 2.8
    heuristics[mst_mask] *= mst_bonus
    
    # Apply NN bonus to edges not already boosted by MST
    nn_not_mst = nn_mask & ~mst_mask
    nn_bonus = 1.8
    heuristics[nn_not_mst] *= nn_bonus
    
    # 5. Adaptive per-node sparsification using percentiles
    # Keep more edges for smaller problems, fewer for larger ones
    base_percentile = max(30, 85 - 15 * np.log10(n)) if n > 10 else 85
    node_masks = []
    
    for i in range(n):
        row_values = heuristics[i].copy()
        row_values[i] = -np.inf  # Ignore diagonal
        
        # Always include MST and NN edges
        essential_mask = mst_mask[i] | nn_mask[i]
        
        # Calculate adaptive percentile for this row
        valid_values = row_values[row_values > 0]
        if len(valid_values) > 0:
            # Use base percentile, but ensure at least essential edges are kept
            threshold = np.percentile(valid_values, base_percentile)
            candidate_mask = row_values >= threshold
        else:
            candidate_mask = np.zeros(n, dtype=bool)
        
        # Combine essential and candidate edges
        node_mask = essential_mask | candidate_mask
        node_masks.append(node_mask)
    
    # Create symmetric combined mask
    combined_mask = np.zeros_like(heuristics, dtype=bool)
    for i in range(n):
        combined_mask[i] = node_masks[i]
    combined_mask = combined_mask | combined_mask.T
    
    # Apply the mask
    heuristics = heuristics * combined_mask
    
    # 6. Global sparsification to remove weakest connections
    if n > 15:
        non_zero_vals = heuristics[heuristics > 0]
        if len(non_zero_vals) > 0:
            # Remove bottom 30-50% of non-zero values, depending on problem size
            cutoff_percentile = max(30, 50 - 5 * np.log10(n))
            global_threshold = np.percentile(non_zero_vals, cutoff_percentile)
            heuristics[heuristics < global_threshold] = 0
    
    # 7. Final normalization and symmetry enforcement
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Ensure perfect symmetry (use max to handle potential asymmetries)
    heuristics = np.maximum(heuristics, heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
