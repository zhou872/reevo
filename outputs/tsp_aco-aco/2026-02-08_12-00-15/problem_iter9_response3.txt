import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection.
    Combines MST, adaptive nearest neighbors, rank-based overlap boosting,
    and triangle inequality sampling with intelligent sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with smoothing
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / (safe_dist + 1e-8)  # Add small epsilon for stability
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree indicators
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.5  # Higher weight for MST edges
    
    # 3. Adaptive nearest neighbors with rank-based weighting
    # Adaptive k based on problem size
    k_nearest = min(max(3, n // 5), 10)
    nn_heuristic = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get distances and sort (excluding self)
        dist_i = safe_dist[i]
        sorted_indices = np.argsort(dist_i)[:k_nearest+1]
        sorted_indices = sorted_indices[sorted_indices != i]
        
        # Logarithmic rank-based weighting (higher rank = lower weight)
        ranks = np.arange(1, len(sorted_indices) + 1)
        weights = 1.0 / np.log(1.5 + ranks)  # Logarithmic decay
        
        # Apply weights to inverse distances
        for idx, j in enumerate(sorted_indices):
            nn_heuristic[i, j] = weights[idx] * base_heuristic[i, j]
    
    # Symmetrize nearest neighbor heuristic
    nn_heuristic = np.maximum(nn_heuristic, nn_heuristic.T)
    
    # 4. Overlap bonus: edges that appear in multiple criteria
    # Combine MST and NN indicators with logarithmic boosting
    overlap_mask = mst_mask & (nn_heuristic > 0)
    
    # 5. Combine all factors with adaptive weights
    heuristics = base_heuristic.copy()
    
    # Apply MST bonus (strong indicator)
    heuristics[mst_mask] *= mst_bonus
    
    # Add nearest neighbor contribution (weighted sum)
    heuristics += 0.7 * nn_heuristic
    
    # Apply logarithmic overlap bonus for edges in both MST and NN
    if np.any(overlap_mask):
        overlap_bonus = 1.0 + np.log1p(base_heuristic[overlap_mask].sum() / overlap_mask.sum())
        heuristics[overlap_mask] *= overlap_bonus
    
    # 6. Triangle inequality sampling (local connectivity check)
    # Sample edges to promote triangle completeness
    sample_size = min(100, n * 3)  # Adaptive sample size
    if n >= 3:
        for _ in range(sample_size):
            # Random triangle vertices
            i, j, k = np.random.choice(n, 3, replace=False)
            
            # Check if edges (i,k) and (j,k) exist in promising set
            ik_promising = heuristics[i, k] > 0
            jk_promising = heuristics[j, k] > 0
            
            # Boost edge (i,j) if it helps complete triangle
            if ik_promising and jk_promising:
                heuristics[i, j] *= 1.15
                heuristics[j, i] = heuristics[i, j]  # Maintain symmetry
    
    # 7. Adaptive sparsification with percentile thresholds
    # Keep edges that are in MST or have high heuristic values
    combined_mask = mst_mask.copy()
    
    # Keep top edges per node based on adaptive percentile
    for i in range(n):
        row_values = heuristics[i]
        non_zero_mask = row_values > 0
        
        if non_zero_mask.sum() > 0:
            non_zero_values = row_values[non_zero_mask]
            
            # Adaptive percentile: higher for larger problems
            percentile = max(70, 95 - n // 20)  # Range from 95% to 70%
            percentile = min(95, max(60, percentile))
            
            threshold = np.percentile(non_zero_values, percentile)
            row_promising = row_values >= threshold
            combined_mask[i] = combined_mask[i] | row_promising
    
    # Ensure symmetry of the mask
    combined_mask = combined_mask | combined_mask.T
    
    # Apply sparsification
    heuristics = heuristics * combined_mask
    
    # 8. Final normalization and symmetry enforcement
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Ensure perfect symmetry
    heuristics = np.maximum(heuristics, heuristics.T)
    
    return heuristics
