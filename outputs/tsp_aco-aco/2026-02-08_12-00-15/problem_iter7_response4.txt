import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection.
    Combines multiple factors including MST inclusion, nearest neighbors,
    triangle violations, and adaptive sparsification with logarithmic boosting.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with logarithmic boosting
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # Logarithmic boosting of top edges
    log_boost_factor = np.log1p(base_heuristic * (n / 10))
    base_heuristic = base_heuristic * log_boost_factor
    
    # 2. Minimum Spanning Tree bonus (multiplicative)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 1.0 + (2.5 * np.log1p(n / 10)) / n  # Adaptive bonus
    
    # 3. Nearest neighbor bonus (multiplicative)
    k = max(2, min(8, int(np.sqrt(n))))  # Adaptive k based on sqrt(n)
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k+1 nearest (including self at inf)
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i]
        nn_mask[i, nearest[:min(k, len(nearest))]] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.0 + (1.8 * np.log1p(n / 5)) / n  # Adaptive bonus
    
    # 4. Triangle violation penalty
    triangle_penalty = np.ones_like(distance_matrix)
    
    # Compute triangle violations efficiently
    for i in range(n):
        for j in range(i+1, n):
            if safe_dist[i, j] < np.inf:
                # Find best alternative path through third node
                min_detour = np.min(
                    safe_dist[i, :] + safe_dist[:, j] - 
                    np.diag(safe_dist)  # Remove self distances
                )
                
                if min_detour > 0:
                    # Penalize if direct edge is significantly longer than detour
                    violation_ratio = safe_dist[i, j] / min_detour
                    if violation_ratio > 1.2:  # Only penalize significant violations
                        triangle_penalty[i, j] = triangle_penalty[j, i] = 1.0 / np.sqrt(violation_ratio)
    
    # 5. Combine all factors multiplicatively
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    heuristics *= triangle_penalty
    
    # 6. Adaptive sparsification
    # Keep edges that are in MST or nearest neighbors
    keep_mask = mst_mask | nn_mask
    
    # Additionally, keep top edges based on percentile threshold
    # Use more aggressive sparsification for larger instances
    sparsity_factor = max(0.3, min(0.7, 0.7 - 0.005 * n))
    percentile_threshold = 100 * sparsity_factor
    
    for i in range(n):
        row_values = heuristics[i]
        valid_values = row_values[row_values > 0]
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, percentile_threshold)
            keep_mask[i] = keep_mask[i] | (row_values >= threshold)
    
    # Apply the combined mask
    heuristics = heuristics * keep_mask
    
    # 7. Symmetry enforcement (ensure the matrix remains symmetric)
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 8. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
