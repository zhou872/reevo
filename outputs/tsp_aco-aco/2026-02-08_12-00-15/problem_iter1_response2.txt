import numpy as np
from scipy.spatial.distance import pdist, squareform
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP that combines multiple factors to evaluate edge promise.
    Returns a matrix of the same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.ones_like(distance_matrix)
    
    # Avoid division by zero
    epsilon = 1e-10
    safe_dist = distance_matrix + epsilon
    
    # 1. Basic reciprocal (shorter edges are better)
    reciprocal = 1.0 / safe_dist
    np.fill_diagonal(reciprocal, 0)
    
    # 2. Nearest neighbor affinity
    # For each node, identify top k nearest neighbors
    k = max(2, min(5, n // 4))
    nearest_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        indices = np.argsort(distance_matrix[i])[1:k+1]
        nearest_mask[i, indices] = True
    
    # Mutual nearest neighbor bonus
    mutual_mask = nearest_mask & nearest_mask.T
    one_sided_mask = nearest_mask ^ mutual_mask  # XOR: one-sided nearest neighbors
    
    # 3. Sparsity factor based on relative distance ranking
    # Rank edges by distance for each node
    rank_matrix = np.zeros_like(distance_matrix, dtype=float)
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])
        ranks = np.zeros(n)
        ranks[sorted_indices] = np.arange(n)
        # Convert to score (higher rank = shorter distance = higher score)
        rank_matrix[i] = (n - ranks) / n
    
    # 4. Triangular inequality consistency score
    # Edges that are consistent with triangle inequality are more promising
    tri_score = np.ones_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            # Check if direct edge is significantly shorter than indirect paths
            min_indirect = np.min(distance_matrix[i] + distance_matrix[:, j])
            if min_indirect > 0:
                tri_score[i, j] = tri_score[j, i] = distance_matrix[i, j] / min_indirect
    
    # 5. Edge betweenness approximation
    # Edges connecting different clusters are more promising
    if n >= 5:
        # Simple clustering: separate into two groups based on farthest points
        # Find two farthest points
        max_idx = np.unravel_index(np.argmax(distance_matrix), distance_matrix.shape)
        group_a = [max_idx[0]]
        group_b = [max_idx[1]]
        
        # Assign other nodes to closest group center
        for node in range(n):
            if node not in group_a and node not in group_b:
                dist_to_a = np.mean([distance_matrix[node, a] for a in group_a])
                dist_to_b = np.mean([distance_matrix[node, b] for b in group_b])
                if dist_to_a < dist_to_b:
                    group_a.append(node)
                else:
                    group_b.append(node)
        
        # Give bonus to edges connecting the two groups
        betweenness_bonus = np.ones_like(distance_matrix)
        for i in group_a:
            for j in group_b:
                betweenness_bonus[i, j] = betweenness_bonus[j, i] = 1.5
    
    # Combine all factors
    heuristics_matrix = reciprocal.copy()
    
    # Apply bonuses and penalties
    heuristics_matrix *= rank_matrix  # Incorporate distance ranking
    heuristics_matrix *= tri_score    # Incorporate triangle inequality consistency
    
    # Apply sparsification - zero out clearly unpromising edges
    # Keep only edges that are in top m% for at least one endpoint
    m_percent = max(20, 100 // (n ** 0.5))  # Adaptive threshold
    
    threshold_matrix = np.zeros_like(heuristics_matrix)
    for i in range(n):
        row = heuristics_matrix[i]
        threshold = np.percentile(row[row > 0], m_percent) if np.any(row > 0) else 0
        threshold_matrix[i] = row >= threshold
    
    # Apply mutual nearest neighbor bonus
    heuristics_matrix[mutual_mask] *= 2.0
    heuristics_matrix[one_sided_mask] *= 1.5
    
    # Apply betweenness bonus if computed
    if n >= 5:
        heuristics_matrix *= betweenness_bonus
    
    # Final sparsification
    keep_mask = threshold_matrix | mutual_mask
    heuristics_matrix = heuristics_matrix * keep_mask
    
    # Normalize to reasonable range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val * 10
    
    # Ensure symmetry
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
