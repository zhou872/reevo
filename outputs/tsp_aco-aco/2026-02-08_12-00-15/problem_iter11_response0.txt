import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge selection heuristics with adaptive sparsification,
    logarithmic nearest neighbor scaling, and triangle-based connectivity boosts.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.2  # Slightly higher bonus for MST edges
    
    # 3. Logarithmic nearest neighbor scaling
    # Base k scales with log(n), capped at reasonable bounds
    log_k = max(2, int(np.log2(n)) + 1)
    k = min(log_k, n - 1)  # Cannot exceed n-1 neighbors
    
    nn_bonus_matrix = np.ones_like(base_heuristic)
    
    for i in range(n):
        # Get distances and sort
        distances = safe_dist[i].copy()
        sorted_indices = np.argsort(distances)[:k+1]  # +1 to include self
        sorted_indices = sorted_indices[sorted_indices != i]  # Remove self
        
        # Logarithmic decay: bonus = log(k+1 - rank) / log(k+1)
        ranks = np.arange(1, len(sorted_indices) + 1)
        decay_factor = np.log(k + 1 - ranks + 1e-10) / np.log(k + 1)
        
        for idx, j in enumerate(sorted_indices):
            # Apply symmetric bonus (average of both directions)
            current_bonus = 1.0 + decay_factor[idx] * 0.8  # Scales 1.0 to 1.8
            nn_bonus_matrix[i, j] = current_bonus
    
    # Symmetrize the NN bonus matrix
    nn_bonus_sym = (nn_bonus_matrix + nn_bonus_matrix.T) / 2
    
    # 4. Overlap bonus for mutual nearest neighbors
    overlap_mask = np.zeros_like(distance_matrix, dtype=bool)
    nn_mask = nn_bonus_matrix > 1.0  # Edges that got any NN bonus
    
    for i in range(n):
        for j in range(i+1, n):
            if nn_mask[i, j] and nn_mask[j, i]:
                overlap_mask[i, j] = overlap_mask[j, i] = True
    
    overlap_bonus = 1.3
    
    # 5. Triangle sampling for connectivity
    triangle_bonus = np.ones_like(base_heuristic)
    if n >= 3:
        # Sample promising edges to form triangles
        threshold = np.percentile(base_heuristic[base_heuristic > 0], 50)
        
        for i in range(n):
            for j in range(i+1, n):
                if base_heuristic[i, j] < threshold:
                    continue
                
                # Find common neighbors that connect to both i and j
                for k in range(n):
                    if k == i or k == j:
                        continue
                    
                    if (base_heuristic[i, k] > threshold and 
                        base_heuristic[j, k] > threshold):
                        # Boost all three edges of the triangle
                        triangle_bonus[i, j] *= 1.05
                        triangle_bonus[i, k] *= 1.05
                        triangle_bonus[j, k] *= 1.05
    
    # 6. Combine all factors
    heuristics = base_heuristic.copy()
    heuristics *= nn_bonus_sym  # Apply NN bonuses
    heuristics[mst_mask] *= mst_bonus
    heuristics[overlap_mask] *= overlap_bonus
    heuristics *= triangle_bonus
    
    # 7. Adaptive percentile sparsification
    # Determine sparsity level adaptively based on n
    sparsity_level = max(25, min(75, 100 - n))  # More sparse for larger n
    
    combined_mask = mst_mask | overlap_mask  # Always keep MST and overlap edges
    
    # Add top percentile edges per node
    for i in range(n):
        row_values = heuristics[i]
        valid_values = row_values[row_values > 0]
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, sparsity_level)
            combined_mask[i] = combined_mask[i] | (row_values >= threshold)
    
    # Also keep edges that appear in promising triangles
    triangle_edges = triangle_bonus > 1.0
    combined_mask = combined_mask | triangle_edges
    
    # Apply the combined mask
    heuristics = heuristics * combined_mask
    
    # 8. Ensure symmetry and normalize
    heuristics = (heuristics + heuristics.T) / 2  # Ensure exact symmetry
    np.fill_diagonal(heuristics, 0.0)
    
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
