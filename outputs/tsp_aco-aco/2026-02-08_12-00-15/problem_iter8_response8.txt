import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with improved MST integration,
    efficient nearest neighbor computation, and adaptive sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with safe handling
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree integration
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # Enhanced MST bonus based on edge centrality
    mst_bonus = np.full_like(base_heuristic, 2.0)
    if n > 3:
        # Edges in MST that connect to high-degree nodes get higher bonus
        node_degree = mst_mask.sum(axis=1)
        for i in range(n):
            for j in range(i+1, n):
                if mst_mask[i, j]:
                    # Bonus proportional to geometric mean of node degrees
                    deg_factor = np.sqrt(node_degree[i] * node_degree[j])
                    mst_bonus[i, j] = mst_bonus[j, i] = 2.0 + 0.5 * np.log1p(deg_factor)
    
    # 3. Efficient nearest neighbor computation
    k = max(2, min(6, int(np.log2(n + 4))))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Vectorized computation for nearest neighbors
    for i in range(n):
        row = safe_dist[i]
        if n <= 50:  # For small problems, use direct sort
            nearest_indices = np.argsort(row)[1:k+1]
        else:  # For larger problems, use argpartition for efficiency
            k_neighbors = min(k, n - 1)
            partition_idx = np.argpartition(row, k_neighbors + 1)[:k_neighbors + 1]
            # Exclude self and take k nearest
            nearest_indices = partition_idx[partition_idx != i][:k_neighbors]
        nn_mask[i, nearest_indices] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Adaptive sparsification based on problem size
    base_sparsity = max(2, min(5, int(np.log2(n))))
    combined_mask = mst_mask | nn_mask
    
    # Add edges based on base heuristic ranking
    for i in range(n):
        row_heuristics = base_heuristic[i].copy()
        np.fill_diagonal(row_heuristics, -np.inf)
        
        # Count current connections
        current_connections = np.sum(combined_mask[i])
        
        # Determine how many additional edges to consider
        target_connections = max(base_sparsity, current_connections)
        if target_connections > current_connections:
            # Find top edges not already in mask
            mask_indices = np.where(combined_mask[i])[0]
            temp_heuristics = row_heuristics.copy()
            temp_heuristics[mask_indices] = -np.inf
            
            # Get indices of top remaining edges
            needed = target_connections - current_connections
            if needed > 0:
                # Use argpartition for efficiency on large problems
                if n - len(mask_indices) - 1 <= 100:  # Small selection space
                    remaining_indices = np.argsort(-temp_heuristics)[:needed]
                else:
                    partition_idx = np.argpartition(-temp_heuristics, needed)[:needed]
                    # Sort the top ones for better selection
                    top_values = temp_heuristics[partition_idx]
                    sorted_idx = np.argsort(-top_values)
                    remaining_indices = partition_idx[sorted_idx]
                
                combined_mask[i, remaining_indices] = True
    
    # Ensure symmetry of the mask
    combined_mask = combined_mask | combined_mask.T
    
    # 5. Minimum connectivity guarantee (at least 3 connections per node)
    for i in range(n):
        connections = np.where(combined_mask[i])[0]
        if len(connections) < 3:
            needed = 3 - len(connections)
            # Find best additional connections
            temp_heuristics = base_heuristic[i].copy()
            temp_heuristics[connections] = -np.inf
            temp_heuristics[i] = -np.inf
            
            additional_indices = np.argsort(-temp_heuristics)[:needed]
            combined_mask[i, additional_indices] = True
    
    # Final symmetry enforcement
    combined_mask = combined_mask | combined_mask.T
    
    # 6. Combine heuristics with bonuses
    heuristics = base_heuristic * mst_bonus
    heuristics[~mst_mask] = base_heuristic[~mst_mask]  # Reset non-MST edges to base
    heuristics[nn_mask] *= 1.5  # Boost nearest neighbors
    
    # Apply the final mask
    heuristics = heuristics * combined_mask
    
    # 7. Final processing
    heuristics = np.maximum(heuristics, heuristics.T)  # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2  # Average for numerical stability
    
    # Normalize to [0, 1] range
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    np.fill_diagonal(heuristics, 0.0)
    
    return heuristics
