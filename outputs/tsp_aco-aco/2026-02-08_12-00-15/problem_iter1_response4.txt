import numpy as np
from scipy.spatial import Delaunay
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP that combines multiple factors to determine edge promise,
    with sparsification to focus on promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Initialize heuristic matrix
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # 1. Inverse distance (baseline)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-8)  # Avoid division by zero
    
    # 2. Node connectivity score based on nearest neighbors
    k = min(5, n // 4 + 1)  # Adaptive k for nearest neighbors
    connectivity = np.zeros_like(distance_matrix)
    for i in range(n):
        # Get k+1 smallest distances (including self)
        idx = np.argpartition(distance_matrix[i], k+1)[:k+1]
        # Remove self if present
        idx = idx[idx != i][:k]
        connectivity[i, idx] = 1 / (1 + np.arange(len(idx)))
    
    # 3. Minimum spanning tree edges get bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = (mst > 0).astype(float) * 2.0
    
    # 4. Convex hull edges get bonus (for Euclidean-like distances)
    try:
        # Generate random coordinates if not Euclidean (for demonstration)
        # In practice, this would require actual coordinates
        # Using random projection for demonstration
        coords = np.random.randn(n, 2)
        hull = Delaunay(coords).convex_hull
        hull_bonus = np.zeros_like(distance_matrix)
        for edge in hull:
            i, j = edge
            hull_bonus[i, j] = hull_bonus[j, i] = 1.5
    except:
        hull_bonus = np.zeros_like(distance_matrix)
    
    # 5. Degree-based penalty: nodes with many good connections share the wealth
    degree_penalty = np.zeros(n)
    for i in range(n):
        good_connections = np.sum(connectivity[i] > 0)
        if good_connections > 0:
            degree_penalty[i] = 1 / np.sqrt(good_connections)
    
    # Combine factors with weights
    heuristics_matrix = (
        1.0 * inv_dist +                     # Inverse distance
        0.8 * connectivity +                 # Nearest neighbors
        1.2 * mst_bonus +                    # MST edges
        0.7 * hull_bonus                     # Convex hull edges
    )
    
    # Apply degree-based normalization
    for i in range(n):
        heuristics_matrix[i] *= degree_penalty[i]
        heuristics_matrix[:, i] *= degree_penalty[i]
    
    # 6. Sparsification: keep only top connections for each node
    sparsity_factor = min(0.3, 3.0 / n)  # Keep fewer edges for larger graphs
    threshold = np.percentile(heuristics_matrix[heuristics_matrix > 0], 
                             100 * (1 - sparsity_factor))
    
    mask = heuristics_matrix < threshold
    heuristics_matrix[mask] = 0
    
    # 7. Symmetrize (for undirected TSP)
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    
    # 8. Normalize for stability
    max_val = heuristics_matrix.max()
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
