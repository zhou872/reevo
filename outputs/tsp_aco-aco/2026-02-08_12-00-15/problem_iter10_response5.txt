import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Simplified yet effective TSP heuristics prioritizing essential features.
    Focuses on inverse distance, MST connectivity, and nearest neighbors with
    adaptive sparsity for computational efficiency.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
        
    Returns:
        Heuristic matrix with same shape as input, values in [0, 1].
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # 1. Base heuristic: inverse distance (closer nodes are more promising)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree connectivity (essential for TSP)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbor factor with adaptive k
    k = max(2, min(5, int(np.sqrt(n))))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i][:k]
        nn_mask[i, nearest] = True
    
    # Make nearest neighbor mask symmetric
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Combine factors with clear priority (MST > NN > base)
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= 2.0  # Strong boost for MST edges
    heuristics[nn_mask] *= 1.5   # Moderate boost for NN edges
    
    # 5. Adaptive sparsity: keep only most promising edges
    edges_per_node = max(3, min(n-1, int(np.sqrt(n) * 2)))
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Always keep MST and NN edges for this node
        must_keep = mst_mask[i] | nn_mask[i]
        
        # Determine how many more edges to keep
        additional_count = max(0, edges_per_node - np.sum(must_keep))
        
        if additional_count > 0:
            # Get top heuristic edges among remaining
            remaining_mask = ~must_keep
            remaining_mask[i] = False  # Exclude self
            remaining_indices = np.where(remaining_mask)[0]
            
            if len(remaining_indices) > 0:
                remaining_values = row[remaining_indices]
                if len(remaining_values) > 0:
                    # Get top edges from remaining
                    top_count = min(additional_count, len(remaining_values))
                    top_indices = np.argsort(remaining_values)[-top_count:]
                    top_nodes = remaining_indices[top_indices]
                    
                    # Create keep mask
                    keep_mask = must_keep.copy()
                    keep_mask[top_nodes] = True
                else:
                    keep_mask = must_keep
            else:
                keep_mask = must_keep
        else:
            keep_mask = must_keep
        
        # Zero out edges not in keep_mask
        heuristics[i, ~keep_mask] = 0
    
    # 6. Ensure symmetry and final processing
    heuristics = np.maximum(heuristics, heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # 7. Normalize non-zero values to [0, 1]
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
