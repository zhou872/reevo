```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Simple yet effective heuristics for TSP edge selection.
    Combines inverse distance, Minimum Spanning Tree inclusion,
    and nearest neighbors with adaptive sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.0  # MST edges are highly promising
    
    # 3. Nearest neighbor bonus
    # Adaptive k: more neighbors for smaller problems
    k = max(2, min(5, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]  # +1 to skip self (inf)
        nearest = nearest[nearest != i]
        nn_mask[i, nearest[:min(k, len(nearest))]] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.5
    
    # 4. Combine all factors
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    
    # 5. Sparsification: keep only promising edges
    # Keep edges that are in MST, nearest neighbors, or have high base score
    combined_mask = mst_mask | nn_mask
    
    # Also keep edges with high base scores (top 25% per node)
    top_percentile = 75
    for i in range(n):
        row_values = heuristics[i]
        valid_values = row_values[row_values > 0]
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, top_percentile)
            combined_mask[i] = combined_mask[i] | (row_values >= threshold)
    
    # Apply the combined mask
    heuristics = heuristics * combined_mask
    
    # 6. Normalize for consistency
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
```
