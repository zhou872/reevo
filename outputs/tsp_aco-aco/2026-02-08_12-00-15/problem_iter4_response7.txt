import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristics for TSP edge selection.
    Focuses on MST inclusion and nearest neighbors with sparsification.
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create safe distance matrix with diagonal set to infinity
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base heuristic: reciprocal of distance
    base_heuristic = 1.0 / safe_dist
    
    # 2. Minimum Spanning Tree inclusion bonus
    mst_mask = _compute_mst_mask(safe_dist)
    
    # 3. Nearest neighbor connectivity
    k = max(2, n // 10)  # Adaptive number of nearest neighbors
    nn_mask = _compute_nn_mask(safe_dist, k)
    
    # 4. Combine factors
    # Start with base heuristic
    heuristics = base_heuristic.copy()
    
    # Apply bonuses
    mst_bonus = 2.0  # Strong bonus for MST edges
    nn_bonus = 1.5   # Moderate bonus for nearest neighbors
    
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    
    # 5. Sparsify: keep only promising edges
    # Keep edges that are in MST, are nearest neighbors, or have high base heuristic
    final_mask = np.zeros_like(heuristics, dtype=bool)
    
    # Always include MST and nearest neighbor edges
    final_mask = final_mask | mst_mask | nn_mask
    
    # Also include top edges based on heuristic values
    if n > 2:
        # Compute threshold for each node
        for i in range(n):
            row_vals = heuristics[i].copy()
            row_vals[i] = -np.inf  # Ignore self
            
            # Keep top edges per node (adaptive)
            top_k = max(2, min(5, n // 4))
            top_indices = np.argpartition(-row_vals, top_k)[:top_k]
            final_mask[i, top_indices] = True
    
    # Make the mask symmetric
    final_mask = final_mask | final_mask.T
    
    # Apply final mask
    heuristics = heuristics * final_mask
    
    # 6. Normalize for stability
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics


def _compute_mst_mask(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute Minimum Spanning Tree mask using Prim's algorithm."""
    n = distance_matrix.shape[0]
    mst_mask = np.zeros((n, n), dtype=bool)
    
    if n <= 1:
        return mst_mask
    
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    
    for _ in range(n - 1):
        # Find minimum edge connecting visited to unvisited nodes
        visited_idx = np.where(visited)[0]
        unvisited_idx = np.where(~visited)[0]
        
        if len(visited_idx) == 0 or len(unvisited_idx) == 0:
            break
        
        # Get distances between visited and unvisited nodes
        min_dist = np.inf
        min_i, min_j = -1, -1
        
        for i in visited_idx:
            for j in unvisited_idx:
                if distance_matrix[i, j] < min_dist:
                    min_dist = distance_matrix[i, j]
                    min_i, min_j = i, j
        
        if min_i != -1:
            mst_mask[min_i, min_j] = True
            mst_mask[min_j, min_i] = True
            visited[min_j] = True
    
    return mst_mask


def _compute_nn_mask(distance_matrix: np.ndarray, k: int) -> np.ndarray:
    """Compute k-nearest neighbors mask."""
    n = distance_matrix.shape[0]
    nn_mask = np.zeros((n, n), dtype=bool)
    
    if n <= 1:
        return nn_mask
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nearest = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, nearest] = True
    
    # Make symmetric (if i is in j's nearest neighbors, include the edge)
    nn_mask = nn_mask | nn_mask.T
    
    return nn_mask
