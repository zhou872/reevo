import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Simplified TSP edge selection heuristics focusing on MST and nearest neighbors
    with adaptive sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create safe distance matrix with diagonal set to infinity
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # Base heuristic: reciprocal of distance
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # Compute Minimum Spanning Tree using Prim's algorithm
    def compute_mst_mask(dist_mat: np.ndarray) -> np.ndarray:
        n_nodes = dist_mat.shape[0]
        mst_mask = np.zeros((n_nodes, n_nodes), dtype=bool)
        
        if n_nodes <= 1:
            return mst_mask
        
        visited = np.zeros(n_nodes, dtype=bool)
        visited[0] = True
        
        for _ in range(n_nodes - 1):
            min_dist = np.inf
            min_i, min_j = -1, -1
            
            # Find minimum edge between visited and unvisited nodes
            visited_indices = np.where(visited)[0]
            unvisited_indices = np.where(~visited)[0]
            
            for i in visited_indices:
                for j in unvisited_indices:
                    if dist_mat[i, j] < min_dist:
                        min_dist = dist_mat[i, j]
                        min_i, min_j = i, j
            
            if min_i != -1:
                mst_mask[min_i, min_j] = True
                mst_mask[min_j, min_i] = True
                visited[min_j] = True
        
        return mst_mask
    
    # Compute nearest neighbors mask
    def compute_nn_mask(dist_mat: np.ndarray, k: int) -> np.ndarray:
        n_nodes = dist_mat.shape[0]
        nn_mask = np.zeros((n_nodes, n_nodes), dtype=bool)
        
        if n_nodes <= 1 or k <= 0:
            return nn_mask
        
        for i in range(n_nodes):
            # Get k nearest neighbors (excluding self)
            row = dist_mat[i].copy()
            nearest_indices = np.argpartition(row, k)[:k]
            nn_mask[i, nearest_indices] = True
        
        # Ensure symmetry
        nn_mask = nn_mask | nn_mask.T
        
        return nn_mask
    
    # Get MST and nearest neighbor masks
    mst_mask = compute_mst_mask(safe_dist)
    k = max(2, min(5, n // 4))  # Adaptive k based on problem size
    nn_mask = compute_nn_mask(safe_dist, k)
    
    # Combine heuristics with adaptive bonuses
    heuristics = base_heuristic.copy()
    
    # Strong bonus for MST edges
    heuristics[mst_mask] *= 3.0
    
    # Moderate bonus for nearest neighbor edges
    heuristics[nn_mask] *= 2.0
    
    # Adaptive sparsification: keep top edges per node
    sparsified = np.zeros_like(heuristics)
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Adaptive number of edges to keep
        keep_count = max(2, min(5, n // 4))
        top_indices = np.argpartition(-row, keep_count)[:keep_count]
        sparsified[i, top_indices] = row[top_indices]
    
    # Ensure symmetry and zero diagonal
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0.0)
    
    # Normalize to [0, 1] range
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
