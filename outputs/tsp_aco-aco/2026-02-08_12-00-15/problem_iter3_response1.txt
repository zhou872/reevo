import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP with triangle inequality penalties and adaptive sparsification.
    Features:
    1. Inverse distance baseline with exponential decay
    2. Minimum Spanning Tree inclusion with edge strength weighting
    3. Nearest neighbor connectivity with adaptive k
    4. Triangle inequality violation penalties
    5. Adaptive sparsification based on problem size
    6. Top edge boosting with sigmoid scaling
    """
    n = distance_matrix.shape[0]
    
    # 0. Preprocessing: handle zero distances to avoid division by zero
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    
    # 1. Inverse distance baseline with exponential decay
    # Use exponential to give more weight to very short edges
    scale = np.median(dist[dist < np.inf])
    inv_dist = np.exp(-dist / scale)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree inclusion with edge strength
    mst = minimum_spanning_tree(dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # Weight MST edges by their relative importance (closer edges get higher weight)
    mst_weights = np.zeros_like(dist)
    for i in range(n):
        for j in range(i+1, n):
            if mst_mask[i, j]:
                # Normalize by median distance to avoid extreme values
                mst_weights[i, j] = mst_weights[j, i] = np.exp(-dist[i, j] / scale)
    
    mst_bonus = 1.0 + 0.5 * mst_weights  # 1.0-1.5 range
    
    # 3. Adaptive nearest neighbor connectivity
    # Use more neighbors for smaller problems, fewer for larger ones
    base_k = max(2, int(np.sqrt(n)))
    k = min(base_k, n-1)
    
    nn_mask = np.zeros_like(dist, dtype=bool)
    for i in range(n):
        # Get k+1 to exclude self, then take k nearest
        nearest = np.argsort(dist[i])[1:k+1]
        nn_mask[i, nearest] = True
    
    # Symmetrize and weight by distance rank
    nn_weights = np.zeros_like(dist)
    for i in range(n):
        sorted_idx = np.argsort(dist[i])[1:k+1]
        for rank, j in enumerate(sorted_idx):
            # Higher weight for closer neighbors (lower rank)
            weight = (k - rank) / k
            nn_weights[i, j] = max(nn_weights[i, j], weight)
    
    # Make symmetric
    nn_weights = np.maximum(nn_weights, nn_weights.T)
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.0 + 0.3 * nn_weights  # 1.0-1.3 range
    
    # 4. Triangle inequality penalties
    # Penalize edges that violate triangle inequality
    tri_penalty = np.ones_like(dist)
    
    if n >= 3:
        for i in range(n):
            for j in range(i+1, n):
                # Find if there's a much shorter path through another node
                min_detour = np.inf
                for k in range(n):
                    if k != i and k != j:
                        detour = dist[i, k] + dist[k, j]
                        if detour < min_detour:
                            min_detour = detour
                
                # If direct edge is significantly longer than best detour, penalize
                if dist[i, j] > 1.2 * min_detour:  # 20% tolerance
                    penalty = min_detour / dist[i, j]
                    tri_penalty[i, j] = tri_penalty[j, i] = penalty
    
    # 5. Combine all factors
    heuristics = inv_dist.copy()
    heuristics *= mst_bonus
    heuristics *= nn_bonus
    heuristics *= tri_penalty
    
    # 6. Adaptive sparsification
    # Keep more edges for small problems, fewer for large ones
    if n <= 20:
        keep_fraction = 0.5
    elif n <= 50:
        keep_fraction = 0.4
    elif n <= 100:
        keep_fraction = 0.3
    else:
        keep_fraction = 0.2
    
    threshold_mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        row_values = heuristics[i]
        # Only consider non-zero values
        non_zero = row_values > 0
        if non_zero.sum() > 0:
            # Keep top keep_fraction of edges for this node
            threshold = np.percentile(row_values[non_zero], 100 * (1 - keep_fraction))
            threshold_mask[i] = row_values >= threshold
    
    # Apply sparsification mask
    heuristics = heuristics * threshold_mask
    
    # 7. Boost top edges using sigmoid-like scaling
    # This makes good edges stand out more
    if heuristics.max() > 0:
        # Normalize first
        heuristics_norm = heuristics / heuristics.max()
        # Apply sigmoid-like transformation: f(x) = x^2 / (0.5 + x)
        heuristics = heuristics_norm ** 2 / (0.5 + heuristics_norm)
        # Renormalize
        if heuristics.max() > 0:
            heuristics = heuristics / heuristics.max()
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
