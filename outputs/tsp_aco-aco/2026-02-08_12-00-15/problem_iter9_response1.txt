import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge selection heuristics.
    Combines MST, nearest neighbors, and triangle inequality checks
    with logarithmic rank-based boosting and adaptive sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.0
    
    # 3. Adaptive nearest neighbor bonus with logarithmic rank
    k = max(2, min(10, n // 3))  # Adaptive k based on problem size
    nn_bonus_matrix = np.zeros_like(base_heuristic)
    
    for i in range(n):
        # Get sorted distances (excluding self)
        sorted_indices = np.argsort(safe_dist[i])
        sorted_indices = sorted_indices[sorted_indices != i][:k]
        
        # Apply logarithmic rank bonus: 1/log(rank+2) gives more weight to top ranks
        for rank, j in enumerate(sorted_indices):
            log_bonus = 1.0 / np.log2(rank + 2)  # Range: 1 to ~0.33 for k=10
            nn_bonus_matrix[i, j] = log_bonus
    
    # Symmetrize nearest neighbor bonus
    nn_bonus_matrix = np.maximum(nn_bonus_matrix, nn_bonus_matrix.T)
    
    # 4. Triangle inequality sampling for edge validation
    triangle_boost = np.ones_like(base_heuristic)
    if n >= 3:
        # Sample random triangles to check triangle inequality
        num_samples = min(1000, n * 20)
        for _ in range(num_samples):
            i, j, k = np.random.choice(n, 3, replace=False)
            
            # Check triangle inequality: if edge (i,j) is much longer than
            # path through k, it's less promising
            alternative_path = safe_dist[i, k] + safe_dist[k, j]
            direct_path = safe_dist[i, j]
            
            if alternative_path < direct_path:
                # Edge violates triangle inequality - reduce its score
                violation_ratio = alternative_path / (direct_path + 1e-10)
                triangle_boost[i, j] *= 0.7 + 0.3 * violation_ratio
                triangle_boost[j, i] = triangle_boost[i, j]
    
    # 5. Combine all factors with overlap boosting
    heuristics = base_heuristic.copy()
    
    # Apply MST bonus
    heuristics[mst_mask] *= mst_bonus
    
    # Apply nearest neighbor bonus multiplicatively
    nn_mask = nn_bonus_matrix > 0
    heuristics[nn_mask] *= (1.0 + nn_bonus_matrix[nn_mask])
    
    # Boost edges that appear in both MST and NN (strong candidates)
    overlap_mask = mst_mask & nn_mask
    heuristics[overlap_mask] *= 1.5
    
    # Apply triangle inequality adjustments
    heuristics *= triangle_boost
    
    # 6. Adaptive sparsification with percentile thresholds
    # Keep edges that are in MST, top NN, or high-scoring
    combined_mask = mst_mask | nn_mask
    
    # Adaptive percentile threshold based on problem size
    base_percentile = 70
    size_factor = min(1.0, n / 50)  # Higher threshold for larger problems
    percentile = base_percentile + 5 * size_factor
    
    # Keep top edges per node based on computed heuristics
    for i in range(n):
        row_values = heuristics[i]
        valid_mask = row_values > 0
        valid_values = row_values[valid_mask]
        
        if len(valid_values) > 1:
            # Calculate adaptive threshold
            threshold = np.percentile(valid_values, percentile)
            
            # Also keep at least 3 edges per node (or n-1 for small n)
            min_edges = min(3, n - 1)
            top_indices = np.argsort(row_values)[-min_edges:]
            row_mask = np.zeros(n, dtype=bool)
            row_mask[top_indices] = True
            
            combined_mask[i] = combined_mask[i] | (row_values >= threshold) | row_mask
    
    # Ensure symmetry by taking max of both directions
    symmetric_mask = np.maximum(combined_mask, combined_mask.T)
    heuristics = heuristics * symmetric_mask
    
    # 7. Final normalization with non-linear scaling
    max_val = heuristics.max()
    if max_val > 0:
        # Use square root scaling to reduce dominance of very high values
        heuristics = np.sqrt(heuristics / max_val)
    
    # Ensure diagonal remains zero
    np.fill_diagonal(heuristics, 0.0)
    
    return heuristics
