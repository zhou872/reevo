import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP that combines multiple factors and sparsifies
    the matrix by setting unpromising edges to zero.
    """
    n = distance_matrix.shape[0]
    
    # Initialize heuristic matrix
    heuristic_matrix = np.zeros_like(distance_matrix, dtype=float)
    
    # 1. Base heuristic: reciprocal of distance (with handling for zeros)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_dist = np.where(distance_matrix > 0, 1.0 / distance_matrix, 0)
    np.fill_diagonal(reciprocal_dist, 0)
    
    # 2. Node centrality: prefer edges connected to centrally located nodes
    # Calculate sum of distances for each node (lower = more central)
    row_sums = np.sum(distance_matrix, axis=1)
    centrality = 1.0 / (row_sums + 1e-10)  # Avoid division by zero
    centrality_weight = 0.3  # Weight for centrality factor
    
    # 3. Local connectivity: prefer edges that connect to nearest neighbors
    # For each node, find its k nearest neighbors
    k = max(3, n // 10)  # Adaptive k based on problem size
    nearest_neighbor_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k+1 smallest distances (including self)
        nearest_indices = np.argpartition(distance_matrix[i], k+1)[:k+1]
        # Remove self from nearest neighbors
        nearest_indices = nearest_indices[nearest_indices != i][:k]
        nearest_neighbor_mask[i, nearest_indices] = True
    
    # 4. Minimum spanning tree bias: encourage edges that might be in MST
    # Approximate MST bias using Prim's algorithm
    mst_bias = np.zeros_like(distance_matrix, dtype=float)
    if n > 1:
        visited = np.zeros(n, dtype=bool)
        visited[0] = True
        mst_weight = 0.2  # Weight for MST bias
        
        # Build approximate MST
        for _ in range(n - 1):
            # Find minimum edge from visited to unvisited nodes
            min_edge = np.inf
            min_i, min_j = 0, 0
            
            visited_indices = np.where(visited)[0]
            unvisited_indices = np.where(~visited)[0]
            
            for i in visited_indices:
                for j in unvisited_indices:
                    if distance_matrix[i, j] < min_edge:
                        min_edge = distance_matrix[i, j]
                        min_i, min_j = i, j
            
            # Mark edge for MST bias
            mst_bias[min_i, min_j] = mst_weight
            mst_bias[min_j, min_i] = mst_weight
            visited[min_j] = True
    
    # 5. Combine factors with weights
    heuristic_matrix = reciprocal_dist.copy()
    
    # Apply centrality: edges connecting to central nodes get boost
    centrality_boost = np.outer(centrality, centrality) * centrality_weight
    heuristic_matrix += centrality_boost
    
    # Add MST bias
    heuristic_matrix += mst_bias
    
    # 6. Sparsification: keep only promising edges
    # Strategy: Keep edges that are in top percentile or connect nearest neighbors
    sparsity_threshold = 0.7  # Keep top 30% of edges for each node
    
    # Create mask for promising edges
    promising_mask = np.zeros_like(heuristic_matrix, dtype=bool)
    
    for i in range(n):
        # Keep nearest neighbors
        promising_mask[i] = promising_mask[i] | nearest_neighbor_mask[i]
        
        # Keep edges above threshold percentile
        row_values = heuristic_matrix[i].copy()
        row_values[i] = -np.inf  # Exclude self
        
        if np.sum(row_values > 0) > 0:  # Check if there are positive values
            threshold = np.percentile(row_values[row_values > 0], 
                                     sparsity_threshold * 100)
            promising_mask[i] = promising_mask[i] | (row_values >= threshold)
    
    # Make symmetric (undirected TSP)
    promising_mask = promising_mask | promising_mask.T
    
    # Apply sparsification
    heuristic_matrix = heuristic_matrix * promising_mask
    
    # 7. Final normalization to make values comparable across instances
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristic_matrix, 0)
    
    return heuristic_matrix
