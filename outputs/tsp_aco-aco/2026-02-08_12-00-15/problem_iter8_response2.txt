import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Simple yet effective TSP heuristics based on essential components:
    1. Inverse distance (fundamental measure)
    2. Minimum Spanning Tree edges
    3. Nearest neighbor connections
    
    The function returns normalized scores where higher values indicate
    edges that are more promising for inclusion in TSP solutions.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
        
    Returns:
        Heuristic matrix with same shape as input, values in [0, 1].
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # 1. Base heuristic: inverse distance (closer nodes are more promising)
    # Copy to avoid modifying input, set diagonal to infinity to avoid division by zero
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree edges (essential for TSP connectivity)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbors (local connectivity pattern)
    # Use adaptive k: smaller k for larger instances
    k = max(2, min(5, int(np.sqrt(n))))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i][:k]
        nn_mask[i, nearest] = True
    
    # Make nearest neighbor mask symmetric
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Combine components
    # Start with base heuristic, boost MST and NN edges
    heuristics = base_heuristic.copy()
    
    # Boost MST edges more strongly as they're essential for connectivity
    heuristics[mst_mask] *= 2.0
    
    # Boost nearest neighbor edges moderately
    heuristics[nn_mask] *= 1.5
    
    # 5. Apply sparsity: keep only promising edges
    # Keep edges that are in MST, nearest neighbors, or have strong base heuristic
    for i in range(n):
        # For each node, keep top edges based on heuristic value
        # Keep more edges for smaller instances
        keep_count = min(n-1, max(3, int(np.sqrt(n) * 1.5)))
        
        row_heuristics = heuristics[i].copy()
        row_heuristics[i] = -np.inf  # Ignore diagonal
        
        # Always keep MST and NN edges
        keep_mask = mst_mask[i] | nn_mask[i]
        
        # Add additional top edges if needed
        if np.sum(keep_mask) < keep_count:
            # Get top heuristic edges not already kept
            all_indices = np.arange(n)
            not_kept = all_indices[~keep_mask & (all_indices != i)]
            
            if len(not_kept) > 0:
                # Get top values among not kept edges
                values = row_heuristics[not_kept]
                if len(values) > 0:
                    # Determine how many more to keep
                    add_count = min(len(values), keep_count - np.sum(keep_mask))
                    
                    # Get indices of top edges to add
                    top_indices = np.argsort(values)[-add_count:]
                    top_nodes = not_kept[top_indices]
                    
                    # Mark these edges to keep
                    keep_mask[top_nodes] = True
        
        # Zero out edges not in keep_mask
        heuristics[i, ~keep_mask] = 0
    
    # Ensure symmetry
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 6. Normalize to [0, 1] range
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
