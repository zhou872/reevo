import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection.
    Incorporates MST and nearest neighbor overlap bonuses, logarithmic
    adaptive sparsification, and triangle-based connectivity sampling.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.5
    
    # 3. Nearest neighbor bonus with adaptive k
    k = max(3, min(8, int(np.log2(n)) + 2))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i][:k]
        nn_mask[i, nearest] = True
    
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.8
    
    # 4. Overlap bonus for edges in both MST and NN
    overlap_mask = mst_mask & nn_mask
    overlap_bonus = 2.2
    
    # 5. Triangle sampling bonus for connectivity
    triangle_bonus = np.ones_like(distance_matrix)
    if n >= 3:
        for i in range(n):
            for j in range(i+1, n):
                # Find the best third node that completes a triangle
                min_tri_sum = np.inf
                for k in range(n):
                    if k != i and k != j:
                        tri_sum = distance_matrix[i, k] + distance_matrix[k, j]
                        if tri_sum < min_tri_sum:
                            min_tri_sum = tri_sum
                
                # Bonus if direct edge is significantly shorter than triangle path
                if min_tri_sum > 0:
                    ratio = min_tri_sum / distance_matrix[i, j]
                    triangle_bonus[i, j] = triangle_bonus[j, i] = min(2.0, 1.0 + 0.3 * np.log1p(ratio - 1))
    
    # 6. Combine all factors multiplicatively
    heuristics = base_heuristic.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    heuristics[overlap_mask] *= overlap_bonus
    heuristics *= triangle_bonus
    
    # 7. Logarithmic adaptive sparsification
    # Keep more edges for small graphs, fewer for large graphs
    log_factor = 2.0 / np.log10(max(n, 10))
    keep_percentile = 100 - 30 * log_factor
    
    combined_mask = mst_mask | nn_mask | overlap_mask
    
    for i in range(n):
        row_values = heuristics[i]
        valid_values = row_values[row_values > 0]
        
        if len(valid_values) > 0:
            threshold = np.percentile(valid_values, keep_percentile)
            mask_i = row_values >= threshold
            combined_mask[i] = combined_mask[i] | mask_i
    
    # Apply the combined mask
    heuristics = heuristics * combined_mask
    
    # 8. Normalize for consistency
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Ensure symmetry (should already be symmetric, but just in case)
    heuristics = (heuristics + heuristics.T) / 2
    
    return heuristics
