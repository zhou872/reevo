import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP that combines multiple factors to evaluate edge potential.
    Features:
    1. Inverse distance baseline
    2. Minimum Spanning Tree inclusion bonus
    3. Nearest neighbor connectivity
    4. Sparsification of unpromising edges
    """
    n = distance_matrix.shape[0]
    
    # 1. Baseline inverse distance
    inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)  # Remove self-connections
    
    # 2. Minimum Spanning Tree (MST) inclusion
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0  # Symmetrize MST edges
    mst_bonus = 2.0  # Bonus factor for MST edges
    
    # 3. Nearest neighbor connectivity
    k = max(2, n // 10)  # Adaptive k-nearest neighbors
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nearest = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, nearest] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    nn_bonus = 1.5  # Bonus factor for nearest neighbor edges
    
    # 4. Combine factors with bonuses
    heuristics = inv_dist.copy()
    heuristics[mst_mask] *= mst_bonus
    heuristics[nn_mask] *= nn_bonus
    
    # 5. Sparsification: keep only top edges for each node
    sparsity_factor = 0.3  # Keep top 30% of edges per node
    threshold_mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        # Threshold based on percentile of non-zero values
        row_values = heuristics[i]
        non_zero_mask = row_values > 0
        if non_zero_mask.any():
            threshold = np.percentile(row_values[non_zero_mask], 
                                    100 * (1 - sparsity_factor))
            threshold_mask[i] = row_values >= threshold
    
    # Apply sparsification mask
    heuristics = heuristics * threshold_mask
    
    # 6. Final normalization for stability
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
