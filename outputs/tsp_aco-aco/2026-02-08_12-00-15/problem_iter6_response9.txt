import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristics with stronger MST emphasis, adaptive sparsification,
    and improved normalization for stochastic solution sampling.
    
    Key improvements over v1:
    1. Even stronger MST edge boosting (factor 6.0)
    2. Dynamic edge retention based on problem density
    3. Logarithmic scaling for MST edges
    4. Final symmetry through max operator
    5. Robust normalization to handle edge cases
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n) with zeros on diagonal
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, with higher values indicating
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    
    if n <= 2:
        # Handle trivial cases
        return np.zeros_like(distance_matrix)
    
    # 1. Baseline: inverse distance with safe handling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / np.where(distance_matrix > 0, distance_matrix, np.inf)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree as solution backbone
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst + mst.T) > 0
    
    # Strong MST boost with logarithmic scaling for robustness
    heuristics = inv_dist.copy()
    mst_boost = 6.0  # Even stronger emphasis on MST backbone
    heuristics[mst_edges] *= mst_boost * (1 + np.log(1 + n) / 10)
    
    # 3. Adaptive per-node edge retention based on problem size
    # More edges for small problems, fewer for large ones
    if n <= 15:
        edges_per_node = min(n - 1, 10)  # Keep most edges for small instances
    elif n <= 30:
        edges_per_node = min(n - 1, 8)
    elif n <= 50:
        edges_per_node = min(n - 1, 6)
    elif n <= 100:
        edges_per_node = min(n - 1, 5)
    else:
        edges_per_node = min(n - 1, 4)
    
    # 4. Keep only top edges for each node
    node_masks = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        row_values = heuristics[i].copy()
        row_values[i] = -np.inf  # Exclude self
        
        if edges_per_node < n - 1:
            # Get indices of top edges using argpartition for efficiency
            top_indices = np.argpartition(-row_values, edges_per_node)[:edges_per_node]
            node_masks[i, top_indices] = True
        else:
            # Keep all except self
            node_masks[i, :] = True
            node_masks[i, i] = False
    
    # 5. Ensure symmetry by keeping edge if selected by either endpoint
    symmetric_mask = node_masks | node_masks.T
    
    # Apply mask to heuristics
    heuristics = heuristics * symmetric_mask
    
    # 6. Additional boost for MST edges that survived sparsification
    # This reinforces the backbone structure
    heuristics[mst_edges & symmetric_mask] *= 1.5
    
    # 7. Final normalization for stability
    heuristics[heuristics < 1e-10] = 0
    
    # Ensure symmetry through max operator (numerical stability)
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # Normalize to [0, 1] range
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Ensure diagonal remains zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
