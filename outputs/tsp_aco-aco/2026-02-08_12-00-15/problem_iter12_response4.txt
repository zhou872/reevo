import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    TSP edge selection heuristics prioritizing MST and nearest neighbors
    with adaptive, scalable sparsification.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape with values in [0, 1],
        where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance with safe diagonal
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree edges (essential connectivity)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbors with adaptive scaling
    # Use logarithmic scaling for k-nearest neighbors
    k_neighbors = max(2, int(np.ceil(np.log2(n))) + 1)
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        neighbors = np.argsort(safe_dist[i])[1:k_neighbors+1]
        nn_mask[i, neighbors] = True
    
    # Symmetrize nearest neighbor mask
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Combined heuristic with tiered bonuses
    heuristics = base_heuristic.copy()
    
    # MST edges get highest priority (tier 1)
    heuristics[mst_mask] *= 3.0
    
    # Nearest neighbor edges not in MST get medium priority (tier 2)
    nn_not_mst = nn_mask & ~mst_mask
    heuristics[nn_not_mst] *= 1.8
    
    # 5. Adaptive sparsification with size-dependent parameters
    # Target edge density decreases with problem size
    if n <= 20:
        target_density = 0.6
    elif n <= 50:
        target_density = 0.4
    elif n <= 100:
        target_density = 0.25
    else:
        target_density = 0.15
    
    # Always keep MST and nearest neighbor edges
    essential_mask = mst_mask | nn_mask
    
    # For non-essential edges, apply threshold-based sparsification
    non_essential = heuristics.copy()
    non_essential[essential_mask] = -np.inf  # Ignore essential edges
    
    # Calculate threshold to achieve target density
    n_potential_edges = n * (n - 1) // 2
    n_target_edges = int(target_density * n_potential_edges)
    
    if n_target_edges > np.sum(essential_mask) // 2:  # Account for symmetry
        # Get top non-essential edges to reach target count
        flat_non_essential = non_essential[np.triu_indices(n, 1)]
        sorted_non_essential = np.sort(flat_non_essential)[::-1]
        
        n_needed = n_target_edges - np.sum(essential_mask) // 2
        if len(sorted_non_essential) > 0 and n_needed > 0:
            threshold = sorted_non_essential[min(n_needed, len(sorted_non_essential)) - 1]
            selected_non_essential = non_essential >= threshold
            selected_mask = essential_mask | selected_non_essential
        else:
            selected_mask = essential_mask
    else:
        selected_mask = essential_mask
    
    # Apply sparsification mask (symmetrize)
    selected_mask = selected_mask | selected_mask.T
    heuristics = heuristics * selected_mask
    
    # 6. Ensure minimum connectivity
    for i in range(n):
        connected = np.where(heuristics[i] > 0)[0]
        if len(connected) < 2:
            # Find nearest nodes not already connected
            sorted_dists = np.argsort(safe_dist[i])
            for j in sorted_dists:
                if j != i and heuristics[i, j] == 0:
                    heuristics[i, j] = base_heuristic[i, j] * 0.5  # Reduced bonus
                    heuristics[j, i] = base_heuristic[j, i] * 0.5
                    if len(np.where(heuristics[i] > 0)[0]) >= 2:
                        break
    
    # 7. Final normalization with softmax-style scaling
    max_val = heuristics.max()
    if max_val > 0:
        # Scale to emphasize relative differences
        scale = max_val / 4.0
        heuristics = np.exp(heuristics / scale)
        heuristics = heuristics / heuristics.max()
    
    # Ensure symmetry and zero diagonal
    heuristics = np.maximum(heuristics, heuristics.T)
    np.fill_diagonal(heuristics, 0.0)
    
    return heuristics
