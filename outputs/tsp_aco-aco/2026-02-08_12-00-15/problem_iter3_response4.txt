import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristics with triangle inequality penalties and adaptive sparsification.
    
    Features:
    1. Multiplicative combination of MST, nearest neighbor, and triangle inequality factors
    2. Adaptive sparsification based on problem size
    3. Triangle inequality penalties for edges that can be shortcut
    4. Top edge boosting for discriminative power
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between cities
        
    Returns:
    --------
    np.ndarray
        Heuristic scores for each edge (same shape as input)
    """
    n = distance_matrix.shape[0]
    
    # 1. Baseline: inverse distance
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-10)  # Avoid division by zero
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree factor
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_factor = np.ones_like(distance_matrix)
    mst_factor[mst_mask] = 2.0  # Boost MST edges
    
    # 3. Nearest neighbor connectivity
    # Adaptive k: larger k for small problems, smaller k for large problems
    k = max(2, min(10, n // 5))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        nearest = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, nearest] = True
    
    nn_mask = nn_mask | nn_mask.T
    nn_factor = np.ones_like(distance_matrix)
    nn_factor[nn_mask] = 1.5
    
    # 4. Triangle inequality penalty
    # Penalize edges that can be significantly shortcut via a third node
    tri_penalty = np.ones_like(distance_matrix)
    epsilon = 0.15  # Tolerance for triangle inequality
    
    # For each pair (i,j), check if there exists k such that
    # dist(i,k) + dist(k,j) < dist(i,j) * (1 - epsilon)
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] == 0:
                continue
                
            # Check a subset of potential shortcut nodes
            # Focus on nodes that are close to i or j
            candidates_i = np.argsort(distance_matrix[i])[1:min(20, n)]
            candidates_j = np.argsort(distance_matrix[j])[1:min(20, n)]
            candidates = np.union1d(candidates_i, candidates_j)
            
            min_detour = np.inf
            for k in candidates:
                if k != i and k != j:
                    detour = distance_matrix[i, k] + distance_matrix[k, j]
                    if detour < min_detour:
                        min_detour = detour
            
            if min_detour < distance_matrix[i, j] * (1 - epsilon):
                # This edge can be shortcut - apply penalty
                penalty = 0.7 + 0.3 * (min_detour / distance_matrix[i, j])
                tri_penalty[i, j] = penalty
                tri_penalty[j, i] = penalty
    
    # 5. Combine all factors multiplicatively
    heuristics = inv_dist * mst_factor * nn_factor * tri_penalty
    
    # 6. Adaptive sparsification based on problem size
    # Keep more edges for small problems, fewer for large problems
    if n <= 30:
        keep_fraction = 0.5
    elif n <= 100:
        keep_fraction = 0.3
    else:
        keep_fraction = 0.2
    
    sparse_mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        row = heuristics[i].copy()
        # Set diagonal to -inf to ensure it's never selected
        row[i] = -np.inf
        
        if np.all(row <= 0):
            continue
            
        # Keep top edges based on percentile
        threshold = np.percentile(row[row > 0], 100 * (1 - keep_fraction))
        sparse_mask[i] = row >= threshold
    
    # Ensure symmetry in sparsification
    sparse_mask = sparse_mask | sparse_mask.T
    
    # Apply sparsification
    heuristics = heuristics * sparse_mask
    
    # 7. Boost top edges for better discrimination
    # Apply non-linear scaling to emphasize strong edges
    if np.max(heuristics) > 0:
        heuristics_normalized = heuristics / np.max(heuristics)
        # Square to boost high values
        heuristics = heuristics_normalized ** 0.7
    
    # 8. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
