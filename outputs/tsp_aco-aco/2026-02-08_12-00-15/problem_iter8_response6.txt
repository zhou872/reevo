import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge selection heuristics focusing on MST and nearest neighbors
    with simple, adaptive bonuses.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create safe distance matrix with diagonal set to infinity
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # Base heuristic: reciprocal of distance
    base_heuristic = 1.0 / safe_dist
    
    # Compute Minimum Spanning Tree mask
    mst_mask = _compute_mst_mask(safe_dist)
    
    # Adaptive nearest neighbors
    k = max(2, min(5, n // 4))
    nn_mask = _compute_nn_mask(safe_dist, k)
    
    # Combine heuristics with simple adaptive bonuses
    heuristics = base_heuristic.copy()
    
    # Apply MST bonus (strong)
    heuristics[mst_mask] *= 2.0
    
    # Apply nearest neighbors bonus (moderate)
    heuristics[nn_mask] *= 1.5
    
    # Keep only top edges for each node to maintain sparsity
    sparsified = np.zeros_like(heuristics)
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Adaptive number of edges to keep
        keep_count = max(2, min(5, n // 4))
        top_indices = np.argpartition(-row, keep_count)[:keep_count]
        sparsified[i, top_indices] = row[top_indices]
    
    # Ensure symmetry and zero diagonal
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0.0)
    
    # Normalize to [0, 1] range
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified


def _compute_mst_mask(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Compute Minimum Spanning Tree mask using Prim's algorithm.
    
    Args:
        distance_matrix: Distance matrix with diagonal set to inf.
    
    Returns:
        Boolean mask indicating edges in the MST.
    """
    n = distance_matrix.shape[0]
    mst_mask = np.zeros((n, n), dtype=bool)
    
    if n <= 1:
        return mst_mask
    
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    
    for _ in range(n - 1):
        min_dist = np.inf
        min_i, min_j = -1, -1
        
        # Find minimum edge between visited and unvisited nodes
        visited_indices = np.where(visited)[0]
        unvisited_indices = np.where(~visited)[0]
        
        for i in visited_indices:
            distances_to_unvisited = distance_matrix[i, unvisited_indices]
            if len(distances_to_unvisited) == 0:
                continue
                
            current_min = np.min(distances_to_unvisited)
            if current_min < min_dist:
                min_dist = current_min
                min_i = i
                min_j = unvisited_indices[np.argmin(distances_to_unvisited)]
        
        if min_i != -1:
            mst_mask[min_i, min_j] = True
            mst_mask[min_j, min_i] = True
            visited[min_j] = True
    
    return mst_mask


def _compute_nn_mask(distance_matrix: np.ndarray, k: int) -> np.ndarray:
    """
    Compute k-nearest neighbors mask.
    
    Args:
        distance_matrix: Distance matrix with diagonal set to inf.
        k: Number of nearest neighbors to consider.
    
    Returns:
        Boolean mask indicating nearest neighbor edges.
    """
    n = distance_matrix.shape[0]
    nn_mask = np.zeros((n, n), dtype=bool)
    
    if n <= 1 or k <= 0:
        return nn_mask
    
    for i in range(n):
        row = distance_matrix[i].copy()
        
        # Get k nearest neighbors (excluding self, already handled by inf diagonal)
        nearest_indices = np.argpartition(row, k)[:k]
        nn_mask[i, nearest_indices] = True
    
    # Ensure symmetry
    nn_mask = nn_mask | nn_mask.T
    
    return nn_mask
