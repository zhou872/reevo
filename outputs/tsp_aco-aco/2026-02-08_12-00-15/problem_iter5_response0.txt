import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with triangle inequality penalty
    and adaptive logarithmic boosting of promising edges.
    
    Args:
        distance_matrix: Symmetric 2D array of pairwise distances.
    
    Returns:
        Heuristic matrix of same shape where higher values indicate 
        more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base heuristic: inverse distance (avoid division by zero)
    safe_dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(safe_dist, np.inf)
    base_heuristic = 1.0 / safe_dist
    np.fill_diagonal(base_heuristic, 0.0)
    
    # 2. Minimum Spanning Tree mask
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbor mask (adaptive k based on problem size)
    k = max(2, min(8, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Vectorized nearest neighbor computation
    for i in range(n):
        nearest = np.argsort(safe_dist[i])[:k+1]
        nearest = nearest[nearest != i]
        nn_mask[i, nearest[:min(k, len(nearest))]] = True
    
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Multiply MST and NN masks as suggested
    combined_mask = mst_mask * nn_mask  # Element-wise multiplication
    
    # 5. Triangle inequality violation penalty
    # Create penalty matrix (1.0 = no penalty, <1.0 = penalty)
    penalty_matrix = np.ones_like(distance_matrix)
    
    # Vectorized check for triangle inequality violations
    # For each triple (i,j,k), check if d(i,j) > d(i,k) + d(k,j)
    # We'll sample a subset of k to make it computationally feasible
    if n > 100:
        # For large graphs, sample 20% of nodes
        k_samples = np.random.choice(n, size=max(10, n//5), replace=False)
    else:
        k_samples = np.arange(n)
    
    for k in k_samples:
        # Compute d(i,k) + d(k,j) for all i,j
        with np.errstate(invalid='ignore'):
            triangle_sum = safe_dist[:, k:k+1] + safe_dist[k:k+1, :]
        
        # Find violations where direct distance > triangle path
        violations = safe_dist > triangle_sum
        # Penalize violations (reduce heuristic value)
        penalty_matrix[violations] *= 0.7
    
    # 6. Adaptive sparsification
    # Determine how many edges to keep per node based on problem size
    edges_per_node = max(3, min(15, int(np.sqrt(n))))
    keep_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get combined score for this node's edges
        edge_scores = base_heuristic[i].copy()
        edge_scores[combined_mask[i]] *= 2.0  # Boost edges in both MST and NN
        
        # Keep top edges per node
        top_indices = np.argsort(edge_scores)[-edges_per_node:]
        keep_mask[i, top_indices] = True
        
        # Also always keep MST edges
        keep_mask[i, mst_mask[i]] = True
    
    # Symmetrize the keep mask
    keep_mask = keep_mask | keep_mask.T
    
    # 7. Combine all factors with penalty
    heuristics = base_heuristic.copy()
    heuristics[combined_mask] *= 2.0  # Boost edges in both MST and NN
    heuristics *= penalty_matrix  # Apply triangle inequality penalty
    
    # Apply sparsification
    heuristics = heuristics * keep_mask
    
    # 8. Logarithmic boosting of top edges
    # Identify top edges globally
    all_values = heuristics[heuristics > 0]
    if len(all_values) > 0:
        threshold = np.percentile(all_values, 85)
        top_edges = heuristics >= threshold
        
        # Apply logarithmic boost: log(1 + score) amplifies high scores more
        boost_factor = np.log1p(heuristics[top_edges] * 2)
        heuristics[top_edges] *= (1.0 + boost_factor / 3.0)
    
    # 9. Final normalization and symmetry enforcement
    heuristics = (heuristics + heuristics.T) / 2.0
    np.fill_diagonal(heuristics, 0.0)
    
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
