import numpy as np
import warnings
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.linalg import eigh

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic combining theoretical guarantees with local optimization principles.
    
    Core Innovations:
    1. Christofides-inspired baseline with matching approximation
    2. Adaptive regret penalization based on instance distance distribution
    3. Multi-scale lookahead with dynamic depth adjustment
    4. Lin-Kernighan inspired local optimization potential assessment
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (shape n x n)
        
    Returns:
        Heuristic matrix of same shape as input, with scores in [0, 1]
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # Suppress warnings for clean output
    warnings.filterwarnings('ignore')
    
    # ===== 1. DYNAMIC PARAMETER ADJUSTMENT =====
    # Extract instance characteristics for adaptive behavior
    triu_indices = np.triu_indices(n, 1)
    distances = distance_matrix[triu_indices]
    dist_mean = np.mean(distances)
    dist_std = np.std(distances)
    dist_cv = dist_std / (dist_mean + 1e-12)  # Coefficient of variation
    
    # Dynamic lookahead depth based on instance size and complexity
    if n <= 20:
        lookahead_hops = 3
    elif n <= 50:
        lookahead_hops = 2
    else:
        lookahead_hops = 1
    
    # Adaptive parameters based on instance characteristics
    regret_power = 1.5 + 0.8 * np.tanh(2 * dist_cv)
    mst_weight = 2.0 + 0.5 * np.tanh(dist_cv - 1.0)
    
    # ===== 2. CHRISTOFIDES APPROXIMATION BASELINE =====
    def _compute_christofides_components(dist_mat: np.ndarray):
        """Compute MST and min-weight matching components for Christofides approximation."""
        n_nodes = dist_mat.shape[0]
        christofides_mask = np.zeros((n_nodes, n_nodes), dtype=bool)
        
        # Minimum Spanning Tree using Prim's algorithm
        visited = np.zeros(n_nodes, dtype=bool)
        min_edge = np.full(n_nodes, np.inf)
        parent = np.full(n_nodes, -1, dtype=int)
        min_edge[0] = 0
        
        for _ in range(n_nodes):
            u = np.argmin(np.where(visited, np.inf, min_edge))
            visited[u] = True
            if parent[u] != -1:
                christofides_mask[u, parent[u]] = christofides_mask[parent[u], u] = True
            
            # Update distances to unvisited neighbors
            for v in range(n_nodes):
                if not visited[v] and dist_mat[u, v] < min_edge[v]:
                    min_edge[v] = dist_mat[u, v]
                    parent[v] = u
        
        # Identify odd-degree vertices in MST
        mst_degrees = np.sum(christofides_mask, axis=1)
        odd_vertices = np.where(mst_degrees % 2 == 1)[0]
        
        # Compute minimum-weight perfect matching approximation
        if len(odd_vertices) >= 2:
            # Greedy matching for odd-degree vertices
            odd_distances = dist_mat[np.ix_(odd_vertices, odd_vertices)]
            np.fill_diagonal(odd_distances, np.inf)
            
            matched = np.zeros(len(odd_vertices), dtype=bool)
            for i in range(len(odd_vertices)):
                if not matched[i]:
                    u = odd_vertices[i]
                    # Find closest unmatched odd vertex
                    remaining = np.where(~matched)[0]
                    if len(remaining) > 1:
                        v_idx = remaining[np.argmin(odd_distances[i, remaining])]
                        v = odd_vertices[v_idx]
                        if v_idx != i:
                            christofides_mask[u, v] = christofides_mask[v, u] = True
                            matched[i] = matched[v_idx] = True
        
        return christofides_mask
    
    christofides_mask = _compute_christofides_components(distance_matrix)
    
    # ===== 3. BASE INVERSE DISTANCE WITH ADAPTIVE SCALING =====
    scale = dist_mean if dist_mean > 0 else 1.0
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = scale / (distance_matrix + 0.15 * scale * np.eye(n))
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # ===== 4. ADAPTIVE REGRET-BASED PENALIZATION =====
    regret_penalty = np.ones((n, n))
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_idx = np.argsort(dists)
        best_dist = dists[sorted_idx[0]]
        
        if best_dist > 0:
            for rank, j in enumerate(sorted_idx):
                if rank == 0:
                    regret_penalty[i, j] = 1.0  # Best edge gets no penalty
                else:
                    # Combined rank and distance ratio penalty
                    norm_rank = rank / (n - 2)  # Normalize to [0, 1]
                    dist_ratio = dists[j] / best_dist
                    penalty = 1 / (1 + regret_power * norm_rank * np.sqrt(dist_ratio))
                    regret_penalty[i, j] = penalty
    
    # Symmetrize using geometric mean
    regret_penalty = np.sqrt(regret_penalty * regret_penalty.T)
    
    # ===== 5. MUTUAL K-NN PROXIMITY BONUS =====
    # Adaptive K based on problem size and density
    k_nn = min(max(3, int(np.sqrt(n) + dist_cv * 2)), n - 1)
    nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:k_nn + 1]
    mutual_nn_bonus = np.ones((n, n))
    
    for i in range(n):
        for j in range(i + 1, n):
            if j in nearest_indices[i] and i in nearest_indices[j]:
                # Both are in each other's K-NN
                rank_i = np.where(nearest_indices[i] == j)[0][0]
                rank_j = np.where(nearest_indices[j] == i)[0][0]
                avg_rank = (rank_i + rank_j) / (2 * (k_nn - 1))
                bonus = 1.0 + 0.6 * (1.0 - avg_rank)
                mutual_nn_bonus[i, j] = mutual_nn_bonus[j, i] = bonus
    
    # ===== 6. MULTI-SCALE LOOKAHEAD PATH EFFICIENCY =====
    lookahead_score = np.ones((n, n))
    if n >= 4 and lookahead_hops >= 2:
        # Dynamic neighborhood size for lookahead
        lookahead_k = min(max(4, int(1.5 * np.sqrt(n) + dist_cv * 3)), n - 1)
        nearest_indices_lookahead = np.argsort(distance_matrix, axis=1)[:, 1:lookahead_k + 1]
        
        for i in range(n):
            for j in range(i + 1, n):
                direct_dist = distance_matrix[i, j]
                if direct_dist <= 0:
                    continue
                
                best_alternative = np.inf
                
                # 2-hop paths through common neighbors
                i_neighbors = set(nearest_indices_lookahead[i].tolist())
                j_neighbors = set(nearest_indices_lookahead[j].tolist())
                common_neighbors = i_neighbors.intersection(j_neighbors)
                
                for k in common_neighbors:
                    if k != i and k != j:
                        alt_path = distance_matrix[i, k] + distance_matrix[k, j]
                        best_alternative = min(best_alternative, alt_path)
                
                # 3-hop paths if enabled and no good 2-hop found
                if lookahead_hops >= 3 and (best_alternative / direct_dist > 1.3):
                    # Consider limited 3-hop paths
                    top_i = list(i_neighbors)[:min(3, len(i_neighbors))]
                    top_j = list(j_neighbors)[:min(3, len(j_neighbors))]
                    
                    for k in top_i:
                        if k == j:
                            continue
                        for l in top_j:
                            if l == i or l == k:
                                continue
                            # Check connectivity between k and l
                            if (k in nearest_indices_lookahead[l] or 
                                l in nearest_indices_lookahead[k]):
                                alt_path = (distance_matrix[i, k] + 
                                           distance_matrix[k, l] + 
                                           distance_matrix[l, j])
                                best_alternative = min(best_alternative, alt_path)
                
                # Calculate efficiency score
                if best_alternative < np.inf:
                    efficiency = best_alternative / direct_dist
                    # Sigmoid transformation: higher score when direct edge is clearly better
                    score = 0.5 + 1.5 / (1.0 + np.exp(-2.0 * (efficiency - 1.0)))
                    lookahead_score[i, j] = lookahead_score[j, i] = score
    
    # ===== 7. LOCAL OPTIMIZATION POTENTIAL (LIN-KERNIGHAN INSPIRED) =====
    optimization_potential = np.ones((n, n))
    if n >= 5:  # Only compute for reasonably sized instances
        # Estimate potential for 2-opt improvements involving each edge
        for i in range(n):
            # Get closest neighbors for i
            i_neighbors = np.argsort(distance_matrix[i])[1:min(6, n)]
            
            for j in range(i + 1, n):
                current_dist = distance_matrix[i, j]
                if current_dist <= 0:
                    continue
                
                # Get closest neighbors for j
                j_neighbors = np.argsort(distance_matrix[j])[1:min(6, n)]
                
                best_gain = 0
                
                # Check 2-opt gains
                for k in i_neighbors:
                    if k == j:
                        continue
                    for l in j_neighbors:
                        if l == i or l == k:
                            continue
                        
                        # Calculate 2-opt gain
                        gain = (current_dist + distance_matrix[k, l] - 
                               distance_matrix[i, k] - distance_matrix[j, l])
                        best_gain = max(best_gain, gain)
                
                # Convert gain to optimization potential score
                if best_gain > 0:
                    # Normalize gain by average distance
                    norm_gain = min(best_gain / dist_mean, 2.0)
                    # Higher potential when gain is significant
                    potential = 1.0 + 0.4 * np.tanh(norm_gain)
                    optimization_potential[i, j] = optimization_potential[j, i] = potential
    
    # ===== 8. ADAPTIVE COMPONENT FUSION =====
    # Dynamic weighting based on instance characteristics
    if dist_cv > 0.5:  # Highly variable distances
        weights = {
            'inverse': 1.0,
            'regret': 1.2,
            'lookahead': 0.8,
            'optimization': 1.1,
            'mutual_nn': 0.9
        }
    elif n > 30:  # Large instances
        weights = {
            'inverse': 1.0,
            'regret': 1.0,
            'lookahead': 1.2,
            'optimization': 0.7,
            'mutual_nn': 0.8
        }
    else:  # Small to medium instances
        weights = {
            'inverse': 1.0,
            'regret': 1.1,
            'lookahead': 1.0,
            'optimization': 1.0,
            'mutual_nn': 1.0
        }
    
    # Combine all components with adaptive weights
    heuristics = inv_dist.copy()
    heuristics *= regret_penalty ** weights['regret']
    heuristics *= lookahead_score ** weights['lookahead']
    heuristics *= mutual_nn_bonus ** weights['mutual_nn']
    heuristics *= optimization_potential ** weights['optimization']
    
    # Strong boost for Christofides approximation edges
    heuristics[christofides_mask] *= 1.8
    
    # Ensure symmetry and zero diagonal
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # ===== 9. CONNECTIVITY-AWARE ADAPTIVE SPARSIFICATION =====
    if n > 3:
        # Estimate local connectivity using top edges
        top_k = min(max(4, int(np.sqrt(n))), n - 1)
        connectivity = np.zeros(n)
        
        for i in range(n):
            row = heuristics[i].copy()
            row[i] = 0
            # Consider top-k edges for connectivity
            if np.sum(row > 0) >= top_k:
                top_values = np.partition(row, -top_k)[-top_k:]
                connectivity[i] = np.mean(top_values[top_values > 0])
        
        # Normalize connectivity
        max_conn = np.max(connectivity)
        if max_conn > 0:
            connectivity /= max_conn
        
        # Apply adaptive thresholding
        for i in range(n):
            row = heuristics[i].copy()
            row[i] = 0
            
            # Always keep Christofides edges
            keep_mask = christofides_mask[i].copy()
            
            # Determine number of edges to keep
            if connectivity[i] > 0.7:  # Well-connected node
                keep_count = min(int(1.5 * np.sqrt(n)), n - 1)
            elif connectivity[i] < 0.3:  # Poorly-connected node
                keep_count = min(int(0.8 * np.sqrt(n)) + 2, n - 1)
            else:
                keep_count = min(int(1.2 * np.sqrt(n)), n - 1)
            
            keep_count = max(keep_count, np.sum(keep_mask) + 2)
            
            # Keep top edges based on heuristic score
            non_zero = row[row > 0]
            if len(non_zero) > keep_count:
                threshold = np.sort(non_zero)[-keep_count]
                low_score_mask = (row < threshold) & (~keep_mask)
                row[low_score_mask] = 0
            
            heuristics[i] = row
        
        # Re-symmetrize
        heuristics = (heuristics + heuristics.T) / 2
    
    # ===== 10. FINAL NORMALIZATION AND VALIDATION =====
    np.fill_diagonal(heuristics, 0)
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Ensure exact symmetry and zero diagonal
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # Safety check: ensure all values are in [0, 1]
    heuristics = np.clip(heuristics, 0, 1)
    
    return heuristics
