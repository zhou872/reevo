import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    
    Features:
    1. Distance-Proximity Weighting: Inverse distance weighting with normalization
    2. MST Structure Bonus: Bonus for edges in minimum spanning tree
    3. Neighbor Affinity Sparsification: Keep only top-k connections per node
    4. Reciprocal Distance Penalty: Penalize edges based on node connectivity
    
    Returns: Matrix of same shape as input with edge promise scores
    """
    n = distance_matrix.shape[0]
    
    # 1. Basic inverse distance (avoid division by zero)
    with np.errstate(divide='ignore'):
        inv_dist = 1.0 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree bonus
    mst_bonus = np.zeros((n, n))
    # Simple Prim's algorithm for MST detection
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    for _ in range(n - 1):
        # Find minimum edge from visited to unvisited
        min_edge = np.inf
        min_i, min_j = 0, 0
        for i in range(n):
            if visited[i]:
                for j in range(n):
                    if not visited[j] and distance_matrix[i, j] < min_edge:
                        min_edge = distance_matrix[i, j]
                        min_i, min_j = i, j
        visited[min_j] = True
        mst_bonus[min_i, min_j] = 1.0
        mst_bonus[min_j, min_i] = 1.0
    
    # 3. Neighbor affinity - keep only promising connections
    neighbor_affinity = np.zeros((n, n))
    k = max(2, n // 4)  # Keep connections to ~25% of nodes
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        distances_i = distance_matrix[i].copy()
        distances_i[i] = np.inf
        nearest_indices = np.argpartition(distances_i, k)[:k]
        
        # Assign scores based on rank (closer = higher score)
        sorted_indices = nearest_indices[np.argsort(distances_i[nearest_indices])]
        for rank, j in enumerate(sorted_indices):
            neighbor_affinity[i, j] = (k - rank) / k
    
    # 4. Reciprocal distance penalty (penalize nodes that are far from others)
    node_centrality = np.zeros(n)
    for i in range(n):
        # Average distance to other nodes (lower = more central)
        node_centrality[i] = 1.0 / (np.sum(distance_matrix[i]) / (n - 1) + 1e-10)
    
    # Normalize centrality
    node_centrality = node_centrality / np.max(node_centrality)
    
    # Combine all factors
    heuristics_matrix = np.zeros((n, n))
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            # Base score from inverse distance
            base_score = inv_dist[i, j]
            
            # Apply MST bonus
            if mst_bonus[i, j] > 0:
                base_score *= 2.0
            
            # Apply neighbor affinity
            affinity_score = (neighbor_affinity[i, j] + neighbor_affinity[j, i]) / 2
            base_score *= (1.0 + affinity_score)
            
            # Apply reciprocal centrality (both nodes should be relatively central)
            centrality_factor = (node_centrality[i] + node_centrality[j]) / 2
            base_score *= centrality_factor
            
            heuristics_matrix[i, j] = base_score
    
    # 5. Sparsification - keep only top connections
    sparsity_factor = 0.3  # Keep top 30% of connections
    threshold = np.percentile(heuristics_matrix[heuristics_matrix > 0], 
                             100 * (1 - sparsity_factor))
    heuristics_matrix[heuristics_matrix < threshold] = 0
    
    # 6. Symmetrize the matrix
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
