import numpy as np
from typing import Optional, Tuple

def heuristics_v2(distance_matrix: np.ndarray,
                  mst_weight: float = 2.0,
                  neighbor_weight: float = 1.5,
                  density_power: float = 0.5,
                  regret_weight: float = 1.2,
                  sparsity_threshold: Optional[float] = None) -> np.ndarray:
    """
    Advanced TSP edge heuristics with regret-based penalties and MST prioritization.
    
    Args:
        distance_matrix: Square matrix of pairwise distances (can be asymmetric)
        mst_weight: Multiplier for edges in Minimum Spanning Tree
        neighbor_weight: Multiplier for edges to nearest neighbors
        density_power: Power for density-based normalization (0.5 = sqrt)
        regret_weight: Weight for regret-based penalties on suboptimal edges
        sparsity_threshold: Relative threshold for zeroing edges (None = auto)
    
    Returns:
        Heuristic matrix with higher values for promising edges
    """
    n = distance_matrix.shape[0]
    
    # Base heuristic: inverse distance
    epsilon = 1e-12
    base_heuristic = 1.0 / (distance_matrix + epsilon)
    np.fill_diagonal(base_heuristic, 0)
    
    # 1. Minimum Spanning Tree detection
    mst_mask = _compute_mst_mask(distance_matrix)
    
    # 2. Nearest neighbor boosting
    neighbor_mask = _compute_neighbor_mask(distance_matrix, k=max(2, n // 10))
    
    # 3. Local density normalization
    density_factors = _compute_density_factors(distance_matrix, power=density_power)
    
    # 4. Asymmetry correction
    asym_correction = _compute_asymmetry_correction(distance_matrix)
    
    # 5. Regret-based penalties
    regret_factors = _compute_regret_factors(distance_matrix, weight=regret_weight)
    
    # Combine all factors multiplicatively
    heuristic = base_heuristic.copy()
    heuristic[mst_mask] *= mst_weight
    heuristic[neighbor_mask] *= neighbor_weight
    heuristic *= density_factors
    heuristic *= asym_correction
    heuristic *= regret_factors
    
    # Sparsification: keep most promising edges
    if sparsity_threshold is None:
        # Auto-threshold: keep top 2n edges plus MST and high-regret edges
        threshold = np.partition(heuristic.flatten(), -2*n)[-2*n]
    else:
        threshold = sparsity_threshold * np.max(heuristic)
    
    # Always preserve MST edges and high-regret edges
    keep_mask = (heuristic >= threshold) | mst_mask | (regret_factors > 1.0)
    heuristic[~keep_mask] = 0.0
    
    # Normalize to prevent extreme values
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic /= max_val
    
    return heuristic


def _compute_mst_mask(dist: np.ndarray) -> np.ndarray:
    """Compute mask for edges in Minimum Spanning Tree using Prim's algorithm."""
    n = dist.shape[0]
    mask = np.zeros_like(dist, dtype=bool)
    
    if n <= 1:
        return mask
    
    # Use symmetric distances for MST (undirected graph)
    sym_dist = (dist + dist.T) / 2
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    
    for _ in range(n - 1):
        # Find minimum edge from visited to unvisited
        min_val = np.inf
        u = v = -1
        
        visited_idx = np.where(visited)[0]
        unvisited_idx = np.where(~visited)[0]
        
        if len(visited_idx) == 0 or len(unvisited_idx) == 0:
            break
        
        # Efficient vectorized search
        for i in visited_idx:
            row = sym_dist[i, unvisited_idx]
            if len(row) > 0:
                local_min = np.min(row)
                if local_min < min_val:
                    min_val = local_min
                    u = i
                    v = unvisited_idx[np.argmin(row)]
        
        if u >= 0 and v >= 0:
            mask[u, v] = mask[v, u] = True
            visited[v] = True
    
    return mask


def _compute_neighbor_mask(dist: np.ndarray, k: int) -> np.ndarray:
    """Compute mask for edges to k-nearest neighbors."""
    n = dist.shape[0]
    mask = np.zeros_like(dist, dtype=bool)
    k = min(k, n - 1)
    
    if k <= 0:
        return mask
    
    for i in range(n):
        # Get k+1 smallest distances (including self)
        indices = np.argpartition(dist[i], k+1)[:k+1]
        neighbors = indices[indices != i][:k]
        mask[i, neighbors] = True
    
    # Ensure symmetry for undirected interpretation
    return mask | mask.T


def _compute_density_factors(dist: np.ndarray, power: float = 0.5) -> np.ndarray:
    """Compute density-based normalization factors."""
    n = dist.shape[0]
    if n <= 2:
        return np.ones_like(dist)
    
    # Estimate local density using nearest 30% of nodes
    k = max(2, int(0.3 * n))
    densities = np.zeros(n)
    
    for i in range(n):
        # Distances to other nodes
        distances = np.delete(dist[i], i)
        if len(distances) >= k:
            k_smallest = np.partition(distances, k)[:k]
            densities[i] = 1.0 / (np.mean(k_smallest) + 1e-10)
        else:
            densities[i] = 1.0 / (np.mean(distances) + 1e-10)
    
    # Normalize densities
    density_mean = np.mean(densities)
    if density_mean > 0:
        densities /= density_mean
    
    # Create normalization matrix using geometric mean
    factors = np.ones_like(dist)
    for i in range(n):
        for j in range(n):
            if i != j:
                geo_mean = np.sqrt(densities[i] * densities[j])
                factors[i, j] = geo_mean ** (-power)
    
    return factors


def _compute_asymmetry_correction(dist: np.ndarray) -> np.ndarray:
    """Compute correction factor for asymmetric distances."""
    n = dist.shape[0]
    correction = np.ones_like(dist)
    
    # Check for asymmetry
    asym = np.max(np.abs(dist - dist.T))
    if asym < 1e-10:
        return correction
    
    # Prefer edges with small forward+backward average
    avg_dist = (dist + dist.T) / 2.0
    valid = avg_dist > 0
    if np.any(valid):
        min_val = np.min(avg_dist[valid])
        max_val = np.max(avg_dist[valid])
        if max_val > min_val:
            normalized = 0.8 + 0.4 * (max_val - avg_dist) / (max_val - min_val)
            correction = np.where(valid, normalized, 1.0)
    
    return correction


def _compute_regret_factors(dist: np.ndarray, weight: float = 1.2) -> np.ndarray:
    """
    Compute regret-based factors penalizing edges with good alternatives.
    
    For each edge (i,j), computes the 'regret' of choosing this edge versus
    the best alternative at both endpoints. Higher regret reduces the score.
    """
    n = dist.shape[0]
    factors = np.ones_like(dist)
    
    if n <= 2:
        return factors
    
    # Find best and second-best alternatives for each node
    best_dists = np.full(n, np.inf)
    second_best_dists = np.full(n, np.inf)
    best_indices = np.zeros(n, dtype=int)
    
    for i in range(n):
        # Get distances to other nodes
        row = dist[i].copy()
        row[i] = np.inf
        
        # Find two smallest distances
        indices = np.argpartition(row, 2)[:2]
        if len(indices) >= 1:
            best_indices[i] = indices[0]
            best_dists[i] = row[indices[0]]
        if len(indices) >= 2:
            second_best_dists[i] = row[indices[1]]
    
    # Compute regret factors for each edge
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            # Regret at node i: how much worse is (i,j) than best alternative?
            if j == best_indices[i]:
                # This is the best edge for i
                regret_i = second_best_dists[i] - dist[i, j]
            else:
                # Compare to the best edge for i
                regret_i = best_dists[i] - dist[i, j]
            
            # Regret at node j: how much worse is (j,i) than best alternative?
            if i == best_indices[j]:
                regret_j = second_best_dists[j] - dist[j, i]
            else:
                regret_j = best_dists[j] - dist[j, i]
            
            # Combined regret (average, normalized by local scale)
            local_scale = max(best_dists[i], best_dists[j], 1e-10)
            combined_regret = (regret_i + regret_j) / (2 * local_scale)
            
            # Convert regret to penalty factor (higher regret → lower factor)
            # Use sigmoid to map regret to [0.5, 1.0] range
            penalty = 1.0 / (1.0 + np.exp(weight * combined_regret))
            factors[i, j] = 0.5 + 0.5 * penalty  # Map to [0.5, 1.0]
    
    return factors
