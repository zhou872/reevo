import numpy as np
from scipy.spatial.distance import pdist, squareform
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.cluster import AgglomerativeClustering
from scipy.cluster.hierarchy import dendrogram, linkage
import warnings
warnings.filterwarnings('ignore')


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge prioritization incorporating
    hierarchical clustering and multi-hop path analysis.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix of same shape as input, normalized to [0,1]
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 2:
        return np.zeros_like(distance_matrix)
    if n == 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result
    
    # === 1. HIERARCHICAL CLUSTERING ANALYSIS ===
    # Perform agglomerative clustering to identify node groups
    clustering = AgglomerativeClustering(
        n_clusters=None,
        distance_threshold=np.median(distance_matrix) * 1.5,
        linkage='ward',
        metric='precomputed'
    )
    
    try:
        cluster_labels = clustering.fit_predict(distance_matrix)
        unique_clusters = np.unique(cluster_labels)
        
        # Compute cluster-based statistics
        cluster_stats = {}
        for cluster_id in unique_clusters:
            cluster_mask = cluster_labels == cluster_id
            if np.sum(cluster_mask) > 1:
                intra_distances = distance_matrix[cluster_mask][:, cluster_mask]
                cluster_stats[cluster_id] = {
                    'size': np.sum(cluster_mask),
                    'mean_intra_dist': np.mean(intra_distances[intra_distances > 0]),
                    'max_intra_dist': np.max(intra_distances)
                }
            else:
                cluster_stats[cluster_id] = {
                    'size': 1,
                    'mean_intra_dist': 0,
                    'max_intra_dist': 0
                }
        
        # Create cluster affinity matrix
        cluster_affinity = np.ones((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                if cluster_labels[i] == cluster_labels[j]:
                    # Intra-cluster edges get boost based on cluster density
                    cluster_id = cluster_labels[i]
                    if cluster_stats[cluster_id]['size'] > 1:
                        density_factor = 1.0 + np.exp(-cluster_stats[cluster_id]['mean_intra_dist'] / 
                                                    np.median(distance_matrix[distance_matrix > 0]))
                        cluster_affinity[i, j] = cluster_affinity[j, i] = min(2.0, density_factor)
                    else:
                        cluster_affinity[i, j] = cluster_affinity[j, i] = 1.2
                else:
                    # Inter-cluster edges: penalize based on cluster separation
                    c1, c2 = cluster_labels[i], cluster_labels[j]
                    inter_dist = distance_matrix[i, j]
                    intra_scale = (cluster_stats[c1]['mean_intra_dist'] + 
                                 cluster_stats[c2]['mean_intra_dist']) / 2
                    separation = inter_dist / (intra_scale + 1e-12)
                    cluster_affinity[i, j] = cluster_affinity[j, i] = 0.5 + 0.5 * np.exp(-separation * 0.3)
    except:
        # Fallback to uniform clustering if hierarchical fails
        cluster_affinity = np.ones((n, n))
    
    # === 2. ADAPTIVE PARAMETER SELECTION ===
    distances = distance_matrix[~np.eye(n, dtype=bool)]
    if len(distances) > 0:
        # Cluster-aware K selection
        if len(unique_clusters) > 1:
            avg_cluster_size = np.mean([stats['size'] for stats in cluster_stats.values()])
            k = max(3, min(12, int(avg_cluster_size * 0.7)))
        else:
            k = max(3, min(15, n // 5))
        
        # Distance statistics for scaling
        median_dist = np.median(distances)
        q10, q90 = np.percentile(distances, [10, 90])
        dist_range = q90 - q10
    else:
        k = min(8, n - 1)
        median_dist = 1.0
        dist_range = 1.0
    
    # === 3. BASE INVERSE DISTANCE WITH CLUSTER-AWARE SCALING ===
    with np.errstate(divide='ignore', invalid='ignore'):
        eps = dist_range * 0.01 if dist_range > 0 else 1e-12
        scaled_dist = distance_matrix / (median_dist + 1e-12)
        inv_dist = 1.0 / (scaled_dist + eps * np.eye(n))
    
    np.fill_diagonal(inv_dist, 0)
    
    # Row-wise softmax normalization
    row_max = np.max(inv_dist, axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    inv_dist = inv_dist / row_max
    
    # === 4. ENHANCED RANK-BASED RECIPROCAL WEIGHTING ===
    # Compute ranks for each node
    sorted_indices = np.argsort(distance_matrix, axis=1)
    ranks = np.argsort(sorted_indices, axis=1)  # 0 = closest, n-1 = farthest
    
    # Mutual rank weighting with cluster awareness
    rank_weight = np.ones((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            r_i_j = ranks[i, j]
            r_j_i = ranks[j, i]
            
            # Base reciprocal weight
            reciprocal = 2.0 / (1.0 + r_i_j + r_j_i)
            
            # Adjust based on cluster relationship
            if 'cluster_labels' in locals() and cluster_labels[i] == cluster_labels[j]:
                # Intra-cluster: more sensitive to ranks
                rank_weight[i, j] = rank_weight[j, i] = reciprocal * 1.5
            else:
                # Inter-cluster: less sensitive
                rank_weight[i, j] = rank_weight[j, i] = reciprocal * 0.8
    
    np.fill_diagonal(rank_weight, 0)
    
    # === 5. THREE-HOP PATH ANALYSIS ===
    # Precompute 2-hop and 3-hop alternative paths
    knn_indices = np.argsort(distance_matrix, axis=1)[:, 1:k+3]  # Extra neighbors for path analysis
    
    three_hop_penalty = np.ones((n, n))
    
    for i in range(n):
        i_neighbors = knn_indices[i]
        if len(i_neighbors) == 0:
            continue
            
        for j in range(i + 1, n):
            direct_dist = distance_matrix[i, j]
            if direct_dist == 0:
                continue
            
            # Best 2-hop alternative
            best_2hop = np.inf
            for m in i_neighbors[:min(5, len(i_neighbors))]:
                if m == j:
                    continue
                alt_dist = distance_matrix[i, m] + distance_matrix[m, j]
                best_2hop = min(best_2hop, alt_dist)
            
            # Best 3-hop alternative
            best_3hop = np.inf
            for m in i_neighbors[:min(4, len(i_neighbors))]:
                if m == j:
                    continue
                m_neighbors = knn_indices[m]
                for l in m_neighbors[:min(4, len(m_neighbors))]:
                    if l == i or l == j:
                        continue
                    alt_dist = distance_matrix[i, m] + distance_matrix[m, l] + distance_matrix[l, j]
                    best_3hop = min(best_3hop, alt_dist)
            
            # Take the better alternative
            best_alt = min(best_2hop, best_3hop)
            
            if best_alt < np.inf:
                # Normalized penalty based on alternative quality
                alt_ratio = best_alt / direct_dist
                if alt_ratio > 1.0:
                    # Good alternative exists - penalize direct edge
                    penalty = 1.0 / (1.0 + np.log(alt_ratio) * 0.5)
                else:
                    # No good alternative - boost direct edge
                    penalty = 1.0 + (1.0 - alt_ratio) * 0.3
                
                # Adjust based on cluster context
                if 'cluster_labels' in locals():
                    if cluster_labels[i] == cluster_labels[j]:
                        # Intra-cluster edges: more alternatives expected
                        penalty = max(0.7, penalty * 0.9)
                    else:
                        # Inter-cluster edges: fewer alternatives
                        penalty = min(1.3, penalty * 1.1)
                
                three_hop_penalty[i, j] = three_hop_penalty[j, i] = penalty
    
    # === 6. MUTUAL K-NN BOOST ===
    mutual_knn = (ranks < k) & (ranks.T < k)
    knn_boost = np.ones((n, n))
    knn_boost[mutual_knn] = 1.5
    
    # Additional boost for intra-cluster mutual KNN
    if 'cluster_labels' in locals():
        for i in range(n):
            for j in range(i + 1, n):
                if mutual_knn[i, j] and cluster_labels[i] == cluster_labels[j]:
                    avg_rank = (ranks[i, j] + ranks[j, i]) / 2
                    additional_boost = 1.0 + (k - avg_rank) / k * 0.5
                    knn_boost[i, j] = knn_boost[j, i] = knn_boost[i, j] * additional_boost
    
    # === 7. COMBINE ALL FACTORS ===
    heuristic_matrix = (inv_dist * rank_weight * three_hop_penalty * 
                       knn_boost * cluster_affinity)
    np.fill_diagonal(heuristic_matrix, 0)
    
    # Ensure symmetry
    heuristic_matrix = (heuristic_matrix + heuristic_matrix.T) / 2
    
    # === 8. HIERARCHICAL SPARSIFICATION ===
    # Preserve MST edges
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Cluster connectivity preservation
    if 'cluster_labels' in locals():
        # Preserve at least 2 edges per node within cluster
        cluster_preserve = np.zeros((n, n), dtype=bool)
        for cluster_id in unique_clusters:
            cluster_nodes = np.where(cluster_labels == cluster_id)[0]
            if len(cluster_nodes) > 1:
                for i in cluster_nodes:
                    # Find top intra-cluster edges for this node
                    intra_mask = cluster_labels == cluster_id
                    intra_mask[i] = False  # Exclude self
                    intra_values = heuristic_matrix[i, intra_mask]
                    if len(intra_values) > 0:
                        top_intra_idx = np.argsort(intra_values)[-2:]  # Top 2
                        actual_nodes = np.where(intra_mask)[0][top_intra_idx]
                        cluster_preserve[i, actual_nodes] = True
        
        preserve_mask = mst_mask | mutual_knn | cluster_preserve
    else:
        preserve_mask = mst_mask | mutual_knn
    
    # Adaptive sparsification
    target_edges_per_node = max(k, int(0.3 * n))
    
    for i in range(n):
        row_values = heuristic_matrix[i].copy()
        row_values[i] = -np.inf
        
        # Determine which edges to preserve for this node
        to_preserve = preserve_mask[i]
        num_preserved = np.sum(to_preserve)
        
        if np.sum(row_values > 0) > target_edges_per_node:
            # Need to sparsify
            additional_needed = max(0, target_edges_per_node - num_preserved)
            
            if additional_needed > 0:
                # Get scores of non-preserved edges
                non_preserved_scores = row_values.copy()
                non_preserved_scores[to_preserve] = -np.inf
                
                if np.any(non_preserved_scores > -np.inf):
                    # Keep top additional edges
                    threshold = np.partition(non_preserved_scores, -additional_needed)[-additional_needed]
                    keep_mask = (row_values >= threshold) | to_preserve
                    heuristic_matrix[i, ~keep_mask] = 0
        
        # Ensure minimum connectivity
        kept_edges = np.sum(heuristic_matrix[i] > 0)
        if kept_edges < 3:  # Minimum 3 edges per node
            # Add top edges from original scores
            row_copy = heuristic_matrix[i].copy()
            row_copy[i] = -np.inf
            row_copy[heuristic_matrix[i] > 0] = -np.inf
            
            if np.any(row_copy > -np.inf):
                needed = 3 - kept_edges
                top_idx = np.argsort(row_copy)[-needed:]
                heuristic_matrix[i, top_idx] = row_copy[top_idx]
    
    # Ensure symmetry after sparsification
    heuristic_matrix = np.maximum(heuristic_matrix, heuristic_matrix.T)
    np.fill_diagonal(heuristic_matrix, 0)
    
    # === 9. FINAL NORMALIZATION ===
    non_zero = heuristic_matrix > 0
    if np.any(non_zero):
        max_val = np.max(heuristic_matrix[non_zero])
        min_val = np.min(heuristic_matrix[non_zero])
        
        if max_val > min_val:
            # Normalize to [0, 1]
            heuristic_matrix[non_zero] = (heuristic_matrix[non_zero] - min_val) / (max_val - min_val)
        else:
            # All non-zero values are the same
            heuristic_matrix[non_zero] = 0.5
    
    return heuristic_matrix
