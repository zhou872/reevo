import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.linalg import eigh
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge prioritization with meta-learned adaptations.
    
    Key Innovations:
    1. Meta-learned component weighting using spectral graph features
    2. Dynamic k-opt neighborhood analysis (2-opt to 3-opt)
    3. Spectral embedding for capturing global structure
    4. Gradient-based penalty for distance curvature adaptation
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values indicating
        more promising edges for inclusion in optimal tours.
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # === 1. ADVANCED INSTANCE CHARACTERIZATION ===
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    dist_cv = dist_std / (dist_mean + 1e-12)
    n_log = np.log(n)
    
    # Compute spectral features for meta-learning
    spectral_features = compute_spectral_features(distance_matrix, n)
    
    # === 2. META-LEARNED COMPONENT WEIGHTING ===
    weights = meta_learn_weights(spectral_features, dist_cv, n)
    
    # === 3. ROBUST INVERSE DISTANCE WITH GRADIENT PENALTY ===
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Add gradient-based penalty for distance sensitivity
    grad_penalty = compute_gradient_penalty(distance_matrix, dist_std)
    inv_dist *= grad_penalty
    
    # === 4. ENHANCED MST WITH SPECTRAL CONNECTIVITY ===
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Spectral connectivity score
    spectral_conn = compute_spectral_connectivity(distance_matrix, mst_mask)
    
    # === 5. ADAPTIVE REGRET WITH NEIGHBORHOOD ANALYSIS ===
    regret_matrix = compute_adaptive_regret(distance_matrix, n, dist_cv)
    
    # === 6. MULTI-SCALE CENTRALITY ===
    centrality_matrix = compute_multi_scale_centrality(distance_matrix, mst_mask)
    
    # === 7. DYNAMIC K-OPT POTENTIAL ESTIMATION ===
    kopt_potential = compute_kopt_potential(distance_matrix, n, dist_mean)
    
    # === 8. META-COMBINED HEURISTICS ===
    heuristics = np.ones_like(distance_matrix)
    
    # Apply weighted components based on meta-learned weights
    components = {
        'inv_dist': inv_dist ** weights['inv_dist_exp'],
        'regret': regret_matrix ** weights['regret_exp'],
        'centrality': centrality_matrix ** weights['centrality_exp'],
        'kopt': kopt_potential ** weights['kopt_exp'],
        'spectral': spectral_conn ** weights['spectral_exp']
    }
    
    for comp_name, comp_value in components.items():
        heuristics *= comp_value
    
    # Boost MST edges with meta-learned factor
    mst_boost = weights['mst_boost']
    heuristics[mst_mask] *= mst_boost
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    
    # === 9. INTELLIGENT SPARSIFICATION ===
    if n > 10:
        heuristics = intelligent_sparsification(heuristics, mst_mask, n, dist_cv)
    
    # === 10. FINAL NORMALIZATION ===
    np.fill_diagonal(heuristics, 0)
    max_val = heuristics.max()
    if max_val > 0:
        heuristics /= max_val
    
    return heuristics


def compute_spectral_features(distance_matrix: np.ndarray, n: int) -> dict:
    """Compute spectral graph features for meta-learning."""
    # Convert distance to similarity for Laplacian
    with np.errstate(divide='ignore', invalid='ignore'):
        similarity = 1 / (distance_matrix + np.eye(n) * 1e-12)
    similarity[np.isinf(similarity)] = 0
    np.fill_diagonal(similarity, 0)
    
    # Laplacian matrix
    D = np.diag(similarity.sum(axis=1))
    L = D - similarity
    
    # Compute eigenvalues (smallest non-zero)
    try:
        eigvals = eigh(L, eigvals_only=True, subset_by_index=[1, min(5, n-1)])
        spectral_gap = eigvals[1] - eigvals[0] if len(eigvals) > 1 else eigvals[0]
    except:
        spectral_gap = 1.0
    
    # Compute clustering coefficient approximation
    tri_counts = np.zeros(n)
    for i in range(n):
        neighbors = np.where(similarity[i] > np.median(similarity[i]))[0]
        if len(neighbors) >= 2:
            subgraph = similarity[np.ix_(neighbors, neighbors)]
            tri_counts[i] = subgraph.sum() / 2
    
    clustering = np.mean(tri_counts) / max(1, np.max(tri_counts))
    
    return {
        'spectral_gap': spectral_gap,
        'clustering': clustering,
        'n': n
    }


def meta_learn_weights(features: dict, dist_cv: float, n: int) -> dict:
    """Meta-learning of component weights based on instance characteristics."""
    sg = features['spectral_gap']
    clust = features['clustering']
    
    # Adaptive weights based on spectral properties
    inv_dist_exp = 0.7 + 0.3 * np.tanh(sg * 2)  # Higher spectral gap → more distance focus
    regret_exp = 0.8 + 0.4 * (1 - np.exp(-dist_cv))  # Higher CV → stronger regret
    centrality_exp = 0.6 + 0.3 * clust  # Higher clustering → more centrality
    kopt_exp = 1.2 - 0.3 * np.tanh(n / 20)  # Larger n → less k-opt focus
    spectral_exp = 0.5 + 0.3 * np.tanh(sg * 3)  # Higher spectral gap → more spectral focus
    
    # MST boost factor
    mst_boost = 1.5 + 0.5 * np.tanh(dist_cv * 2)
    
    return {
        'inv_dist_exp': inv_dist_exp,
        'regret_exp': regret_exp,
        'centrality_exp': centrality_exp,
        'kopt_exp': kopt_exp,
        'spectral_exp': spectral_exp,
        'mst_boost': mst_boost
    }


def compute_gradient_penalty(distance_matrix: np.ndarray, dist_std: float) -> np.ndarray:
    """Penalize edges based on local distance gradients."""
    n = distance_matrix.shape[0]
    penalty = np.ones((n, n))
    
    for i in range(n):
        # Compute distance gradients to neighbors
        dists = distance_matrix[i]
        sorted_idx = np.argsort(dists)
        
        if len(sorted_idx) > 2:
            # Look at gradient between consecutive distances
            sorted_dists = dists[sorted_idx[1:]]  # Exclude self
            gradients = np.diff(sorted_dists) / (sorted_dists[0] + 1e-12)
            
            # Average gradient in top neighborhood
            avg_gradient = np.mean(gradients[:min(5, len(gradients))])
            gradient_factor = 1.0 + np.tanh(avg_gradient)
            
            # Apply penalty based on distance rank
            for rank, j in enumerate(sorted_idx[1:], 1):
                if i != j:
                    rank_penalty = 1.0 / (1.0 + 0.2 * rank * gradient_factor)
                    penalty[i, j] = rank_penalty
    
    return np.maximum(penalty, penalty.T)


def compute_spectral_connectivity(distance_matrix: np.ndarray, mst_mask: np.ndarray) -> np.ndarray:
    """Compute connectivity scores using spectral embedding."""
    n = distance_matrix.shape[0]
    
    # Create adjacency from MST and nearest neighbors
    adj = np.zeros_like(distance_matrix)
    adj[mst_mask] = 1
    
    # Add k-nearest neighbors (k = min(5, n-1))
    k = min(5, n-1)
    for i in range(n):
        nearest = np.argsort(distance_matrix[i])[1:k+1]
        adj[i, nearest] = 1
    
    adj = np.maximum(adj, adj.T)
    
    # Compute personalized PageRank-like scores
    alpha = 0.85
    W = adj / (adj.sum(axis=1, keepdims=True) + 1e-12)
    centrality = np.ones(n) / n
    
    # Power iteration
    for _ in range(20):
        centrality = alpha * W.T @ centrality + (1 - alpha) * np.ones(n) / n
    
    connectivity = np.outer(centrality, centrality)
    return connectivity


def compute_adaptive_regret(distance_matrix: np.ndarray, n: int, dist_cv: float) -> np.ndarray:
    """Compute regret with neighborhood-aware adaptation."""
    regret = np.ones((n, n))
    
    # Adaptive neighborhood size based on instance
    neighborhood_size = max(3, min(10, int(n * 0.2)))
    
    for i in range(n):
        # Get distances and sort
        dists = distance_matrix[i]
        sorted_indices = np.argsort(dists)
        
        # Consider multiple neighborhoods
        for j in range(n):
            if i == j:
                continue
                
            # Find rank of j among all nodes
            rank = np.where(sorted_indices == j)[0][0]
            
            # Compute regret from different perspectives
            # 1. Direct rank-based regret
            direct_regret = 1.0 / (1.0 + 0.5 * rank)
            
            # 2. Neighborhood regret (compare to local cluster)
            neighborhood = sorted_indices[1:neighborhood_size+1]
            if j in neighborhood:
                local_rank = np.where(neighborhood == j)[0][0]
                local_regret = 1.0 / (1.0 + 0.3 * local_rank)
            else:
                local_regret = 0.7  # Penalty for being outside neighborhood
            
            # 3. Distance ratio regret
            if rank > 0:
                best_dist = dists[sorted_indices[1]]
                ratio = dists[j] / best_dist
                ratio_regret = np.exp(-0.5 * (ratio - 1))
            else:
                ratio_regret = 1.0
            
            # Combine regrets with adaptive weights
            regret[i, j] = (0.4 * direct_regret + 0.4 * local_regret + 0.2 * ratio_regret) ** (1 + dist_cv * 0.3)
    
    return np.sqrt(regret * regret.T)


def compute_multi_scale_centrality(distance_matrix: np.ndarray, mst_mask: np.ndarray) -> np.ndarray:
    """Compute centrality using multiple scales and metrics."""
    n = distance_matrix.shape[0]
    
    # 1. Degree centrality (inverse average distance)
    with np.errstate(divide='ignore', invalid='ignore'):
        degree_cent = 1 / np.mean(distance_matrix + np.eye(n) * 1e-12, axis=1)
    degree_cent = degree_cent / np.max(degree_cent)
    
    # 2. Closeness centrality (harmonic mean)
    with np.errstate(divide='ignore', invalid='ignore'):
        harmonic = n / np.sum(1/(distance_matrix + np.eye(n) * 1e-12), axis=1)
    closeness_cent = harmonic / np.max(harmonic)
    
    # 3. MST-based betweenness
    betweenness = np.zeros(n)
    for i in range(n):
        mst_neighbors = np.where(mst_mask[i])[0]
        betweenness[i] = len(mst_neighbors) / (n - 1)
    
    # Combine with adaptive weights
    centrality = 0.4 * degree_cent + 0.3 * closeness_cent + 0.3 * betweenness
    
    # Create edge centrality matrix
    centrality_matrix = np.outer(centrality, centrality)
    
    # Apply non-linear transform
    centrality_matrix = centrality_matrix ** 1.2
    
    return centrality_matrix


def compute_kopt_potential(distance_matrix: np.ndarray, n: int, dist_mean: float) -> np.ndarray:
    """Estimate potential for k-opt improvements (2-opt to 3-opt)."""
    kopt = np.ones((n, n))
    
    # Adaptive k based on instance size
    k_neighbors = max(3, min(8, int(n * 0.15)))
    
    for i in range(n):
        # Get k nearest neighbors for i
        i_neighbors = np.argsort(distance_matrix[i])[1:k_neighbors+1]
        
        for j in range(i+1, n):
            # Get k nearest neighbors for j
            j_neighbors = np.argsort(distance_matrix[j])[1:k_neighbors+1]
            
            current = distance_matrix[i, j]
            best_gain = 0
            
            # 2-opt potential
            for k in i_neighbors:
                if k == j:
                    continue
                for l in j_neighbors:
                    if l == i:
                        continue
                    # Standard 2-opt gain
                    gain = current + distance_matrix[k, l] - distance_matrix[i, k] - distance_matrix[j, l]
                    best_gain = max(best_gain, gain)
            
            # 3-opt potential (simplified)
            if k_neighbors >= 3:
                # Try a simple 3-opt pattern
                for idx in range(min(3, len(i_neighbors))):
                    k = i_neighbors[idx]
                    if k == j:
                        continue
                    # Find intermediate node
                    candidates = [l for l in j_neighbors if l != i and l != k]
                    if len(candidates) >= 2:
                        l1, l2 = candidates[:2]
                        # Estimate 3-opt gain
                        gain_3opt = (current + distance_matrix[k, l1] + distance_matrix[l1, l2] -
                                   distance_matrix[i, k] - distance_matrix[j, l1] - distance_matrix[l1, l2])
                        best_gain = max(best_gain, gain_3opt * 0.7)  # Discount 3-opt gain
            
            # Convert gain to heuristic factor
            if best_gain > 0:
                gain_norm = min(best_gain / dist_mean, 3.0)
                kopt[i, j] = 1.0 + 0.4 * np.tanh(gain_norm)
            else:
                kopt[i, j] = 1.0
    
    kopt = np.maximum(kopt, kopt.T)
    return kopt


def intelligent_sparsification(heuristics: np.ndarray, mst_mask: np.ndarray, n: int, dist_cv: float) -> np.ndarray:
    """Intelligent edge pruning based on multiple criteria."""
    result = heuristics.copy()
    
    # Separate MST and non-MST edges
    non_mst_mask = ~mst_mask
    np.fill_diagonal(non_mst_mask, False)
    
    if np.any(non_mst_mask):
        non_mst_vals = result[non_mst_mask]
        
        if len(non_mst_vals) > 0:
            # Multi-threshold approach
            q25 = np.percentile(non_mst_vals, 25)
            q75 = np.percentile(non_mst_vals, 75)
            iqr = q75 - q25
            
            # Dynamic threshold based on distribution and instance characteristics
            threshold = q25 - 1.5 * iqr * (1.0 - 0.3 * np.tanh(dist_cv))
            
            # Apply soft thresholding for borderline cases
            borderline = non_mst_mask & (result >= threshold * 0.8) & (result < threshold)
            result[borderline] *= 0.5  # Reduce but don't eliminate
            
            # Eliminate clearly bad edges
            eliminate = non_mst_mask & (result < threshold * 0.8)
            result[eliminate] = 0
    
    return result
