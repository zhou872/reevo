import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform
from scipy.sparse import csgraph

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic with three-hop regret estimation, spectral clustering,
    hierarchical sparsification, and meta-heuristic parameter adaptation.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix normalized to [0,1], same shape as input
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 2:
        return np.zeros_like(distance_matrix)
    if n == 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result
    
    # === 1. META-HEURISTIC PARAMETER ADAPTATION ===
    # Extract instance characteristics
    triu_indices = np.triu_indices(n, 1)
    distances_flat = distance_matrix[triu_indices]
    
    if len(distances_flat) == 0:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # Compute comprehensive statistics
    q5, q25, q50, q75, q95 = np.percentile(distances_flat, [5, 25, 50, 75, 95])
    mean_dist = np.mean(distances_flat)
    std_dist = np.std(distances_flat)
    cv = std_dist / (mean_dist + 1e-12)  # Coefficient of variation
    
    # Adaptive parameters based on instance characteristics
    # KNN size: smaller for uniform, larger for clustered
    base_k = max(4, n // 12)
    k = int(base_k * (1.0 + 0.8 * (1 - min(cv, 2.0) / 2.0)))
    k = min(max(4, k), min(20, n // 2))
    
    # Lookahead depth: deeper for larger instances
    lookahead = min(3, 1 + (n > 30) + (n > 60))
    
    # Spectral clustering parameter
    n_clusters = max(2, min(8, n // 15)) if n >= 20 else 1
    
    # === 2. SPECTRAL CLUSTERING FOR SPATIAL AWARENESS ===
    def spectral_clustering_simple(dist_matrix, k_clusters):
        """Fast approximate spectral clustering using normalized cuts"""
        if k_clusters == 1 or n < 10:
            return np.zeros(n, dtype=int)
        
        # Create similarity matrix
        sigma = np.median(dist_matrix) * 0.3
        W = np.exp(-dist_matrix**2 / (2 * sigma**2))
        np.fill_diagonal(W, 0)
        
        # Degree matrix
        D = np.diag(np.sum(W, axis=1))
        D_inv_sqrt = np.diag(1.0 / np.sqrt(np.diag(D) + 1e-12))
        
        # Normalized Laplacian
        L = np.eye(n) - D_inv_sqrt @ W @ D_inv_sqrt
        
        # Compute k smallest eigenvectors
        try:
            eigvals, eigvecs = np.linalg.eigh(L)
            # Sort eigenvectors by eigenvalues
            idx = np.argsort(eigvals)[:k_clusters]
            V = eigvecs[:, idx]
            
            # Normalize rows
            V_norm = V / np.linalg.norm(V, axis=1, keepdims=True)
            
            # Simple k-means on normalized eigenvectors
            from scipy.cluster.vq import kmeans2
            _, labels = kmeans2(V_norm, k_clusters, minit='++', iter=50)
            return labels
        except:
            # Fallback to distance-based clustering
            threshold = q50 * 1.5
            labels = np.zeros(n, dtype=int)
            current_label = 0
            visited = np.zeros(n, dtype=bool)
            
            for i in range(n):
                if not visited[i]:
                    labels[i] = current_label
                    visited[i] = True
                    for j in range(n):
                        if not visited[j] and distance_matrix[i, j] < threshold:
                            labels[j] = current_label
                            visited[j] = True
                    current_label += 1
            return labels
    
    cluster_labels = spectral_clustering_simple(distance_matrix, n_clusters)
    
    # === 3. THREE-HOP REGRET ESTIMATION ===
    # Precompute KNN indices and distances
    knn_indices = np.argsort(distance_matrix, axis=1)[:, 1:k+1]
    knn_distances = np.take_along_axis(distance_matrix, knn_indices, axis=1)
    
    # Initialize regret scores
    regret_scores = np.ones((n, n))
    
    # Precompute 2-hop distances through common neighbors
    # For efficiency, we'll compute this only for promising edges
    promising_edges = np.zeros((n, n), dtype=bool)
    
    # Consider edges where at least one endpoint has the other in top-k
    ranks = np.argsort(np.argsort(distance_matrix, axis=1), axis=1)
    mutual_promising = (ranks < k) | (ranks.T < k)
    np.fill_diagonal(mutual_promising, False)
    
    # Also consider edges between clusters
    for i in range(n_clusters):
        for j in range(i+1, n_clusters):
            mask_i = cluster_labels == i
            mask_j = cluster_labels == j
            # Connect each cluster to nearest neighbor in other cluster
            if np.any(mask_i) and np.any(mask_j):
                # Find minimal distance between clusters
                min_dist = np.min(distance_matrix[mask_i][:, mask_j])
                threshold = min_dist * 1.5
                for idx in np.where(mask_i)[0]:
                    for jdx in np.where(mask_j)[0]:
                        if distance_matrix[idx, jdx] <= threshold:
                            mutual_promising[idx, jdx] = True
    
    # Compute regret for promising edges
    for i in range(n):
        for j in range(i+1, n):
            if not mutual_promising[i, j]:
                continue
                
            direct = distance_matrix[i, j]
            if direct == 0:
                continue
            
            best_alt = np.inf
            
            # 1-hop alternatives (through common neighbors)
            common = np.intersect1d(knn_indices[i], knn_indices[j])
            if len(common) > 0:
                for m in common[:min(3, len(common))]:
                    alt = distance_matrix[i, m] + distance_matrix[m, j]
                    best_alt = min(best_alt, alt)
            
            # 2-hop alternatives (i -> a -> b -> j)
            if lookahead >= 2:
                # Try through i's nearest neighbors
                for a in knn_indices[i, :min(4, k)]:
                    if a == j:
                        continue
                    # Try direct connection or through j's neighbors
                    alt1 = distance_matrix[i, a] + distance_matrix[a, j]
                    best_alt = min(best_alt, alt1)
                    
                    # Try a -> b -> j where b is j's neighbor
                    for b in knn_indices[j, :min(3, k)]:
                        if b == i or b == a:
                            continue
                        alt2 = distance_matrix[i, a] + distance_matrix[a, b] + distance_matrix[b, j]
                        best_alt = min(best_alt, alt2)
            
            # 3-hop alternatives (i -> a -> b -> c -> j)
            if lookahead >= 3 and n > 30:
                # Limited search for larger instances
                for a in knn_indices[i, :min(3, k)]:
                    if a == j:
                        continue
                    for b in knn_indices[a, :min(3, k)]:
                        if b == i or b == j:
                            continue
                        for c in knn_indices[b, :min(2, k)]:
                            if c == i or c == j or c == a:
                                continue
                            alt3 = (distance_matrix[i, a] + distance_matrix[a, b] + 
                                   distance_matrix[b, c] + distance_matrix[c, j])
                            best_alt = min(best_alt, alt3)
            
            if best_alt < np.inf:
                ratio = best_alt / direct
                # Transform ratio to score: 1.0 for ratio=1.0, decreasing for worse alternatives
                if ratio > 2.0:
                    score = 0.5
                elif ratio > 1.0:
                    score = 1.5 - 0.5 * ratio
                else:
                    # Direct edge is very good (alternative is worse)
                    score = 1.0 + 0.5 * (1.0 / ratio - 1.0)
                    score = min(score, 1.5)
                regret_scores[i, j] = regret_scores[j, i] = score
    
    # === 4. SPATIAL SCORING WITH CLUSTER AWARENESS ===
    # Base inverse distance with adaptive scaling
    eps = q5 * 0.01
    inv_dist = 1.0 / (distance_matrix + eps * np.eye(n))
    np.fill_diagonal(inv_dist, 0)
    
    # Row normalization
    row_sums = np.sum(inv_dist, axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    inv_dist_normalized = inv_dist / row_sums
    
    # Reciprocal rank weighting
    ranks = np.argsort(np.argsort(distance_matrix, axis=1), axis=1)
    rank_penalty = 1.0 / (1.0 + 0.3 * ranks)
    np.fill_diagonal(rank_penalty, 0)
    
    # Mutual neighborhood boost
    mutual_topk = (ranks < k) & (ranks.T < k)
    boost = np.ones((n, n))
    boost[mutual_topk] = 1.8
    
    # Cluster-based scoring
    cluster_score = np.ones((n, n))
    if n_clusters > 1:
        # Intra-cluster edges: boost based on cluster density
        for c in range(n_clusters):
            mask = cluster_labels == c
            if np.sum(mask) >= 3:
                # Compute average intra-cluster distance
                intra_dist = distance_matrix[mask][:, mask]
                avg_intra = np.mean(intra_dist[np.triu_indices(np.sum(mask), 1)])
                
                # Boost short intra-cluster edges
                for i in np.where(mask)[0]:
                    for j in np.where(mask)[0]:
                        if i < j:
                            if distance_matrix[i, j] < avg_intra * 1.2:
                                cluster_score[i, j] = cluster_score[j, i] = 1.4
                            else:
                                cluster_score[i, j] = cluster_score[j, i] = 0.8
        
        # Inter-cluster edges: boost based on cluster connectivity
        cluster_centers = []
        for c in range(n_clusters):
            mask = cluster_labels == c
            if np.any(mask):
                # Use median node as center
                center_idx = np.argsort(np.median(distance_matrix[mask][:, mask], axis=1))[0]
                cluster_centers.append(np.where(mask)[0][center_idx])
        
        # Boost edges connecting clusters
        for i in range(len(cluster_centers)):
            for j in range(i+1, len(cluster_centers)):
                ci, cj = cluster_centers[i], cluster_centers[j]
                # Find good connections between these clusters
                mask_i = cluster_labels == i
                mask_j = cluster_labels == j
                
                # Compute median inter-cluster distance
                inter_dist = distance_matrix[mask_i][:, mask_j]
                if inter_dist.size > 0:
                    median_inter = np.median(inter_dist)
                    # Boost edges that are relatively short between clusters
                    for idx in np.where(mask_i)[0]:
                        for jdx in np.where(mask_j)[0]:
                            if distance_matrix[idx, jdx] < median_inter * 1.3:
                                cluster_score[idx, jdx] = cluster_score[jdx, idx] = 1.3
    
    # === 5. COMBINE SCORES ===
    heuristic = (inv_dist_normalized * rank_penalty * boost * 
                regret_scores * cluster_score)
    np.fill_diagonal(heuristic, 0)
    
    # Ensure symmetry
    heuristic = (heuristic + heuristic.T) / 2
    
    # === 6. HIERARCHICAL EDGE SPARSIFICATION ===
    # Level 1: Always preserve edges
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Level 2: Preserve high-quality edges per cluster
    preserve_mask = mst_mask.copy()
    
    if n_clusters > 1:
        # Preserve top intra-cluster connections
        for c in range(n_clusters):
            mask = cluster_labels == c
            cluster_nodes = np.where(mask)[0]
            if len(cluster_nodes) >= 3:
                # Create submatrix for this cluster
                sub_heuristic = heuristic[mask][:, mask]
                sub_nodes = cluster_nodes
                
                # Keep MST of the cluster
                sub_dist = distance_matrix[mask][:, mask]
                sub_mst = minimum_spanning_tree(sub_dist)
                sub_mst_mask = (sub_mst.toarray() > 0) | (sub_mst.toarray().T > 0)
                
                for idx_i, i in enumerate(cluster_nodes):
                    for idx_j, j in enumerate(cluster_nodes):
                        if idx_i < idx_j and sub_mst_mask[idx_i, idx_j]:
                            preserve_mask[i, j] = preserve_mask[j, i] = True
                
                # Keep additional top edges in cluster
                n_keep = min(5, len(cluster_nodes) // 2)
                for i in cluster_nodes:
                    # Get edges within cluster
                    mask_i = (cluster_labels == c) & (np.arange(n) != i)
                    scores_i = heuristic[i, mask_i]
                    if len(scores_i) > n_keep:
                        threshold = np.partition(scores_i, -n_keep)[-n_keep]
                        strong_edges = np.where(mask_i)[0][scores_i >= threshold]
                        for j in strong_edges:
                            preserve_mask[i, j] = preserve_mask[j, i] = True
    
    # Level 3: Preserve critical inter-cluster connections
    if n_clusters > 1:
        # For each cluster, connect to its nearest neighbor cluster
        for c in range(n_clusters):
            mask_c = cluster_labels == c
            # Find nearest cluster
            min_dist = np.inf
            nearest_cluster = -1
            best_pair = (0, 0)
            
            for other_c in range(n_clusters):
                if other_c == c:
                    continue
                mask_other = cluster_labels == other_c
                # Find minimal distance between clusters
                inter_dist = distance_matrix[mask_c][:, mask_other]
                if inter_dist.size > 0:
                    curr_min = np.min(inter_dist)
                    if curr_min < min_dist:
                        min_dist = curr_min
                        nearest_cluster = other_c
                        # Find the actual edge
                        idx = np.where(mask_c)[0][np.argmin(np.min(inter_dist, axis=1))]
                        jdx = np.where(mask_other)[0][np.argmin(inter_dist[np.argmin(np.min(inter_dist, axis=1)), :])]
                        best_pair = (idx, jdx)
            
            if nearest_cluster != -1:
                i, j = best_pair
                preserve_mask[i, j] = preserve_mask[j, i] = True
                
                # Also preserve a few more connections between these clusters
                mask_other = cluster_labels == nearest_cluster
                n_extra = min(2, np.sum(mask_c), np.sum(mask_other))
                if n_extra > 0:
                    # Find n_extra best edges between these clusters
                    cross_scores = heuristic[mask_c][:, mask_other]
                    if cross_scores.size > 0:
                        flat_scores = cross_scores.flatten()
                        if len(flat_scores) > n_extra:
                            threshold = np.partition(flat_scores, -n_extra)[-n_extra]
                            indices = np.where(cross_scores >= threshold)
                            nodes_c = np.where(mask_c)[0]
                            nodes_other = np.where(mask_other)[0]
                            for idx_i, idx_j in zip(indices[0][:n_extra], indices[1][:n_extra]):
                                i2, j2 = nodes_c[idx_i], nodes_other[idx_j]
                                preserve_mask[i2, j2] = preserve_mask[j2, i2] = True
    
    # Level 4: Adaptive per-node preservation
    for i in range(n):
        # Determine how many edges to keep for this node
        target_edges = max(3, min(k + 2, n // 4))
        
        # Count already preserved edges
        current_preserved = np.sum(preserve_mask[i])
        
        if current_preserved < target_edges:
            # Add more edges based on score
            additional = target_edges - current_preserved
            scores = heuristic[i].copy()
            scores[i] = -np.inf  # Exclude self
            scores[preserve_mask[i]] = -np.inf  # Exclude already preserved
            
            if np.any(scores > -np.inf):
                if np.sum(scores > -np.inf) <= additional:
                    # Keep all remaining
                    new_edges = np.where(scores > -np.inf)[0]
                else:
                    # Keep top additional edges
                    threshold = np.partition(scores, -additional)[-additional]
                    new_edges = np.where(scores >= threshold)[0]
                
                for j in new_edges:
                    preserve_mask[i, j] = preserve_mask[j, i] = True
    
    # Apply sparsification
    heuristic[~preserve_mask] = 0
    
    # === 7. FINAL REFINEMENT ===
    # Ensure connectivity through MST
    if np.sum(heuristic > 0) > 0:
        # Add MST edges if missing
        for i in range(n):
            for j in range(i+1, n):
                if mst_mask[i, j] and heuristic[i, j] == 0:
                    # Give MST edges a moderate score
                    heuristic[i, j] = heuristic[j, i] = 0.5
    
    # Normalize to [0, 1]
    non_zero = heuristic > 0
    if np.any(non_zero):
        max_val = np.max(heuristic[non_zero])
        min_val = np.min(heuristic[non_zero])
        if max_val > min_val:
            heuristic[non_zero] = (heuristic[non_zero] - min_val) / (max_val - min_val)
    
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
