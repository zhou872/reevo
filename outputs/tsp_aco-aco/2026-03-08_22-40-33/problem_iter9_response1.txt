import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization using reciprocal proximity
    weighting, rank-based regret, and adaptive sparsification.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix of same shape as input, normalized to [0,1]
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 2:
        return np.zeros_like(distance_matrix)
    if n == 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result
    
    # === 1. ADAPTIVE PARAMETER TUNING ===
    distances = distance_matrix[~np.eye(n, dtype=bool)]
    if len(distances) > 0:
        # Dynamic K based on instance statistics
        density_ratio = np.median(distances) / (np.mean(distances) + 1e-12)
        min_k = max(2, n // 10)
        max_k = min(15, n // 2)
        k = int(np.clip(min_k + (max_k - min_k) * (1 - np.clip(density_ratio, 0.5, 1.5) / 1.5), 
                       min_k, max_k))
        regret_alpha = 0.15 + 0.1 * min(density_ratio, 1.0)
    else:
        k = min(8, n - 1)
        regret_alpha = 0.2
    
    # === 2. RECIPROCAL PROXIMITY WEIGHTING ===
    # Compute reciprocal rankings for each node
    sorted_indices = np.argsort(distance_matrix, axis=1)
    positions = np.argsort(sorted_indices, axis=1)  # Position matrix (0-indexed)
    
    reciprocal_score = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            pos_i_j = positions[i, j]
            pos_j_i = positions[j, i]
            
            # Reciprocal weighting: boost edges where both nodes rank each other highly
            if pos_i_j < k and pos_j_i < k:
                # Mutual top-k: use harmonic mean of reciprocal ranks
                weight = (1.0/(pos_i_j + 1) + 1.0/(pos_j_i + 1)) / 2
                reciprocal_score[i, j] = reciprocal_score[j, i] = 1.0 + weight
            elif pos_i_j < k or pos_j_i < k:
                # One-way top-k: moderate boost
                min_pos = min(pos_i_j, pos_j_i)
                weight = 1.0/(min_pos + 1)
                reciprocal_score[i, j] = reciprocal_score[j, i] = 1.0 + 0.5 * weight
            else:
                reciprocal_score[i, j] = reciprocal_score[j, i] = 1.0
    
    # === 3. LIGHTWEIGHT RANK-BASED REGRET ===
    regret_penalty = np.ones((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            rank_i_j = positions[i, j]
            rank_j_i = positions[j, i]
            
            # Apply rank-based penalty: 1/(1 + α*rank)
            penalty_i = 1.0 / (1.0 + regret_alpha * rank_i_j)
            penalty_j = 1.0 / (1.0 + regret_alpha * rank_j_i)
            
            # Use geometric mean for symmetric penalty
            regret_penalty[i, j] = regret_penalty[j, i] = np.sqrt(penalty_i * penalty_j)
    
    # === 4. BASE DISTANCE WEIGHTING ===
    with np.errstate(divide='ignore', invalid='ignore'):
        eps = np.percentile(distances, 5) * 0.01 if len(distances) > 0 else 1e-12
        inv_dist = 1.0 / (distance_matrix + eps * np.eye(n))
    
    np.fill_diagonal(inv_dist, 0)
    max_inv = np.max(inv_dist[~np.eye(n, dtype=bool)], initial=0)
    if max_inv > 0:
        inv_dist = inv_dist / max_inv
    
    # === 5. COMBINE FACTORS ===
    heuristics = inv_dist * reciprocal_score * regret_penalty
    
    # === 6. CONNECTIVITY-AWARE ENHANCEMENT ===
    # Identify mutual top-k edges for enhancement
    mutual_top_k = (positions < k) & (positions.T < k)
    heuristics[mutual_top_k] *= 1.8  # Strong boost for mutual top-k
    
    # Preserve Minimum Spanning Tree edges
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    heuristics[mst_mask] *= 1.4
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # === 7. ADAPTIVE SPARSIFICATION ===
    # Compute connectivity scores for each node
    connectivity_scores = np.zeros(n)
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf
        non_zero = row[row > 0]
        if len(non_zero) > 0:
            connectivity_scores[i] = np.mean(non_zero)
        else:
            connectivity_scores[i] = 0
    
    # Normalize connectivity scores
    if np.max(connectivity_scores) > 0:
        connectivity_scores = connectivity_scores / np.max(connectivity_scores)
    
    # Adaptive thresholding per node
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf
        
        # Determine threshold based on connectivity
        if connectivity_scores[i] > 0.3:
            # Well-connected nodes: keep more edges
            percentile = 25 - 15 * connectivity_scores[i]
        else:
            # Poorly-connected nodes: keep most edges
            percentile = 40
        
        # Compute threshold (always keep at least k edges)
        valid_values = row[row > 0]
        if len(valid_values) > max(k, 3):
            threshold = np.percentile(valid_values, percentile)
            # Preserve MST and mutual top-k edges regardless of threshold
            preserve_mask = mst_mask[i] | mutual_top_k[i]
            mask = (row < threshold) & (~preserve_mask) & (row >= 0)
            heuristics[i, mask] = 0
    
    # Ensure symmetry after pruning
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # === 8. FINAL NORMALIZATION ===
    non_zero_mask = heuristics > 0
    if np.any(non_zero_mask):
        max_val = np.max(heuristics[non_zero_mask])
        if max_val > 0:
            heuristics = heuristics / max_val
    
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
