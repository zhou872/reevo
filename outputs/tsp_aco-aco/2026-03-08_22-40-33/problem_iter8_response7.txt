import numpy as np
from scipy.sparse import csr_matrix
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse.linalg import eigsh
from scipy.spatial.distance import pdist, squareform
import warnings
from typing import Dict, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge prioritization with learnable adaptations.
    
    Key Innovations:
    1. Hierarchical spectral clustering for multi-scale feature extraction
    2. Dynamic regret computation with adaptive neighborhood sensitivity
    3. Centrality propagation through MST and k-NN graphs
    4. Learnable sparsification with reinforcement-inspired pruning
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (shape n x n)
        
    Returns:
        Heuristic matrix of same shape as input, with scores in [0, 1]
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    warnings.filterwarnings('ignore')
    
    # === 1. HIERARCHICAL INSTANCE CHARACTERIZATION ===
    upper_tri = distance_matrix[np.triu_indices(n, 1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    dist_cv = dist_std / (dist_mean + 1e-12)
    
    # Compute multi-scale spectral features
    spectral_features = compute_multi_scale_spectral(distance_matrix, n)
    
    # === 2. LEARNABLE COMPONENT FUSION ===
    fusion_weights = adaptive_component_fusion(spectral_features, dist_cv, n)
    
    # === 3. ENHANCED INVERSE DISTANCE WITH CURVATURE ADAPTATION ===
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Apply curvature-adaptive scaling
    curvature_factor = compute_distance_curvature(distance_matrix, n)
    inv_dist = inv_dist ** curvature_factor
    
    # === 4. HIERARCHICAL CLUSTERING-BASED REGIONS ===
    cluster_features = hierarchical_cluster_features(distance_matrix, n)
    
    # === 5. DYNAMIC REGRET WITH NEIGHBORHOOD SENSITIVITY ===
    regret_matrix = compute_dynamic_regret(
        distance_matrix, n, dist_cv, cluster_features['cluster_assignments']
    )
    
    # === 6. CENTRALITY PROPAGATION THROUGH MULTI-GRAPH ===
    centrality_matrix = compute_centrality_propagation(
        distance_matrix, n, spectral_features['spectral_gap']
    )
    
    # === 7. LOCAL IMPROVEMENT POTENTIAL ESTIMATION ===
    improvement_potential = estimate_improvement_potential(
        distance_matrix, n, dist_mean, cluster_features['cluster_hierarchy']
    )
    
    # === 8. ADAPTIVE EDGE SPARSIFICATION LEARNING ===
    sparsification_mask = learn_sparsification_mask(
        distance_matrix, n, dist_cv, spectral_features['clustering_coefficient']
    )
    
    # === 9. FUSION WITH LEARNABLE WEIGHTS ===
    # Compute component scores
    components = {
        'inv_dist': inv_dist ** fusion_weights['inv_dist_exp'],
        'regret': regret_matrix ** fusion_weights['regret_exp'],
        'centrality': centrality_matrix ** fusion_weights['centrality_exp'],
        'improvement': improvement_potential ** fusion_weights['improvement_exp'],
        'cluster': cluster_features['intra_cluster_score'] ** fusion_weights['cluster_exp']
    }
    
    # Weighted geometric mean fusion
    heuristics = np.ones_like(distance_matrix)
    total_weight = 0
    
    for comp_name, comp_value in components.items():
        weight = fusion_weights[f'{comp_name}_weight']
        heuristics *= comp_value ** weight
        total_weight += weight
    
    heuristics = heuristics ** (1 / total_weight) if total_weight > 0 else heuristics
    
    # Apply sparsification mask
    heuristics[~sparsification_mask] *= 0.1
    
    # === 10. FINAL REFINEMENT ===
    # Ensure symmetry and zero diagonal
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # Dynamic normalization
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Apply sigmoid transformation for better contrast
    heuristics = 1 / (1 + np.exp(-6 * (heuristics - 0.5)))
    
    return heuristics


def compute_multi_scale_spectral(distance_matrix: np.ndarray, n: int) -> Dict:
    """Compute spectral features at multiple scales."""
    features = {}
    
    # Convert to similarity matrix
    with np.errstate(divide='ignore', invalid='ignore'):
        similarity = 1 / (distance_matrix + np.eye(n) * 1e-12)
    similarity[np.isinf(similarity)] = 0
    np.fill_diagonal(similarity, 0)
    
    # Scale 1: Full graph spectral properties
    try:
        L = np.diag(similarity.sum(axis=1)) - similarity
        eigvals = eigsh(L, k=min(5, n-1), which='SM', return_eigenvectors=False)
        features['spectral_gap'] = eigvals[1] - eigvals[0] if len(eigvals) > 1 else eigvals[0]
    except:
        features['spectral_gap'] = 1.0
    
    # Scale 2: Local clustering coefficient
    features['clustering_coefficient'] = compute_clustering_coefficient(similarity, n)
    
    # Scale 3: Degree distribution statistics
    degrees = np.sum(similarity > np.mean(similarity), axis=1)
    features['degree_skewness'] = np.abs(np.mean((degrees - np.mean(degrees))**3) / (np.std(degrees)**3 + 1e-12))
    
    return features


def compute_clustering_coefficient(similarity: np.ndarray, n: int) -> float:
    """Compute approximate clustering coefficient."""
    if n < 4:
        return 0.5
    
    # Adaptive threshold based on median
    threshold = np.median(similarity[similarity > 0])
    binary_adj = (similarity > threshold).astype(float)
    
    # Count triangles
    A_cubed = np.linalg.matrix_power(binary_adj, 3)
    triangles = np.trace(A_cubed) / 6
    
    # Count connected triples
    degrees = np.sum(binary_adj, axis=1)
    triples = np.sum(degrees * (degrees - 1)) / 2
    
    return triangles / (triples + 1e-12)


def adaptive_component_fusion(spectral_features: Dict, dist_cv: float, n: int) -> Dict:
    """Learn optimal component weights based on instance features."""
    sg = spectral_features['spectral_gap']
    clust = spectral_features['clustering_coefficient']
    skew = spectral_features['degree_skewness']
    
    # Base weights from spectral properties
    weights = {}
    
    # Distance-based component
    weights['inv_dist_exp'] = 0.8 + 0.4 * np.tanh(sg * 1.5)
    weights['inv_dist_weight'] = 1.2 - 0.3 * np.tanh(skew)
    
    # Regret component
    weights['regret_exp'] = 0.7 + 0.5 * (1 - np.exp(-dist_cv))
    weights['regret_weight'] = 0.8 + 0.4 * clust
    
    # Centrality component
    weights['centrality_exp'] = 0.6 + 0.3 * clust
    weights['centrality_weight'] = 1.0 - 0.4 * np.tanh(sg * 2)
    
    # Improvement potential
    weights['improvement_exp'] = 1.1 - 0.3 * np.tanh(n / 25)
    weights['improvement_weight'] = 0.6 + 0.3 * (1 - clust)
    
    # Cluster component
    weights['cluster_exp'] = 0.5 + 0.3 * clust
    weights['cluster_weight'] = 0.7 + 0.3 * np.tanh(skew * 2)
    
    return weights


def compute_distance_curvature(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute adaptive curvature factor for distance scaling."""
    curvature = np.ones((n, n))
    
    for i in range(n):
        dists = distance_matrix[i]
        sorted_idx = np.argsort(dists)
        
        if len(sorted_idx) > 3:
            # Compute local distance gradient
            sorted_dists = dists[sorted_idx[1:4]]  # Top 3 neighbors
            gradients = np.diff(sorted_dists) / (sorted_dists[0] + 1e-12)
            
            # Average curvature measure
            avg_curvature = np.mean(np.abs(gradients))
            
            # Apply adaptive scaling
            scale_factor = 1.0 + 0.5 * np.tanh(avg_curvature)
            
            for j in range(n):
                if i != j:
                    rank = np.where(sorted_idx == j)[0][0]
                    if rank > 0:
                        # Higher curvature -> more emphasis on nearby nodes
                        curvature[i, j] = scale_factor * (1.0 - 0.1 * min(rank, 5))
    
    return np.maximum(curvature, curvature.T)


def hierarchical_cluster_features(distance_matrix: np.ndarray, n: int) -> Dict:
    """Compute features from hierarchical clustering."""
    features = {}
    
    if n <= 10:
        # For small instances, use simple clustering
        features['cluster_assignments'] = np.zeros(n, dtype=int)
        features['intra_cluster_score'] = np.ones((n, n))
        features['cluster_hierarchy'] = None
        return features
    
    # Perform hierarchical clustering (simplified version)
    # Use distance-based clustering with adaptive thresholds
    threshold = np.percentile(distance_matrix[distance_matrix > 0], 30)
    
    # Assign clusters based on nearest neighbors
    k_clusters = min(5, int(np.sqrt(n)))
    cluster_assignments = -np.ones(n, dtype=int)
    current_cluster = 0
    
    for i in range(n):
        if cluster_assignments[i] == -1:
            # Find nearby unassigned nodes
            nearby = np.where((distance_matrix[i] < threshold) & (cluster_assignments == -1))[0]
            if len(nearby) > 0:
                cluster_assignments[nearby] = current_cluster
                current_cluster += 1
    
    # Handle unassigned nodes
    unassigned = cluster_assignments == -1
    if np.any(unassigned):
        cluster_assignments[unassigned] = np.arange(current_cluster, current_cluster + np.sum(unassigned))
    
    features['cluster_assignments'] = cluster_assignments
    
    # Compute intra-cluster scores
    intra_score = np.ones((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if cluster_assignments[i] == cluster_assignments[j]:
                # Boost edges within same cluster
                intra_score[i, j] = intra_score[j, i] = 1.5
            else:
                # Penalize edges between clusters based on distance
                cluster_dist = np.mean(distance_matrix[
                    cluster_assignments == cluster_assignments[i]
                ][:, cluster_assignments == cluster_assignments[j]])
                penalty = 1.0 / (1.0 + 0.5 * cluster_dist / np.mean(distance_matrix))
                intra_score[i, j] = intra_score[j, i] = penalty
    
    features['intra_cluster_score'] = intra_score
    features['cluster_hierarchy'] = (threshold, k_clusters)
    
    return features


def compute_dynamic_regret(distance_matrix: np.ndarray, n: int, 
                          dist_cv: float, cluster_assignments: np.ndarray) -> np.ndarray:
    """Compute regret with adaptive neighborhood and cluster awareness."""
    regret = np.ones((n, n))
    
    # Adaptive neighborhood size
    base_k = max(3, min(8, int(n * 0.15)))
    
    for i in range(n):
        dists = distance_matrix[i]
        sorted_idx = np.argsort(dists)
        
        # Determine effective neighborhood considering clusters
        same_cluster = cluster_assignments == cluster_assignments[i]
        diff_cluster = ~same_cluster
        diff_cluster[i] = False  # Exclude self
        
        # Different regret calculations for same vs different clusters
        for j in range(n):
            if i == j:
                continue
            
            # Find rank in different contexts
            overall_rank = np.where(sorted_idx == j)[0][0]
            
            if same_cluster[j]:
                # Within same cluster: consider local neighborhood
                cluster_neighbors = np.where(same_cluster)[0]
                cluster_neighbors = cluster_neighbors[cluster_neighbors != i]
                if len(cluster_neighbors) > 0:
                    cluster_dists = dists[cluster_neighbors]
                    cluster_rank = np.sum(cluster_dists < dists[j])
                    local_regret = 1.0 / (1.0 + 0.3 * cluster_rank)
                else:
                    local_regret = 1.0
            else:
                # Between clusters: consider global ranking with penalty
                cross_cluster_neighbors = np.where(diff_cluster)[0]
                if len(cross_cluster_neighbors) > 0:
                    cross_dists = dists[cross_cluster_neighbors]
                    cross_rank = np.sum(cross_dists < dists[j])
                    local_regret = 1.0 / (1.0 + 0.5 * cross_rank)
                else:
                    local_regret = 0.8
            
            # Distance ratio regret
            if overall_rank > 0:
                best_dist = dists[sorted_idx[1]]
                ratio = dists[j] / best_dist
                ratio_regret = np.exp(-0.4 * (ratio - 1))
            else:
                ratio_regret = 1.0
            
            # Combine with cluster-aware weighting
            cluster_weight = 0.6 if same_cluster[j] else 0.4
            regret[i, j] = (cluster_weight * local_regret + 
                           (1 - cluster_weight) * ratio_regret) ** (1 + dist_cv * 0.2)
    
    return np.sqrt(regret * regret.T)


def compute_centrality_propagation(distance_matrix: np.ndarray, n: int, 
                                  spectral_gap: float) -> np.ndarray:
    """Compute centrality scores through multi-graph propagation."""
    # Build multiple graph representations
    graphs = []
    
    # 1. MST graph
    mst = minimum_spanning_tree(distance_matrix)
    mst_adj = (mst.toarray() > 0) | (mst.toarray().T > 0)
    graphs.append(mst_adj.astype(float))
    
    # 2. k-NN graph (adaptive k)
    k_nn = min(5, max(2, int(np.sqrt(n))))
    knn_adj = np.zeros((n, n))
    for i in range(n):
        neighbors = np.argsort(distance_matrix[i])[1:k_nn+1]
        knn_adj[i, neighbors] = 1
    knn_adj = np.maximum(knn_adj, knn_adj.T)
    graphs.append(knn_adj)
    
    # 3. Threshold graph
    threshold = np.percentile(distance_matrix[distance_matrix > 0], 40)
    thresh_adj = (distance_matrix < threshold).astype(float)
    np.fill_diagonal(thresh_adj, 0)
    graphs.append(thresh_adj)
    
    # Combine graphs with spectral gap weighting
    combined_centrality = np.zeros(n)
    
    for adj in graphs:
        # Compute PageRank-like centrality
        W = adj / (adj.sum(axis=1, keepdims=True) + 1e-12)
        centrality = np.ones(n) / n
        
        # Power iteration
        alpha = 0.85
        for _ in range(15):
            centrality = alpha * W.T @ centrality + (1 - alpha) * np.ones(n) / n
        
        combined_centrality += centrality * (1.0 + 0.3 * spectral_gap)
    
    combined_centrality /= len(graphs)
    
    # Create edge centrality matrix with non-linear transformation
    centrality_matrix = np.outer(combined_centrality, combined_centrality)
    centrality_matrix = centrality_matrix ** (1.0 + 0.2 * spectral_gap)
    
    return centrality_matrix


def estimate_improvement_potential(distance_matrix: np.ndarray, n: int,
                                  dist_mean: float, cluster_hierarchy: Tuple) -> np.ndarray:
    """Estimate potential for local improvements (2-opt to 3-opt)."""
    potential = np.ones((n, n))
    
    if n < 4:
        return potential
    
    # Adaptive neighborhood based on clustering
    if cluster_hierarchy and cluster_hierarchy[0]:
        threshold, k_clusters = cluster_hierarchy
        k_neighbors = min(6, max(3, int(n / k_clusters)))
    else:
        k_neighbors = min(6, max(3, int(n * 0.15)))
    
    for i in range(n):
        i_neighbors = np.argsort(distance_matrix[i])[1:k_neighbors+1]
        
        for j in range(i+1, n):
            j_neighbors = np.argsort(distance_matrix[j])[1:k_neighbors+1]
            
            current = distance_matrix[i, j]
            best_gain = 0
            
            # 2-opt gain estimation
            for k in i_neighbors:
                if k == j:
                    continue
                for l in j_neighbors:
                    if l == i or l == k:
                        continue
                    
                    gain = current + distance_matrix[k, l] - distance_matrix[i, k] - distance_matrix[j, l]
                    if gain > best_gain:
                        best_gain = gain
            
            # Simple 3-opt pattern check
            if k_neighbors >= 4:
                # Try star patterns
                for k in i_neighbors[:3]:
                    if k == j:
                        continue
                    for l in j_neighbors[:3]:
                        if l == i or l == k:
                            continue
                        # Find connector between k and l
                        common = set(i_neighbors).intersection(j_neighbors)
                        common.discard(i)
                        common.discard(j)
                        common.discard(k)
                        common.discard(l)
                        
                        if common:
                            m = list(common)[0]
                            gain_3opt = (current + distance_matrix[k, m] + distance_matrix[m, l] -
                                       distance_matrix[i, k] - distance_matrix[j, l] - distance_matrix[k, l])
                            best_gain = max(best_gain, gain_3opt * 0.7)
            
            # Convert gain to potential score
            if best_gain > 0:
                normalized_gain = min(best_gain / (dist_mean + 1e-12), 2.0)
                potential[i, j] = 1.0 + 0.5 * np.tanh(normalized_gain)
    
    potential = np.maximum(potential, potential.T)
    return potential


def learn_sparsification_mask(distance_matrix: np.ndarray, n: int,
                             dist_cv: float, clustering_coef: float) -> np.ndarray:
    """Learn which edges to keep based on multiple criteria."""
    mask = np.ones((n, n), dtype=bool)
    np.fill_diagonal(mask, False)
    
    if n <= 10:
        return mask
    
    # Compute edge scores for sparsification
    edge_scores = np.zeros((n, n))
    
    # 1. Distance-based score
    with np.errstate(divide='ignore', invalid='ignore'):
        dist_score = 1 / (distance_matrix + np.eye(n) * 1e-12)
    dist_score[np.isinf(dist_score)] = 0
    np.fill_diagonal(dist_score, 0)
    
    # 2. Local density score
    density_score = np.zeros((n, n))
    for i in range(n):
        local_dists = distance_matrix[i]
        threshold = np.percentile(local_dists[local_dists > 0], 40)
        density_score[i] = (local_dists < threshold).astype(float)
    density_score = np.maximum(density_score, density_score.T)
    
    # 3. Triangle completion score
    triangle_score = np.zeros((n, n))
    if n >= 5:
        for i in range(n):
            for j in range(i+1, n):
                # Check for common neighbors
                common = set(np.where(distance_matrix[i] < np.inf)[0]).intersection(
                         set(np.where(distance_matrix[j] < np.inf)[0]))
                common.discard(i)
                common.discard(j)
                if common:
                    triangle_score[i, j] = triangle_score[j, i] = min(1.0, len(common) / 5)
    
    # Combine scores
    edge_scores = (0.5 * dist_score + 0.3 * density_score + 0.2 * triangle_score) * clustering_coef
    
    # Adaptive thresholding
    upper_tri_scores = edge_scores[np.triu_indices(n, 1)]
    if len(upper_tri_scores) > 0:
        # Dynamic threshold based on distribution and instance characteristics
        q25 = np.percentile(upper_tri_scores, 25)
        q75 = np.percentile(upper_tri_scores, 75)
        iqr = q75 - q25
        
        threshold = q25 - 0.5 * iqr * (1.0 + dist_cv)
        threshold = max(threshold, np.min(upper_tri_scores[upper_tri_scores > 0]))
        
        # Apply threshold
        mask = edge_scores >= threshold
        np.fill_diagonal(mask, False)
        
        # Ensure minimum connectivity
        min_edges = max(2, int(np.sqrt(n)))
        for i in range(n):
            if np.sum(mask[i]) < min_edges:
                # Add top missing edges
                row_scores = edge_scores[i].copy()
                row_scores[~mask[i]] = 0
                needed = min_edges - np.sum(mask[i])
                if needed > 0 and np.any(row_scores > 0):
                    top_indices = np.argsort(row_scores)[-needed:]
                    mask[i, top_indices] = True
                    mask[top_indices, i] = True
    
    return mask
