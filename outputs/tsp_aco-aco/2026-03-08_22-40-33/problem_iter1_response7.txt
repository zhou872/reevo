import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Minimum spanning tree proximity bonus
    2. Sparsification based on k-nearest neighbors
    3. Harmonic mean of distances to common neighbors
    """
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Avoid division by zero for self-distances
    np.fill_diagonal(distance_matrix, np.inf)
    
    # 1. Minimum Spanning Tree proximity factor
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # 2. K-nearest neighbors for sparsification (adaptive k)
    k = max(3, min(10, n // 4))
    nearest_neighbors = np.argsort(distance_matrix, axis=1)[:, 1:k+1]
    
    # 3. Harmonic mean of distances to common neighbors
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] == 0:
                continue
                
            # Base inverse distance
            inv_dist = 1.0 / distance_matrix[i, j]
            
            # MST bonus factor (edges in MST get higher priority)
            mst_bonus = 2.0 if mst_mask[i, j] else 1.0
            
            # Common neighbors harmonic mean factor
            common_neighbors = set(nearest_neighbors[i]) & set(nearest_neighbors[j])
            if common_neighbors:
                avg_inv_dist = np.mean([1.0/distance_matrix[i, c] + 1.0/distance_matrix[j, c] 
                                       for c in common_neighbors])
                harmonic_factor = 1.0 + avg_inv_dist / (2.0 * inv_dist)
            else:
                harmonic_factor = 1.0
            
            # Combined score
            score = inv_dist * mst_bonus * harmonic_factor
            
            # Apply sparsification: keep only if in top-k for at least one endpoint
            if (j in nearest_neighbors[i]) or (i in nearest_neighbors[j]):
                heuristics_matrix[i, j] = score
                heuristics_matrix[j, i] = score
    
    # Restore diagonal
    np.fill_diagonal(heuristics_matrix, 0)
    np.fill_diagonal(distance_matrix, 0)
    
    return heuristics_matrix
