import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection combining multiple factors.
    Returns a matrix of the same shape indicating edge promise scores.
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # Avoid division by zero and self-loops
    np.fill_diagonal(distance_matrix, np.inf)
    
    # 1. Base inverse distance (like v1)
    inv_dist = 1.0 / distance_matrix
    
    # 2. Nearest-neighbor sparsification
    k = max(2, min(5, n // 4))  # Adaptive sparsity level
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        # Keep only k nearest neighbors per node
        nearest_indices = np.argsort(distance_matrix[i])[:k]
        nn_mask[i, nearest_indices] = True
    
    # Symmetrize the mask (undirected graph)
    nn_mask = nn_mask | nn_mask.T
    
    # 3. Distance regret: measure opportunity cost of NOT taking an edge
    regret_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        sorted_dists = np.sort(distance_matrix[i])
        if len(sorted_dists) > 1:
            second_best = sorted_dists[1]  # Closest non-self distance
            for j in range(n):
                if i != j:
                    # Regret is how much worse alternatives are compared to this edge
                    regret = (distance_matrix[i, j] / second_best) if second_best > 0 else 0
                    regret_matrix[i, j] = np.exp(-regret)
    
    # 4. Local clustering coefficient (approximate via triangular potential)
    cluster_bonus = np.zeros_like(distance_matrix)
    if n >= 3:
        for i in range(n):
            for j in range(i+1, n):
                # Check if i and j share common neighbors
                common_neighbors = 0
                for k in range(n):
                    if k != i and k != j:
                        if (distance_matrix[i, k] < np.inf and 
                            distance_matrix[j, k] < np.inf):
                            common_neighbors += 1
                cluster_bonus[i, j] = cluster_bonus[j, i] = common_neighbors / (n-2)
    
    # 5. Two-hop path efficiency
    path_potential = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Find best intermediate node for a two-hop path i->k->j
                min_two_hop = np.inf
                for k in range(n):
                    if k != i and k != j:
                        two_hop = distance_matrix[i, k] + distance_matrix[k, j]
                        min_two_hop = min(min_two_hop, two_hop)
                
                if min_two_hop < np.inf:
                    # Score based on how much better direct edge is vs best two-hop
                    direct = distance_matrix[i, j]
                    if direct > 0:
                        path_potential[i, j] = min_two_hop / direct
    
    # 6. Combine factors with weighted aggregation
    weights = {
        'inv_dist': 0.4,
        'regret': 0.3,
        'cluster': 0.2,
        'path': 0.1
    }
    
    # Normalize each component
    def safe_normalize(arr):
        if np.max(arr) > 0:
            return arr / np.max(arr)
        return arr
    
    norm_inv = safe_normalize(inv_dist)
    norm_regret = safe_normalize(regret_matrix)
    norm_cluster = safe_normalize(cluster_bonus)
    norm_path = safe_normalize(path_potential)
    
    # Combined score
    combined = (weights['inv_dist'] * norm_inv +
                weights['regret'] * norm_regret +
                weights['cluster'] * norm_cluster +
                weights['path'] * norm_path)
    
    # Apply sparsification mask and zero out unpromising edges
    heuristic_matrix = combined * nn_mask
    
    # Final thresholding: keep only top 30% of edges per row
    threshold_matrix = np.zeros_like(heuristic_matrix)
    for i in range(n):
        row = heuristic_matrix[i]
        if np.max(row) > 0:
            threshold = np.percentile(row[row > 0], 70)
            threshold_matrix[i] = np.where(row >= threshold, row, 0)
    
    # Restore diagonal to zero
    np.fill_diagonal(threshold_matrix, 0)
    
    return threshold_matrix


# Example usage and test
if __name__ == "__main__":
    # Create a sample distance matrix
    np.random.seed(42)
    n_nodes = 8
    coords = np.random.rand(n_nodes, 2) * 100
    dist_matrix = squareform(pdist(coords))
    
    # Calculate heuristics
    heuristics = heuristics_v2(dist_matrix.copy())
    
    print(f"Original distance matrix shape: {dist_matrix.shape}")
    print(f"Heuristics matrix shape: {heuristics.shape}")
    print(f"Sparsity: {(heuristics == 0).sum() / heuristics.size:.1%}")
    print(f"Non-zero entries per row: {(heuristics != 0).sum(axis=1).mean():.1f}")
    
    # Show some example scores
    print("\nSample edge scores (i, j, distance, heuristic):")
    for i, j in [(0, 1), (0, 2), (1, 2), (3, 4)]:
        print(f"({i}, {j}): dist={dist_matrix[i, j]:.2f}, "
              f"heuristic={heuristics[i, j]:.4f}")
