import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic function for TSP edge prioritization incorporating MST connectivity
    and enhanced adaptive regret mechanisms.
    
    Key enhancements over v1:
    1. MST Connectivity Assurance: Ensures all MST edges receive minimum priority scores
       to maintain connectivity in sparse solutions
    2. Enhanced Adaptive Regret: Combines global and local regret with MST-based weighting
    3. Dynamic Mutual K-NN: Adaptive neighbor selection based on MST structure
    4. Efficient Sparsification: Preserves connectivity while reducing edge count
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix of same shape as input, with higher values indicating
        more promising edges. Diagonal is zero.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 2:
        return np.zeros_like(distance_matrix)
    if n == 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result
    
    # === INSTANCE CHARACTERIZATION ===
    # Compute instance statistics for adaptive parameter tuning
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    dist_cv = dist_std / (dist_mean + 1e-12)  # Coefficient of variation
    
    # === COMPUTE MINIMUM SPANNING TREE ===
    # Convert to sparse MST representation
    mst_sparse = minimum_spanning_tree(distance_matrix)
    mst_dense = mst_sparse.toarray()
    mst_mask = (mst_dense + mst_dense.T) > 0  # Symmetric MST indicator
    
    # Compute MST degree for each node
    mst_degree = np.sum(mst_mask, axis=1)
    
    # === ADAPTIVE PARAMETER SETTING ===
    # Adaptive k_neighbors based on instance size, variability, and MST structure
    base_k = max(2, int(np.sqrt(n)))
    
    # Adjust based on coefficient of variation and MST connectivity
    avg_mst_degree = np.mean(mst_degree)
    if dist_cv > 0.7:  # High variability
        k_neighbors = max(2, int(base_k * 0.7))
    elif dist_cv < 0.3:  # Low variability
        k_neighbors = min(n-1, int(base_k * 1.3))
    else:
        k_neighbors = base_k
    
    # Adjust further based on MST connectivity
    if avg_mst_degree > 2.5:  # Dense MST structure
        k_neighbors = min(k_neighbors + 2, n - 1)
    
    k_neighbors = min(k_neighbors, n - 1)
    
    # Adaptive sparsity_factor
    if n < 10:
        sparsity_factor = 0.1
    elif dist_cv > 0.8:
        sparsity_factor = 0.4
    elif avg_mst_degree < 1.8:  # Sparse MST
        sparsity_factor = 0.25  # Keep more edges for connectivity
    else:
        sparsity_factor = 0.3
    
    # === BASE INVERSE DISTANCE WEIGHTING ===
    with np.errstate(divide='ignore', invalid='ignore'):
        eps = np.finfo(distance_matrix.dtype).eps * 10
        inv_dist = 1.0 / (distance_matrix + eps * np.eye(n))
    np.fill_diagonal(inv_dist, 0)
    
    # Normalize inverse distances to [0, 1] range
    max_inv = inv_dist.max()
    if max_inv > 0:
        inv_dist = inv_dist / max_inv
    
    # === ENHANCED MUTUAL K-NN PRIORITIZATION WITH MST AWARENESS ===
    # Get k-nearest neighbors for each node
    nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:k_neighbors+1]
    
    # Create k-NN adjacency matrix
    knn_adjacency = np.zeros((n, n), dtype=bool)
    for i in range(n):
        knn_adjacency[i, nearest_indices[i]] = True
    
    # Mutual K-NN: both nodes are in each other's k-nearest neighbors
    mutual_knn = knn_adjacency & knn_adjacency.T
    # One-way K-NN: only one node is in the other's k-nearest neighbors
    oneway_knn = knn_adjacency ^ mutual_knn
    
    # === ENHANCED REGRET MECHANISM WITH MST-BASED WEIGHTING ===
    regret_matrix = np.ones((n, n))
    
    # For each node, compute regret penalties with local MST adjustment
    for i in range(n):
        mask = np.arange(n) != i
        dists_i = distance_matrix[i, mask]
        other_indices = np.where(mask)[0]
        
        if len(other_indices) == 0:
            continue
            
        # Sort by distance and get ranks
        sorted_idx = np.argsort(dists_i)
        ranks = np.argsort(sorted_idx) + 1  # Rank 1 for nearest, 2 for second nearest, etc.
        
        # Compute base regret penalty
        alpha_base = 0.3  # Base regret strength
        
        # Adjust regret strength based on MST degree
        # Nodes with higher MST degree have more alternatives, so stronger regret
        mst_deg_factor = min(2.0, mst_degree[i] / avg_mst_degree) if avg_mst_degree > 0 else 1.0
        alpha = alpha_base * mst_deg_factor
        
        # Compute penalties with MST-aware adjustment
        penalties = 1.0 / (1.0 + alpha * (ranks - 1))
        
        # Apply penalties to regret matrix
        regret_matrix[i, other_indices] = penalties
    
    # Make symmetric by taking geometric mean
    regret_matrix = np.sqrt(regret_matrix * regret_matrix.T)
    
    # === TWO-HOP EFFICIENCY SCORING ===
    efficiency_score = np.ones((n, n))
    
    # Precompute nearest neighbors for efficiency checking
    eff_k = min(k_neighbors + 2, n - 1)
    eff_nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:eff_k+1]
    
    for i in range(n):
        i_neighbors = eff_nearest_indices[i]
        
        for j in range(i + 1, n):
            direct_dist = distance_matrix[i, j]
            
            if direct_dist <= 0:
                efficiency_score[i, j] = efficiency_score[j, i] = 1.0
                continue
            
            j_neighbors = eff_nearest_indices[j]
            
            # Combine candidate intermediate nodes
            candidates = np.union1d(i_neighbors, j_neighbors)
            candidates = candidates[candidates != i]
            candidates = candidates[candidates != j]
            
            if len(candidates) == 0:
                continue
            
            # Find best two-hop alternative
            best_alternative = np.inf
            for k in candidates:
                alt_dist = distance_matrix[i, k] + distance_matrix[k, j]
                if alt_dist < best_alternative:
                    best_alternative = alt_dist
            
            if best_alternative < np.inf:
                efficiency_ratio = best_alternative / direct_dist
                
                # Convert to score with thresholds
                if efficiency_ratio >= 1.5:
                    score = 1.5
                elif efficiency_ratio >= 1.1:
                    score = 1.2
                elif efficiency_ratio >= 0.9:
                    score = 1.0
                else:
                    score = 0.8
                
                efficiency_score[i, j] = efficiency_score[j, i] = score
    
    # === COMBINE FACTORS WITH MST BOOST ===
    # Dynamic weights based on instance characteristics and MST structure
    if dist_cv > 0.6:
        knn_weight = 1.6
        regret_weight = 1.2
        efficiency_weight = 1.0
    else:
        knn_weight = 1.5
        regret_weight = 1.0
        efficiency_weight = 1.1
    
    # Start with base inverse distance
    heuristics = inv_dist.copy()
    
    # Apply efficiency scoring
    heuristics *= efficiency_score ** efficiency_weight
    
    # Apply regret mechanism
    heuristics *= regret_matrix ** regret_weight
    
    # Boost mutual and one-way K-NN edges
    heuristics[mutual_knn] *= knn_weight
    heuristics[oneway_knn] *= 1.2
    
    # Ensure MST edges have minimum priority to maintain connectivity
    mst_boost = 1.0 + (0.5 * (1.0 - heuristics[mst_mask]))  # Boost low-scoring MST edges
    heuristics[mst_mask] = np.maximum(heuristics[mst_mask], 0.3) * mst_boost
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # === CONNECTIVITY-PRESERVING SPARSIFICATION ===
    if sparsity_factor > 0 and n > 3:
        # For each node, keep top edges while preserving MST edges
        for i in range(n):
            row_values = heuristics[i].copy()
            row_values[i] = -np.inf  # Ignore diagonal
            
            # Always keep MST edges for this node
            mst_edges_i = np.where(mst_mask[i])[0]
            
            # Adaptive keep_count
            min_keep = max(k_neighbors, 3, len(mst_edges_i) + 1)
            max_keep = int((1 - sparsity_factor) * (n - 1))
            keep_count = min(max_keep, max(min_keep, n - 1))
            
            if keep_count < n - 1 and keep_count > len(mst_edges_i):
                # Find threshold for top keep_count edges
                # Make sure MST edges are excluded from threshold calculation
                non_mst_mask = ~mst_mask[i]
                non_mst_values = row_values[non_mst_mask]
                
                if len(non_mst_values) > 0:
                    # We need to keep (keep_count - len(mst_edges_i)) non-MST edges
                    non_mst_keep = max(0, keep_count - len(mst_edges_i))
                    if len(non_mst_values) > non_mst_keep:
                        threshold = np.partition(non_mst_values, -non_mst_keep)[-non_mst_keep]
                        # Zero out non-MST edges below threshold
                        zero_mask = non_mst_mask & (row_values < threshold) & (~mst_mask[i])
                        heuristics[i, zero_mask] = 0
        
        # Restore symmetry while preserving MST connectivity
        symmetric_mask = (heuristics > 0) | (heuristics.T > 0) | mst_mask
        heuristics = np.where(symmetric_mask,
                              np.maximum(heuristics, heuristics.T),
                              0)
    
    # === FINAL PROCESSING ===
    # Ensure zero diagonal
    np.fill_diagonal(heuristics, 0)
    
    # Normalize non-zero values to [0, 1] range
    non_zero_mask = heuristics > 0
    if np.any(non_zero_mask):
        max_val = heuristics[non_zero_mask].max()
        if max_val > 0:
            heuristics[non_zero_mask] = heuristics[non_zero_mask] / max_val
    
    return heuristics
