import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization with refined scoring mechanisms.
    
    Key improvements over v1:
    1. Harmonic centrality instead of arithmetic mean for better outlier resilience
    2. Geometric mean rank penalty for smoother regret modeling
    3. Adaptive MST boosting based on edge concentration in graph
    4. Monotonic value propagation for local neighborhood consistency
    5. Sparse sparsification preserving only the most promising connections
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values indicating
        more promising edges for inclusion in optimal tours.
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # === 1. INSTANCE ANALYSIS ===
    # Use harmonic properties for more robust statistics
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(upper_tri)
    dist_median = np.median(upper_tri)
    dist_ratio = dist_mean / (dist_median + 1e-12)
    
    # Adaptive parameters - simpler and more interpretable
    mst_boost = 1.5 + 0.5 * np.tanh(dist_ratio - 1)  # Boost MST more when mean > median
    regret_strength = 1.4 - 0.4 * np.tanh(dist_ratio)  # Stronger regret when distribution is skewed
    sparsity_target = min(0.4, 2.5/n)  # Target sparsity decreases with n
    
    # === 2. HARMONIC BASE SCORE ===
    # Use harmonic mean of distances for better handling of outliers
    with np.errstate(divide='ignore', invalid='ignore'):
        harm_score = 1 / (distance_matrix + np.eye(n) * 1e-12)
    harm_score[np.isinf(harm_score)] = 0
    np.fill_diagonal(harm_score, 0)
    
    # Normalize harmonic scores
    max_harm = harm_score.max()
    if max_harm > 0:
        harm_score /= max_harm
    
    # === 3. GEOMETRIC MEAN RANK PENALTY ===
    rank_penalty = np.ones((n, n))
    
    for i in range(n):
        # Get distances from node i, excluding self
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        dists = distance_matrix[i, mask]
        sorted_idx = np.argsort(dists)
        
        # Geometric mean penalty: penalty = exp(-regret_strength * log(rank+1))
        # This gives smoother decay than reciprocal penalty
        for rank, global_idx in enumerate(np.where(mask)[0][sorted_idx]):
            penalty = np.exp(-regret_strength * np.log(rank + 2))  # +2 to avoid log(1)=0
            rank_penalty[i, global_idx] = penalty
    
    # Symmetrize using geometric mean (more natural for multiplicative factors)
    rank_penalty = np.sqrt(rank_penalty * rank_penalty.T)
    
    # === 4. HARMONIC CENTRALITY ===
    # Harmonic centrality: sum of reciprocals of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        harm_centrality = np.sum(1/(distance_matrix + np.eye(n) * 1e-12), axis=1)
    harm_centrality = harm_centrality / np.max(harm_centrality)
    
    # Create centrality affinity matrix (edges between central nodes are favored)
    centrality_matrix = np.outer(harm_centrality, harm_centrality)
    
    # === 5. MST WITH CONCENTRATION-BASED BOOSTING ===
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Calculate edge concentration in MST (how clustered are MST edges)
    mst_edges = np.sum(mst_mask) // 2
    if mst_edges > 0:
        # Boost factor increases if MST edges are concentrated among high-centrality nodes
        mst_centrality_sum = np.sum(centrality_matrix[mst_mask])
        total_centrality_sum = np.sum(centrality_matrix)
        concentration = mst_centrality_sum / total_centrality_sum
        mst_boost *= (1 + 0.3 * concentration)
    
    # === 6. MONOTONIC VALUE PROPAGATION ===
    # Ensure local consistency: edges to nearby nodes should have similar scores
    propagation_matrix = np.zeros((n, n))
    
    for i in range(n):
        # Find k nearest neighbors (including self for easier indexing)
        k = min(5, n-1)
        neighbor_indices = np.argsort(distance_matrix[i])[1:k+1]
        
        # Propagate values from nearest neighbors
        for j in range(n):
            if i == j:
                continue
                
            # Average the base scores from i's nearest neighbors to j
            neighbor_scores = []
            for nb in neighbor_indices:
                if nb != j:
                    # Use base score (without penalties) from neighbor to j
                    base_score = 1 / (distance_matrix[nb, j] + 1e-12)
                    neighbor_scores.append(base_score)
            
            if neighbor_scores:
                propagation_matrix[i, j] = np.mean(neighbor_scores)
    
    # Normalize propagation matrix
    max_prop = propagation_matrix.max()
    if max_prop > 0:
        propagation_matrix /= max_prop
    
    # Symmetrize propagation matrix
    propagation_matrix = (propagation_matrix + propagation_matrix.T) / 2
    
    # === 7. COMBINE FACTORS WITH ADAPTIVE WEIGHTS ===
    # Start with harmonic base score
    heuristics = harm_score.copy()
    
    # Apply multiplicative factors with adaptive weights
    heuristics *= np.power(rank_penalty, 0.7)  # Moderate rank penalty
    heuristics *= np.power(centrality_matrix, 0.4)  # Moderate centrality influence
    heuristics *= np.power(propagation_matrix + 0.3, 0.5)  # Propagation with baseline
    
    # Apply MST boost
    heuristics[mst_mask] *= mst_boost
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    
    # === 8. SPARSE SPARSIFICATION ===
    if sparsity_target > 0:
        # Calculate target number of edges to keep
        total_edges = n * (n - 1) // 2
        keep_edges = max(n * 2, int(total_edges * (1 - sparsity_target)))
        
        # Get upper triangular values and indices
        triu_indices = np.triu_indices(n, k=1)
        triu_values = heuristics[triu_indices]
        
        if len(triu_values) > keep_edges:
            # Find threshold to keep top 'keep_edges' values
            threshold = np.partition(triu_values, -keep_edges)[-keep_edges]
            
            # Create mask for values below threshold
            zero_mask = heuristics < threshold
            np.fill_diagonal(zero_mask, False)
            
            # Zero out low-value edges
            heuristics[zero_mask] = 0
    
    # === 9. FINAL PROCESSING ===
    np.fill_diagonal(heuristics, 0)
    
    # Final normalization with logarithmic compression for better dynamic range
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = np.log1p(heuristics * 9) / np.log1p(9)  # Map to [0, 1]
    
    return heuristics
