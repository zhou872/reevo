import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple promising factors.
    
    Features:
    1. Multiplicative scoring with distance ranks
    2. Nearest neighbor chain identification
    3. Degree-based edge penalization
    4. Adaptive sparsification with symmetry enforcement
    """
    n = distance_matrix.shape[0]
    heur_matrix = np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance with quadratic emphasis
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix ** 2 + 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Rank-based scoring per node
    rank_weights = np.zeros_like(distance_matrix)
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])
        # Assign weights based on rank: best edge gets weight 1, second gets 0.5, etc.
        for rank, j in enumerate(sorted_indices[1:], 1):  # Skip self
            rank_weights[i, j] = 1.0 / rank
    
    # 3. Combined multiplicative scoring
    combined_score = inv_dist * rank_weights
    
    # 4. Nearest neighbor chain enhancement
    for start_node in range(min(n, 5)):  # Try multiple starting points
        visited = set()
        current = start_node
        
        while len(visited) < min(n, 10):  # Build short chains
            visited.add(current)
            neighbors = np.argsort(distance_matrix[current])
            next_node = None
            
            for nb in neighbors:
                if nb not in visited and nb != current:
                    next_node = nb
                    # Boost edges in this chain
                    chain_boost = 1.0 + 0.5 / (len(visited) + 1)
                    combined_score[current, next_node] *= chain_boost
                    combined_score[next_node, current] *= chain_boost
                    break
            
            if next_node is None:
                break
            current = next_node
    
    # 5. Degree-aware penalization
    degree_estimate = np.sum(combined_score > 0, axis=1)
    degree_factor = np.zeros_like(combined_score)
    for i in range(n):
        # Penalize nodes that are already "popular"
        penalty = 1.0 - 0.1 * min(degree_estimate[i], 5)
        degree_factor[i, :] = penalty
    
    combined_score *= degree_factor
    
    # 6. Sparsification with adaptive thresholds
    sparsified = np.zeros_like(combined_score)
    
    for i in range(n):
        row = combined_score[i].copy()
        
        # Keep only top connections, number depends on problem size
        k = max(2, min(5, n // 4))
        if n > 10:
            k = max(3, n // 5)
        
        # Adaptive threshold based on distribution
        non_zero = row[row > 0]
        if len(non_zero) > 0:
            threshold = np.percentile(non_zero, 70)
            row[row < threshold] = 0
        
        # Ensure we keep at least k edges
        top_k_indices = np.argsort(row)[-k:]
        mask = np.zeros(n, dtype=bool)
        mask[top_k_indices] = True
        sparsified[i] = row * mask
    
    # 7. Enforce symmetry and final cleanup
    heur_matrix = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(heur_matrix, 0)
    
    # 8. Normalize for consistent scale
    max_val = heur_matrix.max()
    if max_val > 0:
        heur_matrix /= max_val
    
    return heur_matrix
