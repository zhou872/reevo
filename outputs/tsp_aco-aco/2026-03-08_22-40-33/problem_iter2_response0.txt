import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection with dynamic weight adaptation,
    multi-alternative regret, and efficient approximations.
    
    Args:
        distance_matrix: Symmetric n x n matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape indicating edge promise scores
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # Avoid division by zero and self-loops
    np.fill_diagonal(distance_matrix, np.inf)
    
    # 1. Base inverse distance
    inv_dist = 1.0 / distance_matrix
    
    # 2. Adaptive nearest-neighbor sparsification
    k_sparse = max(2, min(10, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nearest_indices = np.argsort(distance_matrix[i])[:k_sparse]
        nn_mask[i, nearest_indices] = True
    nn_mask = nn_mask | nn_mask.T  # Symmetrize
    
    # 3. Multi-alternative regret (consider top m alternatives)
    m_regret = min(5, n - 2)
    regret_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        sorted_dists = np.sort(distance_matrix[i])
        if len(sorted_dists) > m_regret:
            # Average of top m alternatives (excluding the edge itself when calculating)
            for j in range(n):
                if i != j:
                    # Get top m distances excluding j
                    alt_dists = []
                    for dist in sorted_dists[1:]:  # Skip self
                        if len(alt_dists) >= m_regret:
                            break
                        alt_dists.append(dist)
                    if alt_dists:
                        alt_avg = np.mean(alt_dists)
                        regret = distance_matrix[i, j] / alt_avg if alt_avg > 0 else 0
                        regret_matrix[i, j] = np.exp(-regret)
    
    # 4. Efficient clustering coefficient approximation using k-nearest neighbors
    cluster_bonus = np.zeros_like(distance_matrix)
    if n >= 3:
        # Precompute neighbor sets for each node
        neighbor_sets = []
        k_cluster = min(10, n - 1)
        for i in range(n):
            neighbors = set(np.argsort(distance_matrix[i])[1:k_cluster+1])
            neighbor_sets.append(neighbors)
        
        # Compute common neighbors for each edge
        for i in range(n):
            for j in range(i+1, n):
                common = len(neighbor_sets[i] & neighbor_sets[j])
                cluster_bonus[i, j] = cluster_bonus[j, i] = common / k_cluster
    
    # 5. Efficient two-hop path efficiency with limited intermediate nodes
    path_potential = np.zeros_like(distance_matrix)
    if n >= 3:
        m_path = min(10, n - 2)
        for i in range(n):
            # Get m nearest neighbors of i (excluding self)
            neighbors_i = np.argsort(distance_matrix[i])[1:m_path+1]
            for j in range(n):
                if i != j:
                    # Compute two-hop paths through each neighbor of i
                    two_hop_dists = []
                    for k in neighbors_i:
                        if k != j:
                            two_hop = distance_matrix[i, k] + distance_matrix[k, j]
                            two_hop_dists.append(two_hop)
                    if two_hop_dists:
                        min_two_hop = min(two_hop_dists)
                        direct = distance_matrix[i, j]
                        if direct > 0:
                            path_potential[i, j] = min_two_hop / direct
    
    # 6. Dynamic weight optimization based on instance characteristics
    # Compute coefficient of variation of distances
    upper_tri_dists = distance_matrix[np.triu_indices(n, k=1)]
    cv = np.std(upper_tri_dists) / (np.mean(upper_tri_dists) + 1e-12)
    
    # Adjust weights based on CV and problem size
    if cv > 1.0:  # High variance in distances
        weights = np.array([0.5, 0.3, 0.15, 0.05])  # Emphasize inverse distance and regret
    else:  # Low variance
        weights = np.array([0.3, 0.25, 0.3, 0.15])  # More balanced, emphasize clustering
    
    # Normalize for large instances
    if n > 100:
        weights = weights * np.array([1.2, 1.1, 0.8, 0.9])  # Adjust for scalability
        weights = weights / np.sum(weights)  # Renormalize
    
    # 7. Normalize each component
    def safe_normalize(arr):
        max_val = np.max(arr)
        if max_val > 0:
            return arr / max_val
        return arr
    
    norm_inv = safe_normalize(inv_dist)
    norm_regret = safe_normalize(regret_matrix)
    norm_cluster = safe_normalize(cluster_bonus)
    norm_path = safe_normalize(path_potential)
    
    # 8. Combine factors with dynamic weights
    combined = (weights[0] * norm_inv +
                weights[1] * norm_regret +
                weights[2] * norm_cluster +
                weights[3] * norm_path)
    
    # 9. Apply sparsification mask and per-row thresholding
    heuristic_matrix = combined * nn_mask
    
    # Keep top edges per row (adaptive threshold)
    threshold_matrix = np.zeros_like(heuristic_matrix)
    for i in range(n):
        row = heuristic_matrix[i]
        non_zero = row[row > 0]
        if len(non_zero) > 0:
            # Keep top 30% of non-zero edges per row
            threshold = np.percentile(non_zero, 70)
            threshold_matrix[i] = np.where(row >= threshold, row, 0)
    
    # Ensure symmetry and zero diagonal
    threshold_matrix = np.maximum(threshold_matrix, threshold_matrix.T)
    np.fill_diagonal(threshold_matrix, 0)
    
    return threshold_matrix
