import numpy as np
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray, 
                  lookahead_depth: int = 3,
                  sparsity_threshold: float = 0.7) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization using multi-hop path efficiency
    and set-based neighborhood analysis.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        lookahead_depth: Maximum depth for multi-hop lookahead (2-3 recommended)
        sparsity_threshold: Threshold for sparsifying unpromising edges (0-1)
        
    Returns:
        Heuristic matrix of same shape as input, with higher values for more promising edges
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # Initialize heuristic matrix
    heuristics = np.zeros_like(distance_matrix)
    np.fill_diagonal(heuristics, 0)
    
    # Base scoring: inverse distance (closer nodes are generally better)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + np.eye(n) * 1e-12)
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Normalize inverse distances to [0, 1] range
    if np.max(inv_dist) > 0:
        inv_dist = inv_dist / np.max(inv_dist)
    
    # ========== REGRET-BASED PENALIZATION ==========
    # Penalize edges that have many good alternatives
    regret_penalty = np.ones((n, n))
    
    for i in range(n):
        # Get distances to all other nodes
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        candidates = np.where(mask)[0]
        
        if len(candidates) >= 2:
            dists = distance_matrix[i, candidates]
            
            # Sort by distance (ascending)
            sorted_indices = np.argsort(dists)
            sorted_dists = dists[sorted_indices]
            
            # Best distance (minimum)
            best_dist = sorted_dists[0]
            
            # Calculate regret for each alternative
            for rank, idx in enumerate(sorted_indices):
                j = candidates[idx]
                current_dist = dists[idx]
                
                if rank == 0:  # Best edge
                    regret_penalty[i, j] = 1.0
                else:
                    # Combined penalty based on rank and distance ratio
                    rank_penalty = 1.0 / (1.0 + 0.5 * rank)
                    dist_ratio = current_dist / (best_dist + 1e-12)
                    ratio_penalty = 1.0 / (1.0 + 0.3 * (dist_ratio - 1.0))
                    
                    regret_penalty[i, j] = rank_penalty * ratio_penalty
    
    # Symmetrize regret penalty
    regret_penalty = (regret_penalty + regret_penalty.T) / 2
    
    # ========== MULTI-HOP PATH EFFICIENCY ==========
    if n >= 4 and lookahead_depth >= 2:
        # Determine number of neighbors to consider for each node
        k_neighbors = min(max(3, int(np.sqrt(n))), n - 1)
        
        # Get nearest neighbors for each node
        nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:k_neighbors+1]
        
        # Precompute neighbor sets for efficient intersection
        neighbor_sets = [set(nearest_indices[i].tolist()) for i in range(n)]
        
        for i in range(n):
            for j in range(i + 1, n):
                direct_dist = distance_matrix[i, j]
                if direct_dist <= 0:
                    continue
                
                best_alternative = np.inf
                
                # 2-HOP LOOKAHEAD: Find intermediate nodes in intersection of neighborhoods
                common_neighbors = neighbor_sets[i].intersection(neighbor_sets[j])
                
                if common_neighbors:
                    for k in common_neighbors:
                        if k != i and k != j:
                            alt_path = distance_matrix[i, k] + distance_matrix[k, j]
                            best_alternative = min(best_alternative, alt_path)
                
                # 3-HOP LOOKAHEAD (optional)
                if lookahead_depth >= 3 and common_neighbors:
                    for k in common_neighbors:
                        if k == i or k == j:
                            continue
                            
                        # Get neighbors of k that are not i, j, or in the 2-hop set
                        k_neighbors_set = neighbor_sets[k].copy()
                        k_neighbors_set.discard(i)
                        k_neighbors_set.discard(j)
                        k_neighbors_set.discard(k)
                        
                        for l in k_neighbors_set:
                            if l not in common_neighbors:
                                alt_path = (distance_matrix[i, k] + 
                                           distance_matrix[k, l] + 
                                           distance_matrix[l, j])
                                best_alternative = min(best_alternative, alt_path)
                
                # Calculate efficiency score
                if best_alternative < np.inf:
                    efficiency = best_alternative / direct_dist
                    
                    # Sigmoid-like transform: edges with fewer good alternatives get higher scores
                    # When efficiency is high (>1.5), the direct edge is much better than alternatives
                    if efficiency > 1.5:
                        efficiency_score = 1.0
                    elif efficiency < 1.0:
                        efficiency_score = 0.2  # Alternative is better
                    else:
                        # Smooth transition between 1.0 and 1.5
                        efficiency_score = 0.2 + 0.8 * (efficiency - 1.0) / 0.5
                else:
                    # No alternative found, edge is potentially important
                    efficiency_score = 0.8
                
                # Combine factors
                base_score = inv_dist[i, j]
                combined = base_score * regret_penalty[i, j] * efficiency_score
                
                heuristics[i, j] = heuristics[j, i] = combined
    
    # ========== ADAPTIVE SPARSIFICATION ==========
    if n >= 3 and sparsity_threshold > 0:
        # Calculate connectivity threshold for each node
        for i in range(n):
            row = heuristics[i].copy()
            np.fill_diagonal(row, -np.inf)  # Ignore diagonal
            
            if np.any(row > 0):
                # Keep top edges for each node
                sorted_vals = np.sort(row[row > 0])[::-1]
                if len(sorted_vals) > 1:
                    # Determine how many edges to keep
                    keep_count = max(2, int(len(sorted_vals) * sparsity_threshold))
                    threshold = sorted_vals[min(keep_count, len(sorted_vals)-1)]
                    
                    # Zero out edges below threshold
                    mask = (row < threshold) & (row > 0)
                    row[mask] = 0
                
                heuristics[i] = row
        
        # Ensure symmetry
        heuristics = np.maximum(heuristics, heuristics.T)
    
    # ========== FINAL NORMALIZATION ==========
    # Normalize non-zero values to [0, 1] range
    non_zero_mask = heuristics > 0
    if np.any(non_zero_mask):
        max_val = heuristics[non_zero_mask].max()
        if max_val > 0:
            heuristics[non_zero_mask] = heuristics[non_zero_mask] / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
