import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray,
                  lookahead_depth: int = 3,
                  regret_power: float = 1.5,
                  density_neighbors: int = 5) -> np.ndarray:
    """
    Improved heuristic function for TSP edge prioritization.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        lookahead_depth: Maximum depth for multi-hop lookahead (2-4 recommended)
        regret_power: Strength of regret penalty (higher = more penalty for non-optimal edges)
        density_neighbors: Number of neighbors to consider for density calculation
        
    Returns:
        Heuristic matrix of same shape as input
    """
    n = distance_matrix.shape[0]
    if n < 3:
        result = np.ones_like(distance_matrix) - np.eye(n)
        return result
    
    # === 1. BASE INVERSE DISTANCE ===
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + np.eye(n))
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. REGRET-BASED PENALIZATION ===
    regret_penalty = np.ones((n, n))
    
    for i in range(n):
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        candidates = np.where(mask)[0]
        
        if len(candidates) >= 2:
            dists = distance_matrix[i, candidates]
            sorted_idx = np.argsort(dists)
            best_dist = dists[sorted_idx[0]]
            
            for rank, pos in enumerate(sorted_idx):
                j = candidates[pos]
                current_dist = dists[pos]
                
                if rank == 0:
                    regret_penalty[i, j] = 1.0
                else:
                    # Combined penalty: rank-based and distance-ratio based
                    norm_rank = rank / (len(candidates) - 1)
                    dist_ratio = current_dist / (best_dist + 1e-12)
                    
                    # Power-law penalty
                    penalty = 1 / (1 + regret_power * norm_rank * np.sqrt(dist_ratio))
                    regret_penalty[i, j] = penalty
    
    # Symmetrize
    regret_penalty = np.sqrt(regret_penalty * regret_penalty.T)
    
    # === 3. MULTI-HOP PATH EFFICIENCY ===
    path_efficiency = np.ones((n, n))
    
    if n >= 4 and lookahead_depth >= 2:
        # Determine number of neighbors to consider
        neighbor_count = min(max(3, int(np.sqrt(n))), n - 1)
        nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:neighbor_count + 1]
        
        for i in range(n):
            for j in range(i + 1, n):
                best_alternative = np.inf
                direct_dist = distance_matrix[i, j]
                
                # Get candidate intermediate nodes
                candidates_set = set()
                for depth in range(2, min(lookahead_depth + 1, n)):
                    if depth == 2:
                        # 2-hop: neighbors of i and j
                        candidates_set.update(nearest_indices[i].tolist())
                        candidates_set.update(nearest_indices[j].tolist())
                    elif depth == 3:
                        # 3-hop: neighbors of neighbors
                        for k in candidates_set.copy():
                            candidates_set.update(nearest_indices[k].tolist())
                
                # Remove i and j
                candidates_set.discard(i)
                candidates_set.discard(j)
                
                # Evaluate paths
                if candidates_set:
                    for path_length in range(2, min(lookahead_depth + 1, len(candidates_set) + 2)):
                        if path_length == 2:
                            # 2-hop paths
                            for k in candidates_set:
                                if k != i and k != j:
                                    alt_path = distance_matrix[i, k] + distance_matrix[k, j]
                                    best_alternative = min(best_alternative, alt_path)
                        
                        elif path_length == 3:
                            # 3-hop paths
                            for k in candidates_set:
                                if k == i or k == j:
                                    continue
                                for l in candidates_set:
                                    if l == i or l == j or l == k:
                                        continue
                                    alt_path = (distance_matrix[i, k] + 
                                               distance_matrix[k, l] + 
                                               distance_matrix[l, j])
                                    best_alternative = min(best_alternative, alt_path)
                
                # Calculate efficiency score
                if best_alternative < np.inf and direct_dist > 0:
                    efficiency = best_alternative / direct_dist
                    # Sigmoid-like transform: edges with no good alternatives get high score
                    score = 1.0 / (1.0 + np.exp(-2.0 * (efficiency - 1.0)))
                    path_efficiency[i, j] = path_efficiency[j, i] = score
    
    # === 4. CLUSTER DENSITY BONUS ===
    density_bonus = np.ones((n, n))
    
    if n > density_neighbors:
        # Calculate local density for each node
        densities = np.zeros(n)
        for i in range(n):
            # Average distance to k nearest neighbors
            sorted_dists = np.sort(distance_matrix[i])[1:density_neighbors+1]
            densities[i] = 1.0 / (np.mean(sorted_dists) + 1e-12)
        
        # Normalize densities
        if np.max(densities) > 0:
            densities = densities / np.max(densities)
        
        # Apply density bonus to edges within dense regions
        for i in range(n):
            for j in range(i + 1, n):
                # Bonus is geometric mean of node densities
                mean_density = np.sqrt(densities[i] * densities[j])
                # Stronger bonus for edges within dense clusters
                bonus = 1.0 + 0.5 * mean_density
                density_bonus[i, j] = density_bonus[j, i] = bonus
    
    # === 5. COMBINE ALL FACTORS ===
    heuristics = inv_dist.copy()
    heuristics *= regret_penalty
    heuristics *= path_efficiency
    heuristics *= density_bonus
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    
    # === 6. SYMMETRIC SPARSIFICATION ===
    if n > 3:
        # For each node, keep top edges
        for i in range(n):
            row = heuristics[i].copy()
            np.fill_diagonal(heuristics, 0)  # Ensure diagonal is zero
            
            # Determine number of edges to keep
            keep_count = min(max(3, int(np.log2(n)) * 2), n - 1)
            
            # Keep top edges
            if keep_count < n - 1:
                sorted_indices = np.argsort(row)[::-1]  # Descending
                threshold_idx = sorted_indices[keep_count]
                threshold_val = row[threshold_idx]
                
                # Zero out edges below threshold
                zero_mask = row < threshold_val
                zero_mask[i] = False  # Keep diagonal zero
                row[zero_mask] = 0
                
                # Apply symmetrically
                heuristics[i] = row
                heuristics[:, i] = row
    
    # Final symmetrization
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # Normalize to [0, 1] range
    non_zero_mask = heuristics > 0
    if np.any(non_zero_mask):
        max_val = heuristics[non_zero_mask].max()
        if max_val > 0:
            heuristics[non_zero_mask] = heuristics[non_zero_mask] / max_val
    
    return heuristics

if __name__ == "__main__":
    np.random.seed(42)
    n_nodes = 20
    coords = np.random.rand(n_nodes, 2) * 100
    dist_matrix = squareform(pdist(coords))
    
    heuristics = heuristics_v2(dist_matrix.copy(), lookahead_depth=3)
    
    print(f"Original distance matrix shape: {dist_matrix.shape}")
    print(f"Heuristics matrix shape: {heuristics.shape}")
    print(f"Sparsity: {(heuristics == 0).sum() / heuristics.size:.1%}")
    
    # Count edges per node
    non_zero_edges = np.sum(heuristics > 0, axis=1)
    print(f"Non-zero edges per node: min={non_zero_edges.min()}, max={non_zero_edges.max()}, mean={non_zero_edges.mean():.1f}")
    
    # Verify symmetry
    is_symmetric = np.allclose(heuristics, heuristics.T)
    print(f"\nMatrix is symmetric: {is_symmetric}")
    
    # Sample edge scores
    print("\nSample edge scores (i, j, distance, heuristic):")
    for i, j in [(0, 1), (0, 5), (2, 8), (3, 7), (10, 15)]:
        if i < n_nodes and j < n_nodes:
            print(f"({i}, {j}): dist={dist_matrix[i, j]:.2f}, heuristic={heuristics[i, j]:.4f}")
