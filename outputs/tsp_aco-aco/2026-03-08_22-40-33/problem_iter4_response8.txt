import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy import stats
from typing import Dict, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Meta-learning enhanced heuristic function for TSP edge prioritization.
    
    Advanced Features:
    1. Meta-learning parameter adaptation based on instance characteristics
    2. K-opt potential estimation (2, 3-opt) with adaptive depth
    3. Reinforcement learning-inspired sparsification with retention learning
    4. Ensemble centrality with graph neural network-inspired propagation
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        
    Returns:
        Heuristic matrix with higher values indicating more promising edges
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # === 1. META-LEARNING INSTANCE CHARACTERIZATION ===
    instance_profile = _meta_learn_instance(distance_matrix, n)
    
    # === 2. ADAPTIVE PARAMETER TUNING VIA META-RULES ===
    params = _adaptive_parameter_tuning(instance_profile, n)
    
    # === 3. ENSEMBLE CENTRALITY WITH GRAPH PROPAGATION ===
    centrality_matrix = _ensemble_centrality(distance_matrix, n, instance_profile)
    
    # === 4. K-OPT POTENTIAL ESTIMATION WITH ADAPTIVE DEPTH ===
    kopt_potential = _kopt_potential_estimation(distance_matrix, n, params)
    
    # === 5. ENHANCED REGRET WITH NEIGHBORHOOD AWARENESS ===
    regret_matrix = _enhanced_regret(distance_matrix, n, params)
    
    # === 6. BASE ATTRACTION WITH ADAPTIVE TRANSFORMATION ===
    base_attraction = _adaptive_base_attraction(distance_matrix, instance_profile)
    
    # === 7. INTEGRATED HEURISTIC CALCULATION ===
    heuristic = base_attraction.copy()
    heuristic *= regret_matrix
    heuristic *= centrality_matrix
    heuristic *= kopt_potential
    
    # MST enhancement with adaptive boosting
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    heuristic[mst_mask] *= params['mst_boost']
    
    # Ensure symmetry
    heuristic = (heuristic + heuristic.T) / 2
    
    # === 8. REINFORCEMENT-LEARNING INSPIRED SPARSIFICATION ===
    heuristic = _rl_sparsification(heuristic, distance_matrix, mst_mask, params)
    
    # === 9. FINAL REFINEMENT AND NORMALIZATION ===
    np.fill_diagonal(heuristic, 0)
    heuristic = _final_normalization(heuristic)
    
    return heuristic


def _meta_learn_instance(dist_mat: np.ndarray, n: int) -> Dict[str, float]:
    """Extract meta-features for instance-aware adaptation."""
    upper_tri = dist_mat[np.triu_indices(n, k=1)]
    
    # Basic statistics
    stats_dict = {
        'mean': np.mean(upper_tri),
        'std': np.std(upper_tri),
        'cv': np.std(upper_tri) / (np.mean(upper_tri) + 1e-12),
        'skew': stats.skew(upper_tri),
        'kurtosis': stats.kurtosis(upper_tri)
    }
    
    # Geometric properties
    nearest_ratios = []
    for i in range(n):
        dists = dist_mat[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)[:3]
        if len(sorted_dists) >= 2:
            nearest_ratios.append(sorted_dists[1] / sorted_dists[0])
    
    stats_dict['nearest_ratio_mean'] = np.mean(nearest_ratios) if nearest_ratios else 1.0
    stats_dict['nearest_ratio_var'] = np.var(nearest_ratios) if nearest_ratios else 0.0
    
    # Cluster tendency
    mst = minimum_spanning_tree(dist_mat)
    mst_dists = mst.data[mst.data > 0]
    stats_dict['mst_span'] = np.max(mst_dists) / (np.min(mst_dists) + 1e-12)
    
    return stats_dict


def _adaptive_parameter_tuning(profile: Dict[str, float], n: int) -> Dict[str, float]:
    """Meta-learning based parameter adaptation."""
    params = {}
    
    # Adaptive MST boost based on instance geometry
    mst_boost_base = 1.5 + 0.5 * np.tanh(2 * profile['cv'])
    mst_boost_adj = 0.3 * (1 - np.exp(-0.5 * profile['mst_span']))
    params['mst_boost'] = mst_boost_base + mst_boost_adj
    
    # Regret sensitivity based on nearest neighbor consistency
    regret_sensitivity = 1.2 + 0.8 * (1 - np.exp(-2 * profile['nearest_ratio_var']))
    params['regret_sensitivity'] = min(regret_sensitivity, 2.5)
    
    # K-opt depth based on instance size and structure
    if n < 20:
        kopt_depth = 2
    elif n < 50:
        kopt_depth = min(3, 2 + int(profile['cv'] > 0.5))
    else:
        kopt_depth = min(4, 2 + int(profile['cv'] > 0.7) + int(profile['skew'] > 0))
    
    params['kopt_depth'] = kopt_depth
    
    # Sparsification aggression based on multiple factors
    size_factor = 0.5 + 0.4 * np.tanh((15 - n) / 10)
    structure_factor = 0.2 * profile['cv'] + 0.1 * abs(profile['skew'])
    params['sparsity_aggression'] = size_factor + structure_factor
    
    # Centrality fusion weights
    params['centrality_weights'] = {
        'degree': 0.35 + 0.1 * profile['cv'],
        'proximity': 0.30 + 0.1 * (1 - profile['cv']),
        'betweenness': 0.35
    }
    
    return params


def _ensemble_centrality(dist_mat: np.ndarray, n: int, 
                         profile: Dict[str, float]) -> np.ndarray:
    """Enhanced centrality fusion with propagation effects."""
    # Degree centrality (inverse distance weighted)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist_sum = np.sum(1.0 / (dist_mat + np.eye(n) * 1e-12), axis=1)
    degree_centrality = inv_dist_sum / np.max(inv_dist_sum + 1e-12)
    
    # Proximity centrality with iterative refinement
    # Initial estimate using median distances
    pos_estimate = np.median(dist_mat, axis=0)
    
    # Refine through weighted averaging (simulating message passing)
    for _ in range(2):  # 2 propagation steps
        weights = 1.0 / (pos_estimate.reshape(-1, 1) + 1e-12)
        weights = weights / np.sum(weights, axis=1, keepdims=True)
        pos_estimate = np.sum(weights * pos_estimate, axis=1)
    
    proximity_centrality = 1.0 / (1.0 + pos_estimate / (profile['mean'] + 1e-12))
    proximity_centrality = proximity_centrality / np.max(proximity_centrality + 1e-12)
    
    # MST-based betweenness with edge importance
    mst = minimum_spanning_tree(dist_mat)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Calculate betweenness approximation
    betweenness = np.zeros(n)
    for i in range(n):
        if np.sum(mst_mask[i]) > 0:
            # Nodes on high-degree MST vertices get higher betweenness
            neighbor_degrees = np.sum(mst_mask, axis=1)[mst_mask[i]]
            betweenness[i] = np.mean(neighbor_degrees) if len(neighbor_degrees) > 0 else 1
    
    betweenness = betweenness / np.max(betweenness + 1e-12)
    
    # Ensemble fusion with learned weights
    centrality = (0.4 * degree_centrality + 
                  0.35 * proximity_centrality + 
                  0.25 * betweenness)
    
    # Create interaction matrix with non-linear combination
    centrality_matrix = np.outer(centrality, centrality)
    
    # Add similarity bonus for nodes with complementary centrality
    centrality_diff = np.abs(centrality.reshape(-1, 1) - centrality.reshape(1, -1))
    similarity_bonus = np.exp(-2 * centrality_diff)
    centrality_matrix *= (0.7 + 0.3 * similarity_bonus)
    
    return centrality_matrix


def _kopt_potential_estimation(dist_mat: np.ndarray, n: int, 
                              params: Dict[str, float]) -> np.ndarray:
    """Estimate k-opt improvement potential for edges."""
    k = params['kopt_depth']
    potential = np.ones((n, n))
    
    # Pre-compute nearest neighbors
    nn_count = min(6, n - 1)
    nearest_indices = np.argsort(dist_mat, axis=1)[:, 1:nn_count+1]
    nearest_dists = np.take_along_axis(dist_mat, nearest_indices, axis=1)
    
    for i in range(n):
        for j in range(i + 1, n):
            if k == 2:
                # 2-opt potential
                gain = _estimate_2opt_gain(dist_mat, i, j, nearest_indices)
                potential[i, j] = 1.0 + 0.3 * np.tanh(gain)
            else:
                # k-opt potential (k=3 or 4)
                gain = _estimate_kopt_gain(dist_mat, i, j, k, nearest_indices)
                potential[i, j] = 1.0 + 0.4 * np.tanh(gain / k)
    
    potential = np.maximum(potential, potential.T)
    return potential


def _estimate_2opt_gain(dist_mat: np.ndarray, i: int, j: int, 
                       nearest_indices: np.ndarray) -> float:
    """Estimate potential gain from 2-opt move involving edge (i,j)."""
    # Find best alternatives for i and j
    i_alts = [k for k in nearest_indices[i] if k != j]
    j_alts = [k for k in nearest_indices[j] if k != i]
    
    if not i_alts or not j_alts:
        return 0.0
    
    alt_i = min(dist_mat[i, k] for k in i_alts[:3])
    alt_j = min(dist_mat[j, k] for k in j_alts[:3])
    
    current = dist_mat[i, j]
    best_gain = (alt_i + alt_j) - current
    
    return max(best_gain, 0)


def _estimate_kopt_gain(dist_mat: np.ndarray, i: int, j: int, k: int,
                       nearest_indices: np.ndarray) -> float:
    """Estimate potential gain from k-opt moves."""
    if k < 3:
        return _estimate_2opt_gain(dist_mat, i, j, nearest_indices)
    
    # For k>2, consider chain of edges
    max_gain = 0.0
    
    # Try a few promising k-opt configurations
    for _ in range(min(3, k-1)):
        # Start with edge (i,j) and try to find improving k-opt
        current_length = dist_mat[i, j]
        alt_length = 0.0
        
        # Build alternative path through nearest neighbors
        nodes = [i, j]
        for step in range(k-1):
            current = nodes[-1]
            # Find best continuation not already in path
            candidates = [c for c in nearest_indices[current] 
                         if c not in nodes and c != i and c != j]
            if candidates:
                next_node = candidates[0]
                alt_length += dist_mat[current, next_node]
                nodes.append(next_node)
            else:
                break
        
        if len(nodes) >= k:
            # Close the cycle
            alt_length += dist_mat[nodes[-1], i]
            gain = current_length - alt_length
            max_gain = max(max_gain, gain)
    
    return max_gain


def _enhanced_regret(dist_mat: np.ndarray, n: int, 
                    params: Dict[str, float]) -> np.ndarray:
    """Enhanced regret penalization with neighborhood context."""
    regret = np.ones((n, n))
    sensitivity = params['regret_sensitivity']
    
    for i in range(n):
        # Get distances and sort
        dists = dist_mat[i].copy()
        dists[i] = np.inf
        sorted_indices = np.argsort(dists)
        sorted_dists = dists[sorted_indices]
        
        if len(sorted_dists) < 2:
            continue
        
        # Calculate rank-based regret with adaptive decay
        for rank, j in enumerate(sorted_indices):
            if rank == 0:
                regret[i, j] = 1.0  # Best edge
            else:
                # Adaptive penalty based on relative distance gap
                rel_gap = (sorted_dists[rank] - sorted_dists[0]) / (sorted_dists[0] + 1e-12)
                rank_penalty = 1.0 / (1.0 + sensitivity * np.log1p(rank))
                gap_penalty = np.exp(-rel_gap)
                
                regret[i, j] = rank_penalty * (0.7 + 0.3 * gap_penalty)
    
    # Symmetrize using harmonic mean (more conservative than geometric)
    regret = 2.0 / (1.0/(regret + 1e-12) + 1.0/(regret.T + 1e-12))
    
    return regret


def _adaptive_base_attraction(dist_mat: np.ndarray, 
                             profile: Dict[str, float]) -> np.ndarray:
    """Base attraction with instance-adaptive transformation."""
    n = dist_mat.shape[0]
    
    # Choose transformation based on instance characteristics
    if profile['cv'] > 0.8:  # High variability
        # Use softer inverse for highly variable instances
        scale = profile['mean']
        attraction = scale / (dist_mat + 0.2 * scale)
    elif profile['skew'] > 1.0:  # Right-skewed
        # Use exponential decay for skewed distributions
        normalized = dist_mat / profile['mean']
        attraction = np.exp(-0.5 * normalized)
    else:
        # Standard inverse with adaptive scaling
        scale = np.percentile(dist_mat[np.triu_indices(n, k=1)], 25)
        attraction = scale / (dist_mat + 0.1 * scale)
    
    np.fill_diagonal(attraction, 0)
    return attraction


def _rl_sparsification(heuristic: np.ndarray, dist_mat: np.ndarray,
                      mst_mask: np.ndarray, params: Dict[str, float]) -> np.ndarray:
    """Reinforcement learning inspired adaptive sparsification."""
    n = heuristic.shape[0]
    result = heuristic.copy()
    aggression = params['sparsity_aggression']
    
    # Always preserve MST edges
    non_mst_mask = ~mst_mask
    np.fill_diagonal(non_mst_mask, False)
    
    if not np.any(non_mst_mask) or aggression <= 0:
        return result
    
    # Extract non-MST values and their distribution
    non_mst_values = result[non_mst_mask]
    
    if len(non_mst_values) == 0:
        return result
    
    # Adaptive threshold selection based on value distribution
    if aggression < 0.3:
        # Conservative: keep more edges
        keep_ratio = 0.7 + 0.3 * (1 - aggression)
    elif aggression < 0.7:
        # Moderate: balance
        keep_ratio = 0.5 + 0.2 * (0.7 - aggression)
    else:
        # Aggressive: keep fewer edges
        keep_ratio = 0.3 + 0.2 * (1 - aggression)
    
    # Ensure we keep at least some edges
    min_edges = max(2 * n, int(0.1 * n * (n - 1) / 2))
    keep_ratio = max(keep_ratio, min_edges / len(non_mst_values))
    
    # Calculate threshold
    threshold = np.percentile(non_mst_values, 100 * (1 - keep_ratio))
    
    # Apply threshold with edge quality consideration
    to_zero = non_mst_mask & (result < threshold)
    
    # But preserve high-quality edges even if below threshold
    # (simulating exploration in RL)
    for i in range(n):
        row_values = result[i]
        row_mask = to_zero[i]
        
        # Keep at least one good alternative per node if possible
        if np.sum(~row_mask & ~mst_mask[i] & (np.arange(n) != i)) == 0:
            # If we would remove all non-MST edges, keep the best one
            candidates = np.where(~mst_mask[i] & (np.arange(n) != i))[0]
            if len(candidates) > 0:
                best = candidates[np.argmax(row_values[candidates])]
                to_zero[i, best] = False
                to_zero[best, i] = False
    
    result[to_zero] = 0
    return result


def _final_normalization(heuristic: np.ndarray) -> np.ndarray:
    """Final normalization and refinement."""
    n = heuristic.shape[0]
    
    # Remove negative values
    heuristic[heuristic < 0] = 0
    
    # Normalize non-zero values to [0, 1] range
    non_zero_mask = heuristic > 0
    if np.any(non_zero_mask):
        h_min = heuristic[non_zero_mask].min()
        h_max = heuristic[non_zero_mask].max()
        
        if h_max > h_min:
            # Sigmoid-like normalization to emphasize differences
            normalized = (heuristic[non_zero_mask] - h_min) / (h_max - h_min)
            # Apply mild S-curve transformation
            heuristic[non_zero_mask] = 0.5 + 0.5 * np.tanh(2 * (normalized - 0.5))
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
