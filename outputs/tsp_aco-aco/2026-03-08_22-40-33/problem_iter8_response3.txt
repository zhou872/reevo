```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.cluster.hierarchy import linkage, fcluster
from typing import Dict, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with multi-head spatial attention,
    dynamic regret optimization, and geometric mean ensemble.
    
    Core innovations:
    1. Multi-head spatial attention capturing different geometric relationships
    2. Dynamic regret computation with adaptive lookahead horizon
    3. Geometric mean ensemble with learned component importance
    4. Hierarchical sparsification with cluster-aware pruning
    
    Args:
        distance_matrix: Square symmetric matrix of pairwise distances
        
    Returns:
        Heuristic matrix with values in [0, 1] indicating edge promise
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        heuristic = np.zeros_like(distance_matrix)
        if n == 2:
            heuristic[0, 1] = heuristic[1, 0] = 1.0
        return heuristic
    
    # ==================== 1. INSTANCE CHARACTERIZATION ====================
    profile = _characterize_instance(distance_matrix, n)
    
    # ==================== 2. MULTI-HEAD SPATIAL ATTENTION ====================
    attention_scores = _multi_head_spatial_attention(distance_matrix, n, profile)
    
    # ==================== 3. DYNAMIC REGRET COMPUTATION ====================
    regret_scores = _dynamic_regret_optimization(distance_matrix, n, profile)
    
    # ==================== 4. STRUCTURAL COMPONENTS ====================
    mst_scores = _enhanced_mst_scoring(distance_matrix, n)
    density_scores = _adaptive_density_scoring(distance_matrix, n, profile)
    centrality_scores = _multi_scale_centrality(distance_matrix, n)
    
    # ==================== 5. GEOMETRIC MEAN ENSEMBLE ====================
    weights = _learn_component_weights(profile, n)
    components = {
        'attention': attention_scores,
        'regret': regret_scores,
        'mst': mst_scores,
        'density': density_scores,
        'centrality': centrality_scores
    }
    
    heuristic = _geometric_mean_ensemble(components, weights, n)
    
    # ==================== 6. HIERARCHICAL SPARSIFICATION ====================
    heuristic = _cluster_aware_sparsification(heuristic, distance_matrix, n, profile)
    
    # ==================== 7. FINAL REFINEMENT ====================
    heuristic = _contrast_enhancement(heuristic, distance_matrix, n)
    
    np.fill_diagonal(heuristic, 0)
    return heuristic


def _characterize_instance(distance_matrix: np.ndarray, n: int) -> Dict[str, float]:
    """Comprehensive instance profiling for adaptive parameter tuning."""
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    # Statistical moments
    stats = {
        'mean': np.mean(distances),
        'std': np.std(distances),
        'skewness': _calculate_skewness(distances),
        'kurtosis': _calculate_kurtosis(distances)
    }
    
    # Spatial distribution metrics
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = mst.data[mst.data > 0]
    stats['mst_span'] = np.max(mst_edges) / (np.min(mst_edges) + 1e-10) if len(mst_edges) > 0 else 1.0
    stats['mst_entropy'] = _calculate_entropy(mst_edges)
    
    # Nearest neighbor analysis
    nn_stats = _nearest_neighbor_analysis(distance_matrix, n)
    stats.update(nn_stats)
    
    # Distance quartiles
    stats['q1'], stats['q2'], stats['q3'] = np.percentile(distances, [25, 50, 75])
    
    # Edge concentration
    stats['concentration'] = _calculate_edge_concentration(distance_matrix, n)
    
    return stats


def _calculate_skewness(data: np.ndarray) -> float:
    """Calculate Fisher-Pearson coefficient of skewness."""
    if len(data) < 3:
        return 0.0
    mean = np.mean(data)
    std = np.std(data)
    if std == 0:
        return 0.0
    return np.mean(((data - mean) / std) ** 3)


def _calculate_kurtosis(data: np.ndarray) -> float:
    """Calculate excess kurtosis."""
    if len(data) < 4:
        return 0.0
    mean = np.mean(data)
    std = np.std(data)
    if std == 0:
        return 0.0
    return np.mean(((data - mean) / std) ** 4) - 3


def _calculate_entropy(data: np.ndarray) -> float:
    """Calculate normalized entropy of distribution."""
    if len(data) == 0:
        return 0.0
    hist, _ = np.histogram(data, bins=min(10, len(data)))
    probs = hist / hist.sum()
    probs = probs[probs > 0]
    entropy = -np.sum(probs * np.log(probs))
    return entropy / np.log(len(probs) + 1e-10)


def _nearest_neighbor_analysis(distance_matrix: np.ndarray, n: int) -> Dict[str, float]:
    """Analyze nearest neighbor relationships."""
    k = min(5, n - 1)
    nn_ratios = []
    nn_consistencies = []
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_indices = np.argsort(dists)[:k]
        sorted_dists = dists[sorted_indices]
        
        # Ratios between consecutive nearest neighbors
        if len(sorted_dists) >= 2:
            for j in range(len(sorted_dists) - 1):
                nn_ratios.append(sorted_dists[j + 1] / sorted_dists[j])
        
        # Consistency check: reciprocal nearest neighbors
        for j in range(min(3, k)):
            neighbor = sorted_indices[j]
            if neighbor < n:
                neighbor_nn = np.argsort(distance_matrix[neighbor])[1]
                nn_consistencies.append(1.0 if neighbor_nn == i else 0.0)
    
    return {
        'nn_ratio_mean': np.mean(nn_ratios) if nn_ratios else 1.0,
        'nn_ratio_std': np.std(nn_ratios) if nn_ratios else 0.0,
        'nn_consistency': np.mean(nn_consistencies) if nn_consistencies else 0.5
    }


def _calculate_edge_concentration(distance_matrix: np.ndarray, n: int) -> float:
    """Measure how concentrated edges are around certain distances."""
    if n < 4:
        return 0.5
    
    # Cluster distances into groups
    triu_distances = distance_matrix[np.triu_indices(n, k=1)]
    if len(triu_distances) < 4:
        return 0.5
    
    try:
        Z = linkage(triu_distances.reshape(-1, 1), method='ward')
        clusters = fcluster(Z, t=2, criterion='maxclust')
        
        # Calculate concentration as max cluster proportion
        unique, counts = np.unique(clusters, return_counts=True)
        concentration = np.max(counts) / len(clusters)
        return min(concentration, 0.95)
    except:
        return 0.5


def _multi_head_spatial_attention(distance_matrix: np.ndarray, n: int, 
                                  profile: Dict[str, float]) -> np.ndarray:
    """Multi-head attention capturing different geometric relationships."""
    attention_scores = np.ones((n, n))
    
    # Head 1: Local proximity attention
    local_scores = _local_proximity_attention(distance_matrix, n)
    
    # Head 2: Regional structure attention
    regional_scores = _regional_structure_attention(distance_matrix, n, profile)
    
    # Head 3: Global pattern attention
    global_scores = _global_pattern_attention(distance_matrix, n)
    
    # Adaptive combination based on instance characteristics
    if profile['std'] / (profile['mean'] + 1e-10) > 0.5:
        # High variance: emphasize regional and global patterns
        weights = [0.2, 0.5, 0.3]
    else:
        # Low variance: emphasize local patterns
        weights = [0.5, 0.3, 0.2]
    
    for i in range(n):
        for j in range(n):
            if i != j:
                attention_scores[i, j] = (
                    weights[0] * local_scores[i, j] +
                    weights[1] * regional_scores[i, j] +
                    weights[2] * global_scores[i, j]
                )
    
    # Normalize
    max_score = attention_scores.max()
    if max_score > 0:
        attention_scores /= max_score
    
    return attention_scores


def _local_proximity_attention(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Attention based on immediate neighborhood relationships."""
    scores = np.ones((n, n))
    k = max(3, n // 10)
    
    for i in range(n):
        # Get k-nearest neighbors
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_indices = np.argsort(dists)[:k]
        
        for j in range(n):
            if i == j:
                continue
            
            if j in nn_indices:
                # Direct neighbor: high score
                rank = np.where(nn_indices == j)[0][0] + 1
                scores[i, j] = 1.0 / (rank ** 0.5)
            else:
                # Check if there's a strong indirect connection
                common_neighbors = set(nn_indices) & set(np.argsort(distance_matrix[j])[:k])
                if common_neighbors:
                    # Strong indirect connection
                    scores[i, j] = 0.7
                else:
                    # Weak connection
                    scores[i, j] = 0.3
    
    return scores


def _regional_structure_attention(distance_matrix: np.ndarray, n: int,
                                  profile: Dict[str, float]) -> np.ndarray:
    """Attention based on regional cluster structure."""
    scores = np.ones((n, n))
    
    if n < 6:
        return scores
    
    # Adaptive clustering based on instance characteristics
    if profile['concentration'] > 0.7:
        n_clusters = max(2, min(5, n // 4))
    else:
        n_clusters = max(2, min(8, n // 3))
    
    try:
        # Perform hierarchical clustering
        Z = linkage(distance_matrix, method='average')
        cluster_labels = fcluster(Z, t=n_clusters, criterion='maxclust')
        
        # Intra-cluster and inter-cluster scoring
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                if cluster_labels[i] == cluster_labels[j]:
                    # Same cluster: boost based on cluster compactness
                    cluster_mask = cluster_labels == cluster_labels[i]
                    cluster_dists = distance_matrix[cluster_mask][:, cluster_mask]
                    if cluster_dists.size > 0:
                        compactness = np.mean(cluster_dists) / (profile['mean'] + 1e-10)
                        scores[i, j] = 1.0 + 0.5 * (1.0 - min(compactness, 1.0))
                else:
                    # Different clusters: score based on cluster proximity
                    cluster_i = cluster_labels[i]
                    cluster_j = cluster_labels[j]
                    
                    # Find minimum distance between clusters
                    mask_i = cluster_labels == cluster_i
                    mask_j = cluster_labels == cluster_j
                    
                    inter_dists = distance_matrix[mask_i][:, mask_j]
                    if inter_dists.size > 0:
                        min_inter_dist = np.min(inter_dists)
                        normalized_dist = min_inter_dist / (profile['mean'] + 1e-10)
                        scores[i, j] = 1.0 / (1.0 + normalized_dist)
    except:
        pass
    
    return scores


def _global_pattern_attention(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Attention based on global geometric patterns."""
    scores = np.ones((n, n))
    
    # Use MST to identify backbone structure
    mst = minimum_spanning_tree(distance_matrix)
    mst_adj = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Compute all-pairs shortest paths on MST
    mst_dist = mst.toarray()
    mst_dist[mst_dist == 0] = np.inf
    np.fill_diagonal(mst_dist, 0)
    
    # Floyd-Warshall on MST (simplified since MST is tree)
    for k in range(n):
        for i in range(n):
            for j in range(n):
                if mst_dist[i, k] + mst_dist[k, j] < mst_dist[i, j]:
                    mst_dist[i, j] = mst_dist[i, k] + mst_dist[k, j]
    
    # Score based on MST distance
    max_mst_dist = mst_dist[mst_dist < np.inf].max()
    if max_mst_dist > 0:
        for i in range(n):
            for j in range(n):
                if i != j and mst_dist[i, j] < np.inf:
                    # Closer in MST structure gets higher score
                    scores[i, j] = 1.0 - (mst_dist[i, j] / max_mst_dist) ** 0.5
    
    return scores


def _dynamic_regret_optimization(distance_matrix: np.ndarray, n: int,
                                 profile: Dict[str, float]) -> np.ndarray:
    """Dynamic regret computation with adaptive lookahead horizon."""
    regret_scores = np.ones((n, n))
    
    # Adaptive parameters based on instance
    if profile['nn_consistency'] > 0.7:
        lookahead = 2  # Consistent neighborhoods: shorter lookahead
    else:
        lookahead = 3  # Inconsistent: longer lookahead
    
    base_k = max(3, min(8, n // 5))
    
    for i in range(n):
        # Get base nearest neighbors
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        nn_i = np.argsort(dists_i)[:base_k]
        
        for j in range(n):
            if i == j:
                continue
            
            direct_cost = distance_matrix[i, j]
            best_alternative = np.inf
            
            # Lookahead through neighbors
            for step in range(1, lookahead + 1):
                if step == 1:
                    # One-step alternatives
                    for k in nn_i:
                        if k != j:
                            alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                            best_alternative = min(best_alternative, alt_cost)
                else:
                    # Multi-step alternatives (limited search)
                    for k in nn_i[:min(3, len(nn_i))]:
                        if k != j:
                            # Get k's nearest neighbors
                            dists_k = distance_matrix[k].copy()
                            dists_k[k] = np.inf
                            dists_k[i] = np.inf  # Avoid going back
                            nn_k = np.argsort(dists_k)[:min(3, n-2)]
                            
                            for l in nn_k:
                                if l != i and l != j:
                                    alt_cost = (distance_matrix[i, k] + 
                                              distance_matrix[k, l] +
                                              distance_matrix[l, j])
                                    best_alternative = min(best_alternative, alt_cost)
            
            # Calculate regret score
            if best_alternative < np.inf and direct_cost > 0:
                regret_ratio = best_alternative / direct_cost
                # Convert to score: lower regret ratio = higher score
                if regret_ratio > 1.0:
                    score = 1.0 / regret_ratio
                else:
                    score = 2.0 - regret_ratio
                
                regret_scores[i, j] = np.clip(score, 0.1, 1.9)
    
    # Normalize
    max_regret = regret_scores.max()
    min_regret = regret_scores.min()
    if max_regret - min_regret > 1e-10:
        regret_scores = (regret_scores - min_regret) / (max_regret - min_regret)
    
    return regret_scores


def _enhanced_mst_scoring(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Enhanced MST-based scoring with edge criticality analysis."""
    scores = np.ones((n, n))
    
    if n < 3:
        return scores
    
    # Compute MST
    mst = minimum_spanning_tree(distance_matrix)
    mst_adj = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Boost for MST edges
    scores[mst_adj] = 1.5
    
    # Additional boost for critical edges (high-degree nodes in MST)
    mst_degrees = mst_adj.sum(axis=1)
    max_degree = mst_degrees.max()
    
    if max_degree > 0:
        for i in range(n):
            for j in range(i + 1, n):
                if mst_adj[i, j]:
                    # Edge connecting high-degree nodes gets extra boost
                    degree_score = (mst_degrees[i] + mst_degrees[j]) / (2 * max_degree)
                    scores[i, j] = scores[j, i] = 1.5 + 0.5 * degree_score
    
    # Penalty for very long edges not in MST
    mean_dist = np.mean(distance_matrix[np.triu_indices(n, k=1)])
    for i in range(n):
        for j in range(i + 1, n):
            if not mst_adj[i, j] and distance_matrix[i, j] > 2 * mean_dist:
                scores[i, j] = scores[j, i] = 0.5
    
    return scores


def _adaptive_density_scoring(distance_matrix: np.ndarray, n: int,
                              profile: Dict[str, float]) -> np.ndarray:
    """Density scoring that adapts to local neighborhood characteristics."""
    scores = np.ones((n, n))
    
    if n < 4:
        return scores
    
    # Adaptive neighborhood size
    k = max(3, min(10, n // 4))
    
    # Compute local densities
    densities = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_dists = np.partition(dists, k)[:k]
        densities[i] = np.median(nn_dists)
    
    # Normalize densities
    density_mean = densities.mean()
    if density_mean > 0:
        rel_densities = densities / density_mean
    
    # Score edges based on relative densities
    for i in range(n):
        for j in range(n):
            if i != j:
                dens_i = rel_densities[i] if 'rel_densities' in locals() else 1.0
                dens_j = rel_densities[j] if 'rel_densities' in locals() else 1.0
                
                # Edge in dense area but relatively short
                if dens_i < 0.8 and dens_j < 0.8:
                    if distance_matrix[i, j] < profile['q1']:
                        scores[i, j] = 1.4
                    else:
                        scores[i, j] = 1.2
                # Edge connecting different density regions
                elif abs(dens_i - dens_j) > 0.5:
                    scores[i, j] = 1.3
                # Edge in sparse area
                elif dens_i > 1.2 and dens_j > 1.2:
                    scores[i, j] = 0.8
    
    return scores


def _multi_scale_centrality(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Centrality scoring at multiple scales."""
    scores = np.ones((n, n))
    
    # Local centrality (degree in k-NN graph)
    k_local = max(2, n // 8)
    local_centrality = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nn_count = np.sum(dists <= np.partition(dists, k_local)[k_local])
        local_centrality[i] = nn_count / (n - 1)
    
    # Global centrality (closeness approximation)
    with np.errstate(divide='ignore', invalid='ignore'):
        harmonic_centrality = np.sum(1.0 / (distance_matrix + np.eye(n) * 1e-10), axis=1)
    
    # Normalize centralities
    if local_centrality.max() > 0:
        local_centrality = local_centrality / local_centrality.max()
    
    if harmonic_centrality.max() > 0:
        harmonic_centrality = harmonic_centrality / harmonic_centrality.max()
    
    # Combine
    centrality = 0.6 * local_centrality + 0.4 * harmonic_centrality
    
    # Convert to edge scores
    for i in range(n):
        for j in range(n):
            if i != j:
                # Higher centrality sum = more promising connection
                cent_sum = centrality[i] + centrality[j]
                scores[i, j] = 0.8 + 0.4 * cent_sum
    
    return scores


def _learn_component_weights(profile: Dict[str, float], n: int) -> Dict[str, float]:
    """Learn optimal weights for component combination."""
    # Base weights based on instance size
    if n < 20:
        base_weights = {
            'attention': 0.25,
            'regret': 0.20,
            'mst': 0.15,
            'density': 0.20,
            'centrality': 0.20
        }
    elif n < 100:
        base_weights = {
            'attention': 0.30,
            'regret': 0.25,
            'mst': 0.15,
            'density': 0.15,
            'centrality': 0.15
        }
    else:
        base_weights = {
            'attention': 0.35,
            'regret': 0.25,
            'mst': 0.10,
            'density': 0.15,
            'centrality': 0.15
        }
    
    # Adjust based on instance characteristics
    adjustments = {
        'high_variance': profile['std'] / (profile['mean'] + 1e-10) > 0.7,
        'clustered': profile['concentration'] > 0.6,
        'consistent_nn': profile['nn_consistency'] > 0.7
    }
    
    if adjustments['high_variance']:
        base_weights['regret'] *= 1.2
        base_weights['attention'] *= 0.9
    
    if adjustments['clustered']:
        base_weights['attention'] *= 1.1
        base_weights['density'] *= 0.8
    
    if adjustments['consistent_nn']:
        base_weights['regret'] *= 0.9
        base_weights['centrality'] *= 1.1
    
    # Normalize
    total = sum(base_weights.values())
    return {k: v / total for k, v in base_weights.items()}


def _geometric_mean_ensemble(components: Dict[str, np.ndarray],
                             weights: Dict[str, float], n: int) -> np.ndarray:
    """Combine components using weighted geometric mean."""
    result = np.ones((n, n))
    
    # Ensure all matrices have positive values
    for key in components:
        components[key] = np.maximum(components[key], 1e-10)
    
    # Compute weighted geometric mean
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            log_sum = 0.0
            total_weight = 0.0
            
            for key, weight in weights.items():
                value = components[key][i, j]
                log_sum += weight * np.log(value)
                total_weight += weight
            
            if total_weight > 0:
                result[i, j] = np.exp(log_sum / total_weight)
    
    return result


def _cluster_aware_sparsification(heuristic: np.ndarray, 
                                  distance_matrix: np.ndarray,
                                  n: int, profile: Dict[str, float]) -> np.ndarray:
    """Intelligent sparsification preserving cluster structure."""
    result = heuristic.copy()
    
    if n < 6:
        return result
    
    # Always preserve MST edges
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Adaptive clustering for sparsification
    if profile['concentration'] > 0.6:
        n_clusters = max(2, min(6, n // 4))
    else:
        n_clusters = max(2, min(8, n // 3))
    
    try:
        # Cluster nodes
        Z = linkage(distance_matrix, method='ward')
        clusters = fcluster(Z, t=n_clusters, criterion='maxclust')
        
        # Preserve top edges within and between clusters
        for i in range(n):
            # Get edges to preserve
            preserve_indices = set()
            
            # Always preserve MST edges
            preserve_indices.update(np.where(mst_mask[i])[0])
            
            # Preserve top intra-cluster edges
            same_cluster = np.where(clusters == clusters[i])[0]
            same_cluster = same_cluster[same_cluster != i]
            
            if len(same_cluster) > 0:
                intra_scores = result[i, same_cluster]
                n_preserve_intra = min(3, len(same_cluster))
                top_intra = np.argsort(intra_scores)[-n_preserve_intra:]
                preserve_indices.update(same_cluster[top_intra])
            
            # Preserve top inter-cluster edges
            other_clusters = np.where(clusters != clusters[i])[0]
            if len(other_clusters) > 0:
                inter_scores = result[i, other_clusters]
                n_preserve_inter = min(2, len(other_clusters))
                top_inter = np.argsort(inter_scores)[-n_preserve_inter:]
                preserve_indices.update(other_clusters[top_inter])
            
            # Zero out non-preserved edges
            for j in range(n):
                if j != i and j not in preserve_indices:
                    result[i, j] = 0
        
        # Ensure symmetry
        result = np.maximum(result, result.T)
        
    except:
        # Fallback: preserve top-k edges per node
        k = max(3, min(8, n // 3))
        for i in range(n):
            row = result[i].copy()
            row[i] = -np.inf
            top_k = np.argsort(row)[-k:]
            for j in range(n):
                if j != i and j not in top_k and not mst_mask[i, j]:
                    result[i, j] = result[j, i] = 0
    
    return result


def _contrast_enhancement(heuristic: np.ndarray, 
                          distance_matrix: np.ndarray,
                          n: int) -> np.ndarray:
    """Enhance contrast between promising and non-promising edges."""
    result = heuristic.copy()
    
    # Apply softmax-like normalization for better contrast
    for i in range(n):
        row = result[i].copy()
        row[i] = -np.inf
        
        if np.any(row > 0):
            max_val = np.max(row)
            if max_val > 0:
                # Exponential scaling
                exp_row = np.exp(3.0 * (row / max_val - 1.0))
                # Normalize to sum to 1
                row_sum = np.sum(exp_row) + 1e-10
                result[i] = exp_row / row_sum
                result[i, i] = 0
    
    # Ensure symmetry
    result = (result + result.T) / 2
    
    # Final normalization to [0, 1]
    max_val = result.max()
    if max_val > 0:
        result = result / max_val
    
    return result
```
