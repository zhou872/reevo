```python
def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Next-generation TSP edge heuristic with neural-inspired adaptive fusion.
    
    Core Innovations:
    1. Spectral instance profiling for global structure capture
    2. Neural network-inspired multi-layer heuristic fusion
    3. Learnable component weighting via instance characteristics
    4. Gradient-aware sparsification with preservation guarantees
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, normalized to [0,1]
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # === 1. ADVANCED INSTANCE PROFILING ===
    profile = instance_profiler(distance_matrix, n)
    
    # === 2. MULTI-LAYER HEURISTIC COMPONENTS ===
    components = compute_multi_layer_components(distance_matrix, profile, n)

    heuristics = adaptive_edge_pruning(heuristics, distance_matrix, profile)

    np.fill_diagonal(heuristics, 0)
    max_val = heuristics.max()

        heuristics /= max_val
    
    return heuristics


def instance_profiler(distance_matrix: np.ndarray, n: int) -> dict:
    """
    Comprehensive instance characterization using spectral and structural features.
    """
    profile = {}
    
    # Basic distance statistics
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    profile['dist_mean'] = np.mean(upper_tri)
    profile['dist_std'] = np.std(upper_tri)
    profile['dist_cv'] = profile['dist_std'] / (profile['dist_mean'] + 1e-12)
    
    # Spectral profile
    profile.update(compute_spectral_profile(distance_matrix, n))
    
    # Structural profile
    profile.update(compute_structural_profile(distance_matrix, n))
    
    # Instance complexity metrics
    profile['complexity'] = estimate_instance_complexity(distance_matrix, n)
    
    return profile


def compute_spectral_profile(distance_matrix: np.ndarray, n: int) -> dict:
    """
    Compute spectral graph features including Laplacian eigenvalues and eigenvector statistics.
    """
    spectral = {}
    
    try:
        # Convert to similarity matrix
        with np.errstate(divide='ignore', invalid='ignore'):
            sim = 1 / (distance_matrix + np.eye(n) * 1e-12)
        sim[np.isinf(sim)] = 0
        np.fill_diagonal(sim, 0)
        
        # Normalized Laplacian
        D = np.diag(sim.sum(axis=1))
        D_inv_sqrt = np.diag(1 / np.sqrt(np.diag(D) + 1e-12))
        L_norm = np.eye(n) - D_inv_sqrt @ sim @ D_inv_sqrt
        
        # Compute eigenvalues
        if n <= 1000:
            eigvals = eigh(L_norm, eigvals_only=True, subset_by_index=[0, min(5, n-1)])
        else:
            # Approximate with random projection for large n
            eigvals = approximate_eigenvalues(L_norm, k=5)
        
        # Spectral features
        spectral['spectral_gap'] = eigvals[1] - eigvals[0] if len(eigvals) > 1 else eigvals[0]
        spectral['eigentail'] = np.mean(eigvals[2:]) if len(eigvals) > 2 else 1.0
        
        # Fiedler vector statistics
        if n <= 500:
            _, eigvecs = eigh(L_norm, subset_by_index=[1, 1])
            fiedler = eigvecs[:, 0]
            spectral['fiedler_std'] = np.std(fiedler)
            spectral['fiedler_skew'] = np.abs(np.mean((fiedler - np.mean(fiedler))**3) / (np.std(fiedler)**3 + 1e-12))
        else:
            spectral['fiedler_std'] = 0.5
            spectral['fiedler_skew'] = 0.0
            
    except Exception:
        # Fallback values
        spectral['spectral_gap'] = 0.1
        spectral['eigentail'] = 1.0
        spectral['fiedler_std'] = 0.5
        spectral['fiedler_skew'] = 0.0
    
    return spectral


def approximate_eigenvalues(matrix: np.ndarray, k: int = 5) -> np.ndarray:
    """
    Approximate largest eigenvalues using power iteration with deflation.
    """
    n = matrix.shape[0]
    eigenvalues = []
    B = matrix.copy()
    
    for i in range(k):
        # Power iteration
        v = np.random.randn(n)
        v = v / np.linalg.norm(v)
        
        for _ in range(20):
            v_new = B @ v
            v_new = v_new / np.linalg.norm(v_new)
            if np.abs(1 - np.abs(v @ v_new)) < 1e-6:
                break
            v = v_new
        
        # Rayleigh quotient
        eigval = v @ B @ v
        
        # Deflation
        B = B - eigval * np.outer(v, v)
        eigenvalues.append(eigval)
    
    return np.array(eigenvalues)


def compute_structural_profile(distance_matrix: np.ndarray, n: int) -> dict:
    """
    Compute structural features including clustering, sparsity, and regularity.
    """
    structural = {}
    
    # Clustering coefficient approximation
    k = min(10, n-1)
    clustering_sum = 0
    valid_nodes = 0
    
    for i in range(min(100, n)):
        # Find k nearest neighbors
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        neighbors = np.argsort(dists)[:k]
        
        if len(neighbors) >= 2:
            # Subgraph density among neighbors
            subgraph = distance_matrix[np.ix_(neighbors, neighbors)]
            tri_count = np.sum(subgraph < np.percentile(distance_matrix[i], 25))
            possible = len(neighbors) * (len(neighbors) - 1)
            clustering_sum += tri_count / possible if possible > 0 else 0
            valid_nodes += 1
    
    structural['clustering'] = clustering_sum / max(valid_nodes, 1)
    
    # Sparsity measure
    threshold = np.percentile(distance_matrix[np.triu_indices(n, k=1)], 25)
    structural['sparsity'] = np.mean(distance_matrix < threshold)
    
    # Regularity measure (coefficient of variation of degrees)
    with np.errstate(divide='ignore', invalid='ignore'):
        degrees = 1 / np.mean(distance_matrix + np.eye(n) * 1e-12, axis=1)
    structural['regularity'] = np.std(degrees) / (np.mean(degrees) + 1e-12)
    
    return structural


def estimate_instance_complexity(distance_matrix: np.ndarray, n: int) -> float:
    """
    Estimate instance complexity using multiple metrics.
    """
    complexity_metrics = []
    
    # Distance distribution complexity
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    hist, _ = np.histogram(upper_tri, bins=min(20, n))
    entropy = -np.sum((hist / hist.sum()) * np.log(hist / hist.sum() + 1e-12))
    complexity_metrics.append(np.tanh(entropy / 5))
    
    # Triangle inequality violation rate
    samples = min(500, n * (n-1) // 2)
    violations = 0
    
    for _ in range(samples):
        i, j, k = np.random.choice(n, 3, replace=False)
        if (distance_matrix[i, j] > 
            distance_matrix[i, k] + distance_matrix[k, j] + 1e-12):
            violations += 1
    
    complexity_metrics.append(violations / samples)
    
    # MST irregularity
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = mst[mst > 0]
    if len(mst_edges) > 1:
        mst_cv = np.std(mst_edges) / (np.mean(mst_edges) + 1e-12)
        complexity_metrics.append(np.tanh(mst_cv))
    
    return np.mean(complexity_metrics)

    """

    """
    components = {}
    
    # Layer 1: Local structure

    components['centrality'] = compute_harmonic_centrality(distance_matrix, n)
    
    # Layer 4: Optimization-aware
    components['kopt'] = compute_advanced_kopt(distance_matrix, profile, n)

def compute_adaptive_inverse(distance_matrix: np.ndarray, profile: dict) -> np.ndarray:
    """
    Inverse distance with adaptive exponent based on instance characteristics.
    """
    n = distance_matrix.shape[0]
    
    # Adaptive exponent
    alpha = 0.8 + 0.4 * np.tanh(profile['complexity'] * 2 - 1)
    
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    return inv_dist ** alpha


def compute_enhanced_mst(distance_matrix: np.ndarray, profile: dict, n: int) -> np.ndarray:
    """
    MST-based heuristic with spectral enhancement.
    """
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # MST score matrix
    mst_score = np.zeros((n, n))
    mst_score[mst_mask] = 1.0
    
    # Spectral enhancement
    if profile['spectral_gap'] > 0.1 and n <= 500:
        try:
            # Compute Fiedler vector for spectral ordering
            with np.errstate(divide='ignore', invalid='ignore'):
                sim = 1 / (distance_matrix + np.eye(n) * 1e-12)
            sim[np.isinf(sim)] = 0
            np.fill_diagonal(sim, 0)
            
            D = np.diag(sim.sum(axis=1))
            D_inv_sqrt = np.diag(1 / np.sqrt(np.diag(D) + 1e-12))
            L_norm = np.eye(n) - D_inv_sqrt @ sim @ D_inv_sqrt
            
            _, eigvecs = eigh(L_norm, subset_by_index=[1, 1])
            fiedler = eigvecs[:, 0]
            
            # Order nodes by Fiedler vector
            order = np.argsort(fiedler)
            
            # Boost edges between adjacent nodes in spectral order
            for i in range(n-1):
                mst_score[order[i], order[i+1]] *= 1.2
                mst_score[order[i+1], order[i]] *= 1.2
        except:
            pass
    
    # Adaptive MST boost
    mst_boost = 1.5 + 0.5 * np.tanh(profile['complexity'] * 2)
    mst_score[mst_mask] *= mst_boost
    
    return mst_score


def compute_multi_scale_regret(distance_matrix: np.ndarray, profile: dict, n: int) -> np.ndarray:
    """
    Multi-scale regret analysis with adaptive neighborhood sizes.
    """
    regret = np.ones((n, n))
    
    # Adaptive neighborhood sizes
    k_local = max(3, min(8, int(n * 0.1)))
    k_global = max(5, min(15, int(n * 0.2)))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_idx = np.argsort(dists)
        
        # Multi-scale ranks
        for j in range(n):
            if i == j:
                continue
            
            # Find ranks at different scales
            local_rank = np.where(sorted_idx[:k_local] == j)[0]
            global_rank = np.where(sorted_idx[:k_global] == j)[0]
            
            # Compute regrets
            local_regret = 1.0 / (1.0 + 0.5 * (local_rank[0] if len(local_rank) > 0 else k_local))
            global_regret = 1.0 / (1.0 + 0.3 * (global_rank[0] if len(global_rank) > 0 else k_global))
            
            # Distance ratio regret
            best_dist = dists[sorted_idx[0]]
            ratio = dists[j] / best_dist
            ratio_regret = np.exp(-0.5 * (ratio - 1))
            
            # Combine with adaptive weights
            w_local = 0.4 + 0.2 * profile['clustering']
            w_global = 0.3 + 0.2 * (1 - profile['clustering'])
            w_ratio = 0.3
            
            regret[i, j] = (w_local * local_regret + 
                           w_global * global_regret + 
                           w_ratio * ratio_regret)
    
    return np.sqrt(regret * regret.T)


def compute_adaptive_clustering(distance_matrix: np.ndarray, profile: dict, n: int) -> np.ndarray:
    """
    Adaptive clustering coefficient based on instance density.
    """
    clustering = np.ones((n, n))
    k = min(10, n-1)
    
    # Precompute k-nearest neighbors
    knn_masks = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        mask = np.zeros(n, dtype=bool)
        mask[np.argsort(dists)[:k]] = True
        knn_masks.append(mask)
    
    # Compute mutual clustering scores
    for i in range(n):
        for j in range(i+1, n):
            # Mutual NN
            mutual = knn_masks[i][j] and knn_masks[j][i]
            
            # Common neighbors
            common = np.sum(knn_masks[i] & knn_masks[j])
            common_ratio = common / k
            
            # Jaccard similarity of neighborhoods
            union = np.sum(knn_masks[i] | knn_masks[j])
            jaccard = common / union if union > 0 else 0
            
            # Combine with density-adaptive weights
            w_mutual = 0.3 + 0.2 * (1 - profile['sparsity'])
            w_common = 0.4
            w_jaccard = 0.3 + 0.2 * profile['sparsity']
            
            clustering[i, j] = clustering[j, i] = (
                w_mutual * float(mutual) +
                w_common * common_ratio +
                w_jaccard * jaccard
            )
    
    return clustering


def compute_spectral_heuristic(distance_matrix: np.ndarray, profile: dict, n: int) -> np.ndarray:
    """
    Spectral embedding based heuristic.
    """
    spectral = np.ones((n, n))
    
    if n <= 300 and profile['spectral_gap'] > 0.05:
        try:
            # Compute spectral embedding
            with np.errstate(divide='ignore', invalid='ignore'):
                sim = 1 / (distance_matrix + np.eye(n) * 1e-12)
            sim[np.isinf(sim)] = 0
            np.fill_diagonal(sim, 0)
            
            # Get top eigenvectors
            k_dim = min(3, n-1)
            eigvals, eigvecs = eigh(sim, subset_by_index=[n-k_dim, n-1])
            
            # Normalize by eigenvalues
            eigvecs = eigvecs * np.sqrt(np.abs(eigvals)) / np.sqrt(np.sum(np.abs(eigvals)))
            
            # Compute cosine similarity in embedding space
            norms = np.linalg.norm(eigvecs, axis=1, keepdims=True)
            norms[norms == 0] = 1
            normalized = eigvecs / norms
            
            spectral = normalized @ normalized.T
            spectral = np.maximum(spectral, 0)
            
        except Exception:
            pass
    
    return spectral


def compute_harmonic_centrality(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """
    Harmonic centrality based on inverse distances.
    """
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dists = 1 / (distance_matrix + np.eye(n) * 1e-12)
    inv_dists[np.isinf(inv_dists)] = 0
    np.fill_diagonal(inv_dists, 0)
    
    # Harmonic centrality scores
    centrality = np.sum(inv_dists, axis=1) / (n - 1)
    
    # Create edge centrality matrix
    centrality_mat = np.outer(centrality, centrality)
    
    # Non-linear enhancement
    centrality_mat = centrality_mat ** 1.5
    
    return centrality_mat


def compute_advanced_kopt(distance_matrix: np.ndarray, profile: dict, n: int) -> np.ndarray:
    """
    Advanced k-opt potential estimation with adaptive depth.
    """
    kopt = np.ones((n, n))
    
    # Adaptive search depth
    depth = min(3, max(1, int(2 + profile['complexity'] * 2)))
    k_neighbors = min(8, max(3, int(n * 0.15)))
    
    for i in range(n):
        i_neighbors = np.argsort(distance_matrix[i])[1:k_neighbors+1]
        
        for j in range(i+1, n):
            j_neighbors = np.argsort(distance_matrix[j])[1:k_neighbors+1]
            current = distance_matrix[i, j]
            
            best_gain = 0
            
            # 2-opt search
            for k in i_neighbors:
                if k == j:
                    continue
                for l in j_neighbors:
                    if l == i or l == k:
                        continue
                    
                    gain = (current + distance_matrix[k, l] - 
                           distance_matrix[i, k] - distance_matrix[j, l])
                    best_gain = max(best_gain, gain)
            
            # 3-opt search if beneficial
            if depth >= 3 and best_gain <= current * 0.1:
                # Simplified 3-opt estimation
                if len(i_neighbors) >= 2 and len(j_neighbors) >= 2:
                    candidates = []
                    for k in i_neighbors[:2]:
                        for l in j_neighbors[:2]:
                            if k != l and k != j and l != i:
                                for m in set(i_neighbors) | set(j_neighbors):
                                    if m not in (i, j, k, l):
                                        gain_3opt = estimate_3opt_gain(
                                            distance_matrix, i, j, k, l, m, current)
                                        candidates.append(gain_3opt)
                    
                    if candidates:
                        best_gain = max(best_gain, max(candidates))
            
            # Convert gain to heuristic
            if best_gain > 0:
                gain_norm = min(best_gain / profile['dist_mean'], 2.0)
                kopt[i, j] = 1.0 + 0.3 * np.tanh(gain_norm)
    
    kopt = np.maximum(kopt, kopt.T)
    return kopt


def estimate_3opt_gain(dist_mat: np.ndarray, i: int, j: int, k: int, l: int, m: int, 
                       current: float) -> float:
    """
    Estimate 3-opt gain for a specific configuration.
    """
    # One possible 3-opt move pattern
    original = (current + dist_mat[k, l] + dist_mat[l, m] + dist_mat[m, i])
    new = (dist_mat[i, k] + dist_mat[k, m] + dist_mat[m, l] + dist_mat[l, j])
    
    return original - new


def compute_gradient_field(distance_matrix: np.ndarray, profile: dict) -> np.ndarray:
    """
    Compute gradient field indicating sensitivity to distance changes.
    """
    n = distance_matrix.shape[0]
    gradient = np.ones((n, n))
    
    for i in range(n):
        dists = distance_matrix[i]
        sorted_idx = np.argsort(dists)
        
        if len(sorted_idx) > 3:
            # Analyze distance gradient
            sorted_dists = dists[sorted_idx[1:4]]
            if sorted_dists[0] > 0:
                ratios = sorted_dists[1:] / sorted_dists[0]
                grad_factor = np.mean(np.abs(ratios - 1))
                
                # Apply gradient sensitivity
                for rank, j in enumerate(sorted_idx[1:], 1):
                    if i != j:
                        sensitivity = 1.0 / (1.0 + 0.3 * rank * grad_factor)
                        gradient[i, j] = sensitivity
    
    gradient = np.maximum(gradient, gradient.T)
    return gradient


def neural_fusion_weights(profile: dict, n: int) -> dict:
    """
    Neural-inspired adaptive weighting of heuristic components.
    """
    weights = {}
    
    # Base weights from instance profile
    sg = profile.get('spectral_gap', 0.1)
    clust = profile.get('clustering', 0.5)
    comp = profile.get('complexity', 0.5)
    sparsity = profile.get('sparsity', 0.5)
    
    # Component exponents (non-linear transformations)
    weights['inv_exp'] = 0.7 + 0.3 * np.tanh(sg * 3 - 1)
    weights['mst_exp'] = 1.0 + 0.4 * np.tanh(comp * 2 - 1)
    weights['regret_exp'] = 0.8 + 0.4 * np.tanh(profile.get('dist_cv', 1.0) * 2)
    weights['cluster_exp'] = 0.6 + 0.4 * clust
    weights['spectral_exp'] = 0.5 + 0.5 * np.tanh(sg * 4)
    weights['central_exp'] = 0.7 + 0.3 * (1 - sparsity)
    weights['kopt_exp'] = 1.2 - 0.4 * np.tanh(n / 50)
    weights['grad_exp'] = 0.9 + 0.2 * np.tanh(comp * 3 - 1.5)
    
    # Interaction weights (how components combine)
    weights['interaction_strength'] = 0.3 + 0.4 * comp
    
    return weights


def fuse_components_neural(components: dict, weights: dict, profile: dict) -> np.ndarray:
    """
    Neural-inspired fusion of components with learned interactions.
    """
    n = next(iter(components.values())).shape[0]
    fused = np.ones((n, n))
    
    # Apply component exponents
    for comp_name, comp_matrix in components.items():
        exp_key = f"{comp_name.split('_')[0]}_exp"
        if exp_key in weights:
            exponent = weights[exp_key]
            fused *= comp_matrix ** exponent
    
    # Add interaction terms
    interaction_strength = weights.get('interaction_strength', 0.5)
    
    # Key interactions
    if 'inverse' in components and 'regret' in components:
        interaction = components['inverse'] * components['regret']
        fused *= (1 + interaction_strength * interaction)
    
    if 'mst' in components and 'spectral' in components:
        interaction = components['mst'] * components['spectral']
        fused *= (1 + 0.5 * interaction_strength * interaction)
    
    # Ensure non-negativity
    fused = np.maximum(fused, 0)
    
    return fused


def gradient_aware_refinement(heuristics: np.ndarray, distance_matrix: np.ndarray, 
                             profile: dict) -> np.ndarray:
    """
    Refine heuristics using gradient information and local consistency.
    """
    n = heuristics.shape[0]
    refined = heuristics.copy()
    
    # Local smoothing
    kernel_size = min(5, int(n * 0.1))
    if kernel_size >= 3:
        for i in range(n):
            # Find nearest neighbors for smoothing
            dists = distance_matrix[i]
            nearest = np.argsort(dists)[1:kernel_size+1]
            
            # Weighted average of heuristic values
            weights = 1 / (dists[nearest] + 1e-12)
            weights = weights / weights.sum()
            
            for j in range(i+1, n):
                if j in nearest:
                    # Local consistency term
                    neighbor_vals = heuristics[nearest, j]
                    local_avg = np.sum(weights * neighbor_vals)
                    
                    # Blend with original
                    alpha = 0.3 + 0.2 * profile.get('clustering', 0.5)
                    refined[i, j] = refined[j, i] = (
                        alpha * heuristics[i, j] + (1 - alpha) * local_avg
                    )
    
    return refined


def adaptive_edge_pruning(heuristics: np.ndarray, distance_matrix: np.ndarray, 
                         profile: dict) -> np.ndarray:
    """
    Intelligent edge pruning based on multiple criteria.
    """
    n = heuristics.shape[0]
    pruned = heuristics.copy()
    
    if n <= 10:
        return pruned
    
    # Compute MST for preservation
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # Separate edges
    non_mst_mask = ~mst_mask
    np.fill_diagonal(non_mst_mask, False)
    
    if np.any(non_mst_mask):
        non_mst_vals = pruned[non_mst_mask]
        
        if len(non_mst_vals) > 0:
            # Adaptive thresholding
            q50 = np.percentile(non_mst_vals, 50)
            q90 = np.percentile(non_mst_vals, 90)
            
            # Threshold based on instance characteristics
            if profile.get('sparsity', 0.5) > 0.7:
                # Dense graphs: keep fewer edges
                threshold = q90 * 0.8
            elif profile.get('complexity', 0.5) > 0.7:
                # Complex instances: keep more edges
                threshold = q50 * 0.9
            else:
                threshold = (q50 + q90) / 2
            
            # Soft thresholding with randomness for exploration
            rng = np.random.RandomState(42)
            for i in range(n):
                for j in range(i+1, n):
                    if non_mst_mask[i, j] and pruned[i, j] < threshold:
                        # Probability of keeping based on distance to threshold
                        keep_prob = np.exp((pruned[i, j] - threshold) / (threshold + 1e-12))
                        if rng.random() > keep_prob:
                            pruned[i, j] = pruned[j, i] = 0
    
    return pruned
```
