import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection.
    Returns a matrix of same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    np.fill_diagonal(distance_matrix, np.inf)  # Avoid self-connections
    heuristics = np.zeros_like(distance_matrix)
    
    # 1. Inverse distance base (like v1)
    inv_dist = 1 / distance_matrix
    
    # 2. Minimum Spanning Tree centrality
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_presence = (mst + mst.T > 0).astype(float) * 2.0
    
    # 3. Local neighborhood sparsity factor
    k = min(5, n // 4)  # Adaptive neighborhood size
    local_density = np.zeros(n)
    for i in range(n):
        sorted_dists = np.sort(distance_matrix[i])
        local_density[i] = np.mean(sorted_dists[1:k+1])  # Exclude self
    
    neighborhood_factor = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i,j] < 1.5 * max(local_density[i], local_density[j]):
                neighborhood_factor[i,j] = 1.0
                neighborhood_factor[j,i] = 1.0
    
    # 4. Angular dispersion (approximated via distance ratios)
    angular_scores = np.ones_like(distance_matrix)
    if n > 3:
        for i in range(n):
            # Find two closest neighbors
            closest = np.argsort(distance_matrix[i])[1:3]
            if len(closest) == 2:
                a, b = closest
                # Law of cosines approximation
                cos_angle = (distance_matrix[i,a]**2 + distance_matrix[i,b]**2 - 
                            distance_matrix[a,b]**2) / (2 * distance_matrix[i,a] * distance_matrix[i,b])
                dispersion = 1.0 - abs(cos_angle)  # Higher for orthogonal edges
                angular_scores[i,a] += dispersion * 0.3
                angular_scores[i,b] += dispersion * 0.3
    
    # 5. Two-hop regret penalty
    regret_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            # Alternative paths through other nodes
            alternatives = []
            for k in range(n):
                if k != i and k != j:
                    alt_path = distance_matrix[i,k] + distance_matrix[k,j]
                    alternatives.append(alt_path)
            if alternatives:
                best_alternative = min(alternatives)
                regret = best_alternative - distance_matrix[i,j]
                regret_matrix[i,j] = regret_matrix[j,i] = np.exp(-regret / np.mean(distance_matrix))
    
    # Combine all factors
    heuristics = (inv_dist * 1.0 +
                  mst_presence * 0.8 +
                  neighborhood_factor * 0.6 +
                  angular_scores * 0.4 +
                  regret_matrix * 0.5)
    
    # 6. Adaptive sparsification
    threshold = np.percentile(heuristics, 70)  Keep top 30%
    heuristics[heuristics < threshold] = 0
    
    # 7. Symmetrize and clean
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
