import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization.
    
    Features:
    1. Rank-based regret penalization with adaptive scaling
    2. Centrality-based edge promotion using inverse average distances
    3. Base weighting using inverse distances
    4. Adaptive sparsification based on instance characteristics
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values indicating
        more promising edges for inclusion in optimal tours.
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # === 1. INSTANCE CHARACTERISTICS ANALYSIS ===
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    dist_cv = dist_std / (dist_mean + 1e-12)
    
    # Adaptive parameters
    regret_factor = 0.8 + 0.4 * np.tanh(dist_cv)
    sparsity_param = 0.7 + 0.3 * np.tanh(8/n - 0.4)
    
    # === 2. BASE INVERSE DISTANCE ===
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # === 3. RANK-BASED REGRET PENALIZATION ===
    regret_weights = np.ones((n, n))
    
    for i in range(n):
        # Get all other nodes
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        other_nodes = np.where(mask)[0]
        
        if len(other_nodes) >= 2:
            # Get distances to other nodes and sort them
            dists = distance_matrix[i, other_nodes]
            sorted_indices = np.argsort(dists)
            
            # Apply rank-based penalty
            for rank, idx in enumerate(sorted_indices):
                j = other_nodes[idx]
                # Penalize worse-ranked edges more heavily
                penalty = 1.0 / (1.0 + regret_factor * rank**0.8)
                regret_weights[i, j] = penalty
    
    # Symmetrize regret weights
    regret_weights = (regret_weights + regret_weights.T) / 2
    
    # === 4. CENTRALITY-BASED EDGE PROMOTION ===
    # Calculate node centrality as inverse of average distance to other nodes
    with np.errstate(divide='ignore', invalid='ignore'):
        centrality = 1 / np.mean(distance_matrix + np.eye(n) * 1e-12, axis=1)
    
    # Normalize centrality to [0, 1] range
    if centrality.max() > centrality.min():
        centrality = (centrality - centrality.min()) / (centrality.max() - centrality.min())
    
    # Create edge centrality score by multiplying node centralities
    centrality_matrix = np.outer(centrality, centrality)
    
    # Boost edges between highly central nodes
    centrality_power = 1.5
    centrality_matrix = centrality_matrix ** centrality_power
    
    # === 5. MINIMUM SPANNING TREE BASELINE ===
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # === 6. COMBINE FACTORS ===
    heuristics = inv_dist.copy()
    heuristics *= regret_weights
    heuristics *= centrality_matrix
    
    # Boost MST edges moderately
    mst_boost = 1.2
    heuristics[mst_mask] *= mst_boost
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    
    # === 7. ADAPTIVE SPARSIFICATION ===
    if sparsity_param > 0.5:  # Only sparsify for larger instances or when appropriate
        # Separate MST and non-MST edges
        non_mst_mask = ~mst_mask
        np.fill_diagonal(non_mst_mask, False)
        
        if np.any(non_mst_mask):
            non_mst_values = heuristics[non_mst_mask]
            if len(non_mst_values) > 0:
                # Calculate dynamic thresholds
                median_val = np.median(non_mst_values)
                mean_val = np.mean(non_mst_values)
                threshold = median_val * 0.8 + mean_val * 0.2
                
                # Apply threshold: zero out edges below threshold
                zero_mask = non_mst_mask & (heuristics < threshold)
                heuristics[zero_mask] = 0
    
    # === 8. FINAL PROCESSING ===
    np.fill_diagonal(heuristics, 0)
    
    # Normalize to [0, 1] range for consistency
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
