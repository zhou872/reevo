import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy import stats

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge prioritization with adaptive
    lookahead depth and cluster-aware scoring.
    
    Key innovations:
    1. Adaptive lookahead depth based on instance characteristics
    2. Cluster-aware edge evaluation using density-based segmentation
    3. Dynamic factor weighting based on instance statistics
    4. Multi-stage scoring with progressive refinement
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values indicating
        more promising edges for inclusion in optimal tours.
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # === 1. INSTANCE ANALYSIS & ADAPTIVE PARAMETERS ===
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    dist_cv = dist_std / (dist_mean + 1e-12)
    n_log = np.log(n)
    
    # Adaptive parameters based on instance characteristics
    regret_strength = 0.8 + 0.4 * np.tanh(dist_cv)
    centrality_weight = 0.6 + 0.4 * (1 - np.exp(-dist_cv))
    lookahead_depth = max(1, min(3, int(2 + 3 * dist_cv - 0.2 * n_log)))
    cluster_sensitivity = 0.4 + 0.3 * np.tanh(10/n - 0.5)
    
    # === 2. DENSITY-BASED CLUSTER IDENTIFICATION ===
    # Estimate local density for each node
    k_density = max(2, n // 5)
    densities = np.zeros(n)
    for i in range(n):
        dists = np.partition(distance_matrix[i], k_density + 1)[:k_density + 1]
        densities[i] = np.mean(dists[dists > 0])
    
    # Adaptive clustering based on density distribution
    if n >= 10:
        density_threshold = np.percentile(densities, 60)
        cluster_mask = densities < density_threshold
        n_clusters = max(2, int(np.sqrt(np.sum(cluster_mask)) + 0.5))
    else:
        cluster_mask = np.ones(n, dtype=bool)
        n_clusters = 1
    
    # === 3. RANK-BASED REGRET WITH CLUSTER AWARENESS ===
    regret_matrix = np.ones((n, n))
    
    for i in range(n):
        mask = np.arange(n) != i
        dists_i = distance_matrix[i, mask]
        sorted_idx = np.argsort(dists_i)
        
        # Create rank-based penalties with cluster adjustment
        ranks = np.arange(1, len(dists_i) + 1)
        base_penalties = 1 / (1 + regret_strength * (ranks - 1))
        
        # Adjust for cluster relationships
        if cluster_mask[i]:
            # For cluster nodes, emphasize connections within cluster
            for idx, j in enumerate(np.where(mask)[0]):
                if cluster_mask[j]:
                    base_penalties[idx] *= 1.2
        
        regret_matrix[i, mask] = base_penalties[np.argsort(sorted_idx)]
    
    regret_matrix = np.sqrt(regret_matrix * regret_matrix.T)
    
    # === 4. ADAPTIVE LOOKAHEAD SPATIAL SCORING ===
    lookahead_matrix = np.ones((n, n))
    
    if lookahead_depth >= 1:
        # Precompute sorted neighbor indices for efficiency
        neighbor_indices = np.zeros((n, min(n-1, 10)), dtype=int)
        for i in range(n):
            mask = np.arange(n) != i
            sorted_neighbors = np.argsort(distance_matrix[i, mask])[:10]
            neighbor_indices[i, :len(sorted_neighbors)] = np.where(mask)[0][sorted_neighbors]
        
        for i in range(n):
            i_neighbors = neighbor_indices[i, :min(5, n-1)]
            i_neighbors = i_neighbors[i_neighbors != 0]
            
            for j in range(i + 1, n):
                j_neighbors = neighbor_indices[j, :min(5, n-1)]
                j_neighbors = j_neighbors[j_neighbors != 0]
                
                # Combine candidate intermediate nodes
                candidates = np.unique(np.concatenate([i_neighbors, j_neighbors]))
                candidates = candidates[candidates != i]
                candidates = candidates[candidates != j]
                
                if len(candidates) == 0:
                    continue
                
                # Multi-step path evaluation
                best_path = np.inf
                for k in candidates:
                    # Two-step path
                    path_len = distance_matrix[i, k] + distance_matrix[k, j]
                    
                    # Optional three-step path for deeper lookahead
                    if lookahead_depth >= 2 and len(candidates) > 1:
                        for l in candidates[candidates != k]:
                            three_step = (distance_matrix[i, k] + 
                                        distance_matrix[k, l] + 
                                        distance_matrix[l, j])
                            path_len = min(path_len, three_step)
                    
                    best_path = min(best_path, path_len)
                
                if best_path < np.inf:
                    edge_ratio = best_path / (distance_matrix[i, j] + 1e-12)
                    # Non-linear transformation emphasizing significant differences
                    lookahead_score = np.exp(-abs(edge_ratio - 1) * 2)
                    lookahead_matrix[i, j] = lookahead_score
    
    lookahead_matrix = np.maximum(lookahead_matrix, lookahead_matrix.T)
    
    # === 5. CLUSTER-AWARE CENTRALITY MEASURES ===
    # Degree centrality with cluster weighting
    with np.errstate(divide='ignore', invalid='ignore'):
        avg_dist = np.mean(distance_matrix + np.eye(n) * 1e-12, axis=1)
        degree_centrality = 1 / avg_dist
    
    # MST-based centrality
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    mst_degree = np.sum(mst_mask, axis=1)
    
    # Cluster-aware centrality adjustment
    cluster_factor = np.ones(n)
    if n_clusters > 1:
        # Nodes connecting clusters get higher centrality
        for i in range(n):
            if cluster_mask[i]:
                # Count connections to other clusters
                cross_cluster = 0
                for j in range(n):
                    if i != j and not cluster_mask[j] and distance_matrix[i, j] < 1.5 * densities[i]:
                        cross_cluster += 1
                cluster_factor[i] = 1.0 + cluster_sensitivity * min(1.0, cross_cluster / 3)
    
    # Combine centrality measures
    degree_centrality = degree_centrality / (np.max(degree_centrality) + 1e-12)
    mst_centrality = mst_degree / (np.max(mst_degree) + 1e-12)
    
    centrality = (centrality_weight * degree_centrality + 
                  (1 - centrality_weight) * mst_centrality)
    centrality = centrality * cluster_factor
    centrality_matrix = np.outer(centrality, centrality)
    
    # === 6. DYNAMIC FACTOR COMBINATION ===
    # Base inverse distance
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Dynamic weights based on instance characteristics
    if dist_cv > 0.5:
        # High variability: emphasize lookahead
        lookahead_weight = 1.2
        regret_weight = 0.8
    else:
        # Low variability: balanced approach
        lookahead_weight = 0.9
        regret_weight = 1.1
    
    # Multiplicative combination with dynamic weights
    heuristics = (inv_dist ** 0.9 * 
                  regret_matrix ** regret_weight * 
                  centrality_matrix ** 0.7 * 
                  lookahead_matrix ** lookahead_weight)
    
    # Boost MST edges and intra-cluster connections
    heuristics[mst_mask] *= 1.4
    
    # Boost edges within clusters
    for i in range(n):
        for j in range(i + 1, n):
            if cluster_mask[i] and cluster_mask[j]:
                density_factor = min(1.5, 1.0 + 0.5 * (1 - distance_matrix[i, j] / densities[i]))
                heuristics[i, j] *= density_factor
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    
    # === 7. ADAPTIVE SPARSIFICATION WITH PROGRESSIVE REFINEMENT ===
    if n > 15:
        # Multi-stage sparsification
        non_mst = ~mst_mask
        np.fill_diagonal(non_mst, False)
        
        if np.any(non_mst):
            non_mst_vals = heuristics[non_mst]
            
            # Stage 1: Remove very low values
            q10 = np.percentile(non_mst_vals, 10)
            stage1_mask = non_mst & (heuristics < q10)
            heuristics[stage1_mask] = 0
            
            # Stage 2: Adaptive threshold based on value distribution
            remaining_vals = heuristics[non_mst & ~stage1_mask]
            if len(remaining_vals) > 0:
                median_val = np.median(remaining_vals)
                iqr = stats.iqr(remaining_vals)
                threshold = median_val + 0.3 * iqr
                
                zero_mask = non_mst & (heuristics > 0) & (heuristics < threshold)
                heuristics[zero_mask] = 0
    
    # === 8. FINAL NORMALIZATION AND PROCESSING ===
    np.fill_diagonal(heuristics, 0)
    
    # Ensure connectivity for small instances
    if n <= 20:
        row_sums = np.sum(heuristics > 0, axis=1)
        min_connections = 3
        for i in range(n):
            if row_sums[i] < min_connections:
                # Add connections to nearest neighbors
                mask = np.arange(n) != i
                nearest = np.argsort(distance_matrix[i, mask])[:min_connections - row_sums[i]]
                nearest_idx = np.where(mask)[0][nearest]
                heuristics[i, nearest_idx] = 0.5
                heuristics[nearest_idx, i] = 0.5
    
    # Normalize to [0, 1] range
    max_val = heuristics.max()
    if max_val > 0:
        heuristics /= max_val
    
    # Apply sigmoid-like transformation for better differentiation
    heuristics = 1 / (1 + np.exp(-5 * (heuristics - 0.5)))
    
    return heuristics
