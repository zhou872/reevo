import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization with three-hop lookahead,
    adaptive local density weighting, and MST-based centrality.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix of same shape as input with values in [0, 1]
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # Adaptive parameter tuning based on instance size
    k = min(max(3, int(np.sqrt(n))), n - 1)
    use_three_hop = n >= 50  # Use three-hop lookahead for larger instances
    
    # 1. Base inverse distance weighting with robust scaling
    with np.errstate(divide='ignore', invalid='ignore'):
        scale = np.median(distance_matrix[np.triu_indices(n, k=1)])
        inv_dist = scale / (distance_matrix + 0.1 * scale * np.eye(n))
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Compute k-nearest neighbors for all nodes
    # Add small noise to break distance ties for consistent sorting
    noise = np.random.randn(n, n) * 1e-10 * distance_matrix.max()
    noisy_dist = distance_matrix + noise
    sorted_indices = np.argsort(noisy_dist, axis=1)[:, 1:k+1]
    
    # 3. Advanced lookahead efficiency scoring with adaptive hop count
    efficiency_score = np.ones((n, n))
    
    # Precompute nearest neighbor distances for faster lookup
    neighbor_distances = np.zeros((n, k))
    for i in range(n):
        neighbor_distances[i] = distance_matrix[i, sorted_indices[i]]
    
    # For each node pair (i, j), find best alternative path
    for i in range(n):
        i_neighbors = sorted_indices[i]
        i_distances = neighbor_distances[i]
        
        for j in range(i + 1, n):
            j_neighbors = sorted_indices[j]
            j_distances = neighbor_distances[j]
            
            direct_dist = distance_matrix[i, j]
            best_alternative = np.inf
            
            # Two-hop alternatives through i's neighbors
            for idx, a_node in enumerate(i_neighbors):
                if a_node != j:
                    # Direct connection from a to j
                    if a_node in j_neighbors:
                        a_to_j_dist = distance_matrix[a_node, j]
                    else:
                        # Find a's nearest connection to j
                        a_neighbors = sorted_indices[a_node]
                        # Check if j is in a's neighbors
                        if j in a_neighbors:
                            a_to_j_dist = distance_matrix[a_node, j]
                        else:
                            # Minimum through a's nearest neighbors
                            a_to_j_dist = min([distance_matrix[m, j] for m in a_neighbors[:2] 
                                             if m != i and m != j], default=np.inf)
                    
                    alternative = i_distances[idx] + a_to_j_dist
                    best_alternative = min(best_alternative, alternative)
            
            # Two-hop alternatives through j's neighbors (symmetric)
            for idx, b_node in enumerate(j_neighbors):
                if b_node != i:
                    if b_node in i_neighbors:
                        b_to_i_dist = distance_matrix[b_node, i]
                    else:
                        b_neighbors = sorted_indices[b_node]
                        if i in b_neighbors:
                            b_to_i_dist = distance_matrix[b_node, i]
                        else:
                            b_to_i_dist = min([distance_matrix[m, i] for m in b_neighbors[:2] 
                                             if m != i and m != j], default=np.inf)
                    
                    alternative = j_distances[idx] + b_to_i_dist
                    best_alternative = min(best_alternative, alternative)
            
            # Three-hop alternatives for larger instances
            if use_three_hop and best_alternative < np.inf * 0.9:
                for idx_a, a_node in enumerate(i_neighbors[:min(3, k)]):
                    if a_node == j:
                        continue
                    
                    a_neighbors = sorted_indices[a_node]
                    a_to_i_dist = i_distances[idx_a]
                    
                    for idx_b, b_node in enumerate(a_neighbors[:min(3, k)]):
                        if b_node == i or b_node == j:
                            continue
                        
                        a_to_b_dist = distance_matrix[a_node, b_node]
                        
                        # Connection from b to j
                        if b_node in j_neighbors:
                            b_to_j_dist = distance_matrix[b_node, j]
                        else:
                            b_neighbors = sorted_indices[b_node]
                            if j in b_neighbors:
                                b_to_j_dist = distance_matrix[b_node, j]
                            else:
                                b_to_j_dist = min([distance_matrix[m, j] for m in b_neighbors[:2] 
                                                 if m != i and m != j], default=np.inf)
                        
                        alternative = a_to_i_dist + a_to_b_dist + b_to_j_dist
                        best_alternative = min(best_alternative, alternative)
            
            # Calculate efficiency score
            if best_alternative < np.inf and direct_dist > 0:
                efficiency_ratio = best_alternative / direct_dist
                efficiency_ratio = np.clip(efficiency_ratio, 0.1, 2.0)
                
                # Convert to score: better alternatives reduce direct edge score
                if efficiency_ratio > 1.0:
                    score = 2.0 / efficiency_ratio
                else:
                    score = efficiency_ratio
                
                efficiency_score[i, j] = efficiency_score[j, i] = np.clip(score, 0.5, 2.0)
    
    # 4. Adaptive local density weighting
    local_density = np.zeros(n)
    for i in range(n):
        # Compute average distance to k nearest neighbors
        local_density[i] = np.mean(neighbor_distances[i, :min(5, k)])
    
    density_factor = np.ones((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            # Edge score is boosted if distance is much lower than average local density
            avg_density = (local_density[i] + local_density[j]) / 2
            if direct_dist > 0:
                density_ratio = direct_dist / avg_density
                # Apply sigmoid-like transformation
                if density_ratio < 0.8:
                    # Much shorter than average - strong boost
                    density_factor[i, j] = density_factor[j, i] = 1.5
                elif density_ratio < 1.0:
                    # Shorter than average - moderate boost
                    density_factor[i, j] = density_factor[j, i] = 1.2
                elif density_ratio > 1.5:
                    # Much longer than average - penalty
                    density_factor[i, j] = density_factor[j, i] = 0.7
    
    # 5. MST-based centrality features
    # Compute minimum spanning tree
    mst = minimum_spanning_tree(distance_matrix)
    mst_adj = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Compute MST degree centrality
    mst_degree = np.sum(mst_adj, axis=1)
    if np.max(mst_degree) > 0:
        mst_degree_norm = mst_degree / np.max(mst_degree)
    else:
        mst_degree_norm = np.ones(n)
    
    # Create centrality factor matrix
    centrality_factor = np.ones((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            # Boost edges connecting high-degree MST nodes
            centrality = (mst_degree_norm[i] + mst_degree_norm[j]) / 2
            centrality_factor[i, j] = centrality_factor[j, i] = 0.8 + 0.4 * centrality
    
    # Additional boost for edges actually in the MST
    mst_boost = np.where(mst_adj, 1.3, 1.0)
    
    # 6. Mutual nearest neighbor boosting
    mutual_nn_boost = np.ones((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            if i in sorted_indices[j, :min(3, k)] and j in sorted_indices[i, :min(3, k)]:
                mutual_nn_boost[i, j] = mutual_nn_boost[j, i] = 1.4
    
    # 7. Combine all factors
    heuristics = inv_dist * efficiency_score * density_factor * centrality_factor * mst_boost * mutual_nn_boost
    
    # 8. Intelligent sparsification
    # Create adjacency mask from mutual k-NN
    adjacency_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        # Include k-nearest neighbors
        adjacency_mask[i, sorted_indices[i]] = True
        # Include nodes that have i as one of their top 3 nearest neighbors
        for j in range(n):
            if i != j and i in sorted_indices[j, :min(3, k)]:
                adjacency_mask[i, j] = True
    
    # Also include MST edges
    adjacency_mask = adjacency_mask | mst_adj
    
    # Symmetrize the mask
    adjacency_mask = adjacency_mask | adjacency_mask.T
    
    # Zero out non-promising edges with adaptive thresholding
    for i in range(n):
        row = heuristics[i]
        keep_mask = adjacency_mask[i]
        
        # For edges not in mask, apply stricter threshold
        if np.any(~keep_mask):
            non_masked_vals = row[~keep_mask & (row > 0)]
            if len(non_masked_vals) > 0:
                # Adaptive threshold based on local edge quality
                masked_median = np.median(row[keep_mask]) if np.any(keep_mask) else 0
                threshold = max(np.median(non_masked_vals) * 0.5, masked_median * 0.3)
                low_value_mask = ~keep_mask & (row < threshold)
                row[low_value_mask] = 0
        
        heuristics[i] = row
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # 9. Final normalization to [0, 1] range with softmax-style contrast enhancement
    non_zero = heuristics > 0
    if np.any(non_zero):
        max_val = np.max(heuristics[non_zero])
        if max_val > 0:
            # Apply softmax-style normalization for better contrast
            exp_heuristics = np.exp(heuristics / max_val * 3)
            row_sums = np.sum(exp_heuristics, axis=1, keepdims=True) + 1e-12
            heuristics = exp_heuristics / row_sums
            np.fill_diagonal(heuristics, 0)
    
    return heuristics
