import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray,
                  mst_weight: float = 2.0,
                  regret_power: float = 1.5,
                  lookahead_neighbors: int = 5,
                  sparsity_adaptive: bool = True) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization with adaptive lookahead.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        mst_weight: Multiplicative bonus for edges in Minimum Spanning Tree
        regret_power: Power for regret-based penalization (higher = more aggressive penalization of suboptimal edges)
        lookahead_neighbors: Number of nearest neighbors to consider for two-hop lookahead
        sparsity_adaptive: Whether to adaptively sparsify edges based on local connectivity
    
    Returns:
        Heuristic matrix of same shape as input
    """
    n = distance_matrix.shape[0]
    
    # Base inverse distance heuristic
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 1. Compute Minimum Spanning Tree using Prim's algorithm
    mst_mask = np.zeros((n, n), dtype=bool)
    
    # Prim's algorithm implementation
    visited = np.zeros(n, dtype=bool)
    min_edge = np.full(n, np.inf)
    parent = np.full(n, -1, dtype=int)
    
    min_edge[0] = 0
    
    for _ in range(n):
        u = -1
        min_val = np.inf
        for i in range(n):
            if not visited[i] and min_edge[i] < min_val:
                min_val = min_edge[i]
                u = i
        
        if u == -1:
            break
            
        visited[u] = True
        
        if parent[u] != -1:
            mst_mask[u, parent[u]] = True
            mst_mask[parent[u], u] = True
        
        for v in range(n):
            if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                min_edge[v] = distance_matrix[u, v]
                parent[v] = u
    
    # 2. Enhanced regret-based penalization with adaptive factors
    regret_penalty = np.ones((n, n))
    
    for i in range(n):
        # Get distances to all other nodes
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        valid_indices = np.where(mask)[0]
        
        if len(valid_indices) >= 2:
            dist_to_neighbors = distance_matrix[i, valid_indices]
            sorted_indices = np.argsort(dist_to_neighbors)
            
            # Base case: best edge gets no penalty
            best_idx = valid_indices[sorted_indices[0]]
            regret_penalty[i, best_idx] = 1.0
            
            # Apply progressive penalty based on rank
            for rank, idx in enumerate(valid_indices[sorted_indices[1:]], start=1):
                # Calculate normalized rank-based penalty
                normalized_rank = rank / (len(valid_indices) - 1)
                
                # Calculate distance ratio penalty
                best_dist = distance_matrix[i, best_idx]
                current_dist = distance_matrix[i, idx]
                dist_ratio = current_dist / max(best_dist, 1e-12)
                
                # Combined penalty using power law
                penalty = 1 / (1 + regret_power * normalized_rank * dist_ratio)
                regret_penalty[i, idx] = penalty
    
    # Make symmetric (geometric mean)
    regret_penalty = np.sqrt(regret_penalty * regret_penalty.T)
    
    # 3. Two-hop lookahead efficiency score
    lookahead_score = np.ones((n, n))
    
    if n >= 3 and lookahead_neighbors > 0:
        # Precompute nearest neighbors for each node
        nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:lookahead_neighbors+1]
        
        for i in range(n):
            for j in range(i+1, n):
                # Find best intermediate node from i's and j's nearest neighbors
                candidates = set(nearest_indices[i].tolist() + nearest_indices[j].tolist())
                candidates.discard(i)
                candidates.discard(j)
                
                if candidates:
                    # Calculate two-hop distances through candidates
                    two_hop_dists = []
                    for k in candidates:
                        two_hop = distance_matrix[i, k] + distance_matrix[k, j]
                        two_hop_dists.append(two_hop)
                    
                    if two_hop_dists:
                        min_two_hop = min(two_hop_dists)
                        direct_dist = distance_matrix[i, j]
                        
                        # Score based on efficiency of direct edge vs best two-hop
                        if direct_dist > 0:
                            efficiency = min_two_hop / direct_dist
                            # Convert to score: higher is better (clamp between 0.1 and 2.0)
                            score = np.clip(efficiency, 0.1, 2.0)
                            lookahead_score[i, j] = lookahead_score[j, i] = score
    
    # 4. Combine factors
    heuristics = inv_dist.copy()
    
    # Apply MST bonus
    heuristics[mst_mask] *= mst_weight
    
    # Apply regret penalty
    heuristics *= regret_penalty
    
    # Apply lookahead efficiency
    heuristics *= lookahead_score
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    
    # 5. Adaptive sparsification
    if sparsity_adaptive and n > 2:
        # Calculate local connectivity score for each node
        connectivity = np.zeros(n)
        for i in range(n):
            # Sort heuristic values for this node
            sorted_heuristics = np.sort(heuristics[i])
            # Take mean of top 3 non-zero values as connectivity measure
            non_zero = sorted_heuristics[sorted_heuristics > 0]
            if len(non_zero) >= 3:
                connectivity[i] = np.mean(non_zero[-3:])
            elif len(non_zero) > 0:
                connectivity[i] = np.mean(non_zero)
        
        # Normalize connectivity scores
        if np.max(connectivity) > 0:
            connectivity = connectivity / np.max(connectivity)
        
        # Adaptive thresholding per node
        for i in range(n):
            row = heuristics[i].copy()
            
            # Keep MST edges regardless
            row[mst_mask[i]] = np.maximum(row[mst_mask[i]], 0)
            
            # For non-MST edges, apply adaptive threshold
            non_mst_mask = ~mst_mask[i]
            non_mst_mask[i] = False  # Exclude diagonal
            
            if np.any(non_mst_mask):
                non_mst_values = row[non_mst_mask]
                if len(non_mst_values) > 0:
                    # Threshold based on connectivity and value distribution
                    if connectivity[i] > 0.5:  # Well-connected node
                        threshold = np.percentile(non_mst_values[non_mst_values > 0], 75)
                    else:  # Poorly-connected node
                        threshold = np.percentile(non_mst_values[non_mst_values > 0], 50)
                    
                    zero_mask = non_mst_mask & (row < threshold)
                    row[zero_mask] = 0
            
            heuristics[i] = row
    
    # Final symmetry enforcement and diagonal zeroing
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    return heuristics

    heuristics = heuristics_v2(dist_matrix.copy())

    for i, j in [(0, 1), (0, 2), (1, 2), (3, 4), (5, 6)]:
        print(f"({i}, {j}): dist={dist_matrix[i, j]:.2f}, heuristic={heuristics[i, j]:.4f}")
    
    # Verify symmetry
    is_symmetric = np.allclose(heuristics, heuristics.T)
    print(f"\nMatrix is symmetric: {is_symmetric}")
