```python
import numpy as np
import warnings
from scipy.spatial.distance import pdist, squareform
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristic combining theoretical foundations with adaptive learning.
    
    Core Innovations:
    1. Enhanced Christofides baseline with improved matching approximation
    2. Two-tier path efficiency lookahead with adaptive depth control
    3. Dynamic component weighting via instance-aware clustering
    4. Connectivity-preserving sparsification with meta-heuristic selection
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (shape n x n)
        
    Returns:
        Heuristic matrix of same shape as input, with scores in [0, 1]
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    warnings.filterwarnings('ignore')
    
    # ===== 1. INSTANCE CHARACTERIZATION =====
    triu_indices = np.triu_indices(n, 1)
    distances = distance_matrix[triu_indices]
    dist_mean = np.mean(distances)
    dist_std = np.std(distances)
    dist_cv = dist_std / (dist_mean + 1e-12)
    
    # ===== 2. ENHANCED CHRISTOFIDES APPROXIMATION =====
    def compute_christofides_components():
        """Improved MST and matching computation with connectivity guarantees."""
        christofides_mask = np.zeros((n, n), dtype=bool)
        
        # Prim's algorithm for MST
        visited = np.zeros(n, dtype=bool)
        min_edge = np.full(n, np.inf)
        parent = np.full(n, -1, dtype=int)
        min_edge[0] = 0
        
        for _ in range(n):
            u = np.argmin(np.where(visited, np.inf, min_edge))
            visited[u] = True
            if parent[u] != -1:
                christofides_mask[u, parent[u]] = christofides_mask[parent[u], u] = True
            
            for v in range(n):
                if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                    min_edge[v] = distance_matrix[u, v]
                    parent[v] = u
        
        # Identify odd-degree vertices in MST
        mst_degrees = np.sum(christofides_mask, axis=1)
        odd_vertices = np.where(mst_degrees % 2 == 1)[0]
        
        # Improved greedy matching with lookahead
        if len(odd_vertices) >= 2:
            odd_distances = distance_matrix[np.ix_(odd_vertices, odd_vertices)]
            np.fill_diagonal(odd_distances, np.inf)
            
            matched = np.zeros(len(odd_vertices), dtype=bool)
            for idx in range(len(odd_vertices)):
                if not matched[idx]:
                    u = odd_vertices[idx]
                    remaining = np.where(~matched)[0]
                    if len(remaining) > 1:
                        # Consider both immediate and next-best matches
                        current_dists = odd_distances[idx, remaining]
                        v_idx = remaining[np.argmin(current_dists)]
                        
                        # Check if this match leaves good options for remaining vertices
                        if len(remaining) > 2:
                            # Look at second-best option
                            sorted_indices = np.argsort(current_dists)
                            if len(sorted_indices) > 1:
                                second_best_idx = remaining[sorted_indices[1]]
                                # Choose match that doesn't isolate remaining vertices
                                if odd_distances[v_idx, second_best_idx] < np.inf * 0.5:
                                    v = odd_vertices[v_idx]
                                    christofides_mask[u, v] = christofides_mask[v, u] = True
                                    matched[idx] = matched[v_idx] = True
        
        return christofides_mask
    
    christofides_mask = compute_christofides_components()
    
    # ===== 3. DYNAMIC COMPONENT WEIGHTING VIA INSTANCE CLUSTERING =====
    def compute_instance_clusters():
        """Quick hierarchical clustering for instance-aware weighting."""
        if n <= 10:
            return np.zeros(n, dtype=int), 1
        
        # Use distance-based clustering with adaptive threshold
        threshold = np.percentile(distances, 40)
        cluster_assignments = -np.ones(n, dtype=int)
        current_cluster = 0
        
        for i in range(n):
            if cluster_assignments[i] == -1:
                # Find nearby unassigned nodes
                nearby = np.where((distance_matrix[i] < threshold) & 
                                 (cluster_assignments == -1))[0]
                if len(nearby) > 0:
                    cluster_assignments[nearby] = current_cluster
                    current_cluster += 1
        
        # Handle unassigned nodes
        unassigned = cluster_assignments == -1
        if np.any(unassigned):
            for i in np.where(unassigned)[0]:
                # Assign to nearest cluster
                if current_cluster > 0:
                    assigned_idx = np.where(cluster_assignments != -1)[0]
                    nearest = assigned_idx[np.argmin(distance_matrix[i, assigned_idx])]
                    cluster_assignments[i] = cluster_assignments[nearest]
                else:
                    cluster_assignments[i] = current_cluster
                    current_cluster += 1
        
        n_clusters = len(np.unique(cluster_assignments))
        return cluster_assignments, n_clusters
    
    cluster_assignments, n_clusters = compute_instance_clusters()
    
    # Compute cluster statistics for weight adaptation
    intra_cluster_avg = 0
    inter_cluster_avg = 0
    for i in range(n):
        for j in range(i+1, n):
            if cluster_assignments[i] == cluster_assignments[j]:
                intra_cluster_avg += distance_matrix[i, j]
            else:
                inter_cluster_avg += distance_matrix[i, j]
    
    n_intra = np.sum(cluster_assignments.reshape(-1, 1) == cluster_assignments) - n
    n_inter = n * (n - 1) // 2 - n_intra
    
    intra_cluster_avg /= max(n_intra, 1)
    inter_cluster_avg /= max(n_inter, 1)
    cluster_ratio = intra_cluster_avg / (inter_cluster_avg + 1e-12)
    
    # Dynamic weight calculation based on clustering
    if n_clusters <= 2 or cluster_ratio > 0.8:
        # Few clusters or similar intra/inter distances
        weights = {
            'inverse': 1.0,
            'regret': 0.9,
            'lookahead': 1.2,
            'optimization': 1.1,
            'mutual_nn': 0.8
        }
    elif n_clusters > max(3, n // 10):
        # Many small clusters
        weights = {
            'inverse': 1.2,
            'regret': 1.0,
            'lookahead': 0.7,
            'optimization': 0.9,
            'mutual_nn': 1.1
        }
    else:
        # Moderate clustering
        weights = {
            'inverse': 1.0,
            'regret': 1.1,
            'lookahead': 1.0,
            'optimization': 1.0,
            'mutual_nn': 1.0
        }
    
    # Adjust based on coefficient of variation
    weights['regret'] *= (1.0 + 0.3 * np.tanh(dist_cv))
    weights['lookahead'] *= (1.0 - 0.2 * np.tanh(dist_cv))
    
    # ===== 4. BASE INVERSE DISTANCE =====
    scale = dist_mean if dist_mean > 0 else 1.0
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = scale / (distance_matrix + 0.15 * scale * np.eye(n))
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # ===== 5. ADAPTIVE REGRET-BASED PENALIZATION =====
    regret_penalty = np.ones((n, n))
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_idx = np.argsort(dists)
        best_dist = dists[sorted_idx[0]]
        
        if best_dist > 0:
            for rank, j in enumerate(sorted_idx):
                if rank == 0:
                    regret_penalty[i, j] = 1.0
                else:
                    norm_rank = rank / (n - 2)
                    dist_ratio = dists[j] / best_dist
                    penalty = 1 / (1 + 1.5 * norm_rank * np.sqrt(dist_ratio))
                    regret_penalty[i, j] = penalty
    
    regret_penalty = np.sqrt(regret_penalty * regret_penalty.T)
    
    # ===== 6. MUTUAL K-NN WITH CLUSTER AWARENESS =====
    k_nn = min(max(3, int(np.sqrt(n) + dist_cv * 2)), n - 1)
    nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:k_nn + 1]
    mutual_nn_bonus = np.ones((n, n))
    
    for i in range(n):
        for j in range(i + 1, n):
            if j in nearest_indices[i] and i in nearest_indices[j]:
                rank_i = np.where(nearest_indices[i] == j)[0][0]
                rank_j = np.where(nearest_indices[j] == i)[0][0]
                avg_rank = (rank_i + rank_j) / (2 * (k_nn - 1))
                
                # Adjust bonus based on clustering
                if cluster_assignments[i] == cluster_assignments[j]:
                    cluster_bonus = 1.3
                else:
                    cluster_bonus = 0.9
                
                bonus = 1.0 + 0.6 * (1.0 - avg_rank) * cluster_bonus
                mutual_nn_bonus[i, j] = mutual_nn_bonus[j, i] = bonus
    
    # ===== 7. TWO-TIER PATH EFFICIENCY LOOKAHEAD =====
    lookahead_score = np.ones((n, n))
    
    # Determine lookahead depth adaptively
    if n <= 15:
        lookahead_hops = 3
    elif n <= 40:
        lookahead_hops = 2
    else:
        lookahead_hops = 1
    
    if n >= 4 and lookahead_hops >= 2:
        lookahead_k = min(max(4, int(1.5 * np.sqrt(n) + dist_cv * 3)), n - 1)
        nearest_lookahead = np.argsort(distance_matrix, axis=1)[:, 1:lookahead_k + 1]
        
        for i in range(n):
            i_neighbors = set(nearest_lookahead[i].tolist())
            
            for j in range(i + 1, n):
                direct_dist = distance_matrix[i, j]
                if direct_dist <= 0:
                    continue
                
                j_neighbors = set(nearest_lookahead[j].tolist())
                best_alternative = np.inf
                
                # Tier 1: Direct 2-hop paths through common neighbors
                common_neighbors = i_neighbors.intersection(j_neighbors)
                for k in common_neighbors:
                    if k != i and k != j:
                        alt_path = distance_matrix[i, k] + distance_matrix[k, j]
                        best_alternative = min(best_alternative, alt_path)
                
                # Tier 2: Extended paths if direct ones are poor
                if lookahead_hops >= 3 and (best_alternative / direct_dist > 1.4):
                    # Consider 3-hop paths with limited search
                    top_i = list(i_neighbors)[:min(4, len(i_neighbors))]
                    top_j = list(j_neighbors)[:min(4, len(j_neighbors))]
                    
                    for k in top_i:
                        if k == j:
                            continue
                        k_neighbors = set(nearest_lookahead[k].tolist()) if k < n else set()
                        for l in top_j:
                            if l == i or l == k:
                                continue
                            # Check if l is in k's neighbors or k in l's neighbors
                            if l in k_neighbors or k in nearest_lookahead[l]:
                                alt_path = (distance_matrix[i, k] + 
                                           distance_matrix[k, l] + 
                                           distance_matrix[l, j])
                                best_alternative = min(best_alternative, alt_path)
                
                # Calculate efficiency score
                if best_alternative < np.inf:
                    efficiency = best_alternative / direct_dist
                    # Adaptive sigmoid based on instance characteristics
                    steepness = 2.5 - 0.5 * np.tanh(dist_cv)
                    score = 0.5 + 1.5 / (1.0 + np.exp(-steepness * (efficiency - 1.0)))
                    lookahead_score[i, j] = lookahead_score[j, i] = score
    
    # ===== 8. LOCAL OPTIMIZATION POTENTIAL WITH 3-OPT EXTENSION =====
    optimization_potential = np.ones((n, n))
    if n >= 5:
        neighbor_limit = min(8, max(4, int(np.sqrt(n))))
        
        for i in range(n):
            i_neighbors = np.argsort(distance_matrix[i])[1:neighbor_limit]
            
            for j in range(i + 1, n):
                current_dist = distance_matrix[i, j]
                if current_dist <= 0:
                    continue
                
                j_neighbors = np.argsort(distance_matrix[j])[1:neighbor_limit]
                best_gain = 0
                
                # 2-opt gains
                for k in i_neighbors:
                    if k == j:
                        continue
                    for l in j_neighbors:
                        if l == i or l == k:
                            continue
                        
                        gain = (current_dist + distance_matrix[k, l] - 
                               distance_matrix[i, k] - distance_matrix[j, l])
                        best_gain = max(best_gain, gain)
                
                # Limited 3-opt pattern checking
                if neighbor_limit >= 6 and n >= 8:
                    # Try star patterns
                    for k in i_neighbors[:4]:
                        if k == j:
                            continue
                        for l in j_neighbors[:4]:
                            if l == i or l == k:
                                continue
                            
                            # Find potential connector
                            common = set(i_neighbors).intersection(set(j_neighbors))
                            common.discard(i)
                            common.discard(j)
                            common.discard(k)
                            common.discard(l)
                            
                            if common:
                                m = list(common)[0]
                                gain_3opt = (current_dist + distance_matrix[k, m] + 
                                           distance_matrix[m, l] - distance_matrix[i, k] - 
                                           distance_matrix[j, l] - distance_matrix[k, l])
                                best_gain = max(best_gain, gain_3opt * 0.7)
                
                # Convert gain to optimization potential
                if best_gain > 0:
                    norm_gain = min(best_gain / dist_mean, 2.0)
                    potential = 1.0 + 0.5 * np.tanh(norm_gain)
                    optimization_potential[i, j] = optimization_potential[j, i] = potential
    
    # ===== 9. COMBINE COMPONENTS WITH DYNAMIC WEIGHTS =====
    heuristics = inv_dist.copy()
    heuristics *= regret_penalty ** weights['regret']
    heuristics *= lookahead_score ** weights['lookahead']
    heuristics *= mutual_nn_bonus ** weights['mutual_nn']
    heuristics *= optimization_potential ** weights['optimization']
    
    # Enhanced boost for Christofides edges
    heuristics[christofides_mask] *= (1.8 + 0.2 * np.tanh(n_clusters))
    
    # Ensure symmetry and zero diagonal
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # ===== 10. CONNECTIVITY-AWARE SPARSIFICATION =====
    if n > 3:
        # Compute local edge density for adaptive thresholding
        edge_density = np.zeros(n)
        top_percentile = min(max(0.3, 2.5 / np.sqrt(n)), 0.7)
        
        for i in range(n):
            row = heuristics[i].copy()
            row[i] = 0
            if np.sum(row > 0) > 0:
                threshold = np.percentile(row[row > 0], (1 - top_percentile) * 100)
                edge_density[i] = np.mean(row[row > threshold]) if np.any(row > threshold) else 0
        
        # Normalize density scores
        max_density = np.max(edge_density)
        if max_density > 0:
            edge_density /= max_density
        
        # Meta-heuristic edge selection
        for i in range(n):
            row = heuristics[i].copy()
            row[i] = 0
            
            # Always keep Christofides edges
            keep_mask = christofides_mask[i].copy()
            
            # Determine number of edges to keep adaptively
            if edge_density[i] > 0.6:
                keep_count = min(int(1.8 * np.sqrt(n)), n - 1)
            elif edge_density[i] < 0.3:
                keep_count = min(int(np.sqrt(n)) + 3, n - 1)
            else:
                keep_count = min(int(1.4 * np.sqrt(n)), n - 1)
            
            keep_count = max(keep_count, np.sum(keep_mask) + 2, 
                           min(4, n - 1))  # Minimum connectivity
            
            # Keep top edges, ensuring connectivity
            non_zero = row[row > 0]
            if len(non_zero) > keep_count:
                sorted_values = np.sort(non_zero)
                threshold = sorted_values[-keep_count]
                
                # Mark low-score edges for removal
                low_score = (row < threshold) & (~keep_mask)
                row[low_score] = 0
            
            # Ensure at least keep_count edges remain
            if np.sum(row > 0) < keep_count and np.sum(row > 0) < n - 1:
                # Add back some high-value edges
                zero_positions = np.where(row == 0)[0]
                zero_positions = zero_positions[zero_positions != i]
                if len(zero_positions) > 0:
                    # Look at original heuristic values for these positions
                    orig_values = heuristics[i, zero_positions]
                    if len(orig_values) > 0:
                        needed = keep_count - np.sum(row > 0)
                        top_zero = zero_positions[np.argsort(orig_values)[-needed:]]
                        row[top_zero] = heuristics[i, top_zero]
            
            heuristics[i] = row
        
        # Re-symmetrize
        heuristics = (heuristics + heuristics.T) / 2
    
    # ===== 11. FINAL NORMALIZATION AND VALIDATION =====
    np.fill_diagonal(heuristics, 0)
    
    # Ensure all values are non-negative
    heuristics = np.maximum(heuristics, 0)
    
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Final symmetry and zero diagonal guarantee
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # Clip to ensure [0, 1] range
    heuristics = np.clip(heuristics, 0, 1)
    
    return heuristics
```
