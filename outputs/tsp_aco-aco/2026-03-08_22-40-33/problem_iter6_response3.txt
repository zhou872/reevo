import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with adaptive lookahead, regret-based penalties,
    and instance-aware sparsification.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix of same shape as input, with higher values indicating
        more promising edges for inclusion in optimal tours.
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # === 1. INSTANCE ANALYSIS FOR ADAPTIVE PARAMETERS ===
    # Analyze instance characteristics to set adaptive parameters
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    dist_cv = dist_std / (dist_mean + 1e-12)  # Coefficient of variation
    
    # Instance density estimation: average ratio of 2nd to 1st nearest neighbor
    density_measure = 1.0
    if n > 10:
        sorted_dists = np.sort(distance_matrix + np.eye(n) * np.inf, axis=1)
        nn_ratio = np.mean(sorted_dists[:, 1] / (sorted_dists[:, 0] + 1e-12))
        density_measure = 1.0 / (nn_ratio + 1e-12)
    
    # Adaptive parameter tuning
    def adaptive_k() -> int:
        """Adapt number of nearest neighbors based on instance size and density."""
        base_k = max(3, min(int(np.sqrt(n)), n - 1))
        # Denser instances benefit from more exploration
        density_factor = 1.0 + 0.5 * np.tanh(density_measure - 1.0)
        return min(int(base_k * density_factor), n - 1)
    
    def adaptive_lookahead_depth() -> int:
        """Adapt lookahead depth based on instance characteristics."""
        # Larger, denser instances benefit from deeper lookahead
        size_factor = min(2, max(1, np.log10(n) / 2))
        density_factor = 1.0 + density_measure
        return min(3, int(size_factor * density_factor))
    
    k = adaptive_k()
    lookahead_depth = adaptive_lookahead_depth()
    
    # === 2. BASE INVERSE DISTANCE WEIGHTING ===
    # Use robust scaling that accounts for instance variance
    scale = np.median(upper_tri) + 0.5 * dist_std
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = scale / (distance_matrix + 0.1 * scale * np.eye(n))
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # === 3. EFFICIENT K-NEAREST NEIGHBORS COMPUTATION ===
    # Add tiny noise to break distance ties deterministically
    np.random.seed(42)  # For reproducibility
    noise = np.random.randn(n, n) * 1e-10 * distance_matrix.max()
    noisy_dist = distance_matrix + noise - noise.T  # Maintain symmetry
    
    # Get k nearest neighbors for each node
    sorted_indices = np.argsort(noisy_dist, axis=1)[:, 1:k+1]
    neighbor_distances = np.zeros((n, k))
    for i in range(n):
        neighbor_distances[i] = distance_matrix[i, sorted_indices[i]]
    
    # === 4. ADAPTIVE LOOKAHEAD EFFICIENCY SCORING ===
    def compute_lookahead_efficiency(i: int, j: int) -> float:
        """Compute multi-hop efficiency for edge (i,j) with adaptive depth."""
        if i == j:
            return 1.0
        
        direct_dist = distance_matrix[i, j]
        
        # Precompute neighbor sets for i and j
        i_neighbors = sorted_indices[i]
        j_neighbors = sorted_indices[j]
        
        best_alternative = np.inf
        
        # Lookahead through intermediate nodes
        if lookahead_depth >= 2:
            # Two-hop lookahead (i -> m -> j)
            for m_idx, m in enumerate(i_neighbors[:k//2]):  # Limit to top neighbors
                if m == j:
                    continue
                
                # Option 1: Direct connection from m to j
                alt1 = neighbor_distances[i, m_idx] + distance_matrix[m, j]
                best_alternative = min(best_alternative, alt1)
                
                # Option 2: Through m's nearest neighbor to j
                if lookahead_depth >= 3 and m_idx < k//3:
                    m_neighbors = sorted_indices[m]
                    for mm in m_neighbors[:2]:  # Top 2 neighbors of m
                        if mm == i or mm == j:
                            continue
                        alt2 = (neighbor_distances[i, m_idx] + 
                               distance_matrix[m, mm] + 
                               distance_matrix[mm, j])
                        best_alternative = min(best_alternative, alt2)
        
        # Also consider paths starting from j
        if lookahead_depth >= 2:
            for m_idx, m in enumerate(j_neighbors[:k//2]):
                if m == i:
                    continue
                
                alt1 = neighbor_distances[j, m_idx] + distance_matrix[m, i]
                best_alternative = min(best_alternative, alt1)
        
        # Handle case where no alternative found
        if best_alternative == np.inf or direct_dist == 0:
            return 1.0
        
        efficiency_ratio = best_alternative / direct_dist
        # Convert to score: lower ratio (better alternative) means worse direct edge
        if efficiency_ratio <= 1.0:
            # Better alternative exists - penalize
            return max(0.3, efficiency_ratio)  # Strong penalty
        else:
            # No better alternative - slight bonus for efficiency
            return min(1.2, 2.0 / efficiency_ratio)
    
    # Vectorized efficiency computation for better performance
    efficiency_scores = np.ones((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            score = compute_lookahead_efficiency(i, j)
            efficiency_scores[i, j] = efficiency_scores[j, i] = score
    
    # === 5. REGRET-BASED PENALTY ===
    def compute_regret_penalty(i: int, j: int) -> float:
        """Compute penalty based on regret (opportunity cost)."""
        if i == j:
            return 1.0
        
        direct_dist = distance_matrix[i, j]
        
        # Find best alternative edge from i (excluding j)
        i_neighbors = sorted_indices[i]
        i_alt_dist = float('inf')
        for neighbor in i_neighbors:
            if neighbor != j:
                i_alt_dist = neighbor_distances[i, np.where(sorted_indices[i] == neighbor)[0][0]]
                break
        
        # Find best alternative edge to j (excluding i)
        j_neighbors = sorted_indices[j]
        j_alt_dist = float('inf')
        for neighbor in j_neighbors:
            if neighbor != i:
                j_alt_dist = neighbor_distances[j, np.where(sorted_indices[j] == neighbor)[0][0]]
                break
        
        # Regret: how much worse is this edge compared to best alternatives?
        if i_alt_dist == float('inf') or j_alt_dist == float('inf'):
            return 1.0
        
        # Normalize regret by instance scale
        avg_alt = (i_alt_dist + j_alt_dist) / 2
        regret_ratio = (direct_dist - avg_alt) / (scale + 1e-12)
        
        # Convert to penalty: higher regret -> lower score
        penalty = np.exp(-2.0 * max(0, regret_ratio))
        return max(0.5, penalty)  # Cap minimum penalty
    
    # Compute regret penalties
    regret_penalties = np.ones((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            penalty = compute_regret_penalty(i, j)
            regret_penalties[i, j] = regret_penalties[j, i] = penalty
    
    # === 6. MUTUAL NEAREST NEIGHBOR BOOST ===
    mutual_nn_boost = np.ones((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            # Check if i and j are in each other's nearest neighbors
            i_in_j = j in sorted_indices[i, :min(3, k)]
            j_in_i = i in sorted_indices[j, :min(3, k)]
            
            if i_in_j and j_in_i:
                # Strong mutual connection
                mutual_nn_boost[i, j] = mutual_nn_boost[j, i] = 1.8
            elif i_in_j or j_in_i:
                # One-way connection
                mutual_nn_boost[i, j] = mutual_nn_boost[j, i] = 1.3
    
    # === 7. COMBINE SCORES WITH ADAPTIVE WEIGHTS ===
    # Adjust weights based on instance characteristics
    if dist_cv > 1.0:  # High variance: emphasize efficiency and regret more
        efficiency_weight = 1.2
        regret_weight = 1.1
        mutual_weight = 0.9
    else:  # More uniform: balance all factors
        efficiency_weight = 1.0
        regret_weight = 1.0
        mutual_weight = 1.0
    
    heuristics = (inv_dist * 
                 (efficiency_scores ** efficiency_weight) * 
                 (regret_penalties ** regret_weight) * 
                 (mutual_nn_boost ** mutual_weight))
    
    # === 8. TWO-STAGE SPARSIFICATION ===
    # Stage 1: Keep edges based on mutual k-NN and local density
    adjacency_mask = np.zeros((n, n), dtype=bool)
    
    # Include mutual nearest neighbors
    for i in range(n):
        # Include k nearest neighbors
        adjacency_mask[i, sorted_indices[i]] = True
        # Include nodes that have i in their top neighbors
        for j in range(n):
            if i != j and i in sorted_indices[j, :min(max(2, k//2), k)]:
                adjacency_mask[i, j] = True
    
    # Symmetrize
    adjacency_mask = adjacency_mask | adjacency_mask.T
    
    # Stage 2: Density-aware thresholding
    for i in range(n):
        row_heuristics = heuristics[i].copy()
        
        # Separate masked and unmasked edges
        masked_vals = row_heuristics[adjacency_mask[i]]
        unmasked_vals = row_heuristics[~adjacency_mask[i] & (np.arange(n) != i)]
        
        if len(masked_vals) > 0 and len(unmasked_vals) > 0:
            # Dynamic threshold based on local edge quality
            if density_measure > 1.2:  # Dense instance: be more selective
                threshold = np.percentile(unmasked_vals, 70)
            else:  # Sparse instance: be more inclusive
                threshold = np.percentile(unmasked_vals, 40)
            
            # Zero out low-scoring unmasked edges
            low_score_mask = ~adjacency_mask[i] & (row_heuristics < threshold) & (np.arange(n) != i)
            row_heuristics[low_score_mask] = 0
        
        heuristics[i] = row_heuristics
    
    # Ensure symmetry and zero diagonal
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # === 9. FINAL NORMALIZATION ===
    # Softmax-style normalization for better contrast
    non_zero_mask = heuristics > 0
    if np.any(non_zero_mask):
        # Scale to [0, 1] range
        min_val = heuristics[non_zero_mask].min()
        max_val = heuristics[non_zero_mask].max()
        if max_val > min_val:
            heuristics[non_zero_mask] = (heuristics[non_zero_mask] - min_val) / (max_val - min_val)
        
        # Apply softmax to rows for probabilistic interpretation
        for i in range(n):
            row = heuristics[i]
            if np.any(row > 0):
                exp_row = np.exp(row / (np.mean(row[row > 0]) + 1e-12))
                row_sum = np.sum(exp_row)
                if row_sum > 0:
                    heuristics[i] = exp_row / row_sum
    
    return heuristics
