import numpy as np
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray, 
                  mst_weight: float = 2.0,
                  regret_factor: float = 1.5,
                  sparsity_factor: float = 0.7) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        mst_weight: Multiplicative bonus for edges in Minimum Spanning Tree
        regret_factor: Factor for regret-based penalty (higher = more penalty for second-best edges)
        sparsity_factor: Fraction of least promising non-MST edges to zero out (0-1)
    
    Returns:
        Heuristic matrix of same shape as input
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance heuristic (with diagonal protection)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Compute Minimum Spanning Tree using Prim's algorithm
    mst_mask = np.zeros((n, n), dtype=bool)
    
    # Prim's algorithm implementation
    visited = np.zeros(n, dtype=bool)
    min_edge = np.full(n, np.inf)
    parent = np.full(n, -1, dtype=int)
    
    # Start from node 0
    min_edge[0] = 0
    
    for _ in range(n):
        # Find unvisited node with minimum edge weight
        u = -1
        min_val = np.inf
        for i in range(n):
            if not visited[i] and min_edge[i] < min_val:
                min_val = min_edge[i]
                u = i
        
        if u == -1:
            break
            
        visited[u] = True
        
        # Add edge to MST if not the first node
        if parent[u] != -1:
            mst_mask[u, parent[u]] = True
            mst_mask[parent[u], u] = True
        
        # Update minimum edges for adjacent nodes
        for v in range(n):
            if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                min_edge[v] = distance_matrix[u, v]
                parent[v] = u
    
    # 3. Regret-based penalty: Penalize edges that are not the best from either endpoint
    regret_penalty = np.ones((n, n))
    
    for i in range(n):
        # Get best and second-best edges from node i (excluding self)
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        valid_indices = np.where(mask)[0]
        
        if len(valid_indices) >= 2:
            # Get distances to valid neighbors
            dist_to_neighbors = distance_matrix[i, valid_indices]
            sorted_indices = np.argsort(dist_to_neighbors)
            
            # Best edge gets no penalty
            best_idx = valid_indices[sorted_indices[0]]
            regret_penalty[i, best_idx] = 1.0
            
            # Second-best gets small penalty
            if len(sorted_indices) > 1:
                second_best_idx = valid_indices[sorted_indices[1]]
                regret_factor_i = 1 + (regret_factor - 1) * (
                    distance_matrix[i, second_best_idx] / 
                    max(distance_matrix[i, best_idx], 1e-12) - 1
                )
                regret_penalty[i, second_best_idx] = 1 / regret_factor_i
            
            # Worse edges get larger penalties
            for k, idx in enumerate(valid_indices[sorted_indices[2:]], start=2):
                penalty = 1 / (1 + regret_factor * (k - 1))
                regret_penalty[i, idx] = penalty
    
    # Make symmetric (take geometric mean)
    regret_penalty = np.sqrt(regret_penalty * regret_penalty.T)
    
    # 4. Combine factors
    heuristics = inv_dist.copy()
    
    # Apply MST bonus
    heuristics[mst_mask] *= mst_weight
    
    # Apply regret penalty
    heuristics *= regret_penalty
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    
    # 5. Sparsify non-MST edges
    if sparsity_factor > 0:
        # Get non-MST edges
        non_mst_mask = ~mst_mask
        np.fill_diagonal(non_mst_mask, False)
        
        # Get heuristic values for non-MST edges
        non_mst_values = heuristics[non_mst_mask]
        
        if len(non_mst_values) > 0:
            # Determine threshold for sparsification
            threshold = np.percentile(non_mst_values, sparsity_factor * 100)
            
            # Zero out values below threshold
            zero_mask = non_mst_mask & (heuristics < threshold)
            heuristics[zero_mask] = 0
    
    return heuristics
