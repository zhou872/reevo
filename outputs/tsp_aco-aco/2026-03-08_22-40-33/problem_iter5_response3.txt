import numpy as np

def heuristics_v2(distance_matrix: np.ndarray,
                  lookahead_depth: int = 3,
                  regret_adaptive: bool = True,
                  sparsity_adaptive: bool = True) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization with Christofides initialization,
    mutual K-NN scoring, and adaptive multi-hop lookahead.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (shape n x n)
        lookahead_depth: Maximum depth for multi-hop lookahead (2-4 recommended)
        regret_adaptive: Whether to adapt regret penalty based on distance distribution
        sparsity_adaptive: Whether to adaptively sparsify unpromising edges
        
    Returns:
        Heuristic matrix of same shape as input, with scores in [0, 1]
    """
    
    # === Helper Functions ===
    def _compute_mst_mask(dist_mat: np.ndarray) -> np.ndarray:
        """Compute Minimum Spanning Tree mask using Prim's algorithm."""
        n = dist_mat.shape[0]
        mst_mask = np.zeros((n, n), dtype=bool)
        visited = np.zeros(n, dtype=bool)
        min_edge = np.full(n, np.inf)
        parent = np.full(n, -1, dtype=int)
        min_edge[0] = 0
        
        for _ in range(n):
            # Find unvisited node with minimum edge
            u = -1
            min_val = np.inf
            for i in range(n):
                if not visited[i] and min_edge[i] < min_val:
                    min_val = min_edge[i]
                    u = i
            if u == -1:
                break
            visited[u] = True
            if parent[u] != -1:
                mst_mask[u, parent[u]] = mst_mask[parent[u], u] = True
            # Update min_edge for neighbors
            for v in range(n):
                if not visited[v] and dist_mat[u, v] < min_edge[v]:
                    min_edge[v] = dist_mat[u, v]
                    parent[v] = u
        return mst_mask
    
    def _christofides_edges(dist_mat: np.ndarray) -> np.ndarray:
        """Approximate Christofides algorithm to identify promising edges."""
        n = dist_mat.shape[0]
        if n < 4:
            return _compute_mst_mask(dist_mat)  # Fallback for small instances
        
        christofides_mask = np.zeros((n, n), dtype=bool)
        # 1. MST edges
        mst_mask = _compute_mst_mask(dist_mat)
        christofides_mask |= mst_mask
        # 2. Identify odd-degree vertices in MST
        mst_degrees = np.sum(mst_mask, axis=1)
        odd_vertices = np.where(mst_degrees % 2 == 1)[0]
        # 3. Simplified minimum weight matching (greedy)
        if len(odd_vertices) >= 2:
            odd_distances = dist_mat[np.ix_(odd_vertices, odd_vertices)]
            avg_distances = np.mean(odd_distances, axis=1)
            sorted_indices = np.argsort(avg_distances)
            sorted_odd = odd_vertices[sorted_indices]
            matched = np.zeros(len(sorted_odd), dtype=bool)
            for i in range(len(sorted_odd)):
                if not matched[i]:
                    u = sorted_odd[i]
                    best_j, best_dist = -1, np.inf
                    for j in range(i + 1, len(sorted_odd)):
                        if not matched[j]:
                            v = sorted_odd[j]
                            dist = dist_mat[u, v]
                            if dist < best_dist:
                                best_dist = dist
                                best_j = j
                    if best_j != -1:
                        v = sorted_odd[best_j]
                        christofides_mask[u, v] = christofides_mask[v, u] = True
                        matched[i] = matched[best_j] = True
        return christofides_mask
    
    # === Main Function Logic ===
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)  # Simple fallback
    
    # 1. Instance characterization for adaptive parameters
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    dist_cv = dist_std / (dist_mean + 1e-12)
    if regret_adaptive:
        regret_power = 1.5 + 0.8 * np.tanh(2 * dist_cv)
    else:
        regret_power = 1.5
    mst_weight = 2.0 + 0.5 * np.tanh(dist_cv - 1.0)
    
    # 2. Christofides and MST initialization
    christofides_mask = _christofides_edges(distance_matrix)
    mst_mask = _compute_mst_mask(distance_matrix)
    
    # 3. Base inverse distance with adaptive scaling
    scale = dist_mean if dist_mean > 0 else 1.0
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = scale / (distance_matrix + 0.15 * scale * np.eye(n))
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 4. Adaptive regret-based penalization
    regret_penalty = np.ones((n, n))
    for i in range(n):
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        candidates = np.where(mask)[0]
        if len(candidates) >= 2:
            dists = distance_matrix[i, candidates]
            sorted_idx = np.argsort(dists)
            best_dist = dists[sorted_idx[0]]
            for rank, pos in enumerate(sorted_idx):
                j = candidates[pos]
                if rank == 0:
                    regret_penalty[i, j] = 1.0
                else:
                    norm_rank = rank / (len(candidates) - 1)
                    dist_ratio = dists[pos] / (best_dist + 1e-12)
                    penalty = 1 / (1 + regret_power * norm_rank * np.sqrt(dist_ratio))
                    regret_penalty[i, j] = penalty
    regret_penalty = np.sqrt(regret_penalty * regret_penalty.T)  # Symmetrize
    
    # 5. Mutual K-NN proximity bonus
    k_nn = min(max(3, int(np.sqrt(n))), n - 1)  # Adaptive neighbor count
    nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:k_nn + 1]
    mutual_nn_bonus = np.ones((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            if j in nearest_indices[i] and i in nearest_indices[j]:
                # Both in each other's K-NN: higher bonus for closer ranks
                rank_i = np.where(nearest_indices[i] == j)[0][0]
                rank_j = np.where(nearest_indices[j] == i)[0][0]
                avg_rank = (rank_i + rank_j) / (2 * (k_nn - 1))
                bonus = 1.0 + 0.5 * (1.0 - avg_rank)  # 1.0 to 1.5 range
                mutual_nn_bonus[i, j] = mutual_nn_bonus[j, i] = bonus
    
    # 6. Enhanced multi-hop lookahead efficiency scoring
    lookahead_score = np.ones((n, n))
    if n >= 4 and lookahead_depth >= 2:
        neighbor_count = min(max(4, int(np.sqrt(n))), n - 1)
        nearest_indices_lookahead = np.argsort(distance_matrix, axis=1)[:, 1:neighbor_count + 1]
        for i in range(n):
            for j in range(i + 1, n):
                best_alternative = np.inf
                # 2-hop lookahead through common neighbors
                candidates_2hop = set(nearest_indices_lookahead[i].tolist() + nearest_indices_lookahead[j].tolist())
                candidates_2hop.discard(i)
                candidates_2hop.discard(j)
                for k in candidates_2hop:
                    alt_path = distance_matrix[i, k] + distance_matrix[k, j]
                    best_alternative = min(best_alternative, alt_path)
                # 3-hop lookahead if enabled
                if lookahead_depth >= 3 and candidates_2hop:
                    for k in candidates_2hop:
                        k_neighbors = set(nearest_indices_lookahead[k].tolist())
                        k_neighbors.discard(i); k_neighbors.discard(j); k_neighbors.discard(k)
                        for l in k_neighbors:
                            if l not in candidates_2hop:
                                alt_path = distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j]
                                best_alternative = min(best_alternative, alt_path)
                # Calculate efficiency score
                if best_alternative < np.inf:
                    direct_dist = distance_matrix[i, j]
                    if direct_dist > 0:
                        efficiency = best_alternative / direct_dist
                        score = np.clip(efficiency, 0.5, 2.0)  # Higher means direct edge is better
                        lookahead_score[i, j] = lookahead_score[j, i] = score
    
    # 7. Combine all factors
    heuristics = inv_dist.copy()
    heuristics *= regret_penalty
    heuristics *= lookahead_score
    heuristics *= mutual_nn_bonus  # Apply mutual K-NN bonus
    
    # Boost edges from Christofides and MST
    heuristics[christofides_mask] *= 1.8
    heuristics[mst_mask] *= mst_weight
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # 8. Adaptive sparsification
    if sparsity_adaptive and n > 3:
        # Estimate local connectivity per node
        connectivity = np.zeros(n)
        for i in range(n):
            row_vals = heuristics[i]
            non_zero = row_vals[row_vals > 0]
            if len(non_zero) > 0:
                connectivity[i] = np.mean(np.sort(non_zero)[-min(4, len(non_zero)):])
        if np.max(connectivity) > 0:
            connectivity /= np.max(connectivity)
        # Apply thresholding
        for i in range(n):
            row = heuristics[i].copy()
            keep_mask = christofides_mask[i] | mst_mask[i]
            row[keep_mask] = np.maximum(row[keep_mask], 0)
            other_mask = ~keep_mask
            other_mask[i] = False  # Exclude diagonal
            if np.any(other_mask):
                other_vals = row[other_mask]
                if len(other_vals) > 0 and np.any(other_vals > 0):
                    # Adaptive percentile threshold
                    if connectivity[i] > 0.7:
                        percentile = 80
                    elif connectivity[i] < 0.3:
                        percentile = 60
                    else:
                        percentile = 70
                    size_factor = min(1.0, 20 / n)
                    percentile *= size_factor
                    threshold = np.percentile(other_vals[other_vals > 0], percentile)
                    zero_mask = other_mask & (row < threshold)
                    row[zero_mask] = 0
            heuristics[i] = row
        heuristics = (heuristics + heuristics.T) / 2  # Re-symmetrize
    
    # 9. Final normalization to [0, 1]
    np.fill_diagonal(heuristics, 0)
    non_zero_mask = heuristics > 0
    if np.any(non_zero_mask):
        max_val = heuristics[non_zero_mask].max()
        if max_val > 0:
            heuristics[non_zero_mask] /= max_val
    
    return heuristics
