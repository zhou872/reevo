import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization with improved
    rank-based weighting and adaptive sparsification.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix of same shape as input, normalized to [0,1]
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 2:
        return np.zeros_like(distance_matrix)
    if n == 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result
    
    # === 1. INSTANCE CHARACTERIZATION ===
    # Use instance properties to adapt parameters
    distances = distance_matrix[~np.eye(n, dtype=bool)]
    if len(distances) > 0:
        median_dist = np.median(distances)
        mean_dist = np.mean(distances)
        sparsity_ratio = median_dist / (mean_dist + 1e-12)
        
        # Dynamic K selection based on instance properties
        min_k = max(3, n // 10)
        max_k = min(15, n // 3)
        k = int(min_k + (max_k - min_k) * (1 - min(sparsity_ratio, 1.5) / 1.5))
        k = max(min_k, min(max_k, k, n - 1))
    else:
        k = min(8, n - 1)
    
    # === 2. BASE INVERSE DISTANCE ===
    with np.errstate(divide='ignore', invalid='ignore'):
        eps = np.percentile(distances, 5) * 0.01 if len(distances) > 0 else 1e-12
        inv_dist = 1.0 / (distance_matrix + eps * np.eye(n))
    
    np.fill_diagonal(inv_dist, 0)
    max_inv = np.max(inv_dist[~np.eye(n, dtype=bool)], initial=0)
    if max_inv > 0:
        inv_dist = inv_dist / max_inv
    
    # === 3. IMPROVED RANK-BASED WEIGHTING ===
    # Compute rank positions using argsort
    sorted_indices = np.argsort(distance_matrix, axis=1)
    positions = np.argsort(sorted_indices, axis=1)
    
    # Create rank-based weight matrix
    rank_weights = np.ones((n, n))
    rank_penalty = 0.25  # Reduced from 0.3 for smoother decay
    
    # Vectorized mutual proximity calculation
    for i in range(n):
        for j in range(i + 1, n):
            rank_i_j = positions[i, j]
            rank_j_i = positions[j, i]
            
            # Mutual proximity enhancement
            if rank_i_j < k and rank_j_i < k:
                # Both in top-k: enhanced weight
                mutual_score = (k - rank_i_j) * (k - rank_j_i)
                max_mutual = k * k
                weight = 1.0 + 2.0 * (mutual_score / max_mutual)
            elif rank_i_j < k or rank_j_i < k:
                # One-way proximity: moderate enhancement
                min_rank = min(rank_i_j, rank_j_i)
                if min_rank < k:
                    weight = 1.0 + (k - min_rank) / (2.0 * k)
                else:
                    weight = 1.0
            else:
                weight = 1.0
            
            # Apply rank-based penalty
            penalty_i = 1.0 / (1.0 + rank_penalty * rank_i_j)
            penalty_j = 1.0 / (1.0 + rank_penalty * rank_j_i)
            combined_penalty = np.sqrt(penalty_i * penalty_j)
            
            rank_weights[i, j] = rank_weights[j, i] = weight * combined_penalty
    
    # === 4. STRUCTURE-AWARE SPARSIFICATION ===
    # First pass: identify structural edges
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Identify mutual nearest neighbors
    mutual_knn = np.zeros((n, n), dtype=bool)
    for i in range(n):
        i_top_k = sorted_indices[i, 1:k+2]  # Include extra for robustness
        for neighbor in i_top_k:
            if i in sorted_indices[neighbor, 1:k+2]:
                mutual_knn[i, neighbor] = True
    
    # Enhanced edge boosting
    boost_factor = np.ones((n, n))
    boost_factor[mst_mask] *= 1.4  # MST edges get stronger boost
    boost_factor[mutual_knn] *= 1.6  # Mutual neighbors get highest boost
    
    # === 5. ADAPTIVE EDGE SCORING ===
    # Combine components with boosting
    heuristic_scores = inv_dist * rank_weights * boost_factor
    np.fill_diagonal(heuristic_scores, 0)
    
    # === 6. DYNAMIC SPARSIFICATION ===
    # Two-stage sparsification: global and local
    heuristics_final = heuristic_scores.copy()
    
    # Stage 1: Global percentile threshold
    global_threshold = np.percentile(heuristic_scores[~np.eye(n, dtype=bool)], 70)
    weak_edges_global = (heuristic_scores < global_threshold) & ~mutual_knn & ~mst_mask
    
    # Stage 2: Local connectivity preservation
    for i in range(n):
        row_values = heuristic_scores[i].copy()
        row_values[i] = -np.inf
        
        # Compute connectivity metrics
        non_zero = row_values > 0
        if np.sum(non_zero) > 0:
            # Keep top connections based on instance size
            keep_count = min(max(k * 2, n // 4), n - 1)
            threshold = np.partition(row_values, -keep_count)[-keep_count]
            
            # Preserve edges above threshold or structural edges
            mask = (row_values >= threshold) | mutual_knn[i] | mst_mask[i]
        else:
            mask = mutual_knn[i] | mst_mask[i]
        
        # Also keep edges that are globally strong
        mask = mask | ~weak_edges_global[i]
        
        # Apply local sparsification
        heuristics_final[i, ~mask] = 0
    
    # Ensure symmetry while preserving strong directional edges
    symmetric_scores = np.maximum(heuristics_final, heuristics_final.T)
    
    # Keep edges that are strong in at least one direction
    directional_strength = (heuristics_final > 0) | (heuristics_final.T > 0)
    final_heuristics = np.where(directional_strength, symmetric_scores, 0)
    
    # === 7. FINAL REFINEMENT ===
    # Soft normalization with outlier handling
    non_zero_mask = final_heuristics > 0
    if np.any(non_zero_mask):
        # Use robust scaling (median/IQR) to handle outliers
        non_zero_vals = final_heuristics[non_zero_mask]
        q75, q25 = np.percentile(non_zero_vals, [75, 25])
        iqr = q75 - q25
        
        if iqr > 0:
            # Robust scaling: (value - median) / IQR + 0.5
            median_val = np.median(non_zero_vals)
            scaled = (final_heuristics - median_val) / iqr + 0.5
            # Clip to [0, 1] range
            final_heuristics = np.clip(scaled, 0, 1)
        else:
            # Uniform scaling if IQR is too small
            max_val = np.max(non_zero_vals)
            if max_val > 0:
                final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    
    return final_heuristics

    # Test with different instance types
    for instance_type in ["cluster", "uniform", "mixed"]:
        if instance_type == "cluster":
            n = 30

            coords[:20] += np.array([30, 30])
            coords[20:] += np.array([70, 70])
        elif instance_type == "uniform":
            n = 30
            coords = np.random.rand(n, 2) * 100
        else:  # mixed
            n = 30
            coords = np.zeros((n, 2))
            coords[:10] = np.random.randn(10, 2) * 5 + [20, 20]
            coords[10:20] = np.random.rand(10, 2) * 60 + [40, 40]
            coords[20:] = np.random.rand(10, 2) * 100
        
        dist_matrix = squareform(pdist(coords))
        heuristics = heuristics_v2(dist_matrix.copy())
        
        print(f"\n=== {instance_type.upper()} Instance (n={n}) ===")
        print(f"Sparsity: {(heuristics == 0).sum() / heuristics.size:.1%}")
        
        edges_per_node = np.sum(heuristics > 0, axis=1)
        print(f"Edges per node: min={edges_per_node.min()}, "
              f"max={edges_per_node.max()}, mean={edges_per_node.mean():.1f}")
        
        is_symmetric = np.allclose(heuristics, heuristics.T)
        print(f"Matrix symmetric: {is_symmetric}")
        
        # Sample scores
        print("Sample edge scores (i, j, distance, heuristic):")
        for i, j in [(0, 1), (0, 10), (5, 15), (10, 25)]:
            if i < n and j < n:
                print(f"({i:2d}, {j:2d}): dist={dist_matrix[i, j]:6.2f}, "
                      f"score={heuristics[i, j]:.4f}")
