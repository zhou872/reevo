```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    Returns a sparse matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (handle zeros on diagonal)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree indicator
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_indicator = (mst + mst.T > 0).astype(float)  # Symmetric indicator
    
    # 3. Node degree in MST (for shortage calculation)
    mst_degree = mst_indicator.sum(axis=1)
    
    # 4. Node centrality (average distance to other nodes)
    avg_distances = distance_matrix.sum(axis=1) / (n - 1)
    centrality = 1 / (avg_distances + 1e-8)  # Higher for central nodes
    
    # 5. Combined scoring
    scores = np.zeros((n, n))
    
    for i in range(n):
        for j in range(i + 1, n):
            if distance_matrix[i, j] == 0:
                continue
                
            # Base inverse distance
            base_score = inv_dist[i, j]
            
            # MST bonus
            mst_bonus = 2.0 if mst_indicator[i, j] > 0 else 1.0
            
            # Connection shortage factor
            # Nodes with lower MST degree need connections more urgently
            shortage_i = max(0, 2 - mst_degree[i])
            shortage_j = max(0, 2 - mst_degree[j])
            shortage_factor = 1 + 0.3 * (shortage_i + shortage_j)
            
            # Centrality weighting
            # Favors edges connecting central nodes or connecting
            # peripheral nodes to central ones
            centrality_weight = 0.5 * (centrality[i] + centrality[j]) / centrality.mean()
            
            # Composite score
            score = (base_score * mst_bonus * 
                     shortage_factor * centrality_weight)
            
            scores[i, j] = scores[j, i] = score
    
    # 6. Sparsification: keep only top promising edges
    # Use 75th percentile as threshold (keep top 25% of non-zero edges)
    non_zero = scores[scores > 0]
    if len(non_zero) > 0:
        threshold = np.percentile(non_zero, 75)
        scores[scores < threshold] = 0
    
    # Ensure diagonal is zero
    np.fill_diagonal(scores, 0)
    
    return scores
```
