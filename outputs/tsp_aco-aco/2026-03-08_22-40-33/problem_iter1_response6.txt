import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection.
    Combines multiple factors to identify promising edges and sparsifies the matrix.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base reciprocal of distance (like v1)
    with np.errstate(divide='ignore'):
        reciprocal = 1 / distance_matrix
    np.fill_diagonal(reciprocal, 0)
    
    # 2. Node degree heuristic - penalize edges from/to nodes with few good alternatives
    # For each node, compute how many "short" edges it has
    threshold_percentile = 25  # Consider edges in the 25th percentile as "short"
    threshold = np.percentile(distance_matrix[distance_matrix > 0], threshold_percentile)
    short_edges_mask = distance_matrix < threshold
    node_degree = np.sum(short_edges_mask, axis=1) - 1  # Subtract self
    
    # Normalize node degree importance
    degree_factor = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Penalize if either node has few good alternatives
                min_degree = min(node_degree[i], node_degree[j])
                degree_factor[i, j] = np.exp(-(2.0 / (min_degree + 1)))  # +1 to avoid division by zero
    
    # 3. Two-step path affinity
    # For edge (i,j), consider if there's a good intermediate node k that makes a good triangle
    path_affinity = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Find best intermediate node (not i or j)
                other_nodes = [k for k in range(n) if k not in (i, j)]
                if other_nodes:
                    # Look for triangles i-k-j with reasonable distances
                    best_affinity = 0
                    for k in other_nodes:
                        triangle_score = 1/(distance_matrix[i, k] + distance_matrix[k, j] + 1e-12)
                        best_affinity = max(best_affinity, triangle_score)
                    path_affinity[i, j] = best_affinity
    
    # 4. Combine factors with weights
    heuristics_matrix = (
        0.5 * reciprocal +          # Base reciprocal distance
        0.2 * degree_factor +       # Node degree consideration
        0.3 * path_affinity         # Two-step path affinity
    )
    
    # 5. Adaptive sparsification
    # Keep only the most promising edges for each node
    sparsified = np.zeros_like(heuristics_matrix)
    keep_per_node = max(3, n // 4)  # Keep at least 3 edges per node, up to n/4
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Ignore self
        # Get indices of top keep_per_node edges
        top_indices = np.argsort(row)[-keep_per_node:]
        # Only keep edges that are in both nodes' top lists (symmetric consideration)
        for j in top_indices:
            if i != j:
                # Check if edge (j,i) is also in j's top list
                j_row = heuristics_matrix[j].copy()
                j_row[j] = -np.inf
                j_top = np.argsort(j_row)[-keep_per_node:]
                if i in j_top:
                    # Mutual preference - keep this edge
                    sparsified[i, j] = heuristics_matrix[i, j]
    
    # 6. Final normalization
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
