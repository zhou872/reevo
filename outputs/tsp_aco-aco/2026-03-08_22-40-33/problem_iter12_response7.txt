import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import squareform, pdist


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristic with dynamic factor weighting, reciprocal proximity,
    adaptive regret scoring, and efficient critical edge preservation.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix of same shape as input, normalized to [0,1]
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 2:
        return np.zeros_like(distance_matrix)
    if n == 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result
    
    # === 1. INSTANCE ANALYSIS & DYNAMIC WEIGHTING ===
    triu_mask = ~np.eye(n, dtype=bool)
    distances = distance_matrix[triu_mask]
    
    if len(distances) > 0:
        mean_dist = np.mean(distances)
        std_dist = np.std(distances)
        median_dist = np.median(distances)
        q1, q3 = np.percentile(distances, [25, 75])
        
        # Density factor: higher means more uniformly distributed
        density_factor = std_dist / (mean_dist + 1e-12)
        
        # Adaptive parameters based on instance characteristics
        base_k = max(2, int(np.sqrt(n)))
        if density_factor > 1.5:  # Uniform distribution
            k = min(base_k, n // 4)
            lookahead = 1
            cluster_weight = 0.7
            regret_weight = 1.3
        elif density_factor < 0.5:  # Clustered distribution
            k = min(base_k * 2, n // 3)
            lookahead = 2
            cluster_weight = 1.5
            regret_weight = 1.1
        else:  # Moderate distribution
            k = min(base_k + 2, n // 3)
            lookahead = 1 if n < 30 else 2
            cluster_weight = 1.2
            regret_weight = 1.2
    else:
        k = min(8, n - 1)
        lookahead = 1
        cluster_weight = 1.0
        regret_weight = 1.0
    
    # === 2. BASE DISTANCE WEIGHTING ===
    with np.errstate(divide='ignore', invalid='ignore'):
        eps = np.percentile(distances, 5) * 0.01 if len(distances) > 0 else 1e-12
        inv_dist = 1.0 / (distance_matrix + eps * np.eye(n))
    
    np.fill_diagonal(inv_dist, 0)
    max_inv = np.max(inv_dist[triu_mask], initial=0)
    if max_inv > 0:
        inv_dist = inv_dist / max_inv
    
    # === 3. RECIPROCAL RANK WEIGHTING ===
    sorted_indices = np.argsort(distance_matrix, axis=1)
    positions = np.argsort(sorted_indices, axis=1)  # Position matrix
    
    reciprocal_score = np.ones((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            rank_i_j = positions[i, j]
            rank_j_i = positions[j, i]
            
            # Harmonic weighting: emphasizes mutual high rankings
            if rank_i_j < k and rank_j_i < k:  # Mutual top-k
                # Stronger boost for better mutual ranks
                weight = 2.0 * k / (rank_i_j + rank_j_i + 1)
                reciprocal_score[i, j] = reciprocal_score[j, i] = 1.0 + weight / k
            elif rank_i_j < k or rank_j_i < k:  # One-way top-k
                min_rank = min(rank_i_j, rank_j_i)
                reciprocal_score[i, j] = reciprocal_score[j, i] = 1.0 + 0.5 / (min_rank + 1)
    
    # === 4. ADAPTIVE REGRET SCORING ===
    regret_score = np.ones((n, n))
    
    # Precompute k-nearest neighbors for efficient lookahead
    knn_indices = sorted_indices[:, 1:k+1]
    
    for i in range(n):
        for j in range(i + 1, n):
            direct_cost = distance_matrix[i, j]
            
            # One-step regret through neighbors
            best_alternative = float('inf')
            
            # Check i's neighbors
            for neighbor in knn_indices[i]:
                if neighbor != j:
                    alt_cost = distance_matrix[i, neighbor] + distance_matrix[neighbor, j]
                    best_alternative = min(best_alternative, alt_cost)
            
            # Check j's neighbors
            for neighbor in knn_indices[j]:
                if neighbor != i:
                    alt_cost = distance_matrix[j, neighbor] + distance_matrix[neighbor, i]
                    best_alternative = min(best_alternative, alt_cost)
            
            # Two-step lookahead if enabled
            if lookahead == 2 and k >= 3:
                for neighbor_i in knn_indices[i][:min(3, k)]:
                    if neighbor_i != j:
                        for neighbor_j in knn_indices[neighbor_i][:min(2, k)]:
                            if neighbor_j != i and neighbor_j != j:
                                alt_cost = (distance_matrix[i, neighbor_i] +
                                          distance_matrix[neighbor_i, neighbor_j] +
                                          distance_matrix[neighbor_j, j])
                                best_alternative = min(best_alternative, alt_cost)
            
            # Dynamic regret scoring based on instance density
            if best_alternative < float('inf') and direct_cost > 0:
                regret_ratio = best_alternative / direct_cost
                
                # More aggressive penalty for clustered instances
                if density_factor < 0.5:  # Clustered
                    if regret_ratio > 1.2:
                        score = 0.7 / (regret_ratio - 0.2)
                    else:
                        score = 1.5 - 0.5 * regret_ratio
                else:  # Uniform or moderate
                    if regret_ratio > 1.5:
                        score = 1.0 / regret_ratio
                    else:
                        score = 1.2 - 0.2 * regret_ratio
                
                regret_score[i, j] = regret_score[j, i] = np.clip(score, 0.5, 1.5)
    
    # === 5. EFFICIENT CLUSTER ANALYSIS ===
    cluster_score = np.ones((n, n))
    
    if n >= 10:
        # Quick clustering using distance thresholds
        intra_threshold = median_dist * 1.2
        inter_threshold = q3 * 1.5
        
        # Simple assignment: nodes closer than intra_threshold are in same "region"
        for i in range(n):
            for j in range(i + 1, n):
                dist = distance_matrix[i, j]
                
                if dist < intra_threshold:
                    # Strong intra-cluster preference
                    cluster_score[i, j] = cluster_score[j, i] = 1.4 - 0.4 * (dist / intra_threshold)
                elif dist < inter_threshold:
                    # Moderate score for reasonable inter-cluster connections
                    cluster_score[i, j] = cluster_score[j, i] = 1.1 - 0.1 * (dist / inter_threshold)
                else:
                    # Penalize very long edges
                    cluster_score[i, j] = cluster_score[j, i] = 0.8
    
    # === 6. DYNAMIC FACTOR COMBINATION ===
    # Adjust weights based on instance size
    size_factor = np.clip(n / 50, 0.5, 2.0)  # Normalize around n=50
    
    # Combine factors with dynamic weighting
    heuristics = inv_dist * (
        reciprocal_score ** (1.0 / size_factor) *
        regret_score ** (regret_weight / size_factor) *
        cluster_score ** (cluster_weight)
    )
    
    # === 7. CRITICAL EDGE ENHANCEMENT ===
    # Preserve MST edges
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    heuristics[mst_mask] *= 1.6
    
    # Boost mutual top-k edges (symmetric top neighbors)
    mutual_top_k = (positions < k) & (positions.T < k)
    heuristics[mutual_top_k] *= 1.4
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # === 8. ADAPTIVE SPARSIFICATION ===
    # Determine target edges per node based on instance size and density
    base_target = max(3, int(np.sqrt(n)))
    if density_factor < 0.5:  # Clustered: need more connectivity
        target_edges = min(base_target + 2, n // 2)
    else:
        target_edges = base_target
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -1  # Exclude diagonal
        
        # Always preserve MST and mutual top-k edges
        preserve_mask = mst_mask[i] | mutual_top_k[i]
        
        # If we need more edges, add top scorers
        if np.sum(preserve_mask) < target_edges:
            # Get non-preserved candidates
            candidates = np.where(~preserve_mask & (np.arange(n) != i))[0]
            if len(candidates) > 0:
                candidate_scores = row[candidates]
                # Select enough to reach target
                n_needed = target_edges - np.sum(preserve_mask)
                if n_needed > 0:
                    top_indices = np.argsort(candidate_scores)[-n_needed:]
                    for idx in top_indices:
                        preserve_mask[candidates[idx]] = True
        
        # Zero out non-preserved edges
        non_preserved = np.where(~preserve_mask & (np.arange(n) != i))[0]
        heuristics[i, non_preserved] = 0
    
    # Ensure symmetry after pruning
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # === 9. FINAL NORMALIZATION ===
    non_zero = heuristics[triu_mask]
    if len(non_zero) > 0 and np.max(non_zero) > 0:
        max_val = np.max(non_zero)
        heuristics = heuristics / max_val
    
    np.fill_diagonal(heuristics, 0)
    
    return heuristics

    np.random.seed(42)
    
    # Test with different instance types
    for instance_type in ["cluster", "uniform", "mixed"]:
        if instance_type == "cluster":
            n = 50
            coords = np.random.randn(n, 2) * 5
            coords[:35] += np.array([30, 30])
            coords[35:] += np.array([70, 70])
        elif instance_type == "uniform":
            n = 50
            coords = np.random.rand(n, 2) * 100
        else:  # mixed
            n = 50
            coords = np.zeros((n, 2))
            coords[:20] = np.random.randn(20, 2) * 5 + [20, 20]
            coords[20:40] = np.random.rand(20, 2) * 60 + [40, 40]
            coords[40:] = np.random.rand(10, 2) * 100
        
        dist_matrix = squareform(pdist(coords))
        heuristics = heuristics_v2(dist_matrix)
        
        print(f"\n=== {instance_type.upper()} Instance (n={n}) ===")
        print(f"Sparsity: {(heuristics == 0).sum() / heuristics.size:.1%}")
        
        edges_per_node = np.sum(heuristics > 0, axis=1)
        print(f"Edges per node: min={edges_per_node.min()}, "
              f"max={edges_per_node.max()}, mean={edges_per_node.mean():.1f}")
        
        # Verify symmetry
        is_symmetric = np.allclose(heuristics, heuristics.T)
        print(f"Symmetric: {is_symmetric}")
        
        # Check diagonal
        diag_zero = np.allclose(np.diag(heuristics), 0)
        print(f"Diagonal zero: {diag_zero}")
        
        # Sample scores
        print("Sample edge scores:")
