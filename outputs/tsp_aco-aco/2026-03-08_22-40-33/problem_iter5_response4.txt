import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray,
                  lookahead_depth: int = 2,
                  regret_adaptive: bool = True,
                  sparsity_adaptive: bool = True) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        lookahead_depth: Maximum depth for multi-hop lookahead (2 recommended)
        regret_adaptive: Whether to adapt regret penalty based on distance distribution
        sparsity_adaptive: Whether to adaptively sparsify edges
        
    Returns:
        Heuristic matrix of same shape as input
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # === 1. INSTANCE CHARACTERIZATION ===
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    dist_cv = dist_std / (dist_mean + 1e-12)
    
    # Adaptive parameters
    if regret_adaptive:
        regret_power = 1.5 + 0.8 * np.tanh(2 * dist_cv)
    else:
        regret_power = 1.5
    
    # === 2. BASE INVERSE DISTANCE ===
    scale = dist_mean if dist_mean > 0 else 1.0
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = scale / (distance_matrix + 0.15 * scale * np.eye(n))
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # === 3. ADAPTIVE REGRET-BASED PENALTY ===
    regret_penalty = np.ones((n, n))
    
    for i in range(n):
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        candidates = np.where(mask)[0]
        
        if len(candidates) >= 2:
            dists = distance_matrix[i, candidates]
            sorted_idx = np.argsort(dists)
            best_dist = dists[sorted_idx[0]]
            
            for rank, pos in enumerate(sorted_idx):
                j = candidates[pos]
                if rank == 0:
                    regret_penalty[i, j] = 1.0
                else:
                    norm_rank = rank / (len(candidates) - 1)
                    dist_ratio = dists[pos] / (best_dist + 1e-12)
                    penalty = 1 / (1 + regret_power * norm_rank * np.sqrt(dist_ratio))
                    regret_penalty[i, j] = penalty
    
    regret_penalty = np.sqrt(regret_penalty * regret_penalty.T)
    
    # === 4. MUTUAL K-NN WITH TWO-HOP EFFICIENCY ===
    # Determine k for nearest neighbors
    k = min(max(3, int(np.sqrt(n))), n - 1)
    
    # Get k-nearest neighbors for each node
    nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:k+1]
    
    # Create mutual K-NN mask
    mutual_knn = np.zeros((n, n), dtype=bool)
    for i in range(n):
        for j in range(i+1, n):
            if (j in nearest_indices[i]) and (i in nearest_indices[j]):
                mutual_knn[i, j] = mutual_knn[j, i] = True
    
    # Calculate two-hop efficiency score
    efficiency_score = np.ones((n, n))
    
    if n >= 4 and lookahead_depth >= 2:
        for i in range(n):
            for j in range(i+1, n):
                best_alternative = np.inf
                
                # Get candidate intermediate nodes
                candidates = set(nearest_indices[i].tolist() + nearest_indices[j].tolist())
                candidates.discard(i)
                candidates.discard(j)
                
                if candidates:
                    for k_cand in candidates:
                        alt_path = distance_matrix[i, k_cand] + distance_matrix[k_cand, j]
                        best_alternative = min(best_alternative, alt_path)
                    
                    direct_dist = distance_matrix[i, j]
                    if direct_dist > 0:
                        efficiency = best_alternative / direct_dist
                        # Convert to score: higher means direct edge is more efficient
                        score = np.clip(efficiency, 0.5, 2.0)
                        efficiency_score[i, j] = efficiency_score[j, i] = score
    
    # === 5. COMBINE FACTORS ===
    heuristics = inv_dist.copy()
    heuristics *= regret_penalty
    heuristics *= efficiency_score
    
    # Boost mutual K-NN edges
    heuristics[mutual_knn] *= 1.5
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    
    # === 6. ADAPTIVE SPARSIFICATION ===
    if sparsity_adaptive and n > 3:
        # Calculate sparsity threshold based on instance characteristics
        density_target = min(0.3, 20/n)  # Target non-zero density
        
        for i in range(n):
            row = heuristics[i].copy()
            np.fill_diagonal(row, 0)  # Ensure diagonal is zero
            
            # Keep mutual K-NN edges
            mutual_mask = mutual_knn[i]
            row[mutual_mask] = np.maximum(row[mutual_mask], 0)
            
            # Determine threshold for other edges
            other_mask = ~mutual_mask
            other_mask[i] = False
            
            if np.any(other_mask):
                other_vals = row[other_mask]
                if len(other_vals) > 0:
                    # Sort and keep top edges based on density target
                    sorted_idx = np.argsort(other_vals)[::-1]
                    keep_count = int(np.ceil(density_target * (n-1)))
                    keep_count = max(2, min(keep_count, np.sum(other_vals > 0)))
                    
                    if keep_count < len(sorted_idx):
                        threshold = other_vals[sorted_idx[keep_count]]
                        zero_mask = other_mask & (row < threshold)
                        row[zero_mask] = 0
            
            heuristics[i] = row
        
        # Final symmetrization
        heuristics = (heuristics + heuristics.T) / 2
    
    # === 7. FINAL PROCESSING ===
    np.fill_diagonal(heuristics, 0)
    
    # Normalize non-zero values to [0, 1] range
    non_zero_mask = heuristics > 0
    if np.any(non_zero_mask):
        max_val = heuristics[non_zero_mask].max()
        if max_val > 0:
            heuristics[non_zero_mask] = heuristics[non_zero_mask] / max_val
    
    return heuristics
