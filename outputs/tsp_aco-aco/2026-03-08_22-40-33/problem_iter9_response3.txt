import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Optimized heuristic function for TSP edge prioritization using reciprocal
    proximity weighting with lightweight rank-based regret and adaptive sparsification.
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix of same shape as input, normalized to [0,1]
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 2:
        return np.zeros_like(distance_matrix)
    if n == 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result
    
    # === 1. ADAPTIVE PARAMETER SELECTION ===
    # Use simpler parameter selection for efficiency
    distances = distance_matrix[~np.eye(n, dtype=bool)]
    if len(distances) > 0:
        # Dynamic K selection based on instance size
        k = max(2, min(8, n // 4))
        # Regret decay parameter (lightweight version)
        regret_alpha = 0.5
    else:
        k = min(5, n - 1)
        regret_alpha = 0.5
    
    # === 2. RECIPROCAL PROXIMITY WEIGHTING ===
    # Sort distances and get reciprocal rankings
    sorted_indices = np.argsort(distance_matrix, axis=1)
    positions = np.argsort(sorted_indices, axis=1)
    
    # Reciprocal proximity score: 1/(1+rank) from both ends
    reciprocal_proximity = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            rank_i_j = positions[i, j]
            rank_j_i = positions[j, i]
            # Reciprocal weighting: boost edges where both rank each other highly
            if rank_i_j < k and rank_j_i < k:
                # Both in top-k: strong mutual preference
                score = 2.0 * k / (k + rank_i_j + rank_j_i)
            elif rank_i_j < k or rank_j_i < k:
                # One-way preference: moderate boost
                min_rank = min(rank_i_j, rank_j_i)
                score = 1.0 + (k - min_rank) / (2 * k)
            else:
                # No strong preference: base score
                score = 1.0
            reciprocal_proximity[i, j] = reciprocal_proximity[j, i] = score
    
    # === 3. LIGHTWEIGHT RANK-BASED REGRET ===
    # Apply penalty based on rank: 1/(1 + α*rank)
    regret_penalty = np.ones((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            rank_i_j = positions[i, j]
            rank_j_i = positions[j, i]
            # Average rank penalty (symmetric)
            avg_rank = (rank_i_j + rank_j_i) / 2
            penalty = 1.0 / (1.0 + regret_alpha * avg_rank)
            regret_penalty[i, j] = regret_penalty[j, i] = penalty
    
    # === 4. DISTANCE SCALING ===
    # Use inverse distance with better numerical stability
    with np.errstate(divide='ignore', invalid='ignore'):
        # Add small epsilon to avoid division by zero
        eps = np.percentile(distances, 5) * 0.01 if len(distances) > 0 else 1e-12
        # Scale distances for better inverse mapping
        scaled_dist = distance_matrix / (np.median(distances) + 1e-12)
        inv_dist = 1.0 / (scaled_dist + eps * np.eye(n))
    
    np.fill_diagonal(inv_dist, 0)
    # Normalize to [0,1]
    max_inv = np.max(inv_dist[~np.eye(n, dtype=bool)], initial=0)
    if max_inv > 0:
        inv_dist = inv_dist / max_inv
    
    # === 5. COMBINE FACTORS ===
    heuristics = inv_dist * reciprocal_proximity * regret_penalty
    np.fill_diagonal(heuristics, 0)
    
    # === 6. CONNECTIVITY-AWARE SPARSIFICATION ===
    # Identify key edges to preserve
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Mutual K-NN edges
    knn_indices = np.argsort(distance_matrix, axis=1)[:, 1:k+1]
    knn_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        knn_mask[i, knn_indices[i]] = True
    mutual_knn = knn_mask & knn_mask.T
    
    # Boost important edges
    heuristics[mst_mask] *= 1.5
    heuristics[mutual_knn] *= 1.3
    
    # === 7. ADAPTIVE EDGE PRUNING ===
    # Determine edges to keep per node based on connectivity
    heuristics_final = heuristics.copy()
    
    for i in range(n):
        row_values = heuristics[i].copy()
        row_values[i] = -np.inf
        
        # Always preserve MST and mutual K-NN edges for this node
        preserve_mask = mst_mask[i] | mutual_knn[i]
        
        # Determine connectivity score for adaptive threshold
        non_zero_vals = row_values[row_values > 0]
        if len(non_zero_vals) > 0:
            # Nodes with lower average scores get more edges
            avg_score = np.mean(non_zero_vals)
            target_edges = max(k, min(int(0.4 * n), int(0.6 * n * avg_score)))
        else:
            target_edges = max(k, int(0.3 * n))
        
        # Count how many edges we need to keep
        current_edges = np.sum(row_values > 0)
        if current_edges > target_edges:
            # Sort non-preserved edges
            non_preserved = row_values.copy()
            non_preserved[preserve_mask] = -np.inf
            
            # Keep top edges among non-preserved
            if np.sum(non_preserved > -np.inf) > 0:
                # Calculate how many additional edges to keep (beyond preserved ones)
                preserved_count = np.sum(preserve_mask)
                additional_to_keep = max(0, target_edges - preserved_count)
                
                if additional_to_keep > 0:
                    # Find threshold for top additional edges
                    sorted_vals = np.sort(non_preserved[non_preserved > -np.inf])
                    if len(sorted_vals) >= additional_to_keep:
                        threshold = sorted_vals[-additional_to_keep]
                        # Zero out edges below threshold (excluding preserved ones)
                        zero_mask = (row_values < threshold) & (~preserve_mask)
                        heuristics_final[i, zero_mask] = 0
        
        # Ensure at least k edges per node (including preserved ones)
        kept_edges = np.sum(heuristics_final[i] > 0)
        if kept_edges < k:
            # Add top edges from original scores to reach k
            row_copy = heuristics[i].copy()
            row_copy[i] = -np.inf
            row_copy[heuristics_final[i] > 0] = -np.inf  # Already kept
            
            if np.any(row_copy > -np.inf):
                # Find top (k - kept_edges) edges
                needed = k - kept_edges
                sorted_idx = np.argsort(row_copy)[::-1][:needed]
                heuristics_final[i, sorted_idx] = heuristics[i, sorted_idx]
    
    # Ensure symmetry
    symmetric_heuristics = np.maximum(heuristics_final, heuristics_final.T)
    np.fill_diagonal(symmetric_heuristics, 0)
    
    # === 8. FINAL NORMALIZATION ===
    non_zero_mask = symmetric_heuristics > 0
    if np.any(non_zero_mask):
        max_val = np.max(symmetric_heuristics[non_zero_mask])
        if max_val > 0:
            symmetric_heuristics = symmetric_heuristics / max_val
    
    return symmetric_heuristics

    np.random.seed(42)
    
    # Test with different instance types
    for instance_type in ["cluster", "uniform", "mixed"]:
        if instance_type == "cluster":
            n = 30
            coords = np.random.randn(n, 2) * 5
            coords[:20] += np.array([30, 30])
            coords[20:] += np.array([70, 70])
        elif instance_type == "uniform":
            n = 30
            coords = np.random.rand(n, 2) * 100
        else:  # mixed
            n = 30
            coords = np.zeros((n, 2))
            coords[:10] = np.random.randn(10, 2) * 5 + [20, 20]
            coords[10:20] = np.random.rand(10, 2) * 60 + [40, 40]
            coords[20:] = np.random.rand(10, 2) * 100
        
        dist_matrix = squareform(pdist(coords))
        heuristics = heuristics_v2(dist_matrix.copy())
        
        print(f"\n=== {instance_type.upper()} Instance (n={n}) ===")
        print(f"Sparsity: {(heuristics == 0).sum() / heuristics.size:.1%}")
        
        edges_per_node = np.sum(heuristics > 0, axis=1)
        print(f"Edges per node: min={edges_per_node.min()}, "
              f"max={edges_per_node.max()}, mean={edges_per_node.mean():.1f}")
        
        is_symmetric = np.allclose(heuristics, heuristics.T)
        print(f"Matrix symmetric: {is_symmetric}")
        
        # Sample scores
        print("Sample edge scores (i, j, distance, heuristic):")
        for i, j in [(0, 1), (0, 10), (5, 15), (10, 25)]:
            if i < n and j < n:
                print(f"({i:2d}, {j:2d}): dist={dist_matrix[i, j]:6.2f}, "
                      f"score={heuristics[i, j]:.4f}")
