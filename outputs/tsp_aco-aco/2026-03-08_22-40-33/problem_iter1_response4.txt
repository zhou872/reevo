import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection probability.
    Combines multiple factors to score edges and sparsifies unpromising ones.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) / (distance_matrix + 1e-12)
    
    # 1. Base inverse distance (handles zeros)
    inv_dist = 1 / (distance_matrix + 1e-12)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Mutual nearest neighbor bonus
    k = min(5, n - 1)  # Consider top k nearest neighbors
    mutual_bonus = np.zeros_like(distance_matrix)
    for i in range(n):
        # Get k-nearest neighbors for node i
        sorted_idx = np.argsort(distance_matrix[i])[1:k+1]
        mutual_bonus[i, sorted_idx] = 1
    mutual_bonus = (mutual_bonus + mutual_bonus.T) / 2  # Symmetrize
    
    # 3. Neighborhood density score
    density_score = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            # Distances to other nodes from i and j
            dist_i = distance_matrix[i]
            dist_j = distance_matrix[j]
            # Average distance to common neighbors (excluding i,j themselves)
            mask = ~np.isin(np.arange(n), [i, j])
            avg_dist_i = np.mean(dist_i[mask]) if np.any(mask) else 0
            avg_dist_j = np.mean(dist_j[mask]) if np.any(mask) else 0
            # Nodes in dense regions should connect more strongly
            density_penalty = (avg_dist_i + avg_dist_j) / 2
            density_score[i, j] = density_score[j, i] = 1 / (density_penalty + 1e-12)
    
    # 4. Rank-based exponential decay
    rank_score = np.zeros_like(distance_matrix)
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])[1:]  # Exclude self
        sorted_distances = distance_matrix[i, sorted_indices]
        # Exponential decay: exp(-rank/λ)
        ranks = np.arange(1, len(sorted_indices)+1)
        lambda_decay = n / 4  # Controls decay rate
        decay_weights = np.exp(-ranks / lambda_decay)
        # Normalize weights
        if np.sum(decay_weights) > 0:
            decay_weights = decay_weights / np.sum(decay_weights)
        rank_score[i, sorted_indices] = decay_weights
    rank_score = (rank_score + rank_score.T) / 2  # Symmetrize
    
    # 5. Combine factors with weighted geometric mean
    w1, w2, w3, w4 = 2.0, 1.0, 0.5, 1.5  # Weights for each factor
    combined = (
        np.power(inv_dist, w1) *
        np.power(mutual_bonus + 0.1, w2) *  # Add small constant to avoid zero
        np.power(density_score, w3) *
        np.power(rank_score + 0.05, w4)  # Add small constant
    )
    np.fill_diagonal(combined, 0)
    
    # 6. Sparsification: keep top percentile of edges
    if n > 3:
        # Get upper triangular values
        upper_tri = combined[np.triu_indices(n, k=1)]
        threshold = np.percentile(upper_tri, 70)  # Keep top 30% of edges
        mask = combined < threshold
        combined[mask] = 0
    
    # 7. Normalize to prevent extreme values
    max_val = np.max(combined)
    if max_val > 0:
        combined = combined / max_val
    
    return combined
