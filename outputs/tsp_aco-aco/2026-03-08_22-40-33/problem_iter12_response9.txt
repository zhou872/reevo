import numpy as np
from scipy.spatial.distance import pdist, squareform
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization.
    
    Key improvements:
    1. Spectral clustering for robust instance characterization
    2. Three-stage adaptive regret modeling with cluster-aware exploration
    3. Cost-benefit optimized multi-hop lookahead
    4. Density-aware sparsification with connectivity guarantees
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (n x n)
        
    Returns:
        Heuristic matrix of same shape as input, normalized to [0,1]
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 2:
        return np.zeros_like(distance_matrix)
    if n == 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result
    
    # === 1. INSTANCE CHARACTERIZATION WITH SPECTRAL CLUSTERING ===
    distances = distance_matrix[~np.eye(n, dtype=bool)]
    median_dist = np.median(distances) if len(distances) > 0 else 1.0
    mean_dist = np.mean(distances) if len(distances) > 0 else 1.0
    dist_cv = np.std(distances) / (mean_dist + 1e-12)
    
    # Compute clustering structure using spectral methods
    clustering_info = _compute_spectral_clustering_info(distance_matrix, n)
    cluster_labels = clustering_info['labels']
    n_clusters = clustering_info['n_clusters']
    cluster_sizes = clustering_info['sizes']
    
    # === 2. ADAPTIVE PARAMETER TUNING ===
    # Determine base K for nearest neighbors
    if n_clusters > 1:
        avg_cluster_size = np.mean(cluster_sizes)
        base_k = int(np.clip(3 + np.log2(avg_cluster_size), 3, 10))
    else:
        base_k = int(np.clip(np.sqrt(n), 3, 15))
    
    # Regret decay parameter based on clustering strength
    if n_clusters > 1:
        intra_inter_ratio = clustering_info['intra_inter_ratio']
        clustering_strength = np.clip(1.0 - intra_inter_ratio, 0.1, 0.9)
    else:
        clustering_strength = 0.3  # Default for uniform instances
    
    regret_decay = 0.4 + 0.4 * clustering_strength
    
    # === 3. BASE INVERSE DISTANCE WITH ADAPTIVE SMOOTHING ===
    with np.errstate(divide='ignore', invalid='ignore'):
        eps = median_dist * 0.01
        inv_dist = 1.0 / (distance_matrix + eps * np.eye(n))
    
    np.fill_diagonal(inv_dist, 0)
    max_inv = np.max(inv_dist[~np.eye(n, dtype=bool)], initial=1.0)
    inv_dist /= max_inv
    
    # === 4. THREE-STAGE ADAPTIVE REGRET MODELING ===
    regret_scores = np.ones((n, n))
    
    # Precompute K-nearest neighbors
    k = min(base_k, n - 1)
    knn_indices = np.argsort(distance_matrix, axis=1)[:, 1:k+1]
    knn_distances = np.take_along_axis(distance_matrix, knn_indices, axis=1)
    
    # Compute local density for each node
    local_density = np.mean(knn_distances, axis=1)
    density_normalized = local_density / (np.max(local_density) + 1e-12)
    
    # Build neighbor sets for efficient lookahead
    neighbor_sets = [set(knn_indices[i]) for i in range(n)]
    
    # Compute MST for connectivity preservation
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Stage 1: Fast rank-based regret for all edges
    positions = np.argsort(np.argsort(distance_matrix, axis=1), axis=1)
    
    for i in range(n):
        for j in range(i + 1, n):
            # Rank-based component
            rank_i = positions[i, j] / (n - 1)
            rank_j = positions[j, i] / (n - 1)
            avg_rank = (rank_i + rank_j) / 2
            
            # Density-aware rank penalty
            density_factor = (density_normalized[i] + density_normalized[j]) / 2
            rank_penalty = np.exp(-2.5 * avg_rank * (1 + 0.5 * density_factor))
            
            regret_scores[i, j] = regret_scores[j, i] = rank_penalty
    
    # Stage 2: Cluster-aware two-hop regret for promising edges
    mutual_knn = np.zeros((n, n), dtype=bool)
    for i in range(n):
        mutual_knn[i, knn_indices[i]] = True
    mutual_knn = mutual_knn & mutual_knn.T
    
    # Stage 3: Deep exploration for critical edges (MST edges and cluster bridges)
    critical_edges = mst_mask | (mutual_knn & (cluster_labels[:, None] != cluster_labels))
    
    for i in range(n):
        for j in range(i + 1, n):
            if not critical_edges[i, j]:
                continue
                
            direct_dist = distance_matrix[i, j]
            if direct_dist <= 0:
                continue
            
            # Determine exploration depth adaptively
            same_cluster = cluster_labels[i] == cluster_labels[j]
            if same_cluster and n_clusters > 1:
                # Deeper exploration within clusters
                cluster_size = cluster_sizes[cluster_labels[i]]
                max_depth = min(3, max(2, int(np.log2(cluster_size))))
            else:
                # Shallow exploration for inter-cluster edges
                max_depth = 2
            
            best_alternative = np.inf
            
            # Multi-hop exploration with cluster constraints
            for depth in range(2, max_depth + 1):
                if depth == 2:
                    # Two-hop paths through common neighbors
                    common = neighbor_sets[i].intersection(neighbor_sets[j])
                    for m in common:
                        if m != i and m != j:
                            alt = distance_matrix[i, m] + distance_matrix[m, j]
                            best_alternative = min(best_alternative, alt)
                else:
                    # Three-hop paths with limited search
                    # Explore through best connections of i and j
                    i_best = list(neighbor_sets[i])[:min(3, len(neighbor_sets[i]))]
                    j_best = list(neighbor_sets[j])[:min(3, len(neighbor_sets[j]))]
                    
                    for a in i_best:
                        if a == j:
                            continue
                        for b in j_best:
                            if b == i or b == a:
                                continue
                            # Check if a and b are connected
                            if b in neighbor_sets[a]:
                                alt = (distance_matrix[i, a] + 
                                       distance_matrix[a, b] + 
                                       distance_matrix[b, j])
                                best_alternative = min(best_alternative, alt)
            
            # Compute regret score if alternative found
            if best_alternative < np.inf:
                regret_ratio = best_alternative / direct_dist
                
                # Cluster-aware regret calculation
                if same_cluster:
                    # Within clusters: be more selective
                    threshold = 1.2
                    penalty_strength = 0.8
                else:
                    # Between clusters: more tolerant
                    threshold = 1.4
                    penalty_strength = 0.6
                
                if regret_ratio > threshold:
                    regret = 1.0 - penalty_strength * (1 - np.exp(-(regret_ratio - threshold)))
                else:
                    regret = 1.0 + (1 - regret_ratio) * 0.3
                
                # Combine with rank-based component
                current_score = regret_scores[i, j]
                combined = 0.7 * regret + 0.3 * current_score
                regret_scores[i, j] = regret_scores[j, i] = np.clip(combined, 0.1, 2.0)
    
    # === 5. DENSITY-AWARE SPARSIFICATION ===
    # Combine base scores
    heuristics_base = inv_dist * regret_scores
    
    # Boost important structural edges
    boost_mutual = 1.5 + 0.3 * clustering_strength
    boost_mst = 1.4 + 0.2 * clustering_strength
    
    heuristics_base[mutual_knn] *= boost_mutual
    heuristics_base[mst_mask] *= boost_mst
    
    # Ensure symmetry
    heuristics_base = (heuristics_base + heuristics_base.T) / 2
    np.fill_diagonal(heuristics_base, 0)
    
    # Adaptive sparsification
    target_edges_per_node = int(np.clip(0.4 * np.sqrt(n), 3, 15))
    
    heuristics_final = heuristics_base.copy()
    
    for i in range(n):
        row = heuristics_base[i].copy()
        row[i] = -np.inf
        
        # Determine node-specific target
        node_density = density_normalized[i]
        if node_density < 0.3:  # Sparse area
            target = int(target_edges_per_node * 1.3)
        elif node_density > 0.7:  # Dense area
            target = int(target_edges_per_node * 0.7)
        else:
            target = target_edges_per_node
        
        # Always preserve MST and mutual KNN edges
        preserve_mask = mst_mask[i] | mutual_knn[i]
        
        if np.sum(row > 0) > target:
            # Separate preserved and candidate edges
            candidates = row.copy()
            candidates[preserve_mask] = -np.inf
            
            # Keep top candidates to reach target
            preserved_count = np.sum(preserve_mask)
            additional_needed = max(0, target - preserved_count)
            
            if additional_needed > 0 and np.any(candidates > -np.inf):
                threshold = np.partition(candidates, -additional_needed)[-additional_needed]
                remove_mask = (row < threshold) & (~preserve_mask)
                heuristics_final[i, remove_mask] = 0
            else:
                # Only keep preserved edges
                remove_mask = (row > -np.inf) & (~preserve_mask)
                heuristics_final[i, remove_mask] = 0
    
    # Ensure symmetry after pruning
    heuristics_final = np.maximum(heuristics_final, heuristics_final.T)
    
    # Final connectivity check
    for i in range(n):
        if np.sum(heuristics_final[i] > 0) < 2:
            # Add back the best missing edge from original scores
            row = heuristics_base[i].copy()
            row[i] = -np.inf
            row[heuristics_final[i] > 0] = -np.inf
            
            if np.max(row) > -np.inf:
                j = np.argmax(row)
                heuristics_final[i, j] = heuristics_base[i, j]
                heuristics_final[j, i] = heuristics_base[j, i]
    
    # === 6. FINAL NORMALIZATION ===
    non_zero_mask = heuristics_final > 0
    if np.any(non_zero_mask):
        max_val = np.max(heuristics_final[non_zero_mask])
        if max_val > 0:
            heuristics_final /= max_val
    
    np.fill_diagonal(heuristics_final, 0)
    
    return heuristics_final

def _compute_spectral_clustering_info(distance_matrix: np.ndarray, n: int) -> dict:
    """
    Compute clustering information using spectral methods.
    Falls back to simple heuristics for large or uniform instances.
    """
    if n < 10:
        # Too small for meaningful clustering
        return {
            'labels': np.zeros(n, dtype=int),
            'n_clusters': 1,
            'sizes': [n],
            'intra_inter_ratio': 0.5
        }
    
    try:
        # Convert distances to similarities
        median_dist = np.median(distance_matrix[np.triu_indices(n, k=1)])
        similarity = np.exp(-distance_matrix / (median_dist + 1e-12))
        np.fill_diagonal(similarity, 0)
        
        # Compute normalized Laplacian
        D = np.diag(np.sum(similarity, axis=1))
        D_inv_sqrt = np.diag(1.0 / np.sqrt(np.diag(D) + 1e-12))
        L = np.eye(n) - D_inv_sqrt @ similarity @ D_inv_sqrt
        
        # Compute eigenvectors
        eigvals, eigvecs = np.linalg.eigh(L)
        
        # Determine number of clusters from eigenvalue gap
        eigval_gaps = np.diff(eigvals[:min(10, n)])
        if len(eigval_gaps) > 0:
            gap_idx = np.argmax(eigval_gaps) + 1
            n_clusters = min(max(2, gap_idx), n // 4)
        else:
            n_clusters = 2
        
        # Use K-means on eigenvectors
        from sklearn.cluster import KMeans
        kmeans = KMeans(n_clusters=n_clusters, n_init=10, random_state=42)
        labels = kmeans.fit_predict(eigvecs[:, 1:n_clusters+1])
        
        # Compute intra-cluster vs inter-cluster distances
        intra_dists = []
        inter_dists = []
        
        for i in range(n):
            for j in range(i + 1, n):
                if labels[i] == labels[j]:
                    intra_dists.append(distance_matrix[i, j])
                else:
                    inter_dists.append(distance_matrix[i, j])
        
        if intra_dists and inter_dists:
            avg_intra = np.mean(intra_dists)
            avg_inter = np.mean(inter_dists)
            intra_inter_ratio = avg_intra / (avg_inter + 1e-12)
        else:
            intra_inter_ratio = 0.5
        
        # Compute cluster sizes
        cluster_sizes = [np.sum(labels == i) for i in range(n_clusters)]
        
        return {
            'labels': labels,
            'n_clusters': n_clusters,
            'sizes': cluster_sizes,
            'intra_inter_ratio': intra_inter_ratio
        }
        
    except Exception:
        # Fallback: simple distance-based clustering
        # Use median distance threshold
        median_dist = np.median(distance_matrix[np.triu_indices(n, k=1)])
        threshold = median_dist * 1.5
        
        labels = np.zeros(n, dtype=int)
        current_cluster = 0
        
        for i in range(n):
            if labels[i] == 0:  # Not assigned
                labels[i] = current_cluster
                # Assign nearby nodes to same cluster
                for j in range(i + 1, n):
                    if labels[j] == 0 and distance_matrix[i, j] < threshold:
                        labels[j] = current_cluster
                current_cluster += 1
        
        n_clusters = current_cluster
        cluster_sizes = [np.sum(labels == i) for i in range(n_clusters)]
        
        return {
            'labels': labels,
            'n_clusters': n_clusters,
            'sizes': cluster_sizes,
            'intra_inter_ratio': 0.6  # Conservative default
        }
