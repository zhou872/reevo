import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge prioritization with multiplicative penalty
    synthesis and sparse centrality boosting.
    
    Features:
    1. Multiplicative synthesis of rank-based regret penalties with adaptive modulation
    2. Sparse centrality boosting focusing on high-potential hub connections
    3. Adaptive sparsification with dynamic thresholding based on edge value distribution
    4. Geometric mean fusion of multiple edge quality indicators
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        
    Returns:
        Heuristic matrix with higher values indicating more promising edges,
        same shape as input with diagonal set to zero.
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # === 1. DISTRIBUTION ANALYSIS FOR ADAPTIVE PARAMETERS ===
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    dist_cv = dist_std / max(dist_mean, 1e-12)
    
    # Adaptive parameters based on instance characteristics
    regret_base = 1.2 + 0.3 * np.tanh(dist_cv - 1.0)
    centrality_strength = 0.7 + 0.3 * np.tanh(8.0 / n - 0.6)
    sparsity_ratio = 0.35 + 0.25 * np.tanh(0.5 * dist_cv)
    
    # === 2. BASE EDGE ATTRACTION (INVERSE DISTANCE WITH LOGARITHMIC COMPRESSION) ===
    with np.errstate(divide='ignore', invalid='ignore'):
        log_inv_dist = np.log1p(1.0 / (distance_matrix + np.eye(n) * 1e-12))
    log_inv_dist[np.isinf(log_inv_dist)] = 0
    np.fill_diagonal(log_inv_dist, 0)
    
    # Normalize base attraction
    if log_inv_dist.max() > log_inv_dist.min():
        log_inv_dist = (log_inv_dist - log_inv_dist.min()) / (log_inv_dist.max() - log_inv_dist.min())
    
    # === 3. RANK-BASED REGRET PENALTY WITH ADAPTIVE MODULATION ===
    regret_penalty = np.ones((n, n))
    
    for i in range(n):
        # Get distances from node i to all others
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        dists_i = distance_matrix[i, mask]
        
        if len(dists_i) >= 2:
            # Sort distances and compute ranks
            sorted_indices = np.argsort(dists_i)
            ranks = np.zeros_like(dists_i)
            ranks[sorted_indices] = np.arange(len(dists_i))
            
            # Adaptive modulation: stronger penalty for mid-ranked edges
            # This creates a "valley" shape emphasizing both best and acceptable alternatives
            rank_weights = 1.0 + (ranks / len(dists_i)) * (regret_base - 1.0)
            
            # Apply to original indices
            valid_idx = np.where(mask)[0]
            for j, rank in zip(valid_idx, ranks):
                penalty = 1.0 / (1.0 + rank_weights[rank] * (rank / len(dists_i)))
                regret_penalty[i, j] = penalty
    
    # Ensure symmetry
    regret_penalty = np.minimum(regret_penalty, regret_penalty.T)
    
    # === 4. SPARSE CENTRALITY BOOSTING ===
    # Calculate node centrality as inverse harmonic mean of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_distances = 1.0 / (distance_matrix + np.eye(n) * 1e-12)
        inv_distances[np.isinf(inv_distances)] = 0
    
    harmonic_means = n / np.sum(inv_distances, axis=1)
    centrality = 1.0 / (harmonic_means + 1e-12)
    
    # Normalize and apply sigmoid to create sparse boosting
    centrality = (centrality - centrality.min()) / (centrality.max() - centrality.min() + 1e-12)
    centrality = 0.2 + 0.8 / (1.0 + np.exp(-8.0 * (centrality - 0.5)))
    
    # Create centrality matrix with multiplicative boosting for hub connections
    centrality_matrix = np.outer(centrality, centrality)
    centrality_boost = 0.3 + 0.7 * centrality_matrix
    
    # === 5. MINIMUM SPANNING TREE REINFORCEMENT ===
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # MST reinforcement factor decays with instance size
    mst_factor = 1.5 + 0.5 * np.exp(-n / 15.0)
    
    # === 6. COMBINE FACTORS WITH GEOMETRIC MEAN ===
    # Geometric mean emphasizes balanced contributions
    heuristics = np.ones((n, n))
    
    # Incorporate base attraction
    heuristics *= np.power(log_inv_dist + 1e-12, 0.4)
    
    # Incorporate regret penalty
    heuristics *= np.power(regret_penalty, 0.3)
    
    # Incorporate centrality boost
    heuristics *= np.power(centrality_boost, centrality_strength)
    
    # Boost MST edges
    heuristics[mst_mask] *= mst_factor
    
    # Zero diagonal
    np.fill_diagonal(heuristics, 0)
    
    # === 7. ADAPTIVE SPARSIFICATION WITH DYNAMIC THRESHOLDING ===
    if sparsity_ratio > 0:
        # Get upper triangle values for threshold determination
        triu_indices = np.triu_indices(n, k=1)
        upper_values = heuristics[triu_indices]
        
        if len(upper_values) > 0:
            # Dynamic threshold based on value distribution and sparsity ratio
            threshold = np.percentile(upper_values, sparsity_ratio * 100)
            
            # Apply threshold symmetrically
            zero_mask = (heuristics < threshold) & ~mst_mask
            zero_mask = zero_mask | zero_mask.T
            np.fill_diagonal(zero_mask, False)
            
            heuristics[zero_mask] = 0
    
    # === 8. FINAL NORMALIZATION AND SYMMETRIZATION ===
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2.0
    
    # Normalize to [0, 1] range
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
