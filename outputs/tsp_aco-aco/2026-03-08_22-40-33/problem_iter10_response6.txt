import numpy as np
from scipy.spatial.distance import pdist, squareform
from scipy.cluster.hierarchy import fcluster, linkage
from scipy.sparse.csgraph import minimum_spanning_tree


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization with improved clustering
    and global approximation guidance.
    
    Key improvements:
    1. Hierarchical clustering for robust spatial grouping
    2. Christofides-based edge boosting with MST and approximate matching
    3. Instance-adaptive parameter tuning based on distance statistics
    4. Density-aware edge preservation with cluster connectivity
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances (shape n x n)
        
    Returns:
        Heuristic matrix of same shape as input, with scores in [0, 1]
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # === 1. INSTANCE FEATURE EXTRACTION ===
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    dist_cv = dist_std / (dist_mean + 1e-12)
    n_features = {
        'cv': dist_cv,
        'size': n,
        'mean_dist': dist_mean,
        'std_dist': dist_std
    }
    
    # === 2. IMPROVED SPATIAL CLUSTERING ===
    cluster_labels = _hierarchical_clustering(distance_matrix, n_features)
    cluster_info = _analyze_clusters(cluster_labels, distance_matrix)
    
    # === 3. BASE INVERSE DISTANCE WITH ADAPTIVE SCALING ===
    with np.errstate(divide='ignore', invalid='ignore'):
        scale = dist_mean if dist_mean > 0 else 1.0
        # Adaptive smoothing based on instance features
        smoothing = 0.05 + 0.15 * np.tanh(2 * dist_cv)
        inv_dist = scale / (distance_matrix + smoothing * scale * np.eye(n))
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # === 4. CLUSTER-AWARE REGRET PENALTY ===
    regret_penalty = _compute_cluster_regret(
        distance_matrix, n, cluster_labels, cluster_info, n_features
    )
    
    # === 5. GLOBAL TSP APPROXIMATION BOOSTS ===
    mst_mask, matching_mask = _christofides_components(distance_matrix, n_features)
    
    # === 6. ADAPTIVE NEIGHBORHOOD ANALYSIS ===
    lookahead_score = _adaptive_neighborhood_score(
        distance_matrix, n, cluster_labels, cluster_info, n_features
    )
    
    # === 7. COMBINE COMPONENTS ===
    heuristics = inv_dist.copy()
    heuristics *= regret_penalty
    heuristics *= lookahead_score
    
    # Boost edges from TSP approximations
    mst_weight = 1.4 + 0.3 * np.tanh(2 - 2 * dist_cv)  # Higher boost for uniform instances
    matching_weight = 1.2 + 0.2 * np.tanh(2 * dist_cv - 1)  # Higher boost for varied instances
    
    heuristics[mst_mask] *= mst_weight
    heuristics[matching_mask] *= matching_weight
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    
    # === 8. DENSITY-AWARE EDGE PRESERVATION ===
    if n > 3:
        heuristics = _cluster_aware_sparsification(
            heuristics, distance_matrix, mst_mask, matching_mask,
            cluster_labels, cluster_info, n_features
        )
    
    # === 9. FINAL NORMALIZATION ===
    np.fill_diagonal(heuristics, 0)
    
    # Normalize to [0, 1] range
    non_zero_mask = heuristics > 0
    if np.any(non_zero_mask):
        max_val = heuristics[non_zero_mask].max()
        if max_val > 0:
            heuristics[non_zero_mask] = heuristics[non_zero_mask] / max_val
    
    # Final symmetry check
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    return heuristics


def _hierarchical_clustering(distance_matrix: np.ndarray, 
                            features: dict) -> np.ndarray:
    """
    Perform hierarchical clustering with adaptive cutoff.
    """
    n = distance_matrix.shape[0]
    
    if n <= 10 or features['cv'] < 0.25:
        return np.zeros(n, dtype=int)
    
    # Convert to condensed form for linkage
    condensed_dist = distance_matrix[np.triu_indices(n, k=1)]
    
    # Perform hierarchical clustering
    Z = linkage(condensed_dist, method='average')
    
    # Adaptive cutoff based on instance features
    if features['cv'] < 0.5:
        # Uniform distances: fewer clusters
        cutoff = np.percentile(Z[:, 2], 70)
    else:
        # Varied distances: more clusters
        cutoff = np.percentile(Z[:, 2], 40)
    
    # Ensure reasonable number of clusters
    max_clusters = min(8, max(2, n // 10))
    cluster_labels = fcluster(Z, t=cutoff, criterion='distance')
    
    # Merge small clusters
    unique_labels, counts = np.unique(cluster_labels, return_counts=True)
    for i, (label, count) in enumerate(zip(unique_labels, counts)):
        if count < max(2, n // 20):
            # Merge with nearest larger cluster
            cluster_mask = cluster_labels == label
            if np.any(cluster_mask):
                # Find nearest cluster centroid
                centroids = []
                for other_label in unique_labels:
                    if other_label != label:
                        other_mask = cluster_labels == other_label
                        if np.any(other_mask):
                            # Compute average distance between clusters
                            inter_dist = distance_matrix[cluster_mask][:, other_mask]
                            centroids.append((other_label, np.mean(inter_dist)))
                
                if centroids:
                    nearest_label = min(centroids, key=lambda x: x[1])[0]
                    cluster_labels[cluster_mask] = nearest_label
    
    # Relabel to consecutive integers
    unique_labels = np.unique(cluster_labels)
    label_map = {old: new for new, old in enumerate(unique_labels)}
    return np.array([label_map[l] for l in cluster_labels])


def _analyze_clusters(cluster_labels: np.ndarray, 
                     distance_matrix: np.ndarray) -> dict:
    """Analyze cluster properties for informed weighting."""
    n = len(cluster_labels)
    n_clusters = len(np.unique(cluster_labels))
    
    if n_clusters == 1:
        return {
            'n_clusters': 1,
            'sizes': [n],
            'densities': [1.0],
            'separations': [0]
        }
    
    cluster_sizes = []
    cluster_densities = []
    cluster_separations = []
    
    for c in range(n_clusters):
        mask = cluster_labels == c
        size = np.sum(mask)
        cluster_sizes.append(size)
        
        if size > 1:
            # Intra-cluster density
            intra_distances = []
            for i in np.where(mask)[0]:
                for j in np.where(mask)[0]:
                    if i < j:
                        intra_distances.append(distance_matrix[i, j])
            density = np.mean(intra_distances) if intra_distances else 1.0
        else:
            density = 1.0
        cluster_densities.append(density)
        
        # Inter-cluster separation
        if size > 0 and size < n:
            other_mask = ~mask
            min_dists = []
            for i in np.where(mask)[0]:
                inter_dists = distance_matrix[i, other_mask]
                if len(inter_dists) > 0:
                    min_dists.append(np.min(inter_dists))
            separation = np.mean(min_dists) if min_dists else 0
        else:
            separation = 0
        cluster_separations.append(separation)
    
    # Normalize densities and separations
    max_density = max(cluster_densities) if cluster_densities else 1.0
    max_separation = max(cluster_separations) if cluster_separations else 1.0
    
    if max_density > 0:
        cluster_densities = [d / max_density for d in cluster_densities]
    if max_separation > 0:
        cluster_separations = [s / max_separation for s in cluster_separations]
    
    return {
        'n_clusters': n_clusters,
        'sizes': cluster_sizes,
        'densities': cluster_densities,
        'separations': cluster_separations
    }


def _compute_cluster_regret(distance_matrix: np.ndarray, n: int,
                           cluster_labels: np.ndarray, 
                           cluster_info: dict,
                           features: dict) -> np.ndarray:
    """
    Compute regret penalty informed by spatial clustering and instance features.
    """
    regret = np.ones((n, n))
    
    # Adaptive parameters based on instance features
    base_power = 1.2 + 1.0 * np.tanh(3 * features['cv'] - 1)
    
    for i in range(n):
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        candidates = np.where(mask)[0]
        
        if len(candidates) < 2:
            continue
        
        dists = distance_matrix[i, candidates]
        sorted_idx = np.argsort(dists)
        best_dist = dists[sorted_idx[0]]
        
        # Get cluster context
        cluster_i = cluster_labels[i]
        same_cluster_mask = cluster_labels[candidates] == cluster_i
        
        for rank, pos in enumerate(sorted_idx):
            j = candidates[pos]
            current_dist = dists[pos]
            
            if rank == 0:
                regret[i, j] = 1.0
            else:
                # Base regret components
                norm_rank = rank / (len(candidates) - 1)
                dist_ratio = current_dist / (best_dist + 1e-12)
                
                # Cluster-aware adjustment
                same_cluster = same_cluster_mask[pos]
                
                if same_cluster and cluster_info['n_clusters'] > 1:
                    # Within clusters: consider cluster density
                    cluster_density = cluster_info['densities'][cluster_i]
                    size_factor = np.sqrt(min(cluster_info['sizes'][cluster_i], 20) / 20)
                    power_adjust = 1.0 + 0.3 * cluster_density - 0.1 * size_factor
                elif not same_cluster and cluster_info['n_clusters'] > 1:
                    # Between clusters: consider separation
                    cluster_j = cluster_labels[j]
                    sep_i = cluster_info['separations'][cluster_i]
                    sep_j = cluster_info['separations'][cluster_j]
                    avg_sep = (sep_i + sep_j) / 2
                    power_adjust = 0.8 + 0.3 * avg_sep
                else:
                    power_adjust = 1.0
                
                adjusted_power = base_power * power_adjust
                
                # Compute regret with adaptive non-linearity
                regret_term = norm_rank * np.sqrt(dist_ratio)
                penalty = 1 / (1 + adjusted_power * regret_term)
                
                # Additional penalty for very poor alternatives
                if dist_ratio > 2.0:
                    penalty *= 0.7
                elif dist_ratio > 1.5:
                    penalty *= 0.85
                
                regret[i, j] = penalty
    
    # Symmetrize using geometric mean
    regret = np.sqrt(regret * regret.T)
    
    return regret


def _christofides_components(distance_matrix: np.ndarray,
                            features: dict) -> tuple:
    """
    Compute MST and approximate minimum-weight matching components.
    """
    n = distance_matrix.shape[0]
    mst_mask = np.zeros((n, n), dtype=bool)
    matching_mask = np.zeros((n, n), dtype=bool)
    
    # Minimum Spanning Tree
    mst = minimum_spanning_tree(distance_matrix)
    mst_dense = mst.toarray()
    mst_mask = (mst_dense > 0) | (mst_dense.T > 0)
    
    # Find odd-degree vertices
    degrees = np.sum(mst_mask, axis=1)
    odd_vertices = np.where(degrees % 2 == 1)[0]
    
    # Greedy minimum-weight matching on odd vertices
    if len(odd_vertices) >= 2:
        matched = np.zeros(len(odd_vertices), dtype=bool)
        
        # Sort vertices by degree for better matching
        vertex_degrees = degrees[odd_vertices]
        sorted_indices = np.argsort(-vertex_degrees)  # High degree first
        
        for idx in sorted_indices:
            if matched[idx]:
                continue
                
            u = odd_vertices[idx]
            best_dist = np.inf
            best_jdx = -1
            
            # Find closest unmatched odd vertex
            for jdx, v in enumerate(odd_vertices):
                if not matched[jdx] and jdx != idx:
                    dist = distance_matrix[u, v]
                    if dist < best_dist:
                        best_dist = dist
                        best_jdx = jdx
            
            if best_jdx != -1:
                v = odd_vertices[best_jdx]
                matching_mask[u, v] = matching_mask[v, u] = True
                matched[idx] = matched[best_jdx] = True
    
    return mst_mask, matching_mask


def _adaptive_neighborhood_score(distance_matrix: np.ndarray, n: int,
                                cluster_labels: np.ndarray,
                                cluster_info: dict,
                                features: dict) -> np.ndarray:
    """
    Compute neighborhood efficiency score with adaptive depth.
    """
    score = np.ones((n, n))
    
    if n < 4:
        return score
    
    # Adaptive neighbor count
    base_k = max(3, int(np.sqrt(n)))
    neighbor_factor = 1.0 + 0.5 * np.tanh(features['cv'] - 0.5)
    k_neighbors = min(n - 1, int(base_k * neighbor_factor))
    
    # Get nearest neighbors for each node
    nearest = np.argsort(distance_matrix, axis=1)[:, 1:k_neighbors + 1]
    
    for i in range(n):
        for j in range(i + 1, n):
            direct = distance_matrix[i, j]
            if direct <= 0:
                continue
            
            # Determine lookahead depth based on clustering
            same_cluster = cluster_labels[i] == cluster_labels[j]
            if same_cluster and cluster_info['n_clusters'] > 1:
                max_depth = 2  # Shorter lookahead within clusters
            else:
                max_depth = 3  # Longer lookahead between clusters
            
            best_alt = np.inf
            
            # 2-hop alternatives (always check)
            common_neighbors = set(nearest[i]).intersection(set(nearest[j]))
            common_neighbors.discard(i)
            common_neighbors.discard(j)
            
            for k in common_neighbors:
                alt = distance_matrix[i, k] + distance_matrix[k, j]
                best_alt = min(best_alt, alt)
            
            # 3-hop alternatives if beneficial
            if max_depth >= 3 and best_alt / direct > 1.2:
                # Limited exploration for efficiency
                for k in nearest[i][:min(3, len(nearest[i]))]:
                    if k == j:
                        continue
                    for l in nearest[j][:min(3, len(nearest[j]))]:
                        if l == i or l == k:
                            continue
                        alt = (distance_matrix[i, k] + 
                              distance_matrix[k, l] + 
                              distance_matrix[l, j])
                        best_alt = min(best_alt, alt)
            
            if best_alt < np.inf:
                ratio = best_alt / direct
                
                # Adaptive scoring based on instance features
                if features['cv'] < 0.4:
                    # Uniform instances: strict scoring
                    center, steepness = 1.3, 2.0
                else:
                    # Varied instances: more forgiving
                    center, steepness = 1.5, 1.5
                
                # Sigmoid-like transformation
                sigmoid = 1.0 / (1.0 + np.exp(-steepness * (ratio - center)))
                efficiency = 0.4 + 1.2 * sigmoid
                
                score[i, j] = score[j, i] = np.clip(efficiency, 0.2, 1.8)
    
    return score


def _cluster_aware_sparsification(heuristics: np.ndarray,
                                 distance_matrix: np.ndarray,
                                 mst_mask: np.ndarray,
                                 matching_mask: np.ndarray,
                                 cluster_labels: np.ndarray,
                                 cluster_info: dict,
                                 features: dict) -> np.ndarray:
    """
    Perform density-aware edge preservation with cluster consideration.
    """
    n = heuristics.shape[0]
    result = heuristics.copy()
    
    # Always preserve MST and matching edges
    preserve_mask = mst_mask | matching_mask
    
    # Calculate local densities
    densities = np.zeros(n)
    k_density = min(max(3, n // 10), 8)
    
    for i in range(n):
        # Get k-nearest distances (excluding self)
        dists = np.partition(distance_matrix[i], k_density + 1)[:k_density + 1]
        dists = dists[dists > 0]
        densities[i] = np.mean(dists) if len(dists) > 0 else 1.0
    
    # Normalize densities
    if np.max(densities) > 0:
        densities = densities / np.max(densities)
    
    # Determine edges to keep per node
    base_keep = max(3, min(8, int(np.sqrt(n))))
    
    for i in range(n):
        row = result[i].copy()
        row[i] = -np.inf
        
        # Always keep preserved edges
        keep_indices = set(np.where(preserve_mask[i])[0])
        
        # Categorize remaining edges by cluster
        same_cluster = cluster_labels == cluster_labels[i]
        
        # Keep within-cluster edges
        same_idx = np.where(same_cluster & (np.arange(n) != i))[0]
        if len(same_idx) > 0:
            same_scores = row[same_idx]
            # Adaptive number based on cluster size and density
            cluster_size = cluster_info['sizes'][cluster_labels[i]]
            size_factor = np.sqrt(min(cluster_size, 20) / 20)
            keep_same = max(1, int(base_keep * 0.6 * size_factor))
            keep_same = min(keep_same, len(same_idx))
            
            if keep_same > 0:
                top_same = np.argsort(same_scores)[-keep_same:]
                keep_indices.update(same_idx[top_same])
        
        # Keep between-cluster edges
        diff_idx = np.where(~same_cluster & (np.arange(n) != i))[0]
        if len(diff_idx) > 0:
            diff_scores = row[diff_idx]
            # Adjust based on local density and instance features
            density_factor = 0.3 + 0.7 * densities[i]
            cv_factor = 1.0 + 0.5 * features['cv']
            keep_diff = max(1, int(base_keep * 0.4 * density_factor * cv_factor))
            keep_diff = min(keep_diff, len(diff_idx))
            
            if keep_diff > 0:
                top_diff = np.argsort(diff_scores)[-keep_diff:]
                keep_indices.update(diff_idx[top_diff])
        
        # Zero out non-kept edges
        for j in range(n):
            if j != i and j not in keep_indices:
                result[i, j] = 0
    
    # Ensure symmetry
    result = np.maximum(result, result.T)
    
    return result
