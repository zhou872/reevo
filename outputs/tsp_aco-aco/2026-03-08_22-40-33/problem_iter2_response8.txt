import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy import stats

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization with adaptive tuning.
    
    Features:
    1. MST-based connectivity with adaptive weighting based on instance characteristics
    2. Regret-based penalty with curvature-adaptive scaling
    3. Node centrality metrics (degree and betweenness proxies)
    4. Local search sensitivity via 2-opt move potential estimation
    
    Args:
        distance_matrix: Symmetric matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape as input, with higher values indicating
        more promising edges for inclusion in optimal tours.
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # === 1. INSTANCE CHARACTERISTICS ANALYSIS ===
    # Compute statistics for adaptive parameter tuning
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    dist_cv = dist_std / (dist_mean + 1e-12)  # Coefficient of variation
    
    # Adaptive parameters based on instance characteristics
    mst_weight = 1.8 + 0.4 * np.tanh(dist_cv)  # Higher CV → more MST emphasis
    regret_factor = 1.3 + 0.4 * (1 - np.exp(-dist_cv))  # Higher CV → stronger regret
    sparsity_factor = 0.6 + 0.2 * np.tanh(10/n - 0.5)  # Sparse more for large n
    
    # === 2. BASE INVERSE DISTANCE WITH ROBUST HANDLING ===
    with np.errstate(divide='ignore', invalid='ignore'):
        # Use logarithmic compression for better numerical stability
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # === 3. MINIMUM SPANNING TREE WITH ADAPTIVE WEIGHTING ===
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # === 4. REGRET-BASED PENALTY WITH CURVATURE ADAPTATION ===
    regret_penalty = np.ones((n, n))
    
    for i in range(n):
        # Get sorted distances from node i
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        valid_indices = np.where(mask)[0]
        
        if len(valid_indices) >= 2:
            dists = distance_matrix[i, valid_indices]
            sorted_idx = np.argsort(dists)
            sorted_dists = dists[sorted_idx]
            
            # Compute gap ratios between consecutive distances
            if sorted_dists[0] > 0:
                gaps = np.diff(sorted_dists) / sorted_dists[0]
                avg_gap = np.mean(gaps[:min(5, len(gaps))])  # Look at top 5 gaps
                
                # Adaptive curvature: larger gaps → stronger regret
                curvature = 1.0 + np.tanh(avg_gap)
                local_regret = regret_factor * curvature
            else:
                local_regret = regret_factor
            
            # Apply penalty based on rank with local regret factor
            for rank, global_idx in enumerate(valid_indices[sorted_idx]):
                penalty = 1 / (1 + local_regret * rank)
                regret_penalty[i, global_idx] = penalty
    
    # Symmetrize regret penalty
    regret_penalty = np.sqrt(regret_penalty * regret_penalty.T)
    
    # === 5. NODE CENTRALITY METRICS ===
    # Degree centrality proxy: inverse of average distance to other nodes
    with np.errstate(divide='ignore', invalid='ignore'):
        degree_centrality = 1 / np.mean(distance_matrix + np.eye(n) * 1e-12, axis=1)
    degree_centrality = degree_centrality / np.max(degree_centrality)
    
    # Betweenness centrality proxy: based on MST structure
    betweenness = np.zeros(n)
    for i in range(n):
        # Count MST edges incident to node i
        mst_neighbors = np.where(mst_mask[i])[0]
        betweenness[i] = len(mst_neighbors) / (n - 1)
    
    # Combine centrality measures
    centrality = 0.6 * degree_centrality + 0.4 * betweenness
    centrality_matrix = np.outer(centrality, centrality)
    
    # === 6. LOCAL SEARCH SENSITIVITY (2-OPT POTENTIAL) ===
    ls_sensitivity = np.ones((n, n))
    
    for i in range(n):
        # Find two closest neighbors for 2-opt estimation
        mask_i = np.ones(n, dtype=bool)
        mask_i[i] = False
        i_neighbors = np.argsort(distance_matrix[i])[1:3]  # Top 2 neighbors
        
        for j in range(i + 1, n):
            # Find two closest neighbors for j
            mask_j = np.ones(n, dtype=bool)
            mask_j[j] = False
            j_neighbors = np.argsort(distance_matrix[j])[1:3]  # Top 2 neighbors
            
            # Estimate potential 2-opt improvement
            current_pair = distance_matrix[i, j]
            
            # Best alternative connections
            alt_i = min([distance_matrix[i, k] for k in j_neighbors if k != i], default=np.inf)
            alt_j = min([distance_matrix[j, k] for k in i_neighbors if k != j], default=np.inf)
            
            if alt_i < np.inf and alt_j < np.inf:
                # Potential gain from 2-opt swap
                potential_gain = (alt_i + alt_j) - current_pair
                if potential_gain > 0:
                    # Higher gain → more sensitive to local search
                    gain_norm = min(potential_gain / dist_mean, 2.0)
                    ls_sensitivity[i, j] = 1.0 + 0.3 * np.tanh(gain_norm)
    
    ls_sensitivity = np.maximum(ls_sensitivity, ls_sensitivity.T)
    
    # === 7. COMBINE ALL FACTORS ===
    heuristics = inv_dist.copy()
    
    # Apply multiplicative factors
    heuristics *= regret_penalty
    heuristics *= centrality_matrix
    heuristics *= ls_sensitivity
    
    # Boost MST edges with adaptive weight
    heuristics[mst_mask] *= mst_weight
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    
    # === 8. ADAPTIVE SPARSIFICATION ===
    if sparsity_factor > 0:
        # Separate treatment for MST vs non-MST edges
        non_mst_mask = ~mst_mask
        np.fill_diagonal(non_mst_mask, False)
        
        if np.any(non_mst_mask):
            non_mst_values = heuristics[non_mst_mask]
            if len(non_mst_values) > 0:
                # Dynamic threshold based on value distribution
                q_low = np.percentile(non_mst_values, sparsity_factor * 50)
                q_high = np.percentile(non_mst_values, 100 - sparsity_factor * 50)
                threshold = (q_low + q_high) / 2
                
                # Zero out non-MST edges below threshold
                zero_mask = non_mst_mask & (heuristics < threshold)
                heuristics[zero_mask] = 0
    
    # === 9. FINAL PROCESSING ===
    np.fill_diagonal(heuristics, 0)
    
    # Normalize to [0, 1] range for consistency
    max_val = heuristics.max()
    if max_val > 0:
        heuristics /= max_val
    
    return heuristics
