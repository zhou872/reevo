import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with reciprocal distance baseline,
    MST edge boosting, density-aware adjustment, and connectivity assurance.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric n x n matrix of pairwise distances
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate more promising edges.
        Diagonal is zero, values are normalized to [0, 1].
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(heuristic, 0)
        max_val = np.max(heuristic)
        if max_val > 0:
            heuristic /= max_val
        return heuristic
    
    # 1. Base heuristic: reciprocal of distances, normalized
    with np.errstate(divide='ignore', invalid='ignore'):
        heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(heuristic, 0)
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic /= max_val
    
    # 2. MST edge identification and boosting
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Strong boost for MST edges (essential connectivity)
    heuristic[mst_mask] *= 3.0
    
    # 3. Density-aware adjustment
    # Compute local densities based on k nearest neighbors
    k_density = min(10, n - 1)
    densities = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if k_density < n - 1:
            nn_dists = np.partition(dists, k_density)[:k_density]
        else:
            nn_dists = dists[dists < np.inf]
        densities[i] = 1.0 / (np.mean(nn_dists) + 1e-12)
    
    # Normalize densities to [0.5, 1.5] range
    min_dens, max_dens = np.min(densities), np.max(densities)
    if max_dens - min_dens > 1e-12:
        densities = 0.5 + (densities - min_dens) / (max_dens - min_dens)
    
    # Apply density adjustment: favor connections between similar-density nodes
    for i in range(n):
        for j in range(i + 1, n):
            density_factor = 2.0 - abs(densities[i] - densities[j])
            heuristic[i, j] *= density_factor
    
    heuristic = (heuristic + heuristic.T) / 2  # Ensure symmetry
    
    # 4. Dynamic neighbor selection with percentile cutoff
    if n < 30:
        percentile_cutoff = 35
    elif n < 100:
        percentile_cutoff = 30
    elif n < 500:
        percentile_cutoff = 25
    else:
        percentile_cutoff = 20
    
    neighborhood_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        threshold = np.percentile(dists, percentile_cutoff)
        close_nodes = dists <= threshold
        
        # Ensure minimum connections (at least 3)
        if close_nodes.sum() < 3:
            k_neighbors = max(3, min(int(np.log2(n)), n - 1))
            nearest_indices = np.argsort(dists)[:k_neighbors]
            close_nodes = np.zeros(n, dtype=bool)
            close_nodes[nearest_indices] = True
        
        neighborhood_mask[i] = close_nodes
    
    neighborhood_mask = neighborhood_mask | neighborhood_mask.T  # Ensure symmetry
    
    # Moderate boost for neighborhood edges
    heuristic[neighborhood_mask] *= 1.8
    
    # 5. Lightweight triangle quality check (sampled)
    if n <= 1000:  # Only for manageable sizes
        sample_edges = min(200, n * (n - 1) // 10)
        edge_indices = np.array(np.triu_indices(n, k=1)).T
        
        if len(edge_indices) > sample_edges:
            edge_scores = heuristic[edge_indices[:, 0], edge_indices[:, 1]]
            top_indices = np.argpartition(-edge_scores, sample_edges)[:sample_edges]
            sampled_edges = edge_indices[top_indices]
        else:
            sampled_edges = edge_indices
        
        # Check triangle inequality for sampled edges
        for i, j in sampled_edges:
            if heuristic[i, j] < 0.1:  # Skip low-probability edges
                continue
            
            # Sample a few common neighbors
            sample_k = min(5, n - 2)
            if sample_k > 0:
                candidates = np.setdiff1d(np.arange(n), [i, j])
                if len(candidates) > sample_k:
                    candidates = np.random.choice(candidates, sample_k, replace=False)
                
                quality_sum = 0
                for k in candidates:
                    # Check if triangle inequality holds (approximately)
                    violation = distance_matrix[i, j] - (distance_matrix[i, k] + distance_matrix[k, j])
                    if violation <= 0:
                        quality_sum += 1.0
                    else:
                        quality_sum += max(0, 1.0 - violation / distance_matrix[i, j])
                
                triangle_quality = quality_sum / sample_k
                heuristic[i, j] = heuristic[j, i] = heuristic[i, j] * (1.0 + 0.15 * triangle_quality)
    
    # 6. Adaptive sparsification
    target_edges = max(4, min(int(np.log2(n) * 2.5), n - 1))
    sparsified = np.zeros_like(heuristic)
    
    for i in range(n):
        row_heuristics = heuristic[i].copy()
        row_heuristics[i] = -np.inf
        
        mst_neighbors = np.where(mst_mask[i])[0]
        additional_needed = max(0, target_edges - len(mst_neighbors))
        
        if additional_needed > 0:
            non_mst_mask = np.ones(n, dtype=bool)
            non_mst_mask[mst_neighbors] = False
            non_mst_mask[i] = False
            non_mst_indices = np.where(non_mst_mask)[0]
            
            if len(non_mst_indices) > 0:
                non_mst_values = row_heuristics[non_mst_indices]
                if len(non_mst_indices) > additional_needed:
                    threshold_idx = np.argpartition(-non_mst_values, additional_needed - 1)[additional_needed - 1]
                    threshold_val = non_mst_values[threshold_idx]
                    keep_non_mst = non_mst_indices[non_mst_values >= threshold_val]
                else:
                    keep_non_mst = non_mst_indices
            else:
                keep_non_mst = []
        else:
            keep_non_mst = []
        
        keep_indices = np.concatenate([mst_neighbors, keep_non_mst])
        sparsified[i, keep_indices] = heuristic[i, keep_indices]
    
    sparsified = np.maximum(sparsified, sparsified.T)  # Ensure symmetry
    
    # 7. Connectivity assurance
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        node = stack.pop()
        neighbors = np.where(sparsified[node] > 0)[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    if not np.all(visited):
        # Connect isolated components
        unvisited = np.where(~visited)[0]
        for node in unvisited:
            dists_to_visited = distance_matrix[node, visited]
            if len(dists_to_visited) > 0:
                closest_idx = np.argmin(dists_to_visited)
                visited_nodes = np.where(visited)[0]
                closest = visited_nodes[closest_idx]
                sparsified[node, closest] = sparsified[closest, node] = 0.2
                visited[node] = True
    
    # 8. Final processing
    np.fill_diagonal(sparsified, 0)
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified /= max_val
    
    return sparsified
