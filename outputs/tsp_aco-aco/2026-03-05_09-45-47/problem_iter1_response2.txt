import numpy as np
from scipy.spatial import KDTree
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection combining multiple factors.
    
    Strategy:
    1. Use reciprocal of distances as base indicator
    2. Enhance with spatial clustering to identify local neighborhoods
    3. Incorporate minimum spanning tree edges as promising connections
    4. Sparsify by setting less promising edges to zero
    """
    n = distance_matrix.shape[0]
    
    if n < 3:
        return 1 / (distance_matrix + np.eye(n) * 1e-12)
    
    # Base heuristic: reciprocal of distances
    reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
    
    # Factor 1: Spatial clustering via KD-tree
    # Generate approximate coordinates from distance matrix using MDS
    from sklearn.manifold import MDS
    mds = MDS(n_components=2, dissimilarity='precomputed', random_state=42)
    coords = mds.fit_transform(distance_matrix)
    
    # Build KD-tree for spatial queries
    tree = KDTree(coords)
    
    # For each node, find k nearest spatial neighbors (k ~ log2(n))
    k = min(max(3, int(np.log2(n)) + 1), n - 1)
    spatial_promising = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Query k+1 neighbors (including self)
        distances, indices = tree.query(coords[i], k=k+1)
        # Exclude self
        indices = indices[1:] if distances[0] == 0 else indices[:k]
        spatial_promising[i, indices] = 1.0
    
    # Factor 2: Minimum Spanning Tree edges
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    mst_promising = mst_edges + mst_edges.T  # Make symmetric
    
    # Factor 3: Local degree enhancement
    # For each node, compute average distance to its m nearest neighbors
    m = min(5, n - 1)
    local_avg_dist = np.zeros(n)
    for i in range(n):
        sorted_dists = np.sort(distance_matrix[i])
        local_avg_dist[i] = np.mean(sorted_dists[1:m+1])  # Exclude self
    
    # Combine factors with weights
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(i+1, n):
            if i == j:
                continue
                
            base_score = reciprocal_weights[i, j]
            
            # Spatial clustering factor
            spatial_factor = 2.0 if spatial_promising[i, j] > 0 else 0.5
            
            # MST factor
            mst_factor = 3.0 if mst_promising[i, j] > 0 else 0.7
            
            # Local connectivity factor
            max_local_avg = max(local_avg_dist[i], local_avg_dist[j])
            dist_ratio = distance_matrix[i, j] / (max_local_avg + 1e-12)
            local_factor = np.exp(-dist_ratio**2)
            
            # Combine all factors
            combined_score = (base_score * spatial_factor * mst_factor * 
                            local_factor * (1 + 0.5 * np.log(1 + 1/base_score)))
            
            heuristics_matrix[i, j] = heuristics_matrix[j, i] = combined_score
    
    # Sparsification: keep only top promising edges
    # For each node, keep connections to top p% of other nodes
    p = min(30, 100 * (np.log2(n) + 2) / n)  # Adaptive percentage
    
    sparsified = np.zeros_like(heuristics_matrix)
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Ignore diagonal
        threshold = np.percentile(row[row > 0], 100 - p) if np.any(row > 0) else 0
        mask = row >= threshold
        sparsified[i] = heuristics_matrix[i] * mask
    
    # Ensure symmetry after sparsification
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Final enhancement: boost scores for edges that form triangles
    triangle_boost = np.zeros_like(sparsified)
    for i in range(n):
        for j in range(i+1, n):
            if sparsified[i, j] > 0:
                # Count how many common neighbors both i and j have
                common_neighbors = np.sum((sparsified[i] > 0) & (sparsified[j] > 0))
                if common_neighbors > 0:
                    triangle_boost[i, j] = triangle_boost[j, i] = 1 + 0.2 * common_neighbors
    
    final_heuristics = sparsified * triangle_boost
    
    # Normalize to similar scale as input
    if np.max(final_heuristics) > 0:
        final_heuristics = final_heuristics / np.max(final_heuristics)
    
    return final_heuristics
