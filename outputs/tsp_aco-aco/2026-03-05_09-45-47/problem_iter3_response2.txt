import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.neighbors import KDTree
from sklearn.manifold import MDS
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection combining multiple factors
    with improved performance and adaptive strategies.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Base heuristic: reciprocal of distances (closer edges are more promising)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Factor 1: Spatial clustering (edges within spatial neighborhoods)
    spatial_promising = _compute_spatial_clustering_v2(distance_matrix, n)
    
    # Factor 2: Minimum Spanning Tree edges (fundamental connections)
    mst_promising = _compute_mst_promises_v2(distance_matrix)
    
    # Factor 3: Local connectivity factors
    local_factors = _compute_local_factors_v2(distance_matrix, n)
    
    # Combine all factors
    heuristics_matrix = _combine_factors_v2(
        reciprocal_weights, spatial_promising, mst_promising, 
        local_factors, distance_matrix
    )
    
    # Adaptive sparsification: keep most promising edges per node
    sparsified = _adaptive_sparsification_v2(heuristics_matrix, n)
    
    # Triangle boosting: enhance edges within dense clusters
    final_heuristics = _triangle_boosting_v2(sparsified, n)
    
    # Normalize to [0, 1] range
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics

def _compute_spatial_clustering_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute spatial clustering factor using efficient coordinate approximation."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    # Adaptive k based on log2(n) with bounds
    k = max(3, min(int(np.log2(n)) + 3, n - 1))
    
    if n <= 500:
        # Use MDS for accurate coordinates when manageable
        try:
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                mds = MDS(
                    n_components=min(10, n-1),
                    dissimilarity='precomputed',
                    random_state=42,
                    n_init=3,
                    max_iter=200,
                    normalized_stress='auto'
                )
                coords = mds.fit_transform(distance_matrix)
                tree = KDTree(coords)
                
                for i in range(n):
                    distances, indices = tree.query(coords[i].reshape(1, -1), k=k+1)
                    indices = indices[0][1:]  # Exclude self
                    spatial_promising[i, indices] = 1.0
        except:
            # Fallback to distance-based approach
            for i in range(n):
                dists = distance_matrix[i].copy()
                dists[i] = np.inf
                indices = np.argpartition(dists, k)[:k]
                spatial_promising[i, indices] = 1.0
    else:
        # Use fast distance-based approach for large n
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            indices = np.argpartition(dists, k)[:k]
            spatial_promising[i, indices] = 1.0
    
    return spatial_promising

def _compute_mst_promises_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute promises based on Minimum Spanning Tree edges with improvements."""
    # Get MST edges
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    # Make symmetric
    mst_promising = mst_edges + mst_edges.T
    
    n = distance_matrix.shape[0]
    
    # For each node, boost connections to nearby nodes
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Get indices of closest nodes (scaled by log2(n))
        num_close = min(5, max(2, int(np.log2(n))))
        close_indices = np.argsort(dists)[:num_close]
        
        for j in close_indices:
            if mst_promising[i, j] == 0:
                # Scale boost based on position in sorted distances
                boost = 0.8 - 0.15 * np.where(close_indices == j)[0][0]
                mst_promising[i, j] = max(mst_promising[i, j], boost)
    
    return mst_promising

def _compute_local_factors_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute local connectivity factors with vectorized operations."""
    # For each node, compute statistics of its local neighborhood
    m = min(10, n - 1)  # Increased neighborhood size
    local_stats = np.zeros((n, 2))  # [mean_dist, std_dist]
    
    # Vectorized computation of local statistics
    dists_copy = distance_matrix.copy()
    np.fill_diagonal(dists_copy, np.inf)
    
    # Get m smallest distances for each node
    sorted_indices = np.argpartition(dists_copy, m, axis=1)[:, :m]
    sorted_dists = np.take_along_axis(dists_copy, sorted_indices, axis=1)
    
    local_stats[:, 0] = np.mean(sorted_dists, axis=1)
    local_stats[:, 1] = np.std(sorted_dists, axis=1) + 1e-12
    
    # Vectorized computation of local factors
    i_indices, j_indices = np.triu_indices(n, k=1)
    
    # Compute stat similarity
    mean_diff = np.abs(local_stats[i_indices, 0] - local_stats[j_indices, 0])
    std_sum = local_stats[i_indices, 1] + local_stats[j_indices, 1]
    stat_similarity = np.exp(-mean_diff / std_sum)
    
    # Compute distance ratio
    mean_dists = (local_stats[i_indices, 0] + local_stats[j_indices, 0]) / 2
    edge_dists = distance_matrix[i_indices, j_indices]
    dist_ratio = edge_dists / (mean_dists + 1e-12)
    
    # Combine factors
    local_vals = stat_similarity * np.exp(-0.3 * dist_ratio**2)  # Reduced penalty
    
    # Create symmetric matrix
    local_factors = np.ones_like(distance_matrix)
    local_factors[i_indices, j_indices] = local_vals
    local_factors[j_indices, i_indices] = local_vals
    
    return local_factors

def _combine_factors_v2(reciprocal_weights: np.ndarray, 
                       spatial_promising: np.ndarray,
                       mst_promising: np.ndarray,
                       local_factors: np.ndarray,
                       distance_matrix: np.ndarray) -> np.ndarray:
    """Combine all heuristic factors efficiently with improved weighting."""
    n = reciprocal_weights.shape[0]
    
    # Start with base reciprocal weights
    heuristics_matrix = reciprocal_weights.copy()
    
    # Apply spatial factor: strong boost for spatial neighbors
    spatial_mask = spatial_promising > 0
    heuristics_matrix[spatial_mask] *= 1.8  # Increased from 2.0 to 1.8
    
    # Apply MST factor: scale based on MST promise value
    mst_mask = mst_promising > 0
    mst_boost = np.where(mst_promising == 2, 2.5,  # Strong boost for direct MST edges
                        np.where(mst_promising >= 1, 1.7,  # Medium boost for nearby
                                1.0))  # No boost otherwise
    heuristics_matrix *= mst_boost
    
    # Apply local factors
    heuristics_matrix *= local_factors
    
    # Additional logarithmic boost for very short edges
    with np.errstate(divide='ignore', invalid='ignore'):
        short_edges = reciprocal_weights > np.percentile(reciprocal_weights[reciprocal_weights > 0], 80)
        heuristics_matrix[short_edges] *= (1 + 0.2 * np.log1p(reciprocal_weights[short_edges]))
    
    # Ensure symmetry
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    return heuristics_matrix

def _adaptive_sparsification_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Adaptively sparsify the heuristic matrix keeping top edges per node."""
    if n <= 10:
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Adaptive sparsity based on log2(n)
    base_keep_ratio = min(0.35, max(0.12, np.log2(n) / 15))
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Ignore diagonal
        
        # Determine number of edges to keep
        num_to_keep = max(4, int(base_keep_ratio * n))
        num_to_keep = min(num_to_keep, n - 1)
        
        # Find top edges efficiently
        if np.any(row > 0):
            # Use argpartition for efficient selection
            top_indices = np.argpartition(-row, num_to_keep)[:num_to_keep]
            sparsified[i, top_indices] = heuristics_matrix[i, top_indices]
    
    # Ensure symmetry by taking maximum (logical OR of connections)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified

def _triangle_boosting_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Boost scores for edges that form triangles (common neighbors)."""
    if n < 3:
        return heuristics_matrix.copy()
    
    boosted = heuristics_matrix.copy()
    
    # Create binary adjacency matrix
    binary_mask = (heuristics_matrix > 0).astype(float)
    
    # Count common neighbors efficiently using matrix multiplication
    common_neighbors = binary_mask @ binary_mask.T
    
    # Boost edges based on number of common neighbors
    for i in range(n):
        for j in range(i+1, n):
            if heuristics_matrix[i, j] > 0:
                common = common_neighbors[i, j]
                if common > 0:
                    # Boost factor increases with common neighbors but saturates
                    boost = 1 + 0.25 * min(common, 6) / (1 + 0.1 * common)
                    boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    
    return boosted
