import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform
from scipy.cluster.hierarchy import linkage, fcluster
from sklearn.cluster import SpectralClustering
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced multi-scale heuristic for TSP edge selection with hierarchical clustering,
    graph structure learning, and adaptive ensemble weighting.
    
    Features:
    1. Multi-scale hierarchical community detection with overlapping assignments
    2. Graph structure learning via random walk statistics and centrality measures
    3. Adaptive ensemble weighting based on instance hardness estimation
    4. Multi-resolution edge promise aggregation
    
    Returns:
        Edge promise matrix with values in [0, 1], same shape as input.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            max_val = np.max(result)
            if max_val > 0:
                result = result / max_val
            return result
    
    # Phase 1: Multi-scale instance characterization
    instance_features = _compute_multi_scale_features(distance_matrix, n)
    
    # Phase 2: Hierarchical community detection with overlapping assignments
    community_layers = _hierarchical_community_detection(distance_matrix, n, instance_features)
    
    # Phase 3: Graph structure learning
    structural_features = _learn_graph_structure(distance_matrix, n, instance_features)
    
    # Phase 4: Generate multi-resolution heuristic ensemble
    heuristic_layers = _generate_multi_resolution_ensemble(
        distance_matrix, n, instance_features, community_layers, structural_features
    )
    
    # Phase 5: Adaptive ensemble fusion with hardness-aware weighting
    combined = _adaptive_ensemble_fusion(
        heuristic_layers, community_layers, instance_features, distance_matrix, n
    )
    
    # Phase 6: Multi-resolution sparsification preserving hierarchical connectivity
    sparsified = _multi_resolution_sparsification(
        combined, community_layers, n, instance_features
    )
    
    # Phase 7: Hierarchical triangle boosting
    boosted = _hierarchical_triangle_boosting(
        sparsified, community_layers, n, instance_features
    )
    
    # Final processing
    np.fill_diagonal(boosted, 0)
    max_val = np.max(boosted)
    if max_val > 0:
        boosted = boosted / max_val
    
    return boosted

def _compute_multi_scale_features(distance_matrix: np.ndarray, n: int) -> dict:
    """Compute multi-scale features characterizing TSP instance hardness."""
    features = {}
    
    # Global statistics
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    features['global_mean'] = np.mean(distances)
    features['global_std'] = np.std(distances)
    features['global_skew'] = np.mean((distances - features['global_mean'])**3) / (features['global_std']**3 + 1e-12)
    
    # Local neighborhood statistics at multiple scales
    scales = [3, 5, 7, min(10, n-1)]
    local_features = {}
    for k in scales:
        if k < n:
            local_dists = np.zeros(n)
            for i in range(n):
                dists = distance_matrix[i].copy()
                dists[i] = np.inf
                local_dists[i] = np.mean(np.partition(dists, k)[:k])
            local_features[f'scale_{k}_mean'] = np.mean(local_dists)
            local_features[f'scale_{k}_cv'] = np.std(local_dists) / (np.mean(local_dists) + 1e-12)
    features['local'] = local_features
    
    # MST-based features
    mst = minimum_spanning_tree(distance_matrix)
    mst_dists = mst.data[mst.data > 0]
    features['mst_mean'] = np.mean(mst_dists)
    features['mst_std'] = np.std(mst_dists)
    
    # Clustering tendency via eigenvalue analysis
    if n <= 500:
        try:
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                # Normalized Laplacian eigenvalues
                W = np.exp(-distance_matrix**2 / (2 * np.median(distances)**2))
                D = np.diag(np.sum(W, axis=1))
                L_norm = np.eye(n) - np.linalg.inv(D) @ W
                eigvals = np.linalg.eigvalsh(L_norm)
                features['spectral_gap'] = eigvals[1] - eigvals[0] if len(eigvals) > 1 else 0
        except:
            features['spectral_gap'] = 0.1
    
    # Multi-scale density estimates
    thresholds = [0.25, 0.5, 0.75]
    for t in thresholds:
        threshold = np.percentile(distances, t * 100)
        features[f'density_{t}'] = np.mean(distances < threshold)
    
    # Instance hardness estimate
    hardness = features['global_std'] / (features['global_mean'] + 1e-12)
    hardness += 1 - min(1.0, features.get('spectral_gap', 0.1) * 10)
    features['hardness'] = min(1.0, hardness)
    
    return features

def _hierarchical_community_detection(distance_matrix: np.ndarray, n: int, features: dict) -> dict:
    """Detect hierarchical communities with overlapping assignments."""
    communities = {}
    
    # Base case: all nodes in one community
    if n < 10:
        communities['level_0'] = np.ones((n, n))
        return communities
    
    # Convert distances to similarities
    median_dist = np.median(distance_matrix[np.triu_indices(n, k=1)])
    similarities = np.exp(-distance_matrix**2 / (2 * (median_dist**2)))
    
    # Multiple community detection methods
    community_matrices = []
    
    # Method 1: Hierarchical clustering with multiple cuts
    if n <= 1000:
        try:
            condensed_dist = squareform(distance_matrix)
            Z = linkage(condensed_dist, method='average')
            
            # Multiple cut levels
            cut_levels = [0.3, 0.5, 0.7]
            for level in cut_levels:
                max_clusters = max(2, int(n * level))
                labels = fcluster(Z, max_clusters, criterion='maxclust')
                comm_matrix = np.zeros((n, n))
                for i in range(n):
                    for j in range(n):
                        if i != j and labels[i] == labels[j]:
                            comm_matrix[i, j] = 1.0
                community_matrices.append(comm_matrix)
        except:
            pass
    
    # Method 2: Spectral clustering at multiple resolutions
    n_clusters_options = [max(2, int(np.sqrt(n))), max(2, n//10), max(2, n//5)]
    for n_clusters in n_clusters_options:
        if n_clusters < n:
            try:
                with warnings.catch_warnings():
                    warnings.simplefilter("ignore")
                    sc = SpectralClustering(
                        n_clusters=n_clusters,
                        affinity='precomputed',
                        random_state=42,
                        assign_labels='discretize'
                    )
                    labels = sc.fit_predict(similarities)
                    comm_matrix = np.zeros((n, n))
                    for i in range(n):
                        for j in range(n):
                            if i != j and labels[i] == labels[j]:
                                comm_matrix[i, j] = 1.0
                    community_matrices.append(comm_matrix)
            except:
                continue
    
    # Method 3: Distance-based overlapping communities
    # Each node belongs to multiple communities based on proximity
    k = min(10, n-1)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        neighbors = np.argsort(dists)[:k]
        for j in neighbors:
            if distance_matrix[i, j] < 1.5 * median_dist:
                # Find common close neighbors
                common_close = set(neighbors)
                dists_j = distance_matrix[j].copy()
                dists_j[j] = np.inf
                neighbors_j = set(np.argsort(dists_j)[:k])
                overlap = len(common_close & neighbors_j)
                if overlap >= k//2:
                    comm_matrix = np.zeros((n, n))
                    # Create community around this edge
                    cluster = list(common_close & neighbors_j)
                    for a in cluster:
                        for b in cluster:
                            if a != b:
                                comm_matrix[a, b] = 1.0
                    community_matrices.append(comm_matrix)
    
    # Combine community matrices with weights
    if community_matrices:
        for i, mat in enumerate(community_matrices):
            communities[f'level_{i}'] = mat
    else:
        # Fallback: single community
        communities['level_0'] = np.ones((n, n))
    
    return communities

def _learn_graph_structure(distance_matrix: np.ndarray, n: int, features: dict) -> dict:
    """Learn graph structural features via random walks and centrality."""
    structural = {}
    
    # Convert to similarity matrix for random walks
    median_dist = np.median(distance_matrix[np.triu_indices(n, k=1)])
    similarity = np.exp(-distance_matrix**2 / (2 * (median_dist**2)))
    np.fill_diagonal(similarity, 0)
    
    # Random walk transition matrix
    row_sums = similarity.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    P = similarity / row_sums
    
    # Random walk statistics
    if n <= 500:
        # Stationary distribution (principal eigenvector)
        try:
            eigvals, eigvecs = np.linalg.eig(P.T)
            stationary_idx = np.argmin(np.abs(eigvals - 1))
            stationary = np.real(eigvecs[:, stationary_idx])
            stationary = np.abs(stationary) / np.sum(np.abs(stationary))
            structural['centrality'] = stationary
        except:
            structural['centrality'] = np.ones(n) / n
    else:
        structural['centrality'] = np.ones(n) / n
    
    # Local clustering coefficients approximation
    k = min(10, n-1)
    clustering = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        neighbors = np.argsort(dists)[:k]
        if len(neighbors) >= 2:
            subgraph = distance_matrix[np.ix_(neighbors, neighbors)]
            if subgraph.size > 0:
                # Simple connectivity measure
                threshold = np.median(subgraph[subgraph > 0]) if np.any(subgraph > 0) else 1
                connected = (subgraph < threshold).sum() / 2
                possible = len(neighbors) * (len(neighbors) - 1) / 2
                clustering[i] = connected / possible if possible > 0 else 0
    structural['clustering'] = clustering
    
    # Edge betweenness approximation
    betweenness = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Simple betweenness proxy: inverse of distance times centrality product
            betweenness[i, j] = structural['centrality'][i] * structural['centrality'][j] / (distance_matrix[i, j] + 1e-12)
    structural['betweenness'] = betweenness + betweenness.T
    
    return structural

def _generate_multi_resolution_ensemble(distance_matrix: np.ndarray, n: int, 
                                       features: dict, communities: dict,
                                       structural: dict) -> dict:
    """Generate heuristic ensemble at multiple resolutions."""
    ensemble = {}
    
    # Base reciprocal heuristic with multi-scale adaptation
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal, 0)
    
    # Adjust based on instance hardness
    exponent = 0.5 + 0.5 * features['hardness']
    reciprocal = np.power(reciprocal, exponent)
    ensemble['reciprocal'] = reciprocal
    
    # MST-based heuristic with structural enhancement
    mst_promising = _enhanced_mst_heuristic_v2(distance_matrix, n, features, structural)
    ensemble['mst'] = mst_promising
    
    # Multi-scale spatial heuristic
    spatial = _multi_scale_spatial_heuristic(distance_matrix, n, features)
    ensemble['spatial'] = spatial
    
    # Structure-aware heuristic using learned features
    structure_heuristic = _structure_aware_heuristic(distance_matrix, n, features, structural)
    ensemble['structure'] = structure_heuristic
    
    # Community-guided heuristic
    community_heuristic = _community_guided_heuristic(distance_matrix, n, features, communities)
    ensemble['community'] = community_heuristic
    
    return ensemble

def _enhanced_mst_heuristic_v2(distance_matrix: np.ndarray, n: int, 
                              features: dict, structural: dict) -> np.ndarray:
    """MST heuristic enhanced with structural features."""
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    mst_promising = mst_edges + mst_edges.T
    
    # Boost MST edges based on structural importance
    centrality_product = np.outer(structural['centrality'], structural['centrality'])
    mst_promising[mst_edges > 0] *= (1 + centrality_product[mst_edges > 0] * 2)
    
    # Add connections considering structural roles
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Adaptive number based on centrality
        num_close = min(max(3, int(5 * structural['centrality'][i] * n / 2)), n-1)
        close_indices = np.argsort(dists)[:num_close]
        
        for rank, j in enumerate(close_indices):
            if mst_promising[i, j] == 0:
                # Boost based on structural similarity
                struct_sim = 1 - abs(structural['clustering'][i] - structural['clustering'][j])
                boost = 0.5 * (1 + struct_sim) / (1 + 0.3 * rank)
                mst_promising[i, j] = boost
    
    return mst_promising

def _multi_scale_spatial_heuristic(distance_matrix: np.ndarray, n: int, features: dict) -> np.ndarray:
    """Spatial heuristic at multiple scales."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    # Multiple neighborhood sizes
    scales = [max(3, int(np.log2(n))), max(5, int(np.sqrt(n))), min(15, n-1)]
    
    for k in scales:
        if k < n:
            for i in range(n):
                dists = distance_matrix[i].copy()
                dists[i] = np.inf
                indices = np.argpartition(dists, k)[:k]
                # Weight by scale
                weight = 1.0 / (1 + np.log1p(k))
                spatial_promising[i, indices] += weight
    
    # Normalize by number of scales
    spatial_promising = spatial_promising / len(scales)
    return spatial_promising

def _structure_aware_heuristic(distance_matrix: np.ndarray, n: int, 
                              features: dict, structural: dict) -> np.ndarray:
    """Heuristic based on learned graph structure."""
    heuristic = np.ones_like(distance_matrix)
    
    # Use centrality and clustering information
    centrality_i, centrality_j = np.meshgrid(structural['centrality'], structural['centrality'], indexing='ij')
    clustering_i, clustering_j = np.meshgrid(structural['clustering'], structural['clustering'], indexing='ij')
    
    # Centrality compatibility
    centrality_compat = 1 - np.abs(centrality_i - centrality_j)
    
    # Clustering similarity
    clustering_sim = 1 - np.abs(clustering_i - clustering_j)
    
    # Distance consistency with local structure
    local_means = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k = min(5, n-1)
        local_means[i] = np.mean(np.partition(dists, k)[:k])
    
    mean_i, mean_j = np.meshgrid(local_means, local_means, indexing='ij')
    expected_dist = (mean_i + mean_j) / 2
    dist_ratio = distance_matrix / (expected_dist + 1e-12)
    dist_consistency = np.exp(-0.5 * np.minimum(dist_ratio, 3)**2)
    
    # Combine factors
    heuristic = centrality_compat * clustering_sim * dist_consistency
    
    # Boost edges with high betweenness
    heuristic *= (1 + 0.5 * structural['betweenness'] / (structural['betweenness'].max() + 1e-12))
    
    np.fill_diagonal(heuristic, 0)
    return heuristic

def _community_guided_heuristic(distance_matrix: np.ndarray, n: int, 
                               features: dict, communities: dict) -> np.ndarray:
    """Heuristic guided by community structure."""
    if len(communities) == 0:
        return np.ones_like(distance_matrix)
    
    # Combine multiple community layers
    comm_heuristic = np.zeros_like(distance_matrix)
    
    for level, comm_matrix in communities.items():
        weight = 1.0 / (1 + int(level.split('_')[1]))  # Decreasing weight for higher levels
        comm_heuristic += weight * comm_matrix
    
    # Normalize
    if comm_heuristic.max() > 0:
        comm_heuristic = comm_heuristic / comm_heuristic.max()
    
    # Add distance-based decay within communities
    median_dist = np.median(distance_matrix[np.triu_indices(n, k=1)])
    dist_decay = np.exp(-distance_matrix**2 / (2 * median_dist**2))
    comm_heuristic = comm_heuristic * dist_decay
    
    return comm_heuristic

def _adaptive_ensemble_fusion(ensemble: dict, communities: dict, 
                            features: dict, distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Fuse ensemble with adaptive, hardness-aware weighting."""
    # Base weights adjusted by instance hardness
    hardness = features['hardness']
    
    # Hard instances: favor structural and community heuristics
    if hardness > 0.7:
        weights = {'reciprocal': 0.3, 'mst': 0.4, 'spatial': 0.2, 'structure': 0.8, 'community': 0.9}
    # Medium hardness: balanced approach
    elif hardness > 0.3:
        weights = {'reciprocal': 0.5, 'mst': 0.6, 'spatial': 0.4, 'structure': 0.6, 'community': 0.7}
    # Easy instances: favor distance-based heuristics
    else:
        weights = {'reciprocal': 0.8, 'mst': 0.7, 'spatial': 0.6, 'structure': 0.4, 'community': 0.3}
    
    # Normalize weights
    total = sum(weights.values())
    for key in weights:
        weights[key] = weights[key] / total
    
    # Initialize combined matrix
    combined = np.zeros_like(distance_matrix)
    
    # Weighted combination
    for name, heuristic in ensemble.items():
        if name in weights:
            combined += weights[name] * heuristic
    
    # Multi-scale community enhancement
    if communities:
        comm_boost = np.zeros_like(combined)
        for level, comm_matrix in communities.items():
            level_num = int(level.split('_')[1])
            weight = 1.0 / (1 + level_num * 0.5)
            
            # Boost edges within communities
            boost_factor = 1.2 + 0.3 * hardness
            comm_boost[comm_matrix > 0] += weight * boost_factor
        
        # Apply community boost
        if comm_boost.max() > 0:
            comm_boost = comm_boost / comm_boost.max()
            combined = combined * (1 + 0.5 * comm_boost)
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    return combined

def _multi_resolution_sparsification(heuristics_matrix: np.ndarray, 
                                   communities: dict, n: int, 
                                   features: dict) -> np.ndarray:
    """Sparsify while preserving multi-resolution connectivity."""
    if n <= 10:
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Adaptive sparsity based on hardness and scale
    base_ratio = 0.2 + 0.1 * (1 - features['hardness'])
    
    # Determine edges to keep per node
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        # Initial candidate edges
        num_candidates = max(4, int(base_ratio * n))
        if num_candidates < n-1:
            candidate_indices = np.argpartition(-row, num_candidates)[:num_candidates]
        else:
            candidate_indices = np.arange(n)
            candidate_indices = candidate_indices[candidate_indices != i]
        
        # Ensure community connectivity
        if communities:
            for level, comm_matrix in communities.items():
                # Find community neighbors
                comm_neighbors = np.where(comm_matrix[i] > 0)[0]
                if len(comm_neighbors) > 0:
                    # Add strongest community connections
                    comm_strengths = heuristics_matrix[i, comm_neighbors]
                    top_comm = comm_neighbors[np.argsort(-comm_strengths)[:min(3, len(comm_neighbors))]]
                    candidate_indices = np.union1d(candidate_indices, top_comm)
        
        # Keep top edges among candidates
        candidate_values = heuristics_matrix[i, candidate_indices.astype(int)]
        num_to_keep = min(len(candidate_indices), max(3, int(0.7 * len(candidate_indices))))
        top_in_candidates = np.argpartition(-candidate_values, num_to_keep)[:num_to_keep]
        keep_indices = candidate_indices[top_in_candidates.astype(int)]
        
        sparsified[i, keep_indices] = heuristics_matrix[i, keep_indices]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified

def _hierarchical_triangle_boosting(heuristics_matrix: np.ndarray,
                                  communities: dict, n: int,
                                  features: dict) -> np.ndarray:
    """Triangle boosting with hierarchical community awareness."""
    if n < 3:
        return heuristics_matrix.copy()
    
    boosted = heuristics_matrix.copy()
    
    # Create adjacency at different thresholds
    thresholds = [0.1, 0.3, 0.5]
    binary_masks = []
    
    for thresh in thresholds:
        binary = (heuristics_matrix > thresh * heuristics_matrix.max()).astype(float)
        binary_masks.append(binary)
    
    # Multi-resolution triangle counting
    for level, binary in enumerate(binary_masks):
        # Count common neighbors
        common = binary @ binary.T
        
        # Boost based on common neighbors and community structure
        boost_strength = 0.15 + 0.1 * level
        for i in range(n):
            for j in range(i+1, n):
                if heuristics_matrix[i, j] > 0:
                    common_ij = common[i, j]
                    if common_ij > 0:
                        # Base triangle boost
                        base_boost = 1 + boost_strength * min(np.sqrt(common_ij), 4)
                        
                        # Community enhancement
                        comm_boost = 1.0
                        if communities:
                            for comm_name, comm_matrix in communities.items():
                                if comm_matrix[i, j] > 0:
                                    comm_level = int(comm_name.split('_')[1])
                                    comm_boost *= (1 + 0.1 / (1 + comm_level))
                        
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * base_boost * comm_boost
    
    return boosted
