import numpy as np
import warnings
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.manifold import MDS
from scipy.spatial import KDTree
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with improved triangle quality
    boosting and spatial clustering strategies.
    
    Args:
        distance_matrix: Symmetric n×n matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # Base heuristic: reciprocal of distances (closer nodes are more promising)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(reciprocal_weights, 0)
    
    # Factor 1: Enhanced spatial clustering
    spatial_promising = _compute_spatial_clustering_v2(distance_matrix, n)
    
    # Factor 2: Minimum Spanning Tree edges
    mst_promising = _compute_mst_promises_v2(distance_matrix)
    
    # Factor 3: Local neighborhood statistics
    local_factors = _compute_local_factors_v2(distance_matrix, n)
    
    # Factor 4: Triangle quality (new: consider both count and quality)
    triangle_quality = _compute_triangle_quality(distance_matrix, n)
    
    # Combine all factors
    heuristics_matrix = _combine_factors_v2(
        reciprocal_weights, spatial_promising, mst_promising,
        local_factors, triangle_quality, distance_matrix
    )
    
    # Adaptive sparsification
    sparsified = _adaptive_sparsification_v2(heuristics_matrix, n)
    
    # Enhanced triangle boosting with quality consideration
    final_heuristics = _triangle_boosting_v2(sparsified, distance_matrix)
    
    # Normalize to [0, 1] range
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics


def _compute_spatial_clustering_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Enhanced spatial clustering with adaptive strategy selection."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    if n <= 1000:
        # Use MDS for accurate coordinates
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(3, n-1),
                dissimilarity='precomputed',
                random_state=42,
                n_init=1,
                max_iter=150  # Increased iterations for better accuracy
            )
            coords = mds.fit_transform(distance_matrix)
    else:
        # Landmark-based approximation with farthest-point sampling
        landmarks = min(100, n // 10)  # Increased landmarks for better coverage
        landmark_indices = _farthest_point_sampling(distance_matrix, landmarks)
        landmark_dists = distance_matrix[landmark_indices]
        
        # Multi-dimensional scaling with landmarks
        coords = np.zeros((n, landmarks))
        for i in range(n):
            coords[i] = distance_matrix[i, landmark_indices]
    
    # Adaptive neighborhood size
    k = max(3, min(int(np.log2(n) * 1.5), n - 1))
    
    # Build KD-tree for efficient queries
    tree = KDTree(coords)
    
    # Find k-nearest neighbors for each node
    if n <= 1000:
        for i in range(n):
            distances, indices = tree.query(coords[i], k=k+1)
            indices = indices[1:] if distances[0] == 0 else indices[:k]
            spatial_promising[i, indices] = 1.0
    else:
        # For large instances, use approximate nearest neighbors
        for i in range(n):
            # Query multiple points at once for efficiency
            if i % 100 == 0:
                batch_indices = range(i, min(i+100, n))
                batch_coords = coords[batch_indices]
                batch_dists, batch_nn = tree.query(batch_coords, k=k)
                for idx, (dists, nn) in enumerate(zip(batch_dists, batch_nn)):
                    actual_idx = batch_indices[idx]
                    spatial_promising[actual_idx, nn] = 1.0
    
    # Make symmetric
    spatial_promising = np.maximum(spatial_promising, spatial_promising.T)
    return spatial_promising


def _farthest_point_sampling(distance_matrix: np.ndarray, k: int) -> np.ndarray:
    """Select k landmarks using farthest-point sampling."""
    n = distance_matrix.shape[0]
    selected = []
    remaining = set(range(n))
    
    # Start with random point
    selected.append(np.random.choice(n))
    remaining.remove(selected[0])
    
    while len(selected) < k and remaining:
        # For each remaining point, compute distance to nearest selected point
        min_dists = np.full(n, np.inf)
        for s in selected:
            min_dists = np.minimum(min_dists, distance_matrix[s])
        
        # Select point with maximum minimum distance
        best_idx = max(remaining, key=lambda x: min_dists[x])
        selected.append(best_idx)
        remaining.remove(best_idx)
    
    return np.array(selected)


def _compute_mst_promises_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Enhanced MST promises with extended neighborhood."""
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    mst_promising = mst_edges + mst_edges.T
    
    n = distance_matrix.shape[0]
    
    # Consider extended neighborhood for each MST node
    for i in range(n):
        if np.sum(mst_edges[i]) > 0:  # Only for nodes in MST
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Get k nearest neighbors where k depends on node degree
            k = min(5, n - 1)
            closest = np.argsort(dists)[:k]
            
            for j in closest:
                if mst_promising[i, j] == 0:
                    # Weight based on relative position
                    rank = np.where(closest == j)[0][0] + 1
                    mst_promising[i, j] = 0.7 / rank
    
    return mst_promising


def _compute_local_factors_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Enhanced local neighborhood statistics."""
    m = min(7, n - 1)  # Slightly larger neighborhood
    local_stats = np.zeros((n, 3))  # [mean_dist, std_dist, median_dist]
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)[:m]
        local_stats[i, 0] = np.mean(sorted_dists)
        local_stats[i, 1] = np.std(sorted_dists) + 1e-12
        local_stats[i, 2] = np.median(sorted_dists)
    
    local_factors = np.ones_like(distance_matrix)
    
    # Vectorized computation for efficiency
    for i in range(n):
        for j in range(i+1, n):
            # Similarity in local statistics
            mean_sim = np.exp(-abs(local_stats[i, 0] - local_stats[j, 0]) / 
                             (local_stats[i, 1] + local_stats[j, 1]))
            median_sim = np.exp(-abs(local_stats[i, 2] - local_stats[j, 2]) / 
                               (local_stats[i, 1] + local_stats[j, 1]))
            
            # Distance relative to local context
            local_mean = (local_stats[i, 0] + local_stats[j, 0]) / 2
            dist_ratio = distance_matrix[i, j] / (local_mean + 1e-12)
            
            # Combined factor
            stat_similarity = (mean_sim + median_sim) / 2
            local_factors[i, j] = local_factors[j, i] = (
                stat_similarity * np.exp(-0.4 * dist_ratio**2)
            )
    
    return local_factors


def _compute_triangle_quality(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute triangle quality factors for edges."""
    if n < 4:
        return np.ones_like(distance_matrix)
    
    triangle_quality = np.ones_like(distance_matrix)
    
    # For medium-sized instances, compute exact triangle quality
    if n <= 2000:
        for i in range(n):
            for j in range(i+1, n):
                # Find common neighbors
                mask = (distance_matrix[i] < np.inf) & (distance_matrix[j] < np.inf)
                mask[i] = mask[j] = False
                common = np.where(mask)[0]
                
                if len(common) > 0:
                    # Compute average triangle quality
                    qualities = []
                    for k in common[:min(20, len(common))]:  # Limit computation
                        triangle_edges = [distance_matrix[i, j], 
                                         distance_matrix[i, k], 
                                         distance_matrix[j, k]]
                        avg_length = np.mean(triangle_edges)
                        # Quality: inverse of average edge length, normalized
                        quality = 1.0 / (avg_length + 1e-12)
                        qualities.append(quality)
                    
                    if qualities:
                        triangle_quality[i, j] = triangle_quality[j, i] = np.mean(qualities)
    
    else:
        # For large instances, use sampling
        sample_size = min(1000, n)
        sampled_pairs = np.random.choice(n, (sample_size, 2), replace=False)
        
        for i, j in sampled_pairs:
            if i >= j:
                continue
                
            # Sample common neighbors
            k_samples = min(10, n-2)
            possible_k = list(set(range(n)) - {i, j})
            if len(possible_k) > 0:
                sampled_k = np.random.choice(possible_k, min(k_samples, len(possible_k)), replace=False)
                
                qualities = []
                for k in sampled_k:
                    triangle_edges = [distance_matrix[i, j], 
                                     distance_matrix[i, k], 
                                     distance_matrix[j, k]]
                    avg_length = np.mean(triangle_edges)
                    quality = 1.0 / (avg_length + 1e-12)
                    qualities.append(quality)
                
                if qualities:
                    avg_quality = np.mean(qualities)
                    # Apply smoothing to nearby edges
                    triangle_quality[i, j] = triangle_quality[j, i] = avg_quality
    
    # Normalize triangle quality
    max_q = np.max(triangle_quality)
    if max_q > 0:
        triangle_quality = triangle_quality / max_q
    
    return triangle_quality


def _combine_factors_v2(reciprocal_weights: np.ndarray,
                       spatial_promising: np.ndarray,
                       mst_promising: np.ndarray,
                       local_factors: np.ndarray,
                       triangle_quality: np.ndarray,
                       distance_matrix: np.ndarray) -> np.ndarray:
    """Enhanced combination of all heuristic factors."""
    n = reciprocal_weights.shape[0]
    heuristics_matrix = np.zeros_like(reciprocal_weights)
    
    # Base scores from reciprocal distances
    base_scores = reciprocal_weights.copy()
    
    # Apply spatial clustering factor
    spatial_factor = np.where(spatial_promising > 0, 2.2, 0.6)
    
    # Apply MST factor with graduated weights
    mst_factor = np.where(mst_promising > 0, 
                         np.where(mst_promising >= 1, 3.0,  # Direct MST edges
                                 np.where(mst_promising > 0.5, 2.0, 1.2)),  # Extended neighborhood
                         0.8)
    
    # Apply triangle quality factor
    quality_factor = 1.0 + 0.5 * triangle_quality  # Boost based on triangle quality
    
    # Combine all factors
    combined_scores = base_scores * spatial_factor * mst_factor * local_factors * quality_factor
    
    # Additional logarithmic boost for very short edges
    with np.errstate(divide='ignore', invalid='ignore'):
        log_boost = 1 + 0.25 * np.log(1 + 1 / (base_scores + 1e-12))
        log_boost = np.nan_to_num(log_boost, nan=1.0, posinf=1.0)
    
    heuristics_matrix = combined_scores * log_boost
    
    # Ensure symmetry
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    return heuristics_matrix


def _adaptive_sparsification_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Enhanced adaptive sparsification with connectivity preservation."""
    if n <= 10:
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Adaptive keep ratio: more edges for smaller graphs
    base_keep_ratio = min(0.5, max(0.2, 0.3 - np.log10(n) / 20))
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        # Ensure connectivity: always keep at least 2 edges
        num_to_keep = max(2, int(base_keep_ratio * n))
        num_to_keep = min(num_to_keep, n - 1)
        
        # Find top edges
        if np.any(row > 0):
            # Get indices of top k edges
            top_indices = np.argpartition(-row, num_to_keep)[:num_to_keep]
            threshold = np.min(row[top_indices])
            
            # Keep edges above threshold
            mask = row >= threshold
            sparsified[i] = heuristics_matrix[i] * mask
    
    # Ensure symmetry by taking maximum
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified


def _triangle_boosting_v2(heuristics_matrix: np.ndarray, 
                         distance_matrix: np.ndarray) -> np.ndarray:
    """Enhanced triangle boosting considering both closure and quality."""
    n = heuristics_matrix.shape[0]
    boosted = heuristics_matrix.copy()
    
    if n <= 500:
        # Exact computation for smaller graphs
        binary_mask = (heuristics_matrix > 0).astype(float)
        
        # Count common neighbors efficiently
        common_neighbors = binary_mask @ binary_mask.T
        
        for i in range(n):
            for j in range(i+1, n):
                if heuristics_matrix[i, j] > 0:
                    common = int(common_neighbors[i, j])
                    if common > 0:
                        # Base boost from triangle closure (capped at 4)
                        closure_boost = 1 + 0.15 * min(common, 4)
                        
                        # Quality-aware boost
                        # Find actual common neighbors for quality computation
                        neighbors_i = set(np.where(binary_mask[i] > 0)[0])
                        neighbors_j = set(np.where(binary_mask[j] > 0)[0])
                        actual_common = list(neighbors_i & neighbors_j)
                        
                        if actual_common:
                            # Sample up to 4 common neighbors for quality computation
                            sample_size = min(4, len(actual_common))
                            sampled = np.random.choice(actual_common, sample_size, replace=False)
                            
                            qualities = []
                            for k in sampled:
                                tri_length = (distance_matrix[i, j] + 
                                             distance_matrix[i, k] + 
                                             distance_matrix[j, k])
                                quality = 3.0 / (tri_length + 1e-12)  # Inverse of average length
                                qualities.append(quality)
                            
                            avg_quality = np.mean(qualities) if qualities else 0
                            quality_boost = 1 + 0.1 * avg_quality
                            
                            # Combined boost
                            total_boost = closure_boost * quality_boost
                            boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * total_boost
    
    else:
        # Sampling for large graphs
        sample_size = min(200, n)
        sampled_nodes = np.random.choice(n, sample_size, replace=False)
        
        for i in sampled_nodes:
            neighbors_i = set(np.where(heuristics_matrix[i] > 0)[0])
            for j in neighbors_i:
                if j <= i:
                    continue
                neighbors_j = set(np.where(heuristics_matrix[j] > 0)[0])
                common = len(neighbors_i & neighbors_j)
                
                if common > 0:
                    # Closure boost (capped at 4)
                    closure_boost = 1 + 0.12 * min(common, 4)
                    
                    # Simplified quality boost based on edge length ratio
                    if distance_matrix[i, j] > 0:
                        # Estimate quality from the edge itself
                        # Shorter edges in dense neighborhoods get higher boost
                        local_density = len(neighbors_i) + len(neighbors_j)
                        quality_factor = 1 + 0.05 * (local_density / distance_matrix[i, j])
                        quality_factor = min(quality_factor, 1.3)  # Cap quality boost
                        
                        total_boost = closure_boost * quality_factor
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * total_boost
    
    return boosted
