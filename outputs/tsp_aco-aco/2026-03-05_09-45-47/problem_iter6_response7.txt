import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Efficient and robust heuristic for TSP edge selection.
    
    Core strategy:
    1. Reciprocal distance as base heuristic
    2. Minimum spanning tree edges as mandatory connections
    3. Adaptive sparsification preserving essential connectivity
    4. Simple neighborhood detection without expensive clustering
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric n x n matrix of pairwise distances
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(heuristic, 0)
        max_val = np.max(heuristic)
        if max_val > 0:
            heuristic /= max_val
        return heuristic
    
    # 1. Base reciprocal heuristic (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        base_heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(base_heuristic, 0)
    
    # Normalize to [0, 1]
    max_val = np.max(base_heuristic)
    if max_val > 0:
        base_heuristic /= max_val
    
    # 2. Identify MST edges (essential connectivity)
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # 3. Simple neighborhood detection via reciprocal ranking
    # Each node connects to its top k neighbors based on reciprocal distance
    k_neighbors = max(3, min(int(np.log2(n)) * 2, n - 1))
    neighborhood_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get indices of top k neighbors (excluding self)
        indices = np.argsort(base_heuristic[i])[-k_neighbors-1:-1]
        neighborhood_mask[i, indices] = True
    
    # Make symmetric
    neighborhood_mask = neighborhood_mask | neighborhood_mask.T
    
    # 4. Combine factors
    heuristic = base_heuristic.copy()
    
    # Boost MST edges significantly
    heuristic[mst_mask] *= 3.0
    
    # Moderate boost for neighborhood edges
    heuristic[neighborhood_mask] *= 1.5
    
    # Ensure MST edges have at least the boosted value even if not in neighborhood
    for i in range(n):
        for j in range(i + 1, n):
            if mst_mask[i, j]:
                # Ensure MST edges have minimum value
                heuristic[i, j] = heuristic[j, i] = max(heuristic[i, j], 0.3)
    
    # 5. Adaptive sparsification
    # Target edges per node adapts with problem size
    target_edges = max(3, min(int(np.log2(n)) * 3, n - 1))
    sparsified = np.zeros_like(heuristic)
    
    for i in range(n):
        # Always keep MST edges
        mst_neighbors = np.where(mst_mask[i])[0]
        
        # Calculate how many additional edges to keep
        additional = max(0, target_edges - len(mst_neighbors))
        
        if additional > 0:
            # Get non-MST edges sorted by heuristic value
            non_mst_mask = np.ones(n, dtype=bool)
            non_mst_mask[mst_neighbors] = False
            non_mst_mask[i] = False  # Exclude self
            
            non_mst_indices = np.where(non_mst_mask)[0]
            if len(non_mst_indices) > 0:
                non_mst_heuristics = heuristic[i, non_mst_indices]
                
                if len(non_mst_indices) > additional:
                    # Keep top 'additional' non-MST edges
                    threshold = np.partition(non_mst_heuristics, -additional)[-additional]
                    keep_non_mst = non_mst_indices[non_mst_heuristics >= threshold]
                else:
                    keep_non_mst = non_mst_indices[non_mst_heuristics > 0]
            else:
                keep_non_mst = []
        else:
            keep_non_mst = []
        
        # Combine all edges to keep
        keep_indices = np.concatenate([mst_neighbors, keep_non_mst])
        sparsified[i, keep_indices] = heuristic[i, keep_indices]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 6. Connectivity verification and repair
    for i in range(n):
        if np.count_nonzero(sparsified[i]) == 0:
            # Connect isolated node to nearest neighbor
            nearest = np.argsort(distance_matrix[i])[1]
            sparsified[i, nearest] = sparsified[nearest, i] = 0.1
    
    # 7. Final normalization
    np.fill_diagonal(sparsified, 0)
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified /= max_val
    
    return sparsified
