import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import KDTree
from sklearn.manifold import MDS
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining spatial structure,
    minimum spanning tree, and adaptive clustering with improved triangle boosting.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # Base heuristic: reciprocal of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Factor 1: Improved spatial clustering
    spatial_promising = _compute_spatial_clustering_v2(distance_matrix, n)
    
    # Factor 2: Minimum Spanning Tree with enhanced connectivity
    mst_promising = _compute_mst_promises_v2(distance_matrix, n)
    
    # Factor 3: Local connectivity with better statistics
    local_factors = _compute_local_factors_v2(distance_matrix, n)
    
    # Factor 4: Triangle potential (pre-sparsification)
    triangle_potential = _compute_triangle_potential(distance_matrix, n)
    
    # Combine all factors
    heuristics_matrix = _combine_factors_v2(
        reciprocal_weights, spatial_promising, mst_promising, 
        local_factors, triangle_potential, distance_matrix
    )
    
    # Adaptive sparsification with size-dependent parameters
    sparsified = _adaptive_sparsification_v2(heuristics_matrix, n)
    
    # Enhanced triangle boosting on sparsified matrix
    boosted = _triangle_boosting_v2(sparsified, n)
    
    # Final normalization
    max_val = np.max(boosted)
    if max_val > 0:
        boosted = boosted / max_val
    
    np.fill_diagonal(boosted, 0)
    return boosted


def _compute_spatial_clustering_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Improved spatial clustering using fast nearest neighbor search."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    if n <= 1500:
        # Use efficient MDS for moderate-sized graphs
        try:
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                mds = MDS(
                    n_components=min(3, n-1),
                    dissimilarity='precomputed',
                    random_state=42,
                    n_init=1,
                    max_iter=50,
                    eps=1e-3
                )
                coords = mds.fit_transform(distance_matrix)
                tree = KDTree(coords)
                k = max(3, min(int(np.log2(n)) + 3, n - 1))
                
                for i in range(n):
                    distances, indices = tree.query(coords[i].reshape(1, -1), k=k+1)
                    indices = indices[0]
                    if distances[0, 0] == 0:  # Exclude self
                        indices = indices[1:]
                    else:
                        indices = indices[:k]
                    spatial_promising[i, indices] = 1.0
        except:
            # Fallback to distance-based approach
            pass
    
    # Always include distance-based nearest neighbors as fallback
    k_dist = max(3, min(int(np.log10(n)) * 2 + 2, n - 1))
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        indices = np.argpartition(dists, k_dist)[:k_dist]
        spatial_promising[i, indices] = np.maximum(spatial_promising[i, indices], 0.7)
    
    # Make symmetric
    spatial_promising = np.maximum(spatial_promising, spatial_promising.T)
    return spatial_promising


def _compute_mst_promises_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Enhanced MST promises with connectivity considerations."""
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    # Strong boost for MST edges
    mst_promising = np.ones_like(distance_matrix) * 0.5
    mst_promising[mst_edges + mst_edges.T > 0] = 3.0
    
    # Additional connectivity: consider nodes that are 2 hops away in MST
    mst_binary = mst_edges + mst_edges.T
    mst_squared = np.minimum(mst_binary @ mst_binary, 1)
    two_hop = mst_squared - np.diag(np.diag(mst_squared)) - mst_binary
    mst_promising[two_hop > 0] = 1.8
    
    return mst_promising


def _compute_local_factors_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Improved local factors with better neighborhood modeling."""
    # Adaptive neighborhood size
    k = max(5, min(int(np.log2(n)) * 2, n - 1))
    
    # Compute local statistics efficiently
    local_means = np.zeros(n)
    local_stds = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.partition(dists, k)[:k]
        local_means[i] = np.mean(sorted_dists)
        local_stds[i] = np.std(sorted_dists) + 1e-12
    
    # Vectorized computation of local factors
    mean_matrix = (local_means[:, None] + local_means[None, :]) / 2
    std_sum = local_stds[:, None] + local_stds[None, :]
    
    # Similarity based on local statistics
    stat_similarity = np.exp(-np.abs(local_means[:, None] - local_means[None, :]) / std_sum)
    
    # Distance factor relative to local means
    dist_ratio = distance_matrix / (mean_matrix + 1e-12)
    dist_factor = np.exp(-0.3 * dist_ratio)
    
    # Combine
    local_factors = stat_similarity * dist_factor
    np.fill_diagonal(local_factors, 0)
    
    return local_factors


def _compute_triangle_potential(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute triangle potential before sparsification."""
    if n < 4:
        return np.ones_like(distance_matrix)
    
    # Normalize distances for triangle comparison
    max_dist = np.max(distance_matrix)
    norm_dist = distance_matrix / (max_dist + 1e-12)
    
    # Compute triangle inequality scores
    triangle_potential = np.zeros_like(distance_matrix)
    
    # For moderate n, compute exact triangle potential
    if n <= 1000:
        for k in range(n):
            # For each intermediate node k, check triangle inequality
            via_k = norm_dist[:, k:k+1] + norm_dist[k:k+1, :]
            triangle_ratio = via_k / (norm_dist + 1e-12)
            # Boost edges that complete many near-equilateral triangles
            good_triangles = (triangle_ratio < 1.2) & (triangle_ratio > 0.8)
            triangle_potential += good_triangles.astype(float)
    
    # Normalize triangle potential
    if np.max(triangle_potential) > 0:
        triangle_potential = triangle_potential / np.max(triangle_potential)
    
    np.fill_diagonal(triangle_potential, 0)
    return triangle_potential


def _combine_factors_v2(reciprocal_weights, spatial_promising, mst_promising,
                       local_factors, triangle_potential, distance_matrix):
    """Enhanced factor combination with balanced weighting."""
    n = reciprocal_weights.shape[0]
    
    # Base score with diminishing returns on very short edges
    base_scores = np.sqrt(reciprocal_weights)  # Square root for diminishing returns
    
    # Combine factors with different weights
    spatial_weight = 1.2
    mst_weight = 1.5
    local_weight = 0.8
    triangle_weight = 1.0
    
    combined = (base_scores * 
                (spatial_weight ** spatial_promising) *
                (mst_weight ** (mst_promising / 3)) *
                (local_weight ** (1 - local_factors)) *  # Inverse because local_factors is similarity
                (1 + triangle_weight * triangle_potential))
    
    # Add penalty for very long edges
    max_dist = np.max(distance_matrix)
    long_edge_penalty = np.exp(-distance_matrix / (0.3 * max_dist))
    combined = combined * long_edge_penalty
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    return combined


def _adaptive_sparsification_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Size-adaptive sparsification with dynamic edge retention."""
    if n <= 10:
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Dynamic edge retention based on log10(n)
    base_ratio = 0.25 + 0.05 * np.log10(n)
    base_ratio = min(0.5, max(0.15, base_ratio))
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        # Determine number of edges to keep
        min_edges = max(3, int(np.log2(n)) + 1)
        max_edges = min(n // 2, 50)  # Upper bound
        num_to_keep = min(max_edges, max(min_edges, int(base_ratio * n)))
        
        # Find top edges
        if np.any(row > 0):
            threshold = np.partition(row, -num_to_keep)[-num_to_keep]
            mask = row >= threshold
            sparsified[i] = heuristics_matrix[i] * mask
    
    # Symmetrize: keep edge if either node wants it
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified


def _triangle_boosting_v2(sparsified_matrix: np.ndarray, n: int) -> np.ndarray:
    """Enhanced triangle boosting on sparsified matrix."""
    boosted = sparsified_matrix.copy()
    
    if n < 4:
        return boosted
    
    # Create binary adjacency from sparsified matrix
    binary_adj = (sparsified_matrix > 0).astype(float)
    
    # Count common neighbors efficiently
    common_neighbors = binary_adj @ binary_adj.T
    
    # Boost edges based on common neighbors with diminishing returns
    for i in range(n):
        for j in range(i+1, n):
            if sparsified_matrix[i, j] > 0:
                common = common_neighbors[i, j]
                if common > 0:
                    # Diminishing returns: log(1 + common)
                    boost = 1 + 0.2 * np.log1p(min(common, 10))
                    boosted[i, j] = boosted[j, i] = sparsified_matrix[i, j] * boost
    
    return boosted
