import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic for TSP edge selection with dynamic percentile neighbor selection,
    MST edge boosting, adaptive sparsification, and connectivity assurance.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric n x n matrix of pairwise distances
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate more promising edges.
        Diagonal is zero, values are normalized to [0, 1].
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases (n <= 3)
    if n <= 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(heuristic, 0)
        max_val = np.max(heuristic)
        if max_val > 0:
            heuristic /= max_val
        return heuristic
    
    # 1. Base heuristic: reciprocal of distances, normalized
    with np.errstate(divide='ignore', invalid='ignore'):
        base_heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(base_heuristic, 0)
    max_val = np.max(base_heuristic)
    if max_val > 0:
        base_heuristic /= max_val
    
    # 2. Identify MST edges for essential connectivity
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # 3. Dynamic percentile neighbor selection
    # Adjust percentile based on graph size to balance connectivity and sparsity
    if n < 30:
        percentile_cutoff = 35  # More neighbors for small graphs
    elif n < 100:
        percentile_cutoff = 30
    elif n < 500:
        percentile_cutoff = 25
    else:
        percentile_cutoff = 20  # Fewer neighbors for large graphs
    
    neighborhood_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        threshold = np.percentile(dists, percentile_cutoff)
        close_nodes = dists <= threshold
        
        # Ensure minimum connections (at least 3)
        if close_nodes.sum() < 3:
            k_neighbors = max(3, min(int(np.log2(n)), n - 1))
            nearest_indices = np.argsort(dists)[:k_neighbors]
            close_nodes = np.zeros(n, dtype=bool)
            close_nodes[nearest_indices] = True
        
        neighborhood_mask[i] = close_nodes
    
    # Ensure symmetry: if i is close to j, j is close to i
    neighborhood_mask = neighborhood_mask | neighborhood_mask.T
    
    # 4. Combine factors with targeted boosting
    heuristic = base_heuristic.copy()
    heuristic[mst_mask] *= 3.0   # Strong boost for MST edges
    heuristic[neighborhood_mask] *= 1.8  # Moderate boost for neighborhood edges
    heuristic = (heuristic + heuristic.T) / 2  # Enforce symmetry
    
    # 5. Adaptive sparsification: control edges per node based on problem size
    target_edges = max(4, min(int(np.log2(n) * 2.5), n - 1))
    sparsified = np.zeros_like(heuristic)
    
    for i in range(n):
        row_heuristics = heuristic[i].copy()
        row_heuristics[i] = -np.inf  # Exclude self
        
        mst_neighbors = np.where(mst_mask[i])[0]
        additional_needed = max(0, target_edges - len(mst_neighbors))
        
        if additional_needed > 0:
            # Identify non-MST candidate edges
            non_mst_mask = np.ones(n, dtype=bool)
            non_mst_mask[mst_neighbors] = False
            non_mst_mask[i] = False
            non_mst_indices = np.where(non_mst_mask)[0]
            
            if len(non_mst_indices) > 0:
                non_mst_values = row_heuristics[non_mst_indices]
                if len(non_mst_indices) > additional_needed:
                    # Select top additional_needed edges by heuristic value
                    threshold_idx = np.argpartition(-non_mst_values, additional_needed - 1)[additional_needed - 1]
                    threshold_val = non_mst_values[threshold_idx]
                    keep_non_mst = non_mst_indices[non_mst_values >= threshold_val]
                else:
                    keep_non_mst = non_mst_indices
            else:
                keep_non_mst = []
        else:
            keep_non_mst = []
        
        # Keep all MST edges and selected non-MST edges
        keep_indices = np.concatenate([mst_neighbors, keep_non_mst])
        sparsified[i, keep_indices] = heuristic[i, keep_indices]
    
    # Symmetrize by taking element-wise maximum
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 6. Connectivity assurance: repair isolated or poorly connected nodes
    for i in range(n):
        degree = np.count_nonzero(sparsified[i])
        if degree == 0:
            # Connect isolated node to its nearest neighbor
            nearest = np.argsort(distance_matrix[i])[1]
            sparsified[i, nearest] = sparsified[nearest, i] = 0.2
        elif degree == 1:
            # Node with single connection: add second nearest
            current_neighbor = np.nonzero(sparsified[i])[0][0]
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            dists[current_neighbor] = np.inf
            second_nearest = np.argmin(dists)
            sparsified[i, second_nearest] = sparsified[second_nearest, i] = 0.15
    
    # 7. Final processing: zero diagonal and normalize
    np.fill_diagonal(sparsified, 0)
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified /= max_val
    
    return sparsified
