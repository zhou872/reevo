import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.manifold import MDS
from sklearn.neighbors import KDTree
import warnings


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic for TSP edge selection with adaptive parameter tuning
    and enhanced local structure analysis.
    
    Key improvements over v1:
    1. Dynamic parameter adaptation based on instance characteristics
    2. Enhanced local connectivity modeling with adaptive neighborhood sizing
    3. Multi-level triangle quality assessment
    4. Improved sparsification with connectivity preservation
    5. Distance distribution-aware normalization
    
    Args:
        distance_matrix: Square symmetric distance matrix (n x n)
        
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Analyze instance characteristics for parameter adaptation
    instance_stats = _analyze_instance(distance_matrix, n)
    
    # Base heuristic: reciprocal of distances with adaptive scaling
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_base = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_base, 0)
    
    # Adaptive spatial clustering with instance-aware parameters
    spatial_factor = _adaptive_spatial_clustering(
        distance_matrix, n, instance_stats
    )
    
    # Enhanced MST-based heuristics
    mst_factor = _enhanced_mst_heuristics(distance_matrix, n, instance_stats)
    
    # Local connectivity modeling with adaptive neighborhoods
    local_factor = _adaptive_local_modeling(distance_matrix, n, instance_stats)
    
    # Combine factors with instance-aware weights
    combined = _weighted_combination(
        reciprocal_base, spatial_factor, mst_factor, local_factor,
        distance_matrix, n, instance_stats
    )
    
    # Intelligent sparsification with connectivity preservation
    sparsified = _connectivity_preserving_sparsification(
        combined, distance_matrix, n, instance_stats
    )
    
    # Multi-level triangle quality enhancement
    triangle_enhanced = _multi_level_triangle_enhancement(
        sparsified, distance_matrix, n, instance_stats
    )
    
    # Instance-aware normalization
    normalized = _instance_aware_normalization(triangle_enhanced, distance_matrix)
    
    np.fill_diagonal(normalized, 0)
    return normalized


def _analyze_instance(distance_matrix: np.ndarray, n: int) -> dict:
    """Analyze instance characteristics for adaptive parameter tuning."""
    stats = {}
    
    # Extract upper triangle distances
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    # Basic statistics
    stats['mean_dist'] = np.mean(distances)
    stats['std_dist'] = np.std(distances)
    stats['cv_dist'] = stats['std_dist'] / (stats['mean_dist'] + 1e-12)
    
    # Percentiles for adaptive thresholding
    stats['p25'] = np.percentile(distances, 25)
    stats['p50'] = np.percentile(distances, 50)
    stats['p75'] = np.percentile(distances, 75)
    
    # Instance size categories
    if n < 50:
        stats['size_category'] = 'small'
    elif n < 500:
        stats['size_category'] = 'medium'
    else:
        stats['size_category'] = 'large'
    
    # Estimate clustering tendency
    from scipy.spatial.distance import pdist
    if n <= 1000:
        coords = _fast_mds_approx(distance_matrix, n)
        if coords.shape[1] > 1:
            coord_dists = pdist(coords)
            stats['spatial_compactness'] = np.std(coord_dists) / (np.mean(coord_dists) + 1e-12)
        else:
            stats['spatial_compactness'] = 1.0
    else:
        stats['spatial_compactness'] = 1.0
    
    # Edge density estimation (for sparsification)
    mst = minimum_spanning_tree(distance_matrix)
    mst_avg_dist = np.mean(mst.data) if len(mst.data) > 0 else stats['mean_dist']
    stats['mst_ratio'] = mst_avg_dist / (stats['mean_dist'] + 1e-12)
    
    return stats


def _fast_mds_approx(distance_matrix: np.ndarray, n: int, max_components: int = 3) -> np.ndarray:
    """Fast MDS approximation with adaptive parameters."""
    if n <= 2:
        return np.zeros((n, 1))
    
    n_components = min(max_components, n - 1)
    
    try:
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=n_components,
                dissimilarity='precomputed',
                random_state=42,
                n_init=1,
                max_iter=30,
                eps=0.1
            )
            return mds.fit_transform(distance_matrix)
    except:
        # Fallback: use landmark-based approximation
        n_landmarks = min(10, n)
        landmarks = np.random.choice(n, n_landmarks, replace=False)
        coords = distance_matrix[:, landmarks]
        # Simple centering
        return coords - np.mean(coords, axis=0)


def _adaptive_spatial_clustering(
    distance_matrix: np.ndarray, 
    n: int, 
    stats: dict
) -> np.ndarray:
    """Adaptive spatial clustering with instance-aware parameters."""
    spatial_factor = np.ones_like(distance_matrix)
    
    # Choose method based on instance size
    if stats['size_category'] == 'small':
        # Exact MDS for small instances
        coords = _fast_mds_approx(distance_matrix, n, min(3, n-1))
        k = min(max(5, n // 10), 15)
    elif stats['size_category'] == 'medium':
        # Balanced approach
        coords = _fast_mds_approx(distance_matrix, n, min(3, n-1))
        k = min(max(3, int(np.log2(n)) + 2), 20)
    else:
        # Landmark-based for large instances
        n_landmarks = min(50, max(10, n // 20))
        landmarks = np.random.choice(n, n_landmarks, replace=False)
        coords = distance_matrix[:, landmarks]
        k = min(max(3, int(np.log10(n)) * 2), 25)
    
    if coords.shape[1] == 0:
        return spatial_factor
    
    # Build KD-tree for efficient queries
    try:
        tree = KDTree(coords)
        
        # Adaptive neighborhood size based on spatial compactness
        base_k = k
        if stats['spatial_compactness'] > 1.5:
            # More dispersed, increase neighborhood
            adjusted_k = min(int(base_k * 1.5), n-1)
        elif stats['spatial_compactness'] < 0.5:
            # More compact, decrease neighborhood
            adjusted_k = max(int(base_k * 0.7), 3)
        else:
            adjusted_k = base_k
        
        for i in range(n):
            # Query for neighbors
            if n <= 1000:
                distances, indices = tree.query(
                    coords[i].reshape(1, -1), 
                    k=min(adjusted_k + 1, n)
                )
                indices = indices[0]
                if distances[0][0] < 1e-12:  # Skip self if present
                    indices = indices[1:adjusted_k+1]
                else:
                    indices = indices[:adjusted_k]
            else:
                # For very large instances, use approximate method
                dists = np.linalg.norm(coords - coords[i], axis=1)
                indices = np.argpartition(dists, adjusted_k)[:adjusted_k]
                indices = indices[indices != i]
            
            # Assign weights based on spatial proximity
            if len(indices) > 0:
                spatial_distances = distance_matrix[i, indices]
                median_spatial = np.median(spatial_distances)
                weights = 2.0 * np.exp(-spatial_distances / (median_spatial + 1e-12))
                spatial_factor[i, indices] = np.maximum(spatial_factor[i, indices], weights)
    except:
        # Fallback to distance-based clustering
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            k_neighbors = min(k, n-1)
            indices = np.argpartition(dists, k_neighbors)[:k_neighbors]
            spatial_factor[i, indices] = 1.5
    
    # Symmetrize
    spatial_factor = np.maximum(spatial_factor, spatial_factor.T)
    return spatial_factor


def _enhanced_mst_heuristics(
    distance_matrix: np.ndarray, 
    n: int, 
    stats: dict
) -> np.ndarray:
    """Enhanced MST-based heuristics with extensions."""
    mst_factor = np.ones_like(distance_matrix)
    
    # Compute MST
    mst = minimum_spanning_tree(distance_matrix)
    mst_binary = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Strong boost for MST edges
    mst_factor[mst_binary] = 2.0 + 0.5 * np.tanh(1 - stats['mst_ratio'])
    
    # Extended neighborhood of MST edges
    # Look at neighbors of MST-connected nodes
    for i in range(n):
        mst_neighbors = np.where(mst_binary[i])[0]
        if len(mst_neighbors) == 0:
            continue
            
        # For each MST neighbor, consider its close connections
        for neighbor in mst_neighbors:
            # Find k closest nodes to this neighbor (excluding i and already in MST)
            dists = distance_matrix[neighbor].copy()
            dists[neighbor] = np.inf
            dists[i] = np.inf
            
            k_extend = min(3, n - 2)
            if k_extend > 0:
                close_indices = np.argpartition(dists, k_extend)[:k_extend]
                for j in close_indices:
                    if not mst_binary[i, j] and mst_factor[i, j] < 1.5:
                        # Adaptive weight based on distance ratio
                        d_ij = distance_matrix[i, j]
                        d_in = distance_matrix[i, neighbor]
                        d_nj = distance_matrix[neighbor, j]
                        
                        # Good if edge completes a short triangle
                        triangle_quality = (d_in + d_nj) / (2 * d_ij + 1e-12)
                        weight = 1.3 + 0.4 * np.exp(-max(0, triangle_quality - 1))
                        mst_factor[i, j] = mst_factor[j, i] = max(mst_factor[i, j], weight)
    
    return mst_factor


def _adaptive_local_modeling(
    distance_matrix: np.ndarray, 
    n: int, 
    stats: dict
) -> np.ndarray:
    """Adaptive local connectivity modeling."""
    local_factor = np.ones_like(distance_matrix)
    
    # Adaptive neighborhood size based on instance characteristics
    if stats['size_category'] == 'small':
        k_local = min(8, n - 1)
    elif stats['size_category'] == 'medium':
        k_local = min(6, max(3, n // 30))
    else:
        k_local = min(5, max(2, n // 50))
    
    # Precompute local statistics for each node
    local_stats = np.zeros((n, 2))  # mean, std of k nearest distances
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k_local < len(dists):
            nn_dists = np.partition(dists, k_local)[:k_local]
        else:
            nn_dists = dists[dists < np.inf]
        
        if len(nn_dists) > 0:
            local_stats[i, 0] = np.mean(nn_dists)
            local_stats[i, 1] = np.std(nn_dists) + 1e-12
    
    # Compute local factor matrix
    for i in range(n):
        for j in range(i + 1, n):
            # Similarity in local neighborhood characteristics
            mean_i, std_i = local_stats[i]
            mean_j, std_j = local_stats[j]
            
            if std_i > 0 and std_j > 0:
                # Nodes with similar local densities are more likely connected
                density_sim = 1.0 / (1.0 + abs(mean_i - mean_j) / (std_i + std_j))
                
                # Distance relative to local means
                d_ij = distance_matrix[i, j]
                avg_local_mean = (mean_i + mean_j) / 2
                dist_ratio = d_ij / (avg_local_mean + 1e-12)
                
                # Combined factor
                if dist_ratio < 2.0:  # Only boost reasonably short edges
                    factor = density_sim * np.exp(-0.8 * dist_ratio)
                    local_factor[i, j] = local_factor[j, i] = max(0.5, factor)
    
    return local_factor


def _weighted_combination(
    reciprocal: np.ndarray,
    spatial: np.ndarray,
    mst: np.ndarray,
    local: np.ndarray,
    distance_matrix: np.ndarray,
    n: int,
    stats: dict
) -> np.ndarray:
    """Weighted combination of heuristic factors."""
    # Base combination (multiplicative)
    combined = reciprocal * spatial * mst * local
    
    # Apply distance-based envelope
    if stats['cv_dist'] > 0:
        # Use instance-specific scaling
        scale = stats['p50']  # Median distance as scaling factor
        dist_envelope = np.exp(-distance_matrix / (scale + 1e-12))
        combined *= (1.0 + 0.5 * dist_envelope)
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    # Zero diagonal
    np.fill_diagonal(combined, 0)
    
    return combined


def _connectivity_preserving_sparsification(
    heuristics: np.ndarray,
    distance_matrix: np.ndarray,
    n: int,
    stats: dict
) -> np.ndarray:
    """Intelligent sparsification that preserves connectivity."""
    if n <= 10:
        return heuristics.copy()
    
    # Adaptive sparsity based on instance size and characteristics
    if stats['size_category'] == 'small':
        keep_ratio = 0.4
    elif stats['size_category'] == 'medium':
        keep_ratio = 0.25
    else:
        keep_ratio = 0.15
    
    # Ensure minimum edges per node
    min_edges = max(3, int(np.log2(n)) + 1)
    max_edges = min(n - 1, int(keep_ratio * n))
    target_edges = max(min_edges, min(max_edges, int(0.3 * n)))
    
    sparsified = np.zeros_like(heuristics)
    
    # First, include MST edges for connectivity
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0) | (mst.toarray().T > 0)
    sparsified[mst_edges] = heuristics[mst_edges]
    
    # For each node, add promising non-MST edges
    for i in range(n):
        current_edges = np.count_nonzero(sparsified[i])
        needed = max(0, target_edges - current_edges)
        
        if needed > 0:
            # Consider non-MST edges with high heuristic values
            non_mst_mask = ~mst_edges[i] & (np.arange(n) != i)
            non_mst_indices = np.where(non_mst_mask)[0]
            
            if len(non_mst_indices) > 0:
                scores = heuristics[i, non_mst_indices]
                
                # Also consider distance to avoid very long edges
                dists = distance_matrix[i, non_mst_indices]
                dist_scores = 1 / (dists + 1e-12)
                
                # Combined score
                combined = scores * (1 + 0.2 * dist_scores / np.max(dist_scores))
                
                # Select top edges
                n_select = min(needed, len(non_mst_indices))
                if n_select > 0:
                    top_idx = np.argpartition(-combined, n_select)[:n_select]
                    selected = non_mst_indices[top_idx]
                    sparsified[i, selected] = heuristics[i, selected]
                    sparsified[selected, i] = heuristics[selected, i]
    
    # Final connectivity check and repair
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        node = stack.pop()
        neighbors = np.where(sparsified[node] > 0)[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    if not np.all(visited):
        # Connect disconnected components via shortest edges
        components = []
        remaining = np.where(~visited)[0]
        
        while len(remaining) > 0:
            start = remaining[0]
            comp = []
            stack = [start]
            visited_comp = np.zeros(n, dtype=bool)
            visited_comp[start] = True
            
            while stack:
                node = stack.pop()
                comp.append(node)
                neighbors = np.where(sparsified[node] > 0)[0]
                for neighbor in neighbors:
                    if not visited_comp[neighbor]:
                        visited_comp[neighbor] = True
                        stack.append(neighbor)
            
            components.append(comp)
            remaining = [node for node in remaining if not visited_comp[node]]
        
        # Connect components with minimum distance edges
        for comp in components[1:]:
            min_dist = np.inf
            best_pair = (0, 0)
            
            for i in components[0]:
                for j in comp:
                    if distance_matrix[i, j] < min_dist:
                        min_dist = distance_matrix[i, j]
                        best_pair = (i, j)
            
            i, j = best_pair
            sparsified[i, j] = sparsified[j, i] = heuristics[i, j]
    
    return sparsified


def _multi_level_triangle_enhancement(
    heuristics: np.ndarray,
    distance_matrix: np.ndarray,
    n: int,
    stats: dict
) -> np.ndarray:
    """Multi-level triangle quality assessment and enhancement."""
    enhanced = heuristics.copy()
    
    if n < 4:
        return enhanced
    
    # Determine sampling strategy based on instance size
    if stats['size_category'] == 'small':
        # Process all edges
        edges = np.argwhere(np.triu(heuristics, k=1) > 0)
        sample_ratio = 1.0
    elif stats['size_category'] == 'medium':
        # Sample edges
        sample_ratio = 0.7
        edges = np.argwhere(np.triu(heuristics, k=1) > 0)
        if len(edges) > 1000:
            sample_size = min(1000, int(len(edges) * sample_ratio))
            idx = np.random.choice(len(edges), sample_size, replace=False)
            edges = edges[idx]
    else:
        # Sparse sampling for large instances
        sample_ratio = 0.3
        edges = np.argwhere(np.triu(heuristics, k=1) > 0)
        if len(edges) > 500:
            sample_size = min(500, int(len(edges) * sample_ratio))
            idx = np.random.choice(len(edges), sample_size, replace=False)
            edges = edges[idx]
    
    # Precompute binary adjacency for faster common neighbor detection
    binary_adj = (heuristics > 0).astype(int)
    
    for edge in edges:
        i, j = edge
        if heuristics[i, j] == 0:
            continue
        
        # Find common neighbors efficiently
        neighbors_i = set(np.where(binary_adj[i])[0])
        neighbors_j = set(np.where(binary_adj[j])[0])
        common = neighbors_i.intersection(neighbors_j)
        
        if not common:
            continue
        
        # Compute triangle quality metrics
        d_ij = distance_matrix[i, j]
        quality_sum = 0
        count = 0
        
        for k in list(common)[:10]:  # Limit to first 10 for efficiency
            d_ik = distance_matrix[i, k]
            d_jk = distance_matrix[j, k]
            
            # Triangle inequality satisfaction
            triangle_sum = d_ij + d_ik + d_jk
            min_perimeter = 2 * max(d_ij, d_ik, d_jk)
            
            if triangle_sum > min_perimeter:  # Valid triangle
                # Quality metric: how compact is the triangle
                perimeter = triangle_sum
                avg_side = perimeter / 3
                
                # Boost factor inversely proportional to perimeter
                # Normalize by instance statistics
                boost = 0.2 * stats['p50'] / (avg_side + 1e-12)
                quality_sum += min(boost, 0.5)  # Cap individual contribution
                count += 1
        
        if count > 0:
            avg_boost = quality_sum / count
            # Apply boost with diminishing returns
            final_boost = 1.0 + min(avg_boost, 0.3 * np.log1p(count))
            enhanced[i, j] = enhanced[j, i] = heuristics[i, j] * final_boost
    
    return enhanced


def _instance_aware_normalization(
    heuristics: np.ndarray,
    distance_matrix: np.ndarray
) -> np.ndarray:
    """Instance-aware normalization of heuristic matrix."""
    n = heuristics.shape[0]
    
    # Zero diagonal
    np.fill_diagonal(heuristics, 0)
    
    # Handle all-zero case
    if np.max(heuristics) <= 0:
        return heuristics
    
    # Scale based on distance distribution
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    heuristics_vals = heuristics[triu_indices]
    
    # Winsorize extreme values
    p95 = np.percentile(heuristics_vals[heuristics_vals > 0], 95)
    if p95 > 0:
        heuristics = np.minimum(heuristics, p95)
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
