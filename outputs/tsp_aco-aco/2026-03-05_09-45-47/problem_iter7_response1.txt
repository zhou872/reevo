import numpy as np
import warnings
from sklearn.manifold import MDS
from sklearn.neighbors import KDTree
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with scalable spatial clustering,
    adaptive sparsification, and efficient triangle boosting via sampling.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Base heuristic: reciprocal of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Compute spatial clustering factor with scalable approximation
    def _compute_spatial_clustering(dist_mat, n):
        spatial_promising = np.zeros_like(dist_mat)
        
        if n <= 1000:
            # Use MDS for accurate coordinates when n is manageable
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                mds = MDS(
                    n_components=min(3, n-1),
                    dissimilarity='precomputed',
                    random_state=42,
                    n_init=1,
                    max_iter=100
                )
                coords = mds.fit_transform(dist_mat)
        else:
            # Use landmark-based approximation for large n
            landmarks = min(50, n // 20)
            landmark_indices = np.random.choice(n, landmarks, replace=False)
            coords = dist_mat[:, landmark_indices]  # n x landmarks matrix
        
        # Build KD-tree for efficient neighbor queries
        tree = KDTree(coords)
        k = max(3, min(int(np.log2(n)) + 2, n - 1))
        
        for i in range(n):
            distances, indices = tree.query(coords[i].reshape(1, -1), k=k+1)
            indices = indices[0]
            distances = distances[0]
            # Exclude self if it's the first result
            if distances[0] == 0 and indices[0] == i:
                indices = indices[1:]
            else:
                indices = indices[:k]
            spatial_promising[i, indices] = 1.0
        
        return spatial_promising
    
    spatial_promising = _compute_spatial_clustering(distance_matrix, n)
    
    # Compute promises based on Minimum Spanning Tree
    def _compute_mst_promises(dist_mat):
        mst = minimum_spanning_tree(dist_mat)
        mst_edges = (mst.toarray() > 0).astype(float)
        mst_promising = mst_edges + mst_edges.T  # Make symmetric
        
        n_loc = dist_mat.shape[0]
        for i in range(n_loc):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            closest = np.argsort(dists)[:3]  # Consider second and third closest
            for j in closest:
                if mst_promising[i, j] == 0:
                    mst_promising[i, j] = 0.5  # Bonus for near-MST edges
        
        return mst_promising
    
    mst_promising = _compute_mst_promises(distance_matrix)
    
    # Compute local connectivity factors based on neighborhood statistics
    def _compute_local_factors(dist_mat, n):
        m = min(5, n - 1)
        local_stats = np.zeros((n, 2))  # [mean_dist, std_dist]
        
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            sorted_dists = np.sort(dists)[:m]
            local_stats[i, 0] = np.mean(sorted_dists)
            local_stats[i, 1] = np.std(sorted_dists) + 1e-12
        
        local_factors = np.ones_like(dist_mat)
        
        for i in range(n):
            for j in range(i+1, n):
                # Similarity based on local statistics
                stat_similarity = np.exp(
                    -np.abs(local_stats[i, 0] - local_stats[j, 0]) /
                    (local_stats[i, 1] + local_stats[j, 1])
                )
                mean_dist = (local_stats[i, 0] + local_stats[j, 0]) / 2
                dist_ratio = dist_mat[i, j] / (mean_dist + 1e-12)
                local_factors[i, j] = local_factors[j, i] = (
                    stat_similarity * np.exp(-0.5 * dist_ratio**2)
                )
        
        return local_factors
    
    local_factors = _compute_local_factors(distance_matrix, n)
    
    # Combine all heuristic factors efficiently
    def _combine_factors(reciprocal_weights, spatial_promising, mst_promising,
                         local_factors, dist_mat):
        base_scores = reciprocal_weights.copy()
        
        spatial_factor = np.where(spatial_promising > 0, 2.0, 0.5)
        mst_factor = np.where(mst_promising > 0,
                             np.where(mst_promising == 2, 3.0, 1.5),
                             0.8)
        
        combined_scores = base_scores * spatial_factor * mst_factor * local_factors
        
        # Logarithmic boost for very short edges
        with np.errstate(divide='ignore', invalid='ignore'):
            log_boost = 1 + 0.3 * np.log(1 + 1 / (base_scores + 1e-12))
            log_boost = np.nan_to_num(log_boost, nan=1.0, posinf=1.0)
        
        heuristics_matrix = combined_scores * log_boost
        heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2  # Ensure symmetry
        return heuristics_matrix
    
    heuristics_matrix = _combine_factors(reciprocal_weights, spatial_promising,
                                         mst_promising, local_factors, distance_matrix)
    
    # Adaptive sparsification: keep top promising edges per node
    def _adaptive_sparsification(heuristics_matrix, n):
        sparsified = np.zeros_like(heuristics_matrix)
        
        if n <= 10:
            return heuristics_matrix.copy()
        
        # Adaptive keep ratio based on log10(n)
        base_keep_ratio = min(0.4, max(0.15, np.log10(n) / 10))
        
        for i in range(n):
            row = heuristics_matrix[i].copy()
            row[i] = -np.inf  # Ignore diagonal
            
            num_to_keep = max(3, int(base_keep_ratio * n))
            num_to_keep = min(num_to_keep, n - 1)
            
            if np.any(row > 0):
                top_indices = np.argpartition(-row, num_to_keep)[:num_to_keep]
                threshold = np.min(row[top_indices])
                mask = row >= threshold
                sparsified[i] = heuristics_matrix[i] * mask
        
        # Symmetrize by taking element-wise maximum
        sparsified = np.maximum(sparsified, sparsified.T)
        return sparsified
    
    sparsified = _adaptive_sparsification(heuristics_matrix, n)
    
    # Triangle boosting with sampling for large instances
    def _triangle_boosting(heuristics_matrix):
        n_boost = heuristics_matrix.shape[0]
        boosted = heuristics_matrix.copy()
        
        if n_boost > 500:
            # Use sampling to reduce computation
            sample_size = min(100, n_boost)
            sampled_nodes = np.random.choice(n_boost, sample_size, replace=False)
            
            for i in sampled_nodes:
                neighbors_i = set(np.where(heuristics_matrix[i] > 0)[0])
                for j in neighbors_i:
                    if j <= i:
                        continue
                    neighbors_j = set(np.where(heuristics_matrix[j] > 0)[0])
                    common = len(neighbors_i & neighbors_j)
                    if common > 0:
                        boost = 1 + 0.15 * common
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
        else:
            # Exact computation for smaller graphs
            binary_mask = (heuristics_matrix > 0).astype(float)
            common_neighbors = binary_mask @ binary_mask.T  # Matrix multiplication for efficiency
            
            for i in range(n_boost):
                for j in range(i+1, n_boost):
                    if heuristics_matrix[i, j] > 0:
                        common = common_neighbors[i, j]
                        if common > 0:
                            boost = 1 + 0.2 * min(common, 5)  # Cap the boost
                            boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
        
        return boosted
    
    final_heuristics = _triangle_boosting(sparsified)
    
    # Normalize to [0, 1] range for consistency
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics
