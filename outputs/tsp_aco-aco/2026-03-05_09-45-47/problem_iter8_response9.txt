import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import KDTree
import warnings
from sklearn.manifold import MDS

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic for TSP edge selection with better vectorization,
    dynamic weight adaptation, and enhanced scalability.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # Base heuristic: reciprocal of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Factor 1: Spatial clustering with adaptive method selection
    spatial_promising = _compute_spatial_clustering_v3(distance_matrix, n)
    
    # Factor 2: Minimum Spanning Tree edges
    mst_promising = _compute_mst_promises_v3(distance_matrix)
    
    # Factor 3: Local connectivity with dynamic neighborhood size
    local_factors = _compute_local_factors_v3(distance_matrix, n)
    
    # Combine factors with adaptive weights based on graph density
    heuristics_matrix = _combine_factors_v3(
        reciprocal_weights, spatial_promising, mst_promising, 
        local_factors, distance_matrix, n
    )
    
    # Adaptive sparsification with log-scaling
    sparsified = _adaptive_sparsification_v3(heuristics_matrix, n)
    
    # Triangle boosting for dense clusters
    final_heuristics = _triangle_boosting_v3(sparsified, n)
    
    # Normalize to [0, 1] range
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics

def _compute_spatial_clustering_v3(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute spatial clustering factor with adaptive method selection."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    # Adaptive k: log2(n) with bounds
    k = max(3, min(int(np.log2(n)) + 2, n - 1))
    
    if n <= 500:
        # Use KDTree for moderate-sized graphs
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(3, n-1),
                dissimilarity='precomputed',
                random_state=42,
                n_init=1,
                max_iter=50
            )
            coords = mds.fit_transform(distance_matrix)
            tree = KDTree(coords)
            
            for i in range(n):
                distances, indices = tree.query(coords[i].reshape(1, -1), k=k+1)
                indices = indices[0][1:]  # Exclude self
                spatial_promising[i, indices[:k]] = 1.0
    else:
        # Use fast approximate method for large graphs
        # Sample landmarks for faster computation
        landmarks = min(50, n // 50)
        if landmarks >= 5:
            landmark_indices = np.random.choice(n, landmarks, replace=False)
            landmark_dists = distance_matrix[landmark_indices]
            
            # Compute approximate distances to landmarks
            coords = distance_matrix[:, landmark_indices]
            
            # Find k nearest neighbors based on landmark distances
            for i in range(n):
                # Compute L2 distances in landmark space
                landmark_diffs = coords - coords[i]
                landmark_distances = np.sqrt(np.sum(landmark_diffs**2, axis=1))
                landmark_distances[i] = np.inf
                
                indices = np.argpartition(landmark_distances, k)[:k]
                spatial_promising[i, indices] = 1.0
        else:
            # Fallback to direct distance matrix for very small landmark sets
            for i in range(n):
                dists = distance_matrix[i].copy()
                dists[i] = np.inf
                indices = np.argpartition(dists, k)[:k]
                spatial_promising[i, indices] = 1.0
    
    return spatial_promising

def _compute_mst_promises_v3(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute MST promises with improved edge weighting."""
    n = distance_matrix.shape[0]
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    # Make symmetric
    mst_promising = mst_edges + mst_edges.T
    
    # Add bonus for close non-MST connections
    # Use vectorized operations for efficiency
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Consider top m closest nodes not in MST
        m = min(4, n - 1)
        closest = np.argsort(dists)[:m]
        
        for j in closest:
            if mst_promising[i, j] == 0:
                # Weight decays with rank
                rank = np.where(closest == j)[0][0] + 1
                mst_promising[i, j] = 0.6 / rank
    
    return mst_promising

def _compute_local_factors_v3(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute local connectivity factors with vectorized operations."""
    # Adaptive neighborhood size based on n
    m = min(max(3, int(np.log2(n))), n - 1)
    
    # Preallocate arrays
    local_stats = np.zeros((n, 2))
    
    # Compute local statistics for each node
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Get m smallest distances
        if m < n - 1:
            sorted_dists = np.partition(dists, m)[:m]
            sorted_dists.sort()
        else:
            sorted_dists = np.sort(dists)
        
        local_stats[i, 0] = np.mean(sorted_dists)
        local_stats[i, 1] = np.std(sorted_dists) + 1e-12
    
    # Vectorized computation of local factors
    mean_dists_i = local_stats[:, 0].reshape(-1, 1)
    mean_dists_j = local_stats[:, 0].reshape(1, -1)
    std_dists_i = local_stats[:, 1].reshape(-1, 1)
    std_dists_j = local_stats[:, 1].reshape(1, -1)
    
    # Similarity based on local statistics
    stat_similarity = np.exp(
        -np.abs(mean_dists_i - mean_dists_j) / 
        (std_dists_i + std_dists_j)
    )
    
    # Distance ratio factor
    mean_pair_dists = (mean_dists_i + mean_dists_j) / 2
    with np.errstate(divide='ignore', invalid='ignore'):
        dist_ratio = distance_matrix / (mean_pair_dists + 1e-12)
    
    # Combined local factors
    local_factors = stat_similarity * np.exp(-0.4 * dist_ratio**2)
    
    # Ensure symmetry and zero diagonal
    local_factors = (local_factors + local_factors.T) / 2
    np.fill_diagonal(local_factors, 0)
    
    return local_factors

def _combine_factors_v3(reciprocal_weights: np.ndarray,
                       spatial_promising: np.ndarray,
                       mst_promising: np.ndarray,
                       local_factors: np.ndarray,
                       distance_matrix: np.ndarray,
                       n: int) -> np.ndarray:
    """Combine factors with weights adapted to graph density."""
    # Compute graph density proxy
    avg_distance = np.mean(distance_matrix[np.triu_indices(n, k=1)])
    density_factor = 1.0 / (1.0 + 0.1 * avg_distance)  # Higher for denser graphs
    
    # Adjust weights based on density
    spatial_weight = 1.5 + 0.5 * density_factor
    non_spatial_weight = 1.0 - 0.3 * density_factor
    
    mst_weight_high = 2.0 + density_factor
    mst_weight_low = 1.2 + 0.3 * density_factor
    non_mst_weight = 0.8 - 0.1 * density_factor
    
    # Apply factors
    spatial_factor = np.where(spatial_promising > 0, 
                             spatial_weight, 
                             non_spatial_weight)
    
    mst_factor = np.where(mst_promising >= 2,  # Direct MST edge
                         mst_weight_high,
                         np.where(mst_promising > 0,  # Indirect MST edge
                                 mst_weight_low,
                                 non_mst_weight))
    
    # Combine all factors
    combined = reciprocal_weights * spatial_factor * mst_factor * local_factors
    
    # Additional boost for very short edges (diminishing returns)
    with np.errstate(divide='ignore', invalid='ignore'):
        short_edge_boost = 1 + 0.25 * np.exp(-reciprocal_weights / 10)
    
    combined = combined * short_edge_boost
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    return combined

def _adaptive_sparsification_v3(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Adaptive sparsification with logarithmic scaling."""
    if n <= 10:
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Base keep ratio: decreases logarithmically with n
    log_n = np.log2(n)
    base_keep_ratio = 0.3 * (1.0 - 0.1 * log_n)
    base_keep_ratio = np.clip(base_keep_ratio, 0.15, 0.35)
    
    # Vectorized top-k selection
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        # Determine number of edges to keep
        num_to_keep = max(3, int(base_keep_ratio * n))
        num_to_keep = min(num_to_keep, n - 1)
        
        if np.any(row > 0):
            # Find threshold for top k edges
            if num_to_keep < len(row):
                threshold = np.partition(-row, num_to_keep)[num_to_keep]
                threshold = -threshold
            else:
                threshold = np.min(row[row > -np.inf])
            
            # Keep edges above threshold
            mask = row >= threshold
            sparsified[i] = heuristics_matrix[i] * mask
    
    # Ensure symmetry by taking maximum
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified

def _triangle_boosting_v3(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Triangle boosting with efficient vectorized implementation."""
    boosted = heuristics_matrix.copy()
    
    if n <= 2:
        return boosted
    
    # Create binary mask of existing edges
    binary_mask = (heuristics_matrix > 0).astype(float)
    
    # Count common neighbors using matrix multiplication
    # This is O(n^3) theoretically but fast in practice due to BLAS optimization
    common_neighbors = binary_mask @ binary_mask.T
    
    # Only boost edges that exist
    edge_mask = (heuristics_matrix > 0) & (np.triu(np.ones_like(heuristics_matrix), 1) > 0)
    
    if n > 1000:
        # For very large graphs, sample edges
        edge_indices = np.where(edge_mask)
        if len(edge_indices[0]) > 10000:
            sample_idx = np.random.choice(len(edge_indices[0]), 10000, replace=False)
            i_idx = edge_indices[0][sample_idx]
            j_idx = edge_indices[1][sample_idx]
        else:
            i_idx, j_idx = edge_indices
    else:
        i_idx, j_idx = np.where(edge_mask)
    
    # Apply boost based on common neighbors
    for i, j in zip(i_idx, j_idx):
        common = common_neighbors[i, j]
        if common > 0:
            # Diminishing returns: sqrt scaling
            boost = 1 + 0.18 * np.sqrt(min(common, 9))
            boosted_val = heuristics_matrix[i, j] * boost
            boosted[i, j] = boosted_val
            boosted[j, i] = boosted_val
    
    return boosted
