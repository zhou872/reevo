import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    if n <= 2:
        # For trivial cases, return inverse distance
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristics = 1 / distance_matrix
            heuristics[np.isinf(heuristics)] = 0
            np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # 1. Compute basic inverse distance (closer nodes are more promising)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        np.fill_diagonal(inv_dist, 0)
    
    # 2. Compute rank-based weights for each node
    rank_weights = np.zeros_like(distance_matrix)
    
    # Determine adaptive sparsity level
    k = max(3, min(10, n // 4))  # Number of nearest neighbors to consider
    
    for i in range(n):
        # Get distances from node i to all others
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf  # Ignore self
        
        # Sort distances and get indices
        sorted_indices = np.argsort(dists_i)
        
        # Assign weights based on rank: 1/rank for top k, 0 for others
        for rank, j in enumerate(sorted_indices[:k]):
            if i != j:
                # Higher weight for better ranks (closer nodes)
                rank_weights[i, j] = 1.0 / (rank + 1)
    
    # 3. Compute combined promise score
    promise = np.zeros_like(distance_matrix)
    
    # Factor 1: Inverse distance (normalized)
    inv_dist_norm = inv_dist / (np.max(inv_dist) + 1e-10)
    
    # Factor 2: Symmetric rank combination
    sym_rank = (rank_weights + rank_weights.T) / 2
    
    # Factor 3: Node centrality adjustment
    # Nodes with smaller average distance to others are more central
    node_centrality = 1 / (np.mean(distance_matrix, axis=1) + 1e-10)
    centrality_matrix = np.outer(node_centrality, node_centrality)
    centrality_norm = centrality_matrix / (np.max(centrality_matrix) + 1e-10)
    
    # Combine all factors
    for i in range(n):
        for j in range(i+1, n):
            # Base promise is product of inverse distance and rank
            base_promise = inv_dist_norm[i, j] * sym_rank[i, j]
            
            # Boost edges connecting peripheral nodes to central nodes
            centrality_boost = 0.5 * centrality_norm[i, j]
            
            # Combine with weighted sum
            promise[i, j] = promise[j, i] = 0.7 * base_promise + 0.3 * centrality_boost
    
    # 4. Sparsification: Set unpromising edges to zero
    # Keep edges that are in top percentile or have promise above threshold
    if n > 10:
        # For larger graphs, be more selective
        threshold = np.percentile(promise[promise > 0], 30)
    else:
        # For smaller graphs, keep more edges
        threshold = np.percentile(promise[promise > 0], 20)
    
    promise[promise < threshold] = 0
    
    # 5. Ensure symmetry and zero diagonal
    promise = (promise + promise.T) / 2  # Ensure exact symmetry
    np.fill_diagonal(promise, 0)
    
    return promise
