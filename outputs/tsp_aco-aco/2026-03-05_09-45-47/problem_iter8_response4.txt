import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import KDTree
from sklearn.manifold import MDS
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with adaptive learning,
    graph structure analysis, and efficient parameter optimization.
    
    Key improvements:
    1. Adaptive parameter tuning based on instance characteristics
    2. Graph structure learning via spectral embedding
    3. Efficient multi-strategy ensemble with dynamic weights
    4. Memory-efficient operations for large instances
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Analyze instance characteristics for adaptive parameter tuning
    def _analyze_instance(dist_mat, n):
        stats = {}
        
        # Basic statistics
        triu_indices = np.triu_indices(n, k=1)
        stats['avg_distance'] = np.mean(dist_mat[triu_indices])
        stats['std_distance'] = np.std(dist_mat[triu_indices])
        stats['cv_distance'] = stats['std_distance'] / (stats['avg_distance'] + 1e-12)
        
        # Estimate graph structure
        m = min(5, n - 1)
        neighbor_stats = []
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            sorted_dists = np.sort(dists)[:m]
            neighbor_stats.append(sorted_dists[0])
        stats['clustering_tendency'] = np.std(neighbor_stats) / (np.mean(neighbor_stats) + 1e-12)
        
        # Determine instance type for parameter adaptation
        if n > 1000:
            stats['type'] = 'large'
        elif stats['cv_distance'] > 0.7:
            stats['type'] = 'irregular'
        elif stats['clustering_tendency'] < 0.3:
            stats['type'] = 'uniform'
        else:
            stats['type'] = 'clustered'
        
        return stats
    
    instance_stats = _analyze_instance(distance_matrix, n)
    
    # Base heuristic: adaptive reciprocal weighting
    def _compute_base_heuristic(dist_mat, n, stats):
        with np.errstate(divide='ignore', invalid='ignore'):
            # Use exponential decay for more robust weighting
            scale = stats['avg_distance']
            base_weights = np.exp(-dist_mat / scale)
            np.fill_diagonal(base_weights, 0)
        
        # Adaptive enhancement based on instance type
        if stats['type'] == 'irregular':
            # Boost short edges more aggressively
            p25 = np.percentile(dist_mat[dist_mat > 0], 25)
            short_mask = dist_mat < p25
            base_weights[short_mask] *= 1.5
        elif stats['type'] == 'clustered':
            # Enhance local connectivity
            for i in range(n):
                row = dist_mat[i].copy()
                row[i] = np.inf
                k_nearest = min(10, n - 1)
                nearest = np.argsort(row)[:k_nearest]
                base_weights[i, nearest] *= 1.2
        
        return base_weights
    
    base_weights = _compute_base_heuristic(distance_matrix, n, instance_stats)
    
    # Graph structure learning via efficient spectral embedding
    def _compute_structural_embedding(dist_mat, n, stats):
        structural_promising = np.zeros_like(dist_mat)
        
        if n <= 500:
            # Use MDS for accurate embedding
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                n_components = min(10, n - 1)
                mds = MDS(
                    n_components=n_components,
                    dissimilarity='precomputed',
                    random_state=42,
                    n_init=1,
                    max_iter=100
                )
                coords = mds.fit_transform(dist_mat)
        else:
            # Use landmark-based spectral approximation
            n_landmarks = min(100, max(30, int(np.log2(n) * 3)))
            landmark_indices = np.random.choice(n, n_landmarks, replace=False)
            
            # Use distances to landmarks as features
            coords = dist_mat[:, landmark_indices]
        
        # Build efficient spatial index
        tree = KDTree(coords)
        
        # Adaptive neighborhood size based on instance type
        if stats['type'] == 'clustered':
            k_neighbors = max(5, int(np.log2(n) * 1.5))
        elif stats['type'] == 'uniform':
            k_neighbors = max(3, int(np.log2(n)))
        else:
            k_neighbors = max(4, int(np.log2(n) * 1.2))
        
        k_neighbors = min(k_neighbors, n - 1)
        
        # Assign promises to nearest neighbors
        for i in range(n):
            distances, indices = tree.query(coords[i].reshape(1, -1), k=k_neighbors + 1)
            indices = indices[0]
            distances = distances[0]
            
            # Adjust weights based on distance rank
            for rank, j in enumerate(indices):
                if j != i:
                    # Exponential decay with rank
                    weight = np.exp(-0.2 * rank)
                    structural_promising[i, j] = weight
        
        return structural_promising
    
    structural_promising = _compute_structural_embedding(distance_matrix, n, instance_stats)
    
    # MST-based promises with extended connectivity analysis
    def _compute_mst_enhanced(dist_mat, n, stats):
        mst = minimum_spanning_tree(dist_mat)
        mst_edges = (mst.toarray() > 0).astype(float)
        
        mst_promising = np.zeros_like(dist_mat)
        
        # Primary MST edges get highest weight
        mst_promising[mst_edges > 0] = 2.0
        
        # Adaptive near-MST connectivity
        if stats['type'] == 'irregular':
            k_near = 5
        else:
            k_near = 3
        
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:k_near]
            
            for rank, j in enumerate(nearest):
                if mst_promising[i, j] == 0:
                    # Weight decays with rank
                    weight = 1.5 / (1.0 + 0.3 * rank)
                    mst_promising[i, j] = weight
        
        # Ensure symmetry
        mst_promising = np.maximum(mst_promising, mst_promising.T)
        
        return mst_promising
    
    mst_promising = _compute_mst_enhanced(distance_matrix, n, instance_stats)
    
    # Local connectivity analysis with adaptive neighborhood
    def _compute_local_connectivity(dist_mat, n, stats):
        # Adaptive neighborhood size
        if stats['type'] == 'clustered':
            k_local = min(8, max(4, int(np.log2(n) * 1.5)))
        else:
            k_local = min(6, max(3, int(np.log2(n))))
        
        local_factors = np.ones_like(dist_mat)
        
        # Precompute local statistics
        local_means = np.zeros(n)
        local_stds = np.zeros(n)
        
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            nearest_dists = np.partition(dists, k_local)[:k_local]
            local_means[i] = np.mean(nearest_dists)
            local_stds[i] = np.std(nearest_dists) + 1e-12
        
        # Vectorized computation of local similarity
        for i in range(n):
            # Mean similarity
            mean_diff = np.abs(local_means[i] - local_means)
            std_sum = local_stds[i] + local_stds
            mean_sim = np.exp(-mean_diff / std_sum)
            
            # Distance adaptation factor
            dist_factor = np.exp(-dist_mat[i, :] / (local_means[i] + 1e-12))
            
            # Combine factors
            local_factors[i, :] = mean_sim * dist_factor
        
        # Ensure symmetry
        local_factors = (local_factors + local_factors.T) / 2
        np.fill_diagonal(local_factors, 0)
        
        return local_factors
    
    local_factors = _compute_local_connectivity(distance_matrix, n, instance_stats)
    
    # Dynamic ensemble combination with learned weights
    def _ensemble_combination(base, structural, mst, local, dist_mat, stats):
        # Adaptive weights based on instance characteristics
        if stats['type'] == 'large':
            weights = {'base': 1.0, 'structural': 1.2, 'mst': 1.8, 'local': 1.1}
        elif stats['type'] == 'irregular':
            weights = {'base': 1.2, 'structural': 1.0, 'mst': 1.6, 'local': 1.3}
        elif stats['type'] == 'clustered':
            weights = {'base': 1.0, 'structural': 1.4, 'mst': 1.5, 'local': 1.4}
        else:  # uniform
            weights = {'base': 1.1, 'structural': 1.1, 'mst': 1.7, 'local': 1.2}
        
        # Apply weights
        structural_weighted = np.where(structural > 0,
                                     base * (1 + (weights['structural'] - 1) * structural),
                                     base * 0.9)
        
        mst_weighted = np.where(mst > 1.5,
                               structural_weighted * weights['mst'],
                               np.where(mst > 0,
                                       structural_weighted * 1.3,
                                       structural_weighted))
        
        local_weighted = mst_weighted * (1 + (weights['local'] - 1) * local)
        
        # Final adjustment with base weight
        combined = local_weighted * weights['base']
        
        return combined
    
    heuristics_matrix = _ensemble_combination(base_weights, structural_promising,
                                              mst_promising, local_factors,
                                              distance_matrix, instance_stats)
    
    # Adaptive sparsification with instance-aware thresholds
    def _adaptive_sparsification(heuristics_mat, n, stats):
        sparsified = heuristics_mat.copy()
        
        if n <= 20:
            return sparsified
        
        # Determine sparsity level based on instance type
        if stats['type'] == 'large':
            keep_ratio = 0.15
        elif stats['type'] == 'irregular':
            keep_ratio = 0.25
        else:
            keep_ratio = 0.2
        
        # Minimum edges to keep per node
        min_edges = max(3, int(np.log2(n)))
        
        for i in range(n):
            row = heuristics_mat[i].copy()
            row[i] = -np.inf
            
            num_to_keep = max(min_edges, int(keep_ratio * n))
            num_to_keep = min(num_to_keep, n - 1)
            
            if np.any(row > 0):
                top_indices = np.argpartition(-row, num_to_keep)[:num_to_keep]
                threshold = np.min(row[top_indices])
                mask = row >= threshold
                sparsified[i] = sparsified[i] * mask
        
        # Symmetrize
        sparsified = np.maximum(sparsified, sparsified.T)
        
        return sparsified
    
    sparsified = _adaptive_sparsification(heuristics_matrix, n, instance_stats)
    
    # Efficient triangle boosting with sampling for large graphs
    def _efficient_triangle_boost(heuristics_mat, dist_mat, n, stats):
        boosted = heuristics_mat.copy()
        
        if n > 1000:
            # Sampling-based approach for very large graphs
            sample_size = min(200, n)
            sampled_nodes = np.random.choice(n, sample_size, replace=False)
            
            for i in sampled_nodes:
                neighbors_i = np.where(heuristics_mat[i] > 0)[0]
                if len(neighbors_i) == 0:
                    continue
                
                for j in neighbors_i:
                    if j <= i:
                        continue
                    
                    # Find common neighbors efficiently
                    neighbors_j = np.where(heuristics_mat[j] > 0)[0]
                    common_neighbors = np.intersect1d(neighbors_i, neighbors_j)
                    
                    if len(common_neighbors) > 0:
                        # Quality-weighted boosting
                        avg_triangle_dist = (dist_mat[i, j] + 
                                           np.mean(dist_mat[i, common_neighbors]) +
                                           np.mean(dist_mat[j, common_neighbors])) / 3
                        
                        # Boost inversely proportional to triangle quality
                        dist_factor = np.exp(-avg_triangle_dist / stats['avg_distance'])
                        boost = 1 + 0.25 * np.tanh(len(common_neighbors) / 3.0) * dist_factor
                        
                        boosted[i, j] = boosted[j, i] = heuristics_mat[i, j] * boost
        else:
            # Exact computation for moderate-sized graphs
            binary_mask = (heuristics_mat > 0).astype(float)
            common_counts = binary_mask @ binary_mask.T  # Matrix multiplication
            
            for i in range(n):
                for j in range(i+1, n):
                    if heuristics_mat[i, j] > 0:
                        common = common_counts[i, j]
                        if common > 0:
                            # Adaptive boosting based on triangle count
                            if stats['type'] == 'clustered':
                                boost = 1 + 0.2 * min(common, 6)
                            else:
                                boost = 1 + 0.15 * min(common, 4)
                            
                            boosted[i, j] = boosted[j, i] = heuristics_mat[i, j] * boost
        
        return boosted
    
    boosted = _efficient_triangle_boost(sparsified, distance_matrix, n, instance_stats)
    
    # Final normalization and symmetry enforcement
    final_heuristics = boosted.copy()
    
    # Ensure symmetry
    final_heuristics = (final_heuristics + final_heuristics.T) / 2
    
    # Normalize to [0, 1] range
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    
    return final_heuristics
