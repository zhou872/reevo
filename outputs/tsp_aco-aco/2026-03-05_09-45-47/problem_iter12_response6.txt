import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.cluster import AgglomerativeClustering

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection incorporating:
    1. Hierarchical clustering for intra-cluster edge emphasis
    2. Adaptive parameter optimization based on matrix statistics
    3. Iterative refinement through simulated local search feedback
    4. Multi-level boosting strategy with dynamic weighting
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric n x n matrix of pairwise distances
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate more promising edges.
        Diagonal is zero, values are normalized to [0, 1].
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(heuristic, 0)
        max_val = np.max(heuristic)
        if max_val > 0:
            heuristic /= max_val
        return heuristic
    
    # Calculate matrix statistics for adaptive parameter tuning
    dist_stats = np.percentile(distance_matrix[np.triu_indices(n, k=1)], [25, 50, 75])
    var_coefficient = np.std(distance_matrix) / np.mean(distance_matrix)
    
    # 1. Base heuristic with adaptive scaling
    with np.errstate(divide='ignore', invalid='ignore'):
        base_heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(base_heuristic, 0)
    
    # Adaptive scaling based on distance distribution
    if var_coefficient > 1.0:  # High variance, use log scaling
        base_heuristic = np.log1p(base_heuristic)
    max_val = np.max(base_heuristic)
    if max_val > 0:
        base_heuristic /= max_val
    
    # 2. Identify MST edges for essential connectivity
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # 3. Hierarchical clustering for intra-cluster emphasis
    n_clusters = max(2, min(10, int(np.sqrt(n))))
    clustering = AgglomerativeClustering(
        n_clusters=n_clusters,
        metric='precomputed',
        linkage='average'
    ).fit(distance_matrix)
    cluster_labels = clustering.labels_
    
    # Create cluster affinity matrix
    cluster_affinity = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            if cluster_labels[i] == cluster_labels[j]:
                # Boost intra-cluster edges based on cluster cohesion
                cluster_size = np.sum(cluster_labels == cluster_labels[i])
                cluster_cohesion = 1.0 / (1.0 + np.log(cluster_size))
                boost = 1.5 + 0.5 * cluster_cohesion
                cluster_affinity[i, j] = cluster_affinity[j, i] = boost
    
    # 4. Dynamic neighborhood selection with adaptive thresholds
    # Adjust based on problem characteristics
    if n < 50:
        base_percentile = 40 - min(15, int(var_coefficient * 10))
        min_neighbors = max(3, n // 10)
    else:
        base_percentile = 30 - min(10, int(var_coefficient * 8))
        min_neighbors = max(4, int(np.log2(n) * 2))
    
    neighborhood_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Adaptive threshold based on local statistics
        local_stats = np.percentile(dists[dists < np.inf], [25, 50, 75])
        threshold = np.percentile(dists, base_percentile)
        
        # Adjust threshold for nodes in sparse regions
        if local_stats[2] / local_stats[0] > 3:  # High local variance
            threshold = min(threshold, local_stats[1] * 1.2)
        
        close_nodes = dists <= threshold
        
        # Ensure minimum connections
        if close_nodes.sum() < min_neighbors:
            k_neighbors = min(min_neighbors, n - 1)
            nearest_indices = np.argsort(dists)[:k_neighbors]
            close_nodes = np.zeros(n, dtype=bool)
            close_nodes[nearest_indices] = True
        
        neighborhood_mask[i] = close_nodes
    
    # Ensure symmetry and combine with MST
    neighborhood_mask = neighborhood_mask | neighborhood_mask.T | mst_mask
    
    # 5. Multi-level boosting strategy
    heuristic = base_heuristic.copy()
    
    # Level 1: Strong boost for MST edges (foundation)
    heuristic[mst_mask] *= 3.2
    
    # Level 2: Moderate boost for intra-cluster edges (local structure)
    cluster_boost_mask = cluster_affinity > 0
    heuristic[cluster_boost_mask] *= cluster_affinity[cluster_boost_mask]
    
    # Level 3: Light boost for neighborhood edges (local connectivity)
    heuristic[neighborhood_mask] *= 1.6
    
    # Level 4: Triangle completion bonus (global structure)
    # Boost edges that complete promising triangles
    strong_edges = heuristic > np.percentile(heuristic[heuristic > 0], 75)
    for i in range(n):
        strong_neighbors = np.where(strong_edges[i])[0]
        for j in strong_neighbors:
            if i < j:
                # Check for common strong connections
                common = np.intersect1d(strong_neighbors, 
                                       np.where(strong_edges[j])[0])
                if len(common) > 0:
                    bonus = 1.0 + 0.1 * len(common)
                    heuristic[i, j] *= bonus
                    heuristic[j, i] = heuristic[i, j]
    
    heuristic = (heuristic + heuristic.T) / 2  # Enforce symmetry
    
    # 6. Simulated iterative refinement
    # Simulate local search feedback by reinforcing promising edge patterns
    refinement_cycles = max(1, min(3, int(np.log10(n))))
    
    for cycle in range(refinement_cycles):
        # Identify promising edges from current heuristic
        threshold = np.percentile(heuristic[heuristic > 0], 70 - cycle * 10)
        promising_edges = heuristic > threshold
        
        # Reinforce patterns: if two nodes share many promising neighbors,
        # their connection becomes more promising
        for i in range(n):
            i_neighbors = np.where(promising_edges[i])[0]
            for j in range(i + 1, n):
                if promising_edges[i, j]:
                    j_neighbors = np.where(promising_edges[j])[0]
                    common_count = len(np.intersect1d(i_neighbors, j_neighbors))
                    if common_count >= 2:
                        reinforcement = 1.0 + 0.08 * common_count
                        heuristic[i, j] *= reinforcement
                        heuristic[j, i] = heuristic[i, j]
    
    # 7. Adaptive sparsification with cluster-aware targeting
    target_edges = max(4, min(int(np.log2(n) * 2.8), n - 1))
    sparsified = np.zeros_like(heuristic)
    
    for i in range(n):
        row_heuristics = heuristic[i].copy()
        row_heuristics[i] = -np.inf
        
        # Required edges: MST and strong intra-cluster connections
        mst_neighbors = np.where(mst_mask[i])[0]
        cluster_neighbors = np.where(cluster_affinity[i] > 1.6)[0]
        required = np.union1d(mst_neighbors, cluster_neighbors)
        
        additional_needed = max(0, target_edges - len(required))
        
        if additional_needed > 0:
            # Select from remaining edges, preferring:
            # 1. High heuristic value
            # 2. Nodes in same cluster
            # 3. Nodes with good triangle completion
            
            remaining_mask = np.ones(n, dtype=bool)
            remaining_mask[required] = False
            remaining_mask[i] = False
            
            if np.any(remaining_mask):
                remaining_indices = np.where(remaining_mask)[0]
                remaining_values = row_heuristics[remaining_indices]
                
                # Adjust values for cluster preference
                cluster_preference = np.array([1.2 if cluster_labels[i] == cluster_labels[j] 
                                             else 1.0 for j in remaining_indices])
                adjusted_values = remaining_values * cluster_preference
                
                if len(remaining_indices) > additional_needed:
                    # Select top additional_needed edges
                    top_indices = np.argpartition(-adjusted_values, 
                                                 additional_needed - 1)[:additional_needed]
                    selected = remaining_indices[top_indices]
                else:
                    selected = remaining_indices
            else:
                selected = np.array([], dtype=int)
        else:
            selected = np.array([], dtype=int)
        
        # Combine required and selected edges
        keep_indices = np.unique(np.concatenate([required, selected]))
        sparsified[i, keep_indices] = heuristic[i, keep_indices]
    
    # Symmetrize
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 8. Final connectivity check and repair
    for i in range(n):
        connected = np.nonzero(sparsified[i])[0]
        if len(connected) == 0:
            # Connect to nearest node
            nearest = np.argsort(distance_matrix[i])[1]
            sparsified[i, nearest] = sparsified[nearest, i] = 0.25
        elif len(connected) == 1:
            # Add second connection
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            dists[connected[0]] = np.inf
            second = np.argmin(dists)
            sparsified[i, second] = sparsified[second, i] = 0.2
    
    # 9. Final normalization
    np.fill_diagonal(sparsified, 0)
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified /= max_val
    
    return sparsified
