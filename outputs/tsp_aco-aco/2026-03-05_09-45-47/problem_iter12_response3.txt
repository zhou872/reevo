import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import KDTree
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic with adaptive sparsification and probabilistic edge sampling.
    
    Key innovations:
    1. Adaptive sparsification based on distance distribution characteristics
    2. Probabilistic edge sampling for massive instances (>5000 nodes)
    3. Distance distribution modeling with adaptive kernel density estimation
    4. Two-stage connectivity preservation with fallback mechanisms
    
    Args:
        distance_matrix: Square symmetric distance matrix (n x n)
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges for TSP tours
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # Analyze instance for adaptive parameter tuning
    instance_meta = _analyze_distance_distribution(distance_matrix, n)
    
    # Base reciprocal heuristic with adaptive scaling
    base_heuristic = _compute_adaptive_reciprocal(distance_matrix, instance_meta)
    
    # Spatial clustering with adaptive method selection
    spatial_heuristic = _adaptive_spatial_clustering_v2(distance_matrix, n, instance_meta)
    
    # MST-enhanced connectivity
    mst_heuristic = _mst_connectivity_boost(distance_matrix, n, instance_meta)
    
    # Adaptive combination with instance-aware weighting
    combined = _instance_aware_fusion(base_heuristic, spatial_heuristic, 
                                     mst_heuristic, distance_matrix, instance_meta)
    
    # Intelligent sparsification with connectivity guarantee
    sparsified = _adaptive_sparsification(combined, distance_matrix, n, instance_meta)
    
    # Local structure enhancement via efficient sampling
    enhanced = _efficient_local_enhancement(sparsified, distance_matrix, n, instance_meta)
    
    # Final adaptive normalization
    normalized = _adaptive_normalization(enhanced, distance_matrix, instance_meta)
    
    return normalized


def _analyze_distance_distribution(distance_matrix: np.ndarray, n: int) -> dict:
    """Comprehensive analysis of distance distribution for adaptive parameter tuning."""
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    # Basic statistics
    stats = {
        'n': n,
        'mean': np.mean(distances),
        'median': np.median(distances),
        'std': np.std(distances),
        'q1': np.percentile(distances, 25),
        'q3': np.percentile(distances, 75),
        'iqr': np.percentile(distances, 75) - np.percentile(distances, 25),
        'skewness': np.mean(((distances - np.mean(distances)) / np.std(distances))**3) if np.std(distances) > 0 else 0,
        'kurtosis': np.mean(((distances - np.mean(distances)) / np.std(distances))**4) if np.std(distances) > 0 else 0
    }
    
    # Size categorization with more granularity
    if n < 100:
        stats['size_class'] = 'tiny'
    elif n < 500:
        stats['size_class'] = 'small'
    elif n < 2000:
        stats['size_class'] = 'medium'
    elif n < 5000:
        stats['size_class'] = 'large'
    else:
        stats['size_class'] = 'massive'
    
    # Density estimation using kernel density
    if n <= 2000:
        # Compute approximate spatial embedding for density estimation
        try:
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                from sklearn.manifold import MDS
                n_components = min(3, n - 1)
                if n_components > 0:
                    mds = MDS(n_components=n_components, dissimilarity='precomputed', 
                             random_state=42, n_init=1, max_iter=20)
                    coords = mds.fit_transform(distance_matrix)
                    # Compute pairwise distances in embedded space
                    from scipy.spatial.distance import pdist
                    embedded_dists = pdist(coords)
                    stats['spatial_variance'] = np.var(embedded_dists) / (np.mean(embedded_dists) + 1e-12)
                else:
                    stats['spatial_variance'] = 1.0
        except:
            stats['spatial_variance'] = 1.0
    else:
        stats['spatial_variance'] = 1.0
    
    # Distance distribution type classification
    cv = stats['std'] / (stats['mean'] + 1e-12)
    if cv < 0.3:
        stats['dist_type'] = 'uniform'
    elif cv < 0.7:
        stats['dist_type'] = 'moderate'
    elif stats['skewness'] > 1.0:
        stats['dist_type'] = 'right_skewed'
    else:
        stats['dist_type'] = 'variable'
    
    # MST properties for connectivity guidance
    if n <= 10000:
        mst = minimum_spanning_tree(distance_matrix)
        mst_dists = mst.data[mst.data > 0]
        if len(mst_dists) > 0:
            stats['mst_mean'] = np.mean(mst_dists)
            stats['mst_median'] = np.median(mst_dists)
            stats['mst_span'] = np.max(mst_dists) / (np.min(mst_dists) + 1e-12)
        else:
            stats['mst_mean'] = stats['mean']
            stats['mst_median'] = stats['median']
            stats['mst_span'] = 1.0
    else:
        # Approximate MST properties for very large instances
        sample_size = min(1000, n)
        sample_nodes = np.random.choice(n, sample_size, replace=False)
        submatrix = distance_matrix[np.ix_(sample_nodes, sample_nodes)]
        mst = minimum_spanning_tree(submatrix)
        mst_dists = mst.data[mst.data > 0]
        if len(mst_dists) > 0:
            stats['mst_mean'] = np.mean(mst_dists) * (stats['mean'] / np.mean(submatrix[np.triu_indices(sample_size, k=1)]))
            stats['mst_median'] = np.median(mst_dists) * (stats['median'] / np.median(submatrix[np.triu_indices(sample_size, k=1)]))
            stats['mst_span'] = np.max(mst_dists) / (np.min(mst_dists) + 1e-12)
        else:
            stats['mst_mean'] = stats['mean']
            stats['mst_median'] = stats['median']
            stats['mst_span'] = 1.0
    
    return stats


def _compute_adaptive_reciprocal(distance_matrix: np.ndarray, stats: dict) -> np.ndarray:
    """Compute reciprocal heuristic with adaptive scaling based on distance distribution."""
    n = distance_matrix.shape[0]
    
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
    
    np.fill_diagonal(reciprocal, 0)
    
    # Adaptive scaling based on distribution type
    if stats['dist_type'] == 'uniform':
        # Less aggressive scaling for uniform distances
        scale = 1.0
    elif stats['dist_type'] == 'right_skewed':
        # More aggressive scaling for skewed distributions
        scale = 1.5
    else:
        scale = 1.2
    
    # Apply non-linear transformation
    max_val = np.max(reciprocal[reciprocal < np.inf])
    if max_val > 0:
        reciprocal = (reciprocal / max_val) ** scale
    
    return reciprocal


def _adaptive_spatial_clustering_v2(distance_matrix: np.ndarray, n: int, stats: dict) -> np.ndarray:
    """Adaptive spatial clustering with method selection based on instance size."""
    spatial = np.ones((n, n))
    
    # Method selection
    if stats['size_class'] in ['tiny', 'small']:
        # Full MDS for small instances
        coords = _fast_embedding(distance_matrix, n, method='mds', dims=min(3, n-1))
        k = min(max(5, n // 20), 15)
    elif stats['size_class'] == 'medium':
        # Landmark-based MDS
        coords = _fast_embedding(distance_matrix, n, method='landmark', n_landmarks=min(50, n//10))
        k = min(max(3, int(np.log2(n)) + 2), 10)
    elif stats['size_class'] == 'large':
        # Distance-based approximation
        coords = None
        k = min(max(2, int(np.log10(n)) + 1), 8)
    else:  # massive
        # Pure distance-based with sampling
        coords = None
        k = min(max(2, int(np.log10(n))), 5)
    
    if coords is not None and coords.shape[1] > 0:
        # Use spatial embedding
        try:
            tree = KDTree(coords)
            for i in range(n):
                # Adaptive k based on local density
                local_k = k
                if stats['spatial_variance'] > 1.5:
                    local_k = min(int(k * 1.5), n-1)
                
                _, indices = tree.query(coords[i].reshape(1, -1), k=min(local_k+1, n))
                indices = indices[0]
                if len(indices) > 1:
                    # Exclude self if present
                    if np.linalg.norm(coords[i] - coords[indices[0]]) < 1e-12:
                        indices = indices[1:local_k+1]
                    else:
                        indices = indices[:local_k]
                    
                    # Assign weights with distance-based decay
                    for rank, j in enumerate(indices):
                        decay = np.exp(-rank / max(1, local_k/2))
                        spatial[i, j] = 1.0 + decay
        except:
            coords = None  # Fallback to distance-based
    
    if coords is None:
        # Distance-based neighborhood
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Adaptive threshold based on local statistics
            if stats['dist_type'] == 'uniform':
                threshold = np.percentile(dists, 30)
            elif stats['dist_type'] == 'right_skewed':
                threshold = np.percentile(dists, 20)
            else:
                threshold = stats['q1']
            
            # Find neighbors within threshold
            neighbors = np.where(dists <= threshold)[0]
            if len(neighbors) < 2:  # Ensure minimum connectivity
                neighbors = np.argsort(dists)[:min(k, n-1)]
            
            for j in neighbors:
                # Weight based on distance rank
                rank = np.where(np.argsort(dists) == j)[0][0] + 1
                spatial[i, j] = 1.0 + 1.0 / rank
    
    # Symmetrize
    spatial = np.maximum(spatial, spatial.T)
    np.fill_diagonal(spatial, 0)
    
    return spatial


def _fast_embedding(distance_matrix: np.ndarray, n: int, method: str = 'mds', 
                   dims: int = 3, n_landmarks: int = 50) -> np.ndarray:
    """Fast embedding methods for spatial approximation."""
    if n <= 2:
        return np.zeros((n, 1))
    
    if method == 'mds' and n <= 1000:
        try:
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                from sklearn.manifold import MDS
                mds = MDS(n_components=min(dims, n-1), dissimilarity='precomputed',
                         random_state=42, n_init=1, max_iter=30, eps=0.2)
                return mds.fit_transform(distance_matrix)
        except:
            pass
    
    # Landmark-based approximation
    n_landmarks = min(n_landmarks, n)
    landmarks = np.random.choice(n, n_landmarks, replace=False)
    
    # Use distances to landmarks as coordinates
    coords = distance_matrix[:, landmarks]
    
    # Simple normalization
    coords = coords - np.mean(coords, axis=0)
    
    return coords


def _mst_connectivity_boost(distance_matrix: np.ndarray, n: int, stats: dict) -> np.ndarray:
    """MST-based connectivity enhancement with adaptive boosting."""
    mst_factor = np.ones((n, n))
    
    if stats['size_class'] == 'massive':
        # Approximate MST for massive instances
        sample_size = min(500, n)
        sample_nodes = np.random.choice(n, sample_size, replace=False)
        submatrix = distance_matrix[np.ix_(sample_nodes, sample_nodes)]
        mst = minimum_spanning_tree(submatrix)
        mst_binary = (mst.toarray() > 0)
        
        # Map back to original indices probabilistically
        for idx_i, i in enumerate(sample_nodes):
            for idx_j, j in enumerate(sample_nodes):
                if idx_i != idx_j and mst_binary[idx_i, idx_j]:
                    # Boost probability decreases with instance size
                    boost = 1.5 * (500 / n)
                    mst_factor[i, j] = mst_factor[j, i] = 1.0 + boost
    else:
        # Exact MST for manageable instances
        mst = minimum_spanning_tree(distance_matrix)
        mst_binary = (mst.toarray() > 0)
        mst_factor[mst_binary] = 2.0
    
    # Extended neighborhood for non-massive instances
    if stats['size_class'] != 'massive':
        for i in range(n):
            mst_neighbors = np.where(mst_binary[i])[0]
            if len(mst_neighbors) > 0:
                # Boost edges to neighbors of MST neighbors
                secondary = set()
                for neighbor in mst_neighbors:
                    # Add closest nodes to this neighbor
                    dists = distance_matrix[neighbor].copy()
                    dists[neighbor] = np.inf
                    k_extend = min(2, n - 2)
                    if k_extend > 0:
                        close = np.argpartition(dists, k_extend)[:k_extend]
                        secondary.update(close)
                
                secondary.discard(i)
                for j in secondary:
                    if mst_factor[i, j] < 1.3:
                        mst_factor[i, j] = mst_factor[j, i] = 1.3
    
    np.fill_diagonal(mst_factor, 0)
    return mst_factor


def _instance_aware_fusion(base: np.ndarray, spatial: np.ndarray, 
                          mst: np.ndarray, distance_matrix: np.ndarray, 
                          stats: dict) -> np.ndarray:
    """Adaptive fusion of heuristic factors based on instance characteristics."""
    n = base.shape[0]
    
    # Determine weights based on instance analysis
    if stats['size_class'] == 'tiny':
        w_base, w_spatial, w_mst = 1.0, 1.2, 1.5
    elif stats['size_class'] == 'small':
        w_base, w_spatial, w_mst = 1.0, 1.1, 1.4
    elif stats['size_class'] == 'medium':
        w_base, w_spatial, w_mst = 1.0, 1.0, 1.3
    elif stats['size_class'] == 'large':
        w_base, w_spatial, w_mst = 1.0, 0.9, 1.2
    else:  # massive
        w_base, w_spatial, w_mst = 1.0, 0.8, 1.1
    
    # Adjust weights based on distance distribution
    if stats['dist_type'] == 'uniform':
        w_spatial *= 0.9  # Spatial less important for uniform
        w_mst *= 1.1      # MST more important
    elif stats['dist_type'] == 'right_skewed':
        w_base *= 1.2     # Base reciprocal more important
    
    # Apply weights
    weighted_base = w_base * base
    weighted_spatial = np.where(spatial > 1, w_spatial * spatial, 1.0)
    weighted_mst = np.where(mst > 1, w_mst * mst, 1.0)
    
    # Combine multiplicatively
    combined = weighted_base * weighted_spatial * weighted_mst
    
    # Distance-based envelope
    scale = stats['median']
    dist_envelope = np.exp(-distance_matrix / (scale + 1e-12))
    combined *= (1.0 + 0.3 * dist_envelope)
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    np.fill_diagonal(combined, 0)
    
    return combined


def _adaptive_sparsification(heuristics: np.ndarray, distance_matrix: np.ndarray, 
                            n: int, stats: dict) -> np.ndarray:
    """Adaptive sparsification with guaranteed connectivity."""
    if n <= 10:
        return heuristics.copy()
    
    # Determine sparsity level based on instance size
    if stats['size_class'] == 'tiny':
        edges_per_node = min(n-1, max(5, n//2))
    elif stats['size_class'] == 'small':
        edges_per_node = min(n-1, max(4, n//3))
    elif stats['size_class'] == 'medium':
        edges_per_node = min(n-1, max(3, int(n**0.7)))
    elif stats['size_class'] == 'large':
        edges_per_node = min(n-1, max(2, int(n**0.6)))
    else:  # massive
        edges_per_node = min(n-1, max(2, int(n**0.5)))
    
    sparsified = np.zeros_like(heuristics)
    
    # Always include MST edges for connectivity
    if stats['size_class'] != 'massive':
        mst = minimum_spanning_tree(distance_matrix)
        mst_edges = (mst.toarray() > 0)
        sparsified[mst_edges] = heuristics[mst_edges]
    
    # For each node, select top edges
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf
        
        # Current MST edges for this node
        if stats['size_class'] != 'massive':
            mst_neighbors = np.where(mst_edges[i])[0]
            current_count = len(mst_neighbors)
        else:
            current_count = 0
        
        # Determine how many more edges to add
        to_add = max(0, edges_per_node - current_count)
        
        if to_add > 0:
            # Select top non-MST edges
            if stats['size_class'] != 'massive':
                non_mst_mask = ~mst_edges[i]
                non_mst_mask[i] = False
                candidates = np.where(non_mst_mask)[0]
            else:
                candidates = np.arange(n)
                candidates = candidates[candidates != i]
            
            if len(candidates) > 0:
                candidate_scores = row[candidates]
                
                # Also consider distance to avoid very long edges
                candidate_dists = distance_matrix[i, candidates]
                dist_scores = 1.0 / (candidate_dists + 1e-12)
                
                # Combined score: 70% heuristic, 30% distance
                combined_scores = 0.7 * candidate_scores + 0.3 * dist_scores
                
                # Select top edges
                n_select = min(to_add, len(candidates))
                top_indices = np.argpartition(-combined_scores, n_select)[:n_select]
                selected = candidates[top_indices]
                
                sparsified[i, selected] = heuristics[i, selected]
                sparsified[selected, i] = heuristics[selected, i]
    
    # Connectivity check and repair
    if n <= 5000:  # Only for manageable instances
        visited = np.zeros(n, dtype=bool)
        stack = [0]
        visited[0] = True
        count = 1
        
        while stack:
            node = stack.pop()
            neighbors = np.where(sparsified[node] > 0)[0]
            for neighbor in neighbors:
                if not visited[neighbor]:
                    visited[neighbor] = True
                    stack.append(neighbor)
                    count += 1
        
        if count < n:
            # Connect disconnected components via shortest edges
            components = []
            remaining = np.where(~visited)[0]
            
            while len(remaining) > 0:
                start = remaining[0]
                comp = []
                comp_stack = [start]
                visited_comp = np.zeros(n, dtype=bool)
                visited_comp[start] = True
                
                while comp_stack:
                    node = comp_stack.pop()
                    comp.append(node)
                    neighbors = np.where(sparsified[node] > 0)[0]
                    for neighbor in neighbors:
                        if not visited_comp[neighbor]:
                            visited_comp[neighbor] = True
                            comp_stack.append(neighbor)
                
                components.append(comp)
                remaining = [node for node in remaining if not visited_comp[node]]
            
            # Connect all components to the main component
            main_comp = np.where(visited)[0]
            for comp in components:
                min_dist = np.inf
                best_pair = (0, 0)
                
                for i in main_comp:
                    for j in comp:
                        if distance_matrix[i, j] < min_dist:
                            min_dist = distance_matrix[i, j]
                            best_pair = (i, j)
                
                i, j = best_pair
                sparsified[i, j] = sparsified[j, i] = heuristics[i, j]
    
    return sparsified


def _efficient_local_enhancement(heuristics: np.ndarray, distance_matrix: np.ndarray, 
                                n: int, stats: dict) -> np.ndarray:
    """Efficient local structure enhancement via sampling."""
    enhanced = heuristics.copy()
    
    if n < 4 or np.count_nonzero(heuristics) == 0:
        return enhanced
    
    # Determine sampling strategy
    if stats['size_class'] in ['tiny', 'small']:
        # Process all promising edges
        edge_mask = heuristics > 0
        edges = np.argwhere(np.triu(edge_mask, k=1))
        sample_ratio = 1.0
    elif stats['size_class'] == 'medium':
        # Sample edges
        edge_mask = heuristics > 0
        edges = np.argwhere(np.triu(edge_mask, k=1))
        if len(edges) > 2000:
            sample_ratio = 0.5
            sample_size = min(2000, len(edges))
            idx = np.random.choice(len(edges), sample_size, replace=False)
            edges = edges[idx]
        else:
            sample_ratio = 1.0
    else:
        # Sparse sampling for large/massive instances
        edge_mask = heuristics > 0
        edges = np.argwhere(np.triu(edge_mask, k=1))
        sample_size = min(1000, len(edges))
        if len(edges) > sample_size:
            idx = np.random.choice(len(edges), sample_size, replace=False)
            edges = edges[idx]
        sample_ratio = sample_size / max(1, len(np.argwhere(np.triu(edge_mask, k=1))))
    
    # Precompute binary adjacency for faster queries
    binary_adj = (heuristics > 0).astype(bool)
    
    for edge in edges:
        i, j = edge
        if heuristics[i, j] == 0:
            continue
        
        # Find common neighbors efficiently
        neighbors_i = np.where(binary_adj[i])[0]
        neighbors_j = np.where(binary_adj[j])[0]
        
        # Use sorted arrays for intersection
        common = np.intersect1d(neighbors_i, neighbors_j)
        
        if len(common) == 0:
            continue
        
        # Sample common neighbors if too many
        max_common = min(10, len(common))
        if len(common) > max_common:
            common = np.random.choice(common, max_common, replace=False)
        
        d_ij = distance_matrix[i, j]
        quality_sum = 0
        count = 0
        
        for k in common:
            d_ik = distance_matrix[i, k]
            d_jk = distance_matrix[j, k]
            
            # Triangle quality: prefer compact triangles
            max_side = max(d_ij, d_ik, d_jk)
            perimeter = d_ij + d_ik + d_jk
            
            if perimeter > 2 * max_side:  # Valid triangle
                # Quality metric: compactness relative to MST scale
                compactness = stats['mst_median'] / (perimeter / 3 + 1e-12)
                quality = min(compactness, 2.0)
                quality_sum += quality
                count += 1
        
        if count > 0:
            avg_quality = quality_sum / count
            # Apply boost with diminishing returns
            boost = 1.0 + 0.15 * avg_quality * np.sqrt(min(count, 4))
            enhanced[i, j] = enhanced[j, i] = heuristics[i, j] * boost
    
    return enhanced


def _adaptive_normalization(heuristics: np.ndarray, distance_matrix: np.ndarray, 
                           stats: dict) -> np.ndarray:
    """Adaptive normalization based on instance characteristics."""
    n = heuristics.shape[0]
    
    # Zero diagonal
    np.fill_diagonal(heuristics, 0)
    
    # Handle all-zero or degenerate cases
    max_val = np.max(heuristics)
    if max_val <= 0:
        return heuristics
    
    # Remove extreme outliers (top 1%)
    if stats['size_class'] not in ['tiny', 'small']:
        triu_vals = heuristics[np.triu_indices(n, k=1)]
        if len(triu_vals) > 100:
            p99 = np.percentile(triu_vals[triu_vals > 0], 99)
            heuristics = np.minimum(heuristics, p99)
    
    # Normalize based on distribution type
    if stats['dist_type'] == 'uniform':
        # Less aggressive normalization for uniform distributions
        norm_power = 0.8
    elif stats['dist_type'] == 'right_skewed':
        # More aggressive to handle skew
        norm_power = 1.2
    else:
        norm_power = 1.0
    
    # Apply normalization
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = (heuristics / max_val) ** norm_power
    
    # Ensure numerical stability
    heuristics = np.nan_to_num(heuristics, nan=0.0, posinf=0.0, neginf=0.0)
    
    return heuristics
