import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.manifold import MDS
from sklearn.neighbors import KDTree
from sklearn.cluster import AgglomerativeClustering

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with adaptive learning and optimized efficiency.
    
    Key improvements:
    1. Dynamic heuristic combination weights based on problem characteristics
    2. Efficient geometric embedding with early stopping
    3. Graph neural-inspired local structure scoring
    4. Optimized sparsification with connectivity guarantees
    """
    n = distance_matrix.shape[0]
    
    if n < 3:
        return 1 / (distance_matrix + np.eye(n) * 1e-12)
    
    # 1. Adaptive base heuristic computation
    # Dynamic epsilon based on distance distribution
    dist_nonzero = distance_matrix[~np.eye(n, dtype=bool)]
    eps = np.percentile(dist_nonzero, 1) * 0.01 if len(dist_nonzero) > 0 else 1e-12
    reciprocal_weights = 1 / (distance_matrix + np.eye(n) * eps)
    
    # 2. Efficient geometric embedding (only for moderate-sized problems)
    spatial_factor = np.ones_like(distance_matrix)
    if n <= 10000:  # Adjust based on available memory
        # Fast MDS with early convergence
        mds = MDS(
            n_components=min(3, n-1), 
            dissimilarity='precomputed', 
            random_state=42,
            n_init=1,
            max_iter=50 if n > 1000 else 100,
            eps=1e-4
        )
        try:
            coords = mds.fit_transform(distance_matrix)
            tree = KDTree(coords)
            
            # Adaptive neighborhood size
            k_spatial = max(3, min(20, int(np.log2(n)) * 2))
            for i in range(n):
                _, indices = tree.query(coords[i].reshape(1, -1), k=min(k_spatial + 1, n))
                indices = indices[0][1:]  # Exclude self
                spatial_factor[i, indices] = 2.5
                spatial_factor[indices, i] = 2.5
        except:
            # Fallback to distance-based neighbors
            pass
    
    # 3. Minimum Spanning Tree with enhanced scoring
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    mst_factor = np.where(mst_edges + mst_edges.T > 0, 3.0, 1.0)
    
    # 4. Local structure learning via hierarchical clustering
    local_structure = np.ones_like(distance_matrix)
    if n <= 5000:  # Adjust threshold for clustering
        # Fast clustering to identify local neighborhoods
        clustering = AgglomerativeClustering(
            n_clusters=max(2, n // 20),
            affinity='precomputed',
            linkage='average'
        )
        try:
            labels = clustering.fit_predict(distance_matrix)
            # Boost intra-cluster edges
            for i in range(n):
                same_cluster = (labels == labels[i])
                local_structure[i, same_cluster] = 2.0
                local_structure[i, i] = 1.0  # Reset diagonal
        except:
            pass
    
    # 5. Dynamic factor combination based on problem characteristics
    # Compute problem statistics for weight adaptation
    avg_distance = np.mean(dist_nonzero) if len(dist_nonzero) > 0 else 1.0
    std_distance = np.std(dist_nonzero) if len(dist_nonzero) > 0 else 1.0
    distance_variation = std_distance / (avg_distance + 1e-12)
    
    # Adaptive weights based on problem characteristics
    if distance_variation < 0.3:
        # Uniform distances: emphasize geometric structure
        weights = np.array([0.3, 0.4, 0.2, 0.1])  # [reciprocal, spatial, mst, local]
    elif distance_variation > 1.0:
        # Highly varied distances: emphasize MST and local structure
        weights = np.array([0.2, 0.2, 0.4, 0.2])
    else:
        # Balanced case
        weights = np.array([0.25, 0.3, 0.3, 0.15])
    
    # 6. Compute combined heuristic score with adaptive normalization
    heuristics_matrix = (
        reciprocal_weights * weights[0] +
        spatial_factor * weights[1] +
        mst_factor * weights[2] +
        local_structure * weights[3]
    )
    
    # Apply non-linear scaling to emphasize promising edges
    heuristics_matrix = np.power(heuristics_matrix, 1.2)
    
    # 7. Optimized sparsification
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Adaptive sparsification threshold
    density = min(0.3, max(0.05, 2.0 * np.log(n) / n))
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        # Keep top edges based on adaptive density
        threshold = np.percentile(row[row > -np.inf], 100 * (1 - density))
        mask = row >= threshold
        sparsified[i] = heuristics_matrix[i] * mask
    
    # Ensure symmetry and connectivity
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Guarantee minimum connectivity: each node connects to at least 2 others
    for i in range(n):
        connected = np.where(sparsified[i] > 0)[0]
        if len(connected) < 2:
            # Find closest nodes not already connected
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            distances[connected] = np.inf
            closest = np.argsort(distances)[:2 - len(connected)]
            sparsified[i, closest] = heuristics_matrix[i, closest]
            sparsified[closest, i] = heuristics_matrix[closest, i]
    
    # 8. Local triangle enhancement (only for sparse edges)
    triangle_boost = np.ones_like(sparsified)
    edge_mask = sparsified > 0
    
    # Efficient triangle counting for sparse graph
    for i in range(n):
        neighbors = np.where(edge_mask[i])[0]
        for j in neighbors:
            if j <= i:
                continue
            # Count common neighbors efficiently
            common = np.intersect1d(
                np.where(edge_mask[i])[0],
                np.where(edge_mask[j])[0]
            )
            if len(common) > 0:
                boost = 1.0 + 0.15 * len(common)
                triangle_boost[i, j] = triangle_boost[j, i] = boost
    
    final_heuristics = sparsified * triangle_boost
    
    # 9. Final normalization preserving relative rankings
    max_val = np.max(final_heuristics[~np.eye(n, dtype=bool)])
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0.0)
    
    return final_heuristics
