import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:

    Enhanced heuristic for TSP edge selection combining multiple graph metrics.
    
    Returns a matrix where each element indicates the promise of including
    that edge in a solution. Combines: inverse distance, nearest neighbor
    sparsity, edge betweenness approximation, local clustering, and MST bias.
    """
    n = distance_matrix.shape[0]
    if n <= 2:

    # 1. Base inverse distance heuristic (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    inv_dist = np.nan_to_num(inv_dist)
    
    # 2. Nearest neighbor sparsification
    k = max(2, min(5, n // 4))  # Adaptive k-nearest
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        # Get k+1 nearest neighbors (including self)
        indices = np.argpartition(distance_matrix[i], k+1)[:k+1]
        indices = indices[indices != i]  # Remove self
        nn_mask[i, indices] = True
    # Keep edges that are mutual nearest neighbors or in top-k for at least one endpoint
    mutual_nn = nn_mask & nn_mask.T
    one_way_nn = nn_mask | nn_mask.T
    nn_weight = np.where(mutual_nn, 2.0, np.where(one_way_nn, 1.0, 0.0))
    
    # 3. Approximate edge betweenness via random paths
    np.random.seed(42)  # For reproducibility
    n_samples = min(100, 10 * n)
    edge_counts = np.zeros_like(distance_matrix)
    
    for _ in range(n_samples):
        start, end = np.random.choice(n, 2, replace=False)
        # Simple greedy path from start to end
        path = [start]
        current = start
        visited = set([start])
        
        while current != end and len(visited) < n:
            # Find unvisited neighbor with smallest distance
            mask = np.ones(n, dtype=bool)
            mask[list(visited)] = False
            mask[current] = False
            
            if not mask.any():
                break
                
            valid_dists = distance_matrix[current].copy()
            valid_dists[~mask] = np.inf
            next_node = np.argmin(valid_dists)
            
            # Count this edge
            edge_counts[current, next_node] += 1
            edge_counts[next_node, current] += 1
            
            path.append(next_node)
            visited.add(next_node)
            current = next_node
    
    betweenness = edge_counts / n_samples
    
    # 4. Local clustering coefficient for edge endpoints
    clustering_weight = np.zeros_like(distance_matrix)
    # Simple approximation: edges between nodes with many common neighbors are promising
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] > 0:
                # Find common neighbors
                i_neighbors = set(np.where(distance_matrix[i] < np.percentile(distance_matrix[i], 25))[0])

                common = len(i_neighbors & j_neighbors)
                clustering_weight[i, j] = clustering_weight[j, i] = common / max(len(i_neighbors), 1)
