import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Compute heuristic indicators for TSP edges combining multiple factors.
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return distance_matrix.copy()
    
    # Factor 1: Inverse distance (short edges are better)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.where(distance_matrix > 0, 1.0 / distance_matrix, 0.0)
    
    # Factor 2: Minimum Spanning Tree edges get bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = (mst > 0).astype(float) * 2.0
    
    # Factor 3: Local connectivity - edges connecting to nodes with few good alternatives
    # Compute how many "good" connections each node has
    threshold = np.percentile(distance_matrix[distance_matrix > 0], 25)
    good_connections = (distance_matrix <= threshold).astype(float)
    np.fill_diagonal(good_connections, 0)
    
    # Nodes with fewer good connections get higher weight for their connections
    connection_counts = good_connections.sum(axis=1, keepdims=True)
    node_weight = 1.0 / (connection_counts + 1e-10)  # +epsilon to avoid division by zero
    connectivity_factor = node_weight + node_weight.T
    
    # Factor 4: Nearest neighbor chains
    # For each node, find k nearest neighbors and give them bonus
    k = max(2, min(5, n // 10))
    nn_bonus = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        indices = np.argsort(distance_matrix[i])[1:k+1]
        nn_bonus[i, indices] = 1.0
        # Symmetric bonus
        for j in indices:
            nn_bonus[j, i] = max(nn_bonus[j, i], 1.0)
    
    # Combine all factors
    heuristics = (
        inv_dist * 2.0 +           # Base inverse distance (double weight)
        mst_bonus * 3.0 +          # MST edges are very promising
        connectivity_factor * 1.5 + # Nodes with few options
        nn_bonus * 2.0             # Nearest neighbors
    )
    
    # Normalize by number of factors
    factor_count = 4
    heuristics /= factor_count
    
    # Sparsify: Keep only promising edges
    # Strategy 1: Keep top edges for each node
    sparsified = np.zeros_like(heuristics)
    keep_per_node = max(2, min(10, n // 4))
    
    for i in range(n):
        # Get indices of top edges for this node
        top_indices = np.argsort(heuristics[i])[-keep_per_node:]
        sparsified[i, top_indices] = heuristics[i, top_indices]
    
    # Strategy 2: Also keep mutual top choices
    mutual_mask = (sparsified > 0) & (sparsified.T > 0)
    sparsified = np.where(mutual_mask, heuristics, sparsified)
    
    # Strategy 3: Keep MST edges regardless
    sparsified = np.where(mst_bonus > 0, heuristics, sparsified)
    
    # Strategy 4: Ensure connectivity - keep at least 2 edges per node
    for i in range(n):
        if np.count_nonzero(sparsified[i]) < 2:
            # Add the next best edge not already included
            sorted_idx = np.argsort(heuristics[i])[::-1]
            for j in sorted_idx:
                if j != i and sparsified[i, j] == 0:
                    sparsified[i, j] = heuristics[i, j]
                    sparsified[j, i] = heuristics[j, i]
                    break
    
    # Zero out diagonal
    np.fill_diagonal(sparsified, 0.0)
    
    return sparsified
