import numpy as np
from scipy.sparse import csr_matrix
from scipy.sparse.csgraph import minimum_spanning_tree
import networkx as nx
from sklearn.cluster import SpectralClustering, KMeans
from sklearn.metrics.pairwise import cosine_similarity
from typing import Dict, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge selection heuristics with scalable approximations,
    adaptive weight tuning, and evolutionary-informed pattern recognition.
    
    Key Innovations:
    1. Scalable spectral clustering with early termination
    2. Adaptive multi-heuristic fusion with instance-aware weighting
    3. Hierarchical sparsification preserving connectivity
    4. Node embedding with distance-based feature engineering
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 3:
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal, 0)
        return reciprocal / np.max(reciprocal) if np.max(reciprocal) > 0 else reciprocal
    
    # 1. Scalable reciprocal distance with adaptive regularization
    base_heuristic = np.zeros_like(distance_matrix)
    epsilon = np.percentile(distance_matrix[distance_matrix > 0], 5) * 0.01
    np.fill_diagonal(base_heuristic, 0)
    
    # Use vectorized computation for base heuristic
    mask = ~np.eye(n, dtype=bool)
    base_heuristic[mask] = 1 / (distance_matrix[mask] + epsilon)
    
    # 2. Scalable spectral clustering with fallback
    cluster_labels = _scalable_clustering(distance_matrix, n)
    
    # Create intra-cluster boost matrix (sparse representation)
    intra_cluster_boost = np.ones((n, n))
    for i in range(n):
        intra_cluster_boost[i] = np.where(cluster_labels == cluster_labels[i], 1.8, 0.8)
    np.fill_diagonal(intra_cluster_boost, 1)
    
    # 3. Minimum Spanning Tree with connectivity guarantee
    mst = minimum_spanning_tree(csr_matrix(distance_matrix))
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # 4. Scalable betweenness centrality approximation
    edge_betweenness = _approximate_betweenness(distance_matrix, n)
    
    # 5. Adaptive node embeddings with feature selection
    node_embeddings = _compute_node_embeddings(distance_matrix, n)
    embedding_similarity = cosine_similarity(node_embeddings)
    np.fill_diagonal(embedding_similarity, 0)
    
    # 6. Mutual nearest neighbors structure
    k_nn = min(max(3, int(np.log2(n))), n - 1)
    mutual_nn_mask = _compute_mutual_nearest_neighbors(distance_matrix, k_nn)
    
    # 7. Adaptive multi-heuristic fusion
    heuristic = _adaptive_heuristic_fusion(
        base_heuristic=base_heuristic,
        intra_cluster_boost=intra_cluster_boost,
        mst_mask=mst_mask,
        edge_betweenness=edge_betweenness,
        embedding_similarity=embedding_similarity,
        mutual_nn_mask=mutual_nn_mask,
        distance_matrix=distance_matrix,
        n=n
    )
    
    # 8. Hierarchical sparsification with connectivity preservation
    heuristic = _hierarchical_sparsification(
        heuristic=heuristic,
        mst_mask=mst_mask,
        distance_matrix=distance_matrix,
        n=n
    )
    
    # 9. Final normalization and validation
    heuristic = np.maximum(heuristic, heuristic.T)
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    np.fill_diagonal(heuristic, 0)
    
    return heuristic

def _scalable_clustering(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Scalable clustering with early termination for large instances."""
    n_clusters = min(max(3, int(np.sqrt(n))), 15)
    
    if n <= 1000:
        # Use spectral clustering for moderate instances
        try:
            spectral = SpectralClustering(
                n_clusters=n_clusters,
                affinity='precomputed',
                random_state=42,
                n_init=3,
                assign_labels='discretize'
            )
            # Convert distances to similarities
            max_dist = np.max(distance_matrix)
            similarities = 1 - distance_matrix / max_dist
            np.fill_diagonal(similarities, 1)
            return spectral.fit_predict(similarities)
        except:
            pass
    
    # Fallback to fast k-means on distance features
    k_features = min(8, n - 1)
    features = np.zeros((n, k_features))
    for i in range(n):
        sorted_dists = np.sort(distance_matrix[i])[1:k_features+1]
        features[i, :len(sorted_dists)] = sorted_dists
    
    return KMeans(n_clusters=n_clusters, random_state=42, n_init=3).fit_predict(features)

def _approximate_betweenness(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Approximate betweenness centrality using random sampling and sparse graph."""
    edge_betweenness = np.ones((n, n))
    
    # Build sparse graph with top-k neighbors for each node
    G = nx.Graph()
    k_edges = min(15, n - 1)
    
    for i in range(n):
        neighbors = np.argsort(distance_matrix[i])[1:k_edges+1]
        for j in neighbors:
            G.add_edge(i, j, weight=distance_matrix[i, j])
    
    if len(G.edges()) > 0:
        # Sample nodes for approximation
        sample_size = min(50, max(10, n // 20))
        if sample_size < n:
            sample_nodes = np.random.choice(n, sample_size, replace=False)
        else:
            sample_nodes = np.arange(n)
        
        try:
            betweenness = nx.betweenness_centrality_subset(
                G,
                sources=list(sample_nodes),
                targets=list(sample_nodes),
                normalized=True,
                weight='weight',
                k=min(sample_size, len(G.nodes()))
            )
            
            # Convert to edge betweenness
            for i, j in G.edges():
                edge_value = (betweenness.get(i, 0) + betweenness.get(j, 0)) / 2
                edge_betweenness[i, j] = edge_betweenness[j, i] = 1.0 + 2.0 * edge_value
        except:
            pass
    
    return edge_betweenness

def _compute_node_embeddings(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute node embeddings using distance-based features."""
    k_representative = min(10, n - 1)
    
    # Select diverse representatives using k-means++ initialization
    if n <= 200:
        # Use all nodes for small instances
        representatives = np.arange(n)
    else:
        # Sample representative nodes using distance-based diversity
        representatives = [np.random.randint(n)]
        for _ in range(1, k_representative):
            # Choose next representative farthest from existing ones
            min_dists = np.min(distance_matrix[:, representatives], axis=1)
            next_rep = np.argmax(min_dists)
            representatives.append(next_rep)
        representatives = np.array(representatives)
    
    # Create embeddings with local and global features
    n_features = len(representatives) + 3
    embeddings = np.zeros((n, n_features))
    
    # Distances to representatives
    embeddings[:, :len(representatives)] = distance_matrix[:, representatives]
    
    # Additional features: local statistics
    for i in range(n):
        # Mean distance to k nearest neighbors
        k_local = min(5, n - 1)
        local_dists = np.sort(distance_matrix[i])[1:k_local+1]
        embeddings[i, len(representatives)] = np.mean(local_dists)
        
        # Distance variance
        embeddings[i, len(representatives)+1] = np.var(distance_matrix[i, distance_matrix[i] > 0])
        
        # Clustering coefficient approximation
        embeddings[i, len(representatives)+2] = _local_clustering_coefficient(distance_matrix, i)
    
    return embeddings

def _local_clustering_coefficient(distance_matrix: np.ndarray, node: int) -> float:
    """Approximate local clustering coefficient for a node."""
    n = distance_matrix.shape[0]
    k_neighbors = min(10, n - 1)
    
    # Get k nearest neighbors
    neighbors = np.argsort(distance_matrix[node])[1:k_neighbors+1]
    if len(neighbors) < 2:
        return 0.5
    
    # Count triangles among neighbors
    triangles = 0
    possible = 0
    
    for i in range(len(neighbors)):
        for j in range(i+1, len(neighbors)):
            possible += 1
            # Check if neighbors are connected (within reasonable distance)
            dist_ij = distance_matrix[neighbors[i], neighbors[j]]
            max_dist = max(distance_matrix[node, neighbors[i]], 
                          distance_matrix[node, neighbors[j]])
            if dist_ij < max_dist * 1.5:
                triangles += 1
    
    return triangles / max(possible, 1)

def _compute_mutual_nearest_neighbors(distance_matrix: np.ndarray, k: int) -> np.ndarray:
    """Compute mutual nearest neighbors mask."""
    n = distance_matrix.shape[0]
    mask = np.zeros((n, n), dtype=bool)
    
    # Precompute nearest neighbors for all nodes
    nn_sets = []
    for i in range(n):
        nn_sets.append(set(np.argsort(distance_matrix[i])[1:k+1]))
    
    # Check mutual nearest neighbors
    for i in range(n):
        for j in range(i+1, n):
            if j in nn_sets[i] and i in nn_sets[j]:
                mask[i, j] = mask[j, i] = True
    
    return mask

def _adaptive_heuristic_fusion(
    base_heuristic: np.ndarray,
    intra_cluster_boost: np.ndarray,
    mst_mask: np.ndarray,
    edge_betweenness: np.ndarray,
    embedding_similarity: np.ndarray,
    mutual_nn_mask: np.ndarray,
    distance_matrix: np.ndarray,
    n: int
) -> np.ndarray:
    """Adaptively fuse multiple heuristics with instance-aware weighting."""
    
    # Adaptive weights based on instance characteristics
    size_factor = min(1.0, n / 100)
    
    # Analyze distance distribution
    flat_dists = distance_matrix[np.triu_indices(n, 1)]
    dist_std = np.std(flat_dists)
    dist_mean = np.mean(flat_dists)
    variability = dist_std / (dist_mean + 1e-10)
    
    # Compute instance-aware weights
    weights = {
        'base': 1.0,
        'cluster': 0.8 + 0.4 * (1 - min(1.0, variability)),
        'mst': 2.0 + 0.5 * size_factor,
        'betweenness': 1.2 + 0.3 * size_factor,
        'embedding': 0.7 + 0.3 * (1 - min(1.0, variability)),
        'mutual_nn': 1.3 + 0.2 * size_factor
    }
    
    # Initialize combined heuristic
    heuristic = base_heuristic.copy()
    
    # Apply weighted contributions
    heuristic *= intra_cluster_boost ** weights['cluster']
    heuristic[mst_mask] *= weights['mst']
    heuristic *= edge_betweenness ** weights['betweenness']
    heuristic *= (1 + embedding_similarity * weights['embedding'])
    heuristic[mutual_nn_mask] *= weights['mutual_nn']
    
    # Add exploration noise
    noise = 1.0 + 0.03 * np.random.randn(n, n)
    np.fill_diagonal(noise, 1)
    heuristic *= noise
    
    return heuristic

def _hierarchical_sparsification(
    heuristic: np.ndarray,
    mst_mask: np.ndarray,
    distance_matrix: np.ndarray,
    n: int
) -> np.ndarray:
    """Hierarchical sparsification preserving connectivity."""
    
    sparsified = np.zeros_like(heuristic)
    
    # Level 1: Always keep MST edges
    sparsified[mst_mask] = heuristic[mst_mask]
    
    # Level 2: Adaptive per-node sparsification
    target_degree = min(max(4, int(np.log2(n) * 2.5)), n - 1)
    
    for i in range(n):
        # Get current row values
        row_values = heuristic[i].copy()
        row_values[i] = -np.inf  # Ignore self
        
        # Keep top-k edges per node
        if np.sum(row_values > 0) > target_degree:
            threshold = np.partition(row_values, -target_degree)[-target_degree]
            keep_mask = row_values >= threshold
        else:
            keep_mask = row_values > 0
        
        # Always keep MST edges
        keep_mask = keep_mask | mst_mask[i]
        
        sparsified[i] = heuristic[i] * keep_mask
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Level 3: Global sparsification threshold
    non_zero = sparsified[sparsified > 0]
    if len(non_zero) > n * 2:  # Too dense
        percentile = 100 - min(95, 70 + 0.2 * n)
        threshold = np.percentile(non_zero, percentile)
        sparsified[sparsified < threshold] = 0
    
    # Ensure connectivity
    components = _find_connected_components(sparsified > 0)
    if len(components) > 1:
        # Connect components with nearest neighbor edges
        sparsified = _connect_components(sparsified, distance_matrix, components)
    
    return sparsified

def _find_connected_components(adj_matrix: np.ndarray) -> list:
    """Find connected components in adjacency matrix."""
    n = adj_matrix.shape[0]
    visited = np.zeros(n, dtype=bool)
    components = []
    
    for i in range(n):
        if not visited[i]:
            component = []
            stack = [i]
            visited[i] = True
            
            while stack:
                node = stack.pop()
                component.append(node)
                neighbors = np.where(adj_matrix[node])[0]
                for neighbor in neighbors:
                    if not visited[neighbor]:
                        visited[neighbor] = True
                        stack.append(neighbor)
            
            components.append(component)
    
    return components

def _connect_components(
    heuristic: np.ndarray,
    distance_matrix: np.ndarray,
    components: list
) -> np.ndarray:
    """Connect disconnected components with shortest edges."""
    connected = heuristic.copy()
    
    # Connect each component to the nearest other component
    for i in range(len(components)):
        comp_i = components[i]
        if i == len(components) - 1:
            break
            
        # Find best connection to next component
        best_dist = np.inf
        best_pair = (0, 0)
        
        for node_i in comp_i:
            for j in range(i + 1, len(components)):
                for node_j in components[j]:
                    dist = distance_matrix[node_i, node_j]
                    if dist < best_dist:
                        best_dist = dist
                        best_pair = (node_i, node_j)
        
        # Add connection with appropriate heuristic value
        i, j = best_pair
        connected[i, j] = connected[j, i] = 1.0 / (best_dist + 1e-12)
    
    return connected
