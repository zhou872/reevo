import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import KDTree
from sklearn.manifold import MDS
from sklearn.cluster import MiniBatchKMeans
import warnings
from typing import Tuple, Dict
from concurrent.futures import ThreadPoolExecutor
import multiprocessing as mp

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with adaptive learning,
    cluster-enhanced triangle boosting, and parallelized computation.
    
    Key improvements over v1:
    1. Adaptive weight learning based on problem characteristics
    2. Clustering-enhanced triangle boosting
    3. Efficient landmark-based approximation for large instances
    4. Parallel processing for computationally intensive steps
    
    Returns:
        Heuristic matrix of same shape as distance_matrix,
        where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # 1. Base reciprocal heuristic
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # 2. Compute feature matrices in parallel where possible
    spatial_promising = _compute_spatial_clustering_parallel(distance_matrix, n)
    mst_promising = _compute_mst_promises(distance_matrix)
    local_factors = _compute_local_factors_parallel(distance_matrix, n)
    
    # 3. Adaptive weight learning
    adaptive_weights = _learn_adaptive_weights(
        reciprocal_weights, spatial_promising, mst_promising,
        local_factors, distance_matrix
    )
    
    # 4. Combine factors with learned weights
    heuristics_matrix = _combine_with_adaptive_weights(
        reciprocal_weights, spatial_promising, mst_promising,
        local_factors, adaptive_weights, distance_matrix
    )
    
    # 5. Adaptive sparsification
    sparsified = _adaptive_sparsification_v2(heuristics_matrix, n)
    
    # 6. Cluster-enhanced triangle boosting
    final_heuristics = _cluster_enhanced_triangle_boosting(
        sparsified, distance_matrix, n
    )
    
    # 7. Final normalization
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics


def _compute_spatial_clustering_parallel(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute spatial clustering factor with parallel processing and landmark approximation."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    if n <= 1000:
        # Use parallel MDS for accurate coordinates
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(3, n-1),
                dissimilarity='precomputed',
                random_state=42,
                n_init=2,
                max_iter=100,
                n_jobs=-1  # Enable parallel MDS computation
            )
            coords = mds.fit_transform(distance_matrix)
    else:
        # Efficient landmark approximation for very large instances
        landmarks = min(100, n // 10)
        landmark_indices = np.random.choice(n, landmarks, replace=False)
        landmark_dists = distance_matrix[landmark_indices]
        
        # Approximate coordinates using distance to landmarks
        coords = np.zeros((n, landmarks))
        for i in range(n):
            coords[i] = np.array([distance_matrix[i, idx] for idx in landmark_indices])
    
    # Build KD-tree for spatial queries
    tree = KDTree(coords)
    
    # Adaptive k: log2(n) but at least 3
    k = max(3, min(int(np.log2(n)) + 2, n - 1))
    
    # Parallel nearest neighbor queries
    def query_neighbors(i):
        if n <= 1000:
            distances, indices = tree.query(coords[i], k=k+1)
            indices = indices[1:] if distances[0] == 0 else indices[:k]
        else:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            indices = np.argpartition(dists, k)[:k]
        return i, indices
    
    with ThreadPoolExecutor(max_workers=mp.cpu_count()) as executor:
        results = list(executor.map(query_neighbors, range(n)))
    
    for i, indices in results:
        spatial_promising[i, indices] = 1.0
    
    return spatial_promising


def _compute_mst_promises(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute promises based on Minimum Spanning Tree edges."""
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    mst_promising = mst_edges + mst_edges.T
    
    n = distance_matrix.shape[0]
    
    # Consider edges that are "almost" in MST
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Get indices of 4 closest nodes (excluding self)
        closest = np.argsort(dists)[:4]
        for j in closest:
            if mst_promising[i, j] == 0:
                mst_promising[i, j] = 0.6 - 0.1 * np.where(closest == j)[0][0]
    
    return mst_promising


def _compute_local_factors_parallel(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute local connectivity factors with parallel processing."""
    # For each node, compute statistics of its local neighborhood
    m = min(5, n - 1)
    local_stats = np.zeros((n, 2))  # [mean_dist, std_dist]
    
    def compute_node_stats(i):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)[:m]
        return np.mean(sorted_dists), np.std(sorted_dists) + 1e-12
    
    with ThreadPoolExecutor(max_workers=mp.cpu_count()) as executor:
        stats_list = list(executor.map(compute_node_stats, range(n)))
    
    for i, (mean_val, std_val) in enumerate(stats_list):
        local_stats[i, 0] = mean_val
        local_stats[i, 1] = std_val
    
    # Compute local factor matrix with vectorized operations
    mean_i = local_stats[:, 0].reshape(-1, 1)
    mean_j = local_stats[:, 0].reshape(1, -1)
    std_i = local_stats[:, 1].reshape(-1, 1)
    std_j = local_stats[:, 1].reshape(1, -1)
    
    # Stat similarity
    stat_similarity = np.exp(-np.abs(mean_i - mean_j) / (std_i + std_j))
    
    # Distance ratio
    mean_dist_matrix = (mean_i + mean_j) / 2
    dist_ratio = distance_matrix / (mean_dist_matrix + 1e-12)
    
    # Combine factors
    local_factors = stat_similarity * np.exp(-0.5 * dist_ratio**2)
    np.fill_diagonal(local_factors, 0)
    
    return local_factors


def _learn_adaptive_weights(reciprocal_weights: np.ndarray,
                           spatial_promising: np.ndarray,
                           mst_promising: np.ndarray,
                           local_factors: np.ndarray,
                           distance_matrix: np.ndarray) -> Dict[str, float]:
    """Learn adaptive weights based on problem characteristics."""
    n = distance_matrix.shape[0]
    
    # Analyze problem characteristics
    triu_indices = np.triu_indices(n, k=1)
    distances_upper = distance_matrix[triu_indices]
    
    mean_dist = np.mean(distances_upper)
    std_dist = np.std(distances_upper)
    cv = std_dist / (mean_dist + 1e-12)  # Coefficient of variation
    
    # Spatial regularity
    if n <= 1000:
        # Compute stress of MDS embedding
        try:
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                mds = MDS(n_components=2, dissimilarity='precomputed',
                         random_state=42, n_init=1, max_iter=50)
                coords = mds.fit_transform(distance_matrix)
                embedded_dists = np.sqrt(((coords[:, np.newaxis] - coords) ** 2).sum(axis=2))
                stress = np.mean((distance_matrix - embedded_dists) ** 2)
                spatial_quality = np.exp(-stress / mean_dist)
        except:
            spatial_quality = 0.5
    else:
        spatial_quality = 0.5
    
    # Compute feature reliabilities
    reciprocal_reliability = 1.0 / (1.0 + cv)  # Less reliable for high variance
    spatial_reliability = spatial_quality
    mst_reliability = 0.8 + 0.2 * cv  # More important for irregular problems
    local_reliability = 0.7 + 0.3 * (1 - cv)  # More reliable for regular problems
    
    # Normalize reliabilities to get weights
    total = (reciprocal_reliability + spatial_reliability + 
             mst_reliability + local_reliability)
    
    weights = {
        'reciprocal': reciprocal_reliability / total,
        'spatial': spatial_reliability / total,
        'mst': mst_reliability / total,
        'local': local_reliability / total
    }
    
    return weights


def _combine_with_adaptive_weights(reciprocal_weights: np.ndarray,
                                  spatial_promising: np.ndarray,
                                  mst_promising: np.ndarray,
                                  local_factors: np.ndarray,
                                  weights: Dict[str, float],
                                  distance_matrix: np.ndarray) -> np.ndarray:
    """Combine all heuristic factors with adaptive weights."""
    n = reciprocal_weights.shape[0]
    
    # Normalize each feature matrix
    def normalize_matrix(mat):
        max_val = np.max(mat)
        if max_val > 0:
            return mat / max_val
        return mat
    
    norm_reciprocal = normalize_matrix(reciprocal_weights)
    norm_spatial = normalize_matrix(spatial_promising)
    norm_mst = normalize_matrix(mst_promising)
    norm_local = normalize_matrix(local_factors)
    
    # Combine with learned weights
    heuristics_matrix = (
        weights['reciprocal'] * norm_reciprocal +
        weights['spatial'] * norm_spatial +
        weights['mst'] * norm_mst +
        weights['local'] * norm_local
    )
    
    # Additional multiplicative interaction for strongly correlated features
    if weights['mst'] > 0.2 and weights['spatial'] > 0.2:
        interaction = norm_mst * norm_spatial
        if np.max(interaction) > 0:
            interaction /= np.max(interaction)
        heuristics_matrix += 0.15 * interaction
    
    # Ensure symmetry
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    return heuristics_matrix


def _adaptive_sparsification_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Adaptive sparsification with connectivity preservation."""
    if n <= 10:
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Adaptive sparsity based on problem size and density
    if n <= 50:
        keep_ratio = 0.5
    elif n <= 200:
        keep_ratio = 0.3
    elif n <= 1000:
        keep_ratio = 0.2
    else:
        keep_ratio = 0.1
    
    # Ensure each node maintains minimum connectivity
    min_edges = max(3, int(np.log2(n)))
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        # Determine number of edges to keep
        num_to_keep = max(min_edges, int(keep_ratio * n))
        num_to_keep = min(num_to_keep, n - 1)
        
        if np.any(row > 0):
            # Get indices of top edges
            top_indices = np.argpartition(-row, num_to_keep)[:num_to_keep]
            threshold = np.min(row[top_indices])
            
            # Keep edges above threshold
            mask = row >= threshold
            sparsified[i] = heuristics_matrix[i] * mask
    
    # Ensure symmetry and connectivity
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Add back essential edges for connectivity
    mst = minimum_spanning_tree(-heuristics_matrix)  # MST on negative to get max edges
    mst_edges = (mst.toarray() > 0).astype(float)
    sparsified = np.maximum(sparsified, mst_edges)
    
    return sparsified


def _cluster_enhanced_triangle_boosting(heuristics_matrix: np.ndarray,
                                       distance_matrix: np.ndarray,
                                       n: int) -> np.ndarray:
    """Boost scores using clustering information and triangle completion."""
    boosted = heuristics_matrix.copy()
    
    if n < 10:
        # Simple triangle boosting for small instances
        return _simple_triangle_boosting(boosted, n)
    
    # 1. Cluster nodes to identify natural groups
    n_clusters = max(2, min(10, int(np.sqrt(n))))
    
    if n > 1000:
        # Use MiniBatchKMeans for large instances
        kmeans = MiniBatchKMeans(n_clusters=n_clusters, random_state=42,
                                batch_size=min(1000, n // 10))
    else:
        # Use regular KMeans for smaller instances
        from sklearn.cluster import KMeans
        kmeans = KMeans(n_clusters=n_clusters, random_state=42, n_init=3)
    
    # Use distance matrix or generate coordinates for clustering
    if n <= 1000:
        try:
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                mds = MDS(n_components=min(3, n_clusters*2),
                         dissimilarity='precomputed',
                         random_state=42, n_init=1, max_iter=50)
                coords = mds.fit_transform(distance_matrix)
                labels = kmeans.fit_predict(coords)
        except:
            # Fallback: use distance matrix rows as features
            labels = kmeans.fit_predict(distance_matrix)
    else:
        # For large instances, use landmark distances
        landmarks = min(50, n // 20)
        landmark_indices = np.random.choice(n, landmarks, replace=False)
        landmark_features = distance_matrix[:, landmark_indices]
        labels = kmeans.fit_predict(landmark_features)
    
    # 2. Boost edges within clusters
    for cluster_id in range(n_clusters):
        cluster_mask = (labels == cluster_id)
        cluster_indices = np.where(cluster_mask)[0]
        
        if len(cluster_indices) > 1:
            # Boost intra-cluster edges
            for i in cluster_indices:
                for j in cluster_indices:
                    if i < j and boosted[i, j] > 0:
                        boost_factor = 1.0 + 0.3 * np.log1p(len(cluster_indices))
                        boosted[i, j] = boosted[j, i] = boosted[i, j] * boost_factor
    
    # 3. Boost inter-cluster edges based on cluster proximity
    # Compute average distance between clusters
    cluster_distances = np.zeros((n_clusters, n_clusters))
    for c1 in range(n_clusters):
        for c2 in range(c1 + 1, n_clusters):
            mask_c1 = (labels == c1)
            mask_c2 = (labels == c2)
            
            if np.any(mask_c1) and np.any(mask_c2):
                # Sample distances between clusters
                sample_size = min(20, np.sum(mask_c1) * np.sum(mask_c2))
                indices_c1 = np.where(mask_c1)[0][:min(10, np.sum(mask_c1))]
                indices_c2 = np.where(mask_c2)[0][:min(10, np.sum(mask_c2))]
                
                sampled_dists = []
                for i in indices_c1:
                    for j in indices_c2:
                        sampled_dists.append(distance_matrix[i, j])
                
                if sampled_dists:
                    cluster_distances[c1, c2] = cluster_distances[c2, c1] = np.median(sampled_dists)
    
    # Boost edges between close clusters
    if n_clusters > 1:
        # Find minimum spanning tree of clusters
        cluster_mst = minimum_spanning_tree(cluster_distances)
        cluster_mst_edges = (cluster_mst.toarray() > 0)
        
        for c1 in range(n_clusters):
            for c2 in range(c1 + 1, n_clusters):
                if cluster_mst_edges[c1, c2] or cluster_mst_edges[c2, c1]:
                    # Boost edges between clusters connected in cluster MST
                    mask_c1 = (labels == c1)
                    mask_c2 = (labels == c2)
                    
                    indices_c1 = np.where(mask_c1)[0]
                    indices_c2 = np.where(mask_c2)[0]
                    
                    for i in indices_c1[:min(5, len(indices_c1))]:
                        for j in indices_c2[:min(5, len(indices_c2))]:
                            if boosted[i, j] > 0:
                                boost_factor = 1.0 + 0.2 / (1.0 + cluster_distances[c1, c2])
                                boosted[i, j] = boosted[j, i] = boosted[i, j] * boost_factor
    
    # 4. Add common neighbor triangle boosting
    boosted = _simple_triangle_boosting(boosted, n)
    
    return boosted


def _simple_triangle_boosting(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Simple triangle boosting based on common neighbors."""
    boosted = heuristics_matrix.copy()
    
    if n > 500:
        # Use sampling for large graphs
        sample_size = min(200, n)
        sampled_nodes = np.random.choice(n, sample_size, replace=False)
        
        for i in sampled_nodes:
            neighbors_i = set(np.where(heuristics_matrix[i] > 0)[0])
            for j in neighbors_i:
                if j <= i:
                    continue
                neighbors_j = set(np.where(heuristics_matrix[j] > 0)[0])
                common = len(neighbors_i & neighbors_j)
                if common > 0:
                    boost = 1 + 0.15 * min(common, 5)
                    boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    else:
        # Exact computation for smaller graphs
        binary_mask = (heuristics_matrix > 0).astype(float)
        common_neighbors = binary_mask @ binary_mask.T
        
        for i in range(n):
            for j in range(i + 1, n):
                if heuristics_matrix[i, j] > 0:
                    common = common_neighbors[i, j]
                    if common > 0:
                        boost = 1 + 0.2 * min(common, 5)
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    
    return boosted
