import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.manifold import MDS
from sklearn.neighbors import KDTree
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple strategies:
    - Reciprocal distance weighting as base
    - Minimum Spanning Tree edge prioritization
    - Size-adaptive sparsification keeping top edges per node
    - Triangle-based clustering coefficient boost
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # Initialize heuristic matrix with reciprocal distances as base
    with np.errstate(divide='ignore', invalid='ignore'):
        heuristic_matrix = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(heuristic_matrix, 0)
    
    # Boost MST edges (they must be in any optimal solution)
    mst_boost = _compute_mst_boost(distance_matrix)
    heuristic_matrix *= (1 + 0.5 * mst_boost)  # Give MST edges 50% boost
    
    # Apply size-adaptive sparsification
    heuristic_matrix = _adaptive_sparsification_v2(heuristic_matrix, n)
    
    # Apply triangle-based clustering boost
    heuristic_matrix = _triangle_clustering_boost(heuristic_matrix)
    
    # Additional sparsification after boosting
    heuristic_matrix = _final_sparsification(heuristic_matrix, n)
    
    # Ensure symmetry
    heuristic_matrix = (heuristic_matrix + heuristic_matrix.T) / 2
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
    
    np.fill_diagonal(heuristic_matrix, 0)
    return heuristic_matrix

def _compute_mst_boost(distance_matrix: np.ndarray) -> np.ndarray:
    """Boost edges that are part of the Minimum Spanning Tree."""
    n = distance_matrix.shape[0]
    mst = minimum_spanning_tree(distance_matrix)
    mst_matrix = mst.toarray()
    
    # Create symmetric MST indicator matrix
    mst_boost = (mst_matrix > 0).astype(float)
    mst_boost = mst_boost + mst_boost.T
    
    return mst_boost

def _adaptive_sparsification_v2(heuristic_matrix: np.ndarray, n: int) -> np.ndarray:
    """
    Size-adaptive sparsification that keeps a dynamic number of top edges per node.
    Based on log10(n) with minimum of 3 edges.
    """
    if n <= 3:
        return heuristic_matrix
    
    sparsified = np.zeros_like(heuristic_matrix)
    
    # Calculate dynamic k based on log10(n) with minimum of 3
    k = max(3, int(np.log10(n)) + 2)
    
    for i in range(n):
        row = heuristic_matrix[i].copy()
        row[i] = -np.inf  # Ignore diagonal
        
        # Get indices of top k edges
        top_indices = np.argpartition(-row, k)[:k]
        
        # Keep only top k edges
        sparsified[i, top_indices] = row[top_indices]
    
    # Make symmetric by keeping edges that appear in either direction
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified

def _triangle_clustering_boost(heuristic_matrix: np.ndarray) -> np.ndarray:
    """
    Boost edges based on triangle clustering coefficient.
    Edges that form more triangles (common neighbors) get higher boost.
    """
    n = heuristic_matrix.shape[0]
    if n < 3:
        return heuristic_matrix
    
    # Create binary adjacency matrix from non-zero heuristic values
    binary_adj = (heuristic_matrix > 0).astype(float)
    
    # Efficient triangle counting using matrix multiplication
    # (A³)_ii gives twice the number of triangles for node i
    # Common neighbors for edge (i,j) = (A²)_ij
    A_squared = binary_adj @ binary_adj
    
    boosted_matrix = heuristic_matrix.copy()
    
    # Boost edges based on number of common neighbors
    for i in range(n):
        for j in range(i+1, n):
            if binary_adj[i, j] > 0:
                common_neighbors = A_squared[i, j]
                if common_neighbors > 0:
                    # Boost factor: 1 + 0.2 per common neighbor, capped at 2.0
                    boost_factor = 1 + min(0.2 * common_neighbors, 1.0)
                    boosted_matrix[i, j] = boosted_matrix[j, i] = heuristic_matrix[i, j] * boost_factor
    
    return boosted_matrix

def _final_sparsification(heuristic_matrix: np.ndarray, n: int) -> np.ndarray:
    """
    Final sparsification to remove very weak edges after all boosts.
    Keeps top edges based on node degree distribution.
    """
    if n <= 5:
        return heuristic_matrix
    
    sparsified = np.zeros_like(heuristic_matrix)
    
    # Calculate degree for each node (number of non-zero edges)
    degrees = np.sum(heuristic_matrix > 0, axis=1)
    
    for i in range(n):
        row = heuristic_matrix[i].copy()
        row[i] = -np.inf
        
        # Keep more edges for nodes with lower degree (to ensure connectivity)
        # Base k on log10(n) but adjust for node degree
        base_k = max(3, int(np.log10(n)) + 2)
        degree_factor = max(0.5, 1.0 - degrees[i] / (2 * n))  # More edges for poorly connected nodes
        k = max(3, int(base_k * degree_factor))
        
        # Get indices of top k edges
        top_indices = np.argpartition(-row, k)[:k]
        
        # Keep only top k edges
        sparsified[i, top_indices] = row[top_indices]
    
    # Final symmetry enforcement
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified
