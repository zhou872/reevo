import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
import warnings
from collections import defaultdict
import math

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with adaptive ensemble learning
    and dynamic resolution scaling.
    
    Key innovations:
    1. Dynamic resolution scaling based on problem complexity
    2. Ensemble of diverse heuristic strategies with learned weights
    3. Machine-learning inspired feature extraction and combination
    4. Connectivity-preserving refinement with quality feedback
    5. Efficient approximations for large-scale instances
    
    Args:
        distance_matrix: Square symmetric distance matrix (n x n)
        
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        result = np.zeros_like(distance_matrix)
        if n == 2:
            result[0, 1] = result[1, 0] = 1.0
        return result
    
    # Phase 1: Dynamic resolution determination
    resolution_profile = _determine_resolution_profile(n, distance_matrix)
    
    # Phase 2: Ensemble of base heuristic generators
    ensemble_results = _generate_ensemble_heuristics(
        distance_matrix, n, resolution_profile
    )
    
    # Phase 3: Adaptive weight learning for ensemble combination
    weights = _learn_ensemble_weights(
        distance_matrix, ensemble_results, n, resolution_profile
    )
    
    # Phase 4: Intelligent ensemble combination
    combined = _intelligent_ensemble_combination(
        ensemble_results, weights, distance_matrix, n
    )
    
    # Phase 5: Resolution-aware refinement
    refined = _resolution_aware_refinement(
        combined, distance_matrix, n, resolution_profile
    )
    
    # Phase 6: Quality feedback enhancement
    enhanced = _quality_feedback_enhancement(
        refined, distance_matrix, n, resolution_profile
    )
    
    # Final processing
    np.fill_diagonal(enhanced, 0)
    max_val = np.max(enhanced)
    if max_val > 0:
        enhanced = enhanced / max_val
    
    return enhanced


def _determine_resolution_profile(n: int, distance_matrix: np.ndarray) -> dict:
    """Determine appropriate resolution and methods based on problem characteristics."""
    profile = {}
    
    # Problem size categories
    if n <= 50:
        profile['size_category'] = 'small'
        profile['spatial_method'] = 'exact'
        profile['clustering_levels'] = 3
        profile['sampling_rate'] = 1.0
    elif n <= 500:
        profile['size_category'] = 'medium'
        profile['spatial_method'] = 'landmark'
        profile['clustering_levels'] = 3
        profile['sampling_rate'] = 0.7
    elif n <= 5000:
        profile['size_category'] = 'large'
        profile['spatial_method'] = 'fast_landmark'
        profile['clustering_levels'] = 2
        profile['sampling_rate'] = 0.3
    else:
        profile['size_category'] = 'very_large'
        profile['spatial_method'] = 'ultra_fast'
        profile['clustering_levels'] = 1
        profile['sampling_rate'] = 0.1
    
    # Problem structure analysis
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    avg_dist = np.mean(distances)
    std_dist = np.std(distances)
    cv = std_dist / (avg_dist + 1e-12)
    
    # Complexity metrics
    profile['distance_cv'] = cv
    profile['sparsity'] = 1.0  # Complete graph
    profile['requires_spatial'] = cv > 0.1
    
    # Memory and computation constraints
    max_edges = n * (n - 1) // 2
    if max_edges > 1e7:  # Very large, need aggressive sparsification
        profile['aggressive_sparsify'] = True
        profile['max_edges_per_node'] = max(3, int(np.log2(n)))
    else:
        profile['aggressive_sparsify'] = False
        profile['max_edges_per_node'] = max(4, int(np.log2(n) * 2))
    
    return profile


def _generate_ensemble_heuristics(
    distance_matrix: np.ndarray, n: int, profile: dict
) -> dict:
    """Generate diverse heuristic strategies for ensemble learning."""
    results = {}
    
    # Strategy 1: Reciprocal distance heuristic (fast baseline)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(reciprocal, 0)
    max_val = np.max(reciprocal)
    if max_val > 0:
        reciprocal = reciprocal / max_val
    results['reciprocal'] = reciprocal
    
    # Strategy 2: MST-based heuristic (connectivity guarantee)
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    mst_heuristic = np.zeros_like(distance_matrix)
    mst_heuristic[mst_mask] = 1.0
    
    # Add near-MST edges
    k_near = min(5, n - 1)
    near_mst = np.zeros((n, n))
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        closest = np.argsort(dists)[:k_near]
        for j in closest:
            if not mst_mask[i, j]:
                near_mst[i, j] = 0.5
    mst_heuristic += near_mst
    mst_heuristic = np.minimum(mst_heuristic, 1.0)
    results['mst'] = mst_heuristic
    
    # Strategy 3: Spatial clustering heuristic (if needed)
    if profile['requires_spatial']:
        spatial_heuristic = _spatial_clustering_heuristic(
            distance_matrix, n, profile
        )
        results['spatial'] = spatial_heuristic
    else:
        results['spatial'] = np.zeros_like(distance_matrix)
    
    # Strategy 4: Density-aware heuristic
    density_heuristic = _density_aware_heuristic(distance_matrix, n)
    results['density'] = density_heuristic
    
    # Strategy 5: Triangle quality heuristic
    if n <= 1000 or profile['sampling_rate'] > 0.3:
        triangle_heuristic = _triangle_quality_heuristic(
            distance_matrix, n, profile
        )
        results['triangle'] = triangle_heuristic
    else:
        results['triangle'] = np.zeros_like(distance_matrix)
    
    # Strategy 6: Graph centrality heuristic
    centrality_heuristic = _graph_centrality_heuristic(distance_matrix, n)
    results['centrality'] = centrality_heuristic
    
    # Ensure all heuristics are symmetric
    for key in results:
        results[key] = (results[key] + results[key].T) / 2
    
    return results


def _spatial_clustering_heuristic(
    distance_matrix: np.ndarray, n: int, profile: dict
) -> np.ndarray:
    """Generate heuristic based on spatial clustering."""
    # Generate spatial coordinates
    if profile['spatial_method'] == 'exact' and n <= 50:
        # Use classical MDS for small instances
        coords = _exact_mds(distance_matrix, n)
    elif profile['spatial_method'] in ['landmark', 'fast_landmark']:
        # Use landmark-based embedding
        n_landmarks = min(max(10, int(np.sqrt(n))), 50)
        if profile['spatial_method'] == 'fast_landmark':
            n_landmarks = min(15, n_landmarks)
        coords = _landmark_embedding(distance_matrix, n, n_landmarks)
    else:
        # Ultra-fast: use pivot distances
        n_pivots = min(5, n)
        pivots = np.random.choice(n, n_pivots, replace=False)
        coords = np.zeros((n, n_pivots))
        for i in range(n):
            coords[i] = distance_matrix[i, pivots]
    
    # Cluster nodes
    n_clusters = min(max(2, n // 20), 20)
    labels = _fast_clustering(coords, n_clusters)
    
    # Generate heuristic based on clusters
    heuristic = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            if labels[i] == labels[j]:
                # Same cluster: boost based on distance
                heuristic[i, j] = 1.0 / (1.0 + distance_matrix[i, j] / np.mean(distance_matrix))
            else:
                # Different clusters: small boost for connections
                heuristic[i, j] = 0.2 / (1.0 + distance_matrix[i, j] / np.mean(distance_matrix))
    
    # Make symmetric
    heuristic = heuristic + heuristic.T
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    return heuristic


def _exact_mds(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute classical MDS coordinates (exact for small n)."""
    # Center the distance matrix
    H = np.eye(n) - np.ones((n, n)) / n
    B = -0.5 * H @ (distance_matrix ** 2) @ H
    
    # Eigen decomposition
    eigenvalues, eigenvectors = np.linalg.eigh(B)
    
    # Keep positive eigenvalues
    idx = eigenvalues > 1e-10
    eigenvalues = eigenvalues[idx]
    eigenvectors = eigenvectors[:, idx]
    
    # Return coordinates in 2D
    n_components = min(2, len(eigenvalues))
    coords = eigenvectors[:, -n_components:] * np.sqrt(eigenvalues[-n_components:])
    
    return coords


def _landmark_embedding(
    distance_matrix: np.ndarray, n: int, n_landmarks: int
) -> np.ndarray:
    """Generate embedding using farthest-point landmark sampling."""
    if n_landmarks >= n:
        n_landmarks = n - 1
    
    # Farthest-point sampling for landmarks
    landmarks = [np.random.randint(n)]
    for _ in range(1, n_landmarks):
        min_dists = np.min(distance_matrix[:, landmarks], axis=1)
        next_landmark = np.argmax(min_dists)
        landmarks.append(next_landmark)
    landmarks = np.array(landmarks)
    
    # Use distances to landmarks as coordinates
    coords = distance_matrix[:, landmarks]
    
    # Simple normalization
    mean_coords = np.mean(coords, axis=0)
    std_coords = np.std(coords, axis=0) + 1e-12
    coords = (coords - mean_coords) / std_coords
    
    return coords


def _fast_clustering(coords: np.ndarray, n_clusters: int) -> np.ndarray:
    """Fast k-means-like clustering."""
    n = coords.shape[0]
    if n_clusters >= n:
        return np.arange(n)
    
    # Initialize centers with farthest-point sampling
    centers = [np.random.randint(n)]
    for _ in range(1, n_clusters):
        dists_to_centers = np.min(
            np.linalg.norm(coords[None, :] - coords[centers, None], axis=-1),
            axis=0
        )
        centers.append(np.argmax(dists_to_centers))
    
    # Simple assignment
    labels = np.zeros(n, dtype=int)
    for i in range(n):
        dists = np.linalg.norm(coords[i] - coords[centers], axis=1)
        labels[i] = np.argmin(dists)
    
    return labels


def _density_aware_heuristic(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Generate heuristic based on local density patterns."""
    heuristic = np.zeros_like(distance_matrix)
    
    # Compute local densities
    k = min(10, max(3, n // 20))
    densities = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if k < n - 1:
            nn_dists = np.partition(dists, k)[:k]
        else:
            nn_dists = dists[dists < np.inf]
        densities[i] = 1.0 / (np.mean(nn_dists) + 1e-12)
    
    # Normalize densities
    max_density = np.max(densities)
    if max_density > 0:
        densities = densities / max_density
    
    # Generate heuristic
    for i in range(n):
        for j in range(i + 1, n):
            # Similar density nodes are preferred
            density_sim = 1.0 - np.abs(densities[i] - densities[j])
            # Combined with distance
            heuristic[i, j] = density_sim / (1.0 + distance_matrix[i, j] / np.mean(distance_matrix))
    
    # Make symmetric
    heuristic = heuristic + heuristic.T
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    return heuristic


def _triangle_quality_heuristic(
    distance_matrix: np.ndarray, n: int, profile: dict
) -> np.ndarray:
    """Generate heuristic based on triangle inequality quality."""
    heuristic = np.zeros_like(distance_matrix)
    
    # Sampling-based approach for efficiency
    sample_size = int(n * profile['sampling_rate'])
    if sample_size < 10:
        sample_size = min(10, n)
    
    if sample_size >= n:
        indices = np.arange(n)
    else:
        indices = np.random.choice(n, sample_size, replace=False)
    
    # Precompute nearest neighbors for sampled nodes
    k_neighbors = min(15, n - 1)
    neighbors = {}
    
    for i in indices:
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if k_neighbors < n - 1:
            neighbor_idx = np.argpartition(dists, k_neighbors)[:k_neighbors]
        else:
            neighbor_idx = np.where(dists < np.inf)[0]
        neighbors[i] = neighbor_idx
    
    # Evaluate triangle quality
    for i in indices:
        for j in neighbors[i]:
            if j <= i:
                continue
            
            # Find common neighbors
            common = np.intersect1d(neighbors.get(i, []), neighbors.get(j, []))
            if len(common) == 0:
                continue
            
            # Evaluate triangle inequality satisfaction
            triangle_quality = 0
            for k in common[:min(5, len(common))]:  # Limit to 5 common neighbors
                # Triangle inequality: d(i,j) ≤ d(i,k) + d(k,j)
                violation = distance_matrix[i, j] - (distance_matrix[i, k] + distance_matrix[k, j])
                if violation <= 0:
                    triangle_quality += 1.0
                else:
                    triangle_quality += max(0, 1.0 - violation / distance_matrix[i, j])
            
            if len(common) > 0:
                triangle_quality = triangle_quality / min(5, len(common))
                heuristic[i, j] = triangle_quality
    
    # Propagate to full matrix (simplified)
    for i in range(n):
        for j in range(i + 1, n):
            if heuristic[i, j] == 0:
                # Estimate from similar nodes
                if i in indices or j in indices:
                    continue
                # Simple fallback
                heuristic[i, j] = 0.5 / (1.0 + distance_matrix[i, j] / np.mean(distance_matrix))
    
    # Make symmetric
    heuristic = heuristic + heuristic.T
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    return heuristic


def _graph_centrality_heuristic(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Generate heuristic based on graph centrality measures."""
    heuristic = np.zeros_like(distance_matrix)
    
    # Simple centrality approximation
    # Closeness centrality approximation
    centrality = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = 0
        centrality[i] = 1.0 / (np.sum(dists) + 1e-12)
    
    # Normalize centrality
    max_centrality = np.max(centrality)
    if max_centrality > 0:
        centrality = centrality / max_centrality
    
    # Generate heuristic: connect central nodes to each other
    for i in range(n):
        for j in range(i + 1, n):
            centrality_product = centrality[i] * centrality[j]
            heuristic[i, j] = centrality_product / (1.0 + distance_matrix[i, j] / np.mean(distance_matrix))
    
    # Make symmetric
    heuristic = heuristic + heuristic.T
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    return heuristic


def _learn_ensemble_weights(
    distance_matrix: np.ndarray,
    ensemble_results: dict,
    n: int,
    profile: dict
) -> dict:
    """Learn optimal weights for combining ensemble heuristics."""
    weights = {}
    
    # Base weights
    weights['reciprocal'] = 1.0  # Always include
    
    # MST weight: important for connectivity
    mst_weight = 1.5 - 0.3 * np.log10(max(n, 10)) / np.log10(100)
    weights['mst'] = np.clip(mst_weight, 1.2, 1.8)
    
    # Spatial weight: depends on structure
    if profile['requires_spatial']:
        # Estimate spatial structure quality
        if 'spatial' in ensemble_results:
            spatial_var = np.var(ensemble_results['spatial'][np.triu_indices(n, k=1)])
            weights['spatial'] = 0.8 + 0.4 * min(1.0, spatial_var * 5)
        else:
            weights['spatial'] = 0.8
    else:
        weights['spatial'] = 0.0
    
    # Density weight: important for uniform vs clustered distributions
    if 'density' in ensemble_results:
        density_var = np.var(ensemble_results['density'][np.triu_indices(n, k=1)])
        weights['density'] = 0.6 + 0.3 * min(1.0, density_var * 3)
    else:
        weights['density'] = 0.6
    
    # Triangle weight: important for metric properties
    if 'triangle' in ensemble_results:
        triangle_mean = np.mean(ensemble_results['triangle'][np.triu_indices(n, k=1)])
        weights['triangle'] = 0.4 + 0.3 * triangle_mean
    else:
        weights['triangle'] = 0.0
    
    # Centrality weight: balances hub-and-spoke vs distributed
    if 'centrality' in ensemble_results:
        centrality_var = np.var(ensemble_results['centrality'][np.triu_indices(n, k=1)])
        weights['centrality'] = 0.3 + 0.2 * min(1.0, centrality_var * 2)
    else:
        weights['centrality'] = 0.0
    
    # Normalize weights to sum to 1
    total_weight = sum(weights.values())
    if total_weight > 0:
        for key in weights:
            weights[key] = weights[key] / total_weight
    
    return weights


def _intelligent_ensemble_combination(
    ensemble_results: dict,
    weights: dict,
    distance_matrix: np.ndarray,
    n: int
) -> np.ndarray:
    """Intelligently combine ensemble heuristics using learned weights."""
    combined = np.zeros_like(distance_matrix)
    
    # Weighted sum of heuristics
    for key, weight in weights.items():
        if key in ensemble_results and weight > 0:
            combined += weight * ensemble_results[key]
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    # Apply non-linear combination for edges with multiple strong signals
    triu_indices = np.triu_indices(n, k=1)
    for i, j in zip(triu_indices[0], triu_indices[1]):
        # Count how many heuristics strongly support this edge
        strong_support = 0
        for key in ensemble_results:
            if key in weights and weights[key] > 0:
                if ensemble_results[key][i, j] > 0.7:
                    strong_support += 1
        
        if strong_support >= 2:
            # Boost edges with multiple strong supports
            combined[i, j] = combined[j, i] = min(1.0, combined[i, j] * (1.0 + 0.2 * strong_support))
    
    # Normalize
    max_val = np.max(combined)
    if max_val > 0:
        combined = combined / max_val
    
    return combined


def _resolution_aware_refinement(
    heuristic_matrix: np.ndarray,
    distance_matrix: np.ndarray,
    n: int,
    profile: dict
) -> np.ndarray:
    """Refine heuristic matrix with resolution-aware methods."""
    refined = heuristic_matrix.copy()
    
    # Ensure MST edges are included
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Boost MST edges
    refined[mst_mask] = np.maximum(refined[mst_mask], 0.5)
    
    # Adaptive sparsification
    if profile['aggressive_sparsify']:
        max_edges = profile['max_edges_per_node']
        for i in range(n):
            row = refined[i].copy()
            row[i] = -np.inf
            
            # Keep MST edges
            mst_neighbors = np.where(mst_mask[i])[0]
            
            # Keep top non-MST edges
            non_mst_mask = ~mst_mask[i]
            non_mst_mask[i] = False
            non_mst_indices = np.where(non_mst_mask)[0]
            
            if len(non_mst_indices) > 0:
                non_mst_values = row[non_mst_indices]
                keep_count = max(0, max_edges - len(mst_neighbors))
                
                if keep_count > 0 and len(non_mst_indices) > keep_count:
                    threshold_idx = np.argpartition(-non_mst_values, keep_count - 1)[keep_count - 1]
                    threshold_val = non_mst_values[threshold_idx]
                    keep_non_mst = non_mst_indices[non_mst_values >= threshold_val]
                else:
                    keep_non_mst = non_mst_indices
            else:
                keep_non_mst = []
            
            # Zero out non-kept edges
            keep_mask = np.zeros(n, dtype=bool)
            keep_mask[np.concatenate([mst_neighbors, keep_non_mst])] = True
            keep_mask[i] = True  # Keep diagonal
            
            refined[i, ~keep_mask] = 0
    
    # Ensure connectivity
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        node = stack.pop()
        neighbors = np.where(refined[node] > 0)[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    if not np.all(visited):
        # Add minimum connections to make graph connected
        unvisited = np.where(~visited)[0]
        for node in unvisited:
            # Connect to closest visited node
            dists_to_visited = distance_matrix[node, visited]
            if len(dists_to_visited) > 0:
                closest_visited_idx = np.argmin(dists_to_visited)
                visited_nodes = np.where(visited)[0]
                closest = visited_nodes[closest_visited_idx]
                refined[node, closest] = refined[closest, node] = 0.3
                visited[node] = True
    
    # Make symmetric
    refined = np.maximum(refined, refined.T)
    
    return refined


def _quality_feedback_enhancement(
    heuristic_matrix: np.ndarray,
    distance_matrix: np.ndarray,
    n: int,
    profile: dict
) -> np.ndarray:
    """Enhance heuristic matrix based on quality feedback."""
    enhanced = heuristic_matrix.copy()
    
    # Only perform enhancement for medium and small instances
    if profile['size_category'] in ['small', 'medium']:
        # Sample promising edges for enhancement
        sample_edges = min(100, n * (n - 1) // 4)
        edge_indices = np.array(np.triu_indices(n, k=1)).T
        
        if len(edge_indices) > sample_edges:
            edge_scores = heuristic_matrix[edge_indices[:, 0], edge_indices[:, 1]]
            top_indices = np.argpartition(-edge_scores, sample_edges)[:sample_edges]
            sampled_edges = edge_indices[top_indices]
        else:
            sampled_edges = edge_indices
        
        # Enhance based on local triangle quality
        for i, j in sampled_edges:
            if enhanced[i, j] == 0:
                continue
            
            # Find common neighbors
            neighbors_i = set(np.where(enhanced[i] > 0)[0])
            neighbors_j = set(np.where(enhanced[j] > 0)[0])
            common = neighbors_i & neighbors_j
            
            if len(common) > 0:
                # Compute average quality of triangles
                qualities = []
                for k in common:
                    if i != k and j != k:
                        triangle_quality = (enhanced[i, k] + enhanced[j, k]) / 2
                        qualities.append(triangle_quality)
                
                if qualities:
                    avg_quality = np.mean(qualities)
                    if avg_quality > 0.5:
                        # Boost edges in high-quality triangles
                        boost = 1.0 + 0.15 * min(len(common), 5)
                        enhanced[i, j] = enhanced[j, i] = min(1.0, enhanced[i, j] * boost)
    
    # Final normalization
    np.fill_diagonal(enhanced, 0)
    max_val = np.max(enhanced)
    if max_val > 0:
        enhanced = enhanced / max_val
    
    return enhanced
