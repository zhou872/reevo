import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.manifold import MDS
from scipy.spatial import KDTree
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple promising indicators.
    
    Key improvements over v1:
    1. More adaptive spatial clustering with saturation
    2. Saturating triangle boosting to prevent overemphasis on dense clusters
    3. Improved factor combination with attention to local/global balance
    4. More aggressive but adaptive sparsification
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # Base heuristic: reciprocal of distances with logarithmic scaling
    with np.errstate(divide='ignore', invalid='ignore'):
        # Use log transformation to reduce extreme values
        safe_dist = distance_matrix + np.eye(n) * 1e-12
        reciprocal_weights = 1 / safe_dist
        np.fill_diagonal(reciprocal_weights, 0)
        
        # Apply logarithmic compression to handle large distance variations
        max_val = np.max(reciprocal_weights[reciprocal_weights < np.inf])
        if max_val > 0:
            log_scaled = np.log1p(reciprocal_weights / max_val)
            reciprocal_weights = log_scaled / np.max(log_scaled)
    
    # Factor 1: Spatial clustering with adaptive neighborhood
    spatial_promising = _compute_spatial_clustering_v2(distance_matrix, n)
    
    # Factor 2: Minimum Spanning Tree with extended neighborhood
    mst_promising = _compute_mst_promises_v2(distance_matrix)
    
    # Factor 3: Local connectivity statistics
    local_factors = _compute_local_factors_v2(distance_matrix, n)
    
    # Combine factors with adaptive weights
    heuristics_matrix = _combine_factors_v2(
        reciprocal_weights, spatial_promising, mst_promising,
        local_factors, distance_matrix
    )
    
    # Aggressive but adaptive sparsification
    sparsified = _adaptive_sparsification_v2(heuristics_matrix, n)
    
    # Saturating triangle boosting (limits effect in very dense clusters)
    boosted = _saturating_triangle_boosting(sparsified)
    
    # Normalize and ensure symmetry
    boosted = (boosted + boosted.T) / 2
    max_val = np.max(boosted)
    if max_val > 0:
        boosted = boosted / max_val
    
    np.fill_diagonal(boosted, 0)
    return boosted


def _compute_spatial_clustering_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Enhanced spatial clustering with adaptive neighborhood selection."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    # Choose method based on size
    if n <= 1000:
        # Use MDS for accurate coordinates
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(5, n-1),
                dissimilarity='precomputed',
                random_state=42,
                n_init=1,
                max_iter=50
            )
            coords = mds.fit_transform(distance_matrix)
    else:
        # Landmark-based approximation for large graphs
        landmarks = min(100, n // 10)
        landmark_indices = np.random.choice(n, landmarks, replace=False)
        landmark_dists = distance_matrix[:, landmark_indices]
        
        # Use distance to landmarks as coordinate approximation
        coords = landmark_dists
    
    # Build KD-tree for efficient spatial queries
    tree = KDTree(coords)
    
    # Adaptive neighborhood size based on log scale
    base_k = max(3, int(np.log2(n)) + 1)
    k = min(base_k * 2, n - 1)  # Larger neighborhood for more context
    
    # Find adaptive number of neighbors for each node
    for i in range(n):
        # Query for k+1 neighbors (including self)
        distances, indices = tree.query(coords[i], k=min(k+1, n))
        
        # Remove self if present
        if distances[0] < 1e-10:
            indices = indices[1:]
            distances = distances[1:]
        else:
            indices = indices[:k]
            distances = distances[:k]
        
        # Assign weights inversely proportional to distance rank
        for rank, (idx, dist) in enumerate(zip(indices, distances)):
            if dist > 0:
                weight = 1.0 / (1.0 + rank * 0.5)  # Decay with rank
                spatial_promising[i, idx] = weight
    
    return spatial_promising


def _compute_mst_promises_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Enhanced MST-based promises with extended neighborhood."""
    n = distance_matrix.shape[0]
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    # Start with MST edges having highest weight
    mst_promising = mst_edges * 2.0  # Boost MST edges
    
    # Also consider k-nearest neighbors for each node
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Get indices of k closest nodes (excluding self)
        k = min(5, n - 1)
        closest_indices = np.argpartition(dists, k)[:k]
        
        for j in closest_indices:
            # If not already in MST, give moderate weight
            if mst_promising[i, j] == 0:
                mst_promising[i, j] = 0.5
    
    # Make symmetric
    mst_promising = np.maximum(mst_promising, mst_promising.T)
    return mst_promising


def _compute_local_factors_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Enhanced local connectivity factors with neighborhood analysis."""
    # Adaptive neighborhood size
    m = min(8, n - 1)
    
    # Compute local statistics for each node
    local_stats = np.zeros((n, 3))  # [mean, std, median]
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)[:m]
        
        if len(sorted_dists) > 0:
            local_stats[i, 0] = np.mean(sorted_dists)
            local_stats[i, 1] = np.std(sorted_dists) + 1e-12
            local_stats[i, 2] = np.median(sorted_dists)
    
    # Compute local factor matrix
    local_factors = np.ones_like(distance_matrix)
    
    for i in range(n):
        for j in range(i + 1, n):
            # Similarity in local distance profiles
            mean_sim = np.exp(-abs(local_stats[i, 0] - local_stats[j, 0]) / 
                            (local_stats[i, 1] + local_stats[j, 1]))
            
            median_sim = np.exp(-abs(local_stats[i, 2] - local_stats[j, 2]) /
                              (local_stats[i, 1] + local_stats[j, 1]))
            
            # Distance relative to local context
            local_context = (local_stats[i, 0] + local_stats[j, 0]) / 2
            dist_factor = np.exp(-distance_matrix[i, j] / (local_context + 1e-12))
            
            # Combine factors
            local_factors[i, j] = local_factors[j, i] = (
                (mean_sim + median_sim) / 2 * dist_factor
            )
    
    return local_factors


def _combine_factors_v2(reciprocal_weights: np.ndarray,
                       spatial_promising: np.ndarray,
                       mst_promising: np.ndarray,
                       local_factors: np.ndarray,
                       distance_matrix: np.ndarray) -> np.ndarray:
    """Intelligent combination of all heuristic factors."""
    n = reciprocal_weights.shape[0]
    combined = np.zeros_like(reciprocal_weights)
    
    # Base weights from reciprocal distances
    base = reciprocal_weights.copy()
    
    # Apply spatial factor with adaptive strength
    spatial_strength = 1.5 + 0.5 * np.log1p(n / 100)  # Stronger for larger graphs
    spatial_adjusted = np.where(spatial_promising > 0,
                               base * (1 + spatial_strength * spatial_promising),
                               base * 0.7)
    
    # Apply MST factor
    mst_strength = 2.0  # Strong boost for MST edges
    mst_adjusted = np.where(mst_promising > 1,
                           spatial_adjusted * mst_strength,
                           np.where(mst_promising > 0,
                                   spatial_adjusted * 1.3,
                                   spatial_adjusted))
    
    # Apply local factors
    combined = mst_adjusted * local_factors
    
    # Add small bonus for very short edges
    with np.errstate(divide='ignore', invalid='ignore'):
        short_edge_bonus = 1 + 0.2 * np.exp(-distance_matrix / np.percentile(distance_matrix, 10))
        combined = combined * short_edge_bonus
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    return combined


def _adaptive_sparsification_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Enhanced adaptive sparsification keeping most promising edges."""
    sparsified = np.zeros_like(heuristics_matrix)
    
    if n <= 10:
        # Keep all edges for very small graphs
        return heuristics_matrix.copy()
    
    # Adaptive sparsity: logarithmic scaling
    base_keep = max(3, int(np.log2(n) * 3))
    keep_per_node = min(base_keep, n - 1)
    
    # Also consider percentile threshold
    percentile = max(70, 100 - n / 10)  # More aggressive for larger graphs
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        # Keep edges above percentile threshold AND top k
        threshold = np.percentile(row[row > -np.inf], percentile)
        top_indices = np.argpartition(-row, keep_per_node)[:keep_per_node]
        top_threshold = np.min(row[top_indices])
        
        # Use maximum of both thresholds
        final_threshold = max(threshold, top_threshold)
        
        # Keep edges above threshold
        mask = row >= final_threshold
        sparsified[i] = heuristics_matrix[i] * mask
    
    # Ensure symmetry by keeping union of kept edges
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified


def _saturating_triangle_boosting(heuristics_matrix: np.ndarray) -> np.ndarray:
    """
    Boost edges based on common neighbors with saturating effect.
    Prevents overemphasis on very dense clusters.
    """
    n = heuristics_matrix.shape[0]
    boosted = heuristics_matrix.copy()
    
    # Create binary mask for existing promising edges
    binary_mask = (heuristics_matrix > 0).astype(float)
    
    if n > 300:
        # Use sampling for large graphs
        sample_size = min(200, n)
        sampled_nodes = np.random.choice(n, sample_size, replace=False)
        
        for i in sampled_nodes:
            neighbors_i = np.where(binary_mask[i] > 0)[0]
            for j in neighbors_i:
                if j <= i:
                    continue
                neighbors_j = np.where(binary_mask[j] > 0)[0]
                common = len(np.intersect1d(neighbors_i, neighbors_j))
                
                if common > 0:
                    # Saturating function: 1 + tanh(common/3)
                    boost = 1 + np.tanh(common / 3.0)
                    boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    else:
        # Exact computation for smaller graphs
        # Count common neighbors efficiently
        common_neighbors = binary_mask @ binary_mask.T
        
        for i in range(n):
            for j in range(i + 1, n):
                if heuristics_matrix[i, j] > 0:
                    common = common_neighbors[i, j]
                    if common > 0:
                        # Saturating boost
                        boost = 1 + np.tanh(common / 3.0)
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    
    return boosted
