import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with adaptive sparsification,
    MST-based prioritization, and improved triangle closure boosting.
    
    Returns a matrix of edge promises where higher values indicate more
    promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # 1. Base reciprocal heuristics
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal, 0)
    
    # 2. Adaptive K-nearest neighbors sparsification
    # Dynamic k based on log2(n) with bounds
    base_k = max(3, int(np.log2(n)) * 2)
    k = min(base_k, n - 1)
    
    knn_matrix = np.zeros_like(distance_matrix, dtype=float)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf  # Exclude self
        if k < n - 1:
            # Efficient selection using partition
            nearest_idx = np.argpartition(dists, k)[:k]
        else:
            nearest_idx = np.argsort(dists)[:k]
        knn_matrix[i, nearest_idx] = 1.0
    
    # Symmetrize KNN edges
    knn_symmetric = (knn_matrix + knn_matrix.T) > 0
    knn_matrix = knn_symmetric.astype(float)
    
    # 3. Minimum spanning tree edges with strong prioritization
    mst = minimum_spanning_tree(distance_matrix)
    mst_binary = (mst.toarray() > 0).astype(float)
    mst_matrix = (mst_binary + mst_binary.T) * 2.5  # Strong weight for MST edges
    
    # 4. Combine base heuristics with KNN and MST boosts
    base_boosted = reciprocal * (1 + 0.5 * knn_matrix)  # Moderate KNN boost
    mst_boost = np.where(mst_matrix > 0, 3.0, 1.0)  # Strong MST boost
    combined = base_boosted * mst_boost
    
    # 5. Enhanced triangle closure boosting
    knn_binary = (knn_matrix > 0).astype(float)
    common_neighbors = knn_binary @ knn_binary.T  # Count common KNN neighbors
    
    # Boost edges based on triangle quality
    for i in range(n):
        for j in range(i + 1, n):
            if combined[i, j] > 0:  # Only boost promising edges
                common = common_neighbors[i, j]
                if common > 0:
                    # Find common neighbors in KNN graph
                    common_indices = np.where((knn_binary[i] > 0) & (knn_binary[j] > 0))[0]
                    common_indices = common_indices[(common_indices != i) & (common_indices != j)]
                    
                    if len(common_indices) > 0:
                        # Sample up to 3 common neighbors for quality assessment
                        sample_size = min(3, len(common_indices))
                        sampled = common_indices[:sample_size]
                        
                        # Compute triangle quality ratios
                        ratios = []
                        for k_node in sampled:
                            a = distance_matrix[i, j]
                            b = distance_matrix[i, k_node]
                            c = distance_matrix[j, k_node]
                            # Ratio measures direct edge efficiency
                            ratio = a / (b + c + 1e-12)
                            ratios.append(np.exp(-ratio))  # Lower ratio → higher boost
                        
                        quality_boost = 1.0 + 0.2 * np.mean(ratios) if ratios else 1.0
                        combined[i, j] *= quality_boost
                        combined[j, i] = combined[i, j]
                    else:
                        # Fallback: boost based on common neighbor count
                        boost = 1.0 + 0.1 * min(common, 4)
                        combined[i, j] *= boost
                        combined[j, i] = combined[i, j]
    
    # 6. Final adaptive sparsification
    final_matrix = np.zeros_like(combined)
    keep_per_node = max(3, int(np.log2(n)))  # Adaptive number of edges to keep
    keep_per_node = min(keep_per_node, n - 1)
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Ignore self
        if keep_per_node < n - 1:
            kth_val = np.partition(row, -keep_per_node)[-keep_per_node]
            mask = row >= kth_val
        else:
            mask = np.ones(n, dtype=bool)
            mask[i] = False
        final_matrix[i] = combined[i] * mask
    
    # 7. Symmetry enforcement and normalization
    final_matrix = np.maximum(final_matrix, final_matrix.T)
    max_val = np.max(final_matrix)
    if max_val > 0:
        final_matrix /= max_val
    
    np.fill_diagonal(final_matrix, 0)
    return final_matrix
