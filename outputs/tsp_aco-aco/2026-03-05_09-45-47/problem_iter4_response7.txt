import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.neighbors import KDTree
from sklearn.manifold import MDS
import warnings
from typing import Tuple, Optional


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with improved efficiency and accuracy.
    
    Combines spatial clustering via coordinate approximation, MST-based connectivity,
    local neighborhood statistics, and adaptive sparsification with triangle boosting.
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes.
    
    Returns:
        Matrix of edge promises with same shape as distance_matrix,
        where higher values indicate more promising edges for TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # Base heuristic: reciprocal of distances (closer = more promising)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(reciprocal_weights, 0)
    
    # Factor 1: Spatial clustering with improved coordinate approximation
    spatial_promising = _compute_spatial_clustering_v2(distance_matrix, n)
    
    # Factor 2: Minimum Spanning Tree with enhanced connectivity
    mst_promising = _compute_mst_promises_v2(distance_matrix, n)
    
    # Factor 3: Optimized local neighborhood factors
    local_factors = _compute_local_factors_v2(distance_matrix, n)
    
    # Efficient combination of all factors
    combined = _combine_factors_v2(
        reciprocal_weights, spatial_promising, mst_promising, 
        local_factors, distance_matrix, n
    )
    
    # Adaptive sparsification with size-aware thresholds
    sparsified = _adaptive_sparsification_v2(combined, n)
    
    # Efficient triangle boosting for clustering enhancement
    final = _triangle_boosting_v2(sparsified, n)
    
    # Normalize to [0, 1] range
    max_val = np.max(final)
    if max_val > 0:
        final = final / max_val
    
    np.fill_diagonal(final, 0)
    return final


def _compute_spatial_clustering_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute spatial clustering factor using efficient coordinate approximation."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    if n <= 500:
        # Use MDS for accurate coordinates when n is manageable
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            n_components = min(10, n - 1)
            mds = MDS(
                n_components=n_components,
                dissimilarity='precomputed',
                random_state=42,
                n_init=1,
                max_iter=100,
                eps=1e-3
            )
            coords = mds.fit_transform(distance_matrix)
    else:
        # Use landmark-based approximation for large graphs
        landmarks = min(100, max(20, n // 50))
        landmark_indices = np.random.choice(n, landmarks, replace=False)
        landmark_dist = distance_matrix[landmark_indices]
        
        # Efficient coordinate approximation using landmarks
        coords = np.zeros((n, landmarks))
        for i in range(n):
            # Vectorized computation for speed
            coords[i] = distance_matrix[i, landmark_indices]
    
    # Build KD-tree for efficient spatial queries
    tree = KDTree(coords)
    
    # Adaptive k: logarithmic in n with sensible bounds
    k_neighbors = min(max(3, int(np.log2(n)) + 2), n - 1)
    
    # Find nearest neighbors for each node
    for i in range(n):
        if n <= 500:
            # Use KD-tree for efficient queries
            distances, indices = tree.query(coords[i:i+1], k=k_neighbors + 1)
            indices = indices[0]
            # Exclude self if present
            if indices[0] == i:
                indices = indices[1:]
            else:
                indices = indices[:k_neighbors]
        else:
            # For large n, use efficient selection
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            # Use argpartition for O(n) instead of O(n log n)
            indices = np.argpartition(dists, k_neighbors)[:k_neighbors]
        
        spatial_promising[i, indices] = 1.0
    
    return spatial_promising


def _compute_mst_promises_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute promises based on Minimum Spanning Tree with enhanced connectivity."""
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    # Make symmetric and assign different weights to MST vs near-MST edges
    mst_promising = np.zeros_like(distance_matrix)
    
    # Strong boost for actual MST edges
    mst_promising[mst_edges > 0] = 2.0
    
    # Also consider edges that connect to MST nodes
    # For each node, consider its k closest connections
    k = min(5, n - 1)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Get k closest nodes
        closest_indices = np.argpartition(dists, k)[:k]
        
        for j in closest_indices:
            if mst_promising[i, j] == 0:
                # Check if either endpoint is in MST (excluding the edge itself)
                if np.any(mst_edges[i] > 0) or np.any(mst_edges[j] > 0):
                    mst_promising[i, j] = 1.0
    
    # Ensure symmetry
    mst_promising = np.maximum(mst_promising, mst_promising.T)
    return mst_promising


def _compute_local_factors_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute local connectivity factors with improved efficiency."""
    # Number of neighbors to consider for local statistics
    m = min(7, n - 1)
    
    # Precompute local statistics for each node
    local_means = np.zeros(n)
    local_stds = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Get m smallest distances
        m_smallest = np.partition(dists, m)[:m]
        local_means[i] = np.mean(m_smallest)
        local_stds[i] = np.std(m_smallest) + 1e-12
    
    # Vectorized computation of local factors
    local_factors = np.ones_like(distance_matrix)
    
    # Precompute mean and std matrices for vectorized operations
    mean_i, mean_j = np.meshgrid(local_means, local_means, indexing='ij')
    std_i, std_j = np.meshgrid(local_stds, local_stds, indexing='ij')
    
    # Compute similarity and distance factors
    with np.errstate(divide='ignore', invalid='ignore'):
        stat_similarity = np.exp(-np.abs(mean_i - mean_j) / (std_i + std_j))
        mean_dist_matrix = (mean_i + mean_j) / 2
        dist_ratio = distance_matrix / (mean_dist_matrix + 1e-12)
        distance_factor = np.exp(-0.5 * np.minimum(dist_ratio, 5)**2)
    
    local_factors = stat_similarity * distance_factor
    np.fill_diagonal(local_factors, 0)
    
    return local_factors


def _combine_factors_v2(reciprocal_weights: np.ndarray, 
                       spatial_promising: np.ndarray, 
                       mst_promising: np.ndarray, 
                       local_factors: np.ndarray, 
                       distance_matrix: np.ndarray,
                       n: int) -> np.ndarray:
    """Combine all heuristic factors efficiently with optimized weighting."""
    heuristics_matrix = np.zeros_like(reciprocal_weights)
    
    # Apply spatial factor: boost edges within spatial clusters
    spatial_boost = np.where(spatial_promising > 0, 1.8, 0.6)
    
    # Apply MST factor with graded boosting
    mst_boost = np.ones_like(mst_promising)
    mst_boost[mst_promising == 2.0] = 2.5  # Actual MST edges
    mst_boost[mst_promising == 1.0] = 1.4  # Near-MST edges
    
    # Combine base factors
    base_combined = reciprocal_weights * spatial_boost * mst_boost * local_factors
    
    # Additional adaptive boost based on edge rank
    for i in range(n):
        row = base_combined[i].copy()
        row[i] = -np.inf
        
        # Rank edges and apply logarithmic boost to top edges
        if np.any(row > 0):
            sorted_indices = np.argsort(-row)
            max_rank = min(10, n - 1)
            for rank, idx in enumerate(sorted_indices[:max_rank]):
                if rank > 0:
                    boost = 1 + 0.2 * np.log1p(max_rank - rank)
                    base_combined[i, idx] *= boost
    
    # Ensure symmetry
    heuristics_matrix = (base_combined + base_combined.T) / 2
    
    return heuristics_matrix


def _adaptive_sparsification_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Adaptively sparsify the heuristic matrix with size-aware thresholds."""
    sparsified = np.zeros_like(heuristics_matrix)
    
    if n <= 15:
        # Keep all edges for very small graphs
        return heuristics_matrix.copy()
    
    # Adaptive sparsity based on graph size
    if n <= 100:
        keep_ratio = 0.35
    elif n <= 500:
        keep_ratio = 0.25
    elif n <= 1000:
        keep_ratio = 0.15
    else:
        keep_ratio = 0.08
    
    # Calculate number of edges to keep per node
    edges_per_node = max(3, int(keep_ratio * n))
    edges_per_node = min(edges_per_node, n - 1)
    
    # Keep top edges for each node
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        if np.any(row > 0):
            # Get indices of top edges
            top_indices = np.argpartition(-row, edges_per_node)[:edges_per_node]
            sparsified[i, top_indices] = heuristics_matrix[i, top_indices]
    
    # Ensure symmetry by taking maximum (union of edges)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified


def _triangle_boosting_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Boost scores for edges that form triangles (common neighbors)."""
    boosted = heuristics_matrix.copy()
    
    if n > 300:
        # Use efficient approximation for large graphs
        # Create binary mask of promising edges
        binary_mask = (heuristics_matrix > 0).astype(np.float32)
        
        # Count common neighbors via matrix multiplication (efficient)
        common_neighbors = binary_mask @ binary_mask.T
        
        # Apply boost where there are common neighbors
        for i in range(n):
            for j in range(i + 1, n):
                if heuristics_matrix[i, j] > 0:
                    common = common_neighbors[i, j]
                    if common > 0:
                        # Diminishing returns for many common neighbors
                        boost = 1 + 0.15 * min(np.sqrt(common), 3)
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    else:
        # Exact computation for smaller graphs
        # Precompute neighbor sets for efficiency
        neighbor_sets = [set(np.where(heuristics_matrix[i] > 0)[0]) for i in range(n)]
        
        for i in range(n):
            for j in range(i + 1, n):
                if heuristics_matrix[i, j] > 0:
                    common = len(neighbor_sets[i] & neighbor_sets[j])
                    if common > 0:
                        boost = 1 + 0.2 * min(common, 4)
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    
    return boosted
