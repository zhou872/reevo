import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection probability.
    Combines inverse distance with MST connectivity and adaptive sparsification.
    Uses multiplicative weighting and dynamic parameter adjustment.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # 1. Base heuristic: inverse distance (handle zeros and diagonal)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree connectivity (boost edges likely in optimal tours)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Nearest neighbor preference (adaptive k based on problem size)
    k = max(2, int(np.ceil(np.log2(n))))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        indices = np.argsort(distance_matrix[i])[1:k+1]  # Exclude self
        nn_mask[i, indices] = True
    nn_mask = nn_mask | nn_mask.T  # Make symmetric
    
    # 4. Combine factors with multiplicative weights
    # MST edges get boosted, non-nearest neighbors get penalized
    weights = np.ones_like(inv_dist)
    weights[mst_mask] *= 1.8  # Boost MST edges more aggressively
    weights[~nn_mask] *= 0.2  # Stronger penalty for non-near neighbors
    
    heuristic = inv_dist * weights
    
    # 5. Adaptive sparsification with MST preservation
    # Always keep MST edges regardless of value
    mst_heuristic = np.zeros_like(heuristic)
    mst_heuristic[mst_mask] = heuristic[mst_mask]
    
    # Keep top adaptive number of edges per node
    keep_per_node = min(max(3, int(np.ceil(np.log2(n) * 1.5))), n - 1)
    
    for i in range(n):
        row = heuristic[i].copy()
        
        # Get top keep_per_node edges (excluding self)
        non_self_mask = np.arange(n) != i
        if np.sum(non_self_mask) > keep_per_node:
            # Partition to get threshold for top keep_per_node values
            threshold = np.partition(row[non_self_mask], -keep_per_node)[-keep_per_node]
            # Keep edges above threshold OR in MST
            keep_mask = (row >= threshold) & non_self_mask
            keep_mask = keep_mask | mst_mask[i]
            row[~keep_mask] = 0
        
        heuristic[i] = row
    
    # 6. Global sparsification while preserving MST
    # Remove very low probability edges globally
    non_zero_vals = heuristic[heuristic > 0]
    if len(non_zero_vals) > 0:
        # Adaptive threshold based on distribution
        median_val = np.median(non_zero_vals)
        std_val = np.std(non_zero_vals)
        global_threshold = max(median_val - 0.5 * std_val, np.percentile(non_zero_vals, 10))
        
        # Apply threshold but preserve MST edges
        low_prob_mask = (heuristic < global_threshold) & (~mst_mask)
        heuristic[low_prob_mask] = 0
    
    # 7. Ensure symmetry (since TSP is undirected)
    heuristic = np.maximum(heuristic, heuristic.T)
    
    # 8. Final normalization for consistency
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    # Ensure diagonal remains zero
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
