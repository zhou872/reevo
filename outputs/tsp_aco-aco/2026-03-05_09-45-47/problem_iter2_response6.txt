import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree, connected_components
from scipy.spatial.distance import pdist, squareform
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection featuring:
    1. Adaptive graph clustering for structural awareness
    2. Dynamic parameter tuning based on problem characteristics
    3. Machine learning-inspired feature combination
    4. Robust sparsification with graph connectivity preservation
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # 1. Adaptive parameter tuning based on problem characteristics
    # Analyze distance distribution to determine appropriate strategies
    dist_flat = distance_matrix[np.triu_indices(n, 1)]
    dist_mean = np.mean(dist_flat)
    dist_std = np.std(dist_flat)
    skewness = np.mean((dist_flat - dist_mean) ** 3) / (dist_std ** 3)
    
    # Dynamic parameters based on problem characteristics
    k_neighbors = max(2, int(np.sqrt(n)))  # Scale with problem size
    cluster_k = min(8, max(2, n // 20))  # Number of clusters
    mutual_boost = 1.5 + 0.5 * np.tanh(skewness)  # Adapt to distribution skew
    intra_cluster_boost = 1.3 + 0.3 * np.log1p(n / 100)  # Scale with size
    
    # 2. Graph clustering for structural awareness
    # Use spectral clustering on normalized distances
    similarity = np.exp(-distance_matrix / (2 * dist_mean ** 2))
    np.fill_diagonal(similarity, 0)
    
    # Compute Laplacian matrix for spectral clustering
    degree = np.diag(np.sum(similarity, axis=1))
    laplacian = degree - similarity
    
    # Use top eigenvectors for clustering (simplified approach)
    eigenvalues, eigenvectors = np.linalg.eigh(laplacian)
    # Get indices of smallest eigenvalues (excluding zero)
    idx_sorted = np.argsort(eigenvalues)[1:cluster_k+1]
    spectral_features = eigenvectors[:, idx_sorted]
    
    # Simple k-means on spectral features (2 iterations for efficiency)
    cluster_centers = spectral_features[np.random.choice(n, cluster_k, replace=False)]
    for _ in range(2):
        # Assign clusters
        distances_to_centers = np.array([
            np.linalg.norm(spectral_features - center, axis=1) 
            for center in cluster_centers
        ])
        clusters = np.argmin(distances_to_centers, axis=0)
        # Update centers
        for i in range(cluster_k):
            if np.sum(clusters == i) > 0:
                cluster_centers[i] = np.mean(spectral_features[clusters == i], axis=0)
    
    # 3. Enhanced mutual nearest neighbor detection
    neighbor_rank = np.argsort(distance_matrix, axis=1)
    mutual_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Adaptive k for each node based on local density
        local_density = np.mean(distance_matrix[i, neighbor_rank[i, 1:6]])
        adaptive_k = min(k_neighbors + int(local_density / dist_mean), n - 1)
        
        i_neighbors = set(neighbor_rank[i, 1:adaptive_k + 1])
        for j in i_neighbors:
            j_adaptive_k = min(k_neighbors + int(
                np.mean(distance_matrix[j, neighbor_rank[j, 1:6]]) / dist_mean
            ), n - 1)
            j_neighbors = set(neighbor_rank[j, 1:j_adaptive_k + 1])
            if i in j_neighbors:
                mutual_mask[i, j] = True
    
    # 4. Multi-factor heuristic computation
    eps = 1e-10
    inv_dist = 1 / (distance_matrix + eps)
    np.fill_diagonal(inv_dist, 0)
    
    # Compute edge betweenness approximation via random walks
    transition = similarity / (np.sum(similarity, axis=1, keepdims=True) + eps)
    # Power iteration for steady state
    pagerank = np.ones(n) / n
    for _ in range(10):
        pagerank = transition.T @ pagerank
    
    edge_importance = np.outer(pagerank, pagerank)
    
    # 5. Cluster-aware scoring
    cluster_scores = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            if clusters[i] == clusters[j]:
                # Intra-cluster: favor edges connecting high-degree nodes within cluster
                intra_mask = clusters == clusters[i]
                intra_degree = np.sum(similarity[intra_mask], axis=1)
                norm_i = intra_degree[i] / (np.max(intra_degree) + eps)
                norm_j = intra_degree[j] / (np.max(intra_degree) + eps)
                cluster_scores[i, j] = intra_cluster_boost * (norm_i + norm_j) / 2
            else:
                # Inter-cluster: penalize unless nodes are gateways
                i_inter = np.mean(distance_matrix[i, clusters != clusters[i]])
                j_inter = np.mean(distance_matrix[j, clusters != clusters[j]])
                inter_penalty = 1.0 / (1.0 + np.exp(-(i_inter + j_inter) / (2 * dist_mean)))
                cluster_scores[i, j] = inter_penalty
    
    cluster_scores = cluster_scores + cluster_scores.T
    
    # 6. Minimum spanning tree reinforcement
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 7. Combined heuristic with adaptive weighting
    # Normalize components
    norm_inv_dist = inv_dist / (np.max(inv_dist) + eps)
    norm_edge_importance = edge_importance / (np.max(edge_importance) + eps)
    norm_cluster_scores = cluster_scores / (np.max(cluster_scores) + eps)
    
    # Adaptive weights based on problem characteristics
    if skewness > 0.5:
        # Skewed distribution: emphasize clustering and MST
        weights = [0.3, 0.3, 0.2, 0.2]  # inv_dist, cluster, edge_importance, mst
    else:
        # Uniform distribution: balanced approach
        weights = [0.4, 0.25, 0.2, 0.15]
    
    heuristics = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            base_score = (
                weights[0] * norm_inv_dist[i, j] +
                weights[1] * norm_cluster_scores[i, j] +
                weights[2] * norm_edge_importance[i, j] +
                weights[3] * float(mst_mask[i, j])
            )
            
            # Apply multiplicative boosts
            if mutual_mask[i, j]:
                base_score *= mutual_boost
            
            # Penalize very long edges unless they're critical for connectivity
            if distance_matrix[i, j] > 2 * dist_mean + dist_std:
                penalty = 1.0 / (1.0 + np.exp(-distance_matrix[i, j] / dist_mean))
                base_score *= penalty
            
            heuristics[i, j] = base_score
    
    # 8. Connectivity-preserving sparsification
    # Ensure each node has at least 2 connections
    for i in range(n):
        row = heuristics[i]
        sorted_indices = np.argsort(row)[::-1]
        # Keep top connections, ensure at least 2
        min_connections = max(2, int(np.sqrt(n)))
        keep_mask = np.zeros(n, dtype=bool)
        keep_mask[sorted_indices[:min_connections]] = True
        
        # Additional connections above adaptive threshold
        non_zero = row[row > 0]
        if len(non_zero) > 0:
            q75 = np.percentile(non_zero, 75)
            q25 = np.percentile(non_zero, 25)
            iqr = q75 - q25
            threshold = q75 + 1.5 * iqr
            keep_mask = keep_mask | (row > threshold)
        
        heuristics[i, ~keep_mask] = 0
    
    # Ensure graph connectivity via MST fallback
    if not is_connected(heuristics > 0):
        # Add MST edges to ensure connectivity
        heuristics = np.maximum(heuristics, mst_mask * 0.1)
    
    # 9. Final symmetry and normalization
    heuristics = np.maximum(heuristics, heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics

def is_connected(adjacency: np.ndarray) -> bool:
    """Check if graph is connected using BFS."""
    n = adjacency.shape[0]
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        node = stack.pop()
        neighbors = np.where(adjacency[node])[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    return np.all(visited)
