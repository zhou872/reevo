def _farthest_point_sampling(distance_matrix: np.ndarray, k: int) -> np.ndarray:
    """Select k landmarks using farthest point sampling."""
    n = distance_matrix.shape[0]
    selected = np.zeros(k, dtype=int)
    
    # Start with a random point
    selected[0] = np.random.randint(0, n)
    
    for i in range(1, k):
        # Compute minimum distance to already selected points
        min_dists = np.min(distance_matrix[:, selected[:i]], axis=1)
        # Select the point farthest from all selected points
        selected[i] = np.argmax(min_dists)
    
    return selected

def _compute_distance_aware_triangle_boosting(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Boost edges forming triangles with boost factor inversely proportional to triangle distances."""
    triangle_boost = np.ones_like(distance_matrix)
    
    if n < 1000:  # Compute exact for manageable sizes
        for i in range(n):
            for j in range(i+1, n):
                # Find common neighbors
                common_neighbors = []
                triangle_distances = []
                
                for k in range(n):
                    if k != i and k != j:
                        # Check if triangle exists
                        if distance_matrix[i, k] < np.inf and distance_matrix[j, k] < np.inf:
                            common_neighbors.append(k)
                            triangle_dist = distance_matrix[i, j] + distance_matrix[i, k] + distance_matrix[j, k]
                            triangle_distances.append(triangle_dist)
                
                if common_neighbors:
                    # Compute boost factor inversely proportional to average triangle distance
                    avg_triangle_dist = np.mean(triangle_distances)
                    num_common = len(common_neighbors)
                    
                    # Capped boost factor as per long-term suggestion
                    base_boost = 1.0 / (avg_triangle_dist + 1e-12)
                    common_boost = 0.2 * min(num_common, 5)
                    boost = 1.0 + min(base_boost, common_boost)
                    
                    triangle_boost[i, j] = triangle_boost[j, i] = boost
    else:
        # Approximate for large graphs using sampling
        sample_size = min(200, n)
        sampled_nodes = np.random.choice(n, sample_size, replace=False)

def _compute_local_factors_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute local connectivity factors with improved statistics."""
    # Adaptive neighborhood size
    m = min(10, max(3, n // 20))
    
    # Precompute local statistics
    local_means = np.zeros(n)
    local_stds = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)[:m]
        local_means[i] = np.mean(sorted_dists)
        local_stds[i] = np.std(sorted_dists) + 1e-12
    
    # Compute local factor matrix using vectorized operations
    i_idx, j_idx = np.triu_indices(n, k=1)
    
    # Compute statistics for all pairs at once
    mean_pairs = (local_means[i_idx] + local_means[j_idx]) / 2
    std_pairs = local_stds[i_idx] + local_stds[j_idx]
    
    # Statistic similarity
    stat_diff = np.abs(local_means[i_idx] - local_means[j_idx])
    stat_similarity = np.exp(-stat_diff / std_pairs)
    
    # Distance ratio
    dist_vals = distance_matrix[i_idx, j_idx]
    dist_ratio = dist_vals / (mean_pairs + 1e-12)
    
    # Combined factor
    local_factors_flat = stat_similarity * np.exp(-0.3 * dist_ratio**2)
    
    # Create symmetric matrix
    local_factors = np.ones_like(distance_matrix)
    local_factors[i_idx, j_idx] = local_factors_flat
    local_factors[j_idx, i_idx] = local_factors_flat
    
    return local_factors

def _adaptive_factor_combination(reciprocal_weights, spatial_promising, mst_promising,
                                triangle_boost, local_factors, distance_matrix, n):
    """Combine heuristic factors with adaptive weights based on graph properties."""
    
    # Compute graph statistics for weight adaptation
    avg_distance = np.mean(distance_matrix[np.triu_indices(n, k=1)])
    distance_variance = np.var(distance_matrix[np.triu_indices(n, k=1)])
    
    # Adaptive weights based on graph characteristics
    if n < 100:
        # For small graphs, emphasize spatial and MST factors
        spatial_weight = 1.5
        mst_weight = 2.0
        triangle_weight = 1.2
        local_weight = 1.0
    elif distance_variance > avg_distance * 0.5:
        # For graphs with high variance, emphasize local factors and triangle boosting
        spatial_weight = 1.2
        mst_weight = 1.5
        triangle_weight = 1.8
        local_weight = 1.5
    else:
        # Balanced weights for regular graphs
        spatial_weight = 1.3
        mst_weight = 1.7
        triangle_weight = 1.5
        local_weight = 1.2
    
    # Apply weights to factors
    weighted_spatial = np.where(spatial_promising > 0, 
                               spatial_weight * spatial_promising, 
                               0.5)
    
    weighted_mst = np.where(mst_promising > 0,
                           mst_weight * mst_promising,
                           0.8)
    
    # Combine all factors
    combined = (reciprocal_weights * 
                weighted_spatial * 
                weighted_mst * 
                triangle_boost * 
                local_factors)
    
    # Additional distance-based adjustment
    with np.errstate(divide='ignore', invalid='ignore'):
        dist_adjustment = 1 + 0.2 * np.log(1 + 1 / (reciprocal_weights + 1e-12))
        dist_adjustment = np.nan_to_num(dist_adjustment, nan=1.0, posinf=1.0)
    
    combined *= dist_adjustment
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    return combined

def _stochastic_edge_sampling(heuristics_matrix, distance_matrix, n):
    """Perform stochastic edge sampling based on heuristic values."""
    if n <= 10:
        return heuristics_matrix
    
    sampled = np.zeros_like(heuristics_matrix)
    
    # Determine sampling probability based on heuristic values
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = 0
        
        if np.sum(row) > 0:
            # Convert heuristic values to probabilities
            probabilities = row / np.sum(row)
            
            # Adaptive number of edges to sample per node
            base_samples = max(3, int(np.log2(n)) + 2)
            max_samples = min(n // 3, 20)
            num_samples = min(base_samples, max_samples)
            
            # Stochastic sampling without replacement
            try:
                # Use weighted random choice
                sampled_indices = np.random.choice(n, size=num_samples, 
                                                   replace=False, p=probabilities)
            except ValueError:
                # Fallback to deterministic top-k if probabilities don't sum to 1
                sampled_indices = np.argsort(row)[-num_samples:]
            
            # Assign sampled edges their heuristic values
            sampled[i, sampled_indices] = heuristics_matrix[i, sampled_indices]
    
    # Ensure symmetry by keeping edges sampled by at least one endpoint
    sampled = np.maximum(sampled, sampled.T)
    
    # Add back some essential edges (MST edges)
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0)
    sampled[mst_edges] = np.maximum(sampled[mst_edges], heuristics_matrix[mst_edges])
    
    return sampled

    np.fill_diagonal(result, 0)
    
    # Normalize to [0, 1] range

    if max_val > 0:
        result = result / max_val
    
    # Small epsilon to avoid exact zeros for sampled edges
