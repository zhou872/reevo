import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.neighbors import KDTree
from sklearn.manifold import MDS
import warnings


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining spatial clustering,
    triangle closure boosting, and adaptive multi-factor fusion.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Factor 1: Reciprocal distances (base heuristic)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Factor 2: Enhanced spatial clustering with adaptive method selection
    spatial_promising = _enhanced_spatial_clustering(distance_matrix, n)
    
    # Factor 3: MST-based connectivity with extended neighborhood
    mst_promising = _enhanced_mst_promises(distance_matrix, n)
    
    # Factor 4: Triangle closure boosting (from long-term suggestions)
    triangle_closure = _triangle_closure_boosting(distance_matrix, n)
    
    # Factor 5: Triangle quality boosting
    triangle_quality = _triangle_quality_boosting(distance_matrix, n)
    
    # Factor 6: Local neighborhood statistics
    local_factors = _improved_local_factors(distance_matrix, n)
    
    # Multi-factor fusion with adaptive weights
    combined_matrix = _adaptive_multi_factor_fusion(
        reciprocal_weights, spatial_promising, mst_promising,
        triangle_closure, triangle_quality, local_factors, n
    )
    
    # Adaptive sparsification with edge preservation
    sparsified = _enhanced_adaptive_sparsification(combined_matrix, distance_matrix, n)
    
    # Final normalization and cleanup
    final_heuristics = _final_processing(sparsified, n)
    
    return final_heuristics


def _enhanced_spatial_clustering(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute spatial clustering factor using adaptive method selection."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    if n <= 800:  # Slightly increased threshold for MDS
        # Use MDS for accurate coordinates
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(3, n-1),
                dissimilarity='precomputed',
                random_state=42,
                n_init=1,
                max_iter=150,  # Increased iterations
                normalized_stress='auto'
            )
            coords = mds.fit_transform(distance_matrix)
    else:
        # Use landmark-based approximation for large instances
        landmarks = min(100, n // 15)  # More landmarks for better accuracy
        landmark_indices = np.random.choice(n, landmarks, replace=False)
        landmark_dists = distance_matrix[landmark_indices]
        
        # Approximate coordinates using distance to landmarks
        coords = np.zeros((n, landmarks))
        for i in range(n):
            coords[i] = distance_matrix[i, landmark_indices]
    
    # Adaptive neighborhood size based on graph size
    k_neighbors = max(4, min(int(np.log2(n)) * 2, n - 1))
    
    # Build KD-tree for efficient spatial queries
    tree = KDTree(coords)
    
    # Find nearest neighbors for each node
    for i in range(n):
        if n <= 800:
            distances, indices = tree.query(coords[i], k=k_neighbors + 1)
            # Exclude self
            indices = indices[1:] if distances[0] == 0 else indices[:k_neighbors]
        else:
            # Use distance matrix directly with argpartition for efficiency
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            indices = np.argpartition(dists, k_neighbors)[:k_neighbors]
        
        # Assign weights based on spatial proximity rank
        for rank, j in enumerate(indices):
            spatial_promising[i, j] = 1.0 / (rank + 1)  # Higher weight for closer neighbors
    
    return spatial_promising


def _enhanced_mst_promises(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute promises based on MST and extended neighborhood."""
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    # Make symmetric and add bonus for MST edges
    mst_promising = mst_edges + mst_edges.T
    
    # Enhanced neighborhood: consider up to 5 closest connections
    neighborhood_size = min(5, n - 1)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Get indices of closest nodes
        closest_indices = np.argsort(dists)[:neighborhood_size]
        
        for j in closest_indices:
            if mst_promising[i, j] == 0:
                # Assign decreasing weights based on closeness rank
                rank = np.where(closest_indices == j)[0][0]
                mst_promising[i, j] = 0.8 / (rank + 1)
                mst_promising[j, i] = 0.8 / (rank + 1)
    
    return mst_promising


def _triangle_closure_boosting(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Boost edges based on triangle closure in KNN graph (capped at 4)."""
    triangle_boost = np.ones_like(distance_matrix)
    
    # Build KNN graph
    k = min(15, n - 1)  # KNN parameter
    knn_graph = np.zeros_like(distance_matrix)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        neighbors = np.argpartition(dists, k)[:k]
        knn_graph[i, neighbors] = 1
    
    # Count common neighbors in KNN graph with efficient matrix multiplication
    knn_binary = (knn_graph > 0).astype(float)
    common_neighbors = knn_binary @ knn_binary.T
    
    for i in range(n):
        for j in range(i + 1, n):
            common = int(min(common_neighbors[i, j], 4))  # Cap at 4
            if common > 0:
                boost = 1.0 + 0.25 * common  # Moderate boosting
                triangle_boost[i, j] = triangle_boost[j, i] = boost
    
    return triangle_boost


def _triangle_quality_boosting(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Boost edges based on triangle quality (inverse of average edge length)."""
    triangle_quality = np.ones_like(distance_matrix)
    
    if n > 500:
        # Adaptive sampling for large instances
        sample_size = min(200, n)
        sample_nodes = np.random.choice(n, sample_size, replace=False)
    else:
        sample_nodes = np.arange(n)
    
    # Consider triangles with common neighbors
    for i in sample_nodes:
        # Find potential neighbors for triangle formation
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        k_neighbors = min(10, n - 1)
        neighbors_i = np.argpartition(dists_i, k_neighbors)[:k_neighbors]
        
        for j in neighbors_i:
            if j <= i:
                continue
                
            # Find common neighbors between i and j
            dists_j = distance_matrix[j].copy()
            dists_j[j] = np.inf
            neighbors_j = np.argpartition(dists_j, k_neighbors)[:k_neighbors]
            common = np.intersect1d(neighbors_i, neighbors_j)
            
            if len(common) > 0:
                # Compute triangle quality for each common neighbor
                qualities = []
                for k in common[:5]:  # Consider up to 5 best triangles
                    avg_length = (distance_matrix[i, j] + 
                                 distance_matrix[i, k] + 
                                 distance_matrix[j, k]) / 3
                    quality = 1.0 / (avg_length + 1e-12)
                    qualities.append(quality)
                
                if qualities:
                    avg_quality = np.mean(qualities)
                    # Normalize quality and apply boost
                    max_dist = np.max(distance_matrix)
                    norm_quality = avg_quality / (1.0 / (max_dist / 3) + 1e-12)
                    boost = 1.0 + 0.3 * min(norm_quality, 2.0)  # Cap boost
                    triangle_quality[i, j] = triangle_quality[j, i] = boost
    
    return triangle_quality


def _improved_local_factors(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute improved local neighborhood statistics."""
    # Adaptive neighborhood size
    m = min(8, n - 1)
    
    # Compute local statistics for each node
    local_stats = np.zeros((n, 3))  # [mean, std, median]
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)[:m]
        local_stats[i, 0] = np.mean(sorted_dists)
        local_stats[i, 1] = np.std(sorted_dists) + 1e-12
        local_stats[i, 2] = np.median(sorted_dists)
    
    # Compute local factors using vectorized operations
    local_factors = np.ones_like(distance_matrix)
    
    # Pre-compute statistic differences
    mean_diffs = np.abs(local_stats[:, 0][:, np.newaxis] - local_stats[:, 0])
    std_sums = local_stats[:, 1][:, np.newaxis] + local_stats[:, 1]
    
    # Compute statistic similarity
    stat_similarity = np.exp(-mean_diffs / std_sums)
    
    # Compute distance ratio factor
    mean_dists = (local_stats[:, 0][:, np.newaxis] + local_stats[:, 0]) / 2
    dist_ratios = distance_matrix / (mean_dists + 1e-12)
    dist_factor = np.exp(-0.5 * dist_ratios**2)
    
    # Combine factors with symmetry
    local_factors = stat_similarity * dist_factor
    np.fill_diagonal(local_factors, 0)
    
    return local_factors


def _adaptive_multi_factor_fusion(reciprocal_weights, spatial_promising,
                                 mst_promising, triangle_closure,
                                 triangle_quality, local_factors, n):
    """Fuse multiple heuristic factors with adaptive weighting."""
    # Base matrix from reciprocal distances
    combined = reciprocal_weights.copy()
    
    # Apply spatial factor (stronger for spatial clusters)
    spatial_mask = spatial_promising > 0
    spatial_weight = np.where(spatial_mask, 2.0, 0.7)
    combined *= spatial_weight
    
    # Apply MST factor (strong boost for fundamental connections)
    mst_mask = mst_promising > 0
    mst_weight = np.where(mst_mask, 
                         np.where(mst_promising == 2, 3.0, 1.8),
                         0.9)
    combined *= mst_weight
    
    # Apply triangle factors
    combined *= triangle_closure
    combined *= triangle_quality
    
    # Apply local factors
    combined *= local_factors
    
    # Distance-based logarithmic boost for very short edges
    with np.errstate(divide='ignore', invalid='ignore'):
        log_boost = 1 + 0.2 * np.log(1 + 1 / (reciprocal_weights + 1e-12))
        log_boost = np.nan_to_num(log_boost, nan=1.0, posinf=1.0)
    
    combined *= log_boost
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    return combined


def _enhanced_adaptive_sparsification(heuristics_matrix, distance_matrix, n):
    """Adaptively sparsify while preserving important edges."""
    if n <= 10:
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Adaptive sparsity based on graph size
    if n < 50:
        keep_ratio = 0.5
    elif n < 200:
        keep_ratio = 0.35
    elif n < 1000:
        keep_ratio = 0.25
    else:
        keep_ratio = 0.15
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        # Determine number of edges to keep
        num_to_keep = max(4, int(keep_ratio * n))
        num_to_keep = min(num_to_keep, n - 1)
        
        if np.any(row > 0):
            # Get top edges
            top_indices = np.argpartition(-row, num_to_keep)[:num_to_keep]
            
            # Preserve exceptionally promising edges regardless of rank
            threshold = np.min(row[top_indices])
            promising_mask = (row >= threshold) | (row > np.percentile(row, 90))
            
            sparsified[i] = heuristics_matrix[i] * promising_mask
    
    # Ensure symmetry by taking union of kept edges
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified


def _final_processing(heuristics_matrix, n):
    """Final normalization and cleanup."""
    # Apply mild smoothing
    smoothed = heuristics_matrix.copy()
    
    # Small boost for edges in dense local neighborhoods
    if n < 1000:
        binary_mask = (smoothed > 0).astype(float)
        neighborhood_density = binary_mask @ binary_mask.T / n
        
        for i in range(n):
            for j in range(i + 1, n):
                if smoothed[i, j] > 0:
                    density = neighborhood_density[i, j]
                    smoothed[i, j] = smoothed[j, i] = smoothed[i, j] * (1 + 0.1 * density)
    
    # Normalize to [0, 1] range
    max_val = np.max(smoothed)
    if max_val > 0:
        smoothed = smoothed / max_val
    
    np.fill_diagonal(smoothed, 0)
    
    return smoothed
