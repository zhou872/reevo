import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.manifold import MDS
from sklearn.neighbors import KDTree
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with distance-aware triangle boosting,
    adaptive spatial clustering, and capped boosting.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Base heuristic: reciprocal of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Factor 1: Spatial clustering with efficient approximation
    spatial_promising = _compute_spatial_clustering_v2(distance_matrix, n)
    
    # Factor 2: Minimum Spanning Tree edges
    mst_promising = _compute_mst_promises_v2(distance_matrix)
    
    # Factor 3: Local connectivity based on neighborhood statistics
    local_factors = _compute_local_factors_v2(distance_matrix, n)
    
    # Combine all factors efficiently
    heuristics_matrix = _combine_factors_v2(
        reciprocal_weights, spatial_promising, mst_promising, 
        local_factors, distance_matrix, n
    )
    
    # Adaptive sparsification: keep most promising edges per node
    sparsified = _adaptive_sparsification_v2(heuristics_matrix, n)
    
    # Triangle boosting with distance-aware and capped boosting
    final_heuristics = _distance_aware_triangle_boosting(sparsified, distance_matrix, n)
    
    # Normalize to [0, 1] range
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics

def _compute_spatial_clustering_v2(distance_matrix, n):
    """Compute spatial clustering factor using MDS for n≤500, else approximation."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    if n <= 500:
        # Use MDS for accurate coordinates when n is manageable
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(3, n-1), 
                dissimilarity='precomputed', 
                random_state=42,
                n_init=1,
                max_iter=100
            )
            coords = mds.fit_transform(distance_matrix)
    else:
        # Use faster approximation for large n
        landmarks = min(50, n // 20)
        landmark_indices = np.random.choice(n, landmarks, replace=False)
        coords = np.zeros((n, landmarks))
        for i in range(n):
            coords[i] = distance_matrix[i, landmark_indices]
    
    # Build KD-tree for spatial queries
    tree = KDTree(coords)
    
    # Adaptive k: log2(n) but at least 3
    k = max(3, min(int(np.log2(n)) + 2, n - 1))
    
    # Find nearest neighbors for each node
    for i in range(n):
        if n <= 500:
            distances, indices = tree.query(coords[i], k=k+1)
            indices = indices[1:] if distances[0] == 0 else indices[:k]
        else:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            indices = np.argpartition(dists, k)[:k]
        
        spatial_promising[i, indices] = 1.0
    
    return spatial_promising

def _compute_mst_promises_v2(distance_matrix):
    """Compute promises based on Minimum Spanning Tree edges."""
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    mst_promising = mst_edges + mst_edges.T
    
    n = distance_matrix.shape[0]
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        closest = np.argsort(dists)[:3]
        for j in closest:
            if mst_promising[i, j] == 0:
                mst_promising[i, j] = 0.5
    
    return mst_promising

def _compute_local_factors_v2(distance_matrix, n):
    """Compute local connectivity factors."""
    m = min(5, n - 1)
    local_stats = np.zeros((n, 2))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)[:m]
        local_stats[i, 0] = np.mean(sorted_dists)
        local_stats[i, 1] = np.std(sorted_dists) + 1e-12
    
    local_factors = np.ones_like(distance_matrix)
    
    for i in range(n):
        for j in range(i+1, n):
            stat_similarity = np.exp(
                -np.abs(local_stats[i, 0] - local_stats[j, 0]) / 
                (local_stats[i, 1] + local_stats[j, 1])
            )
            mean_dist = (local_stats[i, 0] + local_stats[j, 0]) / 2
            dist_ratio = distance_matrix[i, j] / (mean_dist + 1e-12)
            local_factors[i, j] = local_factors[j, i] = (
                stat_similarity * np.exp(-0.5 * dist_ratio**2)
            )
    
    return local_factors

def _combine_factors_v2(reciprocal_weights, spatial_promising, mst_promising, 
                       local_factors, distance_matrix, n):
    """Combine all heuristic factors efficiently."""
    base_scores = reciprocal_weights.copy()
    
    spatial_factor = np.where(spatial_promising > 0, 2.0, 0.5)
    mst_factor = np.where(mst_promising > 0, 
                         np.where(mst_promising == 2, 3.0, 1.5), 
                         0.8)
    
    combined_scores = base_scores * spatial_factor * mst_factor * local_factors
    
    with np.errstate(divide='ignore', invalid='ignore'):
        log_boost = 1 + 0.3 * np.log(1 + 1 / (base_scores + 1e-12))
        log_boost = np.nan_to_num(log_boost, nan=1.0, posinf=1.0)
    
    heuristics_matrix = combined_scores * log_boost
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    return heuristics_matrix

def _adaptive_sparsification_v2(heuristics_matrix, n):
    """Adaptively sparsify the heuristic matrix."""
    sparsified = np.zeros_like(heuristics_matrix)
    
    if n <= 10:
        return heuristics_matrix.copy()
    
    base_keep_ratio = min(0.4, max(0.15, np.log10(n) / 10))
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        num_to_keep = max(3, int(base_keep_ratio * n))
        num_to_keep = min(num_to_keep, n - 1)
        
        if np.any(row > 0):
            top_indices = np.argpartition(-row, num_to_keep)[:num_to_keep]
            threshold = np.min(row[top_indices])
            mask = row >= threshold
            sparsified[i] = heuristics_matrix[i] * mask
    
    sparsified = np.maximum(sparsified, sparsified.T)
    return sparsified

def _distance_aware_triangle_boosting(heuristics_matrix, distance_matrix, n):
    """Boost scores with distance-aware and capped triangle boosting."""
    boosted = heuristics_matrix.copy()
    
    if n > 500:
        # Use sampling for large graphs
        sample_size = min(100, n)
        sampled_nodes = np.random.choice(n, sample_size, replace=False)
        for i in sampled_nodes:
            neighbors_i = set(np.where(heuristics_matrix[i] > 0)[0])
            for j in neighbors_i:
                if j <= i:
                    continue
                neighbors_j = set(np.where(heuristics_matrix[j] > 0)[0])
                common = neighbors_i & neighbors_j
                if common:
                    dist_ij = distance_matrix[i, j]
                    avg_triangle_dist = 0
                    for k in common:
                        dist_ik = distance_matrix[i, k]
                        dist_jk = distance_matrix[j, k]
                        triangle_avg = (dist_ij + dist_ik + dist_jk) / 3
                        avg_triangle_dist += triangle_avg
                    avg_triangle_dist /= len(common)
                    common_count = len(common)
                    cap_boost = 0.2 * min(common_count, 5)  # Capped boosting
                    dist_boost = 0.1 * common_count / (avg_triangle_dist + 1e-12)
                    boost = 1 + min(cap_boost, dist_boost)
                    boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    else:
        # Exact computation for smaller graphs
        neighbors = [set(np.where(heuristics_matrix[i] > 0)[0]) for i in range(n)]
        for i in range(n):
            for j in range(i+1, n):
                if heuristics_matrix[i, j] > 0:
                    common = neighbors[i] & neighbors[j]
                    if common:
                        dist_ij = distance_matrix[i, j]
                        avg_triangle_dist = 0
                        for k in common:
                            dist_ik = distance_matrix[i, k]
                            dist_jk = distance_matrix[j, k]
                            triangle_avg = (dist_ij + dist_ik + dist_jk) / 3
                            avg_triangle_dist += triangle_avg
                        avg_triangle_dist /= len(common)
                        common_count = len(common)
                        cap_boost = 0.2 * min(common_count, 5)
                        dist_boost = 0.1 * common_count / (avg_triangle_dist + 1e-12)
                        boost = 1 + min(cap_boost, dist_boost)
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    
    return boosted
