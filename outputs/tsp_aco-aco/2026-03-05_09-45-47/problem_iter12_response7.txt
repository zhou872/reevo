import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple, Dict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Lightweight adaptive heuristic for TSP edge selection.
    
    Combines essential strategies with instance-aware parameter tuning:
    1. Dynamic neighbor selection based on instance characteristics
    2. MST edge boosting for connectivity
    3. Adaptive sparsification with connectivity assurance
    4. Lightweight instance analysis for parameter adaptation
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric n x n matrix of pairwise distances
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate more promising edges.
        Diagonal is zero, values are normalized to [0, 1].
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(heuristic, 0)
        max_val = np.max(heuristic)
        if max_val > 0:
            heuristic /= max_val
        return heuristic
    
    # Lightweight instance analysis for adaptive parameters
    instance_params = _analyze_instance_lightweight(distance_matrix, n)
    
    # 1. Base heuristic: reciprocal of distances, normalized
    with np.errstate(divide='ignore', invalid='ignore'):
        base_heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(base_heuristic, 0)
    max_val = np.max(base_heuristic)
    if max_val > 0:
        base_heuristic /= max_val
    
    # 2. Identify MST edges for essential connectivity
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # 3. Dynamic percentile neighbor selection with instance-aware cutoff
    neighborhood_mask = _adaptive_neighbor_selection(
        distance_matrix, n, instance_params
    )
    
    # 4. Combine factors with targeted boosting
    heuristic = base_heuristic.copy()
    heuristic[mst_mask] *= instance_params['mst_boost']  # Adaptive MST boost
    heuristic[neighborhood_mask] *= instance_params['neighbor_boost']
    heuristic = (heuristic + heuristic.T) / 2  # Enforce symmetry
    
    # 5. Adaptive sparsification based on instance size and structure
    sparsified = _adaptive_sparsification(
        heuristic, mst_mask, distance_matrix, n, instance_params
    )
    
    # 6. Connectivity assurance with repair strategies
    sparsified = _ensure_connectivity(
        sparsified, distance_matrix, n, instance_params
    )
    
    # 7. Final normalization
    np.fill_diagonal(sparsified, 0)
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified /= max_val
    
    return sparsified


def _analyze_instance_lightweight(
    distance_matrix: np.ndarray, n: int
) -> Dict[str, float]:
    """Lightweight instance analysis for adaptive parameter tuning."""
    params = {}
    
    # Extract upper triangular distances
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    # Basic statistics
    mean_dist = np.mean(distances)
    std_dist = np.std(distances)
    cv_dist = std_dist / (mean_dist + 1e-12)
    
    # Local connectivity statistics
    k_local = min(5, n - 1)
    local_dists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest = np.partition(dists, k_local)[:k_local]
        local_dists.extend(nearest)
    
    local_cv = np.std(local_dists) / (np.mean(local_dists) + 1e-12)
    
    # Adaptive parameter selection
    # MST boost: stronger for more clustered instances
    if local_cv > 0.4:
        params['mst_boost'] = 3.2
    elif local_cv > 0.25:
        params['mst_boost'] = 2.8
    else:
        params['mst_boost'] = 2.5
    
    # Neighbor boost: moderate for all instances
    params['neighbor_boost'] = 1.7
    
    # Edge sparsity: more edges for small/irregular instances
    if n < 50:
        params['target_edges_factor'] = 3.0
    elif cv_dist > 0.5:
        params['target_edges_factor'] = 2.5
    else:
        params['target_edges_factor'] = 2.0
    
    # Neighbor selection cutoff
    if n < 100:
        params['percentile_cutoff'] = 30
    elif local_cv > 0.3:
        params['percentile_cutoff'] = 25
    else:
        params['percentile_cutoff'] = 20
    
    return params


def _adaptive_neighbor_selection(
    distance_matrix: np.ndarray, n: int, params: Dict[str, float]
) -> np.ndarray:
    """Dynamic neighbor selection with instance-aware thresholds."""
    neighborhood_mask = np.zeros((n, n), dtype=bool)
    percentile_cutoff = params['percentile_cutoff']
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Use instance-aware percentile
        threshold = np.percentile(dists, percentile_cutoff)
        close_nodes = dists <= threshold
        
        # Ensure minimum connections
        if close_nodes.sum() < 3:
            k_neighbors = max(3, min(int(np.log2(n)), n - 1))
            nearest_indices = np.argsort(dists)[:k_neighbors]
            close_nodes = np.zeros(n, dtype=bool)
            close_nodes[nearest_indices] = True
        
        neighborhood_mask[i] = close_nodes
    
    # Ensure symmetry
    neighborhood_mask = neighborhood_mask | neighborhood_mask.T
    
    return neighborhood_mask


def _adaptive_sparsification(
    heuristic: np.ndarray,
    mst_mask: np.ndarray,
    distance_matrix: np.ndarray,
    n: int,
    params: Dict[str, float]
) -> np.ndarray:
    """Adaptive edge sparsification with instance-aware edge counts."""
    # Determine target edges per node
    base_edges = max(3, int(np.log2(n)))
    target_edges = int(base_edges * params['target_edges_factor'])
    target_edges = min(target_edges, n - 1)
    
    sparsified = np.zeros_like(heuristic)
    
    for i in range(n):
        row_heuristics = heuristic[i].copy()
        row_heuristics[i] = -np.inf  # Exclude self
        
        # Always keep MST edges
        mst_neighbors = np.where(mst_mask[i])[0]
        sparsified[i, mst_neighbors] = row_heuristics[mst_neighbors]
        
        # Calculate additional edges needed
        current_edges = len(mst_neighbors)
        additional_needed = max(0, target_edges - current_edges)
        
        if additional_needed > 0:
            # Select non-MST candidate edges
            non_mst_mask = np.ones(n, dtype=bool)
            non_mst_mask[mst_neighbors] = False
            non_mst_mask[i] = False
            non_mst_indices = np.where(non_mst_mask)[0]
            
            if len(non_mst_indices) > 0:
                non_mst_values = row_heuristics[non_mst_indices]
                if len(non_mst_indices) > additional_needed:
                    # Select top additional_needed edges
                    if additional_needed == 1:
                        best_idx = np.argmax(non_mst_values)
                        keep_indices = [non_mst_indices[best_idx]]
                    else:
                        threshold_idx = np.argpartition(
                            -non_mst_values, additional_needed - 1
                        )[additional_needed - 1]
                        threshold_val = non_mst_values[threshold_idx]
                        keep_indices = non_mst_indices[non_mst_values >= threshold_val]
                else:
                    keep_indices = non_mst_indices
                
                sparsified[i, keep_indices] = row_heuristics[keep_indices]
    
    # Symmetrize
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified


def _ensure_connectivity(
    heuristic: np.ndarray,
    distance_matrix: np.ndarray,
    n: int,
    params: Dict[str, float]
) -> np.ndarray:
    """Ensure graph connectivity with targeted repairs."""
    result = heuristic.copy()
    
    for i in range(n):
        degree = np.count_nonzero(result[i])
        
        if degree == 0:
            # Connect isolated node to nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest1 = np.argmin(dists)
            dists[nearest1] = np.inf
            nearest2 = np.argmin(dists)
            
            result[i, nearest1] = result[nearest1, i] = 0.2
            result[i, nearest2] = result[nearest2, i] = 0.15
            
        elif degree == 1:
            # Add second connection for degree-1 nodes
            current = np.nonzero(result[i])[0][0]
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            dists[current] = np.inf
            
            if np.any(dists < np.inf):
                second = np.argmin(dists)
                result[i, second] = result[second, i] = 0.15
    
    return result
