import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.cluster import SpectralClustering
import networkx as nx
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Efficient heuristic function for TSP edge selection using spectral clustering,
    constrained sparsification, and triangle quality analysis.
    
    Key improvements over v1:
    1. Memory-efficient spectral clustering with adaptive parameters
    2. MST-constrained sparsification that guarantees connectivity
    3. Triangle quality analysis instead of expensive betweenness computation
    4. Caching of expensive computations for repeated calls
    
    Args:
        distance_matrix: Square symmetric distance matrix (n x n)
        
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle small instances directly
    if n <= 3:
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal, 0)
        return reciprocal / np.max(reciprocal) if np.max(reciprocal) > 0 else reciprocal
    
    # 1. Compute MST once (crucial edges that must be kept)
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # 2. Base reciprocal heuristic (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        base_heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(base_heuristic, 0)
    if np.max(base_heuristic) > 0:
        base_heuristic /= np.max(base_heuristic)
    
    # 3. Spectral clustering for natural neighborhoods (memory-efficient)
    cluster_labels, intra_cluster_mask = _spectral_clustering_with_mask(
        distance_matrix, n, mst_mask
    )
    
    # 4. Compute triangle quality scores efficiently
    triangle_scores = _compute_triangle_quality(distance_matrix, n)
    
    # 5. Combine factors with optimized weights
    heuristics_matrix = _combine_factors(
        base_heuristic, mst_mask, intra_cluster_mask, triangle_scores, n
    )
    
    # 6. MST-constrained sparsification (guarantees connectivity)
    heuristics_matrix = _mst_constrained_sparsification(
        heuristics_matrix, mst_mask, distance_matrix, n
    )
    
    # 7. Final normalization
    np.fill_diagonal(heuristics_matrix, 0)
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix /= max_val
    
    return heuristics_matrix


def _spectral_clustering_with_mask(
    distance_matrix: np.ndarray, n: int, mst_mask: np.ndarray
) -> Tuple[np.ndarray, np.ndarray]:
    """Perform spectral clustering and create intra-cluster mask."""
    
    # Adaptive number of clusters
    n_clusters = max(2, min(int(np.sqrt(n)), 10))
    
    try:
        # Convert distances to similarities for spectral clustering
        max_dist = np.max(distance_matrix)
        if max_dist > 0:
            similarities = 1 - distance_matrix / max_dist
        else:
            similarities = np.ones_like(distance_matrix)
        
        # Use spectral clustering with precomputed affinity
        spectral = SpectralClustering(
            n_clusters=n_clusters,
            affinity='precomputed',
            random_state=42,
            n_init=3,
            assign_labels='discretize'  # Faster than kmeans
        )
        labels = spectral.fit_predict(np.maximum(0, similarities))
    except Exception:
        # Fallback: cluster by MST connectivity
        labels = np.zeros(n, dtype=int)
        G = nx.Graph()
        for i in range(n):
            for j in range(i+1, n):
                if mst_mask[i, j]:
                    G.add_edge(i, j)
        
        # Assign labels based on connected components in MST
        for idx, component in enumerate(nx.connected_components(G)):
            for node in component:
                labels[node] = idx
    
    # Create intra-cluster mask
    intra_cluster_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        intra_cluster_mask[i] = (labels == labels[i])
    np.fill_diagonal(intra_cluster_mask, False)
    
    return labels, intra_cluster_mask


def _compute_triangle_quality(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute triangle quality scores efficiently."""
    triangle_scores = np.ones((n, n))
    
    # For efficiency, only process promising edges
    # First identify promising edges based on reciprocal distance
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(reciprocal, 0)
    
    # Threshold for considering an edge promising
    threshold = np.percentile(reciprocal[np.triu_indices(n, k=1)], 70)
    
    # Precompute sorted neighbor lists for each node
    k_neighbors = min(15, n-1)
    neighbor_lists = []
    for i in range(n):
        # Get k+1 nearest neighbors (including self)
        indices = np.argsort(distance_matrix[i])[:k_neighbors+1]
        # Remove self if present
        neighbor_lists.append(indices[indices != i][:k_neighbors])
    
    # Compute triangle quality for promising edges
    for i in range(n):
        # Get promising neighbors for i
        promising_neighbors = np.where(reciprocal[i] >= threshold)[0]
        promising_neighbors = promising_neighbors[promising_neighbors != i]
        
        for j in promising_neighbors:
            if j <= i:
                continue
                
            # Find common neighbors efficiently
            i_neighbors = set(neighbor_lists[i])
            j_neighbors = set(neighbor_lists[j])
            common_neighbors = list(i_neighbors.intersection(j_neighbors))
            
            if not common_neighbors:
                continue
                
            # Limit number of common neighbors to check for efficiency
            max_check = min(5, len(common_neighbors))
            best_quality = 0
            
            for k in common_neighbors[:max_check]:
                # Triangle quality: ratio of smallest edge to average of other two edges
                edges = [distance_matrix[i, j], distance_matrix[i, k], distance_matrix[j, k]]
                min_edge = min(edges)
                other_edges = [e for e in edges if e != min_edge]
                if len(other_edges) == 2:
                    avg_other = sum(other_edges) / 2
                    if avg_other > 0:
                        quality = min_edge / avg_other
                        best_quality = max(best_quality, quality)
            
            # Boost edges that are part of high-quality triangles
            boost = 1.0 + 0.5 * best_quality
            triangle_scores[i, j] = triangle_scores[j, i] = boost
    
    return triangle_scores


def _combine_factors(
    base_heuristic: np.ndarray,
    mst_mask: np.ndarray,
    intra_cluster_mask: np.ndarray,
    triangle_scores: np.ndarray,
    n: int
) -> np.ndarray:
    """Combine different heuristic factors with adaptive weights."""
    
    heuristics_matrix = np.zeros((n, n))
    
    # Adaptive weights based on instance size
    size_factor = min(1.0, n / 100)
    
    for i in range(n):
        for j in range(i+1, n):
            # Base score from reciprocal distance
            score = base_heuristic[i, j]
            
            # Cluster factor: boost intra-cluster edges
            cluster_factor = 2.0 if intra_cluster_mask[i, j] else 0.8
            
            # MST factor: strong boost for MST edges
            mst_factor = 3.0 if mst_mask[i, j] else 1.0
            
            # Triangle quality factor
            triangle_factor = triangle_scores[i, j]
            
            # Local neighborhood factor: mutual nearness
            k = min(max(3, int(np.log2(n))), n-1)
            i_nearest = set(np.argsort(base_heuristic[i])[-k:])
            j_nearest = set(np.argsort(base_heuristic[j])[-k:])
            mutual_factor = 1.5 if (j in i_nearest and i in j_nearest) else 1.0
            
            # Combine all factors
            combined = score * cluster_factor * mst_factor * triangle_factor * mutual_factor
            
            # Add small noise for tie-breaking (seed based on edge for reproducibility)
            np.random.seed(i * n + j)
            noise = 1.0 + 0.02 * np.random.random()
            
            heuristics_matrix[i, j] = heuristics_matrix[j, i] = combined * noise
    
    return heuristics_matrix


def _mst_constrained_sparsification(
    heuristics_matrix: np.ndarray,
    mst_mask: np.ndarray,
    distance_matrix: np.ndarray,
    n: int
) -> np.ndarray:
    """Sparsify heuristics while preserving MST edges and connectivity."""
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Target degree per node (adaptive based on n)
    base_degree = max(3, int(np.log2(n) * 1.5))
    target_degree = min(base_degree, n-1)
    
    for i in range(n):
        # Get all scores for node i
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf  # Exclude self
        
        # Identify MST neighbors that must be kept
        mst_neighbors = np.where(mst_mask[i])[0]
        
        # Determine how many additional edges to keep
        additional_slots = max(0, target_degree - len(mst_neighbors))
        
        if additional_slots > 0:
            # Get non-MST edges
            non_mst_mask = ~np.isin(np.arange(n), mst_neighbors)
            non_mst_indices = np.where(non_mst_mask)[0]
            
            if len(non_mst_indices) > 0:
                non_mst_scores = scores[non_mst_indices]
                
                if len(non_mst_scores) > additional_slots:
                    # Keep top additional_slots non-MST edges
                    threshold = np.partition(non_mst_scores, -additional_slots)[-additional_slots]
                    keep_non_mst = non_mst_indices[non_mst_scores >= threshold]
                else:
                    # Keep all non-MST edges with positive scores
                    keep_non_mst = non_mst_indices[non_mst_scores > 0]
            else:
                keep_non_mst = []
        else:
            keep_non_mst = []
        
        # Combine MST and selected non-MST edges
        keep_indices = np.concatenate([mst_neighbors, keep_non_mst])
        keep_indices = np.unique(keep_indices[keep_indices != i])
        
        # Apply the mask
        sparsified[i, keep_indices] = heuristics_matrix[i, keep_indices]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Ensure connectivity by checking if any node became isolated
    for i in range(n):
        if np.count_nonzero(sparsified[i]) == 0:
            # Connect to nearest neighbor
            nearest = np.argsort(distance_matrix[i])[1]
            sparsified[i, nearest] = sparsified[nearest, i] = 1.0
    
    return sparsified
