import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.manifold import MDS
from sklearn.neighbors import KDTree
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection using adaptive ensemble methods.
    
    Features:
    1. Multi-factor ensemble with learned weights based on graph properties
    2. Graph structure-aware adaptive parameter selection
    3. Fully vectorized operations for maximum efficiency
    4. Robust handling of diverse TSP instance types
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes.
    
    Returns:
        Matrix of edge promises with same shape as distance_matrix,
        where higher values indicate more promising edges for TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # Analyze graph structure to adapt heuristic weights
    graph_properties = _analyze_graph_structure(distance_matrix, n)
    
    # Ensemble of multiple heuristic factors with learned weights
    factors = _compute_ensemble_factors(distance_matrix, n, graph_properties)
    
    # Adaptive combination with learned weights
    combined = _adaptive_factor_combination(factors, distance_matrix, n, graph_properties)
    
    # Structure-aware sparsification
    sparsified = _graph_aware_sparsification(combined, n, graph_properties)
    
    # Enhanced triangle boosting with diminishing returns
    boosted = _enhanced_triangle_boosting(sparsified, n, graph_properties)
    
    # Final normalization
    final = _normalize_heuristics(boosted, n)
    
    return final

def _analyze_graph_structure(distance_matrix: np.ndarray, n: int) -> dict:
    """Analyze graph properties to adapt heuristic parameters."""
    properties = {}
    
    # Basic statistics
    upper_tri = distance_matrix[np.triu_indices(n, k=1)]
    properties['mean_distance'] = np.mean(upper_tri)
    properties['std_distance'] = np.std(upper_tri)
    properties['cv_distance'] = properties['std_distance'] / (properties['mean_distance'] + 1e-12)
    
    # Distance distribution analysis
    q10, q50, q90 = np.percentile(upper_tri, [10, 50, 90])
    properties['distance_spread'] = q90 / (q10 + 1e-12)
    properties['skewness'] = (q50 - q10) / (q90 - q50 + 1e-12)
    
    # Graph size categories
    if n < 50:
        properties['size_category'] = 'small'
        properties['neighborhood_size'] = min(10, n-1)
    elif n < 200:
        properties['size_category'] = 'medium'
        properties['neighborhood_size'] = min(15, n-1)
    elif n < 1000:
        properties['size_category'] = 'large'
        properties['neighborhood_size'] = min(20, n-1)
    else:
        properties['size_category'] = 'very_large'
        properties['neighborhood_size'] = min(25, n-1)
    
    # Clustering tendency
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        # Sample distances to estimate clustering
        sample_size = min(1000, n*(n-1)//2)
        sample_indices = np.random.choice(len(upper_tri), sample_size, replace=False)
        sample_distances = upper_tri[sample_indices]
        
        # Estimate clustering coefficient from distance distribution
        threshold = np.percentile(sample_distances, 30)
        close_ratio = np.mean(sample_distances < threshold)
        properties['clustering_tendency'] = min(1.0, close_ratio * 2)
    
    return properties

def _compute_ensemble_factors(distance_matrix: np.ndarray, n: int, 
                              properties: dict) -> dict:
    """Compute multiple heuristic factors with adaptive parameters."""
    factors = {}
    
    # 1. Base reciprocal factor with adaptive scaling
    with np.errstate(divide='ignore', invalid='ignore'):
        safe_dist = distance_matrix + np.eye(n) * 1e-12
        reciprocal = 1 / safe_dist
        np.fill_diagonal(reciprocal, 0)
    
    # Adaptive scaling based on distance variation
    if properties['cv_distance'] > 1.5:
        # High variation: use logarithmic compression
        max_val = np.max(reciprocal[reciprocal < np.inf])
        if max_val > 0:
            reciprocal = np.log1p(reciprocal / max_val)
    factors['reciprocal'] = reciprocal
    
    # 2. Spatial clustering factor
    factors['spatial'] = _compute_spatial_clustering(distance_matrix, n, properties)
    
    # 3. MST-based connectivity factor
    factors['mst'] = _compute_mst_factor(distance_matrix, n, properties)
    
    # 4. Local neighborhood consistency factor
    factors['local'] = _compute_local_factor(distance_matrix, n, properties)
    
    # 5. Path continuation factor (based on angle approximation)
    factors['continuation'] = _compute_continuation_factor(distance_matrix, n, properties)
    
    return factors

def _compute_spatial_clustering(distance_matrix: np.ndarray, n: int,
                               properties: dict) -> np.ndarray:
    """Compute spatial clustering factor using adaptive methods."""
    spatial = np.zeros_like(distance_matrix)
    k = properties['neighborhood_size']
    
    if n <= 300:
        # Exact nearest neighbors for small/medium graphs
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            indices = np.argpartition(dists, k)[:k]
            spatial[i, indices] = 1.0
    else:
        # Landmark-based approximation for large graphs
        landmarks = min(100, n // 20)
        landmark_indices = np.random.choice(n, landmarks, replace=False)
        landmark_dists = distance_matrix[:, landmark_indices]
        
        # Approximate coordinates using landmarks
        approx_coords = landmark_dists
        
        # Build KD-tree on landmark space
        tree = KDTree(approx_coords)
        
        # Query for approximate nearest neighbors
        for i in range(n):
            # Query more points than needed for robustness
            query_k = min(k * 2, n - 1)
            distances, indices = tree.query(approx_coords[i:i+1], k=query_k)
            indices = indices[0]
            
            # Filter to k closest based on actual distances
            if i in indices:
                mask = indices != i
                indices = indices[mask]
                distances = distances[0][mask]
            
            # Take top k by approximate distance
            top_k = min(k, len(indices))
            spatial[i, indices[:top_k]] = 1.0
    
    return spatial

def _compute_mst_factor(distance_matrix: np.ndarray, n: int,
                       properties: dict) -> np.ndarray:
    """Compute MST-based connectivity factor."""
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    mst_factor = np.zeros_like(distance_matrix)
    
    # Strong weight for actual MST edges
    mst_factor[mst_edges > 0] = 2.0
    
    # Also consider edges to MST-connected nodes
    mst_connected = np.any(mst_edges > 0, axis=1)
    
    # For nodes not well connected to MST, add nearest neighbors
    for i in range(n):
        if not mst_connected[i] or np.random.random() < 0.3:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            k = min(3, n - 1)
            indices = np.argpartition(dists, k)[:k]
            for j in indices:
                if mst_factor[i, j] == 0:
                    mst_factor[i, j] = 1.0
    
    # Ensure symmetry
    mst_factor = np.maximum(mst_factor, mst_factor.T)
    return mst_factor

def _compute_local_factor(distance_matrix: np.ndarray, n: int,
                         properties: dict) -> np.ndarray:
    """Compute local neighborhood consistency factor."""
    m = properties['neighborhood_size'] // 2
    
    # Compute local statistics efficiently
    local_means = np.zeros(n)
    local_vars = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        m_smallest = np.partition(dists, m)[:m]
        local_means[i] = np.mean(m_smallest)
        local_vars[i] = np.var(m_smallest) + 1e-12
    
    # Vectorized computation of local similarity
    mean_i, mean_j = np.meshgrid(local_means, local_means, indexing='ij')
    var_i, var_j = np.meshgrid(local_vars, local_vars, indexing='ij')
    
    # Gaussian similarity based on local context
    mean_diff = np.abs(mean_i - mean_j)
    pooled_var = var_i + var_j
    
    with np.errstate(divide='ignore', invalid='ignore'):
        context_similarity = np.exp(-0.5 * (mean_diff ** 2) / pooled_var)
        
        # Distance relative to local context
        context_mean = (mean_i + mean_j) / 2
        distance_ratio = distance_matrix / (context_mean + 1e-12)
        distance_factor = np.exp(-0.5 * np.minimum(distance_ratio, 3) ** 2)
    
    local_factor = context_similarity * distance_factor
    np.fill_diagonal(local_factor, 0)
    
    return local_factor

def _compute_continuation_factor(distance_matrix: np.ndarray, n: int,
                                properties: dict) -> np.ndarray:
    """Compute path continuation factor using triangle inequality approximation."""
    continuation = np.ones_like(distance_matrix)
    
    # Only compute for medium/large graphs where it's beneficial
    if n < 50 or properties['size_category'] == 'very_large':
        np.fill_diagonal(continuation, 0)
        return continuation
    
    # Approximate continuation score using triangle inequality
    # For edge (i,j), we want edges (i,k) and (k,j) that could form good triangles
    for i in range(n):
        for j in range(i + 1, n):
            if distance_matrix[i, j] > np.percentile(distance_matrix[i], 70):
                # Not a promising edge, skip detailed computation
                continue
            
            # Sample candidate intermediate nodes
            sample_size = min(20, n - 2)
            candidates = np.setdiff1d(np.arange(n), [i, j])
            if len(candidates) > sample_size:
                candidates = np.random.choice(candidates, sample_size, replace=False)
            
            # Evaluate triangle quality
            triangle_scores = []
            for k in candidates:
                # Check if triangle i-k-j forms a good continuation
                ik = distance_matrix[i, k]
                kj = distance_matrix[k, j]
                ij = distance_matrix[i, j]
                
                # Triangle inequality gives us bound on detour cost
                detour_cost = ik + kj - ij
                if detour_cost > 0:
                    # Good continuation: detour not too expensive
                    score = np.exp(-detour_cost / (ij + 1e-12))
                    triangle_scores.append(score)
            
            if triangle_scores:
                continuation[i, j] = continuation[j, i] = 1.0 + 0.3 * np.mean(triangle_scores)
    
    np.fill_diagonal(continuation, 0)
    return continuation

def _adaptive_factor_combination(factors: dict, distance_matrix: np.ndarray,
                                n: int, properties: dict) -> np.ndarray:
    """Intelligently combine heuristic factors with learned weights."""
    # Learn weights based on graph properties
    if properties['size_category'] == 'small':
        weights = {
            'reciprocal': 0.4,
            'spatial': 0.3,
            'mst': 0.2,
            'local': 0.1,
            'continuation': 0.0
        }
    elif properties['size_category'] == 'medium':
        weights = {
            'reciprocal': 0.3,
            'spatial': 0.3,
            'mst': 0.2,
            'local': 0.1,
            'continuation': 0.1
        }
    elif properties['size_category'] == 'large':
        weights = {
            'reciprocal': 0.25,
            'spatial': 0.3,
            'mst': 0.25,
            'local': 0.1,
            'continuation': 0.1
        }
    else:  # very_large
        weights = {
            'reciprocal': 0.2,
            'spatial': 0.35,
            'mst': 0.3,
            'local': 0.1,
            'continuation': 0.05
        }
    
    # Adjust weights based on clustering tendency
    if properties['clustering_tendency'] > 0.7:
        weights['spatial'] *= 1.2
        weights['local'] *= 0.8
    elif properties['clustering_tendency'] < 0.3:
        weights['mst'] *= 1.2
        weights['spatial'] *= 0.8
    
    # Normalize weights
    total_weight = sum(weights.values())
    for key in weights:
        weights[key] /= total_weight
    
    # Combine factors
    combined = np.zeros_like(distance_matrix)
    for factor_name, factor_matrix in factors.items():
        # Normalize each factor to [0,1] range
        max_val = np.max(factor_matrix)
        if max_val > 0:
            normalized = factor_matrix / max_val
        else:
            normalized = factor_matrix
        
        combined += weights[factor_name] * normalized
    
    # Additional boost for very short edges
    if properties['distance_spread'] > 3:
        # Large spread: strongly favor short edges
        with np.errstate(divide='ignore', invalid='ignore'):
            short_edge_boost = 1 + 0.5 * np.exp(-distance_matrix / np.percentile(distance_matrix, 5))
        combined *= short_edge_boost
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    return combined

def _graph_aware_sparsification(heuristics_matrix: np.ndarray, n: int,
                               properties: dict) -> np.ndarray:
    """Adaptive sparsification based on graph size and structure."""
    if n <= 20:
        # Keep all edges for very small graphs
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Determine sparsity level based on graph size and clustering
    if properties['size_category'] == 'small':
        keep_ratio = 0.4
    elif properties['size_category'] == 'medium':
        keep_ratio = 0.3
    elif properties['size_category'] == 'large':
        keep_ratio = 0.2
    else:
        keep_ratio = 0.15
    
    # Adjust based on clustering tendency
    if properties['clustering_tendency'] > 0.7:
        keep_ratio *= 0.8  # More sparse for clustered graphs
    elif properties['clustering_tendency'] < 0.3:
        keep_ratio *= 1.2  # Less sparse for uniform graphs
    
    edges_per_node = max(3, int(keep_ratio * n))
    edges_per_node = min(edges_per_node, n - 1)
    
    # Keep top edges for each node
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        if np.any(row > 0):
            top_indices = np.argpartition(-row, edges_per_node)[:edges_per_node]
            sparsified[i, top_indices] = heuristics_matrix[i, top_indices]
    
    # Ensure symmetry (keep union)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified

def _enhanced_triangle_boosting(heuristics_matrix: np.ndarray, n: int,
                               properties: dict) -> np.ndarray:
    """Boost edges that form triangles, with diminishing returns."""
    if n < 10:
        return heuristics_matrix.copy()
    
    boosted = heuristics_matrix.copy()
    
    # Create binary mask of promising edges
    binary_mask = (heuristics_matrix > 0).astype(np.float32)
    
    # Efficient common neighbor counting via matrix multiplication
    common_neighbors = binary_mask @ binary_mask.T
    
    # Apply boosting with diminishing returns
    max_boost = 1.5 if properties['clustering_tendency'] > 0.5 else 1.3
    min_boost = 1.05
    
    for i in range(n):
        for j in range(i + 1, n):
            if heuristics_matrix[i, j] > 0:
                common = common_neighbors[i, j]
                if common > 0:
                    # Diminishing returns: sqrt scaling
                    boost_factor = min_boost + (max_boost - min_boost) * np.tanh(common / 5.0)
                    boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost_factor
    
    return boosted

def _normalize_heuristics(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Final normalization and cleanup."""
    result = heuristics_matrix.copy()
    
    # Set diagonal to zero
    np.fill_diagonal(result, 0)
    
    # Normalize to [0,1] range
    max_val = np.max(result)
    if max_val > 0:
        result = result / max_val
    
    # Ensure perfect symmetry (numerical stability)
    result = (result + result.T) / 2
    
    return result
