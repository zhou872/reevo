import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection probability.
    
    Combines multiple factors:
    1. Inverse distance (base attractiveness)
    2. Node connectivity score (how well nodes connect to others)
    3. Edge competition (edges compete within local neighborhoods)
    4. Probabilistic sparsification (unpromising edges set to zero)
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (shorter edges are better)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Node connectivity score - measure of how central/well-connected each node is
    # Nodes that are close to many others get higher scores
    node_scores = np.zeros(n)
    for i in range(n):
        # Average proximity to other nodes (excluding self)
        node_scores[i] = np.sum(1 / (distance_matrix[i] + 1e-10)) - 1 / (distance_matrix[i, i] + 1e-10)
    
    # Normalize node scores
    if np.max(node_scores) > 0:
        node_scores = node_scores / np.max(node_scores)
    
    # 3. Combined edge score: base * node connectivity product
    # Edges connecting two well-connected nodes get higher scores
    edge_scores = inv_dist.copy()
    for i in range(n):
        for j in range(i + 1, n):
            connectivity_factor = (node_scores[i] * node_scores[j]) ** 0.5
            edge_scores[i, j] *= (1.0 + connectivity_factor)
            edge_scores[j, i] = edge_scores[i, j]
    
    # 4. Edge competition - edges compete with their k nearest alternatives
    k = min(5, n - 1)  # Number of competing edges to consider
    competition_factor = np.zeros((n, n))
    
    for i in range(n):
        # Get distances from node i to all others
        distances_i = distance_matrix[i].copy()
        distances_i[i] = np.inf  # Ignore self
        
        # Find k+1 smallest distances (including the best)
        if len(distances_i) > k:
            # Get indices of k+1 smallest distances
            k_smallest_indices = np.argpartition(distances_i, k + 1)[:k + 1]
            k_smallest_distances = distances_i[k_smallest_indices]
            
            # For each edge from i, compute how it compares to alternatives
            for j in range(n):
                if i == j:
                    continue
                    
                # Find this edge's rank among alternatives
                alt_distances = k_smallest_distances.copy()
                # If j is not in the k+1 smallest, add it
                if j not in k_smallest_indices[:k]:
                    alt_distances = np.append(alt_distances, distances_i[j])
                
                # Sort and get rank
                sorted_indices = np.argsort(alt_distances)
                rank = np.where(sorted_indices == np.where(alt_distances == distances_i[j])[0][0])[0][0]
                
                # Competition score: higher for edges that are among the best alternatives
                competition_factor[i, j] = max(0, 1.0 - rank / (len(alt_distances) - 1))
    
    # Apply competition factor (average of both directions)
    for i in range(n):
        for j in range(i + 1, n):
            comp_score = (competition_factor[i, j] + competition_factor[j, i]) / 2.0
            edge_scores[i, j] *= (0.5 + 0.5 * comp_score)
            edge_scores[j, i] = edge_scores[i, j]
    
    # 5. Adaptive sparsification - set unpromising edges to zero
    # Use quantile-based threshold that adapts to the distribution
    
    # Flatten upper triangle (excluding diagonal)
    triu_indices = np.triu_indices(n, k=1)
    edge_values = edge_scores[triu_indices]
    
    if len(edge_values) > 0:
        # Calculate adaptive threshold - keep top 60% of edges
        threshold = np.percentile(edge_values, 40)  # 40th percentile = keep top 60%
        
        # Create sparsified matrix
        result = np.zeros_like(edge_scores)
        
        # Apply threshold: keep edges above threshold, zero out others
        for i in range(n):
            for j in range(i + 1, n):
                if edge_scores[i, j] > threshold:
                    result[i, j] = edge_scores[i, j]
                    result[j, i] = edge_scores[i, j]
        
        # Ensure diagonal is zero
        np.fill_diagonal(result, 0)
        
        # If too sparse, relax the threshold (keep at least n edges per node on average)
        avg_nonzero = np.count_nonzero(result) / (2 * n)  # Divide by 2 for undirected edges
        if avg_nonzero < 3:  # Want at least 3 edges per node on average
            # Use more relaxed threshold (top 80%)
            threshold = np.percentile(edge_values, 20)
            result = np.zeros_like(edge_scores)
            for i in range(n):
                for j in range(i + 1, n):
                    if edge_scores[i, j] > threshold:
                        result[i, j] = edge_scores[i, j]
                        result[j, i] = edge_scores[i, j]
            np.fill_diagonal(result, 0)
    else:
        result = edge_scores
    
    # 6. Final normalization to avoid extreme values
    max_val = np.max(result)
    if max_val > 0:
        result = result / max_val
    
    return result
