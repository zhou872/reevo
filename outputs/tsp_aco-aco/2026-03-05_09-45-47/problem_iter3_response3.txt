import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Heuristic for TSP edge selection focusing on adaptive sparsification and triangle boosting.
    
    The function computes a matrix of edge promises where higher values indicate more promising
    edges for inclusion in a TSP solution. It uses reciprocal distances as a base, adaptively
    sparsifies to keep only the most promising edges per node, and boosts scores for edges that
    form triangles (common neighbors) to encourage cluster connectivity.
    
    Parameters:
    - distance_matrix: A square 2D numpy array of pairwise distances between nodes.
    
    Returns:
    - A numpy array of the same shape as distance_matrix with edge promise scores in [0, 1].
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases (n < 3) by returning reciprocal distances normalized
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        # Normalize to [0, 1] for consistency
        max_val = np.max(result)
        if max_val > 0:
            result = result / max_val
        return result
    
    # Step 1: Base heuristic using reciprocal distances (ignore diagonal)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(reciprocal_weights, 0)
    
    # Step 2: Adaptive sparsification — keep top m edges per node, with m adaptive to log2(n)
    m = max(3, min(n - 1, int(np.log2(n)) + 2))  # Number of edges to keep per node
    mask = np.zeros_like(reciprocal_weights, dtype=bool)
    
    for i in range(n):
        row = reciprocal_weights[i].copy()
        row[i] = -np.inf  # Exclude self from top edges
        # Get indices of top m edges for node i
        top_indices = np.argpartition(-row, m)[:m]
        mask[i, top_indices] = True
    
    # Ensure symmetry: keep edge if it's in top m for either endpoint
    mask = np.logical_or(mask, mask.T)
    sparsified = reciprocal_weights * mask
    
    # Step 3: Triangle boosting — enhance scores for edges with common neighbors
    # Binary matrix indicating non-zero edges after sparsification
    B = (sparsified > 0).astype(float)
    # Matrix of common neighbor counts (excluding self due to zero diagonal in B)
    common_neighbors = B @ B.T
    
    # Boost parameters
    beta = 0.2      # Boost factor per common neighbor
    max_common = 5  # Cap on number of common neighbors considered for boosting
    
    # Vectorized boosting: apply boost factor only where sparsified is non-zero
    boost_factors = 1 + beta * np.minimum(common_neighbors, max_common)
    boosted = sparsified * boost_factors
    
    # Step 4: Normalize to [0, 1] range for consistent scaling
    max_val = np.max(boosted)
    if max_val > 0:
        boosted = boosted / max_val
    
    np.fill_diagonal(boosted, 0)  # Ensure diagonal is zero
    return boosted
