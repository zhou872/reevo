import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.manifold import MDS
from sklearn.neighbors import KDTree
import warnings
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray, random_state: Optional[int] = None) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple strategies.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges for inclusion in TSP tours.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances between nodes.
    random_state : Optional[int]
        Seed for reproducible random operations.
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, with diagonal set to zero.
    """
    n = distance_matrix.shape[0]
    if random_state is not None:
        np.random.seed(random_state)
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Base heuristic: reciprocal of distances (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
        # Normalize to [0, 1]
        if np.max(reciprocal_weights) > 0:
            reciprocal_weights = reciprocal_weights / np.max(reciprocal_weights)
    
    # Factor 1: Spatial clustering with scalable approximation
    spatial_promising = _compute_spatial_clustering_v2(distance_matrix, n)
    
    # Factor 2: Minimum Spanning Tree and near-MST edges
    mst_promising = _compute_mst_promises_v2(distance_matrix)
    
    # Factor 3: Local neighborhood statistics
    local_factors = _compute_local_factors_v2(distance_matrix, n)
    
    # Combine factors with adaptive weighting
    heuristics_matrix = _combine_factors_v2(
        reciprocal_weights, spatial_promising, mst_promising, 
        local_factors, distance_matrix, n
    )
    
    # Adaptive node-wise sparsification
    sparsified = _adaptive_sparsification_v2(heuristics_matrix, n)
    
    # Triangle boosting with sampling for large instances
    final_heuristics = _triangle_boosting_v2(sparsified, n)
    
    # Final normalization and symmetry enforcement
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    final_heuristics = (final_heuristics + final_heuristics.T) / 2
    
    return final_heuristics


def _compute_spatial_clustering_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute spatial clustering factor using scalable coordinate approximation."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    # Use different strategies based on problem size
    if n <= 1000:
        # Use MDS for accurate coordinates
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(5, n-1), 
                dissimilarity='precomputed', 
                random_state=42,
                n_init=1,
                max_iter=50,
                eps=1e-4
            )
            try:
                coords = mds.fit_transform(distance_matrix)
            except:
                # Fallback: use random projection
                coords = np.random.randn(n, 2)
    else:
        # Landmark-based approximation for large instances
        landmarks = min(100, max(20, n // 50))
        landmark_indices = np.random.choice(n, landmarks, replace=False)
        
        # Approximate coordinates via distance to landmarks
        coords = np.zeros((n, landmarks))
        for i in range(n):
            coords[i] = distance_matrix[i, landmark_indices]
    
    # Build KD-tree for efficient nearest neighbor queries
    tree = KDTree(coords, leaf_size=min(40, n//10))
    
    # Adaptive k based on log2(n)
    k = max(3, min(int(2 * np.log2(n)) + 1, n - 1))
    
    # Find nearest neighbors for each node
    for i in range(n):
        if n <= 1000 or landmarks >= n:
            # Exact query for smaller instances
            distances, indices = tree.query(coords[i:i+1], k=min(k+1, n))
            indices = indices[0]
            # Exclude self if present
            if indices[0] == i:
                indices = indices[1:k+1]
            else:
                indices = indices[:k]
        else:
            # Approximate: use distance matrix directly for top k
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            indices = np.argpartition(dists, k)[:k]
        
        # Assign promising scores with distance-based decay
        for j in indices:
            if i != j:
                # Score decays with relative distance rank
                spatial_promising[i, j] = 1.0
    
    return spatial_promising


def _compute_mst_promises_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute promises based on MST edges with extended neighborhood."""
    n = distance_matrix.shape[0]
    mst_promising = np.zeros_like(distance_matrix)
    
    # Compute MST
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    # Strong boost for direct MST edges
    mst_promising = 2.0 * (mst_edges + mst_edges.T)
    
    # Extended neighborhood: include k nearest neighbors for each node
    k_extended = min(5, n - 1)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        closest = np.argpartition(dists, k_extended)[:k_extended]
        
        for j in closest:
            if mst_promising[i, j] == 0:
                # Distance-based decay for near-MST edges
                rel_dist = dists[j] / np.min(dists[closest])
                mst_promising[i, j] = max(mst_promising[i, j], 1.0 / (1.0 + rel_dist))
    
    return mst_promising


def _compute_local_factors_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute local connectivity factors using neighborhood statistics."""
    # Size of local neighborhood to analyze
    m = min(10, n - 1)
    local_stats = np.zeros((n, 2))  # [mean_dist, std_dist]
    
    # Compute local statistics for each node
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.partition(dists, m)[:m]
        local_stats[i, 0] = np.mean(sorted_dists)
        local_stats[i, 1] = np.std(sorted_dists) + 1e-12
    
    # Compute local factor matrix
    local_factors = np.ones_like(distance_matrix)
    
    # Vectorized computation for efficiency
    for i in range(n):
        # Precompute statistics for node i
        mean_i, std_i = local_stats[i]
        
        for j in range(i+1, n):
            mean_j, std_j = local_stats[j]
            
            # Similarity based on local statistics
            stat_diff = np.abs(mean_i - mean_j)
            stat_scale = std_i + std_j
            stat_similarity = np.exp(-stat_diff / stat_scale)
            
            # Distance relative to local context
            mean_dist = (mean_i + mean_j) / 2
            dist_ratio = distance_matrix[i, j] / (mean_dist + 1e-12)
            
            # Combine factors with emphasis on moderate distances
            score = stat_similarity * np.exp(-0.3 * dist_ratio)
            
            local_factors[i, j] = local_factors[j, i] = score
    
    return local_factors


def _combine_factors_v2(reciprocal_weights: np.ndarray, 
                        spatial_promising: np.ndarray,
                        mst_promising: np.ndarray,
                        local_factors: np.ndarray,
                        distance_matrix: np.ndarray,
                        n: int) -> np.ndarray:
    """Combine all heuristic factors with adaptive weighting."""
    # Initialize combined matrix
    combined = np.zeros_like(reciprocal_weights)
    
    # Base weights with distance decay
    base_weights = reciprocal_weights.copy()
    
    # Apply spatial clustering factor
    spatial_weight = np.where(spatial_promising > 0, 
                            1.5 + 0.5 * spatial_promising, 
                            0.3)
    
    # Apply MST factor with stronger emphasis
    mst_weight = np.where(mst_promising > 0,
                        1.0 + mst_promising,
                        0.5)
    
    # Adaptive combination based on problem size
    if n < 100:
        # More balanced combination for small instances
        weights = [1.0, 0.8, 1.2, 0.7]  # [base, spatial, mst, local]
    else:
        # Emphasize spatial and MST for larger instances
        weights = [0.8, 1.0, 1.5, 0.6]
    
    # Combine with weighted geometric mean for smoother blending
    combined = (
        base_weights ** weights[0] *
        spatial_weight ** weights[1] *
        mst_weight ** weights[2] *
        local_factors ** weights[3]
    )
    
    # Additional boost for very short edges
    short_edge_boost = np.exp(-distance_matrix / (np.median(distance_matrix) + 1e-12))
    combined *= (1.0 + 0.3 * short_edge_boost)
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    return combined


def _adaptive_sparsification_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Adaptively sparsify heuristic matrix keeping top connections per node."""
    if n <= 10:
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Adaptive sparsity: keep top p% edges per node
    # p decreases logarithmically with n
    base_p = max(0.1, min(0.4, 0.5 - 0.1 * np.log10(n)))
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Ignore diagonal
        
        # Determine number of edges to keep
        min_keep = max(2, int(0.05 * n))  # At least 2 edges
        max_keep = min(n - 1, int(0.5 * n))  # At most 50% of edges
        num_to_keep = min(max_keep, max(min_keep, int(base_p * n)))
        
        if np.any(row > 0):
            # Get threshold for top k edges
            if num_to_keep < n - 1:
                threshold = np.partition(row, -num_to_keep)[-num_to_keep]
            else:
                threshold = -np.inf
            
            # Keep edges above threshold, with gradual cutoff
            mask = row >= threshold
            sparsified[i] = heuristics_matrix[i] * mask
            
            # Soft penalty for edges just below threshold
            near_threshold = (row >= 0.8 * threshold) & (row < threshold)
            sparsified[i][near_threshold] = heuristics_matrix[i][near_threshold] * 0.3
    
    # Symmetrize: keep edge if either endpoint considers it promising
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified


def _triangle_boosting_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Boost scores for edges that form triangles, with sampling for large n."""
    boosted = heuristics_matrix.copy()
    
    # Skip if matrix is too sparse
    avg_nonzero = np.count_nonzero(heuristics_matrix) / (n * n)
    if avg_nonzero < 0.1:
        return boosted
    
    if n > 500:
        # Use sampling approximation for large instances
        sample_size = min(200, n // 5)
        if sample_size < 10:
            return boosted
            
        sampled_nodes = np.random.choice(n, sample_size, replace=False)
        
        # Build adjacency list for sampled nodes
        adj_lists = []
        for i in sampled_nodes:
            neighbors = np.where(heuristics_matrix[i] > 0)[0]
            adj_lists.append(set(neighbors))
        
        # Check triangles among sampled nodes
        for idx_i, i in enumerate(sampled_nodes):
            neighbors_i = adj_lists[idx_i]
            for idx_j, j in enumerate(sampled_nodes[idx_i+1:], start=idx_i+1):
                if j in neighbors_i:
                    neighbors_j = adj_lists[idx_j]
                    common = len(neighbors_i & neighbors_j)
                    if common > 0:
                        # Boost proportional to number of common neighbors
                        boost = 1.0 + 0.1 * min(common, 10)
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    else:
        # Exact computation for smaller instances
        # Convert to binary adjacency for efficient triangle counting
        binary_adj = (heuristics_matrix > 0).astype(float)
        
        # Count common neighbors via matrix multiplication
        common_neighbors = binary_adj @ binary_adj.T
        
        # Apply boosting
        for i in range(n):
            for j in range(i+1, n):
                if heuristics_matrix[i, j] > 0:
                    common = common_neighbors[i, j]
                    if common > 0:
                        # Diminishing returns for many common neighbors
                        boost = 1.0 + 0.15 * np.log1p(common)
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    
    return boosted
