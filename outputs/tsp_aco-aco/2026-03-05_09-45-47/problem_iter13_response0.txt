import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.manifold import MDS
from sklearn.neighbors import KDTree
import warnings


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection combining spatial clustering,
    triangle quality boosting, and adaptive sparsification.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Base heuristic: reciprocal of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Factor 1: Improved spatial clustering with adaptive landmark selection
    spatial_promising = _compute_spatial_clustering_v2(distance_matrix, n)
    
    # Factor 2: Minimum Spanning Tree edges with extended neighborhood
    mst_promising = _compute_mst_promises_v2(distance_matrix)
    
    # Factor 3: Local connectivity with enhanced neighborhood statistics
    local_factors = _compute_local_factors_v2(distance_matrix, n)
    
    # Factor 4: Triangle quality boosting
    triangle_quality = _compute_triangle_quality(distance_matrix, n)
    
    # Combine all factors efficiently
    heuristics_matrix = _combine_factors_v2(
        reciprocal_weights, spatial_promising, mst_promising, 
        local_factors, triangle_quality, distance_matrix
    )
    
    # Adaptive sparsification with density-aware thresholding
    sparsified = _adaptive_sparsification_v2(heuristics_matrix, n)
    
    # Enhanced triangle boosting with quality-weighted common neighbors
    final_heuristics = _triangle_boosting_v2(sparsified, distance_matrix, n)
    
    # Normalize to [0, 1] range with softmax-like scaling
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics


def _compute_spatial_clustering_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute spatial clustering factor using improved coordinate approximation."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    if n <= 800:
        # Use MDS for accurate coordinates when n is manageable
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(3, n-1), 
                dissimilarity='precomputed', 
                random_state=42,
                n_init=1,
                max_iter=100,
                normalized_stress='auto'
            )
            coords = mds.fit_transform(distance_matrix)
    else:
        # Use farthest-point sampling for landmark selection in large graphs
        landmarks = min(80, n // 15)
        
        # Select initial landmark randomly
        landmarks_indices = [np.random.randint(0, n)]
        
        # Greedily select farthest points from current landmarks
        for _ in range(1, landmarks):
            # Compute minimum distance from each point to current landmarks
            min_dists = np.min(distance_matrix[:, landmarks_indices], axis=1)
            # Choose the point farthest from all current landmarks
            next_landmark = np.argmax(min_dists)
            landmarks_indices.append(next_landmark)
        
        landmark_indices = np.array(landmarks_indices)
        landmark_dists = distance_matrix[landmark_indices]
        
        # Approximate coordinates using distance to landmarks
        coords = np.zeros((n, landmarks))
        for i in range(n):
            coords[i] = np.array([distance_matrix[i, idx] for idx in landmark_indices])
    
    # Build KD-tree for spatial queries
    tree = KDTree(coords)
    
    # Adaptive k based on graph density and size
    k = max(3, min(int(1.5 * np.log2(n)) + 2, n - 1))
    
    # Find nearest neighbors for each node using batch query
    if n <= 800:
        distances, indices = tree.query(coords, k=k+1)
        for i in range(n):
            # Exclude self
            idx = indices[i]
            if distances[i, 0] == 0:
                spatial_promising[i, idx[1:k+1]] = 1.0
            else:
                spatial_promising[i, idx[:k]] = 1.0
    else:
        # Use distance matrix for large n with adaptive threshold
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Adaptive threshold based on distribution
            sorted_dists = np.sort(dists)
            if n > 1000:
                # For very large graphs, use percentile-based threshold
                threshold = np.percentile(sorted_dists, min(10, 100*k/n))
                mask = dists <= threshold
                spatial_promising[i, mask] = 1.0
            else:
                indices = np.argpartition(dists, k)[:k]
                spatial_promising[i, indices] = 1.0
    
    return spatial_promising


def _compute_mst_promises_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute promises based on MST and extended neighborhood."""
    n = distance_matrix.shape[0]
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    # Make symmetric
    mst_promising = mst_edges + mst_edges.T
    
    # Extended neighborhood: consider nodes up to 2 hops away in MST
    # Compute adjacency matrix of MST
    mst_adj = mst_edges + mst_edges.T
    
    # Two-hop connections through MST
    mst_two_hop = np.minimum(mst_adj @ mst_adj, 1)
    np.fill_diagonal(mst_two_hop, 0)
    
    # Add two-hop connections with reduced weight
    mst_promising = np.where(mst_two_hop > 0, 
                            np.maximum(mst_promising, 0.3), 
                            mst_promising)
    
    # Also boost edges to the k-nearest neighbors in original distance
    k = min(5, n - 1)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Get indices of k closest nodes (excluding self)
        closest = np.argsort(dists)[:k]
        for j in closest:
            if mst_promising[i, j] == 0:
                mst_promising[i, j] = 0.5 / (np.where(closest == j)[0][0] + 1)
    
    return mst_promising


def _compute_local_factors_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute enhanced local connectivity factors."""
    # Adaptive neighborhood size based on graph size
    m = min(7, max(3, n // 20))
    
    # Precompute local statistics using vectorized operations
    dists_sorted = np.partition(distance_matrix, range(1, m+2), axis=1)
    local_means = np.mean(dists_sorted[:, 1:m+1], axis=1, keepdims=True)
    local_stds = np.std(dists_sorted[:, 1:m+1], axis=1, keepdims=True) + 1e-12
    
    # Compute similarity matrix
    mean_diff = np.abs(local_means - local_means.T)
    std_sum = local_stds + local_stds.T + 1e-12
    stat_similarity = np.exp(-mean_diff / std_sum)
    
    # Compute distance ratio matrix
    mean_dist_matrix = (local_means + local_means.T) / 2
    dist_ratio = distance_matrix / (mean_dist_matrix + 1e-12)
    
    # Combine factors with Gaussian kernel
    local_factors = stat_similarity * np.exp(-0.5 * dist_ratio**2)
    
    # Set diagonal to 0
    np.fill_diagonal(local_factors, 0)
    
    return local_factors


def _compute_triangle_quality(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute triangle quality scores for edge boosting."""
    triangle_quality = np.ones_like(distance_matrix)
    
    if n > 1000:
        # Use sampling for large graphs
        sample_size = min(200, n)
        sampled_nodes = np.random.choice(n, sample_size, replace=False)
        
        for i in sampled_nodes:
            # Get k-nearest neighbors
            k = min(15, n - 1)
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            neighbors = np.argsort(dists)[:k]
            
            for j in neighbors:
                if j <= i:
                    continue
                
                # Find common neighbors between i and j
                k_neighbors = min(10, n - 2)
                dists_i = distance_matrix[i].copy()
                dists_i[i] = dists_i[j] = np.inf
                neighbors_i = np.argsort(dists_i)[:k_neighbors]
                
                dists_j = distance_matrix[j].copy()
                dists_j[i] = dists_j[j] = np.inf
                neighbors_j = np.argsort(dists_j)[:k_neighbors]
                
                common = set(neighbors_i) & set(neighbors_j)
                
                if common:
                    # Compute average triangle quality
                    qualities = []
                    for k_node in common:
                        # Triangle quality = inverse of average edge length
                        avg_edge_len = (distance_matrix[i, k_node] + 
                                       distance_matrix[j, k_node] + 
                                       distance_matrix[i, j]) / 3
                        qualities.append(1.0 / (avg_edge_len + 1e-12))
                    
                    quality_score = np.mean(qualities) if qualities else 1.0
                    triangle_quality[i, j] = triangle_quality[j, i] = (
                        1.0 + 0.2 * min(len(common), 4) * quality_score
                    )
    else:
        # Compute exactly for smaller graphs
        # Precompute nearest neighbors for each node
        k = min(20, n - 1)
        neighbor_sets = []
        
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            neighbors = set(np.argsort(dists)[:k])
            neighbor_sets.append(neighbors)
        
        # Compute triangle quality for each edge
        for i in range(n):
            for j in range(i+1, n):
                common = neighbor_sets[i] & neighbor_sets[j]
                
                if common:
                    qualities = []
                    for k_node in common:
                        avg_edge_len = (distance_matrix[i, k_node] + 
                                       distance_matrix[j, k_node] + 
                                       distance_matrix[i, j]) / 3
                        qualities.append(1.0 / (avg_edge_len + 1e-12))
                    
                    quality_score = np.mean(qualities) if qualities else 1.0
                    triangle_quality[i, j] = triangle_quality[j, i] = (
                        1.0 + 0.25 * min(len(common), 4) * quality_score
                    )
    
    np.fill_diagonal(triangle_quality, 0)
    return triangle_quality


def _combine_factors_v2(reciprocal_weights, spatial_promising, mst_promising,
                       local_factors, triangle_quality, distance_matrix):
    """Combine all heuristic factors with optimized weights."""
    n = reciprocal_weights.shape[0]
    
    # Apply spatial factor with distance-weighted boost
    spatial_factor = np.where(spatial_promising > 0, 
                             2.0 - 0.5 * spatial_promising,  # Reduce for weaker signals
                             0.3)
    
    # Apply MST factor with distance-based decay
    mst_factor = np.where(mst_promising > 0.5, 3.0,
                         np.where(mst_promising > 0.2, 1.8, 0.7))
    
    # Combine base factors
    base_combined = reciprocal_weights * spatial_factor * mst_factor * local_factors
    
    # Apply triangle quality as multiplicative boost
    quality_adjusted = base_combined * triangle_quality
    
    # Add distance-based logarithmic correction
    with np.errstate(divide='ignore', invalid='ignore'):
        # Use normalized distances for logarithmic term
        max_dist = np.max(distance_matrix)
        if max_dist > 0:
            norm_dists = distance_matrix / max_dist
            log_boost = 1.0 + 0.4 * np.log(1.0 + 1.0 / (norm_dists + 1e-12))
            log_boost = np.nan_to_num(log_boost, nan=1.0, posinf=1.0)
        else:
            log_boost = 1.0
    
    final_matrix = quality_adjusted * log_boost
    
    # Ensure symmetry
    final_matrix = (final_matrix + final_matrix.T) / 2
    
    return final_matrix


def _adaptive_sparsification_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Adaptively sparsify with density-aware thresholding."""
    sparsified = np.zeros_like(heuristics_matrix)
    
    if n <= 10:
        return heuristics_matrix.copy()
    
    # Adaptive sparsity based on graph density
    # Estimate density from heuristics matrix
    density = np.sum(heuristics_matrix > 1e-6) / (n * (n - 1))
    
    # Adjust keep ratio based on density and size
    if density < 0.1:
        base_keep_ratio = 0.3
    elif density < 0.3:
        base_keep_ratio = 0.25
    else:
        base_keep_ratio = 0.2
    
    # Adjust for graph size
    size_factor = 1.0 / np.log1p(n/10)
    keep_ratio = base_keep_ratio * size_factor
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        if np.all(row <= 0):
            continue
            
        # Use adaptive threshold based on distribution
        positive_vals = row[row > 0]
        if len(positive_vals) > 0:
            # Determine number of edges to keep
            num_to_keep = max(3, int(keep_ratio * n))
            num_to_keep = min(num_to_keep, len(positive_vals))
            
            if num_to_keep > 0:
                # Get threshold for top k edges
                if num_to_keep < len(positive_vals):
                    threshold = np.partition(positive_vals, -num_to_keep)[-num_to_keep]
                else:
                    threshold = np.min(positive_vals)
                
                mask = row >= threshold
                sparsified[i] = heuristics_matrix[i] * mask
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified


def _triangle_boosting_v2(heuristics_matrix: np.ndarray, 
                         distance_matrix: np.ndarray,
                         n: int) -> np.ndarray:
    """Enhanced triangle boosting with quality-weighted common neighbors."""
    boosted = heuristics_matrix.copy()
    
    if n > 600:
        # Use efficient approximation for large graphs
        # Sample edges to boost
        edges_to_boost = min(2000, n * 5)
        
        # Get candidate edges (non-zero in heuristics_matrix)
        edge_indices = np.argwhere(heuristics_matrix > 0)
        if len(edge_indices) > edges_to_boost:
            # Sample edges proportional to their current score
            edge_scores = heuristics_matrix[edge_indices[:, 0], edge_indices[:, 1]]
            probs = edge_scores / np.sum(edge_scores)
            sampled_idx = np.random.choice(len(edge_indices), edges_to_boost, 
                                          replace=False, p=probs)
            edge_indices = edge_indices[sampled_idx]
        
        for i, j in edge_indices:
            if i >= j:
                continue
                
            # Find common neighbors efficiently
            row_i = heuristics_matrix[i] > 0
            row_j = heuristics_matrix[j] > 0
            common = np.where(row_i & row_j)[0]
            common = common[(common != i) & (common != j)]
            
            if len(common) > 0:
                # Limit to top 4 common neighbors by edge quality
                if len(common) > 4:
                    # Score common neighbors by average edge length
                    neighbor_scores = []
                    for k in common:
                        avg_len = (distance_matrix[i, k] + distance_matrix[j, k]) / 2
                        neighbor_scores.append(1.0 / avg_len)
                    top_k = np.argsort(neighbor_scores)[-4:]
                    common = common[top_k]
                
                # Compute quality-weighted boost
                total_boost = 1.0
                for k in common:
                    # Triangle quality: inverse of average edge length
                    avg_len = (distance_matrix[i, k] + 
                              distance_matrix[j, k] + 
                              distance_matrix[i, j]) / 3
                    quality = 1.0 / (avg_len + 1e-12)
                    total_boost *= (1.0 + 0.1 * quality)
                
                boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * total_boost
    else:
        # Exact computation for smaller graphs
        binary_mask = (heuristics_matrix > 0).astype(float)
        
        # Count common neighbors
        common_neighbors = binary_mask @ binary_mask.T
        
        for i in range(n):
            for j in range(i+1, n):
                if heuristics_matrix[i, j] > 0:
                    common_count = int(common_neighbors[i, j])
                    if common_count > 0:
                        # Cap at 4 common neighbors as per suggestion
                        effective_count = min(common_count, 4)
                        
                        # Find actual common neighbors
                        row_i = heuristics_matrix[i] > 0
                        row_j = heuristics_matrix[j] > 0
                        common = np.where(row_i & row_j)[0]
                        common = common[(common != i) & (common != j)]
                        
                        if len(common) > 0:
                            # Compute average triangle quality
                            qualities = []
                            for k in common[:effective_count]:  # Take top by count
                                avg_len = (distance_matrix[i, k] + 
                                          distance_matrix[j, k] + 
                                          distance_matrix[i, j]) / 3
                                qualities.append(1.0 / (avg_len + 1e-12))
                            
                            avg_quality = np.mean(qualities) if qualities else 1.0
                            boost = 1.0 + 0.15 * effective_count * avg_quality
                            boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    
    return boosted
