import numpy as np
import networkx as nx
from scipy.sparse import csr_matrix
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.cluster import SpectralClustering, KMeans
from sklearn.metrics.pairwise import cosine_similarity, pairwise_distances

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection combining graph theory,
    spectral methods, and adaptive patterns from solution structures.
    
    Key Innovations:
    1. Spectral clustering for natural neighborhood identification
    2. Edge betweenness centrality to identify critical connections  
    3. Minimum spanning tree for connectivity preservation
    4. Adaptive sparsification with degree awareness
    """
    n = distance_matrix.shape[0]
    
    # Base case for very small instances
    if n < 4:
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal, 0)
        return reciprocal / np.max(reciprocal) if np.max(reciprocal) > 0 else reciprocal
    
    # 1. Base reciprocal distance (fundamental heuristic)
    base_heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(base_heuristic, 0)
    
    # 2. Spectral clustering to identify natural neighborhoods
    n_clusters = min(max(3, int(np.sqrt(n))), 10)
    
    try:
        # Normalize distances for spectral clustering
        max_dist = np.max(distance_matrix)
        if max_dist > 0:
            affinity_matrix = np.maximum(0, 1 - distance_matrix / max_dist)
        else:
            affinity_matrix = np.ones((n, n)) - np.eye(n)
        
        spectral = SpectralClustering(
            n_clusters=n_clusters,
            affinity='precomputed',
            random_state=42,
            n_init=5
        )
        labels = spectral.fit_predict(affinity_matrix)
    except Exception:
        # Fallback to k-means on distance-based features
        k_features = min(5, n - 1)
        features = np.zeros((n, k_features))
        for i in range(n):
            sorted_dists = np.sort(distance_matrix[i])[1:k_features + 1]
            features[i, :len(sorted_dists)] = sorted_dists
        labels = KMeans(n_clusters=n_clusters, random_state=42, n_init=5).fit_predict(features)
    
    # Create intra-cluster mask
    intra_cluster_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        intra_cluster_mask[i] = (labels == labels[i])
    np.fill_diagonal(intra_cluster_mask, False)
    
    # 3. Edge betweenness centrality approximation
    G = nx.Graph()
    # Add edges for top k nearest neighbors for each node
    k_nearest_for_graph = min(10, n - 1)
    for i in range(n):
        neighbors = np.argsort(distance_matrix[i])[1:k_nearest_for_graph + 1]
        for j in neighbors:
            if i < j:
                G.add_edge(i, j, weight=distance_matrix[i, j])
    
    edge_betweenness = np.zeros((n, n))
    if len(G.edges()) > 0:
        # Sample nodes for betweenness approximation
        sample_size = min(50, n)
        sample_nodes = np.random.choice(n, sample_size, replace=False)
        
        # Calculate betweenness centrality for sampled nodes
        betweenness = nx.betweenness_centrality_subset(
            G,
            sources=list(sample_nodes),
            targets=list(sample_nodes),
            normalized=True,
            weight='weight'
        )
        
        # Convert to edge betweenness (average of endpoint betweenness)
        for i, j in G.edges():
            edge_betweenness[i, j] = edge_betweenness[j, i] = (
                betweenness.get(i, 0) + betweenness.get(j, 0)
            ) / 2
    
    # 4. Minimum Spanning Tree edges
    mst = minimum_spanning_tree(csr_matrix(distance_matrix))
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # 5. Node embeddings via distance to representative nodes
    k_representative = min(8, n - 1)
    
    # Select representative nodes using centrality
    if n <= 100:
        centrality = np.sum(1 / (distance_matrix + 1e-12), axis=1)
        representatives = np.argsort(centrality)[-k_representative:]
    else:
        # For larger instances, use random sampling with centrality bias
        centrality = np.sum(1 / (distance_matrix + 1e-12), axis=1)
        prob = centrality / np.sum(centrality)
        representatives = np.random.choice(n, k_representative, replace=False, p=prob)
    
    # Create embeddings as distances to representatives
    node_embeddings = distance_matrix[:, representatives]
    
    # Compute embedding similarity
    embedding_similarity = cosine_similarity(node_embeddings)
    np.fill_diagonal(embedding_similarity, 0)
    
    # 6. Combine all factors with adaptive weights
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Dynamic weights based on instance size
    size_factor = min(1.0, n / 100)
    
    for i in range(n):
        for j in range(i + 1, n):
            # Base score from reciprocal distance
            score = base_heuristic[i, j]
            
            # Cluster factor: boost intra-cluster edges
            cluster_factor = 2.0 if intra_cluster_mask[i, j] else 0.7
            
            # Betweenness factor: edges with high betweenness are important
            betweenness_factor = 1.0 + 2.0 * edge_betweenness[i, j]
            
            # MST factor: edges in MST are crucial
            mst_factor = 3.0 if mst_mask[i, j] else 0.8
            
            # Embedding similarity factor
            embedding_factor = 1.0 + 0.5 * embedding_similarity[i, j]
            
            # Local neighborhood factor (mutual nearest neighbors)
            k_nearest = min(max(3, int(np.log2(n))), n - 1)
            i_nearest = set(np.argsort(distance_matrix[i])[1:k_nearest + 1])
            j_nearest = set(np.argsort(distance_matrix[j])[1:k_nearest + 1])
            mutual_factor = 1.5 if (j in i_nearest and i in j_nearest) else 1.0
            
            # Combine all factors
            combined = (score * cluster_factor * betweenness_factor *
                       mst_factor * embedding_factor * mutual_factor)
            
            # Add small noise to break ties (encourages exploration)
            noise = 1.0 + 0.05 * np.random.random()
            
            heuristics_matrix[i, j] = heuristics_matrix[j, i] = combined * noise
    
    # 7. Adaptive sparsification with degree preservation
    # First, ensure MST edges are kept with high score
    heuristics_matrix[mst_mask] *= 2.0
    
    # Target degree per node (adaptive based on problem size)
    target_degree = min(max(4, int(np.log2(n) * 2)), n - 1)
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    for i in range(n):
        # Get scores for this node
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        # Always keep MST edges
        mst_neighbors = np.where(mst_mask[i])[0]
        
        # Keep top target_degree edges (including MST edges)
        positive_scores = scores[scores > 0]
        if len(positive_scores) > target_degree:
            # Get threshold for top target_degree edges
            threshold = np.partition(scores, -target_degree)[-target_degree]
            keep_mask = (scores >= threshold) | (np.isin(np.arange(n), mst_neighbors))
        else:
            # Keep all positive edges plus MST edges
            keep_mask = (scores > 0) | (np.isin(np.arange(n), mst_neighbors))
        
        sparsified[i] = heuristics_matrix[i] * keep_mask
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 8. Triangle boosting: favor edges that complete promising triangles
    triangle_boost = np.ones_like(sparsified)
    
    # Only compute for nodes with reasonable degree (for efficiency)
    for i in range(n):
        neighbors = np.where(sparsified[i] > 0)[0]
        if len(neighbors) > 2:
            for j in neighbors:
                if j <= i:
                    continue
                # Find common neighbors
                common = np.intersect1d(
                    np.where(sparsified[i] > 0)[0],
                    np.where(sparsified[j] > 0)[0]
                )
                if len(common) > 0:
                    # Boost if the triangle is "tight" (small perimeter)
                    best_boost = 1.0
                    # Check only first few common neighbors for efficiency
                    for k in common[:min(3, len(common))]:
                        perimeter = (distance_matrix[i, j] +
                                   distance_matrix[i, k] +
                                   distance_matrix[j, k])
                        triangle_ratio = distance_matrix[i, j] / (perimeter / 3 + 1e-12)
                        boost = 1.0 + 0.3 * np.exp(-triangle_ratio)
                        best_boost = max(best_boost, boost)
                    triangle_boost[i, j] = triangle_boost[j, i] = best_boost
    
    final_heuristics = sparsified * triangle_boost
    
    # Ensure connectivity by boosting weak connections if needed
    if np.count_nonzero(final_heuristics) < n * 2:  # Too sparse
        for i in range(n):
            if np.count_nonzero(final_heuristics[i]) < 2:
                nearest = np.argsort(distance_matrix[i])[1:3]
                for j in nearest:
                    final_heuristics[i, j] = final_heuristics[j, i] = max(
                        final_heuristics[i, j],
                        base_heuristic[i, j] * 0.5
                    )
    
    # Normalize to [0, 1] range
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    
    return final_heuristics
