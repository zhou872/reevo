import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection probability.
    Combines inverse distance with MST connectivity and local sparsification.
    """
    n = distance_matrix.shape[0]
    
    # 1. Inverse distance base (like v1, but handle zeros)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree connectivity
    # MST tends to include many edges that appear in optimal tours
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0  # Make symmetric
    
    # 3. Nearest neighbor preference
    # For each node, mark its k nearest neighbors
    k = max(2, int(np.ceil(np.log2(n))))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        indices = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, indices] = True
    nn_mask = nn_mask | nn_mask.T  # Make symmetric
    
    # 4. Combine factors with weights
    # Base: inverse distance
    # Boost edges that are in MST
    # Penalize edges that aren't in top-k nearest neighbors
    weights = np.ones_like(inv_dist)
    weights[mst_mask] *= 1.5  # Boost MST edges
    weights[~nn_mask] *= 0.3  # Penalize non-near neighbors
    
    heuristic = inv_dist * weights
    
    # 5. Sparsify: set low-probability edges to zero
    # Keep only top edges per node and global threshold
    # Local sparsification per node
    for i in range(n):
        row = heuristic[i].copy()
        # Keep top (log n) edges per node
        keep_count = min(max(2, int(np.ceil(np.log2(n)))), n-1)
        if keep_count < n-1:
            threshold = np.partition(row, -keep_count)[-keep_count]
            row[row < threshold] = 0
            heuristic[i] = row
    
    # Global sparsification
    non_zero_vals = heuristic[heuristic > 0]
    if len(non_zero_vals) > 0:
        global_threshold = np.percentile(non_zero_vals, 25)  # Keep top 75%
        heuristic[heuristic < global_threshold] = 0
    
    # Ensure symmetry (since TSP is undirected)
    heuristic = np.maximum(heuristic, heuristic.T)
    
    return heuristic
