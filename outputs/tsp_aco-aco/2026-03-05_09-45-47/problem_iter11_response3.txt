import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved heuristic for TSP edge selection emphasizing:
    1. Adaptive sparsification via K-nearest neighbors
    2. Minimum spanning tree edges
    3. Triangle-based closure boosting
    
    Returns a matrix of edge promises (higher = more promising).
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Base: reciprocal of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal, 0)
    
    # 1. SIZE-ADAPTIVE KNN SPARSIFICATION
    knn_matrix = np.zeros_like(distance_matrix)
    # Dynamic k based on log10(n), minimum 3
    k = max(3, int(np.log10(n)) * 2)
    k = min(k, n - 1)  # Cannot exceed n-1
    
    for i in range(n):
        # Get distances for node i
        dists = distance_matrix[i].copy()
        dists[i] = np.inf  # Exclude self
        
        # Get indices of k smallest distances
        if k < n - 1:
            # Use argpartition for O(n) instead of O(n log n) sort
            nearest_idx = np.argpartition(dists, k)[:k]
        else:
            nearest_idx = np.argsort(dists)[:k]
        
        # Set KNN edges
        knn_matrix[i, nearest_idx] = 1.0
    
    # Make symmetric (if i is in j's KNN, boost edge)
    knn_symmetric = (knn_matrix + knn_matrix.T) > 0
    knn_matrix = knn_symmetric.astype(float)
    
    # 2. MINIMUM SPANNING TREE EDGES
    mst = minimum_spanning_tree(distance_matrix)
    mst_binary = (mst.toarray() > 0).astype(float)
    # Make symmetric and give strong weight
    mst_matrix = (mst_binary + mst_binary.T) * 2.0
    
    # 3. COMBINE FACTORS WITH ADAPTIVE WEIGHTS
    combined = np.zeros_like(distance_matrix)
    
    # Base reciprocal weights, scaled by KNN
    base_promising = reciprocal * (1 + knn_matrix)  # Boost KNN edges
    
    # Strong boost for MST edges
    mst_boost = np.where(mst_matrix > 0, 3.0, 1.0)
    
    # Combine
    combined = base_promising * mst_boost
    
    # 4. TRIANGLE-BASED CLOSURE BOOSTING
    # Count common neighbors efficiently
    knn_binary = (knn_matrix > 0).astype(float)
    
    # Matrix multiplication gives common neighbor counts
    common_neighbors = knn_binary @ knn_binary.T
    
    # Boost edges with common neighbors
    for i in range(n):
        for j in range(i + 1, n):
            if combined[i, j] > 0:  # Only boost existing promising edges
                common = common_neighbors[i, j]
                if common > 0:
                    # Boost proportional to common neighbors, capped
                    boost = 1.0 + 0.15 * min(common, 4)
                    combined[i, j] *= boost
                    combined[j, i] = combined[i, j]
    
    # 5. FINAL SPARSIFICATION
    # Keep only top edges per node
    final_matrix = np.zeros_like(combined)
    
    # Adaptive number to keep: based on log10(n)
    keep_per_node = max(3, int(np.log10(n)) * 2)
    keep_per_node = min(keep_per_node, n - 1)
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Ignore self
        
        # Find threshold for top edges
        if keep_per_node < n - 1:
            # Use partition to find threshold efficiently
            kth_largest = np.partition(row, -keep_per_node)[-keep_per_node]
            mask = row >= kth_largest
        else:
            mask = np.ones(n, dtype=bool)
            mask[i] = False
        
        # Apply mask
        final_matrix[i] = combined[i] * mask
    
    # Ensure symmetry by taking max of both directions
    final_matrix = np.maximum(final_matrix, final_matrix.T)
    
    # Normalize to [0, 1]
    max_val = np.max(final_matrix)
    if max_val > 0:
        final_matrix /= max_val
    
    np.fill_diagonal(final_matrix, 0)
    return final_matrix
