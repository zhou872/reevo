import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform
from collections import defaultdict
import random

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with adaptive learning.
    
    Returns a matrix of edge promises combining:
    1. Dynamic weight adaptation based on sampled tour quality
    2. Ensemble sampling of diverse candidate solutions
    3. Graph centrality measures for edge importance
    4. Adaptive sparsification preserving connectivity
    
    Args:
        distance_matrix: Square distance matrix
    
    Returns:
        Heuristic matrix of same shape indicating edge promise
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Initialize adaptive weights for different heuristic components
    weights = {
        'reciprocal': 1.0,
        'mst': 1.5,
        'local': 0.8,
        'betweenness': 1.2,
        'ensemble': 2.0
    }
    
    # Component 1: Base reciprocal heuristic (inverse distance)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # Normalize reciprocal component
    if np.max(reciprocal) > 0:
        reciprocal = reciprocal / np.max(reciprocal)
    
    # Component 2: Minimum spanning tree edges
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    mst_component = mst_edges.astype(float)
    
    # Component 3: Local neighborhood analysis with adaptive k
    k_local = max(3, int(np.sqrt(n)))
    local_promises = np.zeros_like(distance_matrix)
    
    for i in range(n):
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        # Get k nearest neighbors
        nearest_indices = np.argpartition(distances, k_local)[:k_local]
        local_promises[i, nearest_indices] = 1.0
        
        # Extra boost for mutual nearest neighbors
        for j in nearest_indices:
            if i in np.argpartition(distance_matrix[j], k_local)[:k_local]:
                local_promises[i, j] = 1.5
                local_promises[j, i] = 1.5
    
    # Component 4: Edge betweenness centrality approximation
    betweenness = np.zeros_like(distance_matrix)
    
    # Sample random paths to approximate betweenness
    num_paths = min(50, n * 3)
    for _ in range(num_paths):
        # Generate a random permutation (path)
        path = np.random.permutation(n)
        # Add the return to start to complete the cycle
        path = np.append(path, path[0])
        
        # Count edges in this path
        for i in range(len(path) - 1):
            u, v = path[i], path[i+1]
            betweenness[u, v] += 1
            betweenness[v, u] += 1
    
    # Normalize betweenness
    if np.max(betweenness) > 0:
        betweenness = betweenness / np.max(betweenness)
    
    # Component 5: Ensemble sampling of diverse solutions
    ensemble_promises = np.zeros_like(distance_matrix)
    num_samples = min(100, max(20, n * 2))
    
    # Store sampled tours and their lengths for weight adaptation
    sampled_tours = []
    tour_lengths = []
    
    for sample_idx in range(num_samples):
        # Create diverse starting points and strategies
        if sample_idx % 3 == 0:
            # Nearest neighbor from random start
            start = np.random.randint(n)
            unvisited = set(range(n))
            unvisited.remove(start)
            current = start
            tour = [start]
            
            while unvisited:
                # Get distances to unvisited nodes
                mask = np.array([node in unvisited for node in range(n)])
                available_distances = distance_matrix[current].copy()
                available_distances[~mask] = np.inf
                
                # Choose next with some randomness
                if np.random.random() < 0.7:
                    next_node = np.argmin(available_distances)
                else:
                    # Random choice from top candidates
                    sorted_idx = np.argsort(available_distances)
                    k_choice = min(3, len(sorted_idx))
                    next_node = sorted_idx[np.random.randint(k_choice)]
                
                tour.append(next_node)
                unvisited.remove(next_node)
                current = next_node
            
            # Return to start
            tour.append(start)
            
        elif sample_idx % 3 == 1:
            # Random permutation
            tour = list(np.random.permutation(n))
            tour.append(tour[0])
            
        else:
            # Greedy insertion
            start = np.random.randint(n)
            tour = [start, start]
            
            unvisited = set(range(n))
            unvisited.remove(start)
            
            while unvisited:
                best_cost_increase = float('inf')
                best_node = None
                best_position = -1
                
                for node in unvisited:
                    for i in range(len(tour) - 1):
                        u, v = tour[i], tour[i+1]
                        # Cost of inserting node between u and v
                        cost_increase = (distance_matrix[u, node] + 
                                        distance_matrix[node, v] - 
                                        distance_matrix[u, v])
                        
                        if cost_increase < best_cost_increase:
                            best_cost_increase = cost_increase
                            best_node = node
                            best_position = i + 1
                
                tour.insert(best_position, best_node)
                unvisited.remove(best_node)
        
        # Calculate tour length
        tour_length = 0
        for i in range(len(tour) - 1):
            tour_length += distance_matrix[tour[i], tour[i+1]]
        
        sampled_tours.append(tour)
        tour_lengths.append(tour_length)
        
        # Add edges from this tour to ensemble promises
        for i in range(len(tour) - 1):
            u, v = tour[i], tour[i+1]
            ensemble_promises[u, v] += 1
            ensemble_promises[v, u] += 1
    
    # Normalize ensemble promises
    if np.max(ensemble_promises) > 0:
        ensemble_promises = ensemble_promises / np.max(ensemble_promises)
    
    # Adaptive weight adjustment based on sampled tour quality
    if len(tour_lengths) > 1:
        # Calculate correlation between component usage and tour quality
        min_length = min(tour_lengths)
        max_length = max(tour_lengths)
        if max_length > min_length:
            # Normalize tour qualities (lower is better)
            normalized_qualities = 1 - (np.array(tour_lengths) - min_length) / (max_length - min_length)
            
            # For each component, calculate average usage in good tours
            component_usage = defaultdict(list)
            
            for idx, tour in enumerate(sampled_tours):
                quality = normalized_qualities[idx]
                
                # Count component usage in this tour
                tour_edges = set()
                for i in range(len(tour) - 1):
                    u, v = sorted([tour[i], tour[i+1]])
                    tour_edges.add((u, v))
                
                # Check each component's contribution
                comp_values = []
                for u, v in tour_edges:
                    comp_values.append((
                        reciprocal[u, v],
                        mst_component[u, v],
                        local_promises[u, v],
                        betweenness[u, v],
                        ensemble_promises[u, v]
                    ))
                
                if comp_values:
                    avg_values = np.mean(comp_values, axis=0)
                    for comp_idx, comp_name in enumerate(['reciprocal', 'mst', 'local', 'betweenness', 'ensemble']):
                        component_usage[comp_name].append((avg_values[comp_idx], quality))
            
            # Update weights based on correlation with quality
            for comp_name in weights.keys():
                if component_usage[comp_name]:
                    usages, qualities = zip(*component_usage[comp_name])
                    # Simple correlation: if higher usage correlates with better quality, increase weight
                    mean_usage = np.mean(usages)
                    mean_quality = np.mean(qualities)
                    
                    if mean_usage > 0:
                        # Weight adjustment: components that appear more in good tours get higher weight
                        quality_weighted_usage = np.sum([u * q for u, q in component_usage[comp_name]])
                        avg_quality_weighted = quality_weighted_usage / np.sum(qualities) if np.sum(qualities) > 0 else mean_usage
                        
                        # Update weight (smooth adjustment)
                        weights[comp_name] = 0.7 * weights[comp_name] + 0.3 * (avg_quality_weighted / mean_usage if mean_usage > 0 else 1.0)
    
    # Combine all components with adaptive weights
    combined = (weights['reciprocal'] * reciprocal +
                weights['mst'] * mst_component +
                weights['local'] * local_promises +
                weights['betweenness'] * betweenness +
                weights['ensemble'] * ensemble_promises)
    
    # Apply sigmoid normalization for smoother values
    combined = 1 / (1 + np.exp(-combined * 2))
    
    # Dynamic sparsification based on node degree distribution
    if n > 10:
        # Calculate desired sparsity based on problem size
        target_edges = min(n * 3, n * (n - 1) // 4)
        
        # Get all unique edge values
        triu_indices = np.triu_indices(n, k=1)
        edge_values = combined[triu_indices]
        
        if len(edge_values) > target_edges:
            # Find threshold to keep target number of edges
            threshold = np.partition(edge_values, -target_edges)[-target_edges]
            
            # Create sparsity mask
            sparsity_mask = combined >= threshold
            sparsity_mask = sparsity_mask | sparsity_mask.T
            
            # Apply mask
            combined = combined * sparsity_mask
        
        # Ensure minimum connectivity: each node should have at least 2 connections
        for i in range(n):
            connected = np.where(combined[i] > 0)[0]
            if len(connected) < 2:
                # Add the best missing connections
                sorted_idx = np.argsort(combined[i])[::-1]
                added = 0
                for j in sorted_idx:
                    if j != i and combined[i, j] == 0:
                        combined[i, j] = 1.0 / distance_matrix[i, j] if distance_matrix[i, j] > 0 else 0
                        combined[j, i] = combined[i, j]
                        added += 1
                        if added >= 2 - len(connected):
                            break
    
    # Final normalization
    max_val = np.max(combined)
    if max_val > 0:
        combined = combined / max_val
    
    # Ensure symmetry and zero diagonal
    combined = (combined + combined.T) / 2
    np.fill_diagonal(combined, 0)
    
    return combined
