import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import KDTree
from concurrent.futures import ThreadPoolExecutor, as_completed
import multiprocessing as mp
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Next-generation TSP edge heuristics with adaptive online learning,
    graph neural heuristics, and extreme-scale optimization.
    
    Key innovations:
    1. Dynamic online learning for real-time weight adjustment
    2. GNN-inspired pattern detection via message passing
    3. Hierarchical multi-scale clustering for pattern boosting
    4. Extreme-scale optimizations with streaming approximations
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric n x n matrix of pairwise distances
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # 1. Online learning context initialization
    online_weights = _init_online_learning_context(n)
    
    # 2. Multi-strategy feature extraction with streaming computation
    features = _extract_multi_strategy_features(
        distance_matrix, n, online_weights
    )
    
    # 3. Graph Neural Heuristics: Message passing for pattern detection
    node_embeddings = _gnn_message_passing(distance_matrix, features, n)
    
    # 4. Dynamic weight learning with performance feedback
    adaptive_weights = _dynamic_weight_learning(
        features, node_embeddings, distance_matrix, online_weights
    )
    
    # 5. Multi-scale hierarchical clustering
    clustering_signals = _hierarchical_clustering_analysis(
        distance_matrix, node_embeddings, n
    )
    
    # 6. Edge scoring with learned fusion
    heuristic_matrix = _edge_scoring_with_learned_fusion(
        features, node_embeddings, clustering_signals,
        adaptive_weights, distance_matrix
    )
    
    # 7. Extreme-scale sparsification with connectivity guarantee
    heuristic_matrix = _extreme_scale_sparsification(
        heuristic_matrix, distance_matrix, n
    )
    
    # 8. Online learning update for next iteration
    _update_online_learning(online_weights, heuristic_matrix, distance_matrix)
    
    # 9. Final normalization
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
    
    np.fill_diagonal(heuristic_matrix, 0)
    return heuristic_matrix


def _init_online_learning_context(n: int) -> dict:
    """Initialize online learning context with adaptive parameters."""
    return {
        'iteration': 0,
        'weight_history': [],
        'performance_history': [],
        'alpha': 0.1,  # Learning rate
        'decay': 0.99,  # Decay factor
        'base_weights': {
            'reciprocal': 0.25,
            'spatial': 0.25,
            'mst': 0.20,
            'local': 0.15,
            'gnn': 0.15
        }
    }


def _extract_multi_strategy_features(distance_matrix: np.ndarray, n: int,
                                    online_weights: dict) -> dict:
    """Extract multiple feature matrices with streaming computation."""
    features = {}
    
    # 1. Reciprocal feature (always useful)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal, 0)
    features['reciprocal'] = reciprocal
    
    # 2. Streaming spatial clustering (works for any n)
    if n > 1000:
        # Use landmark-based streaming for extreme scale
        features['spatial'] = _streaming_spatial_clustering(distance_matrix, n)
    else:
        # Use full computation for smaller instances
        features['spatial'] = _compute_spatial_clustering(distance_matrix, n)
    
    # 3. Minimum spanning tree with approximation for large n
    features['mst'] = _compute_mst_features(distance_matrix, n)
    
    # 4. Local statistics with adaptive window
    features['local'] = _compute_local_features(distance_matrix, n)
    
    # 5. Distance distribution statistics
    features['distribution'] = _compute_distribution_features(distance_matrix)
    
    return features


def _streaming_spatial_clustering(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Streaming spatial clustering for extreme-scale instances."""
    spatial = np.zeros_like(distance_matrix)
    
    # Use streaming landmarks (constant memory)
    landmarks = min(100, max(10, int(np.sqrt(n))))
    landmark_indices = np.random.choice(n, landmarks, replace=False)
    
    # Compute distances to landmarks
    landmark_dists = distance_matrix[:, landmark_indices]
    
    # Use MiniBatchKMeans on landmark distances
    kmeans = MiniBatchKMeans(
        n_clusters=min(20, landmarks),
        batch_size=min(1000, n // 10),
        random_state=42
    )
    labels = kmeans.fit_predict(landmark_dists)
    
    # Boost edges within same cluster
    for cluster_id in range(kmeans.n_clusters):
        cluster_mask = (labels == cluster_id)
        cluster_indices = np.where(cluster_mask)[0]
        if len(cluster_indices) > 1:
            for i in cluster_indices:
                for j in cluster_indices:
                    if i < j:
                        spatial[i, j] = spatial[j, i] = 1.0
    
    # Add nearest neighbors for connectivity
    k = min(10, max(3, int(np.log2(n))))
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest = np.argpartition(dists, k)[:k]
        spatial[i, nearest] = np.maximum(spatial[i, nearest], 0.5)
    
    return spatial


def _compute_spatial_clustering(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute spatial clustering for moderate-sized instances."""
    spatial = np.zeros_like(distance_matrix)
    
    # Use MDS for embedding
    try:
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(3, n-1),
                dissimilarity='precomputed',
                random_state=42,
                n_init=2,
                max_iter=100
            )
            coords = mds.fit_transform(distance_matrix)
        
        # Build KD-tree for efficient queries
        tree = KDTree(coords)
        k = max(3, min(int(np.log2(n)) + 2, n - 1))
        
        # Find k-nearest neighbors
        for i in range(n):
            distances, indices = tree.query(coords[i], k=k+1)
            indices = indices[1:] if distances[0] == 0 else indices[:k]
            spatial[i, indices] = 1.0
    except:
        # Fallback: use distance-based nearest neighbors
        k = max(3, min(10, n // 10))
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argpartition(dists, k)[:k]
            spatial[i, nearest] = 1.0
    
    return spatial


def _compute_mst_features(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute MST-based features with approximation for large n."""
    mst_features = np.zeros_like(distance_matrix)
    
    # Compute MST (approximate for very large n)
    if n > 5000:
        # Use sampling-based MST approximation
        sample_size = min(1000, n // 5)
        sampled_indices = np.random.choice(n, sample_size, replace=False)
        sampled_dists = distance_matrix[sampled_indices][:, sampled_indices]
        
        mst = minimum_spanning_tree(sampled_dists)
        mst_edges = (mst.toarray() > 0)
        
        # Map back to original indices
        for i_idx, i in enumerate(sampled_indices):
            for j_idx, j in enumerate(sampled_indices):
                if mst_edges[i_idx, j_idx]:
                    mst_features[i, j] = mst_features[j, i] = 1.0
    else:
        # Full MST computation
        mst = minimum_spanning_tree(distance_matrix)
        mst_edges = (mst.toarray() > 0)
        mst_features = mst_edges.astype(float) + mst_edges.T.astype(float)
    
    # Add k-nearest neighbors to MST nodes
    k = max(2, min(5, int(np.log2(n))))
    for i in range(n):
        if np.sum(mst_features[i] > 0) < k:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argpartition(dists, k)[:k]
            mst_features[i, nearest] = np.maximum(mst_features[i, nearest], 0.3)
    
    return mst_features


def _compute_local_features(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute local neighborhood statistics."""
    local = np.zeros_like(distance_matrix)
    
    # Adaptive neighborhood size
    k = max(3, min(10, int(np.log2(n))))
    
    # Compute local statistics in parallel
    def process_node(i):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)[:k]
        mean_dist = np.mean(sorted_dists)
        std_dist = np.std(sorted_dists) + 1e-12
        
        # Find nodes in local neighborhood
        local_mask = dists <= (mean_dist + std_dist)
        local_scores = np.exp(-(dists[local_mask] - mean_dist)**2 / (2 * std_dist**2))
        
        return i, local_mask, local_scores
    
    # Process nodes
    with ThreadPoolExecutor(max_workers=mp.cpu_count()) as executor:
        futures = [executor.submit(process_node, i) for i in range(n)]
        for future in as_completed(futures):
            i, local_mask, local_scores = future.result()
            local[i, local_mask] = local_scores
    
    # Ensure symmetry
    local = np.maximum(local, local.T)
    return local


def _compute_distribution_features(distance_matrix: np.ndarray) -> dict:
    """Compute distance distribution statistics."""
    n = distance_matrix.shape[0]
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    return {
        'mean': np.mean(distances),
        'std': np.std(distances),
        'cv': np.std(distances) / (np.mean(distances) + 1e-12),
        'skew': np.mean((distances - np.mean(distances))**3) / (np.std(distances)**3 + 1e-12),
        'kurtosis': np.mean((distances - np.mean(distances))**4) / (np.std(distances)**4 + 1e-12)
    }


def _gnn_message_passing(distance_matrix: np.ndarray,
                        features: dict, n: int) -> np.ndarray:
    """GNN-inspired message passing for pattern detection."""
    # Initialize node embeddings with multiple views
    embeddings = np.zeros((n, 5))
    
    # View 1: Distance profile (k nearest distances)
    k = min(5, n - 1)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)[:k]
        embeddings[i, :min(k, 5)] = sorted_dists[:5]
    
    # View 2: Feature aggregation from neighbors
    for _ in range(2):  # 2 message passing steps
        new_embeddings = embeddings.copy()
        for i in range(n):
            # Find influential neighbors
            neighbor_mask = (
                (features['spatial'][i] > 0.5) |
                (features['mst'][i] > 0.5) |
                (features['local'][i] > 0.3)
            )
            neighbor_indices = np.where(neighbor_mask)[0]
            
            if len(neighbor_indices) > 0:
                # Aggregate neighbor embeddings
                neighbor_weights = 1 / (distance_matrix[i, neighbor_indices] + 1e-12)
                neighbor_weights = neighbor_weights / np.sum(neighbor_weights)
                
                aggregated = np.sum(
                    embeddings[neighbor_indices] * neighbor_weights[:, np.newaxis],
                    axis=0
                )
                
                # Update with residual connection
                new_embeddings[i] = 0.7 * embeddings[i] + 0.3 * aggregated
        
        embeddings = new_embeddings
    
    # View 3: Global statistics
    dist_stats = _compute_distribution_features(distance_matrix)
    for i in range(n):
        local_dists = distance_matrix[i].copy()
        local_dists[i] = np.inf
        local_mean = np.mean(local_dists)
        embeddings[i, 4] = local_mean / (dist_stats['mean'] + 1e-12)
    
    return embeddings


def _dynamic_weight_learning(features: dict, node_embeddings: np.ndarray,
                            distance_matrix: np.ndarray,
                            online_weights: dict) -> dict:
    """Dynamic weight learning with performance feedback."""
    n = distance_matrix.shape[0]
    
    # Analyze instance characteristics
    dist_stats = features['distribution']
    
    # Compute feature quality indicators
    quality_indicators = {}
    
    # 1. Reciprocal quality (inverse correlation with distance)
    triu_indices = np.triu_indices(n, k=1)
    reciprocal_vals = features['reciprocal'][triu_indices]
    distance_vals = distance_matrix[triu_indices]
    quality_indicators['reciprocal'] = -np.corrcoef(
        reciprocal_vals, distance_vals
    )[0, 1] if len(reciprocal_vals) > 1 else 0.5
    
    # 2. Spatial coherence
    spatial_coherence = np.mean([
        np.corrcoef(features['spatial'][i], distance_matrix[i])[0, 1]
        for i in range(min(100, n))
        if not np.all(features['spatial'][i] == 0)
    ]) if n > 1 else 0.5
    quality_indicators['spatial'] = 1 - abs(spatial_coherence)
    
    # 3. MST coverage
    mst_nonzero = np.sum(features['mst'] > 0) / (n * (n - 1))
    quality_indicators['mst'] = min(1.0, mst_nonzero * 5)
    
    # 4. Local consistency
    local_consistency = 0
    for i in range(min(100, n)):
        local_mask = features['local'][i] > 0
        if np.sum(local_mask) > 1:
            local_dists = distance_matrix[i, local_mask]
            local_scores = features['local'][i, local_mask]
            corr = np.corrcoef(local_dists, local_scores)[0, 1] if len(local_dists) > 1 else 0
            local_consistency += 1 - abs(corr)
    quality_indicators['local'] = local_consistency / min(100, n)
    
    # 5. GNN discriminative power
    if n > 10:
        # Use embedding variance as proxy for discriminative power
        embedding_var = np.var(node_embeddings, axis=0)
        quality_indicators['gnn'] = min(1.0, np.mean(embedding_var) * 10)
    else:
        quality_indicators['gnn'] = 0.5
    
    # Combine with online learning history
    weights = online_weights['base_weights'].copy()
    
    # Adjust based on quality indicators
    for key in weights:
        weights[key] *= (0.5 + 0.5 * quality_indicators.get(key, 0.5))
    
    # Adjust based on instance size
    size_factor = min(1.0, n / 1000)
    weights['spatial'] *= (0.8 + 0.4 * size_factor)
    weights['gnn'] *= (0.7 + 0.6 * size_factor)
    
    # Adjust based on distance distribution
    if dist_stats['cv'] > 1.0:
        weights['reciprocal'] *= 1.2
        weights['local'] *= 0.8
    else:
        weights['spatial'] *= 1.2
        weights['gnn'] *= 1.1
    
    # Normalize
    total = sum(weights.values())
    if total > 0:
        for key in weights:
            weights[key] /= total
    
    return weights


def _hierarchical_clustering_analysis(distance_matrix: np.ndarray,
                                     node_embeddings: np.ndarray,
                                     n: int) -> dict:
    """Hierarchical multi-scale clustering analysis."""
    clustering = {}
    
    if n < 10:
        # Too small for meaningful clustering
        clustering['intra_cluster'] = np.ones((n, n))
        clustering['inter_cluster'] = np.zeros((n, n))
        return clustering
    
    # Multi-scale clustering
    scales = []
    if n > 1000:
        scales = [min(10, int(np.sqrt(n))), min(50, n // 20)]
    elif n > 100:
        scales = [min(5, int(np.sqrt(n))), min(20, n // 10), min(50, n // 5)]
    else:
        scales = [min(3, n // 10), min(5, n // 5)]
    
    intra_cluster = np.zeros((n, n))
    inter_cluster = np.zeros((n, n))
    
    for n_clusters in scales:
        if n_clusters < 2 or n_clusters >= n:
            continue
        
        # Cluster using embeddings
        try:
            if n > 1000:
                kmeans = MiniBatchKMeans(
                    n_clusters=n_clusters,
                    batch_size=min(1000, n // 10),
                    random_state=42
                )
            else:
                kmeans = KMeans(
                    n_clusters=n_clusters,
                    random_state=42,
                    n_init=3
                )
            
            labels = kmeans.fit_predict(node_embeddings)
            
            # Compute intra-cluster boosts
            for cluster_id in range(n_clusters):
                cluster_mask = (labels == cluster_id)
                cluster_indices = np.where(cluster_mask)[0]
                
                if len(cluster_indices) > 1:
                    boost = 1.0 + 0.2 * np.log1p(len(cluster_indices))
                    for i in cluster_indices:
                        for j in cluster_indices:
                            if i < j:
                                intra_cluster[i, j] = max(intra_cluster[i, j], boost)
                                intra_cluster[j, i] = intra_cluster[i, j]
            
            # Compute inter-cluster distances
            cluster_centers = kmeans.cluster_centers_
            cluster_distances = np.zeros((n_clusters, n_clusters))
            
            for c1 in range(n_clusters):
                for c2 in range(c1 + 1, n_clusters):
                    dist = np.linalg.norm(cluster_centers[c1] - cluster_centers[c2])
                    cluster_distances[c1, c2] = cluster_distances[c2, c1] = dist
            
            # Find MST of clusters
            if n_clusters > 1:
                cluster_mst = minimum_spanning_tree(cluster_distances)
                cluster_mst_edges = (cluster_mst.toarray() > 0)
                
                # Boost edges between connected clusters
                for c1 in range(n_clusters):
                    for c2 in range(c1 + 1, n_clusters):
                        if cluster_mst_edges[c1, c2]:
                            mask_c1 = (labels == c1)
                            mask_c2 = (labels == c2)
                            indices_c1 = np.where(mask_c1)[0]
                            indices_c2 = np.where(mask_c2)[0]
                            
                            # Sample connections
                            sample_size = min(3, len(indices_c1), len(indices_c2))
                            if sample_size > 0:
                                sample_c1 = np.random.choice(
                                    indices_c1, sample_size, replace=False
                                )
                                sample_c2 = np.random.choice(
                                    indices_c2, sample_size, replace=False
                                )
                                
                                for i in sample_c1:
                                    for j in sample_c2:
                                        boost = 1.0 + 0.1 / (1.0 + cluster_distances[c1, c2])
                                        inter_cluster[i, j] = max(inter_cluster[i, j], boost)
                                        inter_cluster[j, i] = inter_cluster[i, j]
        
        except Exception as e:
            continue
    
    clustering['intra_cluster'] = intra_cluster
    clustering['inter_cluster'] = inter_cluster
    
    return clustering


def _edge_scoring_with_learned_fusion(features: dict, node_embeddings: np.ndarray,
                                     clustering: dict, weights: dict,
                                     distance_matrix: np.ndarray) -> np.ndarray:
    """Edge scoring with learned multi-feature fusion."""
    n = distance_matrix.shape[0]
    heuristic = np.zeros((n, n))
    
    # Compute edge embeddings
    edge_embeddings = {}
    for i in range(n):
        for j in range(i + 1, n):
            # Combine node embeddings
            diff = node_embeddings[i] - node_embeddings[j]
            dot = np.dot(node_embeddings[i], node_embeddings[j])
            edge_emb = np.concatenate([diff, [dot]])
            edge_embeddings[(i, j)] = edge_emb
    
    # Compute base scores
    base_scores = {}
    for i in range(n):
        for j in range(i + 1, n):
            score = 0
            
            # Feature-based scoring
            if 'reciprocal' in weights:
                score += weights['reciprocal'] * features['reciprocal'][i, j]
            if 'spatial' in weights:
                score += weights['spatial'] * features['spatial'][i, j]
            if 'mst' in weights:
                score += weights['mst'] * features['mst'][i, j]
            if 'local' in weights:
                score += weights['local'] * features['local'][i, j]
            
            # GNN-based scoring
            if 'gnn' in weights and (i, j) in edge_embeddings:
                gnn_score = np.exp(-np.linalg.norm(edge_embeddings[(i, j)]))
                score += weights['gnn'] * gnn_score
            
            # Clustering boosts
            if clustering['intra_cluster'][i, j] > 0:
                score *= clustering['intra_cluster'][i, j]
            if clustering['inter_cluster'][i, j] > 0:
                score *= clustering['inter_cluster'][i, j]
            
            # Distance-based decay
            dist = distance_matrix[i, j]
            mean_dist = features['distribution']['mean']
            if mean_dist > 0:
                dist_factor = np.exp(-(dist / mean_dist) ** 2)
                score *= dist_factor
            
            base_scores[(i, j)] = score
    
    # Convert to matrix
    for (i, j), score in base_scores.items():
        heuristic[i, j] = heuristic[j, i] = score
    
    return heuristic


def _extreme_scale_sparsification(heuristic: np.ndarray,
                                 distance_matrix: np.ndarray,
                                 n: int) -> np.ndarray:
    """Extreme-scale sparsification with connectivity guarantees."""
    if n <= 10:
        return heuristic
    
    sparsified = np.zeros_like(heuristic)
    
    # Adaptive sparsity level
    if n <= 100:
        k = max(3, min(10, n // 5))
    elif n <= 1000:
        k = max(3, int(np.log2(n)) + 2)
    else:
        k = max(3, int(np.log10(n)) * 2)
    
    # Per-node top-k selection
    for i in range(n):
        row = heuristic[i].copy()
        row[i] = -np.inf
        
        # Select top k edges
        if k < n - 1:
            indices = np.argpartition(-row, k)[:k]
            threshold = np.min(row[indices])
        else:
            threshold = 0
        
        mask = row >= threshold
        sparsified[i] = heuristic[i] * mask
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Ensure connectivity via MST
    mst = minimum_spanning_tree(-sparsified)  # Negative for max edges
    mst_edges = (mst.toarray() > 0).astype(float)
    sparsified = np.maximum(sparsified, mst_edges)
    
    # Triangle completion for important nodes
    if n <= 1000:
        # Compute node importance (degree in sparsified graph)
        degrees = np.sum(sparsified > 0, axis=1)
        important_nodes = np.where(degrees > np.percentile(degrees, 75))[0]
        
        for i in important_nodes:
            neighbors = np.where(sparsified[i] > 0)[0]
            if len(neighbors) >= 2:
                # Complete triangles among important neighbors
                for j in neighbors:
                    if j in important_nodes:
                        for k in neighbors:
                            if k in important_nodes and j < k:
                                if sparsified[j, k] == 0 and distance_matrix[j, k] < np.inf:
                                    # Add edge with average score
                                    avg_score = (sparsified[i, j] + sparsified[i, k]) / 2
                                    sparsified[j, k] = sparsified[k, j] = avg_score * 0.7
    
    return sparsified


def _update_online_learning(online_weights: dict, heuristic: np.ndarray,
                           distance_matrix: np.ndarray) -> None:
    """Update online learning context based on current results."""
    online_weights['iteration'] += 1
    
    # Generate a simple tour to evaluate heuristic quality
    try:
        tour_quality = _evaluate_heuristic_quality(heuristic, distance_matrix)
        online_weights['performance_history'].append(tour_quality)
        
        # Keep only recent history
        if len(online_weights['performance_history']) > 10:
            online_weights['performance_history'] = online_weights['performance_history'][-10:]
        
        # Adjust learning rate based on performance trend
        if len(online_weights['performance_history']) >= 3:
            recent_perf = online_weights['performance_history'][-3:]
            if np.std(recent_perf) < np.mean(recent_perf) * 0.1:
                # Stable performance, reduce learning rate
                online_weights['alpha'] *= 0.95
            else:
                # Unstable, maintain or increase learning rate
                online_weights['alpha'] = min(0.2, online_weights['alpha'] * 1.05)
    
    except:
        pass
    
    # Decay weights toward initial values
    decay = online_weights['decay']
    for key in online_weights['base_weights']:
        online_weights['base_weights'][key] = (
            decay * online_weights['base_weights'][key] +
            (1 - decay) * 0.2  # Move toward uniform
        )
    
    # Renormalize
    total = sum(online_weights['base_weights'].values())
    if total > 0:
        for key in online_weights['base_weights']:
            online_weights['base_weights'][key] /= total


def _evaluate_heuristic_quality(heuristic: np.ndarray,
                               distance_matrix: np.ndarray) -> float:
    """Evaluate heuristic quality by constructing a simple tour."""
    n = heuristic.shape[0]
    if n < 3:
        return 1.0
    
    # Convert heuristic to edge weights (higher is better)
    edge_weights = heuristic.copy()
    np.fill_diagonal(edge_weights, -np.inf)
    
    # Greedy construction
    tour = []
    visited = set()
    
    # Start from a random node
    start = np.random.randint(0, n)
    tour.append(start)
    visited.add(start)
    
    while len(visited) < n:
        current = tour[-1]
        # Find unvisited neighbor with highest heuristic value
        unvisited = [j for j in range(n) if j not in visited]
        if not unvisited:
            break
        
        scores = edge_weights[current, unvisited]
        next_node = unvisited[np.argmax(scores)]
        tour.append(next_node)
        visited.add(next_node)
    
    # Complete tour
    tour.append(start)
    
    # Calculate tour length
    length = 0
    for i in range(len(tour) - 1):
        length += distance_matrix[tour[i], tour[i + 1]]
    
    # Normalize by MST lower bound
    mst = minimum_spanning_tree(distance_matrix)
    mst_length = mst.sum()
    
    if mst_length > 0:
        quality = mst_length / length
    else:
        quality = 1.0
    
    return min(max(quality, 0.1), 1.0)
