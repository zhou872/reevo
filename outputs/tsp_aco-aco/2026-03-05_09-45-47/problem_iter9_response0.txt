import numpy as np
from sklearn.manifold import MDS
from scipy.spatial import KDTree
from scipy.sparse.csgraph import minimum_spanning_tree
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with improved triangle boosting.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Base heuristic: reciprocal of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Helper function for spatial clustering
    def compute_spatial_clustering(dist_mat, n_nodes):
        spatial_promising = np.zeros_like(dist_mat)
        if n_nodes <= 500:  # Use MDS for accurate coordinates when n ≤ 500
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                mds = MDS(
                    n_components=min(3, n_nodes-1),
                    dissimilarity='precomputed',
                    random_state=42,
                    n_init=1,
                    max_iter=100
                )
                coords = mds.fit_transform(dist_mat)
            tree = KDTree(coords)
            k = max(3, min(int(np.log2(n_nodes)) + 2, n_nodes - 1))
            for i in range(n_nodes):
                distances, indices = tree.query(coords[i], k=k+1)
                indices = indices[1:] if distances[0] == 0 else indices[:k]
                spatial_promising[i, indices] = 1.0
        else:
            # Use distance-based approximation for larger graphs
            landmarks = min(50, n_nodes // 20)
            landmark_indices = np.random.choice(n_nodes, landmarks, replace=False)
            coords = np.array([[dist_mat[i, idx] for idx in landmark_indices] for i in range(n_nodes)])
            k = max(3, min(int(np.log2(n_nodes)) + 2, n_nodes - 1))
            for i in range(n_nodes):
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                indices = np.argpartition(dists, k)[:k]
                spatial_promising[i, indices] = 1.0
        return spatial_promising
    
    spatial_promising = compute_spatial_clustering(distance_matrix, n)
    
    # Helper function for MST promises
    def compute_mst_promises(dist_mat):
        n_nodes = dist_mat.shape[0]
        mst = minimum_spanning_tree(dist_mat)
        mst_edges = (mst.toarray() > 0).astype(float)
        mst_promising = mst_edges + mst_edges.T
        for i in range(n_nodes):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            closest = np.argsort(dists)[:3]
            for j in closest:
                if mst_promising[i, j] == 0:
                    mst_promising[i, j] = 0.5
        return mst_promising
    
    mst_promising = compute_mst_promises(distance_matrix)
    
    # Helper function for local factors
    def compute_local_factors(dist_mat, n_nodes):
        m = min(5, n_nodes - 1)
        local_stats = np.zeros((n_nodes, 2))
        for i in range(n_nodes):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            sorted_dists = np.sort(dists)[:m]
            local_stats[i, 0] = np.mean(sorted_dists)
            local_stats[i, 1] = np.std(sorted_dists) + 1e-12
        local_factors = np.ones_like(dist_mat)
        for i in range(n_nodes):
            for j in range(i+1, n_nodes):
                stat_similarity = np.exp(
                    -np.abs(local_stats[i, 0] - local_stats[j, 0]) /
                    (local_stats[i, 1] + local_stats[j, 1])
                )
                mean_dist = (local_stats[i, 0] + local_stats[j, 0]) / 2
                dist_ratio = dist_mat[i, j] / (mean_dist + 1e-12)
                local_factors[i, j] = local_factors[j, i] = stat_similarity * np.exp(-0.5 * dist_ratio**2)
        return local_factors
    
    local_factors = compute_local_factors(distance_matrix, n)
    
    # Combine all factors
    def combine_factors(recip_weights, spatial_prom, mst_prom, local_facs, dist_mat):
        n_nodes = recip_weights.shape[0]
        base_scores = recip_weights.copy()
        spatial_factor = np.where(spatial_prom > 0, 2.0, 0.5)
        mst_factor = np.where(mst_prom > 0, np.where(mst_prom == 2, 3.0, 1.5), 0.8)
        combined_scores = base_scores * spatial_factor * mst_factor * local_facs
        with np.errstate(divide='ignore', invalid='ignore'):
            log_boost = 1 + 0.3 * np.log(1 + 1 / (base_scores + 1e-12))
            log_boost = np.nan_to_num(log_boost, nan=1.0, posinf=1.0)
        heuristics_mat = combined_scores * log_boost
        heuristics_mat = (heuristics_mat + heuristics_mat.T) / 2
        return heuristics_mat
    
    heuristics_matrix = combine_factors(reciprocal_weights, spatial_promising, mst_promising, local_factors, distance_matrix)
    
    # Adaptive sparsification
    def adaptive_sparsification(heuristics_mat, n_nodes):
        sparsified = np.zeros_like(heuristics_mat)
        if n_nodes <= 10:
            return heuristics_mat.copy()
        base_keep_ratio = min(0.4, max(0.15, np.log10(n_nodes) / 10))
        for i in range(n_nodes):
            row = heuristics_mat[i].copy()
            row[i] = -np.inf
            num_to_keep = max(3, int(base_keep_ratio * n_nodes))
            num_to_keep = min(num_to_keep, n_nodes - 1)
            if np.any(row > 0):
                top_indices = np.argpartition(-row, num_to_keep)[:num_to_keep]
                threshold = np.min(row[top_indices])
                mask = row >= threshold
                sparsified[i] = heuristics_mat[i] * mask
        sparsified = np.maximum(sparsified, sparsified.T)
        return sparsified
    
    sparsified = adaptive_sparsification(heuristics_matrix, n)
    
    # Distance-aware triangle boosting with capping
    def triangle_boosting_with_distance(sparse_mat, dist_mat, n_nodes):
        boosted = sparse_mat.copy()
        if n_nodes > 500:  # Approximation for large graphs
            sample_size = min(100, n_nodes)
            sampled_nodes = np.random.choice(n_nodes, sample_size, replace=False)
            for i in sampled_nodes:
                neighbors_i = set(np.where(sparse_mat[i] > 0)[0])
                for j in neighbors_i:
                    if j <= i:
                        continue
                    neighbors_j = set(np.where(sparse_mat[j] > 0)[0])
                    common_neighbors = neighbors_i & neighbors_j
                    common = len(common_neighbors)
                    if common > 0:
                        total_inv_dist = 0.0
                        for k in common_neighbors:
                            avg_dist = (dist_mat[i, j] + dist_mat[i, k] + dist_mat[j, k]) / 3
                            inv_dist = 1 / (avg_dist + 1e-12)
                            total_inv_dist += inv_dist
                        avg_inv_dist = total_inv_dist / common
                        # Compute scale factor based on median distance
                        median_dist = np.median(dist_mat[dist_mat > 0]) if np.any(dist_mat > 0) else 1.0
                        dist_boost = min(avg_inv_dist * median_dist, 1.0)  # Cap to 1.0
                        boost = 1 + 0.2 * min(common, 5) * dist_boost  # Capped linear boost
                        boosted[i, j] = boosted[j, i] = sparse_mat[i, j] * boost
        else:
            binary_mask = (sparse_mat > 0).astype(float)
            common_neighbors_count = binary_mask @ binary_mask.T
            median_dist = np.median(distance_matrix[distance_matrix > 0]) if np.any(distance_matrix > 0) else 1.0
            for i in range(n_nodes):
                for j in range(i+1, n_nodes):
                    if sparse_mat[i, j] > 0:
                        common = int(common_neighbors_count[i, j])
                        if common > 0:
                            neighbors_i = np.where(binary_mask[i] > 0)[0]
                            neighbors_j = np.where(binary_mask[j] > 0)[0]
                            common_neighbors = np.intersect1d(neighbors_i, neighbors_j)
                            total_inv_dist = 0.0
                            for k in common_neighbors:
                                avg_dist = (dist_mat[i, j] + dist_mat[i, k] + dist_mat[j, k]) / 3
                                inv_dist = 1 / (avg_dist + 1e-12)
                                total_inv_dist += inv_dist
                            avg_inv_dist = total_inv_dist / len(common_neighbors) if len(common_neighbors) > 0 else 0
                            dist_boost = min(avg_inv_dist * median_dist, 1.0)
                            boost = 1 + 0.2 * min(common, 5) * dist_boost
                            boosted[i, j] = boosted[j, i] = sparse_mat[i, j] * boost
        return boosted
    
    boosted = triangle_boosting_with_distance(sparsified, distance_matrix, n)
    
    # Normalize to [0, 1] range
    max_val = np.max(boosted)
    if max_val > 0:
        boosted = boosted / max_val
    np.fill_diagonal(boosted, 0)
    return boosted
