import numpy as np
from scipy.spatial.distance import pdist, squareform
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP that combines multiple factors to determine
    edge promise, with intelligent sparsification of unpromising edges.
    
    Strategy:
    1. Multi-scale distance analysis combining absolute and rank-based measures
    2. Adaptive k-nearest neighbor sparsification (different k per node)
    3. Edge density awareness to prevent isolation
    4. Stochastic smoothing to avoid deterministic patterns
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Prevent division by zero and self-loops
    np.fill_diagonal(distance_matrix, np.inf)
    epsilon = 1e-10
    
    # Base inverse distance heuristic (normalized)
    inv_dist = 1.0 / (distance_matrix + epsilon)
    inv_dist_norm = inv_dist / np.max(inv_dist)
    
    # 1. Distance rank-based heuristic (relative importance)
    # For each node, rank edges by distance, giving higher weight to closer nodes
    rank_weights = np.zeros_like(distance_matrix)
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])
        # Exponential decay of weights with rank
        ranks = np.arange(n)
        weights = np.exp(-ranks / (n/4))  # Decay factor adapts to problem size
        rank_weights[i, sorted_indices] = weights
        rank_weights[i, i] = 0  # Remove self
    
    # 2. Local density awareness
    # Nodes in dense regions should have more conservative sparsification
    avg_distances = np.mean(distance_matrix, axis=1)
    density_factors = 1.0 / (avg_distances + epsilon)
    density_factors = density_factors / np.max(density_factors)
    
    # 3. Adaptive k-nearest neighbor selection
    # k scales with node density and problem size
    base_k = max(3, int(np.sqrt(n)))  # Base connectivity
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Adaptive k based on local density
        k = int(base_k * (0.7 + 0.6 * density_factors[i]))
        k = min(k, n - 1)
        
        # Get k-nearest neighbors for this node
        distances_i = distance_matrix[i].copy()
        distances_i[i] = np.inf  # Exclude self
        
        # Find threshold distance for top k neighbors
        kth_dist = np.partition(distances_i, k)[k]
        
        # Combine multiple heuristics for promising edges
        for j in range(n):
            if i == j:
                continue
                
            if distance_matrix[i, j] <= kth_dist:
                # Edge qualifies as promising
                # Combine factors: inverse distance, rank, and stochastic component
                dist_factor = inv_dist_norm[i, j]
                rank_factor = rank_weights[i, j]
                
                # Stochastic smoothing to avoid deterministic patterns
                noise = 0.9 + 0.2 * np.random.random()
                
                # Local clustering coefficient (simplified)
                common_neighbors = 0
                for l in range(n):
                    if l != i and l != j:
                        if (distance_matrix[i, l] <= kth_dist and 
                            distance_matrix[j, l] <= np.partition(distance_matrix[j], k)[k]):
                            common_neighbors += 1
                clustering = 0.5 + 0.5 * (common_neighbors / max(1, k))
                
                # Final heuristic value
                heuristics_matrix[i, j] = (dist_factor * 0.4 + 
                                         rank_factor * 0.3 + 
                                         clustering * 0.2) * noise
            else:
                # Edge not in k-nearest neighbors - sparsify
                heuristics_matrix[i, j] = 0
    
    # 4. Symmetrize the matrix (TSP is undirected)
    heuristics_matrix = 0.5 * (heuristics_matrix + heuristics_matrix.T)
    
    # 5. Ensure connectivity: each node must have at least 2 promising edges
    min_edges = 2
    for i in range(n):
        promising_count = np.sum(heuristics_matrix[i] > 0)
        if promising_count < min_edges:
            # Add back the closest edges until we have min_edges
            distances_i = distance_matrix[i].copy()
            distances_i[i] = np.inf
            sorted_indices = np.argsort(distances_i)
            
            for idx in sorted_indices[:min_edges]:
                if heuristics_matrix[i, idx] == 0:
                    # Reactivate this edge with moderate value
                    heuristics_matrix[i, idx] = 0.3 + 0.2 * np.random.random()
                    heuristics_matrix[idx, i] = heuristics_matrix[i, idx]
    
    # 6. Normalize final matrix
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix

# Simple test function to demonstrate usage
def test_heuristics_v2():
    """Test the heuristics function on a sample TSP instance."""
    np.random.seed(42)
    
    # Create a sample distance matrix (symmetric)
    n = 10
    points = np.random.rand(n, 2) * 100
    dist_matrix = squareform(pdist(points))
    
    print(f"Testing heuristics_v2 on {n} cities")
    print("Distance matrix shape:", dist_matrix.shape)
    
    # Generate heuristics
    heuristics = heuristics_v2(dist_matrix.copy())
    
    print("\nHeuristics matrix properties:")
    print(f"Shape: {heuristics.shape}")
    print(f"Non-zero elements: {np.sum(heuristics > 0)} / {n*n}")
    print(f"Non-zero percentage: {100*np.sum(heuristics > 0)/(n*n):.1f}%")
    print(f"Mean non-zero value: {np.mean(heuristics[heuristics > 0]):.3f}")
    print(f"Max value: {np.max(heuristics):.3f}")
    print(f"Min non-zero value: {np.min(heuristics[heuristics > 0]):.3f}")
    
    # Check symmetry
    diff = np.max(np.abs(heuristics - heuristics.T))
    print(f"\nSymmetry check (max diff): {diff:.6f}")
    
    # Check connectivity
    for i in range(n):
        promising = np.sum(heuristics[i] > 0)
        print(f"City {i}: {promising} promising edges")
    
    return heuristics
