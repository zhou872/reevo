import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import KDTree
from sklearn.manifold import MDS
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining spatial clustering,
    minimum spanning tree, and adaptive sparsification with improved factors.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Base heuristic: reciprocal of distances (with smoothing)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Factor 1: Spatial clustering with improved MDS approximation
    spatial_promising = _compute_spatial_clustering_v2(distance_matrix, n)
    
    # Factor 2: Minimum Spanning Tree edges with extended neighborhood
    mst_promising = _compute_mst_promises_v2(distance_matrix)
    
    # Factor 3: Local connectivity with improved statistics
    local_factors = _compute_local_factors_v2(distance_matrix, n)
    
    # Factor 4: Short path potential (2-hop connections)
    path_potential = _compute_path_potential(distance_matrix, n)
    
    # Combine all factors with adaptive weighting
    heuristics_matrix = _combine_factors_v2(
        reciprocal_weights, spatial_promising, mst_promising,
        local_factors, path_potential, distance_matrix, n
    )
    
    # Adaptive sparsification with density-aware thresholding
    sparsified = _adaptive_sparsification_v2(heuristics_matrix, n)
    
    # Saturating triangle boosting with common neighbor analysis
    final_heuristics = _triangle_boosting_v2(sparsified, n)
    
    # Normalize to [0, 1] range with softmax-like scaling
    max_val = np.max(final_heuristics)
    if max_val > 0:
        # Use sigmoid-like normalization to preserve relative differences
        scaled = final_heuristics / max_val
        # Apply soft threshold: boost high values, suppress very low ones
        final_heuristics = np.where(scaled > 0.1, scaled ** 0.8, scaled ** 1.5)
    
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics

def _compute_spatial_clustering_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute spatial clustering factor using improved coordinate approximation."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    if n <= 1000:
        # Use MDS for accurate coordinates
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(10, n-1),
                dissimilarity='precomputed',
                random_state=42,
                n_init=1,
                max_iter=200,
                eps=1e-4
            )
            try:
                coords = mds.fit_transform(distance_matrix)
            except:
                # Fallback to simpler approach
                coords = np.random.randn(n, 2)
    else:
        # Landmark-based approximation with improved sampling
        landmarks = min(100, max(20, n // 50))
        landmark_indices = np.random.choice(n, landmarks, replace=False)
        # Use farthest point sampling for better coverage
        if landmarks > 20:
            for i in range(1, landmarks):
                # Find point farthest from existing landmarks
                dist_to_landmarks = np.min(distance_matrix[landmark_indices[:i]], axis=0)
                landmark_indices[i] = np.argmax(dist_to_landmarks)
        
        # Approximate coordinates using distance to landmarks
        coords = distance_matrix[:, landmark_indices]
        # Normalize coordinates
        coords = coords / (np.max(coords, axis=0, keepdims=True) + 1e-12)
    
    # Build KD-tree for spatial queries
    tree = KDTree(coords)
    
    # Adaptive k based on graph density estimate
    avg_degree = min(20, max(5, int(np.log2(n)) * 2))
    k = min(avg_degree + 5, n - 1)
    
    # Find nearest neighbors for each node with distance weighting
    for i in range(n):
        # Query for k+1 neighbors (including self)
        distances, indices = tree.query(coords[i], k=min(k+1, n))
        
        # Exclude self
        mask = indices != i
        indices = indices[mask][:k]
        distances = distances[mask][:k]
        
        if len(distances) > 0:
            # Weight by inverse distance with saturation
            weights = 1.0 / (distances + 1e-12)
            max_weight = np.max(weights)
            if max_weight > 0:
                weights = weights / max_weight
            # Apply non-linear scaling
            weights = np.sqrt(weights)  # Less extreme weighting
            
            spatial_promising[i, indices] = weights
    
    # Make symmetric
    spatial_promising = np.maximum(sparsified, spatial_promising.T)
    return spatial_promising

def _compute_mst_promises_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute promises based on MST and extended neighborhood."""
    n = distance_matrix.shape[0]
    
    # Compute MST
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    # Make symmetric
    mst_promising = mst_edges + mst_edges.T
    
    # Boost edges that are in the k-nearest neighborhood of MST nodes
    for i in range(n):
        # Get MST neighbors
        mst_neighbors = np.where(mst_edges[i] > 0)[0]
        
        # For each MST neighbor, consider its own nearest neighbors
        for j in mst_neighbors:
            # Find k nearest neighbors of j (excluding i and itself)
            k = min(5, n - 2)
            dists = distance_matrix[j].copy()
            dists[j] = np.inf
            dists[i] = np.inf
            nearest_j = np.argsort(dists)[:k]
            
            # Boost connection from i to j's nearest neighbors
            for neighbor in nearest_j:
                if mst_promising[i, neighbor] == 0:
                    mst_promising[i, neighbor] = 0.3
    
    # Additional boost for very short edges not in MST
    for i in range(n):
        # Consider top 3 shortest edges not already boosted
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Zero out already boosted edges
        dists[mst_promising[i] > 0] = np.inf
        shortest = np.argsort(dists)[:3]
        for j in shortest:
            if dists[j] < np.inf:
                rel_dist = dists[j] / np.mean(distance_matrix[i, distance_matrix[i] < np.inf])
                if rel_dist < 1.5:  # Only boost if significantly shorter than average
                    mst_promising[i, j] = 0.2 * (1.5 - rel_dist)
    
    return mst_promising

def _compute_local_factors_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute local connectivity factors with improved statistics."""
    # Adaptive neighborhood size
    m = min(10, max(3, n // 20))
    
    # Precompute local statistics
    local_stats = np.zeros((n, 3))  # [mean, std, median]
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if n > m:
            # Get m smallest distances
            sorted_dists = np.partition(dists, m)[:m]
        else:
            sorted_dists = dists[dists < np.inf]
        
        if len(sorted_dists) > 0:
            local_stats[i, 0] = np.mean(sorted_dists)
            local_stats[i, 1] = np.std(sorted_dists) + 1e-12
            local_stats[i, 2] = np.median(sorted_dists)
        else:
            local_stats[i, 0] = 1.0
            local_stats[i, 1] = 1.0
            local_stats[i, 2] = 1.0
    
    # Compute local factor matrix efficiently
    local_factors = np.ones_like(distance_matrix)
    
    # Vectorized computation for larger matrices
    if n > 100:
        # Use broadcasting for efficiency
        mean_i = local_stats[:, 0].reshape(-1, 1)
        mean_j = local_stats[:, 0].reshape(1, -1)
        std_i = local_stats[:, 1].reshape(-1, 1)
        std_j = local_stats[:, 1].reshape(1, -1)
        
        # Stat similarity (Gaussian kernel)
        mean_diff = np.abs(mean_i - mean_j)
        std_sum = std_i + std_j
        stat_similarity = np.exp(-mean_diff / (std_sum + 1e-12))
        
        # Distance ratio factor
        mean_dist = (mean_i + mean_j) / 2
        dist_ratio = distance_matrix / (mean_dist + 1e-12)
        
        # Combined factor with distance penalty
        local_factors = stat_similarity * np.exp(-0.3 * dist_ratio)
        
        # Set diagonal to 0
        np.fill_diagonal(local_factors, 0)
    else:
        # Direct computation for smaller matrices
        for i in range(n):
            for j in range(i+1, n):
                # Similarity based on local statistics
                stat_similarity = np.exp(
                    -np.abs(local_stats[i, 0] - local_stats[j, 0]) /
                    (local_stats[i, 1] + local_stats[j, 1])
                )
                
                # Distance relative to local means
                mean_dist = (local_stats[i, 0] + local_stats[j, 0]) / 2
                dist_ratio = distance_matrix[i, j] / (mean_dist + 1e-12)
                
                # Combine with non-linear penalty
                local_factors[i, j] = local_factors[j, i] = (
                    stat_similarity * np.exp(-0.4 * dist_ratio**0.8)
                )
    
    return local_factors

def _compute_path_potential(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute potential for edges that form good 2-hop paths."""
    path_potential = np.zeros_like(distance_matrix)
    
    if n > 500:
        # Use sampling for large graphs
        sample_size = min(200, n)
        sampled = np.random.choice(n, sample_size, replace=False)
        submatrix = distance_matrix[sampled][:, sampled]
        n_sampled = len(sampled)
        
        # Compute 2-hop distances for sampled nodes
        for i_idx, i in enumerate(sampled):
            for j_idx, j in enumerate(sampled[i_idx+1:], i_idx+1):
                j_actual = sampled[j_idx]
                # Find best intermediate node
                intermediate_dists = distance_matrix[i] + distance_matrix[:, j_actual]
                intermediate_dists[i] = np.inf
                intermediate_dists[j_actual] = np.inf
                best_intermediate = np.min(intermediate_dists)
                
                # If direct edge is shorter than best 2-hop path, boost it
                direct = distance_matrix[i, j_actual]
                if direct < best_intermediate * 0.9:  # 10% better
                    path_potential[i, j_actual] = path_potential[j_actual, i] = 1.2
                elif direct < best_intermediate * 1.1:  # Comparable
                    path_potential[i, j_actual] = path_potential[j_actual, i] = 1.0
    else:
        # Full computation for smaller graphs
        for i in range(n):
            for j in range(i+1, n):
                # Find best intermediate node
                intermediate_dists = distance_matrix[i] + distance_matrix[:, j]
                intermediate_dists[i] = np.inf
                intermediate_dists[j] = np.inf
                best_intermediate = np.min(intermediate_dists)
                
                # Boost factor based on how much better direct edge is
                direct = distance_matrix[i, j]
                if best_intermediate > 0:
                    ratio = direct / best_intermediate
                    if ratio < 0.8:
                        path_potential[i, j] = path_potential[j, i] = 1.5
                    elif ratio < 1.0:
                        path_potential[i, j] = path_potential[j, i] = 1.2
                    elif ratio < 1.2:
                        path_potential[i, j] = path_potential[j, i] = 1.0
    
    return path_potential

def _combine_factors_v2(reciprocal_weights, spatial_promising, mst_promising,
                        local_factors, path_potential, distance_matrix, n):
    """Combine all heuristic factors with adaptive weighting."""
    heuristics_matrix = np.zeros_like(reciprocal_weights)
    
    # Base scores from reciprocal weights (with non-linear scaling)
    base_scores = np.sqrt(reciprocal_weights)  # Less extreme than raw reciprocal
    
    # Adaptive factor weights based on graph size and density
    if n < 50:
        spatial_weight = 1.2
        mst_weight = 1.5
        local_weight = 1.0
        path_weight = 0.8
    elif n < 200:
        spatial_weight = 1.5
        mst_weight = 1.2
        local_weight = 0.9
        path_weight = 1.0
    else:
        spatial_weight = 1.8
        mst_weight = 1.0
        local_weight = 0.7
        path_weight = 1.1
    
    # Apply spatial factor with distance-based decay
    spatial_boost = np.where(spatial_promising > 0,
                            1.0 + spatial_weight * spatial_promising,
                            0.3)  # Penalize non-spatial edges
    
    # Apply MST factor with tiered boosting
    mst_boost = np.ones_like(mst_promising)
    mst_boost = np.where(mst_promising >= 1.0, 1.0 + mst_weight, mst_boost)
    mst_boost = np.where((mst_promising > 0) & (mst_promising < 1.0),
                        1.0 + 0.5 * mst_weight * mst_promising, mst_boost)
    
    # Apply local factors
    local_boost = local_factors ** local_weight
    
    # Apply path potential
    path_boost = np.where(path_potential > 0,
                         path_potential ** path_weight,
                         0.7)  # Penalize edges without path potential
    
    # Combine all factors
    combined = (base_scores * spatial_boost * mst_boost *
                local_boost * path_boost)
    
    # Additional boost for very short edges (with saturation)
    with np.errstate(divide='ignore', invalid='ignore'):
        short_edge_boost = 1.0 + 0.5 * np.tanh(1.0 / (base_scores + 1e-12))
        combined *= short_edge_boost
    
    # Ensure symmetry and clean up
    heuristics_matrix = (combined + combined.T) / 2
    heuristics_matrix = np.nan_to_num(heuristics_matrix, nan=0.0, posinf=0.0)
    
    return heuristics_matrix

def _adaptive_sparsification_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Adaptively sparsify the heuristic matrix with density-aware thresholding."""
    if n <= 15:
        # Keep all edges for very small graphs
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Adaptive sparsity based on graph size
    if n < 100:
        keep_ratio = 0.4
    elif n < 500:
        keep_ratio = 0.25
    elif n < 1000:
        keep_ratio = 0.15
    else:
        keep_ratio = 0.08
    
    # Ensure minimum and maximum edges per node
    min_edges = max(3, n // 20)
    max_edges = min(n - 1, max(10, n // 5))
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Ignore diagonal
        
        # Determine number of edges to keep
        num_to_keep = min(max_edges, max(min_edges, int(keep_ratio * n)))
        
        if np.any(row > 0):
            # Use percentile-based threshold
            positive_vals = row[row > 0]
            if len(positive_vals) > num_to_keep:
                threshold = np.percentile(positive_vals, 100 * (1 - num_to_keep / len(positive_vals)))
            else:
                threshold = 0
            
            # Keep edges above threshold
            mask = row >= threshold
            sparsified[i] = heuristics_matrix[i] * mask
    
    # Ensure symmetry by keeping union of edges
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified

def _triangle_boosting_v2(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Boost scores for edges that form triangles with saturating factor."""
    boosted = heuristics_matrix.copy()
    
    if n > 800:
        # Use approximation with sampling
        sample_size = min(150, n)
        sampled_nodes = np.random.choice(n, sample_size, replace=False)
        
        for i in sampled_nodes:
            neighbors_i = np.where(heuristics_matrix[i] > 0)[0]
            if len(neighbors_i) == 0:
                continue
                
            # Sample neighbors for efficiency
            if len(neighbors_i) > 50:
                sampled_neighbors = np.random.choice(neighbors_i, 50, replace=False)
            else:
                sampled_neighbors = neighbors_i
                
            for j in sampled_neighbors:
                if j <= i:
                    continue
                    
                neighbors_j = np.where(heuristics_matrix[j] > 0)[0]
                # Estimate common neighbors via sampling
                if len(neighbors_j) > 0:
                    # Quick intersection estimate
                    common_est = len(set(neighbors_i[:20]) & set(neighbors_j[:20]))
                    if common_est > 0:
                        # Saturating boost: diminishing returns
                        boost = 1.0 + 0.1 * np.log1p(common_est * 3)
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    else:
        # Exact computation for smaller graphs
        binary_mask = (heuristics_matrix > 0).astype(float)
        
        # Count common neighbors efficiently using matrix multiplication
        common_neighbors = binary_mask @ binary_mask.T
        
        for i in range(n):
            for j in range(i+1, n):
                if heuristics_matrix[i, j] > 0:
                    common = common_neighbors[i, j]
                    if common > 0:
                        # Saturating boost: cap at 5 common neighbors
                        effective_common = min(common, 5)
                        # Logarithmic boost to prevent overemphasis
                        boost = 1.0 + 0.15 * np.log1p(effective_common)
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    
    return boosted
