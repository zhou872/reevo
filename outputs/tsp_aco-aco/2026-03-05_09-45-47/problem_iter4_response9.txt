import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import KDTree
import warnings
from sklearn.manifold import MDS
from sklearn.preprocessing import StandardScaler

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with ML-enhanced factor weighting
    and hierarchical landmark sampling for large instances.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Extract instance features for adaptive weighting
    instance_features = _extract_instance_features(distance_matrix, n)
    
    # Compute adaptive weights for different factors
    factor_weights = _compute_adaptive_weights(instance_features, n)
    
    # Base heuristic: reciprocal of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Hierarchical spatial clustering with optimized MDS for all sizes
    spatial_promising = _compute_hierarchical_spatial_clustering(distance_matrix, n, factor_weights)
    
    # Minimum Spanning Tree with probabilistic extensions
    mst_promising = _compute_enhanced_mst_promises(distance_matrix, n, factor_weights)
    
    # Local connectivity with adaptive neighborhood sizing
    local_factors = _compute_adaptive_local_factors(distance_matrix, n, factor_weights)
    
    # Combine factors using ML-inspired weighting scheme
    heuristics_matrix = _ml_combine_factors(
        reciprocal_weights, spatial_promising, mst_promising, 
        local_factors, distance_matrix, factor_weights
    )
    
    # Dynamic sparsification based on graph properties
    sparsified = _dynamic_sparsification(heuristics_matrix, n, instance_features)
    
    # Probabilistic triangle boosting with sampling
    final_heuristics = _probabilistic_triangle_boosting(sparsified, n, factor_weights)
    
    # Adaptive normalization
    final_heuristics = _adaptive_normalization(final_heuristics, instance_features)
    
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics

def _extract_instance_features(distance_matrix: np.ndarray, n: int) -> dict:
    """Extract key features from the distance matrix for adaptive weighting."""
    features = {}
    
    # Basic statistics
    features['n'] = n
    triu_indices = np.triu_indices(n, 1)
    distances = distance_matrix[triu_indices]
    
    features['mean_distance'] = np.mean(distances)
    features['std_distance'] = np.std(distances)
    features['cv_distance'] = features['std_distance'] / (features['mean_distance'] + 1e-12)
    
    # Spatial distribution features
    features['sparsity'] = np.sum(distances < np.percentile(distances, 10)) / len(distances)
    
    # Graph structure features
    mst = minimum_spanning_tree(distance_matrix)
    mst_distances = mst.data
    features['mst_mean'] = np.mean(mst_distances)
    features['mst_std'] = np.std(mst_distances)
    
    # Compute approximate diameter
    sampled_indices = np.random.choice(n, min(100, n), replace=False)
    max_dist = 0
    for i in sampled_indices:
        max_dist = max(max_dist, np.max(distance_matrix[i]))
    features['diameter_approx'] = max_dist
    
    return features

def _compute_adaptive_weights(features: dict, n: int) -> dict:
    """Compute adaptive weights for different heuristic factors."""
    weights = {}
    
    # Base on instance size
    log_n = np.log1p(n)
    
    # Spatial clustering weight: higher for larger, more clustered instances
    weights['spatial'] = 0.5 + 0.3 * np.tanh(log_n / 10) + 0.2 * features['sparsity']
    
    # MST weight: higher for smaller instances, lower for very large ones
    weights['mst'] = 1.0 - 0.4 * np.tanh(log_n / 15)
    
    # Local factors weight: higher for instances with high variance
    weights['local'] = 0.3 + 0.4 * min(features['cv_distance'], 2.0)
    
    # Reciprocal weight: base factor
    weights['reciprocal'] = 1.0
    
    # Sparsification threshold: adaptive based on instance characteristics
    weights['sparsity_threshold'] = min(0.4, max(0.15, 
        0.25 - 0.1 * np.tanh((n - 100) / 200) + 0.1 * features['sparsity']))
    
    # Triangle boosting strength
    weights['triangle_boost'] = 0.15 + 0.1 * min(features['sparsity'] * 3, 1.0)
    
    return weights

def _compute_hierarchical_spatial_clustering(distance_matrix: np.ndarray, n: int, 
                                            weights: dict) -> np.ndarray:
    """Compute spatial clustering factor with hierarchical landmark sampling."""
    spatial_promising = np.zeros((n, n))
    
    if n <= 500:
        # Use full MDS for small to medium instances
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            n_components = min(5, n - 1)
            mds = MDS(
                n_components=n_components,
                dissimilarity='precomputed',
                random_state=42,
                n_init=1,
                max_iter=100,
                eps=1e-3
            )
            coords = mds.fit_transform(distance_matrix)
    elif n <= 5000:
        # Use landmark-based MDS for medium instances
        landmarks = min(100, n // 10)
        landmark_indices = _select_landmarks(distance_matrix, landmarks)
        
        # Compute MDS on landmarks
        landmark_dists = distance_matrix[landmark_indices][:, landmark_indices]
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(3, landmarks - 1),
                dissimilarity='precomputed',
                random_state=42,
                n_init=1,
                max_iter=50
            )
            landmark_coords = mds.fit_transform(landmark_dists)
        
        # Embed all points using landmark coordinates
        coords = _embed_points(distance_matrix, landmark_indices, landmark_coords)
    else:
        # Use ultra-fast sampling for very large instances
        coords = _fast_approximate_embedding(distance_matrix, n)
    
    # Build KD-tree and find adaptive number of neighbors
    tree = KDTree(coords)
    k = _adaptive_k(n, weights)
    
    # Find nearest neighbors for each node
    for i in range(n):
        distances, indices = tree.query(coords[i], k=k+1)
        # Exclude self
        indices = indices[1:] if distances[0] == 0 else indices[:k]
        
        # Weight neighbors by distance rank
        for rank, j in enumerate(indices):
            weight = 1.0 / (1.0 + rank * 0.2)
            spatial_promising[i, j] = weight
    
    # Make symmetric
    spatial_promising = np.maximum(spatial_promising, spatial_promising.T)
    
    return spatial_promising

def _select_landmarks(distance_matrix: np.ndarray, k: int) -> np.ndarray:
    """Select diverse landmarks using max-min sampling."""
    n = distance_matrix.shape[0]
    landmarks = np.zeros(k, dtype=int)
    
    # Start with a random point
    landmarks[0] = np.random.randint(n)
    
    for i in range(1, k):
        # For each point, compute minimum distance to existing landmarks
        min_dists = np.full(n, np.inf)
        for j in range(i):
            min_dists = np.minimum(min_dists, distance_matrix[landmarks[j]])
        
        # Select point with maximum minimum distance
        landmarks[i] = np.argmax(min_dists)
    
    return landmarks

def _embed_points(distance_matrix: np.ndarray, landmark_indices: np.ndarray, 
                 landmark_coords: np.ndarray) -> np.ndarray:
    """Embed all points using landmark coordinates."""
    n = distance_matrix.shape[0]
    k = len(landmark_indices)
    d = landmark_coords.shape[1]
    
    coords = np.zeros((n, d))
    coords[landmark_indices] = landmark_coords
    
    # For non-landmark points, use weighted average of nearest landmarks
    for i in range(n):
        if i in landmark_indices:
            continue
        
        # Find 3 closest landmarks
        landmark_dists = distance_matrix[i, landmark_indices]
        closest = np.argpartition(landmark_dists, 3)[:3]
        
        # Weight by inverse distance
        weights = 1.0 / (landmark_dists[closest] + 1e-12)
        weights = weights / np.sum(weights)
        
        # Weighted average of landmark coordinates
        coords[i] = np.sum(landmark_coords[closest] * weights[:, np.newaxis], axis=0)
    
    return coords

def _fast_approximate_embedding(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Ultra-fast embedding for very large instances."""
    # Sample columns and use them as features
    sample_size = min(20, n // 100)
    sample_indices = np.random.choice(n, sample_size, replace=False)
    
    # Use distances to sampled nodes as coordinates
    coords = distance_matrix[:, sample_indices]
    
    # Normalize columns
    scaler = StandardScaler()
    coords = scaler.fit_transform(coords)
    
    return coords

def _adaptive_k(n: int, weights: dict) -> int:
    """Compute adaptive number of neighbors based on instance size and weights."""
    base_k = max(3, int(np.log2(n)) + 2)
    
    # Adjust based on spatial weight
    spatial_weight = weights.get('spatial', 1.0)
    adjusted_k = int(base_k * (0.7 + 0.3 * spatial_weight))
    
    return min(adjusted_k, n - 1)

def _compute_enhanced_mst_promises(distance_matrix: np.ndarray, n: int, 
                                  weights: dict) -> np.ndarray:
    """Compute MST promises with probabilistic extensions."""
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    # Base MST promises
    mst_promising = mst_edges + mst_edges.T
    
    # Add probabilistic extensions based on local structure
    mst_weight = weights.get('mst', 1.0)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Adaptive number of extensions based on MST weight
        k_ext = max(2, int(3 * mst_weight))
        closest = np.argsort(dists)[:k_ext]
        
        for j in closest:
            if mst_promising[i, j] == 0:
                # Probability decreases with distance rank
                rank = np.where(closest == j)[0][0]
                prob = 0.5 * (1.0 - rank / k_ext) * mst_weight
                mst_promising[i, j] = prob
                mst_promising[j, i] = prob
    
    return mst_promising

def _compute_adaptive_local_factors(distance_matrix: np.ndarray, n: int, 
                                   weights: dict) -> np.ndarray:
    """Compute local connectivity factors with adaptive neighborhood sizing."""
    # Adaptive neighborhood size based on instance features
    local_weight = weights.get('local', 1.0)
    m = max(3, min(n - 1, int(5 * local_weight)))
    
    local_stats = np.zeros((n, 2))  # [mean_dist, std_dist]
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)[:m]
        local_stats[i, 0] = np.mean(sorted_dists)
        local_stats[i, 1] = np.std(sorted_dists) + 1e-12
    
    # Vectorized computation of local factors
    i_idx, j_idx = np.triu_indices(n, 1)
    
    stat_similarity = np.exp(
        -np.abs(local_stats[i_idx, 0] - local_stats[j_idx, 0]) / 
        (local_stats[i_idx, 1] + local_stats[j_idx, 1])
    )
    
    mean_dist = (local_stats[i_idx, 0] + local_stats[j_idx, 0]) / 2
    dist_ratio = distance_matrix[i_idx, j_idx] / (mean_dist + 1e-12)
    
    local_values = stat_similarity * np.exp(-0.5 * dist_ratio**2)
    
    # Create symmetric matrix
    local_factors = np.ones((n, n))
    local_factors[i_idx, j_idx] = local_values
    local_factors[j_idx, i_idx] = local_values
    
    return local_factors

def _ml_combine_factors(reciprocal_weights: np.ndarray, 
                        spatial_promising: np.ndarray,
                        mst_promising: np.ndarray,
                        local_factors: np.ndarray,
                        distance_matrix: np.ndarray,
                        weights: dict) -> np.ndarray:
    """Combine factors using adaptive, ML-inspired weighting scheme."""
    n = reciprocal_weights.shape[0]
    
    # Extract weights
    w_spatial = weights.get('spatial', 1.0)
    w_mst = weights.get('mst', 1.0)
    w_local = weights.get('local', 1.0)
    w_reciprocal = weights.get('reciprocal', 1.0)
    
    # Apply weighted combination
    # Spatial factor with adaptive boosting
    spatial_boost = 1.0 + (w_spatial - 0.5)
    spatial_factor = np.where(spatial_promising > 0, 
                             spatial_boost + spatial_promising * 0.5,
                             0.5)
    
    # MST factor with adaptive strength
    mst_strength = 1.0 + w_mst
    mst_factor = np.where(mst_promising > 0,
                         np.where(mst_promising >= 1, 
                                 mst_strength * 1.5,  # Strong MST edges
                                 1.0 + mst_promising),  # Weaker extensions
                         0.8)
    
    # Local factors with adaptive influence
    local_influence = 0.5 + w_local * 0.5
    local_adjusted = 1.0 + (local_factors - 1.0) * local_influence
    
    # Combine all factors
    combined = (reciprocal_weights ** w_reciprocal) * \
               (spatial_factor ** w_spatial) * \
               (mst_factor ** w_mst) * \
               (local_adjusted ** w_local)
    
    # Additional adaptive short-edge boost
    with np.errstate(divide='ignore', invalid='ignore'):
        short_edge_boost = 1.0 + 0.2 * np.exp(-distance_matrix / np.percentile(distance_matrix, 10))
        short_edge_boost = np.nan_to_num(short_edge_boost, nan=1.0)
    
    combined = combined * short_edge_boost
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    return combined

def _dynamic_sparsification(heuristics_matrix: np.ndarray, n: int, 
                           features: dict) -> np.ndarray:
    """Dynamic sparsification based on graph properties."""
    if n <= 10:
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Adaptive sparsity based on multiple features
    base_ratio = features.get('sparsity', 0.25)
    mst_density = min(1.0, features.get('mst_std', 1.0) / (features.get('mst_mean', 1.0) + 1e-12))
    
    # Adjust ratio based on features
    keep_ratio = min(0.4, max(0.15, 
        base_ratio * (1.0 + 0.3 * mst_density - 0.2 * (n / 1000))))
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        # Adaptive number of edges to keep
        num_to_keep = max(3, int(keep_ratio * n))
        num_to_keep = min(num_to_keep, n - 1)
        
        if np.any(row > 0):
            # Use soft threshold: keep top k + any within 20% of the k-th
            top_indices = np.argpartition(-row, num_to_keep)[:num_to_keep]
            threshold = np.min(row[top_indices]) * 0.8
            
            mask = row >= threshold
            sparsified[i] = heuristics_matrix[i] * mask
    
    # Ensure symmetry with weighted maximum
    sparsified = np.maximum(sparsified, sparsified.T * 0.8)
    
    return sparsified

def _probabilistic_triangle_boosting(heuristics_matrix: np.ndarray, n: int,
                                    weights: dict) -> np.ndarray:
    """Probabilistic triangle boosting with adaptive sampling."""
    boosted = heuristics_matrix.copy()
    
    if n > 1000:
        # Use stochastic sampling for large graphs
        sample_rate = min(0.3, 1000 / n)
        sample_size = max(100, int(n * sample_rate))
        
        if sample_size < n:
            sampled_nodes = np.random.choice(n, sample_size, replace=False)
            
            for i in sampled_nodes:
                neighbors_i = set(np.where(heuristics_matrix[i] > 0)[0])
                for j in neighbors_i:
                    if j > i and j in sampled_nodes:
                        neighbors_j = set(np.where(heuristics_matrix[j] > 0)[0])
                        common = len(neighbors_i & neighbors_j)
                        if common > 0:
                            boost_strength = weights.get('triangle_boost', 0.15)
                            boost = 1.0 + boost_strength * min(common, 5)
                            boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    else:
        # Exact computation for smaller graphs
        binary_mask = (heuristics_matrix > 0).astype(float)
        
        # Count common neighbors using matrix multiplication
        common_neighbors = binary_mask @ binary_mask.T
        
        i_idx, j_idx = np.triu_indices(n, 1)
        mask_nonzero = heuristics_matrix[i_idx, j_idx] > 0
        
        if np.any(mask_nonzero):
            valid_i = i_idx[mask_nonzero]
            valid_j = j_idx[mask_nonzero]
            common = common_neighbors[valid_i, valid_j]
            
            boost_strength = weights.get('triangle_boost', 0.15)
            boosts = 1.0 + boost_strength * np.minimum(common, 5)
            
            boosted[valid_i, valid_j] = heuristics_matrix[valid_i, valid_j] * boosts
            boosted[valid_j, valid_i] = boosted[valid_i, valid_j]
    
    return boosted

def _adaptive_normalization(heuristics_matrix: np.ndarray, features: dict) -> np.ndarray:
    """Adaptive normalization based on instance characteristics."""
    n = heuristics_matrix.shape[0]
    
    # Remove diagonal
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Adaptive normalization strategy
    if features.get('cv_distance', 1.0) > 1.5:
        # High variance: use robust normalization
        q75 = np.percentile(heuristics_matrix, 75)
        if q75 > 0:
            heuristics_matrix = heuristics_matrix / q75
            heuristics_matrix = np.clip(heuristics_matrix, 0, 1)
    else:
        # Standard normalization
        max_val = np.max(heuristics_matrix)
        if max_val > 0:
            heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
