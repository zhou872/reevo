import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Efficient heuristic for TSP edge selection with reciprocal distance base,
    MST edge boosting, adaptive sparsification, and connectivity assurance.
    
    Features:
    1. Simple reciprocal distance baseline
    2. MST edges boosted for connectivity guarantee
    3. Adaptive neighbor selection based on problem size
    4. Intelligent sparsification to focus on promising edges
    5. Connectivity repair for isolated nodes
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric n x n matrix of pairwise distances
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate more promising edges.
        Diagonal is zero, values are normalized to [0, 1].
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(heuristic, 0)
        max_val = np.max(heuristic)
        if max_val > 0:
            heuristic /= max_val
        return heuristic
    
    # 1. Base heuristic: reciprocal of distances, normalized
    with np.errstate(divide='ignore', invalid='ignore'):
        base_heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(base_heuristic, 0)
    max_val = np.max(base_heuristic)
    if max_val > 0:
        base_heuristic /= max_val
    
    # 2. Identify MST edges for essential connectivity
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # 3. Adaptive neighbor selection with size-dependent parameters
    if n < 30:
        k_neighbors = max(3, n - 1)  # Small: keep most connections
    elif n < 100:
        k_neighbors = max(4, int(np.log2(n) * 3))
    elif n < 500:
        k_neighbors = max(4, int(np.log2(n) * 2))
    else:
        k_neighbors = max(4, int(np.log2(n) * 1.5))  # Large: focus on strongest
    
    # Create neighbor mask based on k nearest neighbors
    neighbor_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest = np.argsort(dists)[:k_neighbors]
        neighbor_mask[i, nearest] = True
    
    # Ensure symmetry
    neighbor_mask = neighbor_mask | neighbor_mask.T
    
    # 4. Combine factors with targeted boosting
    heuristic = base_heuristic.copy()
    heuristic[mst_mask] *= 4.0   # Strong boost for MST edges
    heuristic[neighbor_mask] *= 2.0  # Boost for promising neighbors
    heuristic = (heuristic + heuristic.T) / 2  # Enforce symmetry
    
    # 5. Adaptive sparsification: keep top edges per node
    # Dynamic target edges based on problem size
    if n < 50:
        target_edges = max(4, min(n - 1, int(np.log2(n) * 4)))
    elif n < 200:
        target_edges = max(4, min(int(np.log2(n) * 3), n - 1))
    else:
        target_edges = max(4, min(int(np.log2(n) * 2.5), n - 1))
    
    sparsified = np.zeros_like(heuristic)
    
    for i in range(n):
        row_values = heuristic[i].copy()
        row_values[i] = -np.inf  # Exclude self
        
        # Always keep MST edges
        mst_neighbors = np.where(mst_mask[i])[0]
        
        # Determine how many additional edges to keep
        additional_needed = max(0, target_edges - len(mst_neighbors))
        
        if additional_needed > 0:
            # Find promising non-MST edges
            non_mst_mask = np.ones(n, dtype=bool)
            non_mst_mask[mst_neighbors] = False
            non_mst_mask[i] = False
            non_mst_indices = np.where(non_mst_mask)[0]
            
            if len(non_mst_indices) > 0:
                non_mst_values = row_values[non_mst_indices]
                if len(non_mst_indices) > additional_needed:
                    # Select top additional_needed edges by heuristic value
                    threshold_idx = np.argpartition(-non_mst_values, additional_needed - 1)[additional_needed - 1]
                    threshold_val = non_mst_values[threshold_idx]
                    keep_non_mst = non_mst_indices[non_mst_values >= threshold_val]
                else:
                    keep_non_mst = non_mst_indices
            else:
                keep_non_mst = []
        else:
            keep_non_mst = []
        
        # Combine MST and selected non-MST edges
        keep_indices = np.concatenate([mst_neighbors, keep_non_mst])
        sparsified[i, keep_indices] = heuristic[i, keep_indices]
    
    # Symmetrize by taking element-wise maximum
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 6. Connectivity assurance: repair isolated or poorly connected nodes
    # Ensure every node has at least 2 connections for tour feasibility
    for i in range(n):
        connections = np.where(sparsified[i] > 0)[0]
        degree = len(connections)
        
        if degree == 0:
            # Connect isolated node to its 2 nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:2]
            sparsified[i, nearest] = sparsified[nearest, i] = 0.3
        elif degree == 1:
            # Add connection to second nearest (not the current neighbor)
            current_neighbor = connections[0]
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            dists[current_neighbor] = np.inf
            second_nearest = np.argmin(dists)
            sparsified[i, second_nearest] = sparsified[second_nearest, i] = 0.2
    
    # 7. Final normalization
    np.fill_diagonal(sparsified, 0)
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified /= max_val
    
    return sparsified
