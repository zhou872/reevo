import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import KDTree
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with hierarchical multi-strategy learning.
    
    Key innovations:
    1. Hierarchical spatial analysis with adaptive resolution
    2. Online weight adaptation during heuristic computation
    3. Machine-learning inspired feature engineering from graph properties
    4. Connectivity-preserving sparsification with path augmentation
    5. Multi-resolution triangle analysis
    
    Args:
        distance_matrix: Square symmetric distance matrix (n x n)
        
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # Phase 1: Multi-resolution spatial analysis
    spatial_features = _hierarchical_spatial_analysis(distance_matrix, n)
    
    # Phase 2: Graph structure analysis
    structural_features = _compute_structural_features(distance_matrix, n)
    
    # Phase 3: Online adaptive weight learning
    weights = _online_weight_learning(
        distance_matrix, spatial_features, structural_features, n
    )
    
    # Phase 4: Feature integration with learned weights
    heuristics_matrix = _integrate_features_with_feedback(
        distance_matrix, spatial_features, structural_features, weights, n
    )
    
    # Phase 5: Connectivity-aware sparsification
    sparsified = _connectivity_preserving_sparsification(
        heuristics_matrix, distance_matrix, n
    )
    
    # Phase 6: Multi-resolution triangle enhancement
    enhanced = _multi_resolution_triangle_enhancement(sparsified, distance_matrix, n)
    
    # Final normalization
    np.fill_diagonal(enhanced, 0)
    max_val = np.max(enhanced)
    if max_val > 0:
        enhanced = enhanced / max_val
    
    return enhanced


def _hierarchical_spatial_analysis(distance_matrix: np.ndarray, n: int) -> dict:
    """Perform multi-resolution spatial analysis."""
    features = {}
    
    # Level 1: Global structure using fast approximation
    if n <= 300:
        # Use full MDS for small instances
        try:
            from sklearn.manifold import MDS
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                mds = MDS(
                    n_components=min(3, n-1),
                    dissimilarity='precomputed',
                    random_state=42,
                    n_init=1,
                    max_iter=30
                )
                coords = mds.fit_transform(distance_matrix)
                features['coords'] = coords
        except ImportError:
            # Fallback to distance-based approximation
            features['coords'] = _distance_based_coordinates(distance_matrix, n)
    else:
        # Landmark-based approximation for large instances
        features['coords'] = _landmark_based_coordinates(distance_matrix, n)
    
    # Level 2: Local density estimation
    features['density'] = _compute_local_density(distance_matrix, n)
    
    # Level 3: Hierarchical clustering
    features['clusters'] = _hierarchical_clustering(distance_matrix, n)
    
    return features


def _distance_based_coordinates(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Fallback coordinate approximation using distance pivots."""
    n_pivots = min(10, n)
    pivots = np.random.choice(n, n_pivots, replace=False)
    coords = np.zeros((n, n_pivots))
    
    for i in range(n):
        coords[i] = distance_matrix[i, pivots]
    
    # Normalize
    mean_coords = np.mean(coords, axis=0)
    std_coords = np.std(coords, axis=0) + 1e-12
    coords = (coords - mean_coords) / std_coords
    
    return coords


def _landmark_based_coordinates(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Generate coordinates using farthest-point landmark sampling."""
    n_landmarks = min(max(15, int(np.sqrt(n))), 50)
    
    # Farthest-point sampling for landmarks
    landmarks = [np.random.randint(n)]
    for _ in range(1, n_landmarks):
        min_dists = np.min(distance_matrix[:, landmarks], axis=1)
        next_landmark = np.argmax(min_dists)
        landmarks.append(next_landmark)
    landmarks = np.array(landmarks)
    
    # Use distances to landmarks as coordinates
    coords = distance_matrix[:, landmarks]
    
    # Apply PCA-like whitening for better separation
    mean_coords = np.mean(coords, axis=0)
    centered = coords - mean_coords
    
    # Simple whitening using eigenvalue decomposition
    if n_landmarks > 1:
        cov = centered.T @ centered / n
        eigvals, eigvecs = np.linalg.eigh(cov)
        eigvals = np.maximum(eigvals, 1e-12)
        whitening = eigvecs @ np.diag(1.0 / np.sqrt(eigvals)) @ eigvecs.T
        coords = centered @ whitening.T
    else:
        coords = centered / (np.std(centered) + 1e-12)
    
    return coords


def _compute_local_density(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute local density estimates for each node."""
    density = np.zeros(n)
    
    # Adaptive neighborhood size
    k = min(10, max(3, n // 20))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Get k nearest neighbors
        if k < n - 1:
            idx = np.argpartition(dists, k)[:k]
            nn_dists = dists[idx]
        else:
            nn_dists = dists[dists < np.inf]
        
        # Density as inverse of average distance to neighbors
        if len(nn_dists) > 0:
            density[i] = 1.0 / (np.mean(nn_dists) + 1e-12)
    
    # Normalize densities
    max_density = np.max(density)
    if max_density > 0:
        density = density / max_density
    
    return density


def _hierarchical_clustering(distance_matrix: np.ndarray, n: int) -> dict:
    """Perform hierarchical clustering analysis."""
    clusters = {}
    
    if n <= 10:
        # Small instances: single cluster
        clusters['labels'] = np.zeros(n, dtype=int)
        clusters['hierarchy'] = None
        return clusters
    
    # Multi-level clustering
    n_levels = min(3, int(np.log2(n)) - 1)
    cluster_labels = np.zeros((n_levels, n), dtype=int)
    
    # Level 1: Coarse clustering
    n_clusters_1 = max(2, min(10, n // 20))
    cluster_labels[0] = _distance_based_clustering(distance_matrix, n, n_clusters_1)
    
    # Subsequent levels: finer clustering
    for level in range(1, n_levels):
        n_clusters = min(n_clusters_1 * (2 ** level), n // 5)
        if n_clusters > 1:
            cluster_labels[level] = _refine_clustering(
                distance_matrix, cluster_labels[level-1], n_clusters
            )
        else:
            cluster_labels[level] = cluster_labels[level-1]
    
    clusters['labels'] = cluster_labels
    clusters['n_levels'] = n_levels
    
    return clusters


def _distance_based_clustering(distance_matrix: np.ndarray, n: int, n_clusters: int) -> np.ndarray:
    """Simple distance-based clustering."""
    if n_clusters >= n:
        return np.arange(n)
    
    # Initialize with farthest points as centers
    centers = [np.random.randint(n)]
    for _ in range(1, n_clusters):
        min_dists = np.min(distance_matrix[:, centers], axis=1)
        centers.append(np.argmax(min_dists))
    
    # Assign clusters
    labels = np.zeros(n, dtype=int)
    for i in range(n):
        dists_to_centers = distance_matrix[i, centers]
        labels[i] = np.argmin(dists_to_centers)
    
    return labels


def _refine_clustering(distance_matrix: np.ndarray, parent_labels: np.ndarray, n_clusters: int) -> np.ndarray:
    """Refine clustering within parent clusters."""
    unique_parents = np.unique(parent_labels)
    labels = np.zeros_like(parent_labels)
    next_label = 0
    
    for parent in unique_parents:
        mask = parent_labels == parent
        indices = np.where(mask)[0]
        if len(indices) <= 1:
            labels[mask] = next_label
            next_label += 1
            continue
        
        # Cluster within this parent cluster
        sub_dist = distance_matrix[np.ix_(indices, indices)]
        n_sub_clusters = min(n_clusters // len(unique_parents) + 1, len(indices))
        
        if n_sub_clusters == 1:
            labels[mask] = next_label
            next_label += 1
        else:
            sub_labels = _distance_based_clustering(sub_dist, len(indices), n_sub_clusters)
            labels[mask] = sub_labels + next_label
            next_label += np.max(sub_labels) + 1
    
    return labels


def _compute_structural_features(distance_matrix: np.ndarray, n: int) -> dict:
    """Compute graph structural features."""
    features = {}
    
    # Minimum Spanning Tree
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0) | (mst.toarray().T > 0)
    features['mst'] = mst_edges.astype(float)
    
    # Near-MST edges
    k_near = min(5, n - 1)
    near_mst = np.zeros((n, n))
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        closest = np.argsort(dists)[:k_near]
        for j in closest:
            if not mst_edges[i, j]:
                near_mst[i, j] = 0.3
    features['near_mst'] = near_mst
    
    # Degree-based features
    features['degree_potential'] = _compute_degree_potential(distance_matrix, n)
    
    return features


def _compute_degree_potential(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute degree potential for each node."""
    degree_pot = np.zeros(n)
    
    # Estimate optimal degree for TSP (approximately 2 for each node in optimal tour)
    # But we want to identify nodes that might need more connections
    base_degree = 2.0
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Consider local density
        k = min(10, n - 1)
        if k > 0:
            nn_dists = np.partition(dists, k)[:k]
            avg_nn_dist = np.mean(nn_dists)
            std_nn_dist = np.std(nn_dists) + 1e-12
            
            # Nodes in dense regions might need more careful connection
            density_factor = 1.0 / (avg_nn_dist + 1e-12)
            degree_pot[i] = base_degree * (1 + 0.3 * density_factor / (1 + density_factor))
        else:
            degree_pot[i] = base_degree
    
    return degree_pot


def _online_weight_learning(distance_matrix: np.ndarray,
                           spatial_features: dict,
                           structural_features: dict,
                           n: int) -> dict:
    """Learn adaptive weights during computation."""
    weights = {}
    
    # Analyze instance characteristics
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    avg_dist = np.mean(distances)
    std_dist = np.std(distances)
    cv = std_dist / (avg_dist + 1e-12)  # Coefficient of variation
    
    # Skewness of distances
    if len(distances) > 1:
        skewness = np.mean((distances - avg_dist) ** 3) / (std_dist ** 3 + 1e-12)
    else:
        skewness = 0
    
    # Base reciprocal weight
    weights['reciprocal'] = 1.0
    
    # Spatial weight based on structure clarity
    if 'coords' in spatial_features:
        coords = spatial_features['coords']
        # Measure spatial spread
        coord_std = np.std(coords, axis=0)
        spatial_spread = np.mean(coord_std)
        weights['spatial'] = 0.8 + 0.4 * (1 - cv) * (1 + spatial_spread)
    else:
        weights['spatial'] = 0.8 + 0.4 * (1 - cv)
    
    # MST weight: crucial for connectivity
    mst_weight = 2.0 - 0.5 * np.log10(max(n, 10)) / np.log10(100)
    weights['mst'] = np.clip(mst_weight, 1.5, 2.5)
    
    # Cluster weight based on hierarchical structure
    if 'clusters' in spatial_features and spatial_features['clusters']['n_levels'] > 1:
        cluster_factor = min(1.0, spatial_features['clusters']['n_levels'] / 3.0)
        weights['cluster'] = 0.5 + 0.3 * cluster_factor
    else:
        weights['cluster'] = 0.5
    
    # Density weight
    if 'density' in spatial_features:
        density_var = np.var(spatial_features['density'])
        weights['density'] = 0.3 + 0.4 * min(1.0, density_var * 10)
    else:
        weights['density'] = 0.3
    
    # Distance skewness adjustment
    weights['skew_adjust'] = 1.0 + 0.1 * np.tanh(skewness)
    
    return weights


def _integrate_features_with_feedback(distance_matrix: np.ndarray,
                                     spatial_features: dict,
                                     structural_features: dict,
                                     weights: dict,
                                     n: int) -> np.ndarray:
    """Integrate all features with adaptive weights and feedback."""
    # Initialize with reciprocal distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal, 0)
    
    # Start with reciprocal scores
    scores = weights['reciprocal'] * reciprocal
    
    # Apply spatial factors
    if 'coords' in spatial_features:
        coords = spatial_features['coords']
        tree = KDTree(coords)
        k_spatial = min(15, max(3, int(np.log2(n)) * 2))
        
        spatial_mask = np.zeros((n, n))
        for i in range(n):
            distances, indices = tree.query(coords[i].reshape(1, -1), k=k_spatial + 1)
            indices = indices[0]
            if indices[0] == i:
                indices = indices[1:k_spatial+1]
            else:
                indices = indices[:k_spatial]
            spatial_mask[i, indices] = 1.0
        
        spatial_boost = np.where(spatial_mask > 0, weights['spatial'], weights['spatial'] * 0.5)
        scores *= spatial_boost
    
    # Apply cluster factors
    if 'clusters' in spatial_features:
        cluster_labels = spatial_features['clusters']['labels']
        n_levels = spatial_features['clusters']['n_levels']
        
        for level in range(n_levels):
            labels = cluster_labels[level]
            cluster_mask = np.zeros((n, n))
            
            for i in range(n):
                same_cluster = labels == labels[i]
                cluster_mask[i, same_cluster] = 1.0
            
            # Weight decreases with finer clustering levels
            level_weight = weights['cluster'] * (0.7 ** level)
            cluster_boost = np.where(cluster_mask > 0, 1.0 + level_weight, 1.0)
            scores *= cluster_boost
    
    # Apply density factors
    if 'density' in spatial_features:
        density = spatial_features['density']
        density_matrix = np.zeros((n, n))
        for i in range(n):
            for j in range(i+1, n):
                density_sim = 1.0 - np.abs(density[i] - density[j])
                density_matrix[i, j] = density_matrix[j, i] = density_sim
        
        density_boost = 1.0 + weights['density'] * density_matrix
        scores *= density_boost
    
    # Apply MST factors
    if 'mst' in structural_features:
        mst = structural_features['mst']
        mst_boost = np.where(mst > 0, weights['mst'], 1.0)
        scores *= mst_boost
    
    if 'near_mst' in structural_features:
        near_mst = structural_features['near_mst']
        near_mst_boost = np.where(near_mst > 0, 1.0 + 0.3 * weights['mst'], 1.0)
        scores *= near_mst_boost
    
    # Apply degree potential
    if 'degree_potential' in structural_features:
        degree_pot = structural_features['degree_potential']
        for i in range(n):
            scores[i] *= (1.0 + 0.1 * degree_pot[i])
    
    # Apply skewness adjustment
    scores *= weights['skew_adjust']
    
    # Ensure symmetry
    scores = (scores + scores.T) / 2
    
    return scores


def _connectivity_preserving_sparsification(heuristics_matrix: np.ndarray,
                                           distance_matrix: np.ndarray,
                                           n: int) -> np.ndarray:
    """Sparsify while guaranteeing connectivity and good degree distribution."""
    if n <= 10:
        return heuristics_matrix.copy()
    
    # Compute MST for connectivity guarantee
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Initialize with MST edges
    sparsified = np.zeros_like(heuristics_matrix)
    sparsified[mst_edges] = heuristics_matrix[mst_edges]
    
    # Adaptive target degrees
    base_degree = max(3, int(np.log2(n) * 2))
    
    # Add promising non-MST edges
    for i in range(n):
        # Current degree in sparsified graph
        current_degree = np.count_nonzero(sparsified[i])
        target_degree = min(base_degree, n - 1)
        
        if current_degree < target_degree:
            # Consider non-MST edges sorted by heuristic value
            non_mst_mask = ~mst_edges[i] & (np.arange(n) != i)
            non_mst_indices = np.where(non_mst_mask)[0]
            
            if len(non_mst_indices) > 0:
                non_mst_scores = heuristics_matrix[i, non_mst_indices]
                num_to_add = target_degree - current_degree
                
                if num_to_add > 0:
                    if len(non_mst_indices) > num_to_add:
                        top_idx = np.argpartition(-non_mst_scores, num_to_add)[:num_to_add]
                    else:
                        top_idx = np.arange(len(non_mst_indices))
                    
                    selected = non_mst_indices[top_idx]
                    sparsified[i, selected] = heuristics_matrix[i, selected]
                    sparsified[selected, i] = heuristics_matrix[selected, i]
    
    # Ensure no isolated nodes
    for i in range(n):
        if np.count_nonzero(sparsified[i]) == 0:
            # Connect to nearest neighbor
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argmin(dists)
            sparsified[i, nearest] = sparsified[nearest, i] = 1.0
    
    # Additional connectivity check: ensure graph is not disconnected
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        node = stack.pop()
        neighbors = np.where(sparsified[node] > 0)[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    if not np.all(visited):
        # Add minimum connections to make graph connected
        unvisited = np.where(~visited)[0]
        for node in unvisited:
            # Connect to closest visited node
            dists_to_visited = distance_matrix[node, visited]
            if len(dists_to_visited) > 0:
                closest_visited_idx = np.argmin(dists_to_visited)
                visited_nodes = np.where(visited)[0]
                closest = visited_nodes[closest_visited_idx]
                sparsified[node, closest] = sparsified[closest, node] = 1.0
                visited[node] = True
    
    return sparsified


def _multi_resolution_triangle_enhancement(heuristics_matrix: np.ndarray,
                                          distance_matrix: np.ndarray,
                                          n: int) -> np.ndarray:
    """Enhance scores based on triangle structures at multiple resolutions."""
    enhanced = heuristics_matrix.copy()
    
    if n <= 100:
        # Exact computation for small graphs
        binary = (heuristics_matrix > 0).astype(float)
        common_neighbors = binary @ binary.T
        
        for i in range(n):
            for j in range(i+1, n):
                if heuristics_matrix[i, j] > 0:
                    common = int(common_neighbors[i, j])
                    if common > 0:
                        # Quality-aware boost: consider average heuristic of triangles
                        boost = 1.0 + 0.2 * min(common, 5)
                        enhanced[i, j] = enhanced[j, i] = heuristics_matrix[i, j] * boost
    
    elif n <= 1000:
        # Two-level sampling for medium graphs
        # Level 1: Sample nodes for triangle counting
        sample_size_1 = min(200, n // 2)
        sampled_1 = np.random.choice(n, sample_size_1, replace=False)
        
        # Level 2: Sample edges for detailed analysis
        sample_size_2 = min(100, n // 4)
        sampled_2 = np.random.choice(n, sample_size_2, replace=False)
        
        # Count triangles for sampled nodes
        for i in sampled_1:
            neighbors_i = set(np.where(heuristics_matrix[i] > 0)[0])
            for j in neighbors_i:
                if j > i and j in sampled_1:
                    neighbors_j = set(np.where(heuristics_matrix[j] > 0)[0])
                    common = len(neighbors_i & neighbors_j)
                    if common > 0:
                        boost = 1.0 + 0.15 * min(common, 6)
                        enhanced[i, j] = enhanced[j, i] = heuristics_matrix[i, j] * boost
        
        # Quality analysis for a smaller sample
        for i in sampled_2:
            neighbors_i = np.where(heuristics_matrix[i] > 0)[0]
            for j in neighbors_i:
                if j > i and j in sampled_2:
                    # Find common neighbors
                    neighbors_j = np.where(heuristics_matrix[j] > 0)[0]
                    common = np.intersect1d(neighbors_i, neighbors_j)
                    
                    if len(common) > 0:
                        # Compute average triangle quality
                        triangle_qualities = []
                        for k in common:
                            quality = (heuristics_matrix[i, k] + heuristics_matrix[j, k]) / 2
                            triangle_qualities.append(quality)
                        
                        avg_quality = np.mean(triangle_qualities)
                        quality_factor = avg_quality / (np.mean(heuristics_matrix) + 1e-12)
                        boost = 1.0 + 0.1 * min(len(common), 4) * min(quality_factor, 2.0)
                        enhanced[i, j] = enhanced[j, i] = heuristics_matrix[i, j] * boost
    
    else:
        # Fast approximation for large graphs
        sample_size = min(300, n // 3)
        sampled = np.random.choice(n, sample_size, replace=False)
        
        for i in sampled:
            neighbors_i = set(np.where(heuristics_matrix[i] > 0)[0])
            for j in neighbors_i:
                if j > i and j in sampled:
                    neighbors_j = set(np.where(heuristics_matrix[j] > 0)[0])
                    common = len(neighbors_i & neighbors_j)
                    if common > 0:
                        boost = 1.0 + 0.08 * min(common, 4)
                        enhanced[i, j] = enhanced[j, i] = heuristics_matrix[i, j] * boost
    
    return enhanced
