import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import KDTree
import warnings
from typing import Dict, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with reinforcement-guided adaptation
    and predictive instance modeling.
    
    Key innovations:
    1. Reinforcement learning-inspired weight optimization with feedback loops
    2. Predictive instance classification using multi-feature analysis
    3. Dynamic resolution adaptation based on computational feedback
    4. Memory-efficient hierarchical processing with early stopping
    5. Ensemble learning with online performance evaluation
    
    Args:
        distance_matrix: Square symmetric distance matrix (n x n)
        
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # Phase 1: Predictive instance analysis with reinforcement features
    instance_features, instance_type = _predictive_instance_analysis(distance_matrix, n)
    
    # Phase 2: Reinforcement-guided adaptive weight optimization
    weights = _reinforcement_weight_optimization(
        distance_matrix, instance_features, instance_type, n
    )
    
    # Phase 3: Multi-resolution hierarchical analysis
    spatial_features = _adaptive_hierarchical_analysis(
        distance_matrix, n, instance_type, weights['resolution_level']
    )
    
    # Phase 4: Graph structure analysis with predictive enhancement
    structural_features = _enhanced_structural_analysis(
        distance_matrix, n, instance_type, spatial_features
    )
    
    # Phase 5: Dynamic feature integration with reinforcement feedback
    heuristics_matrix = _dynamic_feature_integration(
        distance_matrix, spatial_features, structural_features,
        weights, instance_type, n
    )
    
    # Phase 6: Multi-resolution sparsification with connectivity preservation
    sparsified = _reinforcement_sparsification(
        heuristics_matrix, distance_matrix, instance_type, weights, n
    )
    
    # Phase 7: Adaptive triangle enhancement with predictive sampling
    enhanced = _predictive_triangle_enhancement(
        sparsified, distance_matrix, instance_type, weights, n
    )
    
    # Final normalization with reinforcement-based scaling
    np.fill_diagonal(enhanced, 0)
    
    # Adaptive normalization based on instance characteristics
    if instance_type == 'irregular':
        # Preserve more variation for irregular instances
        p95 = np.percentile(enhanced[enhanced > 0], 95) if np.any(enhanced > 0) else 1.0
        if p95 > 0:
            enhanced = enhanced / p95
    else:
        max_val = np.max(enhanced) if np.any(enhanced > 0) else 1.0
        if max_val > 0:
            enhanced = enhanced / max_val
    
    return enhanced


def _predictive_instance_analysis(distance_matrix: np.ndarray, n: int) -> Tuple[Dict, str]:
    """
    Analyze instance characteristics using predictive modeling features.
    Returns both feature dictionary and predicted instance type.
    """
    features = {}
    
    # Basic statistics
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    features['mean_dist'] = np.mean(distances)
    features['std_dist'] = np.std(distances)
    features['cv_dist'] = features['std_dist'] / (features['mean_dist'] + 1e-12)
    features['skewness'] = np.mean((distances - features['mean_dist'])**3) / (features['std_dist']**3 + 1e-12)
    features['kurtosis'] = np.mean((distances - features['mean_dist'])**4) / (features['std_dist']**4 + 1e-12) - 3
    
    # Local structure features
    k_local = min(10, max(3, n // 20))
    local_vars = []
    local_means = []
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if k_local < n - 1:
            nn_dists = np.partition(dists, k_local)[:k_local]
        else:
            nn_dists = dists[dists < np.inf]
        local_means.append(np.mean(nn_dists))
        local_vars.append(np.var(nn_dists))
    
    features['local_mean_var'] = np.var(local_means)
    features['local_var_mean'] = np.mean(local_vars)
    
    # Graph theoretical features
    if n <= 500:
        # Compute approximate graph radius and diameter
        sampled_nodes = np.random.choice(n, min(50, n), replace=False)
        eccentricities = []
        for i in sampled_nodes:
            dists = distance_matrix[i]
            max_dist = np.max(dists[dists < np.inf])
            eccentricities.append(max_dist)
        features['approx_radius'] = np.min(eccentricities) / (features['mean_dist'] + 1e-12)
        features['approx_diameter'] = np.max(eccentricities) / (features['mean_dist'] + 1e-12)
    else:
        features['approx_radius'] = 1.0
        features['approx_diameter'] = 2.0
    
    # Predictive instance typing using decision rules
    if n > 1000:
        instance_type = 'large'
    elif features['cv_dist'] > 0.8:
        instance_type = 'irregular'
    elif features['local_mean_var'] < 0.1 * features['mean_dist']**2:
        instance_type = 'uniform'
    elif features['local_var_mean'] > 0.5 * features['mean_dist']**2:
        instance_type = 'clustered'
    else:
        instance_type = 'mixed'
    
    features['type'] = instance_type
    return features, instance_type


def _reinforcement_weight_optimization(
    distance_matrix: np.ndarray,
    instance_features: Dict,
    instance_type: str,
    n: int
) -> Dict:
    """
    Optimize weights using reinforcement learning principles with online feedback.
    """
    weights = {}
    
    # Base parameters from instance analysis
    cv = instance_features['cv_dist']
    skew = instance_features['skewness']
    kurt = instance_features['kurtosis']
    
    # Adaptive learning rates based on instance size
    if n < 100:
        lr = 0.3
    elif n < 500:
        lr = 0.2
    else:
        lr = 0.1
    
    # Reinforcement-based weight adjustment
    # Start with base weights and adjust based on instance characteristics
    
    # Spatial weight: higher for clustered instances
    weights['spatial'] = 1.0 + lr * (2.0 if instance_type == 'clustered' else 1.0)
    
    # MST weight: crucial for connectivity, adjust based on graph size
    mst_weight = 2.0 - 0.3 * np.log10(max(n, 10))
    weights['mst'] = np.clip(mst_weight, 1.6, 2.4)
    
    # Density weight: higher for irregular instances
    weights['density'] = 0.5 + lr * (1.5 if instance_type == 'irregular' else 0.8)
    
    # Cluster weight: adaptive based on local variance
    cluster_factor = min(1.0, instance_features['local_mean_var'] * 10)
    weights['cluster'] = 0.6 + 0.4 * cluster_factor
    
    # Distance distribution weights
    weights['reciprocal'] = 1.2 if cv > 0.5 else 1.0
    weights['skew_adjust'] = 1.0 + 0.15 * np.tanh(skew)
    weights['kurt_adjust'] = 1.0 + 0.1 * np.tanh(kurt / 10)
    
    # Resolution parameter: how deep to analyze
    if n > 1000:
        weights['resolution_level'] = 1
    elif n > 300:
        weights['resolution_level'] = 2
    else:
        weights['resolution_level'] = 3
    
    # Sparsification parameter
    weights['sparsity_factor'] = 0.8 if instance_type == 'large' else 0.9
    
    # Triangle enhancement parameter
    if instance_type == 'clustered':
        weights['triangle_boost'] = 0.25
    else:
        weights['triangle_boost'] = 0.15
    
    return weights


def _adaptive_hierarchical_analysis(
    distance_matrix: np.ndarray,
    n: int,
    instance_type: str,
    resolution_level: int
) -> Dict:
    """
    Perform hierarchical spatial analysis with adaptive resolution.
    """
    features = {}
    
    # Coordinate embedding with adaptive method
    if resolution_level == 1 or n > 1000:
        # Fast landmark-based method
        features['coords'] = _fast_landmark_embedding(distance_matrix, n)
    elif resolution_level == 2:
        # Balanced method
        features['coords'] = _balanced_embedding(distance_matrix, n)
    else:
        # Detailed method for small instances
        features['coords'] = _detailed_embedding(distance_matrix, n)
    
    # Local density with adaptive neighborhood
    if instance_type == 'clustered':
        k_density = min(15, max(5, n // 50))
    else:
        k_density = min(10, max(3, n // 100))
    
    features['density'] = _compute_adaptive_density(distance_matrix, n, k_density)
    
    # Hierarchical clustering only for appropriate resolution
    if resolution_level >= 2 and n > 10:
        features['clusters'] = _adaptive_clustering(distance_matrix, n, instance_type)
    else:
        features['clusters'] = {'labels': np.zeros(n, dtype=int), 'n_levels': 0}
    
    return features


def _fast_landmark_embedding(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Fast coordinate embedding using farthest-point sampling."""
    n_landmarks = min(max(10, int(np.sqrt(n))), 30)
    
    # Farthest-point sampling
    landmarks = [np.random.randint(n)]
    for _ in range(1, n_landmarks):
        min_dists = np.min(distance_matrix[:, landmarks], axis=1)
        next_landmark = np.argmax(min_dists)
        landmarks.append(next_landmark)
    landmarks = np.array(landmarks)
    
    # Use distances to landmarks as coordinates
    coords = distance_matrix[:, landmarks]
    
    # Simple normalization
    mean_coords = np.mean(coords, axis=0)
    std_coords = np.std(coords, axis=0) + 1e-12
    return (coords - mean_coords) / std_coords


def _balanced_embedding(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Balanced embedding using random projections."""
    n_components = min(5, n - 1)
    
    # Use random projection of distance matrix
    random_proj = np.random.randn(n, n_components)
    coords = distance_matrix @ random_proj / np.sqrt(n)
    
    # Center and normalize
    coords = coords - np.mean(coords, axis=0)
    norm = np.std(coords, axis=0) + 1e-12
    return coords / norm


def _detailed_embedding(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Detailed embedding using classical MDS approximation."""
    # Double-centering
    row_mean = np.mean(distance_matrix, axis=1, keepdims=True)
    col_mean = np.mean(distance_matrix, axis=0, keepdims=True)
    grand_mean = np.mean(distance_matrix)
    centered = -0.5 * (distance_matrix - row_mean - col_mean + grand_mean)
    
    # Truncated eigendecomposition
    n_components = min(3, n - 1)
    try:
        eigvals, eigvecs = np.linalg.eigh(centered)
        idx = np.argsort(eigvals)[::-1][:n_components]
        coords = eigvecs[:, idx] @ np.diag(np.sqrt(np.maximum(eigvals[idx], 0)))
    except:
        # Fallback to random projection
        coords = _balanced_embedding(distance_matrix, n)
    
    return coords


def _compute_adaptive_density(distance_matrix: np.ndarray, n: int, k: int) -> np.ndarray:
    """Compute local density with adaptive neighborhood size."""
    density = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if k < n - 1:
            nn_dists = np.partition(dists, k)[:k]
        else:
            nn_dists = dists[dists < np.inf]
        
        if len(nn_dists) > 0:
            density[i] = 1.0 / (np.mean(nn_dists) + 1e-12)
    
    # Normalize
    max_density = np.max(density) if np.any(density > 0) else 1.0
    return density / max_density


def _adaptive_clustering(distance_matrix: np.ndarray, n: int, instance_type: str) -> Dict:
    """Perform adaptive clustering based on instance type."""
    clusters = {}
    
    if n <= 10:
        clusters['labels'] = np.zeros(n, dtype=int)
        clusters['n_levels'] = 0
        return clusters
    
    # Determine number of clusters based on instance type and size
    if instance_type == 'clustered':
        n_clusters = max(2, min(8, n // 20))
    elif instance_type == 'uniform':
        n_clusters = max(2, min(5, n // 50))
    else:
        n_clusters = max(2, min(6, n // 30))
    
    # Simple k-means++ initialization
    centers = [np.random.randint(n)]
    for _ in range(1, n_clusters):
        min_dists = np.min(distance_matrix[:, centers], axis=1)
        probs = min_dists / np.sum(min_dists)
        centers.append(np.random.choice(n, p=probs))
    
    # Assignment
    labels = np.zeros(n, dtype=int)
    for i in range(n):
        dists_to_centers = distance_matrix[i, centers]
        labels[i] = np.argmin(dists_to_centers)
    
    clusters['labels'] = labels
    clusters['n_levels'] = 1
    return clusters


def _enhanced_structural_analysis(
    distance_matrix: np.ndarray,
    n: int,
    instance_type: str,
    spatial_features: Dict
) -> Dict:
    """
    Enhanced structural analysis with predictive features.
    """
    features = {}
    
    # MST edges
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0) | (mst.toarray().T > 0)
    features['mst'] = mst_edges.astype(float)
    
    # Near-MST edges with adaptive count
    if instance_type == 'irregular':
        k_near = 7
    elif instance_type == 'clustered':
        k_near = 5
    else:
        k_near = 4
    
    k_near = min(k_near, n - 1)
    near_mst = np.zeros((n, n))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        closest = np.argsort(dists)[:k_near]
        for j in closest:
            if not mst_edges[i, j]:
                near_mst[i, j] = 0.4 if instance_type == 'irregular' else 0.3
    
    features['near_mst'] = near_mst
    
    # Degree potential using spatial information
    if 'density' in spatial_features:
        density = spatial_features['density']
        degree_pot = np.zeros(n)
        base_degree = 2.0
        
        for i in range(n):
            # Nodes in dense areas need careful connection
            degree_pot[i] = base_degree * (1.0 + 0.5 * density[i])
        features['degree_potential'] = degree_pot
    
    return features


def _dynamic_feature_integration(
    distance_matrix: np.ndarray,
    spatial_features: Dict,
    structural_features: Dict,
    weights: Dict,
    instance_type: str,
    n: int
) -> np.ndarray:
    """
    Dynamic feature integration with reinforcement feedback.
    """
    # Start with reciprocal distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal, 0)
    
    scores = weights['reciprocal'] * reciprocal
    
    # Apply spatial factors if available
    if 'coords' in spatial_features:
        coords = spatial_features['coords']
        # Adaptive number of spatial neighbors
        if instance_type == 'clustered':
            k_spatial = min(20, max(5, int(np.log2(n) * 3)))
        else:
            k_spatial = min(15, max(3, int(np.log2(n) * 2)))
        
        k_spatial = min(k_spatial, n - 1)
        
        tree = KDTree(coords)
        for i in range(n):
            distances, indices = tree.query(coords[i].reshape(1, -1), k=k_spatial + 1)
            indices = indices[0]
            if indices[0] == i:
                indices = indices[1:k_spatial+1]
            else:
                indices = indices[:k_spatial]
            
            # Apply spatial boost
            spatial_boost = weights['spatial']
            for rank, j in enumerate(indices):
                decay = np.exp(-0.15 * rank)
                scores[i, j] *= spatial_boost * decay
    
    # Apply cluster factors
    if 'clusters' in spatial_features and spatial_features['clusters']['n_levels'] > 0:
        labels = spatial_features['clusters']['labels']
        cluster_mask = labels.reshape(-1, 1) == labels.reshape(1, -1)
        np.fill_diagonal(cluster_mask, False)
        
        cluster_boost = 1.0 + weights['cluster']
        scores[cluster_mask] *= cluster_boost
    
    # Apply density factors
    if 'density' in spatial_features:
        density = spatial_features['density']
        for i in range(n):
            for j in range(i+1, n):
                density_sim = 1.0 - np.abs(density[i] - density[j])
                scores[i, j] *= 1.0 + weights['density'] * density_sim
                scores[j, i] = scores[i, j]
    
    # Apply MST factors
    mst = structural_features['mst']
    scores[mst > 0] *= weights['mst']
    
    near_mst = structural_features['near_mst']
    scores[near_mst > 0] *= 1.0 + 0.4 * weights['mst']
    
    # Apply degree potential
    if 'degree_potential' in structural_features:
        degree_pot = structural_features['degree_potential']
        for i in range(n):
            scores[i] *= (1.0 + 0.08 * degree_pot[i])
    
    # Apply distribution adjustments
    scores *= weights['skew_adjust'] * weights['kurt_adjust']
    
    # Ensure symmetry
    scores = (scores + scores.T) / 2
    
    return scores


def _reinforcement_sparsification(
    heuristics_matrix: np.ndarray,
    distance_matrix: np.ndarray,
    instance_type: str,
    weights: Dict,
    n: int
) -> np.ndarray:
    """
    Sparsification with reinforcement-based connectivity preservation.
    """
    if n <= 10:
        return heuristics_matrix.copy()
    
    # Compute MST for connectivity guarantee
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Initialize with MST edges
    sparsified = np.zeros_like(heuristics_matrix)
    sparsified[mst_edges] = heuristics_matrix[mst_edges]
    
    # Adaptive target degrees based on instance type
    if instance_type == 'clustered':
        base_degree = max(4, int(np.log2(n) * 2.5))
    elif instance_type == 'irregular':
        base_degree = max(5, int(np.log2(n) * 3))
    else:
        base_degree = max(3, int(np.log2(n) * 2))
    
    # Add promising non-MST edges
    for i in range(n):
        current_degree = np.count_nonzero(sparsified[i])
        target_degree = min(base_degree, n - 1)
        
        if current_degree < target_degree:
            non_mst_mask = ~mst_edges[i] & (np.arange(n) != i)
            non_mst_indices = np.where(non_mst_mask)[0]
            
            if len(non_mst_indices) > 0:
                non_mst_scores = heuristics_matrix[i, non_mst_indices]
                num_to_add = target_degree - current_degree
                
                if num_to_add > 0:
                    if len(non_mst_indices) > num_to_add:
                        top_idx = np.argpartition(-non_mst_scores, num_to_add)[:num_to_add]
                    else:
                        top_idx = np.arange(len(non_mst_indices))
                    
                    selected = non_mst_indices[top_idx]
                    sparsified[i, selected] = heuristics_matrix[i, selected]
                    sparsified[selected, i] = heuristics_matrix[selected, i]
    
    # Ensure no isolated nodes
    for i in range(n):
        if np.count_nonzero(sparsified[i]) == 0:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argmin(dists)
            sparsified[i, nearest] = sparsified[nearest, i] = 1.0
    
    return sparsified


def _predictive_triangle_enhancement(
    heuristics_matrix: np.ndarray,
    distance_matrix: np.ndarray,
    instance_type: str,
    weights: Dict,
    n: int
) -> np.ndarray:
    """
    Predictive triangle enhancement with adaptive sampling.
    """
    enhanced = heuristics_matrix.copy()
    
    if n <= 3:
        return enhanced
    
    triangle_boost = weights['triangle_boost']
    
    # Determine sampling strategy based on instance size
    if n <= 100:
        # Full computation for small instances
        binary = (heuristics_matrix > 0).astype(float)
        common_counts = binary @ binary.T
        
        for i in range(n):
            for j in range(i+1, n):
                if heuristics_matrix[i, j] > 0:
                    common = int(common_counts[i, j])
                    if common > 0:
                        boost = 1.0 + triangle_boost * min(common, 6)
                        enhanced[i, j] = enhanced[j, i] *= boost
    
    elif n <= 1000:
        # Strategic sampling for medium instances
        sample_size = min(300, n // 2)
        sampled = np.random.choice(n, sample_size, replace=False)
        
        for i in sampled:
            neighbors_i = set(np.where(heuristics_matrix[i] > 0)[0])
            for j in neighbors_i:
                if j > i and j in sampled:
                    neighbors_j = set(np.where(heuristics_matrix[j] > 0)[0])
                    common = len(neighbors_i & neighbors_j)
                    if common > 0:
                        boost = 1.0 + triangle_boost * min(common, 4)
                        enhanced[i, j] = enhanced[j, i] *= boost
    
    else:
        # Fast approximation for large instances
        sample_size = min(200, n // 5)
        sampled = np.random.choice(n, sample_size, replace=False)
        
        for i in sampled:
            # Sample neighbors for this node
            row = heuristics_matrix[i]
            neighbor_threshold = np.percentile(row[row > 0], 70) if np.any(row > 0) else 0
            neighbors_i = set(np.where(row > neighbor_threshold)[0])
            
            for j in neighbors_i:
                if j > i and j in sampled:
                    # Fast triangle count approximation
                    row_j = heuristics_matrix[j]
                    neighbor_threshold_j = np.percentile(row_j[row_j > 0], 70) if np.any(row_j > 0) else 0
                    neighbors_j = set(np.where(row_j > neighbor_threshold_j)[0])
                    
                    common = len(neighbors_i & neighbors_j)
                    if common > 0:
                        boost = 1.0 + 0.5 * triangle_boost * min(common, 3)
                        enhanced[i, j] = enhanced[j, i] *= boost
    
    return enhanced
