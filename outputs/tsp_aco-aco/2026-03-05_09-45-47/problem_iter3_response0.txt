import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection emphasizing efficiency and triangle-based clustering.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # 1. Base heuristic: reciprocal of distances (closer = more promising)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(reciprocal, 0)
    
    # 2. MST-based filtering: boost edges that are in or near MST
    mst_boost = _compute_mst_boost(distance_matrix, n)
    
    # 3. Initial combination with MST boost
    combined = reciprocal * mst_boost
    
    # 4. Adaptive sparsification: keep only top edges per node
    sparsified = _adaptive_sparsify(combined, n)
    
    # 5. Triangle-based boosting for clustering
    final = _triangle_boost(sparsified, n)
    
    # Normalize to reasonable scale and ensure non-negativity
    max_val = np.max(final)
    if max_val > 0:
        final = final / max_val
    
    np.fill_diagonal(final, 0)
    return final

def _compute_mst_boost(dist_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute MST-based edge importance scores."""
    mst_boost = np.ones_like(dist_matrix)
    
    # Compute MST and mark its edges
    mst = minimum_spanning_tree(dist_matrix)
    mst_edges = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Strong boost for MST edges
    mst_boost[mst_edges] = 3.0
    
    # Moderate boost for near-MST edges (2nd/3rd nearest per node)
    for i in range(n):
        # Get distances to other nodes
        dists = dist_matrix[i].copy()
        dists[i] = np.inf
        
        # Find 3 closest nodes (excluding self)
        closest_indices = np.argpartition(dists, 3)[:3]
        
        for j in closest_indices:
            if not mst_edges[i, j]:
                mst_boost[i, j] = max(mst_boost[i, j], 1.5)
    
    # Ensure symmetry
    mst_boost = (mst_boost + mst_boost.T) / 2
    return mst_boost

def _adaptive_sparsify(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """Adaptively keep only the most promising edges per node."""
    sparsified = np.zeros_like(heuristics_matrix)
    
    if n <= 10:
        # Keep all edges for very small graphs
        return heuristics_matrix.copy()
    
    # Adaptive edge count: log2(n) ensures scalability
    k = max(3, min(int(np.log2(n)) + 3, n - 1))
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Exclude self
        
        # Get indices of top k edges
        top_k_indices = np.argpartition(-row, k)[:k]
        sparsified[i, top_k_indices] = heuristics_matrix[i, top_k_indices]
    
    # Ensure symmetry by keeping union of edges
    sparsified = np.maximum(sparsified, sparsified.T)
    return sparsified

def _triangle_boost(sparsified_matrix: np.ndarray, n: int) -> np.ndarray:
    """Boost scores for edges that form triangles (common neighbors)."""
    boosted = sparsified_matrix.copy()
    
    if n > 100:
        # Efficient approximation for larger graphs
        binary_mask = (sparsified_matrix > 0).astype(float)
        
        # Count common neighbors via matrix multiplication
        # This efficiently computes |N(i) ∩ N(j)| for all pairs
        common_neighbors = binary_mask @ binary_mask.T
        
        for i in range(n):
            # Get neighbors of i
            neighbors_i = np.where(binary_mask[i] > 0)[0]
            for j in neighbors_i:
                if j <= i:
                    continue
                if sparsified_matrix[i, j] > 0:
                    common = common_neighbors[i, j]
                    if common > 0:
                        # Boost based on number of common neighbors
                        boost = 1 + 0.15 * min(common, 4)
                        boosted[i, j] = boosted[j, i] = sparsified_matrix[i, j] * boost
    else:
        # Exact computation for smaller graphs
        for i in range(n):
            for j in range(i + 1, n):
                if sparsified_matrix[i, j] > 0:
                    # Find common neighbors between i and j
                    neighbors_i = set(np.where(sparsified_matrix[i] > 0)[0])
                    neighbors_j = set(np.where(sparsified_matrix[j] > 0)[0])
                    common = len(neighbors_i.intersection(neighbors_j))
                    if common > 0:
                        boost = 1 + 0.2 * min(common, 5)
                        boosted[i, j] = boosted[j, i] = sparsified_matrix[i, j] * boost
    
    return boosted
