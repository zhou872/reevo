import numpy as np
import warnings
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.manifold import MDS
from sklearn.neighbors import KDTree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Meta-learning enhanced heuristic for TSP edge selection with adaptive
    parameter tuning based on instance features and efficient large-scale approximations.
    
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # Extract instance features for meta-learning
    instance_features = _extract_instance_features(distance_matrix, n)
    
    # Meta-learn parameters based on instance characteristics
    meta_params = _meta_learn_parameters(instance_features, n)
    
    # Base heuristic: reciprocal of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Adaptive multi-scale spatial clustering
    spatial_promising = _adaptive_spatial_clustering(
        distance_matrix, n, meta_params
    )
    
    # Feature-enhanced MST with distance integration
    mst_promising = _feature_enhanced_mst(distance_matrix, n, meta_params)
    
    # Multi-resolution local statistics
    local_factors = _multi_resolution_local_factors(distance_matrix, n, meta_params)
    
    # Meta-adaptive combination
    heuristics_matrix = _meta_adaptive_combination(
        reciprocal_weights, spatial_promising, mst_promising,
        local_factors, distance_matrix, n, meta_params
    )
    
    # Intelligent instance-aware sparsification
    sparsified = _instance_aware_sparsification(
        heuristics_matrix, distance_matrix, n, meta_params
    )
    
    # Structure-aware triangle boosting
    final_heuristics = _structure_aware_triangle_boosting(
        sparsified, distance_matrix, n, meta_params
    )
    
    # Final normalization
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics


def _extract_instance_features(distance_matrix: np.ndarray, n: int) -> dict:
    """Extract comprehensive instance features for meta-learning."""
    features = {}
    
    # Basic statistics
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    features['n'] = n
    features['mean_distance'] = np.mean(distances)
    features['std_distance'] = np.std(distances)
    features['cv_distance'] = features['std_distance'] / (features['mean_distance'] + 1e-12)
    
    # Distance distribution features
    features['q1_distance'] = np.percentile(distances, 25)
    features['median_distance'] = np.median(distances)
    features['q3_distance'] = np.percentile(distances, 75)
    features['iqr_distance'] = features['q3_distance'] - features['q1_distance']
    features['skewness'] = (features['mean_distance'] - features['median_distance']) / (features['std_distance'] + 1e-12)
    
    # Graph structure features
    features['density_ratio'] = features['q1_distance'] / (features['q3_distance'] + 1e-12)
    
    # Triangle inequality features
    if n >= 3:
        triangle_violations = []
        for i in range(min(n, 100)):  # Sample for efficiency
            for j in range(i+1, min(n, 100)):
                for k in range(j+1, min(n, 100)):
                    d_ij = distance_matrix[i, j]
                    d_ik = distance_matrix[i, k]
                    d_jk = distance_matrix[j, k]
                    if d_ij > d_ik + d_jk:
                        triangle_violations.append(d_ij - (d_ik + d_jk))
        
        if triangle_violations:
            features['triangle_violation_mean'] = np.mean(triangle_violations)
            features['triangle_violation_ratio'] = len(triangle_violations) / (100*99*98/6)
        else:
            features['triangle_violation_mean'] = 0
            features['triangle_violation_ratio'] = 0
    
    # Distance matrix rank approximation
    if n <= 1000:
        try:
            cov_matrix = distance_matrix @ distance_matrix.T
            eigenvalues = np.linalg.eigvalsh(cov_matrix)
            eigenvalues = eigenvalues[eigenvalues > 1e-10]
            features['effective_rank'] = (np.sum(eigenvalues) ** 2) / (np.sum(eigenvalues ** 2) + 1e-12)
        except:
            features['effective_rank'] = n
    else:
        features['effective_rank'] = n
    
    return features


def _meta_learn_parameters(features: dict, n: int) -> dict:
    """Meta-learning parameter adaptation based on instance features."""
    params = {}
    
    # Size-based adaptations
    log_n = np.log10(n + 1)
    params['size_category'] = 'small' if n <= 100 else 'medium' if n <= 1000 else 'large'
    
    # Spatial clustering parameters
    if features['effective_rank'] < n * 0.3:
        params['spatial_strategy'] = 'low_dimensional'
        params['spatial_weight'] = 1.5
    elif features['cv_distance'] > 1.0:
        params['spatial_strategy'] = 'high_variance'
        params['spatial_weight'] = 1.2
    else:
        params['spatial_strategy'] = 'balanced'
        params['spatial_weight'] = 1.0
    
    # MST parameters
    if features['density_ratio'] < 0.3:
        params['mst_weight'] = 1.8  # Higher for sparse graphs
        params['mst_extension_factor'] = 1.5
    else:
        params['mst_weight'] = 1.2
        params['mst_extension_factor'] = 1.0
    
    # Local statistics parameters
    params['local_scale_count'] = min(3, int(np.log2(n + 1)))
    params['local_weight'] = 0.7 + 0.3 * np.tanh(1.0 - features['cv_distance'])
    
    # Combination strategy
    if features['triangle_violation_ratio'] > 0.1:
        params['combination_strategy'] = 'distance_heavy'
        params['distance_weight'] = 1.5
    else:
        params['combination_strategy'] = 'balanced'
        params['distance_weight'] = 1.0
    
    # Sparsification parameters
    if params['size_category'] == 'small':
        params['sparsification_ratio'] = 0.4
    elif params['size_category'] == 'medium':
        params['sparsification_ratio'] = 0.25 + 0.05 * np.tanh(2.0 - features['cv_distance'])
    else:
        params['sparsification_ratio'] = 0.15 + 0.1 * np.tanh(1.0 - features['cv_distance'])
    
    params['sparsification_ratio'] = np.clip(params['sparsification_ratio'], 0.1, 0.5)
    
    # Triangle boosting parameters
    if features['triangle_violation_ratio'] < 0.05:
        params['triangle_boost_strength'] = 0.15
        params['max_triangle_boost'] = 0.3
    else:
        params['triangle_boost_strength'] = 0.08
        params['max_triangle_boost'] = 0.15
    
    return params


def _adaptive_spatial_clustering(distance_matrix: np.ndarray, n: int, params: dict) -> np.ndarray:
    """Adaptive spatial clustering based on instance characteristics."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    strategy = params['spatial_strategy']
    size_category = params['size_category']
    
    if size_category == 'small' or (size_category == 'medium' and strategy != 'low_dimensional'):
        # Use MDS-based clustering for small/medium instances
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            n_components = min(3, n-1) if strategy == 'low_dimensional' else min(2, n-1)
            mds = MDS(
                n_components=n_components,
                dissimilarity='precomputed',
                random_state=42,
                n_init=2,
                max_iter=100,
                eps=1e-3
            )
            coords = mds.fit_transform(distance_matrix)
            tree = KDTree(coords)
            
            # Adaptive neighborhood size
            k_base = max(3, min(int(np.log2(n)) + 2, n - 1))
            if strategy == 'high_variance':
                k_near = k_base
                k_medium = min(k_base * 3, n - 1)
            else:
                k_near = min(k_base * 2, n - 1)
                k_medium = min(k_base * 4, n - 1)
            
            for i in range(n):
                # Near neighbors
                _, near_indices = tree.query(coords[i].reshape(1, -1), k=k_near + 1)
                near_indices = near_indices[0][1:]
                spatial_promising[i, near_indices] = 1.0
                
                # Medium neighbors
                if k_medium > k_near:
                    _, medium_indices = tree.query(coords[i].reshape(1, -1), k=k_medium + 1)
                    medium_indices = medium_indices[0][1:]
                    medium_indices = medium_indices[~np.isin(medium_indices, near_indices)]
                    spatial_promising[i, medium_indices] = 0.4
    
    elif size_category == 'medium' or size_category == 'large':
        # Landmark-based approximation with adaptive landmark selection
        if strategy == 'low_dimensional':
            n_landmarks = min(100, n // 10)
            # Select landmarks using farthest point sampling
            landmarks = [np.random.randint(n)]
            for _ in range(1, n_landmarks):
                dist_to_landmarks = np.min(distance_matrix[:, landmarks], axis=1)
                landmarks.append(np.argmax(dist_to_landmarks))
        else:
            n_landmarks = min(200, n // 5)
            landmarks = np.random.choice(n, n_landmarks, replace=False)
        
        landmark_coords = distance_matrix[:, landmarks]
        k = max(4, int(np.log2(n)) + 1)
        
        for i in range(n):
            # Compute distances in landmark space
            dist_in_landmark_space = np.sqrt(
                np.sum((landmark_coords - landmark_coords[i:i+1]) ** 2, axis=1)
            )
            dist_in_landmark_space[i] = np.inf
            
            indices = np.argpartition(dist_in_landmark_space, k)[:k]
            spatial_promising[i, indices] = 1.0
            
            # Add diversity connections
            if k < n // 2:
                diversity_k = min(k * 2, n - 1)
                diversity_indices = np.argpartition(dist_in_landmark_space, diversity_k)[k:diversity_k]
                spatial_promising[i, diversity_indices] = 0.3
    
    # Ensure symmetry
    spatial_promising = np.maximum(spatial_promising, spatial_promising.T)
    return spatial_promising


def _feature_enhanced_mst(distance_matrix: np.ndarray, n: int, params: dict) -> np.ndarray:
    """MST enhancement with feature-aware extensions."""
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    # Symmetric MST matrix
    mst_promising = mst_edges + mst_edges.T
    
    # Compute local distance features
    local_features = np.zeros((n, 2))  # [median, std]
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        valid_dists = dists[dists < np.inf]
        if len(valid_dists) > 0:
            local_features[i, 0] = np.median(valid_dists)
            local_features[i, 1] = np.std(valid_dists) + 1e-12
    
    # Feature-aware MST extension
    extension_factor = params['mst_extension_factor']
    max_extensions = min(5, int(np.log2(n)) + 1)
    
    for i in range(n):
        mst_neighbors = np.where(mst_promising[i] >= 2)[0]
        
        for neighbor in mst_neighbors:
            d_ij = distance_matrix[i, neighbor]
            
            # Compute feature similarity scores
            feature_similarity = np.zeros(n)
            for j in range(n):
                if j != i and j != neighbor and mst_promising[i, j] == 0:
                    # Distance feature similarity
                    mean_local = (local_features[i, 0] + local_features[neighbor, 0]) / 2
                    std_sum = local_features[i, 1] + local_features[j, 1]
                    
                    # Combined score
                    dist_score = np.exp(-distance_matrix[i, j] / (mean_local + 1e-12))
                    feature_score = np.exp(-np.abs(local_features[i, 0] - local_features[j, 0]) / std_sum)
                    
                    feature_similarity[j] = dist_score * feature_score
            
            # Select promising extensions
            if np.any(feature_similarity > 0):
                n_extensions = min(max_extensions, np.sum(feature_similarity > 0))
                top_indices = np.argpartition(-feature_similarity, n_extensions)[:n_extensions]
                
                for rank, j in enumerate(top_indices):
                    if feature_similarity[j] > 0:
                        weight = extension_factor * (0.5 / (rank + 1) ** 0.5)
                        mst_promising[i, j] = max(mst_promising[i, j], weight)
                        mst_promising[j, i] = max(mst_promising[j, i], weight)
    
    return mst_promising


def _multi_resolution_local_factors(distance_matrix: np.ndarray, n: int, params: dict) -> np.ndarray:
    """Multi-resolution local statistics with adaptive scales."""
    scale_count = params['local_scale_count']
    
    # Adaptive neighborhood sizes
    scales = []
    for s in range(scale_count):
        base_size = max(3, int(np.log2(n)) + 1 - s)
        scales.append(min(base_size * (2 ** s), n - 1))
    
    # Compute statistics for each scale
    scale_stats = []
    for scale_size in scales:
        stats = np.zeros((n, 2))  # [mean, std]
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            if n > scale_size:
                scale_dists = np.partition(dists, scale_size)[:scale_size]
            else:
                scale_dists = dists[dists < np.inf]
            
            stats[i, 0] = np.mean(scale_dists)
            stats[i, 1] = np.std(scale_dists) + 1e-12
        
        scale_stats.append(stats)
    
    # Compute multi-resolution factors
    local_factors = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(i + 1, n):
            scale_contributions = []
            
            for s, stats in enumerate(scale_stats):
                # Scale weight (more weight to finer scales)
                scale_weight = 1.0 / (s + 1)
                
                # Mean similarity
                mean_diff = np.abs(stats[i, 0] - stats[j, 0])
                std_sum = stats[i, 1] + stats[j, 1]
                mean_similarity = np.exp(-mean_diff / std_sum)
                
                # Distance ratio factor
                local_mean = (stats[i, 0] + stats[j, 0]) / 2
                dist_ratio = distance_matrix[i, j] / (local_mean + 1e-12)
                dist_factor = np.exp(-0.5 * np.minimum(dist_ratio, 3) ** 2)
                
                scale_contributions.append(scale_weight * mean_similarity * dist_factor)
            
            # Combine scales
            combined = np.sum(scale_contributions) / np.sum([1.0/(s+1) for s in range(scale_count)])
            local_factors[i, j] = local_factors[j, i] = combined
    
    return local_factors


def _meta_adaptive_combination(reciprocal_weights: np.ndarray,
                               spatial_promising: np.ndarray,
                               mst_promising: np.ndarray,
                               local_factors: np.ndarray,
                               distance_matrix: np.ndarray,
                               n: int,
                               params: dict) -> np.ndarray:
    """Meta-adaptive combination of factors based on instance characteristics."""
    # Base combination with diminishing returns
    base_scores = np.sqrt(reciprocal_weights)
    
    # Apply spatial factor with adaptive weight
    spatial_mask = spatial_promising > 0
    spatial_component = base_scores.copy()
    spatial_component[spatial_mask] *= (1.0 + params['spatial_weight'] * spatial_promising[spatial_mask])
    
    # Apply MST factor
    mst_mask = mst_promising > 0
    mst_component = base_scores.copy()
    mst_component[mst_mask] *= (1.0 + params['mst_weight'] * mst_promising[mst_mask])
    
    # Apply local factors
    local_component = base_scores * local_factors
    
    # Strategy-based combination
    if params['combination_strategy'] == 'distance_heavy':
        combined = 0.4 * spatial_component + 0.3 * mst_component + 0.3 * local_component
    else:  # balanced
        combined = 0.35 * spatial_component + 0.35 * mst_component + 0.3 * local_component
    
    # Distance-based enhancement
    median_dist = np.median(distance_matrix[np.triu_indices(n, k=1)])
    dist_envelope = np.exp(-distance_matrix / (params['distance_weight'] * median_dist + 1e-12))
    combined *= (1.0 + 0.15 * dist_envelope)
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    return combined


def _instance_aware_sparsification(heuristics_matrix: np.ndarray,
                                   distance_matrix: np.ndarray,
                                   n: int,
                                   params: dict) -> np.ndarray:
    """Intelligent sparsification adapting to instance characteristics."""
    if n <= 10:
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    base_ratio = params['sparsification_ratio']
    
    # Analyze heuristic distribution
    non_diag = heuristics_matrix[np.triu_indices(n, k=1)]
    heuristic_std = np.std(non_diag)
    heuristic_mean = np.mean(non_diag)
    
    if heuristic_mean > 0:
        spread_factor = heuristic_std / heuristic_mean
        # Adjust ratio based on spread (more selective for concentrated distributions)
        adjusted_ratio = base_ratio * (1.0 - 0.2 * np.tanh(spread_factor))
    else:
        adjusted_ratio = base_ratio
    
    adjusted_ratio = np.clip(adjusted_ratio, 0.1, 0.4)
    
    for i in range(n):
        row_heuristic = heuristics_matrix[i].copy()
        row_heuristic[i] = -np.inf
        
        # Combined score with distance consideration
        with np.errstate(divide='ignore', invalid='ignore'):
            distance_score = 1 / (distance_matrix[i] + 1e-12)
        distance_score[i] = 0
        
        # Normalize and combine
        max_heuristic = np.max(row_heuristic[row_heuristic > -np.inf])
        max_distance = np.max(distance_score[distance_score < np.inf])
        
        if max_heuristic > 0 and max_distance > 0:
            norm_heuristic = row_heuristic / max_heuristic
            norm_distance = distance_score / max_distance
            
            # Adaptive combination weights
            if params['combination_strategy'] == 'distance_heavy':
                combined = 0.6 * norm_heuristic + 0.4 * norm_distance
            else:
                combined = 0.7 * norm_heuristic + 0.3 * norm_distance
        else:
            combined = row_heuristic
        
        combined[i] = -np.inf
        
        # Determine number of edges to keep
        min_edges = max(3, int(np.log2(n)) + 1)
        max_edges = min(n // 2, 50)
        num_to_keep = min(max_edges, max(min_edges, int(adjusted_ratio * n)))
        
        if np.any(combined > -np.inf):
            threshold = np.partition(combined, -num_to_keep)[-num_to_keep]
            keep_mask = combined >= threshold
            sparsified[i, keep_mask] = heuristics_matrix[i, keep_mask]
    
    # Symmetrize
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified


def _structure_aware_triangle_boosting(heuristics_matrix: np.ndarray,
                                        distance_matrix: np.ndarray,
                                        n: int,
                                        params: dict) -> np.ndarray:
    """Triangle boosting with structure awareness."""
    boosted = heuristics_matrix.copy()
    
    if n < 4:
        return boosted
    
    # Identify promising edges
    edge_mask = (heuristics_matrix > 0) & (np.triu(np.ones_like(heuristics_matrix), 1) > 0)
    edges = np.argwhere(edge_mask)
    
    if len(edges) == 0:
        return boosted
    
    # Sample edges for efficiency in large instances
    if len(edges) > 10000:
        sample_size = min(10000, len(edges))
        edges = edges[np.random.choice(len(edges), sample_size, replace=False)]
    
    # Precompute binary adjacency and local densities
    binary_adj = (heuristics_matrix > 0).astype(float)
    local_densities = np.zeros(n)
    
    for i in range(n):
        neighbors = np.where(binary_adj[i] > 0)[0]
        if len(neighbors) > 0:
            # Local density measure
            local_dists = distance_matrix[i, neighbors]
            local_densities[i] = np.mean(np.exp(-local_dists / (np.median(local_dists) + 1e-12)))
    
    # Process edges
    for i, j in edges:
        # Find common neighbors
        neighbors_i = np.where(binary_adj[i] > 0)[0]
        neighbors_j = np.where(binary_adj[j] > 0)[0]
        common = np.intersect1d(neighbors_i, neighbors_j)
        
        if len(common) == 0:
            continue
        
        # Compute structure-aware boost
        d_ij = distance_matrix[i, j]
        total_boost = 0.0
        
        for k in common[:min(10, len(common))]:  # Limit to top triangles
            d_ik = distance_matrix[i, k]
            d_jk = distance_matrix[j, k]
            
            # Triangle quality metrics
            triangle_sum = d_ij + d_ik + d_jk
            max_side = max(d_ij, d_ik, d_jk)
            
            # Well-formed triangle score
            if max_side > 0:
                aspect_ratio = triangle_sum / (3 * max_side + 1e-12)
                triangle_score = np.exp(-np.abs(aspect_ratio - 1.0))
            else:
                triangle_score = 0
            
            # Local structure factor
            avg_density = (local_densities[i] + local_densities[j] + local_densities[k]) / 3
            density_factor = np.tanh(avg_density * 5)  # Scale and cap
            
            # Combined boost
            triangle_boost = params['triangle_boost_strength'] * triangle_score * density_factor
            total_boost += min(triangle_boost, 0.08)  # Cap per triangle
        
        # Apply boost with diminishing returns
        max_boost = params['max_triangle_boost'] * np.log1p(min(len(common), 5))
        total_boost = min(total_boost, max_boost)
        
        if total_boost > 0:
            boost_factor = 1.0 + total_boost
            boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost_factor
    
    return boosted
