import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.stats import rankdata

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors:
    1. Creates a sparse nearest-neighbor graph
    2. Incorporates global structure via minimum spanning tree
    3. Uses edge centrality measures
    4. Applies adaptive thresholding to sparsify the matrix
    """
    n = distance_matrix.shape[0]
    
    # 1. Base heuristic: inverse distance (shorter edges are better)
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    np.fill_diagonal(inv_dist, 0)
    mask = distance_matrix != 0
    inv_dist[mask] = 1.0 / distance_matrix[mask]
    
    # 2. Nearest neighbor factor: prefer edges connecting to k-nearest neighbors
    k = max(3, n // 10)  # Adaptive k based on problem size
    nn_factor = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get k+1 nearest neighbors (including self)
        indices = np.argsort(distance_matrix[i])[:k+1]
        # Assign weights decaying with neighbor rank
        for rank, j in enumerate(indices[1:], start=1):  # Skip self
            decay = np.exp(-rank / k)  # Exponential decay
            nn_factor[i, j] = decay
    
    # 3. Minimum spanning tree factor: edges in MST are important
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_factor = (mst + mst.T) > 0  # Convert to undirected
    
    # 4. Edge centrality: edges connecting different graph regions
    # Approximate via degree product (higher degree nodes connect more regions)
    degrees = np.sum(distance_matrix > 0, axis=1)
    degree_matrix = np.outer(degrees, degrees)
    centrality = degree_matrix / np.max(degree_matrix)
    
    # 5. Combine all factors with weights
    weights = {
        'distance': 0.4,
        'neighbor': 0.3,
        'mst': 0.2,
        'centrality': 0.1
    }
    
    # Normalize each component to [0,1]
    nn_norm = nn_factor / np.max(nn_factor) if np.max(nn_factor) > 0 else nn_factor
    inv_norm = inv_dist / np.max(inv_dist) if np.max(inv_dist) > 0 else inv_dist
    
    # Combined heuristic
    heuristic = (
        weights['distance'] * inv_norm +
        weights['neighbor'] * nn_norm +
        weights['mst'] * mst_factor +
        weights['centrality'] * centrality
    )
    
    # 6. Adaptive thresholding to sparsify
    # Keep top edges for each node based on heuristic value
    sparsified = np.zeros_like(heuristic)
    
    for i in range(n):
        row = heuristic[i]
        # Dynamic threshold: mean + 0.5*std of non-zero values
        non_zero = row[row > 0]
        if len(non_zero) > 0:
            threshold = np.mean(non_zero) + 0.5 * np.std(non_zero)
            mask = row > threshold
            sparsified[i, mask] = row[mask]
    
    # Ensure symmetry for undirected TSP
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Final normalization
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified /= max_val
    
    return sparsified
