import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with adaptive parameter tuning
    and lightweight local search improvements.
    
    Key improvements over v1:
    1. Adaptive percentile cutoff based on instance characteristics
    2. Lightweight triangle quality boosting for local connectivity
    3. Optimized edge selection with better symmetry enforcement
    4. Efficient parameter adaptation without over-engineering
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric n x n matrix of pairwise distances
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(heuristic, 0)
        max_val = np.max(heuristic)
        if max_val > 0:
            heuristic /= max_val
        return heuristic
    
    # 1. Compute instance statistics for adaptive parameter tuning
    def compute_instance_stats(dist_mat: np.ndarray) -> dict:
        """Compute lightweight statistics for adaptive parameter tuning."""
        # Use only upper triangle to avoid double counting
        triu_indices = np.triu_indices(n, k=1)
        distances = dist_mat[triu_indices]
        
        stats = {}
        stats['mean'] = np.mean(distances)
        stats['cv'] = np.std(distances) / (stats['mean'] + 1e-12)
        
        # Determine instance type based on coefficient of variation
        if stats['cv'] > 0.8:
            stats['type'] = 'irregular'
        elif stats['cv'] < 0.3:
            stats['type'] = 'uniform'
        else:
            stats['type'] = 'moderate'
        
        return stats
    
    stats = compute_instance_stats(distance_matrix)
    
    # 2. Adaptive base heuristic with instance-aware parameters
    with np.errinstance(divide='ignore', invalid='ignore'):
        base_heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(base_heuristic, 0)
    
    # Normalize to [0, 1]
    max_val = np.max(base_heuristic)
    if max_val > 0:
        base_heuristic /= max_val
    
    # 3. Adaptive percentile cutoff based on instance characteristics
    if stats['type'] == 'irregular':
        # For irregular instances, use lower percentile to focus on very close neighbors
        percentile_cutoff = 15
        min_neighbors = max(3, int(np.log2(n)))
    elif stats['type'] == 'uniform':
        # For uniform instances, use higher percentile to consider more connections
        percentile_cutoff = 35
        min_neighbors = max(4, int(np.log2(n) * 1.5))
    else:  # moderate
        percentile_cutoff = 25
        min_neighbors = max(3, int(np.log2(n) * 1.2))
    
    # 4. Enhanced neighborhood detection with adaptive parameters
    neighborhood_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Calculate adaptive distance threshold
        threshold = np.percentile(dists, percentile_cutoff)
        
        # Mark nodes within threshold
        close_nodes = dists <= threshold
        
        # Ensure minimum connections
        if close_nodes.sum() < min_neighbors:
            # Fall back to k-nearest neighbors
            k_neighbors = min_neighbors
            nearest_indices = np.argsort(dists)[:k_neighbors]
            close_nodes = np.zeros(n, dtype=bool)
            close_nodes[nearest_indices] = True
        
        neighborhood_mask[i] = close_nodes
    
    # Ensure symmetry
    neighborhood_mask = neighborhood_mask | neighborhood_mask.T
    
    # 5. Identify MST edges
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # 6. Combine factors with adaptive boosting
    heuristic = base_heuristic.copy()
    
    # Strong boost for MST edges
    heuristic[mst_mask] *= 3.2
    
    # Moderate boost for neighborhood edges
    heuristic[neighborhood_mask] *= 1.9
    
    # Ensure symmetry
    heuristic = (heuristic + heuristic.T) / 2
    
    # 7. Optimized edge selection with adaptive sparsification
    # Adaptive target edges based on instance size and type
    if stats['type'] == 'irregular':
        target_edges = max(4, int(np.log2(n) * 2))
    else:
        target_edges = max(5, int(np.log2(n) * 2.5))
    
    target_edges = min(target_edges, n - 1)
    sparsified = np.zeros_like(heuristic)
    
    for i in range(n):
        row_heuristics = heuristic[i].copy()
        row_heuristics[i] = -np.inf
        
        # Always include MST edges
        mst_neighbors = np.where(mst_mask[i])[0]
        
        # Calculate additional edges needed
        additional_needed = max(0, target_edges - len(mst_neighbors))
        
        if additional_needed > 0:
            # Get non-MST edges
            non_mst_mask = np.ones(n, dtype=bool)
            non_mst_mask[mst_neighbors] = False
            non_mst_mask[i] = False
            
            non_mst_indices = np.where(non_mst_mask)[0]
            if len(non_mst_indices) > 0:
                non_mst_values = row_heuristics[non_mst_indices]
                
                if len(non_mst_indices) > additional_needed:
                    # Keep top additional_needed non-MST edges
                    threshold_idx = np.argpartition(-non_mst_values, additional_needed - 1)[additional_needed - 1]
                    threshold_val = non_mst_values[threshold_idx]
                    keep_non_mst = non_mst_indices[non_mst_values >= threshold_val]
                else:
                    keep_non_mst = non_mst_indices
            else:
                keep_non_mst = []
        else:
            keep_non_mst = []
        
        # Combine all edges to keep
        keep_indices = np.concatenate([mst_neighbors, keep_non_mst])
        sparsified[i, keep_indices] = heuristic[i, keep_indices]
    
    # Enhanced symmetry enforcement
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 8. Lightweight triangle quality boosting
    def boost_triangle_quality(heuristic_mat: np.ndarray, dist_mat: np.ndarray, 
                               n_nodes: int, stats: dict) -> np.ndarray:
        """Lightweight boosting based on triangle quality in the sparse graph."""
        boosted = heuristic_mat.copy()
        
        # Only apply to non-trivial graphs
        if n_nodes < 4:
            return boosted
        
        # Binary mask of edges in the sparse heuristic graph
        edge_mask = heuristic_mat > 0
        
        # For each node, consider its neighbors
        for i in range(n_nodes):
            neighbors = np.where(edge_mask[i])[0]
            if len(neighbors) < 2:
                continue
            
            # For each pair of neighbors (potential triangle)
            for j_idx, j in enumerate(neighbors):
                for k in neighbors[j_idx+1:]:
                    # Check if edge (j,k) exists
                    if edge_mask[j, k]:
                        # Calculate triangle perimeter
                        perimeter = dist_mat[i, j] + dist_mat[i, k] + dist_mat[j, k]
                        
                        # Boost inversely proportional to triangle quality
                        # Better triangles (smaller perimeter) get higher boost
                        if stats['type'] == 'irregular':
                            base_boost = 1.12
                        else:
                            base_boost = 1.08
                        
                        # Normalize by average distance
                        norm_perimeter = perimeter / (3 * stats['mean'] + 1e-12)
                        boost_factor = base_boost * np.exp(-0.5 * norm_perimeter)
                        
                        # Apply boost to all three edges
                        boosted[i, j] *= boost_factor
                        boosted[j, i] = boosted[i, j]
                        
                        boosted[i, k] *= boost_factor
                        boosted[k, i] = boosted[i, k]
                        
                        boosted[j, k] *= boost_factor
                        boosted[k, j] = boosted[j, k]
        
        return boosted
    
    # Apply triangle boosting for moderate-sized instances
    if n <= 500:  # Limit for computational efficiency
        boosted = boost_triangle_quality(sparsified, distance_matrix, n, stats)
    else:
        boosted = sparsified.copy()
    
    # 9. Enhanced connectivity repair
    # Check for nodes with insufficient connections
    for i in range(n):
        degree = np.count_nonzero(boosted[i])
        if degree == 0:
            # Connect isolated node to nearest neighbor
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argmin(dists)
            boosted[i, nearest] = boosted[nearest, i] = 0.25
        elif degree == 1:
            # Node with single connection - add second nearest
            current_neighbor = np.nonzero(boosted[i])[0][0]
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            dists[current_neighbor] = np.inf
            second_nearest = np.argmin(dists)
            boosted[i, second_nearest] = boosted[second_nearest, i] = 0.2
        elif degree == 2 and n > 10:
            # For larger graphs, ensure nodes have at least 3 connections
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            current_neighbors = np.nonzero(boosted[i])[0]
            dists[current_neighbors] = np.inf
            third_nearest = np.argmin(dists)
            boosted[i, third_nearest] = boosted[third_nearest, i] = 0.15
    
    # 10. Final processing
    np.fill_diagonal(boosted, 0)
    
    # Final normalization with stability check
    max_val = np.max(boosted)
    if max_val > 0:
        boosted /= max_val
    
    # Ensure final symmetry
    boosted = (boosted + boosted.T) / 2
    
    return boosted
