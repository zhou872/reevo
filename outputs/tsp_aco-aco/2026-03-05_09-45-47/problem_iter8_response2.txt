import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform
import warnings
from typing import Tuple, Dict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection combining:
    1. Adaptive meta-heuristic framework with self-tuning parameters
    2. Graph neural network-inspired feature extraction
    3. Learning-to-rank edge importance prediction (unsupervised)
    4. Dynamic ensemble of specialized heuristics with feedback
    
    Args:
        distance_matrix: Square symmetric distance matrix (n x n)
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            heuristic = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(heuristic, 0)
        max_val = np.max(heuristic)
        if max_val > 0:
            heuristic /= max_val
        return heuristic
    
    # Phase 1: Problem instance analysis
    instance_stats = _analyze_problem_instance(distance_matrix, n)
    
    # Phase 2: Feature extraction (GNN-inspired)
    node_features, edge_features = _extract_gnn_features(distance_matrix, n, instance_stats)
    
    # Phase 3: Ensemble of specialized heuristics
    heuristic_components = _compute_heuristic_ensemble(
        distance_matrix, n, node_features, edge_features, instance_stats
    )
    
    # Phase 4: Dynamic weighting with feedback
    weights = _compute_dynamic_weights(heuristic_components, instance_stats, n)
    
    # Phase 5: Learning-to-rank edge prediction
    heuristic_matrix = _learn_to_rank_edges(
        heuristic_components, weights, edge_features, node_features, instance_stats, n
    )
    
    # Phase 6: Adaptive refinement
    heuristic_matrix = _adaptive_refinement(
        heuristic_matrix, distance_matrix, node_features, edge_features, instance_stats, n
    )
    
    # Final normalization
    np.fill_diagonal(heuristic_matrix, 0)
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix /= max_val
    
    return heuristic_matrix


def _analyze_problem_instance(distance_matrix: np.ndarray, n: int) -> Dict:
    """Analyze problem instance characteristics for adaptive tuning."""
    triu_indices = np.triu_indices(n, k=1)
    distances = distance_matrix[triu_indices]
    
    stats = {
        'mean_dist': np.mean(distances),
        'std_dist': np.std(distances),
        'cv_dist': np.std(distances) / (np.mean(distances) + 1e-12),
        'skewness': _compute_skewness(distances),
        'kurtosis': _compute_kurtosis(distances),
        'mst_sparsity': _compute_mst_sparsity(distance_matrix, n),
        'cluster_tendency': _compute_cluster_tendency(distance_matrix, n),
        'dimensionality': _estimate_dimensionality(distance_matrix, n)
    }
    
    # Determine problem type
    if stats['cv_dist'] < 0.2:
        stats['problem_type'] = 'regular'
    elif stats['cluster_tendency'] > 0.7:
        stats['problem_type'] = 'clustered'
    elif stats['skewness'] > 1.0:
        stats['problem_type'] = 'sparse'
    else:
        stats['problem_type'] = 'mixed'
    
    return stats


def _compute_skewness(distances: np.ndarray) -> float:
    """Compute skewness of distance distribution."""
    mean = np.mean(distances)
    std = np.std(distances) + 1e-12
    skew = np.mean(((distances - mean) / std) ** 3)
    return abs(skew)


def _compute_kurtosis(distances: np.ndarray) -> float:
    """Compute kurtosis of distance distribution."""
    mean = np.mean(distances)
    std = np.std(distances) + 1e-12
    kurt = np.mean(((distances - mean) / std) ** 4) - 3
    return abs(kurt)


def _compute_mst_sparsity(distance_matrix: np.ndarray, n: int) -> float:
    """Compute MST sparsity measure."""
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).sum()
    max_edges = n * (n - 1) // 2
    return mst_edges / max_edges


def _compute_cluster_tendency(distance_matrix: np.ndarray, n: int) -> float:
    """Estimate cluster tendency using Hopkins statistic approximation."""
    if n <= 100:
        # Small sample approximation
        sample_size = min(20, n // 4)
        indices = np.random.choice(n, sample_size, replace=False)
        
        # Distances to nearest neighbors
        min_dists = []
        for i in indices:
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            min_dists.append(np.min(dists))
        
        # Compare with uniform distribution
        uniform_dists = []
        for _ in range(sample_size):
            # Generate random point distances (approximation)
            random_dists = np.random.choice(distance_matrix[np.triu_indices(n, k=1)], sample_size)
            uniform_dists.append(np.min(random_dists))
        
        hopkins = np.sum(min_dists) / (np.sum(min_dists) + np.sum(uniform_dists) + 1e-12)
        return hopkins
    else:
        # Fast approximation for large n
        k = min(10, n // 100)
        min_dists = np.partition(distance_matrix, k, axis=1)[:, 1:k+1].mean()
        random_dists = np.random.choice(distance_matrix[np.triu_indices(n, k=1)], 1000)
        return min_dists / (min_dists + random_dists.mean() + 1e-12)


def _estimate_dimensionality(distance_matrix: np.ndarray, n: int) -> float:
    """Estimate intrinsic dimensionality using correlation dimension approximation."""
    if n <= 50:
        return 2.0  # Default for small instances
    
    # Sample distances
    triu_indices = np.triu_indices(n, k=1)
    sample_size = min(1000, len(triu_indices[0]))
    sample_indices = np.random.choice(len(triu_indices[0]), sample_size, replace=False)
    sample_dists = distance_matrix[triu_indices[0][sample_indices], triu_indices[1][sample_indices]]
    
    # Use two different scales
    r1 = np.percentile(sample_dists, 25)
    r2 = np.percentile(sample_dists, 50)
    
    # Count pairs within radii
    C1 = np.mean(sample_dists <= r1)
    C2 = np.mean(sample_dists <= r2)
    
    if C1 > 0 and C2 > 0:
        dim = (np.log(C2) - np.log(C1)) / (np.log(r2) - np.log(r1) + 1e-12)
        return max(1.0, min(10.0, dim))
    else:
        return 2.0


def _extract_gnn_features(distance_matrix: np.ndarray, n: int, 
                         instance_stats: Dict) -> Tuple[np.ndarray, np.ndarray]:
    """Extract GNN-inspired node and edge features."""
    
    # Node features
    node_features = np.zeros((n, 8))
    
    # Basic node statistics
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Feature 1-3: Distance statistics
        node_features[i, 0] = np.min(dists)  # Minimum distance
        node_features[i, 1] = np.mean(dists)  # Mean distance
        node_features[i, 2] = np.std(dists)  # Distance std
        
        # Feature 4: Local density (inverse of mean distance)
        node_features[i, 3] = 1.0 / (node_features[i, 1] + 1e-12)
        
        # Feature 5: Distance to k-th nearest neighbor (k = log2(n))
        k = min(int(np.log2(n)) + 1, n - 1)
        kth_dist = np.partition(dists, k)[k]
        node_features[i, 4] = kth_dist
        
        # Feature 6: Distance to farthest neighbor
        node_features[i, 5] = np.max(dists)
        
        # Feature 7: Distance ratio (min/max)
        node_features[i, 6] = node_features[i, 0] / (node_features[i, 5] + 1e-12)
        
        # Feature 8: Centrality measure (inverse of sum of distances)
        node_features[i, 7] = 1.0 / (np.sum(dists) + 1e-12)
    
    # Normalize node features
    for j in range(node_features.shape[1]):
        col = node_features[:, j]
        if np.std(col) > 0:
            node_features[:, j] = (col - np.mean(col)) / (np.std(col) + 1e-12)
    
    # Edge features
    edge_features = np.zeros((n, n, 6))
    
    for i in range(n):
        for j in range(i+1, n):
            # Feature 1: Normalized distance
            edge_features[i, j, 0] = edge_features[j, i, 0] = (
                distance_matrix[i, j] / (instance_stats['mean_dist'] + 1e-12)
            )
            
            # Feature 2: Triangle inequality violation score
            # Minimum of |d(i,j) - d(i,k) - d(k,j)| for sampled k
            sample_k = min(5, n-2)
            if sample_k > 0:
                k_indices = np.random.choice([k for k in range(n) if k not in (i, j)], 
                                           sample_k, replace=False)
                violations = []
                for k in k_indices:
                    violation = abs(distance_matrix[i, j] - 
                                   distance_matrix[i, k] - 
                                   distance_matrix[k, j])
                    violations.append(violation)
                edge_features[i, j, 1] = edge_features[j, i, 1] = np.mean(violations)
            
            # Feature 3: Node feature similarity (cosine similarity)
            feat_i = node_features[i]
            feat_j = node_features[j]
            sim = np.dot(feat_i, feat_j) / (np.linalg.norm(feat_i) * np.linalg.norm(feat_j) + 1e-12)
            edge_features[i, j, 2] = edge_features[j, i, 2] = sim
            
            # Feature 4: Local neighborhood overlap
            k_neighbors = min(10, n-1)
            i_neighbors = set(np.argsort(distance_matrix[i])[1:k_neighbors+1])
            j_neighbors = set(np.argsort(distance_matrix[j])[1:k_neighbors+1])
            overlap = len(i_neighbors.intersection(j_neighbors)) / k_neighbors
            edge_features[i, j, 3] = edge_features[j, i, 3] = overlap
            
            # Feature 5: Rank difference
            i_rank = np.argsort(distance_matrix[i])
            j_rank = np.argsort(distance_matrix[j])
            rank_i_of_j = np.where(i_rank == j)[0][0]
            rank_j_of_i = np.where(j_rank == i)[0][0]
            rank_diff = abs(rank_i_of_j - rank_j_of_i) / n
            edge_features[i, j, 4] = edge_features[j, i, 4] = 1.0 - rank_diff
            
            # Feature 6: Expected vs actual distance ratio
            expected_dist = (node_features[i, 1] + node_features[j, 1]) / 2
            edge_features[i, j, 5] = edge_features[j, i, 5] = (
                expected_dist / (distance_matrix[i, j] + 1e-12)
            )
    
    return node_features, edge_features


def _compute_heuristic_ensemble(distance_matrix: np.ndarray, n: int,
                              node_features: np.ndarray, edge_features: np.ndarray,
                              instance_stats: Dict) -> Dict[str, np.ndarray]:
    """Compute ensemble of specialized heuristic components."""
    
    components = {}
    
    # 1. Reciprocal distance heuristic
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(reciprocal, 0)
    if np.max(reciprocal) > 0:
        reciprocal /= np.max(reciprocal)
    components['reciprocal'] = reciprocal
    
    # 2. MST-based heuristic
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    mst_heuristic = np.zeros((n, n))
    mst_heuristic[mst_mask] = 1.0
    
    # Boost edges that are "almost" in MST
    for i in range(n):
        # Consider edges to nodes that are close but not in MST
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        mst_neighbors = np.where(mst_mask[i])[0]
        
        if len(mst_neighbors) > 0:
            max_mst_dist = np.max(distance_matrix[i, mst_neighbors])
            # Give partial score to edges slightly longer than MST edges
            for j in range(n):
                if i != j and not mst_mask[i, j]:
                    if distance_matrix[i, j] <= 1.5 * max_mst_dist:
                        mst_heuristic[i, j] = 0.3
    
    components['mst'] = mst_heuristic
    
    # 3. Cluster-based heuristic
    cluster_heuristic = _compute_cluster_based_heuristic(
        distance_matrix, node_features, instance_stats, n
    )
    components['cluster'] = cluster_heuristic
    
    # 4. Triangle quality heuristic
    triangle_heuristic = _compute_triangle_quality_heuristic(
        distance_matrix, edge_features, instance_stats, n
    )
    components['triangle'] = triangle_heuristic
    
    # 5. Path extension heuristic
    path_heuristic = _compute_path_extension_heuristic(
        distance_matrix, node_features, instance_stats, n
    )
    components['path'] = path_heuristic
    
    # 6. Global structure heuristic
    global_heuristic = _compute_global_structure_heuristic(
        distance_matrix, instance_stats, n
    )
    components['global'] = global_heuristic
    
    return components


def _compute_cluster_based_heuristic(distance_matrix: np.ndarray,
                                   node_features: np.ndarray,
                                   instance_stats: Dict,
                                   n: int) -> np.ndarray:
    """Compute heuristic based on clustering structure."""
    heuristic = np.zeros((n, n))
    
    # Adaptive clustering based on problem type
    if instance_stats['cluster_tendency'] > 0.6:
        # Use density-based clustering
        eps = np.percentile(distance_matrix[np.triu_indices(n, k=1)], 15)
        clusters = {}
        visited = set()
        
        for i in range(n):
            if i not in visited:
                # Find core points
                neighbors = np.where(distance_matrix[i] < eps)[0]
                if len(neighbors) >= 2:  # Core point
                    cluster = set(neighbors)
                    # Expand cluster
                    queue = list(neighbors)
                    while queue:
                        point = queue.pop()
                        if point not in visited:
                            visited.add(point)
                            point_neighbors = np.where(distance_matrix[point] < eps)[0]
                            if len(point_neighbors) >= 2:
                                cluster.update(point_neighbors)
                                queue.extend([p for p in point_neighbors if p not in visited])
                    
                    # Assign cluster scores
                    for node in cluster:
                        clusters.setdefault(node, []).append(len(cluster))
        
        # Create heuristic based on cluster membership
        for i in range(n):
            if i in clusters:
                for j in range(i+1, n):
                    if j in clusters and any(c_i == c_j for c_i in clusters[i] for c_j in clusters[j]):
                        heuristic[i, j] = heuristic[j, i] = 1.0
    else:
        # Use feature-based similarity clustering
        similarity = node_features @ node_features.T
        np.fill_diagonal(similarity, 0)
        
        # Threshold based on similarity distribution
        threshold = np.percentile(similarity[np.triu_indices(n, k=1)], 70)
        heuristic[similarity > threshold] = 1.0
    
    return heuristic


def _compute_triangle_quality_heuristic(distance_matrix: np.ndarray,
                                      edge_features: np.ndarray,
                                      instance_stats: Dict,
                                      n: int) -> np.ndarray:
    """Compute heuristic based on triangle inequality quality."""
    heuristic = np.zeros((n, n))
    
    # Sample edges for triangle analysis
    sample_rate = min(1.0, 5000 / (n * (n-1) / 2))
    triu_indices = np.triu_indices(n, k=1)
    
    if sample_rate >= 1.0 or n <= 100:
        edges_to_check = list(zip(triu_indices[0], triu_indices[1]))
    else:
        n_edges = len(triu_indices[0])
        n_sample = int(sample_rate * n_edges)
        sample_idx = np.random.choice(n_edges, n_sample, replace=False)
        edges_to_check = [(triu_indices[0][i], triu_indices[1][i]) for i in sample_idx]
    
    # Analyze triangle quality for sampled edges
    for i, j in edges_to_check:
        # Sample potential third vertices
        k_samples = min(10, n-2)
        k_indices = np.random.choice([k for k in range(n) if k not in (i, j)], 
                                   k_samples, replace=False)
        
        triangle_scores = []
        for k in k_indices:
            a, b, c = distance_matrix[i, j], distance_matrix[i, k], distance_matrix[j, k]
            
            # Triangle inequality satisfaction
            satisfies = (a <= b + c + 1e-12) and (b <= a + c + 1e-12) and (c <= a + b + 1e-12)
            
            if satisfies:
                # Quality measure: how close to equilateral?
                edges = np.array([a, b, c])
                mean_edge = edges.mean()
                if mean_edge > 0:
                    # Coefficient of variation (inverse is quality)
                    cv = edges.std() / mean_edge
                    quality = 1.0 / (1.0 + cv)
                    triangle_scores.append(quality)
        
        if triangle_scores:
            avg_quality = np.mean(triangle_scores)
            heuristic[i, j] = heuristic[j, i] = avg_quality
    
    # Propagate scores to similar edges
    if n <= 500:  # Only for manageable sizes
        for i in range(n):
            for j in range(i+1, n):
                if heuristic[i, j] == 0:
                    # Find similar edges to get score
                    similar_i = np.where(edge_features[i, :, 2] > 0.7)[0]
                    similar_j = np.where(edge_features[j, :, 2] > 0.7)[0]
                    
                    scores = []
                    for k in similar_i:
                        if k != j and heuristic[i, k] > 0:
                            scores.append(heuristic[i, k])
                    for k in similar_j:
                        if k != i and heuristic[j, k] > 0:
                            scores.append(heuristic[j, k])
                    
                    if scores:
                        heuristic[i, j] = heuristic[j, i] = np.mean(scores)
    
    return heuristic


def _compute_path_extension_heuristic(distance_matrix: np.ndarray,
                                    node_features: np.ndarray,
                                    instance_stats: Dict,
                                    n: int) -> np.ndarray:
    """Compute heuristic based on path extension potential."""
    heuristic = np.zeros((n, n))
    
    # For each node, consider it as a potential path endpoint
    for i in range(n):
        # Get candidate edges from this node
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Consider different types of edges
        k_close = min(5, n-1)
        close_indices = np.argsort(dists)[:k_close]
        
        k_far = min(3, n-1)
        far_indices = np.argsort(-dists)[:k_far]
        
        # Boost close edges
        for j in close_indices:
            heuristic[i, j] = max(heuristic[i, j], 0.8)
        
        # Consider some far edges (for connectivity)
        for j in far_indices:
            # Only if it connects different regions
            if node_features[i, 1] / (node_features[j, 1] + 1e-12) > 2.0:
                heuristic[i, j] = max(heuristic[i, j], 0.3)
        
        # Consider medium-range edges that balance distance
        median_dist = np.median(dists[dists < np.inf])
        medium_mask = (dists > 0.7 * median_dist) & (dists < 1.3 * median_dist)
        medium_indices = np.where(medium_mask)[0]
        
        for j in medium_indices[:min(3, len(medium_indices))]:
            heuristic[i, j] = max(heuristic[i, j], 0.5)
    
    # Make symmetric
    heuristic = np.maximum(heuristic, heuristic.T)
    
    return heuristic


def _compute_global_structure_heuristic(distance_matrix: np.ndarray,
                                      instance_stats: Dict,
                                      n: int) -> np.ndarray:
    """Compute heuristic based on global structure analysis."""
    heuristic = np.zeros((n, n))
    
    # Use different strategies based on problem type
    if instance_stats['problem_type'] == 'regular':
        # For regular problems, prefer edges with distances close to mean
        mean_dist = instance_stats['mean_dist']
        for i in range(n):
            for j in range(i+1, n):
                ratio = distance_matrix[i, j] / mean_dist
                if 0.8 <= ratio <= 1.2:
                    heuristic[i, j] = heuristic[j, i] = 1.0
                elif 0.6 <= ratio <= 1.4:
                    heuristic[i, j] = heuristic[j, i] = 0.5
    
    elif instance_stats['problem_type'] == 'clustered':
        # For clustered problems, boost intra-cluster edges
        # Simple clustering based on distance threshold
        threshold = np.percentile(distance_matrix[np.triu_indices(n, k=1)], 20)
        
        for i in range(n):
            close_nodes = np.where(distance_matrix[i] < threshold)[0]
            for j in close_nodes:
                if i != j:
                    heuristic[i, j] = max(heuristic[i, j], 0.8)
    
    else:  # mixed or sparse
        # Balance local and global connections
        local_thresh = np.percentile(distance_matrix[np.triu_indices(n, k=1)], 30)
        global_thresh = np.percentile(distance_matrix[np.triu_indices(n, k=1)], 70)
        
        for i in range(n):
            # Local connections
            local_nodes = np.where(distance_matrix[i] < local_thresh)[0]
            for j in local_nodes:
                if i != j:
                    heuristic[i, j] = max(heuristic[i, j], 0.7)
            
            # Some global connections
            global_nodes = np.where(distance_matrix[i] > global_thresh)[0]
            for j in global_nodes[:min(2, len(global_nodes))]:
                heuristic[i, j] = max(heuristic[i, j], 0.3)
    
    return heuristic


def _compute_dynamic_weights(heuristic_components: Dict[str, np.ndarray],
                           instance_stats: Dict,
                           n: int) -> Dict[str, float]:
    """Compute dynamic weights for heuristic ensemble."""
    
    # Base weights based on problem characteristics
    if instance_stats['problem_type'] == 'regular':
        base_weights = {
            'reciprocal': 0.25,
            'mst': 0.20,
            'cluster': 0.15,
            'triangle': 0.20,
            'path': 0.10,
            'global': 0.10
        }
    elif instance_stats['problem_type'] == 'clustered':
        base_weights = {
            'reciprocal': 0.20,
            'mst': 0.15,
            'cluster': 0.30,
            'triangle': 0.15,
            'path': 0.10,
            'global': 0.10
        }
    elif instance_stats['problem_type'] == 'sparse':
        base_weights = {
            'reciprocal': 0.30,
            'mst': 0.25,
            'cluster': 0.10,
            'triangle': 0.15,
            'path': 0.10,
            'global': 0.10
        }
    else:  # mixed
        base_weights = {
            'reciprocal': 0.25,
            'mst': 0.20,
            'cluster': 0.20,
            'triangle': 0.15,
            'path': 0.10,
            'global': 0.10
        }
    
    # Adjust based on instance statistics
    adjustments = {}
    
    # Adjust for distance distribution
    if instance_stats['cv_dist'] < 0.3:
        adjustments['reciprocal'] = 1.1
        adjustments['global'] = 1.2
    elif instance_stats['cv_dist'] > 0.7:
        adjustments['mst'] = 1.2
        adjustments['path'] = 1.1
    
    # Adjust for cluster tendency
    if instance_stats['cluster_tendency'] > 0.7:
        adjustments['cluster'] = 1.3
        adjustments['triangle'] = 0.8
    
    # Adjust for sparsity
    if instance_stats['mst_sparsity'] < 0.1:
        adjustments['mst'] = 1.2
        adjustments['global'] = 0.8
    
    # Apply adjustments
    weights = {}
    for key in base_weights:
        weight = base_weights[key]
        if key in adjustments:
            weight *= adjustments[key]
        weights[key] = weight
    
    # Normalize weights
    total = sum(weights.values())
    for key in weights:
        weights[key] /= total
    
    return weights


def _learn_to_rank_edges(heuristic_components: Dict[str, np.ndarray],
                        weights: Dict[str, float],
                        edge_features: np.ndarray,
                        node_features: np.ndarray,
                        instance_stats: Dict,
                        n: int) -> np.ndarray:
    """Learning-to-rank approach for edge prediction."""
    
    # Combine weighted heuristics
    combined = np.zeros((n, n))
    
    for key, weight in weights.items():
        combined += weight * heuristic_components[key]
    
    # Apply feature-based adjustments
    for i in range(n):
        for j in range(i+1, n):
            # Feature-based adjustment
            feat_adjustment = 1.0
            
            # Adjust based on edge features
            # Feature 2: Triangle violation (lower is better)
            if edge_features[i, j, 1] > 0:
                feat_adjustment *= 1.0 / (1.0 + edge_features[i, j, 1])
            
            # Feature 3: Similarity (higher is better)
            feat_adjustment *= (1.0 + 0.5 * edge_features[i, j, 2])
            
            # Feature 4: Overlap (higher is better)
            feat_adjustment *= (1.0 + 0.3 * edge_features[i, j, 3])
            
            # Feature 6: Expected vs actual (higher is better for short edges)
            if edge_features[i, j, 5] > 1.0:
                feat_adjustment *= min(1.5, edge_features[i, j, 5])
            
            combined[i, j] = combined[j, i] = combined[i, j] * feat_adjustment
    
    # Local ranking adjustment
    for i in range(n):
        # Get scores for edges from node i
        scores = combined[i].copy()
        scores[i] = -np.inf
        
        # Apply softmax to get probability distribution
        if np.any(scores > -np.inf):
            exp_scores = np.exp(scores - np.max(scores[scores > -np.inf]))
            exp_scores[i] = 0
            if exp_scores.sum() > 0:
                probs = exp_scores / exp_scores.sum()
                # Adjust scores based on ranking
                sorted_indices = np.argsort(-scores)
                for rank, j in enumerate(sorted_indices):
                    if j != i and scores[j] > -np.inf:
                        # Boost top-ranked edges, suppress lower-ranked
                        rank_factor = 1.0 + 0.5 * np.exp(-rank / 5.0)
                        combined[i, j] = combined[j, i] = max(
                            combined[i, j],
                            probs[j] * rank_factor
                        )
    
    return combined


def _adaptive_refinement(heuristic_matrix: np.ndarray,
                        distance_matrix: np.ndarray,
                        node_features: np.ndarray,
                        edge_features: np.ndarray,
                        instance_stats: Dict,
                        n: int) -> np.ndarray:
    """Adaptive refinement of heuristic matrix."""
    
    refined = heuristic_matrix.copy()
    
    # Phase 1: Connectivity enhancement
    # Ensure each node has at least 2 connections
    for i in range(n):
        if np.count_nonzero(refined[i]) < 2:
            # Add connections to nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:2]
            for j in nearest:
                if refined[i, j] == 0:
                    refined[i, j] = refined[j, i] = 0.5
    
    # Phase 2: Consistency check
    # Boost edges that are mutually promising
    for i in range(n):
        for j in range(i+1, n):
            if refined[i, j] > 0:
                # Check mutual agreement
                i_top = set(np.argsort(-refined[i])[:min(5, n-1)])
                j_top = set(np.argsort(-refined[j])[:min(5, n-1)])
                
                if j in i_top and i in j_top:
                    refined[i, j] = refined[j, i] *= 1.2
    
    # Phase 3: Problem-specific tuning
    if instance_stats['problem_type'] == 'clustered':
        # For clustered problems, enhance intra-cluster edges
        for i in range(n):
            for j in range(i+1, n):
                if edge_features[i, j, 3] > 0.7:  # High overlap
                    refined[i, j] = refined[j, i] *= 1.3
    
    elif instance_stats['dimensionality'] > 3:
        # For high-dimensional problems, boost MST edges more
        mst = minimum_spanning_tree(distance_matrix)
        mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
        refined[mst_mask] *= 1.5
    
    # Phase 4: Smoothing
    # Apply mild smoothing to reduce noise
    if n <= 500:
        smoothed = np.zeros_like(refined)
        for i in range(n):
            for j in range(i+1, n):
                if refined[i, j] > 0:
                    # Consider common neighbors
                    common = 0
                    total = 0
                    for k in range(n):
                        if k != i and k != j:
                            if refined[i, k] > 0 and refined[j, k] > 0:
                                common += 1
                            if refined[i, k] > 0 or refined[j, k] > 0:
                                total += 1
                    
                    if total > 0:
                        smoothing = 0.7 + 0.3 * (common / total)
                        smoothed[i, j] = smoothed[j, i] = refined[i, j] * smoothing
                    else:
                        smoothed[i, j] = smoothed[j, i] = refined[i, j]
        
        refined = smoothed
    
    # Ensure non-negativity
    refined = np.maximum(refined, 0)
    
    return refined
