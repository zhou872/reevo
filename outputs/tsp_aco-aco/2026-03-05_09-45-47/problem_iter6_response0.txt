import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import KDTree
from sklearn.manifold import MDS
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection combining spatial clustering,
    minimum spanning tree, and adaptive sparsification.
    
    Features:
    1. Reciprocal distance as base heuristic
    2. Spatial clustering using efficient coordinate approximation
    3. Minimum spanning tree as fundamental connections
    4. Adaptive sparsification based on graph size
    5. Triangle boosting for dense clusters
    
    Returns:
        Edge promise matrix with values in [0, 1], same shape as input.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Base heuristic: reciprocal of distances (scaled logarithmically)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        # Use logarithmic scaling to reduce dominance of very short edges
        reciprocal_weights = np.log1p(reciprocal_weights)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Spatial clustering for geometric structure
    spatial_promising = _compute_spatial_clustering(distance_matrix, n)
    
    # Minimum Spanning Tree edges (fundamental connections)
    mst_promising = _compute_mst_promises(distance_matrix, n)
    
    # Combine factors efficiently
    heuristics_matrix = _combine_factors(
        reciprocal_weights, spatial_promising, mst_promising, distance_matrix, n
    )
    
    # Adaptive sparsification: keep most promising edges per node
    sparsified = _adaptive_sparsification(heuristics_matrix, n)
    
    # Triangle boosting: enhance edges within dense clusters
    final_heuristics = _triangle_boosting(sparsified, n)
    
    # Normalize to [0, 1] range
    np.fill_diagonal(final_heuristics, 0)
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    return final_heuristics

def _compute_spatial_clustering(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """
    Compute spatial clustering factor using efficient coordinate approximation.
    Uses MDS for small instances and landmark-based approximation for large ones.
    """
    spatial_promising = np.zeros((n, n), dtype=float)
    
    # Adaptive neighborhood size based on graph size
    k_neighbors = max(3, min(int(np.log2(n)) + 2, n - 1))
    
    if n <= 500:  # Use MDS for accurate coordinates
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(3, n-1),
                dissimilarity='precomputed',
                random_state=42,
                n_init=1,
                max_iter=50,
                eps=1e-3
            )
            coords = mds.fit_transform(distance_matrix)
            tree = KDTree(coords)
            
            for i in range(n):
                distances, indices = tree.query(coords[i], k=k_neighbors + 1)
                # Exclude self if it's in the results
                indices = indices[1:] if indices[0] == i else indices[:k_neighbors]
                spatial_promising[i, indices] = 1.0
    else:  # Fast distance-based approach for large instances
        # Use local neighborhood based on distances directly
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            # Get k_neighbors closest nodes
            indices = np.argpartition(dists, k_neighbors)[:k_neighbors]
            spatial_promising[i, indices] = 1.0
    
    return spatial_promising

def _compute_mst_promises(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """
    Compute promises based on Minimum Spanning Tree edges.
    Includes nearby edges that are "almost" in the MST.
    """
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    
    # Make symmetric and boost MST edges
    mst_promising = mst_edges + mst_edges.T
    
    # Also consider edges that are "almost" in MST
    # For each node, consider its k closest neighbors
    k_close = min(4, n - 1)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        closest = np.argpartition(dists, k_close)[:k_close]
        for j in closest:
            if mst_promising[i, j] == 0:
                mst_promising[i, j] = 0.3  # Smaller weight than actual MST edges
    
    return mst_promising

def _combine_factors(reciprocal_weights: np.ndarray,
                    spatial_promising: np.ndarray,
                    mst_promising: np.ndarray,
                    distance_matrix: np.ndarray,
                    n: int) -> np.ndarray:
    """
    Combine all heuristic factors efficiently using vectorized operations.
    """
    # Apply spatial factor: boost edges within spatial clusters
    spatial_boost = np.where(spatial_promising > 0, 1.5, 0.8)
    
    # Apply MST factor: strongly boost MST edges
    mst_boost = np.where(mst_promising >= 2, 2.0,  # Actual MST edges
                         np.where(mst_promising > 0, 1.2, 1.0))  # Nearby edges
    
    # Additional boost for very short edges
    median_dist = np.median(distance_matrix[np.triu_indices(n, k=1)])
    short_edges = distance_matrix < 0.3 * median_dist
    short_boost = np.where(short_edges, 1.3, 1.0)
    
    # Combine all factors
    combined = reciprocal_weights * spatial_boost * mst_boost * short_boost
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    return combined

def _adaptive_sparsification(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """
    Adaptively sparsify the heuristic matrix, keeping more edges for smaller graphs.
    """
    sparsified = np.zeros_like(heuristics_matrix)
    
    if n <= 10:
        # Keep all edges for very small graphs
        return heuristics_matrix.copy()
    
    # Adaptive sparsity: keep fewer edges for larger graphs
    keep_ratio = max(0.15, min(0.4, 0.5 - 0.1 * np.log10(n)))
    num_to_keep = max(3, int(keep_ratio * n))
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Ignore diagonal
        
        # Find top edges
        if np.any(row > 0):
            top_indices = np.argpartition(-row, num_to_keep)[:num_to_keep]
            sparsified[i, top_indices] = heuristics_matrix[i, top_indices]
    
    # Ensure symmetry by taking maximum
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified

def _triangle_boosting(heuristics_matrix: np.ndarray, n: int) -> np.ndarray:
    """
    Boost scores for edges that form triangles (common neighbors).
    Uses efficient matrix multiplication for exact computation on small graphs,
    and sampling for large graphs.
    """
    boosted = heuristics_matrix.copy()
    
    if n <= 300:  # Exact computation for moderate graphs
        binary_mask = (heuristics_matrix > 0).astype(float)
        common_neighbors = binary_mask @ binary_mask.T
        
        for i in range(n):
            for j in range(i+1, n):
                if heuristics_matrix[i, j] > 0:
                    common = common_neighbors[i, j]
                    if common > 0:
                        boost = 1 + 0.15 * min(common, 5)
                        boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    elif n <= 1000:  # Approximate with sampling for larger graphs
        sample_size = min(200, n // 5)
        sampled_nodes = np.random.choice(n, sample_size, replace=False)
        
        for i in sampled_nodes:
            neighbors_i = set(np.where(heuristics_matrix[i] > 0)[0])
            for j in neighbors_i:
                if j <= i:
                    continue
                neighbors_j = set(np.where(heuristics_matrix[j] > 0)[0])
                common = len(neighbors_i & neighbors_j)
                if common > 0:
                    boost = 1 + 0.1 * common
                    boosted[i, j] = boosted[j, i] = heuristics_matrix[i, j] * boost
    
    # For very large graphs (n > 1000), skip triangle boosting for efficiency
    
    return boosted
