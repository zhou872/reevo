import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.manifold import MDS
from sklearn.neighbors import KDTree
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with improved scalability and sparsification.
    Returns a matrix of edge promises with the same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Base heuristic: reciprocal of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Factor 1: Scalable spatial clustering
    def compute_spatial_clustering(dist_mat, size):
        spatial_promising = np.zeros_like(dist_mat)
        if size <= 1000:
            # Use MDS for accurate coordinates
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                mds = MDS(
                    n_components=min(3, size-1),
                    dissimilarity='precomputed',
                    random_state=42,
                    n_init=1,
                    max_iter=100
                )
                coords = mds.fit_transform(dist_mat)
        else:
            # Landmark-based approximation for large n
            landmarks = min(50, size // 20)
            landmark_indices = np.random.choice(size, landmarks, replace=False)
            coords = np.zeros((size, landmarks))
            for i in range(size):
                coords[i] = dist_mat[i, landmark_indices]
        
        # Build KD-tree for efficient queries
        tree = KDTree(coords)
        k = max(3, min(int(np.log2(size)) + 2, size - 1))  # Adaptive k based on log2(n)
        
        for i in range(size):
            if size <= 1000:
                distances, indices = tree.query(coords[i], k=k+1)
                indices = indices[1:] if distances[0] == 0 else indices[:k]
            else:
                dists = dist_mat[i].copy()
                dists[i] = np.inf
                indices = np.argpartition(dists, k)[:k]
            spatial_promising[i, indices] = 1.0
        return spatial_promising
    
    spatial_promising = compute_spatial_clustering(distance_matrix, n)
    
    # Factor 2: Minimum Spanning Tree edges
    def compute_mst_promises(dist_mat):
        mst = minimum_spanning_tree(dist_mat)
        mst_edges = (mst.toarray() > 0).astype(float)
        mst_promising = mst_edges + mst_edges.T
        size = dist_mat.shape[0]
        for i in range(size):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            closest = np.argsort(dists)[:3]
            for j in closest:
                if mst_promising[i, j] == 0:
                    mst_promising[i, j] = 0.5
        return mst_promising
    
    mst_promising = compute_mst_promises(distance_matrix)
    
    # Factor 3: Local connectivity factors
    def compute_local_factors(dist_mat, size):
        m = min(5, size - 1)
        local_stats = np.zeros((size, 2))
        for i in range(size):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            sorted_dists = np.sort(dists)[:m]
            local_stats[i, 0] = np.mean(sorted_dists)
            local_stats[i, 1] = np.std(sorted_dists) + 1e-12
        
        local_factors = np.ones_like(dist_mat)
        for i in range(size):
            for j in range(i+1, size):
                stat_similarity = np.exp(
                    -np.abs(local_stats[i, 0] - local_stats[j, 0]) / 
                    (local_stats[i, 1] + local_stats[j, 1])
                )
                mean_dist = (local_stats[i, 0] + local_stats[j, 0]) / 2
                dist_ratio = dist_mat[i, j] / (mean_dist + 1e-12)
                local_factors[i, j] = local_factors[j, i] = (
                    stat_similarity * np.exp(-0.5 * dist_ratio**2)
                )
        return local_factors
    
    local_factors = compute_local_factors(distance_matrix, n)
    
    # Combine factors efficiently
    def combine_factors(recip_weights, spatial, mst, local, dist_mat):
        base_scores = recip_weights.copy()
        spatial_factor = np.where(spatial > 0, 2.0, 0.5)
        mst_factor = np.where(mst > 0, 
                             np.where(mst == 2, 3.0, 1.5), 
                             0.8)
        combined_scores = base_scores * spatial_factor * mst_factor * local
        with np.errstate(divide='ignore', invalid='ignore'):
            log_boost = 1 + 0.3 * np.log(1 + 1 / (base_scores + 1e-12))
            log_boost = np.nan_to_num(log_boost, nan=1.0, posinf=1.0)
        heuristics_matrix = combined_scores * log_boost
        heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
        return heuristics_matrix
    
    heuristics_matrix = combine_factors(
        reciprocal_weights, spatial_promising, mst_promising, 
        local_factors, distance_matrix
    )
    
    # Improved adaptive sparsification based on log2(n)
    def adaptive_sparsification(heuristics_mat, size):
        if size <= 10:
            return heuristics_mat.copy()
        sparsified = np.zeros_like(heuristics_mat)
        # Base keep ratio using log2(n) for better scalability
        base_keep_ratio = min(0.4, max(0.05, 2 / np.log2(size)))
        for i in range(size):
            row = heuristics_mat[i].copy()
            row[i] = -np.inf
            num_to_keep = max(3, int(base_keep_ratio * size))
            num_to_keep = min(num_to_keep, size - 1)
            if np.any(row > 0):
                top_indices = np.argpartition(-row, num_to_keep)[:num_to_keep]
                threshold = np.min(row[top_indices])
                mask = row >= threshold
                sparsified[i] = heuristics_mat[i] * mask
        sparsified = np.maximum(sparsified, sparsified.T)
        return sparsified
    
    sparsified = adaptive_sparsification(heuristics_matrix, n)
    
    # Triangle boosting with efficient sampling for large n
    def triangle_boosting(heuristics_mat):
        size = heuristics_mat.shape[0]
        boosted = heuristics_mat.copy()
        if size > 500:
            sample_size = min(100, size)
            sampled_nodes = np.random.choice(size, sample_size, replace=False)
            for i in sampled_nodes:
                neighbors_i = set(np.where(heuristics_mat[i] > 0)[0])
                for j in neighbors_i:
                    if j <= i:
                        continue
                    neighbors_j = set(np.where(heuristics_mat[j] > 0)[0])
                    common = len(neighbors_i & neighbors_j)
                    if common > 0:
                        boost = 1 + 0.15 * common
                        boosted[i, j] = boosted[j, i] = heuristics_mat[i, j] * boost
        else:
            binary_mask = (heuristics_mat > 0).astype(float)
            common_neighbors = binary_mask @ binary_mask.T
            for i in range(size):
                for j in range(i+1, size):
                    if heuristics_mat[i, j] > 0:
                        common = common_neighbors[i, j]
                        if common > 0:
                            boost = 1 + 0.2 * min(common, 5)
                            boosted[i, j] = boosted[j, i] = heuristics_mat[i, j] * boost
        return boosted
    
    final_heuristics = triangle_boosting(sparsified)
    
    # Normalize to [0, 1] range
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics
