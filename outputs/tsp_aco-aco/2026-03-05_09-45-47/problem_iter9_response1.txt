import numpy as np
from sklearn.neighbors import KDTree
from sklearn.manifold import MDS
from scipy.sparse.csgraph import minimum_spanning_tree
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection with distance-aware triangle boosting
    and adaptive spatial clustering. Returns a matrix of edge promises where higher
    values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # Base heuristic: reciprocal of distances
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal_weights = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal_weights, 0)
    
    # Factor 1: Adaptive spatial clustering (MDS for n≤500, landmarks for larger)
    spatial_promising = _compute_spatial_clustering_v2(distance_matrix, n)
    
    # Factor 2: Minimum Spanning Tree edges with extended neighborhood
    mst_promising = _compute_mst_promises_v2(distance_matrix)
    
    # Factor 3: Local connectivity statistics
    local_factors = _compute_local_factors_v2(distance_matrix, n)
    
    # Combine factors
    heuristics_matrix = _combine_factors_v2(
        reciprocal_weights, spatial_promising, mst_promising, 
        local_factors, distance_matrix
    )
    
    # Adaptive sparsification with distance-based thresholding
    sparsified = _adaptive_sparsification_v2(heuristics_matrix, distance_matrix, n)
    
    # Distance-aware triangle boosting with capping
    final_heuristics = _triangle_boosting_v2(sparsified, distance_matrix, n)
    
    # Normalize to [0, 1] range
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics = final_heuristics / max_val
    
    np.fill_diagonal(final_heuristics, 0)
    return final_heuristics


def _compute_spatial_clustering_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute spatial clustering factor using adaptive method."""
    spatial_promising = np.zeros_like(distance_matrix)
    
    if n <= 500:
        # Use MDS for accurate coordinates when n is manageable
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            mds = MDS(
                n_components=min(3, n-1), 
                dissimilarity='precomputed', 
                random_state=42,
                n_init=1,
                max_iter=100
            )
            coords = mds.fit_transform(distance_matrix)
    else:
        # Use landmark approximation for large graphs
        landmarks = min(100, n // 50)
        landmark_indices = np.random.choice(n, landmarks, replace=False)
        landmark_dists = distance_matrix[landmark_indices]
        
        # Approximate coordinates using distance to landmarks
        coords = np.zeros((n, landmarks))
        for i in range(n):
            coords[i] = distance_matrix[i, landmark_indices]
    
    # Build KD-tree for spatial queries
    tree = KDTree(coords)
    
    # Adaptive neighborhood size
    k = max(3, min(int(np.log2(n)) + 3, n - 1))
    
    # Find nearest neighbors for each node
    for i in range(n):
        if n <= 500 or coords.shape[1] > 1:
            distances, indices = tree.query(coords[i].reshape(1, -1), k=k+1)
            indices = indices[0]
            # Exclude self if present
            if indices[0] == i:
                indices = indices[1:k+1]
            else:
                indices = indices[:k]
        else:
            # Fallback: use distance matrix
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            indices = np.argpartition(dists, k)[:k]
        
        spatial_promising[i, indices] = 1.0
    
    return spatial_promising


def _compute_mst_promises_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute promises based on MST and extended neighborhood."""
    n = distance_matrix.shape[0]
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    mst_promising = mst_edges + mst_edges.T  # Make symmetric
    
    # Add extended neighborhood (closest 4 nodes)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        closest = np.argsort(dists)[:4]
        for j in closest:
            if mst_promising[i, j] == 0:
                mst_promising[i, j] = 0.7
                mst_promising[j, i] = 0.7
    
    return mst_promising


def _compute_local_factors_v2(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute local connectivity factors with improved similarity measure."""
    # Neighborhood statistics
    m = min(6, n - 1)
    local_stats = np.zeros((n, 2))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        sorted_dists = np.sort(dists)[:m]
        local_stats[i, 0] = np.mean(sorted_dists)
        local_stats[i, 1] = np.std(sorted_dists) + 1e-12
    
    # Compute local factor matrix efficiently
    local_factors = np.ones_like(distance_matrix)
    
    for i in range(n):
        for j in range(i+1, n):
            # Enhanced similarity based on local distributions
            mean_i, std_i = local_stats[i]
            mean_j, std_j = local_stats[j]
            
            stat_similarity = np.exp(
                -0.5 * ((mean_i - mean_j) / (std_i + std_j))**2
            )
            
            # Distance relative to combined local statistics
            comb_mean = 0.5 * (mean_i + mean_j)
            dist_norm = distance_matrix[i, j] / (comb_mean + 1e-12)
            
            # Combined factor
            factor = stat_similarity * np.exp(-0.7 * dist_norm)
            local_factors[i, j] = local_factors[j, i] = max(0.1, factor)
    
    return local_factors


def _combine_factors_v2(reciprocal_weights, spatial_promising, mst_promising,
                        local_factors, distance_matrix):
    """Combine heuristic factors with distance-weighted blending."""
    n = reciprocal_weights.shape[0]
    
    # Base scores with logarithmic enhancement
    base_scores = reciprocal_weights.copy()
    with np.errstate(divide='ignore', invalid='ignore'):
        log_boost = 1 + 0.2 * np.log(1 + 1 / (base_scores + 1e-12))
        log_boost = np.nan_to_num(log_boost, nan=1.0, posinf=1.0)
    
    # Spatial factor: stronger boost for spatial neighbors
    spatial_boost = np.where(spatial_promising > 0, 1.8, 0.6)
    
    # MST factor: hierarchical boosting
    mst_boost = np.where(mst_promising > 1, 2.5,
                        np.where(mst_promising > 0.5, 1.6, 0.9))
    
    # Combine with element-wise multiplication
    combined = base_scores * spatial_boost * mst_boost * local_factors * log_boost
    
    # Ensure symmetry
    combined = 0.5 * (combined + combined.T)
    
    return combined


def _adaptive_sparsification_v2(heuristics_matrix, distance_matrix, n):
    """Adaptive sparsification with distance-based thresholding."""
    if n <= 10:
        return heuristics_matrix.copy()
    
    sparsified = np.zeros_like(heuristics_matrix)
    
    # Adaptive keep ratio based on graph size
    base_ratio = 0.35 - 0.05 * np.log10(max(10, n))
    keep_ratio = np.clip(base_ratio, 0.15, 0.4)
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf
        
        # Number of edges to keep
        k = max(4, int(keep_ratio * n))
        k = min(k, n - 1)
        
        if np.any(row > 0):
            # Get top k indices
            top_indices = np.argpartition(-row, k)[:k]
            threshold = np.min(row[top_indices])
            
            # Apply threshold
            mask = row >= threshold
            sparsified[i] = heuristics_matrix[i] * mask
    
    # Symmetrize
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified


def _triangle_boosting_v2(heuristics_matrix, distance_matrix, n):
    """Distance-aware triangle boosting with capping."""
    boosted = heuristics_matrix.copy()
    
    if n > 600:
        # Sampling for large graphs
        sample_size = min(150, n)
        sampled_nodes = np.random.choice(n, sample_size, replace=False)
        
        for i in sampled_nodes:
            neighbors_i = set(np.where(heuristics_matrix[i] > 0)[0])
            for j in neighbors_i:
                if j <= i:
                    continue
                neighbors_j = set(np.where(heuristics_matrix[j] > 0)[0])
                common = neighbors_i & neighbors_j
                
                if common:
                    total_boost = 0
                    for k in common:
                        # Average distance in triangle
                        avg_dist = (distance_matrix[i, j] + 
                                   distance_matrix[i, k] + 
                                   distance_matrix[j, k]) / 3
                        # Inverse proportional boost
                        triangle_boost = 1.0 / (1.0 + 0.5 * avg_dist)
                        total_boost += min(triangle_boost, 0.3)  # Per-triangle cap
                    
                    # Apply capped total boost
                    total_boost = min(total_boost, 1.2)
                    boosted[i, j] = boosted[j, i] *= (1 + 0.25 * total_boost)
    else:
        # Exact computation for smaller graphs
        binary_mask = (heuristics_matrix > 0).astype(float)
        common_counts = binary_mask @ binary_mask.T
        
        for i in range(n):
            for j in range(i+1, n):
                if heuristics_matrix[i, j] > 0:
                    common = common_counts[i, j]
                    if common > 0:
                        # Estimate average triangle distance
                        row_i = distance_matrix[i]
                        row_j = distance_matrix[j]
                        potential_k = np.where(binary_mask[i] & binary_mask[j])[0]
                        
                        if len(potential_k) > 0:
                            avg_tri_dist = np.mean([
                                (row_i[j] + row_i[k] + row_j[k]) / 3 
                                for k in potential_k[:5]  # Sample first few
                            ])
                            dist_boost = 1.0 / (1.0 + 0.3 * avg_tri_dist)
                        else:
                            dist_boost = 1.0
                        
                        # Combine with common neighbor count (capped)
                        neighbor_boost = 0.18 * min(common, 6)
                        total_boost = min(dist_boost * neighbor_boost, 1.0)
                        
                        boosted[i, j] = boosted[j, i] *= (1 + total_boost)
    
    return boosted
