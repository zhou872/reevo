import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import KDTree
from sklearn.manifold import MDS
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with instance-adaptive optimization
    and spectral graph features.
    
    Key innovations:
    1. Instance-aware parameter tuning using statistical analysis
    2. Spectral graph features for structural insight
    3. Multi-stage heuristic fusion with learned priorities
    4. Quality-aware sparsification preserving solution diversity
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
        return result
    
    # 1. Analyze instance characteristics
    instance_stats = _analyze_instance_v2(distance_matrix, n)
    
    # 2. Compute base heuristics
    reciprocal_base = _compute_reciprocal_base(distance_matrix, n)
    
    # 3. Compute spectral graph features
    spectral_features = _compute_spectral_features(distance_matrix, n, instance_stats)
    
    # 4. Compute geometric heuristics
    geometric_promising = _compute_geometric_heuristics(distance_matrix, n, instance_stats)
    
    # 5. Compute structural heuristics
    structural_promising = _compute_structural_heuristics(distance_matrix, n, instance_stats)
    
    # 6. Fuse heuristics with learned priorities
    fused = _fuse_heuristics_v2(
        reciprocal_base, spectral_features, geometric_promising,
        structural_promising, distance_matrix, instance_stats
    )
    
    # 7. Quality-aware sparsification
    sparsified = _quality_aware_sparsification(fused, distance_matrix, n, instance_stats)
    
    # 8. Community-aware boosting
    boosted = _community_aware_boosting(sparsified, distance_matrix, n, instance_stats)
    
    # 9. Final refinement
    refined = _final_refinement(boosted, distance_matrix, n, instance_stats)
    
    np.fill_diagonal(refined, 0)
    max_val = np.max(refined)
    if max_val > 0:
        refined = refined / max_val
    
    return refined


def _analyze_instance_v2(dist_mat: np.ndarray, n: int) -> dict:
    """Advanced instance analysis with spectral properties."""
    stats = {}
    
    # Extract distance statistics
    triu_indices = np.triu_indices(n, k=1)
    distances = dist_mat[triu_indices]
    
    stats['n'] = n
    stats['median_dist'] = np.median(distances) if len(distances) > 0 else 1.0
    stats['mean_dist'] = np.mean(distances) if len(distances) > 0 else 1.0
    stats['std_dist'] = np.std(distances) if len(distances) > 0 else 1.0
    stats['cv'] = stats['std_dist'] / (stats['mean_dist'] + 1e-12)
    
    # Distance distribution skewness and kurtosis
    if len(distances) > 0:
        stats['skewness'] = np.mean(((distances - stats['mean_dist']) / (stats['std_dist'] + 1e-12))**3)
        stats['kurtosis'] = np.mean(((distances - stats['mean_dist']) / (stats['std_dist'] + 1e-12))**4)
    else:
        stats['skewness'] = 0.0
        stats['kurtosis'] = 3.0
    
    # Size-based parameters
    stats['log_n'] = np.log2(n + 1)
    stats['sqrt_n'] = int(np.sqrt(n))
    
    # Adaptive parameter sets
    if n <= 100:
        stats['use_spectral'] = True
        stats['spatial_dim'] = min(5, n - 1)
        stats['community_resolution'] = 1.0
    elif n <= 1000:
        stats['use_spectral'] = True
        stats['spatial_dim'] = min(10, int(np.sqrt(n)))
        stats['community_resolution'] = 0.8
    else:
        stats['use_spectral'] = False
        stats['spatial_dim'] = min(15, int(np.log2(n) * 2))
        stats['community_resolution'] = 0.6
    
    # Neighbor counts
    stats['k_spatial'] = max(3, min(int(stats['log_n'] * 2), n - 1))
    stats['k_structural'] = max(2, min(int(stats['log_n'] * 1.5), n - 1))
    
    return stats


def _compute_reciprocal_base(dist_mat: np.ndarray, n: int) -> np.ndarray:
    """Compute base reciprocal heuristic with adaptive scaling."""
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / (dist_mat + np.eye(n) * 1e-12)
        np.fill_diagonal(reciprocal, 0)
    
    # Normalize to [0, 1] per node
    for i in range(n):
        row = reciprocal[i]
        max_val = np.max(row)
        if max_val > 0:
            reciprocal[i] = row / max_val
    
    return reciprocal


def _compute_spectral_features(dist_mat: np.ndarray, n: int, stats: dict) -> np.ndarray:
    """Compute spectral graph features using Laplacian eigenvectors."""
    features = np.zeros((n, n))
    
    if not stats['use_spectral'] or n < 10:
        return features
    
    try:
        # Create similarity matrix from distances
        sigma = stats['median_dist'] * 0.5
        similarities = np.exp(-dist_mat**2 / (2 * sigma**2))
        np.fill_diagonal(similarities, 0)
        
        # Compute normalized Laplacian
        degree = np.sum(similarities, axis=1)
        D_inv_sqrt = np.diag(1.0 / np.sqrt(degree + 1e-12))
        L_norm = np.eye(n) - D_inv_sqrt @ similarities @ D_inv_sqrt
        
        # Compute eigenvectors (only for small/medium instances)
        if n <= 500:
            eigvals, eigvecs = np.linalg.eigh(L_norm)
            # Use Fiedler vector (second smallest eigenvector)
            fiedler = eigvecs[:, 1]
            
            # Compute spectral distances
            for i in range(n):
                for j in range(i + 1, n):
                    spectral_dist = abs(fiedler[i] - fiedler[j])
                    # Convert to similarity
                    spectral_sim = np.exp(-spectral_dist * 10)
                    features[i, j] = features[j, i] = spectral_sim
        
        # For larger instances, use landmark-based approximation
        else:
            landmarks = min(50, n // 20)
            landmark_indices = np.random.choice(n, landmarks, replace=False)
            
            # Compute distances to landmarks in spectral space
            landmark_distances = dist_mat[landmark_indices, :].T
            
            # Normalize and use as features
            for i in range(landmarks):
                landmark_distances[:, i] = landmark_distances[:, i] / np.max(landmark_distances[:, i] + 1e-12)
            
            # Compute feature similarity
            for i in range(n):
                for j in range(i + 1, n):
                    diff = np.linalg.norm(landmark_distances[i] - landmark_distances[j])
                    features[i, j] = features[j, i] = np.exp(-diff)
    
    except:
        # Fallback to distance-based features
        pass
    
    return features


def _compute_geometric_heuristics(dist_mat: np.ndarray, n: int, stats: dict) -> np.ndarray:
    """Compute geometric heuristics using spatial embedding."""
    geometric = np.zeros((n, n))
    
    try:
        with warnings.catch_warnings():
            warnings.simplefilter("ignore")
            
            # Use MDS for spatial embedding
            n_components = min(stats['spatial_dim'], n - 1)
            mds = MDS(
                n_components=n_components,
                dissimilarity='precomputed',
                random_state=42,
                n_init=1,
                max_iter=50,
                eps=1e-3
            )
            
            coords = mds.fit_transform(dist_mat)
            tree = KDTree(coords)
            
            # Find nearest neighbors in embedded space
            for i in range(n):
                distances, indices = tree.query(
                    coords[i].reshape(1, -1),
                    k=stats['k_spatial'] + 1
                )
                indices = indices[0]
                
                # Exclude self
                if indices[0] == i:
                    indices = indices[1:stats['k_spatial'] + 1]
                else:
                    indices = indices[:stats['k_spatial']]
                
                # Assign weights based on embedded distance
                for idx, j in enumerate(indices):
                    geometric[i, j] = 1.0 / (1.0 + distances[0][idx])
    
    except:
        # Fallback to distance-based geometric heuristic
        for i in range(n):
            dists = dist_mat[i].copy()
            dists[i] = np.inf
            k = min(stats['k_spatial'], n - 1)
            
            if k < n - 1:
                nearest_idx = np.argpartition(dists, k)[:k]
            else:
                nearest_idx = np.argsort(dists)[:k]
            
            for j in nearest_idx:
                geometric[i, j] = 1.0 / (1.0 + dist_mat[i, j])
    
    # Ensure symmetry
    geometric = (geometric + geometric.T) / 2
    
    return geometric


def _compute_structural_heuristics(dist_mat: np.ndarray, n: int, stats: dict) -> np.ndarray:
    """Compute structural heuristics (MST, centrality, etc.)."""
    structural = np.zeros((n, n))
    
    # 1. MST edges
    mst = minimum_spanning_tree(dist_mat)
    mst_edges = (mst.toarray() > 0) | (mst.toarray().T > 0)
    structural[mst_edges] = 1.0
    
    # 2. Add near-MST edges based on triangle inequality
    for i in range(n):
        dists_i = dist_mat[i].copy()
        dists_i[i] = np.inf
        
        # Get MST neighbors
        mst_neighbors = np.where(mst_edges[i])[0]
        
        # Add edges that complete good triangles with MST edges
        for j in mst_neighbors:
            for k in range(n):
                if k != i and k != j and not mst_edges[i, k]:
                    # Check triangle inequality quality
                    detour = dist_mat[i, j] + dist_mat[j, k]
                    direct = dist_mat[i, k]
                    
                    if detour < direct * 1.3:  # Reasonable detour
                        structural[i, k] = max(structural[i, k], 0.7)
                        structural[k, i] = structural[i, k]
    
    # 3. Degree centrality adjustment
    degree_weights = np.zeros(n)
    for i in range(n):
        degree_weights[i] = np.sum(structural[i] > 0)
    
    if np.max(degree_weights) > 0:
        degree_weights = degree_weights / np.max(degree_weights)
        
        # Boost edges from low-degree nodes
        for i in range(n):
            if degree_weights[i] < 0.5:
                row = structural[i]
                row[row > 0] *= (1.0 + 0.3 * (0.5 - degree_weights[i]))
                structural[i] = row
    
    # Ensure symmetry
    structural = (structural + structural.T) / 2
    
    return structural


def _fuse_heuristics_v2(reciprocal, spectral, geometric, structural,
                        dist_mat: np.ndarray, stats: dict) -> np.ndarray:
    """Fuse multiple heuristics with instance-adaptive weights."""
    n = reciprocal.shape[0]
    fused = np.zeros((n, n))
    
    # Determine weights based on instance characteristics
    if stats['cv'] < 0.2:  # Very uniform
        w_reciprocal = 0.4
        w_spectral = 0.3
        w_geometric = 0.2
        w_structural = 0.1
    elif stats['cv'] < 0.5:  # Moderate variation
        w_reciprocal = 0.3
        w_spectral = 0.4
        w_geometric = 0.2
        w_structural = 0.1
    elif stats['cv'] < 0.8:  # High variation
        w_reciprocal = 0.2
        w_spectral = 0.5
        w_geometric = 0.2
        w_structural = 0.1
    else:  # Very high variation
        w_reciprocal = 0.1
        w_spectral = 0.6
        w_geometric = 0.2
        w_structural = 0.1
    
    # Apply skewness adjustment
    if abs(stats['skewness']) > 1.0:
        # Skewed distributions favor geometric heuristics
        w_geometric += 0.1
        w_reciprocal -= 0.1
    
    # Normalize weights
    total = w_reciprocal + w_spectral + w_geometric + w_structural
    w_reciprocal /= total
    w_spectral /= total
    w_geometric /= total
    w_structural /= total
    
    # Combine weighted heuristics
    fused = (w_reciprocal * reciprocal +
             w_spectral * spectral +
             w_geometric * geometric +
             w_structural * structural)
    
    # Add multiplicative interaction terms
    interaction = (reciprocal * geometric) * 0.5 + (spectral * structural) * 0.5
    fused = fused * 0.8 + interaction * 0.2
    
    # Ensure symmetry
    fused = (fused + fused.T) / 2
    
    return fused


def _quality_aware_sparsification(heuristics_mat: np.ndarray,
                                  dist_mat: np.ndarray,
                                  n: int, stats: dict) -> np.ndarray:
    """Sparsify while preserving solution diversity and quality."""
    if n <= 10:
        return heuristics_mat.copy()
    
    sparsified = np.zeros_like(heuristics_mat)
    
    # Adaptive sparsification based on instance characteristics
    if stats['cv'] < 0.3:
        density_factor = 0.3
        quality_threshold = 0.4
    elif stats['cv'] < 0.7:
        density_factor = 0.4
        quality_threshold = 0.3
    else:
        density_factor = 0.5
        quality_threshold = 0.2
    
    # Adjust based on size
    density_factor *= min(1.0, np.log10(n) / 2)
    
    # Per-node sparsification
    edges_per_node = max(3, int(density_factor * np.sqrt(n)))
    edges_per_node = min(edges_per_node, n - 1)
    
    for i in range(n):
        row = heuristics_mat[i].copy()
        row[i] = -np.inf
        
        # Keep top edges with quality threshold
        if edges_per_node < n - 1:
            kth_largest = np.partition(row, -edges_per_node)[-edges_per_node]
            mask = row >= max(kth_largest, quality_threshold)
        else:
            mask = row >= quality_threshold
        
        # Apply mask
        sparsified[i] = heuristics_mat[i] * mask
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Ensure connectivity
    connected = _ensure_connectivity(sparsified, dist_mat, n)
    
    return connected


def _ensure_connectivity(heuristics_mat: np.ndarray,
                         dist_mat: np.ndarray, n: int) -> np.ndarray:
    """Ensure the sparsified graph is connected."""
    connected = heuristics_mat.copy()
    
    # Quick connectivity check using BFS
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        node = stack.pop()
        neighbors = np.where(connected[node] > 0)[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    # If not connected, add minimal edges to connect components
    if not np.all(visited):
        # Find components
        components = []
        unvisited = set(range(n)) - set(np.where(visited)[0])
        
        while unvisited:
            start = unvisited.pop()
            comp = [start]
            stack = [start]
            local_visited = {start}
            
            while stack:
                node = stack.pop()
                neighbors = np.where(connected[node] > 0)[0]
                for neighbor in neighbors:
                    if neighbor not in local_visited:
                        local_visited.add(neighbor)
                        stack.append(neighbor)
                        comp.append(neighbor)
            
            components.append(comp)
            unvisited -= set(comp)
        
        # Connect components with nearest edges
        for i in range(len(components) - 1):
            comp1 = components[i]
            comp2 = components[i + 1]
            
            # Find closest pair between components
            min_dist = np.inf
            best_pair = (comp1[0], comp2[0])
            
            for node1 in comp1:
                for node2 in comp2:
                    if dist_mat[node1, node2] < min_dist:
                        min_dist = dist_mat[node1, node2]
                        best_pair = (node1, node2)
            
            # Add the connecting edge
            node1, node2 = best_pair
            connected[node1, node2] = connected[node2, node1] = 1.0 / (1.0 + min_dist)
    
    return connected


def _community_aware_boosting(sparse_mat: np.ndarray,
                              dist_mat: np.ndarray,
                              n: int, stats: dict) -> np.ndarray:
    """Boost edges based on community structure."""
    boosted = sparse_mat.copy()
    
    if n < 20:
        return boosted
    
    try:
        # Simple community detection via spectral clustering
        # Create adjacency matrix from heuristics
        adjacency = (sparse_mat > 0).astype(float)
        
        # Compute degree matrix
        degree = np.diag(np.sum(adjacency, axis=1))
        
        # Compute normalized Laplacian
        D_inv_sqrt = np.diag(1.0 / np.sqrt(np.diag(degree) + 1e-12))
        L_norm = np.eye(n) - D_inv_sqrt @ adjacency @ D_inv_sqrt
        
        # Get eigenvectors for clustering
        if n <= 500:
            eigvals, eigvecs = np.linalg.eigh(L_norm)
            
            # Use top k eigenvectors for clustering
            k = min(5, max(2, int(np.log2(n))))
            features = eigvecs[:, :k]
            
            # Simple k-means clustering
            from sklearn.cluster import KMeans
            kmeans = KMeans(n_clusters=k, random_state=42, n_init=5)
            labels = kmeans.fit_predict(features)
            
            # Boost intra-cluster edges and penalize inter-cluster edges
            for i in range(n):
                for j in range(i + 1, n):
                    if sparse_mat[i, j] > 0:
                        if labels[i] == labels[j]:
                            # Intra-cluster: boost
                            boost = 1.0 + stats['community_resolution']
                            boosted[i, j] = boosted[j, i] = sparse_mat[i, j] * boost
                        else:
                            # Inter-cluster: moderate penalty
                            penalty = 0.7
                            boosted[i, j] = boosted[j, i] = sparse_mat[i, j] * penalty
        
        # For large instances, use fast approximate clustering
        else:
            # Use distance-based clustering
            landmarks = min(100, n // 10)
            landmark_indices = np.random.choice(n, landmarks, replace=False)
            
            # Cluster landmarks
            landmark_dists = dist_mat[landmark_indices][:, landmark_indices]
            
            from sklearn.cluster import AgglomerativeClustering
            clustering = AgglomerativeClustering(
                n_clusters=None,
                distance_threshold=stats['median_dist'] * 2,
                linkage='average'
            )
            landmark_labels = clustering.fit_predict(landmark_dists)
            
            # Assign nodes to nearest landmark's cluster
            node_labels = np.zeros(n, dtype=int)
            for i in range(n):
                dists_to_landmarks = dist_mat[i, landmark_indices]
                nearest_landmark = np.argmin(dists_to_landmarks)
                node_labels[i] = landmark_labels[nearest_landmark]
            
            # Apply community-aware boosting
            for i in range(n):
                for j in range(i + 1, n):
                    if sparse_mat[i, j] > 0:
                        if node_labels[i] == node_labels[j]:
                            boost = 1.0 + stats['community_resolution'] * 0.5
                            boosted[i, j] = boosted[j, i] = sparse_mat[i, j] * boost
    
    except:
        # Fallback to triangle quality boosting
        pass
    
    return boosted


def _final_refinement(boosted_mat: np.ndarray,
                      dist_mat: np.ndarray,
                      n: int, stats: dict) -> np.ndarray:
    """Final refinement with local optimization."""
    refined = boosted_mat.copy()
    
    # Local edge swapping heuristic
    if n >= 10:
        iterations = min(3, int(np.log2(n)))
        
        for _ in range(iterations):
            for i in range(n):
                # Get current promising edges
                current_edges = np.where(refined[i] > 0)[0]
                if len(current_edges) < 2:
                    continue
                
                # Consider swapping to better edges
                for j in current_edges:
                    if j <= i:
                        continue
                    
                    # Find alternative edge that might be better
                    for k in range(n):
                        if k != i and k != j and refined[i, k] == 0:
                            # Check if (i,k) + (k,j) is better than (i,j)
                            current_length = dist_mat[i, j]
                            alternative_length = dist_mat[i, k] + dist_mat[k, j]
                            
                            if alternative_length < current_length * 1.1:
                                # Consider adding this edge
                                refined[i, k] = refined[k, i] = max(
                                    refined[i, k],
                                    0.8 * refined[i, j]
                                )
    
    # Smooth the matrix
    alpha = 0.7
    refined = alpha * refined + (1 - alpha) * (refined + refined.T) / 2
    
    # Ensure non-negativity
    refined = np.maximum(refined, 0)
    
    return refined
