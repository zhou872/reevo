import numpy as np
from scipy.spatial.distance import pdist, squareform
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection.
    
    Returns a matrix of edge promises combining:
    1. Reciprocal of distance (closer nodes are more promising)
    2. Minimum spanning tree edges
    3. Local neighborhood connectivity
    4. Sparsification of unpromising edges
    
    Args:
        distance_matrix: Square distance matrix
    
    Returns:
        Heuristic matrix of same shape indicating edge promise
    """
    n = distance_matrix.shape[0]
    
    # 1. Base reciprocal heuristic
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        reciprocal[np.isnan(reciprocal)] = 0
    
    # 2. Minimum spanning tree edges get boosted weight
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)  # Symmetric MST edges
    mst_boost = 2.0  # Boost factor for MST edges
    
    # 3. Local neighborhood analysis
    # For each node, identify top k nearest neighbors
    k = max(3, int(np.sqrt(n)))  # Adaptive k based on problem size
    local_promises = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        distances = distance_matrix[i].copy()
        distances[i] = np.inf  # Exclude self
        nearest_indices = np.argpartition(distances, k)[:k]
        
        # Assign higher promise to nearest neighbors
        local_promises[i, nearest_indices] = 1.0
        
        # Additional boost for the very closest neighbor
        if len(nearest_indices) > 0:
            closest_idx = nearest_indices[0]
            local_promises[i, closest_idx] = 1.5
    
    # 4. Degree-based weighting
    # Nodes with fewer good connections should have their edges boosted
    node_promises = np.sum(local_promises, axis=1)
    degree_weights = 1.0 + (1.0 / (node_promises + 1e-8))  # Inverse weighting
    
    # Create degree weight matrix
    degree_matrix = np.outer(degree_weights, degree_weights)
    
    # 5. Combine all factors
    combined = reciprocal.copy()
    combined *= degree_matrix  # Apply degree weighting
    
    # Boost MST edges
    combined[mst_mask] *= mst_boost
    
    # Incorporate local neighborhood promises
    combined += 0.3 * local_promises
    
    # 6. Sparsification: keep only promising edges
    # Threshold based on percentile of edge promises
    if n > 10:  # Only sparsify for larger problems
        flat_combined = combined[np.triu_indices(n, k=1)]
        threshold = np.percentile(flat_combined, 70)  # Keep top 30%
        
        # Create sparsity mask
        sparsity_mask = combined >= threshold
        sparsity_mask = sparsity_mask | sparsity_mask.T  # Make symmetric
        np.fill_diagonal(sparsity_mask, False)  # No self-loops
        
        # Apply sparsity mask
        combined = combined * sparsity_mask
    
    # 7. Normalize to reasonable range
    max_val = np.max(combined[np.triu_indices(n, k=1)])
    if max_val > 0:
        combined /= max_val
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    # Zero out diagonal
    np.fill_diagonal(combined, 0)
    
    return combined
