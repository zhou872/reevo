import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristic combining multiscale similarity measures,
    structural connectivity analysis, and size-adaptive sparsification.
    
    Returns a matrix of edge promises with same shape as distance_matrix,
    where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n < 3:
        with np.errstate(divide='ignore', invalid='ignore'):
            result = 1 / (distance_matrix + np.eye(n) * 1e-12)
            np.fill_diagonal(result, 0)
            return result
    
    # ----- 1. Base Proximity Factor -----
    with np.errstate(divide='ignore', invalid='ignore'):
        base_promises = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(base_promises, 0)
    
    # ----- 2. Multiscale Similarity Matrix -----
    # Compute similarity at different scales to capture both local and global structure
    similarity_matrix = _compute_multiscale_similarity(distance_matrix, n)
    
    # ----- 3. Structural Connectivity -----
    # Emphasize edges that provide structural connectivity
    structural_importance = _compute_structural_importance(distance_matrix, n)
    
    # ----- 4. Triangle Clustering Boost (Enhanced) -----
    # Strongly boost edges that form triangles (common neighbors)
    triangle_boost = _enhanced_triangle_clustering(distance_matrix, n)
    
    # ----- 5. Combine Factors with Adaptive Weights -----
    combined = _adaptive_weighted_combination(
        base_promises, similarity_matrix, structural_importance, triangle_boost, n
    )
    
    # ----- 6. Size-Adaptive Sparsification -----
    # Keep only the most promising edges per node, with adaptive count
    sparsified = _size_adaptive_sparsification(combined, n)
    
    # ----- 7. Final Normalization -----
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    np.fill_diagonal(sparsified, 0)
    return sparsified


def _compute_multiscale_similarity(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute similarity at multiple scales to capture local and global structure."""
    similarity = np.zeros_like(distance_matrix)
    
    # Scale 1: Direct reciprocal (local proximity)
    with np.errstate(divide='ignore', invalid='ignore'):
        scale1 = 1 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(scale1, 0)
    
    # Scale 2: Two-hop similarity via intermediate nodes
    scale2 = np.zeros_like(distance_matrix)
    max_dist = np.max(distance_matrix)
    normalized_dist = distance_matrix / (max_dist + 1e-12)
    
    # For each node, find similar nodes through intermediate connections
    for i in range(n):
        for j in range(i + 1, n):
            # Find nodes that are close to both i and j
            common_neighbors = np.where(
                (normalized_dist[i] < 0.3) & (normalized_dist[j] < 0.3)
            )[0]
            if len(common_neighbors) > 0:
                # Similarity is higher if they share close common neighbors
                scale2[i, j] = scale2[j, i] = len(common_neighbors) / n
    
    # Scale 3: Global similarity based on distance distribution
    scale3 = np.zeros_like(distance_matrix)
    for i in range(n):
        dists_i = distance_matrix[i]
        for j in range(i + 1, n):
            dists_j = distance_matrix[j]
            # Pearson correlation of distance vectors (captures global similarity)
            corr = np.corrcoef(dists_i, dists_j)[0, 1]
            if not np.isnan(corr):
                scale3[i, j] = scale3[j, i] = max(0, corr)  # Only positive correlations
    
    # Combine scales with weights
    similarity = 0.5 * scale1 + 0.3 * scale2 + 0.2 * scale3
    return similarity


def _compute_structural_importance(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute importance of edges for maintaining structural connectivity."""
    importance = np.zeros_like(distance_matrix)
    
    # Minimum Spanning Tree edges are crucial
    mst = minimum_spanning_tree(distance_matrix)
    mst_binary = (mst.toarray() > 0).astype(float)
    importance += mst_binary + mst_binary.T  # Make symmetric
    
    # Also consider edges that connect different clusters
    # Identify potential clusters via local density
    local_density = np.zeros(n)
    for i in range(n):
        dists = np.sort(distance_matrix[i])[:5]  # 5 nearest neighbors
        local_density[i] = 1 / (np.mean(dists) + 1e-12)
    
    # Connect nodes with different local densities (inter-cluster edges)
    density_diff = np.abs(local_density[:, None] - local_density[None, :])
    max_diff = np.max(density_diff)
    if max_diff > 0:
        inter_cluster = density_diff / max_diff
        importance += 0.5 * inter_cluster
    
    # Boost edges that are part of many short paths
    # Approximate by considering 2-hop paths
    binary_near = (distance_matrix < np.percentile(distance_matrix, 20)).astype(float)
    np.fill_diagonal(binary_near, 0)
    path_count = binary_near @ binary_near  # Count of 2-hop paths via intermediate nodes
    importance += 0.3 * path_count / np.max(path_count + 1e-12)
    
    return importance


def _enhanced_triangle_clustering(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """
    Enhanced triangle clustering that considers both distance and triangle density.
    Implements the Triangle-Based Clustering Boost feature.
    """
    boost = np.ones_like(distance_matrix)
    
    if n > 500:
        # For large graphs, use sampling for efficiency
        sample_size = min(200, n)
        sample_nodes = np.random.choice(n, sample_size, replace=False)
        
        for i in sample_nodes:
            # Find candidate neighbors (closest 20% of nodes)
            k = max(3, n // 5)
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            neighbors = np.argpartition(dists_i, k)[:k]
            
            for j in neighbors:
                if j <= i:
                    continue
                
                # Find common neighbors between i and j
                dists_j = distance_matrix[j].copy()
                dists_j[j] = np.inf
                
                # Consider nodes close to both i and j
                threshold = min(np.percentile(dists_i, 30), np.percentile(dists_j, 30))
                common = np.where(
                    (distance_matrix[i] < threshold) & (distance_matrix[j] < threshold)
                )[0]
                common = common[(common != i) & (common != j)]
                
                if len(common) > 0:
                    # Boost factor depends on number of common neighbors AND their distances
                    triangle_quality = 0
                    for k_node in common:
                        # Quality measure: triangles with more balanced sides are better
                        d_ij = distance_matrix[i, j]
                        d_ik = distance_matrix[i, k_node]
                        d_jk = distance_matrix[j, k_node]
                        triangle_quality += 1 / (1 + np.std([d_ij, d_ik, d_jk]))
                    
                    boost_factor = 1 + 0.25 * len(common) + 0.1 * triangle_quality
                    boost[i, j] = boost[j, i] = boost_factor
    else:
        # Exact computation for smaller graphs
        # Create adjacency matrix for close neighbors
        threshold = np.percentile(distance_matrix, 25)  # Top 25% closest edges
        adjacency = (distance_matrix < threshold).astype(float)
        np.fill_diagonal(adjacency, 0)
        
        # Count triangles efficiently using matrix multiplication
        # (A³)_ii / 2 gives number of triangles involving node i
        triangles_per_node = np.diag(adjacency @ adjacency @ adjacency) / 2
        
        # For each edge, estimate triangle participation
        for i in range(n):
            for j in range(i + 1, n):
                if adjacency[i, j] > 0:
                    # Common neighbors that form triangles
                    common_neighbors = np.where(
                        (adjacency[i] > 0) & (adjacency[j] > 0)
                    )[0]
                    num_triangles = len(common_neighbors)
                    
                    if num_triangles > 0:
                        # Normalize by max possible triangles
                        max_possible = min(triangles_per_node[i], triangles_per_node[j])
                        if max_possible > 0:
                            triangle_ratio = num_triangles / max_possible
                            boost[i, j] = boost[j, i] = 1 + 0.4 * triangle_ratio
    
    return boost


def _adaptive_weighted_combination(base, similarity, structure, triangles, n: int) -> np.ndarray:
    """Combine factors with weights that adapt to graph size."""
    # Weights adapt to graph size
    if n < 50:
        weights = [0.3, 0.3, 0.2, 0.2]  # Balanced for small graphs
    elif n < 200:
        weights = [0.4, 0.3, 0.2, 0.1]  # More weight to base for medium
    else:
        weights = [0.5, 0.2, 0.2, 0.1]  # Strong base weight for large
    
    combined = (
        weights[0] * base / (np.max(base) + 1e-12) +
        weights[1] * similarity / (np.max(similarity) + 1e-12) +
        weights[2] * structure / (np.max(structure) + 1e-12) +
        weights[3] * triangles
    )
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    return combined


def _size_adaptive_sparsification(matrix: np.ndarray, n: int) -> np.ndarray:
    """
    Sparsify by keeping top edges per node, with count adapting to graph size.
    Implements the Size-Adaptive Sparsification feature.
    """
    if n <= 10:
        return matrix.copy()
    
    sparsified = np.zeros_like(matrix)
    
    # Dynamic edge count based on log10(n), with minimum of 3
    edges_per_node = max(3, int(np.log10(n) * 2.5))
    edges_per_node = min(edges_per_node, n - 1)
    
    for i in range(n):
        row = matrix[i].copy()
        row[i] = -np.inf  # Exclude self
        
        if np.any(row > 0):
            # Get indices of top k edges
            k = edges_per_node
            if n > 100:
                # For larger graphs, use percentile threshold
                threshold = np.percentile(row[row > 0], 100 * (1 - k/n))
                mask = row >= threshold
            else:
                # For smaller graphs, exact top k
                top_indices = np.argpartition(-row, k)[:k]
                mask = np.zeros_like(row, dtype=bool)
                mask[top_indices] = True
            
            sparsified[i] = matrix[i] * mask
    
    # Ensure symmetry by taking max of both directions
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # For very sparse matrices, ensure each node has at least 2 connections
    for i in range(n):
        if np.sum(sparsified[i] > 0) < 2:
            # Add the next best edge
            row = matrix[i].copy()
            row[i] = -np.inf
            row[sparsified[i] > 0] = -np.inf  # Exclude already selected
            if np.any(row > 0):
                best_idx = np.argmax(row)
                sparsified[i, best_idx] = matrix[i, best_idx]
                sparsified[best_idx, i] = matrix[best_idx, i]
    
    return sparsified
