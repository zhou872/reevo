import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors:
    1. Mutual nearest neighbor relationships
    2. Normalized inverse distances
    3. Sparsification of unpromising edges
    4. Edge centrality based on node connectivity
    """
    n = distance_matrix.shape[0]
    
    # 1. Basic inverse distance (handle zeros by adding epsilon)
    eps = 1e-10
    inv_dist = 1 / (distance_matrix + eps)
    np.fill_diagonal(inv_dist, 0)  # Ignore self-connections
    
    # 2. Mutual nearest neighbor indicator
    k = max(2, n // 10)  # Consider top 10% neighbors
    neighbor_rank = np.argsort(distance_matrix, axis=1)
    
    # Create mutual nearest neighbor mask
    mutual_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        # Get k nearest neighbors for node i (excluding self)
        i_neighbors = set(neighbor_rank[i, 1:k+1])
        for j in i_neighbors:
            # Check if i is in j's k nearest neighbors
            j_neighbors = set(neighbor_rank[j, 1:k+1])
            if i in j_neighbors:
                mutual_mask[i, j] = True
    
    # 3. Edge centrality: favor edges connecting nodes with high average distance to others
    avg_node_distance = np.mean(distance_matrix, axis=1)
    # Normalize and create edge centrality score
    edge_centrality = (avg_node_distance[:, None] + avg_node_distance[None, :]) / 2
    edge_centrality = edge_centrality / np.max(edge_centrality)
    
    # 4. Local density: penalize edges in dense regions
    local_density = np.zeros(n)
    for i in range(n):
        # Average distance to k nearest neighbors
        local_density[i] = np.mean(distance_matrix[i, neighbor_rank[i, 1:k+1]])
    density_factor = (local_density[:, None] + local_density[None, :]) / 2
    density_factor = density_factor / np.max(density_factor)
    
    # 5. Combine factors
    heuristics = np.zeros_like(distance_matrix)
    
    # Base score: normalized inverse distance
    norm_inv_dist = inv_dist / np.max(inv_dist)
    
    # Combined score with different weights
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            # Base score components
            score = norm_inv_dist[i, j]
            
            # Boost mutual nearest neighbors
            if mutual_mask[i, j]:
                score *= 2.0
            
            # Edge centrality: favor edges connecting remote nodes
            score *= (1.0 + edge_centrality[i, j])
            
            # Density adjustment: favor edges in sparse regions
            score *= (1.5 - 0.5 * density_factor[i, j])
            
            heuristics[i, j] = score
    
    # 6. Sparsification: keep only top promising edges
    # Determine threshold: keep edges with score above median among non-zero scores
    flat_scores = heuristics.flatten()
    non_zero_scores = flat_scores[flat_scores > 0]
    if len(non_zero_scores) > 0:
        threshold = np.percentile(non_zero_scores, 50)  # Median threshold
        heuristics[heuristics < threshold] = 0
    
    # 7. Normalize final heuristics
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
