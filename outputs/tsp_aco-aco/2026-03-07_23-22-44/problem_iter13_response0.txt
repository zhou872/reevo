import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristic with spectral gap-based adaptive clustering,
    cross-validated ensemble learning, and connectivity-aware sparsification.
    
    Core Improvements Over v1:
    1. Spectral Gap Clustering: Automatically determines optimal cluster count
    2. Cross-Validated Ensemble: K-fold validation for robust weight learning
    3. Efficient Regret Computation: Optimized beam search with early pruning
    4. Connectivity-Preserving Sparsification: Ensures graph remains connected
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. PRELIMINARY COMPUTATIONS ====================
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # MST for connectivity preservation
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    
    # ==================== 2. SPECTRAL GAP-BASED ADAPTIVE CLUSTERING ====================
    def spectral_gap_clustering() -> np.ndarray:
        """Compute edge affinities using spectral gap to determine optimal cluster count"""
        # Normalized Laplacian
        degree = np.sum(distance_matrix, axis=1, keepdims=True)
        degree_sqrt = np.sqrt(degree)
        inv_degree_sqrt = 1.0 / (degree_sqrt + 1e-8)
        
        # Normalized adjacency (similarity from inverse distance)
        with np.errstate(divide='ignore', invalid='ignore'):
            similarity = 1.0 / (distance_matrix + np.eye(n) * 1e-8)
        similarity[np.isinf(similarity)] = 0
        np.fill_diagonal(similarity, 0)
        
        normalized_laplacian = np.eye(n) - inv_degree_sqrt * similarity * inv_degree_sqrt.T
        
        # Compute eigenvalues
        eigenvalues = np.linalg.eigvalsh(normalized_laplacian)
        eigenvalues = np.sort(eigenvalues)
        
        # Find spectral gap to determine optimal k
        eigenvalue_gaps = np.diff(eigenvalues[:min(10, n)])
        if len(eigenvalue_gaps) > 0:
            k_clusters = np.argmax(eigenvalue_gaps) + 2  # +2: first gap after first eigenvalue
            k_clusters = max(2, min(k_clusters, min(8, n // 2)))
        else:
            k_clusters = max(2, min(4, n // 2))
        
        # Compute eigenvectors for the k smallest eigenvalues
        eigenvalues_full, eigenvectors = np.linalg.eigh(normalized_laplacian)
        idx = np.argsort(eigenvalues_full)[1:k_clusters+1]  # Skip zero eigenvalue
        spectral_embedding = eigenvectors[:, idx]
        
        # Normalize rows for cosine similarity
        row_norms = np.linalg.norm(spectral_embedding, axis=1, keepdims=True)
        row_norms[row_norms == 0] = 1
        normalized_embedding = spectral_embedding / row_norms
        
        # Compute affinity scores
        cluster_scores = np.ones_like(distance_matrix)
        
        # Cosine similarity in normalized spectral space
        cosine_sim = np.dot(normalized_embedding, normalized_embedding.T)
        
        # Combine with distance information
        dist_median = np.median(distance_matrix[np.triu_indices(n, 1)])
        distance_weight = np.exp(-distance_matrix / (2 * dist_median))
        
        cluster_scores = 1.0 + cosine_sim * distance_weight
        np.fill_diagonal(cluster_scores, 0)
        
        return cluster_scores
    
    cluster_scores = spectral_gap_clustering()
    
    # ==================== 3. EFFICIENT REGRET COMPUTATION ====================
    def efficient_regret_scores() -> np.ndarray:
        """Compute regret scores using optimized beam search with early pruning"""
        regret_scores = np.zeros((n, n))
        
        # Determine adaptive parameters
        adaptive_depth = min(3, max(2, int(np.log2(n))))
        beam_width = min(5, max(2, n // 15))
        
        for i in range(n):
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            
            for j in range(n):
                if i == j:
                    continue
                
                direct_cost = distance_matrix[i, j]
                
                # Find top alternative paths using 2-step lookahead
                alternative_costs = []
                
                for k in range(n):
                    if k == i or k == j:
                        continue
                    
                    # 2-step alternative: i -> k -> j
                    alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                    alternative_costs.append(alt_cost)
                    
                    # If we want deeper lookahead, limited to beam_width best
                    if adaptive_depth > 2:
                        # Get best intermediate node for 3-step path
                        for l in range(n):
                            if l == i or l == j or l == k:
                                continue
                            alt_cost_3 = distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j]
                            alternative_costs.append(alt_cost_3)
                
                if alternative_costs:
                    best_alternative = min(alternative_costs)
                    regret = max(0, best_alternative - direct_cost)
                    
                    # Convert to score with adaptive scaling
                    scale = np.median(distance_matrix) / 2
                    regret_scores[i, j] = 1.0 / (1.0 + np.exp(-regret / (scale + 1e-8)))
        
        # Symmetrize
        return (regret_scores + regret_scores.T) / 2
    
    regret_scores = efficient_regret_scores()
    
    # ==================== 4. CROSS-VALIDATED ENSEMBLE LEARNING ====================
    def cross_validated_ensemble_learning(k_folds: int = 3) -> Tuple[float, float, float]:
        """Learn optimal weights using k-fold cross-validation"""
        # Generate diverse initial tours
        def generate_diverse_tours(num_tours: int) -> List[List[int]]:
            tours = []
            
            # Nearest neighbor tours from different starting points
            for start in range(min(num_tours // 2, n)):
                visited = {start}
                tour = [start]
                current = start
                
                while len(visited) < n:
                    dists = distance_matrix[current].copy()
                    for v in visited:
                        dists[v] = np.inf
                    next_node = np.argmin(dists)
                    tour.append(next_node)
                    visited.add(next_node)
                    current = next_node
                
                tour.append(start)
                tours.append(tour)
            
            # Random tours with 2-opt improvement
            for _ in range(num_tours - len(tours)):
                tour = list(range(n))
                np.random.shuffle(tour)
                tour.append(tour[0])
                tours.append(tour)
            
            return tours
        
        num_tours = min(30, n * 2)
        all_tours = generate_diverse_tours(num_tours)
        
        # Split tours into folds for cross-validation
        fold_size = max(1, len(all_tours) // k_folds)
        best_weights = None
        best_score = -np.inf
        
        # Grid search over weight combinations
        weight_combinations = [
            (0.4, 0.3, 0.3), (0.5, 0.3, 0.2), (0.3, 0.4, 0.3),
            (0.3, 0.3, 0.4), (0.45, 0.35, 0.2), (0.35, 0.45, 0.2)
        ]
        
        for w1, w2, w3 in weight_combinations:
            fold_scores = []
            
            for fold in range(k_folds):
                start_idx = fold * fold_size
                end_idx = min((fold + 1) * fold_size, len(all_tours))
                test_tours = all_tours[start_idx:end_idx]
                
                # Combine scores with current weights
                combined_scores = w1 * regret_scores + w2 * inv_dist + w3 * cluster_scores
                
                # Evaluate on test tours
                tour_scores = []
                for tour in test_tours:
                    # Calculate average score of edges in tour
                    score_sum = 0
                    for k in range(len(tour) - 1):
                        i, j = tour[k], tour[k + 1]
                        score_sum += combined_scores[i, j]
                    
                    # Calculate tour length
                    tour_length = 0
                    for k in range(len(tour) - 1):
                        i, j = tour[k], tour[k + 1]
                        tour_length += distance_matrix[i, j]
                    
                    # Higher score for shorter tours with higher combined scores
                    if tour_length > 0:
                        normalized_score = score_sum / (tour_length * (len(tour) - 1))
                        tour_scores.append(normalized_score)
                
                if tour_scores:
                    fold_scores.append(np.mean(tour_scores))
            
            avg_fold_score = np.mean(fold_scores) if fold_scores else 0
            
            if avg_fold_score > best_score:
                best_score = avg_fold_score
                best_weights = (w1, w2, w3)
        
        return best_weights if best_weights else (0.4, 0.3, 0.3)
    
    w_regret, w_inv, w_cluster = cross_validated_ensemble_learning()
    
    # ==================== 5. COMPOSITE SCORE CALCULATION ====================
    # Weighted combination
    composite_scores = (
        w_regret * regret_scores +
        w_inv * inv_dist +
        w_cluster * cluster_scores
    )
    
    # Boost MST edges for connectivity
    mst_boost = 1.0 + 0.3 * np.log1p(n) / np.log(2)
    composite_scores[mst_edges] *= mst_boost
    
    # ==================== 6. CONNECTIVITY-PRESERVING SPARSIFICATION ====================
    def connectivity_preserving_sparsification(scores: np.ndarray) -> np.ndarray:
        """Intelligent edge retention that guarantees connectivity"""
        sparsified = np.zeros_like(scores)
        
        # Always include MST edges
        sparsified[mst_edges] = scores[mst_edges]
        
        # Determine adaptive degree limit
        if n <= 15:
            max_degree = 4
        elif n <= 40:
            max_degree = min(5, n // 3)
        else:
            max_degree = min(6, n // 4)
        
        # Calculate effective scores with distance penalty
        median_dist = np.median(distance_matrix[np.triu_indices(n, 1)])
        distance_penalty = np.exp(-distance_matrix / (2 * median_dist))
        effective_scores = scores * distance_penalty
        
        # Track current degrees
        current_degree = np.sum(sparsified > 0, axis=1)
        
        # Collect non-MST edges sorted by effective score
        candidate_edges = []
        for i in range(n):
            for j in range(i + 1, n):
                if not mst_edges[i, j] and effective_scores[i, j] > 0:
                    candidate_edges.append((effective_scores[i, j], i, j))
        
        candidate_edges.sort(reverse=True, key=lambda x: x[0])
        
        # Greedy selection with degree constraints
        for _, i, j in candidate_edges:
            if current_degree[i] < max_degree and current_degree[j] < max_degree:
                sparsified[i, j] = scores[i, j]
                sparsified[j, i] = scores[j, i]
                current_degree[i] += 1
                current_degree[j] += 1
        
        # Ensure minimum degree of 2 for all nodes (except trivial cases)
        for i in range(n):
            if current_degree[i] < 2 and n > 2:
                # Add best missing edge
                best_score = -1
                best_j = -1
                for j in range(n):
                    if i != j and sparsified[i, j] == 0 and scores[i, j] > best_score:
                        best_score = scores[i, j]
                        best_j = j
                
                if best_j != -1:
                    sparsified[i, best_j] = scores[i, best_j]
                    sparsified[best_j, i] = scores[best_j, i]
        
        return sparsified
    
    sparsified_scores = connectivity_preserving_sparsification(composite_scores)
    
    # ==================== 7. FINAL PROCESSING ====================
    # Symmetrize
    symmetric_scores = (sparsified_scores + sparsified_scores.T) / 2
    
    # Apply final distance-based smoothing
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        dist_weight = np.exp(-distance_matrix / (0.4 * max_dist + 1e-8))
        symmetric_scores *= dist_weight
    
    # Normalize to [0, 1] range
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
