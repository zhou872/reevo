import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
import math

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic scoring for TSP edges with adaptive parameter tuning and 
    enhanced bidirectional regret analysis.
    
    Key improvements over v1:
    1. Dynamic decay parameter adjustment based on distance variance
    2. Enhanced bidirectional regret using k-nearest-neighbor baselines
    3. Lightweight parameter tuning via problem characteristics analysis
    4. MST connectivity with edge reinforcement based on centrality
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    epsilon = 1e-10
    np.fill_diagonal(distance_matrix, np.inf)  # Temporarily mark diagonal for calculations
    
    # ===== 1. DYNAMIC PARAMETER TUNING =====
    # Analyze problem characteristics to adjust parameters
    flat_dists = distance_matrix[np.triu_indices(n, k=1)]
    if len(flat_dists) > 0:
        distance_mean = np.mean(flat_dists)
        distance_std = np.std(flat_dists)
        distance_cv = distance_std / (distance_mean + epsilon)  # Coefficient of variation
        
        # Determine problem type based on distance distribution
        is_regular = distance_cv < 0.3  # Low variance = regular problem
        
        # Dynamic decay factor based on variance
        if distance_cv < 0.2:
            decay_factor = 0.2  # Strong decay for very regular problems
        elif distance_cv < 0.5:
            decay_factor = 0.3  # Moderate decay
        else:
            decay_factor = 0.4  # Weak decay for irregular problems
        
        # Adaptive k for nearest neighbors
        k_neighbors = min(max(3, int(np.log2(n))), n - 1)
    else:
        decay_factor = 0.3
        k_neighbors = min(4, n - 1)
        is_regular = True
    
    # ===== 2. ENHANCED BIDIRECTIONAL REGRET =====
    regret_matrix = np.ones_like(distance_matrix)
    
    for i in range(n):
        # Get distances from node i
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        
        # Calculate k-nearest baseline for regret calculation
        if k_neighbors > 0 and n > k_neighbors:
            # Get k smallest distances (excluding self)
            k_smallest = np.partition(dists_i, k_neighbors)[:k_neighbors]
            baseline_i = np.mean(k_smallest[k_smallest < np.inf])
        else:
            baseline_i = np.min(dists_i)
        
        if baseline_i > 0:
            # Enhanced regret with smoothed calculation
            regret_i = (distance_matrix[i] - baseline_i) / (baseline_i + epsilon)
            # Apply dynamic exponential decay
            regret_matrix[i] *= np.exp(-regret_i * decay_factor)
    
    # Symmetrize regret using geometric mean
    regret_matrix = np.sqrt(regret_matrix * regret_matrix.T)
    np.fill_diagonal(regret_matrix, 1.0)
    
    # ===== 3. LOCAL DENSITY SCORING =====
    density_scores = np.ones_like(distance_matrix)
    
    if n >= 5:
        # Calculate adaptive local density for each node
        for i in range(n):
            # Get k nearest neighbors for density estimation
            neighbor_dists = np.partition(distance_matrix[i], k_neighbors)[:k_neighbors+1]
            valid_dists = neighbor_dists[neighbor_dists < np.inf]
            
            if len(valid_dists) > 0:
                local_density = np.median(valid_dists)
                
                if local_density > 0:
                    # Dynamic density ratio with upper bound
                    density_ratio = local_density / (distance_matrix[i] + epsilon)
                    
                    # Adjust scaling based on problem regularity
                    if is_regular:
                        max_ratio = 2.5  # Less aggressive for regular problems
                    else:
                        max_ratio = 3.5  # More aggressive for irregular problems
                    
                    density_scores[i] = np.minimum(density_ratio, max_ratio)
    
    # Symmetrize density scores
    density_scores = (density_scores + density_scores.T) / 2
    
    # ===== 4. ADAPTIVE EXPONENTIAL DECAY =====
    # Use distance percentiles for adaptive scaling
    if len(flat_dists) > 0:
        p25 = np.percentile(flat_dists, 25)
        p75 = np.percentile(flat_dists, 75)
        
        # Dynamic scaling based on interquartile range
        if p75 - p25 > 0:
            scale_factor = max(p25, epsilon)
            # Adjust scale based on problem characteristics
            if is_regular:
                scale_factor *= 0.25
            else:
                scale_factor *= 0.35
        else:
            scale_factor = max(np.min(flat_dists), epsilon) * 0.3
    else:
        scale_factor = 1.0 * 0.3
    
    distance_penalty = np.exp(-distance_matrix / scale_factor)
    np.fill_diagonal(distance_penalty, 0)
    
    # ===== 5. MST CONNECTIVITY WITH REINFORCEMENT =====
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # Calculate edge centrality in MST
    mst_centrality = np.zeros_like(distance_matrix)
    if n > 2:
        # Simple centrality: count how many MST edges connect to each node
        node_mst_degree = np.sum(mst_mask, axis=1)
        for i in range(n):
            for j in range(i+1, n):
                if mst_mask[i, j]:
                    # Boost edges that connect high-degree MST nodes
                    centrality = (node_mst_degree[i] + node_mst_degree[j]) / (2 * (n - 1))
                    mst_centrality[i, j] = mst_centrality[j, i] = 1.0 + centrality
    
    # ===== 6. BASE RECIPROCAL DISTANCE =====
    with np.errstate(divide='ignore', invalid='ignore'):
        base_scores = 1.0 / (distance_matrix + epsilon)
    np.fill_diagonal(base_scores, 0)
    
    # Normalize base scores
    max_base = np.max(base_scores)
    if max_base > 0:
        base_scores = base_scores / max_base
    
    # ===== 7. COMBINE COMPONENTS WITH DYNAMIC WEIGHTING =====
    # Adjust weights based on problem characteristics
    if is_regular:
        regret_weight = 0.35
        density_weight = 0.25
        penalty_weight = 0.30
        base_weight = 0.10
    else:
        regret_weight = 0.25
        density_weight = 0.35
        penalty_weight = 0.25
        base_weight = 0.15
    
    combined_scores = (
        base_weight * base_scores +
        regret_weight * regret_matrix +
        density_weight * density_scores +
        penalty_weight * distance_penalty
    )
    
    # Apply MST reinforcement
    combined_scores[mst_mask] *= mst_centrality[mst_mask]
    
    # Ensure minimum value for MST edges
    combined_scores[mst_mask] = np.maximum(combined_scores[mst_mask], 1e-4)
    
    # ===== 8. INTELLIGENT SPARSIFICATION =====
    sparsified = np.zeros_like(combined_scores)
    
    for i in range(n):
        row_scores = combined_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Always preserve MST edges
        mst_edges = mst_mask[i]
        
        # Adaptive candidate pool size with logarithmic scaling
        if n <= 5:
            candidate_count = min(3, n - 1)
        else:
            candidate_count = min(int(2.5 * math.log2(n)), n - 1)
            candidate_count = max(3, candidate_count)
        
        # Preserve MST edges plus top candidates
        mst_count = np.sum(mst_edges)
        non_mst_needed = max(0, candidate_count - mst_count)
        
        if non_mst_needed > 0:
            # Get non-MST scores
            non_mst_mask = ~mst_edges
            non_mst_mask[i] = False
            non_mst_scores = row_scores.copy()
            non_mst_scores[~non_mst_mask] = -np.inf
            
            if np.any(non_mst_scores > -np.inf):
                valid_scores = non_mst_scores[non_mst_scores > -np.inf]
                if len(valid_scores) >= non_mst_needed:
                    # Use threshold-based selection
                    threshold = np.partition(valid_scores, -non_mst_needed)[-non_mst_needed]
                    top_non_mst = non_mst_scores >= threshold
                else:
                    # Keep all non-MST if insufficient
                    top_non_mst = non_mst_mask
                
                keep_mask = mst_edges | top_non_mst
            else:
                keep_mask = mst_edges
        else:
            keep_mask = mst_edges
        
        # Ensure we don't keep self-edges
        keep_mask[i] = False
        sparsified[i][keep_mask] = combined_scores[i][keep_mask]
    
    # ===== 9. FINAL PROCESSING =====
    # Symmetrize
    final_scores = np.maximum(sparsified, sparsified.T)
    
    # Normalize
    max_val = np.max(final_scores)
    if max_val > 0:
        final_scores = final_scores / max_val
    
    # Ensure diagonal is zero and apply final clipping
    np.fill_diagonal(final_scores, 0)
    final_scores = np.clip(final_scores, 0, 1)
    
    # Restore original diagonal
    np.fill_diagonal(distance_matrix, 0)
    
    return final_scores
