import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform
from typing import Tuple, List, Dict
import itertools

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Next-generation TSP heuristic integrating multi-scale regret sampling,
    meta-heuristic weight optimization, graph neural feature extraction,
    and dynamic edge priors from solution structures.
    
    Core Innovations:
    1. Multi-scale Regret Sampling: Adaptive regret computation at multiple resolution levels
    2. Meta-heuristic Weight Optimization: Simulated annealing for component weighting
    3. Graph Neural Feature Extraction: Neural-inspired feature computation without training
    4. Dynamic Edge Priors: Real-time edge importance updates from sampled solutions
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. PRELIMINARY COMPUTATIONS ====================
    # Enhanced inverse distance with power law
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix ** 1.2)
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # MST with additional connectivity information
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    
    # Compute graph statistics
    degree_centrality = np.sum(distance_matrix < np.median(distance_matrix), axis=1)
    degree_centrality = degree_centrality / np.max(degree_centrality) if np.max(degree_centrality) > 0 else np.ones(n)
    
    # ==================== 2. MULTI-SCALE REGRET SAMPLING ====================
    def multi_scale_regret() -> np.ndarray:
        """Compute regret at multiple resolution levels and combine adaptively"""
        regret_scores = np.zeros((n, n))
        
        # Level 1: Local regret (1-step alternatives)
        local_regret = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                direct = distance_matrix[i, j]
                # Find best 2-step alternative
                mask = np.ones(n, dtype=bool)
                mask[i] = mask[j] = False
                if np.any(mask):
                    alternatives = distance_matrix[i, mask] + distance_matrix[mask, j]
                    best_alt = np.min(alternatives)
                    local_regret[i, j] = max(0, best_alt - direct) / (direct + 1e-8)
        
        # Level 2: Regional regret (using graph partitioning)
        regional_regret = np.zeros((n, n))
        # Simple partitioning based on distance quartiles
        dist_quartile = np.percentile(distance_matrix[np.triu_indices(n, 1)], 50)
        clusters = {}
        for i in range(n):
            cluster_id = 0
            if i not in clusters:
                # Grow cluster from i
                clusters[i] = cluster_id
                for j in range(n):
                    if i != j and distance_matrix[i, j] < dist_quartile:
                        clusters[j] = cluster_id
                cluster_id += 1
        
        # Compute inter-cluster and intra-cluster regret differently
        for i in range(n):
            for j in range(i + 1, n):
                if clusters.get(i, -1) == clusters.get(j, -1):
                    # Same cluster: consider cluster-external alternatives
                    mask = np.array([clusters.get(k, -1) != clusters.get(i, -1) for k in range(n)])
                    if np.any(mask):
                        alternatives = distance_matrix[i, mask] + distance_matrix[mask, j]
                        best_alt = np.min(alternatives) if len(alternatives) > 0 else np.inf
                        regional_regret[i, j] = max(0, best_alt - distance_matrix[i, j]) / (distance_matrix[i, j] + 1e-8)
                else:
                    # Different clusters: consider bridge alternatives
                    bridge_nodes = [k for k in range(n) if clusters.get(k, -1) not in [clusters.get(i, -1), clusters.get(j, -1)]]
                    if len(bridge_nodes) > 0:
                        bridge_dists = []
                        for k in bridge_nodes:
                            bridge_dists.append(distance_matrix[i, k] + distance_matrix[k, j])
                        best_bridge = min(bridge_dists) if bridge_dists else np.inf
                        regional_regret[i, j] = max(0, best_bridge - distance_matrix[i, j]) / (distance_matrix[i, j] + 1e-8)
        
        # Combine levels with adaptive weights
        local_weight = min(0.7, 0.3 + 0.4 * (n / 100))
        regional_weight = 1 - local_weight
        
        regret_scores = local_weight * local_regret + regional_weight * regional_regret
        
        # Convert regret to score (higher regret = more promising edge)
        regret_scores = np.exp(-regret_scores / (np.median(regret_scores[regret_scores > 0]) + 1e-8))
        
        # Symmetrize
        regret_scores = (regret_scores + regret_scores.T) / 2
        np.fill_diagonal(regret_scores, 0)
        
        return regret_scores
    
    regret_scores = multi_scale_regret()
    
    # ==================== 3. GRAPH NEURAL FEATURE EXTRACTION ====================
    def neural_features() -> np.ndarray:
        """Compute neural-inspired features for edges without training"""
        feature_scores = np.ones((n, n))
        
        # Feature 1: Edge betweenness approximation
        # Simulate random walks to estimate edge importance
        for i in range(n):
            for j in range(i + 1, n):
                # Estimate connectivity importance via triangle completion
                triangles = 0
                for k in range(n):
                    if k != i and k != j:
                        # Check if edge (i,j) completes triangles
                        if (distance_matrix[i, k] + distance_matrix[k, j] < 1.5 * distance_matrix[i, j]):
                            triangles += 1
                triangle_score = triangles / (n - 2) if n > 2 else 0
                
                # Feature 2: Local clustering coefficient around edge
                neighbors_i = set(np.where(distance_matrix[i] < np.percentile(distance_matrix[i], 30))[0]) - {i}
                neighbors_j = set(np.where(distance_matrix[j] < np.percentile(distance_matrix[j], 30))[0]) - {j}
                common_neighbors = neighbors_i.intersection(neighbors_j)
                clustering_score = len(common_neighbors) / max(len(neighbors_i.union(neighbors_j)), 1)
                
                # Feature 3: Distance rank feature
                rank_i = np.sum(distance_matrix[i, :] < distance_matrix[i, j]) / n
                rank_j = np.sum(distance_matrix[j, :] < distance_matrix[j, i]) / n
                rank_score = 1 - (rank_i + rank_j) / 2
                
                # Combine features
                feature_scores[i, j] = 1.0 + 0.4 * triangle_score + 0.3 * clustering_score + 0.3 * rank_score
                feature_scores[j, i] = feature_scores[i, j]
        
        return feature_scores
    
    neural_scores = neural_features()
    
    # ==================== 4. META-HEURISTIC WEIGHT OPTIMIZATION ====================
    def meta_heuristic_weight_optimization() -> Tuple[float, float, float, float]:
        """Use simulated annealing to optimize component weights"""
        # Component pool
        components = {
            'regret': regret_scores,
            'inv_dist': inv_dist / np.max(inv_dist) if np.max(inv_dist) > 0 else inv_dist,
            'neural': neural_scores / np.max(neural_scores) if np.max(neural_scores) > 0 else neural_scores,
            'mst': mst_edges.astype(float)
        }
        
        # Initial weights
        weights = np.array([0.35, 0.25, 0.25, 0.15])
        best_weights = weights.copy()
        
        # Generate evaluation tours using fast constructive heuristics
        def evaluate_weights(w: np.ndarray) -> float:
            """Evaluate weight combination by building tours"""
            # Combine components
            combined = np.zeros_like(distance_matrix)
            for idx, key in enumerate(['regret', 'inv_dist', 'neural', 'mst']):
                combined += w[idx] * components[key]
            
            # Build tours using greedy algorithm guided by combined scores
            tour_lengths = []
            for start in range(min(5, n)):  # Try multiple starting points
                visited = [start]
                current = start
                
                while len(visited) < n:
                    available = [node for node in range(n) if node not in visited]
                    if not available:
                        break
                    
                    # Select next node based on combined score
                    scores = [combined[current, node] for node in available]
                    next_node = available[np.argmax(scores)]
                    visited.append(next_node)
                    current = next_node
                
                # Complete tour
                visited.append(start)
                
                # Calculate tour length
                length = sum(distance_matrix[visited[i], visited[i+1]] for i in range(len(visited)-1))
                tour_lengths.append(length)
            
            return -np.mean(tour_lengths)  # Negative because we want to minimize length
        
        # Simulated annealing
        current_score = evaluate_weights(weights)
        best_score = current_score
        temperature = 1.0
        cooling_rate = 0.95
        iterations = 20
        
        for _ in range(iterations):
            # Perturb weights
            new_weights = weights + np.random.normal(0, 0.1, 4)
            new_weights = np.maximum(new_weights, 0)
            new_weights = new_weights / np.sum(new_weights)  # Normalize
            
            new_score = evaluate_weights(new_weights)
            
            # Acceptance probability
            if new_score > current_score or np.random.random() < np.exp((new_score - current_score) / temperature):
                weights = new_weights
                current_score = new_score
                
                if current_score > best_score:
                    best_score = current_score
                    best_weights = weights.copy()
            
            temperature *= cooling_rate
        
        return tuple(best_weights)
    
    w_regret, w_inv, w_neural, w_mst = meta_heuristic_weight_optimization()
    
    # ==================== 5. DYNAMIC EDGE PRIORS ====================
    def dynamic_edge_priors(base_scores: np.ndarray) -> np.ndarray:
        """Update edge scores based on sampled high-quality solutions"""
        enhanced = base_scores.copy()
        
        # Generate diverse initial solutions
        solutions = []
        
        # Solution 1: Nearest neighbor from random start
        start = np.random.randint(0, n)
        visited = [start]
        current = start
        while len(visited) < n:
            available = [node for node in range(n) if node not in visited]
            next_node = min(available, key=lambda x: distance_matrix[current, x])
            visited.append(next_node)
            current = next_node
        visited.append(start)
        solutions.append(visited)
        
        # Solution 2: Farthest insertion
        if n > 3:
            start = np.random.randint(0, n)
            tour = [start]
            while len(tour) < n:
                max_dist = -1
                next_node = -1
                insert_pos = -1
                
                for node in set(range(n)) - set(tour):
                    for pos in range(len(tour)):
                        i, j = tour[pos], tour[(pos + 1) % len(tour)]
                        gain = distance_matrix[i, node] + distance_matrix[node, j] - distance_matrix[i, j]
                        if gain > max_dist:
                            max_dist = gain
                            next_node = node
                            insert_pos = pos + 1
                
                if next_node != -1:
                    tour.insert(insert_pos, next_node)
            
            tour.append(tour[0])
            solutions.append(tour)
        
        # Solution 3: Random but guided by current scores
        if n > 5:
            tour = list(range(n))
            np.random.shuffle(tour)
            # Apply 2-opt local search
            improved = True
            while improved:
                improved = False
                for i in range(n - 1):
                    for j in range(i + 2, n):
                        a, b, c, d = tour[i], tour[i+1], tour[j], tour[(j+1)%n]
                        current_cost = distance_matrix[a,b] + distance_matrix[c,d]
                        new_cost = distance_matrix[a,c] + distance_matrix[b,d]
                        if new_cost < current_cost:
                            tour[i+1:j+1] = reversed(tour[i+1:j+1])
                            improved = True
            tour.append(tour[0])
            solutions.append(tour)
        
        # Update scores based on solution edges
        edge_counts = np.zeros_like(enhanced)
        for tour in solutions:
            for idx in range(len(tour) - 1):
                i, j = tour[idx], tour[idx + 1]
                edge_counts[i, j] += 1
                edge_counts[j, i] += 1
        
        # Apply reinforcement
        if np.max(edge_counts) > 0:
            reinforcement = 1.0 + 0.5 * (edge_counts / np.max(edge_counts))
            enhanced *= reinforcement
        
        # MST reinforcement
        mst_reinforcement = np.where(mst_edges, 1.0 + np.log1p(n) / 8.0, 1.0)
        enhanced *= mst_reinforcement
        
        return enhanced
    
    # ==================== 6. COMPOSITE SCORE CALCULATION ====================
    # Weighted combination
    base_composite = (
        w_regret * regret_scores +
        w_inv * inv_dist +
        w_neural * neural_scores +
        w_mst * mst_edges.astype(float)
    )
    
    # Apply dynamic edge priors
    enhanced_composite = dynamic_edge_priors(base_composite)
    
    # ==================== 7. INTELLIGENT SPARSIFICATION ====================
    def intelligent_sparsification(scores: np.ndarray) -> np.ndarray:
        """Select edges preserving connectivity while focusing on promising ones"""
        sparsified = np.zeros_like(scores)
        
        # Always keep MST edges
        sparsified[mst_edges] = scores[mst_edges]
        
        # Adaptive degree limits
        if n <= 10:
            target_degree = min(4, n - 1)
        elif n <= 50:
            target_degree = min(5, n // 3)
        else:
            target_degree = min(6, n // 4)
        
        # Select additional edges
        candidates = []
        for i in range(n):
            for j in range(i + 1, n):
                if not mst_edges[i, j]:
                    # Score adjusted by local graph structure
                    local_density = degree_centrality[i] * degree_centrality[j]
                    adjusted_score = scores[i, j] * (1.0 + 0.2 * local_density)
                    candidates.append((adjusted_score, i, j))
        
        # Sort by adjusted score
        candidates.sort(reverse=True, key=lambda x: x[0])
        
        current_degree = np.sum(sparsified > 0, axis=1)
        for score, i, j in candidates:
            if current_degree[i] < target_degree and current_degree[j] < target_degree:
                sparsified[i, j] = scores[i, j]
                sparsified[j, i] = scores[i, j]
                current_degree[i] += 1
                current_degree[j] += 1
        
        return sparsified
    
    sparsified_scores = intelligent_sparsification(enhanced_composite)
    
    # ==================== 8. FINAL ENHANCEMENTS ====================
    # Ensure symmetry
    final_scores = (sparsified_scores + sparsified_scores.T) / 2
    
    # Distance-based refinement
    median_dist = np.median(distance_matrix[np.triu_indices(n, 1)])
    if median_dist > 0:
        dist_factor = np.exp(-distance_matrix / (2 * median_dist))
        final_scores *= dist_factor
    
    # Normalize
    max_score = np.max(final_scores)
    if max_score > 0:
        final_scores = final_scores / max_score
    
    np.fill_diagonal(final_scores, 0)
    
    return final_scores
