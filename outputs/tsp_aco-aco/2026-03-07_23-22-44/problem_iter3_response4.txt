import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with streamlined computation and improved sparsification.
    
    Key improvements over v1:
    1. Simplified exponential penalty using max distance scaling
    2. More efficient MST-based sparsification with adaptive threshold
    3. Direct regret computation without iterative loops
    4. Streamlined density estimation using fixed k-nearest neighbors
    5. Reduced computational complexity while maintaining effectiveness
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance with robust handling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Simplified local density estimation using fixed k
    k = min(5, n - 1)
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if k > 0 and len(dists) >= k:
            k_nearest = np.partition(dists, k)[:k]
            local_density[i] = 1.0 / (np.mean(k_nearest) + 1e-8)
    
    density_factor = np.sqrt(np.outer(local_density, local_density))
    
    # 3. Efficient regret-based scoring
    regret_scores = np.ones_like(distance_matrix)
    
    # Get two smallest distances for each node
    dists_temp = distance_matrix.copy()
    np.fill_diagonal(dists_temp, np.inf)
    
    if n > 2:
        # Find indices of two smallest distances for each row
        idx_partition = np.argpartition(dists_temp, 2, axis=1)
        
        # Extract the two smallest distances
        row_indices = np.arange(n)[:, None]
        d1 = dists_temp[row_indices, idx_partition[:, 0]]
        d2 = dists_temp[row_indices, idx_partition[:, 1]]
        
        # Calculate regret and apply bonus to best edge
        valid_mask = d1 > 0
        regret = np.zeros(n)
        regret[valid_mask] = (d2[valid_mask] - d1[valid_mask]) / (d1[valid_mask] + 1e-8)
        
        # Apply bounded regret bonus
        regret_bonus = 0.3 * np.tanh(regret)
        for i in range(n):
            if valid_mask[i]:
                best_idx = idx_partition[i, 0]
                regret_scores[i, best_idx] += regret_bonus[i]
    
    # 4. MST-based connectivity scoring
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = mst > 0
    
    # Symmetrize MST mask
    mst_mask_sym = mst_mask | mst_mask.T
    
    # 5. Combine base components
    base_scores = inv_dist * density_factor * regret_scores
    
    # Boost MST edges
    mst_boost = 1.0 + 0.8 * mst_mask_sym  # MST edges get 1.8x boost
    base_scores *= mst_boost
    
    # 6. Adaptive sparsification with MST guarantee
    sparsified = np.zeros_like(base_scores)
    
    # Always keep MST edges
    sparsified[mst_mask_sym] = base_scores[mst_mask_sym]
    
    # For each node, keep top non-MST edges within adaptive threshold
    avg_mst_dist = np.mean(distance_matrix[mst_mask_sym])
    threshold = 3.0 * avg_mst_dist
    
    for i in range(n):
        # Get non-MST edges for this node
        non_mst_mask = ~mst_mask_sym[i] & (np.arange(n) != i)
        
        # Filter by distance threshold
        within_threshold = distance_matrix[i] < threshold
        candidate_mask = non_mst_mask & within_threshold
        
        if np.any(candidate_mask):
            # Get scores for candidate edges
            candidate_scores = base_scores[i, candidate_mask]
            
            # Determine number to keep adaptively
            keep_count = max(2, min(5, int(0.3 * np.sum(candidate_mask))))
            
            if len(candidate_scores) > keep_count:
                # Get indices of top scores
                top_indices = np.argpartition(candidate_scores, -keep_count)[-keep_count:]
                
                # Convert to original indices
                orig_indices = np.where(candidate_mask)[0][top_indices]
                sparsified[i, orig_indices] = base_scores[i, orig_indices]
            else:
                sparsified[i, candidate_mask] = base_scores[i, candidate_mask]
    
    # 7. Exponential penalty for long edges
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        length_penalty = np.exp(-distance_matrix / (0.5 * max_dist))
        final_scores = sparsified * length_penalty
    else:
        final_scores = sparsified
    
    # 8. Symmetrize and normalize
    symmetric_scores = np.maximum(final_scores, final_scores.T)
    
    # Final softmax-like normalization per node
    for i in range(n):
        row = symmetric_scores[i]
        row_sum = np.sum(row)
        if row_sum > 0:
            symmetric_scores[i] = row / row_sum
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
