import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Efficient TSP heuristic combining spectral clustering, regret-based scoring,
    MST connectivity, and adaptive sparsification.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. BASE COMPUTATIONS ====================
    # Inverse distance (closer nodes are better)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Minimum Spanning Tree for connectivity
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    
    # ==================== 2. SPECTRAL CLUSTERING AFFINITY ====================
    def spectral_affinity(k_clusters: int = None) -> np.ndarray:
        """Compute edge affinity using spectral clustering"""
        if n < 4:
            return np.ones_like(distance_matrix)
        
        if k_clusters is None:
            k_clusters = max(2, min(5, int(np.sqrt(n))))
        
        # Use similarity based on inverse distance (with regularization)
        similarity = 1.0 / (1.0 + distance_matrix / np.median(distance_matrix))
        np.fill_diagonal(similarity, 0)
        
        # Normalized Laplacian
        degree = np.sum(similarity, axis=1)
        D_sqrt_inv = np.diag(1.0 / np.sqrt(degree + 1e-8))
        laplacian = np.eye(n) - D_sqrt_inv @ similarity @ D_sqrt_inv
        
        # Compute eigenvectors corresponding to smallest eigenvalues
        try:
            eigenvalues, eigenvectors = np.linalg.eigh(laplacian)
            # Get k_clusters smallest non-zero eigenvectors
            idx = np.argsort(eigenvalues)[1:k_clusters+1]
            spectral_embedding = eigenvectors[:, idx]
        except np.linalg.LinAlgError:
            # Fallback if eigendecomposition fails
            return np.ones_like(distance_matrix)
        
        # Compute cosine similarity in spectral space
        norm_embedding = spectral_embedding / (np.linalg.norm(spectral_embedding, axis=1, keepdims=True) + 1e-8)
        spectral_sim = norm_embedding @ norm_embedding.T
        
        # Combine with distance attenuation
        median_dist = np.median(distance_matrix)
        dist_factor = np.exp(-distance_matrix / (2 * median_dist + 1e-8))
        
        return 1.0 + 0.5 * spectral_sim * dist_factor
    
    cluster_scores = spectral_affinity()
    
    # ==================== 3. NEAREST-NEIGHBOR REGRET SCORING ====================
    def regret_scoring(k_neighbors: int = 5) -> np.ndarray:
        """Compute regret scores based on nearest neighbor alternatives"""
        regret = np.zeros((n, n))
        
        for i in range(n):
            # Get distances from node i (excluding self)
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            
            # Sort distances to find k+1 nearest neighbors
            sorted_indices = np.argsort(dists_i)
            
            # Base regret calculation: compare each edge to k-th best alternative
            for j in range(n):
                if i == j:
                    continue
                
                direct_dist = distance_matrix[i, j]
                
                # Find k-th smallest distance (excluding edge i-j)
                k_best_idx = 0
                count = 0
                while count < min(k_neighbors, n-2):
                    if sorted_indices[k_best_idx] != j:
                        count += 1
                    k_best_idx += 1
                    if k_best_idx >= n-1:
                        break
                
                k_best_dist = dists_i[sorted_indices[k_best_idx-1]]
                
                # Regret is how much worse the k-th best alternative is
                if k_best_dist > direct_dist:
                    regret_val = (k_best_dist - direct_dist) / (direct_dist + 1e-8)
                    regret[i, j] = 1.0 - np.exp(-regret_val / 2)
        
        # Symmetrize and normalize
        regret = (regret + regret.T) / 2
        max_regret = np.max(regret)
        if max_regret > 0:
            regret = regret / max_regret
        
        return regret
    
    # Adaptive k based on problem size
    k_regret = max(3, min(7, n // 5))
    regret_scores = regret_scoring(k_neighbors=k_regret)
    
    # ==================== 4. COMPOSITE SCORE ====================
    # Weighted combination with adaptive weights
    if n <= 10:
        w_cluster, w_regret, w_inv = 0.3, 0.4, 0.3
    elif n <= 50:
        w_cluster, w_regret, w_inv = 0.4, 0.3, 0.3
    else:
        w_cluster, w_regret, w_inv = 0.5, 0.3, 0.2
    
    composite = (
        w_cluster * cluster_scores +
        w_regret * regret_scores +
        w_inv * inv_dist / (np.max(inv_dist) + 1e-8)
    )
    
    # Boost MST edges for connectivity
    mst_boost = 1.0 + 0.3 * mst_edges
    composite *= mst_boost
    
    # ==================== 5. ADAPTIVE SPARSIFICATION ====================
    def intelligent_sparsify(scores: np.ndarray) -> np.ndarray:
        """Sparsify matrix while preserving connectivity"""
        sparsified = np.zeros_like(scores)
        
        # Always include MST edges
        sparsified[mst_edges] = scores[mst_edges]
        
        # Determine degree limits adaptively
        if n <= 10:
            max_degree = min(4, n-1)
        elif n <= 50:
            max_degree = min(5, n // 3)
        else:
            max_degree = min(6, n // 4)
        
        current_degree = np.sum(sparsified > 0, axis=1)
        
        # Candidate edges (excluding MST edges and diagonal)
        candidates = []
        for i in range(n):
            for j in range(i+1, n):
                if not mst_edges[i, j] and scores[i, j] > 0:
                    # Weight by both score and distance proximity
                    dist_weight = np.exp(-distance_matrix[i, j] / np.median(distance_matrix))
                    priority = scores[i, j] * dist_weight
                    candidates.append((priority, i, j))
        
        # Sort by priority
        candidates.sort(reverse=True, key=lambda x: x[0])
        
        # Select edges respecting degree constraints
        for priority, i, j in candidates:
            if current_degree[i] < max_degree and current_degree[j] < max_degree:
                sparsified[i, j] = scores[i, j]
                sparsified[j, i] = scores[i, j]
                current_degree[i] += 1
                current_degree[j] += 1
        
        return sparsified
    
    sparsified = intelligent_sparsify(composite)
    
    # ==================== 6. FINAL PROCESSING ====================
    # Ensure symmetry
    final_scores = (sparsified + sparsified.T) / 2
    
    # Distance-based attenuation
    if np.max(distance_matrix) > 0:
        scale = np.median(distance_matrix)
        dist_decay = np.exp(-distance_matrix / (scale + 1e-8))
        final_scores *= dist_decay
    
    # Normalize to [0, 1]
    max_score = np.max(final_scores)
    if max_score > 0:
        final_scores = final_scores / max_score
    
    np.fill_diagonal(final_scores, 0)
    
    return final_scores
