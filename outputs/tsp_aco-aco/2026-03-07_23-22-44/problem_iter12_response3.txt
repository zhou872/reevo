import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.linalg import eigh
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic with dynamic neural graph features, adaptive lookahead,
    and hierarchical partitioning for improved edge scoring.
    
    Core Innovations:
    1. Dynamic Graph Neural Features: Adaptively extracts structural patterns via 
       multi-scale graph convolution and attention mechanisms without training
    2. Adaptive Two-Step Lookahead: Regret-aware path evaluation with dynamic 
       caching of promising alternatives
    3. Hierarchical Spectral Partitioning: Multi-level cluster detection using 
       recursive spectral bisection with adaptive granularity
    4. Dynamic Weight Ensemble: Self-tuning combination weights based on problem 
       structure analysis
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    epsilon = 1e-12
    dist_copy = distance_matrix.copy()
    np.fill_diagonal(dist_copy, np.inf)
    
    # ===== 1. ENHANCED STATISTICAL ANALYSIS =====
    flat_dists = dist_copy[np.triu_indices(n, k=1)]
    dist_stats = {
        'mean': np.mean(flat_dists),
        'std': np.std(flat_dists),
        'skew': 0,
        'kurtosis': 0,
        'entropy': 0
    }
    
    if len(flat_dists) > 3:
        centered = flat_dists - dist_stats['mean']
        var = dist_stats['std'] ** 2
        if var > epsilon:
            dist_stats['skew'] = np.mean(centered ** 3) / (var ** 1.5)
            dist_stats['kurtosis'] = np.mean(centered ** 4) / (var ** 2) - 3
            
            # Distribution entropy estimation
            hist, _ = np.histogram(flat_dists, bins=min(20, len(flat_dists)//10))
            hist = hist[hist > 0] / len(flat_dists)
            dist_stats['entropy'] = -np.sum(hist * np.log(hist))
    
    cv = dist_stats['std'] / (dist_stats['mean'] + epsilon)
    
    # ===== 2. DYNAMIC PARAMETER INITIALIZATION =====
    # Enhanced classification with entropy-based features
    if cv < 0.25 and abs(dist_stats['skew']) < 0.3 and dist_stats['entropy'] > 1.5:
        params = {
            'lookahead_depth': 2,
            'cluster_granularity': 0.3,
            'regret_aggressiveness': 1.5,
            'neighbor_expansion': 0.6
        }
    elif cv > 0.8 and dist_stats['skew'] < -0.8 and dist_stats['kurtosis'] > 2:
        params = {
            'lookahead_depth': 3,
            'cluster_granularity': 0.8,
            'regret_aggressiveness': 3.0,
            'neighbor_expansion': 0.9
        }
    else:
        params = {
            'lookahead_depth': 2,
            'cluster_granularity': 0.5,
            'regret_aggressiveness': 2.0,
            'neighbor_expansion': 0.75
        }
    
    # ===== 3. GRAPH NEURAL FEATURE EXTRACTION =====
    def extract_graph_features():
        """Extract multi-scale graph features without training"""
        features = np.zeros((n, 5))  # 5 feature dimensions
        
        # Feature 1: Local neighborhood statistics
        k_local = min(5, n-1)
        for i in range(n):
            dists = dist_copy[i].copy()
            dists[i] = np.inf
            if k_local > 0:
                idx = np.argpartition(dists, k_local)[:k_local]
                features[i, 0] = np.mean(dists[idx])  # Mean distance to k nearest
                features[i, 1] = np.std(dists[idx])   # Std of distances
        
        # Feature 2: Spectral embedding using Fiedler vector
        try:
            sigma = np.median(dist_copy[dist_copy < np.inf])
            if sigma > 0:
                affinity = np.exp(-dist_copy / sigma)
                np.fill_diagonal(affinity, 0)
                
                degree = np.sum(affinity, axis=1)
                D_inv_sqrt = np.diag(1.0 / np.sqrt(degree + epsilon))
                L_norm = np.eye(n) - D_inv_sqrt @ affinity @ D_inv_sqrt
                
                # Compute top eigenvectors
                eigenvalues, eigenvectors = eigh(L_norm, subset_by_index=[0, min(4, n-1)])
                features[:, 2:5] = eigenvectors[:, 1:4]  # Skip first eigenvector
        except:
            pass
        
        return features
    
    graph_features = extract_graph_features()
    
    # ===== 4. ADAPTIVE TWO-STEP LOOKAHEAD WITH CACHING =====
    # Precompute neighbor expansion based on dynamic parameter
    k_expand = min(int(params['neighbor_expansion'] * np.log2(n) * 3), n-1)
    k_expand = max(4, min(k_expand, 15))
    
    nearest_idx = np.zeros((n, k_expand), dtype=int)
    nearest_dists = np.zeros((n, k_expand))
    
    for i in range(n):
        dists = dist_copy[i].copy()
        dists[i] = np.inf
        if k_expand > 0:
            idx = np.argpartition(dists, k_expand)[:k_expand]
            sorted_order = np.argsort(dists[idx])
            nearest_idx[i] = idx[sorted_order]
            nearest_dists[i] = dists[idx[sorted_order]]
    
    # Cache for lookahead path evaluations
    lookahead_cache = defaultdict(dict)
    
    def evaluate_lookahead(i, j, depth=2):
        """Evaluate alternative paths with caching"""
        cache_key = (min(i, j), max(i, j))
        if cache_key in lookahead_cache:
            return lookahead_cache[cache_key]
        
        direct = dist_copy[i, j]
        best_alternative = np.inf
        
        if depth == 1:
            # One-step lookahead (already in nearest neighbors)
            alt_costs = []
            for k in nearest_idx[i]:
                if k != j and dist_copy[i, k] < np.inf and dist_copy[k, j] < np.inf:
                    alt_costs.append(dist_copy[i, k] + dist_copy[k, j])
            
            for k in nearest_idx[j]:
                if k != i and dist_copy[i, k] < np.inf and dist_copy[k, j] < np.inf:
                    alt_costs.append(dist_copy[i, k] + dist_copy[k, j])
            
            if alt_costs:
                best_alternative = min(alt_costs)
        
        elif depth >= 2:
            # Two-step lookahead through intermediate pairs
            for k1 in nearest_idx[i][:min(4, k_expand)]:
                if k1 == j:
                    continue
                
                for k2 in nearest_idx[j][:min(4, k_expand)]:
                    if k2 == i or k2 == k1:
                        continue
                    
                    if (dist_copy[i, k1] < np.inf and 
                        dist_copy[k1, k2] < np.inf and 
                        dist_copy[k2, j] < np.inf):
                        
                        alt_cost = (dist_copy[i, k1] + 
                                  dist_copy[k1, k2] + 
                                  dist_copy[k2, j])
                        best_alternative = min(best_alternative, alt_cost)
        
        regret = best_alternative - direct if best_alternative < np.inf else 0
        lookahead_cache[cache_key] = regret
        return regret
    
    # ===== 5. HIERARCHICAL SPECTRAL PARTITIONING =====
    def hierarchical_clustering():
        """Recursive spectral bisection for multi-level clustering"""
        if n < 8:
            return np.zeros(n, dtype=int)
        
        # Build similarity matrix
        sigma = np.median(dist_copy[dist_copy < np.inf])
        if sigma <= 0:
            sigma = dist_stats['mean']
        
        similarity = np.exp(-dist_copy / (sigma + epsilon))
        np.fill_diagonal(similarity, 0)
        
        # Degree matrix
        degree = np.sum(similarity, axis=1)
        D = np.diag(degree)
        
        # Normalized Laplacian
        D_inv_sqrt = np.diag(1.0 / np.sqrt(degree + epsilon))
        L_norm = np.eye(n) - D_inv_sqrt @ similarity @ D_inv_sqrt
        
        # Compute Fiedler vector
        try:
            eigenvalues, eigenvectors = eigh(L_norm, subset_by_index=[0, 1])
            fiedler = eigenvectors[:, 1]
            
            # Initial bisection
            threshold = np.median(fiedler)
            cluster_labels = (fiedler > threshold).astype(int)
            
            # Recursive refinement for larger clusters
            min_cluster_size = max(3, n // 10)
            
            for cluster_id in [0, 1]:
                cluster_mask = cluster_labels == cluster_id
                cluster_size = np.sum(cluster_mask)
                
                if cluster_size >= min_cluster_size * 2:
                    # Extract submatrix
                    cluster_indices = np.where(cluster_mask)[0]
                    sub_dist = dist_copy[np.ix_(cluster_indices, cluster_indices)]
                    
                    # Recursive call on subcluster
                    if len(cluster_indices) > 3:
                        sub_labels = hierarchical_clustering()
                        # Map sublabels to unique new labels
                        max_label = np.max(cluster_labels)
                        cluster_labels[cluster_mask] = max_label + 1 + sub_labels
        except:
            cluster_labels = np.zeros(n, dtype=int)
        
        return cluster_labels
    
    cluster_labels = hierarchical_clustering()
    n_clusters = len(np.unique(cluster_labels))
    
    # ===== 6. DYNAMIC EDGE SCORING COMPONENTS =====
    components = {}
    
    # Component 1: Graph Feature Similarity
    feature_scores = np.zeros_like(dist_copy)
    for i in range(n):
        for j in range(i + 1, n):
            # Cosine similarity in feature space
            f1 = graph_features[i]
            f2 = graph_features[j]
            norm1 = np.linalg.norm(f1) + epsilon
            norm2 = np.linalg.norm(f2) + epsilon
            similarity = np.dot(f1, f2) / (norm1 * norm2)
            
            # Distance-weighted feature similarity
            dist_norm = dist_copy[i, j] / (dist_stats['mean'] + epsilon)
            decay = np.exp(-dist_norm)
            feature_scores[i, j] = feature_scores[j, i] = (1.0 + similarity) * decay
    
    components['features'] = feature_scores
    
    # Component 2: Adaptive Lookahead Regret
    lookahead_scores = np.ones_like(dist_copy)
    for i in range(n):
        for j in range(i + 1, n):
            regret = evaluate_lookahead(i, j, depth=params['lookahead_depth'])
            if regret > 0:
                # Convert regret to score (higher regret = more important edge)
                regret_norm = regret / (dist_copy[i, j] + epsilon)
                lookahead_scores[i, j] = lookahead_scores[j, i] = 1.0 + np.tanh(
                    params['regret_aggressiveness'] * regret_norm
                )
    
    components['lookahead'] = lookahead_scores
    
    # Component 3: Hierarchical Cluster Affinity
    cluster_scores = np.ones_like(dist_copy)
    if n_clusters > 1:
        # Compute intra-cluster distances
        intra_distances = []
        for c in range(n_clusters):
            mask = cluster_labels == c
            if np.sum(mask) > 1:
                sub_dist = dist_copy[np.ix_(mask, mask)]
                intra_distances.extend(sub_dist[sub_dist < np.inf].flatten())
        
        if intra_distances:
            cluster_threshold = np.percentile(intra_distances, 75)
            
            for i in range(n):
                for j in range(i + 1, n):
                    if cluster_labels[i] == cluster_labels[j]:
                        # Boost intra-cluster edges
                        dist_ratio = 1.0 - min(1.0, dist_copy[i, j] / (cluster_threshold + epsilon))
                        boost = 1.0 + params['cluster_granularity'] * dist_ratio
                        cluster_scores[i, j] = cluster_scores[j, i] = boost
                    else:
                        # Moderate penalty for inter-cluster edges
                        cluster_scores[i, j] = cluster_scores[j, i] = 0.8
    
    components['cluster'] = cluster_scores
    
    # Component 4: Structural Connectivity
    mst = minimum_spanning_tree(dist_copy).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # Compute edge betweenness approximation via random walks
    connectivity_scores = np.zeros_like(dist_copy)
    if n <= 100:  # Only compute for manageable sizes
        for i in range(n):
            for j in range(i + 1, n):
                if mst_mask[i, j]:
                    connectivity_scores[i, j] = connectivity_scores[j, i] = 1.5
                else:
                    # Estimate connectivity via neighbor overlap
                    neighbors_i = set(nearest_idx[i, :min(5, k_expand)])
                    neighbors_j = set(nearest_idx[j, :min(5, k_expand)])
                    overlap = len(neighbors_i.intersection(neighbors_j))
                    connectivity_scores[i, j] = connectivity_scores[j, i] = 1.0 + 0.1 * overlap
    else:
        connectivity_scores[mst_mask] = 1.5
        connectivity_scores[~mst_mask] = 1.0
    
    components['connectivity'] = connectivity_scores
    
    # ===== 7. DYNAMIC ENSEMBLE WEIGHTING =====
    # Adaptive weights based on problem characteristics
    if cv < 0.3 and n_clusters <= 2:
        weights = [0.25, 0.35, 0.20, 0.20]  # Emphasis on lookahead for uniform
    elif cv > 0.7 and n_clusters >= 4:
        weights = [0.20, 0.25, 0.35, 0.20]  # Emphasis on clustering
    else:
        weights = [0.30, 0.30, 0.25, 0.15]  # Balanced
    
    # Normalize components
    for key in components:
        max_val = np.max(components[key])
        if max_val > 0:
            components[key] = components[key] / max_val
    
    # Hybrid combination: weighted geometric mean for synergy
    geometric_part = np.ones_like(dist_copy)
    for (key, weight) in zip(components.keys(), weights):
        geometric_part *= (components[key] + epsilon) ** weight
    
    # Additive combination for diversity
    additive_part = np.zeros_like(dist_copy)
    for (key, weight) in zip(components.keys(), weights):
        additive_part += weight * components[key]
    
    # Dynamic blending based on problem entropy
    blend_alpha = 0.3 + 0.4 * (1.0 - np.exp(-dist_stats['entropy']))
    combined_scores = (blend_alpha * geometric_part + 
                      (1 - blend_alpha) * additive_part)
    
    # ===== 8. ADAPTIVE REFINEMENT =====
    final_scores = combined_scores.copy()
    
    # Reinforce MST connectivity
    final_scores[mst_mask] *= 1.3
    
    # Local consistency smoothing
    if n >= 10:
        smoothed = final_scores.copy()
        for i in range(n):
            # Weighted average with nearest neighbors' opinions
            neighbor_weights = 1.0 / (nearest_dists[i, :min(4, k_expand)] + epsilon)
            neighbor_weights = neighbor_weights / np.sum(neighbor_weights + epsilon)
            
            for idx, (j, w) in enumerate(zip(nearest_idx[i, :min(4, k_expand)], neighbor_weights)):
                if j != i:
                    # Blend with neighbor's edge assessments
                    smoothed[i] += 0.2 * w * final_scores[j]
        
        final_scores = 0.7 * final_scores + 0.3 * smoothed
    
    # Adaptive sparsification with degree balancing
    sparsified = np.zeros_like(final_scores)
    target_degree = min(8, max(3, int(2 * np.log2(n))))
    
    for i in range(n):
        row_scores = final_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Always include MST edges
        keep_mask = mst_mask[i].copy()
        
        # Determine number of additional edges needed
        mst_count = np.sum(keep_mask)
        additional = max(0, target_degree - mst_count)
        
        if additional > 0:
            # Select top non-MST edges
            non_mst_mask = ~keep_mask
            non_mst_scores = row_scores[non_mst_mask]
            
            if len(non_mst_scores) > 0:
                threshold = np.partition(
                    non_mst_scores, 
                    -min(additional, len(non_mst_scores))
                )[-min(additional, len(non_mst_scores))]
                
                top_indices = np.where(non_mst_mask)[0][non_mst_scores >= threshold]
                keep_mask[top_indices] = True
        
        sparsified[i, keep_mask] = final_scores[i, keep_mask]
    
    # ===== 9. FINAL PROCESSING =====
    # Symmetrize
    symmetric_scores = (sparsified + sparsified.T) / 2
    
    # Final normalization with adaptive shaping
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    # Problem-specific shaping
    if cv < 0.4:
        symmetric_scores = np.power(symmetric_scores, 0.8)
    elif cv > 0.7:
        symmetric_scores = np.power(symmetric_scores, 1.2)
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
