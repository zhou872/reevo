import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristic with adaptive regret computation, graph spectral analysis,
    dynamic weight optimization, and intelligent edge sparsification.
    
    Core Innovations:
    1. Two-step regret with density-adaptive lookahead
    2. Fiedler vector proximity for clustering structure
    3. Tour-based weight learning with caching
    4. Degree-constrained edge selection preserving connectivity
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. BASE COMPONENTS ====================
    
    # Inverse distance (fundamental component)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Minimum spanning tree for connectivity guarantee
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    
    # ==================== 2. ADAPTIVE REGRET COMPUTATION ====================
    
    def compute_regret_scores() -> np.ndarray:
        """Compute regret scores with adaptive lookahead based on problem density"""
        regret_scores = np.zeros((n, n))
        
        # Calculate problem density (median distance as reference)
        if n > 1:
            triu_dists = distance_matrix[np.triu_indices(n, 1)]
            median_dist = np.median(triu_dists) if len(triu_dists) > 0 else 1.0
        else:
            median_dist = 1.0
        
        # Adaptive lookahead depth based on problem size and density
        if n < 15:
            # For small problems, consider all nodes for alternatives
            k_neighbors = n - 1
            use_two_step = True
        else:
            # For larger problems, use density-based neighbor count
            density = len(triu_dists[triu_dists < median_dist * 1.5]) / len(triu_dists)
            k_neighbors = min(15, max(5, int(n * density * 0.3)))
            use_two_step = True
        
        # Precompute nearest neighbors for each node
        nn_lists = []
        for i in range(n):
            if k_neighbors >= n - 1:
                # Include all nodes except self
                candidates = [j for j in range(n) if j != i]
            else:
                # Get k nearest neighbors
                dists = distance_matrix[i].copy()
                dists[i] = np.inf
                candidates = np.argsort(dists)[:k_neighbors]
            nn_lists.append(candidates)
        
        # Compute regret for each edge
        for i in range(n):
            for j in range(i + 1, n):
                direct_cost = distance_matrix[i, j]
                best_alternative = float('inf')
                
                # Check one-step alternatives (common neighbors)
                common_nn = set(nn_lists[i]).intersection(nn_lists[j])
                if len(common_nn) > 0:
                    for k in common_nn:
                        if k != i and k != j:
                            alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                            if alt_cost < best_alternative:
                                best_alternative = alt_cost
                
                # Two-step alternatives for important edges
                if use_two_step and best_alternative == float('inf'):
                    # Consider paths i->k->l->j where k and l are neighbors
                    for k in nn_lists[i][:min(5, len(nn_lists[i]))]:
                        if k == j:
                            continue
                        for l in nn_lists[j][:min(5, len(nn_lists[j]))]:
                            if l == i or l == k:
                                continue
                            alt_cost = (distance_matrix[i, k] + 
                                       distance_matrix[k, l] + 
                                       distance_matrix[l, j])
                            if alt_cost < best_alternative:
                                best_alternative = alt_cost
                
                # Fallback: direct search if no good alternative found
                if best_alternative == float('inf') and n < 20:
                    for k in range(n):
                        if k != i and k != j:
                            alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                            if alt_cost < best_alternative:
                                best_alternative = alt_cost
                
                # Calculate regret score
                if best_alternative < float('inf'):
                    regret = max(0, best_alternative - direct_cost)
                    # Adaptive scaling based on problem statistics
                    scale = median_dist * 0.5
                    # Sigmoid transformation for stable scores
                    regret_scores[i, j] = 1.0 / (1.0 + np.exp(regret / max(scale, 1e-8)))
        
        # Symmetrize
        regret_scores = (regret_scores + regret_scores.T) / 2
        return regret_scores
    
    regret_scores = compute_regret_scores()
    
    # ==================== 3. SPECTRAL GRAPH ANALYSIS ====================
    
    def compute_spectral_scores() -> np.ndarray:
        """Compute spectral scores using Fiedler vector approximation"""
        if n < 3:
            return np.ones_like(distance_matrix)
        
        # Normalize distances for better spectral properties
        max_dist = np.max(distance_matrix)
        if max_dist > 0:
            norm_dist = distance_matrix / max_dist
        else:
            norm_dist = distance_matrix.copy()
        
        # Create similarity matrix (Gaussian kernel)
        sigma = np.median(norm_dist) * 0.3
        if sigma > 0:
            similarity = np.exp(-norm_dist**2 / (2 * sigma**2))
        else:
            similarity = 1.0 - norm_dist
        
        np.fill_diagonal(similarity, 0)
        
        # Compute Laplacian matrix
        degree = np.sum(similarity, axis=1)
        D_sqrt_inv = np.diag(1.0 / np.sqrt(degree + 1e-10))
        laplacian = np.eye(n) - D_sqrt_inv @ similarity @ D_sqrt_inv
        
        # Approximate Fiedler vector using power iteration
        # Start with random vector orthogonal to all-ones
        v = np.random.randn(n)
        v = v - np.mean(v)  # Make orthogonal to all-ones vector
        
        # Power iteration for dominant eigenvector of shifted Laplacian
        # (inverse iteration for second smallest eigenvalue)
        shift = 0.1
        shifted = laplacian + shift * np.eye(n)
        
        for _ in range(10):
            try:
                v = np.linalg.solve(shifted, v)
                v = v - np.mean(v)  # Maintain orthogonality to all-ones
                v_norm = np.linalg.norm(v)
                if v_norm > 1e-10:
                    v = v / v_norm
            except np.linalg.LinAlgError:
                # Fallback to simple approach
                v = np.random.randn(n)
                v = v - np.mean(v)
                v = v / (np.linalg.norm(v) + 1e-10)
                break
        
        fiedler = v
        
        # Compute spectral scores based on Fiedler vector similarity
        # Nodes with similar Fiedler values are likely in same cluster
        fiedler_diff = np.abs(fiedler[:, None] - fiedler[None, :])
        
        # Convert to similarity: edges between similar nodes get higher scores
        spectral_scores = 1.0 / (1.0 + fiedler_diff * np.sqrt(n))
        
        # Combine with distance information
        dist_weight = np.exp(-norm_dist)
        spectral_scores = 0.5 * spectral_scores + 0.5 * dist_weight
        
        np.fill_diagonal(spectral_scores, 0)
        return spectral_scores
    
    spectral_scores = compute_spectral_scores()
    
    # ==================== 4. DYNAMIC WEIGHT LEARNING ====================
    
    def learn_component_weights() -> Tuple[float, float, float]:
        """Learn optimal weights for components by evaluating sample tours"""
        if n < 4:
            return 0.4, 0.4, 0.2  # Default weights for small problems
        
        num_samples = min(20, max(5, n // 2))
        tour_cache = []
        tour_lengths = []
        
        # Generate diverse sample tours using randomized nearest neighbor
        for sample_idx in range(num_samples):
            start = np.random.randint(0, n)
            visited = np.zeros(n, dtype=bool)
            visited[start] = True
            tour = [start]
            current = start
            
            while len(tour) < n:
                mask = ~visited
                if np.any(mask):
                    # Softmax selection with temperature
                    available_dists = distance_matrix[current, mask]
                    if len(available_dists) > 0:
                        # Adaptive temperature based on progress
                        progress = len(tour) / n
                        temperature = 1.0 - 0.5 * progress  # Start explorative, end greedy
                        beta = 2.0 / (np.median(available_dists) + 1e-8) * temperature
                        
                        probs = np.exp(-beta * available_dists)
                        probs = probs / (probs.sum() + 1e-10)
                        
                        available_indices = np.where(mask)[0]
                        next_idx = np.random.choice(available_indices, p=probs)
                    else:
                        # Fallback
                        available_dists = distance_matrix[current].copy()
                        available_dists[visited] = np.inf
                        next_idx = np.argmin(available_dists)
                    
                    tour.append(next_idx)
                    visited[next_idx] = True
                    current = next_idx
            
            # Complete tour
            tour.append(tour[0])
            tour_cache.append(tour)
            
            # Calculate tour length
            length = 0
            for k in range(len(tour) - 1):
                length += distance_matrix[tour[k], tour[k + 1]]
            tour_lengths.append(length)
        
        # Evaluate each component on sample tours
        component_perf = []
        
        # Evaluate regret scores
        regret_perf = evaluate_component_on_tours(regret_scores, tour_cache, tour_lengths)
        component_perf.append(regret_perf)
        
        # Evaluate inverse distance
        inv_perf = evaluate_component_on_tours(inv_dist, tour_cache, tour_lengths)
        component_perf.append(inv_perf)
        
        # Evaluate spectral scores
        spectral_perf = evaluate_component_on_tours(spectral_scores, tour_cache, tour_lengths)
        component_perf.append(spectral_perf)
        
        # Convert performances to weights using softmax with temperature
        perf_array = np.array(component_perf)
        if np.all(perf_array == 0):
            # Default weights if no meaningful evaluation
            weights = np.array([0.4, 0.4, 0.2])
        else:
            # Normalize performances
            perf_norm = (perf_array - np.min(perf_array)) / (np.max(perf_array) - np.min(perf_array) + 1e-10)
            # Softmax with adaptive temperature
            temperature = 0.3 + 0.2 * np.std(perf_norm)  # Lower temp for clear differences
            exp_scores = np.exp(perf_norm / temperature)
            weights = exp_scores / exp_scores.sum()
        
        return weights[0], weights[1], weights[2]
    
    def evaluate_component_on_tours(scores: np.ndarray, tours: List[List[int]], 
                                   lengths: List[float]) -> float:
        """Evaluate component alignment with good tours"""
        if len(tours) == 0:
            return 0.0
        
        correlations = []
        median_length = np.median(lengths)
        
        for tour, length in zip(tours, lengths):
            # Calculate average score of edges in this tour
            tour_score = 0
            edge_count = 0
            
            for k in range(len(tour) - 1):
                i, j = tour[k], tour[k + 1]
                # Use symmetric average
                tour_score += (scores[i, j] + scores[j, i]) / 2
                edge_count += 1
            
            if edge_count > 0:
                avg_score = tour_score / edge_count
                # Normalize by tour quality (shorter tours are better)
                length_factor = median_length / (length + 1e-8)
                weighted_score = avg_score * min(length_factor, 2.0)  # Cap at 2x
                correlations.append(weighted_score)
        
        return np.mean(correlations) if correlations else 0.0
    
    # Learn optimal weights
    w_regret, w_inv, w_spectral = learn_component_weights()
    
    # ==================== 5. COMPOSITE SCORING ====================
    
    # Weighted combination of components
    composite = (
        w_regret * regret_scores +
        w_inv * inv_dist +
        w_spectral * spectral_scores
    )
    
    # Boost MST edges for connectivity
    mst_boost = 1.0 + np.log1p(n) / 8.0
    composite[mst_edges] *= mst_boost
    
    # ==================== 6. INTELLIGENT SPARSIFICATION ====================
    
    def sparsify_scores(scores: np.ndarray) -> np.ndarray:
        """Intelligently sparsify scores while preserving connectivity"""
        sparsified = np.zeros_like(scores)
        
        # Always include MST edges
        sparsified[mst_edges] = scores[mst_edges]
        
        # Adaptive degree limits based on problem size
        if n <= 20:
            max_degree = min(6, n - 1)
        elif n <= 100:
            max_degree = min(10, n // 3)
        else:
            max_degree = min(15, n // 4)
        
        # Track current degrees
        current_degree = np.sum(sparsified > 0, axis=1)
        
        # For each node, add promising non-MST edges
        for i in range(n):
            # Find candidate edges (non-MST, not self)
            candidates = []
            for j in range(n):
                if i != j and not mst_edges[i, j] and scores[i, j] > 0:
                    # Adjust score by distance penalty
                    dist_ratio = distance_matrix[i, j] / np.median(distance_matrix)
                    penalty = np.exp(-dist_ratio)
                    adjusted_score = scores[i, j] * penalty
                    candidates.append((adjusted_score, j))
            
            # Sort candidates by adjusted score
            candidates.sort(reverse=True, key=lambda x: x[0])
            
            # Add top candidates respecting degree limits
            for score, j in candidates:
                if current_degree[i] < max_degree and current_degree[j] < max_degree:
                    sparsified[i, j] = scores[i, j]
                    current_degree[i] += 1
                    current_degree[j] += 1
        
        # Ensure symmetry (take maximum of both directions)
        sparsified = np.maximum(sparsified, sparsified.T)
        
        return sparsified
    
    sparsified = sparsify_scores(composite)
    
    # ==================== 7. FINAL REFINEMENT ====================
    
    # Apply distance-based attenuation (penalize very long edges)
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        # Adaptive attenuation strength
        attenuation_strength = 0.3 + 0.2 * (n / 50)  # Stronger for larger problems
        attenuation = np.exp(-distance_matrix / (attenuation_strength * max_dist + 1e-8))
        sparsified *= attenuation
    
    # Normalize to [0, 1] range
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    # Zero diagonal
    np.fill_diagonal(sparsified, 0)
    
    # Apply soft saturation for numerical stability
    sparsified = np.tanh(sparsified * 1.2)
    
    return sparsified
