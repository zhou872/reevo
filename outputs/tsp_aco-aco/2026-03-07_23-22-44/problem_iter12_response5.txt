import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List, Tuple
import heapq

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristic with multi-step regret lookahead, graph neural network
    inspired edge scoring, dynamic weight adjustment, and reinforcement learning
    enhanced pattern recognition.
    
    Core Innovations:
    1. Multi-step Regret Lookahead: 3-level recursive regret computation with
       adaptive pruning and reinforcement learning value estimation
    2. Graph Neural Network Edge Scoring: Simulates GNN message passing using
       multi-hop neighborhood features without actual neural network training
    3. Dynamic Weight Adjustment: Real-time ensemble weight optimization during
       edge scoring based on local solution state
    4. Reinforcement Learning Policy Gradient: Learns edge selection patterns
       from high-quality tours using policy gradient approximation
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    epsilon = 1e-12
    
    # ==================== 1. ENHANCED PRELIMINARY COMPUTATIONS ====================
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Enhanced MST with edge importance scores
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    
    # Compute edge centrality based on alternative MSTs
    mst_centrality = np.zeros_like(distance_matrix)
    for i in range(min(10, n * (n-1) // 20)):  # Sample alternative MSTs
        perturbed_dist = distance_matrix * (1 + 0.1 * np.random.randn(n, n))
        perturbed_dist = (perturbed_dist + perturbed_dist.T) / 2
        alt_mst = minimum_spanning_tree(perturbed_dist).toarray()
        mst_centrality += ((alt_mst > 0) | (alt_mst.T > 0))
    mst_centrality = mst_centrality / (mst_centrality.max() + epsilon)
    
    # ==================== 2. MULTI-STEP REGRET LOOKAHEAD ====================
    def multi_step_regret(max_levels: int = 3) -> np.ndarray:
        """Compute regret using recursive multi-step lookahead with RL value estimation"""
        regret_scores = np.zeros((n, n))
        
        # Precompute k-nearest neighbors for each node
        k_nn = min(8, n - 1)
        nearest = []
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            idx = np.argpartition(dists, k_nn)[:k_nn]
            nearest.append(idx)
        
        # RL value estimation cache
        value_cache = {}
        
        def estimate_path_value(mask: int, current: int, depth: int, alpha: float = 0.9) -> float:
            """Estimate value of current partial path using RL approximation"""
            if depth <= 0:
                return 0
            
            cache_key = (mask, current, depth)
            if cache_key in value_cache:
                return value_cache[cache_key]
            
            # Get possible next nodes
            candidates = [j for j in nearest[current] if not (mask & (1 << j))]
            
            if not candidates:
                value = 0
            else:
                # Compute Q-values using immediate reward + discounted future value
                q_values = []
                for j in candidates:
                    immediate_reward = 1.0 / (distance_matrix[current, j] + epsilon)
                    future_mask = mask | (1 << j)
                    future_value = estimate_path_value(future_mask, j, depth-1, alpha)
                    q_value = immediate_reward + alpha * future_value
                    q_values.append(q_value)
                
                # Softmax policy for value estimation
                q_values = np.array(q_values)
                if len(q_values) > 0:
                    exp_q = np.exp(q_values - np.max(q_values))
                    policy = exp_q / (exp_q.sum() + epsilon)
                    value = np.sum(policy * q_values)
                else:
                    value = 0
            
            value_cache[cache_key] = value
            return value
        
        # Compute regret for each edge
        for i in range(n):
            for j in range(i + 1, n):
                if distance_matrix[i, j] == 0:
                    continue
                
                direct_cost = distance_matrix[i, j]
                
                # Start recursive regret computation from both endpoints
                best_alternative = float('inf')
                
                # Try alternative paths starting from i
                start_mask_i = (1 << i) | (1 << j)
                for k in nearest[i]:
                    if k == j:
                        continue
                    
                    # Compute alternative path value (i -> k -> ... -> j)
                    alt_mask = start_mask_i | (1 << k)
                    alt_cost = distance_matrix[i, k]
                    
                    # Estimate remaining path value using RL
                    remaining_value = estimate_path_value(alt_mask, k, max_levels-1)
                    alt_total = alt_cost + (1 - remaining_value) * np.median(distance_matrix)
                    
                    best_alternative = min(best_alternative, alt_total)
                
                # Calculate regret
                if best_alternative < float('inf'):
                    regret = max(0, best_alternative - direct_cost)
                    # Adaptive sigmoid scaling
                    scale = np.percentile(distance_matrix, 25) * 0.7
                    regret_scores[i, j] = 1.0 / (1.0 + np.exp(-regret / (scale + epsilon)))
        
        # Symmetrize
        regret_scores = np.maximum(regret_scores, regret_scores.T)
        np.fill_diagonal(regret_scores, 0)
        return regret_scores
    
    regret_scores = multi_step_regret()
    
    # ==================== 3. GRAPH NEURAL NETWORK EDGE SCORING ====================
    def gnn_inspired_scoring(num_layers: int = 2) -> np.ndarray:
        """Simulate GNN message passing using multi-hop neighborhood features"""
        # Initialize node features (distance-based)
        node_features = np.zeros((n, 3))
        for i in range(n):
            # Feature 1: Average distance to nearest neighbors
            k = min(5, n-1)
            dists = np.partition(distance_matrix[i], k)[:k]
            node_features[i, 0] = np.mean(dists)
            
            # Feature 2: Local density estimate
            node_features[i, 1] = 1.0 / (node_features[i, 0] + epsilon)
            
            # Feature 3: Degree centrality approximation
            threshold = np.median(distance_matrix) * 0.7
            node_features[i, 2] = np.sum(distance_matrix[i] < threshold)
        
        # Normalize features
        for f in range(3):
            if node_features[:, f].std() > 0:
                node_features[:, f] = (node_features[:, f] - node_features[:, f].mean()) / (node_features[:, f].std() + epsilon)
        
        # Simulate GNN layers
        edge_scores = np.zeros_like(distance_matrix)
        
        for _ in range(num_layers):
            # Aggregate neighbor features for each node
            agg_features = np.zeros((n, 3))
            
            for i in range(n):
                # Weight neighbors by distance
                weights = 1.0 / (distance_matrix[i] + epsilon)
                weights[i] = 0
                weights = weights / (weights.sum() + epsilon)
                
                # Weighted aggregation
                for f in range(3):
                    agg_features[i, f] = np.sum(weights * node_features[:, f])
            
            # Combine with original features (simulating residual connections)
            node_features = 0.7 * node_features + 0.3 * agg_features
            
            # Update edge scores based on node feature similarity
            for i in range(n):
                for j in range(i + 1, n):
                    # Cosine similarity between node features
                    sim = np.dot(node_features[i], node_features[j])
                    norm_i = np.linalg.norm(node_features[i])
                    norm_j = np.linalg.norm(node_features[j])
                    
                    if norm_i > 0 and norm_j > 0:
                        cos_sim = sim / (norm_i * norm_j)
                        # Combine with inverse distance
                        dist_factor = 1.0 / (distance_matrix[i, j] + epsilon)
                        edge_scores[i, j] += (cos_sim + 1) * dist_factor
        
        # Normalize
        if edge_scores.max() > 0:
            edge_scores = edge_scores / edge_scores.max()
        
        edge_scores = np.maximum(edge_scores, edge_scores.T)
        np.fill_diagonal(edge_scores, 0)
        return edge_scores
    
    gnn_scores = gnn_inspired_scoring()
    
    # ==================== 4. DYNAMIC WEIGHT ADJUSTMENT ====================
    def dynamic_weight_combination() -> np.ndarray:
        """Combine components with weights adjusted based on local graph properties"""
        # Base weights
        base_weights = np.array([0.35, 0.25, 0.25, 0.15])  # regret, inv_dist, gnn, mst_centrality
        
        # Compute local adjustment factors
        adjustment_factors = np.ones((n, n, 4))
        
        # Density-based adjustment
        local_density = np.zeros(n)
        for i in range(n):
            k = min(10, n-1)
            dists = np.partition(distance_matrix[i], k)[:k]
            local_density[i] = 1.0 / (np.mean(dists) + epsilon)
        
        # Normalize density
        if local_density.max() > 0:
            local_density = local_density / local_density.max()
        
        # Adjust weights based on local properties
        for i in range(n):
            for j in range(i + 1, n):
                # Factor 1: In dense regions, favor regret and gnn scores
                density_factor = (local_density[i] + local_density[j]) / 2
                adjustment_factors[i, j, 0] *= (1 + 0.3 * density_factor)  # Boost regret
                adjustment_factors[i, j, 2] *= (1 + 0.2 * density_factor)  # Boost gnn
                
                # Factor 2: For long edges, favor inverse distance less
                avg_dist = np.median(distance_matrix)
                if distance_matrix[i, j] > 1.5 * avg_dist:
                    adjustment_factors[i, j, 1] *= 0.7  # Reduce inv_dist weight
                
                # Factor 3: For MST central edges, boost mst_centrality
                if mst_centrality[i, j] > 0.8:
                    adjustment_factors[i, j, 3] *= 1.5
        
        # Apply adjustments and combine
        combined = np.zeros_like(distance_matrix)
        
        components = [regret_scores, inv_dist, gnn_scores, mst_centrality]
        
        for i in range(n):
            for j in range(i + 1, n):
                total_weight = 0
                for c in range(4):
                    adjusted_weight = base_weights[c] * adjustment_factors[i, j, c]
                    combined[i, j] += adjusted_weight * components[c][i, j]
                    total_weight += adjusted_weight
                
                if total_weight > 0:
                    combined[i, j] /= total_weight
                
                combined[j, i] = combined[i, j]
        
        return combined
    
    combined_scores = dynamic_weight_combination()
    
    # ==================== 5. REINFORCEMENT LEARNING PATTERN LEARNING ====================
    def rl_pattern_learning(base_scores: np.ndarray) -> np.ndarray:
        """Learn edge selection patterns using policy gradient approximation"""
        reinforced = base_scores.copy()
        
        # Generate diverse high-quality tours using different policies
        policies = ['greedy', 'regret', 'random', 'hybrid']
        high_quality_tours = []
        
        for policy in policies:
            for _ in range(max(3, n // 20)):
                if policy == 'greedy':
                    tour = _greedy_tour_with_policy(distance_matrix, 'nearest')
                elif policy == 'regret':
                    tour = _regret_insertion_tour(distance_matrix)
                elif policy == 'random':
                    tour = list(range(n))
                    np.random.shuffle(tour)
                    tour.append(tour[0])
                else:  # hybrid
                    tour = _hybrid_construction(distance_matrix, base_scores)
                
                # Apply local optimization
                improved = _adaptive_opt_improvement(tour, distance_matrix, base_scores)
                high_quality_tours.append(improved)
        
        # Policy gradient update approximation
        edge_advantages = np.zeros_like(reinforced)
        
        for tour in high_quality_tours:
            # Compute advantage for each edge in tour
            tour_length = sum(distance_matrix[tour[k], tour[k+1]] for k in range(len(tour)-1))
            avg_edge_length = tour_length / (len(tour) - 1)
            
            for k in range(len(tour) - 1):
                i, j = tour[k], tour[k+1]
                # Advantage: how much better than average this edge is
                advantage = (avg_edge_length - distance_matrix[i, j]) / (avg_edge_length + epsilon)
                edge_advantages[i, j] += max(0, advantage)
                edge_advantages[j, i] += max(0, advantage)
        
        # Normalize advantages
        if edge_advantages.max() > 0:
            edge_advantages = edge_advantages / edge_advantages.max()
        
        # Apply advantage-weighted reinforcement
        reinforcement = 1.0 + 0.8 * edge_advantages  # Up to 80% boost
        reinforced *= reinforcement
        
        # Boost edges that appear in multiple high-quality tours
        edge_counts = np.zeros_like(reinforced)
        for tour in high_quality_tours:
            for k in range(len(tour) - 1):
                i, j = tour[k], tour[k+1]
                edge_counts[i, j] += 1
        
        if edge_counts.max() > 0:
            consistency = edge_counts / edge_counts.max()
            reinforced *= (1.0 + 0.4 * consistency)  # Up to 40% consistency bonus
        
        return reinforced
    
    def _greedy_tour_with_policy(dist_mat: np.ndarray, policy: str) -> List[int]:
        """Generate tour using specified greedy policy"""
        start = np.random.randint(0, n)
        tour = [start]
        unvisited = set(range(n)) - {start}
        
        current = start
        while unvisited:
            if policy == 'nearest':
                # Nearest neighbor
                dists = np.array([dist_mat[current, j] if j in unvisited else np.inf 
                                 for j in range(n)])
                next_node = np.argmin(dists)
            elif policy == 'farthest':
                # Farthest insertion
                farthest = max(unvisited, key=lambda x: min(dist_mat[x, y] for y in tour))
                # Insert farthest at best position
                best_pos = 0
                best_cost = float('inf')
                for pos in range(len(tour)):
                    i, j = tour[pos-1], tour[pos] if pos < len(tour) else tour[0]
                    cost = dist_mat[i, farthest] + dist_mat[farthest, j] - dist_mat[i, j]
                    if cost < best_cost:
                        best_cost = cost
                        best_pos = pos
                tour.insert(best_pos, farthest)
                unvisited.remove(farthest)
                continue
            else:
                # Random among nearest 3
                dists = [(dist_mat[current, j], j) for j in unvisited]
                dists.sort()
                next_node = dists[np.random.randint(0, min(3, len(dists)))][1]
            
            tour.append(next_node)
            unvisited.remove(next_node)
            current = next_node
        
        tour.append(tour[0])
        return tour
    
    def _regret_insertion_tour(dist_mat: np.ndarray) -> List[int]:
        """Generate tour using regret-k insertion heuristic"""
        # Start with a subtour of 3 nodes
        nodes = list(range(n))
        np.random.shuffle(nodes)
        tour = nodes[:3]
        tour.append(tour[0])  # Close the subtour
        
        unvisited = set(nodes[3:])
        
        while unvisited:
            best_node = None
            best_pos = None
            max_regret = -float('inf')
            
            for node in unvisited:
                # Compute insertion costs for all positions
                insertion_costs = []
                for pos in range(len(tour) - 1):
                    i, j = tour[pos], tour[pos + 1]
                    cost = dist_mat[i, node] + dist_mat[node, j] - dist_mat[i, j]
                    insertion_costs.append((cost, pos))
                
                # Sort by cost
                insertion_costs.sort()
                
                # Regret-k (k=3)
                k = min(3, len(insertion_costs))
                regret = sum(insertion_costs[m][0] for m in range(1, k)) - insertion_costs[0][0] * (k-1)
                
                if regret > max_regret:
                    max_regret = regret
                    best_node = node
                    best_pos = insertion_costs[0][1]
            
            if best_node is not None:
                tour.insert(best_pos + 1, best_node)
                unvisited.remove(best_node)
        
        return tour
    
    def _hybrid_construction(dist_mat: np.ndarray, scores: np.ndarray) -> List[int]:
        """Hybrid construction using both distance and heuristic scores"""
        # Start with a random edge
        edges = []
        for i in range(n):
            for j in range(i + 1, n):
                # Combined score: inverse distance * heuristic score
                combined = (1.0 / (dist_mat[i, j] + epsilon)) * scores[i, j]
                edges.append((combined, i, j))
        
        edges.sort(reverse=True)
        
        # Greedy construction with degree constraints
        degree = [0] * n
        adj = [[] for _ in range(n)]
        tour_edges = []
        
        for _, i, j in edges:
            if degree[i] < 2 and degree[j] < 2:
                # Check for premature cycle
                if _forms_cycle(adj, i, j, len(tour_edges), n):
                    continue
                
                adj[i].append(j)
                adj[j].append(i)
                degree[i] += 1
                degree[j] += 1
                tour_edges.append((i, j))
                
                if all(d == 2 for d in degree):
                    break
        
        # Convert to tour
        tour = []
        current = 0
        prev = -1
        
        while len(tour) < n:
            tour.append(current)
            neighbors = [n for n in adj[current] if n != prev]
            if not neighbors:
                break
            prev, current = current, neighbors[0]
        
        tour.append(tour[0])
        return tour
    
    def _forms_cycle(adj: List[List[int]], u: int, v: int, edge_count: int, n_nodes: int) -> bool:
        """Check if adding edge (u,v) forms a cycle before completing the tour"""
        if edge_count >= n_nodes - 1:
            return False  # Last edge should form the cycle
        
        # Quick check: if u and v are already connected
        visited = [False] * n_nodes
        stack = [u]
        visited[u] = True
        
        while stack:
            node = stack.pop()
            for neighbor in adj[node]:
                if not visited[neighbor]:
                    if neighbor == v:
                        return True
                    visited[neighbor] = True
                    stack.append(neighbor)
        
        return False
    
    def _adaptive_opt_improvement(tour: List[int], dist_mat: np.ndarray, 
                                 scores: np.ndarray) -> List[int]:
        """Adaptive k-opt improvement guided by heuristic scores"""
        tour = tour[:-1]  # Remove closing edge
        improved = True
        max_iter = min(50, n * 2)
        
        for _ in range(max_iter):
            if not improved:
                break
            improved = False
            
            # Try 2-opt moves
            for i in range(len(tour)):
                for j in range(i + 2, len(tour)):
                    if j == len(tour) - 1 and i == 0:
                        continue
                    
                    a, b = tour[i], tour[(i + 1) % len(tour)]
                    c, d = tour[j], tour[(j + 1) % len(tour)]
                    
                    current = dist_mat[a, b] + dist_mat[c, d]
                    new = dist_mat[a, c] + dist_mat[b, d]
                    
                    # Consider heuristic scores in decision
                    score_current = scores[a, b] + scores[c, d]
                    score_new = scores[a, c] + scores[b, d]
                    
                    # Combined improvement criterion
                    if new < current * 0.95 or (new < current and score_new > score_current):
                        # Reverse segment
                        if (i + 1) % len(tour) < j:
                            tour[(i + 1) % len(tour):j + 1] = reversed(tour[(i + 1) % len(tour):j + 1])
                        else:
                            segment = tour[(i + 1) % len(tour):] + tour[:j + 1]
                            segment.reverse()
                            tour[(i + 1) % len(tour):] = segment[:len(tour) - (i + 1) % len(tour)]
                            tour[:j + 1] = segment[len(tour) - (i + 1) % len(tour):]
                        improved = True
                        break
                
                if improved:
                    break
        
        return tour + [tour[0]]
    
    # Apply RL pattern learning
    reinforced_scores = rl_pattern_learning(combined_scores)
    
    # ==================== 6. ADAPTIVE SPARSIFICATION ====================
    def adaptive_sparsification(scores: np.ndarray) -> np.ndarray:
        """Intelligent edge retention with adaptive connectivity preservation"""
        sparsified = np.zeros_like(scores)
        
        # Phase 1: Always include critical edges
        # MST edges with high centrality
        critical_threshold = np.percentile(mst_centrality[mst_edges], 70) if mst_edges.any() else 0
        critical_edges = mst_edges & (mst_centrality > critical_threshold)
        sparsified[critical_edges] = scores[critical_edges]
        
        # Phase 2: Adaptive degree limits based on local density
        degree_limits = np.zeros(n)
        for i in range(n):
            # Compute local density
            k = min(10, n-1)
            dists = np.partition(distance_matrix[i], k)[:k]
            density = 1.0 / (np.mean(dists) + epsilon)
            
            # Set degree limit: higher in dense regions
            base_limit = min(4, max(2, n // 15))
            degree_limits[i] = base_limit * (1 + 0.5 * density)
        
        # Phase 3: Greedy edge selection with dynamic priorities
        current_degrees = np.sum(sparsified > 0, axis=1)
        
        # Candidate edges
        candidates = []
        for i in range(n):
            for j in range(i + 1, n):
                if not sparsified[i, j] and scores[i, j] > 0:
                    # Priority score considers:
                    # 1. Edge score
                    # 2. Degree availability
                    # 3. Local density match
                    deg_avail = (degree_limits[i] - current_degrees[i]) * (degree_limits[j] - current_degrees[j])
                    priority = scores[i, j] * np.sqrt(deg_avail + 1)
                    candidates.append((-priority, i, j))  # Negative for max-heap
        
        heapq.heapify(candidates)
        
        while candidates:
            neg_priority, i, j = heapq.heappop(candidates)
            priority = -neg_priority
            
            if current_degrees[i] < degree_limits[i] and current_degrees[j] < degree_limits[j]:
                sparsified[i, j] = scores[i, j]
                sparsified[j, i] = scores[j, i]
                current_degrees[i] += 1
                current_degrees[j] += 1
        
        return sparsified
    
    sparsified_scores = adaptive_sparsification(reinforced_scores)
    
    # ==================== 7. FINAL ENHANCEMENTS ====================
    # Ensure symmetry
    symmetric_scores = (sparsified_scores + sparsified_scores.T) / 2
    
    # Adaptive distance decay
    dist_percentiles = np.percentile(distance_matrix[np.triu_indices(n, 1)], [25, 50, 75])
    decay_factor = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(i + 1, n):
            dist = distance_matrix[i, j]
            if dist <= dist_percentiles[0]:  # Short edges
                decay = 1.0
            elif dist <= dist_percentiles[1]:  # Medium edges
                decay = 0.9
            elif dist <= dist_percentiles[2]:  # Longish edges
                decay = 0.7
            else:  # Very long edges
                decay = 0.4
            decay_factor[i, j] = decay
            decay_factor[j, i] = decay
    
    symmetric_scores *= decay_factor
    
    # Final normalization with adaptive clipping
    max_score = symmetric_scores.max()
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    # Adaptive thresholding
    non_zero = symmetric_scores[symmetric_scores > 0]
    if len(non_zero) > 0:
        threshold = np.percentile(non_zero, 15)  # Keep top 85%
        symmetric_scores[symmetric_scores < threshold] = 0
    
    # Ensure connectivity: boost weakest MST edge if needed
    if mst_edges.any():
        mst_edge_scores = symmetric_scores[mst_edges]
        if mst_edge_scores.min() == 0:
            min_mst_idx = np.argmin(mst_edge_scores)
            mst_positions = np.where(mst_edges)
            symmetric_scores[mst_positions[0][min_mst_idx], mst_positions[1][min_mst_idx]] = threshold * 0.8
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
