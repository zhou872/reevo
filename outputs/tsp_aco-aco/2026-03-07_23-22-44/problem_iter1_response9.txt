```python
import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges combining multiple factors:
    1. Weighted inverse distance with local neighborhood consideration
    2. Local sparsity thresholding based on distance percentiles
    3. Nearest neighbor density bonus
    4. Symmetric sparsification to reduce unpromising edges
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # Create a copy and handle zero distances (diagonal)
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    
    # 1. Base heuristic: weighted inverse distance
    # Give higher weight to very short edges
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / dist
        inv_dist = np.nan_to_num(inv_dist, posinf=0, neginf=0)
    
    # 2. Local sparsity thresholding
    # For each node, calculate distance thresholds
    edge_scores = np.zeros_like(dist)
    
    for i in range(n):
        # Get distances from node i to all others
        node_distances = dist[i].copy()
        
        # Calculate distance percentiles
        p25 = np.percentile(node_distances[node_distances < np.inf], 25)
        p50 = np.percentile(node_distances[node_distances < np.inf], 50)
        
        # Score edges based on distance quartiles
        mask_p25 = node_distances <= p25
        mask_p50 = node_distances <= p50
        
        edge_scores[i][mask_p25] = 2.0  # Top quartile: highest score
        edge_scores[i][mask_p50 & ~mask_p25] = 1.0  # Second quartile: medium score
        # Other edges remain 0
    
    # 3. Nearest neighbor density bonus
    # Count how many times each node appears as a nearest neighbor
    nn_counts = np.zeros(n)
    
    # Find k nearest neighbors for each node
    k = min(5, n - 1)
    for i in range(n):
        sorted_indices = np.argsort(dist[i])[:k]
        nn_counts[sorted_indices] += 1
    
    # Apply density bonus to edges connecting popular nodes
    for i in range(n):
        for j in range(n):
            if i != j:
                density_bonus = 0.5 * (nn_counts[i] + nn_counts[j]) / k
                edge_scores[i, j] += density_bonus
    
    # 4. Combine with inverse distance and normalize
    combined = edge_scores * inv_dist
    
    # 5. Symmetric sparsification
    # Only keep edges where both nodes consider each other promising
    symmetric_mask = (edge_scores > 0) & (edge_scores.T > 0)
    
    # Set unpromising edges to zero
    result = combined * symmetric_mask
    
    # Normalize to similar scale as v1
    max_val = np.max(result[result > 0]) if np.any(result > 0) else 1.0
    if max_val > 0:
        result = result / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(result, 0)
    
    return result
```
