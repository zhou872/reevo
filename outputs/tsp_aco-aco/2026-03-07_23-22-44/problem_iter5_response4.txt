import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with improved regret modeling and cluster awareness.
    
    Key improvements over v1:
    1. Dual-stage regret calculation with exponential decay
    2. Proximity cluster detection with intra-cluster bonuses
    3. Node degree normalization to prevent hub over-connection
    4. Dynamic sparsification with MST backbone preservation
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance with logarithmic scaling for stability
    with np.errstate(divide='ignore', invalid='ignore'):
        log_inv_dist = 1.0 / (np.log1p(distance_matrix) + 1e-10)
    log_inv_dist[np.isinf(log_inv_dist)] = 0
    np.fill_diagonal(log_inv_dist, 0)
    
    # 2. Dual-stage regret calculation
    # Stage 1: Local regret (per-node opportunity cost)
    local_regret = np.ones_like(distance_matrix)
    node_min_dist = np.full(n, np.inf)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if np.any(dists < np.inf):
            node_min_dist[i] = np.min(dists[dists > 0])
    
    for i in range(n):
        if node_min_dist[i] < np.inf:
            # Regret ratio with exponential decay
            regret_ratio = distance_matrix[i] / (node_min_dist[i] + 1e-10)
            # Exponential decay: exp(-regret/best_alternative)
            local_regret[i] = np.exp(-regret_ratio)
        local_regret[i, i] = 0
    
    # Stage 2: Global regret (cross-node consistency)
    global_regret = np.ones_like(distance_matrix)
    avg_node_dist = np.mean(distance_matrix[np.triu_indices(n, k=1)])
    
    for i in range(n):
        for j in range(i+1, n):
            # Opportunity cost relative to global alternatives
            alt_cost_i = np.mean(np.partition(distance_matrix[i], 1)[1:3])
            alt_cost_j = np.mean(np.partition(distance_matrix[j], 1)[1:3])
            best_alt = min(alt_cost_i, alt_cost_j)
            
            if best_alt > 0:
                regret = distance_matrix[i, j] / best_alt
                global_regret[i, j] = global_regret[j, i] = np.exp(-regret)
    
    # Combined regret score
    regret_score = local_regret * global_regret
    
    # 3. Proximity cluster detection and bonus
    cluster_bonus = np.ones_like(distance_matrix)
    if n > 3:
        # Identify tight clusters using distance quartiles
        upper_tri = distance_matrix[np.triu_indices(n, k=1)]
        q25 = np.percentile(upper_tri, 25)
        
        for i in range(n):
            # Find close neighbors (potential cluster members)
            close_mask = distance_matrix[i] < q25
            close_mask[i] = False
            
            if np.sum(close_mask) > 1:
                # Intra-cluster bonus
                cluster_bonus[i, close_mask] = 1.2
    
    # 4. MST connectivity guarantee with enhanced bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # MST edges get multiplicative bonus
    mst_bonus = np.ones_like(distance_matrix)
    mst_bonus[mst_mask] = 1.5
    
    # 5. Node degree balancing factor
    degree_balance = np.ones_like(distance_matrix)
    if n > 2:
        # Estimate optimal node degree (2 for TSP tour)
        target_degree = 2
        for i in range(n):
            # Penalize nodes that would become hubs
            potential_connections = np.sum(log_inv_dist[i] > 0)
            if potential_connections > target_degree:
                balance_factor = target_degree / (potential_connections + 1e-10)
                degree_balance[i] = np.sqrt(balance_factor)  # Soft penalty
    
    # 6. Combine all factors
    base_scores = log_inv_dist * regret_score * cluster_bonus * mst_bonus * degree_balance
    
    # 7. Dynamic sparsification with MST preservation
    sparsified = np.zeros_like(base_scores)
    
    for i in range(n):
        row_scores = base_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Always preserve MST edges
        mst_edges = mst_mask[i]
        mst_count = np.sum(mst_edges)
        
        # Dynamic sparsity based on problem size and node centrality
        if n <= 5:
            k = n - 1
        else:
            # Adaptive k: more connections for central nodes
            centrality = np.mean(1.0 / (distance_matrix[i] + 1e-10))
            k_base = max(3, int(np.sqrt(n)))
            k = min(n - 1, k_base + int(centrality * 2))
        
        # Select top k edges including all MST edges
        if mst_count > 0:
            # Ensure MST edges are included
            keep_mask = mst_edges.copy()
            non_mst_count = k - mst_count
            
            if non_mst_count > 0:
                # Get top non-MST edges
                non_mst_scores = row_scores.copy()
                non_mst_scores[mst_edges] = -np.inf
                
                if np.sum(non_mst_scores > -np.inf) >= non_mst_count:
                    threshold = np.partition(
                        non_mst_scores[non_mst_scores > -np.inf], 
                        -non_mst_count
                    )[-non_mst_count]
                    top_non_mst = non_mst_scores >= threshold
                    keep_mask = keep_mask | top_non_mst
                else:
                    # Keep all available non-MST edges
                    keep_mask = keep_mask | (non_mst_scores > -np.inf)
        else:
            # No MST edges for this node (shouldn't happen in connected graph)
            if np.any(row_scores > -np.inf):
                k = min(k, np.sum(row_scores > -np.inf))
                threshold = np.partition(row_scores[row_scores > -np.inf], -k)[-k]
                keep_mask = row_scores >= threshold
            else:
                keep_mask = np.zeros_like(row_scores, dtype=bool)
        
        sparsified[i][keep_mask] = base_scores[i][keep_mask]
    
    # 8. Symmetrize and ensure connectivity
    symmetric_scores = np.maximum(sparsified, sparsified.T)
    
    # Ensure MST edges have minimum viable score
    min_nonzero = np.min(symmetric_scores[symmetric_scores > 0])
    symmetric_scores[mst_mask] = np.maximum(symmetric_scores[mst_mask], min_nonzero * 0.5)
    
    # 9. Final normalization with softmax scaling
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        # Soft normalization preserving relative differences
        symmetric_scores = symmetric_scores / max_score
        # Apply gentle exponential to emphasize top candidates
        symmetric_scores = np.power(symmetric_scores, 0.8)
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
