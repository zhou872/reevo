import numpy as np
from scipy.spatial import Delaunay

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors.
    Returns a sparse matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (closer nodes are generally better)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree (MST) bonus - edges likely to be in optimal solutions
    mst_bonus = np.zeros((n, n))
    # Simple approximation: edges connecting to each node's two nearest neighbors
    for i in range(n):
        # Get two smallest distances (excluding self)
        sorted_idx = np.argsort(distance_matrix[i])
        # Connect to 2 nearest neighbors (positions 1 and 2 since 0 is self)
        for k in range(1, min(3, n)):
            j = sorted_idx[k]
            mst_bonus[i, j] = 1.0
            mst_bonus[j, i] = 1.0
    
    # 3. Distance rank weighting - prioritize edges that are relatively short
    rank_weight = np.zeros((n, n))
    for i in range(n):
        ranks = np.argsort(np.argsort(distance_matrix[i]))
        # Normalize ranks to [0, 1], with 0 for shortest distance
        normalized = 1 - ranks / (n - 1) if n > 1 else 0
        np.fill_diagonal(normalized, 0)
        rank_weight[i] = normalized
    
    # 4. Geographic clustering penalty - avoid edges that stay within dense clusters
    cluster_penalty = np.ones((n, n))
    if n > 3:
        try:
            # Use Delaunay triangulation to identify natural clusters
            # Generate approximate coordinates from distance matrix using MDS
            from sklearn.manifold import MDS
            mds = MDS(n_components=2, dissimilarity='precomputed', random_state=42, normalized_stress='auto')
            coords = mds.fit_transform(distance_matrix)
            
            # Get Delaunay triangulation edges
            tri = Delaunay(coords)
            delaunay_edges = set()
            for simplex in tri.simplices:
                for i in range(3):
                    for j in range(i+1, 3):
                        u, v = simplex[i], simplex[j]
                        delaunay_edges.add((min(u, v), max(u, v)))
            
            # Penalize edges NOT in Delaunay triangulation (less likely to be optimal)
            for i in range(n):
                for j in range(i+1, n):
                    if (i, j) not in delaunay_edges:
                        cluster_penalty[i, j] = 0.7
                        cluster_penalty[j, i] = 0.7
        except:
            # Fallback if Delaunay fails
            pass
    
    # 5. Combine all factors with weights
    heuristics_matrix = (
        inv_dist * 0.4 +          # Base inverse distance
        mst_bonus * 0.3 +         # MST structure bonus
        rank_weight * 0.2 +       # Relative rank weighting
        cluster_penalty * 0.1     # Geographic clustering
    )
    
    # 6. Sparsification: keep only promising edges
    # Strategy 1: Keep top k edges per node
    k = max(3, int(np.sqrt(n)))  # Adaptive k based on problem size
    sparse_matrix = np.zeros((n, n))
    
    for i in range(n):
        # Get indices of top k edges for node i (excluding self)
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Exclude diagonal
        top_k_indices = np.argpartition(row, -k)[-k:]
        
        # Keep these edges (symmetrically)
        for j in top_k_indices:
            if heuristics_matrix[i, j] > 0:
                sparse_matrix[i, j] = heuristics_matrix[i, j]
                sparse_matrix[j, i] = heuristics_matrix[j, i]
    
    # Strategy 2: Additional global threshold (keep top 20% of all edges)
    threshold = np.percentile(sparse_matrix[sparse_matrix > 0], 80) if np.any(sparse_matrix > 0) else 0
    sparse_matrix[sparse_matrix < threshold] = 0
    
    # Ensure symmetry and zero diagonal
    sparse_matrix = np.maximum(sparse_matrix, sparse_matrix.T)
    np.fill_diagonal(sparse_matrix, 0)
    
    return sparse_matrix
