import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with adaptive parameter tuning
    and structural awareness.
    
    Key innovations:
    1. Adaptive exponential regret with distribution-aware decay rates
    2. Multi-scale density clustering using local percentiles
    3. MST-integrated sparsification preserving connectivity guarantees
    4. Centrality-weighted degree balancing
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    epsilon = 1e-12
    
    # ===== 1. DISTRIBUTION ANALYSIS FOR ADAPTIVE PARAMETERS =====
    upper_tri = distance_matrix[np.triu_indices(n, 1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    coef_var = dist_std / (dist_mean + epsilon)
    
    # Adaptive parameters based on distribution characteristics
    if coef_var < 0.3:
        # Regular/uniform distances
        regret_decay = 1.5
        cluster_aggressiveness = 0.3
        sparsity_factor = 0.6
    elif coef_var < 0.7:
        # Moderate variation
        regret_decay = 2.0
        cluster_aggressiveness = 0.5
        sparsity_factor = 0.7
    else:
        # High variation/clustered
        regret_decay = 3.0
        cluster_aggressiveness = 0.7
        sparsity_factor = 0.8
    
    # ===== 2. ADAPTIVE EXPONENTIAL REGRET =====
    regret_scores = np.ones_like(distance_matrix)
    
    # Precompute node-specific statistics
    node_best_alts = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if np.any(dists < np.inf):
            # Use average of 2nd and 3rd best as reference
            sorted_dists = np.partition(dists, 2)[:3]
            node_best_alts[i] = np.mean(sorted_dists[1:3])  # Exclude the absolute minimum
    
    for i in range(n):
        if node_best_alts[i] > 0:
            for j in range(n):
                if i != j:
                    # Multi-reference regret
                    alt_i = node_best_alts[i]
                    alt_j = node_best_alts[j]
                    best_alt = min(alt_i, alt_j)
                    
                    if best_alt > 0:
                        regret_ratio = distance_matrix[i, j] / best_alt
                        # Adaptive exponential decay based on distribution
                        regret_scores[i, j] = np.exp(-regret_decay * (regret_ratio - 1))
        regret_scores[i, i] = 0
    
    # ===== 3. MULTI-SCALE DENSITY CLUSTERING =====
    density_bonus = np.ones_like(distance_matrix)
    
    if n > 3:
        # Compute local density thresholds per node
        local_thresholds = np.zeros(n)
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            # Use percentile based on cluster aggressiveness
            percentile = 100 * (1 - cluster_aggressiveness)
            local_thresholds[i] = np.percentile(dists, percentile)
        
        # Apply intra-cluster bonuses
        for i in range(n):
            close_mask = distance_matrix[i] < local_thresholds[i]
            close_mask[i] = False
            
            if np.any(close_mask):
                # Bonus increases with local density
                close_dists = distance_matrix[i, close_mask]
                avg_close = np.mean(close_dists)
                local_density = 1.0 / (avg_close + epsilon)
                global_density = 1.0 / (dist_mean + epsilon)
                
                # Normalized bonus with saturation
                density_ratio = local_density / (global_density + epsilon)
                bonus = 1.0 + 0.3 * np.tanh(density_ratio - 1.0)
                density_bonus[i, close_mask] = bonus
    
    # ===== 4. LOG-SCALED INVERSE DISTANCE =====
    with np.errstate(divide='ignore', invalid='ignore'):
        # Add small constant to avoid log(0)
        scaled_dist = np.log1p(distance_matrix)
        inv_scaled = 1.0 / (scaled_dist + epsilon)
    
    inv_scaled[np.isinf(inv_scaled)] = 0
    np.fill_diagonal(inv_scaled, 0)
    
    # Normalize to [0, 1]
    max_val = np.max(inv_scaled)
    if max_val > 0:
        inv_scaled = inv_scaled / max_val
    
    # ===== 5. MST CONNECTIVITY GUARANTEE =====
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # Enhanced MST bonus with distance-awareness
    mst_bonus = np.ones_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if mst_mask[i, j]:
                # Stronger bonus for shorter MST edges relative to alternatives
                alt_cost_i = np.mean(np.partition(distance_matrix[i], 1)[1:3])
                alt_cost_j = np.mean(np.partition(distance_matrix[j], 1)[1:3])
                relative_savings = (alt_cost_i + alt_cost_j) / (2 * distance_matrix[i, j] + epsilon)
                mst_bonus[i, j] = mst_bonus[j, i] = 1.0 + 0.5 * np.tanh(relative_savings - 1.0)
    
    # ===== 6. CENTRALITY-WEIGHTED DEGREE BALANCING =====
    centrality = np.zeros(n)
    for i in range(n):
        # Inverse distance centrality
        inv_dists = 1.0 / (distance_matrix[i] + epsilon)
        inv_dists[i] = 0
        centrality[i] = np.sum(inv_dists)
    
    # Normalize centrality
    max_centrality = np.max(centrality)
    if max_centrality > 0:
        centrality = centrality / max_centrality
    
    degree_balance = np.ones_like(distance_matrix)
    target_degree = 2  # TSP tour degree
    for i in range(n):
        # Penalize central nodes more heavily to prevent hubs
        centrality_factor = 0.5 + centrality[i] * 0.5  # [0.5, 1.0]
        # Estimate potential connections from this node
        row_scores_pre = inv_scaled[i] * regret_scores[i] * density_bonus[i] * mst_bonus[i]
        row_scores_pre[i] = 0
        
        # Count potentially promising edges
        threshold = np.percentile(row_scores_pre[row_scores_pre > 0], 50) if np.any(row_scores_pre > 0) else 0
        potential_degree = np.sum(row_scores_pre > threshold)
        
        if potential_degree > target_degree:
            # Apply stronger penalty to central nodes
            balance = target_degree / (potential_degree * centrality_factor + epsilon)
            degree_balance[i] = np.sqrt(balance)  # Soft penalty
    
    # ===== 7. COMBINED SCORE =====
    base_scores = inv_scaled * regret_scores * density_bonus * mst_bonus * degree_balance
    
    # ===== 8. MST-INTEGRATED SPARSIFICATION =====
    sparsified = np.zeros_like(base_scores)
    
    # Compute MST-based connectivity requirements
    mst_degrees = np.sum(mst_mask, axis=1)
    
    for i in range(n):
        row_scores = base_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Always include MST edges
        mst_edges = mst_mask[i]
        mst_count = mst_degrees[i]
        
        # Adaptive connection count based on MST degree and centrality
        if n <= 5:
            k = n - 1
        else:
            # Base connections plus adjustment for centrality
            base_k = max(3, int(np.sqrt(n)))
            centrality_bonus = int(centrality[i] * 2)
            k = min(n - 1, base_k + centrality_bonus)
        
        # Ensure we include all MST edges
        keep_mask = mst_edges.copy()
        
        # Add top non-MST edges if needed
        if mst_count < k:
            non_mst_scores = row_scores.copy()
            non_mst_scores[mst_edges] = -np.inf
            
            # Select top (k - mst_count) non-MST edges
            n_needed = k - mst_count
            if np.sum(non_mst_scores > -np.inf) >= n_needed:
                # Use adaptive threshold based on score distribution
                valid_scores = non_mst_scores[non_mst_scores > -np.inf]
                if len(valid_scores) > 0:
                    # Use percentile-based threshold
                    threshold = np.percentile(valid_scores, 100 * (1 - sparsity_factor))
                    top_mask = non_mst_scores >= max(threshold, np.max(valid_scores) * 0.2)
                    keep_mask = keep_mask | top_mask
        
        sparsified[i][keep_mask] = base_scores[i][keep_mask]
    
    # ===== 9. SYMMETRIZE AND ENHANCE CONNECTIVITY =====
    symmetric_scores = np.maximum(sparsified, sparsified.T)
    
    # Ensure minimum score for MST edges
    non_zero = symmetric_scores[symmetric_scores > 0]
    if len(non_zero) > 0:
        min_score = np.min(non_zero)
        symmetric_scores[mst_mask] = np.maximum(symmetric_scores[mst_mask], min_score * 0.3)
    
    # ===== 10. ADAPTIVE NORMALIZATION =====
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        # Power scaling based on distribution
        if coef_var < 0.5:
            power = 0.9  # Gentle scaling for regular distributions
        else:
            power = 0.7  # Stronger scaling for irregular distributions
        
        symmetric_scores = symmetric_scores / max_score
        symmetric_scores = np.power(symmetric_scores, power)
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
