import numpy as np
from scipy.spatial import distance_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization.
    Returns a matrix of the same shape as input where higher values indicate
    more promising edges for inclusion in a TSP solution.
    """
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # Base inverse distance with smoothing to avoid division by extremely small values
    eps = 1e-10
    inv_dist = 1.0 / (distance_matrix + eps)
    np.fill_diagonal(inv_dist, 0)  # Remove self-connections
    
    # Factor 1: Local neighborhood connectivity
    # For each node, identify top-k nearest neighbors and score edges to them higher
    k = max(2, n // 4)  # Adaptive k based on problem size
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        neighbors = np.argsort(distance_matrix[i])[1:k+1]
        # Scale inverse distance by neighbor rank (closer neighbors get higher weight)
        for rank, j in enumerate(neighbors):
            rank_weight = (k - rank) / k  # Linear decay from 1 to 0
            heuristics[i, j] += inv_dist[i, j] * (1 + 0.5 * rank_weight)
    
    # Factor 2: Global sparsification based on distance percentiles
    # Flatten the upper triangle (excluding diagonal)
    upper_tri_indices = np.triu_indices(n, k=1)
    distances_upper = distance_matrix[upper_tri_indices]
    
    if len(distances_upper) > 0:
        # Calculate dynamic threshold - keep top 30% of closest edges
        threshold = np.percentile(distances_upper, 30)
        
        # Apply symmetric sparsification
        for i in range(n):
            for j in range(i+1, n):
                if distance_matrix[i, j] > threshold:
                    # Strongly penalize long edges
                    heuristics[i, j] *= 0.1
                    heuristics[j, i] *= 0.1
    
    # Factor 3: Node degree regularization - prefer balanced connectivity
    node_scores = np.zeros(n)
    for i in range(n):
        # For each node, look at how many other nodes it could reasonably connect to
        reasonable_edges = np.sum(distance_matrix[i] < np.median(distance_matrix[i]))
        node_scores[i] = 1.0 / (reasonable_edges + 1)  # Lower score for nodes with many options
    
    # Adjust edge scores based on node regularization
    for i in range(n):
        for j in range(n):
            if i != j:
                # Edges connecting nodes with fewer options get bonus
                node_factor = (node_scores[i] + node_scores[j]) / 2
                heuristics[i, j] *= (1 + node_factor)
    
    # Factor 4: Two-tier priority system
    # Identify potentially "critical" edges (very short distances)
    if n > 2:
        critical_threshold = np.percentile(distance_matrix[distance_matrix > 0], 10)
        critical_mask = (distance_matrix > 0) & (distance_matrix <= critical_threshold)
        heuristics[critical_mask] *= 2.0  # Double score for critical edges
    
    # Final normalization and zero-setting for unpromising edges
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Apply final sparsification: zero out bottom 20% of edges
    if n > 1:
        flat_heuristics = heuristics.flatten()
        zero_threshold = np.percentile(flat_heuristics, 20)
        heuristics[heuristics < zero_threshold] = 0
    
    # Ensure symmetry for undirected TSP
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
