import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import cdist
from typing import Optional, Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic incorporating spatial clustering lookahead and
    context-aware regret adjustments with adaptive parameter learning.
    
    Key Innovations:
    1. Spatial clustering for lookahead cost estimation
    2. Context-aware regret with clustering penalties
    3. Dynamic MST importance based on cluster connectivity
    4. Adaptive parameter tuning via geometric features
    
    Args:
        distance_matrix: Symmetric distance matrix of shape (n, n)
        
    Returns:
        Heuristic scores matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # ==================== ADAPTIVE GEOMETRIC ANALYSIS ====================
    # Compute instance characteristics for parameter tuning
    triu_indices = np.triu_indices(n, k=1)
    if len(triu_indices[0]) > 0:
        distances = distance_matrix[triu_indices]
        distance_stats = {
            'mean': np.mean(distances),
            'std': np.std(distances),
            'q25': np.percentile(distances, 25),
            'q75': np.percentile(distances, 75)
        }
    else:
        distance_stats = {'mean': 1.0, 'std': 0.0, 'q25': 1.0, 'q75': 1.0}
    
    # Adaptive parameters based on instance geometry
    norm_factor = max(1e-8, distance_stats['q25'])
    density_k = max(2, min(10, int(np.sqrt(n)) + 1))
    cluster_factor = max(1.0, min(3.0, distance_stats['std'] / (distance_stats['mean'] + 1e-8)))
    
    # ==================== SPATIAL CLUSTERING LOOKAHEAD ====================
    # Estimate spatial positions via multidimensional scaling (simple approximation)
    # This creates pseudo-coordinates for clustering
    if n > 3:
        # Use distance matrix to create approximate 2D coordinates
        # Simple approach: use first two nodes as reference axes
        idx1, idx2 = 0, min(1, n-1)
        if n > 2:
            # Estimate third coordinate using triangle inequality
            pseudo_coords = np.zeros((n, 2))
            pseudo_coords[:, 0] = distance_matrix[:, idx1]
            pseudo_coords[:, 1] = distance_matrix[:, idx2]
            
            # Normalize for clustering stability
            if np.max(pseudo_coords) > 0:
                pseudo_coords = pseudo_coords / np.max(pseudo_coords)
            
            # Determine optimal number of clusters
            n_clusters = max(2, min(8, n // 10 + 2))
            
            # Simple k-means initialization
            cluster_centers = pseudo_coords[np.random.choice(n, n_clusters, replace=False)]
            cluster_labels = np.zeros(n, dtype=int)
            
            # Run a few iterations
            for _ in range(5):
                # Assign points to nearest cluster
                distances_to_centers = cdist(pseudo_coords, cluster_centers)
                cluster_labels = np.argmin(distances_to_centers, axis=1)
                
                # Update centers
                for c in range(n_clusters):
                    if np.sum(cluster_labels == c) > 0:
                        cluster_centers[c] = np.mean(pseudo_coords[cluster_labels == c], axis=0)
        else:
            cluster_labels = np.zeros(n, dtype=int)
    else:
        cluster_labels = np.zeros(n, dtype=int)
    
    # ==================== CONTEXT-AWARE REGRET CALCULATION ====================
    # Enhanced regret with cluster context and lookahead estimation
    regret_matrix = np.ones_like(distance_matrix)
    cluster_penalties = np.zeros_like(distance_matrix)
    
    for i in range(n):
        row_dists = distance_matrix[i].copy()
        row_dists[i] = np.inf
        
        # Find best alternatives in same and different clusters
        best_same_cluster = np.inf
        best_diff_cluster = np.inf
        
        for j in range(n):
            if i == j:
                continue
            
            if cluster_labels[i] == cluster_labels[j]:
                best_same_cluster = min(best_same_cluster, distance_matrix[i, j])
            else:
                best_diff_cluster = min(best_diff_cluster, distance_matrix[i, j])
        
        # Avoid infinite values for small clusters
        if np.isinf(best_same_cluster):
            best_same_cluster = best_diff_cluster if not np.isinf(best_diff_cluster) else 1.0
        if np.isinf(best_diff_cluster):
            best_diff_cluster = best_same_cluster if not np.isinf(best_same_cluster) else 1.0
        
        # Compute regrets with cluster context
        for j in range(n):
            if i == j:
                continue
            
            if cluster_labels[i] == cluster_labels[j]:
                # Intra-cluster edge: compare to best intra-cluster alternative
                alternative_dists = row_dists.copy()
                # Exclude current edge and edges from different clusters
                for k in range(n):
                    if cluster_labels[i] != cluster_labels[k]:
                        alternative_dists[k] = np.inf
                alternative_dists[j] = np.inf
                best_alt = np.min(alternative_dists)
                
                if np.isinf(best_alt):
                    best_alt = best_same_cluster
                
                # Stronger penalty for poor intra-cluster edges
                regret = max(distance_matrix[i, j] - best_alt, 0)
                base_penalty = np.exp(-regret / (best_alt + 1e-8))
                
                # Additional penalty based on cluster density
                cluster_size = np.sum(cluster_labels == cluster_labels[i])
                density_penalty = 1.0 + (cluster_size / n) * 0.5
                regret_matrix[i, j] = base_penalty * density_penalty
                
                # Cluster penalty: encourage keeping good intra-cluster connections
                cluster_penalties[i, j] = 1.2 if distance_matrix[i, j] <= best_same_cluster * 1.5 else 0.8
                
            else:
                # Inter-cluster edge: compare to best inter-cluster alternative
                alternative_dists = row_dists.copy()
                # Exclude current edge and edges from same cluster
                for k in range(n):
                    if cluster_labels[i] == cluster_labels[k]:
                        alternative_dists[k] = np.inf
                alternative_dists[j] = np.inf
                best_alt = np.min(alternative_dists)
                
                if np.isinf(best_alt):
                    best_alt = best_diff_cluster
                
                # More forgiving for inter-cluster edges (they're necessary for connectivity)
                regret = max(distance_matrix[i, j] - best_alt, 0)
                base_penalty = np.exp(-regret / (best_alt * 1.5 + 1e-8))
                
                # Bonus for connecting sparse clusters
                cluster_i_size = np.sum(cluster_labels == cluster_labels[i])
                cluster_j_size = np.sum(cluster_labels == cluster_labels[j])
                sparsity_bonus = 1.0 + (min(cluster_i_size, cluster_j_size) / n) * 0.3
                regret_matrix[i, j] = base_penalty * sparsity_bonus
                
                # Cluster penalty: critical inter-cluster connections get bonus
                if distance_matrix[i, j] <= best_diff_cluster * 1.2:
                    cluster_penalties[i, j] = 1.4
                else:
                    cluster_penalties[i, j] = 0.9
    
    # Symmetrize regret and cluster penalties
    regret_matrix = (regret_matrix + regret_matrix.T) / 2
    cluster_penalties = (cluster_penalties + cluster_penalties.T) / 2
    np.fill_diagonal(regret_matrix, 0)
    np.fill_diagonal(cluster_penalties, 0)
    
    # ==================== DYNAMIC MST STRUCTURAL ANALYSIS ====================
    # Compute MST with cluster-aware importance
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # Analyze MST structure for critical edges
    mst_bonus = np.ones_like(distance_matrix)
    
    # Identify edges that connect different clusters in MST
    mst_inter_cluster = mst_mask & (np.outer(cluster_labels, np.ones(n, dtype=int)) != cluster_labels[:, None])
    
    # Identify edges within dense clusters in MST
    mst_intra_cluster = mst_mask & (np.outer(cluster_labels, np.ones(n, dtype=int)) == cluster_labels[:, None])
    
    # Higher bonus for inter-cluster MST edges (critical connectivity)
    mst_bonus[mst_inter_cluster] = 2.0 * cluster_factor
    
    # Moderate bonus for intra-cluster MST edges
    mst_bonus[mst_intra_cluster] = 1.5
    
    # Additional bonus for MST edges that are also low-regret
    regret_threshold = np.percentile(regret_matrix[regret_matrix > 0], 75) if np.any(regret_matrix > 0) else 0.5
    high_regret_mst = mst_mask & (regret_matrix > regret_threshold)
    mst_bonus[high_regret_mst] *= 1.2
    
    # ==================== DENSITY-AWARE BASE SCORES ====================
    # Inverse distance with local density weighting
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Local density estimation via k-nearest neighbors
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_actual = min(density_k, n - 1)
        if k_actual > 0:
            k_nearest = np.partition(dists, k_actual)[:k_actual]
            local_density[i] = 1.0 / (np.mean(k_nearest) + 1e-12)
    
    # Geometric mean of endpoint densities with cluster adjustment
    density_matrix = np.sqrt(np.maximum(np.outer(local_density, local_density), 1e-12))
    
    # Distance penalty using adaptive normalization
    dist_penalty = np.exp(-distance_matrix / (norm_factor * 0.7))
    np.fill_diagonal(dist_penalty, 0)
    
    # ==================== COMBINE ALL COMPONENTS ====================
    # Combine with careful weighting
    base_scores = inv_dist * density_matrix * dist_penalty
    enhanced_scores = base_scores * regret_matrix * mst_bonus * cluster_penalties
    
    # ==================== ADAPTIVE SPARSIFICATION ====================
    # Preserve critical edges while removing unpromising ones
    sparsified = np.zeros_like(enhanced_scores)
    
    # Base connectivity based on problem characteristics
    if n <= 15:
        base_edges = min(6, n-1)
    elif n <= 100:
        base_edges = int(np.sqrt(n)) + 3
    else:
        base_edges = int(np.log2(n) * 4)
    
    for i in range(n):
        row_scores = enhanced_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Always preserve MST edges for this node
        node_mst_mask = mst_mask[i]
        
        # Determine adaptive edge count based on cluster connectivity
        mst_count = np.sum(node_mst_mask)
        intra_cluster_count = np.sum(cluster_labels == cluster_labels[i])
        
        # Nodes in small clusters need more external connections
        if intra_cluster_count < n // 4:
            target_edges = max(base_edges + 2, mst_count + 2)
        else:
            target_edges = max(base_edges, mst_count + 1)
        
        target_edges = min(target_edges, n-1)
        
        if np.sum(row_scores > -np.inf) > 0:
            # Keep MST edges plus top scoring non-MST edges
            non_mst_scores = row_scores.copy()
            non_mst_scores[node_mst_mask] = -np.inf
            
            if np.sum(non_mst_scores > -np.inf) > 0:
                needed_non_mst = target_edges - mst_count
                if needed_non_mst > 0:
                    valid_non_mst = non_mst_scores[non_mst_scores > -np.inf]
                    if len(valid_non_mst) > 0:
                        if needed_non_mst < len(valid_non_mst):
                            threshold = np.partition(valid_non_mst, -needed_non_mst)[-needed_non_mst]
                        else:
                            threshold = np.min(valid_non_mst)
                        
                        top_non_mst = (non_mst_scores >= threshold) & (non_mst_scores > -np.inf)
                        keep_mask = node_mst_mask | top_non_mst
                    else:
                        keep_mask = node_mst_mask
                else:
                    keep_mask = node_mst_mask
            else:
                keep_mask = node_mst_mask
        else:
            keep_mask = node_mst_mask
        
        sparsified[i][keep_mask] = enhanced_scores[i][keep_mask]
    
    # ==================== FINAL PROCESSING ====================
    # Symmetrize: keep edge if selected by either endpoint
    symmetric_scores = np.maximum(sparsified, sparsified.T)
    
    # Ensure critical edges have minimum scores
    critical_edges = mst_mask | (cluster_penalties > 1.3)
    if np.max(symmetric_scores) > 0:
        min_nonzero = np.min(symmetric_scores[symmetric_scores > 0])
        symmetric_scores[critical_edges] = np.maximum(symmetric_scores[critical_edges], min_nonzero * 0.7)
    
    # Final normalization
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
