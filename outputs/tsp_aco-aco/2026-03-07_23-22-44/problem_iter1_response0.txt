import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics function for TSP edge selection.
    
    Returns a sparse matrix where each element (i,j) represents the 
    relative promise of including edge (i,j) in the final tour.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance heuristic (shorter edges are better)
    with np.errstate(divide='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Nearest-neighbor sparsification
    # Keep only edges to k nearest neighbors for each node
    k = max(3, int(np.log2(n)))  # Adaptive k based on problem size
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        nearest = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, nearest] = True
    
    # Make symmetric (if i is near j, j is likely near i)
    nn_mask = nn_mask | nn_mask.T
    
    # 3. Two-hop path potential
    # For edge (i,j), consider the best alternative path i->k->j
    two_hop_potential = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            # Find the best intermediate node k (excluding i and j)
            mask = np.ones(n, dtype=bool)
            mask[[i, j]] = False
            if np.any(mask):
                best_alternative = np.min(distance_matrix[i, mask] + distance_matrix[mask, j])
                # Edge is promising if direct connection is significantly shorter than two-hop
                two_hop_potential[i, j] = two_hop_potential[j, i] = max(
                    0, best_alternative - distance_matrix[i, j]
                )
    
    # Normalize two-hop potential
    if np.max(two_hop_potential) > 0:
        two_hop_potential = two_hop_potential / np.max(two_hop_potential)
    
    # 4. Minimum Spanning Tree bonus
    try:
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_bonus = ((mst > 0) | (mst.T > 0)).astype(float)
    except:
        mst_bonus = np.ones_like(distance_matrix)
    
    # 5. Edge betweenness approximation
    # Higher score for edges that connect different clusters
    centrality = np.zeros_like(distance_matrix)
    if n > 4:
        # Simple clustering: nodes with similar nearest neighbors
        for i in range(n):
            # Get nearest neighbors of i and j
            for j in range(i+1, n):
                i_nn = set(np.argsort(distance_matrix[i])[1:4])
                j_nn = set(np.argsort(distance_matrix[j])[1:4])
                # Edge gets bonus if it connects different neighborhoods
                overlap = len(i_nn & j_nn)
                centrality[i, j] = centrality[j, i] = 1.0 / (1.0 + overlap)
    
    # 6. Combine all factors
    heuristics_matrix = (
        0.4 * inv_dist * nn_mask +  # Inverse distance weighted by NN sparsity
        0.3 * two_hop_potential +   # Two-hop path advantage
        0.2 * mst_bonus +           # MST inclusion bonus
        0.1 * centrality            # Betweenness centrality
    )
    
    # 7. Additional sparsification: keep only top edges for each node
    final_mask = np.zeros_like(heuristics_matrix, dtype=bool)
    for i in range(n):
        # Keep top m edges per node
        m = max(2, int(np.sqrt(n)))
        top_indices = np.argsort(heuristics_matrix[i])[-m:]
        final_mask[i, top_indices] = True
    
    # Make symmetric
    final_mask = final_mask | final_mask.T
    
    # Apply final sparsification
    heuristics_matrix = heuristics_matrix * final_mask
    
    # 8. Final normalization to avoid extreme values
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
