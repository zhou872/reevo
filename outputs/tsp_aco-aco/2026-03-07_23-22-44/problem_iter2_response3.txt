import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic incorporating:
    1. Inverse distance weighted by local point density
    2. Minimum Spanning Tree connectivity bonus
    3. Regret-based alternative cost penalties
    4. Topology-aware adaptive sparsification
    
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # ==================== 1. Inverse distance with density weighting ====================
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Local density estimation via k-nearest neighbors
    k_density = min(5, max(2, n // 10))
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k_density)[:k_density]
        local_density[i] = 1.0 / (np.mean(k_nearest) + 1e-8)
    
    density_weights = np.sqrt(np.outer(local_density, local_density))
    base_scores = inv_dist * density_weights
    
    # ==================== 2. MST connectivity bonus ====================
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_active = (mst > 0) | (mst.T > 0)
    # Strong bonus for MST edges (structurally essential)
    mst_bonus = np.where(mst_active, 2.0, 1.0)
    enhanced_scores = base_scores * mst_bonus
    
    # ==================== 3. Regret-based alternative cost penalties ====================
    regret_scores = np.zeros_like(enhanced_scores)
    
    for i in range(n):
        # Get distances to other nodes
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Find two nearest neighbors
        idx_sorted = np.argsort(dists)
        d1_idx, d2_idx = idx_sorted[0], idx_sorted[1]
        d1, d2 = dists[d1_idx], dists[d2_idx]
        
        # Regret for nearest neighbor: cost of using second best instead
        regret_nearest = max(d2 - d1, 1e-8)
        regret_scores[i, d1_idx] = np.exp(-regret_nearest / (d1 + 1e-8))
        
        # For other edges: penalty based on how much worse than best alternative
        for j in range(n):
            if i != j and j != d1_idx:
                # Alternative cost: minimum distance to i from nodes other than j
                alt_mask = np.ones(n, dtype=bool)
                alt_mask[j] = False
                alt_mask[i] = False
                if np.any(alt_mask):
                    alt_min = np.min(distance_matrix[i, alt_mask])
                    regret = max(distance_matrix[i, j] - alt_min, 0)
                    regret_scores[i, j] = np.exp(-regret / (alt_min + 1e-8))
    
    # Symmetrize regret scores
    regret_symmetric = (regret_scores + regret_scores.T) / 2
    
    # Combine with enhanced scores
    combined_scores = enhanced_scores * (1.0 + regret_symmetric)
    
    # ==================== 4. Topology-aware adaptive sparsification ====================
    # Determine MST degree for each node
    mst_degree = np.sum(mst_active, axis=1)
    
    # Adaptive edge retention: nodes with higher MST degree keep more connections
    sparsified = np.zeros_like(combined_scores)
    for i in range(n):
        row_scores = combined_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Dynamic k based on MST degree and problem size
        base_k = max(2, int(np.sqrt(n)))
        adaptive_k = min(n - 1, base_k + int(mst_degree[i] * 1.5))
        
        # Keep top adaptive_k edges
        if adaptive_k < len(row_scores):
            threshold = np.partition(row_scores, -adaptive_k)[-adaptive_k]
            mask = row_scores >= threshold
        else:
            mask = np.ones_like(row_scores, dtype=bool)
            mask[i] = False
        
        sparsified[i, mask] = combined_scores[i, mask]
    
    # Additional smooth penalty for long edges
    max_dist = np.max(distance_matrix)
    length_penalty = np.exp(-distance_matrix / (0.3 * max_dist + 1e-8))
    final_scores = sparsified * length_penalty
    
    # ==================== 5. Symmetrization and normalization ====================
    # Ensure symmetry for undirected TSP
    symmetric = (final_scores + final_scores.T) / 2
    np.fill_diagonal(symmetric, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(symmetric)
    if max_val > 0:
        symmetric = symmetric / max_val
    
    return symmetric
