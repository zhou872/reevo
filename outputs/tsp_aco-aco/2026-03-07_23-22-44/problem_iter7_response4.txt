import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Efficient TSP heuristic with exponential regret, adaptive weighting,
    MST connectivity bonus, and intelligent sparsification.
    
    Core Innovations:
    1. Exponential Regret Penalization: Smooth opportunity cost scoring
    2. Dynamic Ensemble Weighting: Problem-adaptive component fusion
    3. MST Connectivity Bonus: Prioritizing spanning structure edges
    4. Distance-based Sparsification: Focused edge retention
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. CORE COMPONENT COMPUTATION ====================
    
    # Inverse distance (closer = better)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # MST connectivity bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    
    # ==================== 2. EXPONENTIAL REGRET PENALIZATION ====================
    def compute_regret_scores() -> np.ndarray:
        """Compute regret-based scores using exponential penalization"""
        regret_scores = np.zeros((n, n))
        
        for i in range(n):
            # Get distances from node i
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            
            for j in range(i + 1, n):
                direct_cost = distance_matrix[i, j]
                
                # Find best alternative edges from i and j
                # Alternative from i (excluding j)
                alt_dists_i = dists_i.copy()
                alt_dists_i[j] = np.inf
                best_alt_i = np.min(alt_dists_i)
                
                # Alternative from j (excluding i)
                dists_j = distance_matrix[j].copy()
                dists_j[i] = np.inf
                dists_j[j] = np.inf
                best_alt_j = np.min(dists_j)
                
                # Combined alternative (simplified two-step)
                best_alternative = best_alt_i + best_alt_j
                
                # Compute opportunity cost (regret)
                regret = max(0, best_alternative - 2 * direct_cost)
                
                # Exponential decay: exp(-regret / best_alternative_distance)
                # Use direct_cost as scaling reference
                scale = max(direct_cost, 1e-8)
                regret_score = np.exp(-regret / scale)
                
                regret_scores[i, j] = regret_score
                regret_scores[j, i] = regret_score
        
        return regret_scores
    
    regret_scores = compute_regret_scores()
    
    # ==================== 3. DYNAMIC ENSEMBLE WEIGHTING ====================
    def compute_ensemble_weights() -> Tuple[float, float, float]:
        """Compute adaptive weights based on problem characteristics"""
        
        # Problem size factor
        size_factor = min(1.0, n / 100.0)  # Normalize by reference size
        
        # Distance statistics
        triu_indices = np.triu_indices(n, 1)
        distances = distance_matrix[triu_indices]
        
        if len(distances) == 0:
            return 0.4, 0.4, 0.2
        
        dist_std = np.std(distances)
        dist_mean = np.mean(distances)
        cv = dist_std / (dist_mean + 1e-8)  # Coefficient of variation
        
        # Sparsity indicator (not truly sparse, but distance variation)
        sparsity_indicator = min(1.0, cv * 2.0)
        
        # Base weights
        w_regret = 0.4
        w_inv = 0.3
        w_mst = 0.3
        
        # Adjust based on problem characteristics
        # Larger problems benefit more from regret
        w_regret += 0.2 * size_factor
        
        # High distance variation favors inverse distance
        w_inv += 0.2 * sparsity_indicator
        
        # MST importance increases with problem size
        w_mst += 0.1 * size_factor
        
        # Normalize to sum to 1
        total = w_regret + w_inv + w_mst
        w_regret /= total
        w_inv /= total
        w_mst /= total
        
        return w_regret, w_inv, w_mst
    
    w_regret, w_inv, w_mst = compute_ensemble_weights()
    
    # ==================== 4. COMPOSITE SCORE ====================
    # Create MST bonus component
    mst_bonus = np.where(mst_edges, 1.5, 1.0)
    
    # Weighted combination
    composite_scores = (
        w_regret * regret_scores +
        w_inv * inv_dist * mst_bonus  # Apply MST bonus to inverse distance
    )
    
    # Additional MST boost for connectivity
    composite_scores[mst_edges] *= 1.2
    
    # ==================== 5. INTELLIGENT SPARSIFICATION ====================
    def smart_sparsify(scores: np.ndarray) -> np.ndarray:
        """Retain most promising edges while preserving connectivity"""
        sparsified = np.zeros_like(scores)
        
        # Always keep MST edges for connectivity
        sparsified[mst_edges] = scores[mst_edges]
        
        # Determine number of edges to keep per node
        base_edges = max(3, min(5, int(np.log2(n)) + 2))
        
        # For each node, select top-k non-MST edges
        for i in range(n):
            # Get scores for this node (excluding self and MST edges)
            node_scores = scores[i].copy()
            node_scores[i] = -np.inf
            node_scores[mst_edges[i]] = -np.inf  # Already included
            
            # Find top-k candidates
            k = min(base_edges, n - 1 - np.sum(mst_edges[i]))
            if k > 0:
                top_indices = np.argpartition(node_scores, -k)[-k:]
                top_indices = top_indices[node_scores[top_indices] > 0]
                
                # Add top edges (bidirectional)
                sparsified[i, top_indices] = scores[i, top_indices]
                sparsified[top_indices, i] = scores[top_indices, i]
        
        return sparsified
    
    sparsified_scores = smart_sparsify(composite_scores)
    
    # ==================== 6. FINAL ENHANCEMENTS ====================
    # Distance-based attenuation (penalize very long edges)
    if n > 1:
        # Normalize distances
        max_dist = np.max(distance_matrix[distance_matrix < np.inf])
        if max_dist > 0:
            dist_penalty = np.exp(-distance_matrix / (0.5 * max_dist))
            sparsified_scores *= dist_penalty
    
    # Symmetrize final matrix
    symmetric_scores = (sparsified_scores + sparsified_scores.T) / 2
    
    # Normalize to [0, 1] range
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
