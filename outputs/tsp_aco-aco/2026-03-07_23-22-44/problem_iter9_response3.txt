import numpy as np
from typing import List
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Mutated TSP heuristic with simplified regret calculation, fixed ensemble weighting,
    spectral clustering, and fast tour boosting.
    
    Core Mutations:
    1. Simple Regret Calculation: Replace beam search with efficient pairwise alternative cost
    2. Fixed Ensemble Weighting: Use predetermined weights instead of online learning
    3. Spectral Clustering Affinity: Simplified Laplacian-based edge scoring
    4. Fast Tour Boosting: Use nearest-neighbor tours for edge reinforcement
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. BASIC COMPONENTS ====================
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # MST for connectivity
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    
    # ==================== 2. SIMPLE REGRET CALCULATION ====================
    def simple_regret() -> np.ndarray:
        """Compute regret as difference between edge cost and best alternatives"""
        regret_scores = np.zeros((n, n))
        
        # Precompute sorted distances for each node
        sorted_indices = []
        sorted_dists = []
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            idx = np.argsort(dists)
            sorted_indices.append(idx)
            sorted_dists.append(dists[idx])
        
        for i in range(n):
            for j in range(i + 1, n):
                # Best alternative from i (excluding j)
                alt_i_cost = sorted_dists[i][1] if sorted_indices[i][0] == j else sorted_dists[i][0]
                
                # Best alternative from j (excluding i)
                alt_j_cost = sorted_dists[j][1] if sorted_indices[j][0] == j else sorted_dists[j][0]
                
                # Regret formula: how much worse are the alternatives?
                direct_cost = distance_matrix[i, j]
                regret = (alt_i_cost + alt_j_cost) - 2 * direct_cost
                
                # Convert to score with sigmoid normalization
                scale = np.median(distance_matrix[np.triu_indices(n, 1)])
                regret_scores[i, j] = 1.0 / (1.0 + np.exp(-regret / (scale + 1e-8)))
        
        # Symmetrize
        return regret_scores + regret_scores.T
    
    regret_scores = simple_regret()
    
    # ==================== 3. SIMPLIFIED SPECTRAL CLUSTERING ====================
    def spectral_affinity() -> np.ndarray:
        """Compute edge affinity using Laplacian eigenvectors"""
        # Use fixed number of clusters
        k = max(2, min(3, n // 4))
        
        # Normalized Laplacian
        degree = np.sum(distance_matrix, axis=1)
        degree_sqrt = np.sqrt(degree)
        with np.errstate(divide='ignore', invalid='ignore'):
            norm_laplacian = np.eye(n) - distance_matrix / (degree_sqrt[:, None] * degree_sqrt[None, :])
        norm_laplacian = np.nan_to_num(norm_laplacian, nan=0.0)
        
        # Compute eigenvectors
        try:
            _, eigenvectors = np.linalg.eigh(norm_laplacian)
            spectral_embedding = eigenvectors[:, 1:k+1]
        except:
            # Fallback to random embedding if eigendecomposition fails
            spectral_embedding = np.random.randn(n, k)
        
        # Compute cosine similarities
        cluster_scores = np.ones((n, n))
        norms = np.linalg.norm(spectral_embedding, axis=1, keepdims=True)
        safe_norms = np.where(norms == 0, 1, norms)
        normalized_embedding = spectral_embedding / safe_norms
        
        similarity = normalized_embedding @ normalized_embedding.T
        
        # Combine with distance decay
        median_dist = np.median(distance_matrix)
        dist_decay = np.exp(-distance_matrix / (2 * median_dist + 1e-8))
        
        cluster_scores = 1.0 + similarity * dist_decay
        np.fill_diagonal(cluster_scores, 0)
        
        return cluster_scores
    
    cluster_scores = spectral_affinity()
    
    # ==================== 4. FIXED WEIGHT COMBINATION ====================
    # Fixed weights based on empirical performance
    w_regret = 0.4
    w_inv = 0.3
    w_cluster = 0.3
    
    composite = (
        w_regret * regret_scores +
        w_inv * inv_dist +
        w_cluster * cluster_scores
    )
    
    # ==================== 5. FAST TOUR BOOSTING ====================
    def fast_tour_boosting(scores: np.ndarray) -> np.ndarray:
        """Boost edges that appear in quick nearest-neighbor tours"""
        boosted = scores.copy()
        
        # Generate tours using nearest neighbor from random starts
        n_tours = min(10, n)
        edge_counts = np.zeros((n, n))
        
        for _ in range(n_tours):
            start = np.random.randint(0, n)
            visited = np.zeros(n, dtype=bool)
            tour = [start]
            visited[start] = True
            current = start
            
            while len(tour) < n:
                # Find nearest unvisited
                dists = distance_matrix[current].copy()
                dists[visited] = np.inf
                next_node = np.argmin(dists)
                
                # Count edge
                edge_counts[current, next_node] += 1
                edge_counts[next_node, current] += 1
                
                tour.append(next_node)
                visited[next_node] = True
                current = next_node
            
            # Count return edge
            edge_counts[current, start] += 1
            edge_counts[start, current] += 1
        
        # Normalize and apply boost
        if np.max(edge_counts) > 0:
            edge_boost = edge_counts / np.max(edge_counts)
            boosted *= (1.0 + 0.5 * edge_boost)  # Up to 50% boost
        
        # Additional boost for MST edges
        boosted[mst_edges] *= 1.2
        
        return boosted
    
    boosted_scores = fast_tour_boosting(composite)
    
    # ==================== 6. INTELLIGENT SPARSIFICATION ====================
    def smart_sparsify(scores: np.ndarray) -> np.ndarray:
        """Keep most promising edges while preserving connectivity"""
        sparsified = np.zeros_like(scores)
        
        # Always keep MST edges
        sparsified[mst_edges] = scores[mst_edges]
        
        # Determine degree limit
        base_degree = 3 if n <= 20 else 4
        max_degree = min(base_degree, n - 1)
        
        # For each node, keep top-k non-MST edges
        for i in range(n):
            # Get candidate edges
            mask = ~mst_edges[i] & (scores[i] > 0)
            candidates = scores[i][mask]
            indices = np.where(mask)[0]
            
            if len(candidates) > 0:
                # Sort by score
                order = np.argsort(-candidates)
                keep = min(len(candidates), max_degree)
                
                for idx in order[:keep]:
                    j = indices[idx]
                    if np.sum(sparsified[j] > 0) < max_degree:
                        sparsified[i, j] = scores[i, j]
                        sparsified[j, i] = scores[j, i]
        
        return sparsified
    
    sparsified = smart_sparsify(boosted_scores)
    
    # ==================== 7. FINAL PROCESSING ====================
    # Ensure symmetry
    symmetric = (sparsified + sparsified.T) / 2
    
    # Apply distance-based decay
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        decay = np.exp(-distance_matrix / (0.5 * max_dist + 1e-8))
        symmetric *= decay
    
    # Normalize
    max_val = np.max(symmetric)
    if max_val > 0:
        symmetric = symmetric / max_val
    
    np.fill_diagonal(symmetric, 0)
    
    return symmetric
