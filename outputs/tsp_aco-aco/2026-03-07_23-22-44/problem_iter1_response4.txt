import numpy as np
from scipy.spatial import ConvexHull

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection probability.
    
    Features:
    1. Sparsified nearest neighbor: Only top-k nearest edges get non-zero values
    2. Convex hull bonus: Edges on the approximate convex hull get extra weight
    3. Geometric sparsity threshold: Filter based on distance quantiles
    4. Distance quartile scoring: Combine multiple distance-based factors
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # Create a copy to avoid modifying the input
    heuristics_mat = np.zeros_like(distance_matrix, dtype=float)
    
    # 1. Basic inverse distance (starting point)
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    np.divide(1.0, distance_matrix, out=inv_dist, where=distance_matrix != 0)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Sparsity: Keep only top-k nearest neighbors for each node
    k = max(2, min(5, n // 4))  # Adaptive k based on problem size
    
    for i in range(n):
        # Get distances from node i to all others
        dist_from_i = distance_matrix[i].copy()
        dist_from_i[i] = np.inf  # Ignore self
        
        # Find k-nearest neighbors
        nearest_indices = np.argpartition(dist_from_i, k)[:k]
        
        # Only keep edges to k-nearest neighbors
        for j in nearest_indices:
            if j != i:
                heuristics_mat[i, j] += inv_dist[i, j]
    
    # 3. Convex hull bonus (approximation using extreme points)
    if n >= 4:
        try:
            # Find approximate "extreme" nodes using min/max coordinates
            # This is a heuristic approximation of convex hull nodes
            row_sums = distance_matrix.sum(axis=1)
            extreme_candidates = np.argsort(row_sums)
            
            # Take nodes with min and max total distances as extreme points
            n_extreme = min(8, n // 2)
            extreme_nodes = np.concatenate([
                extreme_candidates[:n_extreme//2],
                extreme_candidates[-n_extreme//2:]
            ])
            
            # Boost edges between extreme nodes
            for i in extreme_nodes:
                for j in extreme_nodes:
                    if i != j and heuristics_mat[i, j] > 0:
                        heuristics_mat[i, j] *= 1.5
        except:
            pass  # Fallback if convex hull approximation fails
    
    # 4. Distance quartile filtering
    # Compute distance statistics for sparsity
    non_zero_dists = distance_matrix[np.triu_indices(n, k=1)]
    if len(non_zero_dists) > 0:
        q1 = np.percentile(non_zero_dists, 25)
        q3 = np.percentile(non_zero_dists, 75)
        
        # Penalize very long edges (above Q3)
        long_edge_mask = distance_matrix > q3
        heuristics_mat[long_edge_mask] *= 0.3
        
        # Boost moderately short edges (between Q1 and median)
        median = np.median(non_zero_dists)
        moderate_edge_mask = (distance_matrix > q1) & (distance_matrix <= median)
        heuristics_mat[moderate_edge_mask] *= 1.2
    
    # 5. Symmetrize the matrix (undirected TSP)
    heuristics_mat = np.maximum(heuristics_mat, heuristics_mat.T)
    
    # 6. Final sparsification: Zero out very small values
    threshold = np.percentile(heuristics_mat[heuristics_mat > 0], 20) if np.any(heuristics_mat > 0) else 0
    heuristics_mat[heuristics_mat < threshold] = 0
    
    # 7. Normalize non-zero values to reasonable range
    max_val = heuristics_mat.max()
    if max_val > 0:
        heuristics_mat = heuristics_mat / max_val * 10
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics_mat, 0)
    
    return heuristics_mat
