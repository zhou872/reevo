import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray,
                  density_k: Optional[int] = None,
                  regret_temperature: Optional[float] = None,
                  lookahead_factor: float = 0.3) -> np.ndarray:
    """
    Enhanced TSP edge heuristic with lightweight triangle lookahead for critical edges.
    
    Features:
    1. Exponential regret penalization with adaptive temperature scaling
    2. Density-weighted inverse distance using adaptive k-nearest neighbors
    3. MST structural reinforcement with distance-decay bonuses
    4. Lightweight triangle lookahead for critical edges (MST edges)
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        density_k: Number of neighbors for density estimation (auto-tuned if None)
        regret_temperature: Scaling for regret exponential (auto-tuned if None)
        lookahead_factor: Influence weight for triangle lookahead scores [0, 1]
        
    Returns:
        Matrix of edge priorities in [0,1], same shape as input
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # ==================== ADAPTIVE PARAMETER TUNING ====================
    if density_k is None:
        density_k = max(2, min(8, n // 15 + 2))
    
    if regret_temperature is None:
        off_diag = distance_matrix[~np.eye(n, dtype=bool)]
        regret_temperature = np.percentile(off_diag, 25) if len(off_diag) > 0 else 1.0
    
    # ==================== 1. DENSITY-WEIGHTED BASE SCORES ====================
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(inv_dist, 0)
    
    # Local density estimation
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_actual = min(density_k, n - 1)
        if k_actual > 0:
            k_nearest = np.partition(dists, k_actual)[:k_actual]
            local_density[i] = 1.0 / (np.mean(k_nearest) + 1e-12)
    
    density_weight = np.sqrt(np.maximum(np.outer(local_density, local_density), 1e-12))
    base_scores = inv_dist * density_weight
    
    # ==================== 2. MST STRUCTURAL REINFORCEMENT ====================
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # MST distance decay bonus
    mst_distances = np.where(mst_mask, distance_matrix, np.inf)
    mst_median = np.median(mst_distances[mst_mask]) if np.any(mst_mask) else 1.0
    mst_norm = np.exp(-mst_distances / (mst_median + 1e-12))
    mst_bonus = 1.0 + np.where(mst_mask, 1.5 * mst_norm, 0)
    
    # ==================== 3. EXPONENTIAL REGRET PENALTY ====================
    regret_scores = np.ones_like(distance_matrix)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if n > 1:
            idx_sorted = np.argsort(dists)
            best_idx = idx_sorted[0]
            second_idx = idx_sorted[1] if n > 2 else best_idx
            best_dist, second_dist = dists[best_idx], dists[second_idx]
            
            for j in range(n):
                if i == j:
                    continue
                    
                if j == best_idx:
                    regret = max(second_dist - best_dist, 0)
                else:
                    regret = max(distance_matrix[i, j] - best_dist, 0)
                
                regret_penalty = np.exp(-regret / (regret_temperature + 1e-12))
                regret_scores[i, j] = regret_penalty
    
    regret_scores = (regret_scores + regret_scores.T) / 2
    
    # ==================== 4. LIGHTWEIGHT TRIANGLE LOOKAHEAD ====================
    lookahead_scores = np.ones_like(distance_matrix)
    
    if n > 3 and np.any(mst_mask):
        # Precompute nearest neighbors for efficiency
        nn_count = min(5, n - 2)
        nearest = np.argsort(distance_matrix, axis=1)[:, 1:nn_count + 1]
        
        for i in range(n):
            for j in range(i + 1, n):
                if not mst_mask[i, j]:
                    continue  # Focus lookahead on critical MST edges
                
                # Find best third node k to form triangle i-j-k-i
                min_triangle = float('inf')
                
                # Check k from i's and j's nearest neighbors
                candidates = set(nearest[i]).union(nearest[j])
                candidates.discard(i)
                candidates.discard(j)
                
                for k in candidates:
                    if k == i or k == j:
                        continue
                    triangle_cost = (distance_matrix[i, j] + 
                                     distance_matrix[j, k] + 
                                     distance_matrix[k, i])
                    min_triangle = min(min_triangle, triangle_cost)
                
                if min_triangle < float('inf'):
                    # Convert to reward (lower triangle cost = higher reward)
                    lookahead_reward = 1.0 / (min_triangle + 1e-12)
                    # Apply soft normalization relative to base score
                    base_val = base_scores[i, j] + 1e-12
                    normalized_reward = lookahead_reward / (base_val + lookahead_reward)
                    lookahead_scores[i, j] = lookahead_scores[j, i] = (
                        1.0 + lookahead_factor * normalized_reward)
    
    # ==================== 5. COMBINE COMPONENTS ====================
    reinforced_scores = base_scores * mst_bonus
    regret_adjusted = reinforced_scores * regret_scores
    enhanced_scores = regret_adjusted * lookahead_scores
    
    # ==================== 6. TOPOLOGY-AWARE SPARSIFICATION ====================
    mst_degree = np.sum(mst_mask, axis=1)
    degree_importance = 1.0 + mst_degree / np.max(mst_degree + 1e-12)
    
    sparsified = np.zeros_like(enhanced_scores)
    for i in range(n):
        row_scores = enhanced_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Adaptive edge retention based on MST centrality
        base_edges = max(2, int(np.sqrt(n)))
        adaptive_k = min(n - 1, int(base_edges * degree_importance[i]))
        
        if adaptive_k > 0:
            threshold_idx = -min(adaptive_k, len(row_scores) - 1)
            threshold = np.partition(row_scores, threshold_idx)[threshold_idx]
            mask = row_scores >= threshold
            sparsified[i, mask] = enhanced_scores[i, mask]
    
    # ==================== 7. FINAL PROCESSING ====================
    symmetric = (sparsified + sparsified.T) / 2
    np.fill_diagonal(symmetric, 0)
    
    # Global normalization to [0,1]
    max_val = np.max(symmetric)
    if max_val > 0:
        symmetric = symmetric / max_val
    
    return symmetric
