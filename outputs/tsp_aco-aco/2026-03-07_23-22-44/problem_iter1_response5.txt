import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection.
    Combines multiple factors and sparsifies unpromising edges.
    """
    n = len(distance_matrix)
    heuristics = np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance (like v1)
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)  # Avoid division by zero
    inv_dist -= np.diag(np.diag(inv_dist))  # Remove diagonal
    
    # 2. Nearest neighbor sparsification
    k = max(3, min(10, n // 4))  # Adaptive k-nearest neighbors
    nearest_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        indices = np.argsort(distance_matrix[i])[1:k+1]  # Exclude self
        nearest_mask[i, indices] = True
    
    # 3. MST-based edge prioritization
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst + mst.T) > 0  # Symmetrize
    
    # 4. Two-hop connectivity bonus
    two_hop_bonus = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Find best intermediate node
                intermediates = [distance_matrix[i, k] + distance_matrix[k, j] 
                               for k in range(n) if k not in (i, j)]
                if intermediates:
                    direct = distance_matrix[i, j]
                    best_indirect = min(intermediates)
                    two_hop_bonus[i, j] = best_indirect / (direct + 1e-12)
    
    # 5. Rank-based weighting
    rank_weights = np.zeros_like(distance_matrix)
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])
        for rank, j in enumerate(sorted_indices[1:], 1):  # Exclude self
            rank_weights[i, j] = (n - rank) / n  # Higher rank = higher weight
    
    # Combine factors
    heuristics = inv_dist.copy()
    
    # Apply sparsification: keep only promising edges
    promising_mask = nearest_mask | mst_edges | (two_hop_bonus > 1.2)
    heuristics[~promising_mask] = 0
    
    # Boost factors for promising edges
    boost = np.ones_like(heuristics)
    boost[mst_edges] *= 2.0          # MST edges get double weight
    boost[two_hop_bonus > 1.5] *= 1.5  # Strong two-hop connections
    boost *= rank_weights            # Incorporate rank weights
    
    heuristics *= boost
    
    # 6. Symmetrize and normalize
    heuristics = np.maximum(heuristics, heuristics.T)
    max_val = heuristics.max()
    if max_val > 0:
        heuristics /= max_val
    
    # Final sparsification: keep top edges per node
    final_mask = np.zeros_like(heuristics, dtype=bool)
    top_k = max(2, min(5, n // 3))
    for i in range(n):
        indices = np.argsort(heuristics[i])[-top_k:]
        final_mask[i, indices] = True
        final_mask[indices, i] = True
    
    heuristics[~final_mask] = 0
    
    return heuristics
