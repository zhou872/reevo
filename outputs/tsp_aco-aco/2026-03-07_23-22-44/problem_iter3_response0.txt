import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with MST-constrained sparsification.
    
    Key improvements over v1:
    1. Exponential penalty scaled by distance median rather than adaptive parameter
    2. MST edges are guaranteed preservation through dedicated sparsification step
    3. Simplified regret calculation without hyperbolic tangent saturation
    4. More aggressive long-edge penalization with clearer distance normalization
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance with stability handling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Exponential penalty using median distance as normalization
    # More interpretable than v1's adaptive scaling
    median_dist = np.median(distance_matrix[distance_matrix > 0])
    if median_dist == 0:  # Edge case: all zero distances
        exp_penalty = np.ones_like(distance_matrix)
    else:
        # Strong exponential decay: edges at median distance get ~13% of base score
        exp_penalty = np.exp(-distance_matrix / (0.5 * median_dist))
    
    # 3. MST connectivity preservation
    mst = minimum_spanning_tree(distance_matrix).toarray()
    # Make MST matrix symmetric for consistency
    mst_sym = np.maximum(mst, mst.T)
    
    # 4. Simplified regret-based node preference
    # Focus on how much worse the second-best connection is
    node_preference = np.ones_like(distance_matrix)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if len(dists) >= 2:
            # Find two nearest neighbors
            sorted_idx = np.argsort(dists)
            d1, d2 = dists[sorted_idx[0]], dists[sorted_idx[1]]
            
            # Simple linear regret: (second_best - best) / best
            if d1 > 0:
                regret = (d2 - d1) / d1
                # Apply to best edge only, bounded to [0, 1]
                node_preference[i, sorted_idx[0]] += min(0.5, regret * 0.3)
    
    # 5. Combine components
    base_scores = inv_dist * exp_penalty * node_preference
    
    # 6. MST-constrained adaptive sparsification
    sparsified = np.zeros_like(base_scores)
    
    for i in range(n):
        row_scores = base_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Always include MST edges for this node
        mst_mask = mst_sym[i] > 0
        
        # Determine how many additional edges to keep
        # Keep more edges in sparse graphs, fewer in dense ones
        if n > 3:
            # Use interquartile range of distances as density proxy
            non_inf_dists = distance_matrix[i][distance_matrix[i] < np.inf]
            if len(non_inf_dists) > 3:
                q75, q25 = np.percentile(non_inf_dists, [75, 25])
                iqr = q75 - q25
                avg_dist = np.mean(non_inf_dists)
                
                # High IQR relative to mean indicates sparse structure
                if avg_dist > 0:
                    density_factor = iqr / avg_dist
                    # Keep more edges in sparse graphs (high density_factor)
                    keep_ratio = 0.15 + 0.25 * min(1.0, density_factor)
                else:
                    keep_ratio = 0.2
            else:
                keep_ratio = 0.3
        else:
            keep_ratio = 0.5
        
        # Ensure minimum connectivity
        min_edges = max(2, int(0.1 * (n - 1)))
        max_edges = max(3, int(0.4 * (n - 1)))
        
        # Count MST edges for this node
        mst_count = np.sum(mst_mask)
        k_additional = max(0, min_edges - mst_count)
        k_keep = min(max_edges, max(k_additional, int(keep_ratio * (n - 1))))
        
        # Select top edges (including MST edges automatically)
        combined_mask = mst_mask.copy()
        if k_keep > mst_count:
            # Find top non-MST edges
            non_mst_scores = row_scores.copy()
            non_mst_scores[mst_mask] = -np.inf
            
            if np.any(non_mst_scores > -np.inf):
                # Get threshold for top k_keep-mst_count edges
                if k_keep - mst_count < len(non_mst_scores):
                    threshold = np.partition(non_mst_scores, -(k_keep - mst_count))[-(k_keep - mst_count)]
                    top_non_mst = non_mst_scores >= threshold
                else:
                    top_non_mst = non_mst_scores > -np.inf
                
                combined_mask = combined_mask | top_non_mst
        
        # Apply sparsification
        sparsified[i][combined_mask] = base_scores[i][combined_mask]
    
    # 7. Symmetrize: keep edge if either endpoint selects it
    symmetric_scores = np.maximum(sparsified, sparsified.T)
    
    # 8. Final normalization
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
