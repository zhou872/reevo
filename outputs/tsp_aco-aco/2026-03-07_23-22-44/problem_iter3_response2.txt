import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristic scoring with improved sparsification and penalty mechanisms.
    
    Key improvements over v1:
    1. MST-constrained sparsification: Always retain MST edges with adaptive bonus
    2. Exponential penalty using maximum distance scaling (0.5*max_d factor)
    3. Simplified regret calculation using k-nearest neighborhood differential
    4. Global edge filtering with MST-length bound (3× average MST edge distance)
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance scoring
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Calculate MST for connectivity constraints
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_binary = mst > 0
    
    # Compute average MST edge distance for bound calculation
    mst_distances = distance_matrix[mst_binary]
    avg_mst_dist = np.mean(mst_distances) if len(mst_distances) > 0 else np.mean(distance_matrix)
    
    # 3. Adaptive neighborhood analysis with simplified regret
    neighbor_regret = np.ones_like(distance_matrix)
    k = min(5, n - 1)  # Fixed k for simplicity
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if len(dists) >= k + 1:
            # Find k+1 smallest distances
            kth_idx = np.argpartition(dists, k)[:k+1]
            kth_dists = dists[kth_idx]
            
            # Regret: difference from nearest to k-th nearest
            if kth_dists[0] > 0:
                regret = (kth_dists[-1] - kth_dists[0]) / kth_dists[0]
                # Apply to all k nearest neighbors
                for idx in kth_idx[:k]:
                    neighbor_regret[i, idx] += 0.2 * np.tanh(regret)
    
    # 4. Local density estimation (simplified)
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if n > 1:
            # Use median distance as density proxy
            median_dist = np.median(dists[dists < np.inf])
            local_density[i] = 1.0 / (median_dist + 1e-8)
    
    # 5. Combine base components
    density_weight = np.sqrt(np.outer(local_density, local_density))
    base_scores = inv_dist * density_weight * neighbor_regret
    
    # 6. MST-constrained sparsification
    sparsified = np.zeros_like(base_scores)
    
    # Global edge filtering: keep edges within length bound
    length_bound = 3.0 * avg_mst_dist
    within_bound = distance_matrix < length_bound
    
    # For each node, select promising edges
    for i in range(n):
        row_scores = base_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Always keep MST edges for this node
        mst_edges = mst_binary[i]
        
        # Combine MST requirement with length bound
        candidate_mask = np.zeros(n, dtype=bool)
        candidate_mask[mst_edges] = True  # Keep MST edges
        
        # Add non-MST edges within length bound
        non_mst_within_bound = within_bound[i] & ~mst_edges
        candidate_mask[non_mst_within_bound] = True
        
        # Calculate keep ratio based on candidate count
        num_candidates = np.sum(candidate_mask)
        if num_candidates > 0:
            # Adaptive keep ratio: more candidates -> keep fewer proportionally
            keep_ratio = 0.3 + 0.2 * np.exp(-0.1 * num_candidates)
            k_keep = max(2, int(keep_ratio * num_candidates))
            
            # Select top k_keep candidates by score
            candidate_scores = row_scores.copy()
            candidate_scores[~candidate_mask] = -np.inf
            
            if k_keep < num_candidates:
                threshold = np.partition(candidate_scores, -k_keep)[-k_keep]
                selected = candidate_scores >= threshold
            else:
                selected = candidate_mask
            
            sparsified[i][selected] = base_scores[i][selected]
    
    # Symmetrize: keep edge if promising from either direction
    symmetric_scores = np.maximum(sparsified, sparsified.T)
    
    # 7. Exponential penalty using maximum distance scaling
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        # Use 0.5*max_d as scaling factor (as per suggestion)
        scale = 0.5 * max_dist
        length_penalty = np.exp(-distance_matrix / (scale + 1e-8))
        final_scores = symmetric_scores * length_penalty
    else:
        final_scores = symmetric_scores
    
    # 8. Apply MST connectivity bonus
    mst_bonus = 1.5  # Bonus factor for MST edges
    final_scores[mst_binary] *= mst_bonus
    
    # 9. Final normalization
    max_score = np.max(final_scores)
    if max_score > 0:
        final_scores = final_scores / max_score
    
    np.fill_diagonal(final_scores, 0)
    
    return final_scores
