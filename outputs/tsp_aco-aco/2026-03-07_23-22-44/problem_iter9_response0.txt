import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List, Tuple
import math

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Efficient TSP heuristic focusing on spatial partitioning and proximity-weighted regret.
    
    Core Innovations:
    1. Spatial Partitioning Index: Organizes nodes into spatial cells for local edge evaluation
    2. Proximity Weighted Regret: Combines distance-based regret with spatial clustering
    3. Local Search Integration: Uses quick 2-opt moves to identify promising edge patterns
    4. Confidence-Based Sparsification: Intelligently zeroes low-confidence edges
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # ==================== 1. BASIC PREPROCESSING ====================
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Normalize inverse distances to [0, 1]
    if np.max(inv_dist) > 0:
        inv_dist = inv_dist / np.max(inv_dist)
    
    # ==================== 2. SPATIAL PARTITIONING INDEX ====================
    def spatial_partition_score() -> np.ndarray:
        """Organize nodes into spatial cells and compute intra-cell affinity"""
        # Estimate optimal number of partitions
        n_partitions = max(2, int(np.sqrt(n)))
        
        # Create partitions based on distance quartiles
        partition_scores = np.zeros((n, n))
        
        # For small instances, use simpler approach
        if n <= 20:
            return np.ones_like(distance_matrix)
        
        # Compute centroid distances as simple spatial proxy
        centroid = np.mean(distance_matrix, axis=1)
        sorted_indices = np.argsort(centroid)
        
        # Create partitions along the sorted axis
        partition_size = max(2, n // n_partitions)
        partitions = []
        for i in range(0, n, partition_size):
            partitions.append(sorted_indices[i:min(i+partition_size, n)])
        
        # Assign high scores to intra-partition edges, moderate to adjacent partitions
        for idx, part in enumerate(partitions):
            for i in part:
                for j in part:
                    if i != j:
                        # Intra-partition edges get high score
                        partition_scores[i, j] = 1.0
                
                # Edges to adjacent partitions get moderate score
                if idx > 0:
                    for j in partitions[idx-1]:
                        if i != j:
                            partition_scores[i, j] = max(partition_scores[i, j], 0.7)
                if idx < len(partitions)-1:
                    for j in partitions[idx+1]:
                        if i != j:
                            partition_scores[i, j] = max(partition_scores[i, j], 0.7)
        
        # Ensure symmetry
        partition_scores = np.maximum(partition_scores, partition_scores.T)
        
        # Apply distance decay
        median_dist = np.median(distance_matrix[np.triu_indices(n, 1)])
        dist_decay = np.exp(-distance_matrix / (2 * median_dist + 1e-8))
        partition_scores *= dist_decay
        
        return partition_scores
    
    spatial_scores = spatial_partition_score()
    
    # ==================== 3. PROXIMITY WEIGHTED REGRET ====================
    def proximity_weighted_regret() -> np.ndarray:
        """Compute regret weighted by spatial proximity and alternative path quality"""
        regret_scores = np.zeros((n, n))
        
        # For each node, find k nearest neighbors
        k_neighbors = min(10, n-1)
        
        for i in range(n):
            # Get distances from node i
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            
            # Find nearest neighbors
            neighbor_indices = np.argsort(dists_i)[:k_neighbors]
            
            for j in neighbor_indices:
                if i == j:
                    continue
                    
                direct_cost = distance_matrix[i, j]
                
                # Find best alternative path of length 2 (i-k-j)
                best_alternative = float('inf')
                for k in range(n):
                    if k == i or k == j:
                        continue
                    alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                    if alt_cost < best_alternative:
                        best_alternative = alt_cost
                
                # Calculate regret
                regret = max(0, best_alternative - direct_cost)
                
                # Normalize regret by median distance
                median_dist = np.median(distance_matrix[np.triu_indices(n, 1)])
                normalized_regret = regret / (median_dist + 1e-8)
                
                # Convert to score (higher regret = higher score)
                regret_scores[i, j] = 1.0 - np.exp(-normalized_regret)
        
        # Symmetrize and apply spatial weighting
        regret_scores = (regret_scores + regret_scores.T) / 2
        regret_scores *= spatial_scores  # Weight by spatial proximity
        
        return regret_scores
    
    regret_scores = proximity_weighted_regret()
    
    # ==================== 4. LOCAL SEARCH PATTERN EXTRACTION ====================
    def extract_local_patterns() -> np.ndarray:
        """Extract edge patterns from quick local search improvements"""
        pattern_scores = np.zeros((n, n))
        
        if n < 4:
            return pattern_scores
        
        # Generate a few initial tours and improve them with 2-opt
        n_tours = min(10, max(3, n//5))
        
        for _ in range(n_tours):
            # Start with a random tour
            tour = list(range(n))
            np.random.shuffle(tour)
            tour.append(tour[0])
            
            # Quick 2-opt improvement (limited iterations)
            improved = _quick_two_opt(tour, distance_matrix, max_iterations=20)
            
            # Boost edges in improved tour
            for k in range(len(improved)-1):
                i, j = improved[k], improved[k+1]
                pattern_scores[i, j] += 1.0
                pattern_scores[j, i] += 1.0
        
        # Normalize pattern scores
        if np.max(pattern_scores) > 0:
            pattern_scores = pattern_scores / np.max(pattern_scores)
        
        # Apply MST-based filtering
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0) | (mst.T > 0)
        pattern_scores = np.where(mst_mask, pattern_scores * 1.5, pattern_scores)
        
        return pattern_scores
    
    def _quick_two_opt(tour: List[int], dist_matrix: np.ndarray, 
                      max_iterations: int = 20) -> List[int]:
        """Perform limited 2-opt improvements"""
        current_tour = tour[:-1]  # Remove closing edge
        n_nodes = len(current_tour)
        improved = True
        iterations = 0
        
        while improved and iterations < max_iterations:
            improved = False
            best_gain = 0
            best_i, best_j = -1, -1
            
            # Check only a subset of possible swaps
            for i in range(n_nodes - 3):
                for j in range(i + 2, min(i + 7, n_nodes)):
                    a, b = current_tour[i], current_tour[(i + 1) % n_nodes]
                    c, d = current_tour[j], current_tour[(j + 1) % n_nodes]
                    
                    current_cost = dist_matrix[a, b] + dist_matrix[c, d]
                    new_cost = dist_matrix[a, c] + dist_matrix[b, d]
                    gain = current_cost - new_cost
                    
                    if gain > best_gain:
                        best_gain = gain
                        best_i, best_j = i, j
            
            if best_gain > 0:
                # Perform the swap
                i, j = best_i, best_j
                current_tour[i+1:j+1] = reversed(current_tour[i+1:j+1])
                improved = True
            
            iterations += 1
        
        return current_tour + [current_tour[0]]
    
    pattern_scores = extract_local_patterns()
    
    # ==================== 5. CONFIDENCE-BASED SPARSIFICATION ====================
    def confidence_sparsification(base_scores: np.ndarray) -> np.ndarray:
        """Intelligently zero low-confidence edges while preserving connectivity"""
        sparsified = np.zeros_like(base_scores)
        
        # Always include MST edges with high confidence
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0) | (mst.T > 0)
        sparsified[mst_mask] = base_scores[mst_mask]
        
        # Determine adaptive connectivity level
        if n <= 10:
            target_degree = 4
        elif n <= 50:
            target_degree = min(5, n // 4)
        else:
            target_degree = min(6, n // 5)
        
        # For each node, select top edges by confidence
        for i in range(n):
            # Get candidate edges (excluding MST edges already included)
            candidates = []
            for j in range(n):
                if i != j and not mst_mask[i, j] and base_scores[i, j] > 0:
                    # Confidence score combines base score and distance
                    dist_factor = np.exp(-distance_matrix[i, j] / np.median(distance_matrix))
                    confidence = base_scores[i, j] * (0.7 + 0.3 * dist_factor)
                    candidates.append((confidence, j))
            
            # Sort by confidence
            candidates.sort(reverse=True, key=lambda x: x[0])
            
            # Select top edges, considering degree constraints
            selected = 0
            for confidence, j in candidates:
                if selected < target_degree:
                    sparsified[i, j] = base_scores[i, j]
                    selected += 1
        
        # Ensure symmetry
        sparsified = np.maximum(sparsified, sparsified.T)
        
        return sparsified
    
    # ==================== 6. COMPOSITE SCORE CALCULATION ====================
    # Dynamic weighting based on problem size
    if n <= 20:
        w_regret, w_spatial, w_pattern = 0.5, 0.3, 0.2
    elif n <= 100:
        w_regret, w_spatial, w_pattern = 0.4, 0.4, 0.2
    else:
        w_regret, w_spatial, w_pattern = 0.3, 0.5, 0.2
    
    # Combine components
    composite = (
        w_regret * regret_scores +
        w_spatial * spatial_scores +
        w_pattern * pattern_scores
    )
    
    # Apply inverse distance as a final weighting factor
    composite *= inv_dist
    
    # Apply confidence-based sparsification
    sparsified = confidence_sparsification(composite)
    
    # ==================== 7. FINAL PROCESSING ====================
    # Ensure symmetry
    symmetric = (sparsified + sparsified.T) / 2
    
    # Apply mild distance-based decay for very long edges
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        # Only penalize edges significantly longer than average
        avg_dist = np.mean(distance_matrix[np.triu_indices(n, 1)])
        long_edge_mask = distance_matrix > 2 * avg_dist
        decay_factor = np.where(long_edge_mask, 0.7, 1.0)
        symmetric *= decay_factor
    
    # Normalize to [0, 1] range
    max_val = np.max(symmetric)
    if max_val > 0:
        symmetric = symmetric / max_val
    
    np.fill_diagonal(symmetric, 0)
    
    return symmetric
