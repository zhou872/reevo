import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with regret mechanisms and adaptive sparsification.
    
    Features:
    1. Regret-based scoring: Prioritizes edges with few good alternatives
    2. Cluster density awareness: Rewards edges connecting sparse regions
    3. Dynamic percentile filtering: Adaptively prunes long edges
    4. Two-stage sparsification: Balances connectivity and selectivity
    
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Initialize with inverse distance (closer is better)
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    np.divide(1.0, distance_matrix, out=inv_dist, where=distance_matrix != 0)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Adaptive neighbor selection with dynamic sparsity
    # Base connectivity ensures at least 2 connections per node
    base_k = max(2, min(5, int(np.log2(n))))
    dynamic_k = base_k + int(0.1 * n)  # Scale with problem size
    
    # Initialize scoring matrix
    score_mat = np.zeros_like(distance_matrix, dtype=float)
    neighbor_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # 3. First pass: Basic neighbor connectivity with distance weighting
    for i in range(n):
        dist_row = distance_matrix[i].copy()
        dist_row[i] = np.inf
        k_effective = min(dynamic_k, n - 1)
        
        # Get k nearest neighbors
        nearest = np.argpartition(dist_row, k_effective)[:k_effective]
        neighbor_mask[i, nearest] = True
        
        # Initial scoring: inverse distance
        for j in nearest:
            if j != i:
                score_mat[i, j] = inv_dist[i, j]
    
    # 4. Regret-based penalty: Penalize edges with good alternatives
    regret_penalty = np.zeros_like(score_mat)
    
    for i in range(n):
        # Find neighbors for current node
        neighbors = np.where(neighbor_mask[i])[0]
        if len(neighbors) < 2:
            continue
            
        # Calculate regret as quality difference between best and second-best alternative
        neighbor_dists = distance_matrix[i, neighbors]
        sorted_idx = np.argsort(neighbor_dists)
        
        if len(sorted_idx) >= 2:
            best_dist = neighbor_dists[sorted_idx[0]]
            second_best = neighbor_dists[sorted_idx[1]]
            regret = second_best - best_dist
            
            # Normalize regret by average distance
            avg_dist = np.mean(neighbor_dists)
            if avg_dist > 0:
                normalized_regret = regret / avg_dist
                
                # Apply penalty inversely: edges with high regret (unique good options) get boosted
                # Edges with low regret (many alternatives) get penalized
                for j_idx, j in enumerate(neighbors):
                    if j == i:
                        continue
                    
                    # Distance ratio compared to best
                    dist_ratio = distance_matrix[i, j] / best_dist if best_dist > 0 else 1
                    
                    # Regret factor: higher for edges closer to best
                    regret_factor = np.exp(-dist_ratio * normalized_regret)
                    
                    # Penalty increases with more alternatives (low regret)
                    regret_penalty[i, j] = regret_factor
    
    # Apply regret adjustment
    score_mat *= regret_penalty
    
    # 5. Cluster density bonus: Reward edges connecting sparse regions
    density_bonus = np.zeros_like(score_mat)
    
    if n >= 4:
        # Estimate local density using nearest neighbor distances
        local_density = np.zeros(n)
        for i in range(n):
            # Use 3 nearest neighbors to estimate density
            dist_row = distance_matrix[i].copy()
            dist_row[i] = np.inf
            nearest_dists = np.partition(dist_row, 3)[:3]
            local_density[i] = 1.0 / np.mean(nearest_dists) if np.mean(nearest_dists) > 0 else 1.0
        
        # Normalize density estimates
        if np.max(local_density) > 0:
            local_density = local_density / np.max(local_density)
        
        # Apply density bonus: edges connecting low-density nodes get higher bonus
        for i in range(n):
            for j in range(i + 1, n):
                if not neighbor_mask[i, j]:
                    continue
                
                # Bonus is inversely proportional to average density
                avg_density = (local_density[i] + local_density[j]) / 2
                density_factor = 1.0 + (1.0 - avg_density) * 0.5
                
                density_bonus[i, j] = density_bonus[j, i] = density_factor
    
    score_mat *= density_bonus
    
    # 6. Minimum spanning tree reinforcement
    try:
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0) | (mst.T > 0)
        
        # Boost MST edges moderately
        score_mat[mst_mask] *= 1.3
    except:
        pass
    
    # 7. Dynamic percentile filtering
    # Use distance statistics to adaptively prune edges
    if n > 2:
        # Collect non-zero distances from active edges
        active_distances = []
        for i in range(n):
            active_idx = np.where(neighbor_mask[i])[0]
            active_distances.extend(distance_matrix[i, active_idx])
        
        if len(active_distances) > 0:
            q1 = np.percentile(active_distances, 25)
            q3 = np.percentile(active_distances, 75)
            
            # Penalize very long edges (above Q3)
            long_edge_penalty = np.zeros_like(score_mat)
            long_edge_mask = distance_matrix > q3
            long_edge_penalty[long_edge_mask] = 0.4
            
            # Moderate boost for edges in middle range
            median = np.median(active_distances)
            moderate_mask = (distance_matrix > q1) & (distance_matrix <= median)
            moderate_boost = np.zeros_like(score_mat)
            moderate_boost[moderate_mask] = 1.2
            
            # Apply adjustments
            score_mat = score_mat * (1 - long_edge_penalty) * moderate_boost
    
    # 8. Two-stage adaptive sparsification
    # First stage: Ensure minimum connectivity
    final_mask = neighbor_mask.copy()
    
    # Second stage: Prune to maintain selectivity
    for i in range(n):
        # Keep only top-m edges per node, where m adapts to node degree
        active_scores = score_mat[i, final_mask[i]]
        if len(active_scores) > 0:
            # Adaptive threshold: keep edges above median score
            median_score = np.median(active_scores)
            keep_mask = score_mat[i] >= median_score
            final_mask[i] = final_mask[i] & keep_mask
    
    # Ensure symmetry
    final_mask = final_mask | final_mask.T
    
    # Apply final mask
    score_mat = score_mat * final_mask
    
    # 9. Final normalization
    max_val = score_mat.max()
    if max_val > 0:
        score_mat = score_mat / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(score_mat, 0)
    
    return score_mat
