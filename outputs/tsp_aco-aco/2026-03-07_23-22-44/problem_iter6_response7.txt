import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import squareform
from scipy.cluster.hierarchy import fcluster, linkage

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edge selection with multi-step regret analysis
    and adaptive local weighting.
    
    Key innovations:
    1. Multi-step (2-3 step) regret lookahead with dynamic depth adjustment
    2. Fast spectral-inspired clustering using hierarchical methods
    3. Local property-based dynamic weighting (density, degree distribution)
    4. Connectivity-preserving sparsification with adaptive edge retention
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # 1. Base inverse distance with robust scaling
    non_zero_mask = ~np.eye(n, dtype=bool) & (distance_matrix > 0)
    if np.any(non_zero_mask):
        median_dist = np.median(distance_matrix[non_zero_mask])
        # Use robust scaling with interquartile range for better outlier handling
        q75, q25 = np.percentile(distance_matrix[non_zero_mask], [75, 25])
        iqr_dist = q75 - q25 if q75 > q25 else median_dist
        scale_factor = max(median_dist, iqr_dist * 0.5)
    else:
        scale_factor = 1.0
    
    scaled_dist = distance_matrix / (scale_factor + 1e-10)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (scaled_dist + 1e-10)
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Multi-step regret analysis with adaptive depth
    regret_scores = np.ones_like(distance_matrix)
    
    # Adaptive regret depth based on problem size
    if n <= 10:
        max_depth = 3  # For small problems, consider 3-step alternatives
    elif n <= 50:
        max_depth = 2  # For medium problems, 2-step is sufficient
    else:
        max_depth = 2  # For large problems, use efficient 2-step
    
    for i in range(n):
        for j in range(i + 1, n):
            if distance_matrix[i, j] == 0:
                regret_scores[i, j] = regret_scores[j, i] = 0
                continue
            
            best_alternative = np.inf
            
            # Consider alternative paths with increasing depth
            for depth in range(2, max_depth + 1):
                if depth == 2:
                    # Two-step paths i->k->j
                    for k in range(n):
                        if k != i and k != j:
                            alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                            best_alternative = min(best_alternative, alt_cost)
                elif depth == 3:
                    # Three-step paths i->k->l->j (sampled for efficiency)
                    # Sample a subset of intermediate nodes for large n
                    if n > 50:
                        # Sample sqrt(n) nodes for efficiency
                        sample_size = int(np.sqrt(n))
                        k_indices = np.random.choice([x for x in range(n) if x != i and x != j], 
                                                     min(sample_size, n-2), replace=False)
                    else:
                        k_indices = [k for k in range(n) if k != i and k != j]
                    
                    for k in k_indices:
                        for l in range(n):
                            if l != i and l != j and l != k:
                                alt_cost = (distance_matrix[i, k] + 
                                           distance_matrix[k, l] + 
                                           distance_matrix[l, j])
                                best_alternative = min(best_alternative, alt_cost)
                                # Early termination if we find a very good alternative
                                if alt_cost < distance_matrix[i, j] * 0.8:
                                    break
            
            if best_alternative < np.inf and best_alternative > 0:
                # Convert regret to score using sigmoid-like function
                # Higher score when direct edge is much better than alternatives
                regret_ratio = distance_matrix[i, j] / best_alternative
                # Exponential decay with smoothing
                regret_scores[i, j] = regret_scores[j, i] = np.exp(-3.0 * max(0, regret_ratio - 1))
            else:
                regret_scores[i, j] = regret_scores[j, i] = 1.0
    
    # 3. Fast hierarchical clustering for local structure detection
    cluster_bonus = np.ones_like(distance_matrix)
    
    if n >= 4:
        # Use hierarchical clustering with average linkage
        # Convert to condensed form for linkage
        condensed_dist = squareform(distance_matrix, checks=False)
        
        # Adaptive number of clusters based on problem size
        n_clusters = max(2, int(np.sqrt(n)))
        
        try:
            # Perform hierarchical clustering
            Z = linkage(condensed_dist, method='average')
            cluster_labels = fcluster(Z, n_clusters, criterion='maxclust')
            
            # Create cluster affinity matrix
            for i in range(n):
                for j in range(i + 1, n):
                    if cluster_labels[i] == cluster_labels[j]:
                        # Bonus for intra-cluster edges
                        # Adjust bonus based on relative distance within cluster
                        cluster_indices = np.where(cluster_labels == cluster_labels[i])[0]
                        if len(cluster_indices) > 1:
                            intra_cluster_dists = []
                            for k in cluster_indices:
                                for l in cluster_indices:
                                    if k < l:
                                        intra_cluster_dists.append(distance_matrix[k, l])
                            
                            if intra_cluster_dists:
                                median_intra = np.median(intra_cluster_dists)
                                if median_intra > 0:
                                    relative_dist = distance_matrix[i, j] / median_intra
                                    # Stronger bonus for edges much shorter than cluster median
                                    bonus = 1.0 + 0.7 * np.exp(-relative_dist * 2.0)
                                    cluster_bonus[i, j] = cluster_bonus[j, i] = bonus
        except:
            # Fallback to simple nearest-neighbor clustering if hierarchical fails
            pass
    
    # 4. Dynamic local weighting based on node properties
    local_weights = np.ones((n, n))
    
    if n >= 3:
        # Compute local density for each node
        local_density = np.zeros(n)
        for i in range(n):
            # Use k-nearest neighbors (k = min(5, n//4))
            k = max(2, min(5, n // 4))
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest_indices = np.argsort(dists)[:k]
            if len(nearest_indices) > 0:
                local_density[i] = np.mean(distance_matrix[i, nearest_indices])
        
        # Normalize densities
        if np.max(local_density) > np.min(local_density):
            normalized_density = (local_density - np.min(local_density)) / \
                                (np.max(local_density) - np.min(local_density))
        else:
            normalized_density = np.ones(n) * 0.5
        
        # Adjust edge weights based on endpoint densities
        for i in range(n):
            for j in range(i + 1, n):
                # Edges between high-density nodes get higher weight
                density_factor = 0.5 * (normalized_density[i] + normalized_density[j])
                # Edges between nodes with similar density get additional boost
                similarity_factor = 1.0 - abs(normalized_density[i] - normalized_density[j])
                local_weights[i, j] = local_weights[j, i] = 0.8 + 0.4 * density_factor * similarity_factor
    
    # 5. MST connectivity preservation
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 6. Combine components with dynamic local weighting
    base_scores = inv_dist * regret_scores * cluster_bonus * local_weights
    
    # Boost MST edges proportionally to their quality
    mst_bonus = np.ones_like(base_scores)
    if np.any(mst_mask):
        # Compute quality scores for MST edges
        mst_scores = base_scores[mst_mask]
        if len(mst_scores) > 0 and np.max(mst_scores) > np.min(mst_scores):
            # Normalize MST edge scores
            mst_min, mst_max = np.min(mst_scores), np.max(mst_scores)
            normalized_mst = (mst_scores - mst_min) / (mst_max - mst_min + 1e-10)
            
            # Apply proportional bonus
            bonus_values = 1.0 + 0.6 * normalized_mst
            mst_bonus[mst_mask] = bonus_values
    
    enhanced_scores = base_scores * mst_bonus
    
    # 7. Connectivity-preserving sparsification
    sparsified = np.zeros_like(enhanced_scores)
    
    # Calculate adaptive edge retention
    if n <= 5:
        target_edges = n * (n - 1) // 2  # Keep all edges for very small problems
    elif n <= 20:
        target_edges = int(0.6 * n * (n - 1) // 2)
    else:
        # For larger problems, retain edges proportional to n*log(n)
        target_edges = int(2.0 * n * np.log(n + 1))
    
    # Ensure minimum edges for connectivity
    min_edges = max(n - 1, target_edges // 2)
    target_edges = max(target_edges, min_edges)
    target_edges = min(target_edges, n * (n - 1) // 2)
    
    # Keep top edges globally
    upper_tri_indices = np.triu_indices(n, k=1)
    upper_scores = enhanced_scores[upper_tri_indices]
    
    if len(upper_scores) > target_edges:
        threshold = np.partition(upper_scores, -target_edges)[-target_edges]
        keep_mask_upper = upper_scores >= threshold
    else:
        keep_mask_upper = np.ones_like(upper_scores, dtype=bool)
    
    # Create symmetric keep matrix
    keep_matrix = np.zeros_like(enhanced_scores, dtype=bool)
    keep_matrix[upper_tri_indices] = keep_mask_upper
    keep_matrix = keep_matrix | keep_matrix.T
    
    # Ensure MST edges are always kept
    keep_matrix = keep_matrix | mst_mask
    
    # Apply sparsification
    sparsified[keep_matrix] = enhanced_scores[keep_matrix]
    
    # 8. Final refinement and normalization
    # Apply mild distance attenuation to very long edges
    if np.any(non_zero_mask):
        p90_dist = np.percentile(distance_matrix[non_zero_mask], 90)
        if p90_dist > 0:
            dist_attenuation = np.exp(-distance_matrix / (p90_dist + 1e-10))
            sparsified = sparsified * dist_attenuation
    
    # Ensure symmetry and zero diagonal
    sparsified = (sparsified + sparsified.T) / 2.0
    np.fill_diagonal(sparsified, 0)
    
    # Normalize non-zero scores to [0, 1]
    non_zero_mask = sparsified > 0
    if np.any(non_zero_mask):
        min_val = np.min(sparsified[non_zero_mask])
        max_val = np.max(sparsified[non_zero_mask])
        if max_val > min_val:
            sparsified[non_zero_mask] = (sparsified[non_zero_mask] - min_val) / (max_val - min_val)
    
    return sparsified
