import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List, Tuple
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristic with spectral affinity smoothing, fixed weight ensemble,
    adaptive density-weighted scoring, and intelligent sparsification.
    
    Key Improvements over v1:
    1. Uses spectral affinity smoothing with Fiedler vector for cluster detection
    2. Fixed weight ensemble for consistent performance
    3. Adaptive density-weighted scoring using k-nearest neighbors
    4. Simplified yet effective sparsification with MST preservation
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Suppress warnings for division by zero/invalid
    warnings.filterwarnings('ignore', category=RuntimeWarning)
    
    # ==================== 1. INVERSE DISTANCE WITH SAFETY ====================
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # ==================== 2. SPECTRAL AFFINITY SMOOTHING ====================
    def spectral_affinity_smoothing() -> np.ndarray:
        """Use graph Laplacian and Fiedler vector to identify natural clusters"""
        # Convert distance to similarity using Gaussian kernel
        sigma = np.median(distance_matrix[np.triu_indices(n, 1)])
        similarity = np.exp(-distance_matrix**2 / (2 * sigma**2))
        np.fill_diagonal(similarity, 0)
        
        # Compute normalized Laplacian
        degree = np.sum(similarity, axis=1)
        D_sqrt_inv = np.diag(1.0 / np.sqrt(degree + 1e-8))
        L_norm = np.eye(n) - D_sqrt_inv @ similarity @ D_sqrt_inv
        
        # Compute Fiedler vector (second smallest eigenvector)
        eigenvalues, eigenvectors = np.linalg.eigh(L_norm)
        fiedler_idx = np.argsort(eigenvalues)[1]  # Second smallest
        fiedler_vector = eigenvectors[:, fiedler_idx]
        
        # Compute spectral affinity scores
        spectral_scores = np.zeros((n, n))
        
        # Normalize Fiedler vector
        fiedler_norm = (fiedler_vector - np.min(fiedler_vector)) / (
            np.max(fiedler_vector) - np.min(fiedler_vector) + 1e-8)
        
        for i in range(n):
            for j in range(i + 1, n):
                # Similar nodes in spectral space get higher scores
                spectral_diff = abs(fiedler_norm[i] - fiedler_norm[j])
                # Use exponential decay: similar nodes → high score
                spectral_sim = np.exp(-spectral_diff / 0.1)
                
                # Combine with distance information
                dist_ratio = distance_matrix[i, j] / sigma
                distance_weight = np.exp(-dist_ratio)
                
                spectral_scores[i, j] = spectral_sim * distance_weight
                spectral_scores[j, i] = spectral_scores[i, j]
        
        return spectral_scores
    
    spectral_scores = spectral_affinity_smoothing()
    
    # ==================== 3. ADAPTIVE DENSITY-WEIGHTED SCORING ====================
    def adaptive_density_weighted_scoring() -> np.ndarray:
        """Weight inverse distances by local density using k-nearest neighbors"""
        k = min(5, n - 1)
        
        # Compute local densities using k-nearest neighbors
        densities = np.zeros(n)
        for i in range(n):
            # Get distances to other nodes
            dists = distance_matrix[i].copy()
            dists[i] = np.inf  # Exclude self
            
            # Find k smallest distances
            k_smallest = np.partition(dists, k)[:k]
            avg_knn_dist = np.mean(k_smallest[k_smallest < np.inf])
            
            # Density inversely proportional to average distance to neighbors
            densities[i] = 1.0 / (avg_knn_dist + 1e-8)
        
        # Normalize densities
        if np.max(densities) > 0:
            densities = densities / np.max(densities)
        
        # Create density-weighted scores
        density_weighted = np.zeros((n, n))
        for i in range(n):
            for j in range(i + 1, n):
                # Geometric mean of endpoint densities
                density_factor = np.sqrt(densities[i] * densities[j])
                
                # Combine with inverse distance
                inv_dist_ij = inv_dist[i, j] if distance_matrix[i, j] > 0 else 0
                density_weighted[i, j] = density_factor * inv_dist_ij
                density_weighted[j, i] = density_weighted[i, j]
        
        return density_weighted
    
    density_weighted = adaptive_density_weighted_scoring()
    
    # ==================== 4. FIXED WEIGHT ENSEMBLE COMBINATION ====================
    # Fixed weights as suggested: spectral, inverse distance, density-weighted
    w_spectral = 0.4
    w_inv_dist = 0.3
    w_density = 0.3
    
    # Normalize each component to [0,1] before combining
    def normalize_component(mat: np.ndarray) -> np.ndarray:
        max_val = np.max(mat)
        if max_val > 0:
            return mat / max_val
        return mat
    
    spectral_norm = normalize_component(spectral_scores)
    inv_dist_norm = normalize_component(inv_dist)
    density_norm = normalize_component(density_weighted)
    
    # Combine with fixed weights
    combined = (
        w_spectral * spectral_norm +
        w_inv_dist * inv_dist_norm +
        w_density * density_norm
    )
    
    # Ensure symmetry
    combined = (combined + combined.T) / 2
    
    # ==================== 5. DISTANCE-BASED ATTENUATION ====================
    # Penalize very long edges more aggressively
    max_dist = np.max(distance_matrix[np.triu_indices(n, 1)])
    if max_dist > 0:
        # Use exponential decay with adaptive scaling
        dist_ratio = distance_matrix / (max_dist + 1e-8)
        attenuation = np.exp(-3.0 * dist_ratio**2)  # Gaussian-like decay
        combined *= attenuation
    
    # ==================== 6. INTELLIGENT SPARSIFICATION ====================
    def intelligent_sparsification(scores: np.ndarray) -> np.ndarray:
        """Sparsify while preserving MST edges and connectivity"""
        sparsified = np.zeros_like(scores)
        
        # Always include MST edges for connectivity
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = (mst > 0) | (mst.T > 0)
        sparsified[mst_edges] = scores[mst_edges]
        
        # Determine adaptive degree limit
        if n <= 10:
            max_degree = 3
        elif n <= 50:
            max_degree = min(4, n // 5)
        else:
            max_degree = min(5, n // 10)
        
        current_degree = np.sum(sparsified > 0, axis=1)
        
        # Collect promising non-MST edges
        candidates = []
        for i in range(n):
            for j in range(i + 1, n):
                if not mst_edges[i, j] and scores[i, j] > 0:
                    # Consider both score and distance
                    score_norm = scores[i, j] / (np.max(scores) + 1e-8)
                    dist_norm = distance_matrix[i, j] / (max_dist + 1e-8)
                    # Favor high-score, short edges
                    effective_score = score_norm * np.exp(-dist_norm)
                    candidates.append((effective_score, i, j))
        
        # Sort by effective score
        candidates.sort(reverse=True, key=lambda x: x[0])
        
        # Greedy selection respecting degree constraints
        for score, i, j in candidates:
            if (current_degree[i] < max_degree and 
                current_degree[j] < max_degree):
                sparsified[i, j] = scores[i, j]
                sparsified[j, i] = scores[j, i]
                current_degree[i] += 1
                current_degree[j] += 1
        
        return sparsified
    
    sparsified = intelligent_sparsification(combined)
    
    # ==================== 7. FINAL PROCESSING ====================
    # Final distance-based refinement
    median_dist = np.median(distance_matrix[np.triu_indices(n, 1)])
    if median_dist > 0:
        dist_factor = np.exp(-distance_matrix / (2 * median_dist))
        sparsified *= dist_factor
    
    # Ensure symmetry and zero diagonal
    sparsified = (sparsified + sparsified.T) / 2
    np.fill_diagonal(sparsified, 0)
    
    # Normalize to [0, 1]
    max_score = np.max(sparsified)
    if max_score > 0:
        sparsified = sparsified / max_score
    
    # Restore warnings
    warnings.filterwarnings('default', category=RuntimeWarning)
    
    return sparsified
