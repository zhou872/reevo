import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic scoring for TSP edges with exponential regret penalization.
    
    Key improvements over v1:
    1. Enhanced regret calculation using exponential penalization
    2. MST edge guarantee with multiplicative bonus
    3. More aggressive adaptive sparsification
    4. Symmetric edge scoring with connectivity preservation
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance with safe handling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. MST connectivity - identify edges in minimum spanning tree
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 3. Exponential regret penalization
    regret_scores = np.ones_like(distance_matrix)
    for i in range(n):
        # Get distances from node i (excluding self)
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        
        # Find the best and second-best distances for regret calculation
        if np.any(dists_i < np.inf):
            # Get smallest distance (best alternative)
            min_dist = np.min(dists_i)
            # Get indices of edges to calculate regret
            valid_mask = dists_i < np.inf
            # Calculate regret: how much worse this edge is than the best alternative
            regret = (distance_matrix[i] - min_dist) / min_dist
            # Apply exponential decay to regret scores
            # Large regret → low score, small regret → high score
            regret_factor = np.exp(-regret)
            regret_factor[i] = 0  # Zero out diagonal
            regret_factor[~valid_mask] = 0  # Zero out invalid distances
            regret_scores[i] = regret_factor
    
    # Make regret symmetric by averaging
    regret_scores = (regret_scores + regret_scores.T) / 2
    
    # 4. Combine base components
    base_scores = inv_dist * regret_scores
    
    # 5. Apply MST bonus: multiply MST edges by factor 1.5
    mst_bonus = np.ones_like(base_scores)
    mst_bonus[mst_mask] = 1.5
    base_scores = base_scores * mst_bonus
    
    # 6. Adaptive sparsification with MST guarantee
    sparsified = np.zeros_like(base_scores)
    
    for i in range(n):
        row_scores = base_scores[i].copy()
        row_scores[i] = -np.inf  # Ignore self
        
        # Determine adaptive sparsity level based on problem size
        if n <= 5:
            keep_k = max(2, n - 1)
        elif n <= 10:
            keep_k = max(3, int(0.6 * (n - 1)))
        elif n <= 20:
            keep_k = max(4, int(0.4 * (n - 1)))
        elif n <= 50:
            keep_k = max(5, int(0.3 * (n - 1)))
        else:
            keep_k = max(6, int(0.2 * (n - 1)))
        
        # Always keep MST edges for this node
        mst_edges_i = mst_mask[i]
        mst_count = np.sum(mst_edges_i)
        
        # Ensure we keep at least MST edges
        keep_k = max(keep_k, mst_count)
        keep_k = min(keep_k, n - 1)  # Cannot keep more than n-1 edges
        
        # Select top-k edges (including all MST edges)
        if keep_k > 0:
            # Get threshold for top-k edges
            if keep_k < n - 1:
                # Use partition to get k-th largest element efficiently
                kth_largest = np.partition(row_scores, -keep_k)[-keep_k]
                top_k_mask = row_scores >= kth_largest
            else:
                top_k_mask = row_scores > -np.inf
            
            # Force inclusion of MST edges
            top_k_mask = top_k_mask | mst_edges_i
            
            # Apply mask
            sparsified[i][top_k_mask] = base_scores[i][top_k_mask]
    
    # 7. Symmetrize: keep edge if selected by either endpoint
    symmetric_scores = np.maximum(sparsified, sparsified.T)
    
    # 8. Ensure MST edges are never zero
    symmetric_scores[mst_mask] = np.maximum(symmetric_scores[mst_mask], 1e-8)
    
    # 9. Final normalization
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
