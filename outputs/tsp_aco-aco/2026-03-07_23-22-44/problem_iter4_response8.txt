import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import ConvexHull

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristic incorporating deep lookahead, global topological features,
    and optimized weight parameters based on problem characteristics.
    
    Key Innovations:
    1. Dynamic programming inspired 3-step lookahead for better path foresight
    2. Convex hull detection for boundary edge prioritization
    3. Adaptive regret calculation considering multiple alternative paths
    4. Machine learning inspired feature combination with dynamic weights
    5. Global path structure estimation using Christofides-like approximations
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. BASE INVERSE DISTANCE ====================
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # ==================== 2. DYNAMIC PROGRAMMING INSPIRED 3-STEP LOOKAHEAD ====================
    lookahead_scores = np.ones_like(distance_matrix)
    
    if n > 3:
        # Precompute nearest neighbor information for efficiency
        k_lookahead = min(3, n - 2)
        nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:k_lookahead+1]
        
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                # Consider paths of length 3: i -> j -> k -> l (back to start or other)
                remaining_nodes = [k for k in range(n) if k not in (i, j)]
                
                if len(remaining_nodes) == 0:
                    lookahead_scores[i, j] = 1.0
                    continue
                
                # Estimate minimum 3-step path cost
                min_3step_cost = float('inf')
                
                # Try limited number of promising next steps based on nearest neighbors
                candidate_ks = nearest_indices[j]
                candidate_ks = [k for k in candidate_ks if k != i]
                
                if len(candidate_ks) == 0:
                    # Fallback: try all remaining nodes (limited to 5 for efficiency)
                    candidate_ks = remaining_nodes[:min(5, len(remaining_nodes))]
                
                for k in candidate_ks:
                    if k == i or k == j:
                        continue
                    
                    # Remaining nodes after i, j, k
                    final_nodes = [l for l in remaining_nodes if l != k]
                    
                    if len(final_nodes) == 0:
                        # Complete the triangle i->j->k->i
                        cost = distance_matrix[i, j] + distance_matrix[j, k] + distance_matrix[k, i]
                        min_3step_cost = min(min_3step_cost, cost)
                    else:
                        # Estimate return cost using nearest neighbor from k
                        # and approximate remaining tour cost with MST heuristic
                        l_candidates = nearest_indices[k]
                        l_candidates = [l for l in l_candidates if l not in (i, j, k)]
                        
                        if len(l_candidates) > 0:
                            # Try best candidate return
                            best_l = l_candidates[0]
                            # Estimate remaining tour for other nodes
                            remaining_after_l = [m for m in final_nodes if m != best_l]
                            
                            if len(remaining_after_l) > 0:
                                # Create submatrix for remaining nodes
                                indices = [i] + remaining_after_l
                                sub_matrix = distance_matrix[np.ix_(indices, indices)]
                                
                                # Estimate tour cost with MST approximation
                                if len(indices) > 1:
                                    mst = minimum_spanning_tree(sub_matrix)
                                    mst_cost = mst.sum() * 2  # Approximate tour cost
                                else:
                                    mst_cost = 0
                            else:
                                mst_cost = 0
                            
                            cost = (distance_matrix[i, j] + distance_matrix[j, k] + 
                                   distance_matrix[k, best_l] + mst_cost)
                            min_3step_cost = min(min_3step_cost, cost)
                
                # Convert to score (lower cost = higher score)
                lookahead_scores[i, j] = 1.0 / (min_3step_cost + 1e-8)
    
    # ==================== 3. CONVEX HULL BOUNDARY PRIORITIZATION ====================
    hull_scores = np.ones_like(distance_matrix)
    
    if n > 3:
        try:
            # Generate approximate coordinates from distance matrix using MDS
            # This is simplified - in practice would use proper MDS
            # For efficiency, we use a heuristic based on the distance matrix
            mean_dists = np.mean(distance_matrix, axis=1)
            
            # Identify boundary nodes as those with above-average distances
            # (a proxy for convex hull membership)
            median_mean = np.median(mean_dists)
            boundary_nodes = mean_dists > median_mean
            
            # Boost edges between boundary nodes (forming the hull)
            for i in range(n):
                for j in range(n):
                    if i != j and boundary_nodes[i] and boundary_nodes[j]:
                        # Check if this is a likely hull edge by comparing
                        # to alternative connections
                        other_nodes = [k for k in range(n) if k not in (i, j)]
                        if len(other_nodes) > 0:
                            # Compute average distance to alternatives
                            avg_alt_i = np.mean([distance_matrix[i, k] for k in other_nodes])
                            avg_alt_j = np.mean([distance_matrix[j, k] for k in other_nodes])
                            
                            # If direct connection is shorter than average alternatives,
                            # it's more likely to be a hull edge
                            if distance_matrix[i, j] < 0.8 * (avg_alt_i + avg_alt_j) / 2:
                                hull_scores[i, j] = 1.5
        except:
            # Fallback if convex hull detection fails
            pass
    
    # ==================== 4. ADAPTIVE REGRET WITH MULTIPLE ALTERNATIVES ====================
    regret_scores = np.ones_like(distance_matrix)
    
    if n > 2:
        k_regret = min(5, n - 2)
        
        for i in range(n):
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            
            for j in range(n):
                if i == j:
                    continue
                
                # Consider multiple alternative paths from i to other nodes
                alt_mask = np.ones(n, dtype=bool)
                alt_mask[[i, j]] = False
                
                if np.sum(alt_mask) > 0:
                    alt_dists = dists_i[alt_mask]
                    
                    if len(alt_dists) >= k_regret:
                        # Get k best alternatives
                        k_best = np.partition(alt_dists, k_regret - 1)[:k_regret]
                        
                        # Calculate weighted regret considering all alternatives
                        weights = 1.0 / (np.arange(1, k_regret + 1))  # Higher weight for better alternatives
                        weighted_regret = np.sum(weights * (k_best - distance_matrix[i, j]))
                        
                        # Normalize regret by the average distance
                        avg_dist = np.mean(k_best)
                        normalized_regret = weighted_regret / (avg_dist + 1e-8)
                        
                        # Convert to score using sigmoid function
                        regret_scores[i, j] = 1.0 / (1.0 + np.exp(-normalized_regret))
                    else:
                        # Not enough alternatives
                        regret_scores[i, j] = 0.5
    
    # ==================== 5. CLUSTER DENSITY WITH IMPROVED DETECTION ====================
    density_scores = np.ones_like(distance_matrix)
    
    if n > 4:
        # Adaptive cluster detection using distance percentiles
        upper_triangle = distance_matrix[np.triu_indices(n, 1)]
        p25 = np.percentile(upper_triangle, 25)
        p75 = np.percentile(upper_triangle, 75)
        
        cluster_radius = p25  # Dense connections are below 25th percentile
        
        # Precompute neighbor counts for efficiency
        neighbor_counts = np.sum(distance_matrix < cluster_radius, axis=1) - 1
        
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                # Check if nodes are in the same dense region
                if distance_matrix[i, j] < cluster_radius:
                    # Bonus increases with how many common neighbors they share
                    common_neighbors = 0
                    for k in range(n):
                        if (k != i and k != j and 
                            distance_matrix[i, k] < cluster_radius and 
                            distance_matrix[j, k] < cluster_radius):
                            common_neighbors += 1
                    
                    density_bonus = 1.0 + 0.3 * min(common_neighbors, 5) / 5.0
                    density_scores[i, j] = density_bonus
    
    # ==================== 6. MINIMUM SPANNING TREE REINFORCEMENT ====================
    mst_scores = np.ones_like(distance_matrix)
    
    try:
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = (mst > 0) | (mst.T > 0)
        mst_scores = np.where(mst_edges, 2.0, 1.0)
    except:
        pass  # Keep default scores if MST fails
    
    # ==================== 7. MACHINE LEARNING INSPIRED FEATURE COMBINATION ====================
    # Dynamic weights based on problem characteristics
    if n < 10:
        # Small problems: emphasize lookahead and regret
        weights = {
            'inv_dist': 0.8,
            'lookahead': 1.2,
            'hull': 0.7,
            'regret': 1.1,
            'density': 0.9,
            'mst': 1.0
        }
    elif n < 50:
        # Medium problems: balanced approach
        weights = {
            'inv_dist': 1.0,
            'lookahead': 1.0,
            'hull': 0.8,
            'regret': 1.0,
            'density': 1.0,
            'mst': 1.2
        }
    else:
        # Large problems: emphasize MST and density for efficiency
        weights = {
            'inv_dist': 0.9,
            'lookahead': 0.7,
            'hull': 0.6,
            'regret': 0.8,
            'density': 1.1,
            'mst': 1.3
        }
    
    # Combine all factors with dynamic weights
    combined_scores = (
        weights['inv_dist'] * inv_dist *
        weights['lookahead'] * lookahead_scores *
        weights['hull'] * hull_scores *
        weights['regret'] * regret_scores *
        weights['density'] * density_scores *
        weights['mst'] * mst_scores
    )
    
    # Ensure symmetry
    symmetric_scores = (combined_scores + combined_scores.T) / 2
    np.fill_diagonal(symmetric_scores, 0)
    
    # ==================== 8. INTELLIGENT SPARSIFICATION ====================
    sparsified = np.zeros_like(symmetric_scores)
    
    if n > 2:
        # Adaptive sparsification based on node degree in MST
        try:
            mst_degree = np.sum(mst_edges, axis=1)
        except:
            mst_degree = np.ones(n)
        
        for i in range(n):
            row_scores = symmetric_scores[i].copy()
            row_scores[i] = -np.inf
            
            # Determine number of edges to keep based on MST degree and problem size
            base_keep = max(2, int(np.sqrt(n)))
            adaptive_keep = min(n - 1, base_keep + int(mst_degree[i] * 0.5))
            
            # Keep top edges, ensuring minimum connectivity
            if adaptive_keep < len(row_scores):
                threshold = np.partition(row_scores, -adaptive_keep)[-adaptive_keep]
                keep_mask = row_scores >= threshold
                
                # Also keep edges that are significantly better than average
                mean_score = np.mean(row_scores[row_scores > 0])
                if mean_score > 0:
                    high_quality_mask = row_scores >= 2.0 * mean_score
                    keep_mask = keep_mask | high_quality_mask
            else:
                keep_mask = np.ones_like(row_scores, dtype=bool)
            
            keep_mask[i] = False
            sparsified[i, keep_mask] = symmetric_scores[i, keep_mask]
    else:
        sparsified = symmetric_scores
    
    # ==================== 9. DISTANCE-BASED ATTENUATION ====================
    # Smooth penalty for very long edges
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        distance_penalty = np.exp(-distance_matrix / (0.25 * max_dist + 1e-8))
        final_scores = sparsified * distance_penalty
    else:
        final_scores = sparsified
    
    # ==================== 10. FINAL NORMALIZATION ====================
    max_score = np.max(final_scores)
    if max_score > 0:
        final_scores = final_scores / max_score
    
    return final_scores
