import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List, Tuple, Optional
import random

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristic with hierarchical clustering, population-based edge sampling,
    multi-scale regret analysis, and adaptive reinforcement learning.
    
    Core Innovations:
    1. Hierarchical Spatial Clustering: Multi-level cluster analysis for adaptive
       granularity in edge evaluation
    2. Population-Based Edge Sampling: Ant colony optimization to capture global
       solution patterns from multiple constructive agents
    3. Multi-Scale Regret Analysis: Regret computation across hierarchical cluster levels
       for comprehensive edge assessment
    4. Adaptive Reinforcement Learning: Dynamic component weighting based on solution
       quality feedback during sampling
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges (0-1 range)
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # ==================== 1. DISTANCE DISTRIBUTION ANALYSIS ====================
    flat_dists = distance_matrix[np.triu_indices(n, k=1)]
    dist_median = np.median(flat_dists)
    dist_q1 = np.percentile(flat_dists, 25)
    dist_q3 = np.percentile(flat_dists, 75)
    iqr = dist_q3 - dist_q1
    is_uniform = iqr / (dist_median + 1e-8) < 0.5
    
    # ==================== 2. HIERARCHICAL SPATIAL CLUSTERING ====================
    def hierarchical_clustering() -> Tuple[List[np.ndarray], List[List[int]]]:
        """Create hierarchical cluster levels for multi-scale analysis"""
        levels = []
        cluster_assignments = []
        
        # Level 0: All nodes as one cluster
        level0_clusters = [list(range(n))]
        cluster_assignments.append([0] * n)
        levels.append(distance_matrix)
        
        # Level 1: Coarse clustering (sqrt(n) clusters)
        n_clusters_l1 = max(2, min(int(np.sqrt(n)), n // 4))
        if n >= 8:
            # Use farthest-first seeding
            seeds = []
            remaining = set(range(n))
            
            # First seed: median-distance point
            center_dists = np.sum(distance_matrix, axis=1)
            first_seed = np.argmin(np.abs(center_dists - np.median(center_dists)))
            seeds.append(first_seed)
            remaining.remove(first_seed)
            
            # Add seeds using max-min distance
            for _ in range(min(n_clusters_l1 - 1, len(remaining))):
                max_min_dist = -1
                best_candidate = -1
                for candidate in remaining:
                    min_dist = min(distance_matrix[candidate, s] for s in seeds)
                    if min_dist > max_min_dist:
                        max_min_dist = min_dist
                        best_candidate = candidate
                if best_candidate != -1:
                    seeds.append(best_candidate)
                    remaining.remove(best_candidate)
            
            # Assign nodes to nearest seed
            l1_labels = np.zeros(n, dtype=int)
            for i in range(n):
                if i in seeds:
                    l1_labels[i] = seeds.index(i)
                else:
                    nearest_seed = min(seeds, key=lambda s: distance_matrix[i, s])
                    l1_labels[i] = seeds.index(nearest_seed)
            
            cluster_assignments.append(l1_labels.tolist())
            
            # Compute level 1 distance matrix (cluster centroids)
            l1_dist = np.zeros((n_clusters_l1, n_clusters_l1))
            for i in range(n_clusters_l1):
                nodes_i = np.where(l1_labels == i)[0]
                for j in range(i + 1, n_clusters_l1):
                    nodes_j = np.where(l1_labels == j)[0]
                    # Mean distance between clusters
                    if len(nodes_i) > 0 and len(nodes_j) > 0:
                        inter_dists = distance_matrix[nodes_i][:, nodes_j]
                        l1_dist[i, j] = l1_dist[j, i] = np.mean(inter_dists)
            levels.append(l1_dist)
        
        # Level 2: Fine clustering (more clusters for large instances)
        if n >= 20:
            n_clusters_l2 = max(4, min(n // 2, n_clusters_l1 * 3))
            # K-means style initialization from level 1 clusters
            l2_labels = np.zeros(n, dtype=int)
            
            # Initialize with nodes from level 1 clusters
            init_seeds = []
            for cluster_id in range(n_clusters_l1):
                cluster_nodes = np.where(l1_labels == cluster_id)[0]
                if len(cluster_nodes) > 0:
                    # Select representative nodes within cluster
                    sub_dist = distance_matrix[np.ix_(cluster_nodes, cluster_nodes)]
                    if len(cluster_nodes) > 1:
                        # Choose node with minimum total distance to others in cluster
                        total_dists = np.sum(sub_dist, axis=1)
                        representative = cluster_nodes[np.argmin(total_dists)]
                        init_seeds.append(representative)
            
            # Add random seeds if needed
            if len(init_seeds) < n_clusters_l2:
                remaining = set(range(n)) - set(init_seeds)
                additional = min(n_clusters_l2 - len(init_seeds), len(remaining))
                init_seeds.extend(random.sample(list(remaining), additional))
            
            # Assign nodes to nearest seed
            for i in range(n):
                if i in init_seeds:
                    l2_labels[i] = init_seeds.index(i)
                else:
                    nearest = min(init_seeds, key=lambda s: distance_matrix[i, s])
                    l2_labels[i] = init_seeds.index(nearest)
            
            cluster_assignments.append(l2_labels.tolist())
            
            # Compute level 2 distance matrix
            l2_dist = np.zeros((n_clusters_l2, n_clusters_l2))
            for i in range(n_clusters_l2):
                nodes_i = np.where(l2_labels == i)[0]
                for j in range(i + 1, n_clusters_l2):
                    nodes_j = np.where(l2_labels == j)[0]
                    if len(nodes_i) > 0 and len(nodes_j) > 0:
                        inter_dists = distance_matrix[nodes_i][:, nodes_j]
                        l2_dist[i, j] = l2_dist[j, i] = np.mean(inter_dists)
            levels.append(l2_dist)
        
        return levels, cluster_assignments
    
    cluster_levels, cluster_assignments = hierarchical_clustering()
    
    # ==================== 3. MULTI-SCALE REGRET ANALYSIS ====================
    def multi_scale_regret() -> np.ndarray:
        """Compute regret scores across hierarchical cluster levels"""
        regret_matrix = np.zeros((n, n))
        
        if n <= 3:
            return regret_matrix
        
        # Base regret at node level
        for i in range(n):
            # Get k nearest neighbors
            k = min(8, n - 1)
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            neighbor_indices = np.argsort(dists_i)[:k]
            
            for j in neighbor_indices:
                if i == j:
                    continue
                
                direct = distance_matrix[i, j]
                best_alt = np.inf
                
                # Consider alternatives through other neighbors
                for k_node in neighbor_indices:
                    if k_node != j:
                        alt_cost = distance_matrix[i, k_node] + distance_matrix[k_node, j]
                        if alt_cost < best_alt:
                            best_alt = alt_cost
                
                if best_alt < np.inf:
                    regret = max(best_alt - direct, 0)
                    # Normalize by local scale
                    local_scale = np.percentile(distance_matrix[i], 25)
                    if local_scale > 0:
                        norm_regret = regret / local_scale
                        regret_matrix[i, j] = 1.0 / (1.0 + norm_regret)
        
        # Apply hierarchical refinement
        if len(cluster_levels) > 1:
            # Level 1 cluster-based refinement
            if len(cluster_assignments) > 1:
                l1_labels = cluster_assignments[1]
                for i in range(n):
                    for j in range(n):
                        if i != j:
                            same_cluster = (l1_labels[i] == l1_labels[j])
                            if same_cluster:
                                # Boost intra-cluster edges
                                regret_matrix[i, j] *= 1.2
                            else:
                                # Moderate inter-cluster edges
                                regret_matrix[i, j] *= 0.8
            
            # Level 2 cluster-based refinement (if available)
            if len(cluster_assignments) > 2:
                l2_labels = cluster_assignments[2]
                for i in range(n):
                    for j in range(n):
                        if i != j:
                            same_fine_cluster = (l2_labels[i] == l2_labels[j])
                            same_coarse_cluster = (l1_labels[i] == l1_labels[j])
                            
                            if same_fine_cluster and not same_coarse_cluster:
                                # Edge connects different coarse clusters but same fine cluster
                                regret_matrix[i, j] *= 1.1
        
        # Symmetrize
        regret_matrix = (regret_matrix + regret_matrix.T) / 2
        return regret_matrix
    
    regret_scores = multi_scale_regret()
    
    # ==================== 4. POPULATION-BASED EDGE SAMPLING ====================
    def ant_colony_sampling() -> np.ndarray:
        """Use ant colony optimization to sample promising edges"""
        pheromone = np.ones((n, n))
        np.fill_diagonal(pheromone, 0)
        
        # ACO parameters
        n_ants = min(20, max(5, n // 4))
        n_iterations = min(15, max(3, n // 10))
        alpha = 1.0  # Pheromone importance
        beta = 5.0   # Distance importance
        rho = 0.1    # Evaporation rate
        q0 = 0.7     # Exploitation probability
        
        best_tour_length = float('inf')
        
        for iteration in range(n_iterations):
            ant_tours = []
            ant_lengths = []
            
            for ant in range(n_ants):
                # Start from random node
                start = random.randint(0, n - 1)
                tour = [start]
                unvisited = set(range(n)) - {start}
                
                while unvisited:
                    current = tour[-1]
                    
                    # Calculate probabilities for unvisited nodes
                    probabilities = []
                    candidates = list(unvisited)
                    
                    if len(candidates) == 0:
                        break
                    
                    # Compute heuristic information (inverse distance)
                    heuristic = np.zeros(len(candidates))
                    for idx, node in enumerate(candidates):
                        heuristic[idx] = 1.0 / (distance_matrix[current, node] + 1e-8)
                    
                    # Compute pheromone and heuristic combination
                    pheromone_values = np.array([pheromone[current, node] for node in candidates])
                    combined = (pheromone_values ** alpha) * (heuristic ** beta)
                    
                    if np.sum(combined) > 0:
                        probabilities = combined / np.sum(combined)
                    else:
                        probabilities = np.ones(len(candidates)) / len(candidates)
                    
                    # Choose next node
                    if random.random() < q0:
                        # Exploitation: choose best
                        next_node = candidates[np.argmax(combined)]
                    else:
                        # Exploration: probabilistic choice
                        next_node = random.choices(candidates, weights=probabilities)[0]
                    
                    tour.append(next_node)
                    unvisited.remove(next_node)
                
                # Close the tour
                tour.append(tour[0])
                
                # Apply 2-opt local search
                if n >= 4:
                    tour = apply_2opt(tour, distance_matrix)
                
                # Calculate tour length
                tour_length = sum(distance_matrix[tour[i], tour[i+1]] for i in range(len(tour)-1))
                ant_tours.append(tour)
                ant_lengths.append(tour_length)
                
                # Update best tour
                if tour_length < best_tour_length:
                    best_tour_length = tour_length
            
            # Update pheromone
            pheromone *= (1 - rho)  # Evaporation
            
            # Add pheromone from ant tours
            for tour, length in zip(ant_tours, ant_lengths):
                delta = 1.0 / length
                for i in range(len(tour) - 1):
                    u, v = tour[i], tour[i+1]
                    pheromone[u, v] += delta
                    pheromone[v, u] += delta
        
        # Normalize pheromone to [0, 1]
        if np.max(pheromone) > 0:
            pheromone = pheromone / np.max(pheromone)
        
        return pheromone
    
    def apply_2opt(tour: List[int], dist_matrix: np.ndarray) -> List[int]:
        """Apply 2-opt improvement to a tour"""
        current_tour = tour[:-1]
        n_nodes = len(current_tour)
        improved = True
        max_iterations = min(50, n_nodes * 2)
        iterations = 0
        
        while improved and iterations < max_iterations:
            improved = False
            best_gain = 0
            best_i, best_j = -1, -1
            
            for i in range(n_nodes):
                for j in range(i + 2, min(i + 10, n_nodes)):
                    if j == n_nodes - 1:
                        continue
                    
                    a, b = current_tour[i], current_tour[(i + 1) % n_nodes]
                    c, d = current_tour[j], current_tour[(j + 1) % n_nodes]
                    
                    current = dist_matrix[a, b] + dist_matrix[c, d]
                    new = dist_matrix[a, c] + dist_matrix[b, d]
                    gain = current - new
                    
                    if gain > best_gain:
                        best_gain = gain
                        best_i, best_j = i, j
            
            if best_gain > 0:
                # Reverse segment between i+1 and j
                i, j = best_i, best_j
                if i < j:
                    current_tour[i+1:j+1] = reversed(current_tour[i+1:j+1])
                improved = True
            
            iterations += 1
        
        return current_tour + [current_tour[0]]
    
    colony_scores = ant_colony_sampling()
    
    # ==================== 5. ADAPTIVE REINFORCEMENT LEARNING ====================
    def compute_composite_score() -> np.ndarray:
        """Dynamically combine components with reinforcement learning"""
        # Base inverse distance
        with np.errstate(divide='ignore', invalid='ignore'):
            inv_dist = 1.0 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        np.fill_diagonal(inv_dist, 0)
        if np.max(inv_dist) > 0:
            inv_dist = inv_dist / np.max(inv_dist)
        
        # MST-based scoring
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0) | (mst.T > 0)
        mst_scores = np.zeros_like(distance_matrix)
        mst_scores[mst_mask] = 1.0
        
        # Dynamic weighting based on reinforcement
        # Evaluate component effectiveness through sampling
        components = [regret_scores, colony_scores, inv_dist, mst_scores]
        component_names = ['regret', 'colony', 'inverse', 'mst']
        
        # Test each component with simple greedy construction
        component_performance = []
        for comp_idx, comp in enumerate(components):
            # Build a simple tour using component as edge weights
            tour = [0]
            used = set([0])
            
            while len(tour) < n:
                current = tour[-1]
                # Get available nodes sorted by component score
                available = [j for j in range(n) if j not in used]
                if not available:
                    break
                
                # Choose next node probabilistically based on component score
                scores = [comp[current, j] for j in available]
                if np.sum(scores) > 0:
                    probs = scores / np.sum(scores)
                    next_node = random.choices(available, weights=probs)[0]
                else:
                    next_node = random.choice(available)
                
                tour.append(next_node)
                used.add(next_node)
            
            # Close tour
            tour.append(tour[0])
            
            # Calculate tour length
            tour_length = sum(distance_matrix[tour[i], tour[i+1]] for i in range(len(tour)-1))
            component_performance.append(1.0 / (1.0 + tour_length))
        
        # Normalize performance scores
        if np.sum(component_performance) > 0:
            weights = component_performance / np.sum(component_performance)
        else:
            weights = np.ones(len(components)) / len(components)
        
        # Adaptive adjustment based on problem size and uniformity
        if n <= 20:
            # Small instances: emphasize regret and colony
            weights = weights * np.array([1.2, 1.2, 0.8, 0.8])
        elif is_uniform:
            # Uniform distances: emphasize spatial patterns
            weights = weights * np.array([1.0, 1.1, 0.9, 1.0])
        else:
            # Non-uniform: emphasize distance-based components
            weights = weights * np.array([1.1, 1.0, 1.1, 0.8])
        
        # Renormalize
        weights = weights / np.sum(weights)
        
        # Combine components
        composite = np.zeros_like(distance_matrix)
        for comp, weight in zip(components, weights):
            composite += weight * comp
        
        return composite
    
    composite = compute_composite_score()
    
    # ==================== 6. INTELLIGENT SPARSIFICATION ====================
    def intelligent_sparsification(scores: np.ndarray) -> np.ndarray:
        """Sparsify while preserving connectivity and important edges"""
        sparsified = np.zeros_like(scores)
        
        # Always include MST edges
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0) | (mst.T > 0)
        sparsified[mst_mask] = scores[mst_mask]
        
        # Adaptive degree limits
        if n <= 10:
            max_degree = 4
        elif n <= 50:
            max_degree = min(5, n // 3)
        elif n <= 200:
            max_degree = min(6, n // 4)
        else:
            max_degree = min(8, n // 5)
        
        # Collect candidate edges
        candidates = []
        for i in range(n):
            for j in range(i + 1, n):
                if not mst_mask[i, j] and scores[i, j] > 0:
                    # Score adjustment based on hierarchical clustering
                    cluster_bonus = 1.0
                    if len(cluster_assignments) > 1:
                        l1_labels = cluster_assignments[1]
                        if l1_labels[i] == l1_labels[j]:
                            cluster_bonus = 1.3  # Intra-cluster boost
                    
                    adjusted_score = scores[i, j] * cluster_bonus
                    candidates.append((adjusted_score, i, j))
        
        # Sort candidates
        candidates.sort(reverse=True, key=lambda x: x[0])
        
        # Current degrees
        current_degrees = np.sum(sparsified > 0, axis=1)
        
        # Add edges respecting degree limits and connectivity
        for score, i, j in candidates:
            if current_degrees[i] < max_degree and current_degrees[j] < max_degree:
                sparsified[i, j] = scores[i, j]
                sparsified[j, i] = scores[i, j]
                current_degrees[i] += 1
                current_degrees[j] += 1
        
        return sparsified
    
    sparsified = intelligent_sparsification(composite)
    
    # ==================== 7. FINAL REFINEMENT ====================
    # Ensure symmetry
    final = (sparsified + sparsified.T) / 2
    
    # Apply distance-based decay for very long edges
    avg_dist = np.mean(flat_dists)
    if avg_dist > 0:
        long_threshold = 2.5 * avg_dist
        decay = np.where(distance_matrix > long_threshold,
                        np.exp(-(distance_matrix - long_threshold) / avg_dist),
                        1.0)
        final *= decay
    
    # Boost intra-cluster edges one final time
    if len(cluster_assignments) > 1:
        l1_labels = cluster_assignments[1]
        for i in range(n):
            for j in range(i + 1, n):
                if l1_labels[i] == l1_labels[j]:
                    final[i, j] *= 1.1
                    final[j, i] *= 1.1
    
    # Normalize to [0, 1]
    max_val = np.max(final)
    if max_val > 0:
        final = final / max_val
    
    np.fill_diagonal(final, 0)
    
    return final
