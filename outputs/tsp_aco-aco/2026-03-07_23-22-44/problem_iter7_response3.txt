import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Efficient TSP heuristic focusing on core strategies:
    1. Exponential Regret Penalization: Smooth opportunity cost scoring
    2. Dynamic Ensemble Weighting: Adaptive component balancing
    3. MST Connectivity Bonus: Minimum Spanning Tree edge prioritization
    4. Distance-Scaled Clustering: Spatial grouping with distance awareness
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # ==================== 1. CORE COMPONENTS ====================
    
    # Inverse distance (simplified computation)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Minimum Spanning Tree connectivity bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = 1.0 + np.log1p(n) / 10  # Logarithmic scaling with problem size
    mst_component = np.where((mst > 0) | (mst.T > 0), mst_bonus, 1.0)
    
    # ==================== 2. EXPONENTIAL REGRET PENALIZATION ====================
    
    def compute_exponential_regret() -> np.ndarray:
        """Compute regret scores with exponential decay for smooth penalization"""
        regret_scores = np.zeros((n, n))
        
        # Compute regret using two-step alternatives (efficient O(n³) implementation)
        for i in range(n):
            # Precompute distances from node i (excluding self)
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            
            for j in range(n):
                if i == j:
                    continue
                
                direct_cost = distance_matrix[i, j]
                
                # Find best two-step alternative i→k→j
                best_alternative = float('inf')
                for k in range(n):
                    if k == i or k == j:
                        continue
                    alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                    if alt_cost < best_alternative:
                        best_alternative = alt_cost
                
                if best_alternative < float('inf'):
                    # Exponential regret penalization as per knowledge graph
                    regret = max(0, best_alternative - direct_cost)
                    scale = max(1e-8, distance_matrix[i, j])
                    # exp(-regret / best_alternative_distance) as suggested
                    regret_scores[i, j] = np.exp(-regret / scale)
        
        # Symmetrize and normalize
        regret_scores = (regret_scores + regret_scores.T) / 2
        max_regret = np.max(regret_scores)
        if max_regret > 0:
            regret_scores /= max_regret
        
        return regret_scores
    
    regret_scores = compute_exponential_regret()
    
    # ==================== 3. DISTANCE-SCALED CLUSTERING ====================
    
    def compute_cluster_affinity() -> np.ndarray:
        """Compute cluster affinity with distance-aware scaling"""
        # Use distance percentile for adaptive clustering
        dist_flat = distance_matrix[np.triu_indices(n, 1)]
        median_dist = np.median(dist_flat)
        q75_dist = np.percentile(dist_flat, 75)
        
        # Simple distance-based clustering: edges within reasonable distance get bonus
        cluster_scores = np.ones_like(distance_matrix)
        
        for i in range(n):
            for j in range(i + 1, n):
                dist = distance_matrix[i, j]
                
                # Exponential decay based on distance percentile
                if dist <= median_dist:
                    # Within median distance - strong cluster affinity
                    decay = np.exp(-dist / (2 * median_dist + 1e-8))
                    cluster_scores[i, j] = 1.0 + decay
                elif dist <= q75_dist:
                    # Moderate distance - moderate affinity
                    decay = np.exp(-dist / (median_dist + 1e-8))
                    cluster_scores[i, j] = 1.0 + 0.5 * decay
                else:
                    # Long distance - reduced affinity
                    decay = np.exp(-dist / (q75_dist + 1e-8))
                    cluster_scores[i, j] = 1.0 + 0.2 * decay
                
                cluster_scores[j, i] = cluster_scores[i, j]
        
        return cluster_scores
    
    cluster_scores = compute_cluster_affinity()
    
    # ==================== 4. DYNAMIC ENSEMBLE WEIGHTING ====================
    
    def compute_dynamic_weights() -> Tuple[float, float, float, float]:
        """Compute adaptive weights based on problem characteristics"""
        
        # Analyze problem characteristics
        dist_flat = distance_matrix[np.triu_indices(n, 1)]
        dist_mean = np.mean(dist_flat)
        dist_std = np.std(dist_flat)
        
        # Feature 1: Problem size scaling
        size_factor = min(1.0, 10.0 / n)  # Smaller problems get more balanced weights
        
        # Feature 2: Distance variability
        cv = dist_std / (dist_mean + 1e-8) if dist_mean > 0 else 1.0
        var_factor = min(1.0, cv * 2)  # Higher variability emphasizes regret
        
        # Feature 3: Sparsity indicator (using distance distribution)
        dist_q25 = np.percentile(dist_flat, 25)
        dist_q75 = np.percentile(dist_flat, 75)
        iqr = dist_q75 - dist_q25
        sparsity_factor = iqr / (dist_mean + 1e-8) if dist_mean > 0 else 1.0
        
        # Base weights with dynamic adjustments
        w_regret_base = 0.35
        w_inv_base = 0.25
        w_cluster_base = 0.25
        w_mst_base = 0.15
        
        # Adjust based on characteristics
        w_regret = w_regret_base * (1.0 + 0.5 * var_factor) * size_factor
        w_inv = w_inv_base * (1.0 + 0.3 * (1.0 - sparsity_factor)) * size_factor
        w_cluster = w_cluster_base * (1.0 + 0.4 * sparsity_factor) * size_factor
        w_mst = w_mst_base * (1.0 + 0.2 * (1.0 - var_factor)) * size_factor
        
        # Normalize to sum to 1
        total = w_regret + w_inv + w_cluster + w_mst
        return (w_regret/total, w_inv/total, w_cluster/total, w_mst/total)
    
    w_regret, w_inv, w_cluster, w_mst = compute_dynamic_weights()
    
    # ==================== 5. COMPOSITE SCORE CALCULATION ====================
    
    # Normalize components for fair weighting
    def safe_normalize(matrix: np.ndarray) -> np.ndarray:
        """Normalize matrix to [0,1] range safely"""
        max_val = np.max(matrix)
        if max_val > 0:
            return matrix / max_val
        return matrix
    
    inv_dist_norm = safe_normalize(inv_dist)
    regret_norm = safe_normalize(regret_scores)
    cluster_norm = safe_normalize(cluster_scores)
    
    # Weighted combination with MST multiplicative bonus
    base_composite = (
        w_regret * regret_norm +
        w_inv * inv_dist_norm +
        w_cluster * cluster_norm
    )
    
    # Apply MST connectivity bonus multiplicatively
    composite_scores = base_composite * mst_component
    
    # ==================== 6. INTELLIGENT SPARSIFICATION ====================
    
    def smart_sparsify(scores: np.ndarray) -> np.ndarray:
        """Sparsify while preserving connectivity and promising edges"""
        sparsified = np.zeros_like(scores)
        
        # Always preserve MST edges
        mst_mask = (mst > 0) | (mst.T > 0)
        sparsified[mst_mask] = scores[mst_mask]
        
        # Determine adaptive degree limits
        if n <= 10:
            max_degree = 4
        elif n <= 50:
            max_degree = min(5, n // 4)
        else:
            max_degree = min(6, n // 5)
        
        # Track current degrees
        current_degree = np.sum(sparsified > 0, axis=1)
        
        # Collect candidate non-MST edges
        candidates = []
        for i in range(n):
            for j in range(i + 1, n):
                if not mst_mask[i, j] and scores[i, j] > 0:
                    # Score adjusted by distance penalty
                    median_dist = np.median(distance_matrix)
                    dist_penalty = np.exp(-distance_matrix[i, j] / (2 * median_dist + 1e-8))
                    effective_score = scores[i, j] * dist_penalty
                    candidates.append((effective_score, i, j))
        
        # Sort by effective score (descending)
        candidates.sort(reverse=True, key=lambda x: x[0])
        
        # Greedy selection respecting degree constraints
        for score, i, j in candidates:
            if current_degree[i] < max_degree and current_degree[j] < max_degree:
                sparsified[i, j] = scores[i, j]
                sparsified[j, i] = scores[j, i]
                current_degree[i] += 1
                current_degree[j] += 1
        
        return sparsified
    
    sparsified_scores = smart_sparsify(composite_scores)
    
    # ==================== 7. FINAL PROCESSING ====================
    
    # Ensure symmetry
    symmetric_scores = (sparsified_scores + sparsified_scores.T) / 2
    
    # Final distance-based attenuation
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        # Stronger attenuation for very long edges
        dist_decay = np.exp(-distance_matrix / (0.5 * max_dist + 1e-8))
        symmetric_scores *= dist_decay
    
    # Normalize to [0, 1] range
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores /= max_score
    
    # Zero diagonal
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
