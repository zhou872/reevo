import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Efficient TSP heuristic combining exponential regret penalization,
    dynamic ensemble weighting, MST connectivity bonus, and adaptive sparsification.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. BASIC COMPONENTS ====================
    # Inverse distance (closer nodes get higher scores)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # MST connectivity bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = np.where((mst > 0) | (mst.T > 0), 1.5, 1.0)
    
    # ==================== 2. EXPONENTIAL REGRET PENALIZATION ====================
    regret_scores = np.zeros((n, n))
    
    for i in range(n):
        for j in range(i + 1, n):
            # Best two-step alternative path (i -> k -> j)
            alternative_costs = distance_matrix[i, :] + distance_matrix[:, j]
            alternative_costs[i] = np.inf  # Exclude i itself
            alternative_costs[j] = np.inf  # Exclude j itself
            best_alternative = np.min(alternative_costs)
            
            regret = max(0, best_alternative - distance_matrix[i, j])
            
            # Apply exponential decay: exp(-regret / best_alternative)
            if best_alternative > 0:
                scale_factor = best_alternative
            else:
                scale_factor = np.median(distance_matrix[np.triu_indices(n, 1)]) + 1e-8
            regret_scores[i, j] = np.exp(-regret / scale_factor)
            regret_scores[j, i] = regret_scores[i, j]
    
    np.fill_diagonal(regret_scores, 0)
    
    # ==================== 3. DYNAMIC ENSEMBLE WEIGHTING ====================
    # Compute problem characteristics
    triu_indices = np.triu_indices(n, 1)
    distances_flat = distance_matrix[triu_indices]
    
    # Coefficient of variation of distances
    dist_mean = np.mean(distances_flat)
    dist_std = np.std(distances_flat)
    coeff_var = dist_std / (dist_mean + 1e-8)
    
    # Sparsity indicator (relative spread of distances)
    dist_range = np.ptp(distances_flat) / (dist_mean + 1e-8)
    
    # Determine weights based on problem characteristics
    w_inv = 1.0 / (1.0 + coeff_var)  # Inverse distance weight decreases with high variance
    w_regret = 0.5 + 0.5 * coeff_var  # Regret weight increases with variance
    w_mst = 0.3 + 0.7 * np.exp(-dist_range)  # MST weight increases when distances are similar
    
    # Normalize weights to sum to 1
    total_weight = w_inv + w_regret + w_mst
    w_inv /= total_weight
    w_regret /= total_weight
    w_mst /= total_weight
    
    # ==================== 4. COMPOSITE SCORE ====================
    # Combine components with dynamic weights
    composite = (
        w_inv * inv_dist +
        w_regret * regret_scores +
        w_mst * mst_bonus
    )
    
    # Apply MST bonus multiplicatively (in addition to additive component)
    composite *= mst_bonus
    
    # ==================== 5. ADAPTIVE SPARSIFICATION ====================
    # Determine adaptive degree limit
    if n <= 10:
        max_degree = 3
    elif n <= 30:
        max_degree = 4
    else:
        max_degree = min(5, n // 6)
    
    sparsified = np.zeros_like(composite)
    current_degree = np.zeros(n, dtype=int)
    
    # Always include MST edges first
    mst_mask = (mst > 0) | (mst.T > 0)
    sparsified[mst_mask] = composite[mst_mask]
    current_degree = np.sum(sparsified > 0, axis=1)
    
    # Collect remaining candidate edges
    candidates = []
    for i in range(n):
        for j in range(i + 1, n):
            if not mst_mask[i, j] and composite[i, j] > 0:
                candidates.append((composite[i, j], i, j))
    
    # Sort by composite score (descending)
    candidates.sort(reverse=True, key=lambda x: x[0])
    
    # Greedy selection with degree constraints
    for score, i, j in candidates:
        if current_degree[i] < max_degree and current_degree[j] < max_degree:
            sparsified[i, j] = score
            sparsified[j, i] = score
            current_degree[i] += 1
            current_degree[j] += 1
    
    # ==================== 6. FINAL PROCESSING ====================
    # Symmetrize (already symmetric, but ensure)
    symmetric_scores = (sparsified + sparsified.T) / 2
    
    # Apply distance-based attenuation
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        dist_decay = np.exp(-distance_matrix / (0.4 * max_dist + 1e-8))
        symmetric_scores *= dist_decay
    
    # Normalize to [0, 1]
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
