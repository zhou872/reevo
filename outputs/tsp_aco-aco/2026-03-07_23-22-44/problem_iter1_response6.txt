import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import cdist

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Inverse distance weighted by local point density
    2. Minimum Spanning Tree connectivity bonus
    3. Nearest-neighbor based sparsification
    4. Symmetric edge scoring for undirected graphs
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Basic inverse distance (avoid division by zero)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Local density estimation using k-nearest neighbors
    k = min(5, n - 1)
    local_density = np.zeros(n)
    for i in range(n):
        # Get distances to other nodes, exclude self
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Average distance to k nearest neighbors
        k_nearest = np.partition(dists, k)[:k]
        local_density[i] = 1.0 / (np.mean(k_nearest) + 1e-8)
    
    # 3. Density-weighted distance score
    density_weights = np.sqrt(np.outer(local_density, local_density))
    weighted_scores = inv_dist * density_weights
    
    # 4. Minimum Spanning Tree connectivity bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = np.where((mst > 0) | (mst.T > 0), 1.5, 1.0)
    enhanced_scores = weighted_scores * mst_bonus
    
    # 5. Symmetrize the scores (TSP is undirected)
    symmetric_scores = (enhanced_scores + enhanced_scores.T) / 2
    
    # 6. Sparsification: keep only top edges for each node
    sparsity_factor = 0.3  # Keep top 30% of edges per node
    k_keep = max(2, int(sparsity_factor * (n - 1)))
    
    sparsified = np.zeros_like(symmetric_scores)
    for i in range(n):
        # Get scores for node i (excluding self)
        row_scores = symmetric_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Find threshold for top k_keep edges
        if k_keep < len(row_scores):
            threshold = np.partition(row_scores, -k_keep)[-k_keep]
            mask = row_scores >= threshold
        else:
            mask = np.ones_like(row_scores, dtype=bool)
            mask[i] = False
        
        sparsified[i][mask] = symmetric_scores[i][mask]
    
    # 7. Additional penalty for very long edges
    max_dist = np.max(distance_matrix)
    edge_length_penalty = np.exp(-distance_matrix / (0.5 * max_dist + 1e-8))
    final_scores = sparsified * edge_length_penalty
    
    # Normalize to [0, 1] range for stability
    max_score = np.max(final_scores)
    if max_score > 0:
        final_scores = final_scores / max_score
    
    return final_scores
