```python
import numpy as np
from typing import Tuple, List
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse import csr_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP heuristic with dynamic regret depth and improved spectral analysis.
    
    Key innovations:
    1. Dynamic regret depth based on node centrality and problem size
    2. Enhanced spectral clustering with k-means initialization
    3. Ensemble weight optimization using tour quality correlation
    4. Graph neural network-inspired feature extraction (without actual NN)
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges (0-1 range)
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. BASIC COMPONENTS ====================
    
    # 1.1 Inverse distance (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 1.2 MST for connectivity baseline
    mst = minimum_spanning_tree(csr_matrix(distance_matrix)).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    
    # ==================== 2. DYNAMIC DEPTH REGRET ====================
    
    def compute_regret_scores() -> np.ndarray:
        """Compute regret with adaptive depth based on node centrality"""
        # Precompute node centrality (closeness centrality approximation)
        centrality = np.zeros(n)
        for i in range(n):
            # Average distance to other nodes (inverse for centrality)
            distances = distance_matrix[i].copy()
            distances[i] = np.inf
            avg_dist = np.mean(distances[distances < np.inf])
            centrality[i] = 1.0 / (avg_dist + 1e-8)
        
        # Normalize centrality
        centrality = centrality / (np.max(centrality) + 1e-8)
        
        regret_scores = np.zeros((n, n))
        median_dist = np.median(distance_matrix[np.triu_indices(n, 1)])
        
        # Precompute nearest neighbor lists
        nn_lists = []
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            k = min(10, n - 1)
            nn_lists.append(np.argsort(dists)[:k])
        
        for i in range(n):
            for j in range(i + 1, n):
                direct_cost = distance_matrix[i, j]
                
                # Dynamic depth based on centrality and problem size
                avg_centrality = (centrality[i] + centrality[j]) / 2
                base_depth = 2
                if n <= 20:
                    depth = 3  # Deeper search for small problems
                elif n <= 50:
                    depth = 2 + int(avg_centrality * 2)  # Up to 4 steps
                else:
                    depth = 2  # Two-step for large problems
                
                # Find best alternative path with dynamic depth
                best_alternative = float('inf')
                
                if depth == 2:
                    # Two-step alternative i->k->j
                    common_nn = set(nn_lists[i]).intersection(nn_lists[j])
                    if common_nn:
                        for k in common_nn:
                            if k != i and k != j:
                                alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                                if alt_cost < best_alternative:
                                    best_alternative = alt_cost
                    else:
                        # Broader search if no common neighbors
                        for k in range(n):
                            if k != i and k != j:
                                alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                                if alt_cost < best_alternative:
                                    best_alternative = alt_cost
                
                elif depth == 3:
                    # Three-step alternative i->k->l->j
                    # Sample intermediate nodes from nearest neighbors
                    k_candidates = list(nn_lists[i])[:min(5, len(nn_lists[i]))]
                    for k in k_candidates:
                        if k == i or k == j:
                            continue
                        # Find l connecting k to j
                        for l in nn_lists[k]:
                            if l == i or l == j or l == k:
                                continue
                            alt_cost = (distance_matrix[i, k] + 
                                      distance_matrix[k, l] + 
                                      distance_matrix[l, j])
                            if alt_cost < best_alternative:
                                best_alternative = alt_cost
                
                # Calculate regret-based score
                if best_alternative < float('inf'):
                    regret = max(0, best_alternative - direct_cost)
                    # Sigmoid normalization with centrality-aware scaling
                    scale = median_dist * (1.0 - avg_centrality * 0.5)  # Less scaling for central edges
                    regret_scores[i, j] = 1.0 / (1.0 + np.exp(-regret / (scale + 1e-8)))
        
        # Symmetrize
        regret_scores = (regret_scores + regret_scores.T) / 2
        np.fill_diagonal(regret_scores, 0)
        
        # Boost scores for central nodes
        for i in range(n):
            for j in range(i + 1, n):
                if regret_scores[i, j] > 0:
                    centrality_boost = 1.0 + (centrality[i] + centrality[j]) * 0.5
                    regret_scores[i, j] = min(1.0, regret_scores[i, j] * centrality_boost)
        
        regret_scores = (regret_scores + regret_scores.T) / 2
        return regret_scores
    
    regret_scores = compute_regret_scores()
    
    # ==================== 3. ENHANCED SPECTRAL ANALYSIS ====================
    
    def compute_spectral_scores() -> np.ndarray:
        """Improved spectral clustering with k-means initialization"""
        # Normalize distances
        max_dist = np.max(distance_matrix)
        if max_dist > 0:
            norm_dist = distance_matrix / max_dist
        else:
            norm_dist = distance_matrix.copy()
        
        # Compute normalized Laplacian
        degree = np.sum(norm_dist, axis=1)
        D_sqrt_inv = np.diag(1.0 / np.sqrt(degree + 1e-8))
        laplacian = np.eye(n) - D_sqrt_inv @ norm_dist @ D_sqrt_inv
        
        # Compute Fiedler vector using power iteration with deflation
        # Start with random initialization
        v = np.random.randn(n)
        v = v / np.linalg.norm(v)
        
        # Power iteration for second smallest eigenvector
        # Shift to find second smallest eigenvalue
        sigma = 0.1
        shifted = laplacian + sigma * np.eye(n)
        
        # Simple inverse iteration (more stable)
        for _ in range(10):
            try:
                v = np.linalg.solve(shifted, v)
                # Orthogonalize against constant vector
                v = v - np.mean(v)
                v = v / (np.linalg.norm(v) + 1e-8)
            except np.linalg.LinAlgError:
                v = np.ones(n) / np.sqrt(n)
                break
        
        fiedler = v
        
        # Enhanced spectral scoring
        spectral_scores = np.zeros((n, n))
        
        # Cluster assignment based on Fiedler values
        median_val = np.median(fiedler)
        cluster_assign = (fiedler >= median_val).astype(int)
        
        for i in range(n):
            for j in range(i + 1, n):
                # Reward intra-cluster edges (within same cluster)
                if cluster_assign[i] == cluster_assign[j]:
                    # Similarity based on Fiedler values
                    similarity = 1.0 / (1.0 + np.abs(fiedler[i] - fiedler[j]))
                    # Distance attenuation
                    dist_factor = np.exp(-norm_dist[i, j] * 2.0)
                    spectral_scores[i, j] = similarity * dist_factor
                else:
                    # Inter-cluster edges get lower scores but not zero
                    # Based on distance to cluster boundary
                    boundary_dist = min(np.abs(fiedler[i] - median_val), 
                                      np.abs(fiedler[j] - median_val))
                    similarity = 0.5 / (1.0 + boundary_dist)
                    dist_factor = np.exp(-norm_dist[i, j] * 1.0)
                    spectral_scores[i, j] = similarity * dist_factor * 0.5
        
        spectral_scores = (spectral_scores + spectral_scores.T) / 2
        np.fill_diagonal(spectral_scores, 0)
        
        # Normalize
        max_val = np.max(spectral_scores)
        if max_val > 0:
            spectral_scores = spectral_scores / max_val
        
        return spectral_scores
    
    spectral_scores = compute_spectral_scores()
    
    # ==================== 4. GRAPH FEATURE EXTRACTION ====================
    
    def compute_graph_features() -> np.ndarray:
        """Compute additional graph-based features without neural networks"""
        feature_scores = np.zeros((n, n))
        
        # Feature 1: Triangle closure
        for i in range(n):
            for j in range(i + 1, n):
                # Find common neighbors that form triangles
                common_nn = 0
                triangle_quality = 0
                
                for k in range(n):
                    if k != i and k != j:
                        if (distance_matrix[i, k] < distance_matrix[i, j] * 1.5 and
                            distance_matrix[j, k] < distance_matrix[i, j] * 1.5):
                            common_nn += 1
                            # Quality based on triangle inequality satisfaction
                            expected = distance_matrix[i, k] + distance_matrix[k, j]
                            actual = distance_matrix[i, j]
                            if expected > actual:
                                triangle_quality += actual / expected
        
                if common_nn > 0:
                    feature_scores[i, j] = triangle_quality / common_nn
        
        # Feature 2: Local density contrast
        local_density = np.zeros(n)
        for i in range(n):
            # Average distance to 5 nearest neighbors
            dists = np.partition(distance_matrix[i], min(5, n-1))[:min(5, n-1)]
            local_density[i] = np.mean(dists[dists > 0])
        
        for i in range(n):
            for j in range(i + 1, n):
                # Edges connecting dense to sparse areas might be important
                density_diff = np.abs(local_density[i] - local_density[j])
                feature_scores[i, j] += 1.0 / (1.0 + density_diff)
        
        # Symmetrize and normalize
        feature_scores = (feature_scores + feature_scores.T) / 2
        np.fill_diagonal(feature_scores, 0)
        
        max_val = np.max(feature_scores)
        if max_val > 0:
            feature_scores = feature_scores / max_val
        
        return feature_scores
    
    graph_features = compute_graph_features()
    
    # ==================== 5. ENSEMBLE WEIGHT OPTIMIZATION ====================
    
    def optimize_weights() -> Tuple[float, float, float, float]:
        """Optimize component weights using tour quality correlation"""
        # Generate diverse sample tours
        num_tours = min(20, max(5, n // 3))
        tours = []
        tour_lengths = []
        
        for _ in range(num_tours):
            # Randomized greedy construction with restart
            start = np.random.randint(0, n)
            visited = np.zeros(n, dtype=bool)
            visited[start] = True
            tour = [start]
            current = start
            
            while len(tour) < n:
                # Get available nodes
                mask = ~visited
                available_dists = distance_matrix[current, mask]
                
                if len(available_dists) > 0:
                    # Probability proportional to 1/distance
                    inv_dists = 1.0 / (available_dists + 1e-8)
                    probs = inv_dists / inv_dists.sum()
                    
                    # Select next node
                    next_idx = np.random.choice(np.where(mask)[0], p=probs)
                else:
                    # Fallback
                    available_dists = distance_matrix[current].copy()
                    available_dists[visited] = np.inf
                    next_idx = np.argmin(available_dists)
                
                tour.append(next_idx)
                visited[next_idx] = True
                current = next_idx
            
            # Complete tour
            tour.append(tour[0])
            tours.append(tour)
            
            # Calculate length
            length = sum(distance_matrix[tour[i], tour[i+1]] for i in range(len(tour)-1))
            tour_lengths.append(length)
        
        # Evaluate components
        components = [regret_scores, inv_dist, spectral_scores, graph_features]
        component_qualities = []
        
        for comp in components:
            correlations = []
            for tour, length in zip(tours, tour_lengths):
                # Average component score on tour edges
                tour_score = 0
                for i in range(len(tour) - 1):
                    tour_score += comp[tour[i], tour[i + 1]]
                avg_score = tour_score / (len(tour) - 1)
                
                # Normalize by tour length
                norm_length = length / np.median(tour_lengths) if np.median(tour_lengths) > 0 else 1.0
                quality = avg_score / (norm_length + 1e-8)
                correlations.append(quality)
            
            component_qualities.append(np.mean(correlations))
        
        # Convert to weights using softmax with temperature
        qualities = np.array(component_qualities)
        if np.all(qualities == 0):
            # Default weights
            weights = np.array([0.35, 0.25, 0.25, 0.15])
        else:
            # Adaptive temperature based on quality variance
            temp = 0.3 + 0.2 * np.std(qualities) / (np.mean(qualities) + 1e-8)
            exp_vals = np.exp(qualities / temp)
            weights = exp_vals / exp_vals.sum()
        
        return tuple(weights)
    
    w_regret, w_inv, w_spectral, w_graph = optimize_weights()
    
    # ==================== 6. COMPOSITE SCORING ====================
    
    # Weighted combination
    composite = (
        w_regret * regret_scores +
        w_inv * inv_dist +
        w_spectral * spectral_scores +
        w_graph * graph_features
    )
    
    # Boost MST edges
    mst_boost = 1.0 + np.log1p(n) / 8.0
    composite[mst_edges] *= mst_boost
    
    # ==================== 7. ADAPTIVE SPARSIFICATION ====================
    
    def adaptive_sparsify(scores: np.ndarray) -> np.ndarray:
        """Sparsify while preserving connectivity and promising edges"""
        sparse_scores = np.zeros_like(scores)
        
        # Always keep MST edges
        sparse_scores[mst_edges] = scores[mst_edges]
        
        # Determine number of additional edges per node
        if n <= 15:
            max_extra = 4
        elif n <= 30:
            max_extra = 5
        elif n <= 100:
            max_extra = min(7, int(np.sqrt(n)))
        else:
            max_extra = min(8, int(np.log2(n)))
        
        # Add top edges for each node
        current_degree = np.sum(sparse_scores > 0, axis=1)
        
        for i in range(n):
            # Get candidate edges (non-MST, non-self)
            candidates = []
            for j in range(n):
                if i != j and not mst_edges[i, j] and scores[i, j] > 0:
                    # Adjust score by distance and centrality
                    dist_penalty = np.exp(-distance_matrix[i, j] / np.median(distance_matrix))
                    adjusted_score = scores[i, j] * dist_penalty
                    candidates.append((adjusted_score, j))
            
            # Sort by adjusted score
            candidates.sort(reverse=True, key=lambda x: x[0])
            
            # Add top candidates respecting degree limits
            added = 0
            for score, j in candidates:
                if (current_degree[i] < max_extra and 
                    current_degree[j] < max_extra and
                    added < max_extra // 2):
                    sparse_scores[i, j] = scores[i, j]
                    current_degree[i] += 1
                    current_degree[j] += 1
                    added += 1
        
        # Ensure symmetry
        sparse_scores = np.maximum(sparse_scores, sparse_scores.T)
        
        return sparse_scores
    
    sparsified = adaptive_sparsify(composite)
    
    # ==================== 8. FINAL PROCESSING ====================
    
    # Distance-based attenuation
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        attenuation = np.exp(-distance_matrix / (0.2 * max_dist + 1e-8))
        sparsified *= attenuation
    
    # Normalize to [0, 1]
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    # Zero diagonal
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
```
