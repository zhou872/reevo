import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection probability.
    
    Enhanced Features:
    1. Minimum Spanning Tree (MST) edge weighting: Boost edges likely to be in optimal tours
    2. Christofides-like approximation: Incorporate matching heuristics for better guidance
    3. Dynamic k-nearest neighbors: Adaptive k based on local node density
    4. Node density-adaptive sparsification: More intelligent edge filtering
    
    Returns a heuristic matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # Initialize heuristic matrix
    heuristics_mat = np.zeros_like(distance_matrix, dtype=float)
    
    # 1. Minimum Spanning Tree edge weights
    mst_weights = compute_mst_weights(distance_matrix)
    heuristics_mat += mst_weights * 2.0  # Strong emphasis on MST edges
    
    # 2. Christofides-like matching heuristic
    matching_bonus = compute_matching_heuristic(distance_matrix, mst_weights)
    heuristics_mat += matching_bonus * 1.5
    
    # 3. Density-adaptive nearest neighbors
    density_k = compute_density_adaptive_k(distance_matrix)
    
    # Inverse distance as base metric
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    np.divide(1.0, distance_matrix, out=inv_dist, where=distance_matrix != 0)
    np.fill_diagonal(inv_dist, 0)
    
    # Apply density-adaptive k-nearest neighbors
    for i in range(n):
        k = density_k[i]
        if k > 0:
            dist_from_i = distance_matrix[i].copy()
            dist_from_i[i] = np.inf
            nearest_indices = np.argpartition(dist_from_i, k)[:k]
            for j in nearest_indices:
                if j != i:
                    # Weight by both inverse distance and node importance
                    heuristics_mat[i, j] += inv_dist[i, j] * (1.0 + 1.0/k)
    
    # 4. Node density-adaptive sparsification
    if n >= 3:
        # Compute local density for each node
        local_densities = np.zeros(n)
        for i in range(n):
            # Median distance to other nodes as density measure
            local_densities[i] = 1.0 / (np.median(distance_matrix[i, distance_matrix[i] > 0]) + 1e-10)
        
        # Normalize densities
        if np.max(local_densities) > 0:
            local_densities = local_densities / np.max(local_densities)
        
        # Apply density-aware penalization: penalize long edges in dense regions less
        for i in range(n):
            for j in range(i+1, n):
                density_factor = (local_densities[i] + local_densities[j]) / 2.0
                if distance_matrix[i, j] > np.percentile(distance_matrix, 75):
                    # Long edges in dense regions get less penalty
                    penalty = 0.5 + 0.3 * density_factor  # Range: 0.5 to 0.8
                    heuristics_mat[i, j] *= penalty
                    heuristics_mat[j, i] *= penalty
                elif distance_matrix[i, j] < np.percentile(distance_matrix, 25):
                    # Short edges get bonus, especially in sparse regions
                    bonus = 1.0 + 0.5 * (1.0 - density_factor)  # Range: 1.0 to 1.5
                    heuristics_mat[i, j] *= bonus
                    heuristics_mat[j, i] *= bonus
    
    # 5. Symmetrize for undirected TSP
    heuristics_mat = (heuristics_mat + heuristics_mat.T) / 2.0
    
    # 6. Final sparsification based on edge significance
    if n > 2:
        # Keep only edges with above-median heuristic value
        non_zero = heuristics_mat[heuristics_mat > 0]
        if len(non_zero) > 0:
            median_val = np.median(non_zero)
            heuristics_mat[heuristics_mat < median_val] = 0
    
    # 7. Normalize to [0, 1] range
    max_val = heuristics_mat.max()
    if max_val > 0:
        heuristics_mat = heuristics_mat / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics_mat, 0)
    
    return heuristics_mat


def compute_mst_weights(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute Minimum Spanning Tree edge weights using Prim's algorithm."""
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    mst_weights = np.zeros_like(distance_matrix, dtype=float)
    
    # Prim's algorithm for MST
    visited = np.zeros(n, dtype=bool)
    min_edge = np.full(n, np.inf)
    parent = np.full(n, -1, dtype=int)
    
    # Start from node 0
    min_edge[0] = 0
    
    for _ in range(n):
        # Find minimum edge from visited to unvisited nodes
        u = -1
        min_val = np.inf
        for i in range(n):
            if not visited[i] and min_edge[i] < min_val:
                min_val = min_edge[i]
                u = i
        
        if u == -1:
            break
            
        visited[u] = True
        
        # Add edge to MST if not root
        if parent[u] != -1:
            mst_weights[u, parent[u]] = 1.0
            mst_weights[parent[u], u] = 1.0
        
        # Update min_edge for adjacent nodes
        for v in range(n):
            if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                min_edge[v] = distance_matrix[u, v]
                parent[v] = u
    
    return mst_weights


def compute_matching_heuristic(distance_matrix: np.ndarray, mst_weights: np.ndarray) -> np.ndarray:
    """Compute Christofides-like matching heuristic for odd-degree nodes."""
    n = distance_matrix.shape[0]
    if n < 4:
        return np.zeros_like(distance_matrix)
    
    matching_bonus = np.zeros_like(distance_matrix, dtype=float)
    
    # Identify nodes with odd degree in MST approximation
    mst_degrees = np.sum(mst_weights > 0, axis=1)
    odd_nodes = np.where(mst_degrees % 2 == 1)[0]
    
    if len(odd_nodes) < 2:
        return matching_bonus
    
    # Simplified greedy matching (Christofides approximation)
    unmatched = set(odd_nodes.tolist())
    
    while len(unmatched) >= 2:
        # Find best pair among unmatched odd nodes
        best_pair = None
        best_dist = np.inf
        
        for u in unmatched:
            for v in unmatched:
                if u != v and distance_matrix[u, v] < best_dist:
                    best_dist = distance_matrix[u, v]
                    best_pair = (u, v)
        
        if best_pair:
            u, v = best_pair
            # Add matching bonus (inverse distance weighted)
            bonus = 1.0 / (distance_matrix[u, v] + 1e-10)
            matching_bonus[u, v] = bonus
            matching_bonus[v, u] = bonus
            
            unmatched.remove(u)
            unmatched.remove(v)
        else:
            break
    
    return matching_bonus


def compute_density_adaptive_k(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute adaptive k for each node based on local density."""
    n = distance_matrix.shape[0]
    if n < 3:
        return np.array([max(1, n-1)] * n, dtype=int)
    
    k_values = np.zeros(n, dtype=int)
    
    for i in range(n):
        # Compute median distance to other nodes
        other_dists = distance_matrix[i, distance_matrix[i] > 0]
        if len(other_dists) > 0:
            median_dist = np.median(other_dists)
            
            # Count nodes within 1.5x median distance
            close_count = np.sum(distance_matrix[i] <= 1.5 * median_dist) - 1  # -1 for self
            
            # Adaptive k: base + density factor
            base_k = max(2, min(5, n // 8))
            density_k = max(0, min(close_count, n // 4))
            
            k_values[i] = base_k + density_k // 2
        else:
            k_values[i] = max(2, n-1)
    
    return k_values
