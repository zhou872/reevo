import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple, List

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristic combining spectral clustering, density weighting,
    and fixed ensemble combination with intelligent sparsification.
    
    Core Features:
    1. Spectral Affinity Smoothing: Uses Fiedler vector for cluster detection
    2. Fixed Weight Ensemble: Combines components with predetermined weights
    3. Adaptive Density-Weighting: Weights scores by local node density
    4. Degree-Constrained Sparsification: Preserves connectivity while reducing edges
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. BASE COMPONENTS ====================
    
    # Inverse distance (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    if np.max(inv_dist) > 0:
        inv_dist = inv_dist / np.max(inv_dist)
    
    # MST for connectivity preservation
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    
    # ==================== 2. SPECTRAL AFFINITY SMOOTHING ====================
    
    def compute_spectral_affinity(k_clusters: int = None) -> np.ndarray:
        """Compute edge affinity using spectral clustering with Fiedler vector"""
        if k_clusters is None:
            k_clusters = max(2, min(5, int(np.sqrt(n))))
        
        # Convert distances to similarities (exponential kernel)
        sigma = np.median(distance_matrix)
        if sigma > 0:
            similarity = np.exp(-distance_matrix / (2 * sigma**2))
        else:
            similarity = 1.0 - np.eye(n)  # Fallback: complete graph minus self
        
        # Graph Laplacian
        degree = np.sum(similarity, axis=1)
        laplacian = np.diag(degree) - similarity
        
        # Eigen decomposition (focused on small eigenvalues)
        try:
            eigenvalues, eigenvectors = np.linalg.eigh(laplacian)
        except np.linalg.LinAlgError:
            # Fallback: use inverse distance as affinity
            return inv_dist
        
        # Use eigenvectors for k_clusters smallest non-zero eigenvalues
        idx = np.argsort(eigenvalues)[1:k_clusters+1]  # Skip zero eigenvalue
        spectral_embedding = eigenvectors[:, idx]
        
        # Compute affinity based on spectral similarity
        affinity = np.ones_like(distance_matrix)
        
        # Fast computation using broadcasting
        for i in range(k_clusters):
            vec = spectral_embedding[:, i:i+1]
            affinity += 0.5 * np.abs(np.dot(vec, vec.T))
        
        # Normalize and apply distance decay
        if np.max(affinity) > 0:
            affinity = affinity / np.max(affinity)
        
        # Blend with distance information
        dist_factor = np.exp(-distance_matrix / (2 * sigma))
        spectral_scores = 0.7 * affinity + 0.3 * dist_factor
        
        np.fill_diagonal(spectral_scores, 0)
        return spectral_scores
    
    spectral_scores = compute_spectral_affinity()
    
    # ==================== 3. ADAPTIVE DENSITY-WEIGHTED SCORING ====================
    
    def compute_density_weighted(k_neighbors: int = 5) -> np.ndarray:
        """Weight inverse distances by local node density"""
        # Compute local density for each node
        densities = np.zeros(n)
        
        # Find k-nearest neighbors for each node
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            # Get distances to k nearest neighbors
            kth_dist = np.partition(dists, k_neighbors)[k_neighbors]
            # Density is inverse of average distance to k nearest
            densities[i] = 1.0 / (kth_dist + 1e-8)
        
        # Normalize densities
        if np.max(densities) > 0:
            densities = densities / np.max(densities)
        
        # Create density-weighted scores
        density_scores = np.zeros_like(distance_matrix)
        
        # Use geometric mean of endpoint densities
        for i in range(n):
            for j in range(i + 1, n):
                geo_mean = np.sqrt(densities[i] * densities[j])
                # Weight inverse distance by geometric mean
                density_scores[i, j] = inv_dist[i, j] * geo_mean
                density_scores[j, i] = density_scores[i, j]
        
        # Normalize
        if np.max(density_scores) > 0:
            density_scores = density_scores / np.max(density_scores)
        
        return density_scores
    
    density_scores = compute_density_weighted()
    
    # ==================== 4. FIXED WEIGHT ENSEMBLE ====================
    
    # Fixed weights from knowledge graph suggestion
    weights = np.array([0.4, 0.3, 0.3])  # [inv_dist, spectral, density]
    weights = weights / np.sum(weights)  # Ensure normalization
    
    # Combine components
    combined_scores = (
        weights[0] * inv_dist +
        weights[1] * spectral_scores +
        weights[2] * density_scores
    )
    
    # ==================== 5. DEGREE-CONSTRAINED SPARSIFICATION ====================
    
    def degree_constrained_sparsify(scores: np.ndarray) -> np.ndarray:
        """Intelligent edge selection preserving connectivity"""
        sparsified = np.zeros_like(scores)
        
        # Always include MST edges for connectivity
        sparsified[mst_edges] = scores[mst_edges]
        
        # Determine adaptive degree limit
        if n <= 10:
            max_degree = 4
        elif n <= 50:
            max_degree = max(3, n // 10)
        else:
            max_degree = max(3, n // 15)
        
        # Track current degree for each node
        current_degree = np.sum(sparsified > 0, axis=1)
        
        # Prepare candidate edges (excluding MST edges)
        candidates = []
        for i in range(n):
            for j in range(i + 1, n):
                if not mst_edges[i, j] and scores[i, j] > 0:
                    # Apply distance penalty to long edges
                    median_dist = np.median(distance_matrix)
                    if median_dist > 0:
                        dist_penalty = np.exp(-distance_matrix[i, j] / (2 * median_dist))
                    else:
                        dist_penalty = 1.0
                    effective_score = scores[i, j] * dist_penalty
                    candidates.append((effective_score, i, j))
        
        # Sort by effective score (descending)
        candidates.sort(reverse=True, key=lambda x: x[0])
        
        # Greedy selection respecting degree constraints
        for score, i, j in candidates:
            if current_degree[i] < max_degree and current_degree[j] < max_degree:
                sparsified[i, j] = scores[i, j]
                sparsified[j, i] = scores[j, i]
                current_degree[i] += 1
                current_degree[j] += 1
        
        return sparsified
    
    sparsified = degree_constrained_sparsify(combined_scores)
    
    # ==================== 6. FINAL PROCESSING ====================
    
    # Ensure symmetry
    symmetric_scores = (sparsified + sparsified.T) / 2
    
    # Final distance-based attenuation
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        dist_decay = np.exp(-distance_matrix / (0.5 * max_dist + 1e-8))
        symmetric_scores *= dist_decay
    
    # Normalize to [0, 1] range
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    # Zero out diagonal
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
