import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with MST-constrained sparsification.
    
    Core improvements over v1:
    1. Simplified exponential penalty using fixed scaling (0.5*max_distance)
    2. MST-constrained sparsification: retain MST edges + promising short edges
    3. Two-level connectivity bonus: MST edges and edges under distance threshold
    4. Cleaner implementation with fewer adaptive parameters
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance scoring
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Simple local density estimation using fixed k-nearest neighbors
    k = min(5, n - 1)
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if k > 0:
            k_nearest = np.partition(dists, k)[:k]
            local_density[i] = 1.0 / (np.mean(k_nearest) + 1e-8)
    
    # 3. Simple regret-based scoring (bonus for clear best choices)
    regret_bonus = np.ones_like(distance_matrix)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if len(dists) >= 2:
            # Find two smallest distances
            indices = np.argpartition(dists, 2)[:2]
            d1, d2 = dists[indices[0]], dists[indices[1]]
            
            if d1 > 0:
                regret = (d2 - d1) / (d1 + 1e-8)
                # Apply moderate bonus (0-20%) based on regret
                best_idx = indices[0]
                regret_bonus[i, best_idx] += 0.2 * np.tanh(regret)
    
    # 4. Density-weighted base scores
    density_weights = np.sqrt(np.outer(local_density, local_density))
    base_scores = inv_dist * density_weights * regret_bonus
    
    # 5. Apply exponential edge length penalty (suggestion #3)
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        length_penalty = np.exp(-distance_matrix / (0.5 * max_dist))
        base_scores = base_scores * length_penalty
    
    # 6. MST-based sparsification (suggestions #1-2)
    sparsified = np.zeros_like(base_scores)
    
    # Compute MST
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = mst > 0
    
    # Calculate average MST edge distance
    if np.sum(mst_mask) > 0:
        avg_mst_dist = np.mean(distance_matrix[mst_mask])
    else:
        avg_mst_dist = np.median(distance_matrix[distance_matrix > 0])
    
    # Distance threshold for promising edges (3x average MST edge as in suggestion #2)
    dist_threshold = 3.0 * avg_mst_dist
    
    # Connectivity bonus: MST edges get higher bonus
    connectivity_bonus = np.ones_like(distance_matrix)
    connectivity_bonus[mst_mask] = 1.5  # MST edges get 50% bonus
    
    # Also apply moderate bonus to edges under distance threshold
    short_edge_mask = distance_matrix < dist_threshold
    connectivity_bonus[short_edge_mask] = np.maximum(
        connectivity_bonus[short_edge_mask], 1.2
    )
    
    base_scores = base_scores * connectivity_bonus
    
    # Create sparsification mask
    # Keep: 1) MST edges, 2) Short promising edges
    sparsity_mask = mst_mask | (short_edge_mask & (base_scores > 0))
    
    # Also keep top edges per node (ensuring each node has some connections)
    for i in range(n):
        # Current connections from MST and short edges
        current_connections = np.sum(sparsity_mask[i])
        
        # Ensure minimum connectivity
        min_connections = min(3, n - 1)
        if current_connections < min_connections:
            # Add top scoring edges to reach minimum
            row_scores = base_scores[i].copy()
            row_scores[i] = -np.inf
            row_scores[sparsity_mask[i]] = -np.inf  # Already included
            
            if np.any(row_scores > 0):
                k_needed = min_connections - current_connections
                if k_needed > 0:
                    # Get indices of top k_needed scores
                    top_indices = np.argpartition(row_scores, -k_needed)[-k_needed:]
                    sparsity_mask[i, top_indices] = True
    
    # Apply sparsification mask
    sparsified[sparsity_mask] = base_scores[sparsity_mask]
    
    # 7. Symmetrize (keep edge if either endpoint considers it promising)
    symmetric_scores = np.maximum(sparsified, sparsified.T)
    
    # 8. Final normalization
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
