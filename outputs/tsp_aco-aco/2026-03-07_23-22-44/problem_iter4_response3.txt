import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.cluster import DBSCAN
from sklearn.neighbors import LocalOutlierFactor


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristic with adaptive clustering and dynamic scoring.
    
    Features:
    1. Adaptive density clustering with outlier-aware scaling
    2. Multi-level regret weighting with dynamic thresholds
    3. Hierarchical edge scoring based on structural importance
    4. Context-aware sparsification preserving topological diversity
    
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # ==================== 1. Adaptive Density Clustering ====================
    # Dynamic bandwidth selection based on local point distribution
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # Estimate local density using adaptive k-NN
    k_density = min(7, max(3, n // 15))
    local_density = np.zeros(n)
    local_variability = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_nearest = np.partition(dists, k_density)[:k_density]
        local_density[i] = 1.0 / (np.median(k_nearest) + 1e-8)
        local_variability[i] = np.std(k_nearest) / (np.mean(k_nearest) + 1e-8)
    
    # Adaptive density weighting with variability discount
    density_weights = np.sqrt(np.outer(local_density, local_density))
    variability_mask = np.exp(-0.5 * np.outer(local_variability, local_variability))
    density_weights *= variability_mask
    
    base_scores = inv_dist * density_weights
    
    # ==================== 2. MST Structural Importance ====================
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_active = (mst > 0) | (mst.T > 0)
    mst_degree = np.sum(mst_active, axis=1)
    
    # Adaptive MST bonus based on node importance
    degree_normalized = mst_degree / np.max(mst_degree) if np.max(mst_degree) > 0 else np.ones(n)
    mst_bonus = np.where(mst_active, 1.8 + 0.4 * degree_normalized[:, None], 1.0)
    enhanced_scores = base_scores * mst_bonus
    
    # ==================== 3. Dynamic Regret Weighting ====================
    regret_scores = np.zeros_like(enhanced_scores)
    
    # Dynamic regret thresholds based on distance distribution
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Calculate dynamic thresholds
        sorted_dists = np.sort(dists)
        if len(sorted_dists) >= 3:
            q1_idx = max(0, len(sorted_dists) // 4 - 1)
            q3_idx = min(len(sorted_dists) - 1, 3 * len(sorted_dists) // 4 - 1)
            dynamic_scale = max(sorted_dists[q3_idx] - sorted_dists[q1_idx], 1e-8)
        else:
            dynamic_scale = np.max(dists) - np.min(dists[dists < np.inf]) if np.any(dists < np.inf) else 1.0
        
        # Two-level regret calculation
        for j in range(n):
            if i != j:
                # Level 1: Regret compared to nearest alternative
                alt_mask = np.ones(n, dtype=bool)
                alt_mask[j] = False
                alt_mask[i] = False
                
                if np.any(alt_mask):
                    alt_min = np.min(distance_matrix[i, alt_mask])
                    regret_1 = max(distance_matrix[i, j] - alt_min, 0)
                    
                    # Level 2: Regret compared to cluster-appropriate alternative
                    # Estimate appropriate alternative based on local density
                    density_diff = np.abs(local_density[i] - local_density[j]) / (local_density[i] + local_density[j] + 1e-8)
                    density_factor = np.exp(-2 * density_diff)
                    
                    # Find alternative with similar density characteristics
                    density_similarities = 1.0 / (1.0 + np.abs(local_density[alt_mask] - local_density[j]))
                    weighted_alts = distance_matrix[i, alt_mask] * density_similarities
                    weighted_min = np.min(weighted_alts) if len(weighted_alts) > 0 else alt_min
                    
                    regret_2 = max(distance_matrix[i, j] - weighted_min, 0)
                    
                    # Combine regrets with dynamic weighting
                    regret = 0.6 * regret_1 + 0.4 * regret_2
                    regret_score = np.exp(-regret / (dynamic_scale + 1e-8))
                    
                    # Apply density-aware adjustment
                    density_match = 1.0 - 0.3 * density_diff
                    regret_scores[i, j] = regret_score * density_match
    
    # Symmetrize and combine regret scores
    regret_symmetric = (regret_scores + regret_scores.T) / 2
    regret_adjusted_scores = enhanced_scores * (1.0 + 0.8 * regret_symmetric)
    
    # ==================== 4. Cluster-Aware Enhancement ====================
    # Detect natural clusters in the point distribution
    if n >= 10:
        try:
            # Use DBSCAN with adaptive parameters
            eps_estimate = np.median(distance_matrix[distance_matrix > 0])
            clustering = DBSCAN(eps=eps_estimate * 1.2, min_samples=max(2, n // 20), metric='precomputed')
            labels = clustering.fit_predict(distance_matrix)
            
            if len(np.unique(labels[labels != -1])) >= 2:
                # Give bonus to intra-cluster edges, penalty to inter-cluster
                cluster_bonus = np.ones((n, n))
                for i in range(n):
                    for j in range(i + 1, n):
                        if labels[i] == labels[j] and labels[i] != -1:
                            # Strong bonus for same cluster (scaled by cluster density)
                            cluster_size = np.sum(labels == labels[i])
                            cluster_density = local_density[labels == labels[i]].mean()
                            bonus_factor = 1.5 + 0.5 * np.tanh(cluster_density - 1.0)
                            cluster_bonus[i, j] = cluster_bonus[j, i] = bonus_factor
                        elif labels[i] != labels[j] and labels[i] != -1 and labels[j] != -1:
                            # Moderate penalty for crossing clusters
                            cluster_bonus[i, j] = cluster_bonus[j, i] = 0.7
                
                regret_adjusted_scores *= cluster_bonus
        except:
            pass
    
    # ==================== 5. Context-Aware Sparsification ====================
    # Hierarchical edge retention based on multiple criteria
    sparsified = np.zeros_like(regret_adjusted_scores)
    
    for i in range(n):
        row_scores = regret_adjusted_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Determine retention threshold based on:
        # 1. MST connectivity (high degree = more connections)
        # 2. Local density (dense regions need more connections)
        # 3. Cluster context (if applicable)
        
        base_k = max(2, int(np.sqrt(n)))
        
        # Dynamic retention factor
        retention_factor = 1.0
        retention_factor += 0.5 * (mst_degree[i] / max(2, np.median(mst_degree))) if np.median(mst_degree) > 0 else 0
        retention_factor += 0.3 * (local_density[i] / np.median(local_density)) if np.median(local_density) > 0 else 0
        
        adaptive_k = min(n - 1, int(base_k * retention_factor))
        
        # Keep top edges, but ensure MST edges are always included
        if adaptive_k < len(row_scores):
            threshold = np.partition(row_scores, -adaptive_k)[-adaptive_k]
            mask = row_scores >= threshold
        else:
            mask = np.ones_like(row_scores, dtype=bool)
            mask[i] = False
        
        # Force inclusion of MST edges
        mask |= mst_active[i]
        
        sparsified[i, mask] = regret_adjusted_scores[i, mask]
    
    # ==================== 6. Geometric Smoothing ====================
    # Apply distance-based smoothing with adaptive length scaling
    median_dist = np.median(distance_matrix[distance_matrix > 0])
    if median_dist > 0:
        length_factor = np.exp(-distance_matrix / (2.0 * median_dist))
    else:
        length_factor = np.ones_like(distance_matrix)
    
    final_scores = sparsified * length_factor
    
    # ==================== 7. Symmetrization and Normalization ====================
    symmetric = (final_scores + final_scores.T) / 2
    np.fill_diagonal(symmetric, 0)
    
    # Adaptive normalization preserving relative differences
    max_val = np.max(symmetric)
    if max_val > 0:
        # Soft normalization preserving rank order
        symmetric = np.tanh(symmetric / (max_val * 0.8))
    
    return symmetric
