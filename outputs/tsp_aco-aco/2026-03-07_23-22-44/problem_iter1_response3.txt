import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection that combines multiple factors
    and sparsifies the matrix by zeroing out unpromising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Basic inverse distance (like v1) with regularization
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + 1e-10)  # Add small epsilon to avoid division by zero
    np.fill_diagonal(inv_dist, 0)  # Zero out self-connections
    
    # 2. Minimum Spanning Tree (MST) based weighting
    # Edges that appear in MST are more promising
    from scipy.sparse.csgraph import minimum_spanning_tree
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_weight = np.where(mst > 0, 1.0, 0.3)  # Boost MST edges by 3.3x
    
    # 3. Local connectivity bonus - edges connecting to nodes with few good alternatives
    # For each node, find distance to k-th nearest neighbor
    k = min(5, n - 1)
    sorted_indices = np.argsort(distance_matrix, axis=1)
    kth_distances = distance_matrix[np.arange(n), sorted_indices[:, k]]
    
    # Normalize and create connectivity factor
    kth_factor = kth_distances[:, np.newaxis] / (distance_matrix + 1e-10)
    kth_factor = np.minimum(kth_factor, 3.0)  # Cap the factor
    
    # 4. Nearest neighbor sparsification mask
    # Keep only edges to the m nearest neighbors for each node
    m = max(3, n // 4)  # Keep more neighbors for smaller instances
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nn_indices = sorted_indices[i, 1:m+1]  # Exclude self
        nn_mask[i, nn_indices] = True
    
    # Make mask symmetric (if i is in j's nearest neighbors or vice versa)
    nn_mask = nn_mask | nn_mask.T
    
    # 5. Combine all factors
    scores = inv_dist * mst_weight * kth_factor
    
    # 6. Apply sparsification and regularization
    scores = np.where(nn_mask, scores, 0)
    
    # 7. Normalize scores to [0, 1] range for better numerical stability
    max_score = np.max(scores)
    if max_score > 0:
        scores = scores / max_score
    
    # 8. Final zeroing of very weak edges (below threshold)
    threshold = np.percentile(scores[scores > 0], 20) if np.any(scores > 0) else 0
    scores[scores < threshold] = 0
    
    return scores
