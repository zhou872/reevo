import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple, List
import itertools

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic matrix for TSP edge selection incorporating:
    1. Multi-step regret analysis: Evaluates edge criticality considering longer-term alternatives
    2. Monte Carlo edge evaluation: Samples partial tours to estimate edge contribution
    3. Local cluster affinity: Identifies natural groupings using fast geometric clustering
    4. Connectivity-aware sparsification: Preserves edges essential for tour construction
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape indicating edge promise (0-1 range)
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance with numerical stability
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Multi-step regret analysis
    regret_scores = _compute_multi_step_regret(distance_matrix)
    
    # 3. Monte Carlo edge evaluation
    mc_scores = _monte_carlo_edge_evaluation(distance_matrix)
    
    # 4. Local cluster affinity detection
    cluster_scores = _compute_local_cluster_affinity(distance_matrix)
    
    # 5. Minimum Spanning Tree connectivity reinforcement
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    
    # 6. Intelligent combination with adaptive weighting
    # Weights adapt based on problem size and score distributions
    if n <= 10:
        # Small problems: emphasize regret and exact properties
        weights = [0.3, 0.4, 0.2, 0.1]  # regret, mc, cluster, inverse
    elif n <= 50:
        # Medium problems: balanced approach
        weights = [0.25, 0.35, 0.25, 0.15]
    else:
        # Large problems: emphasize sampling and clustering
        weights = [0.2, 0.4, 0.3, 0.1]
    
    # Normalize components to comparable scales
    components = [regret_scores, mc_scores, cluster_scores, inv_dist]
    normalized_components = []
    
    for comp in components:
        comp_max = np.max(comp)
        if comp_max > 0:
            normalized_components.append(comp / comp_max)
        else:
            normalized_components.append(comp)
    
    # Weighted combination
    combined_scores = np.zeros_like(distance_matrix)
    for comp, weight in zip(normalized_components, weights):
        combined_scores += weight * comp
    
    # Boost MST edges (critical for connectivity)
    mst_boost = np.where(mst_edges, 1.5, 1.0)
    enhanced_scores = combined_scores * mst_boost
    
    # 7. Symmetrize for undirected TSP
    symmetric_scores = (enhanced_scores + enhanced_scores.T) / 2
    np.fill_diagonal(symmetric_scores, 0)
    
    # 8. Connectivity-aware sparsification
    sparsified = _connectivity_aware_sparsification(symmetric_scores, distance_matrix, mst_edges)
    
    # 9. Final distance-based attenuation (penalize very long edges)
    median_dist = np.median(distance_matrix[np.triu_indices(n, 1)])
    distance_attenuation = np.exp(-distance_matrix / (2.0 * median_dist + 1e-8))
    final_scores = sparsified * distance_attenuation
    
    # 10. Normalize to [0, 1] range
    max_score = np.max(final_scores)
    if max_score > 0:
        final_scores /= max_score
    
    return final_scores


def _compute_multi_step_regret(dist_matrix: np.ndarray) -> np.ndarray:
    """Compute regret scores considering 2-step alternatives."""
    n = dist_matrix.shape[0]
    regret_scores = np.zeros_like(dist_matrix)
    
    for i in range(n):
        # Get distances from node i
        dists_i = dist_matrix[i].copy()
        dists_i[i] = np.inf
        
        # For each potential edge (i,j)
        for j in range(n):
            if i == j:
                continue
            
            # Consider 2-step alternatives (i,k,j) for k != i,j
            alternative_costs = []
            for k in range(n):
                if k == i or k == j:
                    continue
                # Cost of path i-k-j (2 steps)
                alt_cost = dist_matrix[i, k] + dist_matrix[k, j]
                alternative_costs.append(alt_cost)
            
            if alternative_costs:
                # Find best 2-step alternative
                best_2step = min(alternative_costs)
                # Regret is how much worse the direct edge is compared to best 2-step
                regret = max(0, best_2step - dist_matrix[i, j])
                # Convert regret to score: higher regret means more critical edge
                # Use sigmoid to bound the score
                regret_scores[i, j] = 1.0 / (1.0 + np.exp(-regret / (np.mean(alternative_costs) + 1e-8)))
    
    return regret_scores


def _monte_carlo_edge_evaluation(dist_matrix: np.ndarray, n_samples: int = 100) -> np.ndarray:
    """Evaluate edges by sampling partial tours that include them."""
    n = dist_matrix.shape[0]
    if n < 4:
        return np.ones_like(dist_matrix)
    
    edge_scores = np.zeros_like(dist_matrix)
    np.fill_diagonal(edge_scores, 0)
    
    # Number of samples per edge adapts to problem size
    samples_per_edge = min(n_samples, max(10, n * 2))
    
    for i in range(n):
        for j in range(i + 1, n):
            if dist_matrix[i, j] == 0:
                continue
            
            tour_qualities = []
            
            for _ in range(samples_per_edge):
                # Start a partial tour with edge (i,j)
                remaining = list(range(n))
                remaining.remove(i)
                remaining.remove(j)
                np.random.shuffle(remaining)
                
                # Build tour: i -> j -> random permutation -> back to i
                tour = [i, j] + remaining + [i]
                
                # Calculate tour length
                tour_length = 0
                for k in range(len(tour) - 1):
                    tour_length += dist_matrix[tour[k], tour[k + 1]]
                
                tour_qualities.append(1.0 / (tour_length + 1e-8))
            
            # Score is average quality of tours containing this edge
            if tour_qualities:
                edge_scores[i, j] = np.mean(tour_qualities)
                edge_scores[j, i] = edge_scores[i, j]
    
    return edge_scores


def _compute_local_cluster_affinity(dist_matrix: np.ndarray) -> np.ndarray:
    """Identify local clusters using adaptive neighborhood analysis."""
    n = dist_matrix.shape[0]
    cluster_scores = np.ones_like(dist_matrix)
    
    if n < 4:
        return cluster_scores
    
    # Adaptive neighborhood size
    k_neighbors = max(2, min(5, n // 4))
    
    for i in range(n):
        # Find k nearest neighbors
        distances = dist_matrix[i].copy()
        distances[i] = np.inf
        neighbor_indices = np.argsort(distances)[:k_neighbors]
        
        # Calculate average distance to neighbors
        if len(neighbor_indices) > 0:
            avg_neighbor_dist = np.mean(dist_matrix[i, neighbor_indices])
            
            # Assign bonus to edges within neighborhood
            for j in neighbor_indices:
                if i != j:
                    # Bonus decays with distance relative to average neighborhood distance
                    relative_dist = dist_matrix[i, j] / (avg_neighbor_dist + 1e-8)
                    bonus = np.exp(-relative_dist * 0.5)
                    cluster_scores[i, j] = 1.0 + bonus
                    cluster_scores[j, i] = cluster_scores[i, j]
    
    return cluster_scores


def _connectivity_aware_sparsification(scores: np.ndarray, 
                                       dist_matrix: np.ndarray,
                                       mst_edges: np.ndarray) -> np.ndarray:
    """Intelligent sparsification that preserves connectivity."""
    n = scores.shape[0]
    result = np.zeros_like(scores)
    
    # Phase 1: Always include MST edges (ensures connectivity)
    result[mst_edges] = scores[mst_edges]
    
    # Phase 2: Greedy addition of promising edges
    # Determine adaptive degree limits based on problem size
    if n <= 10:
        max_degree = 4
    elif n <= 50:
        max_degree = 5
    else:
        max_degree = 6
    
    current_degree = np.sum(result > 0, axis=1)
    
    # Create candidate list (excluding MST edges and self)
    candidates = []
    for i in range(n):
        for j in range(i + 1, n):
            if not mst_edges[i, j] and scores[i, j] > 0:
                # Penalize very long edges
                max_dist = np.max(dist_matrix)
                length_penalty = np.exp(-dist_matrix[i, j] / (0.5 * max_dist + 1e-8))
                effective_score = scores[i, j] * length_penalty
                candidates.append((effective_score, i, j))
    
    # Sort candidates by effective score
    candidates.sort(reverse=True, key=lambda x: x[0])
    
    # Greedy selection with degree constraints
    for score, i, j in candidates:
        if current_degree[i] < max_degree and current_degree[j] < max_degree:
            result[i, j] = scores[i, j]
            result[j, i] = scores[j, i]
            current_degree[i] += 1
            current_degree[j] += 1
    
    return result
