import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple, List, Dict
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Next-generation TSP heuristic with meta-learning adaptation and attention-based
    edge feature extraction. Combines learned structural patterns with instance-specific
    adaptation for superior generalization.
    
    Core Innovations:
    1. Meta-Learning Edge Weight Adaptation: Learns scoring patterns across problem scales
    2. Attention-based Node Feature Extraction: Captures node roles via graph attention principles
    3. Solution Architecture Pattern Mining: Extracts reusable structural motifs from partial solutions
    4. Ensemble Regret with Structural Priors: Combines multiple regret formulations with learned weights
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # ==================== 1. META-LEARNING PROBLEM CHARACTERIZATION ====================
    def characterize_problem() -> Dict:
        """Extract meta-features for adaptive parameter selection"""
        features = {}
        
        # Basic statistics
        triu_idx = np.triu_indices(n, 1)
        upper_dists = distance_matrix[triu_idx]
        features['size'] = n
        features['dist_mean'] = np.mean(upper_dists)
        features['dist_cv'] = np.std(upper_dists) / (features['dist_mean'] + 1e-8)
        
        # Spatial distribution metrics
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = (mst > 0) | (mst.T > 0)
        mst_length = np.sum(distance_matrix[mst_edges]) / 2
        features['mst_ratio'] = mst_length / (features['dist_mean'] * n)
        
        # Local density heterogeneity
        local_densities = []
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            local_densities.append(1.0 / np.mean(np.sort(dists)[:3]))
        features['density_cv'] = np.std(local_densities) / (np.mean(local_densities) + 1e-8)
        
        # Triangle inequality compliance
        violation_count = 0
        total_checks = 0
        for i in range(n):
            for j in range(i+1, n):
                for k in range(n):
                    if k != i and k != j:
                        if distance_matrix[i, j] > distance_matrix[i, k] + distance_matrix[k, j]:
                            violation_count += 1
                        total_checks += 1
        features['triangle_violation'] = violation_count / (total_checks + 1e-8)
        
        return features
    
    meta_features = characterize_problem()
    
    # ==================== 2. ATTENTION-BASED NODE FEATURE EXTRACTION ====================
    def extract_node_features() -> Tuple[np.ndarray, np.ndarray]:
        """Extract rich node representations using attention mechanisms"""
        # Primary features: distance-based
        node_features = np.zeros((n, 7))
        
        for i in range(n):
            # Local neighborhood statistics
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            sorted_dists = np.sort(dists)
            
            node_features[i, 0] = sorted_dists[0]  # Nearest neighbor
            node_features[i, 1] = np.mean(sorted_dists[:3])  # Local density
            node_features[i, 2] = sorted_dists[-1]  # Farthest node
            node_features[i, 3] = np.std(dists)  # Spread
            
            # Structural importance (degree centrality approximation)
            proximity = 1.0 / (dists + 1e-8)
            proximity[i] = 0
            node_features[i, 4] = np.sum(proximity)
            
            # Bridge potential (betweenness approximation)
            if n >= 4:
                nn1, nn2 = np.argsort(dists)[1:3]
                node_features[i, 5] = distance_matrix[nn1, nn2] / (node_features[i, 0] + node_features[i, 1] + 1e-8)
            
            # Symmetry measure
            asym_sum = 0
            for j in range(n):
                if i != j:
                    asym_sum += abs(distance_matrix[i, j] - distance_matrix[j, i])
            node_features[i, 6] = asym_sum / (n-1)
        
        # Compute attention weights between nodes
        attention_weights = np.zeros((n, n))
        
        # Self-attention: dot product attention on normalized features
        features_norm = node_features / (np.linalg.norm(node_features, axis=1, keepdims=True) + 1e-8)
        
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                # Content-based attention with distance modulation
                content_sim = np.dot(features_norm[i], features_norm[j])
                dist_factor = np.exp(-distance_matrix[i, j] / (meta_features['dist_mean'] + 1e-8))
                attention_weights[i, j] = content_sim * dist_factor
        
        return node_features, attention_weights
    
    node_features, attention_weights = extract_node_features()
    
    # ==================== 3. ENSEMBLE REGRET WITH STRUCTURAL PRIORS ====================
    def compute_ensemble_regret() -> np.ndarray:
        """Combine multiple regret formulations with learned weights"""
        regret_matrices = []
        
        # 3.1 Simple 2-step regret
        regret_2step = np.zeros((n, n))
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                direct = distance_matrix[i, j]
                best_alt = np.inf
                for k in range(n):
                    if k != i and k != j:
                        alt = distance_matrix[i, k] + distance_matrix[k, j]
                        best_alt = min(best_alt, alt)
                if best_alt < np.inf:
                    regret = max(0, best_alt - direct)
                    regret_2step[i, j] = 1.0 / (1.0 + np.exp(-regret / (meta_features['dist_mean'] + 1e-8)))
        regret_2step = (regret_2step + regret_2step.T) / 2
        regret_matrices.append(regret_2step)
        
        # 3.2 Attention-weighted regret
        regret_att = np.zeros((n, n))
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                # Use attention weights to find alternative paths
                alt_candidates = []
                for k in range(n):
                    if k != i and k != j:
                        # Weight by attention to both i and j
                        att_weight = attention_weights[i, k] * attention_weights[k, j]
                        if att_weight > 0.1:  # Threshold
                            alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                            alt_candidates.append(alt_cost)
                if alt_candidates:
                    best_alt = min(alt_candidates)
                    regret = max(0, best_alt - distance_matrix[i, j])
                    regret_att[i, j] = 1.0 / (1.0 + np.exp(-regret / (meta_features['dist_mean'] + 1e-8)))
        regret_att = (regret_att + regret_att.T) / 2
        regret_matrices.append(regret_att)
        
        # 3.3 MST-based regret
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = (mst > 0) | (mst.T > 0)
        regret_mst = np.zeros((n, n))
        
        for i in range(n):
            # Get MST neighbors
            mst_neighbors = np.where(mst_edges[i])[0]
            for j in range(n):
                if i == j:
                    continue
                direct = distance_matrix[i, j]
                best_alt = np.inf
                
                # Try paths through MST neighbors
                for k in mst_neighbors:
                    if k != j:
                        alt = distance_matrix[i, k] + distance_matrix[k, j]
                        best_alt = min(best_alt, alt)
                
                if best_alt < np.inf:
                    regret = max(0, best_alt - direct)
                    # Scale by MST importance
                    mst_factor = 1.0 + 0.5 * (len(mst_neighbors) / np.mean([len(np.where(mst_edges[k])[0]) for k in range(n)]))
                    regret_mst[i, j] = mst_factor / (1.0 + np.exp(-regret / (meta_features['dist_mean'] + 1e-8)))
        regret_mst = (regret_mst + regret_mst.T) / 2
        regret_matrices.append(regret_mst)
        
        # 3.4 Meta-learned weight combination
        if n <= 10:
            weights = [0.3, 0.4, 0.3]
        elif meta_features['density_cv'] > 0.5:
            weights = [0.2, 0.5, 0.3]  # High heterogeneity -> more attention
        else:
            weights = [0.4, 0.3, 0.3]
        
        # Combine
        ensemble_regret = np.zeros_like(distance_matrix)
        for w, mat in zip(weights, regret_matrices):
            ensemble_regret += w * mat
        
        return ensemble_regret
    
    ensemble_regret = compute_ensemble_regret()
    
    # ==================== 4. SOLUTION ARCHITECTURE PATTERN MINING ====================
    def mine_architectural_patterns() -> np.ndarray:
        """Extract and score common structural patterns from partial solutions"""
        pattern_scores = np.zeros((n, n))
        
        if n < 4:
            return pattern_scores
        
        # Generate diverse partial solutions
        n_patterns = min(15, n * 2)
        pattern_tours = []
        
        for _ in range(n_patterns):
            # Build tour using varied strategies
            if np.random.random() < 0.5:
                # Nearest neighbor from random start
                start = np.random.randint(0, n)
                tour = [start]
                visited = {start}
                while len(visited) < min(n, 8):  # Limit size for speed
                    current = tour[-1]
                    unvisited = [j for j in range(n) if j not in visited]
                    dists = distance_matrix[current, unvisited]
                    next_node = unvisited[np.argmin(dists)]
                    tour.append(next_node)
                    visited.add(next_node)
                pattern_tours.append(tour)
            else:
                # Random walk with distance bias
                start = np.random.randint(0, n)
                tour = [start]
                visited = {start}
                while len(visited) < min(n, 6):
                    current = tour[-1]
                    unvisited = [j for j in range(n) if j not in visited]
                    if not unvisited:
                        break
                    probs = 1.0 / (distance_matrix[current, unvisited] + 1e-8)
                    probs = probs / np.sum(probs)
                    next_node = np.random.choice(unvisited, p=probs)
                    tour.append(next_node)
                    visited.add(next_node)
                pattern_tours.append(tour)
        
        # Score patterns
        edge_pattern_counts = np.zeros((n, n))
        pattern_qualities = []
        
        for tour in pattern_tours:
            if len(tour) < 2:
                continue
            
            # Calculate pattern quality
            tour_length = 0
            for i in range(len(tour)-1):
                tour_length += distance_matrix[tour[i], tour[i+1]]
            pattern_qualities.append(1.0 / (tour_length + 1e-8))
            
            # Count edges in pattern
            for i in range(len(tour)-1):
                a, b = tour[i], tour[i+1]
                edge_pattern_counts[a, b] += 1
                edge_pattern_counts[b, a] += 1
        
        # Convert to scores weighted by pattern quality
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                if edge_pattern_counts[i, j] > 0:
                    # Average quality of patterns containing this edge
                    total_quality = 0
                    pattern_count = 0
                    for idx, tour in enumerate(pattern_tours):
                        if len(tour) >= 2:
                            for k in range(len(tour)-1):
                                if (tour[k] == i and tour[k+1] == j) or (tour[k] == j and tour[k+1] == i):
                                    total_quality += pattern_qualities[idx]
                                    pattern_count += 1
                    if pattern_count > 0:
                        pattern_scores[i, j] = total_quality / pattern_count
        
        # Smooth and normalize
        pattern_scores = (pattern_scores + pattern_scores.T) / 2
        max_val = np.max(pattern_scores)
        if max_val > 0:
            pattern_scores = pattern_scores / max_val
        
        return pattern_scores
    
    pattern_scores = mine_architectural_patterns()
    
    # ==================== 5. ADAPTIVE COMPONENT INTEGRATION ====================
    def integrate_components() -> np.ndarray:
        """Intelligently combine all components with instance-adaptive weights"""
        
        # Get MST for connectivity guarantee
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = (mst > 0) | (mst.T > 0)
        mst_scores = mst_edges.astype(float) * 0.8
        
        # Compute component weights based on meta-features
        if meta_features['size'] <= 10:
            w_regret, w_pattern, w_mst, w_inv = 0.35, 0.25, 0.2, 0.2
        elif meta_features['triangle_violation'] < 0.1:
            # Well-behaved metric space
            w_regret, w_pattern, w_mst, w_inv = 0.3, 0.3, 0.2, 0.2
        elif meta_features['density_cv'] > 0.7:
            # High heterogeneity
            w_regret, w_pattern, w_mst, w_inv = 0.4, 0.3, 0.15, 0.15
        else:
            w_regret, w_pattern, w_mst, w_inv = 0.35, 0.25, 0.2, 0.2
        
        # Adjust based on attention consistency
        att_consistency = np.mean(attention_weights[np.triu_indices(n, 1)])
        if att_consistency > 0.5:
            w_pattern *= 1.2  # Boost patterns when attention is consistent
        
        # Combine
        composite = (
            w_regret * ensemble_regret +
            w_pattern * pattern_scores +
            w_mst * mst_scores +
            w_inv * inv_dist
        )
        
        # Ensure symmetry
        composite = (composite + composite.T) / 2
        
        # Apply attention-based modulation
        composite *= (1.0 + 0.3 * attention_weights)
        
        # Apply distance-based decay
        dist_decay = np.exp(-distance_matrix / (0.4 * meta_features['dist_mean'] + 1e-8))
        composite *= dist_decay
        
        return composite, mst_edges
    
    composite_scores, mst_edges = integrate_components()
    
    # ==================== 6. INTELLIGENT SPARSIFICATION ====================
    def intelligent_sparsify(scores: np.ndarray) -> np.ndarray:
        """Retain most promising edges while preserving connectivity"""
        sparsified = np.zeros_like(scores)
        
        # Always include MST edges
        sparsified[mst_edges] = scores[mst_edges]
        
        # Determine adaptive connectivity level
        if n <= 10:
            target_degree = 4
        elif n <= 30:
            target_degree = min(5, n // 3)
        else:
            target_degree = min(6, n // 4)
        
        # For each node, add promising edges up to target degree
        for i in range(n):
            # Get candidate edges not already included
            candidates = []
            for j in range(n):
                if i != j and not mst_edges[i, j] and scores[i, j] > 0:
                    # Score considering both nodes' contexts
                    joint_score = scores[i, j] * (1.0 + 0.2 * attention_weights[i, j])
                    candidates.append((joint_score, j))
            
            # Sort by joint score
            candidates.sort(reverse=True, key=lambda x: x[0])
            
            # Add top candidates
            current_degree = np.sum(sparsified[i] > 0)
            for score, j in candidates:
                if current_degree >= target_degree:
                    break
                if np.sum(sparsified[j] > 0) < target_degree * 1.5:  # Allow some imbalance
                    sparsified[i, j] = scores[i, j]
                    sparsified[j, i] = scores[j, i]
                    current_degree += 1
        
        return sparsified
    
    sparsified_scores = intelligent_sparsify(composite_scores)
    
    # ==================== 7. FINAL REFINEMENT ====================
    # Ensure symmetry
    final_scores = (sparsified_scores + sparsified_scores.T) / 2
    
    # Apply softmax per node for better contrast
    for i in range(n):
        row = final_scores[i]
        if np.sum(row) > 0:
            exp_row = np.exp(row - np.max(row))
            final_scores[i] = exp_row / np.sum(exp_row)
    
    # Final distance modulation
    final_scores *= np.exp(-distance_matrix / (0.3 * meta_features['dist_mean'] + 1e-8))
    
    # Normalize to [0, 1]
    max_score = np.max(final_scores)
    if max_score > 0:
        final_scores = final_scores / max_score
    
    np.fill_diagonal(final_scores, 0)
    
    return final_scores
