import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with MST-preserving sparsification.
    
    Key improvements:
    1. MST connectivity guaranteed through explicit preservation
    2. Stable exponential penalty normalized by global median distance
    3. Simplified regret calculation using distance ratios
    4. Symmetric edge selection with node-adaptive sparsity
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance with safe division
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Exponential penalty using global median for stable normalization
    non_zero_dists = distance_matrix[np.triu_indices(n, k=1)]
    if len(non_zero_dists) > 0:
        median_dist = np.median(non_zero_dists)
    else:
        median_dist = 1.0  # Fallback for edge cases
    
    # Strong exponential decay: edges at median distance get ~13% weight
    exp_penalty = np.exp(-distance_matrix / (0.5 * median_dist))
    np.fill_diagonal(exp_penalty, 0)
    
    # 3. Simplified regret: ratio of edge distance to node's minimum distance
    regret_factor = np.ones_like(distance_matrix)
    for i in range(n):
        # Get distances excluding self
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        min_dist = np.min(dists[dists > 0])
        
        if min_dist > 0:
            # Simple ratio: closer edges get higher score
            ratio = min_dist / distance_matrix[i]
            ratio[i] = 0  # Zero diagonal
            # Cap at 2.0 to avoid extreme values
            regret_factor[i] = np.minimum(ratio, 2.0)
    
    # Make regret symmetric by averaging both directions
    regret_factor = (regret_factor + regret_factor.T) / 2
    
    # 4. MST connectivity guarantee
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_sym = np.maximum(mst, mst.T) > 0
    
    # 5. Combine components
    base_scores = inv_dist * exp_penalty * regret_factor
    
    # 6. Node-adaptive sparsification with MST preservation
    sparsified = np.zeros_like(base_scores)
    
    for i in range(n):
        row_scores = base_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Always include MST edges for this node
        mst_mask = mst_sym[i]
        
        # Determine adaptive sparsity level
        if n <= 3:
            keep_fraction = 0.5
        else:
            # Simple linear scaling: keep fewer edges in larger problems
            keep_fraction = 0.3 - 0.1 * min(1.0, (n - 10) / 50)
            keep_fraction = max(0.1, keep_fraction)
        
        # Calculate number of edges to keep (minimum 2, maximum n-1)
        mst_count = np.sum(mst_mask)
        non_mst_to_keep = max(0, int(keep_fraction * (n - 1)) - mst_count)
        total_to_keep = min(n - 1, max(2, mst_count + non_mst_to_keep))
        
        # Select edges: MST edges + top non-MST edges
        if total_to_keep > mst_count:
            # Get scores for non-MST edges
            non_mst_scores = row_scores.copy()
            non_mst_scores[mst_mask] = -np.inf
            
            if np.any(non_mst_scores > -np.inf):
                # Get threshold for top non-MST edges
                needed_non_mst = total_to_keep - mst_count
                if needed_non_mst > 0 and np.sum(non_mst_scores > -np.inf) > 0:
                    if needed_non_mst < np.sum(non_mst_scores > -np.inf):
                        threshold = np.partition(
                            non_mst_scores[non_mst_scores > -np.inf], 
                            -needed_non_mst
                        )[-needed_non_mst]
                        top_non_mst = non_mst_scores >= threshold
                    else:
                        top_non_mst = non_mst_scores > -np.inf
                    
                    keep_mask = mst_mask | top_non_mst
                else:
                    keep_mask = mst_mask
            else:
                keep_mask = mst_mask
        else:
            keep_mask = mst_mask
        
        # Apply selection
        sparsified[i][keep_mask] = base_scores[i][keep_mask]
    
    # 7. Symmetrize: keep edge if selected by either endpoint
    symmetric_scores = np.maximum(sparsified, sparsified.T)
    
    # 8. Ensure MST edges have non-zero scores
    symmetric_scores[mst_sym] = np.maximum(symmetric_scores[mst_sym], 1e-6)
    
    # 9. Final normalization
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
