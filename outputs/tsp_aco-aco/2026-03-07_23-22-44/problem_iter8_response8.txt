import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristic with improved connectivity preservation and
    hierarchical structure analysis.
    
    Core improvements over v1:
    1. Enhanced connectivity preservation through hierarchical MST components
    2. Multi-resolution spatial sampling with adaptive granularity
    3. Balanced hub penalization using degree-aware centrality
    4. Tour structure awareness via connectivity path analysis
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix in [0, 1] range, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    epsilon = 1e-12
    
    # ===== 1. DISTRIBUTION-AWARE PARAMETER TUNING =====
    upper_tri = distance_matrix[np.triu_indices(n, 1)]
    dist_mean = np.mean(upper_tri)
    dist_std = np.std(upper_tri)
    coef_var = dist_std / (dist_mean + epsilon)
    
    # Adaptive parameters based on problem characteristics
    if coef_var < 0.25:
        # Regular uniform distribution
        regret_power = 1.8
        cluster_sensitivity = 0.2
        connectivity_weight = 0.7
    elif coef_var < 0.6:
        # Moderate variation
        regret_power = 2.2
        cluster_sensitivity = 0.4
        connectivity_weight = 0.8
    else:
        # Highly variable/clustered
        regret_power = 2.8
        cluster_sensitivity = 0.6
        connectivity_weight = 0.9
    
    # ===== 2. MULTI-RESOLUTION SPATIAL SAMPLING =====
    def compute_multi_resolution_scores() -> np.ndarray:
        """Compute edge scores at multiple spatial resolutions"""
        base_scores = np.ones_like(distance_matrix)
        
        # Resolution 1: Global perspective (direct distance scaling)
        with np.errstate(divide='ignore', invalid='ignore'):
            global_inv = 1.0 / distance_matrix
        global_inv[np.isinf(global_inv)] = 0
        np.fill_diagonal(global_inv, 0)
        
        # Resolution 2: Local neighborhood perspective
        local_scores = np.zeros_like(distance_matrix)
        k_neighbors = min(6, n - 1)
        
        for i in range(n):
            if k_neighbors > 0:
                # Get k nearest neighbors
                dists = distance_matrix[i].copy()
                dists[i] = np.inf
                neighbor_idx = np.argpartition(dists, k_neighbors)[:k_neighbors]
                
                # Local density estimation
                local_dist = np.mean(dists[neighbor_idx])
                if local_dist > 0:
                    # Score edges within local neighborhood
                    local_scale = local_dist * 1.5  # Slightly beyond immediate neighbors
                    local_mask = distance_matrix[i] <= local_scale
                    local_scores[i, local_mask] = 1.0 - (distance_matrix[i, local_mask] / local_scale)
        
        # Resolution 3: Regional perspective (connectivity clusters)
        if n >= 5:
            # Use MST to identify regional structure
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_mask = (mst + mst.T) > 0
            
            # Identify edges that connect different MST components
            component_bonus = np.ones_like(distance_matrix)
            for i in range(n):
                # Find nodes in same MST component
                same_component = np.where(mst_mask[i])[0]
                for j in range(i + 1, n):
                    if j not in same_component:
                        # Edge connects different components - potential bridge
                        component_bonus[i, j] = component_bonus[j, i] = 1.2
        
        # Combine resolutions with adaptive weights
        resolution_weights = [0.4, 0.3, 0.3]  # Global, Local, Regional
        if n < 10:
            resolution_weights = [0.6, 0.4, 0.0]  # Small problems don't need regional analysis
        
        # Normalize components
        if np.max(global_inv) > 0:
            global_inv = global_inv / np.max(global_inv)
        
        local_max = np.max(local_scores)
        if local_max > 0:
            local_scores = local_scores / local_max
        
        base_scores = (resolution_weights[0] * global_inv +
                      resolution_weights[1] * local_scores +
                      resolution_weights[2] * component_bonus)
        
        return base_scores
    
    base_scores = compute_multi_resolution_scores()
    
    # ===== 3. ENHANCED CONNECTIVITY PRESERVATION =====
    def enhance_connectivity(scores: np.ndarray) -> np.ndarray:
        """Ensure connectivity through hierarchical MST analysis"""
        # Compute primary MST
        mst_primary = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst_primary + mst_primary.T) > 0
        
        # Identify critical connectivity edges
        connectivity_scores = scores.copy()
        
        # Boost MST edges with distance-aware scaling
        for i in range(n):
            mst_neighbors = np.where(mst_mask[i])[0]
            for j in mst_neighbors:
                if i < j:
                    # Compute how critical this MST edge is
                    alt_dists = []
                    for k in range(n):
                        if k != i and k != j:
                            # Alternative path through k
                            alt_path = distance_matrix[i, k] + distance_matrix[k, j]
                            alt_dists.append(alt_path)
                    
                    if alt_dists:
                        direct_dist = distance_matrix[i, j]
                        best_alt = min(alt_dists)
                        # Criticality: ratio of best alternative to direct distance
                        criticality = best_alt / (direct_dist + epsilon)
                        # Boost score based on criticality
                        boost = min(0.5, np.tanh(criticality - 1.0))
                        connectivity_scores[i, j] = min(1.0, scores[i, j] * (1.0 + boost * connectivity_weight))
                        connectivity_scores[j, i] = connectivity_scores[i, j]
        
        return connectivity_scores
    
    enhanced_scores = enhance_connectivity(base_scores)
    
    # ===== 4. HIERARCHICAL STRUCTURE SCORING =====
    def compute_hierarchical_scores(scores: np.ndarray) -> np.ndarray:
        """Score edges based on hierarchical structure of the problem"""
        hierarchical_scores = scores.copy()
        
        if n >= 8:
            # Identify potential clusters using distance thresholds
            cluster_threshold = np.percentile(upper_tri, 30)  # 30th percentile as cluster threshold
            
            # Intra-cluster edges (short distances)
            intra_mask = distance_matrix <= cluster_threshold
            np.fill_diagonal(intra_mask, False)
            
            # Inter-cluster edges (medium distances)
            inter_lower = cluster_threshold * 1.5
            inter_upper = cluster_threshold * 3.0
            inter_mask = (distance_matrix > cluster_threshold) & (distance_matrix <= inter_upper)
            
            # Adjust scores based on hierarchy level
            # Intra-cluster edges get moderate boost
            hierarchical_scores[intra_mask] *= 1.2
            
            # Critical inter-cluster edges get selective boost
            if np.any(inter_mask):
                # Boost edges that are the best connection between their endpoints
                for i in range(n):
                    for j in range(i + 1, n):
                        if inter_mask[i, j]:
                            # Check if this is the best connection between these nodes
                            same_cluster_i = np.where(intra_mask[i])[0]
                            same_cluster_j = np.where(intra_mask[j])[0]
                            
                            # If nodes are in different clusters, boost connecting edge
                            if not np.any(np.isin(same_cluster_i, same_cluster_j)):
                                # This edge connects different clusters
                                boost_factor = 1.0 + 0.3 * cluster_sensitivity
                                hierarchical_scores[i, j] = min(1.0, scores[i, j] * boost_factor)
                                hierarchical_scores[j, i] = hierarchical_scores[i, j]
        
        return hierarchical_scores
    
    hierarchical_scores = compute_hierarchical_scores(enhanced_scores)
    
    # ===== 5. BALANCED HUB PENALIZATION =====
    def apply_hub_penalization(scores: np.ndarray) -> np.ndarray:
        """Penalize over-connected nodes to encourage balanced tours"""
        balanced_scores = scores.copy()
        
        # Compute node centrality (inverse distance sum)
        centrality = np.zeros(n)
        for i in range(n):
            inv_dists = 1.0 / (distance_matrix[i] + epsilon)
            inv_dists[i] = 0
            centrality[i] = np.sum(inv_dists)
        
        # Normalize centrality
        max_centrality = np.max(centrality)
        if max_centrality > 0:
            centrality = centrality / max_centrality
        
        # Penalize edges from highly central nodes
        target_degree = 2.2  # Slightly above 2 to allow flexibility
        
        for i in range(n):
            # Estimate potential degree for this node
            row_scores = scores[i].copy()
            row_scores[i] = 0
            
            # Threshold based on score distribution
            if np.any(row_scores > 0):
                score_threshold = np.percentile(row_scores[row_scores > 0], 60)
                
                # Count edges above threshold
                potential_edges = np.sum(row_scores > score_threshold)
                
                # Apply penalty if node is too central and has too many potential edges
                if potential_edges > target_degree:
                    # Penalty increases with centrality and excess edges
                    excess_ratio = potential_edges / target_degree
                    centrality_factor = 0.3 + 0.7 * centrality[i]  # [0.3, 1.0]
                    penalty = min(0.8, (excess_ratio - 1.0) * centrality_factor * 0.5)
                    
                    # Apply penalty to all edges from this node
                    mask = row_scores > score_threshold
                    balanced_scores[i, mask] = scores[i, mask] * (1.0 - penalty)
        
        return balanced_scores
    
    balanced_scores = apply_hub_penalization(hierarchical_scores)
    
    # ===== 6. TOUR STRUCTURE AWARENESS =====
    def apply_tour_constraints(scores: np.ndarray) -> np.ndarray:
        """Apply soft constraints based on tour structure requirements"""
        final_scores = scores.copy()
        
        # Ensure symmetry
        final_scores = (final_scores + final_scores.T) / 2.0
        
        # Guarantee minimum connectivity through MST
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst + mst.T) > 0
        
        # Ensure all MST edges have non-zero scores
        mst_min_score = np.max(final_scores) * 0.2
        final_scores[mst_mask] = np.maximum(final_scores[mst_mask], mst_min_score)
        
        # Penalize very long edges (beyond 3x average)
        long_edge_threshold = dist_mean * 3.0
        long_mask = distance_matrix > long_edge_threshold
        np.fill_diagonal(long_mask, False)
        final_scores[long_mask] *= 0.3
        
        return final_scores
    
    final_scores = apply_tour_constraints(balanced_scores)
    
    # ===== 7. ADAPTIVE SPARSIFICATION =====
    def adaptive_sparsify(scores: np.ndarray) -> np.ndarray:
        """Retain most promising edges while ensuring connectivity"""
        sparsified = np.zeros_like(scores)
        
        # Always include MST edges
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst + mst.T) > 0
        
        # Determine adaptive k for each node
        for i in range(n):
            row_scores = scores[i].copy()
            row_scores[i] = -np.inf
            
            # Base k depends on problem size
            if n <= 10:
                base_k = n - 1
            elif n <= 30:
                base_k = max(4, int(np.log2(n) * 2))
            else:
                base_k = max(5, int(np.sqrt(n)))
            
            # Adjust k based on local density
            local_dists = distance_matrix[i].copy()
            local_dists[i] = np.inf
            if np.any(local_dists < np.inf):
                local_avg = np.mean(local_dists[local_dists < np.inf])
                density_factor = dist_mean / (local_avg + epsilon)
                k = min(n - 1, int(base_k * (0.7 + 0.3 * density_factor)))
            else:
                k = base_k
            
            # Select top k edges, ensuring MST edges are included
            keep_mask = mst_mask[i].copy()
            
            # Add top non-MST edges
            non_mst_scores = row_scores.copy()
            non_mst_scores[mst_mask[i]] = -np.inf
            
            if np.sum(non_mst_scores > -np.inf) > 0:
                # Get threshold for top k - mst_edges
                mst_count = np.sum(mst_mask[i])
                needed = max(0, k - mst_count)
                
                if needed > 0:
                    valid_scores = non_mst_scores[non_mst_scores > -np.inf]
                    if len(valid_scores) >= needed:
                        threshold = np.partition(valid_scores, -needed)[-needed]
                        top_mask = non_mst_scores >= threshold
                        keep_mask = keep_mask | top_mask
            
            sparsified[i, keep_mask] = scores[i, keep_mask]
        
        # Ensure symmetry
        sparsified = np.maximum(sparsified, sparsified.T)
        
        # Fill diagonal with zeros
        np.fill_diagonal(sparsified, 0)
        
        return sparsified
    
    sparsified_scores = adaptive_sparsify(final_scores)
    
    # ===== 8. FINAL NORMALIZATION =====
    # Ensure non-negativity
    sparsified_scores = np.maximum(sparsified_scores, 0)
    
    # Normalize to [0, 1]
    max_score = np.max(sparsified_scores)
    if max_score > 0:
        sparsified_scores = sparsified_scores / max_score
    
    # Apply final power scaling based on distribution
    if coef_var < 0.4:
        power = 0.85  # Gentle for uniform
    elif coef_var < 0.7:
        power = 0.75  # Moderate
    else:
        power = 0.65  # Strong for clustered
    
    sparsified_scores = np.power(sparsified_scores, power)
    
    return sparsified_scores
