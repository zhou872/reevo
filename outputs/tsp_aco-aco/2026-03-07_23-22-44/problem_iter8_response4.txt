```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List, Tuple
import random

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Next-generation TSP heuristic with evolutionary optimization, neural-inspired edge saliency,
    hybrid graph embeddings, and multi-fidelity adaptive ensemble learning.
    
    Core Innovations:
    1. Evolutionary Parameter Optimization: Genetic algorithm to optimize heuristic parameters
    2. Neural Edge Saliency Approximation: Multi-layer perceptron-inspired edge scoring
    3. Hybrid Graph Embedding: Combines spectral, distance, and topological embeddings
    4. Multi-fidelity Adaptive Ensemble: Learns weights from tours at different quality levels
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. PRELIMINARY COMPUTATIONS ====================
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # MST for connectivity
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    
    # ==================== 2. EVOLUTIONARY PARAMETER OPTIMIZATION ====================
    def evolve_parameters(pop_size: int = 20, generations: int = 15) -> dict:
        """Evolve optimal parameters using genetic algorithm"""
        
        # Define parameter search space
        param_ranges = {
            'regret_depth': (2, min(5, max(2, int(np.log2(n))))),
            'regret_beam': (3, min(10, max(3, n // 5))),
            'cluster_weight': (0.1, 0.8),
            'mst_bonus': (1.0, 2.0),
            'sparsity_factor': (0.2, 0.8)
        }
        
        # Initialize population
        population = []
        for _ in range(pop_size):
            params = {}
            for key, (low, high) in param_ranges.items():
                if isinstance(low, int):
                    params[key] = random.randint(low, high)
                else:
                    params[key] = low + random.random() * (high - low)
            population.append(params)
        
        best_params = None
        best_fitness = -float('inf')
        
        for generation in range(generations):
            # Evaluate fitness
            fitness_scores = []
            for params in population:
                fitness = evaluate_parameters(params, distance_matrix)
                fitness_scores.append(fitness)
                
                if fitness > best_fitness:
                    best_fitness = fitness
                    best_params = params.copy()
            
            # Selection (tournament selection)
            new_population = []
            tournament_size = 3
            
            for _ in range(pop_size):
                tournament = random.sample(range(pop_size), tournament_size)
                winner = max(tournament, key=lambda i: fitness_scores[i])
                new_population.append(population[winner].copy())
            
            # Crossover and mutation
            for i in range(0, pop_size, 2):
                if i + 1 < pop_size:
                    parent1 = new_population[i]
                    parent2 = new_population[i + 1]
                    
                    # Uniform crossover
                    child1, child2 = {}, {}
                    for key in param_ranges:
                        if random.random() < 0.5:
                            child1[key] = parent1[key]
                            child2[key] = parent2[key]
                        else:
                            child1[key] = parent2[key]
                            child2[key] = parent1[key]
                    
                    # Mutation
                    for child in [child1, child2]:
                        if random.random() < 0.2:
                            key = random.choice(list(param_ranges.keys()))
                            low, high = param_ranges[key]
                            if isinstance(low, int):
                                child[key] = random.randint(low, high)
                            else:
                                child[key] = low + random.random() * (high - low)
                    
                    new_population[i] = child1
                    new_population[i + 1] = child2
            
            population = new_population
        
        return best_params
    
    def evaluate_parameters(params: dict, dist_matrix: np.ndarray) -> float:
        """Evaluate parameter fitness using fast heuristic"""
        n_nodes = dist_matrix.shape[0]
        
        # Quick regret computation
        regret_scores = np.zeros((n_nodes, n_nodes))
        for i in range(n_nodes):
            for j in range(i + 1, n_nodes):
                # Simple regret approximation
                dist_ij = dist_matrix[i, j]
                alternatives = []
                for k in range(min(5, n_nodes)):
                    if k != i and k != j:
                        alt_cost = dist_matrix[i, k] + dist_matrix[k, j]
                        alternatives.append(alt_cost)
                if alternatives:
                    best_alt = min(alternatives)
                    regret = max(0, best_alt - dist_ij)
                    regret_scores[i, j] = 1.0 / (1.0 + regret / (dist_ij + 1e-8))
                    regret_scores[j, i] = regret_scores[i, j]
        
        # Quick composite
        with np.errstate(divide='ignore', invalid='ignore'):
            inv_d = 1.0 / dist_matrix
        inv_d[np.isinf(inv_d)] = 0
        np.fill_diagonal(inv_d, 0)
        
        mst = minimum_spanning_tree(dist_matrix).toarray()
        mst_e = (mst > 0) | (mst.T > 0)
        
        composite = (0.5 * regret_scores + 0.3 * inv_d / np.max(inv_d) + 
                    0.2 * mst_e.astype(float))
        
        # Generate simple tour using composite
        start = random.randint(0, n_nodes - 1)
        visited = {start}
        tour = [start]
        current = start
        
        while len(visited) < n_nodes:
            # Use composite for next node selection
            candidates = [j for j in range(n_nodes) if j not in visited]
            if not candidates:
                break
            scores = [composite[current, j] for j in candidates]
            next_node = candidates[np.argmax(scores)]
            tour.append(next_node)
            visited.add(next_node)
            current = next_node
        
        # Calculate tour length
        tour_length = 0
        for i in range(len(tour) - 1):
            tour_length += dist_matrix[tour[i], tour[i + 1]]
        
        return 1.0 / (tour_length + 1e-8)
    
    # Get optimized parameters
    opt_params = evolve_parameters()
    
    # ==================== 3. NEURAL EDGE SALIENCY APPROXIMATION ====================
    def neural_edge_saliency() -> np.ndarray:
        """Compute edge importance using neural-inspired multi-feature processing"""
        
        # Feature extraction
        features = []
        
        # Feature 1: Normalized distance
        max_dist = np.max(distance_matrix)
        if max_dist > 0:
            norm_dist = 1.0 - distance_matrix / max_dist
            np.fill_diagonal(norm_dist, 0)
            features.append(norm_dist.flatten())
        
        # Feature 2: Local clustering coefficient approximation
        clustering_feature = np.zeros_like(distance_matrix)
        for i in range(n):
            # Get neighbors within median distance
            median_dist = np.median(distance_matrix[i])
            neighbors = [j for j in range(n) if j != i and 
                        distance_matrix[i, j] < 1.5 * median_dist]
            
            if len(neighbors) >= 2:
                # Approximate local density
                neighbor_dists = [distance_matrix[i, j] for j in neighbors]
                avg_neighbor_dist = np.mean(neighbor_dists)
                for j in neighbors:
                    clustering_feature[i, j] = np.exp(-distance_matrix[i, j] / 
                                                     (2 * avg_neighbor_dist))
        
        features.append(clustering_feature.flatten())
        
        # Feature 3: Minimum spanning tree proximity
        mst_proximity = np.zeros_like(distance_matrix)
        for i in range(n):
            for j in range(i + 1, n):
                if mst_edges[i, j]:
                    mst_proximity[i, j] = 1.0
                else:
                    # Distance to nearest MST edge
                    min_mst_dist = float('inf')
                    for k in range(n):
                        if mst_edges[i, k]:
                            path_dist = distance_matrix[i, k] + distance_matrix[k, j]
                            min_mst_dist = min(min_mst_dist, path_dist)
                    mst_proximity[i, j] = np.exp(-min_mst_dist / (2 * max_dist))
                mst_proximity[j, i] = mst_proximity[i, j]
        
        features.append(mst_proximity.flatten())
        
        # Neural-inspired processing (simulated multi-layer perceptron)
        saliency = np.zeros_like(distance_matrix)
        
        # Simple feed-forward simulation
        for i in range(n):
            for j in range(i + 1, n):
                # Input features for this edge
                f1 = 1.0 - distance_matrix[i, j] / max_dist if max_dist > 0 else 0
                f2 = clustering_feature[i, j]
                f3 = mst_proximity[i, j]
                
                # Hidden layer simulation (simplified)
                h1 = np.tanh(0.8 * f1 + 0.4 * f2 + 0.3 * f3)
                h2 = np.tanh(0.3 * f1 + 0.6 * f2 + 0.5 * f3)
                h3 = np.tanh(0.5 * f1 + 0.2 * f2 + 0.7 * f3)
                
                # Output layer
                output = 0.4 * h1 + 0.3 * h2 + 0.3 * h3
                saliency[i, j] = (output + 1) / 2  # Scale to [0, 1]
                saliency[j, i] = saliency[i, j]
        
        return saliency
    
    neural_scores = neural_edge_saliency()
    
    # ==================== 4. HYBRID GRAPH EMBEDDING ====================
    def hybrid_embedding() -> np.ndarray:
        """Create hybrid embedding combining multiple graph representations"""
        
        embedding_scores = np.ones_like(distance_matrix)
        
        # Spectral embedding
        try:
            degree_matrix = np.diag(np.sum(distance_matrix, axis=1))
            laplacian = degree_matrix - distance_matrix
            eigenvalues, eigenvectors = np.linalg.eigh(laplacian)
            
            k_clusters = max(2, min(5, int(np.sqrt(n))))
            idx = np.argsort(eigenvalues)[1:k_clusters+1]
            spectral_embedding = eigenvectors[:, idx]
            
            # Spectral affinity
            for i in range(n):
                for j in range(i + 1, n):
                    vec_i = spectral_embedding[i]
                    vec_j = spectral_embedding[j]
                    similarity = np.dot(vec_i, vec_j) / (np.linalg.norm(vec_i) * 
                                                        np.linalg.norm(vec_j) + 1e-8)
                    spectral_factor = 1.0 + 0.5 * similarity
                    embedding_scores[i, j] *= spectral_factor
                    embedding_scores[j, i] = embedding_scores[i, j]
        except:
            pass
        
        # Distance-based embedding (MDS approximation)
        dist_embedding = np.zeros((n, 2))
        for i in range(n):
            # Simple 2D embedding based on distances to a few reference points
            ref1 = random.randint(0, n - 1)
            ref2 = random.randint(0, n - 1)
            while ref2 == ref1:
                ref2 = random.randint(0, n - 1)
            
            dist_embedding[i, 0] = distance_matrix[i, ref1]
            dist_embedding[i, 1] = distance_matrix[i, ref2]
        
        # Distance embedding factor
        for i in range(n):
            for j in range(i + 1, n):
                diff = np.linalg.norm(dist_embedding[i] - dist_embedding[j])
                avg_dist = np.mean([np.linalg.norm(dist_embedding[i]), 
                                   np.linalg.norm(dist_embedding[j])])
                if avg_dist > 0:
                    embedding_factor = np.exp(-diff / (2 * avg_dist))
                    embedding_scores[i, j] *= (1.0 + 0.3 * embedding_factor)
                    embedding_scores[j, i] = embedding_scores[i, j]
        
        return embedding_scores
    
    embedding_scores = hybrid_embedding()
    
    # ==================== 5. ADVANCED REGRET COMPUTATION ====================
    def advanced_regret(beam_width: int, max_depth: int) -> np.ndarray:
        """Compute regret with optimized parameters"""
        regret_scores = np.zeros((n, n))
        
        # Adaptive depth based on problem size and evolved parameter
        adaptive_depth = min(max_depth, max(2, int(np.log2(n))))
        
        for i in range(n):
            for j in range(i + 1, n):
                direct_cost = distance_matrix[i, j]
                
                # Beam search for alternatives
                beams = []
                
                # Start alternatives
                for k in range(n):
                    if k == i or k == j:
                        continue
                    alt_cost = distance_matrix[i, k]
                    visited = {i, k}
                    path = [i, k]
                    beams.append((alt_cost, visited, path))
                
                # Beam search
                for depth in range(1, adaptive_depth):
                    new_beams = []
                    
                    for cost, visited, path in beams:
                        current = path[-1]
                        
                        # Get candidate next nodes
                        candidates = []
                        for next_node in range(n):
                            if next_node not in visited and next_node != current:
                                alt = cost + distance_matrix[current, next_node]
                                candidates.append((alt, next_node))
                        
                        # Sort and take top beam_width
                        candidates.sort(key=lambda x: x[0])
                        for alt_cost, next_node in candidates[:beam_width]:
                            new_cost = alt_cost
                            new_visited = visited.copy()
                            new_visited.add(next_node)
                            new_path = path + [next_node]
                            
                            new_beams.append((new_cost, new_visited, new_path))
                    
                    if new_beams:
                        new_beams.sort(key=lambda x: x[0])
                        beams = new_beams[:beam_width]
                    else:
                        break
                
                # Find best alternative to j
                best_alternative = float('inf')
                for cost, visited, path in beams:
                    if path[-1] == j and len(path) > 2:
                        best_alternative = min(best_alternative, cost)
                
                if best_alternative < float('inf'):
                    regret = max(0, best_alternative - direct_cost)
                    scale = np.median(distance_matrix[np.triu_indices(n, 1)])
                    regret_scores[i, j] = 1.0 / (1.0 + np.exp(-regret / (scale + 1e-8)))
                    regret_scores[j, i] = regret_scores[i, j]
        
        return regret_scores
    
    # Use evolved parameters for regret
    regret_scores = advanced_regret(
        beam_width=opt_params['regret_beam'],
        max_depth=opt_params['regret_depth']
    )
    
    # ==================== 6. MULTI-FIDELITY ADAPTIVE ENSEMBLE ====================
    def multi_fidelity_learning() -> Tuple[float, float, float, float]:
        """Learn weights from tours at different quality levels"""
        
        # Generate tours at multiple fidelity levels
        tour_groups = []
        
        # Level 1: Random tours (low fidelity)
        random_tours = []
        for _ in range(min(10, n)):
            tour = list(range(n))
            random.shuffle(tour)
            tour.append(tour[0])
            random_tours.append(tour)
        tour_groups.append(random_tours)
        
        # Level 2: Greedy tours (medium fidelity)
        greedy_tours = []
        for _ in range(min(10, n)):
            start = random.randint(0, n - 1)
            visited = {start}
            tour = [start]
            current = start
            
            while len(visited) < n:
                candidates = [j for j in range(n) if j not in visited]
                next_node = min(candidates, key=lambda j: distance_matrix[current, j])
                tour.append(next_node)
                visited.add(next_node)
                current = next_node
            
            tour.append(start)
            greedy_tours.append(tour)
        tour_groups.append(greedy_tours)
        
        # Level 3: 2-opt improved tours (high fidelity)
        def two_opt_improvement(tour: List[int]) -> List[int]:
            """Simple 2-opt improvement"""
            improved = True
            current_tour = tour[:-1]
            n_nodes = len(current_tour)
            
            while improved:
                improved = False
                best_gain = 0
                best_i, best_j = -1, -1
                
                for i in range(n_nodes - 1):
                    for j in range(i + 2, n_nodes):
                        a, b = current_tour[i], current_tour[(i + 1) % n_nodes]
                        c, d = current_tour[j], current_tour[(j + 1) % n_nodes]
                        
                        current_cost = distance_matrix[a, b] + distance_matrix[c, d]
                        new_cost = distance_matrix[a, c] + distance_matrix[b, d]
                        gain = current_cost - new_cost
                        
                        if gain > best_gain:
                            best_gain = gain
                            best_i, best_j = i, j
                
                if best_gain > 0:
                    i, j = best_i, best_j
                    current_tour[i + 1:j + 1] = reversed(current_tour[i + 1:j + 1])
                    improved = True
            
            return current_tour + [current_tour[0]]
        
        improved_tours = []
        base_tours = random_tours + greedy_tours
        for tour in base_tours[:min(10, len(base_tours))]:
            improved = two_opt_improvement(tour)
            improved_tours.append(improved)
        tour_groups.append(improved_tours)
        
        # Evaluate components at each fidelity level
        components = [regret_scores, inv_dist, neural_scores, embedding_scores]
        level_weights = []
        
        for level, tours in enumerate(tour_groups):
            comp_scores = []
            for comp in components:
                # Score component on this level's tours
                total_score = 0
                for tour in tours:
                    tour_score = 0
                    tour_length = 0
                    for k in range(len(tour) - 1):
                        i, j = tour[k], tour[k + 1]
                        tour_score += comp[i, j]
                        tour_length += distance_matrix[i, j]
                    
                    avg_score = tour_score / (len(tour) - 1)
                    # Higher weight for shorter tours at higher fidelity levels
                    weight = 1.0 / (tour_length + 1e-8)
                    if level == 2:  # High fidelity
                        weight *= 2.0
                    
                    total_score += avg_score * weight
                
                comp_scores.append(total_score / len(tours))
            
            # Normalize component scores for this level
            total = sum(comp_scores)
            if total > 0:
                level_weights.append([s / total for s in comp_scores])
            else:
                level_weights.append([0.25] * 4)
        
        # Combine level weights with level importance
        # Higher fidelity levels get more weight
        level_importance = [0.2, 0.3, 0.5]  # Sum to 1.0
        final_weights = np.zeros(4)
        for level, importance in zip(range(3), level_importance):
            final_weights += importance * np.array(level_weights[level])
        
        # Normalize final weights
        final_weights /= np.sum(final_weights)
        
        return tuple(final_weights)
    
    w_regret, w_inv, w_neural, w_embed = multi_fidelity_learning()
    
    # ==================== 7. COMPOSITE SCORE CALCULATION ====================
    # Weighted combination using learned weights
    base_composite = (
        w_regret * regret_scores +
        w_inv * inv_dist / np.max(inv_dist) +
        w_neural * neural_scores +
        w_embed * embedding_scores
    )
    
    # Apply evolved MST bonus
    mst_bonus = np.where(mst_edges, opt_params['mst_bonus'], 1.0)
    enhanced_composite = base_composite * mst_bonus
    
    # ==================== 8. ADAPTIVE SPARSIFICATION ====================
    def adaptive_sparsification(scores: np.ndarray) -> np.ndarray:
        """Intelligent edge retention using evolved sparsity factor"""
        sparsified = np.zeros_like(scores)
        
        # Always include MST edges
        sparsified[mst_edges] = scores[mst_edges]
        
        # Evolved sparsity factor
        sparsity_factor = opt_params['sparsity_factor']
        
        # Determine degree limit
        if n <= 10:
            max_degree = 4
        elif n <= 50:
            max_degree = min(5, int(n * sparsity_factor))
        else:
            max_degree = min(6, int(n * sparsity_factor * 0.5))
        
        current_degree = np.sum(sparsified > 0, axis=1)
        
        # Collect candidates
        candidates = []
        for i in range(n):
            for j in range(i + 1, n):
                if not mst_edges[i, j] and scores[i, j] > 0:
                    median_dist = np.median(distance_matrix)
                    dist_penalty = np.exp(-distance_matrix[i, j] / (2 * median_dist))
                    effective_score = scores[i, j] * dist_penalty
                    candidates.append((effective_score, i, j))
        
        # Sort and select
        candidates.sort(reverse=True, key=lambda x: x[0])
        
        for score, i, j in candidates:
            if current_degree[i] < max_degree and current_degree[j] < max_degree:
                sparsified[i, j] = scores[i, j]
                sparsified[j, i] = sparsified[i, j]
                current_degree[i] += 1
                current_degree[j] += 1
        
        return sparsified
    
    sparsified_scores = adaptive_sparsification(enhanced_composite)
    
    # ==================== 9. FINAL PROCESSING ====================
    # Ensure symmetry
    symmetric_scores = (sparsified_scores + sparsified_scores.T) / 2
    
    # Final distance-based smoothing
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        dist_decay = np.exp(-distance_matrix / (0.3 * max_dist + 1e-8))
        symmetric_scores *= dist_decay
    
    # Normalize
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
```
