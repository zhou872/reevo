import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray,
                  density_k: Optional[int] = None,
                  regret_temperature: Optional[float] = None,
                  lookahead_factor: float = 0.25) -> np.ndarray:
    """
    Advanced TSP edge heuristic with selective two-step lookahead for high-potential edges.
    
    Features:
    1. Efficient nearest-neighbor regret approximation with adaptive scaling
    2. Density-weighted inverse distance using k-nearest neighbors
    3. Selective two-step lookahead (triangle + 2-opt) for critical edges
    4. Topology-aware sparsification preserving connectivity
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        density_k: Number of neighbors for density estimation (auto-tuned if None)
        regret_temperature: Scaling for regret exponential (auto-tuned if None)
        lookahead_factor: Influence weight for lookahead scores [0, 0.5]
        
    Returns:
        Matrix of edge priorities in [0,1], same shape as input
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # ==================== ADAPTIVE PARAMETER TUNING ====================
    if density_k is None:
        density_k = max(2, min(8, n // 10 + 2))
    
    if regret_temperature is None:
        off_diag = distance_matrix[~np.eye(n, dtype=bool)]
        regret_temperature = np.percentile(off_diag, 25) if len(off_diag) > 0 else 1.0
    
    # ==================== 1. DENSITY-WEIGHTED BASE SCORES ====================
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(inv_dist, 0)
    
    # Local density estimation using k-nearest neighbors
    local_density = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        k_actual = min(density_k, n - 1)
        if k_actual > 0:
            k_nearest = np.partition(dists, k_actual)[:k_actual]
            local_density[i] = 1.0 / (np.mean(k_nearest) + 1e-12)
    
    density_weight = np.sqrt(np.maximum(np.outer(local_density, local_density), 1e-12))
    base_scores = inv_dist * density_weight
    
    # ==================== 2. MST STRUCTURAL REINFORCEMENT ====================
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # MST distance-aware bonus
    mst_distances = np.where(mst_mask, distance_matrix, np.inf)
    mst_median = np.median(mst_distances[mst_mask]) if np.any(mst_mask) else 1.0
    mst_norm = np.exp(-mst_distances / (mst_median + 1e-12))
    mst_bonus = 1.0 + np.where(mst_mask, 1.5 * mst_norm, 0)
    
    # ==================== 3. EFFICIENT REGRET APPROXIMATION ====================
    regret_scores = np.ones_like(distance_matrix)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if n > 1:
            # Use nearest neighbor for efficient regret
            best_idx = np.argmin(dists)
            best_dist = dists[best_idx]
            
            for j in range(n):
                if i == j:
                    continue
                
                regret = max(distance_matrix[i, j] - best_dist, 0)
                regret_penalty = np.exp(-regret / (regret_temperature + 1e-12))
                regret_scores[i, j] = regret_penalty
    
    regret_scores = (regret_scores + regret_scores.T) / 2
    
    # ==================== 4. SELECTIVE TWO-STEP LOOKAHEAD ====================
    lookahead_scores = np.ones_like(distance_matrix)
    
    if n > 3 and np.any(mst_mask):
        # Precompute nearest neighbors for efficiency
        nn_count = min(6, n - 2)
        nearest = np.argsort(distance_matrix, axis=1)[:, 1:nn_count + 1]
        
        # Identify high-potential edges: MST edges + top 20% of non-MST edges
        upper_triu = np.triu_indices(n, 1)
        non_mst_mask = ~mst_mask[upper_triu]
        
        if np.any(non_mst_mask):
            non_mst_scores = base_scores[upper_triu][non_mst_mask]
            threshold = np.percentile(non_mst_scores, 80) if len(non_mst_scores) > 0 else 0
            high_potential = np.zeros((n, n), dtype=bool)
            high_potential[upper_triu] = (base_scores[upper_triu] >= threshold) & non_mst_mask
            high_potential = high_potential | high_potential.T
        
        for i in range(n):
            for j in range(i + 1, n):
                if not (mst_mask[i, j] or high_potential[i, j]):
                    continue
                
                # Step 1: Triangle lookahead (1-step)
                best_triangle = float('inf')
                triangle_candidates = set(nearest[i]).union(nearest[j])
                triangle_candidates.discard(i)
                triangle_candidates.discard(j)
                
                for k in triangle_candidates:
                    if k == i or k == j:
                        continue
                    triangle_cost = (distance_matrix[i, j] + 
                                     distance_matrix[j, k] + 
                                     distance_matrix[k, i])
                    best_triangle = min(best_triangle, triangle_cost)
                
                # Step 2: 2-opt lookahead for MST edges (2-step)
                best_2opt_gain = 0.0
                if mst_mask[i, j] and nn_count >= 4:
                    # Consider candidate swaps with neighbors of i and j
                    i_neighbors = [nbr for nbr in nearest[i] if nbr != j][:3]
                    j_neighbors = [nbr for nbr in nearest[j] if nbr != i][:3]
                    
                    for k in i_neighbors:
                        for l in j_neighbors:
                            if k == l or k == j or l == i:
                                continue
                            # Classic 2-opt gain calculation
                            current = distance_matrix[i, j] + distance_matrix[k, l]
                            alternative = distance_matrix[i, k] + distance_matrix[j, l]
                            gain = current - alternative
                            best_2opt_gain = max(best_2opt_gain, gain)
                
                # Combine lookahead insights
                if best_triangle < float('inf'):
                    triangle_reward = 1.0 / (best_triangle + 1e-12)
                    base_val = base_scores[i, j] + 1e-12
                    triangle_factor = triangle_reward / (base_val + triangle_reward)
                    
                    # Apply 2-opt penalty for MST edges (higher gain = less promising)
                    if best_2opt_gain > 0:
                        penalty = np.exp(-best_2opt_gain / (regret_temperature + 1e-12))
                        combined_factor = 1.0 + lookahead_factor * (triangle_factor * penalty)
                    else:
                        combined_factor = 1.0 + lookahead_factor * triangle_factor
                    
                    lookahead_scores[i, j] = lookahead_scores[j, i] = combined_factor
    
    # ==================== 5. COMBINE COMPONENTS ====================
    reinforced_scores = base_scores * mst_bonus
    regret_adjusted = reinforced_scores * regret_scores
    enhanced_scores = regret_adjusted * lookahead_scores
    
    # ==================== 6. TOPOLOGY-AWARE SPARSIFICATION ====================
    # Compute edge importance based on MST centrality and local density
    mst_degree = np.sum(mst_mask, axis=1)
    degree_importance = 1.0 + mst_degree / np.max(mst_degree + 1e-12)
    
    sparsified = np.zeros_like(enhanced_scores)
    for i in range(n):
        row_scores = enhanced_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Adaptive edge retention: more edges for high-degree nodes
        base_edges = max(3, int(np.sqrt(n) * 0.8))
        adaptive_k = min(n - 1, int(base_edges * degree_importance[i]))
        
        if adaptive_k > 0:
            threshold_idx = -min(adaptive_k, len(row_scores) - 1)
            threshold = np.partition(row_scores, threshold_idx)[threshold_idx]
            mask = row_scores >= threshold
            sparsified[i, mask] = enhanced_scores[i, mask]
    
    # Ensure connectivity for small components
    components = np.arange(n)
    for i in range(n):
        if np.sum(sparsified[i] > 0) == 0:
            # Add best edge for isolated node
            best_j = np.argmax(base_scores[i])
            sparsified[i, best_j] = enhanced_scores[i, best_j]
            sparsified[best_j, i] = enhanced_scores[best_j, i]
    
    # ==================== 7. FINAL PROCESSING ====================
    symmetric = (sparsified + sparsified.T) / 2
    np.fill_diagonal(symmetric, 0)
    
    # Apply soft distance decay for very long edges
    dist_decay = np.exp(-distance_matrix / (np.median(distance_matrix[~np.eye(n, dtype=bool)]) + 1e-12))
    symmetric *= dist_decay
    
    # Global normalization to [0,1]
    max_val = np.max(symmetric)
    if max_val > 0:
        symmetric = symmetric / max_val
    
    return symmetric
