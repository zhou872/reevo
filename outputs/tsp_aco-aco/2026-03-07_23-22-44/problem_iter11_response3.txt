import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP heuristic combining spectral analysis, fixed ensemble weights,
    density-adaptive scoring, and intelligent sparsification.
    
    Key Improvements over v1:
    1. Replaces complex beam search with efficient spectral clustering
    2. Uses fixed ensemble weights for computational efficiency
    3. Implements adaptive density-weighted scoring
    4. Simplified sparsification with degree constraints
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances
    
    Returns:
    --------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # ==================== 1. BASIC COMPONENTS ====================
    # Inverse distance (normalized)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    if np.max(inv_dist) > 0:
        inv_dist = inv_dist / np.max(inv_dist)
    
    # MST for connectivity preservation
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst > 0) | (mst.T > 0)
    
    # ==================== 2. SPECTRAL AFFINITY SMOOTHING ====================
    def compute_spectral_affinity() -> np.ndarray:
        """Compute edge scores using graph Laplacian and Fiedler vector"""
        # Convert distances to similarities using Gaussian kernel
        median_dist = np.median(distance_matrix[np.triu_indices(n, 1)])
        sigma = median_dist / 2.0
        similarity = np.exp(-distance_matrix**2 / (2 * sigma**2))
        np.fill_diagonal(similarity, 0)
        
        # Graph Laplacian
        degree = np.sum(similarity, axis=1)
        D = np.diag(degree)
        L = D - similarity
        
        # Compute eigenvectors (use k=min(5, n-1) eigenvectors)
        k = min(5, n - 1)
        try:
            eigenvalues, eigenvectors = np.linalg.eigh(L)
            # Sort by eigenvalue
            idx = np.argsort(eigenvalues)
            eigenvectors = eigenvectors[:, idx]
            
            # Use Fiedler vector (second smallest eigenvalue) and next few
            spectral_embeddings = eigenvectors[:, 1:k+1]
            
            # Compute affinity based on spectral similarity
            spectral_scores = np.zeros((n, n))
            
            for i in range(n):
                for j in range(i + 1, n):
                    # Cosine similarity in spectral space
                    vec_i = spectral_embeddings[i]
                    vec_j = spectral_embeddings[j]
                    norm_i = np.linalg.norm(vec_i)
                    norm_j = np.linalg.norm(vec_j)
                    
                    if norm_i > 0 and norm_j > 0:
                        cos_sim = np.dot(vec_i, vec_j) / (norm_i * norm_j)
                        # Convert to positive score [0, 1]
                        spectral_scores[i, j] = (cos_sim + 1) / 2
                    else:
                        spectral_scores[i, j] = 0.5
                    
                    spectral_scores[j, i] = spectral_scores[i, j]
            
            return spectral_scores
            
        except np.linalg.LinAlgError:
            # Fallback to distance-based scores if spectral computation fails
            return 0.5 * np.ones_like(distance_matrix)
    
    spectral_scores = compute_spectral_affinity()
    
    # ==================== 3. ADAPTIVE DENSITY-WEIGHTED SCORING ====================
    def compute_density_weighted_scores() -> np.ndarray:
        """Compute scores weighted by local density using k-nearest neighbors"""
        # Determine k for k-NN density estimation
        k = min(5, n - 1)
        
        # Compute local densities
        densities = np.zeros(n)
        for i in range(n):
            # Get distances to other nodes
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Average distance to k nearest neighbors
            k_nearest = np.partition(dists, k)[:k]
            densities[i] = 1.0 / (np.mean(k_nearest) + 1e-8)
        
        # Normalize densities
        if np.max(densities) > 0:
            densities = densities / np.max(densities)
        
        # Compute density-weighted scores
        density_scores = np.zeros_like(distance_matrix)
        
        for i in range(n):
            for j in range(i + 1, n):
                # Geometric mean of endpoint densities
                density_weight = np.sqrt(densities[i] * densities[j])
                
                # Inverse distance weighted by density
                base_score = 1.0 / (distance_matrix[i, j] + 1e-8)
                density_scores[i, j] = base_score * density_weight
                density_scores[j, i] = density_scores[i, j]
        
        # Normalize
        if np.max(density_scores) > 0:
            density_scores = density_scores / np.max(density_scores)
        
        return density_scores
    
    density_scores = compute_density_weighted_scores()
    
    # ==================== 4. FIXED WEIGHT ENSEMBLE COMBINATION ====================
    # Fixed weights as suggested in long-term recommendations
    w_regret = 0.0  # Not computed in this version
    w_spectral = 0.4
    w_density = 0.3
    w_inverse = 0.3
    
    # Note: We don't compute regret in this version, redistribute its weight
    # Redistribute the regret weight proportionally
    total_other = w_spectral + w_density + w_inverse
    if total_other > 0:
        w_spectral = 0.4 / total_other * (w_spectral + w_regret)
        w_density = 0.3 / total_other * (w_density + w_regret)
        w_inverse = 0.3 / total_other * (w_inverse + w_regret)
    
    # Weighted combination
    combined_scores = (
        w_spectral * spectral_scores +
        w_density * density_scores +
        w_inverse * inv_dist
    )
    
    # ==================== 5. DEGREE-CONSTRAINED SPARSIFICATION ====================
    def degree_constrained_sparsification(scores: np.ndarray) -> np.ndarray:
        """Intelligent sparsification preserving connectivity"""
        sparsified = np.zeros_like(scores)
        
        # Always include MST edges for connectivity
        sparsified[mst_edges] = scores[mst_edges]
        
        # Determine degree limits based on problem size
        if n <= 10:
            max_degree = 4
        elif n <= 50:
            max_degree = min(5, n // 4)
        else:
            max_degree = min(6, n // 5)
        
        # Track current degree for each node
        current_degree = np.sum(sparsified > 0, axis=1)
        
        # Collect candidate edges (excluding MST edges)
        candidates = []
        for i in range(n):
            for j in range(i + 1, n):
                if not mst_edges[i, j] and scores[i, j] > 0:
                    candidates.append((scores[i, j], i, j))
        
        # Sort candidates by score (descending)
        candidates.sort(reverse=True, key=lambda x: x[0])
        
        # Greedy selection respecting degree constraints
        for score, i, j in candidates:
            if current_degree[i] < max_degree and current_degree[j] < max_degree:
                sparsified[i, j] = score
                sparsified[j, i] = score
                current_degree[i] += 1
                current_degree[j] += 1
        
        return sparsified
    
    sparsified_scores = degree_constrained_sparsification(combined_scores)
    
    # ==================== 6. FINAL ENHANCEMENTS ====================
    # Ensure symmetry
    symmetric_scores = (sparsified_scores + sparsified_scores.T) / 2
    
    # Apply distance-based attenuation (long edges are less promising)
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        dist_factor = np.exp(-distance_matrix / (0.3 * max_dist + 1e-8))
        symmetric_scores *= dist_factor
    
    # Apply MST boost (critical edges get extra weight)
    mst_boost = 1.0 + 0.2 * mst_edges  # 20% boost for MST edges
    symmetric_scores *= mst_boost
    
    # Normalize to [0, 1]
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    np.fill_diagonal(symmetric_scores, 0)
    
    return symmetric_scores
