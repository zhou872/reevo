    np.fill_diagonal(inv_dist, 0)
    
    # 1. Edge savings heuristic (like in Christofides algorithm)
    # For edge (i,j): savings = d(i,0) + d(0,j) - d(i,j)
    # Using node 0 as reference, but could use any node
    savings = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Find a third node k that's different from i and j
                k = (i + 1) % n if (i + 1) % n != j else (i + 2) % n
                savings[i, j] = (distance_matrix[i, k] + 
                               distance_matrix[k, j] - 
                               distance_matrix[i, j])
    
    # Normalize savings
    savings_norm = np.zeros_like(savings)
    if np.any(savings > 0):

            np.max(savings[savings > 0]) - np.min(savings[savings > 0]) + 1e-10)
    
    # 2. Local cluster density bonus
    # For each node, compute average distance to its k nearest neighbors
    k = min(5, n - 1)
    density_bonus = np.zeros((n, n))
    
    for i in range(n):

        distances_i[i] = np.inf
        nearest_indices = np.argpartition(distances_i, k)[:k]

            for j in range(i + 1, n):

                    # Strong bonus for MST edges
                    mst_bonus[i, j] = mst_bonus[j, i] = 2.0
    except:
        pass  # If MST fails, continue without it
