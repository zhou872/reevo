import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Next-generation TSP edge heuristic with automated parameter optimization,
    efficient multi-step lookahead, and hybrid scoring mechanisms.
    
    Key innovations over v1:
    1. Automated parameter learning through statistical moment analysis
    2. Efficient lookahead via neighbor pruning and memoization
    3. Hybrid scoring combining additive and geometric integration
    4. Sparse connectivity reinforcement with degree balancing
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric distance matrix of shape (n, n)
        
    Returns:
    --------
    np.ndarray
        Heuristic scores matrix of same shape, higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    epsilon = 1e-12
    np.fill_diagonal(distance_matrix, np.inf)
    
    # ===== 1. AUTOMATED PARAMETER LEARNING =====
    # Analyze distance distribution moments for parameter optimization
    flat_dists = distance_matrix[np.triu_indices(n, k=1)]
    dist_stats = {
        'mean': np.mean(flat_dists),
        'std': np.std(flat_dists),
        'skew': 0,
        'kurtosis': 0
    }
    
    if len(flat_dists) > 3:
        # Higher moment analysis for sophisticated parameter tuning
        centered = flat_dists - dist_stats['mean']
        var = dist_stats['std'] ** 2
        if var > epsilon:
            dist_stats['skew'] = np.mean(centered ** 3) / (var ** 1.5)
            dist_stats['kurtosis'] = np.mean(centered ** 4) / (var ** 2) - 3
    
    # Parameter optimization based on statistical moments
    # Regularity: low std/mean ratio and moderate skew/kurtosis
    # Clustering: moderate-high std/mean, negative skew, positive kurtosis
    # Uniform: very low skew and kurtosis
    cv = dist_stats['std'] / (dist_stats['mean'] + epsilon)
    
    if cv < 0.3 and abs(dist_stats['skew']) < 0.5 and abs(dist_stats['kurtosis']) < 1:
        # Uniform/regular problem
        lookahead_factor = 0.7
        cluster_sensitivity = 0.3
        regret_exponent = 1.8
        hybrid_alpha = 0.6  # Weight for additive vs geometric combination
    elif cv > 0.6 and dist_stats['skew'] < -0.5 and dist_stats['kurtosis'] > 1:
        # Strongly clustered problem
        lookahead_factor = 0.9
        cluster_sensitivity = 0.8
        regret_exponent = 2.5
        hybrid_alpha = 0.3
    else:
        # Mixed problem
        lookahead_factor = 0.8
        cluster_sensitivity = 0.5
        regret_exponent = 2.0
        hybrid_alpha = 0.5
    
    # ===== 2. EFFICIENT LOOKAHEAD PREPROCESSING =====
    # Precompute nearest neighbors for efficient lookahead
    k_neighbors = min(8, n - 1) if n > 10 else n - 1
    nearest_idx = np.zeros((n, k_neighbors), dtype=int)
    nearest_dists = np.zeros((n, k_neighbors))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if k_neighbors > 0:
            # Get k nearest neighbors
            idx = np.argpartition(dists, k_neighbors)[:k_neighbors]
            sorted_order = np.argsort(dists[idx])
            nearest_idx[i] = idx[sorted_order]
            nearest_dists[i] = dists[idx[sorted_order]]
    
    # ===== 3. HYBRID BASE SCORING =====
    # Combine multiple scoring strategies
    base_scores = np.zeros_like(distance_matrix)
    
    # Strategy 1: Inverse distance with adaptive power
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    
    # Strategy 2: Rank-based scoring
    rank_scores = np.zeros_like(distance_matrix)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        ranks = np.argsort(dists)
        # Convert ranks to scores: best rank gets 1.0, worst gets epsilon
        for rank, j in enumerate(ranks):
            if i != j:
                # Exponential decay with rank
                rank_scores[i, j] = np.exp(-rank / max(1, (n-2)/4))
    
    # Strategy 3: MST-based connectivity scoring
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_connectivity = np.zeros_like(distance_matrix)
    mst_connectivity[mst_mask] = 1.0
    
    # Combine strategies with learned weights
    base_weights = [0.5, 0.3, 0.2]  # Inverse, Rank, MST
    if cv > 0.7:  # More emphasis on ranks for clustered problems
        base_weights = [0.4, 0.4, 0.2]
    
    inv_dist_norm = inv_dist / (np.max(inv_dist) + epsilon)
    rank_scores_norm = rank_scores / (np.max(rank_scores) + epsilon)
    
    base_scores = (base_weights[0] * inv_dist_norm +
                   base_weights[1] * rank_scores_norm +
                   base_weights[2] * mst_connectivity)
    
    # ===== 4. EFFICIENT MULTI-STEP LOOKAHEAD =====
    lookahead_scores = np.ones_like(distance_matrix)
    
    if n >= 4:
        # Use precomputed neighbors for efficient lookahead
        for i in range(n):
            for j in range(i + 1, n):
                direct = distance_matrix[i, j]
                
                # Two-step lookahead through nearest neighbors
                alt_costs = []
                for k in nearest_idx[i]:
                    if k != j and distance_matrix[i, k] < np.inf and distance_matrix[k, j] < np.inf:
                        alt_costs.append(distance_matrix[i, k] + distance_matrix[k, j] - direct)
                
                for k in nearest_idx[j]:
                    if k != i and distance_matrix[i, k] < np.inf and distance_matrix[k, j] < np.inf:
                        alt_costs.append(distance_matrix[i, k] + distance_matrix[k, j] - direct)
                
                if alt_costs:
                    avg_detour = np.mean(alt_costs)
                    if avg_detour > 0:
                        # Normalize detour cost relative to direct distance
                        detour_ratio = avg_detour / (direct + epsilon)
                        lookahead_scores[i, j] = lookahead_scores[j, i] = np.tanh(detour_ratio * lookahead_factor)
    
    # ===== 5. ADAPTIVE CLUSTER DETECTION =====
    cluster_scores = np.ones_like(distance_matrix)
    
    if n >= 6:
        # Use MST with adaptive threshold for cluster detection
        mst_edges = []
        for i in range(n):
            for j in range(i + 1, n):
                if mst_mask[i, j]:
                    mst_edges.append(distance_matrix[i, j])
        
        if mst_edges:
            cluster_threshold = np.percentile(mst_edges, 70)  # Top 30% longest MST edges
            
            # Identify clusters by removing long MST edges
            cluster_id = np.zeros(n, dtype=int) - 1
            current_cluster = 0
            
            for i in range(n):
                if cluster_id[i] == -1:
                    # Start new cluster with BFS
                    stack = [i]
                    cluster_id[i] = current_cluster
                    
                    while stack:
                        node = stack.pop()
                        for neighbor in np.where(mst_mask[node])[0]:
                            if (cluster_id[neighbor] == -1 and 
                                distance_matrix[node, neighbor] <= cluster_threshold):
                                cluster_id[neighbor] = current_cluster
                                stack.append(neighbor)
                    
                    current_cluster += 1
            
            # Apply cluster-based scoring
            for i in range(n):
                for j in range(i + 1, n):
                    if cluster_id[i] == cluster_id[j]:
                        # Intra-cluster: boost based on cluster sensitivity
                        intra_boost = 1.0 + cluster_sensitivity * (1.0 - distance_matrix[i, j] / cluster_threshold)
                        cluster_scores[i, j] = cluster_scores[j, i] = intra_boost
                    else:
                        # Inter-cluster: moderate penalty
                        cluster_scores[i, j] = cluster_scores[j, i] = 0.7
    
    # ===== 6. ENHANCED REGRET MODELING =====
    regret_scores = np.ones_like(distance_matrix)
    
    # Compute node-specific regret
    node_min_dists = np.zeros(n)
    for i in range(n):
        valid_dists = distance_matrix[i, distance_matrix[i] < np.inf]
        if len(valid_dists) > 0:
            node_min_dists[i] = np.min(valid_dists)
        else:
            node_min_dists[i] = dist_stats['mean']
    
    for i in range(n):
        for j in range(i + 1, n):
            # Local regret: how much worse than best alternative
            local_regret = distance_matrix[i, j] / (node_min_dists[i] + node_min_dists[j] + epsilon)
            
            # Global regret: alternative through best neighbor
            global_regret = 1.0
            if nearest_dists[i, 0] < np.inf and nearest_dists[j, 0] < np.inf:
                alt_path = (nearest_dists[i, 0] + nearest_dists[j, 0] + 
                           distance_matrix[nearest_idx[i, 0], nearest_idx[j, 0]])
                global_regret = distance_matrix[i, j] / (alt_path + epsilon)
            
            # Combine local and global regret
            combined_regret = (local_regret + global_regret) / 2
            regret_scores[i, j] = regret_scores[j, i] = np.exp(-combined_regret ** regret_exponent)
    
    # ===== 7. HYBRID SCORING INTEGRATION =====
    # Normalize all components
    components = {
        'base': base_scores,
        'lookahead': lookahead_scores,
        'cluster': cluster_scores,
        'regret': regret_scores
    }
    
    for key in components:
        max_val = np.max(components[key])
        if max_val > 0:
            components[key] = components[key] / max_val
    
    # Hybrid combination: weighted geometric mean + additive component
    geometric_part = np.ones_like(distance_matrix)
    weights = [0.3, 0.25, 0.25, 0.2]  # Base, Lookahead, Cluster, Regret
    
    for idx, (key, weight) in enumerate(zip(components.keys(), weights)):
        geometric_part *= (components[key] + epsilon) ** weight
    
    # Additive combination for complementary information
    additive_part = np.zeros_like(distance_matrix)
    for idx, (key, weight) in enumerate(zip(components.keys(), weights)):
        additive_part += weight * components[key]
    
    # Hybrid combination parameterized by hybrid_alpha
    combined_scores = (hybrid_alpha * geometric_part + 
                      (1 - hybrid_alpha) * additive_part)
    
    # ===== 8. SPARSE CONNECTIVITY REINFORCEMENT =====
    # Ensure MST connectivity with degree balancing
    final_scores = combined_scores.copy()
    
    # Boost MST edges
    final_scores[mst_mask] *= 1.5
    
    # Degree balancing: penalize nodes with too many high-scoring edges
    node_scores = np.zeros(n)
    for i in range(n):
        row_scores = final_scores[i].copy()
        row_scores[i] = 0
        node_scores[i] = np.sum(row_scores > np.percentile(row_scores, 70))
    
    if np.max(node_scores) > 0:
        node_scores_norm = node_scores / np.max(node_scores)
        for i in range(n):
            if node_scores_norm[i] > 0.7:  # Over-connected node
                penalty = 1.0 - 0.3 * node_scores_norm[i]
                final_scores[i] *= penalty
    
    # Adaptive sparsification
    sparsified = np.zeros_like(final_scores)
    for i in range(n):
        row_scores = final_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Always keep MST edges
        keep_mask = mst_mask[i].copy()
        
        # Determine number of additional edges to keep
        base_k = min(6, max(3, int(np.log2(n) * 2)))
        if n <= 15:
            base_k = min(8, n - 1)
        
        mst_count = np.sum(keep_mask)
        needed = max(0, base_k - mst_count)
        
        if needed > 0:
            # Select top non-MST edges
            non_mst_scores = row_scores.copy()
            non_mst_scores[keep_mask] = -np.inf
            
            valid_scores = non_mst_scores[non_mst_scores > -np.inf]
            if len(valid_scores) > 0:
                threshold = np.partition(valid_scores, -min(needed, len(valid_scores)))[-min(needed, len(valid_scores))]
                top_edges = non_mst_scores >= threshold
                keep_mask = keep_mask | top_edges
        
        sparsified[i, keep_mask] = final_scores[i, keep_mask]
    
    # ===== 9. FINAL PROCESSING =====
    # Symmetrize
    symmetric_scores = np.maximum(sparsified, sparsified.T)
    
    # Final normalization
    max_score = np.max(symmetric_scores)
    if max_score > 0:
        symmetric_scores = symmetric_scores / max_score
    
    # Apply adaptive smoothing
    if cv < 0.4:
        symmetric_scores = np.power(symmetric_scores, 0.9)
    elif cv > 0.7:
        symmetric_scores = np.power(symmetric_scores, 0.7)
    else:
        symmetric_scores = np.power(symmetric_scores, 0.8)
    
    np.fill_diagonal(symmetric_scores, 0)
    np.fill_diagonal(distance_matrix, 0)
    
    return symmetric_scores
