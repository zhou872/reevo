import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with symmetric regret computation.
    Combines multiple factors to prioritize promising edges for TSP solutions.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix) if n == 1 else np.array([])
    
    # Work on a copy to avoid modifying input
    dist = distance_matrix.copy()
    
    # Set diagonal to infinity to avoid self-edges
    np.fill_diagonal(dist, np.inf)
    
    # 1. Base inverse distance weighting
    epsilon = 1e-10
    inv_dist = 1 / (dist + epsilon)
    
    # 2. MST edge bonus multiplier
    # MST edges get higher multiplier, others get baseline
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    mst_bonus = np.ones_like(dist)
    mst_bonus[mst_mask] = 2.0  # MST edges get 2x bonus
    
    # 3. Symmetric regret penalization
    # Compute regret from both node perspectives and average
    regret_penalty = np.ones_like(dist)
    
    # Pre-compute best and second-best for each node
    best_dists = np.zeros(n)
    second_best_dists = np.zeros(n)
    
    for i in range(n):
        # Get sorted distances for node i
        sorted_dists = np.sort(dist[i])
        best_dists[i] = sorted_dists[0]  # Best (smallest) distance
        second_best_dists[i] = sorted_dists[1]  # Second best distance
    
    # Compute symmetric regret for each edge
    for i in range(n):
        for j in range(i+1, n):  # Only compute upper triangle
            if dist[i, j] < np.inf:
                # Regret from i's perspective
                regret_i = (dist[i, j] - best_dists[i]) / \
                          (second_best_dists[i] - best_dists[i] + epsilon)
                
                # Regret from j's perspective
                regret_j = (dist[j, i] - best_dists[j]) / \
                          (second_best_dists[j] - best_dists[j] + epsilon)
                
                # Average regret and convert to penalty (0-1 range)
                avg_regret = (regret_i + regret_j) / 2
                penalty = np.exp(-avg_regret)
                
                # Apply symmetrically
                regret_penalty[i, j] = penalty
                regret_penalty[j, i] = penalty
    
    # 4. Combine factors
    scores = inv_dist * mst_bonus * regret_penalty
    
    # 5. Adaptive sparsification - keep most promising edges
    # Determine k based on graph size (more edges for larger graphs)
    k = max(2, min(n // 2, 10))  # Keep between 2 and 10 edges per node
    
    # Create sparsification mask
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        # Get top k edges for this node (excluding self)
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf  # Exclude diagonal
        
        # Get indices of top k scores
        if n > k:
            # Use argpartition for efficiency
            top_k_idx = np.argpartition(row_scores, -k)[-k:]
        else:
            # For very small graphs, keep all edges
            top_k_idx = [j for j in range(n) if j != i]
        
        mask[i, top_k_idx] = True
        # Ensure symmetry
        for idx in top_k_idx:
            mask[idx, i] = True
    
    # Ensure MST edges are always included for connectivity
    mask = mask | mst_mask
    
    # Apply sparsification
    scores = scores * mask
    
    # 6. Normalize scores to [0, 1] range
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    # Zero out diagonal
    np.fill_diagonal(scores, 0)
    
    return scores
