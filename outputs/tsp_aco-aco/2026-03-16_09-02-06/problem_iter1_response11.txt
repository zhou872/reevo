import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    Returns a matrix of the same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # Base inverse distance (avoid division by zero)
    epsilon = 1e-10
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # Factor 1: Local neighbor importance
    # Calculate average distance to k nearest neighbors for each node
    k_neighbors = min(5, n-1)
    neighbor_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        indices = np.argsort(distance_matrix[i])[1:k_neighbors+1]  # Exclude self
        neighbor_mask[i, indices] = True
        neighbor_mask[indices, i] = True  # Symmetric
    
    # Factor 2: Minimum Spanning Tree inclusion
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_binary = (mst > 0) | (mst.T > 0)  # Make symmetric
    
    # Factor 3: Savings heuristic (Clarke-Wright)
    # For each edge (i,j), savings = d(i,0) + d(0,j) - d(i,j)
    # Use node 0 as depot, then average over all possible depots
    savings = np.zeros_like(distance_matrix)
    for depot in range(min(3, n)):  # Try first 3 nodes as depot
        for i in range(n):
            for j in range(i+1, n):
                if i != depot and j != depot:
                    s = (distance_matrix[i, depot] + 
                         distance_matrix[depot, j] - 
                         distance_matrix[i, j])
                    savings[i, j] += max(s, 0)
                    savings[j, i] += max(s, 0)
    savings = savings / min(3, n)  # Average over depots
    
    # Normalize factors to [0,1]
    def normalize(arr):
        arr_min, arr_max = arr.min(), arr.max()
        if arr_max - arr_min > epsilon:
            return (arr - arr_min) / (arr_max - arr_min)
        return np.zeros_like(arr)
    
    inv_dist_norm = normalize(inv_dist)
    savings_norm = normalize(savings)
    
    # Combine factors with weights
    heuristics = (
        0.4 * inv_dist_norm +          # Base inverse distance
        0.3 * savings_norm +           # Savings heuristic
        0.2 * neighbor_mask.astype(float) +  # Neighbor importance
        0.1 * mst_binary.astype(float)      # MST inclusion
    )
    
    # Sparsification: keep only promising edges
    # Strategy 1: Keep edges that are in top percentiles for both endpoints
    percentile = 75
    thresholds = np.percentile(heuristics, percentile, axis=1)
    mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        # Keep edges above threshold for node i
        mask[i] |= heuristics[i] > thresholds[i]
        
        # Also keep edges that are strong for the other endpoint
        for j in range(n):
            if i != j and heuristics[j, i] > thresholds[j]:
                mask[i, j] = True
    
    # Strategy 2: Enforce degree constraints
    # Ensure each node has at least min_edges and at most max_edges connections
    min_edges, max_edges = 2, 4
    
    for i in range(n):
        row_heuristics = heuristics[i].copy()
        row_heuristics[i] = -np.inf  # Exclude self
        
        # Keep at least min_edges strongest connections
        strong_indices = np.argsort(row_heuristics)[-min_edges:]
        mask[i, strong_indices] = True
        
        # Limit to at most max_edges
        if np.sum(mask[i]) > max_edges:
            # Keep only the top max_edges
            values = heuristics[i].copy()
            values[~mask[i]] = -np.inf
            top_indices = np.argsort(values)[-max_edges:]
            new_mask = np.zeros(n, dtype=bool)
            new_mask[top_indices] = True
            mask[i] = new_mask
    
    # Apply sparsification mask
    heuristics = heuristics * mask.astype(float)
    
    # Final cleanup
    np.fill_diagonal(heuristics, 0)
    
    # Ensure symmetry (undirected graph)
    heuristics = np.maximum(heuristics, heuristics.T)
    
    return heuristics
