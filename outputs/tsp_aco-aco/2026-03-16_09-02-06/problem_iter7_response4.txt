import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with improved symmetric regret
    and adaptive sparsification strategies.
    
    Features:
    1. Symmetric Regret Penalization: Computes regret from both node perspectives
    2. MST Connectivity Bonus: Boosts edges in minimum spanning tree
    3. Inverse Distance Weighting: Prioritizes shorter edges
    4. Adaptive Edge Sparsification: Dynamic edge preservation based on graph size
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create a working copy with diagonal set to infinity
    dmat = distance_matrix.copy().astype(float)
    np.fill_diagonal(dmat, np.inf)
    
    # 1. Base inverse distance weighting
    inv_dist = 1 / (dmat + 1e-10)
    
    # 2. MST connectivity bonus
    mst = minimum_spanning_tree(dmat)
    mst_mask = (mst.toarray() > 0)
    mst_bonus = np.ones_like(dmat)
    mst_bonus[mst_mask] = 2.0  # Double the score for MST edges
    
    # 3. Improved symmetric regret penalization
    regret_matrix = np.ones_like(dmat)
    
    # Precompute best and second-best distances for each node
    best_dists = np.zeros(n)
    second_best_dists = np.zeros(n)
    
    for i in range(n):
        # Get distances from node i
        dists_i = dmat[i].copy()
        
        # Find two smallest distances (excluding infinity for diagonal)
        sorted_vals = np.sort(dists_i[dists_i < np.inf])
        if len(sorted_vals) >= 2:
            best_dists[i] = sorted_vals[0]
            second_best_dists[i] = sorted_vals[1]
        elif len(sorted_vals) == 1:
            best_dists[i] = sorted_vals[0]
            second_best_dists[i] = sorted_vals[0] * 1.1  # Small buffer for single neighbor
    
    # Compute symmetric regret for each edge
    for i in range(n):
        for j in range(i+1, n):  # Only upper triangle for symmetric matrix
            if dmat[i, j] == np.inf:
                continue
                
            # Regret from i's perspective
            if second_best_dists[i] > best_dists[i]:
                regret_i = (dmat[i, j] - best_dists[i]) / (second_best_dists[i] - best_dists[i])
            else:
                regret_i = 0
                
            # Regret from j's perspective
            if second_best_dists[j] > best_dists[j]:
                regret_j = (dmat[j, i] - best_dists[j]) / (second_best_dists[j] - best_dists[j])
            else:
                regret_j = 0
            
            # Symmetric average with exponential decay
            symmetric_regret = (regret_i + regret_j) / 2
            penalty = np.exp(-symmetric_regret)
            regret_matrix[i, j] = penalty
            regret_matrix[j, i] = penalty
    
    # 4. Combine all factors
    scores = inv_dist * mst_bonus * regret_matrix
    
    # 5. Adaptive edge sparsification
    # Determine number of edges to preserve per node
    if n <= 3:
        k = n - 1  # Keep all edges for very small graphs
    else:
        # Logarithmic scaling: preserve more edges for larger graphs
        k = max(2, int(np.log2(n) * 1.5))
    
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        # Get scores for this node, excluding self
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        # Keep top k edges
        if np.any(row_scores > 0):  # Only if there are positive scores
            top_k_idx = np.argpartition(row_scores, -k)[-k:]
            mask[i, top_k_idx] = True
            # Ensure symmetry
            mask[top_k_idx, i] = True
    
    # Always include MST edges for connectivity
    mask = mask | mst_mask
    
    # Apply sparsification
    scores = scores * mask
    
    # Normalize to [0, 1] range
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    # Zero out diagonal and ensure symmetry
    np.fill_diagonal(scores, 0)
    
    return scores
