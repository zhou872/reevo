import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import squareform
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with global graph metrics and
    dynamic factor balancing.
    
    Features:
    1. Symmetric regret with local clustering coefficient awareness
    2. Betweenness centrality weighting for critical nodes
    3. Dynamic MST importance scaling based on graph diameter
    4. Adaptive factor weights based on problem characteristics
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape, with higher values indicating more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        if n == 0:
            return np.array([])
        elif n == 1:
            return np.zeros((1, 1))
        else:  # n == 2
            scores = np.ones((2, 2))
            np.fill_diagonal(scores, 0)
            return scores
    
    # Create a copy with infinite diagonal
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # Calculate problem characteristics for dynamic weighting
    graph_density = n / (n * (n - 1) / 2)  # Always 1 for complete graphs, but kept for generality
    diameter = np.max(dist[dist < np.inf])
    avg_distance = np.mean(dist[dist < np.inf])
    
    # Dynamic weighting parameters
    if n < 10:
        # Small instances: emphasize regret and centrality
        w_regret = 1.2
        w_centrality = 1.1
        w_mst = 0.9
        w_cluster = 0.8
    elif n < 50:
        # Medium instances: balanced approach
        w_regret = 1.0
        w_centrality = 1.0
        w_mst = 1.0
        w_cluster = 1.0
    else:
        # Large instances: emphasize MST and clustering
        w_regret = 0.9
        w_centrality = 0.8
        w_mst = 1.2
        w_cluster = 1.1
    
    # Adjust weights based on problem structure
    if diameter / avg_distance > 5:
        # Highly elongated instance: emphasize MST for backbone
        w_mst *= 1.3
    
    # 1. Base inverse distance with log scaling
    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        inv_dist = np.log1p(1 / (dist + 1e-10))
        inv_dist[dist == np.inf] = 0
    
    # 2. MST connectivity with dynamic importance
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    
    # Scale MST importance based on relative edge length
    mst_bonus = np.ones_like(dist)
    if mst_mask.any():
        mst_edges = dist[mst_mask]
        if mst_edges.size > 1:
            # Normalize MST edge lengths
            mst_norm = mst_edges / mst_edges.max()
            # Give higher bonus to shorter MST edges
            mst_scores = 1.0 + (1.0 - mst_norm) * w_mst
            mst_bonus[mst_mask] = mst_scores
    
    # 3. Enhanced symmetric regret with clustering awareness
    regret_matrix = np.ones_like(dist)
    
    # Precompute clustering coefficients (local density measure)
    clustering_coeff = np.zeros(n)
    k = min(5, n - 1)
    
    for i in range(n):
        # Get distances to k nearest neighbors
        distances = dist[i].copy()
        distances[i] = np.inf
        if k > 0:
            knn_distances = np.partition(distances, k)[:k]
            # Clustering coefficient based on distance variance
            if k > 1 and np.std(knn_distances) > 0:
                clustering_coeff[i] = 1.0 / (1.0 + np.std(knn_distances) / np.mean(knn_distances))
            else:
                clustering_coeff[i] = 0.5
    
    # Compute symmetric regret
    for i in range(n):
        for j in range(i + 1, n):
            if i == j:
                continue
            
            # Regret calculation from both perspectives
            d_ij = dist[i, j]
            
            # Get alternative distances for i (excluding j)
            alts_i = dist[i].copy()
            alts_i[j] = np.inf
            alts_i[i] = np.inf
            best_i = np.min(alts_i[alts_i < np.inf])
            
            # Get alternative distances for j (excluding i)
            alts_j = dist[j].copy()
            alts_j[i] = np.inf
            alts_j[j] = np.inf
            best_j = np.min(alts_j[alts_j < np.inf])
            
            # Calculate regret scores
            if best_i > 0 and best_j > 0:
                regret_i = max(0, (d_ij - best_i) / (d_ij + best_i + 1e-10))
                regret_j = max(0, (d_ij - best_j) / (d_ij + best_j + 1e-10))
                
                # Combine with clustering awareness
                cluster_factor = 1.0 - abs(clustering_coeff[i] - clustering_coeff[j])
                combined_regret = (regret_i + regret_j) / 2 * cluster_factor
                
                # Apply exponential penalty with dynamic weight
                penalty = np.exp(-w_regret * combined_regret)
                regret_matrix[i, j] = regret_matrix[j, i] = penalty
    
    # 4. Betweenness centrality approximation
    centrality_bonus = np.ones_like(dist)
    
    # Approximate centrality using farness (inverse of closeness)
    farness = np.zeros(n)
    for i in range(n):
        valid_dists = dist[i][dist[i] < np.inf]
        if len(valid_dists) > 0:
            farness[i] = np.sum(valid_dists)
    
    if np.max(farness) > 0:
        # Nodes with high farness are more "central" in the sense they're hard to reach
        # We want to connect them earlier
        norm_farness = farness / np.max(farness)
        
        for i in range(n):
            for j in range(i + 1, n):
                if i == j:
                    continue
                
                # Bonus for connecting two central nodes
                centrality_score = 0.5 + (norm_farness[i] + norm_farness[j]) / 2
                centrality_bonus[i, j] = centrality_bonus[j, i] = (
                    1.0 + w_centrality * (centrality_score - 0.5)
                )
    
    # 5. Local clustering bonus
    cluster_bonus = np.ones_like(dist)
    
    for i in range(n):
        for j in range(i + 1, n):
            if i == j:
                continue
            
            # Bonus for edges within dense clusters
            avg_cluster = (clustering_coeff[i] + clustering_coeff[j]) / 2
            # Also consider if they share common neighbors
            common_neighbor_bonus = 0
            
            # Check if i and j have common close neighbors
            if k > 0:
                # Get k-nearest neighbors for both
                neighbors_i = np.argsort(dist[i])[:k]
                neighbors_j = np.argsort(dist[j])[:k]
                common = len(set(neighbors_i) & set(neighbors_j))
                common_neighbor_bonus = common / k
            
            cluster_bonus[i, j] = cluster_bonus[j, i] = (
                1.0 + w_cluster * (avg_cluster + common_neighbor_bonus) / 2
            )
    
    # 6. Combine all factors
    scores = (
        inv_dist ** 0.7 *          # Base attraction
        mst_bonus *                # Structural importance
        regret_matrix *            # Opportunity cost
        centrality_bonus ** 0.8 *  # Critical nodes
        cluster_bonus              # Local structure
    )
    
    # 7. Adaptive sparsification with centrality awareness
    base_k = max(2, int(np.sqrt(n) + 0.5))
    
    # Adjust k based on centrality distribution
    if np.std(norm_farness) > 0.3:
        # Diverse centrality: keep more edges
        base_k = min(base_k + 1, n // 2)
    
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        # Determine k for this node based on its centrality
        k_i = base_k
        if norm_farness[i] > 0.7:  # Very central node
            k_i = min(k_i + 2, n // 2)
        elif norm_farness[i] < 0.3:  # Peripheral node
            k_i = max(2, k_i - 1)
        
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        if k_i > 0 and k_i < n:
            # Keep top k_i edges
            top_indices = np.argpartition(row_scores, -k_i)[-k_i:]
            mask[i, top_indices] = True
            # Ensure symmetry
            for idx in top_indices:
                mask[idx, i] = True
    
    # Ensure MST connectivity and minimum degree
    mask = mask | mst_mask
    
    # Ensure each node has at least 2 connections
    for i in range(n):
        if np.sum(mask[i]) < 2:
            # Add best available edges
            row_scores = scores[i].copy()
            row_scores[mask[i]] = -np.inf
            row_scores[i] = -np.inf
            if np.any(row_scores > -np.inf):
                needed = 2 - np.sum(mask[i])
                top_indices = np.argsort(row_scores)[-needed:]
                mask[i, top_indices] = True
                for idx in top_indices:
                    mask[idx, i] = True
    
    # Apply sparsification
    scores = scores * mask
    
    # Final normalization
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    np.fill_diagonal(scores, 0)
    
    return scores
