import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.cluster.hierarchy import linkage, fcluster
from scipy.spatial.distance import squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with improved regret calculation,
    lightweight clustering, and adaptive sparsification.
    
    Key improvements over v1:
    1. Two-step regret calculation considering neighbor-of-neighbor costs
    2. Simple hierarchical clustering for cluster-aware bonuses
    3. Density-adaptive sparsification that preserves connectivity
    4. Symmetric regret penalty with improved normalization
    
    Returns a matrix of edge scores with same shape as input.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        scores = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(scores, 0)
        return scores
    
    # Create working distance matrix
    dist = distance_matrix.astype(float, copy=True)
    np.fill_diagonal(dist, np.inf)
    
    # 1. Base inverse distance with adaptive smoothing
    dist_mean = np.mean(dist[dist < np.inf])
    inv_dist = 1 / (dist + dist_mean * 0.1)
    
    # 2. MST inclusion bonus
    mst_input = dist.copy()
    np.fill_diagonal(mst_input, 0)
    mst = minimum_spanning_tree(mst_input)
    mst_mask = mst.toarray() > 0
    mst_bonus = np.where(mst_mask, 1.5, 1.0)  # MST edges get 50% bonus
    
    # 3. Enhanced symmetric regret with two-step lookahead
    regret_matrix = np.ones_like(dist)
    
    # Precompute nearest neighbors for each node
    k_neighbors = min(5, n - 1)
    neighbor_indices = np.argsort(dist, axis=1)[:, 1:k_neighbors+1]
    
    # Compute two-step regret for each edge (i,j)
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            # One-step regret (same as v1)
            row_i = dist[i].copy()
            row_i[i] = np.inf
            best_i = np.min(row_i)
            row_i[row_i == best_i] = np.inf
            second_best_i = np.min(row_i)
            regret_i = (dist[i, j] - best_i) / max(second_best_i - best_i, 1e-10)
            
            row_j = dist[j].copy()
            row_j[j] = np.inf
            best_j = np.min(row_j)
            row_j[row_j == best_j] = np.inf
            second_best_j = np.min(row_j)
            regret_j = (dist[j, i] - best_j) / max(second_best_j - best_j, 1e-10)
            
            # Two-step regret: consider neighbors of j (excluding i)
            j_neighbors = [nbr for nbr in neighbor_indices[j] if nbr != i]
            if len(j_neighbors) > 0:
                # Average cost from i to j's other good neighbors
                avg_to_j_neighbors = np.mean([dist[i, nbr] for nbr in j_neighbors])
                two_step_regret = (dist[i, j] + dist_mean) / (avg_to_j_neighbors + 1e-10)
            else:
                two_step_regret = 1.0
            
            # Combine regrets (weighted average favoring one-step)
            combined_regret = (regret_i + regret_j) / 2 * 0.7 + (two_step_regret - 1) * 0.3
            
            # Convert to penalty factor using exponential decay
            regret_matrix[i, j] = np.exp(-combined_regret)
    
    # 4. Simple clustering bonus using hierarchical clustering
    if n >= 4:
        try:
            # Use hierarchical clustering on distances
            condensed_dist = squareform(dist, checks=False)
            Z = linkage(condensed_dist, method='average')
            n_clusters = max(2, min(int(np.sqrt(n)), n // 2))
            clusters = fcluster(Z, n_clusters, criterion='maxclust') - 1
            
            # Create cluster bonus matrix
            cluster_bonus = np.ones_like(dist)
            for i in range(n):
                for j in range(i+1, n):
                    if clusters[i] == clusters[j]:
                        bonus = 1.2  # Intra-cluster edges get 20% bonus
                        cluster_bonus[i, j] = cluster_bonus[j, i] = bonus
        except:
            cluster_bonus = np.ones_like(dist)
    else:
        cluster_bonus = np.ones_like(dist)
    
    # 5. Combine all factors
    scores = inv_dist * mst_bonus * regret_matrix * cluster_bonus
    
    # 6. Adaptive sparsification with connectivity preservation
    # Base k = sqrt(n), adjusted by graph density
    dist_noninf = dist[dist < np.inf]
    if len(dist_noninf) > 0:
        density_factor = np.median(dist_noninf) / np.max(dist_noninf)
        k = max(2, int(np.sqrt(n) * (1.5 - density_factor)))  # More edges for sparse graphs
    else:
        k = max(2, int(np.sqrt(n)))
    
    # Create sparsification mask
    mask = np.zeros_like(scores, dtype=bool)
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        # Keep top k edges
        if np.any(row_scores > -np.inf):
            top_k = min(k, np.sum(row_scores > -np.inf))
            threshold = np.partition(row_scores, -top_k)[-top_k]
            mask[i, row_scores >= threshold] = True
    
    # Ensure symmetry and preserve MST edges
    mask = mask | mask.T | mst_mask
    
    # Apply mask
    scores = scores * mask
    
    # 7. Normalize to [0, 1] range while preserving zero diagonal
    np.fill_diagonal(scores, 0)
    valid_scores = scores[scores > 0]
    if len(valid_scores) > 1:
        min_score, max_score = np.min(valid_scores), np.max(valid_scores)
        if max_score > min_score:
            scores[scores > 0] = (scores[scores > 0] - min_score) / (max_score - min_score)
        else:
            scores[scores > 0] = 1.0
    elif len(valid_scores) == 1:
        scores[scores > 0] = 1.0
    
    return scores
