import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.cluster.hierarchy import linkage, fcluster
from scipy.spatial.distance import squareform
from typing import Tuple, Dict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Next-generation heuristic scoring for TSP with enhanced lookahead,
    meta-learning inspired parameter adaptation, and graph neural network
    inspired structural embeddings.
    
    Key Innovations:
    1. Adaptive 3-opt Lookahead Simulation: Performs lightweight 3-opt simulations
       to evaluate edge quality with deeper local search context
    2. Meta-heuristic Parameter Optimization: Dynamically tunes heuristic weights
       using problem-specific characteristics through a rule-based meta-optimizer
    3. Graph Neural Network-inspired Embedding: Computes structural embeddings
       via random walks and node similarity to capture complex dependencies
    4. Multi-resolution Cluster Analysis: Uses hierarchical clustering at multiple
       thresholds to identify nested structural patterns
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create working distance matrix with infinity diagonal
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # ===== 1. GRAPH CHARACTERIZATION & META-PARAMETER OPTIMIZATION =====
    graph_stats = _compute_graph_statistics(dist, n)
    
    # Meta-learning inspired parameter optimization
    params = _optimize_parameters(graph_stats, n)
    
    # ===== 2. MULTI-RESOLUTION CLUSTER ANALYSIS =====
    cluster_hierarchy = _multi_resolution_clustering(dist, n, graph_stats)
    
    # ===== 3. GRAPH NEURAL NETWORK-INSPIRED EMBEDDINGS =====
    # Compute structural embeddings via random walks
    embeddings = _compute_structural_embeddings(dist, n, params['embedding_dim'])
    
    # ===== 4. BASE COMPONENTS WITH EMBEDDING-AWARE ADJUSTMENTS =====
    # Base inverse distance with adaptive exponent
    inv_power = params['inv_power']
    inv_dist = 1 / (dist + 1e-12) ** inv_power
    
    # Enhanced regret calculation with embedding similarity
    regret_penalty = _compute_embedding_aware_regret(dist, n, embeddings, cluster_hierarchy, params)
    
    # ===== 5. ADVANCED LOOKAHEAD: ADAPTIVE 3-OPT SIMULATION =====
    lookahead_scores = _adaptive_3opt_lookahead(dist, n, cluster_hierarchy, params, graph_stats)
    
    # ===== 6. STRUCTURAL CONNECTIVITY BONUSES =====
    mst_scores = _compute_structural_connectivity_bonus(dist, n, cluster_hierarchy, embeddings)
    
    # ===== 7. EMBEDDING SIMILARITY & MUTUAL AFFINITY =====
    affinity_scores = _compute_embedding_affinity(embeddings, n, params)
    
    # ===== 8. DYNAMIC COMPONENT FUSION =====
    # Use meta-optimized weights for component fusion
    scores = (
        inv_dist ** params['w_inv_dist'] *
        regret_penalty ** params['w_regret'] *
        lookahead_scores ** params['w_lookahead'] *
        mst_scores ** params['w_connectivity'] *
        affinity_scores ** params['w_affinity']
    )
    
    # ===== 9. INTELLIGENT SPARSIFICATION =====
    scores = _intelligent_sparsification(
        scores, dist, n, cluster_hierarchy, mst_scores > 1.0, params
    )
    
    # ===== 10. FINAL NORMALIZATION =====
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    # Apply soft thresholding for better discrimination
    scores = 1 / (1 + np.exp(-params['sigmoid_scale'] * (scores - params['sigmoid_threshold'])))
    
    np.fill_diagonal(scores, 0)
    return scores


# ===== HELPER FUNCTIONS =====

def _compute_graph_statistics(dist: np.ndarray, n: int) -> Dict:
    """Compute comprehensive graph statistics for parameter optimization."""
    stats = {}
    
    # Basic statistics
    valid_dists = dist[dist < np.inf]
    stats['avg_distance'] = np.mean(valid_dists) if len(valid_dists) > 0 else 0
    stats['std_distance'] = np.std(valid_dists) if len(valid_dists) > 0 else 0
    
    # Neighborhood statistics
    sorted_dists = np.sort(dist, axis=1)
    stats['avg_nearest'] = np.mean(sorted_dists[:, 1])
    stats['avg_second_nearest'] = np.mean(sorted_dists[:, 2]) if n > 2 else stats['avg_nearest']
    
    # Sparsity and regularity measures
    stats['sparsity'] = min(1.0, max(0.1, stats['avg_nearest'] / (stats['avg_distance'] + 1e-12)))
    
    # Symmetry check (TSP is symmetric, but we measure any asymmetry)
    asymmetry = np.max(np.abs(dist - dist.T) / (dist + 1e-12))
    stats['asymmetry'] = asymmetry if np.isfinite(asymmetry) else 0
    
    # Degree distribution statistics
    degrees = np.sum(dist < np.inf, axis=1)
    stats['degree_variation'] = np.std(degrees) / (np.mean(degrees) + 1e-12)
    
    return stats


def _optimize_parameters(stats: Dict, n: int) -> Dict:
    """Meta-heuristic parameter optimization based on graph characteristics."""
    params = {}
    
    # Base parameters with adaptive scaling
    sparsity = stats['sparsity']
    degree_var = stats['degree_variation']
    
    # Inverse distance exponent (higher for sparse graphs)
    params['inv_power'] = 1.0 + 0.8 * sparsity - 0.3 * min(1.0, degree_var)
    
    # Embedding dimension adapts to graph size
    params['embedding_dim'] = min(20, max(5, int(np.log2(n + 1) * 3)))
    
    # Lookahead intensity (more intensive for moderate-sized dense graphs)
    if n < 50:
        params['lookahead_intensity'] = 2.0
    elif sparsity < 0.4:
        params['lookahead_intensity'] = 1.5
    else:
        params['lookahead_intensity'] = 1.0
    
    # Component weights based on multi-objective optimization rules
    if n < 30:
        params['w_inv_dist'] = 0.4
        params['w_regret'] = 0.9
        params['w_lookahead'] = 1.4
        params['w_connectivity'] = 1.3
        params['w_affinity'] = 0.5
    elif sparsity < 0.3:
        params['w_inv_dist'] = 0.6
        params['w_regret'] = 1.1
        params['w_lookahead'] = 0.8
        params['w_connectivity'] = 1.2
        params['w_affinity'] = 0.7
    else:
        params['w_inv_dist'] = 0.8
        params['w_regret'] = 1.2
        params['w_lookahead'] = 1.0
        params['w_connectivity'] = 1.0
        params['w_affinity'] = 0.9
    
    # Sparsification parameters
    params['edges_per_node_base'] = max(2, int(np.sqrt(n) * (1.0 + 0.4 * sparsity)))
    params['min_cluster_connections'] = max(1, int(np.log2(n + 1)))
    
    # Normalization parameters
    params['sigmoid_scale'] = 6.0
    params['sigmoid_threshold'] = 0.4 + 0.2 * sparsity
    
    return params


def _multi_resolution_clustering(dist: np.ndarray, n: int, stats: Dict) -> Dict:
    """Perform hierarchical clustering at multiple resolutions."""
    if n < 4:
        return {'coarse': np.zeros(n, dtype=int), 'medium': np.zeros(n, dtype=int), 'fine': np.zeros(n, dtype=int)}
    
    # Convert to condensed form for clustering
    condensed_dist = squareform(dist, checks=False)
    Z = linkage(condensed_dist, method='average')
    
    # Three resolution levels
    thresholds = [
        stats['avg_distance'] * 1.5,  # Coarse
        stats['avg_distance'] * 0.8,  # Medium
        stats['avg_distance'] * 0.4   # Fine
    ]
    
    clusters = {}
    clusters['coarse'] = fcluster(Z, thresholds[0], criterion='distance')
    clusters['medium'] = fcluster(Z, thresholds[1], criterion='distance')
    clusters['fine'] = fcluster(Z, thresholds[2], criterion='distance')
    
    return clusters


def _compute_structural_embeddings(dist: np.ndarray, n: int, dim: int) -> np.ndarray:
    """Compute node embeddings via random walks (GNN-inspired)."""
    embeddings = np.zeros((n, dim))
    
    if n < 3:
        return embeddings
    
    # Transition matrix with teleportation (PageRank-like)
    epsilon = 1e-12
    row_sums = np.sum(dist, axis=1, keepdims=True)
    transition = (1 / (dist + epsilon)) / (row_sums + epsilon)
    np.fill_diagonal(transition, 0)
    
    # Add self-loops and normalize
    transition = 0.8 * transition + 0.2 * np.eye(n)
    transition = transition / np.sum(transition, axis=1, keepdims=True)
    
    # Random walk propagation for embeddings
    for d in range(dim):
        # Initialize with random or degree-based weights
        if d == 0:
            embeddings[:, d] = np.sum(dist < np.inf, axis=1)  # Degree
        elif d == 1:
            embeddings[:, d] = np.mean(dist, axis=1)  # Average distance
        else:
            # Random walk propagation for higher dimensions
            walk_steps = min(5, n // 2)
            current = np.random.randn(n)
            for _ in range(walk_steps):
                current = transition @ current
            embeddings[:, d] = current
    
    # Normalize embeddings
    norms = np.linalg.norm(embeddings, axis=1, keepdims=True)
    embeddings = embeddings / (norms + 1e-12)
    
    return embeddings


def _compute_embedding_aware_regret(dist: np.ndarray, n: int, embeddings: np.ndarray, 
                                   clusters: Dict, params: Dict) -> np.ndarray:
    """Compute regret penalty enhanced with embedding similarity."""
    regret_penalty = np.ones_like(dist)
    
    # Precompute local statistics
    sorted_dists = np.sort(dist, axis=1)
    second_best = sorted_dists[:, 1]
    local_avgs = np.array([np.mean(row[row < np.inf]) for row in dist])
    
    # Compute embedding similarity matrix
    embedding_sim = embeddings @ embeddings.T
    np.fill_diagonal(embedding_sim, 0)
    
    for i in range(n):
        for j in range(i + 1, n):
            # Base regret calculation
            regret_i = max(0, dist[i, j] - second_best[i])
            regret_j = max(0, dist[i, j] - second_best[j])
            
            # Normalize by local averages
            norm_regret_i = regret_i / (local_avgs[i] + 1e-12)
            norm_regret_j = regret_j / (local_avgs[j] + 1e-12)
            
            # Combine with embedding similarity
            sim_factor = 1.0 + 0.5 * embedding_sim[i, j]
            combined_regret = (norm_regret_i + norm_regret_j) / (2 * sim_factor)
            
            # Cluster-aware adjustment
            cluster_bonus = 1.0
            if clusters['coarse'][i] != clusters['coarse'][j]:
                # Inter-cluster edges: reduce penalty
                cluster_bonus = 1.3
            elif clusters['fine'][i] != clusters['fine'][j]:
                # Different sub-clusters: moderate bonus
                cluster_bonus = 1.1
            
            # Final penalty with exponential decay
            penalty = np.exp(-1.5 * combined_regret) * cluster_bonus
            
            regret_penalty[i, j] = min(1.5, penalty)  # Cap at 1.5
            regret_penalty[j, i] = regret_penalty[i, j]
    
    return regret_penalty


def _adaptive_3opt_lookahead(dist: np.ndarray, n: int, clusters: Dict, 
                            params: Dict, stats: Dict) -> np.ndarray:
    """Perform adaptive 3-opt lookahead simulations."""
    lookahead_scores = np.ones_like(dist)
    
    if n < 6:
        return lookahead_scores
    
    intensity = params['lookahead_intensity']
    
    # Precompute nearest neighbors for efficiency
    k_neighbors = min(8, n - 1)
    nearest_indices = np.argsort(dist, axis=1)[:, 1:k_neighbors+1]
    
    # Adaptive sampling based on intensity
    n_samples = int(intensity * 10)
    
    for i in range(n):
        # Consider edges from i to its promising connections
        candidates = nearest_indices[i]
        if len(candidates) == 0:
            continue
            
        for j in candidates:
            if j <= i:
                continue
                
            best_improvement = 0
            
            # Sample random 3-opt configurations
            for _ in range(n_samples):
                # Randomly select 4 distinct nodes (not i or j)
                available = [x for x in range(n) if x not in [i, j]]
                if len(available) < 4:
                    break
                    
                selected = np.random.choice(available, size=4, replace=False)
                k, l, m, p = selected
                
                # Try different 3-opt configurations
                configurations = [
                    # Remove (i,j), (k,l), (m,p)
                    [(i,j), (k,l), (m,p)],
                    # Remove (i,j), (k,m), (l,p)
                    [(i,j), (k,m), (l,p)],
                    # Remove (i,j), (k,p), (l,m)
                    [(i,j), (k,p), (l,m)]
                ]
                
                for edges in configurations:
                    (a1, a2), (b1, b2), (c1, c2) = edges
                    
                    # Current cost
                    current = dist[a1, a2] + dist[b1, b2] + dist[c1, c2]
                    
                    # Try promising reconnections (subset of 3-opt possibilities)
                    reconnections = [
                        # (a1,b1), (a2,c1), (b2,c2)
                        (dist[a1, b1] + dist[a2, c1] + dist[b2, c2]),
                        # (a1,b2), (a2,c1), (b1,c2)
                        (dist[a1, b2] + dist[a2, c1] + dist[b1, c2]),
                        # (a1,c1), (a2,b1), (b2,c2)
                        (dist[a1, c1] + dist[a2, b1] + dist[b2, c2]),
                    ]
                    
                    for new_cost in reconnections:
                        improvement = current - new_cost
                        if improvement > best_improvement:
                            best_improvement = improvement
            
            # Convert improvement to score
            if best_improvement > 0:
                norm_improvement = min(1.0, best_improvement / (stats['avg_distance'] + 1e-12))
                score = 1.0 + 0.8 * norm_improvement
            else:
                # Small penalty if no improvement found
                score = 0.85
                
            # Cluster-aware adjustment
            if clusters['medium'][i] != clusters['medium'][j]:
                # Inter-cluster edges get bonus for lookahead
                score = min(1.5, score * 1.1)
                
            lookahead_scores[i, j] = score
            lookahead_scores[j, i] = score
    
    return lookahead_scores


def _compute_structural_connectivity_bonus(dist: np.ndarray, n: int, 
                                          clusters: Dict, embeddings: np.ndarray) -> np.ndarray:
    """Compute connectivity bonuses with structural awareness."""
    connectivity_bonus = np.ones_like(dist)
    
    # Minimum Spanning Tree edges
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    
    # Base MST bonus
    connectivity_bonus[mst_mask] = 1.5
    
    # Enhanced bonus for MST edges connecting different coarse clusters
    coarse_clusters = clusters['coarse']
    for i in range(n):
        for j in range(i + 1, n):
            if mst_mask[i, j]:
                if coarse_clusters[i] != coarse_clusters[j]:
                    connectivity_bonus[i, j] = 2.0
                elif clusters['fine'][i] != clusters['fine'][j]:
                    # Different sub-clusters
                    connectivity_bonus[i, j] = 1.7
    
    # Additional bonus for edges that are in many shortest paths (approximation)
    # We approximate via degree product (simpler than betweenness)
    degrees = np.sum(dist < np.inf, axis=1)
    degree_product = np.outer(degrees, degrees)
    degree_bonus = 1.0 + 0.3 * (degree_product / np.max(degree_product))
    
    # Combine with connectivity bonus
    connectivity_bonus = connectivity_bonus * degree_bonus
    
    return connectivity_bonus


def _compute_embedding_affinity(embeddings: np.ndarray, n: int, params: Dict) -> np.ndarray:
    """Compute edge affinity scores based on embedding similarity."""
    if n < 2:
        return np.ones((n, n))
    
    # Cosine similarity between embeddings
    norms = np.linalg.norm(embeddings, axis=1, keepdims=True)
    normalized = embeddings / (norms + 1e-12)
    similarity = normalized @ normalized.T
    
    # Convert to affinity scores
    affinity = 0.5 + 0.5 * similarity  # Map to [0, 1]
    np.fill_diagonal(affinity, 0)
    
    # Mutual top-K reinforcement
    k = max(2, int(np.sqrt(n)))
    mutual_priority = np.zeros_like(affinity)
    
    for i in range(n):
        row_affinity = affinity[i].copy()
        row_affinity[i] = -1
        top_k_idx = np.argsort(row_affinity)[-k:]
        
        for j in top_k_idx:
            col_affinity = affinity[j].copy()
            col_affinity[j] = -1
            j_top_k = np.argsort(col_affinity)[-k:]
            
            if i in j_top_k:
                mutual_priority[i, j] = 1.0
    
    # Combine base affinity with mutual priority
    affinity = affinity * (1.0 + 0.4 * mutual_priority)
    
    return affinity


def _intelligent_sparsification(scores: np.ndarray, dist: np.ndarray, n: int,
                               clusters: Dict, mst_mask: np.ndarray, params: Dict) -> np.ndarray:
    """Intelligent sparsification preserving structural connectivity."""
    if n <= 2:
        return scores
    
    # Create initial mask
    final_mask = np.zeros_like(scores, dtype=bool)
    edges_per_node = params['edges_per_node_base']
    
    # Keep top edges per node
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        # Adaptive number of edges based on cluster size
        cluster_size = np.sum(clusters['coarse'] == clusters['coarse'][i])
        adaptive_edges = max(2, int(edges_per_node * (1.0 + 0.3 * np.log2(cluster_size + 1))))
        
        top_indices = np.argsort(row_scores)[-adaptive_edges:]
        final_mask[i, top_indices] = True
    
    # Ensure symmetry
    final_mask = final_mask | final_mask.T
    
    # Always keep MST edges
    final_mask = final_mask | mst_mask
    
    # Ensure inter-cluster connectivity
    coarse_clusters = clusters['coarse']
    unique_clusters = np.unique(coarse_clusters)
    
    if len(unique_clusters) > 1:
        for cluster_id in unique_clusters:
            cluster_nodes = np.where(coarse_clusters == cluster_id)[0]
            other_nodes = np.where(coarse_clusters != cluster_id)[0]
            
            if len(cluster_nodes) > 0 and len(other_nodes) > 0:
                # For each cluster, ensure at least min connections to other clusters
                for node in cluster_nodes[:params['min_cluster_connections']]:
                    # Find best connection to other clusters
                    mask = coarse_clusters != cluster_id
                    if np.any(mask):
                        available_scores = scores[node] * mask
                        if np.any(available_scores > 0):
                            best_other = np.argmax(available_scores)
                            final_mask[node, best_other] = True
                            final_mask[best_other, node] = True
    
    # Apply mask with smooth transition
    masked_scores = scores * final_mask
    
    # Add small residual for unmasked edges (smoother than hard cutoff)
    residual = 0.05 * scores.max()
    scores = masked_scores + residual * (~final_mask)
    
    return scores
