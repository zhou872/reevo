import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import distance_matrix


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics combining multiple scoring strategies.
    
    This function computes heuristic scores for each edge in a TSP instance,
    indicating how promising it is to include that edge in a solution.
    Higher values indicate more promising edges.
    
    Core strategies:
    1. Distance Inverse Weighting: Base scoring from inverse distances
    2. Two-step Lookahead Spatial Scoring: Considers neighborhood quality
    3. Regret-based Penalty: Penalizes edges with poor alternatives
    4. Cluster Density Bonus: Rewards edges in dense local clusters
    
    Parameters
    ----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances between n nodes
        
    Returns
    -------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) if n == 2 else np.zeros_like(distance_matrix)
    
    # Working copy with diagonal set to infinity
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # ========== 1. BASE INVERSE DISTANCE SCORING ==========
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / (dist + 1e-12)
    
    # ========== 2. TWO-STEP LOOKAHEAD SPATIAL SCORING ==========
    lookahead_scores = np.ones_like(dist)
    
    # Precompute k-nearest neighbors for each node (k = min(5, n-1))
    k = min(5, n - 1)
    neighbor_indices = np.argsort(dist, axis=1)[:, 1:k+1]
    
    for i in range(n):
        for j in range(i + 1, n):
            if dist[i, j] < np.inf:
                # Find alternative best connections for i and j
                alt_for_i = None
                for idx in neighbor_indices[i]:
                    if idx != j:
                        alt_for_i = idx
                        break
                
                alt_for_j = None
                for idx in neighbor_indices[j]:
                    if idx != i:
                        alt_for_j = idx
                        break
                
                if alt_for_i is not None and alt_for_j is not None:
                    # Calculate two-step score
                    immediate_score = 1.0 / (dist[i, j] + 1e-12)
                    alt_i_score = 1.0 / (dist[i, alt_for_i] + 1e-12) if dist[i, alt_for_i] > 0 else 0
                    alt_j_score = 1.0 / (dist[j, alt_for_j] + 1e-12) if dist[j, alt_for_j] > 0 else 0
                    
                    # Combine immediate score with neighborhood quality
                    lookahead_scores[i, j] = lookahead_scores[j, i] = (
                        immediate_score + 0.3 * (alt_i_score + alt_j_score)
                    )
    
    # ========== 3. REGRET-BASED PENALTY ==========
    regret_scores = np.ones_like(dist)
    
    for i in range(n):
        # Get distances from node i
        row_dists = dist[i].copy()
        
        # Find best and second best distances
        sorted_indices = np.argsort(row_dists)
        if len(sorted_indices) >= 2:
            best_idx = sorted_indices[1]  # Index 0 is self (inf)
            second_best_idx = sorted_indices[2] if len(sorted_indices) >= 3 else best_idx
            
            best_dist = row_dists[best_idx]
            second_best_dist = row_dists[second_best_idx]
            
            if second_best_dist - best_dist > 1e-12:
                for j in range(n):
                    if i != j and dist[i, j] < np.inf:
                        # Calculate regret: how much worse is this than the best option?
                        regret = (dist[i, j] - best_dist) / (second_best_dist - best_dist)
                        # Apply exponential penalty for high regret
                        regret_scores[i, j] = np.exp(-0.5 * regret)
    
    # Make regret symmetric
    regret_scores = np.minimum(regret_scores, regret_scores.T)
    
    # ========== 4. CLUSTER DENSITY BONUS ==========
    density_scores = np.ones_like(dist)
    
    # Calculate local density for each node
    densities = np.zeros(n)
    for i in range(n):
        # Average distance to k nearest neighbors
        if len(neighbor_indices[i]) > 0:
            avg_dist = np.mean(dist[i, neighbor_indices[i]])
            densities[i] = 1.0 / (avg_dist + 1e-12)
    
    # Normalize densities
    if np.max(densities) > 0:
        densities = densities / np.max(densities)
    
    # Apply density bonus to edges between nodes in dense regions
    for i in range(n):
        for j in range(i + 1, n):
            if dist[i, j] < np.inf:
                # Bonus is geometric mean of node densities
                density_bonus = 1.0 + 0.5 * np.sqrt(densities[i] * densities[j])
                density_scores[i, j] = density_scores[j, i] = density_bonus
    
    # ========== 5. COMBINE ALL SCORES ==========
    # Normalize components first
    components = [inv_dist, lookahead_scores, regret_scores, density_scores]
    normalized_components = []
    
    for comp in components:
        comp_max = comp.max()
        if comp_max > 0:
            normalized_components.append(comp / comp_max)
        else:
            normalized_components.append(comp)
    
    # Geometric mean for combination (preserves zero values)
    combined_scores = np.ones_like(dist)
    for comp in normalized_components:
        combined_scores *= comp
    
    # ========== 6. MST REINFORCEMENT ==========
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    combined_scores[mst_mask] *= 1.5  # Bonus for MST edges
    
    # ========== 7. INTELLIGENT SPARSIFICATION ==========
    # Keep only promising edges
    sparsity_mask = np.zeros_like(combined_scores, dtype=bool)
    
    # For each node, keep connections to top candidates
    for i in range(n):
        # Determine how many connections to keep for this node
        keep_count = min(max(3, n // 4), n - 1)
        
        # Get scores for this node (excluding self)
        node_scores = combined_scores[i].copy()
        node_scores[i] = -np.inf
        
        # Find threshold for top keep_count scores
        if keep_count > 0 and keep_count < n:
            threshold = np.partition(node_scores, -keep_count)[-keep_count]
            sparsity_mask[i, node_scores >= threshold] = True
    
    # Ensure symmetry
    sparsity_mask = sparsity_mask | sparsity_mask.T
    
    # Always keep MST edges
    sparsity_mask = sparsity_mask | mst_mask
    
    # Apply sparsification
    final_scores = combined_scores * sparsity_mask
    
    # ========== 8. FINAL NORMALIZATION ==========
    np.fill_diagonal(final_scores, 0)
    
    # Row-wise normalization to [0, 1]
    row_sums = final_scores.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    final_scores = final_scores / row_sums
    
    # Ensure symmetry
    final_scores = np.maximum(final_scores, final_scores.T)
    
    return final_scores
