import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with improved factor combination:
    1. Inverse distance with adaptive epsilon
    2. MST edge inclusion with continuous bonus
    3. Candidate pool sparsification with dynamic threshold
    4. Symmetric normalization preserving relative rankings
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix) if n == 1 else np.array([])
    
    # Create working copy to avoid modifying input
    dist = distance_matrix.copy().astype(float)
    
    # Adaptive epsilon based on distance statistics
    flat_dist = dist[~np.eye(n, dtype=bool)]
    eps = np.percentile(flat_dist, 10) * 0.01  # 1% of 10th percentile
    
    # 1. Base inverse distance with adaptive smoothing
    np.fill_diagonal(dist, np.inf)
    inv_dist = 1 / (dist + eps)
    
    # 2. MST edge bonus - continuous bonus based on MST edge importance
    mst = minimum_spanning_tree(dist)
    mst_array = mst.toarray()
    mst_edges = mst_array > 0
    
    # Calculate MST bonus as scaled inverse of edge rank in MST ordering
    if np.any(mst_edges):
        mst_values = dist[mst_edges]
        sorted_indices = np.argsort(mst_values)
        rank_bonus = np.zeros_like(mst_values)
        for idx, pos in enumerate(sorted_indices):
            rank_bonus[pos] = 2.0 - 0.5 * (idx / max(1, len(sorted_indices) - 1))
        
        mst_bonus = np.ones_like(dist)
        mst_bonus[mst_edges] = rank_bonus
    else:
        mst_bonus = np.ones_like(dist)
    
    # 3. Candidate pool sparsification with dynamic threshold
    # For each node, keep edges within factor of minimum distance
    mask = np.zeros_like(dist, dtype=bool)
    factor = 1.5 + 0.5 * np.log10(max(2, n))  # Adaptive factor
    
    for i in range(n):
        row = dist[i].copy()
        row[i] = np.inf
        min_dist = np.min(row)
        
        # Keep edges within factor * min_dist
        candidate_mask = row <= factor * min_dist
        # Ensure at least 2 candidates per node
        if np.sum(candidate_mask) < 2:
            # Get indices of two smallest distances
            indices = np.argsort(row)[:2]
            candidate_mask = np.zeros(n, dtype=bool)
            candidate_mask[indices] = True
        
        mask[i, :] = candidate_mask
    
    # Ensure symmetry
    mask = mask | mask.T
    
    # Always include MST edges for connectivity
    mask = mask | mst_edges
    
    # 4. Degree-aware penalization
    # Penalize edges from nodes with many candidates
    degree_penalty = np.ones_like(dist)
    for i in range(n):
        node_degree = np.sum(mask[i])
        if node_degree > 0:
            penalty = 1.0 / (1.0 + 0.2 * (node_degree - 2))  # Base degree is 2
            degree_penalty[i, :] *= penalty
            degree_penalty[:, i] *= penalty
    
    # Combine all factors
    scores = inv_dist * mst_bonus * degree_penalty
    
    # Apply sparsification mask
    scores = scores * mask
    
    # Symmetric normalization preserving zeros
    np.fill_diagonal(scores, 0)
    max_val = np.max(scores)
    if max_val > 0:
        scores = scores / max_val
    
    return scores
