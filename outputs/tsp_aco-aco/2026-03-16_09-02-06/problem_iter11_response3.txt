import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with multiple scoring strategies combined.
    
    Core improvements:
    1. Dynamic regret thresholding based on distance distribution
    2. Average-based two-step lookahead using k-nearest neighbors
    3. Two-step lookahead spatial scoring for neighborhood links
    4. Cluster density-based bonuses for intra-cluster edges
    
    Parameters
    ----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances between n nodes
        
    Returns
    -------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # Working copy with diagonal set to infinity
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # ========== 1. STATISTICAL ANALYSIS ==========
    flat_dists = dist[np.triu_indices(n, k=1)]
    dist_mean = np.mean(flat_dists)
    dist_std = np.std(flat_dists)
    dist_cv = dist_std / (dist_mean + 1e-12)  # Coefficient of variation
    
    # ========== 2. BASE INVERSE DISTANCE ==========
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.where(dist < np.inf, 1.0 / (dist + 1e-12), 0)
    
    # ========== 3. DYNAMIC REGRET THRESHOLDING ==========
    regret_matrix = np.ones_like(dist)
    
    for i in range(n):
        dist_i = dist[i].copy()
        
        # Find k nearest neighbors for robust statistics
        k_neighbors = min(6, n - 1)
        if k_neighbors >= 2:
            neighbor_indices = np.argsort(dist_i)[1:k_neighbors+1]
            neighbor_dists = dist_i[neighbor_indices]
            
            # Calculate robust best and second-best
            if len(neighbor_dists) >= 2:
                best_dist = neighbor_dists[0]
                # Use median of next few for robust second best
                second_best = np.median(neighbor_dists[1:min(4, len(neighbor_dists))])
                
                # Dynamic threshold based on coefficient of variation
                threshold = max(0.3, 0.1 + 0.4 * np.tanh(dist_cv))
                
                # Apply regret penalty only when exceeding threshold
                for j in range(n):
                    if i != j and dist_i[j] < np.inf:
                        if second_best - best_dist > 1e-12:
                            regret = (dist_i[j] - best_dist) / (second_best - best_dist)
                            if regret > threshold:
                                # Exponential penalty scaled by CV
                                regret_matrix[i, j] = np.exp(-0.4 * (regret - threshold))
    
    # Symmetrize regret
    regret_symmetric = np.sqrt(regret_matrix * regret_matrix.T)
    
    # ========== 4. AVERAGE-BASED TWO-STEP LOOKAHEAD ==========
    avg_lookahead = np.ones_like(dist)
    k_lookahead = min(4, n - 2)  # Number of neighbors to average
    
    for i in range(n):
        for j in range(i + 1, n):
            if dist[i, j] < np.inf:
                # Calculate average distance from j to its k nearest neighbors (excluding i)
                dist_j = dist[j].copy()
                dist_j[i] = np.inf
                
                if np.sum(dist_j < np.inf) >= k_lookahead:
                    # Find k nearest neighbors of j (excluding i)
                    j_neighbors = np.argsort(dist_j)[:k_lookahead]
                    avg_next_dist = np.mean(dist_j[j_neighbors])
                    
                    # Normalize by global average and compute penalty
                    normalized_avg = avg_next_dist / (dist_mean + 1e-12)
                    # Inverse relationship: lower avg_next_dist is better
                    avg_lookahead[i, j] = avg_lookahead[j, i] = 1.0 / (1.0 + 0.3 * normalized_avg)
    
    # ========== 5. TWO-STEP LOOKAHEAD SPATIAL SCORING ==========
    spatial_scores = np.ones_like(dist)
    
    for i in range(n):
        dist_i = dist[i].copy()
        
        # Find best alternative connections for i
        if np.sum(dist_i < np.inf) >= 2:
            sorted_i = np.argsort(dist_i)
            best_alt_i = dist_i[sorted_i[1]]  # Best excluding diagonal
            second_best_alt = dist_i[sorted_i[2]] if len(sorted_i) > 2 else best_alt_i
            
        for j in range(i + 1, n):
            if dist[i, j] < np.inf:
                dist_j = dist[j].copy()
                
                # Find best alternative connections for j
                if np.sum(dist_j < np.inf) >= 2:
                    sorted_j = np.argsort(dist_j)
                    best_alt_j = dist_j[sorted_j[1]]  # Best excluding diagonal
                    second_best_alt_j = dist_j[sorted_j[2]] if len(sorted_j) > 2 else best_alt_j
                    
                    # Combine immediate distance with alternative connection quality
                    immediate_score = 1.0 / (1.0 + dist[i, j] / dist_mean)
                    alt_quality = 0.3 * (best_alt_i + best_alt_j) / (2 * dist_mean)
                    spatial_scores[i, j] = spatial_scores[j, i] = immediate_score + alt_quality
    
    # ========== 6. CLUSTER DENSITY BONUS ==========
    # Identify potential clusters using distance percentiles
    density_bonus = np.ones_like(dist)
    if n >= 10:
        # Calculate local density for each node
        local_density = np.zeros(n)
        k_density = min(5, n - 1)
        
        for i in range(n):
            dist_i = dist[i].copy()
            # Average distance to k nearest neighbors
            if np.sum(dist_i < np.inf) >= k_density:
                nearest = np.partition(dist_i, k_density)[:k_density]
                local_density[i] = 1.0 / (np.mean(nearest) + 1e-12)
        
        # Apply density bonus to edges within dense clusters
        density_threshold = np.percentile(local_density, 70)
        dense_nodes = local_density > density_threshold
        
        for i in range(n):
            if dense_nodes[i]:
                for j in range(i + 1, n):
                    if dense_nodes[j] and dist[i, j] < np.inf:
                        # Bonus for intra-cluster edges
                        density_factor = (local_density[i] * local_density[j]) / (density_threshold**2 + 1e-12)
                        density_bonus[i, j] = density_bonus[j, i] = 1.0 + 0.5 * np.tanh(density_factor - 1.0)
    
    # ========== 7. MST REINFORCEMENT ==========
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # ========== 8. COMBINE ALL SCORES ==========
    # Weighted combination of different scoring strategies
    combined = (inv_dist * 0.4 + 
                regret_symmetric * 0.2 + 
                avg_lookahead * 0.2 + 
                spatial_scores * 0.2) * density_bonus
    
    # Boost MST edges
    mst_boost = 1.0 + 0.8 * np.tanh(15 / n)  # Larger boost for smaller problems
    combined[mst_mask] *= mst_boost
    
    # ========== 9. ADAPTIVE SPARSIFICATION ==========
    # Determine degree targets based on problem size
    base_degree = max(2, int(np.log2(n) + 1))
    max_degree = min(12, n // 2)
    
    sparsity_mask = np.zeros_like(combined, dtype=bool)
    
    for i in range(n):
        # Adaptive degree based on node centrality
        row_vals = combined[i].copy()
        row_vals[i] = -np.inf
        
        # Use local statistics to determine cutoff
        if np.sum(row_vals > 0) > base_degree:
            # Dynamic threshold based on value distribution
            sorted_vals = np.sort(row_vals[row_vals > -np.inf])
            if len(sorted_vals) > base_degree:
                # Keep top base_degree to max_degree edges
                keep_k = min(max_degree, max(base_degree, len(sorted_vals) // 3))
                threshold = sorted_vals[-keep_k]
                sparsity_mask[i, row_vals >= threshold] = True
    
    # Ensure symmetry and preserve MST
    sparsity_mask = (sparsity_mask | sparsity_mask.T) | mst_mask
    
    # Apply sparsification
    final_scores = combined * sparsity_mask
    
    # ========== 10. FINAL NORMALIZATION ==========
    np.fill_diagonal(final_scores, 0)
    
    # Row-wise softmax normalization
    row_max = final_scores.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    row_exp = np.exp(final_scores / row_max - 1)
    row_sum = row_exp.sum(axis=1, keepdims=True)
    final_scores = np.where(row_sum > 0, row_exp / row_sum, 0)
    
    # Global scaling to [0, 1]
    global_max = final_scores.max()
    if global_max > 0:
        final_scores = final_scores / global_max
    
    return final_scores
