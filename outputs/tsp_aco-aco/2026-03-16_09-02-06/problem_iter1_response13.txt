import numpy as np
from scipy.spatial import cKDTree
from sklearn.cluster import KMeans

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining spatial clustering,
    regret analysis, neighbor density, and sparsity.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric matrix of distances between nodes
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape indicating edge promise
    """
    n = distance_matrix.shape[0]
    np.fill_diagonal(distance_matrix, np.inf)  # Avoid self-connections
    
    # 1. Base inverse distance with adaptive scaling
    base_heuristic = 1 / distance_matrix
    
    # 2. Spatial clustering bonus
    positions = np.random.randn(n, 2)  # Placeholder for actual coordinates
    kmeans = KMeans(n_clusters=min(10, n//4), n_init=10, random_state=42)
    cluster_labels = kmeans.fit_predict(positions)
    
    # Bonus for intra-cluster edges, penalty for inter-cluster
    cluster_bonus = np.zeros_like(base_heuristic)
    for i in range(n):
        for j in range(i+1, n):
            if cluster_labels[i] == cluster_labels[j]:
                # Intra-cluster bonus inversely proportional to cluster size
                cluster_size = np.sum(cluster_labels == cluster_labels[i])
                cluster_bonus[i, j] = cluster_bonus[j, i] = 2.0 / np.sqrt(cluster_size)
            else:
                # Inter-cluster penalty based on cluster separation
                mean_i = kmeans.cluster_centers_[cluster_labels[i]]
                mean_j = kmeans.cluster_centers_[cluster_labels[j]]
                cluster_dist = np.linalg.norm(mean_i - mean_j)
                penalty = np.exp(-cluster_dist / 5.0)
                cluster_bonus[i, j] = cluster_bonus[j, i] = 0.3 * penalty
    
    # 3. Neighbor density weighting
    kdtree = cKDTree(positions)
    # For each point, get distances to k nearest neighbors
    k = min(15, n-1)
    neighbor_dists, _ = kdtree.query(positions, k=k+1)
    avg_local_density = 1 / np.mean(neighbor_dists[:, 1:], axis=1)
    
    density_weight = np.zeros_like(base_heuristic)
    for i in range(n):
        for j in range(i+1, n):
            # Edge weight proportional to geometric mean of endpoint densities
            density_factor = np.sqrt(avg_local_density[i] * avg_local_density[j])
            # Normalize by maximum density
            density_weight[i, j] = density_weight[j, i] = density_factor / np.max(avg_local_density)
    
    # 4. Path regret penalty
    # For each node, find its two nearest neighbors
    nearest_dists = np.partition(distance_matrix, 2, axis=1)[:, :3]
    regret_penalty = np.zeros_like(base_heuristic)
    
    for i in range(n):
        # Sort distances to get 1st and 2nd nearest
        sorted_dists = np.sort(distance_matrix[i])
        if len(sorted_dists) >= 3:
            nearest = sorted_dists[1]  # 2nd smallest (1st is 0/inf)
            second_nearest = sorted_dists[2]
            
            for j in range(n):
                if i != j:
                    # Penalty proportional to how much worse this edge is compared to nearest
                    regret = distance_matrix[i, j] - nearest
                    max_regret = second_nearest - nearest
                    if max_regret > 0:
                        penalty = 1.0 - np.minimum(regret / max_regret, 1.0)
                        regret_penalty[i, j] = penalty
    
    # 5. Combine factors
    combined_heuristic = (
        base_heuristic * 
        (1.2 + cluster_bonus) * 
        (0.8 + 0.4 * density_weight) * 
        (0.7 + 0.6 * regret_penalty)
    )
    
    # 6. Adaptive sparsification with proximity threshold
    # For each node, only keep top k edges based on distance
    k_sparse = min(20, max(5, n // 3))
    sparse_heuristic = np.zeros_like(combined_heuristic)
    
    for i in range(n):
        # Get indices of k_sparse nearest neighbors by distance
        valid_indices = np.where(distance_matrix[i] < np.inf)[0]
        if len(valid_indices) > k_sparse:
            # Get distances to valid neighbors
            neighbor_dists = distance_matrix[i, valid_indices]
            # Find threshold for top k_sparse
            threshold_idx = np.argpartition(neighbor_dists, k_sparse)[k_sparse]
            threshold = neighbor_dists[threshold_idx]
            
            # Keep only edges within threshold
            mask = distance_matrix[i] <= threshold
            sparse_heuristic[i] = combined_heuristic[i] * mask
        else:
            sparse_heuristic[i] = combined_heuristic[i]
    
    # 7. Symmetrize and finalize
    # Ensure symmetry by taking maximum of both directions
    final_heuristic = np.maximum(sparse_heuristic, sparse_heuristic.T)
    np.fill_diagonal(final_heuristic, 0)
    
    return final_heuristic
