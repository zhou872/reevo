import numpy as np
from scipy.spatial import ConvexHull

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge priority heuristic with multi-step lookahead, regret-based
    optimization, and global topological features.
    
    Features:
    1. Multi-step lookahead edge potential (3-step path evaluation)
    2. Regret-based edge prioritization (opportunity cost analysis)
    3. Minimum spanning tree boost (global structure preservation)
    4. Convex hull boundary bonus (handles perimeter edges)
    
    Returns: Heuristic matrix of same shape as distance_matrix
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.zeros_like(distance_matrix)
    
    epsilon = 1e-12
    
    # 1. Base inverse distance (standard foundation)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.where(distance_matrix > epsilon, 1.0 / distance_matrix, 0)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Multi-step lookahead edge potential (3-step paths)
    lookahead_potential = np.zeros_like(distance_matrix)
    
    # Precompute sorted neighbor indices for faster lookups
    sorted_neighbors = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        # Consider each potential next node j
        for j_idx in range(1, min(6, n)):  # Top 5 neighbors
            j = sorted_neighbors[i, j_idx]
            if i == j:
                continue
                
            # Lookahead to 3-step paths i->j->k->l
            best_3step = np.inf
            
            # Try potential middle nodes k
            for k_idx in range(1, min(6, n)):  # Top 5 neighbors of j
                k = sorted_neighbors[j, k_idx]
                if k == i or k == j:
                    continue
                    
                # Find best continuation l from k (excluding i, j, k)
                for l_idx in range(1, min(6, n)):  # Top 5 neighbors of k
                    l = sorted_neighbors[k, l_idx]
                    if l == i or l == j or l == k:
                        continue
                    
                    path_cost = (distance_matrix[i, j] + 
                                distance_matrix[j, k] + 
                                distance_matrix[k, l])
                    avg_step = path_cost / 3.0
                    
                    # Normalize by distance[i,j] to get efficiency ratio
                    efficiency = distance_matrix[i, j] / (avg_step + epsilon)
                    if efficiency > 0 and efficiency < best_3step:
                        best_3step = efficiency
            
            if best_3step < np.inf:
                # Higher potential for edges leading to efficient 3-step paths
                lookahead_potential[i, j] = 1.0 / (best_3step + epsilon)
    
    # Make symmetric
    lookahead_potential = np.maximum(lookahead_potential, lookahead_potential.T)
    
    # 3. Regret-based edge prioritization
    regret_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get distances to all other nodes
        distances = distance_matrix[i].copy()
        distances[i] = np.inf
        
        # Sort to find best alternatives
        sorted_dists = np.sort(distances)
        if len(sorted_dists) >= 3:
            best_dist = sorted_dists[0]
            second_best = sorted_dists[1]
            third_best = sorted_dists[2]
            
            # Regret of not taking the best edge
            base_regret = second_best - best_dist
            
            for j in range(n):
                if i == j:
                    continue
                    
                dist_ij = distance_matrix[i, j]
                
                if dist_ij == best_dist:
                    # This is the best edge - high priority
                    regret_matrix[i, j] = 1.0 + (third_best - second_best) / (best_dist + epsilon)
                else:
                    # Calculate regret of choosing this edge vs best
                    regret = (dist_ij - best_dist) / (base_regret + epsilon)
                    # Transform: lower regret (closer to best) gets higher score
                    regret_matrix[i, j] = 1.0 / (1.0 + regret)
    
    # 4. Minimum Spanning Tree boost (Prim's algorithm)
    mst_boost = np.zeros_like(distance_matrix)
    
    if n > 1:
        # Initialize MST construction
        in_mst = np.zeros(n, dtype=bool)
        min_edge_to = np.full(n, np.inf)
        min_edge_from = np.full(n, -1, dtype=int)
        
        # Start from node 0
        min_edge_to[0] = 0
        
        for _ in range(n):
            # Find node not in MST with minimum edge weight
            u = -1
            min_val = np.inf
            for v in range(n):
                if not in_mst[v] and min_edge_to[v] < min_val:
                    min_val = min_edge_to[v]
                    u = v
            
            if u == -1:
                break
                
            in_mst[u] = True
            
            # Add edge to MST if it exists
            if min_edge_from[u] != -1:
                v = min_edge_from[u]
                mst_boost[u, v] = mst_boost[v, u] = 1.0
            
            # Update minimum edges for neighbors
            for v in range(n):
                if not in_mst[v] and distance_matrix[u, v] < min_edge_to[v]:
                    min_edge_to[v] = distance_matrix[u, v]
                    min_edge_from[v] = u
    
    # 5. Convex hull boundary bonus (estimate from distances)
    hull_bonus = np.zeros_like(distance_matrix)
    
    if n >= 4:
        # Estimate which nodes might be on convex hull using relative distances
        # Nodes with large distances to many others are likely on hull
        avg_distances = np.mean(distance_matrix, axis=1)
        hull_candidates = avg_distances > np.percentile(avg_distances, 70)
        
        # Boost connections between hull candidates
        for i in range(n):
            if hull_candidates[i]:
                # Find other hull candidates
                for j in range(i+1, n):
                    if hull_candidates[j]:
                        # Bonus proportional to how "extreme" both nodes are
                        bonus = (avg_distances[i] * avg_distances[j]) / np.mean(avg_distances)**2
                        hull_bonus[i, j] = hull_bonus[j, i] = min(bonus, 2.0)
    
    # 6. Dynamic parameter tuning based on problem characteristics
    n_edges = n * (n-1) // 2
    avg_distance = np.mean(distance_matrix[distance_matrix > 0])
    distance_std = np.std(distance_matrix[distance_matrix > 0])
    
    # Adaptive weights based on problem structure
    if distance_std / (avg_distance + epsilon) > 0.5:
        # High variance: emphasize lookahead and regret
        weights = [0.25, 0.3, 0.25, 0.2]  # inv_dist, lookahead, regret, mst, hull
    else:
        # More uniform: balanced approach
        weights = [0.3, 0.25, 0.2, 0.25]
    
    # 7. Combine all factors
    heuristic = (weights[0] * inv_dist +
                 weights[1] * lookahead_potential +
                 weights[2] * regret_matrix +
                 weights[3] * (mst_boost + hull_bonus))
    
    # 8. Adaptive sparsification
    # Keep more edges for small problems, fewer for large ones
    edges_per_node = max(3, int(np.sqrt(n) * 1.5))
    
    sparsified = np.zeros_like(heuristic)
    for i in range(n):
        # Get top edges for this node
        row_heuristics = heuristic[i].copy()
        row_heuristics[i] = -np.inf  # Exclude self
        
        if np.sum(row_heuristics > 0) > edges_per_node:
            threshold = np.sort(row_heuristics)[-edges_per_node]
            mask = row_heuristics >= threshold
        else:
            mask = row_heuristics > 0
        
        sparsified[i, mask] = heuristic[i, mask]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 9. Final normalization
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
