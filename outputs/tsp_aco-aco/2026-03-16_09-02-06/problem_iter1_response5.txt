import numpy as np
from scipy.spatial import distance_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection priorities.
    Combines multiple factors to identify promising edges and sparsifies the matrix.
    """
    n = distance_matrix.shape[0]
    
    # Base inverse distance (handling zeros)
    epsilon = 1e-10
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # Factor 1: Nearest neighbor rankings
    nn_rank = np.zeros_like(distance_matrix)
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])
        for rank, j in enumerate(sorted_indices[1:]):  # Skip self
            nn_rank[i, j] = 1 / (rank + 1)  # Higher score for closer ranks
    
    # Factor 2: Savings heuristic (Clarke-Wright inspired)
    savings = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if i == j: continue
            # Find nearest neighbors for i and j (excluding each other)
            mask_i = np.ones(n, dtype=bool)
            mask_i[[i, j]] = False
            mask_j = np.ones(n, dtype=bool)
            mask_j[[i, j]] = False
            
            nn_i = np.min(distance_matrix[i, mask_i])
            nn_j = np.min(distance_matrix[j, mask_j])
            
            # Savings = direct_distance - (nn_i + nn_j)
            savings[i, j] = max(0, (nn_i + nn_j) - distance_matrix[i, j])
            savings[j, i] = savings[i, j]
    
    # Factor 3: Node centrality penalty
    degree_penalty = np.zeros_like(distance_matrix)
    node_scores = np.zeros(n)
    
    # Quick estimate of node "busyness" based on nearest neighbors
    for i in range(n):
        # Average distance to k nearest neighbors
        k = min(5, n-1)
        nearest_dists = np.partition(distance_matrix[i], k)[:k+1]
        node_scores[i] = np.mean(nearest_dists[nearest_dists > 0])
    
    for i in range(n):
        for j in range(n):
            if i == j: continue
            # Penalize edges between two "busy" nodes
            penalty = (node_scores[i] + node_scores[j]) / 2
            degree_penalty[i, j] = 1 / (1 + penalty)
    
    # Combine factors with weights
    w1, w2, w3 = 0.4, 0.4, 0.2
    combined = (w1 * inv_dist + 
                w2 * nn_rank + 
                w3 * (savings / (np.max(savings) + epsilon)) * degree_penalty)
    
    # Sparsification: keep only top edges
    sparsified = np.zeros_like(combined)
    
    # Strategy 1: Keep top k edges per node
    k = min(3, n-1)  # Keep at most 3 best edges per node
    for i in range(n):
        # Get indices of top k edges (excluding self)
        row = combined[i].copy()
        row[i] = -np.inf
        top_k_indices = np.argpartition(row, -k)[-k:]
        sparsified[i, top_k_indices] = row[top_k_indices]
    
    # Strategy 2: Symmetrize (if edge is good in one direction, keep both ways)
    sym_mask = (sparsified > 0) | (sparsified.T > 0)
    sparsified = sparsified * sym_mask
    
    # Strategy 3: Global threshold (keep top 20% of all edges)
    flat = sparsified.flatten()
    threshold = np.percentile(flat[flat > 0], 80)
    sparsified[sparsified < threshold] = 0
    
    return sparsified


# Example usage for testing
if __name__ == "__main__":
    # Create a sample distance matrix
    points = np.random.rand(10, 2) * 100
    dist_mat = distance_matrix(points, points)
    
    heuristics = heuristics_v2(dist_mat)
    print("Original shape:", dist_mat.shape)
    print("Heuristics shape:", heuristics.shape)
    print("Non-zero entries:", np.count_nonzero(heuristics))
    print("Max heuristic value:", np.max(heuristics))
