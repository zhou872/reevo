import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with improved sparsification and ranking.
    
    Key features:
    1. Reciprocal Distance Weighting: Uses 1/(d + ε) for base scores
    2. Two-Sided Percentile Pruning: Prunes edges that rank poorly for both endpoints
    3. Consensus Ranking Boost: Rewards edges where both nodes rank each other highly
    4. Progressive Edge Preservation: Maintains essential edges while aggressively pruning others
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # Create a working copy with diagonal set to infinity
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # 1. Base reciprocal distance with smoothing
    epsilon = 1e-10
    base_scores = 1 / (dist + epsilon)
    
    # 2. Two-sided percentile pruning
    # Identify edges that rank poorly for both endpoints
    percentile_threshold = 0.3  # Keep only top 30% for each node
    
    # Create ranking matrices
    rank_matrix = np.zeros_like(base_scores)
    for i in range(n):
        row_scores = base_scores[i].copy()
        row_scores[i] = -np.inf  # Exclude self
        
        # Get sorted indices by descending score
        sorted_idx = np.argsort(-row_scores)
        # Assign ranks (0 = best)
        for rank, idx in enumerate(sorted_idx):
            rank_matrix[i, idx] = rank
    
    # Normalize ranks to [0, 1] range (lower = better)
    max_rank = n - 1  # Excluding self
    if max_rank > 0:
        normalized_ranks = rank_matrix / max_rank
    else:
        normalized_ranks = np.zeros_like(rank_matrix)
    
    # Calculate mutual ranking penalty
    # Penalize edges where both nodes rank each other poorly
    mutual_penalty = np.ones_like(base_scores)
    for i in range(n):
        for j in range(i + 1, n):
            if normalized_ranks[i, j] > percentile_threshold and normalized_ranks[j, i] > percentile_threshold:
                # Both rank this edge poorly - apply penalty
                mutual_penalty[i, j] = mutual_penalty[j, i] = 0.1
    
    # Apply mutual penalty
    scores = base_scores * mutual_penalty
    
    # 3. Consensus ranking boost
    # Boost edges where nodes rank each other similarly well
    consensus_boost = np.ones_like(scores)
    for i in range(n):
        for j in range(i + 1, n):
            if normalized_ranks[i, j] <= 0.1 and normalized_ranks[j, i] <= 0.1:
                # Both nodes rank this edge in top 10% - apply boost
                consensus_boost[i, j] = consensus_boost[j, i] = 1.5
            elif normalized_ranks[i, j] <= 0.2 or normalized_ranks[j, i] <= 0.2:
                # At least one node ranks it in top 20% - moderate boost
                consensus_boost[i, j] = consensus_boost[j, i] = 1.2
    
    scores *= consensus_boost
    
    # 4. Progressive edge preservation with hybrid criteria
    # Use multiple preservation criteria
    preservation_mask = np.zeros_like(scores, dtype=bool)
    
    # 4a. Degree-aware local preservation
    k_local = max(2, int(np.sqrt(n)))  # Adaptive local preservation
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        # Preserve top k_local edges for each node
        top_k_idx = np.argpartition(row_scores, -k_local)[-k_local:]
        preservation_mask[i, top_k_idx] = True
    
    # 4b. Global percentile preservation
    # Get all edge scores (excluding diagonal and symmetric duplicates)
    triu_indices = np.triu_indices(n, k=1)
    edge_scores = scores[triu_indices]
    if len(edge_scores) > 0:
        global_threshold = np.percentile(edge_scores, 80)  # Top 20% globally
        global_mask = scores >= global_threshold
        preservation_mask = preservation_mask | global_mask
    
    # 4c. Ensure MST edges are preserved for connectivity
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    preservation_mask = preservation_mask | mst_mask
    
    # Ensure symmetry
    preservation_mask = preservation_mask | preservation_mask.T
    
    # Apply preservation mask
    scores = scores * preservation_mask
    
    # 5. Normalize scores
    max_score = np.max(scores)
    if max_score > 0:
        scores = scores / max_score
    
    # Zero out diagonal
    np.fill_diagonal(scores, 0)
    
    return scores
