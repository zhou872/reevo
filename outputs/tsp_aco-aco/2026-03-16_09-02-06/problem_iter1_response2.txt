import numpy as np
from scipy.spatial import Delaunay
from typing import Tuple, List

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple spatial and graph-based factors.
    Returns a matrix of edge promise indicators with sparsification (unpromising edges set to zero).
    """
    n = len(distance_matrix)
    if n <= 2:
        return 1 / (distance_matrix + np.eye(n) * 1e-12)
    
    # 1. Base inverse distance (handling zeros on diagonal)
    base_scores = 1 / (distance_matrix + np.eye(n) * 1e-12)
    np.fill_diagonal(base_scores, 0)
    
    # 2. Spatial cluster detection using Delaunay triangulation
    spatial_weights = np.zeros((n, n))
    try:
        # Create synthetic coordinates from distance matrix using MDS approximation
        from sklearn.manifold import MDS
        mds = MDS(n_components=2, dissimilarity='precomputed', random_state=42)
        coords = mds.fit_transform(distance_matrix)
        
        # Delaunay triangulation to identify natural spatial neighbors
        tri = Delaunay(coords)
        for simplex in tri.simplices:
            for i in range(3):
                for j in range(i+1, 3):
                    u, v = simplex[i], simplex[j]
                    spatial_weights[u, v] += 1
                    spatial_weights[v, u] += 1
        
        # Normalize spatial weights
        if spatial_weights.max() > 0:
            spatial_weights = spatial_weights / spatial_weights.max()
    except:
        spatial_weights = np.ones((n, n)) - np.eye(n)
    
    # 3. Local neighborhood connectivity score
    neighborhood_scores = np.zeros((n, n))
    k_neighbors = min(5, n-1)  # Number of nearest neighbors to consider
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        distances_i = distance_matrix[i].copy()
        distances_i[i] = np.inf
        nearest_indices = np.argpartition(distances_i, k_neighbors)[:k_neighbors]
        
        # Score based on reciprocal nearest neighbor relationships
        for j in nearest_indices:
            # Check if j also considers i among its nearest neighbors
            distances_j = distance_matrix[j].copy()
            distances_j[j] = np.inf
            j_nearest = np.argpartition(distances_j, k_neighbors)[:k_neighbors]
            
            mutual_bonus = 1.5 if i in j_nearest else 1.0
            # Weight by rank in neighbor list
            rank_i = np.where(np.argsort(distances_i)[:k_neighbors] == j)[0][0] + 1
            rank_j = np.where(np.argsort(distances_j)[:k_neighbors] == i)[0][0] + 1 if i in j_nearest else k_neighbors
            rank_score = 1.0 / (0.5 * (rank_i + rank_j))
            
            neighborhood_scores[i, j] = mutual_bonus * rank_score
    
    # 4. Node centrality measure (modified closeness centrality)
    centrality = np.zeros(n)
    for i in range(n):
        # Harmonic centrality (handles disconnected components better)
        non_zero_dists = distance_matrix[i][distance_matrix[i] > 0]
        if len(non_zero_dists) > 0:
            centrality[i] = len(non_zero_dists) / np.sum(1 / non_zero_dists)
    
    centrality_weights = np.outer(centrality, centrality)
    centrality_weights = centrality_weights / centrality_weights.max() if centrality_weights.max() > 0 else np.ones((n, n))
    
    # 5. Combine factors with adaptive weighting
    combined = (
        base_scores * 0.4 +  # Base distance term
        spatial_weights * 0.3 +  # Spatial structure term
        neighborhood_scores * 0.2 +  # Local connectivity term
        centrality_weights * 0.1  # Global structure term
    )
    
    # 6. Rank-based sparsification
    sparsified = combined.copy()
    sparsity_threshold = 0.3  # Keep top 30% of edges for each node
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf  # Ignore self-connection
        
        # Determine threshold for this row
        threshold = np.percentile(row[row > -np.inf], (1 - sparsity_threshold) * 100)
        
        # Zero out edges below threshold
        sparsified[i][sparsified[i] < threshold] = 0
    
    # Ensure symmetry (undirected graph)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 7. Final normalization
    row_sums = sparsified.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1  # Avoid division by zero
    normalized = sparsified / row_sums
    
    return normalized

# Optional: Helper function for demonstration
def generate_sample_tsp(n_cities: int = 10, seed: int = 42) -> np.ndarray:
    """Generate a sample distance matrix for testing."""
    np.random.seed(seed)
    # Random 2D coordinates
    coords = np.random.rand(n_cities, 2)
    
    # Euclidean distance matrix
    dist_matrix = np.zeros((n_cities, n_cities))
    for i in range(n_cities):
        for j in range(n_cities):
            if i != j:
                dist_matrix[i, j] = np.linalg.norm(coords[i] - coords[j])
    
    return dist_matrix
