import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform
from sklearn.cluster import KMeans

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic scoring for TSP edges with dynamic adaptation and multi-hop
    anticipation strategies.
    
    Core Innovations:
    1. Dynamic Regret Thresholding: Adjusts regret sensitivity based on graph 
       sparsity and distance distribution using coefficient of variation.
    2. Two-Hop Neighborhood Anticipation: Evaluates edge quality through the
       connectivity potential of neighboring nodes via secondary adjacency scoring.
    3. Cluster-Aware Edge Prioritization: Uses K-means clustering on distance
       embeddings to differentially score intra-cluster and inter-cluster edges.
    4. Adaptive Factor Balancing: Dynamically weights heuristic components
       based on graph size, density, and sparsity characteristics.
    
    Returns:
        np.ndarray: Heuristic score matrix with same shape as input.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # ========== 1. INITIALIZATION & SPARSITY ANALYSIS ==========
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # Compute graph sparsity metrics
    triu_indices = np.triu_indices(n, k=1)
    distances_flat = dist[triu_indices]
    mean_dist = np.mean(distances_flat)
    std_dist = np.std(distances_flat)
    cv_dist = std_dist / (mean_dist + 1e-12)  # Coefficient of variation
    
    # ========== 2. DYNAMIC REGRET THRESHOLDING ==========
    # Adjust regret sensitivity based on graph sparsity
    regret_sensitivity = 1.5 / (1.0 + cv_dist)  # Less sensitive for sparse graphs
    
    # Compute dynamic regret penalties
    regret_penalty = np.ones_like(dist)
    for i in range(n):
        sorted_dists = np.sort(dist[i])
        if len(sorted_dists) >= 2:
            best, second_best = sorted_dists[0], sorted_dists[1]
            regret_gap = second_best - best
            
            # Dynamic threshold: higher for dense graphs (small cv)
            threshold = regret_gap * (0.5 + 0.5 * cv_dist)
            
            for j in range(n):
                if i != j:
                    regret = dist[i, j] - best
                    if regret > threshold:
                        # Exponential penalty scaled by sensitivity
                        norm_regret = max(0, (regret - threshold) / (regret_gap + 1e-12))
                        penalty = np.exp(-regret_sensitivity * norm_regret)
                        regret_penalty[i, j] = min(regret_penalty[i, j], penalty)
    
    # ========== 3. TWO-HOP NEIGHBORHOOD ANTICIPATION ==========
    # Precompute node centrality and neighborhood metrics
    k_neighbors = max(2, int(np.sqrt(n)))
    neighbor_quality = np.zeros(n)
    
    for i in range(n):
        # Get distances to k nearest neighbors
        neighbor_dists = np.partition(dist[i], k_neighbors)[:k_neighbors+1]
        neighbor_dists = neighbor_dists[neighbor_dists < np.inf]
        neighbor_quality[i] = np.mean(1.0 / (neighbor_dists + 1e-12)) if len(neighbor_dists) > 0 else 0
    
    # Two-hop scoring: edge quality influenced by neighbor connectivity
    two_hop_factor = np.ones_like(dist)
    for i in range(n):
        for j in range(i + 1, n):
            # Consider the best connections from i and j's neighborhoods
            i_neighbors = np.argsort(dist[i])[1:k_neighbors+1]
            j_neighbors = np.argsort(dist[j])[1:k_neighbors+1]
            
            # Average neighbor quality for both endpoints
            avg_i_quality = np.mean(neighbor_quality[i_neighbors]) if len(i_neighbors) > 0 else 0
            avg_j_quality = np.mean(neighbor_quality[j_neighbors]) if len(j_neighbors) > 0 else 0
            
            # Boost edges that connect to high-quality neighborhoods
            neighborhood_boost = 1.0 + 0.2 * (avg_i_quality + avg_j_quality)
            two_hop_factor[i, j] = two_hop_factor[j, i] = neighborhood_boost
    
    # ========== 4. CLUSTER-AWARE EDGE PRIORITIZATION ==========
    # Create distance-based embeddings for clustering
    scaled_dist = dist.copy()
    scaled_dist[scaled_dist == np.inf] = np.max(scaled_dist[scaled_dist < np.inf])
    
    # Determine optimal number of clusters
    n_clusters = max(2, min(5, n // 4))
    if n >= n_clusters * 2:
        try:
            # Use spectral embedding for better distance-based clustering
            from sklearn.manifold import SpectralEmbedding
            embedder = SpectralEmbedding(n_components=min(10, n-1), 
                                        affinity='precomputed')
            embeddings = embedder.fit_transform(1.0 / (scaled_dist + 1e-12))
            
            # Apply K-means on embeddings
            kmeans = KMeans(n_clusters=n_clusters, random_state=42, n_init=10)
            clusters = kmeans.fit_predict(embeddings)
        except:
            # Fallback to distance-based clustering
            from sklearn.cluster import AgglomerativeClustering
            clustering = AgglomerativeClustering(n_clusters=n_clusters, 
                                                affinity='precomputed',
                                                linkage='average')
            clusters = clustering.fit_predict(scaled_dist)
        
        # Assign cluster bonuses: intra-cluster edges get moderate boost,
        # critical inter-cluster edges (connecting to nearest cluster) get strong boost
        cluster_bonus = np.ones_like(dist)
        for i in range(n):
            for j in range(i + 1, n):
                if clusters[i] == clusters[j]:
                    # Intra-cluster: moderate bonus
                    cluster_bonus[i, j] = cluster_bonus[j, i] = 1.2
                else:
                    # Check if this is a near inter-cluster connection
                    i_cluster_mask = (clusters == clusters[i])
                    j_cluster_mask = (clusters == clusters[j])
                    
                    # Distance to nearest node in other cluster
                    i_to_j_cluster = np.min(dist[i, j_cluster_mask])
                    j_to_i_cluster = np.min(dist[j, i_cluster_mask])
                    
                    # Boost if this edge is among the best inter-cluster connections
                    if (dist[i, j] <= 1.2 * i_to_j_cluster and 
                        dist[i, j] <= 1.2 * j_to_i_cluster):
                        cluster_bonus[i, j] = cluster_bonus[j, i] = 1.5
    else:
        cluster_bonus = np.ones_like(dist)
    
    # ========== 5. ADAPTIVE FACTOR BALANCING ==========
    # Adjust component weights based on graph properties
    density_factor = n / (mean_dist + 1e-12) if mean_dist > 0 else 1.0
    size_factor = np.log(n + 1)
    
    # Dynamic weights for different components
    w_base = 1.0
    w_regret = 0.8 + 0.4 * (1.0 - cv_dist)  # Higher weight for dense graphs
    w_neighbor = 0.5 + 0.3 * size_factor / 10  # Increases with graph size
    w_cluster = 0.7 + 0.3 * (cv_dist)  # Higher weight for sparse graphs
    
    # Combine components with adaptive weights
    base_inv = 1.0 / (dist + 1e-12)
    scores = (base_inv ** w_base) * \
             (regret_penalty ** w_regret) * \
             (two_hop_factor ** w_neighbor) * \
             (cluster_bonus ** w_cluster)
    
    # ========== 6. MST CONNECTIVITY ENFORCEMENT ==========
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    scores[mst_mask] *= 1.8  # Strong reinforcement for MST edges
    
    # ========== 7. ADAPTIVE SPARSIFICATION ==========
    # Dynamic edge retention based on graph characteristics
    base_keep = int(np.sqrt(n) * (1.0 + cv_dist))
    k_keep = max(3, min(base_keep, n // 2))
    
    final_mask = np.zeros_like(scores, dtype=bool)
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        top_indices = np.argsort(row_scores)[-k_keep:]
        final_mask[i, top_indices] = True
    
    # Ensure symmetry and MST inclusion
    final_mask = final_mask | final_mask.T | mst_mask
    
    # Apply mask
    scores = scores * final_mask
    
    # ========== 8. POST-PROCESSING ==========
    # Normalize scores
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    np.fill_diagonal(scores, 0.0)
    
    return scores
