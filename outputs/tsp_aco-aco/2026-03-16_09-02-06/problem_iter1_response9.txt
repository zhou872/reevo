import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge priority heuristic combining multiple graph-based features.
    Returns a sparse matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (handling zeros)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Spectral graph centrality (Fiedler vector)
    if n > 2:
        # Create normalized Laplacian
        D = np.diag(np.sum(distance_matrix, axis=1))
        L = D - distance_matrix
        # Use pseudo-inverse for stability
        try:
            eigvals, eigvecs = np.linalg.eigh(L)
            # Get Fiedler vector (second smallest eigenvalue)
            idx = np.argsort(eigvals)[1]
            fiedler = np.abs(eigvecs[:, idx])
            # Edge centrality: product of endpoint centralities
            centrality = np.outer(fiedler, fiedler)
        except np.linalg.LinAlgError:
            centrality = np.ones((n, n))
    else:
        centrality = np.ones((n, n))
    
    # 3. Local neighborhood sparsity factor
    # Compute average distance to k nearest neighbors
    k = max(2, int(np.log2(n)))
    sorted_dists = np.partition(distance_matrix, k, axis=1)[:, :k+1]
    local_density = np.mean(sorted_dists, axis=1, keepdims=True)
    # Edges connecting sparse regions are more promising
    sparsity_bonus = np.sqrt(np.outer(local_density, local_density))
    
    # 4. Two-hop path feasibility
    # For each edge (i,j), look for best intermediate node k
    two_hop = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Find best intermediate node (excluding i and j)
            mask = np.ones(n, dtype=bool)
            mask[[i, j]] = False
            if np.any(mask):
                best_indirect = np.min(
                    distance_matrix[i, mask] + distance_matrix[mask, j]
                )
                # Penalty if direct edge is much worse than 2-hop
                regret = max(0, best_indirect - distance_matrix[i, j])
                two_hop[i, j] = np.exp(-regret / distance_matrix[i, j]) if distance_matrix[i, j] > 0 else 0
    
    # 5. Combine factors with weights
    heuristics = (
        0.4 * inv_dist +
        0.2 * centrality +
        0.2 * sparsity_bonus / (sparsity_bonus.mean() + 1e-8) +
        0.2 * two_hop
    )
    
    # 6. Adaptive sparsification
    # Keep only top edges per node
    keep_per_node = max(3, int(np.sqrt(n)))
    mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Exclude self
        top_indices = np.argpartition(row, -keep_per_node)[-keep_per_node:]
        mask[i, top_indices] = True
    
    # Symmetrize mask (if edge is kept in one direction, keep both)
    mask = mask | mask.T
    
    # Apply sparsification
    heuristics[~mask] = 0
    
    # 7. Normalize by node degree (prefer balanced connections)
    row_sums = heuristics.sum(axis=1, keepdims=True)
    heuristics = np.divide(heuristics, row_sums + 1e-8, where=row_sums > 0)
    
    return heuristics
