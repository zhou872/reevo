import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with symmetric regret computation.
    
    Improvements over v1:
    1. Symmetric regret calculation using average of both node perspectives
    2. Adaptive sparsification with connectivity preservation
    3. Enhanced MST bonus based on edge importance in spanning tree
    
    Args:
        distance_matrix: Symmetric n×n matrix of distances between nodes
        
    Returns:
        n×n matrix of edge scores (0-1 normalized)
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Prepare distance matrix (treat diagonal as infinite distance)
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    
    # 2. Base inverse distance with smoothing
    inv_dist = 1 / (dist + 1e-10)
    
    # 3. MST edge bonus - edges in MST get higher priority
    mst = minimum_spanning_tree(dist)
    mst_edges = (mst.toarray() > 0)
    
    # Enhanced MST bonus: give stronger bonus to critical edges in small MSTs
    mst_weight = 2.0 + np.log(max(1, n / 10))  # Adjust based on graph size
    mst_bonus = np.ones_like(dist)
    mst_bonus[mst_edges] = mst_weight
    
    # 4. Symmetric regret calculation
    regret_matrix = np.zeros_like(dist)
    
    # Compute best and second-best distances for each node
    best_dists = np.zeros(n)
    second_best_dists = np.zeros(n)
    
    for i in range(n):
        distances_i = dist[i].copy()
        distances_i[i] = np.inf
        sorted_dists = np.sort(distances_i)
        best_dists[i] = sorted_dists[0]
        second_best_dists[i] = sorted_dists[1]
    
    # Compute symmetric regret for each edge
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            # Regret from i's perspective
            regret_i = (dist[i, j] - best_dists[i]) / max(1e-10, second_best_dists[i] - best_dists[i])
            
            # Regret from j's perspective  
            regret_j = (dist[j, i] - best_dists[j]) / max(1e-10, second_best_dists[j] - best_dists[j])
            
            # Symmetric average regret
            avg_regret = (regret_i + regret_j) / 2
            
            # Convert to penalty factor (lower regret = higher score)
            regret_matrix[i, j] = np.exp(-avg_regret)
    
    # 5. Combined scoring
    scores = inv_dist * mst_bonus * regret_matrix
    
    # 6. Adaptive sparsification with connectivity guarantee
    # Determine number of edges to keep per node
    if n <= 10:
        k = max(2, n - 2)  # Keep most edges for small graphs
    else:
        k = max(3, int(np.sqrt(n)))  # Adaptive k based on graph size
    
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        # Get top k edges for this node
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf  # Exclude self
        
        # Use argpartition for efficiency
        top_k_idx = np.argpartition(row_scores, -k)[-k:]
        mask[i, top_k_idx] = True
        mask[top_k_idx, i] = True  # Maintain symmetry
    
    # Ensure MST edges are always included
    mask = mask | mst_edges
    
    # Apply sparsification while preserving symmetry
    scores = scores * mask
    
    # 7. Normalize scores to [0, 1] range
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    # Set diagonal to zero
    np.fill_diagonal(scores, 0)
    
    return scores
