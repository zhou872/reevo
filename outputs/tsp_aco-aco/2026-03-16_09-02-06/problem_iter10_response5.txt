import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with dynamic regret thresholds and
    two-step lookahead for better edge utility prediction.
    
    Features:
    1. Dynamic regret thresholds based on node density
    2. Two-step lookahead considering neighbor distances
    3. Adaptive sparsification preserving connectivity
    4. MST bonus for fundamental edges
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create working copy with diagonal set to infinity
    D = distance_matrix.copy().astype(float)
    np.fill_diagonal(D, np.inf)
    
    # 1. Base inverse distance with adaptive smoothing
    eps = np.finfo(float).eps * 10
    inv_dist = 1 / (D + eps)
    
    # 2. MST connectivity bonus (edges in MST get 1.5x boost)
    mst = minimum_spanning_tree(D)
    mst_bonus = np.ones_like(D)
    mst_mask = (mst.toarray() > 0)
    mst_bonus[mst_mask] = 1.5
    
    # 3. Dynamic density-aware regret penalty
    # Compute local density for each node (inverse of average distance to neighbors)
    local_density = np.zeros(n)
    for i in range(n):
        valid_dists = D[i, D[i] < np.inf]
        if len(valid_dists) > 0:
            local_density[i] = 1 / (np.mean(valid_dists) + eps)
    
    regret_matrix = np.ones_like(D)
    for i in range(n):
        dists_i = D[i].copy()
        dists_i[i] = np.inf
        
        # Find best two connections from node i
        mask_i = dists_i < np.inf
        if mask_i.sum() >= 2:
            valid_dists = dists_i[mask_i]
            sorted_dists = np.sort(valid_dists)
            best_i, second_best_i = sorted_dists[:2]
            
            # Dynamic regret threshold based on local density
            # Denser regions have stricter regret thresholds
            density_factor = 1.0 + local_density[i] / (np.mean(local_density) + eps)
            regret_threshold = 0.1 * density_factor
            
            for j in range(n):
                if i != j and D[i, j] < np.inf:
                    # Normalized regret from i's perspective
                    raw_regret = (D[i, j] - best_i) / (second_best_i - best_i + eps)
                    
                    # Apply dynamic threshold with sigmoid smoothing
                    regret_penalty = 1 / (1 + np.exp(10 * (raw_regret - regret_threshold)))
                    regret_matrix[i, j] *= regret_penalty
    
    # 4. Two-step lookahead cost adjustment
    lookahead_matrix = np.ones_like(D)
    for i in range(n):
        for j in range(n):
            if i != j and D[i, j] < np.inf:
                # Consider best continuation from node j (excluding return to i)
                dists_j = D[j].copy()
                dists_j[i] = np.inf  # Exclude the edge back to i
                dists_j[j] = np.inf
                
                mask_j = dists_j < np.inf
                if mask_j.sum() > 0:
                    # Find best continuation from j
                    best_from_j = np.min(dists_j[mask_j])
                    
                    # Two-step cost: current edge + best continuation
                    two_step_cost = D[i, j] + best_from_j
                    
                    # Normalize by average two-step cost in the graph
                    avg_two_step = np.mean(D[D < np.inf]) * 2
                    lookahead_factor = 1.0 / (1.0 + two_step_cost / (avg_two_step + eps))
                    lookahead_matrix[i, j] = lookahead_factor
    
    # 5. Combined scoring with all components
    scores = inv_dist * mst_bonus * regret_matrix * lookahead_matrix
    
    # 6. Adaptive sparsification with connectivity guarantee
    if n > 3:
        # Dynamic sparsification: keep more edges for sparse areas
        avg_dist = np.mean(D[D < np.inf])
        base_k = max(2, int(np.log2(n)) + 1)
        
        # Adjust k based on graph sparsity
        sparsity_factor = avg_dist / (np.max(D[D < np.inf]) + eps)
        k = min(n - 1, max(base_k, int(base_k * (1 + sparsity_factor))))
        
        mask = np.zeros_like(scores, dtype=bool)
        for i in range(n):
            row_scores = scores[i].copy()
            row_scores[i] = -np.inf
            
            # Ensure at least k edges if possible
            valid_edges = np.sum(row_scores > -np.inf)
            current_k = min(k, valid_edges)
            
            if current_k > 0:
                top_k_idx = np.argpartition(row_scores, -current_k)[-current_k:]
                mask[i, top_k_idx] = True
        
        # Ensure symmetry and MST connectivity
        mask = mask | mask.T | mst_mask
        scores = scores * mask
    
    # 7. Final normalization
    max_score = np.max(scores[scores > 0]) if np.any(scores > 0) else 1.0
    if max_score > 0:
        scores = scores / max_score
    
    np.fill_diagonal(scores, 0)
    return scores
