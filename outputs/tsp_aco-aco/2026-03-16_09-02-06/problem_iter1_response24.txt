import numpy as np
from scipy.spatial import ConvexHull

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple geometric and graph-based factors.
    Returns a sparse matrix where only promising edges have non-zero scores.
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) / (distance_matrix + np.eye(n))
    
    # 1. Base inverse distance (handle zeros on diagonal)
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    np.fill_diagonal(inv_dist, 0)
    mask = distance_matrix != 0
    inv_dist[mask] = 1.0 / distance_matrix[mask]
    
    # 2. Minimum Spanning Tree bonus (Prim's algorithm)
    mst_bonus = np.zeros((n, n))
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    for _ in range(n - 1):
        # Find minimum edge from visited to unvisited
        min_val = np.inf
        u, v = -1, -1
        for i in np.where(visited)[0]:
            for j in np.where(~visited)[0]:
                if distance_matrix[i, j] < min_val:
                    min_val = distance_matrix[i, j]
                    u, v = i, j
        # Mark edge as part of MST
        mst_bonus[u, v] = mst_bonus[v, u] = 2.0
        visited[v] = True
    
    # 3. Convex hull boundary weighting
    hull_weight = np.ones((n, n))
    try:
        # Generate approximate coordinates using multidimensional scaling
        from sklearn.manifold import MDS
        mds = MDS(n_components=2, dissimilarity='precomputed', random_state=42)
        coords = mds.fit_transform(distance_matrix)
        hull = ConvexHull(coords)
        hull_vertices = hull.vertices
        # Edges on convex hull get bonus
        for i in range(len(hull_vertices)):
            v1, v2 = hull_vertices[i], hull_vertices[(i+1)%len(hull_vertices)]
            hull_weight[v1, v2] = hull_weight[v2, v1] = 1.5
    except:
        pass  # Fallback if convex hull fails
    
    # 4. Distance-median ratio scoring
    median_dist = np.median(distance_matrix[~np.eye(n, dtype=bool)])
    dist_ratio = median_dist / (distance_matrix + 1e-10)
    dist_ratio = np.clip(dist_ratio, 0.2, 5.0)  # Bound the ratio
    
    # 5. Nearest neighbor sparsification
    nn_mask = np.zeros((n, n), dtype=bool)
    k_neighbors = max(2, min(5, n//4))  # Adaptive k
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        neighbors = np.argsort(distance_matrix[i])[1:k_neighbors+1]
        nn_mask[i, neighbors] = True
    nn_mask = nn_mask | nn_mask.T  # Make symmetric
    
    # Combine all factors
    scores = inv_dist * mst_bonus * hull_weight * dist_ratio
    
    # Apply sparsification: zero out non-promising edges
    scores[~nn_mask] = 0
    
    # Normalize non-zero scores
    non_zero_mask = scores > 0
    if np.any(non_zero_mask):
        scores[non_zero_mask] /= np.max(scores[non_zero_mask])
    
    # Ensure symmetry and zero diagonal
    scores = (scores + scores.T) / 2
    np.fill_diagonal(scores, 0)
    
    return scores
