import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges combining multiple factors:
    1. Inverse distance (base attractiveness)
    2. Mutual regret penalization (edges poorly ranked by both endpoints)
    3. Hybrid sparsification (local top-m + global top-p)
    4. Minimum Spanning Tree connectivity guarantee
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # 1. Base inverse distance with smoothing
    distance_matrix = distance_matrix.copy()
    np.fill_diagonal(distance_matrix, np.inf)
    inv_dist = 1 / (distance_matrix + 1e-10)
    
    # 2. Compute MST bonus (edges in MST get 2x multiplier)
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0)
    mst_bonus = np.ones_like(distance_matrix)
    mst_bonus[mst_mask] = 2.0
    
    # 3. Mutual regret penalization
    regret_penalty = np.ones_like(distance_matrix)
    
    # For each node, identify edges in top 30% by inverse distance
    top_threshold_percent = 30
    inv_dist_no_diag = inv_dist.copy()
    np.fill_diagonal(inv_dist_no_diag, -np.inf)
    
    for i in range(n):
        # Get inverse distances for node i
        row_inv = inv_dist_no_diag[i].copy()
        
        if len(row_inv[row_inv > -np.inf]) == 0:
            continue
            
        # Calculate threshold for top 30%
        valid_values = row_inv[row_inv > -np.inf]
        threshold = np.percentile(valid_values, 100 - top_threshold_percent)
        
        # Mark edges below threshold for this node
        below_threshold_mask = (row_inv < threshold) & (row_inv > -np.inf)
        regret_penalty[i, below_threshold_mask] *= 0.1
    
    # Apply symmetry for mutual penalization
    # An edge gets penalized only if it's below threshold for BOTH endpoints
    mutual_bad_mask = (regret_penalty < 0.2) & (regret_penalty.T < 0.2)
    regret_penalty[mutual_bad_mask] = 0.1
    regret_penalty[~mutual_bad_mask] = 1.0
    
    # 4. Combined scoring (before sparsification)
    scores = inv_dist * mst_bonus * regret_penalty
    
    # 5. Hybrid sparsification
    # Local: top m edges per node
    m = max(2, int(np.sqrt(n)))
    # Global: top p percentile of all edges
    p_percentile = 30
    
    # Create sparsification mask
    mask = np.zeros_like(scores, dtype=bool)
    
    # Local top-m sparsification
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        top_m_idx = np.argsort(row_scores)[-m:]
        mask[i, top_m_idx] = True
        mask[top_m_idx, i] = True  # Maintain symmetry
    
    # Global top-p percentile sparsification
    # Flatten scores (excluding diagonal and already masked edges)
    flat_scores = scores[~np.eye(n, dtype=bool) & ~mask]
    if len(flat_scores) > 0:
        global_threshold = np.percentile(flat_scores, 100 - p_percentile)
        global_mask = scores >= global_threshold
        mask = mask | global_mask
    
    # Ensure MST edges are always included
    mask = mask | mst_mask
    
    # Apply sparsification
    scores = scores * mask
    
    # 6. Normalize to [0, 1] range
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    np.fill_diagonal(scores, 0)  # Zero out diagonal
    return scores
