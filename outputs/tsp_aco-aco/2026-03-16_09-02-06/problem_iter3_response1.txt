import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with improved strategies:
    1. Inverse distance as base attractiveness
    2. Mutual top-k penalization (edges must be in top percentage for both endpoints)
    3. Hybrid sparsification (top m per node AND global top p percentile)
    4. MST edges forced for connectivity guarantee
    
    Returns a matrix of edge scores with same shape as input.
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # Create a copy to avoid modifying the original
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)  # Remove self-loops
    
    # 1. Base inverse distance with smoothing
    inv_dist = 1 / (dist + 1e-10)
    
    # 2. Mutual top-k penalization
    # For each node, identify edges that are in top percentage for BOTH endpoints
    top_percentage = 0.3  # Top 30%
    k_per_node = max(1, int(top_percentage * (n - 1)))
    
    # Initialize penalty matrix (1 = full weight, 0.1 = penalized)
    penalty_matrix = np.ones_like(inv_dist)
    
    for i in range(n):
        # Get sorted indices for node i (descending by inverse distance)
        i_sorted = np.argsort(inv_dist[i])[::-1]
        # Only consider non-self edges
        i_top_set = set(i_sorted[:k_per_node])
        
        for j in range(i + 1, n):
            # Check if edge (i,j) is in top k for node j
            j_sorted = np.argsort(inv_dist[j])[::-1]
            j_top_set = set(j_sorted[:k_per_node])
            
            # Penalize if edge is NOT in top percentage for BOTH nodes
            if (i not in j_top_set) or (j not in i_top_set):
                penalty_matrix[i, j] = 0.1
                penalty_matrix[j, i] = 0.1
    
    # Apply penalty
    scores = inv_dist * penalty_matrix
    
    # 3. Hybrid sparsification
    m = max(2, int(np.sqrt(n)))  # Local: top m per node
    p = 0.2  # Global: top 20% percentile
    
    # Local mask: top m edges per node
    local_mask = np.zeros_like(scores, dtype=bool)
    for i in range(n):
        row = scores[i].copy()
        row[i] = -np.inf  # Exclude self
        top_m_indices = np.argpartition(row, -m)[-m:]
        local_mask[i, top_m_indices] = True
        # Symmetric
        local_mask[top_m_indices, i] = True
    
    # Global mask: top p percentile of all edges
    # Get upper triangle values (avoid double counting)
    upper_tri = scores[np.triu_indices(n, k=1)]
    if len(upper_tri) > 0:
        threshold = np.percentile(upper_tri, (1 - p) * 100)
        global_mask = scores >= threshold
        # Ensure symmetry
        global_mask = global_mask | global_mask.T
    else:
        global_mask = np.ones_like(scores, dtype=bool)
    
    # Combine masks
    sparsity_mask = local_mask | global_mask
    
    # 4. MST connectivity guarantee
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    
    # Final mask: union of all masks
    final_mask = sparsity_mask | mst_mask
    
    # Apply final mask
    scores = scores * final_mask
    
    # Normalize to [0, 1] range
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    # Ensure diagonal is zero
    np.fill_diagonal(scores, 0)
    
    return scores
