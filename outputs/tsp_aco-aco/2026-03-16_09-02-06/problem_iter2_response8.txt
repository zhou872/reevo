import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection with regret penalization
    and adaptive spatial features.
    
    Features:
    1. Regret-based edge penalization - prioritizes edges where alternatives are poor
    2. Neighborhood density scaling - lookahead spatial averaging
    3. Adaptive k-NN sparsification - density-aware neighbor selection
    4. Harmonic centrality bonus - node importance for connectivity
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape, with higher values indicating
        more promising edges for inclusion in TSP tour
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    # Avoid division by zero for diagonal
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base inverse distance with farthest penalty
    inv_dist = 1.0 / safe_dist
    max_dist = np.max(safe_dist)
    dist_penalty = 1.0 - (safe_dist / max_dist)
    base_heuristic = inv_dist * dist_penalty
    
    # 2. Regret-based edge penalization
    # For each node, compute regret = 2nd_best_dist - best_dist
    # Edges with higher regret (poor alternatives) get bonus
    regret_matrix = np.zeros_like(base_heuristic)
    
    for i in range(n):
        # Get sorted distances from node i (excluding self)
        sorted_dists = np.sort(safe_dist[i])
        if len(sorted_dists) >= 2:
            best_dist = sorted_dists[0]
            second_best = sorted_dists[1]
            regret_i = second_best - best_dist
            
            # Normalize regret by average distance
            avg_dist_i = np.mean(safe_dist[i])
            norm_regret_i = regret_i / avg_dist_i if avg_dist_i > 0 else 0
            
            # Apply regret bonus to edges from node i
            for j in range(n):
                if i != j:
                    # Edge gets bonus proportional to how good it is relative to alternatives
                    if safe_dist[i, j] == best_dist:
                        regret_matrix[i, j] = 1.0 + 0.5 * norm_regret_i
                    else:
                        # Alternative edges get reduced bonus based on distance ratio
                        alt_ratio = best_dist / safe_dist[i, j] if safe_dist[i, j] > 0 else 0
                        regret_matrix[i, j] = 0.5 + 0.5 * alt_ratio
    
    # 3. Neighborhood density scaling (lookahead spatial averaging)
    # Scale edges by average distance of target nodes to their neighbors
    density_scale = np.zeros_like(base_heuristic)
    
    for i in range(n):
        # Compute average distance from i to its k nearest neighbors
        k_density = min(5, n - 1)
        nearest_dists = np.sort(safe_dist[i])[:k_density]
        avg_density_i = np.mean(nearest_dists) if len(nearest_dists) > 0 else 1.0
        
        for j in range(n):
            if i != j:
                # Compute average density for j
                nearest_dists_j = np.sort(safe_dist[j])[:k_density]
                avg_density_j = np.mean(nearest_dists_j) if len(nearest_dists_j) > 0 else 1.0
                
                # Edges connecting dense regions (low average distance) get higher scores
                joint_density = (avg_density_i + avg_density_j) / 2
                density_scale[i, j] = 1.0 / (1.0 + joint_density)
    
    # Normalize density scale
    max_density_scale = np.max(density_scale)
    if max_density_scale > 0:
        density_scale = density_scale / max_density_scale
    
    # 4. Harmonic centrality bonus
    node_centrality = np.zeros(n)
    for i in range(n):
        # Use harmonic mean for better sensitivity
        with np.errstate(divide='ignore'):
            harmonic_mean = n / np.sum(1.0 / safe_dist[i])
        node_centrality[i] = 1.0 / harmonic_mean if not np.isinf(harmonic_mean) else 0
    
    # Normalize centrality scores
    if np.max(node_centrality) > 0:
        node_centrality = node_centrality / np.max(node_centrality)
    
    centrality_bonus = np.zeros_like(base_heuristic)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Combine centrality of both endpoints
                centrality_bonus[i, j] = 0.3 + 0.7 * (node_centrality[i] + node_centrality[j]) / 2
    
    # 5. Adaptive k-selection for sparsification based on local density
    sparse_mask = np.zeros_like(base_heuristic, dtype=bool)
    
    # Compute local density for each node
    local_density = np.zeros(n)
    for i in range(n):
        # Measure density as inverse of average distance to nearest neighbors
        k_density = min(5, n - 1)
        nearest_dists = np.sort(safe_dist[i])[:k_density]
        local_density[i] = 1.0 / np.mean(nearest_dists) if np.mean(nearest_dists) > 0 else 1.0
    
    # Normalize densities
    if np.max(local_density) > 0:
        local_density = local_density / np.max(local_density)
    
    for i in range(n):
        # Adaptive k based on local density
        base_k = max(2, min(5, n // 4))
        adaptive_k = max(2, int(base_k * (0.5 + 0.5 * local_density[i])))
        
        # Get indices of adaptive_k smallest distances
        nearest = np.argsort(safe_dist[i])[:adaptive_k]
        sparse_mask[i, nearest] = True
    
    # 6. Combine all components
    combined = base_heuristic * regret_matrix * density_scale * centrality_bonus
    
    # 7. Apply adaptive sparsification and enforce symmetry
    heuristic = np.zeros_like(combined)
    
    for i in range(n):
        for j in range(n):
            if i == j:
                heuristic[i, j] = 0
            elif sparse_mask[i, j] or sparse_mask[j, i]:
                # Include edge if it's in adaptive top-k for either endpoint
                # Take geometric mean for symmetry
                heuristic[i, j] = np.sqrt(combined[i, j] * combined[j, i])
                heuristic[j, i] = heuristic[i, j]
            else:
                heuristic[i, j] = 0
    
    # Final normalization
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
