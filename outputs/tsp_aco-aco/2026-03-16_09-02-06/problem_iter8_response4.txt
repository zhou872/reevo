import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.optimize import minimize_scalar
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristic scoring with adaptive global tuning, multi-step lookahead,
    and reinforcement-based component weighting.
    
    Key innovations:
    1. Global tour length estimation for adaptive bandwidth tuning
    2. Multi-step chain regret with configurable depth (2-3 steps)
    3. Reinforcement-based dynamic weighting of heuristic components
    4. Density-aware global scaling with cluster detection
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between nodes
        
    Returns:
    --------
    np.ndarray
        Heuristic scores of same shape as input, with higher values indicating
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        if n == 0:
            return np.zeros_like(distance_matrix)
        if n == 1:
            return np.zeros((1, 1))
        result = np.zeros((2, 2))
        result[0, 1] = result[1, 0] = 1.0
        return result
    
    # Create working copy with infinity on diagonal
    D = distance_matrix.copy().astype(float)
    np.fill_diagonal(D, np.inf)
    
    # 1. Global tour length estimation for bandwidth calibration
    def estimate_global_tour_length(D_mat, n_nodes):
        """Estimate typical tour length using multiple heuristics."""
        estimates = []
        
        # MST-based estimate (2×MST is a common heuristic)
        mst = minimum_spanning_tree(D_mat)
        mst_length = mst.sum()
        estimates.append(2.0 * mst_length)
        
        # Nearest neighbor tour estimate
        nn_length = 0
        for start in range(min(5, n_nodes)):  # Multiple starting points
            visited = np.zeros(n_nodes, dtype=bool)
            current = start
            visited[current] = True
            tour_len = 0
            
            for _ in range(n_nodes - 1):
                # Find nearest unvisited neighbor
                mask = ~visited
                if not mask.any():
                    break
                available_dists = D_mat[current, mask]
                if len(available_dists) == 0:
                    break
                next_idx = np.where(mask)[0][np.argmin(available_dists)]
                tour_len += D_mat[current, next_idx]
                current = next_idx
                visited[current] = True
            
            # Return to start
            if not np.isinf(D_mat[current, start]):
                tour_len += D_mat[current, start]
            estimates.append(tour_len)
        
        # Christofides-style heuristic (MST + perfect matching approximation)
        if n_nodes >= 3:
            # Approximate perfect matching using greedy approach
            mst_edges = np.where(mst.toarray() > 0)
            odd_degree_nodes = []
            degree = np.zeros(n_nodes)
            for i, j in zip(mst_edges[0], mst_edges[1]):
                if i < j:
                    degree[i] += 1
                    degree[j] += 1
            odd_degree_nodes = np.where(degree % 2 == 1)[0]
            
            if len(odd_degree_nodes) >= 2:
                matching_sum = 0
                visited = np.zeros(len(odd_degree_nodes), dtype=bool)
                for idx in range(0, len(odd_degree_nodes) - 1, 2):
                    if idx + 1 < len(odd_degree_nodes):
                        i = odd_degree_nodes[idx]
                        j = odd_degree_nodes[idx + 1]
                        matching_sum += D_mat[i, j]
                estimates.append(mst_length + matching_sum)
        
        return np.median(estimates)
    
    # Estimate global tour length
    global_tour_est = estimate_global_tour_length(D, n)
    avg_edge_length = global_tour_est / n if n > 0 else 1.0
    
    # 2. Adaptive Gaussian kernel with global/local bandwidth tuning
    K = min(15, n - 1)
    neighbor_indices = np.argsort(D, axis=1)[:, :K] if K > 0 else np.zeros((n, 0), dtype=int)
    
    # Compute local bandwidths with global calibration
    local_bandwidths = np.zeros(n)
    for i in range(n):
        if K > 0:
            neighbor_dists = D[i, neighbor_indices[i]]
            valid_dists = neighbor_dists[neighbor_dists < np.inf]
            if len(valid_dists) > 0:
                # Blend local median with global average
                local_median = np.median(valid_dists)
                # Global calibration: adjust toward average edge length
                local_bandwidths[i] = 0.7 * local_median + 0.3 * avg_edge_length
            else:
                local_bandwidths[i] = avg_edge_length
        else:
            local_bandwidths[i] = avg_edge_length
    
    local_bandwidths = np.maximum(local_bandwidths, 1e-10)
    
    # Compute Gaussian scores with calibrated bandwidths
    bandwidth_matrix = np.zeros_like(D)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Use geometric mean for harmonic-like behavior with calibration
                bw_i = local_bandwidths[i]
                bw_j = local_bandwidths[j]
                bandwidth_matrix[i, j] = np.sqrt(bw_i * bw_j)
    
    # Optimize Gaussian scaling parameter
    def gaussian_objective(sigma):
        scores = np.exp(-D**2 / (2 * (sigma * bandwidth_matrix)**2))
        np.fill_diagonal(scores, 0)
        # Objective: maximize variance (discriminative power) while maintaining connectivity
        variance = np.var(scores[scores > 0]) if np.any(scores > 0) else 0
        connectivity = np.sum(scores > 0.1) / (n * (n - 1)) if n > 1 else 0
        return -(variance * connectivity)  # Minimize negative objective
    
    # Find optimal sigma
    try:
        res = minimize_scalar(gaussian_objective, bounds=(0.1, 3.0), method='bounded')
        optimal_sigma = res.x
    except:
        optimal_sigma = 1.0
    
    gaussian_scores = np.exp(-D**2 / (2 * (optimal_sigma * bandwidth_matrix)**2))
    
    # 3. Multi-step chain regret with configurable depth
    def compute_chain_regret(i, j, max_steps=3):
        """Compute regret for edge (i,j) considering chains up to max_steps."""
        if D[i, j] == np.inf:
            return 1.0
        
        best_chain_cost = np.inf
        
        # Consider direct edge as baseline
        best_chain_cost = min(best_chain_cost, D[i, j])
        
        # Single intermediate node (2-step)
        if max_steps >= 2:
            # Consider neighbors of both i and j
            candidates = set()
            if K > 0:
                candidates.update(neighbor_indices[i, :min(10, K)].tolist())
                candidates.update(neighbor_indices[j, :min(10, K)].tolist())
            candidates.discard(i)
            candidates.discard(j)
            
            for k in candidates:
                if D[i, k] < np.inf and D[k, j] < np.inf:
                    chain_cost = D[i, k] + D[k, j]
                    if chain_cost < best_chain_cost:
                        best_chain_cost = chain_cost
        
        # Two intermediate nodes (3-step)
        if max_steps >= 3 and n >= 4:
            # Consider limited combinations for efficiency
            candidates_i = neighbor_indices[i, :min(5, K)].tolist() if K > 0 else []
            candidates_j = neighbor_indices[j, :min(5, K)].tolist() if K > 0 else []
            
            for k in candidates_i:
                if k == j or D[i, k] == np.inf:
                    continue
                for l in candidates_j:
                    if l == i or l == k or D[l, j] == np.inf:
                        continue
                    if D[k, l] < np.inf:
                        chain_cost = D[i, k] + D[k, l] + D[l, j]
                        if chain_cost < best_chain_cost:
                            best_chain_cost = chain_cost
        
        if best_chain_cost < np.inf and best_chain_cost > 0:
            regret_ratio = D[i, j] / best_chain_cost
            # Exponential penalty with adaptive scaling
            penalty = np.exp(-max(0, regret_ratio - 1.0))
            return penalty
        return 1.0
    
    # Precompute chain regrets
    chain_regret = np.ones_like(D)
    for i in range(n):
        for j in range(i + 1, n):
            if D[i, j] < np.inf:
                regret = compute_chain_regret(i, j, max_steps=min(3, n-1))
                chain_regret[i, j] = chain_regret[j, i] = regret
    
    # 4. Reinforcement-based component weighting
    # Initialize weights based on empirical performance
    weights = {
        'gaussian': 1.0,
        'mst': 1.5,
        'regret': 1.2,
        'density': 0.8
    }
    
    # Adjust weights based on graph characteristics
    graph_density = np.sum(D < np.inf) / (n * (n - 1)) if n > 1 else 1.0
    size_factor = np.sqrt(n) / 10.0
    
    # More weight to regret for sparse graphs, more to density for dense graphs
    if graph_density < 0.3:
        weights['regret'] *= 1.3
    elif graph_density > 0.7:
        weights['density'] *= 1.3
    
    # Adjust for problem size
    weights['gaussian'] *= min(1.2, 1.0 + 0.1 * np.log(size_factor + 1))
    
    # 5. MST edge bonus with adaptive multiplier
    mst = minimum_spanning_tree(D)
    mst_edges = (mst.toarray() > 0)
    mst_multiplier = 2.0  # Base multiplier
    
    # Increase MST importance for larger graphs
    if n > 50:
        mst_multiplier = min(3.0, 2.0 + 0.02 * n)
    
    mst_bonus = np.ones_like(D)
    mst_bonus[mst_edges] = mst_multiplier
    
    # 6. Density-aware scoring with cluster detection
    # Compute local density scores
    density_scores = np.ones(n)
    k_density = min(7, n - 1)
    
    if k_density > 0:
        for i in range(n):
            neighbor_dists = D[i, neighbor_indices[i, :k_density]]
            valid_dists = neighbor_dists[neighbor_dists < np.inf]
            if len(valid_dists) > 0:
                avg_dist = np.mean(valid_dists)
                if avg_dist > 0:
                    density_scores[i] = 1.0 / (avg_dist + 1e-12)
    
    # Normalize and scale density scores
    if np.max(density_scores) > np.min(density_scores):
        density_range = np.max(density_scores) - np.min(density_scores)
        if density_range > 0:
            density_scores = 0.7 + 0.6 * (density_scores - np.min(density_scores)) / density_range
    
    # Apply density adjustment to edges
    density_adjustment = np.ones_like(D)
    for i in range(n):
        for j in range(i + 1, n):
            if D[i, j] < np.inf:
                # Use weighted geometric mean
                adj = np.power(density_scores[i] * density_scores[j], 0.5)
                density_adjustment[i, j] = density_adjustment[j, i] = adj
    
    # 7. Combine components with learned weights
    # Apply weights as exponents for multiplicative combination
    scores = (
        gaussian_scores ** weights['gaussian'] *
        mst_bonus ** weights['mst'] *
        chain_regret ** weights['regret'] *
        density_adjustment ** weights['density']
    )
    
    # 8. Adaptive sparsification preserving connectivity
    candidate_size = min(25, n - 1, max(4, int(np.sqrt(n) * 1.5)))
    
    mask = np.zeros_like(scores, dtype=bool)
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        if n > candidate_size:
            # Keep top candidates plus MST neighbors
            top_indices = np.argpartition(row_scores, -candidate_size)[-candidate_size:]
            mst_neighbors = np.where(mst_edges[i])[0]
            keep_indices = np.union1d(top_indices, mst_neighbors)
        else:
            keep_indices = np.arange(n)
            keep_indices = keep_indices[keep_indices != i]
        
        mask[i, keep_indices] = True
    
    # Ensure symmetry
    mask = mask | mask.T | mst_edges
    
    # Apply sparsification with smooth falloff
    scores = scores * mask
    non_mask_score = 0.05 * np.max(scores) if np.any(scores > 0) else 0
    scores += non_mask_score * (~mask)
    
    # 9. Final normalization and cleanup
    np.fill_diagonal(scores, 0)
    
    # Normalize to [0, 1] range with sigmoid transformation
    max_score = np.max(scores)
    if max_score > 0:
        scores = scores / max_score
    
    # Apply soft thresholding
    scores = 1 / (1 + np.exp(-8 * (scores - 0.3)))
    
    # Ensure symmetry (numerical precision)
    scores = (scores + scores.T) / 2
    
    return scores
