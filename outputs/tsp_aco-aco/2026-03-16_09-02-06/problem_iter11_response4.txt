import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with combined scoring strategies.
    
    Core improvements:
    1. Dynamic regret thresholding based on distance distribution
    2. Average-based two-step lookahead with k-nearest neighbors
    3. Two-step spatial scoring considering alternative connections
    4. MST reinforcement with adaptive sparsification
    
    Parameters
    ----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances between n nodes
        
    Returns
    -------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # Working copy with diagonal set to infinity
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # ========== 1. PROBLEM CHARACTERISTICS ==========
    # Extract upper triangular distances for statistics
    triu_indices = np.triu_indices(n, k=1)
    flat_dists = dist[triu_indices]
    
    # Global statistics
    global_avg = np.mean(flat_dists)
    global_std = np.std(flat_dists)
    cv = global_std / (global_avg + 1e-12)  # Coefficient of variation
    
    # ========== 2. BASE INVERSE DISTANCE ==========
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.where(dist < np.inf, 1 / (dist + 1e-12), 0)
    
    # ========== 3. DYNAMIC REGRET THRESHOLDING ==========
    # Adjust regret sensitivity based on graph sparsity (CV)
    # Higher CV means more varied distances -> higher regret threshold
    regret_threshold = 0.3 + 0.5 * np.tanh(cv * 2)  # Ranges ~0.3-0.8
    
    regret_matrix = np.ones_like(dist)
    
    for i in range(n):
        dist_i = dist[i].copy()
        
        # Find best and second-best distances for each node
        # Use more neighbors for robust statistics in sparse graphs
        k_regret = min(7, n - 1)
        sorted_indices = np.argsort(dist_i)
        valid_indices = sorted_indices[dist_i[sorted_indices] < np.inf]
        
        if len(valid_indices) >= 2:
            # Get k nearest distances for robust statistics
            k_nearest = min(k_regret, len(valid_indices))
            neighbor_dists = dist_i[valid_indices[:k_nearest]]
            
            best_dist = neighbor_dists[0]
            
            # Dynamic threshold calculation
            if len(neighbor_dists) >= 3:
                # Use median of remaining distances as threshold reference
                second_best_ref = np.median(neighbor_dists[1:])
            else:
                second_best_ref = neighbor_dists[1] if len(neighbor_dists) > 1 else best_dist
            
            # Calculate gap for thresholding
            gap = second_best_ref - best_dist
            
            for j in range(n):
                if i != j and dist_i[j] < np.inf:
                    if gap > 1e-12:
                        # Normalized regret
                        regret = (dist_i[j] - best_dist) / gap
                        
                        # Apply exponential penalty only when regret exceeds threshold
                        if regret > regret_threshold:
                            # Stronger penalty for larger regrets
                            penalty_strength = 1.5 + 0.5 * cv
                            regret_matrix[i, j] = np.exp(-penalty_strength * (regret - regret_threshold))
                        else:
                            # Mild penalty or no penalty for edges near optimal
                            regret_matrix[i, j] = 1.0 - 0.3 * (regret / regret_threshold)
                    else:
                        # Minimal variation case
                        regret_matrix[i, j] = 0.8  # Slight penalty for uniform distances
    
    # Symmetrize regret matrix
    regret_symmetric = np.sqrt(regret_matrix * regret_matrix.T)
    
    # ========== 4. AVERAGE-BASED TWO-STEP LOOKAHEAD ==========
    avg_lookahead = np.ones_like(dist)
    
    for i in range(n):
        for j in range(i + 1, n):
            if dist[i, j] < np.inf:
                # Consider j's perspective after taking edge (i,j)
                dist_j = dist[j].copy()
                dist_j[i] = np.inf  # Exclude the edge we came from
                
                # Find k nearest neighbors of j (excluding i)
                k_lookahead = min(5, n - 2)
                valid_j = dist_j[dist_j < np.inf]
                
                if len(valid_j) >= k_lookahead:
                    # Get k nearest distances from j
                    j_nearest = np.partition(valid_j, k_lookahead - 1)[:k_lookahead]
                    avg_next_dist = np.mean(j_nearest)
                    
                    # Normalize by global average
                    if global_avg > 0:
                        normalized_avg = avg_next_dist / global_avg
                        # Penalize inversely: smaller avg_next_dist is better
                        avg_lookahead[i, j] = avg_lookahead[j, i] = 1.0 / (1.0 + normalized_avg)
    
    # ========== 5. TWO-STEP SPATIAL SCORING ==========
    spatial_scores = np.ones_like(dist)
    
    for i in range(n):
        # Pre-compute best alternative for each node
        dist_i_full = dist[i].copy()
        best_alt_i = np.inf
        if np.any(dist_i_full < np.inf):
            # Best alternative connection from i (excluding self)
            mask_i = dist_i_full < np.inf
            if np.any(mask_i):
                best_alt_i = np.min(dist_i_full[mask_i])
    
    for i in range(n):
        dist_i = dist[i].copy()
        if np.any(dist_i < np.inf):
            best_alt_i_val = np.min(dist_i[dist_i < np.inf])
            
            for j in range(i + 1, n):
                if dist[i, j] < np.inf:
                    dist_j = dist[j].copy()
                    dist_j[i] = np.inf
                    
                    if np.any(dist_j < np.inf):
                        # Best alternative from j (excluding i)
                        best_alt_j = np.min(dist_j[dist_j < np.inf])
                        
                        # Combine immediate distance with alternative connections
                        # Weight factor for alternative connections (e.g., 0.3)
                        alt_weight = 0.3
                        
                        # Normalize by global average
                        if global_avg > 0:
                            immediate_norm = dist[i, j] / global_avg
                            alt_norm = (best_alt_i_val + best_alt_j) / (2 * global_avg)
                            
                            # Score favors smaller combined value
                            combined = immediate_norm + alt_weight * alt_norm
                            spatial_scores[i, j] = spatial_scores[j, i] = 1.0 / (1.0 + combined)
    
    # ========== 6. COMBINE LOOKAHEAD SCORES ==========
    lookahead_combined = avg_lookahead * spatial_scores
    
    # ========== 7. MST REINFORCEMENT ==========
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Adaptive MST bonus: stronger for smaller problems
    mst_bonus = 1.4 + 0.6 * np.tanh(15 / n)
    
    # ========== 8. COMBINE ALL SCORES ==========
    # Weight the components
    regret_weight = 0.4 + 0.3 * np.tanh(cv)  # 0.4-0.7
    lookahead_weight = min(0.35, 2.0 / np.sqrt(n))  # 0.1-0.35
    
    base_scores = inv_dist * (
        (1 - regret_weight - lookahead_weight) + 
        regret_weight * regret_symmetric + 
        lookahead_weight * lookahead_combined
    )
    
    # Apply MST bonus
    base_scores[mst_mask] *= mst_bonus
    
    # ========== 9. INTELLIGENT SPARSIFICATION ==========
    # Dynamic degree targets based on problem size and distance distribution
    min_degree = max(2, int(np.log2(n + 1)))
    max_degree = min(12, n // 3 + 2)
    
    # Node-specific degree targets
    centrality = np.zeros(n)
    for i in range(n):
        valid_dists = dist[i][dist[i] < np.inf]
        if len(valid_dists) > 0:
            # Nodes with smaller average distances are more central
            centrality[i] = 1.0 / (np.mean(valid_dists) + 1e-12)
    
    if np.max(centrality) > 0:
        centrality = centrality / np.max(centrality)
    
    # Degree targets: more central nodes keep more connections
    target_degrees = np.clip(
        min_degree + centrality * (max_degree - min_degree),
        min_degree, max_degree
    ).astype(int)
    
    # Create sparsification mask
    sparsity_mask = np.zeros_like(base_scores, dtype=bool)
    
    for i in range(n):
        target_k = target_degrees[i]
        if target_k > 0:
            row_scores = base_scores[i].copy()
            row_scores[i] = -np.inf
            
            # Ensure we have enough valid edges
            valid_scores = row_scores[row_scores > -np.inf]
            if len(valid_scores) > 0:
                actual_k = min(target_k, len(valid_scores))
                if actual_k > 0:
                    threshold = np.partition(row_scores, -actual_k)[-actual_k]
                    sparsity_mask[i, row_scores >= threshold] = True
    
    # Ensure symmetry and preserve MST edges
    sparsity_mask = (sparsity_mask | sparsity_mask.T) | mst_mask
    
    # Apply sparsification: set unpromising edges to zero
    final_scores = base_scores * sparsity_mask
    
    # ========== 10. FINAL NORMALIZATION ==========
    np.fill_diagonal(final_scores, 0)
    
    # Row-wise min-max normalization
    for i in range(n):
        row = final_scores[i]
        valid_vals = row[row > 0]
        if len(valid_vals) > 0:
            min_val = np.min(valid_vals)
            max_val = np.max(valid_vals)
            if max_val > min_val:
                final_scores[i] = np.where(row > 0, (row - min_val) / (max_val - min_val), 0)
    
    # Global scaling to [0, 1]
    max_global = final_scores.max()
    if max_global > 0:
        final_scores = final_scores / max_global
    
    return final_scores
