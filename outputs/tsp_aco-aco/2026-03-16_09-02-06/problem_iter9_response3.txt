import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with improved sparsification and adaptive weighting.
    Features:
    1. Cluster-aware sparsification: Detect natural clusters and preserve intra-cluster edges
    2. Dynamic threshold truncation: Adaptive threshold based on score distribution
    3. Inter-node attractiveness gradient: Score edges based on local connectivity patterns
    4. Connectivity-preserving rewiring: Ensure graph remains connected after sparsification
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # Backup original diagonal for restoration
    np.fill_diagonal(distance_matrix, np.inf)
    
    # 1. Base inverse distance with adaptive smoothing
    epsilon = np.percentile(distance_matrix[distance_matrix != np.inf], 10) * 0.01
    inv_dist = 1 / (distance_matrix + epsilon)
    
    # 2. Enhanced MST scoring with edge importance weights
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = mst.toarray() > 0
    
    # MST edges get bonus proportional to their contribution to total MST weight
    mst_weights = mst.toarray()
    total_mst_weight = mst_weights.sum()
    mst_bonus = np.ones_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if mst_edges[i, j]:
                edge_importance = 1.0 + (total_mst_weight - mst_weights[i, j]) / total_mst_weight
                mst_bonus[i, j] = edge_importance
                mst_bonus[j, i] = edge_importance
    
    # 3. Inter-node attractiveness gradient
    # Calculate node centrality based on average distance to other nodes
    node_centrality = np.zeros(n)
    for i in range(n):
        valid_dists = distance_matrix[i][distance_matrix[i] != np.inf]
        node_centrality[i] = 1.0 / (np.mean(valid_dists) + epsilon)
    
    # Attractiveness gradient: edges connecting nodes with different centrality get adjusted scores
    grad_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j:
                centrality_diff = abs(node_centrality[i] - node_centrality[j])
                grad_factor = 1.0 + centrality_diff * 0.5  # Reward diverse centrality connections
                grad_matrix[i, j] = grad_factor
    
    # 4. Cluster-aware edge preservation
    # Simple clustering based on spatial proximity
    if n >= 5:
        # Use geometric median distances for cluster detection
        from scipy.spatial.distance import cdist
        # Create feature matrix from distance rows
        features = distance_matrix.copy()
        np.fill_diagonal(features, 0)
        
        # Simple threshold-based clustering
        avg_dist = np.mean(features[features != np.inf])
        cluster_mask = features < avg_dist * 0.6
        
        # Boost intra-cluster edges
        cluster_bonus = np.ones_like(distance_matrix)
        cluster_bonus[cluster_mask] = 1.5
    else:
        cluster_bonus = np.ones_like(distance_matrix)
    
    # 5. Combined scoring with adaptive weights
    base_score = inv_dist * mst_bonus * grad_matrix * cluster_bonus
    
    # 6. Dynamic sparsification with connectivity preservation
    # Create initial sparse mask
    mask = np.zeros_like(base_score, dtype=bool)
    
    # Determine adaptive k per node based on score distribution
    for i in range(n):
        row_scores = base_score[i].copy()
        row_scores[i] = -np.inf
        
        # Dynamic threshold: median of non-zero scores
        non_zero_scores = row_scores[row_scores > 0]
        if len(non_zero_scores) > 0:
            threshold = np.percentile(non_zero_scores, 30)  # Keep top 70%
            mask[i] = row_scores >= threshold
        else:
            # Fallback: keep 2 best edges
            top_idx = np.argsort(row_scores)[-2:]
            mask[i, top_idx] = True
    
    # Ensure symmetry
    mask = mask | mask.T
    
    # 7. Connectivity-preserving rewiring
    # Force include MST edges and ensure graph connectivity
    mask = mask | mst_edges
    
    # Additional connectivity check: ensure each node has at least 2 connections
    for i in range(n):
        if mask[i].sum() < 2:
            # Add best missing edges
            row_scores = base_score[i].copy()
            row_scores[mask[i]] = -np.inf
            row_scores[i] = -np.inf
            needed = 2 - mask[i].sum()
            if needed > 0 and np.any(row_scores > -np.inf):
                top_idx = np.argsort(row_scores)[-needed:]
                mask[i, top_idx] = True
                mask[top_idx, i] = True
    
    # 8. Apply final sparsification
    scores = base_score * mask
    
    # 9. Adaptive normalization
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    else:
        scores = np.zeros_like(scores)
    
    # 10. Final cleanup
    np.fill_diagonal(scores, 0)
    
    # Restore original diagonal in input matrix
    np.fill_diagonal(distance_matrix, 0)
    
    return scores
