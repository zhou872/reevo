import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection that combines multiple factors
    and sparsifies the matrix by setting unpromising edges to zero.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # Prevent division by zero and self-connections
    np.fill_diagonal(distance_matrix, np.inf)
    
    # 1. Base inverse distance component
    inv_dist = 1 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    
    # 2. Nearest neighbor ranking component
    # For each node, rank edges by distance (1 = closest)
    ranks = np.zeros_like(distance_matrix)
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])
        ranks[i, sorted_indices] = np.arange(n)
    # Convert to score: closer ranks get higher scores
    rank_score = (n - ranks) / n
    
    # 3. Clustering potential component
    # Measure how central an edge is in connecting clusters
    avg_dist_to_others = np.mean(distance_matrix, axis=1)
    node_potential = 1 / avg_dist_to_others
    node_potential = node_potential / np.max(node_potential)
    # Edge clustering score is product of endpoint potentials
    cluster_score = np.outer(node_potential, node_potential)
    
    # 4. Regret component - difference between best and second best
    regret = np.zeros_like(distance_matrix)
    for i in range(n):
        sorted_dists = np.sort(distance_matrix[i])
        if len(sorted_dists) >= 2:
            best, second_best = sorted_dists[0], sorted_dists[1]
            regret_i = second_best - best
            # Normalize regret
            max_regret = np.max(sorted_dists) - np.min(sorted_dists)
            if max_regret > 0:
                regret[i, :] = (regret_i / max_regret) * (distance_matrix[i] == best)
    
    # Combine components with weights
    heuristics_matrix = (
        0.4 * inv_dist +           # Inverse distance (40%)
        0.3 * rank_score +         # Rank-based (30%)
        0.2 * cluster_score +      # Clustering potential (20%)
        0.1 * regret               # Regret component (10%)
    )
    
    # 5. Progressive sparsification
    # Keep only top-k edges for each node
    k = max(2, min(5, n // 3))  # Adaptive k based on problem size
    
    sparsified = np.zeros_like(heuristics_matrix)
    for i in range(n):
        # Get indices of top k edges (excluding self)
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Exclude diagonal
        top_k_indices = np.argpartition(row, -k)[-k:]
        
        # Keep only top k edges
        sparsified[i, top_k_indices] = heuristics_matrix[i, top_k_indices]
    
    # Ensure symmetry (since TSP edges are undirected)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Additional sparsification: remove edges below threshold
    non_zero_vals = sparsified[sparsified > 0]
    if len(non_zero_vals) > 0:
        threshold = np.percentile(non_zero_vals, 30)  # Keep top 70%
        sparsified[sparsified < threshold] = 0
    
    # Normalize final matrix
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    np.fill_diagonal(sparsified, 0)
    return sparsified
