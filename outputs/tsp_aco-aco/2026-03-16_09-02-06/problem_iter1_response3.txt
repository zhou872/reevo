import numpy as np
from scipy.sparse.csgraph import laplacian
from scipy.spatial.distance import cdist

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection combining multiple factors:
    1. Mutual nearest neighbor strength with exponential decay
    2. Spectral graph properties via Fiedler vector alignment
    3. Regret-based weighting for promising alternatives
    4. Local clustering coefficient for community structure
    5. Strategic sparsification of unpromising edges
    """
    n = distance_matrix.shape[0]
    if n < 4:
        # Fallback for very small instances
        return 1 / (distance_matrix + np.eye(n) * 1e-10)
    
    # 1. Mutual nearest neighbors with exponential decay
    k = min(10, n - 1)
    rank_weights = np.exp(-0.5 * np.arange(k))
    
    # Get k-nearest neighbors for each node
    nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:k+1]
    nearest_distances = np.take_along_axis(distance_matrix, nearest_indices, axis=1)
    
    # Create mutual neighbor matrix
    mutual_neighbor = np.zeros_like(distance_matrix)
    for i in range(n):
        for idx, j in enumerate(nearest_indices[i]):
            if i in nearest_indices[j, :k//2]:  # Mutual if within top k//2 of j
                mutual_neighbor[i, j] = rank_weights[idx] * rank_weights[
                    np.where(nearest_indices[j] == i)[0][0]
                ]
    
    # 2. Spectral embedding via Fiedler vector
    try:
        L = laplacian(1 / (distance_matrix + np.eye(n) * 1e-10), normed=True)
        eigvals, eigvecs = np.linalg.eigh(L)
        fiedler_idx = np.argsort(eigvals)[1]  # Second smallest eigenvalue
        fiedler_vec = eigvecs[:, fiedler_idx]
        spectral_sim = 1 / (1 + np.abs(fiedler_vec[:, None] - fiedler_vec[None, :]))
        np.fill_diagonal(spectral_sim, 0)
    except:
        spectral_sim = np.ones_like(distance_matrix)
    
    # 3. Regret-based weighting
    nearest_vals = np.take_along_axis(distance_matrix, nearest_indices, axis=1)
    if k >= 2:
        regret = nearest_vals[:, 1] - nearest_vals[:, 0]  # Difference between 1st and 2nd best
        regret_weight = 1 + np.tanh(regret[:, None] * 0.1)
    else:
        regret_weight = np.ones((n, 1))
    
    # 4. Local clustering coefficient (approximate)
    cluster_coef = np.zeros(n)
    threshold = np.median(distance_matrix[distance_matrix > 0])
    adj_approx = (distance_matrix < threshold).astype(float)
    np.fill_diagonal(adj_approx, 0)
    
    for i in range(n):
        neighbors = np.where(adj_approx[i] > 0)[0]
        if len(neighbors) >= 2:
            # Subgraph of neighbors
            subgraph = adj_approx[neighbors][:, neighbors]
            cluster_coef[i] = np.sum(subgraph) / (len(neighbors) * (len(neighbors) - 1))
    
    cluster_bonus = 1 + 2 * (cluster_coef[:, None] * cluster_coef[None, :])
    
    # 5. Combine factors with careful weighting
    inverse_dist = 1 / (distance_matrix + np.eye(n) * 1e-10)
    np.fill_diagonal(inverse_dist, 0)
    
    base_score = inverse_dist * (1 + mutual_neighbor)
    spectral_weighted = base_score * spectral_sim
    regret_weighted = spectral_weighted * regret_weight
    final_scores = regret_weighted * cluster_bonus
    
    # 6. Strategic sparsification
    # Keep top edges per node and mutual high scores
    edges_per_node = min(8, n - 1)
    mask = np.zeros_like(final_scores, dtype=bool)
    
    for i in range(n):
        # Keep top edges_per_node by score
        top_indices = np.argsort(final_scores[i])[-edges_per_node:]
        mask[i, top_indices] = True
        # Also keep edges where both endpoints rank each other highly
        mutual_top = np.where(mutual_neighbor[i] > 0.3)[0]
        mask[i, mutual_top] = True
    
    # Symmetrize mask and apply
    mask = mask | mask.T
    final_scores = final_scores * mask
    
    # Normalize for stability
    max_val = np.max(final_scores)
    if max_val > 0:
        final_scores = final_scores / max_val
    
    np.fill_diagonal(final_scores, 0)
    return final_scores
