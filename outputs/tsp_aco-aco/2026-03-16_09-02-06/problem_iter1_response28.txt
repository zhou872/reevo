```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges combining multiple factors:
    1. Inverse distance (base attractiveness)
    2. Minimum Spanning Tree inclusion bonus
    3. Regret-based penalty (difference between best and second-best edges)
    4. Degree-aware sparsification (preserving connectivity)
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # 1. Base inverse distance with smoothing
    np.fill_diagonal(distance_matrix, np.inf)
    inv_dist = 1 / (distance_matrix + 1e-10)
    
    # 2. MST edge bonus - edges in MST get 2x bonus
    mst = minimum_spanning_tree(distance_matrix)
    mst_bonus = (mst.toarray() > 0).astype(float) + 1.0  # 1 for non-MST, 2 for MST
    
    # 3. Regret calculation for each node
    regret_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        # Get distances from node i to all others
        distances_i = distance_matrix[i].copy()
        distances_i[i] = np.inf  # Ignore self
        
        # Find best and second-best distances
        sorted_idx = np.argsort(distances_i)
        best_dist = distances_i[sorted_idx[0]]
        second_best_dist = distances_i[sorted_idx[1]]
        
        # Regret for each edge (i,j) is how much worse it is than the best alternative
        for j in range(n):
            if i != j:
                regret = (distance_matrix[i, j] - best_dist) / (second_best_dist - best_dist + 1e-10)
                regret_matrix[i, j] = np.exp(-regret)  # Convert to penalty factor (0-1)
    
    # 4. Combined scoring
    scores = inv_dist * mst_bonus * regret_matrix
    
    # 5. Degree-aware sparsification
    # Preserve top-k edges for each node while maintaining connectivity
    k = max(2, n // 4)  # Keep at least 2 edges per node, more for small graphs
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        # Get top k edges for this node
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf  # Exclude self
        top_k_idx = np.argpartition(row_scores, -k)[-k:]
        mask[i, top_k_idx] = True
        mask[top_k_idx, i] = True  # Symmetric
    
    # Ensure MST edges are always included for connectivity
    mst_mask = (mst.toarray() > 0)
    mask = mask | mst_mask
    
    # Apply sparsification
    scores = scores * mask
    
    # Normalize scores to [0, 1] range
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    np.fill_diagonal(scores, 0)  # Zero out diagonal
    return scores
```
