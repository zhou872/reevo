import numpy as np
from scipy.spatial import KDTree
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    Returns a priority matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # Initialize priority matrix
    priority = np.zeros_like(distance_matrix, dtype=float)
    
    # 1. Inverse distance (base factor)
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    np.fill_diagonal(inv_dist, 0)
    mask = distance_matrix > 0
    inv_dist[mask] = 1.0 / distance_matrix[mask]
    
    # 2. Local cluster density bonus
    # Points in dense clusters should connect to nearby points
    k = min(5, n-1)
    density_bonus = np.zeros_like(distance_matrix)
    
    # Use KDTree for efficient spatial queries
    points = np.random.rand(n, 2)  # For spatial clustering, we need coordinates
    # In practice, you might have actual coordinates. Using random for demonstration.
    tree = KDTree(points)
    
    for i in range(n):
        # Find k nearest neighbors
        distances, indices = tree.query(points[i], k=k+1)
        # Give bonus to edges connecting to nearest neighbors
        for dist, j in zip(distances[1:], indices[1:]):
            density_bonus[i, j] = 1.0 / (1.0 + dist)
            density_bonus[j, i] = density_bonus[i, j]
    
    # 3. Minimum Spanning Tree factor
    mst_factor = np.zeros_like(distance_matrix)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_present = mst > 0
    mst_factor[mst_present] = 0.8  # Strong bias for MST edges
    
    # 4. Neighbor coherence score
    coherence = np.zeros_like(distance_matrix)
    for i in range(n):
        # For each node, look at its nearest neighbors' nearest neighbors
        row = distance_matrix[i].copy()
        row[i] = np.inf
        nearest_idx = np.argpartition(row, min(3, n-1))[:min(3, n-1)]
        
        for j in range(n):
            if i == j:
                continue
            # Check if j is in the neighborhood of i's neighbors
            neighbor_overlap = 0
            for neighbor in nearest_idx:
                if neighbor != j:
                    neighbor_row = distance_matrix[neighbor].copy()
                    neighbor_row[neighbor] = np.inf
                    neighbor_nearest = np.argpartition(neighbor_row, min(3, n-1))[:min(3, n-1)]
                    if j in neighbor_nearest:
                        neighbor_overlap += 1
            
            coherence[i, j] = neighbor_overlap / len(nearest_idx)
    
    # 5. Combine factors with weights
    priority = (
        0.4 * inv_dist +           # Base inverse distance
        0.25 * density_bonus +     # Spatial clustering
        0.2 * mst_factor +         # MST bias
        0.15 * coherence           # Neighbor coherence
    )
    
    np.fill_diagonal(priority, 0)  # Zero out diagonal
    
    # 6. Regret-based sparsification
    # For each node, calculate regret (difference between best and second-best edges)
    regrets = np.zeros(n)
    for i in range(n):
        row = priority[i].copy()
        row[i] = -np.inf  # Ignore self
        sorted_idx = np.argsort(row)[::-1]
        if len(sorted_idx) >= 2:
            best_val = row[sorted_idx[0]]
            second_best_val = row[sorted_idx[1]]
            regrets[i] = best_val - second_best_val
    
    # Normalize regrets and use as threshold factor
    if np.max(regrets) > 0:
        regrets_norm = regrets / np.max(regrets)
    else:
        regrets_norm = np.ones(n)
    
    # Apply sparsification: for each node, keep only edges above threshold
    threshold_matrix = np.zeros_like(priority)
    for i in range(n):
        # Keep top k edges based on regret threshold
        k_keep = max(2, int(0.3 * n))  # Keep at least 2 edges, up to 30% of nodes
        row = priority[i].copy()
        row[i] = -np.inf
        
        # Adjust threshold based on regret
        if regrets_norm[i] > 0.5:  # High regret -> be more selective
            threshold = np.percentile(row[row > -np.inf], 75)
        else:  # Low regret -> keep more edges
            threshold = np.percentile(row[row > -np.inf], 50)
        
        # Apply threshold
        mask = row < threshold
        row[mask] = 0
        threshold_matrix[i] = row
    
    # Symmetrize the matrix (keep edge if either endpoint finds it promising)
    priority = np.maximum(threshold_matrix, threshold_matrix.T)
    
    # Final cleanup: ensure matrix is non-negative
    priority = np.maximum(priority, 0)
    
    return priority
