import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristic scoring with improved components:
    1. Gaussian kernel distance attraction for smoother distance scaling
    2. Multiplicative MST edge bonus (2x for MST edges)
    3. Exponential regret penalty based on normalized edge disadvantage
    4. Degree-constrained candidate set sparsification for computational efficiency
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between nodes
        
    Returns:
    --------
    np.ndarray
        Heuristic scores of same shape as input, with higher values indicating
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create working copy with infinity on diagonal to avoid self-connections
    D = distance_matrix.copy().astype(float)
    np.fill_diagonal(D, np.inf)
    
    # 1. Gaussian kernel distance attraction
    # Use median distance as scaling parameter for robustness to outliers
    median_dist = np.median(D[D < np.inf])
    sigma = median_dist / 2 if median_dist > 0 else 1.0
    distance_attraction = np.exp(-D**2 / (2 * sigma**2))
    
    # 2. MST edge bonus (multiplicative factor of 2)
    mst = minimum_spanning_tree(D)
    mst_edges = (mst.toarray() > 0)
    mst_bonus = np.ones_like(D)
    mst_bonus[mst_edges] = 2.0
    
    # 3. Regret-based penalty using exponential decay
    regret_penalty = np.ones_like(D)
    
    for i in range(n):
        # Get distances from node i to all others
        dist_i = D[i].copy()
        
        if np.sum(np.isfinite(dist_i)) < 2:
            continue
            
        # Find two smallest distances (best and second-best)
        sorted_idx = np.argsort(dist_i)
        valid_idx = sorted_idx[:2]
        best_dist, second_best = dist_i[valid_idx[0]], dist_i[valid_idx[1]]
        
        # Avoid division by zero or very small differences
        scale = max(second_best - best_dist, 1e-10)
        
        # Calculate regret for each edge from node i
        for j in range(n):
            if i != j and np.isfinite(dist_i[j]):
                regret = (dist_i[j] - best_dist) / scale
                regret_penalty[i, j] = np.exp(-regret)
    
    # 4. Combine factors multiplicatively
    scores = distance_attraction * mst_bonus * regret_penalty
    
    # 5. Degree-constrained candidate set sparsification
    # Determine candidate set size: min(20, n-1) but at least 2
    candidate_size = min(20, n - 1, max(2, int(np.sqrt(n))))
    
    # Initialize mask for sparsification
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        # Get scores for edges from node i (excluding self)
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        # Select top-k candidate edges
        if n > candidate_size:
            # Use argpartition for efficient top-k selection
            top_k_indices = np.argpartition(row_scores, -candidate_size)[-candidate_size:]
        else:
            # Include all edges for small graphs
            top_k_indices = np.arange(n)
            top_k_indices = top_k_indices[top_k_indices != i]
        
        # Mark selected edges in mask (symmetric)
        mask[i, top_k_indices] = True
        for j in top_k_indices:
            mask[j, i] = True
    
    # Ensure MST edges are always included
    mask = mask | mst_edges
    
    # Apply sparsification and zero out non-candidate edges
    scores = scores * mask
    
    # Normalize scores to [0, 1] range for interpretability
    max_score = np.max(scores)
    if max_score > 0:
        scores = scores / max_score
    
    # Zero out diagonal (self-connections)
    np.fill_diagonal(scores, 0)
    
    return scores
