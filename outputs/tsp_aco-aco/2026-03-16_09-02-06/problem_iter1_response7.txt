import numpy as np
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection.
    
    Combines multiple factors to score each edge:
    1. Inverse distance (base attractiveness)
    2. Degree-aware sparsification (keeps only promising connections)
    3. MST connectivity bonus (favors backbone edges)
    4. Local clustering penalty (avoids isolated clusters)
    """
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    
    # 1. Base inverse distance with regularization
    eps = 1e-10
    inv_dist = 1 / (distance_matrix + eps)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Degree-constrained sparsification
    # Keep only top-k edges per node (k = max(2, 10% of nodes))
    k = max(2, int(0.1 * n))
    for i in range(n):
        row = inv_dist[i].copy()
        # Get indices of top k edges (excluding self)
        top_k_indices = np.argpartition(row, -k)[-k:]
        mask = np.zeros(n, dtype=bool)
        mask[top_k_indices] = True
        row[~mask] = 0
        heuristics[i] = row
    
    # 3. MST connectivity bonus
    # Compute Minimum Spanning Tree edges using Prim's algorithm
    def prim_mst_edges(dist_mat):
        n_nodes = dist_mat.shape[0]
        visited = [False] * n_nodes
        visited[0] = True
        mst_edges = []
        
        # Priority: (distance, from, to)
        for _ in range(n_nodes - 1):
            min_dist = np.inf
            u = v = -1
            for i in range(n_nodes):
                if not visited[i]:
                    continue
                for j in range(n_nodes):
                    if not visited[j] and dist_mat[i, j] < min_dist:
                        min_dist = dist_mat[i, j]
                        u, v = i, j
            if u != -1:
                mst_edges.append((u, v))
                visited[v] = True
        return mst_edges
    
    mst_edges = prim_mst_edges(distance_matrix)
    mst_bonus = np.mean(heuristics[heuristics > 0]) * 2 if np.any(heuristics > 0) else 1.0
    
    for u, v in mst_edges:
        heuristics[u, v] += mst_bonus
        heuristics[v, u] += mst_bonus
    
    # 4. Local clustering penalty
    # Penalize edges that connect nodes with very different nearest-neighbor sets
    nn_sets = [set(np.argsort(distance_matrix[i])[1:4]) for i in range(n)]
    
    for i in range(n):
        for j in range(i + 1, n):
            if heuristics[i, j] > 0:
                # Penalty based on Jaccard dissimilarity of nearest neighbors
                intersection = len(nn_sets[i] & nn_sets[j])
                union = len(nn_sets[i] | nn_sets[j])
                if union > 0:
                    jaccard_sim = intersection / union
                    # Reduce score if nodes have very different local neighborhoods
                    heuristics[i, j] *= (0.5 + 0.5 * jaccard_sim)
                    heuristics[j, i] = heuristics[i, j]
    
    # Final sparsification: set values below threshold to zero
    threshold = np.percentile(heuristics[heuristics > 0], 70) if np.any(heuristics > 0) else 0
    heuristics[heuristics < threshold] = 0
    
    return heuristics
