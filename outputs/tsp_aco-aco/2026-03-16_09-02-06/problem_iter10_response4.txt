import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree, connected_components
from scipy.spatial.distance import pdist, squareform


def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with adaptive instance profiling, cached multi-step lookahead,
    and cluster-aware spatial scoring for robust performance across diverse TSP instances.
    
    Key innovations:
    1. Adaptive instance profiling with multiple geometric features
    2. Cached multi-step lookahead (2-3 steps) with efficient computation
    3. Cluster-aware spatial scoring using MST-based clustering
    4. Regret-weighted edge prioritization with adaptive scaling
    5. Dynamic component weighting based on instance characteristics
    
    Parameters
    ----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between nodes
        
    Returns
    -------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 2:
        if n == 0:
            return np.zeros_like(distance_matrix)
        if n == 1:
            return np.zeros((1, 1))
        result = np.zeros((2, 2))
        result[0, 1] = result[1, 0] = 1.0
        return result
    
    # Create working copy with diagonal masked
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # 1. Enhanced instance profiling
    non_inf_dists = dist[dist < np.inf]
    if len(non_inf_dists) == 0:
        return np.zeros_like(dist)
    
    # Basic statistics
    dist_mean = np.mean(non_inf_dists)
    dist_std = np.std(non_inf_dists)
    dist_median = np.median(non_inf_dists)
    dist_q75 = np.percentile(non_inf_dists, 75)
    dist_q25 = np.percentile(non_inf_dists, 25)
    iqr = dist_q75 - dist_q25
    
    # Instance characteristics
    cv = dist_std / (dist_mean + 1e-10)  # Coefficient of variation
    uniformity = 1.0 / (1.0 + cv)  # Higher for more uniform distances
    sparsity = (dist_mean - dist_q25) / (iqr + 1e-10)  # Distance distribution skew
    
    # 2. MST-based clustering for spatial awareness
    mst = minimum_spanning_tree(dist)
    mst_array = mst.toarray()
    
    # Determine optimal number of clusters
    n_clusters = max(2, min(int(np.sqrt(n)), n // 4))
    
    # Extract MST edges and sort by length (descending)
    mst_edges = np.stack(np.where(mst_array > 0), axis=1)
    if len(mst_edges) > 0:
        mst_lengths = np.array([dist[i, j] for i, j in mst_edges])
        sorted_indices = np.argsort(-mst_lengths)
        
        # Remove longest edges to create clusters
        edges_to_remove = min(n_clusters - 1, len(mst_edges) - 1)
        cluster_mst = mst_array.copy()
        
        for idx in range(edges_to_remove):
            i, j = mst_edges[sorted_indices[idx]]
            cluster_mst[i, j] = cluster_mst[j, i] = 0
        
        # Get cluster labels
        _, cluster_labels = connected_components(cluster_mst, directed=False)
    else:
        cluster_labels = np.zeros(n, dtype=int)
    
    # Compute cluster statistics
    n_clusters_actual = np.max(cluster_labels) + 1
    cluster_sizes = np.bincount(cluster_labels)
    cluster_density = np.zeros(n_clusters_actual)
    
    for c in range(n_clusters_actual):
        nodes = np.where(cluster_labels == c)[0]
        if len(nodes) > 1:
            # Mean intra-cluster distance
            intra_dists = []
            for i in range(len(nodes)):
                for j in range(i + 1, len(nodes)):
                    intra_dists.append(dist[nodes[i], nodes[j]])
            if intra_dists:
                cluster_density[c] = 1.0 / (np.mean(intra_dists) + 1e-10)
    
    # Normalize cluster densities
    if np.max(cluster_density) > 0:
        cluster_density = cluster_density / np.max(cluster_density)
    
    # 3. Adaptive parameter initialization
    # Weights based on instance characteristics
    base_weights = {
        'mst': 0.35 - 0.15 * cv,
        'density': 0.25 + 0.15 * uniformity,
        'regret': 0.25 + 0.15 * cv,
        'lookahead': 0.15 + 0.1 * (1 - uniformity)
    }
    
    # Adjust for cluster characteristics
    if n_clusters_actual > 1:
        cluster_variation = np.std(cluster_density) / (np.mean(cluster_density) + 1e-10)
        base_weights['density'] *= (1.0 + 0.2 * cluster_variation)
    
    # Normalize weights
    total = sum(base_weights.values())
    for key in base_weights:
        base_weights[key] /= total
    
    # 4. Base inverse distance with adaptive smoothing
    smoothing = dist_mean * (0.05 + 0.1 * cv)
    inv_dist = 1 / (dist + smoothing)
    
    # 5. MST connectivity bonus
    mst_bonus = np.ones_like(dist)
    mst_bonus[mst_array > 0] = 2.0  # Double bonus for MST edges
    
    # 6. Cached neighbor statistics for efficient multi-step lookahead
    k_neighbors = min(max(3, int(np.log2(n) + 2)), n - 1)
    neighbor_cache = []
    
    for i in range(n):
        sorted_indices = np.argsort(dist[i])[:k_neighbors + 1]
        sorted_indices = sorted_indices[sorted_indices != i][:k_neighbors]
        neighbor_dists = dist[i, sorted_indices]
        
        neighbor_cache.append({
            'indices': sorted_indices,
            'dists': neighbor_dists,
            'avg_dist': np.mean(neighbor_dists) if len(neighbor_dists) > 0 else dist_mean
        })
    
    # 7. Enhanced multi-step lookahead with caching
    lookahead_scores = np.ones_like(dist)
    
    for i in range(n):
        cache_i = neighbor_cache[i]
        neighbors_i = cache_i['indices']
        
        if len(neighbors_i) == 0:
            continue
        
        for j in neighbors_i:
            if i == j:
                continue
            
            cache_j = neighbor_cache[j]
            neighbors_j = cache_j['indices']
            
            # Direct connection score
            direct_score = 1.0 / (1.0 + dist[i, j] / (cache_i['avg_dist'] + 1e-10))
            
            # 2-step lookahead via best intermediate node
            best_2step = np.inf
            for k in neighbors_i:
                if k == j:
                    continue
                if dist[k, j] < np.inf:
                    cost = dist[i, k] + dist[k, j]
                    if cost < best_2step:
                        best_2step = cost
            
            # 3-step lookahead (limited search for efficiency)
            best_3step = np.inf
            if len(neighbors_i) >= 2 and len(neighbors_j) >= 2:
                for k in neighbors_i[:3]:  # Top 3 neighbors of i
                    if k == j:
                        continue
                    for l in neighbors_j[:3]:  # Top 3 neighbors of j
                        if l == i or l == k:
                            continue
                        if dist[k, l] < np.inf:
                            cost = dist[i, k] + dist[k, l] + dist[l, j]
                            if cost < best_3step:
                                best_3step = cost
            
            # Compute regret ratios
            regret_2step = dist[i, j] / best_2step if best_2step < np.inf and best_2step > 0 else 1.0
            regret_3step = dist[i, j] / best_3step if best_3step < np.inf and best_3step > 0 else 1.0
            
            # Combined lookahead score
            lookahead_penalty = np.exp(-max(0, min(regret_2step, regret_3step) - 1.0))
            lookahead_scores[i, j] = direct_score * lookahead_penalty
    
    # Ensure symmetry
    lookahead_scores = (lookahead_scores + lookahead_scores.T) / 2
    
    # 8. Cluster-aware spatial scoring
    spatial_bonus = np.ones_like(dist)
    
    # Intra-cluster bonus
    for i in range(n):
        for j in range(i + 1, n):
            if cluster_labels[i] == cluster_labels[j]:
                # Same cluster: higher bonus for denser clusters
                cluster_id = cluster_labels[i]
                density_factor = 0.5 + 0.5 * cluster_density[cluster_id]
                spatial_bonus[i, j] = spatial_bonus[j, i] = 1.0 + density_factor
            else:
                # Different clusters: baseline with slight penalty
                spatial_bonus[i, j] = spatial_bonus[j, i] = 0.8
    
    # 9. Enhanced regret penalization
    dist_copy = dist.copy()
    np.fill_diagonal(dist_copy, np.inf)
    
    # Get k-best distances for each node
    k_regret = min(5, n - 1)
    sorted_idx = np.argsort(dist_copy, axis=1)[:, :k_regret]
    
    regret_matrix = np.ones_like(dist)
    for i in range(n):
        if k_regret > 1:
            best_dist = dist_copy[i, sorted_idx[i, 0]]
            second_best = dist_copy[i, sorted_idx[i, 1]] if k_regret > 1 else best_dist
            
            for j in range(n):
                if i != j and dist_copy[i, j] < np.inf:
                    # Adaptive regret scaling based on instance characteristics
                    regret_scale = 2.0 + 2.0 * cv
                    regret = (dist_copy[i, j] - best_dist) / (second_best - best_dist + 1e-10)
                    regret_matrix[i, j] = np.exp(-regret_scale * np.clip(regret, 0, 3))
    
    # Symmetrize regret matrix
    regret_matrix = (regret_matrix + regret_matrix.T) / 2
    
    # 10. Combine components with adaptive weights
    scores = inv_dist * (
        base_weights['mst'] * mst_bonus +
        base_weights['density'] * spatial_bonus +
        base_weights['regret'] * regret_matrix +
        base_weights['lookahead'] * lookahead_scores
    )
    
    # 11. Smart sparsification with connectivity guarantees
    # Determine target edges per node
    base_edges = max(3, int(np.sqrt(n)))
    density_factor = 1.0 + 0.3 * (1 - uniformity)
    target_edges = min(n - 1, int(base_edges * density_factor))
    
    # Initialize mask with MST edges
    mask = mst_array > 0
    mask = mask | mask.T
    
    # Add promising edges based on scores
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        # Sort by score
        sorted_indices = np.argsort(row_scores)[::-1]
        
        # Count existing edges
        current_edges = np.sum(mask[i])
        needed_edges = max(0, target_edges - current_edges)
        
        # Add top-scoring edges
        added = 0
        for j in sorted_indices:
            if added >= needed_edges:
                break
            if not mask[i, j] and i != j:
                mask[i, j] = True
                mask[j, i] = True
                added += 1
    
    # Ensure connectivity
    conn_mask = mask.astype(int)
    n_components, labels = connected_components(conn_mask, directed=False)
    
    if n_components > 1:
        # Connect components using best available edges
        comp_edges = []
        for comp_i in range(n_components):
            for comp_j in range(comp_i + 1, n_components):
                nodes_i = np.where(labels == comp_i)[0]
                nodes_j = np.where(labels == comp_j)[0]
                
                # Find best edge between components
                best_score = -np.inf
                best_pair = (0, 0)
                for i in nodes_i:
                    for j in nodes_j:
                        if scores[i, j] > best_score:
                            best_score = scores[i, j]
                            best_pair = (i, j)
                
                if best_score > -np.inf:
                    comp_edges.append((best_pair, best_score))
        
        # Connect components in order of edge quality
        comp_edges.sort(key=lambda x: x[1], reverse=True)
        
        for (i, j), _ in comp_edges:
            if n_components <= 1:
                break
            if not mask[i, j]:
                mask[i, j] = True
                mask[j, i] = True
                # Recompute connectivity
                conn_mask = mask.astype(int)
                n_components, _ = connected_components(conn_mask, directed=False)
    
    # Apply final mask
    final_scores = scores * mask
    
    # 12. Robust normalization
    np.fill_diagonal(final_scores, 0)
    
    valid_scores = final_scores[final_scores > 0]
    if len(valid_scores) > 0:
        # Use percentile-based normalization
        q95 = np.percentile(valid_scores, 95)
        q05 = np.percentile(valid_scores, 5)
        
        if q95 > q05:
            scaled = (final_scores - q05) / (q95 - q05 + 1e-10)
            final_scores = np.clip(scaled, 0, 1)
        else:
            # All scores similar, normalize by max
            max_score = np.max(valid_scores)
            if max_score > 0:
                final_scores = final_scores / max_score
    
    # Apply soft thresholding for better discriminability
    final_scores = 1 / (1 + np.exp(-6 * (final_scores - 0.5)))
    
    return final_scores
