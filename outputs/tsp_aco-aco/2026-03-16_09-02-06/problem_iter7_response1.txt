import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges combining multiple factors with improved regret calculation.
    1. Inverse distance (base attractiveness)
    2. Minimum Spanning Tree inclusion bonus
    3. Symmetric regret-based penalty (averaged from both node perspectives)
    4. Adaptive sparsification with connectivity preservation
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # Work with a copy to avoid modifying the original
    dist_mat = distance_matrix.copy()
    np.fill_diagonal(dist_mat, np.inf)
    
    # 1. Base inverse distance with smoothing
    inv_dist = 1 / (dist_mat + 1e-10)
    
    # 2. MST edge bonus - edges in MST get bonus factor
    # Prepare matrix for MST (needs zeros on diagonal)
    mst_input = dist_mat.copy()
    np.fill_diagonal(mst_input, 0)
    mst = minimum_spanning_tree(mst_input)
    mst_bonus = (mst.toarray() > 0).astype(float) * 1.5 + 1.0  # 2.5x for MST, 1x for non-MST
    
    # 3. Improved symmetric regret calculation
    regret_matrix = np.zeros_like(dist_mat)
    
    # Precompute best and second best distances for each node
    best_dists = np.zeros(n)
    second_best_dists = np.zeros(n)
    best_indices = np.zeros(n, dtype=int)
    second_best_indices = np.zeros(n, dtype=int)
    
    for i in range(n):
        distances_i = dist_mat[i].copy()
        distances_i[i] = np.inf
        
        # Find two smallest distances
        idx = np.argpartition(distances_i, 2)
        sorted_idx = idx[np.argsort(distances_i[idx])[:2]]
        
        best_indices[i] = sorted_idx[0]
        second_best_indices[i] = sorted_idx[1]
        best_dists[i] = distances_i[sorted_idx[0]]
        second_best_dists[i] = distances_i[sorted_idx[1]]
    
    # Compute symmetric regret for each edge
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            # Regret from i's perspective
            if j == best_indices[i]:
                # This is the best edge from i, minimal regret
                regret_i = 0.0
            else:
                regret_i = (dist_mat[i, j] - best_dists[i]) / max(second_best_dists[i] - best_dists[i], 1e-10)
            
            # Regret from j's perspective
            if i == best_indices[j]:
                regret_j = 0.0
            else:
                regret_j = (dist_mat[j, i] - best_dists[j]) / max(second_best_dists[j] - best_dists[j], 1e-10)
            
            # Symmetric regret: average of both perspectives
            avg_regret = (regret_i + regret_j) / 2.0
            
            # Convert regret to penalty factor (lower regret = higher score)
            # Using sigmoid-like transformation to keep values in reasonable range
            regret_matrix[i, j] = 1.0 / (1.0 + avg_regret)
    
    # 4. Combined scoring with balanced weighting
    scores = inv_dist * mst_bonus * regret_matrix
    
    # 5. Adaptive sparsification with connectivity guarantee
    # Determine k based on graph size: keep more edges for smaller graphs
    k = max(2, int(np.sqrt(n)))  # Adaptive k based on sqrt(n)
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        # Keep top k edges
        threshold = np.partition(row_scores, -k)[-k]
        mask[i, row_scores >= threshold] = True
    
    # Ensure symmetry
    mask = mask | mask.T
    
    # Ensure MST edges are always included
    mst_mask = (mst.toarray() > 0)
    mask = mask | mst_mask
    
    # Apply sparsification
    scores = scores * mask
    
    # Normalize scores to [0, 1] range
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    np.fill_diagonal(scores, 0)
    return scores
