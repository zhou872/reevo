import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic scoring for TSP edges with adaptive strategies:
    1. Multipath regret analysis considering best 3 alternatives
    2. Cluster-aware normalization using node density
    3. Dynamic degree-based sparsification with connectivity preservation
    4. Adaptive MST bonus scaling based on graph size and structure
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # Create working copy to avoid modifying input
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    
    # 1. Base inverse distance with adaptive smoothing
    avg_dist = np.nanmean(dist[dist < np.inf])
    inv_dist = 1 / (dist + avg_dist * 0.01)
    
    # 2. Adaptive MST edge bonus based on graph size and density
    mst = minimum_spanning_tree(dist)
    mst_edges = (mst.toarray() > 0)
    
    # Calculate MST bonus factor based on graph characteristics
    mst_density = mst_edges.sum() / (n * (n - 1)) if n > 1 else 0
    base_bonus = 1.5 + 0.5 * np.log1p(n)  # Scales with log of size
    mst_bonus = np.ones_like(dist)
    mst_bonus[mst_edges] = base_bonus
    
    # 3. Multipath regret analysis
    regret_matrix = np.ones_like(dist)
    
    for i in range(n):
        distances_i = dist[i].copy()
        distances_i[i] = np.inf
        
        # Get best 3 distances for more robust regret calculation
        sorted_vals = np.sort(distances_i)
        if len(sorted_vals) >= 3:
            best_vals = sorted_vals[:3]
            # Calculate regret relative to average of best 3 alternatives
            best_avg = np.mean(best_vals)
            for j in range(n):
                if i != j and distances_i[j] < np.inf:
                    # Quadratic penalty for larger deviations
                    regret = (dist[i, j] - best_avg) / (best_vals[-1] - best_vals[0] + 1e-10)
                    regret_matrix[i, j] = np.exp(-regret * (1 + 0.5 * regret))
    
    # 4. Cluster-aware scoring using node density
    density_scores = np.ones((n, n))
    if n > 2:
        # Estimate local density for each node
        avg_neighbor_dist = np.zeros(n)
        for i in range(n):
            valid_dists = dist[i, dist[i] < np.inf]
            if len(valid_dists) > 0:
                avg_neighbor_dist[i] = np.mean(np.partition(valid_dists, min(3, len(valid_dists)-1))[:min(3, len(valid_dists))])
        
        # Adjust scores based on relative densities
        for i in range(n):
            for j in range(n):
                if i != j and dist[i, j] < np.inf:
                    density_ratio = avg_neighbor_dist[i] / (avg_neighbor_dist[j] + 1e-10)
                    density_scores[i, j] = 1.0 / (1.0 + np.abs(np.log1p(density_ratio)))
    
    # 5. Combined scoring with weighted factors
    scores = inv_dist * mst_bonus * regret_matrix * density_scores
    
    # 6. Dynamic degree-aware sparsification
    # Adaptive k based on graph size and density
    avg_degree_target = min(6, max(3, int(np.sqrt(n))))
    k_per_node = min(avg_degree_target, n - 1)
    
    # First pass: keep top edges for each node
    mask = np.zeros_like(scores, dtype=bool)
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        if n > k_per_node:
            # Keep top k_per_node edges
            threshold = np.partition(row_scores, -k_per_node)[-k_per_node]
            mask[i, row_scores >= threshold] = True
        else:
            # Keep all edges for small graphs
            mask[i, :] = True
            mask[i, i] = False
    
    # Ensure symmetry
    mask = mask | mask.T
    
    # Always include MST edges for connectivity
    mask = mask | mst_edges
    
    # 7. Progressive sparsification for larger graphs
    if n > 20:
        # Additional filtering: only keep edges with above-average scores
        score_threshold = np.percentile(scores[mask], 30)
        mask = mask & (scores >= score_threshold)
        
        # Ensure each node has at least 2 connections
        for i in range(n):
            if mask[i].sum() < 2:
                # Add back best edges if needed
                row_scores = scores[i].copy()
                row_scores[i] = -np.inf
                row_scores[~mask[i]] = -np.inf
                if row_scores.max() > -np.inf:
                    best_idx = np.argmax(row_scores)
                    mask[i, best_idx] = True
                    mask[best_idx, i] = True
    
    # Apply sparsification
    scores = scores * mask
    
    # 8. Adaptive normalization
    non_zero_scores = scores[scores > 0]
    if len(non_zero_scores) > 0:
        # Use robust normalization (clip extreme values)
        median_score = np.median(non_zero_scores)
        mad_score = np.median(np.abs(non_zero_scores - median_score))
        
        if mad_score > 0:
            # Robust z-score scaling to [0, 1] range
            robust_scores = (scores - median_score) / (mad_score * 1.4826)
            scores = 1 / (1 + np.exp(-robust_scores * 0.5))  # Sigmoid scaling
        else:
            # Fallback to min-max if no variance
            max_score = scores.max()
            if max_score > 0:
                scores = scores / max_score
    
    # Ensure diagonal is zero
    np.fill_diagonal(scores, 0)
    
    return scores
