import numpy as np
from scipy.spatial.distance import squareform, pdist
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Inverse distance weighting with saturation
    2. Sparsification based on nearest neighbors
    3. Node degree normalization
    4. Triangular inequality threshold filtering
    """
    n = distance_matrix.shape[0]
    eps = 1e-8
    
    # 1. Inverse distance with saturation
    # Use sigmoid-like function to avoid extreme values
    max_dist = np.max(distance_matrix)
    normalized_dist = distance_matrix / (max_dist + eps)
    inverse_weight = 1.0 / (normalized_dist + 0.1)  # Add constant to avoid singularities
    
    # 2. Nearest neighbor sparsification
    # For each node, keep edges to its k nearest neighbors
    k = max(3, min(10, n // 4))  # Adaptive k based on problem size
    mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k+1 smallest distances (including self)
        indices = np.argsort(distance_matrix[i])[:k+1]
        # Remove self from neighbors
        indices = indices[indices != i][:k]
        mask[i, indices] = True
    
    # Make symmetric (undirected TSP)
    mask = mask | mask.T
    
    # 3. Node degree normalization
    # Give higher weight to edges connecting nodes with fewer good alternatives
    node_promising_edges = np.sum(inverse_weight * mask, axis=1, keepdims=True)
    degree_factor = 1.0 / (node_promising_edges + eps)
    
    # Combine factors with appropriate weights
    combined_score = inverse_weight * mask
    combined_score *= degree_factor  # Normalize by node's alternatives
    combined_score *= degree_factor.T  # Symmetric normalization
    
    # 4. Triangular inequality threshold
    # Penalize edges that violate triangle inequality significantly
    # Only compute for non-zero edges to save computation
    nonzero_mask = combined_score > 0
    row_idx, col_idx = np.where(nonzero_mask)
    
    for r, c in zip(row_idx, col_idx):
        if r >= c:  # Process each undirected edge once
            continue
            
        # Find a third node that makes triangle inequality nearly violated
        intermediate_nodes = np.setdiff1d(np.arange(n), [r, c])
        if len(intermediate_nodes) > 0:
            # Compute shortest path through any intermediate node
            indirect_paths = distance_matrix[r, intermediate_nodes] + distance_matrix[intermediate_nodes, c]
            min_indirect = np.min(indirect_paths)
            
            # If direct edge is much longer than indirect path, penalize
            if distance_matrix[r, c] > 1.5 * min_indirect:
                penalty = min_indirect / (distance_matrix[r, c] + eps)
                combined_score[r, c] *= penalty * penalty  # Quadratic penalty
                combined_score[c, r] = combined_score[r, c]
    
    # Set diagonal to zero (no self-loops)
    np.fill_diagonal(combined_score, 0)
    
    # Ensure symmetry (numerical errors might cause small asymmetries)
    combined_score = (combined_score + combined_score.T) / 2
    
    # Final sparsification: keep only top p% of edges
    if n > 20:  # Only for larger instances
        threshold = np.percentile(combined_score[combined_score > 0], 30)
        combined_score[combined_score < threshold] = 0
    
    return combined_score
