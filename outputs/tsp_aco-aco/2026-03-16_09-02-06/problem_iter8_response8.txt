import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Optimized heuristic scoring for TSP edges based on three synergistic principles:
    1. Symmetric normalized regret penalization
    2. Multiplicative fusion of fundamental factors
    3. Lightweight adaptive sparsification with MST connectivity guarantee
    
    Returns a matrix of prior indicators (0-1) for edge inclusion probability.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Work on copy with infinity diagonal to avoid self-edges
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    epsilon = 1e-12
    
    # 1. BASE INVERSE DISTANCE WEIGHTING
    inv_dist = 1 / (dist + epsilon)
    
    # 2. MINIMUM SPANNING TREE CONNECTIVITY GUARANTEE
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    mst_bonus = np.ones_like(dist)
    mst_bonus[mst_mask] = 2.0  # MST edges receive 2x bonus
    
    # 3. SYMMETRIC NORMALIZED REGRET PENALIZATION
    # Compute best and second-best distances for each node
    sorted_dists = np.sort(dist, axis=1)
    best_dists = sorted_dists[:, 0]
    second_best_dists = sorted_dists[:, 1]
    
    # Compute normalized regret gap for stability
    regret_gap = second_best_dists - best_dists + epsilon
    
    # Vectorized regret computation
    regret_i = (dist - best_dists[:, np.newaxis]) / regret_gap[:, np.newaxis]
    regret_j = (dist - best_dists[np.newaxis, :]) / regret_gap[np.newaxis, :]
    
    # Symmetric average regret with exponential penalty
    avg_regret = (regret_i + regret_j) / 2
    regret_penalty = np.exp(-avg_regret)
    
    # Zero diagonal in penalty matrix
    np.fill_diagonal(regret_penalty, 0)
    
    # 4. MULTIPLICATIVE FACTOR FUSION
    scores = inv_dist * mst_bonus * regret_penalty
    
    # 5. ADAPTIVE TOP-K SPARSIFICATION
    # Dynamic K based on graph size (sublinear scaling)
    base_k = int(np.sqrt(n))
    k = max(2, min(base_k, 15))  # Cap at 15 for large graphs
    
    # Create sparsification mask
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf  # Exclude diagonal
        
        # Keep top K edges per node
        if n > k:
            top_indices = np.argpartition(row_scores, -k)[-k:]
        else:
            top_indices = np.where(row_scores > -np.inf)[0]
        
        mask[i, top_indices] = True
    
    # Ensure symmetry
    mask = mask | mask.T
    
    # Guarantee MST connectivity
    mask = mask | mst_mask
    
    # Apply sparsification
    scores = scores * mask
    
    # 6. NORMALIZE TO [0, 1] RANGE
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    np.fill_diagonal(scores, 0)
    
    return scores
