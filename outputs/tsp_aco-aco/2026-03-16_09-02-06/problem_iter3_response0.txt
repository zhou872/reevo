import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with improved factors:
    1. Reciprocal distance with exponential scaling
    2. Mutual edge ranking penalty (both endpoints must consider edge promising)
    3. Adaptive top-k sparsification per node based on sqrt(n)
    4. Symmetric connectivity preservation
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create a copy to avoid modifying input
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    
    # 1. Reciprocal distance with exponential scaling
    # Use exponential to give stronger preference to very close nodes
    inv_dist = np.exp(-dist / (dist[dist < np.inf].mean() + 1e-10))
    
    # 2. MST bonus (slightly reduced from v1)
    mst = minimum_spanning_tree(dist)
    mst_bonus = (mst.toarray() > 0).astype(float) * 1.3 + 0.7  # MST: 2.0, others: 0.7
    
    # 3. Mutual edge ranking penalty
    mutual_penalty = np.ones_like(dist)
    
    # For each node, rank edges by inverse distance
    for i in range(n):
        # Get sorted indices by reciprocal distance (descending)
        sorted_indices = np.argsort(-inv_dist[i])
        
        # Only consider edges to other nodes (exclude diagonal)
        valid_indices = [idx for idx in sorted_indices if idx != i]
        
        # If we have enough edges, penalize those beyond top 30% for both endpoints
        if len(valid_indices) >= 3:
            cutoff_idx = int(0.3 * len(valid_indices))
            bottom_indices = valid_indices[cutoff_idx:]
            
            for j in bottom_indices:
                # Check if edge (i,j) is also in bottom 30% for node j
                sorted_indices_j = np.argsort(-inv_dist[j])
                valid_indices_j = [idx for idx in sorted_indices_j if idx != j]
                
                if len(valid_indices_j) >= 3:
                    cutoff_idx_j = int(0.3 * len(valid_indices_j))
                    bottom_indices_j = valid_indices_j[cutoff_idx_j:]
                    
                    if i in bottom_indices_j:
                        mutual_penalty[i, j] = 0.1
    
    # 4. Combined scoring
    scores = inv_dist * mst_bonus * mutual_penalty
    
    # 5. Adaptive sparsification
    # Use sqrt(n) as suggested, with minimum of 2
    k = max(2, int(np.sqrt(n)))
    
    # Create initial mask
    mask = np.zeros_like(scores, dtype=bool)
    
    # For each node, select top k edges
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf  # Exclude self
        
        # Get indices of top k scores
        top_k_indices = np.argpartition(row_scores, -k)[-k:]
        mask[i, top_k_indices] = True
    
    # Ensure symmetry: if (i,j) is selected, (j,i) should also be selected
    mask = mask | mask.T
    
    # Ensure MST edges are always included for connectivity
    mst_mask = (mst.toarray() > 0)
    mask = mask | mst_mask
    
    # Apply sparsification
    scores = scores * mask
    
    # 6. Global percentile filtering (hybrid sparsification)
    # Keep edges that are globally in top p percentile
    p = 20  # Top 20%
    if scores[scores > 0].size > 0:
        flat_scores = scores[scores > 0]
        threshold = np.percentile(flat_scores, 100 - p)
        global_mask = scores >= threshold
        scores = scores * global_mask
    
    # Ensure diagonal is zero
    np.fill_diagonal(scores, 0)
    
    # Normalize non-zero scores to [0, 1]
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    return scores
