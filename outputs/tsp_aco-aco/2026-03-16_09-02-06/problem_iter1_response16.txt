import numpy as np
from scipy.spatial import KDTree
from sklearn.cluster import KMeans

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization.
    Returns a matrix of edge promise indicators with same shape as input.
    Combines multiple factors to prioritize promising edges and sparsifies
    unpromising ones by setting them to zero.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance factor
    inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Nearest-neighbor confidence factor
    # Each edge gets bonus if it's among k-nearest neighbors for both endpoints
    k = min(5, n - 1)
    nn_bonus = np.zeros_like(distance_matrix)
    
    # For each node, identify k nearest neighbors
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])
        valid_indices = sorted_indices[1:k+1]  # Exclude self
        nn_bonus[i, valid_indices] = 1
    
    # Mutual nearest neighbor bonus (both directions)
    mutual_nn = (nn_bonus + nn_bonus.T) == 2
    nn_bonus = mutual_nn.astype(float) * 2  # Strong bonus for mutual NN
    
    # 3. Cluster-based edge density factor
    # Use KMeans to identify clusters, favoring intra-cluster edges
    n_clusters = min(5, n // 2)
    if n >= 10:
        coords = np.random.randn(n, 2)  # Dummy coordinates for clustering
        kmeans = KMeans(n_clusters=n_clusters, random_state=42, n_init=10)
        clusters = kmeans.fit_predict(coords)
        
        cluster_bonus = np.zeros_like(distance_matrix)
        for i in range(n):
            for j in range(i+1, n):
                if clusters[i] == clusters[j]:
                    # Edges within same cluster get bonus
                    cluster_bonus[i, j] = cluster_bonus[j, i] = 1.5
    else:
        cluster_bonus = np.ones_like(distance_matrix)  # No clustering for small n
    
    # 4. Edge centrality penalty
    # Penalize edges that are too "central" (connect to many nodes)
    node_centrality = 1 / (distance_matrix.sum(axis=1) + 1e-10)
    centrality_penalty = np.outer(node_centrality, node_centrality)
    
    # 5. Sparsification: Combine factors and zero out unpromising edges
    # Weighted combination
    edge_promise = (
        0.4 * inv_dist + 
        0.3 * nn_bonus + 
        0.2 * cluster_bonus + 
        0.1 * centrality_penalty
    )
    
    # Sparsification: Keep only top percentile of edges
    if n >= 3:
        # Calculate threshold to keep top 30% of edges (excluding diagonal)
        upper_tri = edge_promise[np.triu_indices(n, 1)]
        threshold = np.percentile(upper_tri, 70)
        
        # Create sparsity mask
        sparsity_mask = edge_promise > threshold
        
        # Apply mask (zero out unpromising edges)
        edge_promise = edge_promise * sparsity_mask
    
    # Ensure symmetry and zero diagonal
    edge_promise = (edge_promise + edge_promise.T) / 2
    np.fill_diagonal(edge_promise, 0)
    
    # Normalize to reasonable scale
    if edge_promise.max() > 0:
        edge_promise = edge_promise / edge_promise.max()
    
    return edge_promise
