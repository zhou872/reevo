import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse import csr_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges combining multiple factors:
    1. Inverse distance (base attractiveness)
    2. Minimum Spanning Tree inclusion bonus
    3. Symmetric regret penalization (averaged from both node perspectives)
    4. Spectral sparsification with connectivity preservation
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix) if n == 1 else np.array([])
    
    # Work with a copy to avoid modifying input
    D = distance_matrix.copy()
    np.fill_diagonal(D, np.inf)
    
    # 1. Base inverse distance with smoothing
    inv_dist = 1 / (D + 1e-12)
    
    # 2. MST edge bonus - edges in MST get 3x bonus (enhanced from 2x)
    mst = minimum_spanning_tree(D)
    mst_edges = (mst.toarray() > 0)
    mst_bonus = np.where(mst_edges, 3.0, 1.0)
    
    # 3. Symmetric regret calculation
    regret_matrix = np.zeros_like(D)
    
    # Precompute best and second-best for each node
    sorted_indices = np.argsort(D, axis=1)
    best_dists = D[np.arange(n), sorted_indices[:, 0]]
    second_best_dists = D[np.arange(n), sorted_indices[:, 1]]
    
    for i in range(n):
        for j in range(i+1, n):  # Only compute once per edge pair
            if D[i, j] == np.inf:
                continue
                
            # Regret from i's perspective
            if D[i, j] == best_dists[i]:
                regret_i = 0.0
            else:
                regret_i = (D[i, j] - best_dists[i]) / (second_best_dists[i] - best_dists[i] + 1e-12)
            
            # Regret from j's perspective
            if D[i, j] == best_dists[j]:
                regret_j = 0.0
            else:
                regret_j = (D[i, j] - best_dists[j]) / (second_best_dists[j] - best_dists[j] + 1e-12)
            
            # Symmetric regret (geometric mean)
            symmetric_regret = np.sqrt((regret_i + 1e-12) * (regret_j + 1e-12))
            penalty = np.exp(-2.0 * symmetric_regret)  # Stronger penalty
            
            regret_matrix[i, j] = penalty
            regret_matrix[j, i] = penalty
    
    # 4. Combined scoring with balanced weights
    scores = inv_dist * mst_bonus * regret_matrix
    
    # 5. Spectral-inspired sparsification
    # Compute degree-normalized scores for spectral properties
    degree = np.sum(D < np.inf, axis=1)
    degree_matrix = np.diag(degree)
    laplacian = degree_matrix - scores
    
    # Use eigenvectors for sparsification guidance
    if n <= 500:  # Eigen decomposition for moderate sizes
        try:
            eigvals, eigvecs = np.linalg.eigh(laplacian)
            fiedler_vector = eigvecs[:, 1]  # Second smallest eigenvector
            
            # Cluster nodes based on Fiedler vector sign
            cluster_mask = np.outer(fiedler_vector >= 0, fiedler_vector >= 0) | \
                          np.outer(fiedler_vector < 0, fiedler_vector < 0)
            
            # Enhance intra-cluster edges, penalize inter-cluster
            cluster_bonus = np.where(cluster_mask, 1.2, 0.8)
            scores *= cluster_bonus
        except:
            pass  # Fall back to degree-based sparsification
    
    # Adaptive sparsification preserving connectivity
    k = max(2, int(np.sqrt(n)))  # Adaptive k based on sqrt(n)
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        # Keep top-k edges and all MST edges connected to this node
        top_k_idx = np.argpartition(row_scores, -k)[-k:]
        mask[i, top_k_idx] = True
        
        # Ensure MST connectivity
        mst_neighbors = np.where(mst_edges[i])[0]
        mask[i, mst_neighbors] = True
        mask[mst_neighbors, i] = True
    
    # Ensure symmetry and connectivity
    mask = mask | mask.T | mst_edges
    
    # Apply sparsification with smooth decay for non-promising edges
    scores = scores * mask
    scores += 0.1 * scores.max() * (~mask)  # Small residual for all edges
    
    # 6. Normalize with softmax-like transformation
    score_max = scores.max()
    if score_max > 0:
        scores = scores / score_max
    
    # Apply sigmoid transformation for better discrimination
    scores = 1 / (1 + np.exp(-5 * (scores - 0.5)))
    
    np.fill_diagonal(scores, 0.0)
    return scores
