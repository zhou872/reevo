import numpy as np
from scipy.spatial import KDTree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection probability.
    
    Combines multiple factors to assess edge promise:
    1. Inverse distance (closer nodes are generally better)
    2. Two-step lookahead (considering neighbor's neighbors)
    3. Regret penalty (penalizing edges that are poor choices for both endpoints)
    4. Cluster density bonus (edges within dense clusters get bonus)
    
    Returns a matrix of the same shape as input, with promising edges
    having higher values and unpromising edges set to zero.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Avoid division by zero for diagonal
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base inverse distance component
    inv_dist = 1.0 / safe_dist
    
    # 2. Two-step lookahead spatial scoring
    # For each node, consider the quality of connecting through this edge
    lookahead_scores = np.zeros_like(distance_matrix)
    
    # Find k nearest neighbors for each node (excluding self)
    k = min(10, n-1)
    neighbor_indices = np.argsort(safe_dist, axis=1)[:, 1:k+1]
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Consider best alternative connections for both nodes
            if k > 0:
                # Best alternative for i (excluding j)
                i_alt = neighbor_indices[i][0] if neighbor_indices[i][0] != j else neighbor_indices[i][1] if k > 1 else i
                # Best alternative for j (excluding i)
                j_alt = neighbor_indices[j][0] if neighbor_indices[j][0] != i else neighbor_indices[j][1] if k > 1 else j
                
                # Two-step score: quality of i-j plus quality of connecting their neighborhoods
                two_step_score = (inv_dist[i, j] + 
                                0.3 * inv_dist[i, i_alt] + 
                                0.3 * inv_dist[j, j_alt])
                lookahead_scores[i, j] = two_step_score
    
    # 3. Regret-based penalty
    regret_penalty = np.ones_like(distance_matrix)
    for i in range(n):
        # Sort edges from this node by quality
        sorted_indices = np.argsort(inv_dist[i])[::-1]
        if len(sorted_indices) >= 3:
            # Mark edges beyond top 30% as potentially regrettable
            threshold_idx = int(0.3 * len(sorted_indices))
            for rank, j in enumerate(sorted_indices):
                if i != j and rank > threshold_idx:
                    # Also check if this edge is bad for j too
                    if np.where(np.argsort(inv_dist[j])[::-1] == i)[0] > threshold_idx:
                        regret_penalty[i, j] = 0.1  # Strong penalty for mutual bad choice
    
    # 4. Cluster density bonus using KDTree for spatial analysis
    density_bonus = np.ones_like(distance_matrix)
    if n >= 10:
        # Create coordinates from distance matrix (approximate)
        # Using MDS-like approximation for spatial structure
        try:
            from sklearn.manifold import MDS
            mds = MDS(n_components=2, dissimilarity='precomputed', random_state=42)
            coords = mds.fit_transform(distance_matrix)
            tree = KDTree(coords)
            
            # For each point, find density of local neighborhood
            radius = np.percentile(distance_matrix[distance_matrix > 0], 25)
            densities = np.zeros(n)
            for i in range(n):
                neighbors = tree.query_ball_point(coords[i], r=radius)
                densities[i] = len(neighbors) / n
            
            # Apply density bonus: edges within dense regions get boost
            for i in range(n):
                for j in range(n):
                    if i != j:
                        avg_density = (densities[i] + densities[j]) / 2
                        density_bonus[i, j] = 1.0 + 0.5 * avg_density
        except:
            # Fallback if MDS fails
            density_bonus = np.ones_like(distance_matrix)
    
    # Combine all factors
    combined = inv_dist * lookahead_scores * regret_penalty * density_bonus
    
    # Normalize to reasonable range
    max_val = np.max(combined)
    if max_val > 0:
        combined = combined / max_val
    
    # Sparsification: zero out unpromising edges
    # Keep top edges for each node and edges with score above threshold
    threshold = np.percentile(combined[combined > 0], 70) if np.any(combined > 0) else 0
    
    sparse_result = np.zeros_like(combined)
    for i in range(n):
        # Keep top k edges for each node
        top_k = min(5, n-1)
        if top_k > 0:
            top_indices = np.argsort(combined[i])[-top_k:]
            sparse_result[i, top_indices] = combined[i, top_indices]
        
        # Also keep edges above threshold
        above_threshold = combined[i] > threshold
        sparse_result[i, above_threshold] = np.maximum(
            sparse_result[i, above_threshold], 
            combined[i, above_threshold]
        )
    
    # Symmetrize (optional but keeps matrix symmetric like input)
    sparse_result = (sparse_result + sparse_result.T) / 2
    
    # Zero out diagonal
    np.fill_diagonal(sparse_result, 0)
    
    return sparse_result
