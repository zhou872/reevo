import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics function for TSP edge selection.
    
    Features:
    1. Exponential distance discounting with adaptive scaling
    2. Two-way nearest neighbor compatibility
    3. Cluster-based connectivity scoring
    4. Dynamic threshold sparsification
    """
    n = distance_matrix.shape[0]
    
    # 1. Exponential distance discounting with adaptive scaling
    # Normalize distances to [0,1] range for stable exponential scaling
    max_dist = distance_matrix.max()
    min_dist = distance_matrix[distance_matrix > 0].min()  # Exclude zeros
    
    # Adaptive scaling factor based on distance distribution
    mean_dist = np.mean(distance_matrix)
    scale_factor = max(1.0, mean_dist / min_dist)
    
    # Exponential discounting with adaptive decay
    exp_discount = np.exp(-distance_matrix / (scale_factor * min_dist))
    
    # 2. Two-way nearest neighbor compatibility
    # For each node, find its k nearest neighbors
    k = min(5, n // 4)  # Adaptive k based on problem size
    
    # Initialize neighbor compatibility matrix
    neighbor_compat = np.zeros((n, n))
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        neighbors_i = np.argsort(distance_matrix[i])[1:k+1]
        
        # For each neighbor j, check if i is also among j's k nearest neighbors
        for j in neighbors_i:
            neighbors_j = np.argsort(distance_matrix[j])[1:k+1]
            if i in neighbors_j:
                # Mutual nearest neighbors get bonus
                neighbor_compat[i, j] = 2.0
            else:
                # One-way nearest neighbor gets moderate bonus
                neighbor_compat[i, j] = 1.0
    
    # Make symmetric
    neighbor_compat = np.maximum(neighbor_compat, neighbor_compat.T)
    
    # 3. Cluster-based connectivity scoring
    # Identify clusters using simple thresholding
    cluster_threshold = np.percentile(distance_matrix, 30)
    
    # For each edge, calculate intra-cluster vs inter-cluster score
    cluster_score = np.zeros((n, n))
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            # Check if this is likely an intra-cluster edge
            if distance_matrix[i, j] < cluster_threshold:
                # Count how many other nodes are close to both i and j
                close_to_i = distance_matrix[i] < cluster_threshold
                close_to_j = distance_matrix[j] < cluster_threshold
                common_close = np.sum(close_to_i & close_to_j) - 2  # Exclude i and j
                
                # Intra-cluster edges with many common neighbors get higher score
                cluster_score[i, j] = 1.0 + common_close / (n - 2)
            else:
                # For inter-cluster edges, check if they connect different clusters
                # by comparing their nearest neighbors
                neighbors_i = set(np.argsort(distance_matrix[i])[1:k+1])
                neighbors_j = set(np.argsort(distance_matrix[j])[1:k+1])
                
                if len(neighbors_i.intersection(neighbors_j)) == 0:
                    # Connect nodes from different clusters - potentially important
                    cluster_score[i, j] = 0.5
                else:
                    cluster_score[i, j] = 0.2
    
    # 4. Combine all factors with weights
    base_weight = 0.4
    neighbor_weight = 0.3
    cluster_weight = 0.3
    
    combined = (base_weight * exp_discount + 
                neighbor_weight * neighbor_compat + 
                cluster_weight * cluster_score)
    
    # 5. Dynamic threshold sparsification
    # Calculate adaptive threshold based on edge scores
    upper_tri = combined[np.triu_indices(n, k=1)]
    
    if len(upper_tri) > 0:
        # Use percentile-based threshold (keep top 60% of edges)
        threshold = np.percentile(upper_tri, 40)
        
        # Sparsify: set values below threshold to zero
        sparsified = combined.copy()
        sparsified[sparsified < threshold] = 0
        
        # Ensure diagonal remains zero
        np.fill_diagonal(sparsified, 0)
        
        return sparsified
    else:
        # Fallback: return combined matrix without sparsification
        np.fill_diagonal(combined, 0)
        return combined
