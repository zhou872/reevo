import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import squareform, pdist

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with adaptive sparsification.
    Combines inverse distance, MST bonus, bidirectional regret, and degree-aware pruning.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create working copy to avoid modifying input
    D = distance_matrix.copy().astype(float)
    np.fill_diagonal(D, np.inf)
    
    # 1. Base inverse distance with adaptive smoothing
    eps = np.percentile(D[D != np.inf], 10) * 0.01 if n > 2 else 1e-8
    inv_dist = 1 / (D + eps)
    
    # 2. MST bonus with diminishing returns for large graphs
    mst = minimum_spanning_tree(D)
    mst_bonus = 1.0 + np.exp(-n/20)  # Bonus decays from ~2.0 to 1.0 as n increases
    bonus_matrix = np.ones_like(D)
    mst_mask = (mst.toarray() > 0)
    bonus_matrix[mst_mask] = mst_bonus
    
    # 3. Bidirectional regret penalty
    regret_matrix = np.ones_like(D)
    for i in range(n):
        # Get distances from node i
        dist_i = D[i].copy()
        
        # Find two best connections
        mask_i = dist_i < np.inf
        if mask_i.sum() >= 2:
            valid_dists = dist_i[mask_i]
            sorted_dists = np.sort(valid_dists)
            best_i, second_best_i = sorted_dists[:2]
            
            # Calculate regret from i's perspective
            for j in range(n):
                if i != j and D[i, j] < np.inf:
                    regret_i = (D[i, j] - best_i) / (second_best_i - best_i + eps)
                    # Also consider regret from j's perspective
                    dist_j = D[j].copy()
                    mask_j = dist_j < np.inf
                    if mask_j.sum() >= 2:
                        valid_dists_j = dist_j[mask_j]
                        sorted_dists_j = np.sort(valid_dists_j)
                        best_j, second_best_j = sorted_dists_j[:2]
                        regret_j = (D[i, j] - best_j) / (second_best_j - best_j + eps)
                        # Combine regrets multiplicatively
                        regret_matrix[i, j] *= np.exp(-(regret_i + regret_j)/2)
    
    # 4. Combined scoring
    scores = inv_dist * bonus_matrix * regret_matrix
    
    # 5. Adaptive neighborhood sparsification
    if n > 3:
        # Dynamic k based on graph size and density
        base_k = max(2, int(np.log2(n)) + 1)
        
        # Create symmetric mask
        mask = np.zeros_like(scores, dtype=bool)
        
        for i in range(n):
            # Get top edges for this node
            row_scores = scores[i].copy()
            row_scores[i] = -np.inf
            row_scores[row_scores == 0] = -np.inf
            
            # Ensure at least base_k edges if possible
            valid_edges = np.sum(row_scores > -np.inf)
            k = min(base_k, valid_edges)
            
            if k > 0:
                # Get indices of top k edges
                top_k_idx = np.argpartition(row_scores, -k)[-k:]
                mask[i, top_k_idx] = True
        
        # Make symmetric and ensure MST connectivity
        mask = mask | mask.T | mst_mask
        
        # Apply sparsification
        scores = scores * mask
    
    # Normalize scores to [0, 1] range
    valid_scores = scores[scores > 0]
    if len(valid_scores) > 0:
        max_score = valid_scores.max()
        if max_score > 0:
            scores = scores / max_score
    
    np.fill_diagonal(scores, 0)
    return scores
