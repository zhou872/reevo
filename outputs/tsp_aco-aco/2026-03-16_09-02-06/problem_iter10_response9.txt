import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with adaptive regret and lookahead scoring.
    
    Core improvements over v1:
    1. Adaptive regret weighting based on problem size and distance distribution
    2. Two-step lookahead cost approximation using nearest neighbors
    3. Dynamic degree constraints for intelligent edge preservation
    4. MST reinforcement with problem-aware scaling
    
    Parameters
    ----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances between n nodes
        
    Returns
    -------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # Working copy with diagonal set to infinity
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # ========== 1. ADAPTIVE PARAMETER CALCULATION ==========
    # Analyze problem characteristics for adaptive heuristics
    flat_dists = dist[np.triu_indices(n, k=1)]
    dist_stats = {
        'mean': np.mean(flat_dists),
        'std': np.std(flat_dists),
        'cv': np.std(flat_dists) / (np.mean(flat_dists) + 1e-12),  # Coefficient of variation
        'size': n
    }
    
    # Adaptive parameters based on problem characteristics
    regret_weight = 0.3 + 0.4 * np.tanh(dist_stats['cv'])  # Higher CV → stronger regret weighting
    lookahead_weight = min(0.2, 1.5 / np.sqrt(n))  # Larger n → less lookahead influence
    
    # ========== 2. BASE INVERSE DISTANCE SCORING ==========
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.where(dist < np.inf, 1 / (dist + 1e-12), 0)
    
    # ========== 3. ADAPTIVE REGRET CALCULATION ==========
    regret_matrix = np.ones_like(dist)
    
    for i in range(n):
        # Get distances from node i
        dist_i = dist[i].copy()
        
        # Find k nearest neighbors for robust regret calculation
        k_neighbors = min(5, n - 1)
        neighbor_indices = np.argsort(dist_i)[1:k_neighbors+1]
        neighbor_dists = dist_i[neighbor_indices]
        
        if len(neighbor_dists) >= 2:
            # Use robust statistics for regret calculation
            best_dist = neighbor_dists[0]
            second_best = np.mean(neighbor_dists[1:3]) if len(neighbor_dists) >= 3 else neighbor_dists[1]
            
            # Calculate adaptive regret for each edge
            for j in range(n):
                if i != j and dist_i[j] < np.inf:
                    if second_best - best_dist > 1e-12:
                        regret = (dist_i[j] - best_dist) / (second_best - best_dist)
                        # Convert to penalty with adaptive scaling
                        regret_matrix[i, j] = np.exp(-regret_weight * regret)
                    else:
                        regret_matrix[i, j] = 1.0
    
    # Make regret symmetric by taking geometric mean
    regret_symmetric = np.sqrt(regret_matrix * regret_matrix.T)
    
    # ========== 4. TWO-STEP LOOKAHEAD SCORING ==========
    lookahead_scores = np.ones_like(dist)
    
    for i in range(n):
        for j in range(i + 1, n):
            if dist[i, j] < np.inf:
                # Lookahead: approximate cost after taking edge (i,j)
                # Find best continuation from j (excluding i)
                dist_j = dist[j].copy()
                dist_j[i] = np.inf  # Exclude the edge we just came from
                
                if np.any(dist_j < np.inf):
                    # Best next step cost from j
                    best_next = np.min(dist_j[dist_j < np.inf])
                    # Two-step cost approximation
                    two_step_cost = dist[i, j] + best_next
                    
                    # Compare with alternative two-step paths from i
                    dist_i = dist[i].copy()
                    dist_i[j] = np.inf
                    if np.any(dist_i < np.inf):
                        best_alt_next = np.min(dist_i[dist_i < np.inf])
                        alt_target = np.argmin(dist_i)
                        # Best two-step alternative (i → alt_target → best from alt_target)
                        dist_alt = dist[alt_target].copy()
                        dist_alt[i] = np.inf
                        if np.any(dist_alt < np.inf):
                            best_from_alt = np.min(dist_alt[dist_alt < np.inf])
                            alt_two_step_cost = dist[i, alt_target] + best_from_alt
                            
                            # Score based on relative two-step cost
                            if alt_two_step_cost > 0:
                                lookahead_ratio = two_step_cost / alt_two_step_cost
                                lookahead_scores[i, j] = lookahead_scores[j, i] = np.exp(-lookahead_weight * (lookahead_ratio - 1))
    
    # ========== 5. MST REINFORCEMENT ==========
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Adaptive MST bonus based on problem size
    mst_bonus = 1.5 + 0.5 * np.tanh(20 / n)  # Larger bonus for smaller problems
    
    # ========== 6. COMBINE SCORES ==========
    base_scores = inv_dist * regret_symmetric * lookahead_scores
    base_scores[mst_mask] *= mst_bonus
    
    # ========== 7. DYNAMIC DEGREE-CONSTRAINED SPARSIFICATION ==========
    # Determine target degree for each node based on problem characteristics
    min_degree = max(2, int(np.log2(n)))
    max_degree = min(15, n // 3)
    
    # Node-specific degree targets based on centrality
    centrality = np.zeros(n)
    for i in range(n):
        valid_dists = dist[i][dist[i] < np.inf]
        if len(valid_dists) > 0:
            centrality[i] = 1.0 / (np.mean(valid_dists) + 1e-12)
    
    if np.max(centrality) > 0:
        centrality = centrality / np.max(centrality)
    
    target_degrees = np.clip(
        min_degree + centrality * (max_degree - min_degree),
        min_degree, max_degree
    ).astype(int)
    
    # Create sparsification mask
    sparsity_mask = np.zeros_like(base_scores, dtype=bool)
    
    for i in range(n):
        target_k = target_degrees[i]
        row_scores = base_scores[i].copy()
        row_scores[i] = -np.inf
        
        if target_k > 0:
            # Select top target_k edges
            threshold = np.partition(row_scores, -target_k)[-target_k]
            sparsity_mask[i, row_scores >= threshold] = True
    
    # Ensure symmetry and preserve MST edges
    sparsity_mask = (sparsity_mask | sparsity_mask.T) | mst_mask
    
    # Apply sparsification
    final_scores = base_scores * sparsity_mask
    
    # ========== 8. FINAL NORMALIZATION ==========
    np.fill_diagonal(final_scores, 0)
    
    # Row-wise normalization
    row_max = final_scores.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    final_scores = final_scores / row_max
    
    # Global normalization
    max_val = final_scores.max()
    if max_val > 0:
        final_scores = final_scores / max_val
    
    return final_scores
