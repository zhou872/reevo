import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge priority heuristic combining multiple factors:
    1. Nearest-neighbor chain reinforcement
    2. Degree-constrained sparsification
    3. Second-best alternative penalty
    4. Local cluster density bonus
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance (standard approach)
    inv_dist = np.zeros_like(distance_matrix, dtype=np.float64)
    mask = distance_matrix > 1e-12
    inv_dist[mask] = 1.0 / distance_matrix[mask]
    
    # 2. Degree-aware sparsification
    # Keep only top-k connections per node based on inverse distance
    k = max(2, int(np.sqrt(n)))  # Adaptive sparsity level
    priority = np.zeros_like(inv_dist)
    
    for i in range(n):
        # Get indices of k+1 best connections (including self)
        top_indices = np.argsort(inv_dist[i])[::-1][:k+1]
        # Remove self from consideration
        top_indices = top_indices[top_indices != i][:k]
        priority[i, top_indices] = inv_dist[i, top_indices]
    
    # Make symmetric (undirected graph assumption)
    priority = np.maximum(priority, priority.T)
    
    # 3. Nearest-neighbor chain reinforcement
    for i in range(n):
        # Find nearest neighbor
        valid_mask = priority[i] > 0
        if not np.any(valid_mask):
            continue
        nearest_idx = np.argmax(priority[i] * valid_mask)
        
        # Find nearest neighbor's nearest neighbor (excluding current)
        other_mask = priority[nearest_idx] > 0
        other_mask[i] = False  # Don't consider current node
        if np.any(other_mask):
            nn_of_nn = np.argmax(priority[nearest_idx] * other_mask)
            # Boost connection if they form a mutual chain
            if nn_of_nn != i:
                priority[i, nearest_idx] *= 1.5
                priority[nearest_idx, i] = priority[i, nearest_idx]
    
    # 4. Second-best alternative penalty
    for i in range(n):
        valid_mask = priority[i] > 0
        if np.sum(valid_mask) < 2:
            continue
        # Get top two connections
        top_two = np.argsort(priority[i])[::-1][:2]
        if len(top_two) == 2:
            ratio = priority[i, top_two[1]] / (priority[i, top_two[0]] + 1e-12)
            # Penalize edges where there's a much better alternative
            if ratio < 0.3:  # Second best is significantly worse
                priority[i, top_two[1]] *= 0.7
                priority[top_two[1], i] = priority[i, top_two[1]]
    
    # 5. Local cluster density bonus
    cluster_radius = np.percentile(distance_matrix[distance_matrix > 0], 25)
    
    for i in range(n):
        # Find nodes within cluster radius
        within_radius = distance_matrix[i] < cluster_radius
        within_radius[i] = False  # Exclude self
        cluster_nodes = np.where(within_radius)[0]
        
        if len(cluster_nodes) > 1:
            # Boost intra-cluster connections
            for j in cluster_nodes:
                if priority[i, j] > 0:
                    # Bonus proportional to cluster density
                    density_bonus = 1.0 + 0.3 * len(cluster_nodes) / n
                    priority[i, j] *= density_bonus
                    priority[j, i] = priority[i, j]
    
    # 6. Final sparsification and normalization
    # Keep only edges in top percentile
    threshold = np.percentile(priority[priority > 0], 20) if np.any(priority > 0) else 0
    priority[priority <= threshold] = 0
    
    # Normalize by maximum value for stability
    max_val = priority.max()
    if max_val > 0:
        priority = priority / max_val
    
    # Ensure symmetry
    priority = (priority + priority.T) / 2
    
    return priority
