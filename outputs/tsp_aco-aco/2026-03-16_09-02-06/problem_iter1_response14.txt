import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. MST-based importance (edges in minimum spanning tree get high scores)
    2. Local neighborhood density (edges connecting to dense clusters)
    3. Path continuity potential (edges that complete promising partial tours)
    4. Distance-based sparsification (unpromising edges set to zero)
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / distance_matrix
    
    # 1. Base inverse distance score
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.where(distance_matrix > 0, 1 / distance_matrix, 0)
    
    # 2. Minimum Spanning Tree importance
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    mst_bonus = 2.0  # Bonus multiplier for MST edges
    
    # 3. Local neighborhood density factor
    # For each node, compute average distance to k nearest neighbors
    k = min(5, n - 1)
    sorted_indices = np.argsort(distance_matrix, axis=1)[:, 1:k+1]
    avg_local_dist = np.zeros(n)
    for i in range(n):
        avg_local_dist[i] = np.mean(distance_matrix[i, sorted_indices[i]])
    
    # Density score: favors edges connecting nodes in dense regions
    density_score = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] > 0:
                # Nodes in dense regions (small avg distance) get higher scores
                density_bonus = 1.0 / (avg_local_dist[i] * avg_local_dist[j])
                density_score[i, j] = density_score[j, i] = density_bonus
    
    # Normalize density score
    if np.max(density_score) > 0:
        density_score = density_score / np.max(density_score)
    
    # 4. Path continuity factor (favor edges that connect to nodes with few good connections)
    # Compute node centrality as sum of inverse distances
    node_centrality = np.sum(inv_dist, axis=1) - np.diag(inv_dist)
    
    # Continuity score: favors edges connecting less central nodes
    continuity_score = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] > 0:
                # Lower centrality nodes need stronger connections
                continuity = 1.0 / (1 + node_centrality[i] * node_centrality[j])
                continuity_score[i, j] = continuity_score[j, i] = continuity
    
    # 5. Combined score
    heuristics_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] > 0:
                base = inv_dist[i, j]
                
                # Apply MST bonus
                if mst_mask[i, j]:
                    base *= mst_bonus
                
                # Add density and continuity factors
                combined = (base * 0.5 + 
                           density_score[i, j] * 0.25 + 
                           continuity_score[i, j] * 0.25)
                
                heuristics_matrix[i, j] = heuristics_matrix[j, i] = combined
    
    # 6. Sparsification: keep only top edges for each node
    # For each node, keep connections to top m neighbors
    m = min(8, n - 1)
    mask = np.zeros_like(heuristics_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of top m scores for node i
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Exclude self
        top_indices = np.argpartition(row, -m)[-m:]
        mask[i, top_indices] = True
    
    # Symmetrize the mask (if i connects to j, j should connect to i)
    mask = mask | mask.T
    
    # Apply sparsification
    heuristics_matrix = np.where(mask, heuristics_matrix, 0)
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix

# Example usage
if __name__ == "__main__":
    # Test with a simple example
    np.random.seed(42)
    n = 10
    points = np.random.rand(n, 2)
    from scipy.spatial.distance import pdist, squareform
    dist_matrix = squareform(pdist(points))
    
    heuristics = heuristics_v2(dist_matrix)
    print("Heuristics matrix shape:", heuristics.shape)
    print("Non-zero elements:", np.count_nonzero(heuristics))
    print("Max value:", np.max(heuristics))
    print("Min non-zero value:", heuristics[heuristics > 0].min())
