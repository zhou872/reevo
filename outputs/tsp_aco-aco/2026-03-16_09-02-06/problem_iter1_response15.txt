import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection.
    
    Features:
    1. Distance inverse with penalty for farthest edges
    2. One-sided nearest neighbor sparsification
    3. Edge centrality bonus based on node degree potential
    4. Symmetry enforcement for undirected TSP
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix of same shape, with higher values indicating
        more promising edges for inclusion in TSP tour
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    # Avoid division by zero for diagonal
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base inverse distance with farthest penalty
    inv_dist = 1.0 / safe_dist
    max_dist = np.max(safe_dist)
    dist_penalty = 1.0 - (safe_dist / max_dist)
    base_heuristic = inv_dist * dist_penalty
    
    # 2. One-sided nearest neighbor sparsification
    # Keep only k nearest neighbors for each node
    k = max(2, min(5, n // 3))  # Adaptive k based on problem size
    sparse_mask = np.zeros_like(base_heuristic, dtype=bool)
    
    for i in range(n):
        # Get indices of k smallest distances (excluding self)
        nearest = np.argsort(safe_dist[i])[:k]
        sparse_mask[i, nearest] = True
    
    # 3. Edge centrality bonus
    # Nodes with smaller average distance to others are more "central"
    node_centrality = np.zeros(n)
    for i in range(n):
        # Harmonic mean of distances for better sensitivity to small values
        with np.errstate(divide='ignore'):
            harmonic_mean = n / np.sum(1.0 / safe_dist[i])
        node_centrality[i] = 1.0 / harmonic_mean if not np.isinf(harmonic_mean) else 0
    
    # Normalize centrality scores
    if np.max(node_centrality) > 0:
        node_centrality = node_centrality / np.max(node_centrality)
    
    # Apply centrality bonus to edges
    centrality_bonus = np.zeros_like(base_heuristic)
    for i in range(n):
        for j in range(n):
            if i != j:
                centrality_bonus[i, j] = 0.5 + 0.5 * (node_centrality[i] + node_centrality[j])
    
    # 4. Combine components and enforce symmetry
    heuristic = np.zeros_like(base_heuristic)
    
    for i in range(n):
        for j in range(n):
            if i == j:
                heuristic[i, j] = 0
            elif sparse_mask[i, j] or sparse_mask[j, i]:
                # Include edge if it's in top-k for either endpoint
                combined_score = base_heuristic[i, j] * centrality_bonus[i, j]
                # Enforce symmetry for undirected TSP
                heuristic[i, j] = combined_score
                heuristic[j, i] = combined_score
            else:
                heuristic[i, j] = 0
    
    # Final normalization to keep values in reasonable range
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val
    
    return heuristic
