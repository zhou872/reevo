import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.cluster.hierarchy import linkage, fcluster
from scipy.spatial.distance import squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic scoring for TSP edges with multi-step dynamic programming regret,
    learnable factor fusion, and adaptive neural-inspired feature extraction.
    
    Key innovations:
    1. Multi-step DP regret with path lookahead using efficient DP tables
    2. Learnable factor weights optimized via graph statistics
    3. Graph-structural feature extraction via node embeddings
    4. Density-adaptive connectivity preservation with reinforcement learning principles
    
    Returns a matrix of edge scores with same shape as input.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        scores = np.ones_like(distance_matrix, dtype=float)
        np.fill_diagonal(scores, 0)
        return scores
    
    # Create working distance matrix
    dist = distance_matrix.astype(float, copy=True)
    np.fill_diagonal(dist, np.inf)
    
    # Compute basic graph statistics for learnable weights
    valid_dists = dist[dist < np.inf]
    dist_mean = np.mean(valid_dists)
    dist_std = np.std(valid_dists) if len(valid_dists) > 1 else dist_mean * 0.1
    graph_density = np.sum(dist < np.inf * 0.9) / (n * (n - 1))
    
    # 1. Base inverse distance with variance-aware smoothing
    smoothing = max(dist_mean * 0.05, dist_std * 0.02)
    inv_dist = 1 / (dist + smoothing)
    
    # 2. MST inclusion bonus with edge importance weighting
    mst_input = dist.copy()
    np.fill_diagonal(mst_input, 0)
    mst = minimum_spanning_tree(mst_input)
    mst_mask = mst.toarray() > 0
    
    # Compute MST edge importance based on centrality
    mst_centrality = np.zeros(n)
    for i in range(n):
        if np.any(mst_mask[i]):
            mst_centrality[i] = np.sum(mst_mask[i])
    
    mst_bonus = np.ones_like(dist)
    for i in range(n):
        for j in range(i+1, n):
            if mst_mask[i, j]:
                # Bonus proportional to combined centrality
                centrality_factor = 0.5 + 0.5 * (mst_centrality[i] + mst_centrality[j]) / (2 * np.max(mst_centrality))
                mst_bonus[i, j] = mst_bonus[j, i] = 1.0 + 0.8 * centrality_factor
    
    # 3. Multi-step Dynamic Programming Regret with lookahead depth L
    L = min(3, n - 1)  # Lookahead depth, max 3 for efficiency
    regret_matrix = np.ones_like(dist)
    
    if L >= 2:
        # Precompute DP tables for efficient lookahead
        # dp1[i][k] = best distance from i to any node in 1 step
        # dp2[i][k] = best distance from i to any node in 2 steps
        dp1 = dist.copy()
        dp2 = np.full_like(dist, np.inf)
        
        # Fill DP tables
        for i in range(n):
            # Sort neighbors by distance
            sorted_idx = np.argsort(dist[i])
            sorted_dist = dist[i, sorted_idx]
            
            # One-step lookahead (already in dp1)
            # Two-step lookahead via dynamic programming
            for j in range(n):
                if i == j:
                    continue
                # Find best intermediate node k
                min_two_step = np.inf
                for k in range(n):
                    if k == i or k == j:
                        continue
                    two_step = dist[i, k] + dist[k, j]
                    if two_step < min_two_step:
                        min_two_step = two_step
                dp2[i, j] = min_two_step
        
        # Compute multi-step regret for each edge
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                # One-step regret (classic)
                row_i = dist[i].copy()
                row_i[i] = np.inf
                best_i = np.min(row_i)
                row_i[row_i == best_i] = np.inf
                second_best_i = np.min(row_i)
                regret_1 = (dist[i, j] - best_i) / max(second_best_i - best_i, 1e-10)
                
                # Two-step regret using DP table
                if L >= 2 and n >= 3:
                    # Best two-step path from i (without using j as first)
                    best_two_step = np.inf
                    for k in range(n):
                        if k == i or k == j:
                            continue
                        two_step = dist[i, k] + dist[k, j]
                        if two_step < best_two_step:
                            best_two_step = two_step
                    
                    # Best alternative two-step path from i
                    alt_two_step = np.inf
                    for k in range(n):
                        if k == i:
                            continue
                        for l in range(n):
                            if l == i or l == k:
                                continue
                            alt = dist[i, k] + dist[k, l]
                            if alt < alt_two_step:
                                alt_two_step = alt
                    
                    regret_2 = (dist[i, j] + best_i - alt_two_step) / max(alt_two_step, 1e-10)
                else:
                    regret_2 = 0
                
                # Three-step regret if applicable
                if L >= 3 and n >= 4:
                    # Approximate three-step via DP principles
                    # Find best three-step path starting with i->j
                    best_three_step_with_j = dist[i, j] + dp2[j, i]  # Return to start or similar
                    
                    # Find best three-step path from i (without j)
                    best_three_step = np.inf
                    for k in range(n):
                        if k == i:
                            continue
                        for l in range(n):
                            if l == i or l == k:
                                continue
                            for m in range(n):
                                if m == i or m == k or m == l:
                                    continue
                                three_step = dist[i, k] + dist[k, l] + dist[l, m]
                                if three_step < best_three_step:
                                    best_three_step = three_step
                    
                    regret_3 = (best_three_step_with_j - best_three_step) / max(best_three_step, 1e-10)
                else:
                    regret_3 = 0
                
                # Combine multi-step regrets with learned weights
                # Weights adapt based on graph characteristics
                w1 = max(0.3, 0.7 - graph_density * 0.4)  # More weight to one-step in dense graphs
                w2 = 0.4 + graph_density * 0.3  # More weight to two-step in sparse graphs
                w3 = 0.3 - graph_density * 0.2  # Less weight to three-step in dense graphs
                
                total_weight = w1 + w2 + w3
                w1, w2, w3 = w1/total_weight, w2/total_weight, w3/total_weight
                
                combined_regret = w1 * regret_1 + w2 * regret_2 + w3 * regret_3
                
                # Convert to penalty using adaptive exponential decay
                decay_rate = 1.0 + dist_std / dist_mean if dist_mean > 0 else 1.0
                regret_matrix[i, j] = np.exp(-combined_regret * decay_rate)
    
    # 4. Graph Neural Feature Extraction via node embeddings
    # Create simple node embeddings based on distance profile
    node_features = np.zeros((n, 4))
    for i in range(n):
        valid_dists_i = dist[i][dist[i] < np.inf]
        if len(valid_dists_i) > 0:
            node_features[i, 0] = np.mean(valid_dists_i)
            node_features[i, 1] = np.std(valid_dists_i) if len(valid_dists_i) > 1 else 0
            node_features[i, 2] = np.min(valid_dists_i)
            node_features[i, 3] = len(valid_dists_i) / (n - 1)
    
    # Compute similarity bonus based on feature compatibility
    feature_bonus = np.ones_like(dist)
    for i in range(n):
        for j in range(i+1, n):
            if i != j:
                # Feature compatibility: nodes with complementary profiles get bonus
                feat_diff = np.abs(node_features[i] - node_features[j])
                compatibility = 1.0 - 0.3 * np.mean(feat_diff / (node_features[i] + node_features[j] + 1e-10))
                feature_bonus[i, j] = feature_bonus[j, i] = max(0.7, min(1.3, compatibility))
    
    # 5. Hierarchical clustering with adaptive cluster count
    if n >= 4:
        try:
            condensed_dist = squareform(dist, checks=False)
            Z = linkage(condensed_dist, method='ward')
            
            # Adaptive cluster count based on graph statistics
            base_clusters = max(2, int(np.sqrt(n) * (1.0 + graph_density)))
            clusters = fcluster(Z, base_clusters, criterion='maxclust') - 1
            
            # Intra-cluster bonus with distance normalization
            cluster_bonus = np.ones_like(dist)
            for i in range(n):
                for j in range(i+1, n):
                    if clusters[i] == clusters[j]:
                        # Normalize by cluster size and edge distance
                        cluster_size = np.sum(clusters == clusters[i])
                        norm_factor = min(1.0, 2.0 / cluster_size)
                        distance_factor = 1.0 - min(0.5, dist[i, j] / (2 * dist_mean))
                        cluster_bonus[i, j] = cluster_bonus[j, i] = 1.0 + 0.4 * norm_factor * distance_factor
        except:
            cluster_bonus = np.ones_like(dist)
    else:
        cluster_bonus = np.ones_like(dist)
    
    # 6. Learnable factor fusion with adaptive weights
    # Weights adapt based on graph characteristics
    w_inv = 0.4 - graph_density * 0.1  # Less weight to inverse distance in dense graphs
    w_mst = 0.3 + dist_std / dist_mean * 0.2 if dist_mean > 0 else 0.3  # More MST weight in varied graphs
    w_regret = 0.5 + graph_density * 0.2  # More regret weight in dense graphs
    w_feature = 0.3 - dist_std / dist_mean * 0.1 if dist_mean > 0 else 0.3  # Less feature weight in uniform graphs
    w_cluster = 0.2 + (1 - graph_density) * 0.3  # More cluster weight in sparse graphs
    
    # Normalize weights
    total_w = w_inv + w_mst + w_regret + w_feature + w_cluster
    w_inv, w_mst, w_regret, w_feature, w_cluster = [w/total_w for w in [w_inv, w_mst, w_regret, w_feature, w_cluster]]
    
    # Combined scoring
    scores = (
        w_inv * inv_dist +
        w_mst * mst_bonus +
        w_regret * regret_matrix +
        w_feature * feature_bonus +
        w_cluster * cluster_bonus
    )
    
    # 7. Reinforcement Learning-inspired adaptive sparsification
    # Base connectivity based on graph density
    base_k = max(2, int(np.sqrt(n) * (0.8 + 0.4 * graph_density)))
    
    # Create adaptive mask
    mask = np.zeros_like(scores, dtype=bool)
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        # Adaptive k per node based on local connectivity
        local_density = np.sum(dist[i] < dist_mean * 1.5) / (n - 1)
        node_k = max(2, int(base_k * (0.7 + 0.6 * local_density)))
        
        if np.any(row_scores > -np.inf):
            top_k = min(node_k, np.sum(row_scores > -np.inf))
            if top_k > 0:
                threshold = np.partition(row_scores, -top_k)[-top_k]
                mask[i, row_scores >= threshold] = True
    
    # Ensure symmetry and preserve MST edges
    mask = mask | mask.T | mst_mask
    
    # Reinforcement learning style: preserve edges with high regret penalty
    regret_threshold = np.percentile(regret_matrix[regret_matrix < 1.0], 20)
    high_regret_mask = regret_matrix <= regret_threshold
    mask = mask | high_regret_mask
    
    # Final connectivity check
    for i in range(n):
        if mask[i].sum() < 2:
            row_scores = scores[i].copy()
            row_scores[mask[i]] = -np.inf
            row_scores[i] = -np.inf
            needed = 2 - mask[i].sum()
            if needed > 0 and np.any(row_scores > -np.inf):
                top_idx = np.argsort(row_scores)[-needed:]
                mask[i, top_idx] = True
                mask[top_idx, i] = True
    
    # Apply mask
    scores = scores * mask
    
    # 8. Normalize with outlier-robust scaling
    np.fill_diagonal(scores, 0)
    valid_scores = scores[scores > 0]
    
    if len(valid_scores) > 1:
        # Use percentile-based normalization to reduce outlier effects
        p5, p95 = np.percentile(valid_scores, [5, 95])
        if p95 > p5:
            scores[scores > 0] = np.clip((scores[scores > 0] - p5) / (p95 - p5), 0, 1)
        else:
            scores[scores > 0] = 1.0
    elif len(valid_scores) == 1:
        scores[scores > 0] = 1.0
    
    return scores
