import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection.
    Combines multiple factors and sparsifies unpromising edges.
    """
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix)
    
    # Prevent division by zero and self-loops
    np.fill_diagonal(distance_matrix, np.inf)
    
    # 1. Base inverse distance (normalized)
    inv_dist = 1 / distance_matrix
    inv_dist_norm = inv_dist / np.max(inv_dist)
    
    # 2. Clarke-Wright savings heuristic variant
    # Savings(i,j) = d(i,center) + d(center,j) - d(i,j)
    # Use average node as virtual "center"
    center_dist = np.mean(distance_matrix, axis=1)
    savings = center_dist[:, None] + center_dist[None, :] - distance_matrix
    savings_norm = savings / np.max(savings)
    
    # 3. Local clustering coefficient proxy
    # For each node, how many of its k-nearest neighbors are connected to each other
    k = min(5, n // 4)
    cluster_score = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nearest = np.argsort(distance_matrix[i])[1:k+1]
        if len(nearest) > 1:
            # Compute average distance among neighbors
            neighbor_dists = distance_matrix[np.ix_(nearest, nearest)]
            cluster_density = 1 / (1 + np.mean(neighbor_dists[np.triu_indices_from(neighbor_dists, k=1)]))
            for j in nearest:
                if i != j:
                    cluster_score[i, j] = cluster_density
    
    cluster_norm = cluster_score / np.max(cluster_score) if np.max(cluster_score) > 0 else cluster_score
    
    # 4. Edge betweenness approximation
    # Edge (i,j) is important if it's on short paths between many pairs
    betweenness = np.zeros_like(distance_matrix)
    
    # For each node pair (p,q), identify if edge (i,j) is on their shortest 2-step path
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Check if (i,j) is part of short detours for other pairs
            for k in range(n):
                if k == i or k == j:
                    continue
                # Path k -> i -> j and i -> j -> k
                path1 = distance_matrix[k, i] + distance_matrix[i, j]
                path2 = distance_matrix[i, j] + distance_matrix[j, k]
                
                # If this edge creates a shorter detour than direct connection
                direct1 = distance_matrix[k, j]
                direct2 = distance_matrix[i, k]
                
                if path1 < direct1 * 1.5:  # Within 50% longer than direct
                    betweenness[i, j] += 1 / (1 + path1 - direct1)
                if path2 < direct2 * 1.5:
                    betweenness[i, j] += 1 / (1 + path2 - direct2)
    
    betweenness_norm = betweenness / np.max(betweenness) if np.max(betweenness) > 0 else betweenness
    
    # 5. Combine factors with weights
    weights = [0.35, 0.25, 0.2, 0.2]  # Sum to 1.0
    combined = (weights[0] * inv_dist_norm +
                weights[1] * savings_norm +
                weights[2] * cluster_norm +
                weights[3] * betweenness_norm)
    
    # 6. Sparsification: Keep only promising edges
    # Strategy: Keep top edges for each node + edges with high combined score
    sparsified = np.zeros_like(combined)
    
    # Keep top m edges for each node
    m = min(10, n // 2)
    for i in range(n):
        top_indices = np.argsort(combined[i])[-m:]
        sparsified[i, top_indices] = combined[i, top_indices]
    
    # Also keep edges with overall high score
    threshold = np.percentile(combined[combined > 0], 70)  # Top 30%
    high_score_mask = combined > threshold
    sparsified[high_score_mask] = np.maximum(sparsified[high_score_mask], combined[high_score_mask])
    
    # Ensure symmetry (optional for undirected TSP)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Zero out diagonal
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
