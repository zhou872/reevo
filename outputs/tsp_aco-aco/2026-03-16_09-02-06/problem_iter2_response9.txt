import numpy as np
from scipy.spatial import ConvexHull
from sklearn.manifold import MDS

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge priority heuristic with regret-based lookahead and adaptive clustering.
    
    Features:
    1. Regret-based exclusion cost: Measures penalty of not using an edge by comparing
       alternatives for both endpoints
    2. Adaptive density clustering: Dynamically identifies clusters based on global
       distance distribution rather than fixed radius
    3. Two-step local optimality: Considers secondary connections when evaluating edges
    4. Dynamic sparsification: Adjusts connectivity based on problem size and structure
    
    Returns a sparse priority matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # Base inverse distance (handle zeros)
    inv_dist = np.zeros_like(distance_matrix, dtype=np.float64)
    mask = distance_matrix > 1e-12
    inv_dist[mask] = 1.0 / distance_matrix[mask]
    np.fill_diagonal(inv_dist, 0)
    
    # 1. Dynamic sparsification - adapt connectivity to problem size
    # More connected for small problems, sparser for large ones
    base_k = max(3, int(np.sqrt(n)))
    k = min(base_k, n - 1)
    
    priority = np.zeros_like(inv_dist)
    for i in range(n):
        # Keep top-k connections based on inverse distance
        top_indices = np.argsort(inv_dist[i])[::-1][:k+1]
        top_indices = top_indices[top_indices != i][:k]
        priority[i, top_indices] = inv_dist[i, top_indices]
    
    # Ensure symmetry
    priority = np.maximum(priority, priority.T)
    
    # 2. Adaptive density clustering
    # Use distance distribution to identify natural clusters
    if n >= 4:
        # Flatten distances (excluding diagonal)
        flat_dists = distance_matrix[~np.eye(n, dtype=bool)]
        
        # Use quartiles to identify cluster thresholds
        q25 = np.percentile(flat_dists, 25)
        q50 = np.percentile(flat_dists, 50)
        
        # Create multiple cluster levels
        cluster_levels = []
        for threshold in [q25, q50]:
            cluster_masks = []
            for i in range(n):
                mask = distance_matrix[i] < threshold
                mask[i] = False
                cluster_masks.append(np.where(mask)[0])
            cluster_levels.append(cluster_masks)
        
        # Boost intra-cluster connections based on cluster size
        for level_idx, cluster_masks in enumerate(cluster_levels):
            level_weight = 1.2 - 0.2 * level_idx  # Higher weight for tighter clusters
            
            for i in range(n):
                cluster_nodes = cluster_masks[i]
                if len(cluster_nodes) > 1:
                    for j in cluster_nodes:
                        if priority[i, j] > 0:
                            # Bonus increases with cluster density
                            density_factor = 1.0 + level_weight * len(cluster_nodes) / n
                            priority[i, j] *= density_factor
                            priority[j, i] = priority[i, j]
    
    # 3. Regret-based lookahead for edge exclusion cost
    for i in range(n):
        # Get all candidate edges for this node
        candidates = np.where(priority[i] > 0)[0]
        if len(candidates) < 2:
            continue
        
        # Find best and second-best edges for this node
        candidate_vals = priority[i, candidates]
        best_idx = np.argmax(candidate_vals)
        best_val = candidate_vals[best_idx]
        
        # Remove best to find second best
        mask_second = np.ones(len(candidate_vals), dtype=bool)
        mask_second[best_idx] = False
        if np.any(mask_second):
            second_best_val = np.max(candidate_vals[mask_second])
            
            # Calculate regret for each candidate edge
            for idx, j in enumerate(candidates):
                current_val = candidate_vals[idx]
                
                # Regret measures how much worse alternatives are if we don't use this edge
                if j == candidates[best_idx]:
                    # For the best edge: regret is difference to second best
                    regret = (best_val - second_best_val) / (best_val + 1e-12)
                else:
                    # For other edges: how much worse than best alternative
                    regret = (best_val - current_val) / (best_val + 1e-12)
                
                # Apply regret-based adjustment
                # Higher regret edges (better alternatives exist) get penalized
                # But very high regret for best edge means it's crucial
                if j == candidates[best_idx] and regret > 0.3:
                    # Crucial edge gets boost
                    priority[i, j] *= (1.0 + 0.5 * regret)
                elif regret > 0.5:
                    # Poor alternative gets penalized
                    priority[i, j] *= (1.0 - 0.3 * regret)
                
                # Ensure symmetry
                priority[j, i] = priority[i, j]
    
    # 4. Two-step local optimality check
    # Consider secondary connections when evaluating edges
    for i in range(n):
        candidates = np.where(priority[i] > 0)[0]
        if len(candidates) < 2:
            continue
        
        for j in candidates:
            if i >= j:  # Process each edge once
                continue
            
            # Look at j's other connections
            j_candidates = np.where(priority[j] > 0)[0]
            j_candidates = j_candidates[j_candidates != i]
            
            if len(j_candidates) > 0:
                # Check if edge (i,j) helps connect to good secondary nodes
                secondary_benefit = 0
                for k in j_candidates[:min(3, len(j_candidates))]:
                    if priority[i, k] > 0 or priority[j, k] > 0:
                        secondary_benefit += 0.1
                
                if secondary_benefit > 0:
                    bonus = 1.0 + 0.15 * secondary_benefit
                    priority[i, j] *= bonus
                    priority[j, i] = priority[i, j]
    
    # 5. Convex hull reinforcement for large instances
    if n >= 10:
        try:
            # Use MDS to get approximate 2D coordinates
            mds = MDS(n_components=2, dissimilarity='precomputed', 
                     random_state=42, max_iter=200)
            coords = mds.fit_transform(distance_matrix)
            
            # Find convex hull vertices
            hull = ConvexHull(coords)
            hull_vertices = hull.vertices
            
            # Boost edges on convex hull
            for idx in range(len(hull_vertices)):
                v1 = hull_vertices[idx]
                v2 = hull_vertices[(idx + 1) % len(hull_vertices)]
                
                if priority[v1, v2] > 0:
                    priority[v1, v2] *= 1.3
                    priority[v2, v1] = priority[v1, v2]
        except:
            pass  # Fallback if convex hull computation fails
    
    # Final sparsification: keep only the most promising edges
    # Use adaptive threshold based on edge distribution
    if np.any(priority > 0):
        non_zero = priority[priority > 0]
        
        # Determine threshold: keep more edges for small problems
        if n <= 20:
            threshold = np.percentile(non_zero, 10)  # Keep 90%
        elif n <= 50:
            threshold = np.percentile(non_zero, 20)  # Keep 80%
        else:
            threshold = np.percentile(non_zero, 30)  # Keep 70%
        
        priority[priority <= threshold] = 0
    
    # Normalize to [0, 1] range
    max_val = priority.max()
    if max_val > 0:
        priority = priority / max_val
    
    # Ensure perfect symmetry
    priority = (priority + priority.T) / 2
    np.fill_diagonal(priority, 0)
    
    return priority
