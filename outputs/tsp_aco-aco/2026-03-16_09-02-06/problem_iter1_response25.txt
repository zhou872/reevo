import numpy as np
from scipy.spatial import KDTree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection combining multiple factors:
    1. Inverse distance weighting with normalization
    2. Nearest-neighbor chain reinforcement
    3. Penalization of isolated nodes/clusters
    4. Selective sparsification of unlikely edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        # Base case: return inverse distance for trivial instances
        return np.where(distance_matrix > 0, 1 / distance_matrix, 0)
    
    # 1. Base inverse distance with smoothing
    # Add small epsilon to avoid division by zero and overemphasis of tiny distances
    eps = np.finfo(distance_matrix.dtype).eps * 10
    inv_dist = 1 / (distance_matrix + eps)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Normalize by node-specific scaling factors
    # Compute average inverse distance for each node to its k nearest neighbors
    k = min(5, n - 1)
    nearest_indices = np.argpartition(distance_matrix, k+1, axis=1)[:, 1:k+1]
    row_masks = np.ogrid[:n, :k][0]
    nearest_dists = distance_matrix[row_masks, nearest_indices]
    node_scale = np.mean(1 / (nearest_dists + eps), axis=1, keepdims=True)
    
    # Scale inverse distances by node-specific factors
    # This gives higher weight to edges from nodes with generally poor connections
    scaled_inv = inv_dist / (node_scale + eps)
    
    # 3. Nearest-neighbor chain reinforcement
    # For each node, identify its two nearest neighbors
    nn1 = np.argsort(distance_matrix, axis=1)[:, 1]
    nn2 = np.argsort(distance_matrix, axis=1)[:, 2]
    
    # Create reinforcement matrix for edges that form potential "chains"
    chain_bonus = np.zeros_like(distance_matrix)
    for i in range(n):
        # Edge i -> nn1(i) gets bonus
        chain_bonus[i, nn1[i]] += 0.3
        # Edge i -> nn2(i) gets smaller bonus
        chain_bonus[i, nn2[i]] += 0.15
        # Reciprocal edges also get bonus (symmetry)
        chain_bonus[nn1[i], i] += 0.3
        chain_bonus[nn2[i], i] += 0.15
    
    # 4. Cluster/connectivity penalty
    # Compute "isolation score" for each node based on MST-like connectivity
    # Use Prim's algorithm to estimate connectivity
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    connectivity = np.zeros(n)
    
    for _ in range(1, n):
        # Find min distance from visited to unvisited nodes
        mask = visited[:, None] & ~visited[None, :]
        if not np.any(mask):
            break
        min_dists = np.where(mask, distance_matrix, np.inf)
        min_idx = np.unravel_index(np.argmin(min_dists), distance_matrix.shape)
        connectivity[min_idx[1]] = distance_matrix[min_idx]
        visited[min_idx[1]] = True
    
    # Penalize edges connecting nodes with high connectivity costs
    connectivity_penalty = connectivity[:, None] + connectivity[None, :]
    connectivity_factor = np.exp(-connectivity_penalty / np.mean(connectivity_penalty))
    
    # 5. Combine all factors
    combined = scaled_inv * (1 + chain_bonus) * connectivity_factor
    
    # 6. Selective sparsification
    # For each node, keep only top-m edges and small edges that might connect clusters
    m = min(max(3, int(np.log2(n))), n - 1)
    
    sparsified = np.zeros_like(combined)
    
    # Strategy A: Keep top-m edges by combined score for each node
    for i in range(n):
        top_indices = np.argpartition(combined[i], -m)[-m:]
        sparsified[i, top_indices] = combined[i, top_indices]
    
    # Strategy B: Also preserve very short edges regardless of other factors
    # These might connect nearby clusters
    dist_threshold = np.percentile(distance_matrix[distance_matrix > 0], 10)
    short_edges = distance_matrix <= dist_threshold
    np.fill_diagonal(short_edges, False)
    
    # For short edges that weren't already kept, add them with reduced weight
    missed_short = short_edges & (sparsified == 0)
    if np.any(missed_short):
        sparsified[missed_short] = 0.5 * inv_dist[missed_short]
    
    # 7. Symmetrize the result (TSP is undirected)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 8. Final normalization to [0, 1] range
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
