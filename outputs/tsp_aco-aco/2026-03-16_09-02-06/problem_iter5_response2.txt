import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with improved symmetric regret calculation
    and adaptive sparsification.
    
    Features:
    1. Inverse distance weighting (base attractiveness)
    2. Minimum spanning tree connectivity bonus
    3. Symmetric regret penalization (considering both endpoints)
    4. Adaptive degree-aware sparsification
    
    Args:
        distance_matrix: Square matrix of distances between nodes
        
    Returns:
        Heuristic matrix of same shape, with higher values indicating more promising edges
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # Create a copy to avoid modifying input
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    
    # 1. Base inverse distance with smoothing
    inv_dist = 1 / (dist + 1e-10)
    
    # 2. MST connectivity bonus
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    mst_bonus = np.ones_like(dist)
    mst_bonus[mst_mask] = 2.0  # MST edges get 2x bonus
    
    # 3. Symmetric regret calculation
    regret_matrix = np.ones_like(dist)
    
    # Precompute best and second-best distances for each node
    best_dists = np.full(n, np.inf)
    second_best_dists = np.full(n, np.inf)
    
    for i in range(n):
        distances_i = dist[i].copy()
        distances_i[i] = np.inf  # Ignore self
        sorted_vals = np.sort(distances_i)
        best_dists[i] = sorted_vals[0]
        second_best_dists[i] = sorted_vals[1]
    
    # Compute symmetric regret for each edge
    for i in range(n):
        for j in range(i+1, n):  # Only compute once for each undirected edge
            if i == j:
                continue
                
            # Regret from i's perspective
            regret_i = 0
            if second_best_dists[i] > best_dists[i]:
                regret_i = (dist[i, j] - best_dists[i]) / (second_best_dists[i] - best_dists[i])
            
            # Regret from j's perspective
            regret_j = 0
            if second_best_dists[j] > best_dists[j]:
                regret_j = (dist[j, i] - best_dists[j]) / (second_best_dists[j] - best_dists[j])
            
            # Combined symmetric regret penalty
            avg_regret = (regret_i + regret_j) / 2
            penalty = np.exp(-avg_regret)
            
            regret_matrix[i, j] = penalty
            regret_matrix[j, i] = penalty
    
    # 4. Combined scoring
    scores = inv_dist * mst_bonus * regret_matrix
    
    # 5. Adaptive degree-aware sparsification
    # Keep more edges for smaller graphs, fewer for larger
    base_k = max(2, int(np.sqrt(n)))
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        # Get adaptive k for this node based on its connectivity pattern
        row_distances = dist[i].copy()
        row_distances[i] = np.inf
        std_dev = np.std(row_distances[row_distances < np.inf])
        
        # Adjust k based on distance distribution
        if std_dev > np.mean(row_distances[row_distances < np.inf]) * 0.5:
            k = base_k + 1  # More edges for heterogeneous distances
        else:
            k = base_k
        
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf  # Exclude self
        
        # Keep top k edges for this node
        if k < n-1:
            top_k_idx = np.argpartition(row_scores, -k)[-k:]
            mask[i, top_k_idx] = True
            # Ensure symmetry
            for idx in top_k_idx:
                mask[idx, i] = True
    
    # Ensure MST edges are always included
    mask = mask | mst_mask
    
    # Apply sparsification
    scores = scores * mask
    
    # Normalize scores to [0, 1] range
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    np.fill_diagonal(scores, 0)  # Zero out diagonal
    
    return scores
