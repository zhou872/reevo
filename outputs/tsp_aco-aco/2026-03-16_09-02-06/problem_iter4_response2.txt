import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristic integrating Christofides algorithm insights.
    
    Features:
    1. MST Connectivity Bonus: Edges in minimum spanning tree receive base bonus
    2. Two-step Regret Enhancement: Measures importance by comparing direct edge to
       best two-step alternative path
    3. Greedy Odd-Vertex Matching: Approximates Christofides matching on MST odd-degree
       vertices with greedy nearest-neighbor matching
    4. Logarithmic Sparsification: Maintains connectivity via MST while preserving
       top O(log n) edges per node
    
    Returns priority matrix with higher values indicating more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance with diagonal handling
    inv_dist = np.zeros_like(distance_matrix, dtype=np.float64)
    np.fill_diagonal(distance_matrix, np.inf)
    inv_dist = 1.0 / (distance_matrix + 1e-12)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(distance_matrix)
    mst_edges = (mst.toarray() > 0).astype(float)
    mst_bonus = 1.0 + mst_edges  # MST edges get factor 2, others 1
    
    # 3. Identify odd-degree vertices in MST for matching bonus
    odd_vertices = []
    if n >= 4:
        mst_degree = np.count_nonzero(mst_edges, axis=1)
        odd_vertices = np.where(mst_degree % 2 == 1)[0]
    
    # Greedy matching on odd vertices (nearest neighbor pairing)
    matching_bonus = np.ones_like(distance_matrix, dtype=np.float64)
    if len(odd_vertices) >= 2:
        odd_mask = np.zeros(n, dtype=bool)
        odd_mask[odd_vertices] = True
        remaining = set(odd_vertices.tolist())
        
        while len(remaining) >= 2:
            i = min(remaining)
            remaining.remove(i)
            
            # Find nearest remaining odd vertex
            best_j = -1
            best_dist = np.inf
            for j in remaining:
                if distance_matrix[i, j] < best_dist:
                    best_dist = distance_matrix[i, j]
                    best_j = j
            
            if best_j != -1:
                remaining.remove(best_j)
                matching_bonus[i, best_j] = 1.5
                matching_bonus[best_j, i] = 1.5
    
    # 4. Two-step regret calculation (comparison to best alternative path)
    regret_factor = np.ones_like(distance_matrix, dtype=np.float64)
    
    if n <= 200:  # Exact calculation for moderate-sized instances
        for i in range(n):
            for j in range(n):
                if i != j:
                    # Find minimum two-step path i -> k -> j
                    best_two_step = np.inf
                    for k in range(n):
                        if k != i and k != j:
                            two_step = distance_matrix[i, k] + distance_matrix[k, j]
                            if two_step < best_two_step:
                                best_two_step = two_step
                    
                    # Regret factor: ratio of alternative to direct edge
                    if distance_matrix[i, j] > 0:
                        regret = best_two_step / distance_matrix[i, j]
                        regret_factor[i, j] = min(2.0, 1.0 + 0.5 * (regret - 1.0))
    else:  # Approximate for large instances using node-based regret
        for i in range(n):
            distances_i = distance_matrix[i].copy()
            distances_i[i] = np.inf
            
            sorted_idx = np.argsort(distances_i)
            best_dist = distances_i[sorted_idx[0]]
            second_best_dist = distances_i[sorted_idx[1]]
            
            for j in range(n):
                if i != j:
                    regret = (distance_matrix[i, j] - best_dist) / (second_best_dist - best_dist + 1e-10)
                    regret_factor[i, j] = np.exp(-regret)
    
    # 5. Combine all factors
    scores = inv_dist * mst_bonus * matching_bonus * regret_factor
    
    # 6. Adaptive sparsification preserving MST connectivity
    # Keep MST edges plus top O(log n) edges per node
    k = max(2, int(np.log2(n)) + 1)
    mask = mst_edges.copy()  # Always include MST edges
    
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        top_k_idx = np.argpartition(row_scores, -k)[-k:]
        for idx in top_k_idx:
            mask[i, idx] = 1
            mask[idx, i] = 1
    
    scores = scores * mask
    
    # 7. Final normalization and symmetry enforcement
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    scores = (scores + scores.T) / 2.0
    np.fill_diagonal(scores, 0)
    
    return scores
