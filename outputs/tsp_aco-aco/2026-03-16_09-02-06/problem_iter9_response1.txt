import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with improved strategies:
    1. Symmetrized regret averaging (consider both endpoints)
    2. Adaptive degree sparsification with edge betweenness
    3. MST connectivity with enhanced weighting
    4. Edge centrality measures
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    if n == 1:
        return np.array([[0]])
    
    # Create a copy to avoid modifying input
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    
    # 1. Enhanced inverse distance with power law
    inv_dist = 1 / (dist ** 1.25 + 1e-10)
    
    # 2. MST with weighted bonus based on edge importance
    mst = minimum_spanning_tree(dist)
    mst_array = (mst.toarray() > 0).astype(float)
    
    # Calculate MST edge importance as inverse of MST weight
    mst_weights = np.zeros_like(mst_array)
    mst_mask = mst_array > 0
    mst_weights[mst_mask] = 1 / (dist[mst_mask] + 1e-10)
    # Normalize MST weights
    if mst_weights[mst_mask].size > 0:
        max_mst_weight = mst_weights[mst_mask].max()
        if max_mst_weight > 0:
            mst_weights[mst_mask] = 1 + (mst_weights[mst_mask] / max_mst_weight)
    
    # 3. Symmetrized regret calculation
    regret_matrix = np.zeros_like(dist)
    
    # Precompute best and second best for each node
    best_dists = np.zeros(n)
    second_best_dists = np.zeros(n)
    
    for i in range(n):
        distances_i = dist[i].copy()
        distances_i[i] = np.inf
        sorted_vals = np.sort(distances_i)
        best_dists[i] = sorted_vals[0]
        second_best_dists[i] = sorted_vals[1] if len(sorted_vals) > 1 else sorted_vals[0]
    
    # Calculate symmetric regret
    for i in range(n):
        for j in range(i+1, n):
            if i != j:
                # Regret from i's perspective
                regret_i = (dist[i, j] - best_dists[i]) / (second_best_dists[i] - best_dists[i] + 1e-10)
                # Regret from j's perspective
                regret_j = (dist[j, i] - best_dists[j]) / (second_best_dists[j] - best_dists[j] + 1e-10)
                # Symmetrized regret (average)
                avg_regret = (regret_i + regret_j) / 2
                # Convert to attractiveness (higher is better)
                regret_matrix[i, j] = regret_matrix[j, i] = np.exp(-avg_regret)
    
    # 4. Edge betweenness approximation for centrality
    centrality = np.zeros_like(dist)
    if n > 2:
        # Simple betweenness approximation using shortest paths through 2-hop neighbors
        for k in range(n):
            for i in range(n):
                if i != k:
                    for j in range(i+1, n):
                        if j != k and j != i:
                            # If k lies on shortest 2-hop path between i and j
                            ikj_dist = dist[i, k] + dist[k, j]
                            if ikj_dist < dist[i, j] * 1.1:  # Within 10% of direct
                                centrality[i, j] += 1
                                centrality[j, i] += 1
        
        # Normalize centrality scores
        if centrality.max() > 0:
            centrality = centrality / centrality.max()
    
    # 5. Combined scoring
    scores = inv_dist * mst_weights * regret_matrix * (1 + 0.5 * centrality)
    
    # 6. Adaptive degree sparsification
    mask = np.zeros_like(scores, dtype=bool)
    
    # Determine adaptive k per node based on score distribution
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        # Calculate dynamic threshold
        non_inf_scores = row_scores[row_scores > -np.inf]
        if len(non_inf_scores) > 0:
            mean_score = np.mean(non_inf_scores)
            std_score = np.std(non_inf_scores)
            threshold = max(mean_score - 0.5 * std_score, np.percentile(non_inf_scores, 25))
        else:
            threshold = -np.inf
        
        # Apply threshold and ensure minimum connectivity
        candidate_mask = row_scores >= threshold
        if candidate_mask.sum() < 2:  # Ensure at least 2 connections
            top_idx = np.argsort(row_scores)[-2:]
            candidate_mask = np.zeros(n, dtype=bool)
            candidate_mask[top_idx] = True
        
        # Limit maximum connections to avoid over-connection
        max_connections = min(n // 2 + 1, n - 1)
        if candidate_mask.sum() > max_connections:
            top_idx = np.argsort(row_scores)[-max_connections:]
            candidate_mask = np.zeros(n, dtype=bool)
            candidate_mask[top_idx] = True
        
        mask[i, :] = candidate_mask
    
    # Ensure symmetry
    mask = mask | mask.T
    
    # 7. Ensure MST connectivity (critical edges)
    mask = mask | (mst_array > 0)
    
    # 8. Apply sparsification and final normalization
    scores = scores * mask
    
    # Softmax normalization for better probability interpretation
    if scores.sum() > 0:
        # Shift scores to avoid overflow
        max_score = scores.max()
        if max_score > 0:
            exp_scores = np.exp((scores - max_score) * 2)  # Scale for sharper distribution
            scores = exp_scores / exp_scores.sum(axis=1, keepdims=True)
    
    np.fill_diagonal(scores, 0)
    return scores
