import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection that combines multiple factors
    and sparsifies the matrix by zeroing unpromising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base reciprocal distance (preserve original information)
    with np.errstate(divide='ignore', invalid='ignore'):
        reciprocal = 1 / distance_matrix
        reciprocal[np.isinf(reciprocal)] = 0
        np.fill_diagonal(reciprocal, 0)
    
    # 2. Adaptive sparsification based on node density
    # For each node, keep connections to its k nearest neighbors
    k = max(2, int(np.sqrt(n)))  # Adaptive neighborhood size
    
    # Create nearest neighbor mask
    mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        # Get indices of k+1 smallest distances (including self)
        nearest = np.argpartition(distance_matrix[i], k+1)[:k+1]
        nearest = nearest[nearest != i]  # Remove self
        mask[i, nearest] = True
    
    # 3. Mutual preference score
    # Count how many nodes prefer each edge as one of their best connections
    preference = np.zeros_like(distance_matrix)
    
    # For each node, rank its connections
    for i in range(n):
        # Get distances to all other nodes
        dists = distance_matrix[i].copy()
        dists[i] = np.inf  # Ignore self
        
        # Create preference scores based on rank
        ranks = np.argsort(np.argsort(dists))
        # Convert ranks to scores (higher score for better ranks)
        pref_scores = (n - 1 - ranks) / (n - 1)
        preference[i] = pref_scores
    
    # Mutual preference: geometric mean of both nodes' preferences
    mutual_pref = np.sqrt(preference * preference.T)
    
    # 4. Triangle inequality penalty
    # Penalize edges that are significantly longer than indirect paths
    triangle_penalty = np.ones_like(distance_matrix)
    
    # For each edge (i,j), check if there exists a node k that provides
    # a much shorter two-step path i->k->j
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Find the best intermediate node (excluding i and j)
            others = [k for k in range(n) if k not in (i, j)]
            if others:
                indirect_lengths = distance_matrix[i, others] + distance_matrix[others, j]
                best_indirect = np.min(indirect_lengths)
                
                # If direct edge is much worse than best indirect path, penalize it
                if distance_matrix[i, j] > 1.5 * best_indirect:
                    penalty = best_indirect / distance_matrix[i, j]
                    triangle_penalty[i, j] = max(0.1, penalty)  # Cap penalty at 0.1
    
    # 5. Degree awareness
    # Nodes with fewer good connections get higher weights for their good edges
    degree_awareness = np.ones_like(distance_matrix)
    
    # For each node, compute how many "good" connections it has
    for i in range(n):
        # Good connections are those in the reciprocal top quartile
        good_threshold = np.percentile(reciprocal[i][reciprocal[i] > 0], 75)
        good_count = np.sum(reciprocal[i] >= good_threshold)
        
        # If node has few good connections, boost its best edges
        if good_count < 3 and good_count > 0:
            boost_factor = 3.0 / good_count
            best_edges = reciprocal[i] >= good_threshold
            degree_awareness[i, best_edges] = boost_factor
    
    # 6. Combine all factors
    heuristics_matrix = reciprocal * mutual_pref * triangle_penalty * degree_awareness
    
    # Apply sparsification mask - zero out non-promising edges
    heuristics_matrix = heuristics_matrix * (mask | mask.T)  # Keep if either endpoint considers it promising
    
    # Normalize to preserve scale relative to original distances
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        scale_factor = np.percentile(non_zero, 90) / np.percentile(reciprocal[reciprocal > 0], 90)
        heuristics_matrix *= scale_factor
    
    # Ensure symmetry and clean up
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
