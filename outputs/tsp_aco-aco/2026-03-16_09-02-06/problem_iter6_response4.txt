import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse.csgraph import connected_components

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with improved adaptive mechanisms and efficiency.
    
    Key features:
    1. Exponential regret penalization for smooth edge downweighting
    2. Local spatial density features to balance tour connectivity
    3. Adaptive factor weighting based on graph variability
    4. Efficient connectivity-preserving sparsification
    
    Parameters
    ----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between nodes
        
    Returns
    -------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix) if n == 0 else 1 / (distance_matrix + np.eye(n))
    
    # Create working copy with diagonal masked
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    
    # 1. Base inverse distance with adaptive smoothing
    median_dist = np.median(dist[dist < np.inf])
    inv_dist = 1 / (dist + 0.1 * median_dist)
    
    # 2. MST connectivity bonus (vectorized)
    mst = minimum_spanning_tree(dist)
    mst_array = mst.toarray()
    mst_bonus = np.ones_like(dist)
    mst_bonus[mst_array > 0] = 2.0  # Double bonus for MST edges
    
    # 3. Local spatial density feature (vectorized)
    k_neighbors = min(5, n - 1)
    # Get k-nearest neighbor distances for each node
    sorted_dists = np.partition(dist, k_neighbors, axis=1)[:, :k_neighbors+1]
    valid_mask = sorted_dists < np.inf
    local_density = np.zeros(n)
    
    for i in range(n):
        valid_dists = sorted_dists[i, valid_mask[i]]
        if len(valid_dists) > 0:
            local_density[i] = 1.0 / np.mean(valid_dists)
    
    if local_density.max() > 0:
        local_density = local_density / local_density.max()
    
    # Create spatial factor using broadcasting
    sparsity_factor = 1.0 - 0.4 * (local_density[:, np.newaxis] + local_density[np.newaxis, :])
    spatial_bonus = np.clip(1.0 + sparsity_factor, 0.6, 1.4)
    np.fill_diagonal(spatial_bonus, 1.0)
    
    # 4. Exponential regret penalization (vectorized)
    # Find best and second-best distances for each node
    dist_copy = dist.copy()
    dist_copy[np.diag_indices(n)] = np.inf
    
    # Get best and second-best distances
    sorted_idx = np.argsort(dist_copy, axis=1)
    best_dists = dist_copy[np.arange(n), sorted_idx[:, 0]]
    second_best_dists = dist_copy[np.arange(n), sorted_idx[:, 1]]
    
    # Compute regret matrix
    regret_numer = dist - best_dists[:, np.newaxis]
    regret_denom = second_best_dists[:, np.newaxis] - best_dists[:, np.newaxis] + 1e-10
    regret = regret_numer / regret_denom
    regret_penalty = np.exp(-np.clip(regret, 0, 3))
    
    # Symmetrize regret penalty
    regret_penalty = (regret_penalty + regret_penalty.T) / 2
    
    # 5. Adaptive weighting based on graph properties
    non_inf_dists = dist[dist < np.inf]
    if len(non_inf_dists) > 0:
        dist_mean = np.mean(non_inf_dists)
        dist_std = np.std(non_inf_dists)
        cv = dist_std / (dist_mean + 1e-10)
        
        # Adjust weights based on coefficient of variation
        if cv < 0.3:  # Uniform distances
            mst_weight = 0.6
            spatial_weight = 0.3
            regret_weight = 0.1
        elif cv < 0.7:  # Moderate variability
            mst_weight = 0.4
            spatial_weight = 0.4
            regret_weight = 0.2
        else:  # High variability
            mst_weight = 0.2
            spatial_weight = 0.3
            regret_weight = 0.5
    else:
        mst_weight, spatial_weight, regret_weight = 0.4, 0.3, 0.3
    
    # 6. Combine factors with adaptive weights
    base_scores = inv_dist * (
        mst_weight * mst_bonus +
        spatial_weight * spatial_bonus +
        regret_weight * regret_penalty
    ) / (mst_weight + spatial_weight + regret_weight)
    
    # 7. Connectivity-preserving sparsification
    # Determine target edges per node adaptively
    target_edges = max(3, min(8, int(np.sqrt(n)) + 2))
    
    # Create initial mask with MST edges
    mask = mst_array > 0
    mask = mask | mask.T  # Ensure symmetry
    
    # Add promising edges based on scores
    for i in range(n):
        row_scores = base_scores[i].copy()
        row_scores[i] = -np.inf
        
        # Sort by score
        sorted_indices = np.argsort(row_scores)[::-1]
        
        # Count how many edges we already have from MST
        current_edges = np.sum(mask[i])
        needed_edges = max(0, target_edges - current_edges)
        
        # Add top-scoring edges if not already in mask
        added = 0
        for j in sorted_indices:
            if added >= needed_edges:
                break
            if not mask[i, j] and i != j:
                mask[i, j] = True
                mask[j, i] = True
                added += 1
    
    # Ensure graph connectivity
    conn_mask = mask.astype(int)
    n_components, labels = connected_components(conn_mask, directed=False)
    
    while n_components > 1:
        # Connect components by adding best edge between them
        comp_scores = np.zeros((n_components, n_components))
        comp_scores.fill(-np.inf)
        
        # Find best edge between each pair of components
        for comp_i in range(n_components):
            for comp_j in range(comp_i + 1, n_components):
                nodes_i = np.where(labels == comp_i)[0]
                nodes_j = np.where(labels == comp_j)[0]
                
                # Find best edge between these components
                best_score = -np.inf
                best_pair = (0, 0)
                for i in nodes_i:
                    for j in nodes_j:
                        if base_scores[i, j] > best_score:
                            best_score = base_scores[i, j]
                            best_pair = (i, j)
                
                comp_scores[comp_i, comp_j] = best_score
                comp_scores[comp_j, comp_i] = best_score
        
        # Find the best connection overall
        best_comp_i, best_comp_j = np.unravel_index(np.argmax(comp_scores), comp_scores.shape)
        if comp_scores[best_comp_i, best_comp_j] > -np.inf:
            # Find the actual edge to add
            nodes_i = np.where(labels == best_comp_i)[0]
            nodes_j = np.where(labels == best_comp_j)[0]
            best_score = -np.inf
            best_edge = (0, 0)
            for i in nodes_i:
                for j in nodes_j:
                    if base_scores[i, j] > best_score:
                        best_score = base_scores[i, j]
                        best_edge = (i, j)
            
            mask[best_edge] = True
            mask[best_edge[1], best_edge[0]] = True
            
            # Recompute connectivity
            conn_mask = mask.astype(int)
            n_components, labels = connected_components(conn_mask, directed=False)
        else:
            break
    
    # Apply final mask
    final_scores = base_scores * mask
    
    # 8. Final normalization with outlier protection
    np.fill_diagonal(final_scores, 0)
    
    valid_scores = final_scores[final_scores > 0]
    if len(valid_scores) > 0:
        q90 = np.percentile(valid_scores, 90)
        if q90 > 0:
            final_scores = np.clip(final_scores / q90, 0, 1)
    
    return final_scores
