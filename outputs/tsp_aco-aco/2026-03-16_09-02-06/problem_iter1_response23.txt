import numpy as np
from scipy.spatial import distance_matrix
from collections import defaultdict

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization.
    
    Combines multiple factors to assign priority scores to edges:
    1. Inverse distance with neighbor scaling
    2. Edge sparsification based on distance quantiles
    3. Degree-constrained sampling to avoid over-connecting nodes
    4. Local clustering coefficient to identify promising regions
    
    Args:
        distance_matrix: Square matrix of pairwise distances between nodes
        
    Returns:
        Priority matrix of same shape as input, where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    
    # 1. Inverse distance with neighbor scaling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-9)  # Avoid division by zero
        
    # Scale by nearest neighbor factor
    nearest_neighbor_dist = np.partition(distance_matrix, 1, axis=1)[:, 1:3].mean(axis=1)
    scaling_factors = nearest_neighbor_dist[:, None] * nearest_neighbor_dist[None, :]
    inv_dist_scaled = inv_dist * scaling_factors
    
    # 2. Edge sparsification via quantile thresholding
    # Keep only edges in the top 30% of promising connections
    threshold = np.quantile(inv_dist_scaled[~np.eye(n, dtype=bool)], 0.7)
    mask = inv_dist_scaled >= threshold
    
    # 3. Degree-constrained sampling
    # Penalize edges that would connect high-degree nodes
    degree_weights = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            if mask[i, j]:
                # Estimate potential degrees if this edge is included
                i_neighbors = np.sum(mask[i, :]) - 1  # Exclude self
                j_neighbors = np.sum(mask[j, :]) - 1
                max_degree_penalty = max(i_neighbors, j_neighbors) / (n - 1)
                degree_weights[i, j] = degree_weights[j, i] = 1 - max_degree_penalty
    
    # 4. Local clustering coefficient bonus
    # Reward edges that connect to well-clustered regions
    clustering_bonus = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            if mask[i, j]:
                # Find common neighbors
                common_neighbors = np.sum(mask[i, :] & mask[j, :])
                possible_neighbors = min(np.sum(mask[i, :]) - 1, np.sum(mask[j, :]) - 1)
                if possible_neighbors > 0:
                    clustering_bonus[i, j] = clustering_bonus[j, i] = common_neighbors / possible_neighbors
    
    # Combine all factors
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    # Base: inverse distance with neighbor scaling
    heuristic_matrix = inv_dist_scaled * mask
    
    # Apply degree constraints
    heuristic_matrix *= degree_weights
    
    # Add clustering bonus
    heuristic_matrix += clustering_bonus * 0.5  # Weighted addition
    
    # Normalize for stability
    max_val = np.max(heuristic_matrix[~np.eye(n, dtype=bool)])
    if max_val > 0:
        heuristic_matrix /= max_val
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristic_matrix, 0)
    
    return heuristic_matrix
