import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with neural-inspired embeddings and meta-heuristic optimization.
    
    Core Innovations:
    1. Multi-Scale Regret Analysis: Combines immediate neighbor regret with 
       cluster-level regret using hierarchical graph decomposition.
    2. Graph Neural Embedding Approximation: Simulates GNN-like node embeddings
       using iterative neighborhood aggregation without requiring training.
    3. Adaptive Topological Sparsification: Dynamically prunes edges while
       preserving both local and global connectivity patterns.
    4. Meta-Heuristic Parameter Adaptation: Uses problem characteristics to
       auto-tune heuristic parameters through empirical rules.
    
    Parameters
    ----------
    distance_matrix : np.ndarray
        Square symmetric matrix of pairwise distances between n nodes
        
    Returns
    -------
    np.ndarray
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # Preprocess distance matrix
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # ========== 1. META-HEURISTIC PARAMETER ADAPTATION ==========
    # Analyze problem characteristics to auto-tune parameters
    triu_indices = np.triu_indices(n, k=1)
    distances_flat = dist[triu_indices]
    
    # Calculate problem statistics for parameter tuning
    mean_dist = np.mean(distances_flat)
    std_dist = np.std(distances_flat)
    cv_dist = std_dist / (mean_dist + 1e-12)  # Coefficient of variation
    sparsity = n / (mean_dist + 1e-12)
    
    # Adaptive parameters based on problem characteristics
    alpha = 0.3 + 0.4 * np.tanh(cv_dist)  # Weight for neural embeddings
    beta = 0.8 - 0.3 * np.tanh(sparsity / n)  # Regret sensitivity
    gamma = 0.25 + 0.2 * np.log1p(n) / 10  # Cluster weight
    
    # ========== 2. GRAPH NEURAL EMBEDDING APPROXIMATION ==========
    # Simulate GNN-like node embeddings through iterative aggregation
    embeddings = np.eye(n)  # Start with one-hot encodings
    
    # Multiple aggregation rounds (simulating GNN layers)
    n_layers = min(3, int(np.log2(n)) + 1)
    for layer in range(n_layers):
        new_embeddings = np.zeros_like(embeddings)
        
        for i in range(n):
            # Get k nearest neighbors for aggregation
            k_neighbors = min(5, n - 1)
            neighbor_indices = np.argsort(dist[i])[1:k_neighbors+1]
            
            if len(neighbor_indices) > 0:
                # Weight neighbors by inverse distance
                neighbor_weights = 1.0 / (dist[i, neighbor_indices] + 1e-12)
                neighbor_weights = neighbor_weights / (neighbor_weights.sum() + 1e-12)
                
                # Aggregate neighbor embeddings (simulating message passing)
                aggregated = np.zeros_like(embeddings[i])
                for idx, weight in zip(neighbor_indices, neighbor_weights):
                    aggregated += weight * embeddings[idx]
                
                # Combine with self embedding (residual connection)
                new_embeddings[i] = 0.7 * embeddings[i] + 0.3 * aggregated
        
        embeddings = new_embeddings.copy()
    
    # Compute edge scores from embeddings (simulating edge scoring head)
    embedding_scores = np.zeros_like(dist)
    for i in range(n):
        for j in range(i + 1, n):
            # Cosine similarity between node embeddings
            sim = np.dot(embeddings[i], embeddings[j])
            sim = sim / (np.linalg.norm(embeddings[i]) * np.linalg.norm(embeddings[j]) + 1e-12)
            embedding_scores[i, j] = embedding_scores[j, i] = (sim + 1) / 2  # Scale to [0,1]
    
    # ========== 3. MULTI-SCALE REGRET ANALYSIS ==========
    # Level 1: Immediate neighbor regret
    neighbor_regret = np.zeros_like(dist)
    for i in range(n):
        # Get distances to all neighbors
        valid_dists = dist[i][dist[i] < np.inf]
        if len(valid_dists) >= 2:
            sorted_dists = np.sort(valid_dists)
            best, second_best = sorted_dists[0], sorted_dists[1]
            
            for j in range(n):
                if i != j and dist[i, j] < np.inf:
                    regret = (dist[i, j] - best) / (second_best - best + 1e-12)
                    neighbor_regret[i, j] = max(0, min(1.0, regret))
    
    # Level 2: Cluster-level regret using hierarchical clustering
    cluster_regret = np.ones_like(dist)
    if n >= 8:
        # Create hierarchical clusters using distance thresholds
        thresholds = np.percentile(distances_flat, [25, 50, 75])
        
        for threshold in thresholds:
            # Create adjacency at this threshold
            adj = (dist <= threshold).astype(float)
            np.fill_diagonal(adj, 0)
            
            # Find connected components
            visited = np.zeros(n, dtype=bool)
            clusters = []
            
            for node in range(n):
                if not visited[node]:
                    # BFS to find cluster
                    stack = [node]
                    cluster = []
                    
                    while stack:
                        current = stack.pop()
                        if not visited[current]:
                            visited[current] = True
                            cluster.append(current)
                            # Add neighbors within threshold
                            neighbors = np.where(adj[current] > 0)[0]
                            for neighbor in neighbors:
                                if not visited[neighbor]:
                                    stack.append(neighbor)
                    
                    if len(cluster) > 1:
                        clusters.append(cluster)
            
            # Reset visited for next threshold
            visited.fill(False)
            
            # Calculate inter-cluster regret
            if len(clusters) > 1:
                for c_idx, cluster in enumerate(clusters):
                    for node in cluster:
                        # Best connection to each other cluster
                        for other_idx, other_cluster in enumerate(clusters):
                            if c_idx != other_idx:
                                # Find best connection from node to other cluster
                                best_to_cluster = np.inf
                                for other_node in other_cluster:
                                    if dist[node, other_node] < best_to_cluster:
                                        best_to_cluster = dist[node, other_node]
                                
                                # Compare with actual connections
                                for other_node in other_cluster:
                                    if other_node > node:
                                        actual = dist[node, other_node]
                                        cluster_reg = (actual - best_to_cluster) / (best_to_cluster + 1e-12)
                                        # Combine with existing cluster regret
                                        cluster_regret[node, other_node] = min(
                                            cluster_regret[node, other_node],
                                            max(0, min(1.0, cluster_reg))
                                        )
                                        cluster_regret[other_node, node] = cluster_regret[node, other_node]
    
    # Combine regret levels
    combined_regret = beta * neighbor_regret + (1 - beta) * cluster_regret
    
    # ========== 4. ADAPTIVE TOPOLOGICAL SPARSIFICATION ==========
    # Base inverse distance scoring
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.where(dist < np.inf, 1 / (dist + 1e-12), 0)
    
    # Combine all components
    base_scores = inv_dist * (1.0 - combined_regret)
    
    # Add neural embedding component
    neural_component = alpha * embedding_scores
    
    # MST reinforcement
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # Initial combined scores
    combined_scores = base_scores + neural_component
    combined_scores[mst_mask] *= 1.5  # Boost MST edges
    
    # ========== 5. INTELLIGENT EDGE PRUNING ==========
    # Dynamic connectivity preservation
    min_degree = max(2, int(np.log2(n)) + 1)
    max_degree = min(15, n // 3)
    
    final_scores = np.zeros_like(combined_scores)
    
    for i in range(n):
        # Determine target degree for this node
        centrality = np.sum(1.0 / (dist[i][dist[i] < np.inf] + 1e-12))
        centrality = centrality / (np.max([np.sum(1.0 / (dist[j][dist[j] < np.inf] + 1e-12)) 
                                          for j in range(n)]) + 1e-12)
        
        target_degree = int(min_degree + centrality * (max_degree - min_degree))
        target_degree = min(max(target_degree, min_degree), n - 1)
        
        # Select top edges
        row_scores = combined_scores[i].copy()
        row_scores[i] = -np.inf
        
        if target_degree > 0:
            # Get indices of top target_degree edges
            threshold = np.partition(row_scores, -target_degree)[-target_degree]
            mask = row_scores >= threshold
            final_scores[i, mask] = combined_scores[i, mask]
    
    # Ensure symmetry
    final_scores = np.maximum(final_scores, final_scores.T)
    
    # Always include MST edges
    final_scores[mst_mask] = np.maximum(final_scores[mst_mask], 
                                       np.percentile(final_scores[final_scores > 0], 75) 
                                       if np.any(final_scores > 0) else 1.0)
    
    # ========== 6. FINAL NORMALIZATION ==========
    np.fill_diagonal(final_scores, 0)
    
    # Softmax-style normalization per row
    row_max = final_scores.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    final_scores = final_scores / row_max
    
    # Global normalization
    max_val = final_scores.max()
    if max_val > 0:
        final_scores = final_scores / max_val
    
    return final_scores
