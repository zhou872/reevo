import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection combining multiple factors:
    1. Mutual nearest-neighbor relationships
    2. Clarke-Wright savings heuristic
    3. Distance-based sparsification
    4. Normalized reciprocal weighting
    """
    n = distance_matrix.shape[0]
    
    # Base reciprocal of distances (avoid division by zero)
    eps = 1e-10
    inv_dist = 1 / (distance_matrix + eps * np.eye(n))
    
    # 1. Mutual nearest neighbor bonus
    k = max(1, n // 10)  # Consider top 10% as nearest neighbors
    nn_bonus = np.zeros((n, n))
    for i in range(n):
        # Get k+1 smallest distances (excluding self)
        nearest = np.argsort(distance_matrix[i])[1:k+2]
        for j in nearest:
            if i != j:
                # Check if mutual (j also considers i among its k nearest)
                j_nearest = np.argsort(distance_matrix[j])[1:k+2]
                if i in j_nearest:
                    nn_bonus[i, j] = 2.0  # Mutual bonus
                else:
                    nn_bonus[i, j] = 1.0  # One-way bonus
    
    # 2. Clarke-Wright savings heuristic component
    savings = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # For each node k (excluding i and j), compute potential savings
            for k in range(n):
                if k not in (i, j):
                    savings[i, j] += distance_matrix[i, k] + distance_matrix[k, j] - distance_matrix[i, j]
            savings[j, i] = savings[i, j]
    
    # Normalize savings to [0, 1] range
    if savings.max() > savings.min():
        savings = (savings - savings.min()) / (savings.max() - savings.min())
    
    # 3. Distance-based sparsification threshold
    # Consider only edges below 75th percentile distance (shorter edges are more promising)
    flat_dist = distance_matrix[np.triu_indices(n, k=1)]
    threshold = np.percentile(flat_dist, 75)
    mask = distance_matrix <= threshold
    mask[np.diag_indices(n)] = False  # Remove self-edges
    
    # 4. Combine factors with weighted sum
    # Weights can be adjusted based on problem characteristics
    w_inv = 0.4    # Reciprocal distance weight
    w_nn = 0.3     # Nearest neighbor weight
    w_save = 0.3   # Savings weight
    
    heuristic_matrix = (
        w_inv * inv_dist / inv_dist.max() +  # Normalized reciprocal
        w_nn * nn_bonus / nn_bonus.max() +   # Normalized NN bonus
        w_save * savings                     # Already normalized
    )
    
    # Apply sparsification mask
    heuristic_matrix *= mask.astype(float)
    
    # Ensure symmetry (for undirected TSP)
    heuristic_matrix = (heuristic_matrix + heuristic_matrix.T) / 2
    
    return heuristic_matrix
