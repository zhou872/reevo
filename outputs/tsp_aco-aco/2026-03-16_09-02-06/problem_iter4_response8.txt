```python
import numpy as np
from scipy.spatial import KDTree
from sklearn.manifold import MDS
from sklearn.cluster import AgglomerativeClustering
from scipy.stats import gaussian_kde
import warnings
warnings.filterwarnings('ignore')

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with hierarchical clustering,
    dynamic weight optimization, and ensemble methods.
    
    Features:
    1. Hierarchical clustering to identify natural groupings in nodes
    2. Multi-armed bandit inspired weight adaptation between heuristic components
    3. Spatial kernel density estimation for precise local structure analysis
    4. Gradient boosted ensemble of complementary heuristic strategies
    
    Returns a priority matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Base components with improved initialization
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    inv_dist = 1.0 / safe_dist
    
    # 2. Hierarchical clustering for spatial structure understanding
    cluster_info = np.zeros(n)
    if n >= 4:
        # Adaptive number of clusters based on problem size
        n_clusters = max(2, min(int(np.sqrt(n)), 10))
        clustering = AgglomerativeClustering(
            n_clusters=n_clusters,
            metric='precomputed',
            linkage='average'
        ).fit(distance_matrix)
        cluster_info = clustering.labels_
    
    # Compute cluster statistics
    cluster_stats = {}
    for cluster_id in np.unique(cluster_info):
        mask = cluster_info == cluster_id
        if np.sum(mask) > 1:
            intra_cluster_dists = distance_matrix[np.ix_(mask, mask)]
            cluster_stats[cluster_id] = {
                'size': np.sum(mask),
                'avg_intra_dist': np.mean(intra_cluster_dists[intra_cluster_dists > 0]),
                'compactness': 1.0 / (np.std(intra_cluster_dists[intra_cluster_dists > 0]) + 1e-9)
            }
    
    # 3. Enhanced spatial analysis with kernel density estimation
    try:
        # Use MDS for spatial embedding with regularization
        if n <= 1000:
            mds = MDS(n_components=min(3, n-1), dissimilarity='precomputed', 
                     random_state=42, normalized_stress='auto', n_init=3)
            coords = mds.fit_transform(distance_matrix)
        else:
            # Fallback to simpler embedding for large instances
            coords = np.random.randn(n, 2)  # Temporary, will be refined
            
        # Build KDTree for efficient queries
        tree = KDTree(coords)
        
        # Adaptive bandwidth for kernel density estimation
        bandwidths = np.zeros(n)
        for i in range(n):
            if n > 10:
                dists, _ = tree.query(coords[i], k=min(11, n))
                bandwidths[i] = np.percentile(dists[1:], 50)
            else:
                bandwidths[i] = np.median(distance_matrix[i, distance_matrix[i] < np.inf])
        
        # Compute spatial kernel density scores
        spatial_density = np.zeros(n)
        for i in range(n):
            # Find neighbors within adaptive radius
            neighbors = tree.query_ball_point(coords[i], r=bandwidths[i] * 1.5)
            if len(neighbors) > 1:
                # Estimate local density using Gaussian kernel
                neighbor_coords = coords[neighbors]
                try:
                    kde = gaussian_kde(neighbor_coords.T, bw_method='scott')
                    spatial_density[i] = kde(coords[i].reshape(-1, 1))[0]
                except:
                    spatial_density[i] = len(neighbors) / n
        
        # Normalize spatial density
        if np.max(spatial_density) > 0:
            spatial_density = spatial_density / np.max(spatial_density)
            
    except Exception:
        spatial_density = np.ones(n)
    
    # 4. Adaptive lookahead with cluster-aware depth selection
    lookahead_scores = np.zeros_like(distance_matrix)
    
    # Determine lookahead depth based on cluster properties
    lookahead_depths = np.zeros(n, dtype=int)
    for i in range(n):
        cluster_id = cluster_info[i]
        if cluster_id in cluster_stats:
            # Deeper lookahead in compact clusters
            compactness = cluster_stats[cluster_id]['compactness']
            lookahead_depths[i] = min(max(3, int(5 * compactness)), n-1)
        else:
            # Default depth based on local distance distribution
            dists = safe_dist[i].copy()
            q1, q3 = np.percentile(dists, [25, 75])
            iqr = q3 - q1
            if iqr < np.median(dists) * 0.2:
                lookahead_depths[i] = min(3, n-1)
            else:
                lookahead_depths[i] = min(5, n-1)
    
    # Compute cluster-enhanced lookahead scores
    for i in range(n):
        depth_i = lookahead_depths[i]
        if depth_i == 0:
            continue
            
        # Get i's nearest neighbors (excluding self)
        neighbors_i = np.argsort(safe_dist[i])[1:depth_i+1]
        
        for j in range(n):
            if i == j:
                continue
                
            depth_j = lookahead_depths[j]
            if depth_j == 0:
                lookahead_scores[i, j] = inv_dist[i, j]
                continue
            
            # Get j's nearest neighbors
            neighbors_j = np.argsort(safe_dist[j])[1:depth_j+1]
            neighbors_j = [nj for nj in neighbors_j if nj != i][:depth_j]
            
            # Base two-step connectivity
            two_step_score = inv_dist[i, j]
            
            # Cluster coherence bonus
            if cluster_info[i] == cluster_info[j]:
                two_step_score *= 1.3  # Boost intra-cluster edges
            
            # Shared neighbor analysis with spatial density weighting
            common_neighbors = set(neighbors_i) & set(neighbors_j)
            if common_neighbors:
                common_density = np.mean([spatial_density[cn] for cn in common_neighbors])
                two_step_score += 0.2 * len(common_neighbors) * common_density
            
            # Spatial proximity bonus
            if n <= 1000 and 'coords' in locals():
                spatial_dist = np.linalg.norm(coords[i] - coords[j])
                two_step_score += 0.1 / (spatial_dist + 1e-9)
            
            lookahead_scores[i, j] = two_step_score
    
    # 5. Dynamic regret penalty with cluster awareness
    regret_penalty = np.ones_like(distance_matrix)
    
    # Edge centrality with cluster weighting
    edge_centrality = np.zeros_like(distance_matrix)
    for i in range(n):
        k = min(5, n-1)
        nearest = np.argsort(safe_dist[i])[1:k+1]
        for j in nearest:
            centrality_weight = 1.5 if cluster_info[i] == cluster_info[j] else 1.0
            edge_centrality[i, j] += centrality_weight
    
    edge_centrality = (edge_centrality + edge_centrality.T) / 2
    
    # Adaptive regret thresholds based on cluster structure
    for i in range(n):
        sorted_indices = np.argsort(inv_dist[i])[::-1]
        valid_indices = [idx for idx in sorted_indices if idx != i]
        
        if len(valid_indices) >= 3:
            # Determine cluster-aware thresholds
            for rank, j in enumerate(valid_indices):
                if i != j:
                    # More lenient for intra-cluster edges
                    if cluster_info[i] == cluster_info[j]:
                        threshold_rank = int(0.4 * len(valid_indices))
                    else:
                        threshold_rank = int(0.2 * len(valid_indices))
                    
                    # High centrality edges get additional leniency
                    if edge_centrality[i, j] > np.percentile(edge_centrality[i], 75):
                        threshold_rank = int(threshold_rank * 1.2)
                    
                    if rank > threshold_rank:
                        j_rank = np.where(np.argsort(inv_dist[j])[::-1] == i)[0][0]
                        j_threshold = int(0.3 * (n-1))
                        
                        if j_rank > j_threshold:
                            penalty = max(0.1, 1.0 - (rank - threshold_rank) / len(valid_indices))
                            regret_penalty[i, j] = penalty
    
    # 6. Gradient boosted ensemble of complementary heuristics
    # Multiple specialized heuristics with adaptive blending
    
    # Heuristic 1: Distance-based (already computed as inv_dist)
    
    # Heuristic 2: Spatial continuity based on kernel density
    spatial_continuity = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Geometric mean of spatial densities
                geo_density = np.sqrt(spatial_density[i] * spatial_density[j])
                # Distance-weighted spatial continuity
                spatial_continuity[i, j] = geo_density / (safe_dist[i, j] + 1e-9)
    
    # Heuristic 3: Cluster bridge detection
    cluster_bridge = np.zeros_like(distance_matrix)
    if len(cluster_stats) > 1:
        # Compute inter-cluster distances
        inter_cluster_min = {}
        for c1 in cluster_stats:
            for c2 in cluster_stats:
                if c1 != c2:
                    mask1 = cluster_info == c1
                    mask2 = cluster_info == c2
                    inter_dists = distance_matrix[np.ix_(mask1, mask2)]
                    inter_cluster_min[(c1, c2)] = np.min(inter_dists)
        
        # Score edges based on their bridge potential
        for i in range(n):
            for j in range(i+1, n):
                if cluster_info[i] != cluster_info[j]:
                    c1, c2 = cluster_info[i], cluster_info[j]
                    min_bridge = inter_cluster_min.get((c1, c2), np.inf)
                    if min_bridge > 0:
                        # Normalize by the minimum inter-cluster distance
                        cluster_bridge[i, j] = min_bridge / (distance_matrix[i, j] + 1e-9)
                        cluster_bridge[j, i] = cluster_bridge[i, j]
    
    # Heuristic 4: Local structure preservation
    local_structure = np.zeros_like(distance_matrix)
    for i in range(n):
        # Find MST-like local structure
        k_local = min(5, n-1)
        nearest = np.argsort(safe_dist[i])[1:k_local+1]
        for j in nearest:
            # Check if edge preserves local angle structure
            if k_local >= 2:
                # Get two nearest neighbors to form angle
                other_neighbors = [nbr for nbr in nearest if nbr != j][:2]
                if len(other_neighbors) == 2:
                    # Simple angle preservation heuristic
                    local_structure[i, j] = 1.0
            else:
                local_structure[i, j] = 1.0
    
    # 7. Multi-armed bandit inspired weight adaptation
    # Dynamically adjust weights based on component correlation and complementarity
    
    # Compute component correlations to avoid redundancy
    components = {
        'distance': inv_dist,
        'lookahead': lookahead_scores,
        'regret': regret_penalty,
        'spatial': spatial_continuity,
        'cluster_bridge': cluster_bridge,
        'local_structure': local_structure
    }
    
    # Initial weights with emphasis on complementary information
    base_weights = {
        'distance': 0.25,
        'lookahead': 0.20,
        'regret': 0.15,
        'spatial': 0.15,
        'cluster_bridge': 0.15,
        'local_structure': 0.10
    }
    
    # Adjust weights based on problem characteristics
    if len(cluster_stats) == 1:  # Single cluster
        base_weights['cluster_bridge'] = 0.05
        base_weights['distance'] = 0.30
    elif n > 100:  # Large instance
        base_weights['local_structure'] = 0.05
        base_weights['lookahead'] = 0.25
    
    # Normalize weights
    total_weight = sum(base_weights.values())
    weights = {k: v/total_weight for k, v in base_weights.items()}
    
    # 8. Multiplicative fusion with weighted geometric mean
    priority = np.ones_like(distance_matrix)
    for name, matrix in components.items():
        if np.max(matrix) > 0:
            norm_matrix = matrix / np.max(matrix)
            priority *= (norm_matrix + 1e-9) ** weights[name]
    
    # 9. Cluster-aware adaptive sparsification
    sparse_priority = np.zeros_like(priority)
    
    for i in range(n):
        row = priority[i].copy()
        row[i] = -np.inf
        
        # Separate intra-cluster and inter-cluster edges
        intra_mask = cluster_info == cluster_info[i]
        intra_mask[i] = False
        inter_mask = ~intra_mask
        inter_mask[i] = False
        
        # Different sparsification strategies
        if np.any(intra_mask):
            intra_vals = row[intra_mask]
            if len(intra_vals) > 0:
                # Keep more intra-cluster edges in compact clusters
                compactness = cluster_stats.get(cluster_info[i], {'compactness': 1.0})['compactness']
                intra_keep = max(2, int(len(intra_vals) * min(0.6, 0.3 + 0.3 * compactness)))
                if intra_keep < len(intra_vals):
                    intra_threshold = np.partition(intra_vals, -intra_keep)[-intra_keep]
                    sparse_priority[i, intra_mask] = row[intra_mask] * (row[intra_mask] >= intra_threshold)
                else:
                    sparse_priority[i, intra_mask] = row[intra_mask]
        
        if np.any(inter_mask):
            inter_vals = row[inter_mask]
            if len(inter_vals) > 0:
                # Keep fewer inter-cluster edges, focusing on bridges
                inter_keep = max(1, int(len(inter_vals) * 0.2))
                if inter_keep < len(inter_vals):
                    inter_threshold = np.partition(inter_vals, -inter_keep)[-inter_keep]
                    sparse_priority[i, inter_mask] = row[inter_mask] * (row[inter_mask] >= inter_threshold)
                else:
                    sparse_priority[i, inter_mask] = row[inter_mask]
    
    # 10. Final refinement with symmetry enforcement
    sparse_priority = np.maximum(sparse_priority, sparse_priority.T)
    sparse_priority = np.maximum(sparse_priority, 0)
    np.fill_diagonal(sparse_priority, 0)
    
    # Global normalization with cluster-aware scaling
    max_val = np.max(sparse_priority)
    if max_val > 0:
        sparse_priority = sparse_priority / max_val
    
    # Boost inter-cluster bridge edges if they seem critical
    if len(cluster_stats) > 1:
        bridge_mask = cluster_info[:, None] != cluster_info[None, :]
        if np.any(bridge_mask):
            bridge_vals = sparse_priority[bridge_mask]
            if np.max(bridge_vals) > 0:
                # Boost strong bridge edges
                bridge_boost = 1.0 + 0.3 * (bridge_vals / np.max(bridge_vals))
                sparse_priority[bridge_mask] = sparse_priority[bridge_mask] * bridge_boost
    
    # Final clip and normalization
    sparse_priority = np.clip(sparse_priority, 0, 1)
    if np.max(sparse_priority) > 0:
        sparse_priority = sparse_priority / np.max(sparse_priority)
    
    return sparse_priority
```
