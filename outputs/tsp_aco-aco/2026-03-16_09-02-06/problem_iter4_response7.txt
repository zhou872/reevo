import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with regret-based penalization
    and spatial lookahead strategies.
    
    Features:
    1. Regret-based Alternative Cost Penalization: Penalizes edges with high
       alternative connection costs using second-nearest neighbor distances.
    2. Spatial Lookahead Neighbor Scaling: Scales edge scores by the average
       neighbor distance of target nodes to anticipate future path quality.
    3. Mutual Top-K Edge Prioritization: Prioritizes edges that are in the
       top-K shortest connections for both endpoints.
    4. MST Connectivity Guarantee: Ensures all minimum spanning tree edges
       are retained for graph connectivity.
    
    Returns:
        np.ndarray: Heuristic score matrix with same shape as input.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create working distance matrix with infinity diagonal
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # 1. Base inverse distance scoring
    inv_dist = 1 / (dist + 1e-12)
    
    # 2. Regret-based alternative cost penalization
    # For each node, compute the cost of the second-best alternative edge
    second_best_dist = np.zeros(n)
    regret_penalty = np.ones_like(dist)
    
    for i in range(n):
        # Get sorted distances (excluding self)
        sorted_dists = np.sort(dist[i])
        if len(sorted_dists) >= 2:
            second_best_dist[i] = sorted_dists[1]  # Second smallest distance
        else:
            second_best_dist[i] = np.inf
    
    # Compute regret penalty: penalize edges that are much worse than the second best
    for i in range(n):
        for j in range(i + 1, n):
            if second_best_dist[i] < np.inf and second_best_dist[j] < np.inf:
                # Regret factor: how much worse this edge is compared to second best alternatives
                regret_i = max(0, dist[i, j] - second_best_dist[i])
                regret_j = max(0, dist[i, j] - second_best_dist[j])
                avg_regret = (regret_i + regret_j) / 2
                
                # Normalize regret by node's average distance
                avg_dist_i = np.mean(dist[i][dist[i] < np.inf])
                avg_dist_j = np.mean(dist[j][dist[j] < np.inf])
                norm_regret = avg_regret / ((avg_dist_i + avg_dist_j) / 2 + 1e-12)
                
                # Apply penalty: edges with high alternative cost regret get reduced scores
                penalty = np.exp(-norm_regret)
                regret_penalty[i, j] = penalty
                regret_penalty[j, i] = penalty
    
    # 3. Spatial lookahead neighbor scaling
    # Pre-compute average neighbor distance for each node
    avg_neighbor_dist = np.zeros(n)
    for i in range(n):
        valid_dists = dist[i][dist[i] < np.inf]
        avg_neighbor_dist[i] = np.mean(valid_dists) if len(valid_dists) > 0 else 1.0
    
    # Compute lookahead scaling factor
    lookahead_scale = np.ones_like(dist)
    for i in range(n):
        for j in range(i + 1, n):
            # Scale by target node's average neighbor distance
            # Shorter average neighbor distance indicates better connection opportunities
            scale = 1 / (1 + 0.5 * (avg_neighbor_dist[i] + avg_neighbor_dist[j]) / 
                        np.mean(avg_neighbor_dist))
            lookahead_scale[i, j] = scale
            lookahead_scale[j, i] = scale
    
    # 4. MST connectivity bonus
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    mst_bonus = 1.0 + 0.5 * mst_mask  # 1.5x bonus for MST edges
    
    # 5. Mutual top-K prioritization
    top_k = max(2, int(np.sqrt(n)))
    mutual_priority = np.zeros_like(dist)
    
    for i in range(n):
        # Get top-K edges for node i
        row_dists = dist[i].copy()
        row_dists[i] = np.inf
        top_k_indices = np.argsort(row_dists)[:top_k]
        
        for j in top_k_indices:
            # Check if edge (i,j) is in top-K for node j as well
            col_dists = dist[j].copy()
            col_dists[j] = np.inf
            j_top_k = np.argsort(col_dists)[:top_k]
            
            if i in j_top_k:
                mutual_priority[i, j] = 1.0
                mutual_priority[j, i] = 1.0
    
    # 6. Combine all factors
    scores = (inv_dist * 
              regret_penalty * 
              lookahead_scale * 
              mst_bonus * 
              (1.0 + 0.3 * mutual_priority))  # 30% bonus for mutually top-K edges
    
    # 7. Adaptive sparsification
    # Keep top edges per node based on combined score
    k_keep = max(2, int(np.sqrt(n)))
    final_mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        top_indices = np.argsort(row_scores)[-k_keep:]
        final_mask[i, top_indices] = True
    
    # Ensure symmetry
    final_mask = final_mask | final_mask.T
    
    # Always keep MST edges
    final_mask = final_mask | mst_mask
    
    # Apply mask
    scores = scores * final_mask
    
    # 8. Normalize to [0, 1] range
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    np.fill_diagonal(scores, 0)
    
    return scores
