import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges with mutual regret penalization
    and hybrid sparsification strategies.
    
    Features:
    1. Mutual Top-Percentile Penalization: Edges failing to be in top percentile
       for both endpoints are penalized (reduced to 10% weight).
    2. Adaptive Degree Sparsification: Keep top k = max(2, int(sqrt(n))) edges per node.
    3. MST Connectivity Guarantee: All minimum spanning tree edges are always retained.
    4. Global Elite Edge Preservation: Top 10% of edges globally are preserved regardless
       of local degree constraints.
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # 1. Base inverse distance with smoothing
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    inv_dist = 1 / (dist + 1e-10)
    
    # 2. MST bonus (MST edges get 2x, others 1x)
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    mst_bonus = mst_mask.astype(float) * 1.0 + 1.0  # 2.0 for MST, 1.0 otherwise
    
    # 3. Mutual top-percentile penalization
    penalty = np.ones_like(inv_dist)
    top_percentile = 0.3  # Keep top 30% per node
    
    # For each node, identify edges in top percentile
    top_mask = np.zeros_like(inv_dist, dtype=bool)
    for i in range(n):
        row = inv_dist[i].copy()
        row[i] = -np.inf  # Exclude self
        k = max(1, int(top_percentile * (n - 1)))
        if k > 0:
            threshold = np.partition(row, -k)[-k]
            top_mask[i] = row >= threshold
    
    # Penalize edges that are NOT in top percentile for BOTH endpoints
    mutual_poor = (~top_mask) & (~top_mask.T)
    penalty[mutual_poor] = 0.1
    
    # 4. Combined scoring
    scores = inv_dist * mst_bonus * penalty
    
    # 5. Hybrid sparsification
    # 5a. Adaptive degree sparsification (top k per node)
    k_degree = max(2, int(np.sqrt(n)))
    degree_mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        row = scores[i].copy()
        row[i] = -np.inf
        top_k_idx = np.argpartition(row, -k_degree)[-k_degree:]
        degree_mask[i, top_k_idx] = True
    
    # Symmetrize degree mask
    degree_mask = degree_mask | degree_mask.T
    
    # 5b. Global elite preservation (top 10% of all edges)
    upper_tri = np.triu_indices(n, k=1)
    if len(upper_tri[0]) > 0:
        elite_scores = scores[upper_tri]
        elite_threshold = np.percentile(elite_scores, 90)  # Top 10%
        global_mask = scores >= elite_threshold
    else:
        global_mask = np.zeros_like(scores, dtype=bool)
    
    # 5c. Final mask: degree OR global elite OR MST
    final_mask = degree_mask | global_mask | mst_mask
    
    # Apply sparsification
    scores = scores * final_mask
    
    # 6. Normalize and final cleanup
    max_score = scores.max()
    if max_score > 0:
        scores = scores / max_score
    
    np.fill_diagonal(scores, 0)
    return scores
