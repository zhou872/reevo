import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges combining multiple factors:
    1. Greedy local edge weighting (modified inverse distance)
    2. Minimum Spanning Tree connectivity reinforcement
    3. Exponential regret penalization for non-optimal edges
    4. Degree-aware edge pruning with connectivity preservation
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create a working copy to avoid modifying the input
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    
    # 1. Modified inverse distance with smoother scaling
    # Using sqrt instead of linear inverse to reduce dominance of very short edges
    inv_dist = 1 / np.sqrt(dist + 1e-8)
    
    # 2. MST connectivity reinforcement
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    # Stronger reinforcement for MST edges: multiplicative factor 2.5
    mst_reinforcement = np.ones_like(dist)
    mst_reinforcement[mst_mask] = 2.5
    
    # 3. Exponential regret penalization
    regret_penalty = np.ones_like(dist)
    for i in range(n):
        # Get distances from node i to all others
        distances_i = dist[i].copy()
        distances_i[i] = np.inf
        
        # Sort distances to find best and second-best
        sorted_vals = np.sort(distances_i)
        best = sorted_vals[0]
        second_best = sorted_vals[1]
        
        # Calculate regret for each edge (i,j)
        for j in range(n):
            if i != j:
                if second_best - best > 1e-8:  # Avoid division by zero
                    normalized_regret = (dist[i, j] - best) / (second_best - best)
                    # Exponential decay: penalize edges that are much worse than best
                    regret_penalty[i, j] = np.exp(-2.0 * normalized_regret)
                else:
                    # When best and second_best are very close, minimal penalty
                    regret_penalty[i, j] = 0.9
    
    # 4. Combined scoring
    scores = inv_dist * mst_reinforcement * regret_penalty
    
    # 5. Degree-aware edge pruning with connectivity preservation
    # Dynamic k: keep more edges for larger graphs
    k = max(3, min(n // 3, 10))  # Keep between 3 and 10 edges per node
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        # Get scores for this node's edges
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf  # Exclude self
        
        # Preserve top-k edges for this node
        if n > k:
            top_k_idx = np.argpartition(row_scores, -k)[-k:]
            mask[i, top_k_idx] = True
            # Ensure symmetry
            mask[top_k_idx, i] = True
        else:
            # For very small graphs, preserve all edges except self
            mask[i, :] = True
            mask[i, i] = False
    
    # Ensure MST edges are always included for connectivity
    mask = mask | mst_mask
    
    # Apply pruning mask
    pruned_scores = scores * mask
    
    # 6. Normalize to [0, 1] range
    max_val = pruned_scores.max()
    if max_val > 0:
        pruned_scores = pruned_scores / max_val
    
    # Set diagonal to zero
    np.fill_diagonal(pruned_scores, 0)
    
    return pruned_scores
