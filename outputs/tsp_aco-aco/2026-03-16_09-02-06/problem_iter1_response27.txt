import numpy as np
from scipy.spatial import distance_matrix as scipy_distance_matrix
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.ndimage import gaussian_filter

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization.
    
    Combines multiple factors to determine edge promise:
    1. Nearest-neighbor chain connectivity
    2. Local clustering density
    3. Relative distance to alternative paths
    4. Edge betweenness in minimum spanning tree
    
    Returns a sparse priority matrix where unpromising edges are set to zero.
    """
    n = distance_matrix.shape[0]
    
    # Basic inverse distance (handling zeros)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.true_divide(1.0, distance_matrix)
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.isnan(inv_dist)] = 0
    
    # 1. Nearest-neighbor chain factor
    # Find k nearest neighbors for each node
    k = max(2, min(5, n // 4))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Exclude self-distance
        indices = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, indices] = True
    
    # Make symmetric (if i is in j's NN, j is in i's NN)
    nn_mask = np.logical_or(nn_mask, nn_mask.T)
    
    # 2. Local clustering density factor
    # Calculate local density around each node
    local_density = np.zeros(n)
    for i in range(n):
        # Mean distance to k nearest neighbors
        nn_dists = np.sort(distance_matrix[i])[1:k+1]
        local_density[i] = 1.0 / (np.mean(nn_dists) + 1e-10)
    
    # Normalize density
    density_norm = local_density / (np.max(local_density) + 1e-10)
    
    # Density factor for edges: connects dense regions
    density_factor = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            density_factor[i, j] = density_factor[j, i] = (
                density_norm[i] * density_norm[j]
            )
    
    # 3. Relative distance penalty
    # For each edge, compare to alternative paths through one intermediate node
    relative_penalty = np.ones_like(distance_matrix)
    
    for i in range(n):
        for j in range(i+1, n):
            # Find best alternative path through third node
            alternatives = []
            for k in range(n):
                if k != i and k != j:
                    alt_dist = distance_matrix[i, k] + distance_matrix[k, j]
                    alternatives.append(alt_dist)
            
            if alternatives:
                best_alt = np.min(alternatives)
                # Penalize if direct edge is much longer than alternative
                if distance_matrix[i, j] > best_alt:
                    penalty = best_alt / (distance_matrix[i, j] + 1e-10)
                    relative_penalty[i, j] = relative_penalty[j, i] = penalty
    
    # 4. Minimum spanning tree betweenness
    # Build MST and give priority to edges that appear in it
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0).astype(float)
    
    # Also consider edges that are "almost" in MST
    # Calculate second-best edges for each node
    mst_bonus = mst_mask.copy()
    
    # For nodes not connected in MST, find their nearest neighbor in MST
    for i in range(n):
        mst_connected = np.where(mst_mask[i] > 0)[0]
        if len(mst_connected) == 0:  # Isolated in MST (shouldn't happen in connected graph)
            continue
        
        # Find closest node not already connected in MST
        for j in range(n):
            if i != j and mst_mask[i, j] == 0:
                # Check if connecting through MST gives good alternative
                min_path_via_mst = np.inf
                for k in mst_connected:
                    # Estimate path length through MST-connected nodes
                    path_length = distance_matrix[i, k] + distance_matrix[k, j]
                    min_path_via_mst = min(min_path_via_mst, path_length)
                
                # If direct edge is competitive with MST path, give some bonus
                if distance_matrix[i, j] <= min_path_via_mst * 1.2:
                    mst_bonus[i, j] = 0.5
    
    # Combine all factors with weights
    weights = {
        'inv_dist': 2.0,
        'density': 1.5,
        'nn_chain': 3.0,
        'mst': 2.5,
        'penalty': 0.8
    }
    
    # Base score
    score = weights['inv_dist'] * inv_dist
    
    # Apply nearest-neighbor chain factor (sparsification)
    score *= weights['nn_chain'] * nn_mask.astype(float)
    
    # Add density factor
    score += weights['density'] * density_factor
    
    # Apply relative distance penalty
    score *= weights['penalty'] * relative_penalty
    
    # Add MST bonus
    score += weights['mst'] * mst_bonus
    
    # Additional sparsification: keep only top edges for each node
    final_mask = np.zeros_like(score, dtype=bool)
    edges_per_node = max(2, min(6, n // 3))
    
    for i in range(n):
        # Get top edges for this node
        top_indices = np.argsort(score[i])[-edges_per_node:]
        final_mask[i, top_indices] = True
    
    # Make symmetric
    final_mask = np.logical_or(final_mask, final_mask.T)
    
    # Apply final mask and zero out diagonal
    result = score * final_mask
    np.fill_diagonal(result, 0)
    
    # Smooth the result for better gradation
    result = gaussian_filter(result, sigma=0.5, mode='reflect')
    
    # Normalize to [0, 1] range for consistency
    max_val = np.max(result)
    if max_val > 0:
        result = result / max_val
    
    return result

# Optional: Helper function for testing
def generate_sample_tsp(n_cities=20, seed=42):
    """Generate sample TSP data for testing."""
    np.random.seed(seed)
    points = np.random.rand(n_cities, 2) * 100
    dist_mat = scipy_distance_matrix(points, points)
    return dist_mat

if __name__ == "__main__":
    # Example usage
    sample_dist = generate_sample_tsp(15)
    heuristic_matrix = heuristics_v2(sample_dist)
    
    print(f"Heuristic matrix shape: {heuristic_matrix.shape}")
    print(f"Non-zero entries: {np.count_nonzero(heuristic_matrix)} / {heuristic_matrix.size}")
    print(f"Mean priority: {np.mean(heuristic_matrix[heuristic_matrix > 0]):.4f}")
