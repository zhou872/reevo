import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection.
    
    Combines multiple factors to assess edge promise:
    1. Inverse distance weighting (emphasizes short edges)
    2. Local clustering to identify community structure
    3. Two-hop neighborhood density for connectivity patterns
    4. Sparsification by setting unpromising edges to zero
    """
    n = distance_matrix.shape[0]
    heuristic = np.zeros((n, n))
    
    # 1. Base inverse distance with smoothing
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        np.fill_diagonal(inv_dist, 0)
    
    # Normalize inverse distances
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # 2. Local clustering coefficient (approximate for complete graph)
    # Higher coefficient suggests nodes belong to tight clusters
    cluster_coeff = np.zeros(n)
    for i in range(n):
        # For each node, look at its k nearest neighbors
        k = max(3, int(np.sqrt(n)))
        # Get indices of k nearest neighbors (excluding self)
        neighbors = np.argsort(distance_matrix[i])[1:k+1]
        if len(neighbors) < 2:
            continue
        
        # Estimate local density by average distance to neighbors
        avg_neighbor_dist = np.mean(distance_matrix[i, neighbors])
        # Convert to coefficient (lower distance = higher coefficient)
        cluster_coeff[i] = 1.0 / (1.0 + avg_neighbor_dist / distance_matrix.mean())
    
    # 3. Two-hop neighborhood analysis
    two_hop_score = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Find common neighbors
            # For efficiency, consider top nearest neighbors only
            k_hops = min(10, n-1)
            i_neighbors = np.argsort(distance_matrix[i])[1:k_hops+1]
            j_neighbors = np.argsort(distance_matrix[j])[1:k_hops+1]
            
            common = np.intersect1d(i_neighbors, j_neighbors)
            if len(common) > 0:
                # Edge gets bonus if it connects nodes with many common neighbors
                # (suggests it's within a cluster)
                avg_common_dist = np.mean([
                    distance_matrix[i, c] + distance_matrix[j, c] 
                    for c in common[:5]  # Limit to top 5 for stability
                ]) if len(common) > 0 else 0
                
                if avg_common_dist > 0:
                    two_hop_score[i, j] = two_hop_score[j, i] = min(
                        1.0, 2.0 * distance_matrix.mean() / avg_common_dist
                    )
    
    # 4. Combine factors
    for i in range(n):
        for j in range(i+1, n):
            if i == j:
                continue
            
            # Base inverse distance
            base_score = inv_dist[i, j]
            
            # Clustering factor: edges between nodes with high cluster coefficients
            cluster_factor = 0.5 * (cluster_coeff[i] + cluster_coeff[j])
            
            # Two-hop neighborhood factor
            hop_factor = two_hop_score[i, j]
            
            # Combined score with weights
            combined = (
                0.6 * base_score + 
                0.2 * cluster_factor + 
                0.2 * hop_factor
            )
            
            heuristic[i, j] = heuristic[j, i] = combined
    
    # 5. Sparsification: keep only promising edges
    # Determine threshold adaptively
    if n > 10:
        # For larger graphs, keep edges for each node's top connections
        edges_per_node = max(3, int(np.log2(n)) * 2)
        for i in range(n):
            # Get indices of this node's edges
            row = heuristic[i].copy()
            row[i] = -np.inf  # Exclude self
            
            # Find threshold for top edges_per_node connections
            if edges_per_node < len(row) - 1:
                threshold = np.partition(row, -edges_per_node)[-edges_per_node]
                # Set below-threshold edges to zero
                heuristic[i][heuristic[i] < threshold] = 0
                # Ensure symmetry
                heuristic[:, i][heuristic[i] < threshold] = 0
    else:
        # For small graphs, keep edges above median
        non_diag = heuristic[~np.eye(n, dtype=bool)]
        if len(non_diag) > 0:
            threshold = np.median(non_diag)
            mask = heuristic < threshold
            heuristic[mask] = 0
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
