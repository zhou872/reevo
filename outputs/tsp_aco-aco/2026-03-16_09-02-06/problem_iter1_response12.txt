import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors.
    Returns a sparse matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Avoid division by zero for diagonal elements
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base closeness score (inverse distance)
    closeness = 1.0 / safe_dist
    
    # 2. Local connectivity factor: how close each edge is to the node's minimum distance
    min_dists = np.min(safe_dist, axis=1, keepdims=True)
    closeness_ratio = min_dists / safe_dist
    
    # 3. Global sparsity factor: penalize edges much longer than average
    avg_dist = np.mean(safe_dist[safe_dist < np.inf])
    avg_penalty = np.exp(-safe_dist / (2 * avg_dist))
    
    # 4. Two-step neighbor consideration: edges connecting to well-connected nodes
    neighbor_scores = np.zeros_like(distance_matrix)
    for i in range(n):
        # Consider second-order neighbors (neighbors of neighbors)
        for j in range(n):
            if i != j:
                # Find best intermediate node k that makes i-j-k promising
                k_scores = closeness[i, :] * closeness[:, j]
                k_scores[i] = k_scores[j] = 0
                neighbor_scores[i, j] = np.max(k_scores) if len(k_scores) > 0 else 0
    
    # 5. Combine factors with adaptive weights
    # Weight factors based on problem size (more neighbors for larger problems)
    neighbor_weight = min(0.3, 2.0 / n)
    base_score = closeness * closeness_ratio * avg_penalty
    combined = base_score + neighbor_weight * neighbor_scores
    
    # 6. Adaptive sparsification: keep only top promising edges
    # Keep edges within top percentiles or exceeding threshold
    for i in range(n):
        row = combined[i, :]
        
        # Determine adaptive threshold for this row
        non_inf_mask = safe_dist[i, :] < np.inf
        if not np.any(non_inf_mask):
            continue
            
        valid_scores = row[non_inf_mask]
        if len(valid_scores) == 0:
            continue
            
        # Keep top 40% of edges or those above mean + 0.5*std
        threshold = max(
            np.percentile(valid_scores, 60),  # Top 40%
            np.mean(valid_scores) + 0.5 * np.std(valid_scores)
        )
        
        # Apply threshold
        mask = row < threshold
        combined[i, mask] = 0
        
        # Ensure at least 2 connections per node for feasibility
        non_zero = np.sum(combined[i, :] > 0)
        if non_zero < 2 and np.any(non_inf_mask):
            # Reinstate the two best edges
            top_indices = np.argsort(row)[-2:]
            for idx in top_indices:
                if idx != i:
                    combined[i, idx] = row[idx]
    
    # 7. Symmetrize the matrix (undirected TSP)
    combined = np.maximum(combined, combined.T)
    
    # 8. Final zero-out: remove extremely weak edges
    overall_mean = np.mean(combined[combined > 0]) if np.any(combined > 0) else 0
    if overall_mean > 0:
        combined[combined < 0.1 * overall_mean] = 0
    
    # Set diagonal to zero
    np.fill_diagonal(combined, 0)
    
    return combined
