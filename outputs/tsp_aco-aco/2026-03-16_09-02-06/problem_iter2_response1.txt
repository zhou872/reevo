import numpy as np
from scipy.cluster.hierarchy import linkage, fcluster
from scipy.spatial.distance import squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization.
    Features dynamic weight adaptation, regret-based edge penalties,
    hierarchical clustering using actual distances, and adaptive sparsification.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    epsilon = 1e-10
    
    # 1. Dynamic weight initialization based on problem size
    # Larger problems benefit more from clustering, smaller ones from local heuristics
    size_factor = min(1.0, n / 50)  # Normalize by reference size of 50
    w_inv = 0.35 - 0.15 * size_factor      # Decrease weight for inverse distance
    w_nn = 0.25 + 0.10 * size_factor       # Increase weight for nearest neighbor
    w_cluster = 0.20 + 0.15 * size_factor  # Increase weight for clustering
    w_regret = 0.20                       # Constant weight for regret
    
    # 2. Base inverse distance factor (normalized)
    inv_dist = 1 / np.clip(distance_matrix, epsilon, None)
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() > epsilon:
        inv_dist = inv_dist / inv_dist.max()
    
    # 3. Enhanced mutual nearest-neighbor factor
    k = min(max(3, int(0.1 * n)), 8)  # Adaptive k: 10% of n, min 3, max 8
    nn_bonus = np.zeros((n, n))
    
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])
        valid_nn = sorted_indices[1:k+1]  # Exclude self
        nn_bonus[i, valid_nn] = 1
    
    # Mutual nearest neighbor with strength based on reciprocal rank
    mutual_strength = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):
            if nn_bonus[i, j] > 0 and nn_bonus[j, i] > 0:
                # Strength based on position in each other's sorted lists
                rank_i = np.where(np.argsort(distance_matrix[i]) == j)[0][0]
                rank_j = np.where(np.argsort(distance_matrix[j]) == i)[0][0]
                strength = (1.0 / (rank_i + 1) + 1.0 / (rank_j + 1)) / 2
                mutual_strength[i, j] = mutual_strength[j, i] = strength
    
    # 4. Adaptive hierarchical clustering using actual distances
    cluster_bonus = np.zeros((n, n))
    
    if n >= 6:
        # Convert distance matrix to condensed form for hierarchical clustering
        condensed_dist = squareform(distance_matrix)
        
        # Perform hierarchical clustering with average linkage
        Z = linkage(condensed_dist, method='average')
        
        # Determine optimal number of clusters (between 2 and min(8, n//3))
        n_clusters = min(max(2, int(np.sqrt(n))), 8, n // 3)
        clusters = fcluster(Z, n_clusters, criterion='maxclust')
        
        # Assign cluster bonus based on cluster membership
        for i in range(n):
            for j in range(i + 1, n):
                if clusters[i] == clusters[j]:
                    # Bonus inversely proportional to cluster size
                    cluster_size = np.sum(clusters == clusters[i])
                    bonus = 1.5 / (1 + np.log(cluster_size))
                    cluster_bonus[i, j] = cluster_bonus[j, i] = bonus
    
    # 5. Regret-based edge penalty (lookahead to second-best options)
    regret_penalty = np.zeros((n, n))
    
    for i in range(n):
        # Find best and second-best edges from node i
        sorted_dist = np.sort(distance_matrix[i])
        if n > 2:
            best_dist = sorted_dist[1]  # Skip self
            second_best_dist = sorted_dist[2]
        else:
            best_dist = sorted_dist[1]
            second_best_dist = best_dist
        
        # Calculate regret for not taking the best edge
        base_regret = second_best_dist - best_dist
        
        for j in range(n):
            if j != i:
                # Regret increases if this edge is much worse than alternatives
                if distance_matrix[i, j] > best_dist + epsilon:
                    regret = base_regret / (distance_matrix[i, j] - best_dist + epsilon)
                    regret_penalty[i, j] = max(0, 1 - regret)  # Convert to penalty score
    
    # Symmetrize regret penalty
    regret_penalty = (regret_penalty + regret_penalty.T) / 2
    
    # 6. Combine factors with dynamic weights
    edge_promise = (
        w_inv * inv_dist +
        w_nn * mutual_strength +
        w_cluster * cluster_bonus +
        w_regret * regret_penalty
    )
    
    # 7. Adaptive sparsification
    if n >= 4:
        # Keep top edges based on node-specific thresholds
        sparsified = np.zeros_like(edge_promise)
        
        for i in range(n):
            # Calculate adaptive threshold per node
            node_edges = edge_promise[i]
            node_edges_sorted = np.sort(node_edges)[::-1]
            
            # Keep edges above median or top 40%, whichever is more restrictive
            median_val = np.median(node_edges_sorted[node_edges_sorted > 0])
            top_k = max(2, int(0.4 * n))
            threshold = max(median_val, node_edges_sorted[top_k - 1])
            
            # Apply threshold
            mask = node_edges >= threshold
            mask[i] = False  # Ensure diagonal is zero
            sparsified[i] = node_edges * mask
        
        # Ensure symmetry
        sparsified = np.maximum(sparsified, sparsified.T)
    else:
        sparsified = edge_promise
    
    # 8. Final normalization and diagonal zeroing
    np.fill_diagonal(sparsified, 0)
    
    if sparsified.max() > epsilon:
        sparsified = sparsified / sparsified.max()
    
    return sparsified
