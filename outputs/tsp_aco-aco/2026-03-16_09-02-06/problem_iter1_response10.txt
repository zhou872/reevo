import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics function for TSP edge scoring with sparsification.
    
    Combines multiple factors:
    1. Distance inverse (base preference for short edges)
    2. Local sparsity measure (edges connecting to nodes with few good options)
    3. Edge competition penalty (edges competing for the same nodes)
    4. Minimum spanning tree bias (edges in MST get bonus)
    
    Returns a sparse matrix where unpromising edges are set to zero.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base distance inverse (avoid division by zero)
    with np.errstate(divide='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-12)
    inv_dist[np.eye(n, dtype=bool)] = 0
    
    # 2. Local sparsity scoring
    # For each node, find how sparse its neighborhood is
    sparsity_scores = np.zeros((n, n))
    k = max(2, n // 10)  # Consider top k nearest neighbors
    
    for i in range(n):
        # Get distances from node i to all others
        dists = distance_matrix[i].copy()
        dists[i] = np.inf  # Ignore self
        
        # Find indices of k nearest neighbors
        nn_indices = np.argpartition(dists, k)[:k]
        
        # Calculate sparsity: node gets higher score if it has few good options
        avg_nn_dist = np.mean(dists[nn_indices])
        sparsity = 1.0 / (avg_nn_dist + 1e-12)
        
        # Distribute sparsity to nearest neighbors
        for j in nn_indices:
            sparsity_scores[i, j] = sparsity * (1.0 / (dists[j] + 1e-12))
    
    # 3. Edge competition penalty
    # Penalize edges that face strong competition from other edges
    competition_penalty = np.zeros((n, n))
    
    for i in range(n):
        # For each potential edge (i,j), check competition from other nodes
        for j in range(n):
            if i == j:
                continue
                
            # Find edges competing for node j
            competitors = []
            for k in range(n):
                if k != i and k != j:
                    competitors.append(distance_matrix[k, j])
            
            if competitors:
                # Stronger penalty if many good alternatives exist for node j
                min_comp_dist = np.min(competitors)
                penalty = distance_matrix[i, j] / (min_comp_dist + 1e-12)
                competition_penalty[i, j] = 1.0 / (1.0 + penalty)
    
    # 4. Minimum spanning tree bias
    mst_bias = np.zeros((n, n))
    try:
        # Compute minimum spanning tree
        mst = minimum_spanning_tree(distance_matrix)
        mst_mask = mst.toarray() != 0
        
        # Edges in MST get bonus
        mst_bias[mst_mask] = 2.0  # Boost factor for MST edges
    except:
        # Fallback if MST computation fails
        mst_bias = np.ones((n, n))
    
    # Combine all factors
    combined_scores = (
        inv_dist * 1.0 +          # Base distance preference
        sparsity_scores * 0.5 +   # Local sparsity consideration
        competition_penalty * 0.3 +  # Edge competition adjustment
        mst_bias * 0.7            # MST bias
    )
    
    # 5. Sparsification: keep only promising edges
    # Strategy: Keep top edges for each node + globally top edges
    
    # Keep top m edges for each node
    m = max(3, n // 8)
    node_top_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        row_scores = combined_scores[i].copy()
        row_scores[i] = -np.inf  # Ignore diagonal
        
        # Get indices of top m edges from this node
        top_indices = np.argpartition(-row_scores, m)[:m]
        node_top_mask[i, top_indices] = True
    
    # Keep globally top p% of edges
    p = 0.2  # Keep top 20% globally
    flat_scores = combined_scores.flatten()
    threshold = np.percentile(flat_scores, 100 * (1 - p))
    global_top_mask = combined_scores >= threshold
    
    # Final mask: edge must be in top for at least one node OR globally top
    final_mask = node_top_mask | node_top_mask.T | global_top_mask
    
    # Apply sparsification
    sparsified_scores = np.zeros((n, n))
    sparsified_scores[final_mask] = combined_scores[final_mask]
    
    # Ensure symmetry for undirected TSP
    sparsified_scores = (sparsified_scores + sparsified_scores.T) / 2
    
    return sparsified_scores
