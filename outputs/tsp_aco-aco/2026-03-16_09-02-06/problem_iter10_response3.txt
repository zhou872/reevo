import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Improved TSP edge heuristic with balanced computational efficiency and effectiveness.
    
    Core Strategies:
    1. MST Reinforcement: Strongly boost edges belonging to the minimum spanning tree
    2. Cluster Density Scoring: Fast geometric clustering with density-aware edge weighting
    3. Connectivity-Preserving Sparsification: Dynamic edge pruning while maintaining graph connectivity
    4. Distance Reciprocal Weighting: Inverse distance scoring with adaptive regularization
    
    Returns:
        np.ndarray: Heuristic score matrix with same shape as input
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # === 1. PREPROCESSING ===
    dist = distance_matrix.copy().astype(float)
    np.fill_diagonal(dist, np.inf)
    
    # Adaptive epsilon for numerical stability
    flat_dists = dist[dist < np.inf]
    eps = np.percentile(flat_dists, 5) * 0.01 if len(flat_dists) > 0 else 1e-6
    
    # === 2. BASE INVERSE DISTANCE SCORING ===
    inv_dist = 1 / (dist + eps)
    np.fill_diagonal(inv_dist, 0)
    
    # === 3. MINIMUM SPANNING TREE REINFORCEMENT ===
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    
    # Compute MST edge importance relative to average distances
    mst_weights = mst.toarray()
    avg_dist = np.mean(flat_dists)
    mst_bonus = np.where(mst_mask, 1.0 + 0.8 * avg_dist / (mst_weights + eps), 1.0)
    
    # === 4. FAST GEOMETRIC CLUSTERING ===
    cluster_bonus = np.ones_like(dist)
    
    if n >= 8:
        # Simple distance-based clustering using nearest neighbors
        k_clusters = max(2, min(6, n // 5))
        cluster_size = n // k_clusters
        
        # Initial cluster centers: random selection
        np.random.seed(42)  # Deterministic for reproducibility
        centers = np.random.choice(n, size=k_clusters, replace=False)
        
        # Assign nodes to nearest center
        assignments = np.argmin(dist[centers], axis=0)
        
        # Boost intra-cluster edges
        for c in range(k_clusters):
            cluster_nodes = np.where(assignments == c)[0]
            if len(cluster_nodes) > 1:
                # Compute average intra-cluster distance
                if len(cluster_nodes) > 1:
                    cluster_dist = dist[cluster_nodes][:, cluster_nodes]
                    cluster_avg = np.mean(cluster_dist[cluster_dist < np.inf])
                    
                    # Stronger boost for compact clusters
                    boost = 1.0 + 0.5 * avg_dist / (cluster_avg + eps)
                    for i in cluster_nodes:
                        for j in cluster_nodes:
                            if i != j:
                                cluster_bonus[i, j] = boost
    
    # === 5. NODE DEGREE BALANCING ===
    # Simple centrality measure based on average distance
    node_centrality = np.zeros(n)
    for i in range(n):
        valid_dists = dist[i][dist[i] < np.inf]
        node_centrality[i] = 1.0 / (np.mean(valid_dists) + eps)
    
    # Normalize centrality
    if node_centrality.max() > 0:
        node_centrality = node_centrality / node_centrality.max()
    
    # Adjust scores to favor connections between central and peripheral nodes
    centrality_adjust = np.ones_like(dist)
    for i in range(n):
        for j in range(i + 1, n):
            diff = abs(node_centrality[i] - node_centrality[j])
            # Moderate boost for connecting different centrality regions
            centrality_adjust[i, j] = centrality_adjust[j, i] = 1.0 + 0.3 * diff
    
    # === 6. COMBINED SCORING ===
    scores = inv_dist * mst_bonus * cluster_bonus * centrality_adjust
    
    # === 7. CONNECTIVITY-PRESERVING SPARSIFICATION ===
    # Dynamic edge retention based on graph size and density
    base_edges = int(np.sqrt(n) * 2)
    k_keep = max(3, min(base_edges, n // 2))
    
    sparse_mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        # Keep top edges for each node
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf
        
        top_indices = np.argsort(row_scores)[-k_keep:]
        sparse_mask[i, top_indices] = True
        
        # Ensure connectivity to nearest neighbors
        nearest = np.argsort(dist[i])[1:min(3, n)]
        sparse_mask[i, nearest] = True
    
    # Ensure symmetry and MST inclusion
    sparse_mask = sparse_mask | sparse_mask.T | mst_mask
    
    # Additional connectivity check
    for i in range(n):
        if sparse_mask[i].sum() < 2:
            # Add best missing connections
            row_scores = scores[i].copy()
            row_scores[sparse_mask[i]] = -np.inf
            row_scores[i] = -np.inf
            
            if np.any(row_scores > -np.inf):
                top_idx = np.argsort(row_scores)[-1]
                sparse_mask[i, top_idx] = True
                sparse_mask[top_idx, i] = True
    
    # Apply sparsification
    scores = scores * sparse_mask
    
    # === 8. FINAL NORMALIZATION ===
    score_max = scores.max()
    if score_max > 0:
        scores = scores / score_max
    
    np.fill_diagonal(scores, 0)
    return scores
