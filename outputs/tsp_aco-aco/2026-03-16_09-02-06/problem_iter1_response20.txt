import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge prioritization combining multiple factors:
    1. Distance-bounded inverse weighting with adaptive sparsification
    2. Two-level node centrality (local and global)
    3. Minimum spanning tree edge preference
    4. Symmetric regret-based penalty
    """
    n = distance_matrix.shape[0]
    
    # 1. Basic inverse distance with robust handling
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.divide(1.0, distance_matrix, 
                           out=np.zeros_like(distance_matrix, dtype=float),
                           where=distance_matrix > 0)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Calculate node centralities
    # Local centrality: average distance to nearest neighbors
    k_local = min(3, n-1)
    local_centrality = np.zeros(n)
    for i in range(n):
        sorted_dists = np.sort(distance_matrix[i])
        local_centrality[i] = np.mean(sorted_dists[1:k_local+1])  # Skip self
    
    # Global centrality: closeness centrality approximation
    row_sums = distance_matrix.sum(axis=1)
    global_centrality = 1 / (row_sums + 1e-10)
    
    # Combine centralities for node priority
    node_priority = (global_centrality - global_centrality.min()) / \
                    (global_centrality.max() - global_centrality.min() + 1e-10)
    node_priority += (1 - (local_centrality - local_centrality.min()) / \
                     (local_centrality.max() - local_centrality.min() + 1e-10))
    node_priority /= 2.0
    
    # 3. Generate Minimum Spanning Tree (Prim's algorithm)
    mst_mask = np.zeros_like(distance_matrix, dtype=bool)
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    
    for _ in range(n-1):
        min_edge = np.inf
        u, v = -1, -1
        
        # Find minimum edge between visited and unvisited nodes
        visited_indices = np.where(visited)[0]
        unvisited_indices = np.where(~visited)[0]
        
        for i in visited_indices:
            for j in unvisited_indices:
                if distance_matrix[i, j] < min_edge:
                    min_edge = distance_matrix[i, j]
                    u, v = i, j
        
        if u != -1 and v != -1:
            mst_mask[u, v] = True
            mst_mask[v, u] = True
            visited[v] = True
    
    # 4. Calculate regret penalties
    # For each node, find difference between 2nd best and best edge
    regret_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf  # Ignore self
        sorted_indices = np.argsort(row)
        
        if len(sorted_indices) >= 2:
            best_idx = sorted_indices[0]
            second_best_idx = sorted_indices[1]
            regret = row[second_best_idx] - row[best_idx]
            
            # Apply penalty to best edge (higher regret = more penalty)
            if regret > 0:
                regret_matrix[i, best_idx] = -np.log1p(regret)
                regret_matrix[best_idx, i] = -np.log1p(regret)
    
    # 5. Combine all factors
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
    
    for i in range(n):
        for j in range(i+1, n):  # Upper triangle only, symmetric
            if distance_matrix[i, j] == 0:
                continue
                
            # Base score from inverse distance
            base_score = inv_dist[i, j]
            
            # Node centrality factor (geometric mean)
            centrality_factor = np.sqrt(node_priority[i] * node_priority[j])
            
            # MST bonus
            mst_bonus = 2.0 if mst_mask[i, j] else 1.0
            
            # Regret adjustment
            regret_adj = 1.0 + 0.5 * np.tanh(-regret_matrix[i, j])
            
            # Distance-bounded factor (penalize very long edges)
            avg_dist = distance_matrix.mean()
            dist_factor = np.exp(-distance_matrix[i, j] / (2 * avg_dist))
            
            # Combine factors
            score = base_score * centrality_factor * mst_bonus * regret_adj * dist_factor
            
            # Apply symmetric assignment
            heuristics_matrix[i, j] = score
            heuristics_matrix[j, i] = score
    
    # 6. Adaptive sparsification
    # Keep only top edges for each node based on connectivity needs
    sparsity_factor = min(0.3, 5.0/n)  # Adjust based on problem size
    
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Ignore diagonal
        
        # Determine threshold
        sorted_row = np.sort(row[row > 0])
        if len(sorted_row) > 0:
            keep_count = max(2, int(len(sorted_row) * sparsity_factor))
            threshold = sorted_row[-keep_count] if keep_count > 0 else 0
            
            # Zero out unpromising edges
            mask = heuristics_matrix[i] < threshold
            heuristics_matrix[i, mask] = 0
            heuristics_matrix[mask, i] = 0  # Maintain symmetry
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to reasonable range
    max_val = heuristics_matrix.max()
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
