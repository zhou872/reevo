import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges combining multiple factors:
    1. Inverse distance (base attractiveness)
    2. Minimum Spanning Tree inclusion bonus
    3. Regret-based penalty (difference between best and second-best edges)
    4. Degree-aware sparsification (preserving connectivity)
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between nodes
        
    Returns:
    --------
    np.ndarray
        Heuristic scores of same shape as input, with unpromising edges set to zero
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # Create a working copy to avoid modifying the original
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)  # Exclude self-connections
    
    # 1. Base inverse distance with smoothing
    inv_dist = 1 / (dist + 1e-10)
    
    # 2. MST edge bonus - compute MST on original distances (excluding diagonal)
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst.toarray() > 0)
    mst_bonus = np.ones_like(dist)
    mst_bonus[mst_mask] = 2.0  # Bonus for MST edges
    
    # 3. Regret calculation for each node
    regret_matrix = np.ones_like(dist)
    for i in range(n):
        # Get distances from node i to all others
        distances_i = dist[i].copy()
        
        # Find best and second-best distances
        sorted_idx = np.argsort(distances_i)
        best_idx, second_best_idx = sorted_idx[0], sorted_idx[1]
        best_dist = distances_i[best_idx]
        second_best_dist = distances_i[second_best_idx]
        
        # Calculate regret for each edge from node i
        if second_best_dist - best_dist > 1e-10:  # Avoid division by zero
            for j in range(n):
                if i != j and distances_i[j] > best_dist:
                    regret = (distances_i[j] - best_dist) / (second_best_dist - best_dist)
                    regret_matrix[i, j] = np.exp(-regret)  # Convert to penalty (0-1)
        else:
            # All edges are equally good/bad
            for j in range(n):
                if i != j:
                    regret_matrix[i, j] = 1.0
    
    # 4. Combined scoring
    scores = inv_dist * mst_bonus * regret_matrix
    
    # 5. Degree-aware sparsification
    # Calculate expected degree for each node based on current scores
    degree_weights = np.sum(scores, axis=1) + np.sum(scores, axis=0)
    degree_penalty = 1 - degree_weights / (2 * np.max(degree_weights) + 1e-10)
    
    # Apply degree penalty symmetrically
    penalty_matrix = np.minimum.outer(degree_penalty, degree_penalty)
    np.fill_diagonal(penalty_matrix, 0)  # Zero out diagonal
    scores = scores * penalty_matrix
    
    # Preserve top-k edges for each node while maintaining connectivity
    k = max(2, n // 4)  # Keep at least 2 edges per node, more for small graphs
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        # Get top k edges for this node
        row_scores = scores[i].copy()
        row_scores[i] = -np.inf  # Exclude self
        top_k_idx = np.argpartition(row_scores, -k)[-k:]
        mask[i, top_k_idx] = True
    
    # Make the mask symmetric
    mask = mask | mask.T
    
    # Ensure MST edges are always included for connectivity
    mask = mask | mst_mask
    
    # Apply sparsification
    scores = scores * mask
    
    # 6. Final normalization
    max_score = np.max(scores)
    if max_score > 0:
        scores = scores / max_score
    
    np.fill_diagonal(scores, 0)  # Zero out diagonal
    return scores
