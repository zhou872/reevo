import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection combining multiple factors.
    Returns a matrix of edge selection priorities with unpromising edges set to zero.
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-10
    
    # 1. Base inverse distance (handling zeros)
    with np.errstate(divide='ignore'):
        inv_dist = np.where(distance_matrix > epsilon, 1 / distance_matrix, 0)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Local connectivity strength (averaged 2-step connectivity)
    connectivity = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Find best intermediate node for 2-hop path i->k->j
            min_two_step = np.inf
            for k in range(n):
                if k != i and k != j:
                    two_step = distance_matrix[i, k] + distance_matrix[k, j]
                    if two_step < min_two_step:
                        min_two_step = two_step
            
            # If 2-hop path is significantly better than direct connection
            if min_two_step < 0.9 * distance_matrix[i, j]:
                connectivity[i, j] = connectivity[j, i] = 1.0
            elif min_two_step < 1.1 * distance_matrix[i, j]:
                connectivity[i, j] = connectivity[j, i] = 0.5
    
    # 3. Regional sparsity analysis (clustering-based importance)
    # Calculate average distances to identify cluster centers
    avg_distances = np.mean(distance_matrix, axis=1)
    cluster_centers = avg_distances < np.percentile(avg_distances, 30)
    
    # Bonus for edges connected to cluster centers
    cluster_bonus = np.zeros((n, n))
    for i in range(n):
        if cluster_centers[i]:
            # Connect cluster centers to their nearest 30% of nodes
            sorted_idx = np.argsort(distance_matrix[i])
            top_k = int(0.3 * n)
            for j in sorted_idx[1:top_k+1]:
                cluster_bonus[i, j] = cluster_bonus[j, i] = 0.7
    
    # 4. Two-hop potential analysis (path extension quality)
    path_potential = np.zeros((n, n))
    for i in range(n):
        # Find best and second-best connections from each node
        sorted_idx = np.argsort(distance_matrix[i])
        best_j = sorted_idx[1]
        second_best_k = sorted_idx[2] if n > 2 else sorted_idx[1]
        
        # Calculate potential of extending through this edge
        for j in range(n):
            if j != i and distance_matrix[i, j] < np.inf:
                # Find best continuation from j (excluding i)
                best_continuation = np.inf
                for k in range(n):
                    if k != i and k != j:
                        if distance_matrix[j, k] < best_continuation:
                            best_continuation = distance_matrix[j, k]
                
                # Edge is promising if it leads to good continuations
                if best_continuation < 1.5 * distance_matrix[i, j]:
                    path_potential[i, j] = 0.8 * (distance_matrix[i, j] / (best_continuation + epsilon))
    
    # 5. Combine all factors with weights
    heuristic = (0.4 * inv_dist + 
                 0.2 * connectivity + 
                 0.2 * cluster_bonus + 
                 0.2 * path_potential)
    
    # 6. Sparsify by keeping only top connections for each node
    sparsified = np.zeros((n, n))
    for i in range(n):
        # Keep top 25% of edges or at least 3 edges per node
        k = max(3, int(0.25 * n))
        indices = np.argsort(heuristic[i])[-k:]
        for j in indices:
            if j != i and heuristic[i, j] > 0:
                sparsified[i, j] = heuristic[i, j]
    
    # Ensure symmetry for undirected TSP
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified
