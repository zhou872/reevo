import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges combining multiple factors with sparsification.
    
    Features:
    1. MST-based edge importance
    2. Local clustering density
    3. Angular sector analysis
    4. Degree-aware sparsification
    """
    n = distance_matrix.shape[0]
    
    # Base inverse distance (like v1)
    with np.errstate(divide='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 1. Minimum Spanning Tree contribution (Prim's algorithm)
    mst_mask = np.zeros((n, n), dtype=bool)
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    min_edge = np.full(n, np.inf)
    min_edge[0] = 0
    
    for _ in range(n - 1):
        # Find minimum edge connecting visited to unvisited
        u = -1
        min_val = np.inf
        for i in range(n):
            if not visited[i] and min_edge[i] < min_val:
                min_val = min_edge[i]
                u = i
        
        if u == -1:
            break
            
        visited[u] = True
        
        # Update minimum edges
        for v in range(n):
            if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                min_edge[v] = distance_matrix[u, v]
                mst_mask[u, v] = True
                mst_mask[v, u] = True
    
    mst_bonus = mst_mask.astype(float) * 2.0
    
    # 2. Local clustering density
    cluster_score = np.zeros((n, n))
    k = min(5, n // 3)  # Adaptive neighborhood size
    
    for i in range(n):
        # Find k nearest neighbors
        sorted_indices = np.argsort(distance_matrix[i])
        nearest = sorted_indices[1:k+1]
        
        # Score based on mutual connectivity in neighborhood
        for j in range(n):
            if i == j:
                continue
                
            # Count common neighbors
            common_neighbors = 0
            for neighbor in nearest:
                if distance_matrix[neighbor, j] < distance_matrix[i, j] * 1.5:
                    common_neighbors += 1
            
            cluster_score[i, j] = common_neighbors / k
    
    # 3. Angular sector analysis
    angle_score = np.zeros((n, n))
    if n > 3:
        # Compute geometric center
        center = np.mean(np.random.randn(n, 2), axis=0)  # Random reference
        
        # Calculate angular sectors
        angles = np.zeros(n)
        for i in range(n):
            # Pseudo-angle based on distances to two farthest nodes
            farthest = np.argsort(distance_matrix[i])[-2:]
            angles[i] = np.mean(distance_matrix[i, farthest])
        
        # Normalize angles
        if np.max(angles) > 0:
            angles = angles / np.max(angles) * 2 * np.pi
        
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                # Prefer edges that cover angular space
                angle_diff = np.abs(angles[i] - angles[j])
                angle_diff = min(angle_diff, 2*np.pi - angle_diff)
                angle_score[i, j] = np.exp(-angle_diff / (np.pi / 4))
    
    # 4. Combine scores with weighted sum
    weights = [1.0, 0.7, 0.5]  # Inverse distance, MST, clustering, angular
    combined = (weights[0] * inv_dist + 
                weights[1] * mst_bonus + 
                weights[2] * cluster_score)
    
    if n > 3:
        combined += 0.3 * angle_score
    
    # 5. Degree-aware sparsification
    # Calculate promising edges per node
    edges_per_node = max(3, int(np.sqrt(n)))
    
    # Keep top edges for each node
    sparsified = np.zeros_like(combined)
    for i in range(n):
        # Get top edges for this node
        row = combined[i].copy()
        row[i] = -np.inf  # Exclude self
        top_indices = np.argsort(row)[-edges_per_node:]
        
        # Apply sigmoid to keep strongest connections
        threshold = np.percentile(row[top_indices], 30)
        for j in top_indices:
            if combined[i, j] > threshold:
                # Symmetric scoring
                sparsified[i, j] = combined[i, j]
                sparsified[j, i] = max(sparsified[j, i], combined[i, j])
    
    # 6. Final normalization
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
