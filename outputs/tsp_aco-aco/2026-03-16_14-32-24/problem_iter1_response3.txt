import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Base inverse distance weighting
    2. Bonus for edges in nearest-neighbor chains
    3. Bonus for edges in Minimum Spanning Tree
    4. Sparsification of statistically unpromising edges
    """
    n = distance_matrix.shape[0]
    if n == 0:
        return np.array([])
    
    # 1. Base inverse distance with regularization
    base_scores = 1 / (distance_matrix + 1e-10)
    np.fill_diagonal(base_scores, 0)
    
    # 2. Nearest-neighbor chain bonus
    neighbor_bonus = np.zeros_like(distance_matrix)
    k = min(5, n - 1)  # Number of nearest neighbors to consider
    
    for i in range(n):
        # Get k nearest neighbors
        distances = distance_matrix[i].copy()
        distances[i] = np.inf  # Exclude self
        nearest_indices = np.argpartition(distances, k)[:k]
        
        # Assign higher bonus to closer neighbors
        sorted_indices = nearest_indices[np.argsort(distances[nearest_indices])]
        for rank, j in enumerate(sorted_indices):
            decay = 0.7 ** rank  # Exponential decay with rank
            neighbor_bonus[i, j] += decay
    
    # 3. Minimum Spanning Tree bonus
    mst_bonus = np.zeros_like(distance_matrix)
    try:
        mst = minimum_spanning_tree(distance_matrix)
        mst_dense = mst.toarray()
        mst_mask = (mst_dense + mst_dense.T) > 0  # Symmetrize
        mst_bonus[mst_mask] = 1.0
    except:
        pass  # Fallback if MST computation fails
    
    # 4. Combine scores with weights
    combined = (
        0.5 * base_scores +
        0.3 * neighbor_bonus +
        0.2 * mst_bonus
    )
    
    # 5. Statistical sparsification
    # Compute row-wise statistics (excluding diagonal)
    row_means = np.zeros(n)
    row_stds = np.zeros(n)
    
    for i in range(n):
        row = distance_matrix[i]
        mask = np.ones(n, dtype=bool)
        mask[i] = False
        valid_distances = row[mask]
        if len(valid_distances) > 0:
            row_means[i] = np.mean(valid_distances)
            row_stds[i] = np.std(valid_distances)
    
    # Sparsify edges that are statistical outliers
    sparsity_mask = np.zeros_like(combined, dtype=bool)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Edge is unpromising if distance > mean + 1.5*std for either endpoint
                if (distance_matrix[i, j] > row_means[i] + 1.5 * row_stds[i] or
                    distance_matrix[i, j] > row_means[j] + 1.5 * row_stds[j]):
                    sparsity_mask[i, j] = True
    
    # Set unpromising edges to zero (but preserve MST edges)
    combined[sparsity_mask & ~mst_mask] = 0
    
    # Ensure symmetry (average both directions)
    combined = 0.5 * (combined + combined.T)
    
    return combined
