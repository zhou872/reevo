import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between nodes.
        
    Returns:
    --------
    np.ndarray
        Heuristic scores for each edge, same shape as input.
        Lower scores indicate less promising edges (some may be zero).
    """
    n = distance_matrix.shape[0]
    
    # Avoid division by zero and self-loops
    np.fill_diagonal(distance_matrix, np.inf)
    
    # 1. Basic inverse distance component
    inv_dist = 1 / distance_matrix
    
    # 2. Local density factor: favor edges connected to nodes with many nearby neighbors
    k = max(3, min(10, n // 5))  # Adaptive k for small/large instances
    nearest_dists = np.partition(distance_matrix, k, axis=1)[:, :k]
    density = 1 / (nearest_dists.mean(axis=1) + 1e-10)
    density_factor = np.outer(density, density)
    
    # 3. Minimum Spanning Tree preference
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_preference = np.where(mst + mst.T > 0, 1.2, 1.0)  # Boost MST edges
    
    # 4. Global vs local distance balance
    global_mean = np.mean(distance_matrix[distance_matrix < np.inf])
    local_mean = np.mean(np.partition(distance_matrix, 2, axis=1)[:, 1:3], axis=1)
    balance = 0.5 * (global_mean / (local_mean + 1e-10))
    balance_factor = np.outer(balance, balance)
    
    # Combine factors
    scores = inv_dist * density_factor * mst_preference * balance_factor
    
    # 5. Sparsification: zero out least promising edges
    # Keep top edges per node (at least 2 for connectivity)
    keep_per_node = max(2, min(5, n // 3))
    mask = np.zeros_like(scores, dtype=bool)
    
    for i in range(n):
        row = scores[i].copy()
        row[i] = -np.inf  # Exclude self
        top_indices = np.argpartition(row, -keep_per_node)[-keep_per_node:]
        mask[i, top_indices] = True
        mask[top_indices, i] = True  # Symmetric
    
    sparsified = scores * mask
    
    # Normalize to similar scale as original distances
    max_val = sparsified[sparsified > 0].max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    # Restore diagonal to zero
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
