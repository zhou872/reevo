import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge evaluation with multi-node lookahead regret,
    instance-adaptive parameter learning, and Christofides-inspired global structural biases.
    
    Core Innovations:
    1. Multi-node Regret with 2-step Lookahead: Evaluates edge quality by considering
       alternative 2-edge paths through intermediate nodes.
    2. Adaptive Parameter Learning: Dynamically adjusts weighting parameters based on
       instance-specific distance statistics.
    3. Christofides-inspired Structural Bonuses: Incorporates Minimum Spanning Tree and
       matching-based insights for global solution structure.
    4. Cluster Connectivity Analysis: Identifies and rewards edges connecting dense
       clusters while penalizing isolated connections.
    """
    n = distance_matrix.shape[0]
    
    # Safe distance matrix (avoid self-loops and division issues)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # === 1. INSTANCE-ADAPTIVE PARAMETER LEARNING ===
    # Analyze distance distribution to adapt heuristics
    flat_distances = safe_dist[np.isfinite(safe_dist)]
    dist_mean = np.mean(flat_distances)
    dist_std = np.std(flat_distances)
    dist_skew = np.mean(((flat_distances - dist_mean) / (dist_std + eps))**3)
    
    # Adaptive parameters based on instance characteristics
    # Compact instances (low std relative to mean) get different treatment
    compactness = dist_std / (dist_mean + eps)
    k_candidate = max(4, int(np.sqrt(n) * (1.5 if compactness > 0.5 else 1.2)))
    regret_power = 1.0 + 0.5 * np.clip(dist_skew, -1, 1)  # Skewness affects regret sensitivity
    density_weight = 0.15 + 0.1 * compactness  # More density focus for compact instances
    
    # === 2. ADAPTIVE CANDIDATE LIST WITH SYMMETRY ===
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        neighbors = np.argsort(safe_dist[i])[:k_candidate]
        candidate_mask[i, neighbors] = True
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # === 3. MULTI-NODE REGRET WITH 2-STEP LOOKAHEAD ===
    regret_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        dists_i = safe_dist[i].copy()
        
        # Get best direct connection
        best_idx = np.argmin(dists_i)
        best_dist = dists_i[best_idx]
        
        # Calculate 2-step alternatives for all nodes
        for j in range(n):
            if i == j:
                continue
                
            # Find best 2-step path i->k->j (excluding i and j as intermediates)
            mask = np.ones(n, dtype=bool)
            mask[i] = mask[j] = False
            
            if np.any(mask):
                # Vectorized computation of 2-step paths
                two_step_paths = dists_i[mask] + safe_dist[mask, j]
                best_two_step = np.min(two_step_paths) if len(two_step_paths) > 0 else np.inf
                
                # Regret: how much worse is direct edge compared to best 2-step alternative
                direct_cost = safe_dist[i, j]
                
                if best_two_step < np.inf:
                    # Normalized regret measure
                    regret = (direct_cost - best_two_step) / (best_dist + eps)
                    regret_matrix[i, j] = max(0, regret) ** regret_power
                else:
                    regret_matrix[i, j] = 0
            else:
                regret_matrix[i, j] = 0
    
    # Apply exponential decay to regret (higher regret → lower score)
    regret_penalty = np.exp(-2.0 * regret_matrix)
    
    # === 4. ADAPTIVE RANK-WEIGHTED SCORING ===
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        sorted_indices = np.argsort(safe_dist[i])
        
        # Adaptive decay based on distance distribution around node i
        top_dists = safe_dist[i, sorted_indices[1:min(10, n-1)]]
        if len(top_dists) > 1:
            local_std = np.std(top_dists)
            local_mean = np.mean(top_dists)
            cv = local_std / (local_mean + eps)
            decay_rate = 0.3 + 0.6 * np.tanh(cv)  # Sigmoid-like adjustment
        else:
            decay_rate = 0.7
        
        # Assign exponential rank weights
        for rank, j in enumerate(sorted_indices):
            if i != j:
                rank_weights[i, j] = decay_rate ** rank
    
    # === 5. ENHANCED SPATIAL DENSITY ANALYSIS ===
    density_bonus = np.zeros_like(distance_matrix)
    
    # Kernel density estimation with adaptive bandwidth
    k_density = min(8, n-1)
    bandwidths = np.zeros(n)
    
    for i in range(n):
        k_nearest = np.partition(safe_dist[i], k_density)[:k_density+1]
        bandwidths[i] = np.median(k_nearest[k_nearest > 0])
    
    # Identify cluster structures
    cluster_indicators = np.zeros(n)
    for i in range(n):
        close_neighbors = safe_dist[i] < bandwidths[i] * 1.5
        cluster_indicators[i] = np.sum(close_neighbors) - 1  # Exclude self
    
    # Density bonus with cluster awareness
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Adaptive bandwidth for the pair
                bandwidth = np.sqrt(bandwidths[i] * bandwidths[j])
                
                # Gaussian kernel density around both endpoints
                sigma = bandwidth / 2
                close_to_i = np.exp(-0.5 * (safe_dist[i] / sigma) ** 2)
                close_to_j = np.exp(-0.5 * (safe_dist[j] / sigma) ** 2)
                
                # Joint density measure
                joint_density = np.sum(close_to_i * close_to_j) - 2
                
                # Cluster connectivity bonus
                cluster_bonus = np.sqrt(cluster_indicators[i] * cluster_indicators[j]) / (n + eps)
                
                if joint_density > 0:
                    density_bonus[i, j] = (np.log1p(joint_density) + cluster_bonus) * density_weight
                else:
                    density_bonus[i, j] = -0.05  # Small penalty for isolated edges
    
    # Symmetrize density bonus
    density_bonus = np.maximum(density_bonus, density_bonus.T)
    
    # === 6. CHRISTOFIDES-INSPIRED STRUCTURAL BIASES ===
    structural_bonus = np.zeros_like(distance_matrix)
    
    if n <= 2000:  # Computationally feasible limit
        try:
            # Minimum Spanning Tree edges get base bonus
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_edges = (mst + mst.T) > 0
            
            # Identify odd-degree nodes in MST approximation
            mst_degree = np.sum(mst_edges, axis=1)
            odd_nodes = np.where(mst_degree % 2 == 1)[0]
            
            # Bonus for edges that could form a matching between odd nodes
            if len(odd_nodes) > 0:
                # Compute minimum weight perfect matching approximation
                odd_dist = distance_matrix[np.ix_(odd_nodes, odd_nodes)]
                np.fill_diagonal(odd_dist, np.inf)
                
                # Greedy matching approximation
                matched = np.zeros(len(odd_nodes), dtype=bool)
                matching_bonus = np.zeros_like(distance_matrix)
                
                for idx in range(len(odd_nodes)):
                    if not matched[idx]:
                        available = np.where(~matched)[0]
                        if len(available) > 1:
                            # Find best unmatched neighbor
                            best_match = np.argmin(odd_dist[idx, available])
                            if available[best_match] != idx:
                                i_orig = odd_nodes[idx]
                                j_orig = odd_nodes[available[best_match]]
                                matching_bonus[i_orig, j_orig] = 0.25
                                matching_bonus[j_orig, i_orig] = 0.25
                                matched[idx] = True
                                matched[available[best_match]] = True
                
                # Combine MST and matching bonuses
                structural_bonus = mst_edges * 0.2 + matching_bonus
        except:
            # Fallback to simple MST bonus
            try:
                mst = minimum_spanning_tree(distance_matrix).toarray()
                mst_edges = (mst + mst.T) > 0
                structural_bonus = mst_edges * 0.2
            except:
                pass
    
    # === 7. BASE SCORE WITH INSTANCE-NORMALIZED INVERSE DISTANCE ===
    inv_dist = 1.0 / safe_dist
    # Scale by instance statistics for better normalization
    inv_dist = inv_dist * dist_mean * (1.0 + 0.5 * compactness)
    
    # === 8. COMBINE ALL FACTORS WITH ADAPTIVE WEIGHTING ===
    base_score = inv_dist * rank_weights
    score_with_regret = base_score * regret_penalty
    
    # Adaptive combination based on instance size and compactness
    if n > 100:
        # For larger instances, emphasize regret and structure more
        combo_weight = 0.7 + 0.3 * compactness
        heuristics_matrix = score_with_regret * (1.0 + density_bonus * combo_weight) * (1.0 + structural_bonus)
    else:
        # For smaller instances, balanced approach
        heuristics_matrix = score_with_regret * (1.0 + density_bonus) * (1.0 + structural_bonus)
    
    # === 9. ENHANCED SPARSIFICATION WITH CONNECTIVITY GUARANTEE ===
    # Apply candidate mask
    heuristics_matrix[~candidate_mask] = 0
    
    # Ensure connectivity: each node must have at least 2 connections
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero_count = np.count_nonzero(row)
        
        if nonzero_count < 2:
            # Force connections to nearest neighbors
            top_indices = np.argsort(safe_dist[i])[1:3]  # Two nearest (excluding self)
            boost_val = np.max(row) * 1.2 if np.max(row) > 0 else 0.5
            heuristics_matrix[i, top_indices] = boost_val
            heuristics_matrix[top_indices, i] = boost_val
    
    # Additional connectivity: ensure graph is not too sparse
    avg_connections = np.mean(np.count_nonzero(heuristics_matrix, axis=1))
    if avg_connections < 2.5 and n > 10:
        # Add connections based on combined score
        combined_scores = base_score * (1.0 + density_bonus)
        for i in range(n):
            if np.count_nonzero(heuristics_matrix[i]) < 3:
                potential = np.argsort(combined_scores[i])[-3:]
                for j in potential:
                    if heuristics_matrix[i, j] == 0 and i != j:
                        heuristics_matrix[i, j] = combined_scores[i, j] * 0.5
                        heuristics_matrix[j, i] = combined_scores[j, i] * 0.5
    
    # === 10. SYMMETRIZE AND FINAL NORMALIZATION ===
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1] range with instance-aware scaling
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix /= max_val
    
    # Apply soft threshold to very small values
    heuristics_matrix[heuristics_matrix < 1e-3] = 0
    
    return heuristics_matrix
