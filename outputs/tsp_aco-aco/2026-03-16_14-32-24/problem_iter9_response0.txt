import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with simplified regret modeling, adaptive
    exponential rank weighting, and efficient spatial clustering.
    
    Features:
    1. Simplified Distance Regret: Penalizes edges based on normalized difference
       from the second-best alternative distance for each node.
    2. Adaptive Rank-Weighted Scoring: Uses coefficient of variation of nearest
       distances to determine exponential decay rate for ranking.
    3. Spatial Cluster Density: Estimates local point density using adaptive
       bandwidth for each node pair.
    4. Symmetry-Enforced Sparsification: Maintains symmetric candidate edges with
       minimum connectivity guarantees.
    """
    n = distance_matrix.shape[0]
    eps = 1e-12
    
    # Safe distance matrix (exclude self-loops)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Multi-factor sparsification with symmetry enforcement
    k_candidate = max(4, int(np.sqrt(n) * 1.5))
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        neighbors = np.argsort(safe_dist[i])[:k_candidate]
        candidate_mask[i, neighbors] = True
    
    # Ensure symmetric candidate list
    candidate_mask = candidate_mask | candidate_mask.T
    
    # 2. Simplified regret-based penalty
    regret_penalty = np.ones_like(distance_matrix)
    
    for i in range(n):
        dists = safe_dist[i].copy()
        if len(dists) >= 2:
            # Get first and second smallest distances (excluding self)
            sorted_dists = np.sort(dists[dists < np.inf])
            best_dist = sorted_dists[0]
            second_best = sorted_dists[1] if len(sorted_dists) > 1 else best_dist
            
            for j in range(n):
                if i != j:
                    edge_dist = dists[j]
                    if np.isfinite(edge_dist):
                        # Simplified regret formula from suggestions
                        normalized_diff = (edge_dist - second_best) / (second_best + eps)
                        regret_penalty[i, j] = np.exp(-np.clip(normalized_diff, 0, 10))
    
    # 3. Exponential rank-weighted scoring with coefficient of variation
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get first 5 nearest distances (excluding self)
        sorted_dists = np.sort(safe_dist[i])
        nearest_dists = sorted_dists[1:min(6, n)]
        
        if len(nearest_dists) >= 2:
            # Calculate coefficient of variation
            mean_dist = np.mean(nearest_dists)
            std_dist = np.std(nearest_dists)
            cv = std_dist / (mean_dist + eps)
            
            # Map CV to decay rate using sigmoid, clipped to [0.3, 0.85]
            sigmoid_cv = 1 / (1 + np.exp(-5 * (cv - 0.5)))  # Centered at 0.5
            decay_rate = 0.3 + 0.55 * sigmoid_cv
            decay_rate = np.clip(decay_rate, 0.3, 0.85)
        else:
            decay_rate = 0.7
        
        # Assign exponential rank weights
        sorted_idx = np.argsort(safe_dist[i])
        for rank, j in enumerate(sorted_idx):
            if i != j:
                rank_weights[i, j] = decay_rate ** rank
    
    # 4. Spatial density clustering with adaptive bandwidth
    density_bonus = np.zeros_like(distance_matrix)
    
    # Adaptive bandwidth using nearest neighbor distances
    k_density = min(5, n - 1)
    bandwidths = np.zeros(n)
    
    for i in range(n):
        k_nearest = np.partition(safe_dist[i], k_density)[:k_density + 1]
        bandwidths[i] = np.median(k_nearest[k_nearest > 0])
    
    # Compute density bonus for candidate edges only
    for i in range(n):
        for j in range(i + 1, n):
            if candidate_mask[i, j]:
                # Combined bandwidth for the pair
                bandwidth = np.sqrt(bandwidths[i] * bandwidths[j])
                if bandwidth > eps:
                    # Count points in joint neighborhood
                    in_neighborhood_i = safe_dist[i] <= (2 * bandwidth)
                    in_neighborhood_j = safe_dist[j] <= (2 * bandwidth)
                    joint_count = np.sum(in_neighborhood_i & in_neighborhood_j) - 2
                    
                    # Bonus proportional to log of cluster size
                    if joint_count > 0:
                        density_bonus[i, j] = np.log1p(joint_count) * 0.2
                    else:
                        density_bonus[i, j] = -0.05
    
    # Make symmetric
    density_bonus = np.maximum(density_bonus, density_bonus.T)
    
    # 5. MST structural bias (for moderate-sized instances)
    mst_bonus = np.zeros_like(distance_matrix)
    if n <= 1000:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_mask = (mst + mst.T) > 0
            mst_bonus = mst_mask * 0.25
        except:
            pass
    
    # 6. Combine factors with base inverse distance
    inv_dist = 1.0 / safe_dist
    mean_dist = np.mean(safe_dist[safe_dist < np.inf])
    scaled_inv_dist = inv_dist * mean_dist
    
    base_score = scaled_inv_dist * rank_weights
    score_with_regret = base_score * regret_penalty
    heuristics_matrix = score_with_regret * (1.0 + density_bonus) * (1.0 + mst_bonus)
    
    # 7. Apply sparsification mask
    heuristics_matrix[~candidate_mask] = 0
    
    # Ensure minimum connectivity (at least 2 edges per node)
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero_count = np.count_nonzero(row)
        
        if nonzero_count < 2:
            # Force top 2 connections
            top_indices = np.argsort(row)[-2:]
            max_val = np.max(row)
            boost_val = max_val * 1.2 if max_val > 0 else 1.0
            
            heuristics_matrix[i, top_indices] = boost_val
            heuristics_matrix[top_indices, i] = boost_val
    
    # 8. Final symmetrization and normalization
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    max_val = np.max(heuristics_matrix)
    if max_val > eps:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
