import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Distance-rank composite scoring
    2. Two-level neighbor-based sparsification  
    3. Minimum spanning tree penalty
    4. Edge variance penalty
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance matrix
    inv_dist = np.zeros_like(distance_matrix, dtype=np.float64)
    np.fill_diagonal(inv_dist, 0)
    mask = distance_matrix != 0
    inv_dist[mask] = 1.0 / distance_matrix[mask]
    
    # 2. Distance-rank composite score
    # For each node, rank edges by distance (closest = highest rank)
    rank_scores = np.zeros_like(distance_matrix)
    for i in range(n):
        valid_indices = np.arange(n) != i
        if np.any(valid_indices):
            dist_to_i = distance_matrix[i, valid_indices]
            sorted_indices = np.argsort(dist_to_i)
            ranks = np.zeros(n - 1)
            # Assign ranks inversely (closest gets highest rank)
            ranks[sorted_indices] = np.arange(n - 1, 0, -1)
            rank_scores[i, valid_indices] = ranks / (n - 1)  # Normalize
    
    # Composite of inverse distance and rank
    composite = inv_dist * (0.6 + 0.4 * rank_scores)
    
    # 3. Two-level neighbor sparsification
    # First level: keep only k-nearest neighbors
    k1 = min(4, n - 1)
    neighbor_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        valid_indices = np.arange(n) != i
        if np.sum(valid_indices) > 0:
            dists = distance_matrix[i, valid_indices]
            indices = np.where(valid_indices)[0]
            if len(dists) > 0:
                kth_idx = min(k1 - 1, len(dists) - 1)
                threshold = np.partition(dists, kth_idx)[kth_idx]
                neighbor_mask[i, indices] = dists <= threshold
    
    # Second level: mutual nearest neighbors
    mutual_mask = neighbor_mask & neighbor_mask.T
    
    # Combined mask (mutual neighbors get priority, then one-way neighbors)
    final_mask = mutual_mask.copy()
    for i in range(n):
        one_way_count = np.sum(neighbor_mask[i]) - np.sum(mutual_mask[i])
        if one_way_count > 0:
            # Add some one-way connections if we have room
            candidates = np.where(neighbor_mask[i] & ~mutual_mask[i])[0]
            if len(candidates) > 0:
                dists = distance_matrix[i, candidates]
                max_add = min(2, len(candidates))
                add_idx = np.argsort(dists)[:max_add]
                final_mask[i, candidates[add_idx]] = True
    
    # 4. Minimum spanning tree penalty
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = mst > 0
    # Edges in MST get bonus, others get penalty
    mst_bonus = 1.2 * mst_mask.astype(float)
    composite *= mst_bonus
    
    # 5. Edge variance penalty
    # Penalize edges with high variance to neighboring edges
    edge_variance_penalty = np.ones_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            # Find neighboring edges (edges sharing nodes i or j)
            neighbor_edges = []
            for k in range(n):
                if k != i and k != j:
                    neighbor_edges.extend([distance_matrix[i, k], distance_matrix[j, k]])
            
            if len(neighbor_edges) > 1:
                mean_dist = np.mean(neighbor_edges)
                var_penalty = 1.0 / (1.0 + np.abs(distance_matrix[i, j] - mean_dist) / mean_dist)
                edge_variance_penalty[i, j] = edge_variance_penalty[j, i] = var_penalty
    
    composite *= edge_variance_penalty
    
    # Apply sparsification mask and normalize
    heuristics_matrix = np.zeros_like(distance_matrix)
    heuristics_matrix[final_mask] = composite[final_mask]
    
    # Final normalization
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        max_val = np.max(non_zero)
        if max_val > 0:
            heuristics_matrix[heuristics_matrix > 0] /= max_val
    
    # Ensure symmetry
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    
    return heuristics_matrix
