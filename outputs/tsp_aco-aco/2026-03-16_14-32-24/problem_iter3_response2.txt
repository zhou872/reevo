import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge evaluation with simplified regret modeling,
    reciprocal rank weighting, and improved density bonuses.
    
    Features:
    1. Reciprocal Rank Weighting: Use 1/(rank+1) instead of exponential decay
    2. Simplified Regret Penalty: Based on second-best alternative distance
    3. Adaptive Neighbor Density Bonus: Count nodes within median distance
    4. Dynamic Candidate Sparsification: Top k edges per node with symmetry
    """
    n = distance_matrix.shape[0]
    
    # Safe distance matrix (avoid self-loops)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # Base inverse distance (normalized by mean distance)
    inv_dist = 1.0 / safe_dist
    mean_dist = np.mean(safe_dist[safe_dist < np.inf])
    inv_dist = inv_dist * mean_dist
    
    # 1. Adaptive candidate list with symmetry enforcement
    # Use sqrt(n) candidates per node, but at least 3
    k_candidate = max(3, int(np.sqrt(n)))
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get top k candidates by distance (nearest neighbors)
        neighbors = np.argsort(safe_dist[i])[:k_candidate]
        candidate_mask[i, neighbors] = True
    
    # Ensure candidate list is symmetric
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # 2. Simplified Regret-Based Penalty
    regret_penalty = np.ones_like(distance_matrix)
    
    for i in range(n):
        dists = safe_dist[i].copy()
        if len(dists) >= 2:
            # Find second best distance for this node
            second_best = np.partition(dists, 1)[1]
            
            for j in range(n):
                if i != j:
                    edge_dist = dists[j]
                    # Penalty increases with distance relative to second best
                    regret = (edge_dist - second_best) / (mean_dist + eps)
                    # Apply exponential decay to penalty factor
                    regret_penalty[i, j] = np.exp(-max(0, regret))
    
    # 3. Reciprocal Rank Weighting
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get sorted indices by distance
        sorted_idx = np.argsort(safe_dist[i])
        
        # Assign reciprocal rank weights: 1/(rank+1)
        for rank, j in enumerate(sorted_idx):
            if i != j:
                rank_weights[i, j] = 1.0 / (rank + 1)
    
    # 4. Adaptive Neighbor Density Bonus
    density_bonus = np.zeros_like(distance_matrix)
    
    # Compute median distance for each node (for density threshold)
    median_distances = np.zeros(n)
    for i in range(n):
        valid_dists = safe_dist[i][safe_dist[i] < np.inf]
        median_distances[i] = np.median(valid_dists) if len(valid_dists) > 0 else mean_dist
    
    # Compute density for each node (nodes within median distance)
    node_densities = np.zeros(n)
    for i in range(n):
        within_threshold = (safe_dist[i] <= median_distances[i]) & (safe_dist[i] < np.inf)
        node_densities[i] = np.sum(within_threshold)
    
    # Normalize densities
    max_density = np.max(node_densities)
    if max_density > 0:
        node_densities = node_densities / max_density
    
    # Apply density bonus to edges
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Geometric mean of endpoint densities
                geo_mean = np.sqrt(node_densities[i] * node_densities[j] + eps)
                # Bonus proportional to geometric mean
                density_bonus[i, j] = 0.2 * geo_mean
    
    # Make symmetric
    density_bonus = np.maximum(density_bonus, density_bonus.T)
    
    # 5. Minimum Spanning Tree bias (if computationally feasible)
    if n <= 1000:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_bias = (mst + mst.T) > 0
            mst_bonus = mst_bias * 0.15
        except:
            mst_bonus = np.zeros_like(distance_matrix)
    else:
        mst_bonus = np.zeros_like(distance_matrix)
    
    # 6. Combine all factors
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base score with rank weighting and regret penalty
    base_score = inv_dist * rank_weights
    score_with_regret = base_score * regret_penalty
    
    # Add density and MST bonuses
    heuristics_matrix = score_with_regret * (1.0 + density_bonus) * (1.0 + mst_bonus)
    
    # 7. Apply candidate mask
    heuristics_matrix[~candidate_mask] = 0
    
    # 8. Ensure minimum connectivity
    # Each node should have at least 2 non-zero connections for TSP feasibility
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero_count = np.count_nonzero(row)
        
        if nonzero_count < 2:
            # Find top 2 edges by base score (before masking)
            top_indices = np.argsort(score_with_regret[i])[-2:]
            for idx in top_indices:
                if idx != i:
                    boost_val = np.max(heuristics_matrix[i]) * 1.2 if np.max(heuristics_matrix[i]) > 0 else 0.1
                    heuristics_matrix[i, idx] = boost_val
                    heuristics_matrix[idx, i] = boost_val
    
    # 9. Symmetrize and normalize
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
