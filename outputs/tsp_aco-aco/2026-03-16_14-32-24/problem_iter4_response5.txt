import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import List, Tuple
import itertools

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with adaptive multi-strategy fusion, pheromone-inspired
    reinforcement, local search integration, and dynamic problem-specific tuning.
    
    Core Innovations:
    1. Multi-strategy adaptive fusion with problem-aware weighting
    2. Pheromone-inspired reinforcement learning via local search perturbations
    3. Integration of 2-opt local search to identify promising edges
    4. Dynamic parameter tuning based on problem geometry and scale
    """
    n = distance_matrix.shape[0]
    if n < 3:
        heuristics_matrix = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics_matrix, 0)
        return heuristics_matrix
    
    # ========== PROBLEM ANALYSIS & DYNAMIC TUNING ==========
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # Analyze problem characteristics
    flat_dists = safe_dist[np.isfinite(safe_dist)]
    dist_std = np.std(flat_dists)
    dist_skew = np.mean((flat_dists - np.mean(flat_dists))**3) / (dist_std**3 + 1e-10)
    
    # Dynamic parameters based on problem characteristics
    scale_factor = np.log10(n + 1) / np.log10(100)
    
    # Candidate list size - adaptive to problem size
    base_k = max(3, int(np.sqrt(n)))
    adaptive_k = min(base_k + int(scale_factor * 3), n - 1)
    
    # Strategy weights based on problem analysis
    if dist_skew > 0.5:
        # Skewed distribution: emphasize local search and reinforcement
        strategy_weights = {
            'base': 0.2, 'regret': 0.2, 'lookahead': 0.15,
            'local_search': 0.3, 'pheromone': 0.15
        }
    elif dist_std / np.mean(flat_dists) > 0.3:
        # High variance: emphasize regret and lookahead
        strategy_weights = {
            'base': 0.15, 'regret': 0.3, 'lookahead': 0.25,
            'local_search': 0.2, 'pheromone': 0.1
        }
    else:
        # Balanced problem
        strategy_weights = {
            'base': 0.25, 'regret': 0.25, 'lookahead': 0.2,
            'local_search': 0.2, 'pheromone': 0.1
        }
    
    # ========== CORE HEURISTIC COMPONENTS ==========
    
    # 1. Base inverse distance with exponential decay
    inv_dist = 1.0 / safe_dist
    np.fill_diagonal(inv_dist, 0)
    
    # Rank-based weighting
    rank_weight = np.zeros_like(distance_matrix)
    decay = 0.6 + 0.2 * np.exp(-scale_factor)
    
    for i in range(n):
        sorted_idx = np.argsort(safe_dist[i])
        for rank, j in enumerate(sorted_idx):
            if i != j:
                rank_weight[i, j] = decay ** rank
    
    base_component = inv_dist * rank_weight
    
    # 2. Enhanced regret with opportunity cost
    regret_component = np.zeros_like(distance_matrix)
    
    for i in range(n):
        dists = safe_dist[i].copy()
        if len(dists) < 3:
            continue
            
        sorted_dists = np.sort(dists)
        best_idx = np.where(dists == sorted_dists[0])[0][0]
        second_best = sorted_dists[1]
        third_best = sorted_dists[2]
        
        for j in range(n):
            if i != j:
                if j == best_idx:
                    regret = 0
                else:
                    # Normalized regret considering multiple alternatives
                    regret = (dists[j] - second_best) / (third_best - second_best + 1e-10)
                
                # Apply sigmoid to bound regret
                regret_component[i, j] = 1 / (1 + np.exp(5 * (regret - 0.5)))
    
    # 3. Two-step lookahead with path coherence
    lookahead_component = np.zeros_like(distance_matrix)
    
    # Pre-compute nearest neighbors for efficiency
    k_neighbors = min(8, n - 1)
    nearest = np.argsort(safe_dist, axis=1)[:, 1:k_neighbors+1]
    
    for i in range(n):
        for j in range(i+1, n):
            # Find best continuation from both ends
            best_from_i = float('inf')
            best_from_j = float('inf')
            
            # From i (avoiding j)
            i_neighbors = [nbr for nbr in nearest[i] if nbr != j]
            if i_neighbors:
                best_from_i = np.min(safe_dist[i, i_neighbors])
            
            # From j (avoiding i)
            j_neighbors = [nbr for nbr in nearest[j] if nbr != i]
            if j_neighbors:
                best_from_j = np.min(safe_dist[j, j_neighbors])
            
            if best_from_i < float('inf') and best_from_j < float('inf'):
                # Coherence score based on continuation quality
                path_coherence = (safe_dist[i, j] + best_from_i + best_from_j) / 3
                median_dist = np.median([safe_dist[i, k] for k in range(n) if k != i])
                lookahead_component[i, j] = lookahead_component[j, i] = np.exp(-path_coherence / (median_dist + 1e-10))
    
    # ========== LOCAL SEARCH INTEGRATION ==========
    
    def fast_2opt_perturbation(tour: List[int], dist_matrix: np.ndarray, max_iter: int = 100) -> List[int]:
        """Apply 2-opt local search to improve a tour."""
        n_nodes = len(tour)
        best_tour = tour.copy()
        best_length = sum(dist_matrix[best_tour[i], best_tour[(i+1) % n_nodes]] for i in range(n_nodes))
        
        improved = True
        iterations = 0
        
        while improved and iterations < max_iter:
            improved = False
            for i in range(n_nodes):
                for j in range(i+2, n_nodes):
                    if j == n_nodes - 1 and i == 0:
                        continue
                    
                    # Calculate gain from 2-opt move
                    a, b = best_tour[i], best_tour[(i+1) % n_nodes]
                    c, d = best_tour[j], best_tour[(j+1) % n_nodes]
                    
                    current = dist_matrix[a, b] + dist_matrix[c, d]
                    new = dist_matrix[a, c] + dist_matrix[b, d]
                    
                    if new < current - 1e-10:
                        # Reverse segment between i+1 and j
                        best_tour[i+1:j+1] = best_tour[i+1:j+1][::-1]
                        best_length = best_length - current + new
                        improved = True
                        break
                if improved:
                    break
            iterations += 1
        
        return best_tour
    
    def construct_initial_tours(dist_matrix: np.ndarray, n_tours: int = 5) -> List[List[int]]:
        """Construct multiple initial tours using different heuristics."""
        n_nodes = dist_matrix.shape[0]
        tours = []
        
        # Strategy 1: Nearest neighbor with random start
        for _ in range(min(n_tours, n_nodes)):
            start = np.random.randint(0, n_nodes)
            remaining = set(range(n_nodes))
            tour = [start]
            remaining.remove(start)
            
            while remaining:
                last = tour[-1]
                # Find nearest unvisited neighbor
                nearest = min(remaining, key=lambda x: dist_matrix[last, x])
                tour.append(nearest)
                remaining.remove(nearest)
            
            tours.append(tour)
        
        # Strategy 2: Christofides algorithm approximation
        if n_nodes >= 4:
            try:
                # Minimum spanning tree
                mst = minimum_spanning_tree(dist_matrix)
                
                # Find odd-degree vertices
                degrees = np.sum(mst.toarray() > 0, axis=1)
                odd_vertices = np.where(degrees % 2 == 1)[0]
                
                if len(odd_vertices) >= 2:
                    # Minimum weight matching on odd vertices
                    sub_dist = dist_matrix[odd_vertices][:, odd_vertices]
                    from scipy.optimize import linear_sum_assignment
                    row_ind, col_ind = linear_sum_assignment(sub_dist)
                    
                    # Combine MST and matching
                    combined = mst.toarray().copy()
                    for i, j in zip(row_ind, col_ind):
                        if i != j:
                            v1, v2 = odd_vertices[i], odd_vertices[j]
                            combined[v1, v2] = max(combined[v1, v2], dist_matrix[v1, v2])
                            combined[v2, v1] = combined[v1, v2]
                    
                    # Find Eulerian tour and shortcut
                    # Simplified: use MST edges as base
                    mst_tour = list(np.where(mst.toarray().flatten() > 0)[0])
                    if len(mst_tour) >= n_nodes:
                        tours.append(mst_tour[:n_nodes])
            except:
                pass
        
        return tours
    
    # Generate and improve multiple tours
    local_search_component = np.zeros_like(distance_matrix)
    n_initial_tours = min(10, max(3, n // 10))
    
    initial_tours = construct_initial_tours(safe_dist, n_initial_tours)
    
    edge_counts = np.zeros_like(distance_matrix, dtype=int)
    
    for tour in initial_tours:
        # Improve with 2-opt
        improved_tour = fast_2opt_perturbation(tour, safe_dist, max_iter=50)
        
        # Count edges in improved tour
        n_nodes = len(improved_tour)
        for idx in range(n_nodes):
            i = improved_tour[idx]
            j = improved_tour[(idx + 1) % n_nodes]
            edge_counts[i, j] += 1
            edge_counts[j, i] += 1
    
    # Normalize edge frequencies
    if np.max(edge_counts) > 0:
        local_search_component = edge_counts / np.max(edge_counts)
    
    # ========== PHEROMONE-INSPIRED REINFORCEMENT ==========
    
    pheromone_component = np.zeros_like(distance_matrix)
    
    # Initialize pheromones based on inverse distance
    pheromone = 1.0 / (safe_dist + 1e-10)
    np.fill_diagonal(pheromone, 0)
    
    # Evaporation and reinforcement based on local search results
    evaporation = 0.3 + 0.4 * scale_factor  # Higher evaporation for larger problems
    reinforcement = 0.5
    
    # Evaporate all pheromones
    pheromone *= (1 - evaporation)
    
    # Reinforce edges found in good tours
    reinforcement_mask = local_search_component > 0.5
    pheromone[reinforcement_mask] += reinforcement * local_search_component[reinforcement_mask]
    
    # Additional reinforcement for edges in minimum spanning tree
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_mask = mst > 0
    pheromone[mst_mask] += reinforcement * 0.3
    
    # Normalize pheromone component
    if np.max(pheromone) > 0:
        pheromone_component = pheromone / np.max(pheromone)
    
    # ========== ADAPTIVE MULTI-STRATEGY FUSION ==========
    
    # Normalize components to similar scales
    components = {
        'base': base_component,
        'regret': regret_component,
        'lookahead': lookahead_component,
        'local_search': local_search_component,
        'pheromone': pheromone_component
    }
    
    # Normalize each component
    for key in components:
        if np.max(components[key]) > 0:
            components[key] = components[key] / np.max(components[key])
    
    # Combine with adaptive weights
    heuristics_matrix = np.zeros_like(distance_matrix)
    for key, weight in strategy_weights.items():
        heuristics_matrix += weight * components[key]
    
    # ========== INTELLIGENT SPARSIFICATION ==========
    
    # Adaptive candidate selection
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Keep top adaptive_k edges per node
    for i in range(n):
        if adaptive_k < n - 1:
            # Get top edges, considering both heuristics and distance
            combined_scores = 0.7 * heuristics_matrix[i] + 0.3 * (1.0 / (safe_dist[i] + 1e-10))
            combined_scores[i] = -np.inf
            
            top_indices = np.argsort(combined_scores)[-adaptive_k:]
            candidate_mask[i, top_indices] = True
        else:
            candidate_mask[i, :] = True
            candidate_mask[i, i] = False
    
    # Ensure connectivity: add edges from minimum spanning tree
    candidate_mask = candidate_mask | (mst > 0)
    
    # Ensure symmetry
    candidate_mask = candidate_mask | candidate_mask.T
    
    # Apply sparsification
    heuristics_matrix[~candidate_mask] = 0
    
    # ========== FINAL REFINEMENT ==========
    
    # Smooth the heuristics using neighborhood averaging
    smoothed = heuristics_matrix.copy()
    for i in range(n):
        neighbors = np.where(candidate_mask[i])[0]
        if len(neighbors) > 0:
            for j in neighbors:
                if j > i:
                    # Average with common neighbors
                    common = np.where(candidate_mask[i] & candidate_mask[j])[0]
                    if len(common) > 0:
                        avg_val = np.mean([heuristics_matrix[i, k] + heuristics_matrix[k, j] for k in common]) / 2
                        smoothed[i, j] = smoothed[j, i] = 0.7 * heuristics_matrix[i, j] + 0.3 * avg_val
    
    heuristics_matrix = smoothed
    
    # Final normalization
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        min_val, max_val = np.min(non_zero), np.max(non_zero)
        if max_val > min_val:
            heuristics_matrix[heuristics_matrix > 0] = (heuristics_matrix[heuristics_matrix > 0] - min_val) / (max_val - min_val)
    
    # Ensure symmetry and zero diagonal
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
