import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    
    Features:
    1. K-Nearest Neighbor sparsity: Keep only top k connections per node
    2. Triangular inequality violation penalty: Penalize edges that create detours
    3. Minimum spanning tree bonus: Reward edges likely in optimal solutions
    4. Node centrality weighting: Prioritize connections to centrally located nodes
    """
    n = distance_matrix.shape[0]
    
    # 1. Inverse distance baseline (handle zeros to avoid division by zero)
    with np.errstate(divide='ignore'):
        inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. K-Nearest Neighbor sparsity mask
    k = max(3, int(np.log2(n)))  # Dynamic k based on problem size
    knn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k+1 smallest distances (including self)
        idx = np.argpartition(distance_matrix[i], k+1)[:k+1]
        # Remove self from neighbors
        idx = idx[idx != i][:k]
        knn_mask[i, idx] = True
    
    # Make mask symmetric (undirected graph assumption)
    knn_mask = knn_mask | knn_mask.T
    
    # 3. Triangular inequality violation penalty
    # For each triangle (i,j,k), penalty if d(i,j) > d(i,k) + d(k,j)
    tri_penalty = np.zeros_like(distance_matrix)
    
    # Sample triangles to avoid O(n³) complexity
    sample_size = min(1000, n)
    sample_nodes = np.random.choice(n, size=sample_size, replace=False)
    
    for i in range(n):
        for k in sample_nodes:
            if i != k:
                violation = distance_matrix[i, :] > (distance_matrix[i, k] + distance_matrix[k, :])
                tri_penalty[i, violation] += 1
    
    # Normalize and convert to penalty factor (0.1 to 1.0)
    tri_penalty = 1.0 - 0.9 * (tri_penalty / tri_penalty.max())
    np.fill_diagonal(tri_penalty, 0)
    
    # 4. Minimum spanning tree bonus
    mst_bonus = np.zeros_like(distance_matrix)
    
    # Compute MST using Kruskal's algorithm
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = mst > 0
    
    # Bonus for MST edges and their 2-hop neighbors
    for i in range(n):
        mst_neighbors = np.where(mst_edges[i])[0]
        for j in mst_neighbors:
            # Direct MST edge bonus
            mst_bonus[i, j] += 2.0
            # 2-hop neighbors through MST
            for k in np.where(mst_edges[j])[0]:
                if k != i:
                    mst_bonus[i, k] += 0.5
    
    # 5. Node centrality weighting
    # Central nodes have smaller average distance to all other nodes
    centrality = 1 / (np.mean(distance_matrix, axis=1) + 1e-10)
    centrality = centrality / centrality.max()
    
    # Weight edges by geometric mean of endpoint centralities
    centrality_weight = np.sqrt(np.outer(centrality, centrality))
    
    # 6. Combine all factors with weighted geometric mean
    factors = [
        inv_dist,           # Base inverse distance
        tri_penalty,        # Triangular inequality penalty
        mst_bonus + 1.0,   # MST bonus (add 1 to avoid zero)
        centrality_weight   # Node centrality
    ]
    
    weights = [0.4, 0.3, 0.2, 0.1]  # Weight for each factor
    
    # Compute weighted geometric mean
    log_heuristic = np.zeros_like(distance_matrix)
    for factor, weight in zip(factors, weights):
        with np.errstate(divide='ignore', invalid='ignore'):
            log_heuristic += weight * np.log(factor + 1e-10)
    
    heuristic = np.exp(log_heuristic)
    
    # 7. Apply K-NN sparsity mask and ensure symmetry
    heuristic = heuristic * knn_mask
    heuristic = 0.5 * (heuristic + heuristic.T)  # Symmetrize
    
    # 8. Soft thresholding: set bottom 30% of non-zero values to zero
    non_zero_mask = heuristic > 0
    if non_zero_mask.any():
        threshold = np.percentile(heuristic[non_zero_mask], 30)
        heuristic[heuristic < threshold] = 0
    
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
