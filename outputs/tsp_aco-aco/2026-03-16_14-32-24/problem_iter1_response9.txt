import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Two-hop neighborhood scoring (looks at indirect connections)
    2. Distance quantile penalization (downgrades excessively long edges)
    3. Connectivity density bonus (favors edges in dense regions)
    4. Edge sparsification (zeros out unpromising edges)
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix)
    
    # 1. Base inverse distance (handle zeros on diagonal)
    inv_dist = 1 / (distance_matrix + np.eye(n))
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Two-hop neighborhood scoring
    # For edge (i,j), consider best alternative connections through intermediate nodes
    two_hop_bonus = np.zeros((n, n))
    for k in range(n):
        # Exclude self and current edge
        mask = np.ones(n, dtype=bool)
        mask[k] = False
        # Find minimum distance through intermediate node k (excluding i,j themselves)
        for i in range(n):
            if i == k:
                continue
            mask[i] = False
            if np.sum(mask) > 0:
                # Best alternative path i->k->j (if exists)
                alt_paths = inv_dist[i, k] * inv_dist[k, mask]
                if len(alt_paths) > 0:
                    two_hop_bonus[i, mask] += np.max(alt_paths)
            mask[i] = True
    
    # 3. Distance quantile penalization
    # Compute distance percentiles for each node
    quantile_penalty = np.ones((n, n))
    for i in range(n):
        # Get distances from node i to all others (excluding self)
        distances_i = distance_matrix[i, np.arange(n) != i]
        if len(distances_i) > 0:
            q75 = np.percentile(distances_i, 75)
            q90 = np.percentile(distances_i, 90)
            for j in range(n):
                if i != j:
                    d = distance_matrix[i, j]
                    if d > q90:
                        quantile_penalty[i, j] = 0.1
                    elif d > q75:
                        quantile_penalty[i, j] = 0.3
                    else:
                        quantile_penalty[i, j] = 1.0
    
    # 4. Connectivity density bonus
    # Nodes in dense regions get bonus for connections
    density_bonus = np.zeros((n, n))
    # Compute local density for each node (inverse of average distance to nearest neighbors)
    nearest_avg_dist = np.zeros(n)
    for i in range(n):
        # Get distances to k nearest neighbors (excluding self)
        k_neighbors = min(5, n-1)
        distances_i = np.partition(distance_matrix[i, :], k_neighbors)[:k_neighbors+1]
        nearest_avg_dist[i] = np.mean(distances_i[distances_i > 0])
    
    density_factor = 1 / (nearest_avg_dist + 1e-8)
    density_factor = density_factor / np.max(density_factor)  # Normalize
    
    # Apply density bonus: edges between dense nodes get higher bonus
    for i in range(n):
        for j in range(n):
            if i != j:
                density_bonus[i, j] = 0.5 + 0.5 * (density_factor[i] * density_factor[j])
    
    # 5. Combine all factors
    combined_heuristic = inv_dist * (1 + two_hop_bonus) * quantile_penalty * density_bonus
    
    # 6. Edge sparsification - keep only promising edges
    sparsified = np.zeros_like(combined_heuristic)
    
    # For each node, keep top-k edges based on heuristic value
    k = max(3, min(10, n // 3))  # Adaptive k based on problem size
    for i in range(n):
        row = combined_heuristic[i, :]
        # Get indices of top k values (excluding self)
        valid_indices = np.arange(n) != i
        valid_values = row[valid_indices]
        if len(valid_values) >= k:
            threshold = np.partition(valid_values, -k)[-k]
            top_k_mask = row >= threshold
            top_k_mask[i] = False  # Exclude self
            sparsified[i, top_k_mask] = combined_heuristic[i, top_k_mask]
    
    # Ensure symmetry (keep edge if either endpoint considers it promising)
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
