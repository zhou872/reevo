import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform
import warnings
warnings.filterwarnings('ignore', category=RuntimeWarning)

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with deep regret chains, ant-colony inspired
    pheromone simulation, and reinforcement learning parameter optimization.
    
    Returns a matrix of edge promise scores between 0 and 1, where higher
    scores indicate edges more likely to be in optimal solutions.
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # Prepare safe distance matrix
    dmat = distance_matrix.copy()
    np.fill_diagonal(dmat, np.inf)
    eps = 1e-12
    
    # ========== REINFORCEMENT LEARNING PARAMETER OPTIMIZATION ==========
    # Learn instance characteristics to adapt parameters
    flat_dist = dmat[np.isfinite(dmat)]
    if len(flat_dist) == 0:
        return np.zeros_like(distance_matrix)
    
    dist_mean = np.mean(flat_dist)
    dist_std = np.std(flat_dist)
    dist_q25, dist_q75 = np.percentile(flat_dist, [25, 75])
    iqr = dist_q75 - dist_q25
    
    # Instance type classification
    is_uniform = dist_std / (dist_mean + eps) < 0.3
    is_clustered = iqr / (dist_mean + eps) > 0.5
    compactness = dist_std / (dist_mean + eps)
    
    # RL-inspired parameter adaptation
    if n < 50:
        # Small instances: precise calculations
        lookahead_steps = 3
        beam_width = 8
        k_candidate = min(15, n-1)
    elif n < 200:
        # Medium instances: balanced approach
        lookahead_steps = 2 if is_clustered else 3
        beam_width = 6
        k_candidate = min(10, n-1)
    else:
        # Large instances: efficient calculations
        lookahead_steps = 2
        beam_width = 4
        k_candidate = min(8, n-1)
    
    # Density-based adjustments
    if is_clustered:
        lookahead_steps = min(lookahead_steps + 1, 4)
        beam_width = max(beam_width, 5)
    
    # ========== ANT-COLONY PHEROMONE SIMULATION ==========
    pheromone = np.ones_like(dmat)
    np.fill_diagonal(pheromone, 0)
    
    # Construct multiple heuristic tours for pheromone update
    n_tours = min(5, max(2, n // 20))
    
    for tour_idx in range(n_tours):
        # Start from different nodes
        start = tour_idx % n
        unvisited = set(range(n))
        current = start
        tour = [current]
        unvisited.remove(current)
        
        # Build tour with randomized greedy
        while unvisited:
            # Get candidate next nodes
            candidates = list(unvisited)
            dists = dmat[current, candidates]
            
            # Add some randomness to exploration
            if len(candidates) > 1:
                # Softmax selection
                inv_dists = 1.0 / (dists + eps)
                probs = inv_dists / np.sum(inv_dists)
                
                # Temperature decreases as tour progresses
                temp = 0.5 + 0.5 * (len(unvisited) / n)
                scaled_probs = np.exp(np.log(probs + eps) / temp)
                scaled_probs = scaled_probs / np.sum(scaled_probs)
                
                # Sample next node
                next_idx = np.random.choice(len(candidates), p=scaled_probs)
                next_node = candidates[next_idx]
            else:
                next_node = candidates[0]
            
            tour.append(next_node)
            unvisited.remove(next_node)
            current = next_node
        
        # Update pheromone based on tour quality
        tour_length = 0
        for i in range(len(tour)):
            j = (i + 1) % len(tour)
            tour_length += dmat[tour[i], tour[j]]
        
        # Pheromone deposition: better tours deposit more
        deposition = 1.0 / (tour_length + eps)
        
        for i in range(len(tour)):
            j = (i + 1) % len(tour)
            pheromone[tour[i], tour[j]] += deposition
            pheromone[tour[j], tour[i]] += deposition
    
    # Normalize and evaporate pheromone
    pheromone_max = np.max(pheromone)
    if pheromone_max > 0:
        pheromone = pheromone / pheromone_max
    
    # ========== DEEP REGRET CHAINS WITH BEAM SEARCH ==========
    regret_scores = np.zeros_like(dmat)
    
    # Precompute nearest neighbors for efficiency
    k_neighbors = min(k_candidate * 2, n - 1)
    neighbor_indices = np.argsort(dmat, axis=1)[:, 1:k_neighbors+1]
    neighbor_distances = np.take_along_axis(dmat, neighbor_indices, axis=1)
    
    for i in range(n):
        # Skip if node has no valid connections
        if np.all(np.isinf(dmat[i])):
            continue
            
        # Get direct distances from i
        dists_i = dmat[i].copy()
        
        for j in range(n):
            if i == j or np.isinf(dists_i[j]):
                continue
            
            # Beam search for multi-step alternative paths
            beam = [(j, dists_i[j], [i, j])]  # (current, total_cost, path)
            
            for step in range(lookahead_steps - 1):
                new_beam = []
                
                for node, cost, path in beam:
                    # Get candidate continuations
                    candidates = neighbor_indices[node]
                    
                    # Filter already visited nodes (except maybe return to i)
                    valid_mask = ~np.isin(candidates, path)
                    if step == lookahead_steps - 2:  # Last step can return to i
                        valid_mask = valid_mask | (candidates == i)
                    
                    valid_candidates = candidates[valid_mask]
                    
                    if len(valid_candidates) == 0:
                        continue
                    
                    # Get distances to valid candidates
                    candidate_dists = dmat[node, valid_candidates]
                    
                    # Keep top beam_width candidates
                    top_k = min(beam_width, len(valid_candidates))
                    top_indices = np.argpartition(candidate_dists, top_k)[:top_k]
                    
                    for idx in top_indices:
                        next_node = valid_candidates[idx]
                        new_cost = cost + candidate_dists[idx]
                        new_path = path + [next_node]
                        new_beam.append((next_node, new_cost, new_path))
                
                if not new_beam:
                    break
                
                # Prune beam
                new_beam.sort(key=lambda x: x[1])
                beam = new_beam[:beam_width]
            
            # Calculate regret: difference between direct edge and best alternative
            if beam:
                best_alt_cost = min(beam, key=lambda x: x[1])[1]
                direct_cost = dists_i[j]
                
                if best_alt_cost < np.inf and direct_cost > 0:
                    # Normalized regret
                    regret = (direct_cost - best_alt_cost) / (direct_cost + eps)
                    regret_scores[i, j] = max(0, regret)
                else:
                    regret_scores[i, j] = 0
    
    # Convert regret to score (higher regret = better edge)
    regret_boost = 1.0 + np.tanh(regret_scores * 2.0)  # Scale to [1, 2]
    
    # ========== CLUSTER-AWARE STRUCTURAL BONUSES ==========
    structural_bonus = np.zeros_like(dmat)
    
    # Minimum Spanning Tree edges (global structure)
    if n <= 5000:  # Computational limit for MST
        try:
            mst = minimum_spanning_tree(dmat).toarray()
            mst_edges = (mst + mst.T) > 0
            structural_bonus += mst_edges * 0.3
        except:
            pass
    
    # Local clustering coefficient
    if n > 10 and n <= 1000:
        # Identify potential clusters via distance thresholds
        threshold = np.percentile(flat_dist, 25)  # 25th percentile as closeness threshold
        
        cluster_affinity = np.zeros(n)
        for i in range(n):
            close_neighbors = np.sum(dmat[i] < threshold)
            cluster_affinity[i] = close_neighbors / (n - 1)
        
        # Bonus for edges within dense regions
        for i in range(n):
            for j in range(i+1, n):
                if dmat[i, j] < threshold:
                    density_score = np.sqrt(cluster_affinity[i] * cluster_affinity[j])
                    structural_bonus[i, j] += density_score * 0.2
                    structural_bonus[j, i] += density_score * 0.2
    
    # ========== ADAPTIVE BASE SCORING ==========
    # Inverse distance with instance-aware normalization
    inv_dist = 1.0 / (dmat + dist_mean * 0.1)  # Add small offset for stability
    
    # Rank-based weighting
    rank_weights = np.zeros_like(dmat)
    for i in range(n):
        sorted_idx = np.argsort(dmat[i])
        for rank, j in enumerate(sorted_idx):
            if i != j and rank < k_candidate:
                # Exponential decay based on rank
                decay = np.exp(-rank / max(1, k_candidate / 2))
                rank_weights[i, j] = decay
    
    base_score = inv_dist * rank_weights
    
    # ========== COMBINE ALL COMPONENTS ==========
    # Adaptive weighting based on instance characteristics
    if is_uniform:
        # Uniform instances: emphasize regret and pheromone
        pheromone_weight = 0.4
        regret_weight = 0.4
        structure_weight = 0.2
    elif is_clustered:
        # Clustered instances: emphasize structure and pheromone
        pheromone_weight = 0.3
        regret_weight = 0.3
        structure_weight = 0.4
    else:
        # Mixed instances: balanced approach
        pheromone_weight = 0.35
        regret_weight = 0.35
        structure_weight = 0.3
    
    # Combine scores
    heuristics_matrix = (
        base_score * 
        (1.0 + pheromone * pheromone_weight) * 
        regret_boost * 
        (1.0 + structural_bonus * structure_weight)
    )
    
    # ========== INTELLIGENT SPARSIFICATION ==========
    # Keep top edges per node
    edges_per_node = max(3, min(k_candidate, int(np.sqrt(n)) + 2))
    
    candidate_mask = np.zeros_like(dmat, dtype=bool)
    for i in range(n):
        scores = heuristics_matrix[i].copy()
        scores[i] = -np.inf
        
        if np.all(scores <= 0):
            # Fallback: connect to nearest neighbors
            top_idx = np.argsort(dmat[i])[1:edges_per_node+1]
            candidate_mask[i, top_idx] = True
        else:
            top_idx = np.argpartition(scores, -edges_per_node)[-edges_per_node:]
            candidate_mask[i, top_idx] = True
    
    # Ensure symmetry and connectivity
    candidate_mask = candidate_mask | candidate_mask.T
    
    # Force minimum degree of 2 for connectivity
    for i in range(n):
        if np.sum(candidate_mask[i]) < 2:
            # Add nearest neighbors
            nearest = np.argsort(dmat[i])[1:3]
            candidate_mask[i, nearest] = True
            candidate_mask[nearest, i] = True
    
    # Apply sparsification
    heuristics_matrix[~candidate_mask] = 0
    
    # ========== FINAL PROCESSING ==========
    # Symmetrize
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    
    # Row-wise normalization to [0, 1]
    row_max = np.max(heuristics_matrix, axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    heuristics_matrix = heuristics_matrix / row_max
    
    # Global normalization
    overall_max = np.max(heuristics_matrix)
    if overall_max > 0:
        heuristics_matrix = heuristics_matrix / overall_max
    
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
