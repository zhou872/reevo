import numpy as np
from scipy.spatial import ConvexHull
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection.
    Combines multiple geometric and graph-based factors to identify promising edges,
    with sparsification to focus computational effort on high-potential candidates.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # Base inverse distance
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 1. Nearest-Neighbor Clustering with Sparsification
    nn_factor = np.zeros_like(distance_matrix)
    k = max(2, min(5, n // 4))  # Adaptive neighborhood size
    for i in range(n):
        neighbors = np.argsort(distance_matrix[i])[1:k+1]
        nn_factor[i, neighbors] = 1
        nn_factor[neighbors, i] = 1  # Symmetrize
    
    # 2. Triangle Inequality Edge Potential
    tri_factor = np.zeros_like(distance_matrix)
    # Sample random triplets to approximate triangle inequality violations
    n_samples = min(1000, n * 3)
    for _ in range(n_samples):
        i, j, k = np.random.choice(n, 3, replace=False)
        detour_dist = distance_matrix[i, k] + distance_matrix[k, j]
        if detour_dist < 1.2 * distance_matrix[i, j]:  # Within 20% detour tolerance
            tri_factor[i, j] += 1
            tri_factor[j, i] += 1
    tri_factor = np.sqrt(tri_factor / tri_factor.max()) if tri_factor.max() > 0 else tri_factor
    
    # 3. Convex Hull Perimeter Bonus (if coordinates can be approximated)
    hull_factor = np.ones_like(distance_matrix)
    try:
        # Use multidimensional scaling to approximate 2D coordinates
        from sklearn.manifold import MDS
        mds = MDS(n_components=2, dissimilarity='precomputed', random_state=42)
        coords = mds.fit_transform(distance_matrix)
        hull = ConvexHull(coords)
        hull_edges = [(hull.vertices[i], hull.vertices[(i+1)%len(hull.vertices)]) 
                      for i in range(len(hull.vertices))]
        for i, j in hull_edges:
            hull_factor[i, j] = 2.0
            hull_factor[j, i] = 2.0
    except:
        pass  # Fallback if convex hull computation fails
    
    # 4. Minimum Spanning Tree Core
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_factor = (mst > 0).astype(float) * 1.5 + 1.0
    
    # Combine factors with weights
    weights = [0.4, 0.25, 0.2, 0.15]  # Sum to 1.0
    heuristic = (weights[0] * inv_dist * nn_factor +
                 weights[1] * tri_factor +
                 weights[2] * hull_factor +
                 weights[3] * mst_factor)
    
    # Sparsification: Keep only top edges per node
    sparsity = 0.3  # Keep ~30% of edges
    threshold = np.percentile(heuristic, 100 * (1 - sparsity))
    heuristic[heuristic < threshold] = 0
    
    # Ensure symmetry and zero diagonal
    heuristic = (heuristic + heuristic.T) / 2
    np.fill_diagonal(heuristic, 0)
    
    return heuristic / heuristic.max() if heuristic.max() > 0 else heuristic
