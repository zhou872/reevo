import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge prioritization.
    
    Combines multiple factors to score edge promise:
    1. Normalized inverse distance
    2. Nearest-neighbor connections bonus
    3. Minimum spanning tree edge prioritization
    4. Sparsification of unpromising edges
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.ones_like(distance_matrix)
    
    # 1. Inverse distance with normalization
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-10)  # Avoid division by zero
    inv_dist = inv_dist / inv_dist.max()  # Normalize to [0, 1]
    
    # 2. Nearest-neighbor bonus
    nearest_bonus = np.zeros_like(distance_matrix)
    for i in range(n):
        # Find k nearest neighbors (excluding self)
        k = min(5, n - 1)
        sorted_indices = np.argsort(distance_matrix[i])[1:k+1]
        nearest_bonus[i, sorted_indices] = 1 - np.arange(k) / k
    
    # 3. Minimum spanning tree prioritization
    mst_priority = np.zeros_like(distance_matrix)
    
    # Create symmetric version for MST calculation
    symmetric_dist = (distance_matrix + distance_matrix.T) / 2
    mst = minimum_spanning_tree(symmetric_dist).toarray()
    mst_priority = (mst + mst.T) > 0  # Convert to undirected edges
    
    # 4. Combined scoring with sparsification
    edge_scores = (
        0.4 * inv_dist +
        0.3 * nearest_bonus +
        0.3 * mst_priority
    )
    
    # 5. Sparsification: keep only promising edges
    # For each node, keep top k edges
    k_sparse = min(10, n - 1)
    sparse_mask = np.zeros_like(edge_scores, dtype=bool)
    
    for i in range(n):
        # Get top k edges from this node (excluding self)
        top_k_indices = np.argsort(edge_scores[i])[-k_sparse:]
        top_k_indices = top_k_indices[top_k_indices != i]
        sparse_mask[i, top_k_indices] = True
    
    # Also keep symmetric connections
    sparse_mask = sparse_mask | sparse_mask.T
    
    # Apply sparsification
    edge_scores = edge_scores * sparse_mask
    
    # 6. Final enhancement: local clustering bonus
    # Reward edges that connect to nodes with similar neighbor sets
    cluster_bonus = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            if not sparse_mask[i, j]:
                continue
                
            # Get neighbor sets (excluding self)
            i_neighbors = set(np.where(sparse_mask[i])[0]) - {i}
            j_neighbors = set(np.where(sparse_mask[j])[0]) - {j}
            
            if i_neighbors and j_neighbors:
                # Jaccard similarity of neighbor sets
                intersection = len(i_neighbors & j_neighbors)
                union = len(i_neighbors | j_neighbors)
                if union > 0:
                    similarity = intersection / union
                    cluster_bonus[i, j] = cluster_bonus[j, i] = similarity * 0.2
    
    edge_scores = edge_scores + cluster_bonus
    
    # Ensure diagonal is zero
    np.fill_diagonal(edge_scores, 0)
    
    return edge_scores


def validate_heuristics(distance_matrix: np.ndarray):
    """
    Validation and demonstration of the heuristics function.
    """
    n = distance_matrix.shape[0]
    heuristics = heuristics_v2(distance_matrix)
    
    print(f"Matrix shape: {heuristics.shape}")
    print(f"Non-zero entries: {np.count_nonzero(heuristics)} / {n*n}")
    print(f"Non-zero percentage: {100*np.count_nonzero(heuristics)/(n*n):.1f}%")
    print(f"Min value: {heuristics[heuristics>0].min():.4f}" if np.any(heuristics>0) else "Min value: 0")
    print(f"Max value: {heuristics.max():.4f}")
    print(f"Mean non-zero: {heuristics[heuristics>0].mean():.4f}" if np.any(heuristics>0) else "Mean non-zero: 0")
    
    return heuristics
