import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge evaluation with lookahead mechanisms
    and adaptive reinforcement components.
    
    Features:
    1. Two-step Lookahead Edge Contribution: Evaluates edge value within potential 3-node chains
    2. Adaptive Hybrid Regret: Combines absolute and rank-based regret with smoothing
    3. Neighborhood Structural Density: Measures local clustering patterns around edge endpoints
    4. Reinforcement-guided Sparsification: Dynamic candidate selection with solution-quality adaptation
    """
    n = distance_matrix.shape[0]
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # 1. Adaptive candidate list with reinforcement-inspired adjustment
    base_k = max(4, int(np.sqrt(n) * 1.5))
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Dynamic candidate count based on local distance distribution
        dists = safe_dist[i]
        mean_dist = np.mean(dists[dists < np.inf])
        std_dist = np.std(dists[dists < np.inf])
        
        # Adjust candidate count based on dispersion
        dispersion = std_dist / (mean_dist + eps)
        k_adj = int(base_k * (0.7 + 0.6 * np.tanh(1.5 - 3 * dispersion)))
        k_adj = min(max(k_adj, 4), n-1)
        
        neighbors = np.argsort(dists)[:k_adj]
        candidate_mask[i, neighbors] = True
    
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # 2. Two-step Lookahead Edge Contribution
    lookahead_score = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(i+1, n):
            if not candidate_mask[i, j]:
                continue
            
            # Consider best connections for i and j excluding each other
            i_dists = safe_dist[i].copy()
            j_dists = safe_dist[j].copy()
            i_dists[j] = np.inf
            j_dists[i] = np.inf
            
            # Find best two alternatives for each endpoint
            i_alt1 = np.min(i_dists)
            i_alt_idx = np.argmin(i_dists)
            i_dists[i_alt_idx] = np.inf
            i_alt2 = np.min(i_dists)
            
            j_alt1 = np.min(j_dists)
            j_alt_idx = np.argmin(j_dists)
            j_dists[j_alt_idx] = np.inf
            j_alt2 = np.min(j_dists)
            
            # Evaluate edge (i,j) in context of potential 3-node chains
            # Consider both possible orientations: i→j→k and l→i→j
            chain_cost1 = safe_dist[i, j] + min(i_alt1 + j_alt2, i_alt2 + j_alt1)
            chain_cost2 = safe_dist[i, j] + min(i_alt1, j_alt1)
            
            # Base chain quality relative to individual alternatives
            alt_quality = (i_alt1 + j_alt1) / 2
            chain_quality = min(chain_cost1, chain_cost2) / 2
            
            # Score favors edges that form efficient chains
            if chain_quality < alt_quality * 1.2:
                lookahead_score[i, j] = alt_quality / (chain_quality + eps)
            else:
                lookahead_score[i, j] = 1.0 / (1.0 + (chain_quality - alt_quality) / mean_dist)
    
    lookahead_score = np.maximum(lookahead_score, lookahead_score.T)
    
    # 3. Adaptive Hybrid Regret
    regret_matrix = np.zeros_like(distance_matrix)
    
    # Precompute rank-based and absolute regret components
    for i in range(n):
        dists = safe_dist[i]
        sorted_dists = np.sort(dists)
        sorted_indices = np.argsort(dists)
        
        if len(sorted_dists) >= 3:
            # Absolute regret component
            best_dist = sorted_dists[0]
            alt_dist = sorted_dists[min(3, len(sorted_dists)-1)]  # 3rd best
            
            # Rank-based regret component
            ranks = np.zeros(n)
            for rank, idx in enumerate(sorted_indices):
                if i != idx:
                    ranks[idx] = rank / (len(sorted_indices) - 1)
            
            # Hybrid regret with adaptive weighting
            for j in range(n):
                if i != j:
                    abs_regret = (dists[j] - best_dist) / (alt_dist + eps)
                    rank_regret = ranks[j]
                    
                    # Adaptive weight based on distance distribution
                    dist_std = np.std(sorted_dists[:5])
                    weight = 0.4 + 0.3 * np.tanh(1.0 - dist_std / mean_dist)
                    weight = np.clip(weight, 0.3, 0.7)
                    
                    hybrid_regret = weight * abs_regret + (1 - weight) * rank_regret
                    regret_matrix[i, j] = np.tanh(hybrid_regret)  # Smooth to [-1, 1]
    
    # 4. Neighborhood Structural Density
    density_bonus = np.zeros_like(distance_matrix)
    
    # Precompute local neighborhood statistics
    k_density = min(6, n-1)
    neighborhood_radii = np.zeros(n)
    neighborhood_shapes = np.zeros(n)
    
    for i in range(n):
        # Get k-nearest distances
        k_nearest = np.partition(safe_dist[i], k_density)[:k_density+1]
        valid_dists = k_nearest[k_nearest < np.inf]
        neighborhood_radii[i] = np.mean(valid_dists)
        
        # Measure neighborhood shape (cluster vs spread)
        if len(valid_dists) > 2:
            shape_metric = np.std(valid_dists) / (np.mean(valid_dists) + eps)
            neighborhood_shapes[i] = np.tanh(shape_metric * 2)  # [-1, 1]
    
    # Compute structural density for candidate edges
    for i in range(n):
        for j in range(i+1, n):
            if not candidate_mask[i, j]:
                continue
            
            # Combined neighborhood analysis
            radius_ij = np.sqrt(neighborhood_radii[i] * neighborhood_radii[j])
            
            # Count nodes in overlapping neighborhood
            sigma = radius_ij / 2
            kernel_i = np.exp(-0.5 * (safe_dist[i] / sigma) ** 2)
            kernel_j = np.exp(-0.5 * (safe_dist[j] / sigma) ** 2)
            overlap = kernel_i * kernel_j
            
            # Structural bonus based on neighborhood compatibility
            shape_compat = 1.0 - 0.3 * abs(neighborhood_shapes[i] - neighborhood_shapes[j])
            density_val = np.sum(overlap) - 2  # Exclude endpoints
            
            if density_val > 0:
                # Bonus increases with overlap and shape compatibility
                density_bonus[i, j] = shape_compat * np.log1p(density_val) * 0.18
            else:
                density_bonus[i, j] = -0.1 * shape_compat
    
    density_bonus = np.maximum(density_bonus, density_bonus.T)
    
    # 5. Base inverse distance with adaptive scaling
    inv_dist = 1.0 / safe_dist
    global_mean = np.mean(safe_dist[safe_dist < np.inf])
    inv_dist = inv_dist * global_mean
    
    # 6. Dynamic rank weighting with reinforcement-inspired decay
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        sorted_idx = np.argsort(safe_dist[i])
        distances = safe_dist[i, sorted_idx[1:min(8, n-1)]]
        
        if len(distances) > 2:
            # Adaptive decay based on distance progression
            first_quartile = np.percentile(distances, 25)
            third_quartile = np.percentile(distances, 75)
            progression = (third_quartile - first_quartile) / (first_quartile + eps)
            
            decay_base = 0.35 + 0.3 * np.tanh(2.0 - 4.0 * progression)
            decay_base = np.clip(decay_base, 0.25, 0.75)
        else:
            decay_base = 0.5
        
        # Assign weights with modified exponential decay
        for rank, j in enumerate(sorted_idx):
            if i != j:
                # Slightly flatten decay curve for top ranks
                adjusted_rank = rank ** 0.8
                rank_weights[i, j] = decay_base ** adjusted_rank
    
    # 7. Combine components with reinforcement-guided weighting
    # Base score component
    base_score = inv_dist * rank_weights
    
    # Apply regret penalty (transformed to [0.2, 1.8] range)
    regret_penalty = 1.0 + 0.8 * (1.0 - regret_matrix) / 2.0
    
    # Combine all factors with lookahead as primary modulator
    heuristics_matrix = np.zeros_like(distance_matrix)
    heuristics_matrix = base_score * regret_penalty * (1.0 + density_bonus) * lookahead_score
    
    # 8. Reinforcement-guided sparsification with connectivity assurance
    # Apply candidate mask
    heuristics_matrix[~candidate_mask] = 0
    
    # Minimum Spanning Tree bias (for medium-sized instances)
    if n <= 1500:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_edges = (mst + mst.T) > 0
            # Boost MST edges moderately
            heuristics_matrix[mst_edges] *= 1.2
        except:
            pass
    
    # Ensure each node has sufficient connections
    min_connections = max(2, int(np.log2(n)))
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero_indices = np.where(row > 0)[0]
        
        if len(nonzero_indices) < min_connections:
            # Find additional promising connections
            all_scores = base_score[i] * lookahead_score[i]
            all_scores[i] = -np.inf
            top_indices = np.argsort(all_scores)[-min_connections:]
            
            # Boost selected connections
            current_max = np.max(row) if len(nonzero_indices) > 0 else 0.1
            boost_value = max(current_max * 1.2, 0.15)
            heuristics_matrix[i, top_indices] = boost_value
            heuristics_matrix[top_indices, i] = boost_value
    
    # 9. Symmetrize and normalize
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize using robust statistics
    non_zero = heuristics_matrix[heuristics_matrix > 0]
    if len(non_zero) > 0:
        p95 = np.percentile(non_zero, 95)
        scale = max(np.max(heuristics_matrix), p95 * 1.3)
        heuristics_matrix = heuristics_matrix / scale
        heuristics_matrix = np.clip(heuristics_matrix, 0, 1)
    
    return heuristics_matrix
