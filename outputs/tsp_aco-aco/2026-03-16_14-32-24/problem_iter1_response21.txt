import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection combining multiple factors.
    Returns a sparse matrix of edge priorities where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance component (like v1 but with smoothing)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / (distance_matrix + 1e-12)  # Add small epsilon to avoid division by zero
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree bonus (edges in MST are more promising)
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = (mst + mst.T > 0).astype(float) * 2.0
    
    # 3. Nearest neighbor component (edges to/from nearest neighbors)
    nn_factor = np.zeros_like(distance_matrix)
    k_nearest = min(5, n - 1)  # Consider k nearest neighbors
    
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])[1:k_nearest+1]  # Exclude self
        nn_factor[i, sorted_indices] = 1.5
        nn_factor[sorted_indices, i] = np.maximum(nn_factor[sorted_indices, i], 1.5)
    
    # 4. Edge centrality penalty (penalize edges that are too central/average)
    mean_dist = np.mean(distance_matrix[np.triu_indices(n, k=1)])
    std_dist = np.std(distance_matrix[np.triu_indices(n, k=1)])
    
    # Penalty for edges with distance close to mean (less distinctive)
    z_scores = np.abs((distance_matrix - mean_dist) / (std_dist + 1e-12))
    centrality_penalty = 1.0 / (1.0 + np.exp(-(z_scores - 1.0)))  # Sigmoid centered at 1 std
    
    # 5. Combine components with weights
    heuristics = (
        0.5 * inv_dist / np.max(inv_dist) +      # Normalized inverse distance
        0.3 * mst_bonus +                        # MST bonus
        0.2 * nn_factor -                        # Nearest neighbor factor
        0.1 * centrality_penalty                 # Centrality penalty
    )
    
    # 6. Sparsification: keep only top promising edges
    # For each node, keep connections to top promising neighbors
    sparsity_factor = 0.4  # Keep ~40% of edges per node
    k_keep = max(2, int(sparsity_factor * (n - 1)))
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Exclude diagonal from consideration
        threshold = np.sort(row)[-k_keep]  # k_keep-th largest value
        heuristics[i] = np.where(heuristics[i] >= threshold, heuristics[i], 0)
    
    # Ensure symmetry (since TSP is symmetric)
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # Normalize final matrix
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
