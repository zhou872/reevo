import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with refined regret modeling, adaptive geometric
    ranking, and geometric mean density bonus for better edge prioritization.
    
    Key improvements over v1:
    1. More nuanced regret modeling with distance-ratio weighting
    2. Direct geometric rank decay without std-based adaptation
    3. Simplified density bonus using geometric mean of node densities
    4. Optimized sparsification with adaptive candidate count
    """
    n = distance_matrix.shape[0]
    
    # Safe distance matrix handling
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # 1. Adaptive candidate list construction
    # Use adaptive candidate count based on problem size
    base_k = max(3, int(np.sqrt(n)))
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        neighbors = np.argsort(safe_dist[i])[:base_k]
        candidate_mask[i, neighbors] = True
    
    # Ensure symmetry in candidate selection
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # 2. Enhanced Regret-Aware Scoring
    regret_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        dists = safe_dist[i].copy()
        valid_dists = dists[dists < np.inf]
        
        if len(valid_dists) >= 2:
            best_dist = np.min(valid_dists)
            second_best = np.partition(valid_dists, 1)[1] if len(valid_dists) > 1 else best_dist
            
            for j in range(n):
                if i != j and dists[j] < np.inf:
                    edge_dist = dists[j]
                    
                    # Ratio-based regret with smoothing
                    if edge_dist <= second_best + eps:
                        # Minimal penalty for competitive edges
                        regret = 0.05 * (edge_dist - best_dist) / (second_best + eps)
                    else:
                        # Progressive penalty based on distance ratio
                        regret = 0.3 + 0.7 * (edge_dist - second_best) / (np.max(valid_dists) + eps)
                    
                    # Additional penalty for edges much worse than local alternatives
                    local_ratio = edge_dist / (best_dist + eps)
                    if local_ratio > 2.0:
                        regret *= (1.0 + 0.5 * (local_ratio - 2.0))
                    
                    regret_matrix[i, j] = min(regret, 1.0)
    
    # 3. Geometric Rank-Weighted Scoring
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        sorted_idx = np.argsort(safe_dist[i])
        
        # Fixed geometric decay rate for ranking
        decay_rate = 0.65  # Slightly more aggressive than v1
        
        for rank, j in enumerate(sorted_idx):
            if i != j:
                rank_weights[i, j] = decay_rate ** rank
    
    # 4. Geometric Mean Density Bonus
    density_bonus = np.zeros_like(distance_matrix)
    
    # Compute node densities based on median distance threshold
    node_densities = np.zeros(n)
    for i in range(n):
        valid_dists = safe_dist[i][safe_dist[i] < np.inf]
        if len(valid_dists) > 0:
            median_dist = np.median(valid_dists)
            # Count nodes within median distance (excluding self)
            density = np.sum(valid_dists <= median_dist + eps)
            node_densities[i] = density
    
    # Normalize densities to [0, 1]
    max_density = np.max(node_densities)
    if max_density > 0:
        node_densities = node_densities / max_density
    
    # Apply geometric mean density bonus for candidate edges
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Geometric mean of normalized densities
                geo_mean = np.sqrt(node_densities[i] * node_densities[j] + eps)
                
                # Scale bonus based on geometric mean
                if geo_mean > 0.1:
                    # Stronger bonus for denser regions
                    density_bonus[i, j] = 0.25 * geo_mean
                else:
                    # Mild penalty for isolated regions
                    density_bonus[i, j] = -0.05
    
    # Make symmetric
    density_bonus = np.maximum(density_bonus, density_bonus.T)
    
    # 5. Base Inverse Distance with Normalization
    inv_dist = 1.0 / safe_dist
    # Normalize by median for better scaling
    median_dist = np.median(safe_dist[safe_dist < np.inf])
    inv_dist = inv_dist * median_dist
    
    # 6. MST Structural Guidance (lightweight)
    if n <= 800:  # Slightly lower threshold than v1 for efficiency
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_bias = (mst + mst.T) > 0
            mst_bonus = mst_bias * 0.25  # Reduced from v1
        except:
            mst_bonus = np.zeros_like(distance_matrix)
    else:
        mst_bonus = np.zeros_like(distance_matrix)
    
    # 7. Multi-Factor Combination
    # Base score combines inverse distance with rank weighting
    base_score = inv_dist * rank_weights
    
    # Apply regret penalty with exponential decay
    regret_penalty = np.exp(-1.8 * regret_matrix)  # Adjusted decay rate
    score_with_regret = base_score * regret_penalty
    
    # Incorporate density and MST bonuses multiplicatively
    combined_score = score_with_regret * (1.0 + density_bonus) * (1.0 + mst_bonus)
    
    # 8. Dynamic Sparsification with Connectivity Guarantee
    # Apply candidate mask
    heuristics_matrix = combined_score.copy()
    heuristics_matrix[~candidate_mask] = 0
    
    # Ensure minimum connectivity: each node must have at least 2 non-zero edges
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero_indices = np.nonzero(row)[0]
        
        if len(nonzero_indices) < 2:
            # Find top 2 edges by combined score (not just distance)
            top_indices = np.argsort(combined_score[i])[-2:]
            
            # Restore these edges with slight boost
            heuristics_matrix[i, top_indices] = combined_score[i, top_indices] * 1.15
            heuristics_matrix[top_indices, i] = combined_score[top_indices, i] * 1.15
    
    # 9. Final Symmetrization and Normalization
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1] range while preserving zero entries
    max_val = np.max(heuristics_matrix)
    if max_val > eps:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
