import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection using multiple factors:
    1. Inverse distance baseline
    2. Bidirectional proximity ranking (how close nodes are to each other's neighbor lists)
    3. Local neighborhood sparsification
    4. Geometric connectivity pattern bonus
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) / (distance_matrix + np.eye(n))
    
    # Avoid division by zero for diagonal
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # Base inverse distance (closer = better)
    inv_dist = 1.0 / safe_dist
    
    # Factor 1: Bidirectional proximity rank
    # For each node, sort other nodes by distance
    sorted_indices = np.argsort(safe_dist, axis=1)
    
    # Create rank matrix: rank[i,j] = position of j in i's sorted list (0 = closest)
    rank_matrix = np.zeros_like(distance_matrix, dtype=int)
    for i in range(n):
        rank_matrix[i, sorted_indices[i]] = np.arange(n)
    
    # Symmetric rank score: geometric mean of two directional ranks
    # Lower combined rank = more promising
    rank_sum = rank_matrix + rank_matrix.T
    rank_factor = 1.0 / (1.0 + 0.5 * rank_sum)  # +1 to avoid division by zero
    
    # Factor 2: Local density adjustment
    # Nodes in dense regions should connect to nearby nodes
    avg_distances = np.mean(safe_dist, axis=1)
    density_factor = np.outer(avg_distances, avg_distances)
    density_factor = 1.0 / np.sqrt(density_factor + 1e-10)
    density_factor = density_factor / np.max(density_factor)
    
    # Factor 3: Angle-based connectivity pattern
    # Compute angular distribution for potential triangle completion
    if n >= 3:
        angle_bonus = np.zeros_like(distance_matrix)
        for i in range(n):
            for j in range(i+1, n):
                # Find third node that forms the best triangle
                if n <= 100:  # Computationally feasible
                    other_dists = safe_dist[i] + safe_dist[j]
                    other_dists[i] = other_dists[j] = np.inf
                    best_third = np.argmin(other_dists)
                    # Bonus if edge completes a tight triangle
                    triangle_perim = safe_dist[i, j] + safe_dist[i, best_third] + safe_dist[j, best_third]
                    angle_bonus[i, j] = angle_bonus[j, i] = 1.0 / (1.0 + triangle_perim / np.mean(safe_dist))
                else:
                    angle_bonus[i, j] = angle_bonus[j, i] = 1.0
    else:
        angle_bonus = np.ones_like(distance_matrix)
    
    # Combine factors
    heuristics_matrix = (inv_dist ** 0.4) * (rank_factor ** 0.3) * \
                       (density_factor ** 0.2) * (angle_bonus ** 0.1)
    
    # Sparsification: keep only promising edges
    # Keep top k edges per node based on heuristics score
    k = max(2, int(np.sqrt(n)))  # Adaptive sparsity
    
    sparsified = np.zeros_like(heuristics_matrix)
    for i in range(n):
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Exclude self
        top_k_indices = np.argpartition(row, -k)[-k:]
        sparsified[i, top_k_indices] = heuristics_matrix[i, top_k_indices]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Normalize and set diagonal to zero
    np.fill_diagonal(sparsified, 0.0)
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
