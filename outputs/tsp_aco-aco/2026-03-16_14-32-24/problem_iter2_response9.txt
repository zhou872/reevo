import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial import ConvexHull
from sklearn.cluster import KMeans

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic for TSP edge selection with multi-step lookahead,
    dynamic weighting, and cluster-aware scoring.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between nodes
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, with unpromising edges set to zero
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        heuristics = np.ones_like(distance_matrix) / (distance_matrix + 1e-10)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # Create a copy with inf on diagonal
    D = distance_matrix.copy()
    np.fill_diagonal(D, np.inf)
    
    # 1. Multi-step lookahead potential (up to 3-step paths)
    lookahead_scores = np.zeros((n, n))
    
    # Consider 2-step and 3-step paths
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            # 2-step paths: i -> k -> j
            k_mask = np.ones(n, dtype=bool)
            k_mask[[i, j]] = False
            savings_2step = D[i, :] + D[:, j] - D[i, j]
            best_2step = np.max(savings_2steps[k_mask]) if np.any(k_mask) else 0
            
            # 3-step paths: i -> k1 -> k2 -> j (only for n >= 5)
            if n >= 5:
                # Consider top 10% of intermediate nodes for efficiency
                candidate_nodes = np.argsort(D[i, :] + D[:, j])[:max(3, n//10)]
                candidate_nodes = [k for k in candidate_nodes if k not in [i, j]]
                
                if len(candidate_nodes) >= 2:
                    savings_3step = []
                    for k1 in candidate_nodes:
                        for k2 in candidate_nodes:
                            if k1 != k2:
                                savings = (D[i, k1] + D[k1, k2] + D[k2, j] - D[i, j])
                                savings_3step.append(savings)
                    best_3step = np.max(savings_3step) if savings_3step else 0
                else:
                    best_3step = 0
            else:
                best_3step = 0
            
            # Combine multi-step potentials with decay factor
            lookahead_scores[i, j] = 0.7 * best_2step + 0.3 * best_3step
    
    # Normalize lookahead scores
    lookahead_max = np.max(lookahead_scores)
    if lookahead_max > 0:
        lookahead_scores = lookahead_scores / lookahead_max
    
    # 2. Dynamic weight optimization based on problem characteristics
    # Compute problem statistics for weight adjustment
    dist_flat = D[D < np.inf]
    dist_mean = np.mean(dist_flat)
    dist_std = np.std(dist_flat)
    density = 1.0 / (dist_mean + 1e-10)
    
    # Adjust weights based on problem characteristics
    if dist_std / dist_mean > 1.0:  # High variance
        w_inv_dist = 0.4
        w_lookahead = 0.4
        w_mst = 0.1
        w_cluster = 0.1
    elif density > 0.1:  # Dense problem
        w_inv_dist = 0.3
        w_lookahead = 0.3
        w_mst = 0.2
        w_cluster = 0.2
    else:  # Sparse or balanced problem
        w_inv_dist = 0.5
        w_lookahead = 0.3
        w_mst = 0.1
        w_cluster = 0.1
    
    # 3. Base inverse distance weighting
    inv_dist = 1 / (D + 1e-10)
    inv_dist_max = np.max(inv_dist)
    if inv_dist_max > 0:
        inv_dist = inv_dist / inv_dist_max
    
    # 4. Minimum Spanning Tree bonus
    try:
        mst = minimum_spanning_tree(D).toarray()
        mst_bonus = ((mst + mst.T) > 0).astype(float)
    except:
        mst_bonus = np.zeros((n, n))
    
    # 5. Cluster-aware scoring
    cluster_scores = np.ones((n, n))
    
    # Estimate coordinates from distance matrix using MDS approximation
    if n >= 4:
        try:
            # Simple 2D embedding based on distance to two reference points
            ref1 = np.argmin(np.sum(D, axis=1))
            ref2 = np.argmax(D[ref1, :])
            
            # Use distances to reference points as pseudo-coordinates
            pseudo_coords = np.column_stack((D[ref1, :], D[ref2, :]))
            
            # Cluster nodes into k = sqrt(n) groups
            k = max(2, int(np.sqrt(n)))
            kmeans = KMeans(n_clusters=k, random_state=42, n_init=10)
            labels = kmeans.fit_predict(pseudo_coords)
            
            # Bonus for intra-cluster edges, penalty for inter-cluster
            for i in range(n):
                for j in range(n):
                    if i != j:
                        if labels[i] == labels[j]:
                            # Same cluster: bonus based on cluster density
                            cluster_members = np.where(labels == labels[i])[0]
                            cluster_density = len(cluster_members) / n
                            cluster_scores[i, j] = 1.0 + 0.5 * cluster_density
                        else:
                            # Different clusters: penalty based on distance between cluster centers
                            center_i = kmeans.cluster_centers_[labels[i]]
                            center_j = kmeans.cluster_centers_[labels[j]]
                            cluster_dist = np.linalg.norm(center_i - center_j)
                            cluster_scores[i, j] = 0.5 + 0.3 * np.exp(-cluster_dist / 10.0)
        except:
            # Fallback if clustering fails
            cluster_scores = np.ones((n, n))
    
    # Normalize cluster scores
    cluster_max = np.max(cluster_scores)
    if cluster_max > 0:
        cluster_scores = cluster_scores / cluster_max
    
    # 6. Combine all factors with dynamic weights
    heuristics = (
        w_inv_dist * inv_dist +
        w_lookahead * lookahead_scores +
        w_mst * mst_bonus +
        w_cluster * cluster_scores
    )
    
    # 7. Node centrality-based sparsification
    # Compute node centrality (average distance to other nodes)
    centrality = np.zeros(n)
    for i in range(n):
        valid_dists = D[i, D[i, :] < np.inf]
        centrality[i] = 1.0 / (np.mean(valid_dists) + 1e-10)
    
    # Normalize centrality
    centrality = centrality / (np.max(centrality) + 1e-10)
    
    # Adaptive sparsification: keep more edges for central nodes
    mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        # Number of edges to keep depends on centrality
        k = max(3, int(5 * centrality[i] + 0.5))
        k = min(k, n - 1)
        
        # Get top k edges for this node
        top_k_indices = np.argpartition(heuristics[i], -k)[-k:]
        mask[i, top_k_indices] = True
    
    # Ensure symmetry
    mask = np.logical_or(mask, mask.T)
    heuristics = heuristics * mask
    
    # 8. Final processing
    np.fill_diagonal(heuristics, 0)
    
    # Ensure non-negativity and normalize
    heuristics = np.maximum(heuristics, 0)
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
