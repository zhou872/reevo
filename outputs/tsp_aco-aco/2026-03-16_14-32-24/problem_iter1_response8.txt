import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection that combines multiple factors
    and returns a sparsified priority matrix.
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-10  # Avoid division by zero
    
    # 1. Base inverse distance weighting (like v1)
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree (MST) inclusion bonus
    mst_matrix = _compute_mst_edges(distance_matrix)
    
    # 3. Mutual k-nearest neighbors bonus
    k = max(2, min(5, n // 4))  # Adaptive k based on problem size
    knn_matrix = _compute_mutual_knn(distance_matrix, k)
    
    # 4. Combine factors with weighted blending
    weights = {
        'inv_dist': 1.0,
        'mst': 3.0,      # Strong bonus for MST edges
        'knn': 2.0       # Moderate bonus for mutual nearest neighbors
    }
    
    priority_matrix = (
        weights['inv_dist'] * inv_dist +
        weights['mst'] * mst_matrix +
        weights['knn'] * knn_matrix
    )
    
    # 5. Sparsification: keep only top-m edges per node
    m = max(3, min(8, n // 3))  # Adaptive m
    sparsified = _sparsify_by_top_m(priority_matrix, m)
    
    return sparsified

def _compute_mst_edges(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute Minimum Spanning Tree edges using Prim's algorithm."""
    n = distance_matrix.shape[0]
    mst_matrix = np.zeros_like(distance_matrix)
    
    if n <= 1:
        return mst_matrix
    
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    
    for _ in range(n - 1):
        min_edge = np.inf
        u, v = -1, -1
        
        # Find minimum edge between visited and unvisited nodes
        for i in range(n):
            if visited[i]:
                for j in range(n):
                    if not visited[j] and distance_matrix[i, j] < min_edge:
                        min_edge = distance_matrix[i, j]
                        u, v = i, j
        
        if u != -1 and v != -1:
            visited[v] = True
            mst_matrix[u, v] = 1
            mst_matrix[v, u] = 1
    
    return mst_matrix

def _compute_mutual_knn(distance_matrix: np.ndarray, k: int) -> np.ndarray:
    """Compute mutual k-nearest neighbors matrix."""
    n = distance_matrix.shape[0]
    knn_matrix = np.zeros_like(distance_matrix)
    
    if k >= n:
        k = n - 1
    
    # Get k-nearest neighbors for each node (excluding self)
    for i in range(n):
        # Get indices of k smallest distances (excluding diagonal)
        indices = np.argsort(distance_matrix[i])[1:k+1]
        knn_matrix[i, indices] = 1
    
    # Keep only mutual edges (both directions)
    mutual_knn = np.minimum(knn_matrix, knn_matrix.T)
    
    return mutual_knn

def _sparsify_by_top_m(priority_matrix: np.ndarray, m: int) -> np.ndarray:
    """Keep only top-m edges per node, zero out the rest."""
    n = priority_matrix.shape[0]
    sparsified = priority_matrix.copy()
    
    # Zero out diagonal to avoid self-loops
    np.fill_diagonal(sparsified, -np.inf)
    
    for i in range(n):
        row = sparsified[i].copy()
        
        # Find indices of m largest values
        if m < n - 1:
            threshold_idx = np.argsort(row)[-m]
            threshold = row[threshold_idx]
            
            # Zero out values below threshold
            mask = row < threshold
            sparsified[i, mask] = 0
        
        # Ensure symmetry
        for j in range(n):
            if i != j and (sparsified[i, j] == 0 or sparsified[j, i] == 0):
                sparsified[i, j] = 0
                sparsified[j, i] = 0
    
    # Restore diagonal to zero
    np.fill_diagonal(sparsified, 0)
    
    return sparsified
