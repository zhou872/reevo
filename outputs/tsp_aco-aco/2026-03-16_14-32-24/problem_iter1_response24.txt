import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.cluster import KMeans

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    
    Features:
    1. Candidate set sparsification based on k-nearest neighbors
    2. Minimum spanning tree edge bonus
    3. Normalized inverse distance weighting
    4. Inter-cluster connectivity scoring
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Avoid division by zero and self-connections
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Normalized inverse distance (baseline)
    inv_dist = 1.0 / safe_dist
    row_sums = inv_dist.sum(axis=1, keepdims=True)
    norm_inv = inv_dist / (row_sums + 1e-12)
    
    # 2. Candidate set sparsification (k-nearest neighbors)
    k = min(max(5, n // 5), n - 1)  # Adaptive k
    heuristics = np.zeros_like(distance_matrix)
    
    # For each node, get k-nearest neighbors
    for i in range(n):
        indices = np.argsort(safe_dist[i])[:k]
        heuristics[i, indices] = norm_inv[i, indices]
    
    # Make symmetric
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 3. Minimum spanning tree bonus
    try:
        mst = minimum_spanning_tree(safe_dist).toarray()
        mst_mask = (mst > 0) | (mst.T > 0)
        heuristics[mst_mask] *= 2.0  # Double importance for MST edges
    except:
        pass  # Fallback if MST fails
    
    # 4. Cluster connectivity scoring
    if n >= 4:
        n_clusters = min(max(2, n // 10), 10)
        try:
            kmeans = KMeans(n_clusters=n_clusters, n_init=10, random_state=42)
            labels = kmeans.fit_predict(distance_matrix)
            
            # Bonus for intra-cluster edges
            for cluster in range(n_clusters):
                mask = (labels == cluster)
                cluster_indices = np.where(mask)[0]
                if len(cluster_indices) > 1:
                    heuristics[np.ix_(cluster_indices, cluster_indices)] *= 1.5
            
            # Penalty for very long inter-cluster edges
            cluster_centers = kmeans.cluster_centers_
            inter_cluster_dist = np.zeros((n_clusters, n_clusters))
            for i in range(n_clusters):
                for j in range(i+1, n_clusters):
                    inter_cluster_dist[i,j] = inter_cluster_dist[j,i] = np.linalg.norm(
                        cluster_centers[i] - cluster_centers[j]
                    )
            
            max_inter = inter_cluster_dist.max() if inter_cluster_dist.max() > 0 else 1
            for i in range(n):
                for j in range(i+1, n):
                    if labels[i] != labels[j]:
                        cluster_dist = inter_cluster_dist[labels[i], labels[j]]
                        penalty = 1.0 - (cluster_dist / max_inter) * 0.5
                        heuristics[i,j] = heuristics[j,i] *= max(0.2, penalty)
        except:
            pass  # Fallback if clustering fails
    
    # Final sparsification: zero out bottom 30% of edges for each node
    threshold = np.percentile(heuristics[heuristics > 0], 30) if np.any(heuristics > 0) else 0
    heuristics[heuristics < threshold] = 0
    
    # Ensure diagonal is zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
