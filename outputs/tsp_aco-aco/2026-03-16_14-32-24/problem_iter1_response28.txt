import numpy as np
from scipy.spatial import KDTree
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection.
    
    Combines multiple geometric and graph-theoretic insights:
    1. K-nearest neighbor sparsification to focus on locally promising edges
    2. Two-hop path analysis to identify edges that complete efficient triangles
    3. Local density weighting to favor edges in sparse regions
    4. Minimum spanning tree alignment to emphasize backbone connections
    """
    n = distance_matrix.shape[0]
    if n <= 3:
        return 1.0 / (distance_matrix + 1e-12)
    
    # Avoid division by zero and self-connections
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. K-nearest neighbor mask (keep only top k connections per node)
    k = min(5, n - 1)
    mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        indices = np.argsort(safe_dist[i])[:k]
        mask[i, indices] = True
        mask[indices, i] = True  # Symmetry
    
    # 2. Two-hop path potential via triangle completion
    triangle_potential = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            if not mask[i, j]:
                continue
            # Find best intermediate node k to form i->k->j triangle
            intermediate_costs = safe_dist[i, :] + safe_dist[:, j]
            intermediate_costs[[i, j]] = np.inf  # Exclude endpoints
            best_k = np.argmin(intermediate_costs)
            triangle_score = safe_dist[i, j] / (safe_dist[i, best_k] + safe_dist[best_k, j] + 1e-12)
            triangle_potential[i, j] = triangle_potential[j, i] = 1.0 / (1.0 + triangle_score)
    
    # 3. Local density weighting (inverse distance weighted by neighborhood sparsity)
    tree = KDTree(np.arange(n).reshape(-1, 1))  # Dummy coordinates for connectivity
    densities = np.zeros(n)
    for i in range(n):
        # Average distance to k nearest neighbors as density measure
        neighbor_dists = np.sort(safe_dist[i])[:k]
        densities[i] = np.mean(neighbor_dists)
    
    density_weight = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            if not mask[i, j]:
                continue
            # Prefer edges connecting sparse regions (high average neighbor distance)
            density_bonus = (densities[i] + densities[j]) / (2 * np.mean(densities) + 1e-12)
            density_weight[i, j] = density_weight[j, i] = density_bonus
    
    # 4. Minimum spanning tree alignment
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_bonus = np.zeros_like(distance_matrix)
    mst_bonus[mst > 0] = 2.0  # Strong bonus for MST edges
    mst_bonus += mst_bonus.T
    np.fill_diagonal(mst_bonus, 0)
    
    # Combine all factors
    inverse_dist = 1.0 / (safe_dist + 1e-12)
    
    heuristic = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            if not mask[i, j]:
                continue
            base_score = inverse_dist[i, j]
            triangle_factor = 0.7 + 0.3 * triangle_potential[i, j]
            density_factor = 0.5 + 0.5 * density_weight[i, j]
            mst_factor = 1.0 + 0.5 * mst_bonus[i, j]
            
            heuristic[i, j] = heuristic[j, i] = (
                base_score * triangle_factor * density_factor * mst_factor
            )
    
    # Normalize to similar scale as original inverse distances
    if np.max(heuristic) > 0:
        heuristic = heuristic / np.max(heuristic) * np.max(inverse_dist[mask])
    
    np.fill_diagonal(heuristic, 0)
    return heuristic
