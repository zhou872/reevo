import numpy as np
from scipy.spatial import ConvexHull

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    Returns a matrix where higher values indicate more promising edges.
    Unpromising edges are set to zero to sparsify the matrix.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (like v1)
    with np.errstate(divide='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Node degree penalty: penalize edges from high-degree candidate nodes
    # Nodes with many short edges might be overconnected
    closeness = np.sum(inv_dist, axis=1)
    degree_penalty = 1 / (1 + np.outer(closeness, closeness) / (n ** 2))
    
    # 3. Local density bonus: edges between nodes in sparse regions
    # Compute average distance to k nearest neighbors
    k = min(5, n - 1)
    sorted_indices = np.argsort(distance_matrix, axis=1)[:, 1:k+1]
    avg_nn_dist = np.mean(distance_matrix[np.arange(n)[:, None], sorted_indices], axis=1)
    density_bonus = np.outer(avg_nn_dist, avg_nn_dist)
    density_bonus = density_bonus / np.max(density_bonus) if np.max(density_bonus) > 0 else 1
    
    # 4. Minimum Spanning Tree (MST) inspired sparsification
    # Prim's algorithm to find MST edges
    mst_mask = np.zeros((n, n), dtype=bool)
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    
    for _ in range(n - 1):
        # Find minimum edge between visited and unvisited
        visited_mask = visited[:, None] & ~visited[None, :]
        if not np.any(visited_mask):
            break
            
        # Create mask for valid edges
        valid_mask = visited_mask & (distance_matrix > 0)
        if not np.any(valid_mask):
            break
            
        # Find minimum distance among valid edges
        min_idx = np.unravel_index(
            np.argmin(np.where(valid_mask, distance_matrix, np.inf)),
            distance_matrix.shape
        )
        mst_mask[min_idx] = True
        mst_mask[min_idx[1], min_idx[0]] = True
        visited[min_idx[1]] = True
    
    # 5. Convex hull bonus: edges likely to be on the outer boundary
    hull_bonus = np.ones((n, n))
    if n >= 3:
        try:
            # Use 2D projection for convex hull (assume nodes are in 2D+ space)
            # For high-dimensional data, use PCA or random projection
            points = np.random.randn(n, 2)  # Placeholder - in practice use actual coordinates
            if hasattr(distance_matrix, 'shape'):
                # If we had coordinates, we would use them here
                # For now, create synthetic coordinates from distance matrix using MDS-like approach
                from sklearn.manifold import MDS
                mds = MDS(n_components=2, dissimilarity='precomputed', random_state=42)
                points = mds.fit_transform(distance_matrix)
            
            hull = ConvexHull(points)
            hull_vertices = set(hull.vertices)
            
            # Bonus for edges between hull vertices
            for i in range(n):
                for j in range(i + 1, n):
                    if i in hull_vertices and j in hull_vertices:
                        # Check if they're adjacent on the hull
                        for simplex in hull.simplices:
                            if (i in simplex and j in simplex):
                                hull_bonus[i, j] = 2.0
                                hull_bonus[j, i] = 2.0
                                break
        except:
            # If convex hull fails, use uniform bonus
            pass
    
    # 6. Combine all factors
    heuristics_matrix = inv_dist * degree_penalty * density_bonus * hull_bonus
    
    # 7. Boost MST edges
    heuristics_matrix[mst_mask] *= 3.0
    
    # 8. Sparsify: keep only top promising edges
    # Zero out diagonal
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Keep edges that are in top percentile or are MST edges
    if n > 2:
        # Calculate threshold
        flat_heuristics = heuristics_matrix.flatten()
        flat_heuristics = flat_heuristics[flat_heuristics > 0]
        
        if len(flat_heuristics) > 0:
            # Keep top 30% of non-zero edges plus all MST edges
            threshold = np.percentile(flat_heuristics, 70)
            mask = (heuristics_matrix < threshold) & ~mst_mask
            heuristics_matrix[mask] = 0
    
    # 9. Symmetrize (ensure matrix is symmetric for undirected TSP)
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    
    return heuristics_matrix
