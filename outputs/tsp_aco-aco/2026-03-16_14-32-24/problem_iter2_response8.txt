import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with dynamic parameter adaptation, lookahead mechanisms,
    and insertion-based hybridization for improved global coherence.
    
    Features:
    1. Dynamic parameter adaptation based on problem scale and structure
    2. Two-step lookahead for path coherence assessment
    3. Insertion heuristic hybridization for local refinement
    4. Multi-layer feature fusion with adaptive weighting
    """
    n = distance_matrix.shape[0]
    if n < 3:
        # Trivial cases
        heuristics_matrix = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics_matrix, 0)
        return heuristics_matrix
    
    # Avoid division by zero and self-connections
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # ========== DYNAMIC PARAMETER ADAPTATION ==========
    # Adapt parameters based on problem size and structure
    scale_factor = np.log10(n + 1) / np.log10(50)  # Normalized scale factor
    
    # Dynamic decay factor: smaller for larger problems to focus on top candidates
    decay_factor = 0.5 + 0.3 * np.exp(-scale_factor / 2)
    
    # Candidate list size: increases sublinearly with problem size
    base_k = max(3, int(np.sqrt(n)))
    adaptive_k = min(base_k + int(scale_factor * 2), n - 1)
    
    # ========== CORE HEURISTIC COMPONENTS ==========
    
    # 1. Base inverse distance with dynamic rank weighting
    inverse_dist = 1.0 / safe_dist
    np.fill_diagonal(inverse_dist, 0)
    
    rank_weight = np.zeros_like(distance_matrix)
    for i in range(n):
        sorted_idx = np.argsort(safe_dist[i])
        for rank, j in enumerate(sorted_idx):
            if i != j:
                rank_weight[i, j] = decay_factor ** rank
    
    # 2. Enhanced regret with connectivity preservation
    regret_penalty = np.zeros_like(distance_matrix)
    node_alternatives = np.zeros(n)
    
    for i in range(n):
        dists = safe_dist[i].copy()
        sorted_dists = np.sort(dists)
        
        if len(sorted_dists) >= 3:
            best = sorted_dists[0]
            second_best = sorted_dists[1]
            third_best = sorted_dists[2]
            
            # Alternative quality metric
            node_alternatives[i] = (second_best + third_best) / (2 * best + 1e-10)
            
            for j in range(n):
                if i != j:
                    if np.isclose(dists[j], best):
                        regret = 0
                    else:
                        regret = (dists[j] - second_best) / (third_best - second_best + 1e-10)
                    
                    # Scale regret by node's alternative quality
                    regret_penalty[i, j] = regret * (1 + node_alternatives[i])
    
    # 3. Two-step lookahead path coherence
    lookahead_score = np.zeros_like(distance_matrix)
    
    # Pre-compute nearest neighbors for efficiency
    k_lookahead = min(5, n - 1)
    nearest_neighbors = np.argsort(safe_dist, axis=1)[:, 1:k_lookahead+1]
    
    for i in range(n):
        for j in range(n):
            if i != j:
                # First step: edge (i,j)
                step1_dist = safe_dist[i, j]
                
                # Second step: best continuation from j (excluding i)
                mask = np.ones(k_lookahead, dtype=bool)
                mask[np.where(nearest_neighbors[j] == i)[0]] = False if i in nearest_neighbors[j] else True
                valid_next = nearest_neighbors[j][mask] if mask.any() else []
                
                if len(valid_next) > 0:
                    # Find best continuation from j
                    best_next_dist = np.min(safe_dist[j, valid_next])
                    
                    # Coherence score: favors edges that lead to good continuations
                    lookahead_ratio = (step1_dist + best_next_dist) / (2 * np.median(safe_dist[i]) + 1e-10)
                    lookahead_score[i, j] = np.exp(-lookahead_ratio)
                else:
                    lookahead_score[i, j] = 0.5
    
    # 4. Insertion heuristic hybridization
    insertion_bonus = np.zeros_like(distance_matrix)
    
    # Run fast nearest insertion heuristic
    def nearest_insertion_heuristic(dist_matrix: np.ndarray) -> np.ndarray:
        n_nodes = dist_matrix.shape[0]
        if n_nodes <= 2:
            return np.ones((n_nodes, n_nodes), dtype=bool)
        
        # Start with a small subtour
        remaining = set(range(n_nodes))
        tour = [list(remaining)[0]]
        remaining.remove(tour[0])
        
        # Add nearest node to create initial 2-node tour
        nearest = np.argmin(dist_matrix[tour[0]])
        tour.append(nearest)
        remaining.remove(nearest)
        
        # Insert remaining nodes using nearest insertion
        while remaining:
            best_node = None
            best_position = -1
            best_increase = np.inf
            
            # Find node with minimal insertion cost
            for node in remaining:
                for pos in range(len(tour)):
                    # Calculate insertion cost
                    i = tour[pos]
                    j = tour[(pos + 1) % len(tour)]
                    increase = dist_matrix[i, node] + dist_matrix[node, j] - dist_matrix[i, j]
                    
                    if increase < best_increase:
                        best_increase = increase
                        best_node = node
                        best_position = pos
            
            # Insert the node
            if best_node is not None:
                tour.insert(best_position + 1, best_node)
                remaining.remove(best_node)
        
        # Create edge mask from tour
        edge_mask = np.zeros((n_nodes, n_nodes), dtype=bool)
        for idx in range(len(tour)):
            i = tour[idx]
            j = tour[(idx + 1) % len(tour)]
            edge_mask[i, j] = True
            edge_mask[j, i] = True
        
        return edge_mask
    
    insertion_edges = nearest_insertion_heuristic(safe_dist)
    
    # Calculate insertion bonus based on insertion cost savings
    for i in range(n):
        for j in range(n):
            if i != j and insertion_edges[i, j]:
                # Estimate insertion cost saving
                if len(nearest_neighbors[i]) > 0 and len(nearest_neighbors[j]) > 0:
                    # Average distance to other potential connections
                    avg_dist_i = np.mean(safe_dist[i, nearest_neighbors[i]])
                    avg_dist_j = np.mean(safe_dist[j, nearest_neighbors[j]])
                    insertion_bonus[i, j] = 1.0 + (avg_dist_i + avg_dist_j) / (2 * safe_dist[i, j] + 1e-10)
                else:
                    insertion_bonus[i, j] = 1.2
    
    # 5. Adaptive density and clustering awareness
    density_bonus = np.zeros_like(distance_matrix)
    
    # Local density estimation using adaptive radius
    density_radius = np.percentile(safe_dist[safe_dist < np.inf], 25)
    
    node_density = np.zeros(n)
    for i in range(n):
        close_nodes = safe_dist[i] <= density_radius
        node_density[i] = np.sum(close_nodes) - 1  # Exclude self
    
    # Normalize densities
    if np.max(node_density) > 0:
        node_density = node_density / np.max(node_density)
    
    for i in range(n):
        for j in range(n):
            if i != j:
                # Bonus for connecting nodes in similar density regions
                density_similarity = 1.0 - abs(node_density[i] - node_density[j])
                # Additional bonus if both are in dense regions
                dense_connection = min(node_density[i], node_density[j])
                density_bonus[i, j] = 0.5 * density_similarity + 0.5 * dense_connection
    
    # ========== ADAPTIVE FEATURE FUSION ==========
    
    # Dynamic weighting based on problem characteristics
    problem_sparsity = np.median(safe_dist) / np.mean(safe_dist)
    
    if problem_sparsity > 1.1:
        # Sparse problem: emphasize lookahead and insertion
        weights = {'base': 0.3, 'regret': 0.2, 'lookahead': 0.3, 'insertion': 0.2, 'density': 0.0}
    elif problem_sparsity < 0.9:
        # Dense problem: emphasize clustering and regret
        weights = {'base': 0.2, 'regret': 0.3, 'lookahead': 0.2, 'insertion': 0.1, 'density': 0.2}
    else:
        # Balanced problem
        weights = {'base': 0.25, 'regret': 0.25, 'lookahead': 0.2, 'insertion': 0.2, 'density': 0.1}
    
    # Combine components with adaptive weights
    base_component = inverse_dist * rank_weight
    regret_component = base_component * np.exp(-regret_penalty)
    lookahead_component = regret_component * lookahead_score
    density_component = lookahead_component * (1 + density_bonus)
    
    heuristics_matrix = (
        weights['base'] * base_component +
        weights['regret'] * regret_component +
        weights['lookahead'] * lookahead_component +
        weights['insertion'] * insertion_bonus +
        weights['density'] * density_component
    )
    
    # ========== INTELLIGENT SPARSIFICATION ==========
    
    # Adaptive candidate list
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Keep top adaptive_k edges per node
    for i in range(n):
        if adaptive_k < n - 1:
            top_indices = np.argsort(heuristics_matrix[i])[-adaptive_k:]
            candidate_mask[i, top_indices] = True
        else:
            candidate_mask[i, :] = True
            candidate_mask[i, i] = False
    
    # Ensure connectivity: add edges from insertion heuristic
    candidate_mask = candidate_mask | insertion_edges
    
    # Ensure symmetry
    candidate_mask = candidate_mask | candidate_mask.T
    
    # Apply sparsification
    heuristics_matrix[~candidate_mask] = 0
    
    # ========== POST-PROCESSING ==========
    
    # Normalize non-zero entries
    non_zero_mask = heuristics_matrix > 0
    if np.any(non_zero_mask):
        non_zero_vals = heuristics_matrix[non_zero_mask]
        if non_zero_vals.max() > non_zero_vals.min():
            heuristics_matrix[non_zero_mask] = (
                (non_zero_vals - non_zero_vals.min()) / 
                (non_zero_vals.max() - non_zero_vals.min() + 1e-10)
            )
    
    # Final symmetrization
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    # Zero diagonal
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
