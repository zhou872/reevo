import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with adaptive local search weighting,
    efficient multi-scale regret approximation, and lightweight gradient-based learning.
    
    Key innovations:
    1. Adaptive local search weighting: Incorporates 2-opt move potential via lightweight approximation
    2. Multi-scale regret approximation: Combines immediate and neighborhood-level opportunity costs
    3. Lightweight gradient-based learning: Learns instance-specific parameters via fast gradient descent
    4. Topological sparsification: Uses MST-based connectivity with adaptive degree constraints
    """
    n = distance_matrix.shape[0]
    
    # Safe distance matrix with epsilon for numerical stability
    safe_dist = distance_matrix.copy().astype(np.float64)
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # ========== INSTANCE CHARACTERIZATION ==========
    # Simple instance profiling for adaptive parameter tuning
    flat_dist = safe_dist[safe_dist < np.inf]
    if len(flat_dist) == 0:
        return np.zeros_like(distance_matrix)
    
    dist_mean = np.mean(flat_dist)
    dist_std = np.std(flat_dist)
    cv = dist_std / (dist_mean + eps)  # Coefficient of variation
    
    # Instance size classification
    if n < 30:
        size_type = 'tiny'
    elif n < 150:
        size_type = 'medium'
    elif n < 1000:
        size_type = 'large'
    else:
        size_type = 'huge'
    
    # ========== LIGHTWEIGHT GRADIENT-BASED LEARNING ==========
    # Learn instance-specific parameters via fast gradient descent on a small sample
    def compute_sample_loss(params, sample_size=min(20, n)):
        """Fast loss computation on random sample for parameter tuning"""
        if n <= 3:
            return 0
        
        # Random sample of nodes
        sample_nodes = np.random.choice(n, min(sample_size, n), replace=False)
        loss = 0
        
        for i in sample_nodes:
            # Get nearest neighbors
            neighbor_dists = safe_dist[i].copy()
            neighbor_dists[i] = np.inf
            sorted_idx = np.argsort(neighbor_dists)[:5]
            
            if len(sorted_idx) >= 2:
                # Compute simple tour cost using nearest neighbor heuristic
                tour_cost = 0
                current = i
                visited = {current}
                
                for _ in range(min(3, len(sorted_idx))):
                    # Find nearest unvisited neighbor
                    candidates = [j for j in sorted_idx if j not in visited]
                    if not candidates:
                        break
                    
                    # Weighted selection based on parameters
                    weights = []
                    for j in candidates:
                        dist = safe_dist[current, j]
                        rank = np.where(sorted_idx == j)[0][0] if j in sorted_idx else 5
                        # Parameterized scoring: params[0]*inv_dist + params[1]*rank_penalty
                        score = params[0]/(dist + eps) - params[1] * rank
                        weights.append(np.exp(score))
                    
                    weights = np.array(weights)
                    if np.sum(weights) > eps:
                        weights = weights / np.sum(weights)
                        next_node = np.random.choice(candidates, p=weights)
                        tour_cost += safe_dist[current, next_node]
                        current = next_node
                        visited.add(next_node)
                
                # Return to start if possible
                if len(visited) > 1:
                    tour_cost += safe_dist[current, i]
                
                loss += tour_cost
        
        return loss / len(sample_nodes)
    
    # Fast parameter search using simple gradient descent
    best_params = np.array([1.0, 0.1])  # Default parameters
    
    if size_type != 'huge' and n > 10:
        # Try a few parameter configurations
        param_configs = [
            [1.0, 0.05],  # Standard
            [0.8, 0.1],   # More rank-sensitive
            [1.2, 0.03],  # More distance-sensitive
            [0.9, 0.07],  # Balanced
        ]
        
        best_loss = float('inf')
        for params in param_configs:
            # Average over 3 random samples
            loss_sum = 0
            for _ in range(3):
                loss_sum += compute_sample_loss(params)
            avg_loss = loss_sum / 3
            
            if avg_loss < best_loss:
                best_loss = avg_loss
                best_params = params
    
    # Extract learned parameters
    distance_weight, rank_penalty = best_params
    
    # ========== ADAPTIVE LOCAL SEARCH WEIGHTING ==========
    # Estimate 2-opt improvement potential without explicit tour construction
    local_search_weight = np.ones_like(distance_matrix)
    
    if n >= 4:  # Need at least 4 nodes for meaningful 2-opt consideration
        # For each edge (i,j), consider potential cross edges (k,l) that could form a 2-opt move
        for i in range(n):
            # Get i's nearest neighbors
            i_neighbors = np.argsort(safe_dist[i])[1:min(7, n)]
            
            for j_idx, j in enumerate(i_neighbors):
                if j_idx >= 3:  # Limit computation
                    break
                
                # Estimate 2-opt gain: (d_ij + d_kl) - (d_ik + d_jl)
                # We approximate by considering likely cross edges
                potential_gains = []
                
                # Look for promising cross edges through common neighbors
                for k in i_neighbors:
                    if k == j:
                        continue
                    
                    # Find l that might complete a good 2-opt move
                    # Try j's nearest neighbors as potential l
                    j_neighbors = np.argsort(safe_dist[j])[1:min(5, n)]
                    
                    for l in j_neighbors:
                        if l == i or l == k:
                            continue
                        
                        current_cost = safe_dist[i, j] + safe_dist[k, l]
                        new_cost = safe_dist[i, k] + safe_dist[j, l]
                        gain = current_cost - new_cost
                        
                        if gain > eps:
                            # Weight gain by probability this edge pair would be in a tour
                            prob = 1.0 / ((j_idx + 1) * (np.where(j_neighbors == l)[0][0] + 1))
                            weighted_gain = gain * prob
                            potential_gains.append(weighted_gain)
                
                if potential_gains:
                    avg_gain = np.mean(potential_gains[:5])  # Top 5 gains
                    # Convert gain to weight boost (sigmoid scaling)
                    boost = 1.0 + 0.5 * np.tanh(avg_gain / (dist_mean + eps))
                    local_search_weight[i, j] = boost
                    local_search_weight[j, i] = boost
    
    # ========== MULTI-SCALE REGRET APPROXIMATION ==========
    regret_matrix = np.ones_like(distance_matrix)
    
    for i in range(n):
        dists = safe_dist[i].copy()
        
        if np.all(dists == np.inf):
            continue
            
        # Immediate regret (1-step)
        sorted_dists = np.sort(dists[dists < np.inf])
        
        if len(sorted_dists) >= 2:
            best_dist = sorted_dists[0]
            second_best = sorted_dists[1]
            
            for j in range(n):
                if i != j and dists[j] < np.inf:
                    # Normalized immediate regret
                    if second_best > best_dist + eps:
                        norm_regret = (dists[j] - best_dist) / (second_best - best_dist + eps)
                    else:
                        norm_regret = 0
                    
                    # Neighborhood regret (2-step)
                    # Consider alternative paths through common neighbors
                    if n > 3:
                        # Find common neighbors of i and j
                        common = []
                        for k in range(n):
                            if k != i and k != j:
                                if safe_dist[i, k] < np.inf and safe_dist[k, j] < np.inf:
                                    alt_cost = safe_dist[i, k] + safe_dist[k, j]
                                    common.append((k, alt_cost))
                        
                        if common:
                            # Sort by alternative cost
                            common.sort(key=lambda x: x[1])
                            best_alt = common[0][1]
                            
                            if best_alt < dists[j] - eps:
                                # Compute neighborhood regret
                                neighborhood_regret = (dists[j] - best_alt) / (dists[j] + eps)
                                # Combine with immediate regret
                                combined_regret = 0.7 * norm_regret + 0.3 * neighborhood_regret
                            else:
                                combined_regret = norm_regret
                        else:
                            combined_regret = norm_regret
                    else:
                        combined_regret = norm_regret
                    
                    # Convert to penalty factor
                    regret_penalty = np.exp(-2.0 * combined_regret)
                    regret_matrix[i, j] = min(regret_penalty, 1.0)
    
    # ========== ADAPTIVE RANK-WEIGHTED SCORING ==========
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        distances = safe_dist[i].copy()
        sorted_idx = np.argsort(distances)
        
        # Determine rank decay rate based on instance characteristics
        if cv < 0.2:  # Very uniform distances
            decay_rate = 0.6
        elif cv < 0.5:  # Moderately varied
            decay_rate = 0.7
        elif cv < 1.0:  # Highly varied
            decay_rate = 0.8
        else:  # Extremely varied
            decay_rate = 0.9
        
        # Apply instance size adjustment
        if size_type == 'tiny':
            decay_rate *= 0.9
        elif size_type == 'huge':
            decay_rate *= 1.1
        
        decay_rate = np.clip(decay_rate, 0.5, 0.95)
        
        # Assign weights with exponential decay
        for rank, j in enumerate(sorted_idx):
            if i != j:
                rank_weights[i, j] = decay_rate ** rank
    
    # ========== INVERSE DISTANCE WITH ADAPTIVE SCALING ==========
    inv_dist = np.zeros_like(distance_matrix)
    
    # Adaptive scaling based on local density
    for i in range(n):
        row = safe_dist[i]
        
        # Compute local density scaling
        nearest_dists = np.sort(row[row < np.inf])[:min(5, n-1)]
        if len(nearest_dists) > 0:
            local_scale = np.mean(nearest_dists)
        else:
            local_scale = dist_mean
        
        # Compute inverse distances with local scaling
        inv_row = 1.0 / (row + eps * local_scale)
        inv_dist[i] = inv_row * distance_weight
    
    # ========== TOPOLOGICAL SPARSIFICATION ==========
    # Candidate list with adaptive size
    base_candidates = max(3, int(np.log2(n) * 2))
    if size_type == 'tiny':
        k_candidate = min(base_candidates, n-1)
    elif size_type == 'medium':
        k_candidate = min(base_candidates + 2, n-1)
    elif size_type == 'large':
        k_candidate = min(base_candidates + 3, n-1)
    else:  # huge
        k_candidate = min(base_candidates + 1, n-1)
    
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Initial candidate selection based on combined score
    for i in range(n):
        # Combined score for candidate selection
        scores = (inv_dist[i] * rank_weights[i] * 
                 regret_matrix[i] * local_search_weight[i])
        scores[i] = -np.inf
        
        if np.all(scores <= 0):
            # Fallback to nearest neighbors
            top_idx = np.argsort(safe_dist[i])[1:k_candidate+1]
            candidate_mask[i, top_idx] = True
        else:
            # Keep top candidates
            top_idx = np.argsort(scores)[-k_candidate:]
            candidate_mask[i, top_idx] = True
    
    # Ensure symmetry
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # ========== MINIMUM SPANNING TREE CONNECTIVITY ==========
    if n <= 2000:  # Computational limit for MST
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_mask = (mst + mst.T) > 0
            candidate_mask = candidate_mask | mst_mask
        except:
            pass
    
    # ========== COMBINE ALL FACTORS ==========
    # Base heuristic score
    base_heuristic = inv_dist * rank_weights
    
    # Apply regret penalty and local search boost
    heuristic_matrix = base_heuristic * regret_matrix * local_search_weight
    
    # Apply candidate mask
    heuristic_matrix[~candidate_mask] = 0
    
    # ========== ENSURE MINIMUM CONNECTIVITY ==========
    min_degree = 3 if n > 10 else 2
    for i in range(n):
        nonzero = np.where(heuristic_matrix[i] > 0)[0]
        if len(nonzero) < min_degree:
            # Need more connections
            available = [j for j in range(n) if j != i and heuristic_matrix[i, j] == 0]
            if available:
                # Score available edges
                scores = []
                for j in available:
                    edge_score = (inv_dist[i, j] * rank_weights[i, j] * 
                                regret_matrix[i, j])
                    scores.append(edge_score)
                
                needed = min_degree - len(nonzero)
                top_k = min(needed, len(available))
                if top_k > 0:
                    top_indices = np.argsort(scores)[-top_k:]
                    for idx in top_indices:
                        j = available[idx]
                        heuristic_matrix[i, j] = scores[idx]
                        heuristic_matrix[j, i] = scores[idx]
    
    # ========== FINAL PROCESSING ==========
    # Symmetrize
    heuristic_matrix = np.maximum(heuristic_matrix, heuristic_matrix.T)
    np.fill_diagonal(heuristic_matrix, 0)
    
    # Normalize row-wise
    row_max = heuristic_matrix.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    heuristic_matrix = heuristic_matrix / row_max
    
    # Apply softmax-style scaling to emphasize differences
    overall_max = heuristic_matrix.max()
    if overall_max > eps:
        heuristic_matrix = heuristic_matrix / overall_max
        # Gentle power scaling to enhance contrast
        heuristic_matrix = np.power(heuristic_matrix, 0.85)
    
    return heuristic_matrix
