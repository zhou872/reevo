import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
import warnings

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with efficient regret modeling, 
    adaptive statistical weighting, and optimized sparsification.
    
    Core Features:
    1. Two-step regret approximation via neighborhood sampling
    2. Coefficient-of-variation adaptive decay rates for rank weighting
    3. Hybrid sparsification combining candidate lists and MST connectivity
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # Safe distance matrix (avoid self-loops)
    safe_dist = distance_matrix.copy().astype(np.float64)
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # ========== ADAPTIVE PARAMETER SELECTION ==========
    # Determine candidate list size based on problem scale
    if n <= 50:
        k_candidate = min(15, n - 1)
        regret_neighbors = min(10, n - 1)
    elif n <= 200:
        k_candidate = min(int(5 + np.sqrt(n)), n - 1)
        regret_neighbors = min(15, n - 1)
    else:
        k_candidate = min(int(3 * np.log2(n)), n - 1)
        regret_neighbors = min(20, n - 1)
    
    # ========== HYBRID SPARSIFICATION ==========
    # Candidate mask using k-nearest neighbors
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    # Get k_candidate nearest neighbors for each node
    neighbor_indices = np.argsort(safe_dist, axis=1)[:, 1:k_candidate+1]
    
    # Efficiently set candidate edges
    rows = np.arange(n)[:, None].repeat(k_candidate, axis=1)
    candidate_mask[rows.flatten(), neighbor_indices.flatten()] = True
    
    # Ensure symmetry
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # Add MST edges for connectivity (for moderate-sized instances)
    if n <= 1000 and n >= 3:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_mask = (mst + mst.T) > 0
            candidate_mask = candidate_mask | mst_mask
        except:
            pass  # Skip if MST computation fails
    
    # ========== TWO-STEP REGRET APPROXIMATION ==========
    regret_penalty = np.ones_like(safe_dist)
    
    # Precompute neighbor indices for regret calculation
    regret_neighbor_idx = np.argsort(safe_dist, axis=1)[:, 1:regret_neighbors+1]
    
    for i in range(n):
        dists_i = safe_dist[i].copy()
        
        # For each potential edge (i, j) in candidate set
        for j_idx in range(n):
            if i == j_idx or not candidate_mask[i, j_idx]:
                continue
                
            direct_cost = dists_i[j_idx]
            
            # Sample alternative paths through intermediate nodes
            alt_costs = []
            
            # Consider intermediate nodes that are neighbors of both i and j
            common_neighbors = np.intersect1d(
                regret_neighbor_idx[i], 
                regret_neighbor_idx[j_idx], 
                assume_unique=True
            )
            
            # Remove i and j from consideration
            common_neighbors = common_neighbors[common_neighbors != i]
            common_neighbors = common_neighbors[common_neighbors != j_idx]
            
            # Sample up to 3 common neighbors
            if len(common_neighbors) > 0:
                sample_size = min(3, len(common_neighbors))
                sampled = common_neighbors[:sample_size] if len(common_neighbors) <= 5 else \
                         np.random.choice(common_neighbors, sample_size, replace=False)
                
                for k in sampled:
                    alt_cost = safe_dist[i, k] + safe_dist[k, j_idx]
                    alt_costs.append(alt_cost)
            
            # If no common neighbors, try random neighbors of i
            if len(alt_costs) == 0 and regret_neighbors > 0:
                for k in regret_neighbor_idx[i, :min(3, regret_neighbors)]:
                    if k != j_idx:
                        alt_costs.append(safe_dist[i, k] + safe_dist[k, j_idx])
            
            # Calculate regret penalty
            if alt_costs:
                best_alt = min(alt_costs)
                if best_alt < direct_cost:
                    # Positive regret: alternative is better
                    regret = (direct_cost - best_alt) / (direct_cost + eps)
                    # Sigmoid transform for smooth penalty
                    regret_penalty[i, j_idx] = 1.0 - 0.5 * (1.0 / (1.0 + np.exp(-5 * regret)))
                else:
                    # Negative or zero regret: direct edge is good
                    regret_penalty[i, j_idx] = 1.0 + 0.1 * np.tanh((best_alt - direct_cost) / (direct_cost + eps))
    
    # ========== COEFFICIENT-OF-VARIATION ADAPTIVE WEIGHTING ==========
    rank_weights = np.zeros_like(safe_dist)
    
    for i in range(n):
        # Get valid distances for node i
        dists = safe_dist[i, safe_dist[i] < np.inf]
        
        if len(dists) >= 3:
            # Compute coefficient of variation on nearest distances
            nearest_dists = np.sort(dists)[:min(5, len(dists))]
            cv = np.std(nearest_dists) / (np.mean(nearest_dists) + eps)
            
            # Adaptive decay rate based on CV
            # High CV (varied distances) -> slower decay (preserve more candidates)
            # Low CV (uniform distances) -> faster decay (focus on nearest)
            decay_rate = 0.4 + 0.5 * (1.0 / (1.0 + np.exp(-8 * (cv - 0.3))))
            decay_rate = np.clip(decay_rate, 0.4, 0.9)
        else:
            decay_rate = 0.7  # Default
        
        # Apply exponential rank weighting
        sorted_indices = np.argsort(safe_dist[i])
        for rank, j in enumerate(sorted_indices):
            if i != j:
                rank_weights[i, j] = decay_rate ** rank
    
    # ========== BASE INVERSE DISTANCE SCORING ==========
    # Scale inverse distance by global statistics for stability
    valid_dists = safe_dist[safe_dist < np.inf]
    if len(valid_dists) > 0:
        mean_dist = np.mean(valid_dists)
        std_dist = np.std(valid_dists)
    else:
        mean_dist = 1.0
        std_dist = 0.0
    
    inv_dist = np.zeros_like(safe_dist)
    valid_mask = safe_dist < np.inf
    # Use robust inverse with regularization
    inv_dist[valid_mask] = mean_dist / (safe_dist[valid_mask] + 0.1 * std_dist + eps)
    
    # ========== COMBINE FACTORS ==========
    heuristics_matrix = np.zeros_like(safe_dist)
    
    # Base combination with adaptive component weights
    base_score = inv_dist * rank_weights
    
    # Adjust regret weight based on problem size
    if n <= 100:
        regret_weight = 0.7
    elif n <= 500:
        regret_weight = 0.5
    else:
        regret_weight = 0.3
    
    heuristics_matrix = base_score * (1.0 - regret_weight + regret_weight * regret_penalty)
    
    # Apply candidate mask
    heuristics_matrix[~candidate_mask] = 0
    
    # ========== ENSURE MINIMUM CONNECTIVITY ==========
    for i in range(n):
        row_heuristics = heuristics_matrix[i]
        nonzero_mask = row_heuristics > 0
        
        if np.sum(nonzero_mask) < 2:
            # Find 2 nearest neighbors not already connected
            sorted_indices = np.argsort(safe_dist[i])
            added = 0
            
            for j in sorted_indices:
                if i != j and added < 2:
                    # Boost this edge
                    boost_value = np.max(heuristics_matrix[i]) * 1.5 if np.max(heuristics_matrix[i]) > 0 else 0.5
                    heuristics_matrix[i, j] = boost_value
                    heuristics_matrix[j, i] = boost_value
                    added += 1
    
    # ========== FINAL PROCESSING ==========
    # Symmetrize
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    
    # Normalize rows
    row_max = np.max(heuristics_matrix, axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    heuristics_matrix = heuristics_matrix / row_max
    
    # Global normalization to [0, 1]
    overall_max = np.max(heuristics_matrix)
    if overall_max > eps:
        heuristics_matrix = heuristics_matrix / overall_max
    
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
