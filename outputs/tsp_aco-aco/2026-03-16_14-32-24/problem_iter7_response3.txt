import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge evaluation with smooth regret modeling,
    distribution-adaptive weighting, and hierarchical sparsification.
    
    Features:
    1. Smooth Regret Penalty: Sigmoid-based continuous penalty scaling
    2. Distribution-Adaptive Weighting: Exponential rank weights adjusted by distance statistics
    3. Gaussian Kernel Density: Pairwise density estimation with covariance-weighted bandwidth
    4. Hierarchical Sparsification: Multi-level candidate selection with symmetry constraints
    """
    n = distance_matrix.shape[0]
    
    # Safe distance matrix (avoid self-loops and division by zero)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # 1. Hierarchical candidate selection with adaptive thresholds
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Primary candidates: nearest neighbors (sqrt(n) scaling)
    k_primary = max(3, int(np.sqrt(n)))
    for i in range(n):
        neighbors = np.argsort(safe_dist[i])[:k_primary]
        candidate_mask[i, neighbors] = True
    
    # Secondary candidates: distance-based threshold (median + 1.5 * IQR)
    for i in range(n):
        row_dists = safe_dist[i, safe_dist[i] < np.inf]
        if len(row_dists) > 0:
            q1, q3 = np.percentile(row_dists, [25, 75])
            iqr = q3 - q1
            threshold = q3 + 1.5 * iqr
            secondary_mask = safe_dist[i] <= threshold
            candidate_mask[i, secondary_mask] = True
    
    # Ensure symmetry and at least 3 connections per node
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    for i in range(n):
        if np.sum(candidate_mask[i]) < 3:
            # Add additional nearest neighbors if needed
            additional = np.argsort(safe_dist[i])[3:6]
            candidate_mask[i, additional] = True
            candidate_mask[additional, i] = True
    
    # 2. Smooth Regret Penalty with sigmoid transition
    regret_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        dists = safe_dist[i].copy()
        if len(dists) < 2:
            continue
            
        # Reference distances
        min_dist = np.min(dists)
        med_dist = np.median(dists)
        max_dist = np.max(dists)
        
        # Adaptive scaling factor based on distance spread
        scale = max(med_dist - min_dist, eps)
        
        # Smooth regret function: sigmoid transition from 0 to 1
        for j in range(n):
            if i != j:
                rel_dist = (dists[j] - min_dist) / scale
                # Sigmoid with adjustable steepness
                steepness = 2.0
                regret = 1.0 / (1.0 + np.exp(-steepness * (rel_dist - 1.0)))
                regret_matrix[i, j] = regret
    
    # 3. Distribution-Adaptive Rank Weighting
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        sorted_idx = np.argsort(safe_dist[i])
        valid_dists = safe_dist[i, sorted_idx[1:min(8, n-1)]]
        
        if len(valid_dists) > 1:
            # Adaptive decay based on coefficient of variation
            mean_dist = np.mean(valid_dists)
            std_dist = np.std(valid_dists)
            cv = std_dist / (mean_dist + eps)
            
            # Base decay increases with uniformity (low CV -> sharper decay)
            base_decay = 0.4 + 0.3 * np.tanh(2.0 - 5.0 * cv)  # Range ~0.3-0.7
            decay_rate = np.clip(base_decay, 0.25, 0.85)
        else:
            decay_rate = 0.6
        
        # Exponential rank weighting with rank offset
        for rank, j in enumerate(sorted_idx):
            if i != j:
                # Offset rank by 1 to avoid zero weight for nearest neighbor
                rank_weights[i, j] = decay_rate ** (rank + 1)
    
    # 4. Gaussian Kernel Density with covariance-weighted bandwidth
    density_bonus = np.zeros_like(distance_matrix)
    
    # Precompute node-specific statistics
    node_radii = np.zeros(n)
    for i in range(n):
        # Radius defined as mean of k-nearest distances
        k_radius = min(4, n-1)
        nearest_dists = np.partition(safe_dist[i], k_radius)[:k_radius+1]
        node_radii[i] = np.mean(nearest_dists[nearest_dists > 0])
    
    # Compute pairwise density scores only for candidate edges
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Combined bandwidth using geometric mean of node radii
                bandwidth = np.sqrt(node_radii[i] * node_radii[j])
                sigma = bandwidth / np.sqrt(2)  # Adjusted for 2D kernel
                
                # Gaussian kernel contributions from all nodes
                kernel_i = np.exp(-0.5 * (safe_dist[i] / sigma) ** 2)
                kernel_j = np.exp(-0.5 * (safe_dist[j] / sigma) ** 2)
                
                # Joint kernel density (excluding the endpoints themselves)
                joint_kernel = kernel_i * kernel_j
                joint_density = np.sum(joint_kernel) - 2 * joint_kernel[i]  # Subtract double-counted self
                
                # Adaptive bonus scaling
                if joint_density > 0:
                    # Logarithmic scaling with diminishing returns
                    density_scale = 0.2
                    density_bonus[i, j] = density_scale * np.log1p(joint_density / n)
                else:
                    density_bonus[i, j] = -0.08  # Mild penalty for isolation
    
    density_bonus = np.maximum(density_bonus, density_bonus.T)
    
    # 5. Minimum Spanning Tree guidance (scaled by problem size)
    mst_bonus = np.zeros_like(distance_matrix)
    if n <= 2000:  # Increased threshold for modern hardware
        try:
            # Use MST only if it adds meaningful information
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_edges = (mst + mst.T) > 0
            
            # Scale MST bonus inversely with problem size
            mst_weight = 0.25 * np.exp(-n / 500.0)  # Decays from 0.25 to ~0.03
            mst_bonus = mst_edges * mst_weight
        except:
            pass
    
    # 6. Composite scoring with smooth penalty application
    # Base score: inverse distance with rank weighting
    inv_dist = 1.0 / safe_dist
    mean_dist = np.mean(safe_dist[safe_dist < np.inf])
    inv_dist = inv_dist * mean_dist  # Normalization
    
    base_score = inv_dist * rank_weights
    
    # Apply regret penalty with exponential decay
    regret_scale = 2.5  # Controls penalty strength
    penalty_factor = np.exp(-regret_scale * regret_matrix)
    
    # Combine all components
    combined_score = base_score * penalty_factor * (1.0 + density_bonus) * (1.0 + mst_bonus)
    
    # 7. Multi-stage sparsification
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Stage 1: Apply candidate mask
    heuristics_matrix = combined_score * candidate_mask
    
    # Stage 2: Ensure connectivity constraints
    for i in range(n):
        row = heuristics_matrix[i]
        if np.count_nonzero(row) < 2:
            # Find two most promising edges based on combined score (before masking)
            top_scores = combined_score[i].copy()
            top_scores[i] = -np.inf  # Exclude self
            top_indices = np.argsort(top_scores)[-2:]
            
            # Boost these edges to above average level
            avg_score = np.mean(heuristics_matrix[heuristics_matrix > 0]) if np.any(heuristics_matrix > 0) else 0.1
            heuristics_matrix[i, top_indices] = max(avg_score * 1.3, np.max(row))
            heuristics_matrix[top_indices, i] = max(avg_score * 1.3, np.max(row))
    
    # 8. Symmetrization and normalization
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1] with outlier clipping
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        # Use 99th percentile to reduce outlier influence
        p99 = np.percentile(heuristics_matrix[heuristics_matrix > 0], 99) if np.any(heuristics_matrix > 0) else max_val
        scale_factor = min(max_val, p99 * 1.5)  # Cap at 1.5 * p99
        heuristics_matrix = heuristics_matrix / scale_factor
        heuristics_matrix = np.clip(heuristics_matrix, 0, 1)
    
    return heuristics_matrix
