import numpy as np
from scipy.spatial import ConvexHull
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between nodes
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, with unpromising edges set to zero
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) / (distance_matrix + 1e-10)
    
    # 1. Base inverse distance (handle zeros)
    inv_dist = 1.0 / (distance_matrix + np.eye(n) + 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Nearest neighbor sparsification
    nn_count = np.zeros((n, n))
    k = min(5, n - 1)  # Number of nearest neighbors to consider
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nearest = np.argsort(distance_matrix[i])[1:k+1]
        nn_count[i, nearest] = 1
        nn_count[nearest, i] = 1
    
    # 3. Minimum spanning tree bonus
    try:
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_bonus = ((mst + mst.T) > 0).astype(float) * 2.0
    except:
        mst_bonus = np.ones((n, n))
    
    # 4. Convex hull distance weighting
    # Points on convex hull are often connected in optimal tours
    try:
        coords = np.random.randn(n, 2)  # Placeholder - in practice would use actual coordinates
        hull = ConvexHull(coords)
        hull_mask = np.zeros((n, n))
        for simplex in hull.simplices:
            for i in range(len(simplex)):
                a, b = simplex[i], simplex[(i+1)%len(simplex)]
                hull_mask[a, b] = hull_mask[b, a] = 1
        hull_weight = 1.0 + 1.5 * hull_mask
    except:
        hull_weight = np.ones((n, n))
    
    # 5. Local density penalty
    # Penalize edges in dense clusters (nodes with many close neighbors)
    density_penalty = np.ones((n, n))
    avg_dist = distance_matrix.mean(axis=1, keepdims=True)
    local_density = (distance_matrix < avg_dist).sum(axis=1)
    density_factor = 1.0 / (1.0 + 0.1 * np.maximum(local_density[:, None], local_density[None, :]))
    density_penalty *= density_factor
    
    # 6. Combine all factors
    heuristics = inv_dist * nn_count * mst_bonus * hull_weight * density_penalty
    
    # 7. Additional sparsification: keep only top edges for each node
    sparsified = np.zeros_like(heuristics)
    keep_per_node = min(8, n - 1)  # Keep at most 8 edges per node
    for i in range(n):
        top_indices = np.argsort(heuristics[i])[-keep_per_node:]
        sparsified[i, top_indices] = heuristics[i, top_indices]
        sparsified[top_indices, i] = heuristics[top_indices, i]
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    np.fill_diagonal(sparsified, 0)
    
    # Normalize for numerical stability
    max_val = sparsified.max()
    if max_val > 0:
        sparsified /= max_val
    
    return sparsified
