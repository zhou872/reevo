import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge evaluation focusing on regret-based penalization,
    adaptive sparsification, and spatial density analysis.
    
    Features:
    1. Regret-Based Penalization: Prioritizes edges with minimal opportunity cost
    2. Adaptive Sparsification: Dynamic candidate list scaling with sqrt(n)
    3. Spatial Density Bonus: Geometric mean of node densities within median distance
    4. Exponential Decay Weighting: Rank-based importance assignment
    """
    n = distance_matrix.shape[0]
    
    # Safe distance matrix (avoid self-loops)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # Initialize heuristic matrix
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
    
    # 1. Adaptive candidate sparsification
    k = max(3, int(np.sqrt(n)))
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Select k nearest neighbors
        neighbors = np.argsort(safe_dist[i])[:k]
        candidate_mask[i, neighbors] = True
    
    # Ensure symmetry
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # 2. Compute node densities within median distance
    density_scores = np.zeros(n)
    for i in range(n):
        # Calculate median distance to other nodes
        valid_dists = safe_dist[i][safe_dist[i] < np.inf]
        if len(valid_dists) > 0:
            median_dist = np.median(valid_dists)
            # Count nodes within median distance (excluding self)
            density_scores[i] = np.sum(safe_dist[i] <= median_dist)
    
    # Normalize density scores
    if np.max(density_scores) > 0:
        density_scores = density_scores / np.max(density_scores)
    
    # 3. Regret-based penalization with exponential decay
    average_distance = np.mean(safe_dist[safe_dist < np.inf])
    
    for i in range(n):
        # Get distances from node i
        dists = safe_dist[i].copy()
        
        # Find second best distance (minimum excluding the best)
        if len(dists) >= 2:
            sorted_dists = np.sort(dists)
            second_best = sorted_dists[1] if sorted_dists[0] == 0 else sorted_dists[0]
        else:
            second_best = average_distance
        
        for j in range(n):
            if i != j and candidate_mask[i, j]:
                # Base inverse distance
                inv_dist = 1.0 / (safe_dist[i, j] + 1e-12)
                
                # Regret penalty: (distance - second_best) / average_distance
                regret_penalty = (safe_dist[i, j] - second_best) / (average_distance + 1e-12)
                
                # Apply exponential decay to regret penalty
                regret_factor = np.exp(-np.abs(regret_penalty))
                
                # Density bonus: geometric mean of node densities
                density_bonus = np.sqrt(density_scores[i] * density_scores[j] + 1e-12)
                
                # Rank-based weighting: exponential decay by rank
                # Get rank of edge (i,j) among i's connections
                if candidate_mask[i].any():
                    sorted_neighbors = np.argsort(safe_dist[i])
                    rank = np.where(sorted_neighbors == j)[0]
                    if len(rank) > 0:
                        rank_weight = np.exp(-rank[0] / max(k, 1))
                    else:
                        rank_weight = 0.5
                else:
                    rank_weight = 0.5
                
                # Combine factors
                score = (inv_dist * regret_factor * 
                        (1.0 + 0.3 * density_bonus) * 
                        rank_weight)
                
                heuristics_matrix[i, j] = score
    
    # 4. Symmetrize the matrix
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    # 5. Ensure connectivity: each node must have at least 2 connections
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero = np.nonzero(row)[0]
        if len(nonzero) < 2:
            # Find top 2 edges by inverse distance
            inv_dists = 1.0 / (safe_dist[i] + 1e-12)
            inv_dists[i] = 0  # Exclude self
            top_indices = np.argsort(inv_dists)[-2:]
            for idx in top_indices:
                if heuristics_matrix[i, idx] == 0:
                    heuristics_matrix[i, idx] = np.mean(heuristics_matrix[heuristics_matrix > 0])
                    heuristics_matrix[idx, i] = heuristics_matrix[i, idx]
    
    # 6. Normalize to [0, 1] range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    # Set diagonal to zero
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
