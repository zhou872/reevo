import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge evaluation with refined regret modeling,
    adaptive rank weighting, and optimized sparsification.
    
    Key Improvements over v1:
    1. More efficient candidate list construction with vectorized operations
    2. Simplified regret calculation focusing on second-best alternative
    3. Optimized rank weighting using coefficient of variation
    4. Reduced computational overhead while maintaining accuracy
    """
    n = distance_matrix.shape[0]
    
    # Safe distance matrix (avoid self-loops)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # 1. Multi-Factor Sparsification with Symmetry Enforcement
    # Dynamic candidate list size based on problem size
    k_candidate = max(4, int(np.sqrt(n)) * 2)
    
    # Vectorized candidate selection
    candidate_indices = np.argsort(safe_dist, axis=1)[:, :k_candidate]
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Set candidate edges
    rows = np.arange(n)[:, None].repeat(k_candidate, axis=1)
    candidate_mask[rows.flatten(), candidate_indices.flatten()] = True
    
    # Enforce symmetry
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # 2. Exponential Rank-Weighted Scoring with Coefficient of Variation
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get sorted distances for current node
        sorted_dists = np.sort(safe_dist[i])
        valid_dists = sorted_dists[sorted_dists < np.inf]
        
        if len(valid_dists) >= 5:
            # Use first 5 distances for coefficient of variation
            cv_dists = valid_dists[:5]
            coef_var = np.std(cv_dists) / (np.mean(cv_dists) + eps)
            
            # Map coefficient of variation to decay rate using sigmoid
            # sigmoid(10*(coef_var-0.5)) maps 0.5 to 0.5, with clipping
            sigmoid_val = 1 / (1 + np.exp(-10 * (coef_var - 0.5)))
            decay_rate = 0.3 + 0.55 * sigmoid_val  # Range: [0.3, 0.85]
            decay_rate = np.clip(decay_rate, 0.3, 0.85)
        else:
            decay_rate = 0.7  # Default value
        
        # Assign exponential weights based on rank
        sorted_indices = np.argsort(safe_dist[i])
        for rank, j in enumerate(sorted_indices):
            if i != j:
                rank_weights[i, j] = decay_rate ** rank
    
    # 3. Simplified Regret-Based Penalty
    regret_penalty = np.ones_like(distance_matrix)
    
    for i in range(n):
        # Find best and second-best distances for node i
        dists = safe_dist[i].copy()
        
        if len(dists[dists < np.inf]) >= 2:
            # Get two smallest distances
            sorted_dists = np.sort(dists)
            best_dist = sorted_dists[0]
            second_best_dist = sorted_dists[1]
            
            # Calculate penalty for each edge
            for j in range(n):
                if i != j and candidate_mask[i, j]:
                    edge_dist = dists[j]
                    
                    # Simplified regret calculation
                    if edge_dist == best_dist:
                        penalty = 1.0
                    else:
                        # Normalized difference relative to second-best
                        norm_diff = (edge_dist - second_best_dist) / (second_best_dist + eps)
                        # Exponential penalty with smoothing
                        penalty = np.exp(-max(0, norm_diff))
                    
                    regret_penalty[i, j] = penalty
    
    # 4. Base distance inverse scoring with normalization
    inv_dist = np.zeros_like(distance_matrix)
    valid_mask = safe_dist < np.inf
    
    # Scale inverse distance by mean distance for stability
    mean_dist = np.mean(safe_dist[valid_mask])
    inv_dist[valid_mask] = mean_dist / (safe_dist[valid_mask] + eps)
    
    # 5. Combine factors
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base combination: inverse distance × rank weights × regret penalty
    base_score = inv_dist * rank_weights
    heuristics_matrix = base_score * regret_penalty
    
    # Apply candidate mask
    heuristics_matrix[~candidate_mask] = 0
    
    # 6. Ensure minimum connectivity (at least 2 edges per node)
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero_count = np.count_nonzero(row)
        
        if nonzero_count < 2:
            # Find top 2 edges by distance (excluding zero edges)
            sorted_indices = np.argsort(safe_dist[i])
            top_edges = []
            
            for j in sorted_indices:
                if i != j and len(top_edges) < 2:
                    top_edges.append(j)
            
            # Boost these edges
            for j in top_edges:
                boost_value = np.max(heuristics_matrix[i]) * 1.2 + 0.1
                heuristics_matrix[i, j] = boost_value
                heuristics_matrix[j, i] = boost_value
    
    # 7. Lightweight MST bias for moderate-sized problems
    if n <= 1000 and n >= 3:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_mask = (mst + mst.T) > 0
            
            # Apply modest bonus to MST edges
            mst_bonus = 0.25
            heuristics_matrix[mst_mask] *= (1 + mst_bonus)
        except:
            pass  # Skip MST if computation fails
    
    # 8. Symmetrize and normalize
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_matrix)
    if max_val > eps:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
