import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with adaptive sigmoid regret, landmark-based
    graph embeddings, and dynamic component weighting.
    
    Returns:
        Matrix of edge promise scores [0,1] where higher values indicate
        edges more likely to be in optimal solutions.
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # Safe distance matrix with diagonal removed
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # === 1. INSTANCE CHARACTERIZATION ===
    # Extract distance statistics for adaptive parameters
    flat_dists = safe_dist[safe_dist < np.inf]
    if len(flat_dists) == 0:
        return np.zeros_like(distance_matrix)
    
    dist_mean = np.mean(flat_dists)
    dist_std = np.std(flat_dists)
    dist_q1, dist_q3 = np.percentile(flat_dists, [25, 75])
    dist_iqr = dist_q3 - dist_q1
    
    # Measure distribution shape
    cv = dist_std / (dist_mean + eps)  # Coefficient of variation
    skewness = 3 * (dist_mean - np.median(flat_dists)) / (dist_std + eps)
    
    # === 2. ADAPTIVE CANDIDATE SELECTION ===
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Primary candidates: dynamic number based on instance size and uniformity
    k_primary = max(3, int(np.sqrt(n) * (1.2 if cv < 0.5 else 0.8)))
    for i in range(n):
        neighbors = np.argsort(safe_dist[i])[:k_primary]
        candidate_mask[i, neighbors] = True
    
    # Secondary threshold: adaptive based on distribution skew
    for i in range(n):
        row_dists = safe_dist[i, safe_dist[i] < np.inf]
        if len(row_dists) > 0:
            # Adjust threshold based on skewness
            if skewness > 0.5:  # Right-skewed: use higher threshold
                threshold = dist_q3 + 2.0 * dist_iqr
            elif skewness < -0.5:  # Left-skewed: use tighter threshold
                threshold = dist_q3 + 1.0 * dist_iqr
            else:  # Symmetric
                threshold = dist_q3 + 1.5 * dist_iqr
            
            secondary_mask = safe_dist[i] <= threshold
            candidate_mask[i, secondary_mask] = True
    
    # Ensure symmetry and minimum connectivity
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    for i in range(n):
        if np.sum(candidate_mask[i]) < 3:
            additional = np.argsort(safe_dist[i])[3:6]
            candidate_mask[i, additional] = True
            candidate_mask[additional, i] = True
    
    # === 3. ADAPTIVE SIGMOID REGRET ===
    regret_matrix = np.zeros_like(distance_matrix)
    
    # Dynamic sigmoid steepness based on distance concentration
    if dist_iqr > 0:
        # Higher CV -> more spread out distances -> shallower sigmoid
        steepness_base = 2.0
        concentration = 1.0 / (1.0 + cv)  # [0.5, 1] range
        steepness = steepness_base * concentration
    else:
        steepness = 2.0
    
    for i in range(n):
        dists = safe_dist[i].copy()
        if len(dists[dists < np.inf]) < 2:
            continue
            
        # Reference points: minimum and adaptive pivot
        min_dist = np.min(dists)
        pivot_dist = np.percentile(dists[dists < np.inf], 75)
        
        # Normalize distances
        scale = max(pivot_dist - min_dist, eps)
        
        for j in range(n):
            if i != j and dists[j] < np.inf:
                rel_dist = (dists[j] - min_dist) / scale
                # Adaptive sigmoid with dynamic steepness and offset
                offset = 0.8  # Slight right shift to favor shorter edges
                regret = 1.0 / (1.0 + np.exp(-steepness * (rel_dist - offset)))
                regret_matrix[i, j] = regret
    
    # === 4. LANDMARK-BASED GRAPH EMBEDDING ===
    embedding_bonus = np.zeros_like(distance_matrix)
    
    # Select landmarks: sqrt(n) nodes with maximum dispersion
    n_landmarks = max(3, min(int(np.sqrt(n)), n))
    if n_landmarks == n:
        landmarks = list(range(n))
    else:
        # Greedy max-min dispersion for landmark selection
        landmarks = [np.random.randint(n)]
        for _ in range(1, n_landmarks):
            # For each candidate node, compute minimum distance to existing landmarks
            min_dists = np.min(safe_dist[:, landmarks], axis=1)
            min_dists[landmarks] = -np.inf  # Exclude already selected
            landmarks.append(np.argmax(min_dists))
    
    # Compute landmark distance vectors for each node
    landmark_vectors = safe_dist[:, landmarks].copy()
    
    # Normalize each landmark dimension
    for l_idx in range(n_landmarks):
        col = landmark_vectors[:, l_idx]
        valid = col < np.inf
        if np.any(valid):
            col_mean = np.mean(col[valid])
            col_std = np.std(col[valid]) + eps
            landmark_vectors[valid, l_idx] = (col[valid] - col_mean) / col_std
    
    # Compute similarity only for candidate edges
    for i in range(n):
        vec_i = landmark_vectors[i]
        valid_i = ~np.isnan(vec_i)
        
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                vec_j = landmark_vectors[j]
                valid_j = ~np.isnan(vec_j)
                
                # Valid dimensions where both vectors have finite values
                valid_dims = valid_i & valid_j
                if np.sum(valid_dims) >= 2:
                    # Cosine similarity in landmark space
                    v1 = vec_i[valid_dims]
                    v2 = vec_j[valid_dims]
                    norm1 = np.linalg.norm(v1)
                    norm2 = np.linalg.norm(v2)
                    
                    if norm1 > 0 and norm2 > 0:
                        similarity = np.dot(v1, v2) / (norm1 * norm2)
                        # Transform to bonus: similar nodes (positive correlation) get bonus
                        embedding_bonus[i, j] = max(0, similarity) * 0.25
                    else:
                        embedding_bonus[i, j] = -0.05  # Mild penalty
                else:
                    embedding_bonus[i, j] = -0.1  # Stronger penalty for insufficient data
    
    embedding_bonus = np.maximum(embedding_bonus, embedding_bonus.T)
    
    # === 5. DISTRIBUTION-AWARE RANK WEIGHTING ===
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        sorted_idx = np.argsort(safe_dist[i])
        valid_ranks = min(12, n-1)
        valid_dists = safe_dist[i, sorted_idx[1:valid_ranks+1]]
        
        if len(valid_dists) > 1:
            # Adjust decay based on distance monotonicity
            monotonicity = 0
            for k in range(1, len(valid_dists)):
                if valid_dists[k] > valid_dists[k-1]:
                    monotonicity += 1
            monotonicity /= (len(valid_dists) - 1)
            
            # Base decay: steeper for monotonic sequences
            base_decay = 0.3 + 0.4 * monotonicity
            decay_rate = np.clip(base_decay, 0.25, 0.8)
        else:
            decay_rate = 0.5
        
        # Exponential weighting with rank
        for rank, j in enumerate(sorted_idx):
            if i != j:
                rank_weights[i, j] = decay_rate ** (rank + 1)
    
    # === 6. MINIMUM SPANNING TREE BONUS ===
    mst_bonus = np.zeros_like(distance_matrix)
    if n <= 5000:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_edges = (mst + mst.T) > 0
            
            # Scale MST importance based on instance uniformity
            if cv < 0.3:
                mst_weight = 0.15  # Less important for uniform instances
            else:
                mst_weight = 0.25  # More important for varied instances
            
            mst_bonus = mst_edges * mst_weight
        except:
            pass
    
    # === 7. DYNAMIC COMPONENT COMBINATION ===
    # Base score: normalized inverse distance
    inv_dist = 1.0 / safe_dist
    mean_valid = np.mean(safe_dist[safe_dist < np.inf])
    inv_dist = inv_dist * mean_valid  # Scale to [0, ~1] range
    
    base_score = inv_dist * rank_weights
    
    # Component weights adjusted by instance characteristics
    if cv < 0.3:  # Uniform distances
        regret_weight = 0.35
        embed_weight = 0.20
        mst_weight = 0.15
        base_weight = 0.30
    elif cv > 0.7:  # Highly varied distances
        regret_weight = 0.40
        embed_weight = 0.25
        mst_weight = 0.20
        base_weight = 0.15
    else:  # Mixed characteristics
        regret_weight = 0.35
        embed_weight = 0.25
        mst_weight = 0.20
        base_weight = 0.20
    
    # Apply regret penalty with exponential decay
    penalty_factor = np.exp(-2.5 * regret_matrix)
    
    # Combine components with adaptive weighting
    combined_score = (
        base_score * base_weight * penalty_factor +
        embedding_bonus * embed_weight +
        mst_bonus * mst_weight
    )
    
    # === 8. TWO-TIER SPARSIFICATION ===
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Tier 1: Apply candidate mask
    heuristics_matrix = combined_score * candidate_mask
    
    # Tier 2: Ensure each node has at least 2 connections
    for i in range(n):
        row = heuristics_matrix[i]
        if np.count_nonzero(row) < 2:
            # Find top 2 edges from combined score (pre-masking)
            top_scores = combined_score[i].copy()
            top_scores[i] = -np.inf
            top_indices = np.argsort(top_scores)[-2:]
            
            # Set to median of non-zero scores
            non_zero = heuristics_matrix[heuristics_matrix > 0]
            if len(non_zero) > 0:
                median_val = np.median(non_zero)
                heuristics_matrix[i, top_indices] = median_val
                heuristics_matrix[top_indices, i] = median_val
            else:
                # Fallback: set to reasonable value
                heuristics_matrix[i, top_indices] = 0.5
                heuristics_matrix[top_indices, i] = 0.5
    
    # === 9. FINAL PROCESSING ===
    # Symmetrize
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize with robust scaling
    valid_scores = heuristics_matrix[heuristics_matrix > 0]
    if len(valid_scores) > 0:
        p90 = np.percentile(valid_scores, 90)
        if p90 > 0:
            heuristics_matrix = np.clip(heuristics_matrix / p90, 0, 1)
    
    return heuristics_matrix
