import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection.
    
    Features:
    1. Combines multiple promising factors including:
       - Inverse distance (like v1)
       - Local/global average distance ratios
       - Minimum spanning tree component
       - Greedy path consistency
    2. Sparsifies by setting unpromising edges to zero
    3. Returns matrix of same shape with heuristic scores
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.ones_like(distance_matrix)
    
    # Initialize heuristic matrix
    heuristics = np.zeros_like(distance_matrix)
    
    # Avoid division by zero by replacing zeros (diagonal) with inf
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # Feature 1: Inverse distance (baseline)
    inv_dist = 1.0 / safe_dist
    
    # Feature 2: Local average ratio
    # For each node, compare edge distance to average distance from that node
    row_avgs = np.mean(safe_dist, axis=1, keepdims=True)
    col_avgs = np.mean(safe_dist, axis=0, keepdims=True)
    local_ratio = (row_avgs / safe_dist + col_avgs.T / safe_dist) / 2
    
    # Feature 3: Minimum spanning tree inspired scoring
    # Give bonus to edges that are close to minimum distances
    min_per_row = np.min(safe_dist, axis=1, keepdims=True)
    min_per_col = np.min(safe_dist, axis=0, keepdims=True)
    mst_bonus = (min_per_row / safe_dist + min_per_col.T / safe_dist) / 2
    
    # Feature 4: Two-step lookahead consistency
    # Consider if edge (i,j) connects to other promising edges
    consistency = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Look at other nodes k to see if edge (i,j) fits well in local structure
            other_nodes = [k for k in range(n) if k not in (i, j)]
            if other_nodes:
                # Check if distance(i,j) is less than average of distances through k
                alt_paths = []
                for k in other_nodes:
                    if safe_dist[i, k] < np.inf and safe_dist[k, j] < np.inf:
                        alt_paths.append(safe_dist[i, k] + safe_dist[k, j])
                if alt_paths:
                    avg_alt = np.mean(alt_paths)
                    consistency[i, j] = avg_alt / (2 * safe_dist[i, j]) if safe_dist[i, j] > 0 else 0
    
    # Combine features with weights
    weights = [0.4, 0.3, 0.2, 0.1]  # Tuned weights for each feature
    combined = (weights[0] * inv_dist +
                weights[1] * local_ratio +
                weights[2] * mst_bonus +
                weights[3] * consistency)
    
    # Sparsification: keep only top edges for each node
    # For each node, keep edges that are in top k or above threshold
    k = max(3, int(np.sqrt(n)))  # Adaptive k based on problem size
    threshold_factor = 0.7  # Keep edges above this fraction of max score per row
    
    for i in range(n):
        row_scores = combined[i, :].copy()
        row_scores[i] = -np.inf  # Exclude self
        
        # Method 1: Keep top k edges
        if row_scores.size > k:
            kth_largest = np.partition(row_scores, -k)[-k]
            top_k_mask = row_scores >= kth_largest
        else:
            top_k_mask = np.ones_like(row_scores, dtype=bool)
        
        # Method 2: Keep edges above threshold
        max_score = np.max(row_scores)
        if max_score > 0:
            threshold = threshold_factor * max_score
            threshold_mask = row_scores >= threshold
        else:
            threshold_mask = np.ones_like(row_scores, dtype=bool)
        
        # Combine both criteria
        keep_mask = top_k_mask & threshold_mask
        heuristics[i, :] = combined[i, :] * keep_mask
    
    # Symmetrize (since TSP is usually symmetric)
    heuristics = (heuristics + heuristics.T) / 2
    
    # Normalize for stability
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    # Set diagonal to zero
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
