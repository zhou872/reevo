import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge evaluation incorporating two-step lookahead
    mechanisms and adaptive instance-aware weighting.
    
    Core Innovations:
    1. Two-step Regret Lookahead: Evaluates edges based on downstream impact via
       greedy continuation paths from both endpoints
    2. Adaptive Cluster Potential: Dynamically identifies promising clusters and
       rewards intra-cluster edges based on multi-scale density estimation
    3. Instance Geometry Awareness: Adjusts factor weights based on the problem's
       spatial characteristics (uniformity, clustering tendency)
    4. Progressive Edge Pruning: Multi-stage candidate filtering with early
       elimination of clearly suboptimal edges
    """
    n = distance_matrix.shape[0]
    
    # Handle trivial cases
    if n <= 3:
        heuristics_matrix = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics_matrix, 0)
        return heuristics_matrix
    
    # Safe distance matrix
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # 1. Instance Geometry Analysis
    # Compute spatial characteristics to adapt heuristic weights
    flat_distances = safe_dist[safe_dist < np.inf]
    mean_dist = np.mean(flat_distances)
    std_dist = np.std(flat_distances)
    cv_dist = std_dist / (mean_dist + eps)  # Coefficient of variation
    
    # Clustering tendency: median nearest neighbor ratio
    nn_dists = np.zeros(n)
    for i in range(n):
        nn_dists[i] = np.min(safe_dist[i, safe_dist[i] < np.inf])
    clustering_ratio = np.median(nn_dists) / mean_dist
    
    # Determine adaptive weights based on instance characteristics
    # More uniform instances benefit more from lookahead, clustered instances from density
    if cv_dist < 0.3:
        w_lookahead = 1.8
        w_density = 0.7
    elif clustering_ratio < 0.6:
        w_lookahead = 1.2
        w_density = 1.5
    else:
        w_lookahead = 1.0
        w_density = 1.0
    
    # 2. Progressive Candidate Filtering
    # Stage 1: Conservative initial filter (top k1 neighbors)
    k1 = max(5, int(np.sqrt(n) * 1.2))
    candidate_mask1 = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        neighbors = np.argsort(safe_dist[i])[:k1]
        candidate_mask1[i, neighbors] = True
    candidate_mask1 = np.maximum(candidate_mask1, candidate_mask1.T)
    
    # Stage 2: Refined filter based on local competitiveness
    candidate_mask2 = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        row_dist = safe_dist[i]
        # Calculate competitiveness threshold
        sorted_dists = np.sort(row_dist[row_dist < np.inf])
        if len(sorted_dists) >= 10:
            p10_dist = sorted_dists[min(9, len(sorted_dists)-1)]
            competitiveness_thresh = p10_dist * 1.5
        else:
            competitiveness_thresh = np.mean(sorted_dists) * 1.8
        
        # Keep edges that are competitive
        competitive = row_dist < competitiveness_thresh
        candidate_mask2[i] = competitive & candidate_mask1[i]
    
    candidate_mask = np.maximum(candidate_mask2, candidate_mask2.T)
    
    # 3. Two-step Regret Lookahead
    lookahead_scores = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(i+1, n):
            if not candidate_mask[i, j]:
                continue
                
            # Greedy two-step continuation from i (excluding j)
            dist_i = safe_dist[i].copy()
            dist_i[j] = np.inf  # Exclude current edge
            # First step from i
            best_i1 = np.min(dist_i)
            if best_i1 < np.inf:
                k = np.argmin(dist_i)
                # Second step from k (excluding i and j)
                dist_k = safe_dist[k].copy()
                dist_k[i] = np.inf
                dist_k[j] = np.inf
                best_i2 = np.min(dist_k) if np.any(dist_k < np.inf) else 0
                lookahead_i = best_i1 + best_i2
            else:
                lookahead_i = 0
            
            # Greedy two-step continuation from j (excluding i)
            dist_j = safe_dist[j].copy()
            dist_j[i] = np.inf
            # First step from j
            best_j1 = np.min(dist_j)
            if best_j1 < np.inf:
                l = np.argmin(dist_j)
                # Second step from l (excluding i and j)
                dist_l = safe_dist[l].copy()
                dist_l[i] = np.inf
                dist_l[j] = np.inf
                best_j2 = np.min(dist_l) if np.any(dist_l < np.inf) else 0
                lookahead_j = best_j1 + best_j2
            else:
                lookahead_j = 0
            
            # Total lookahead cost: current edge + average continuation
            lookahead_cost = safe_dist[i, j] + 0.5 * (lookahead_i + lookahead_j)
            
            # Alternative: cost if we don't take this edge
            # Use second-best from i and j
            dist_i_alt = safe_dist[i].copy()
            dist_j_alt = safe_dist[j].copy()
            dist_i_alt[j] = np.inf
            dist_j_alt[i] = np.inf
            
            sorted_i = np.sort(dist_i_alt[dist_i_alt < np.inf])
            sorted_j = np.sort(dist_j_alt[dist_j_alt < np.inf])
            
            alt_cost_i = sorted_i[0] if len(sorted_i) > 0 else 0
            alt_cost_j = sorted_j[0] if len(sorted_j) > 0 than 0 else 0
            alternative_cost = (alt_cost_i + alt_cost_j) / 2
            
            # Regret: cost difference normalized by mean distance
            regret = (lookahead_cost - alternative_cost) / (mean_dist + eps)
            
            # Convert to score: exp(-regret^2) for smooth decay
            lookahead_scores[i, j] = np.exp(-w_lookahead * max(0, regret) ** 2)
    
    lookahead_scores = np.maximum(lookahead_scores, lookahead_scores.T)
    
    # 4. Adaptive Cluster Potential
    density_scores = np.zeros_like(distance_matrix)
    
    # Multi-scale density estimation
    scales = [min(3, n-1), min(5, n-1), min(8, n-1)]
    node_densities = np.zeros((len(scales), n))
    
    for scale_idx, k in enumerate(scales):
        for i in range(n):
            k_nearest = np.partition(safe_dist[i], k)[:k+1]
            valid_dists = k_nearest[k_nearest < np.inf]
            if len(valid_dists) > 0:
                node_densities[scale_idx, i] = k / (np.sum(valid_dists) + eps)
    
    # Normalize across scales
    for scale_idx in range(len(scales)):
        max_val = np.max(node_densities[scale_idx])
        if max_val > 0:
            node_densities[scale_idx] /= max_val
    
    # Combined density across scales (weighted by scale)
    combined_density = np.zeros(n)
    scale_weights = [0.2, 0.3, 0.5]  # Emphasize larger scales
    for i in range(n):
        combined_density[i] = sum(w * node_densities[idx, i] 
                                  for idx, w in enumerate(scale_weights))
    
    # Edge density score: geometric mean with distance adjustment
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Base density score
                density_mean = np.sqrt(combined_density[i] * combined_density[j] + eps)
                
                # Distance relative to local neighborhood
                local_mean_i = np.mean(safe_dist[i, candidate_mask[i]])
                local_mean_j = np.mean(safe_dist[j, candidate_mask[j]])
                rel_dist = (safe_dist[i, j] / (local_mean_i + eps) + 
                           safe_dist[i, j] / (local_mean_j + eps)) / 2
                
                # Combined score with distance penalty
                density_scores[i, j] = density_mean / (1 + 0.5 * rel_dist)
    
    density_scores = np.maximum(density_scores, density_scores.T)
    
    # Apply adaptive weight
    density_scores = density_scores ** w_density
    
    # 5. Base Inverse Distance with Rank Weighting
    inv_dist = np.zeros_like(distance_matrix)
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Normalized inverse distance
        row_mean = np.mean(safe_dist[i, candidate_mask[i]])
        inv_dist[i, :] = row_mean / (safe_dist[i, :] + eps)
        
        # Rank-based weighting
        sorted_idx = np.argsort(safe_dist[i])
        for rank, j in enumerate(sorted_idx):
            if i != j:
                # Adaptive decay based on local distance variance
                local_dists = safe_dist[i, candidate_mask[i]]
                if len(local_dists) > 1:
                    local_cv = np.std(local_dists) / (np.mean(local_dists) + eps)
                    decay_rate = 0.6 + 0.3 * np.clip(local_cv, 0, 1)
                else:
                    decay_rate = 0.7
                rank_weights[i, j] = decay_rate ** rank
    
    np.fill_diagonal(inv_dist, 0)
    
    # 6. Minimum Spanning Tree Bias (for moderate instances)
    mst_bonus = np.ones_like(distance_matrix)
    if n <= 2000:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_mask = (mst + mst.T) > 0
            # Modest bonus for MST edges
            mst_bonus[mst_mask] = 1.2
        except:
            pass
    
    # 7. Combine All Factors
    base_score = inv_dist * rank_weights
    heuristics_matrix = base_score * lookahead_scores * density_scores * mst_bonus
    
    # 8. Apply Candidate Mask
    heuristics_matrix[~candidate_mask] = 0
    
    # 9. Ensure Connectivity
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero_count = np.count_nonzero(row)
        
        if nonzero_count < 2:
            # Force top 2 edges by original distance
            top_indices = np.argsort(safe_dist[i])[:2]
            avg_nonzero = np.mean(row[row > 0]) if np.any(row > 0) else 1.0
            heuristics_matrix[i, top_indices] = avg_nonzero * 1.15
            heuristics_matrix[top_indices, i] = avg_nonzero * 1.15
    
    # 10. Final Processing
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
