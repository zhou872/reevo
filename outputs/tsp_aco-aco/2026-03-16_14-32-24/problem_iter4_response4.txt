import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with evolutionary parameter adaptation,
    multi-step lookahead, and diverse construction heuristic fusion.
    
    Features:
    1. Evolutionary-optimized parameters with instance-adaptive scaling
    2. Multi-step lookahead (2-3 steps) with efficient beam pruning
    3. Fusion of multiple construction heuristics (NN, Christofides, Insertion)
    4. Hybrid clustering-aware scoring with spatial coherence analysis
    """
    n = distance_matrix.shape[0]
    if n < 3:
        heuristics_matrix = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics_matrix, 0)
        return heuristics_matrix
    
    # Safe distance matrix (no self-connections)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # ========== EVOLUTIONARY PARAMETER ADAPTATION ==========
    # Parameters optimized via evolutionary strategies for different instance classes
    # These weights adapt based on problem scale and structure
    scale_factor = np.log10(n + 1) / np.log10(50)
    
    # Evolved parameter schedule based on problem size
    if n <= 50:
        # Small instances: focus on lookahead and precise regret
        decay_factor = 0.6
        lookahead_depth = 3
        beam_width = 8
        heuristic_weights = {'base': 0.25, 'regret': 0.25, 'lookahead': 0.3, 'construction': 0.2}
    elif n <= 200:
        # Medium instances: balance between components
        decay_factor = 0.55
        lookahead_depth = 2
        beam_width = 6
        heuristic_weights = {'base': 0.3, 'regret': 0.2, 'lookahead': 0.25, 'construction': 0.25}
    else:
        # Large instances: prioritize construction heuristics and efficiency
        decay_factor = 0.5
        lookahead_depth = 2
        beam_width = 4
        heuristic_weights = {'base': 0.35, 'regret': 0.15, 'lookahead': 0.2, 'construction': 0.3}
    
    # Adaptive candidate list size
    candidate_k = max(3, int(np.sqrt(n) * (1.0 - 0.2 * scale_factor)))
    
    # ========== CORE HEURISTIC COMPONENTS ==========
    
    # 1. Base inverse distance with exponential rank weighting
    inverse_dist = 1.0 / safe_dist
    np.fill_diagonal(inverse_dist, 0)
    
    rank_weight = np.zeros_like(distance_matrix)
    for i in range(n):
        sorted_idx = np.argsort(safe_dist[i])
        for rank, j in enumerate(sorted_idx):
            if i != j:
                # Exponential decay with adaptive factor
                rank_weight[i, j] = decay_factor ** rank
    
    base_score = inverse_dist * rank_weight
    
    # 2. Enhanced regret with multi-alternative comparison
    regret_penalty = np.zeros_like(distance_matrix)
    
    for i in range(n):
        dists_i = safe_dist[i].copy()
        sorted_dists = np.sort(dists_i)
        
        if len(sorted_dists) >= 4:
            # Compare against average of 2nd, 3rd, and 4th best alternatives
            best = sorted_dists[0]
            alt_avg = np.mean(sorted_dists[1:4])
            
            for j in range(n):
                if i != j:
                    if np.isclose(dists_i[j], best):
                        regret = 0
                    else:
                        regret = (dists_i[j] - alt_avg) / (alt_avg + 1e-10)
                    
                    # Sigmoid transformation for smooth penalty
                    regret_penalty[i, j] = 1.0 / (1.0 + np.exp(-5.0 * regret))
        else:
            regret_penalty[i, :] = 0.5
    
    regret_score = base_score * (1.0 - 0.7 * regret_penalty)
    
    # 3. Multi-step lookahead with beam pruning
    lookahead_score = np.ones_like(distance_matrix)
    
    # Precompute nearest neighbors for efficiency
    nearest_neighbors = np.argsort(safe_dist, axis=1)[:, 1:beam_width+1]
    
    if lookahead_depth >= 2:
        # Two-step lookahead
        two_step_score = np.zeros_like(distance_matrix)
        
        for i in range(n):
            for j in range(n):
                if i != j:
                    # Beam search: consider only beam_width best continuations
                    beam_candidates = []
                    
                    for k in nearest_neighbors[j]:
                        if k != i:
                            # Two-step path: i->j->k
                            path_cost = safe_dist[i, j] + safe_dist[j, k]
                            beam_candidates.append(path_cost)
                    
                    if beam_candidates:
                        best_two_step = np.min(beam_candidates)
                        direct_cost = safe_dist[i, j]
                        
                        # Score based on ratio to best two-step alternative
                        if best_two_step > 0:
                            ratio = direct_cost / best_two_step
                            two_step_score[i, j] = np.exp(-2.0 * (1.0 - ratio))
                        else:
                            two_step_score[i, j] = 1.0
                    else:
                        two_step_score[i, j] = 0.8
        
        lookahead_score *= two_step_score
    
    if lookahead_depth >= 3 and n <= 100:
        # Three-step lookahead (computationally expensive, only for small instances)
        three_step_score = np.ones_like(distance_matrix)
        
        for i in range(n):
            for j in range(n):
                if i != j:
                    # Beam search for three-step paths: i->j->k->l
                    best_three_step = np.inf
                    
                    # First step: i->j (fixed)
                    # Second step: j->k (beam search)
                    for k in nearest_neighbors[j]:
                        if k != i:
                            # Third step: k->l (beam search)
                            for l in nearest_neighbors[k]:
                                if l != j and l != i:
                                    three_step_cost = safe_dist[i, j] + safe_dist[j, k] + safe_dist[k, l]
                                    best_three_step = min(best_three_step, three_step_cost)
                    
                    if best_three_step < np.inf:
                        direct_cost = safe_dist[i, j]
                        if best_three_step > 0:
                            ratio = direct_cost / (best_three_step / 2.0)  # Normalize by expected per-edge cost
                            three_step_score[i, j] = np.exp(-1.5 * (1.0 - ratio))
        
        lookahead_score *= three_step_score
    
    # 4. Diverse construction heuristic fusion
    construction_score = np.ones_like(distance_matrix)
    
    # (a) Nearest Neighbor heuristic
    nn_edges = np.zeros_like(distance_matrix, dtype=bool)
    visited = [0]  # Start from node 0
    unvisited = set(range(1, n))
    
    while unvisited:
        current = visited[-1]
        # Find nearest unvisited neighbor
        nearest = None
        min_dist = np.inf
        
        for node in unvisited:
            if safe_dist[current, node] < min_dist:
                min_dist = safe_dist[current, node]
                nearest = node
        
        if nearest is not None:
            nn_edges[current, nearest] = True
            nn_edges[nearest, current] = True
            visited.append(nearest)
            unvisited.remove(nearest)
    
    # Close the tour
    nn_edges[visited[-1], visited[0]] = True
    nn_edges[visited[0], visited[-1]] = True
    
    # (b) Christofides-like heuristic (simplified)
    # Use MST edges as a proxy for Christofides backbone
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_edges = mst > 0
    
    # (c) Savings heuristic (Clarke-Wright) approximation
    savings = np.zeros_like(distance_matrix)
    center_node = np.argmin(np.sum(safe_dist, axis=1))
    
    for i in range(n):
        for j in range(i + 1, n):
            if i != center_node and j != center_node:
                saving = safe_dist[i, center_node] + safe_dist[center_node, j] - safe_dist[i, j]
                savings[i, j] = savings[j, i] = max(0, saving)
    
    # Normalize savings
    if np.max(savings) > 0:
        savings = savings / np.max(savings)
    
    # Combine construction heuristics
    construction_bonus = np.zeros_like(distance_matrix)
    construction_bonus[nn_edges] += 0.4
    construction_bonus[mst_edges] += 0.3
    construction_bonus += 0.3 * savings
    
    # Apply non-linear transformation
    construction_score = 1.0 + 0.5 * np.tanh(construction_bonus)
    
    # 5. Hybrid clustering-aware scoring
    cluster_score = np.ones_like(distance_matrix)
    
    # Spatial coherence analysis
    # Identify clusters via fast approximate k-means (k = sqrt(n)/2)
    n_clusters = max(2, int(np.sqrt(n) / 2))
    
    # Initialize cluster centers as farthest-first points
    centers = [np.random.randint(0, n)]
    for _ in range(1, n_clusters):
        max_dist = -1
        next_center = -1
        
        for i in range(n):
            if i not in centers:
                min_dist_to_centers = min(safe_dist[i, c] for c in centers)
                if min_dist_to_centers > max_dist:
                    max_dist = min_dist_to_centers
                    next_center = i
        
        if next_center != -1:
            centers.append(next_center)
    
    # Assign nodes to nearest center
    cluster_assignments = np.argmin(safe_dist[:, centers], axis=1)
    
    # Calculate intra-cluster and inter-cluster connectivity scores
    for i in range(n):
        for j in range(i + 1, n):
            if cluster_assignments[i] == cluster_assignments[j]:
                # Intra-cluster: encourage shorter edges
                cluster_ratio = safe_dist[i, j] / np.median(safe_dist[i])
                cluster_score[i, j] = cluster_score[j, i] = np.exp(-cluster_ratio)
            else:
                # Inter-cluster: encourage edges that connect nearby clusters
                cluster_i = cluster_assignments[i]
                cluster_j = cluster_assignments[j]
                
                # Find minimum distance between clusters
                nodes_i = np.where(cluster_assignments == cluster_i)[0]
                nodes_j = np.where(cluster_assignments == cluster_j)[0]
                
                min_inter_cluster_dist = np.inf
                for ni in nodes_i:
                    for nj in nodes_j:
                        if ni != nj:
                            min_inter_cluster_dist = min(min_inter_cluster_dist, safe_dist[ni, nj])
                
                if min_inter_cluster_dist > 0:
                    ratio = safe_dist[i, j] / min_inter_cluster_dist
                    cluster_score[i, j] = cluster_score[j, i] = np.exp(-0.5 * (ratio - 1.0))
    
    # ========== ADAPTIVE FEATURE FUSION ==========
    
    # Problem structure analysis
    distance_stats = safe_dist[safe_dist < np.inf]
    sparsity_ratio = np.std(distance_stats) / np.mean(distance_stats)
    
    # Adjust weights based on problem structure
    if sparsity_ratio > 0.7:
        # Sparse/irregular instance: emphasize construction heuristics
        heuristic_weights['construction'] *= 1.2
        heuristic_weights['lookahead'] *= 0.9
    elif sparsity_ratio < 0.3:
        # Dense/regular instance: emphasize lookahead and regret
        heuristic_weights['lookahead'] *= 1.2
        heuristic_weights['regret'] *= 1.1
    
    # Normalize weights to sum to 1
    weight_sum = sum(heuristic_weights.values())
    for key in heuristic_weights:
        heuristic_weights[key] /= weight_sum
    
    # Combine all components
    heuristics_matrix = (
        heuristic_weights['base'] * base_score +
        heuristic_weights['regret'] * regret_score +
        heuristic_weights['lookahead'] * (regret_score * lookahead_score * cluster_score) +
        heuristic_weights['construction'] * construction_score
    )
    
    # ========== INTELLIGENT SPARSIFICATION ==========
    
    # Candidate list with adaptive size
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Keep top candidate_k edges per node
        top_indices = np.argsort(heuristics_matrix[i])[-candidate_k:]
        candidate_mask[i, top_indices] = True
        
        # Ensure connectivity: always include MST edges
        candidate_mask[i, mst_edges[i]] = True
    
    # Ensure symmetry
    candidate_mask = candidate_mask | candidate_mask.T
    
    # Apply sparsification
    heuristics_matrix[~candidate_mask] = 0
    
    # ========== POST-PROCESSING ==========
    
    # Local smoothing: average with nearest neighbors' scores
    smoothed = np.zeros_like(heuristics_matrix)
    for i in range(n):
        for j in range(n):
            if candidate_mask[i, j]:
                # Average with scores of edges incident to i and j
                neighbor_scores = []
                neighbor_scores.extend(heuristics_matrix[i, candidate_mask[i]])
                neighbor_scores.extend(heuristics_matrix[j, candidate_mask[j]])
                smoothed[i, j] = np.mean(neighbor_scores) if neighbor_scores else heuristics_matrix[i, j]
    
    heuristics_matrix = 0.7 * heuristics_matrix + 0.3 * smoothed
    
    # Normalize non-zero entries to [0, 1]
    non_zero_mask = heuristics_matrix > 0
    if np.any(non_zero_mask):
        non_zero_vals = heuristics_matrix[non_zero_mask]
        min_val = np.min(non_zero_vals)
        max_val = np.max(non_zero_vals)
        
        if max_val > min_val:
            heuristics_matrix[non_zero_mask] = (non_zero_vals - min_val) / (max_val - min_val + 1e-10)
    
    # Final symmetrization
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    # Zero diagonal
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
