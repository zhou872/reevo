import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Nearest-neighbor gradient: Measures how much better an edge is compared to alternatives
    2. Two-hop density: Considers path quality through intermediate nodes
    3. Cost-regret sparsification: Zeroes out edges with poor combined metrics
    4. Symmetry normalization: Ensures balanced consideration of directed edges
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (handling zeros)
    eps = 1e-10
    inv_dist = 1 / (distance_matrix + eps)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Nearest-neighbor gradient factor
    # For each node, how much better is this edge than the average alternative?
    row_mins = np.min(distance_matrix + np.eye(n) * np.inf, axis=1, keepdims=True)
    gradient_factor = row_mins / (distance_matrix + eps)
    gradient_factor = np.clip(gradient_factor, 0, 2)  # Cap at 2x better than minimum
    
    # 3. Two-hop path density factor
    # Measures if an edge connects to nodes with good onward connections
    density_matrix = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Find best onward connection from both ends
            mask_i = np.ones(n, dtype=bool)
            mask_i[[i, j]] = False
            if np.any(mask_i):
                best_from_i = np.min(distance_matrix[i, mask_i])
            else:
                best_from_i = distance_matrix[i, j]
                
            mask_j = np.ones(n, dtype=bool)
            mask_j[[i, j]] = False
            if np.any(mask_j):
                best_from_j = np.min(distance_matrix[j, mask_j])
            else:
                best_from_j = distance_matrix[i, j]
                
            density_matrix[i, j] = 2 / (best_from_i + best_from_j + eps)
    
    # Normalize density factor
    if np.max(density_matrix) > 0:
        density_matrix /= np.max(density_matrix)
    
    # 4. Combine factors with weighted geometric mean (preserves zero relationships)
    combined = np.zeros((n, n))
    mask = distance_matrix > 0
    combined[mask] = (
        inv_dist[mask] ** 0.4 * 
        gradient_factor[mask] ** 0.3 * 
        density_matrix[mask] ** 0.3
    )
    
    # 5. Symmetry-aware normalization
    # Consider both directions but don't double count
    symmetric_score = (combined + combined.T) / 2
    np.fill_diagonal(symmetric_score, 0)
    
    # 6. Sparsification: Keep only top edges
    # Strategy: For each node, keep edges to top 30% of other nodes by score
    percentile_threshold = 70  # Keep edges better than 70th percentile per row
    sparse_matrix = np.zeros((n, n))
    
    for i in range(n):
        row_scores = symmetric_score[i].copy()
        threshold = np.percentile(row_scores[row_scores > 0], percentile_threshold)
        sparse_matrix[i] = np.where(row_scores >= threshold, row_scores, 0)
    
    # Also keep globally top 15% of all edges
    all_scores = symmetric_score[symmetric_score > 0].flatten()
    if len(all_scores) > 0:
        global_threshold = np.percentile(all_scores, 85)
        sparse_matrix = np.where(symmetric_score >= global_threshold, sparse_matrix, 0)
    
    # Ensure symmetry after sparsification
    final_matrix = (sparse_matrix + sparse_matrix.T) / 2
    np.fill_diagonal(final_matrix, 0)
    
    return final_matrix
