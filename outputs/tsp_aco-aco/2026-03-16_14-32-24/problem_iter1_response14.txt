import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors.
    Returns a matrix of the same shape indicating edge promise scores.
    """
    n = distance_matrix.shape[0]
    np.fill_diagonal(distance_matrix, np.inf)  # Avoid self-loops
    
    # 1. Base inverse distance weighting (with smoothing)
    epsilon = 1e-10
    inv_dist = 1 / (distance_matrix + epsilon)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = (mst + mst.T) > 0  # Symmetrize MST edges
    
    # 3. Two-step edge potential: consider intermediate nodes
    two_step_potential = np.zeros((n, n))
    for i in range(n):
        # Find best intermediate node k for path i -> k -> j
        for j in range(n):
            if i == j:
                continue
            # Look for k that minimizes i->k + k->j - i->j (savings)
            k_mask = np.ones(n, dtype=bool)
            k_mask[[i, j]] = False
            potential_savings = distance_matrix[i, :] + distance_matrix[:, j] - distance_matrix[i, j]
            two_step_potential[i, j] = np.max(potential_savings[k_mask])
    
    # Normalize two-step potential
    if np.max(two_step_potential) > 0:
        two_step_potential = two_step_potential / np.max(two_step_potential)
    
    # 4. Combine factors with weights
    weights = [0.5, 0.3, 0.2]  # Tune these weights
    heuristics = (
        weights[0] * inv_dist +
        weights[1] * mst_bonus +
        weights[2] * two_step_potential
    )
    
    # 5. Sparsification: keep only promising edges
    # Keep top-k edges per node based on nearest neighbors
    k = max(3, n // 4)  # Adaptive k
    mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get top k edges for this node (excluding diagonal)
        top_k_indices = np.argpartition(heuristics[i], -k)[-k:]
        mask[i, top_k_indices] = True
    
    # Symmetrize the mask
    mask = np.logical_or(mask, mask.T)
    heuristics = heuristics * mask
    
    # 6. Final normalization and zero diagonal
    np.fill_diagonal(heuristics, 0)
    if np.max(heuristics) > 0:
        heuristics = heuristics / np.max(heuristics)
    
    return heuristics
