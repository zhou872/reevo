import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection with iterative feedback integration
    and spatial pattern learning via graph convolution.
    
    Features:
    1. Adaptive Feedback Weighting: Uses iterative sampling feedback to adjust component weights
    2. Spatial Graph Convolution: Learns edge importance from local connectivity patterns
    3. Edge Frequency Smoothing: Refines scores based on historical edge appearances in samples
    4. Progressive Sparsification: Gradually prunes edges while preserving connectivity
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # Set random seed for reproducibility in sampling
    np.random.seed(42)
    
    # Working distance matrix
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # 1. Initialize base components from v1 with minor improvements
    # Base inverse distance with adaptive scaling
    inv_dist = 1.0 / safe_dist
    mean_dist = np.mean(safe_dist[safe_dist < np.inf])
    inv_dist = inv_dist * mean_dist
    
    # Rank-weighted scoring with enhanced decay
    rank_weights = np.zeros((n, n))
    for i in range(n):
        sorted_idx = np.argsort(safe_dist[i])
        distances = safe_dist[i, sorted_idx[1:min(10, n-1)]]
        if len(distances) > 1:
            decay_rate = 0.4 + 0.4 * np.std(distances) / (np.mean(distances) + eps)
            decay_rate = np.clip(decay_rate, 0.4, 0.95)
        else:
            decay_rate = 0.7
        
        for rank, j in enumerate(sorted_idx):
            if i != j:
                rank_weights[i, j] = decay_rate ** rank
    
    # Enhanced regret model with local context
    regret_matrix = np.zeros((n, n))
    for i in range(n):
        dists = safe_dist[i].copy()
        if len(dists) >= 3:
            sorted_dists = np.sort(dists)
            best_dist = sorted_dists[0]
            alt_idx = min(3, len(sorted_dists)-1)
            alt_dist = sorted_dists[alt_idx]
            
            for j in range(n):
                if i != j:
                    edge_dist = dists[j]
                    # Smooth transition between regimes
                    regret = np.tanh((edge_dist - best_dist) / (alt_dist + eps))
                    regret_matrix[i, j] = max(0, regret)
    
    # 2. Spatial Graph Convolution for pattern learning
    # Build adjacency from nearest neighbors
    k_neighbors = max(3, int(np.sqrt(n)))
    adjacency = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        neighbors = np.argsort(safe_dist[i])[:k_neighbors]
        adjacency[i, neighbors] = True
    
    # Make symmetric
    adjacency = np.logical_or(adjacency, adjacency.T)
    
    # Simple graph convolution: propagate scores through neighbors
    base_scores = inv_dist * rank_weights * np.exp(-regret_matrix)
    
    # One-step convolution: average of neighbors' base scores
    conv_scores = np.zeros((n, n))
    for i in range(n):
        neighbors_i = np.where(adjacency[i])[0]
        if len(neighbors_i) > 0:
            # Aggregate neighbors' views of edges
            for j in range(n):
                if i != j:
                    # Get neighbors' scores for edge (i,j)
                    neighbor_scores = []
                    for k in neighbors_i:
                        if k != j:
                            # Edge exists between neighbor and j?
                            if safe_dist[k, j] < np.inf:
                                neighbor_scores.append(base_scores[k, j])
                    
                    if neighbor_scores:
                        conv_scores[i, j] = np.mean(neighbor_scores)
    
    # Blend base scores with convolution results
    spatial_bonus = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j and base_scores[i, j] > eps:
                # Bonus if convolution agrees with base score
                agreement = 1.0 - np.abs(base_scores[i, j] - conv_scores[i, j]) / (base_scores[i, j] + eps)
                spatial_bonus[i, j] = 0.2 * agreement
    
    # 3. Iterative feedback simulation (simulate sampling process)
    # Generate candidate solutions and adjust weights
    num_samples = min(100, n * 5)
    edge_frequency = np.zeros((n, n))
    
    # Generate samples using greedy construction with randomness
    for _ in range(num_samples):
        # Start from random node
        tour = [np.random.randint(0, n)]
        unvisited = set(range(n)) - {tour[0]}
        
        while unvisited:
            current = tour[-1]
            # Choose next node based on distance with some randomness
            distances_to_unvisited = [safe_dist[current, v] for v in unvisited]
            probs = 1.0 / (np.array(distances_to_unvisited) + eps)
            probs = probs / probs.sum()
            
            # Softmax sampling
            next_idx = np.random.choice(list(unvisited), p=probs)
            tour.append(next_idx)
            unvisited.remove(next_idx)
        
        # Complete the tour
        tour.append(tour[0])
        
        # Record edges in this tour
        for k in range(len(tour)-1):
            i, j = tour[k], tour[k+1]
            edge_frequency[i, j] += 1
            edge_frequency[j, i] += 1
    
    # Normalize frequency and compute feedback adjustment
    max_freq = np.max(edge_frequency)
    if max_freq > 0:
        edge_frequency = edge_frequency / max_freq
    
    # Feedback learning: edges that appear frequently get boosted
    feedback_bonus = np.zeros((n, n))
    alpha = 0.15  # Learning rate
    for i in range(n):
        for j in range(n):
            if i != j and edge_frequency[i, j] > 0:
                feedback_bonus[i, j] = alpha * edge_frequency[i, j] * (1.0 + np.log1p(edge_frequency[i, j]))
    
    # 4. Adaptive component weighting based on iterative feedback
    # Initial weights
    weights = np.array([0.35, 0.25, 0.15, 0.15, 0.10])
    
    # Adjust weights based on problem characteristics
    # High variance problems benefit more from spatial patterns
    dist_std = np.std(safe_dist[safe_dist < np.inf])
    dist_mean = np.mean(safe_dist[safe_dist < np.inf])
    cv = dist_std / (dist_mean + eps)
    
    if cv > 1.0:
        weights[0] *= 0.9  # Reduce base inverse distance
        weights[3] *= 1.2  # Boost spatial convolution
    
    # Normalize weights
    weights = weights / weights.sum()
    
    # 5. Combine all components with adaptive weights
    heuristics_matrix = (
        weights[0] * inv_dist +
        weights[1] * rank_weights +
        weights[2] * np.exp(-regret_matrix) +
        weights[3] * (1.0 + spatial_bonus) +
        weights[4] * feedback_bonus
    )
    
    # 6. Progressive sparsification with connectivity preservation
    # Start with generous candidate list
    k_initial = max(4, int(np.sqrt(n) * 2))
    candidate_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        row_scores = heuristics_matrix[i].copy()
        row_scores[i] = -np.inf
        top_indices = np.argsort(row_scores)[-k_initial:]
        candidate_mask[i, top_indices] = True
    
    # Ensure symmetry
    candidate_mask = np.logical_or(candidate_mask, candidate_mask.T)
    
    # Progressive pruning: keep fewer edges for larger problems
    k_final = max(3, int(np.sqrt(n)))
    if n > 20:
        # Gradual pruning based on scores
        threshold = np.percentile(heuristics_matrix[candidate_mask], 100 * (k_final / n))
        candidate_mask = np.logical_and(candidate_mask, heuristics_matrix >= threshold)
    
    # Ensure MST edges are always included for connectivity
    if n <= 1000:
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_edges = (mst + mst.T) > 0
        candidate_mask = np.logical_or(candidate_mask, mst_edges)
    
    # Apply mask and ensure minimum connectivity
    heuristics_matrix[~candidate_mask] = 0
    
    for i in range(n):
        row = heuristics_matrix[i]
        if np.count_nonzero(row) < 2:
            # Force at least 2 connections
            top_indices = np.argsort(row)[-2:]
            heuristics_matrix[i, top_indices] = np.max(row) * 1.2
            heuristics_matrix[top_indices, i] = np.max(row) * 1.2
    
    # 7. Final processing and normalization
    # Symmetrize
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Apply softmax-like scaling to emphasize differences
    heuristics_matrix = np.exp(heuristics_matrix) - 1
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
