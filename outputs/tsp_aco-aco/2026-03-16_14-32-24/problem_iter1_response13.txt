import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection.
    Combines multiple factors to score edges and sparsifies the matrix
    by setting unpromising edges to zero.
    """
    n = distance_matrix.shape[0]
    if n <= 3:
        # Fallback for very small instances
        return np.ones_like(distance_matrix) / (distance_matrix + np.eye(n))
    
    # 1. Base inverse distance (handling zeros on diagonal)
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    mask = distance_matrix > 0
    inv_dist[mask] = 1.0 / distance_matrix[mask]
    
    # 2. Constrained k-nearest neighbor sparsification
    k = max(2, int(np.sqrt(n)))  # Adaptive k based on problem size
    knn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k+1 smallest distances (including self)
        indices = np.argpartition(distance_matrix[i], k+1)[:k+1]
        # Remove self if present
        indices = indices[indices != i][:k]
        knn_mask[i, indices] = True
    
    # Make symmetric (undirected TSP)
    knn_mask = knn_mask | knn_mask.T
    
    # 3. Local clustering coefficient bonus
    # For each edge (i,j), count how many common neighbors they share
    clustering_bonus = np.zeros_like(distance_matrix, dtype=float)
    for i in range(n):
        for j in range(i+1, n):
            if knn_mask[i, j]:
                common_neighbors = np.sum(knn_mask[i] & knn_mask[j])
                # Normalize by maximum possible common neighbors
                max_common = min(np.sum(knn_mask[i]), np.sum(knn_mask[j])) - 1
                if max_common > 0:
                    clustering_bonus[i, j] = clustering_bonus[j, i] = 0.5 * (common_neighbors / max_common)
    
    # 4. Global minimum spanning tree alignment
    # Compute MST and give bonus to edges in it
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = (mst > 0).astype(float) * 0.3
    
    # 5. Distance quantile thresholding
    # Calculate distance distribution and threshold
    flat_dists = distance_matrix[np.triu_indices(n, k=1)]
    if len(flat_dists) > 0:
        # Keep only edges below 75th percentile distance
        threshold = np.quantile(flat_dists, 0.75)
        dist_threshold_mask = distance_matrix <= threshold
        # Don't include diagonal
        np.fill_diagonal(dist_threshold_mask, False)
    else:
        dist_threshold_mask = np.ones_like(distance_matrix, dtype=bool)
        np.fill_diagonal(dist_threshold_mask, False)
    
    # Combine all factors
    heuristics = inv_dist * (1.0 + clustering_bonus + mst_bonus)
    
    # Apply sparsification masks
    final_mask = knn_mask & dist_threshold_mask
    heuristics = heuristics * final_mask
    
    # Ensure symmetry and zero diagonal
    heuristics = np.maximum(heuristics, heuristics.T)
    np.fill_diagonal(heuristics, 0.0)
    
    return heuristics
