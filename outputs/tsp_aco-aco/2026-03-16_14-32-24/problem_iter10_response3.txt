import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic scoring for TSP edge evaluation using bilateral regret
    minimization, adaptive distance spectrum weighting, and intelligent sparsification.
    
    Key Innovations:
    1. Bilateral Regret Minimization: Combines regret from both nodes for each edge
    2. Adaptive Weighting: Uses distance distribution statistics to adjust rank decay
    3. Neighborhood Sparsification: Adaptive candidate selection based on distance ratios
    4. Structural Bias: MST-based reinforcement with degree preservation
    """
    n = distance_matrix.shape[0]
    eps = 1e-12
    
    # 1. Initialize safe distance matrix
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 2. Adaptive candidate list using distance ratio threshold
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        dists = safe_dist[i].copy()
        if len(dists[dists < np.inf]) >= 2:
            sorted_dists = np.sort(dists[dists < np.inf])
            best = sorted_dists[0]
            second_best = sorted_dists[1]
            
            # Dynamic threshold: include edges within factor of best distance
            threshold_factor = 2.5 - 0.5 * min(1.0, (second_best - best) / (best + eps))
            threshold = best * threshold_factor
            
            candidate_mask[i, dists <= threshold] = True
    
    # Enforce symmetry
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # 3. Bilateral Regret Minimization
    regret_matrix = np.ones_like(distance_matrix)
    
    # Precompute best and second best for all nodes
    best_dists = np.zeros(n)
    second_best_dists = np.zeros(n)
    
    for i in range(n):
        dists = safe_dist[i].copy()
        sorted_dists = np.sort(dists[dists < np.inf])
        if len(sorted_dists) >= 2:
            best_dists[i] = sorted_dists[0]
            second_best_dists[i] = sorted_dists[1]
        elif len(sorted_dists) == 1:
            best_dists[i] = sorted_dists[0]
            second_best_dists[i] = sorted_dists[0] * 1.5
        else:
            best_dists[i] = np.mean(safe_dist[safe_dist < np.inf])
            second_best_dists[i] = best_dists[i] * 1.5
    
    # Compute bilateral regret for each candidate edge
    for i in range(n):
        for j in range(n):
            if i != j and candidate_mask[i, j]:
                d_ij = safe_dist[i, j]
                
                # Regret from i's perspective
                if second_best_dists[i] - best_dists[i] > eps:
                    regret_i = (d_ij - best_dists[i]) / (second_best_dists[i] - best_dists[i] + eps)
                else:
                    regret_i = (d_ij - best_dists[i]) / (best_dists[i] + eps)
                
                # Regret from j's perspective
                if second_best_dists[j] - best_dists[j] > eps:
                    regret_j = (d_ij - best_dists[j]) / (second_best_dists[j] - best_dists[j] + eps)
                else:
                    regret_j = (d_ij - best_dists[j]) / (best_dists[j] + eps)
                
                # Combined bilateral regret (geometric mean)
                combined_regret = np.sqrt((regret_i + eps) * (regret_j + eps))
                
                # Exponential penalty with smoothing
                regret_matrix[i, j] = np.exp(-combined_regret)
    
    # 4. Adaptive Rank Weighting with Distance Spectrum Analysis
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        dists = safe_dist[i].copy()
        sorted_idx = np.argsort(dists)
        
        # Analyze distance distribution
        valid_dists = dists[dists < np.inf]
        if len(valid_dists) >= 5:
            # Use quartiles to understand distribution shape
            q1 = np.percentile(valid_dists, 25)
            q3 = np.percentile(valid_dists, 75)
            iqr = q3 - q1
            
            # Determine decay rate based on distribution characteristics
            if iqr < eps:
                decay_rate = 0.3  # Uniform distances
            else:
                skewness = (np.mean(valid_dists) - np.median(valid_dists)) / (np.std(valid_dists) + eps)
                # Map skewness to decay rate: positive skew -> faster decay
                decay_rate = 0.3 + 0.55 / (1 + np.exp(-5 * skewness))
        else:
            decay_rate = 0.7
        
        # Apply exponential rank weighting
        for rank, j in enumerate(sorted_idx):
            if i != j and candidate_mask[i, j]:
                rank_weights[i, j] = decay_rate ** rank
    
    # 5. Inverse Distance Scoring with Adaptive Normalization
    inv_dist = np.zeros_like(distance_matrix)
    
    for i in range(n):
        row = safe_dist[i]
        if np.any(row < np.inf):
            # Use median for robust scaling
            valid_dists = row[row < np.inf]
            scale_factor = np.median(valid_dists) if len(valid_dists) > 0 else 1.0
            
            # Inverse distance with adaptive scaling
            inv_dist[i] = scale_factor / (row + eps)
    
    # 6. Minimum Spanning Tree Structural Reinforcement
    mst_bias = np.zeros_like(distance_matrix)
    if n <= 3000 and n >= 3:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_edges = (mst + mst.T) > 0
            
            # Stronger bias for MST edges that connect low-degree nodes
            degree_bias = np.zeros(n)
            for i in range(n):
                degree_bias[i] = np.sum(candidate_mask[i]) / (n - 1)
            
            for i in range(n):
                for j in range(n):
                    if mst_edges[i, j]:
                        # Higher bias for edges connecting nodes with few candidates
                        bias_strength = 0.4 * (1 - 0.5 * (degree_bias[i] + degree_bias[j]))
                        mst_bias[i, j] = bias_strength
        except:
            pass
    
    # 7. Multi-Factor Integration
    # Base score combines inverse distance and rank weights
    base_score = inv_dist * rank_weights
    
    # Apply regret penalty
    score_with_regret = base_score * regret_matrix
    
    # Add MST bias
    heuristics_matrix = score_with_regret * (1.0 + mst_bias)
    
    # 8. Connectivity Preservation and Sparsification
    heuristics_matrix[~candidate_mask] = 0
    
    # Ensure minimum degree of 3 for each node
    for i in range(n):
        row_scores = heuristics_matrix[i]
        nonzero_count = np.sum(row_scores > 0)
        
        if nonzero_count < 3:
            # Find additional promising edges
            potential_edges = []
            for j in range(n):
                if i != j and not candidate_mask[i, j]:
                    # Score based on combined factors
                    edge_score = (rank_weights[i, j] * regret_matrix[i, j] * 
                                 (1.0 / (safe_dist[i, j] + eps)))
                    potential_edges.append((j, edge_score))
            
            if potential_edges:
                potential_edges.sort(key=lambda x: x[1], reverse=True)
                needed = min(3 - nonzero_count, len(potential_edges))
                
                for k in range(needed):
                    j, score_val = potential_edges[k]
                    boost_value = np.max(row_scores) * 1.5 if np.max(row_scores) > 0 else 1.0
                    heuristics_matrix[i, j] = boost_value
                    heuristics_matrix[j, i] = boost_value
                    candidate_mask[i, j] = candidate_mask[j, i] = True
    
    # 9. Symmetrization and Final Normalization
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize with power transformation to emphasize differences
    max_val = np.max(heuristics_matrix)
    if max_val > eps:
        heuristics_matrix = heuristics_matrix / max_val
        # Apply adaptive power scaling based on score distribution
        score_std = np.std(heuristics_matrix[heuristics_matrix > 0])
        if score_std > 0.1:
            power = 0.7  # More compression for varied scores
        else:
            power = 0.9  # Less compression for similar scores
        heuristics_matrix = np.power(heuristics_matrix, power)
    
    return heuristics_matrix
