import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic matrix for TSP edge selection with extended regret modeling,
    dynamic parameter tuning, and lookahead path analysis.
    
    Features:
    1. Extended k-regret penalty using multiple alternative edges for robustness
    2. Dynamic exponential decay factors based on instance-specific distance distribution
    3. Two-step lookahead path potential via optimal intermediate node selection
    4. Adaptive cluster density scoring with normalization by global statistics
    """
    n = distance_matrix.shape[0]
    if n <= 3:
        return 1.0 / (distance_matrix + 1e-12)
    
    # Preprocess distances
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    inverse_dist = 1.0 / safe_dist
    np.fill_diagonal(inverse_dist, 0)
    
    # Adaptive parameters based on instance characteristics
    k_candidate = max(4, int(np.sqrt(n)))  # Candidate list size
    k_regret = min(5, n - 2)  # Number of alternatives for regret calculation
    median_dist = np.median(safe_dist[safe_dist < np.inf])
    std_dist = np.std(safe_dist[safe_dist < np.inf])
    
    # 1. Extended k-regret penalty calculation
    regret_penalty = np.zeros_like(distance_matrix)
    
    for i in range(n):
        dists_i = safe_dist[i].copy()
        sorted_idx = np.argsort(dists_i)
        
        if len(sorted_idx) > k_regret:
            # Calculate k-regret: average of top k alternatives
            top_k_alternatives = dists_i[sorted_idx[1:k_regret+1]]
            avg_alternative = np.mean(top_k_alternatives)
            
            for j in range(n):
                if i != j:
                    # Extended regret considering multiple alternatives
                    if np.isclose(dists_i[j], dists_i[sorted_idx[0]]):
                        regret = 0
                    else:
                        # Regret relative to average of k best alternatives
                        regret = dists_i[j] - avg_alternative
                    
                    # Normalize by instance statistics
                    regret_penalty[i, j] = max(0, regret) / (median_dist + 1e-10)
    
    # 2. Dynamic exponential rank weighting with instance-specific decay
    rank_weight = np.zeros_like(distance_matrix)
    
    for i in range(n):
        sorted_idx = np.argsort(safe_dist[i])
        
        # Dynamic decay factor based on distance distribution
        if len(sorted_idx) >= 3:
            dist_ratio = safe_dist[i, sorted_idx[2]] / (safe_dist[i, sorted_idx[1]] + 1e-10)
            decay_factor = 0.5 + 0.3 * np.tanh(2.0 - dist_ratio)  # Adaptive between 0.5-0.8
        else:
            decay_factor = 0.7
        
        for rank, j in enumerate(sorted_idx):
            if i != j:
                # Exponential decay with dynamic factor
                rank_weight[i, j] = decay_factor ** rank
    
    # 3. Two-step lookahead path potential
    path_potential = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(i + 1, n):
            # Find best intermediate node for i->k->j path
            intermediate_costs = safe_dist[i, :] + safe_dist[:, j]
            intermediate_costs[[i, j]] = np.inf
            
            if np.min(intermediate_costs) < np.inf:
                best_two_step = np.min(intermediate_costs)
                direct_cost = safe_dist[i, j]
                
                # Potential factor: how much better is direct edge vs two-step
                if best_two_step > 0:
                    potential_ratio = direct_cost / best_two_step
                    # Sigmoid transformation for smooth scaling
                    path_potential[i, j] = path_potential[j, i] = 2.0 / (1.0 + np.exp(-5.0 * (1.0 - potential_ratio)))
    
    # 4. Adaptive cluster density scoring
    density_score = np.zeros_like(distance_matrix)
    
    # Calculate local densities using adaptive neighborhood
    local_densities = np.zeros(n)
    density_neighbors = min(8, n - 1)
    
    for i in range(n):
        neighbor_dists = np.sort(safe_dist[i])[:density_neighbors]
        local_densities[i] = np.mean(neighbor_dists)
    
    # Global density statistics for normalization
    global_density_mean = np.mean(local_densities)
    global_density_std = np.std(local_densities)
    
    for i in range(n):
        for j in range(i + 1, n):
            # Score edges connecting sparse regions higher
            if global_density_std > 0:
                z_i = (local_densities[i] - global_density_mean) / global_density_std
                z_j = (local_densities[j] - global_density_mean) / global_density_std
                
                # Bonus for connecting sparse regions, penalty for dense-to-dense
                density_bias = 1.0 + 0.3 * np.tanh(-0.5 * (z_i + z_j))
                density_score[i, j] = density_score[j, i] = density_bias
    
    # 5. Minimum Spanning Tree alignment (reinforcement)
    mst = minimum_spanning_tree(safe_dist).toarray()
    mst_bonus = np.zeros_like(distance_matrix)
    mst_bonus[mst > 0] = 1.5  # Reinforce MST edges
    
    # 6. Candidate list sparsification
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        neighbors = np.argsort(safe_dist[i])[:k_candidate]
        candidate_mask[i, neighbors] = True
    candidate_mask = candidate_mask | candidate_mask.T
    
    # 7. Combine all factors with adaptive weighting
    heuristic = np.zeros_like(distance_matrix)
    
    # Base score combines inverse distance with rank weighting
    base_score = inverse_dist * rank_weight
    
    # Apply extended regret penalty
    regret_factor = np.exp(-2.0 * regret_penalty)  # Stronger penalty
    
    # Incorporate lookahead path potential
    path_factor = 0.6 + 0.4 * path_potential  # 60-100% range
    
    # Combine all components
    combined = base_score * regret_factor * path_factor * density_score * (1.0 + mst_bonus)
    
    # Apply candidate mask
    heuristic = combined * candidate_mask
    
    # Normalize to maintain scale consistency
    max_val = np.max(heuristic)
    if max_val > 0:
        heuristic = heuristic / max_val * np.median(inverse_dist[inverse_dist > 0])
    
    # Ensure symmetry and zero diagonal
    heuristic = (heuristic + heuristic.T) / 2
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
