from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

        candidate_mask[i, neighbors] = True

    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)

            # Get the best and second best distances for regret calculation
            sorted_dists = np.sort(dists)

            second_best_dist = sorted_dists[1]

                    # Regret: opportunity cost of not choosing the second best
                    regret = (edge_dist - second_best_dist) / (mean_dist + eps)
                    # Apply exponential decay to soften extreme penalties
                    regret_penalty[i, j] = np.exp(-0.7 * max(0, regret))

    density_bonus = np.zeros_like(distance_matrix)
    
    # Estimate local density for each node using k-nearest neighbor distances
    k_density = min(5, n-1)
    local_densities = np.zeros(n)
    
    for i in range(n):
        # Find k-nearest distances
        k_nearest = np.partition(safe_dist[i], k_density)[:k_density+1]
        # Inverse of average distance to k-nearest neighbors as density measure
        if np.sum(k_nearest > 0) > 0:
            local_densities[i] = k_density / (np.sum(k_nearest) + eps)
        else:
            local_densities[i] = 1.0
    
    # Normalize densities
    if np.max(local_densities) > 0:
        local_densities = local_densities / np.max(local_densities)
    
    # Assign density bonus to edges
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Bonus based on geometric mean of endpoint densities
                geo_mean_density = np.sqrt(local_densities[i] * local_densities[j] + eps)
                # Additional factor: reward edges that are relatively short for both endpoints
                rel_dist_i = safe_dist[i, j] / (np.mean(safe_dist[i, safe_dist[i] < np.inf]) + eps)

                rel_factor = 1.0 / (0.5 + 0.25 * (rel_dist_i + rel_dist_j))

    # Normalize to [0, 1] range for consistent scaling

    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
