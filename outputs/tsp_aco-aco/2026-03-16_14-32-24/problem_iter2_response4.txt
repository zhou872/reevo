import numpy as np
from scipy import stats

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic matrix for TSP edge prioritization with regret-based
    penalties and two-step lookahead spatial averaging.
    
    Combines multiple sophisticated factors to identify promising edges,
    with adaptive weighting based on instance characteristics.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Square matrix of pairwise distances between nodes
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, with low-potential edges set to zero
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    epsilon = 1e-10
    
    # 1. Dynamic weight adaptation based on instance characteristics
    # Compute instance statistics to adapt heuristic weights
    distances_flattened = distance_matrix[np.triu_indices(n, k=1)]
    dist_mean = np.mean(distances_flattened)
    dist_std = np.std(distances_flattened)
    dist_cv = dist_std / (dist_mean + epsilon)  # Coefficient of variation
    
    # Adaptive weights based on problem characteristics
    if dist_cv < 0.3:  # Uniform distances
        alpha, beta, gamma, delta = 0.3, 0.4, 0.2, 0.1
    elif dist_cv < 0.7:  # Moderate variation
        alpha, beta, gamma, delta = 0.4, 0.3, 0.2, 0.1
    else:  # High variation
        alpha, beta, gamma, delta = 0.5, 0.2, 0.2, 0.1
    
    # 2. Base inverse distance with entropy guidance
    inv_dist = 1 / (distance_matrix + epsilon * (distance_matrix == 0))
    np.fill_diagonal(inv_dist, 0)
    
    # Entropy-based node importance
    distance_entropy = np.zeros(n)
    for i in range(n):
        valid_dists = distance_matrix[i][distance_matrix[i] > 0]
        if len(valid_dists) > 1:
            distance_entropy[i] = stats.entropy(valid_dists + epsilon)
    
    if distance_entropy.max() > 0:
        entropy_factor = 1 - (distance_entropy / distance_entropy.max())
    else:
        entropy_factor = np.ones(n)
    
    # 3. Regret-based alternative penalty
    # For each node, compute penalty based on quality of alternative connections
    regret_penalty = np.ones((n, n))
    for i in range(n):
        # Get sorted distances (excluding self)
        sorted_dists = np.sort(distance_matrix[i])
        valid_dists = sorted_dists[sorted_dists > 0][:min(3, n-1)]
        
        if len(valid_dists) > 1:
            # Compute regret as difference from best alternative
            base_regret = valid_dists[1] - valid_dists[0]
            for j in range(n):
                if i != j:
                    # Penalize edges that are much worse than best alternative
                    dist_ratio = distance_matrix[i, j] / (valid_dists[0] + epsilon)
                    regret_penalty[i, j] = np.exp(-delta * dist_ratio * base_regret / (dist_mean + epsilon))
    
    # 4. Two-step lookahead spatial averaging
    lookahead_score = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Consider two-step paths through intermediate nodes
                if n > 2:
                    # Exclude i and j from potential intermediates
                    other_nodes = [k for k in range(n) if k not in (i, j)]
                    if other_nodes:
                        # Average of two-step path lengths through intermediates
                        two_step_paths = []
                        for k in other_nodes:
                            path_length = distance_matrix[i, k] + distance_matrix[k, j]
                            two_step_paths.append(path_length)
                        
                        avg_two_step = np.mean(two_step_paths) if two_step_paths else distance_matrix[i, j]
                        # Reward edges that are shorter than average two-step alternatives
                        lookahead_score[i, j] = avg_two_step / (distance_matrix[i, j] + epsilon)
    
    # 5. Local clustering coefficient (improved version)
    k_neighbors = max(2, min(5, n // 4))
    cluster_weight = np.zeros((n, n))
    
    for i in range(n):
        # Find k nearest neighbors for each node
        neighbor_indices = np.argsort(distance_matrix[i])[1:k_neighbors+1]
        
        # Create weighted adjacency among neighbors
        for idx_i, n_i in enumerate(neighbor_indices):
            for n_j in neighbor_indices[idx_i+1:]:
                if distance_matrix[n_i, n_j] > 0:
                    # Use inverse square distance for stronger locality
                    weight = 1 / (distance_matrix[n_i, n_j]**2 + epsilon)
                    cluster_weight[i, n_i] += weight
                    cluster_weight[i, n_j] += weight
    
    # Normalize cluster weights
    if cluster_weight.max() > 0:
        cluster_weight = cluster_weight / cluster_weight.max()
    
    # 6. Symmetric combination with adaptive weights
    heuristic_matrix = np.zeros((n, n))
    
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] > 0:
                # Base score with entropy adjustment
                base_score = inv_dist[i, j] * ((entropy_factor[i] + entropy_factor[j]) / 2)
                
                # Cluster density bonus
                cluster_bonus = (cluster_weight[i, j] + cluster_weight[j, i]) / 2
                
                # Combine with regret penalty and lookahead score
                combined = (
                    (base_score ** alpha) *
                    ((1 + cluster_bonus) ** beta) *
                    (regret_penalty[i, j] ** gamma) *
                    (lookahead_score[i, j] ** delta)
                )
                
                heuristic_matrix[i, j] = combined
                heuristic_matrix[j, i] = combined
    
    # 7. Adaptive sparsification based on problem size
    if n < 10:
        k_sparse = min(4, n - 1)
    elif n < 50:
        k_sparse = min(6, n // 5)
    else:
        k_sparse = min(8, n // 10)
    
    mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get indices of top k heuristic values for this node
        row_heuristics = heuristic_matrix[i]
        top_k_indices = np.argsort(row_heuristics)[-k_sparse:]
        mask[i, top_k_indices] = True
    
    # Make mask symmetric and apply
    mask = mask | mask.T
    heuristic_matrix[~mask] = 0
    
    # 8. Final normalization with softmax-like transformation
    max_val = heuristic_matrix.max()
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
        
        # Softmax transformation to emphasize differences
        if n > 2:
            temp = 0.1  # Temperature parameter
            exp_vals = np.exp(heuristic_matrix / temp)
            row_sums = exp_vals.sum(axis=1, keepdims=True)
            heuristic_matrix = exp_vals / (row_sums + epsilon)
    
    np.fill_diagonal(heuristic_matrix, 0)
    return heuristic_matrix
