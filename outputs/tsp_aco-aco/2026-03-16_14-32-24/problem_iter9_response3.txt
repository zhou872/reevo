import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with simplified regret modeling,
    adaptive exponential rank weighting, and robust sparsification.
    
    Features:
    1. Exponential Rank-Weighted Scoring with Coefficient of Variation
    2. Simplified Regret-Based Penalty
    3. Multi-Factor Sparsification with Symmetry Enforcement
    4. Lightweight MST structural bias for small instances
    """
    n = distance_matrix.shape[0]
    
    # Safe distance matrix (avoid self-loops and division by zero)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # 1. Multi-Factor Sparsification with Symmetry Enforcement
    # Adaptive candidate list size based on instance size
    k_candidate = max(4, int(np.sqrt(n)))
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Build candidate lists for each node
    for i in range(n):
        # Get top k candidates by distance (nearest neighbors)
        neighbors = np.argsort(safe_dist[i])[:k_candidate]
        candidate_mask[i, neighbors] = True
    
    # Ensure candidate list is symmetric
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # 2. Exponential Rank-Weighted Scoring with Coefficient of Variation
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get distances to all other nodes
        distances = safe_dist[i].copy()
        valid_distances = distances[distances < np.inf]
        
        if len(valid_distances) >= 5:
            # Take first 5 nearest distances
            nearest_dists = np.sort(valid_distances)[:5]
            # Calculate coefficient of variation
            cv = np.std(nearest_dists) / (np.mean(nearest_dists) + eps)
            # Map CV to decay rate using sigmoid function
            decay_rate = 0.3 + 0.55 / (1 + np.exp(-cv))
            decay_rate = np.clip(decay_rate, 0.3, 0.85)
        else:
            decay_rate = 0.7
        
        # Assign weights with exponential decay by rank
        sorted_idx = np.argsort(distances)
        for rank, j in enumerate(sorted_idx):
            if i != j:
                rank_weights[i, j] = decay_rate ** rank
    
    # 3. Simplified Regret-Based Penalty
    regret_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        distances = safe_dist[i].copy()
        
        # Find the two smallest distances (excluding self)
        sorted_dists = np.sort(distances[distances < np.inf])
        if len(sorted_dists) >= 2:
            best_dist = sorted_dists[0]
            second_best = sorted_dists[1]
            
            for j in range(n):
                if i != j and distances[j] < np.inf:
                    # Calculate normalized difference
                    diff_norm = (distances[j] - second_best) / (second_best + eps)
                    # Apply exponential penalty
                    regret_matrix[i, j] = np.exp(-diff_norm)
    
    # 4. Lightweight MST structural bias (only for manageable sizes)
    mst_bonus = np.zeros_like(distance_matrix)
    if n <= 500:  # Conservative limit for MST computation
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_mask = (mst + mst.T) > 0
            mst_bonus = mst_mask * 0.2  # Light bonus for MST edges
        except:
            pass
    
    # 5. Combine all factors
    # Base score using inverse distance (normalized)
    inv_dist = 1.0 / safe_dist
    mean_dist = np.mean(safe_dist[safe_dist < np.inf])
    inv_dist = inv_dist * mean_dist
    
    # Combine: base * rank_weights * regret * (1 + mst_bonus)
    heuristics_matrix = inv_dist * rank_weights * regret_matrix * (1.0 + mst_bonus)
    
    # 6. Apply candidate mask and sparsify
    heuristics_matrix[~candidate_mask] = 0
    
    # Ensure each node has at least 2 connections for basic feasibility
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero_count = np.count_nonzero(row)
        
        if nonzero_count < 2:
            # Force at least 2 connections
            sorted_vals = np.argsort(row)
            # Get indices with non-zero values first, then zeros
            zero_indices = sorted_vals[row[sorted_vals] == 0]
            non_zero_indices = sorted_vals[row[sorted_vals] > 0]
            
            # If we have some non-zero, fill up to 2
            if len(non_zero_indices) > 0:
                # Already have at least 1, need 1 more
                if len(non_zero_indices) == 1 and len(zero_indices) > 0:
                    best_zero = zero_indices[-1]  # The "least zero" (closest to being non-zero)
                    heuristics_matrix[i, best_zero] = np.max(row) * 0.8
                    heuristics_matrix[best_zero, i] = np.max(row) * 0.8
            else:
                # No connections at all - take 2 closest
                closest = np.argsort(safe_dist[i])[1:3]  # Skip self
                avg_val = np.mean(heuristics_matrix[heuristics_matrix > 0]) if np.any(heuristics_matrix > 0) else 0.5
                heuristics_matrix[i, closest] = avg_val
                heuristics_matrix[closest, i] = avg_val
    
    # 7. Final symmetrization and normalization
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
