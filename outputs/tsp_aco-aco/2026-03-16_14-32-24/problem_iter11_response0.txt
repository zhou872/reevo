import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge evaluation with improved regret modeling,
    adaptive rank weighting, and refined spatial analysis.
    
    Features:
    1. Local-Regret Normalized Penalty: Considers local vs global alternatives
    2. Geometric Rank-Weighted Scoring: Adaptive exponential rank weighting
    3. Spatial Density Bonus: Kernel-based density estimation with adaptive bandwidth
    4. Multi-Factor Sparsification: Dynamic candidate lists with connectivity preservation
    """
    n = distance_matrix.shape[0]
    
    # Safe distance matrix (avoid self-loops and division by zero)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # Base inverse distance (normalized by mean distance)
    inv_dist = 1.0 / safe_dist
    mean_dist = np.mean(safe_dist[safe_dist < np.inf])
    inv_dist = inv_dist * mean_dist  # Scale to comparable magnitude
    
    # 1. Adaptive candidate list with symmetry enforcement
    # Use O(sqrt(n)) candidates per node, but at least 4
    k_candidate = max(4, int(np.sqrt(n) * 1.5))
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get top k candidates by distance (nearest neighbors)
        neighbors = np.argsort(safe_dist[i])[:k_candidate]
        candidate_mask[i, neighbors] = True
    
    # Ensure candidate list is symmetric
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # 2. Local-Regret Normalized Penalty
    regret_matrix = np.zeros_like(distance_matrix)
    
    # Precompute best and alternative distances for each node
    for i in range(n):
        dists = safe_dist[i].copy()
        sorted_dists = np.sort(dists)
        
        if len(sorted_dists) >= 3:
            # Best (closest) distance
            best_dist = sorted_dists[0]
            
            # Alternative distance: median of top k/2 alternatives
            alt_idx = min(k_candidate//2, len(sorted_dists)-1)
            alt_dist = sorted_dists[alt_idx]
            
            # Calculate local regret: normalized difference from best
            # with smoothing for near-best edges
            for j in range(n):
                if i != j:
                    edge_dist = dists[j]
                    if edge_dist <= alt_dist:
                        # Minimal penalty for top-tier edges
                        regret = 0.1 * (edge_dist - best_dist) / (alt_dist + eps)
                    else:
                        # Progressive penalty for worse edges
                        regret = 0.5 + 0.5 * (edge_dist - alt_dist) / (np.max(dists) + eps)
                    
                    regret_matrix[i, j] = regret
    
    # 3. Geometric Rank-Weighted Scoring with adaptive decay
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get sorted indices by distance
        sorted_idx = np.argsort(safe_dist[i])
        
        # Adaptive decay based on distance distribution
        distances = safe_dist[i, sorted_idx[1:min(10, n-1)]]
        if len(distances) > 1:
            decay_rate = 0.3 + 0.5 * np.std(distances) / (np.mean(distances) + eps)
            decay_rate = np.clip(decay_rate, 0.3, 0.9)
        else:
            decay_rate = 0.7
        
        # Assign weights with exponential decay
        for rank, j in enumerate(sorted_idx):
            if i != j:
                rank_weights[i, j] = decay_rate ** rank
    
    # 4. Spatial Density Bonus with adaptive bandwidth
    density_bonus = np.zeros_like(distance_matrix)
    
    # Adaptive bandwidth: median of distances to k-nearest neighbors
    k_density = min(5, n-1)
    bandwidths = np.zeros(n)
    
    for i in range(n):
        # Find k-nearest distances (excluding self)
        k_nearest = np.partition(safe_dist[i], k_density)[:k_density+1]
        bandwidths[i] = np.median(k_nearest[k_nearest > 0])
    
    # Compute density contributions using Gaussian kernel
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Adaptive bandwidth for pair
                bandwidth = np.sqrt(bandwidths[i] * bandwidths[j])
                
                # Measure how many nodes are "close" to both endpoints
                # Using Gaussian kernel similarity
                sigma = bandwidth / 2  # Kernel width
                
                # Count nodes within 2*sigma of both i and j
                close_to_i = np.exp(-0.5 * (safe_dist[i] / sigma) ** 2)
                close_to_j = np.exp(-0.5 * (safe_dist[j] / sigma) ** 2)
                
                # Joint density measure
                joint_density = np.sum(close_to_i * close_to_j) - 2  # Subtract i and j
                
                # Bonus scales with log of joint density
                if joint_density > 0:
                    density_bonus[i, j] = np.log1p(joint_density) * 0.15
                else:
                    density_bonus[i, j] = -0.1  # Penalty for isolated edges
    
    # Make density bonus symmetric
    density_bonus = np.maximum(density_bonus, density_bonus.T)
    
    # 5. Minimum Spanning Tree structural bias (lightweight)
    if n <= 1000:  # MST is O(n^2), skip for very large instances
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_bias = (mst + mst.T) > 0
            mst_bonus = mst_bias * 0.3
        except:
            mst_bonus = np.zeros_like(distance_matrix)
    else:
        mst_bonus = np.zeros_like(distance_matrix)
    
    # 6. Combine all factors
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base score with rank weighting
    base_score = inv_dist * rank_weights
    
    # Apply regret penalty (exponential decay)
    regret_penalty = np.exp(-2.0 * (1.0 - regret_matrix))
    score_with_regret = base_score * regret_penalty
    
    # Add density and MST bonuses
    heuristics_matrix = score_with_regret * (1.0 + density_bonus) * (1.0 + mst_bonus)
    
    # 7. Multi-factor sparsification
    # Apply candidate mask
    heuristics_matrix[~candidate_mask] = 0
    
    # Ensure each node has at least 2 connections
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero_count = np.count_nonzero(row)
        
        if nonzero_count < 2:
            # Force top 2 edges for connectivity
            top_indices = np.argsort(row)[-2:]
            heuristics_matrix[i, top_indices] = np.max(row) * 1.1  # Slightly boost
            heuristics_matrix[top_indices, i] = np.max(row) * 1.1
    
    # 8. Symmetrize and normalize
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
