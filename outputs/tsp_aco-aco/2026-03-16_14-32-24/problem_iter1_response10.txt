import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Distance inverse with nearest neighbor density weighting
    2. Clarke-Wright savings heuristic for sparsification
    3. Edge centrality from minimum spanning tree
    4. Two-hop connection potential
    """
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
    
    # 1. Distance inverse with nearest neighbor density weighting
    inv_dist = np.divide(1.0, distance_matrix + np.eye(n) * 1e-12)  # Avoid division by zero
    
    # Calculate nearest neighbor density for each node
    k = max(2, n // 10)  # Adaptive k for nearest neighbors
    nearest_neighbor_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])
        nearest_neighbor_mask[i, sorted_indices[1:k+1]] = True
    
    # Weight by how often edge appears in nearest neighbors
    mutual_nearest = nearest_neighbor_mask & nearest_neighbor_mask.T
    density_weight = np.zeros_like(distance_matrix)
    density_weight[mutual_nearest] = 2.0
    density_weight[nearest_neighbor_mask ^ mutual_nearest] = 1.0
    
    # 2. Clarke-Wright savings heuristic for sparsification
    savings_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            # Classic Clarke-Wright savings: d(i,0) + d(0,j) - d(i,j)
            # Using average of all nodes as "depot" for robustness
            depot_dist_i = np.mean(distance_matrix[i])
            depot_dist_j = np.mean(distance_matrix[j])
            savings = depot_dist_i + depot_dist_j - distance_matrix[i, j]
            if savings > 0:
                savings_matrix[i, j] = savings
                savings_matrix[j, i] = savings
    
    # 3. Edge centrality from minimum spanning tree
    mst = minimum_spanning_tree(distance_matrix)
    mst_array = mst.toarray()
    mst_bonus = (mst_array > 0) | (mst_array.T > 0)
    
    # 4. Two-hop connection potential
    two_hop_potential = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i+1, n):
            # Find minimum two-hop path through intermediate nodes
            min_two_hop = np.min(distance_matrix[i, :] + distance_matrix[:, j])
            two_hop_potential[i, j] = distance_matrix[i, j] / (min_two_hop + 1e-12)
            two_hop_potential[j, i] = two_hop_potential[i, j]
    
    # Combine all factors
    base_score = inv_dist * density_weight
    
    # Apply savings-based sparsification: only keep edges with positive savings
    savings_mask = savings_matrix > 0
    
    # Apply MST bonus to selected edges
    mst_boost = np.where(mst_bonus, 1.5, 1.0)
    
    # Combine with two-hop potential (lower is better for direct connection)
    connection_factor = np.clip(2.0 - two_hop_potential, 0.5, 2.0)
    
    # Final heuristic score
    heuristics_matrix = base_score * savings_mask * mst_boost * connection_factor
    
    # Normalize to [0, 1] range for stability
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    # Set diagonal to zero
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
