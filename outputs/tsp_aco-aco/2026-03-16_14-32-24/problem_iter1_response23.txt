import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge prioritization.
    
    Combines inverse distance weighting with adaptive threshold-based sparsification:
    1. Computes node-specific distance thresholds using medians
    2. Applies symmetric pruning to unpromising long edges
    3. Returns sparsified inverse distance matrix as edge priorities
    """
    n = distance_matrix.shape[0]
    
    # Compute median distance for each node (excluding self)
    medians = np.zeros(n)
    for i in range(n):
        # Extract distances to other nodes (excluding diagonal)
        d = np.delete(distance_matrix[i, :], i)
        medians[i] = np.median(d)
    
    # Adaptive threshold scaling factor
    # Balances sparsity with maintaining graph connectivity
    factor = 1.5
    medians = medians * factor
    
    # Create threshold matrix using maximum of endpoint medians
    max_median_matrix = np.maximum.outer(medians, medians)
    
    # Create pruning mask: keep edges within threshold
    mask = distance_matrix <= max_median_matrix
    np.fill_diagonal(mask, False)  # Remove self-loops
    
    # Compute inverse distances with safe division
    inv_dist = np.divide(
        1, 
        distance_matrix, 
        out=np.zeros_like(distance_matrix), 
        where=distance_matrix != 0
    )
    
    # Apply mask to sparsify and get final priorities
    priority_matrix = inv_dist * mask
    
    return priority_matrix
