import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection with multi-step lookahead and
    regret-based scoring. Returns a matrix of edge promise scores.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Create a working copy with diagonal set to infinity
    dmat = distance_matrix.copy()
    np.fill_diagonal(dmat, np.inf)
    
    # 1. Base inverse distance with dynamic scaling
    epsilon = np.median(dmat[dmat < np.inf]) * 1e-6 if np.any(dmat < np.inf) else 1e-10
    inv_dist = 1 / (dmat + epsilon)
    
    # 2. Minimum Spanning Tree connectivity
    mst = minimum_spanning_tree(dmat).toarray()
    mst_bonus = (mst + mst.T) > 0
    
    # 3. Multi-step path savings potential (2-step and 3-step lookahead)
    savings_2step = np.zeros((n, n))
    savings_3step = np.zeros((n, n))
    
    # Precompute nearest neighbor sets for efficiency
    k_neighbors = max(3, min(15, n // 4))
    neighbor_indices = np.argsort(dmat, axis=1)[:, 1:k_neighbors+1]
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            # Two-step savings: i -> k -> j
            mask_2step = np.ones(n, dtype=bool)
            mask_2step[[i, j]] = False
            if np.any(mask_2step):
                savings = dmat[i, :] + dmat[:, j] - dmat[i, j]
                savings_2step[i, j] = np.max(savings[mask_2step])
            
            # Three-step savings: i -> k -> l -> j (using limited neighbor sets)
            if n >= 4:
                best_saving = -np.inf
                # Consider intermediate nodes from restricted sets
                for k in neighbor_indices[i]:
                    if k == j:
                        continue
                    # Limited set for second intermediate
                    for l in neighbor_indices[j]:
                        if l == i or l == k:
                            continue
                        path_cost = dmat[i, k] + dmat[k, l] + dmat[l, j]
                        saving = dmat[i, j] - path_cost
                        if saving > best_saving:
                            best_saving = saving
                savings_3step[i, j] = max(best_saving, 0)
    
    # Normalize savings metrics
    for savings in [savings_2step, savings_3step]:
        max_val = np.max(savings)
        if max_val > 0:
            savings /= max_val
    
    # 4. Regret-based opportunity cost
    regret_scores = np.zeros((n, n))
    for i in range(n):
        # Get distances from node i, excluding self
        dist_i = dmat[i].copy()
        dist_i[i] = np.inf
        
        # Find two best edges from i
        best_indices = np.argpartition(dist_i, 2)[:2]
        best_dists = dist_i[best_indices]
        
        if len(best_dists) >= 2:
            # Regret = difference between second-best and current edge
            second_best = np.max(best_dists)
            for j in range(n):
                if i != j:
                    regret = max(0, second_best - dmat[i, j])
                    # Convert to score: high regret -> high penalty
                    regret_scores[i, j] = 1.0 / (1.0 + regret)
    
    # 5. Dynamic weight tuning based on problem characteristics
    # Weights adjust based on problem size and distance distribution
    if n < 10:
        weights = [0.4, 0.2, 0.2, 0.15, 0.05]  # More local for small problems
    elif n < 50:
        weights = [0.35, 0.25, 0.2, 0.15, 0.05]
    else:
        weights = [0.3, 0.3, 0.2, 0.15, 0.05]  # More global for large problems
    
    # Adjust weights based on distance variance (spatial structure)
    dist_std = np.std(dmat[dmat < np.inf])
    dist_mean = np.mean(dmat[dmat < np.inf])
    if dist_std / dist_mean > 1.0:  # High variance
        weights[0] *= 1.2  # Boost inverse distance
        weights[1] *= 0.8  # Reduce MST reliance
    
    # Normalize weights
    weights = np.array(weights) / np.sum(weights)
    
    # 6. Combine all factors
    heuristics = (
        weights[0] * inv_dist +
        weights[1] * mst_bonus +
        weights[2] * savings_2step +
        weights[3] * savings_3step +
        weights[4] * regret_scores
    )
    
    # 7. Adaptive edge sparsification
    # Keep top edges per node, number adapts to problem size
    k_edges = max(3, min(n // 2, int(np.sqrt(n)) * 2))
    mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get top k edges, excluding self
        row_scores = heuristics[i].copy()
        row_scores[i] = -np.inf
        top_k_indices = np.argpartition(row_scores, -k_edges)[-k_edges:]
        mask[i, top_k_indices] = True
    
    # Ensure symmetry and preserve MST edges
    mask = np.logical_or(mask, mask.T)
    mask = np.logical_or(mask, mst_bonus)
    heuristics = heuristics * mask
    
    # 8. Final processing
    np.fill_diagonal(heuristics, 0)
    
    # Row-wise normalization
    row_max = np.max(heuristics, axis=1, keepdims=True)
    row_max[row_max == 0] = 1  # Avoid division by zero
    heuristics = heuristics / row_max
    
    # Apply softmax-like transformation to emphasize differences
    heuristics = np.exp(heuristics) - 1
    
    # Final normalization to [0, 1] range
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
