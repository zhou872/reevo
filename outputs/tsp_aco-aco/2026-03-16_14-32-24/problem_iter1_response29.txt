import numpy as np
from scipy import stats

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    
    Features:
    1. Mutual nearest neighbor bonus for edges where nodes are each other's top neighbors
    2. Local cluster density weighting for edges in dense regions
    3. Distance variance penalty for edges with highly variable alternative connections
    4. Sparsification by keeping only top-k edges per node (k = ceil(n/3))
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (handle zeros on diagonal)
    np.fill_diagonal(distance_matrix, np.inf)  # Temporarily set diagonal to inf for clean division
    inv_dist = 1 / distance_matrix
    np.fill_diagonal(distance_matrix, 0)  # Restore original
    
    # 2. Mutual nearest neighbor bonus
    k_neighbors = max(2, n // 4)  # Number of nearest neighbors to consider
    mutual_bonus = np.zeros_like(distance_matrix)
    
    # Get k nearest neighbors for each node
    nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:k_neighbors+1]
    
    for i in range(n):
        for j in range(i+1, n):
            if i in nearest_indices[j] and j in nearest_indices[i]:
                # Strong bonus for mutual nearest neighbors
                mutual_bonus[i, j] = mutual_bonus[j, i] = 2.0
    
    # 3. Local cluster density weighting
    density_weight = np.zeros_like(distance_matrix)
    local_radius = np.percentile(distance_matrix[distance_matrix > 0], 25)  # 25th percentile as radius
    
    for i in range(n):
        # Find nodes within local radius
        local_nodes = distance_matrix[i] <= local_radius
        local_nodes[i] = False  # Exclude self
        
        if np.sum(local_nodes) > 1:
            # Calculate density factor
            density_factor = np.sum(local_nodes) / n
            density_weight[i, local_nodes] = 1.0 + density_factor
    
    # 4. Distance variance penalty
    variance_penalty = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Variance of distances from node i to all others
        var_i = np.var(distance_matrix[i, distance_matrix[i] > 0])
        for j in range(n):
            if i != j:
                # Variance of distances from node j to all others
                var_j = np.var(distance_matrix[j, distance_matrix[j] > 0])
                # Penalize edges where either node has high variance (uncertain connections)
                penalty = 1.0 / (1.0 + (var_i + var_j) / 2.0)
                variance_penalty[i, j] = penalty
    
    # 5. Combine all factors
    heuristics_matrix = inv_dist * (1 + mutual_bonus) * density_weight * variance_penalty
    
    # 6. Sparsify: keep only top-k edges per node
    k_sparse = max(2, int(np.ceil(n / 3)))
    sparse_matrix = np.zeros_like(heuristics_matrix)
    
    for i in range(n):
        # Get indices of top-k edges for node i (excluding self)
        row = heuristics_matrix[i].copy()
        row[i] = -np.inf  # Exclude diagonal
        top_k_indices = np.argpartition(row, -k_sparse)[-k_sparse:]
        
        # Keep only top-k edges
        sparse_matrix[i, top_k_indices] = heuristics_matrix[i, top_k_indices]
        sparse_matrix[top_k_indices, i] = heuristics_matrix[top_k_indices, i]
    
    # Ensure symmetry (average both directions)
    final_matrix = (sparse_matrix + sparse_matrix.T) / 2
    np.fill_diagonal(final_matrix, 0)
    
    return final_matrix
