import numpy as np
from scipy.spatial import KDTree
from scipy.sparse import csr_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges combining multiple spatial and structural features.
    
    Strategy:
    1. Dynamic neighbor density estimation for adaptive sparsification
    2. Symmetry-aware edge pruning based on mutual nearness
    3. Rank-based weighting with geometric progression
    4. Normalized distance inversion with outlier adjustment
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix)
    
    # 1. Normalize distances and handle zeros
    eps = 1e-10
    dist_norm = distance_matrix.copy()
    np.fill_diagonal(dist_norm, np.inf)
    min_dist = np.min(dist_norm)
    max_dist = np.max(dist_norm)
    dist_range = max_dist - min_dist + eps
    
    # 2. Compute adaptive neighborhood density
    # Dynamic k based on problem size and distance distribution
    k_base = max(3, min(10, n // 4))
    density_scores = np.zeros(n)
    
    for i in range(n):
        # Use quickselect for partial sorting
        neighbor_dists = np.partition(distance_matrix[i], k_base)[:k_base+1]
        valid_dists = neighbor_dists[neighbor_dists > 0]
        if len(valid_dists) > 1:
            density_scores[i] = np.mean(valid_dists[:k_base])
        else:
            density_scores[i] = np.mean(distance_matrix[i, distance_matrix[i] > 0])
    
    # 3. Symmetry-aware mutual nearness matrix
    mutual_nearness = np.zeros_like(distance_matrix)
    for i in range(n):
        # Find symmetric nearest neighbors
        row_dists = distance_matrix[i].copy()
        row_dists[i] = np.inf
        
        # Adaptive threshold based on local density
        threshold = np.percentile(row_dists[row_dists < np.inf], 
                                 min(75, 100 * k_base / n))
        
        # Mark promising connections
        promising_mask = row_dists <= threshold
        mutual_nearness[i] = promising_mask.astype(float)
    
    # 4. Rank-based weighting with geometric decay
    rank_weights = np.zeros_like(distance_matrix)
    decay_factor = 0.7
    
    for i in range(n):
        valid_indices = np.where(distance_matrix[i] > 0)[0]
        if len(valid_indices) == 0:
            continue
            
        # Get ranks (1-based) for valid distances
        sorted_indices = np.argsort(distance_matrix[i, valid_indices])
        ranks = np.zeros(len(valid_indices))
        ranks[sorted_indices] = np.arange(1, len(valid_indices) + 1)
        
        # Apply geometric decay to ranks
        for idx, j in enumerate(valid_indices):
            rank = ranks[idx]
            if rank <= k_base:
                weight = decay_factor ** (rank - 1)
                # Symmetry bonus for mutual high ranks
                if mutual_nearness[i, j] > 0 and mutual_nearness[j, i] > 0:
                    weight *= 1.5
                rank_weights[i, j] = weight
    
    # 5. Combined scoring with sparsification
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            # Base inverse distance with normalization
            inv_dist = (max_dist - distance_matrix[i, j] + min_dist) / dist_range
            
            # Density adjustment: favor connections between dense regions
            density_factor = 1.0 / (1.0 + abs(density_scores[i] - density_scores[j]))
            
            # Combined score
            base_score = inv_dist * density_factor * rank_weights[i, j]
            
            # Apply mutual nearness filter with hysteresis
            if mutual_nearness[i, j] > 0:
                # Strong promotion for symmetric connections
                if mutual_nearness[j, i] > 0:
                    heuristic_matrix[i, j] = base_score * 2.0
                else:
                    heuristic_matrix[i, j] = base_score
            elif rank_weights[i, j] > 0.3:  # Keep some moderately ranked edges
                heuristic_matrix[i, j] = base_score * 0.5
            # Otherwise score remains 0 (sparsified)
    
    # 6. Final adjustments and symmetry enforcement
    # Ensure matrix is symmetric for undirected TSP
    heuristic_matrix = np.maximum(heuristic_matrix, heuristic_matrix.T * 0.8)
    
    # Global sparsification: keep only top connections per node
    sparsity_factor = min(0.3, 5.0 / n)
    for i in range(n):
        row = heuristic_matrix[i].copy()
        row[i] = -np.inf
        threshold = np.partition(row, int((n-1) * (1 - sparsity_factor)))[int((n-1) * (1 - sparsity_factor))]
        heuristic_matrix[i, heuristic_matrix[i] < threshold] = 0
    
    # Normalize for stability
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
    
    np.fill_diagonal(heuristic_matrix, 0)
    return heuristic_matrix
