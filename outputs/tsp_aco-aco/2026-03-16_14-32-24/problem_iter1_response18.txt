import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics function for TSP edge selection.
    
    Returns a matrix where higher values indicate more promising edges,
    with unpromising edges set to zero via sparsification.
    """
    n = distance_matrix.shape[0]
    
    # Avoid division by zero and self-loops
    eps = 1e-12
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base inverse distance (normalized)
    inv_dist = 1 / safe_dist
    
    # 2. Nearest neighbor bonus
    k = max(2, n // 10)  # Dynamic k: top 10% of nearest edges per node
    nn_bonus = np.zeros((n, n))
    for i in range(n):
        # Get k nearest neighbors for node i
        sorted_idx = np.argsort(safe_dist[i])[:k]
        nn_bonus[i, sorted_idx] = 1.0
    
    # 3. Minimum Spanning Tree structural bias
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bias = (mst + mst.T) > 0  # Symmetrize MST edges
    
    # 4. Triangular inequality coefficient
    # Measures how much an edge violates triangle inequality
    tri_penalty = np.ones((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Find minimum indirect distance through any k
                min_indirect = np.min(safe_dist[i, :] + safe_dist[:, j])
                direct = safe_dist[i, j]
                if min_indirect < direct:
                    tri_penalty[i, j] = direct / (min_indirect + eps)
    
    # Combine factors
    heuristics_matrix = (
        inv_dist *                      # Base inverse distance
        (1 + 0.3 * nn_bonus) *          # Nearest neighbor bonus
        (1 + 0.5 * mst_bias) *          # MST structural bias
        (1 / np.sqrt(tri_penalty))      # Penalty for triangle violation
    )
    
    # 5. Sparsification using candidate lists
    candidate_threshold = np.percentile(heuristics_matrix, 70)  # Keep top 30%
    mask = heuristics_matrix > candidate_threshold
    
    # Ensure connectivity: keep at least 2 edges per node
    for i in range(n):
        row = heuristics_matrix[i]
        if np.sum(mask[i]) < 2:
            # Keep top 2 edges for this node
            top_indices = np.argsort(row)[-2:]
            mask[i, top_indices] = True
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * mask
    
    # Symmetrize the final matrix (TSP edges are undirected)
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    
    return heuristics_matrix
