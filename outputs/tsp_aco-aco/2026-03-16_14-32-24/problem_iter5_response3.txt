import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge evaluation with improved multi-factor
    combination, adaptive distribution-based weighting, and efficient sparsification.
    
    Core Improvements:
    1. Multi-Factor Edge Sparsification: Dynamically combines multiple heuristics
       to identify and preserve promising edges while aggressively pruning others.
    2. Adaptive Distance Distribution Weighting: Uses the entire distance distribution
       rather than just ranks, with emphasis on local vs global distance patterns.
    3. Cluster Density Geometric Mean: Computes node cluster strength using geometric
       mean of neighborhood densities for more balanced regional coverage.
    4. Stochastic Candidate Preservation: Preserves randomness in candidate selection
       to avoid over-commitment to deterministic patterns.
    """
    n = distance_matrix.shape[0]
    if n < 4:
        # For very small instances, return simple normalized inverse distances
        simple_heuristic = 1.0 / (distance_matrix + np.eye(n) * 1e-12)
        np.fill_diagonal(simple_heuristic, 0)
        max_val = np.max(simple_heuristic)
        if max_val > 0:
            simple_heuristic /= max_val
        return simple_heuristic
    
    # Prepare safe distance matrix (avoid self-loops and division by zero)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # 1. Multi-resolution candidate selection
    # Use different candidate counts for different instance sizes
    if n < 50:
        k_candidate = min(8, n - 1)
    elif n < 200:
        k_candidate = max(5, int(np.sqrt(n)))
    else:
        k_candidate = max(4, int(np.sqrt(n) * 0.8))
    
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors as deterministic candidates
        neighbors = np.argsort(safe_dist[i])[:k_candidate]
        candidate_mask[i, neighbors] = True
        
        # Add stochastic candidates: sample a few additional edges with probability
        # inversely proportional to distance rank
        if n > 20:
            remaining = [j for j in range(n) if j != i and j not in neighbors]
            if remaining:
                # Weight by inverse rank (lower rank = higher probability)
                ranks = np.arange(len(remaining)) + k_candidate + 1
                probs = 1.0 / (ranks ** 0.7)
                probs /= probs.sum()
                
                # Sample 1-2 additional candidates
                n_stochastic = min(2, len(remaining))
                stochastic_idx = np.random.choice(len(remaining), n_stochastic, 
                                                 p=probs, replace=False)
                for idx in stochastic_idx:
                    candidate_mask[i, remaining[idx]] = True
    
    # Ensure symmetric candidate list
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # 2. Adaptive distance distribution weighting
    # Instead of simple inverse, use sigmoid-like transformation
    # that emphasizes differences in promising distance ranges
    dist_weights = np.zeros_like(distance_matrix)
    
    # Compute per-node distance statistics
    node_stats = []
    for i in range(n):
        dists = safe_dist[i]
        q25, q50, q75 = np.percentile(dists, [25, 50, 75])
        node_stats.append((q25, q50, q75))
    
    for i in range(n):
        q25_i, q50_i, q75_i = node_stats[i]
        for j in range(n):
            if i == j:
                continue
                
            dist = safe_dist[i, j]
            q25_j, q50_j, q75_j = node_stats[j]
            
            # Combined quartile from both endpoints
            combined_q25 = min(q25_i, q25_j)
            combined_q50 = (q50_i + q50_j) / 2
            combined_q75 = max(q75_i, q75_j)
            
            if dist <= combined_q25:
                # Top tier: very promising edges
                weight = 1.0 - 0.3 * (dist / (combined_q25 + eps))
            elif dist <= combined_q50:
                # Good tier: promising but not excellent
                weight = 0.7 - 0.4 * ((dist - combined_q25) / (combined_q50 - combined_q25 + eps))
            elif dist <= combined_q75:
                # Medium tier: average edges
                weight = 0.3 - 0.2 * ((dist - combined_q50) / (combined_q75 - combined_q50 + eps))
            else:
                # Poor tier: unlikely to be good
                weight = 0.1 - 0.09 * ((dist - combined_q75) / (np.max(safe_dist[i]) - combined_q75 + eps))
            
            dist_weights[i, j] = max(0.01, weight)
    
    # 3. Cluster density geometric mean
    density_bonus = np.zeros_like(distance_matrix)
    
    # Compute node densities using adaptive bandwidth
    node_densities = np.zeros(n)
    k_density = min(max(3, n // 20), 10)
    
    for i in range(n):
        # Get distances to k_density nearest neighbors
        k_nearest = np.partition(safe_dist[i], k_density)[:k_density+1]
        bandwidth = np.median(k_nearest[k_nearest > 0])
        
        # Count nodes within bandwidth (excluding self)
        within_bandwidth = np.sum(safe_dist[i] <= bandwidth * 1.5)
        node_densities[i] = within_bandwidth - 1  # Exclude self
    
    # Normalize densities to [0.5, 2.0] range
    if np.max(node_densities) > 0:
        min_d, max_d = np.min(node_densities), np.max(node_densities)
        if max_d > min_d:
            node_densities = 0.5 + 1.5 * (node_densities - min_d) / (max_d - min_d)
    
    # Apply geometric mean of endpoint densities as bonus
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                geo_mean = np.sqrt(node_densities[i] * node_densities[j])
                # Bonus scales with geometric mean
                density_bonus[i, j] = 0.3 * (geo_mean - 1.0)  # Positive for >1, negative for <1
    
    density_bonus = np.maximum(density_bonus, density_bonus.T)
    
    # 4. Minimum spanning tree preservation (for moderate-sized instances)
    mst_bonus = np.zeros_like(distance_matrix)
    if n <= 2000:  # Practical limit for MST computation
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_mask = (mst + mst.T) > 0
            mst_bonus = mst_mask.astype(float) * 0.4
        except:
            pass
    
    # 5. Combine factors with adaptive weighting
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base matrix combines distance weights and candidate mask
    base_matrix = dist_weights * candidate_mask.astype(float)
    
    # Apply density bonus multiplicatively
    heuristics_matrix = base_matrix * (1.0 + density_bonus)
    
    # Add MST bonus additively (edges in MST get extra boost)
    heuristics_matrix += mst_bonus
    
    # 6. Adaptive sparsification with connectivity guarantee
    # For each node, keep top candidates plus any MST edges
    for i in range(n):
        row = heuristics_matrix[i]
        
        # Determine number of edges to keep per node
        if n < 30:
            keep_count = max(3, min(6, n // 3))
        else:
            keep_count = max(3, int(np.sqrt(n) * 0.7))
        
        # Get indices of top edges
        top_indices = np.argsort(row)[-keep_count:]
        
        # Create new mask for this node
        new_mask = np.zeros(n, dtype=bool)
        new_mask[top_indices] = True
        
        # Also preserve MST edges for this node
        if n <= 2000:
            mst_edges_i = np.where(mst_bonus[i] > 0)[0]
            new_mask[mst_edges_i] = True
        
        # Apply mask
        heuristics_matrix[i, ~new_mask] = 0
    
    # Ensure symmetry
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # 7. Final connectivity check and normalization
    # Ensure each node has at least 2 connections
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero_count = np.count_nonzero(row)
        
        if nonzero_count < 2:
            # Find the two best edges by base distance weight
            backup_scores = dist_weights[i].copy()
            backup_scores[i] = 0  # Exclude self
            top_two = np.argsort(backup_scores)[-2:]
            heuristics_matrix[i, top_two] = backup_scores[top_two]
            heuristics_matrix[top_two, i] = backup_scores[top_two]
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
