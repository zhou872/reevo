import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge evaluation with improved regret modeling,
    adaptive rank weighting, and refined spatial analysis.
    
    Features:
    1. Adaptive Local-Regret Penalty: Normalized regret relative to local alternatives
    2. Exponential Rank-Weighted Scoring: Adaptive decay based on distance distribution
    3. Geometric Mean Density Bonus: Density estimation via geometric mean of endpoint densities
    4. Dynamic Candidate Sparsification: Adaptive candidate selection with connectivity preservation
    """
    n = distance_matrix.shape[0]
    
    # Safe distance matrix (avoid self-loops and division by zero)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # 1. Dynamic candidate list with adaptive size
    # Base candidate count on log scale, ensuring minimum of 4
    k_candidate = max(4, int(np.log2(n) * 2.5))
    k_candidate = min(k_candidate, n // 4)  # Upper bound for very large instances
    
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get top k candidates by distance (nearest neighbors)
        neighbors = np.argsort(safe_dist[i])[:k_candidate]
        candidate_mask[i, neighbors] = True
    
    # Ensure candidate list is symmetric
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # 2. Adaptive Local-Regret Penalty with improved normalization
    regret_matrix = np.zeros_like(distance_matrix)
    
    # Precompute local statistics for each node
    for i in range(n):
        dists = safe_dist[i].copy()
        valid_dists = dists[dists < np.inf]
        
        if len(valid_dists) >= 3:
            # Get local statistics
            sorted_dists = np.sort(valid_dists)
            best_dist = sorted_dists[0]
            
            # Use 75th percentile of top k candidates as alternative reference
            ref_idx = min(int(k_candidate * 0.75), len(sorted_dists) - 1)
            ref_dist = sorted_dists[ref_idx]
            
            # Calculate local regret with smooth transition
            for j in range(n):
                if i != j and candidate_mask[i, j]:
                    edge_dist = dists[j]
                    
                    if edge_dist <= ref_dist:
                        # Smooth penalty for edges near best
                        rel_pos = (edge_dist - best_dist) / (ref_dist - best_dist + eps)
                        regret = 0.1 * np.sqrt(rel_pos)
                    else:
                        # Progressive penalty for worse edges
                        worst_dist = np.max(valid_dists)
                        rel_pos = (edge_dist - ref_dist) / (worst_dist - ref_dist + eps)
                        regret = 0.3 + 0.7 * rel_pos
                    
                    regret_matrix[i, j] = regret
    
    # 3. Exponential Rank-Weighted Scoring with adaptive decay rate
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get sorted indices by distance
        valid_indices = np.where(safe_dist[i] < np.inf)[0]
        if len(valid_indices) == 0:
            continue
            
        sorted_indices = valid_indices[np.argsort(safe_dist[i, valid_indices])]
        distances = safe_dist[i, sorted_indices]
        
        # Adaptive decay based on local distance variability
        if len(distances) >= 5:
            # Use first 5 distances for decay rate calculation
            local_dists = distances[:5]
            cv = np.std(local_dists) / (np.mean(local_dists) + eps)  # Coefficient of variation
            decay_rate = 0.4 + 0.5 * np.tanh(cv - 0.5)  # Sigmoid adjustment
            decay_rate = np.clip(decay_rate, 0.3, 0.85)
        else:
            decay_rate = 0.6
        
        # Assign weights with exponential decay
        for rank, j in enumerate(sorted_indices):
            if i != j:
                rank_weights[i, j] = decay_rate ** rank
    
    # 4. Geometric Mean Density Bonus with adaptive radius
    density_bonus = np.zeros_like(distance_matrix)
    
    # Compute node density using adaptive radius
    node_densities = np.zeros(n)
    for i in range(n):
        # Use median distance to k_density nearest neighbors as radius
        k_density = min(6, n-1)
        valid_dists = safe_dist[i, safe_dist[i] < np.inf]
        if len(valid_dists) > 0:
            radius = np.median(np.partition(valid_dists, k_density)[:k_density])
            
            # Count nodes within radius (excluding self)
            within_radius = np.sum(safe_dist[i] <= radius) - 1  # Subtract self
            node_densities[i] = within_radius / (n - 1)  # Normalize
    
    # Apply geometric mean density bonus for candidate edges
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Geometric mean of normalized densities
                geo_mean = np.sqrt(node_densities[i] * node_densities[j] + eps)
                
                # Bonus scales with square root of geometric mean
                if geo_mean > 0.05:  # Threshold to avoid noise
                    density_bonus[i, j] = np.sqrt(geo_mean) * 0.25
                else:
                    density_bonus[i, j] = -0.05  # Small penalty for isolated edges
    
    # Make density bonus symmetric
    density_bonus = np.maximum(density_bonus, density_bonus.T)
    
    # 5. Base inverse distance with adaptive scaling
    inv_dist = np.zeros_like(distance_matrix)
    for i in range(n):
        valid_mask = safe_dist[i] < np.inf
        if np.any(valid_mask):
            # Local scaling by median distance
            local_median = np.median(safe_dist[i, valid_mask])
            inv_dist[i, valid_mask] = local_median / (safe_dist[i, valid_mask] + eps)
    
    # 6. Combine factors with improved weighting
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base score with rank weighting
    base_score = inv_dist * rank_weights
    
    # Apply regret penalty with exponential decay
    regret_penalty = np.exp(-1.5 * regret_matrix)
    score_with_regret = base_score * regret_penalty
    
    # Add density bonus multiplicatively
    heuristics_matrix = score_with_regret * (1.0 + density_bonus)
    
    # 7. MST-based reinforcement for smaller instances
    if n <= 500:
        try:
            mst = minimum_spanning_tree(distance_matrix).toarray()
            mst_mask = (mst + mst.T) > 0
            # Boost MST edges by 20%
            heuristics_matrix[mst_mask] *= 1.2
        except:
            pass
    
    # 8. Enhanced sparsification and connectivity preservation
    heuristics_matrix[~candidate_mask] = 0
    
    # Ensure each node has at least 3 promising connections
    min_connections = min(3, n-1)
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero_indices = np.where(row > 0)[0]
        
        if len(nonzero_indices) < min_connections:
            # Find top min_connections edges by base score
            top_indices = np.argsort(base_score[i])[-min_connections:]
            for idx in top_indices:
                if i != idx:
                    # Set to 60th percentile of non-zero values or minimum threshold
                    nonzero_vals = row[row > 0]
                    if len(nonzero_vals) > 0:
                        target_val = np.percentile(nonzero_vals, 60) if len(nonzero_vals) > 2 else np.mean(nonzero_vals)
                    else:
                        target_val = np.max(base_score[i]) * 0.1
                    heuristics_matrix[i, idx] = max(heuristics_matrix[i, idx], target_val)
                    heuristics_matrix[idx, i] = max(heuristics_matrix[idx, i], target_val)
    
    # 9. Final symmetrization and normalization
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1] range with robust scaling
    max_val = np.max(heuristics_matrix)
    if max_val > eps:
        # Use 99th percentile to avoid extreme values dominating
        robust_max = np.percentile(heuristics_matrix[heuristics_matrix > 0], 99) if np.sum(heuristics_matrix > 0) > 0 else max_val
        scale_factor = min(max_val, robust_max * 1.5)
        heuristics_matrix = heuristics_matrix / scale_factor
        heuristics_matrix = np.clip(heuristics_matrix, 0, 1)
    
    return heuristics_matrix
