import numpy as np
from scipy import stats

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge prioritization.
    
    Combines multiple factors to identify promising edges and sparsifies
    the matrix by setting low-potential edges to zero.
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance (handling zero distances)
    epsilon = 1e-10
    inv_dist = 1 / (distance_matrix + epsilon * (distance_matrix == 0))
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Local clustering coefficient - edges within dense clusters get bonus
    k_neighbors = max(2, min(5, n // 4))  # Adaptive neighborhood size
    cluster_weight = np.zeros((n, n))
    for i in range(n):
        # Find k nearest neighbors for each node
        neighbor_indices = np.argsort(distance_matrix[i])[1:k_neighbors+1]
        # Create adjacency among neighbors
        for idx_i, n_i in enumerate(neighbor_indices):
            for n_j in neighbor_indices[idx_i+1:]:
                if distance_matrix[n_i, n_j] > 0:
                    cluster_weight[i, n_i] += 1 / distance_matrix[n_i, n_j]
                    cluster_weight[i, n_j] += 1 / distance_matrix[n_i, n_j]
    
    # Normalize cluster weights
    if cluster_weight.max() > 0:
        cluster_weight = cluster_weight / cluster_weight.max()
    
    # 3. Entropy-based normalization - penalize nodes with uniform distances
    distance_entropy = np.zeros(n)
    for i in range(n):
        valid_dists = distance_matrix[i][distance_matrix[i] > 0]
        if len(valid_dists) > 1:
            distance_entropy[i] = stats.entropy(valid_dists + epsilon)
    
    # Normalize entropy and convert to weighting factor
    if distance_entropy.max() > 0:
        entropy_factor = 1 - (distance_entropy / distance_entropy.max())
    else:
        entropy_factor = np.ones(n)
    
    # 4. Symmetric edge potential - average of both directions
    symmetric_potential = (inv_dist + inv_dist.T) / 2
    
    # 5. Combine all factors with weighted geometric mean
    alpha, beta, gamma = 0.5, 0.3, 0.2  # Weight parameters
    heuristic_matrix = np.zeros((n, n))
    
    for i in range(n):
        for j in range(i+1, n):  # Only upper triangle for symmetric matrix
            if distance_matrix[i, j] > 0:
                # Base inverse distance with entropy adjustment
                base_score = inv_dist[i, j] * (
                    (entropy_factor[i] + entropy_factor[j]) / 2
                )
                
                # Add cluster density bonus
                cluster_bonus = (cluster_weight[i, j] + cluster_weight[j, i]) / 2
                
                # Combine with symmetric potential
                combined = (
                    (base_score ** alpha) *
                    ((1 + cluster_bonus) ** beta) *
                    (symmetric_potential[i, j] ** gamma)
                )
                
                heuristic_matrix[i, j] = combined
                heuristic_matrix[j, i] = combined
    
    # 6. Sparsification - keep only top edges
    # Keep edges where at least one endpoint considers it among top k
    k_sparse = max(2, min(6, n // 3))
    mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get indices of top k heuristic values for this node
        row_heuristics = heuristic_matrix[i]
        top_k_indices = np.argsort(row_heuristics)[-k_sparse:]
        mask[i, top_k_indices] = True
    
    # Make mask symmetric and apply
    mask = mask | mask.T
    heuristic_matrix[~mask] = 0
    
    # 7. Final normalization
    max_val = heuristic_matrix.max()
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
    
    return heuristic_matrix
