import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection.
    
    Returns a matrix of prior indicators where higher values indicate
    more promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (handling zeros to avoid division by infinity)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.diag_indices(n)] = 0
    
    # 2. Minimum Spanning Tree edge bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = np.where(mst + mst.T > 0, 1.5, 1.0)
    
    # 3. Local connectivity score
    # For each node, identify promising edges to its nearest neighbors
    k = max(3, n // 10)  # Adaptive number of nearest neighbors
    local_mask = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get k+1 nearest neighbors (including self)
        indices = np.argsort(distance_matrix[i])[:k+1]
        # Remove self
        indices = indices[indices != i]
        local_mask[i, indices] = 1
    
    # Symmetrize local mask
    local_mask = np.maximum(local_mask, local_mask.T)
    
    # 4. Cluster cohesion bonus using normalized distances
    # Normalize distances to [0,1] range for consistent thresholds
    max_dist = np.max(distance_matrix)
    if max_dist > 0:
        norm_dist = distance_matrix / max_dist
    else:
        norm_dist = distance_matrix
    
    # Compute threshold: median of normalized distances among promising edges
    promising_edges = np.where(local_mask, norm_dist, np.nan)
    median_threshold = np.nanmedian(promising_edges)
    
    if np.isnan(median_threshold):
        median_threshold = 0.5
    
    # Soft thresholding: sigmoid function for continuous transition
    threshold_factor = 1.0 / (1.0 + np.exp(10.0 * (norm_dist - median_threshold)))
    
    # 5. Combine all factors with sparsification
    heuristic_matrix = inv_dist * mst_bonus * local_mask * threshold_factor
    
    # 6. Apply soft sparsification: zero out edges below adaptive threshold
    non_zero_entries = heuristic_matrix[heuristic_matrix > 0]
    if len(non_zero_entries) > 0:
        # Keep top 30% of edges or at least n edges
        threshold_value = np.percentile(non_zero_entries, 70)
        min_threshold = np.sort(non_zero_entries)[-min(n, len(non_zero_entries))]
        threshold_value = max(threshold_value, min_threshold)
        
        heuristic_matrix[heuristic_matrix < threshold_value] = 0
    
    # 7. Symmetrize final heuristic matrix
    heuristic_matrix = np.maximum(heuristic_matrix, heuristic_matrix.T)
    
    # Normalize for consistent scale
    max_val = np.max(heuristic_matrix)
    if max_val > 0:
        heuristic_matrix = heuristic_matrix / max_val
    
    return heuristic_matrix
