import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection that combines multiple factors
    and sparsifies unpromising edges.
    
    Features:
    1. Candidate list sparsification (keep top k edges per node)
    2. Regret-based connectivity (penalize edges that leave nodes with poor alternatives)
    3. Distance-rank weighted scoring (better ranks get higher weights)
    4. Neighbor density bonus (edges in dense regions get bonuses)
    """
    n = distance_matrix.shape[0]
    
    # Avoid division by zero and self-connections
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    inverse_dist = 1.0 / safe_dist
    np.fill_diagonal(inverse_dist, 0)
    
    # 1. Create candidate lists (top k edges for each node)
    k = max(3, int(np.sqrt(n)))  # Adaptive candidate list size
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        neighbors = np.argsort(safe_dist[i])[:k]
        candidate_mask[i, neighbors] = True
    
    # Make candidate list symmetric (if i considers j, j should consider i)
    candidate_mask = candidate_mask | candidate_mask.T
    
    # 2. Calculate regret-based penalties
    regret_penalty = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get distances to all other nodes
        dists = safe_dist[i].copy()
        
        # Find the best and second best distances (for regret calculation)
        sorted_idx = np.argsort(dists)
        if len(sorted_idx) >= 2:
            best_dist = dists[sorted_idx[0]]
            second_best_dist = dists[sorted_idx[1]]
            
            # Calculate regret for each edge: how much worse than second best
            for j in range(n):
                if i != j:
                    # If this is the best edge, regret is 0 (no penalty)
                    if np.isclose(dists[j], best_dist):
                        regret = 0
                    else:
                        # Regret is difference between this edge and second best
                        regret = dists[j] - second_best_dist
                    
                    # Normalize regret by average distance
                    avg_dist = np.mean(dists[dists < np.inf])
                    regret_penalty[i, j] = regret / (avg_dist + 1e-10)
    
    # 3. Calculate rank-based weights
    rank_weight = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get sorted indices by distance
        sorted_idx = np.argsort(safe_dist[i])
        
        # Assign weights based on rank (exponential decay)
        for rank, j in enumerate(sorted_idx):
            if i != j:
                # Higher rank (closer) gets higher weight
                decay_factor = 0.7  # Controls how fast weights decay with rank
                rank_weight[i, j] = decay_factor ** rank
    
    # 4. Calculate neighbor density bonus
    density_bonus = np.zeros_like(distance_matrix)
    
    # For each node, count how many other nodes are within median distance
    median_dist = np.median(safe_dist[safe_dist < np.inf])
    
    for i in range(n):
        # Find nodes within median distance
        close_nodes = safe_dist[i] <= median_dist
        density_i = np.sum(close_nodes) - 1  # Exclude self
        
        for j in range(n):
            if i != j:
                # If both nodes are in dense regions, give bonus
                close_nodes_j = safe_dist[j] <= median_dist
                density_j = np.sum(close_nodes_j) - 1
                
                # Bonus proportional to geometric mean of densities
                if density_i > 0 and density_j > 0:
                    geo_mean_density = np.sqrt(density_i * density_j)
                    density_bonus[i, j] = np.log1p(geo_mean_density) / 10
    
    # 5. Combine all factors
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base: inverse distance with rank weighting
    base_score = inverse_dist * rank_weight
    
    # Apply regret penalty (lower regret is better, so subtract)
    regret_adjusted = base_score * np.exp(-regret_penalty)
    
    # Add density bonus
    heuristics_matrix = regret_adjusted * (1 + density_bonus)
    
    # 6. Apply sparsification: zero out non-candidate edges
    heuristics_matrix[~candidate_mask] = 0
    
    # Normalize to keep values in reasonable range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    # Ensure symmetry (since TSP is undirected)
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    # Zero out diagonal
    np.fill_diagonal(heuristics_matrix, 0)
    
    return heuristics_matrix
