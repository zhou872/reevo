import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Mutated heuristics for TSP edge evaluation with streamlined regret modeling,
    adaptive density scoring, and efficient candidate sparsification.
    
    Features:
    1. Regret-Based Penalty: Exponential decay based on second-best distance normalization.
    2. Adaptive Density Bonus: Geometric mean of node densities from median k-NN distances.
    3. Candidate List Sparsification: Top-k edges per node with k = max(3, sqrt(n)), enforced symmetrically.
    4. Symmetric Normalization: Ensures matrix symmetry and normalizes scores to [0,1].
    """
    n = distance_matrix.shape[0]
    # Safe distance matrix (avoid self-loops)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    eps = 1e-12
    
    # Base inverse distance normalized by global mean distance
    mean_dist = np.mean(safe_dist[safe_dist < np.inf])
    inv_dist = mean_dist / safe_dist  # Symmetric if input is symmetric
    
    # 1. Adaptive candidate sparsification
    k_candidate = max(3, int(np.sqrt(n)))  # From knowledge graph
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        neighbors = np.argsort(safe_dist[i])[:k_candidate]
        candidate_mask[i, neighbors] = True
    # Enforce symmetry
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # 2. Regret-based penalty with exponential decay
    # Compute second-best distance for each node
    second_best_dists = np.zeros(n)
    for i in range(n):
        dists = safe_dist[i]
        sorted_dists = np.sort(dists)
        if len(sorted_dists) > 1:
            second_best_dists[i] = sorted_dists[1]  # Second smallest distance
        else:
            second_best_dists[i] = mean_dist  # Fallback
    
    # Compute penalty matrix: (distance - second_best) / mean_dist
    penalty_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j:
                penalty_matrix[i, j] = (safe_dist[i, j] - second_best_dists[i]) / (mean_dist + eps)
    # Apply exponential decay: decay_factor = exp(-penalty)
    decay_factor = np.exp(-penalty_matrix)
    
    # 3. Adaptive density bonus
    # Node density as inverse of median distance to k-nearest neighbors
    k_density = min(5, n - 1)
    densities = np.zeros(n)
    for i in range(n):
        dists = safe_dist[i]
        # Get k-nearest distances (excluding inf)
        k_nearest = np.partition(dists, k_density)[:k_density + 1]
        k_nearest = k_nearest[k_nearest < np.inf]
        if len(k_nearest) > 0:
            median_dist = np.median(k_nearest)
            if median_dist > 0:
                densities[i] = 1.0 / median_dist
    # Density bonus: geometric mean of densities for candidate edges
    density_bonus = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            if candidate_mask[i, j]:
                geo_mean = np.sqrt(densities[i] * densities[j])
                density_bonus[i, j] = geo_mean * 0.1  # Scaling factor
                density_bonus[j, i] = density_bonus[i, j]  # Symmetrize
    
    # 4. Combine factors
    heuristics_matrix = inv_dist * decay_factor * (1.0 + density_bonus)
    
    # Apply candidate mask to sparsify
    heuristics_matrix[~candidate_mask] = 0
    
    # Ensure each node has at least 2 connections for basic connectivity
    for i in range(n):
        row = heuristics_matrix[i]
        nonzero_count = np.count_nonzero(row)
        if nonzero_count < 2:
            top_indices = np.argsort(row)[-2:]
            max_val = np.max(row)
            if max_val > 0:
                boost_val = max_val * 1.1
            else:
                boost_val = 1.0
            heuristics_matrix[i, top_indices] = boost_val
            heuristics_matrix[top_indices, i] = boost_val
    
    # Symmetrize and zero diagonal
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix /= max_val
    
    return heuristics_matrix
