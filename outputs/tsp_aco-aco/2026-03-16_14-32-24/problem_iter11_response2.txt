import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge evaluation heuristics with streamlined computation,
    improved regret modeling, and adaptive normalization.
    
    Key Improvements:
    1. Simplified regret calculation using normalized exponential penalty
    2. Adaptive rank weighting with variance-based decay
    3. Efficient spatial clustering via joint neighborhood density
    4. Improved candidate list construction with symmetry preservation
    """
    n = distance_matrix.shape[0]
    eps = 1e-12
    
    # Safe distance matrix (avoid division by zero)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Adaptive candidate list - O(sqrt(n)) nearest neighbors
    k_candidate = max(4, int(np.sqrt(n)))
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Build candidate list using nearest neighbors
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        neighbors = np.argsort(safe_dist[i])[:k_candidate]
        candidate_mask[i, neighbors] = True
    
    # Enforce symmetry in candidate list
    candidate_mask = np.maximum(candidate_mask, candidate_mask.T)
    
    # 2. Normalized Exponential Regret Penalty
    regret_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        dists = safe_dist[i].copy()
        
        # Find best and second-best distances for each node
        if n > 2:
            # Get two smallest distances (excluding self/inf)
            sorted_dists = np.partition(dists, 2)[:3]
            d_best = sorted_dists[0]
            d_second = sorted_dists[1] if sorted_dists[1] < np.inf else d_best
            
            # Compute regret penalty for each edge
            for j in range(n):
                if i != j:
                    d_ij = dists[j]
                    if d_second > d_best + eps:
                        # Normalized exponential regret penalty
                        regret = np.exp(-(d_ij - d_best) / (d_second - d_best + eps))
                    else:
                        # Fallback: simple inverse weighting for equal distances
                        regret = 1.0 / (1.0 + d_ij - d_best)
                    
                    regret_matrix[i, j] = regret
    
    # 3. Adaptive Rank-Weighted Scoring
    rank_weights = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get sorted distances (excluding self)
        sorted_indices = np.argsort(safe_dist[i])
        sorted_dists = safe_dist[i, sorted_indices]
        
        # Adaptive decay rate based on distance variability
        if n > 5:
            # Use variability in nearest 5 distances
            nearest_dists = sorted_dists[1:min(6, n)]
            if len(nearest_dists) > 1:
                dist_std = np.std(nearest_dists)
                dist_mean = np.mean(nearest_dists)
                # Decay adapts to variability: more uniform -> faster decay
                decay_rate = 0.4 + 0.4 * np.tanh(dist_std / (dist_mean + eps))
                decay_rate = np.clip(decay_rate, 0.3, 0.8)
            else:
                decay_rate = 0.6
        else:
            decay_rate = 0.7
        
        # Assign exponential rank weights
        for rank, j in enumerate(sorted_indices):
            if i != j:
                rank_weights[i, j] = decay_rate ** rank
    
    # 4. Joint Spatial Density Bonus
    density_bonus = np.zeros_like(distance_matrix)
    
    # Precompute node densities using k-nearest neighbors
    k_density = min(5, n-1)
    node_densities = np.zeros(n)
    
    for i in range(n):
        if n > 1:
            # Estimate local density as inverse of average distance to k nearest neighbors
            k_nearest = np.partition(safe_dist[i], k_density)[:k_density+1]
            node_densities[i] = 1.0 / (np.mean(k_nearest[k_nearest < np.inf]) + eps)
        else:
            node_densities[i] = 1.0
    
    # Normalize densities
    if np.max(node_densities) > 0:
        node_densities = node_densities / np.max(node_densities)
    
    # Compute joint density bonus for candidate edges
    for i in range(n):
        for j in range(i+1, n):
            if candidate_mask[i, j]:
                # Joint density as product of individual densities
                joint_density = node_densities[i] * node_densities[j]
                
                # Additional bonus for edges connecting high-density regions
                if joint_density > 0.25:  # Both endpoints are relatively dense
                    # Bonus proportional to geometric mean of densities
                    density_bonus[i, j] = np.sqrt(joint_density) * 0.2
                else:
                    # Small penalty for isolated edges
                    density_bonus[i, j] = -0.05
    
    # Make symmetric
    density_bonus = np.maximum(density_bonus, density_bonus.T)
    
    # 5. Combine factors with improved weighting
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    # Base inverse distance (normalized by median)
    inv_dist = 1.0 / safe_dist
    median_dist = np.median(safe_dist[safe_dist < np.inf])
    inv_dist = inv_dist * median_dist
    
    # Combine: inverse distance × rank weights × regret penalty
    base_score = inv_dist * rank_weights * regret_matrix
    
    # Apply density bonus (additive, not multiplicative)
    heuristics_matrix = base_score * (1.0 + density_bonus)
    
    # 6. Sparsify using candidate mask
    heuristics_matrix[~candidate_mask] = 0
    
    # 7. Ensure connectivity: each node must have at least 2 non-zero edges
    for i in range(n):
        row_nonzero = np.nonzero(heuristics_matrix[i])[0]
        if len(row_nonzero) < 2:
            # Find top 2 edges by base score (including zero ones)
            top_indices = np.argsort(base_score[i])[-2:]
            # Set to max of existing non-zero values or a small positive value
            max_val = np.max(heuristics_matrix[i])
            boost_val = max(max_val, np.max(base_score[i]) * 0.1)
            heuristics_matrix[i, top_indices] = boost_val
            heuristics_matrix[top_indices, i] = boost_val
    
    # 8. Symmetrize final matrix
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    np.fill_diagonal(heuristics_matrix, 0)
    
    # 9. Adaptive normalization
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        # Use sigmoid-like normalization for better contrast
        scale = 1.0 / max_val
        heuristics_matrix = heuristics_matrix * scale
        # Apply soft threshold: boost values above median
        median_val = np.median(heuristics_matrix[heuristics_matrix > 0])
        if median_val > 0:
            heuristics_matrix = np.where(
                heuristics_matrix > median_val,
                heuristics_matrix * 1.2,
                heuristics_matrix * 0.8
            )
    
    # Final normalization to [0, 1]
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
