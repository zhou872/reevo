import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple spatial and 
    topological factors with intelligent sparsification.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (closer nodes are generally better)
    inv_dist = np.divide(1.0, distance_matrix + np.eye(n) * 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Distance gradient penalty: penalize edges connecting to very differently distanced nodes
    row_means = distance_matrix.mean(axis=1, keepdims=True)
    col_means = distance_matrix.mean(axis=0, keepdims=True)
    distance_consistency = 1.0 / (1.0 + np.abs(row_means - col_means.T) / row_means.mean())
    
    # 3. Mutual nearest-neighbor bonus
    k = max(2, n // 10)  # Adaptive neighbor count
    top_k_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        idx = np.argsort(distance_matrix[i])[1:k+1]  # Exclude self
        top_k_mask[i, idx] = True
    
    mutual_nn = top_k_mask & top_k_mask.T
    mutual_bonus = 1.0 + 2.0 * mutual_nn
    
    # 4. Connectivity-centric reweighting: prefer edges that connect to nodes with few good alternatives
    node_quality = inv_dist.sum(axis=1)
    connectivity_need = 1.0 / (1e-10 + node_quality[:, None] * node_quality[None, :])
    connectivity_need = connectivity_need / connectivity_need.mean()
    
    # 5. Combine all factors
    heuristic_matrix = (inv_dist * 
                       distance_consistency * 
                       mutual_bonus * 
                       connectivity_need)
    
    # 6. Intelligent sparsification: keep only promising edges
    # Keep top edges per node + ensure connectivity
    sparsity_mask = np.zeros_like(heuristic_matrix, dtype=bool)
    
    # Keep top m edges per node (m adaptive based on problem size)
    m = max(3, n // 8)
    for i in range(n):
        idx = np.argsort(heuristic_matrix[i])[-m:]
        sparsity_mask[i, idx] = True
    
    # Ensure graph connectivity: keep minimum spanning tree edges
    # Prim's algorithm for MST (ensures connectivity)
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    while not visited.all():
        # Find minimum edge from visited to unvisited
        mask_visited = visited[:, None] & ~visited[None, :]
        if not mask_visited.any():
            break
        valid_edges = distance_matrix.copy()
        valid_edges[~mask_visited] = np.inf
        i, j = np.unravel_index(np.argmin(valid_edges), distance_matrix.shape)
        sparsity_mask[i, j] = True
        sparsity_mask[j, i] = True
        visited[j] = True
    
    # Apply sparsity mask and normalize
    heuristic_matrix[~sparsity_mask] = 0
    heuristic_matrix = (heuristic_matrix - heuristic_matrix.min()) / (
        heuristic_matrix.max() - heuristic_matrix.min() + 1e-10)
    
    # Symmetrize (for undirected TSP)
    heuristic_matrix = np.maximum(heuristic_matrix, heuristic_matrix.T)
    
    return heuristic_matrix
