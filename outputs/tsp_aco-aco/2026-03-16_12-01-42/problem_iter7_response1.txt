import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with improved regret modeling
    and adaptive fusion of multiple quality indicators.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base inverse distance with stability epsilon ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. IMPROVED: Normalized Regret Penalization ===
    # Prioritize edges with high regret (poor alternatives) using scale-invariant measure
    regret_factor = np.ones_like(distance_matrix)
    
    for i in range(n):
        # Get distances from node i to all others (excluding self)
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        
        # Find two nearest neighbors for normalized regret calculation
        nearest_indices = np.argsort(dist_i)[:2]
        nearest_dists = dist_i[nearest_indices]
        
        if len(nearest_dists) < 2:
            continue
            
        # For each potential edge from i
        for j in range(n):
            if i == j:
                continue
                
            d_ij = distance_matrix[i, j]
            
            # Compute normalized regret: (alternative - current) / current
            if j == nearest_indices[0]:
                # If j is the nearest, use second nearest as alternative
                alternative_dist = nearest_dists[1]
                if d_ij > epsilon:
                    norm_regret = (alternative_dist - d_ij) / d_ij
                else:
                    norm_regret = 0
            else:
                # If j is not the nearest, use nearest as alternative
                alternative_dist = nearest_dists[0]
                if d_ij > epsilon:
                    norm_regret = (alternative_dist - d_ij) / d_ij
                else:
                    norm_regret = 0
            
            # Convert to boost factor: edges with high regret get higher priority
            # Use sigmoid-like transformation to bound the effect
            regret_boost = 1.0 + 1.5 * np.tanh(norm_regret)
            regret_factor[i, j] = regret_boost
    
    # === 3. Two-step Lookahead Spatial Scoring ===
    # Consider not just immediate neighbors but neighbors of neighbors
    spatial_scores = np.zeros_like(distance_matrix)
    k_spatial = min(5, n - 1)
    
    # Pre-compute extended spatial quality for each node
    node_quality = np.zeros(n)
    for i in range(n):
        # Get distances to other nodes
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Find k_spatial nearest neighbors
        nearest_indices = np.argsort(dists)[:k_spatial]
        nearest_dists = dists[nearest_indices]
        
        # Compute two-step quality: average of neighbor's connection qualities
        two_step_quality = 0
        for idx in nearest_indices:
            # Get second-order distances
            neighbor_dists = distance_matrix[idx].copy()
            neighbor_dists[idx] = np.inf
            # Quality of this neighbor is harmonic mean of its nearest distances
            neighbor_nearest = np.partition(neighbor_dists, k_spatial)[:k_spatial]
            with np.errstate(divide='ignore', invalid='ignore'):
                neighbor_quality = k_spatial / np.sum(1 / (neighbor_nearest + epsilon))
            two_step_quality += neighbor_quality
        
        if len(nearest_indices) > 0:
            node_quality[i] = two_step_quality / len(nearest_indices)
    
    # Combine node qualities for edges
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Geometric mean to emphasize balanced connections
            spatial_scores[i, j] = np.sqrt(node_quality[i] * node_quality[j])
    
    # Normalize spatial scores
    if np.max(spatial_scores) > 0:
        non_zero = spatial_scores[spatial_scores > 0]
        if len(non_zero) > 0:
            spatial_min, spatial_max = np.min(non_zero), np.max(non_zero)
            if spatial_max > spatial_min:
                spatial_scores = 0.5 + (spatial_scores - spatial_min) / (spatial_max - spatial_min)
    
    # === 4. Efficient MST Preservation Factor ===
    mst_factor = np.ones_like(distance_matrix)
    if n > 1:
        # Build MST using Prim's algorithm
        in_mst = np.zeros(n, dtype=bool)
        in_mst[0] = True
        min_edge = distance_matrix[0].copy()
        min_edge[0] = np.inf
        parent = np.zeros(n, dtype=int)
        
        for _ in range(n - 1):
            # Find minimum edge connecting MST to non-MST
            valid_mask = ~in_mst
            if not np.any(valid_mask):
                break
            min_val = np.min(min_edge[valid_mask])
            min_idx = np.argwhere(min_edge == min_val)[0][0]
            
            # Mark as in MST
            in_mst[min_idx] = True
            p = parent[min_idx]
            
            # Boost MST edges based on spatial score
            mst_boost = 1.0 + 0.8 * spatial_scores[p, min_idx]
            mst_factor[p, min_idx] = mst_boost
            mst_factor[min_idx, p] = mst_boost
            
            # Update min_edge for remaining vertices
            for v in range(n):
                if not in_mst[v] and distance_matrix[min_idx, v] < min_edge[v]:
                    min_edge[v] = distance_matrix[min_idx, v]
                    parent[v] = min_idx
    
    # === 5. IMPROVED: Multiplicative Factor Fusion ===
    # Combine all factors multiplicatively with controlled exponents
    heuristics = inv_dist.copy()
    heuristics *= regret_factor  # Regret-based adjustment
    heuristics *= spatial_scores  # Spatial lookahead
    heuristics *= mst_factor  # MST preservation
    
    # === 6. Adaptive Sparsification with Regret Preservation ===
    adaptive_k = min(10, max(3, n // 3))
    
    for i in range(n):
        row = heuristics[i].copy()
        
        if adaptive_k < n - 1:
            # Separate edges by regret level
            high_regret_mask = regret_factor[i] > 1.2  # Edges with significant regret
            normal_mask = ~high_regret_mask & (np.arange(n) != i)
            
            # Always keep high-regret edges
            keep_mask = high_regret_mask.copy()
            
            # Fill remaining slots with best normal edges
            normal_indices = np.where(normal_mask)[0]
            if len(normal_indices) > 0:
                normal_scores = row[normal_indices]
                k_normal = min(len(normal_indices), adaptive_k - np.sum(keep_mask))
                if k_normal > 0:
                    top_normal = np.argsort(normal_scores)[-k_normal:]
                    keep_mask[normal_indices[top_normal]] = True
            
            # Ensure at least 2 edges are kept
            if np.sum(keep_mask) < 2:
                # Fall back to top edges by combined score
                top2 = np.argsort(row)[-2:]
                keep_mask[top2] = True
            
            # Apply sparsification
            row[~keep_mask] = 0
            heuristics[i] = row
    
    # === 7. Symmetrization with Mutual Preference ===
    # Emphasize edges where both endpoints have high scores
    symmetric_heuristics = np.zeros_like(heuristics)
    for i in range(n):
        for j in range(n):
            if i == j:
                symmetric_heuristics[i, j] = 0
            else:
                # Weight more heavily if both directions agree
                if heuristics[i, j] > 0 and heuristics[j, i] > 0:
                    symmetric_heuristics[i, j] = np.sqrt(heuristics[i, j] * heuristics[j, i])
                else:
                    symmetric_heuristics[i, j] = max(heuristics[i, j], heuristics[j, i] * 0.5)
    
    # === 8. Final Normalization ===
    np.fill_diagonal(symmetric_heuristics, 0)
    
    # Scale to [0, 1] range
    if np.max(symmetric_heuristics) > 0:
        symmetric_heuristics = symmetric_heuristics / np.max(symmetric_heuristics)
    
    return symmetric_heuristics
