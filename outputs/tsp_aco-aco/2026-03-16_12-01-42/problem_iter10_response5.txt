import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse import csr_matrix
from collections import defaultdict
import math

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge prioritization with graph-theoretic centrality,
    hierarchical clustering density, and adaptive multi-objective fusion.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base Inverse Distance with Adaptive Scaling ===
    # Scale distances for better numerical stability
    scale_factor = np.median(distance_matrix[distance_matrix > 0])
    scaled_dist = distance_matrix / (scale_factor + epsilon)
    inv_dist = 1 / (scaled_dist + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Graph-Theoretic Edge Betweenness Approximation ===
    betweenness_scores = np.ones_like(distance_matrix)
    
    if n > 3:
        # Approximate edge betweenness using random walks
        num_walks = min(1000, n * 20)
        walk_length = min(10, n // 2 + 1)
        
        for _ in range(num_walks):
            # Start from random node
            current = np.random.randint(n)
            visited = set([current])
            
            for step in range(walk_length):
                # Transition probability inversely proportional to distance
                probs = 1 / (distance_matrix[current] + epsilon)
                probs[current] = 0
                probs = probs / (probs.sum() + epsilon)
                
                # Sample next node
                next_node = np.random.choice(n, p=probs)
                
                # Update betweenness for edge (current, next_node)
                betweenness_scores[current, next_node] += 1
                betweenness_scores[next_node, current] += 1
                
                current = next_node
                visited.add(current)
        
        # Normalize betweenness scores
        max_between = np.max(betweenness_scores)
        if max_between > 0:
            betweenness_scores = betweenness_scores / max_between
    
    # === 3. Hierarchical Cluster Density Scoring ===
    cluster_scores = np.ones_like(distance_matrix)
    
    if n > 4:
        # Build hierarchical clusters using MST
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_array = mst.toarray()
            
            # Extract clusters at different levels
            clusters_by_level = []
            for threshold in np.percentile(distance_matrix[distance_matrix > 0], [25, 50, 75]):
                # Connect nodes within threshold distance
                connected = distance_matrix <= threshold
                np.fill_diagonal(connected, False)
                
                # Find connected components
                visited = [False] * n
                clusters = []
                
                for i in range(n):
                    if not visited[i]:
                        stack = [i]
                        cluster = []
                        
                        while stack:
                            node = stack.pop()
                            if not visited[node]:
                                visited[node] = True
                                cluster.append(node)
                                for j in range(n):
                                    if connected[node, j] and not visited[j]:
                                        stack.append(j)
                        
                        if len(cluster) > 1:
                            clusters.append(cluster)
                
                clusters_by_level.append(clusters)
            
            # Compute cluster density scores
            for i in range(n):
                for j in range(i + 1, n):
                    if distance_matrix[i, j] > 0:
                        score = 0
                        weight_sum = 0
                        
                        for level_idx, clusters in enumerate(clusters_by_level):
                            weight = 1.0 / (level_idx + 2)  # Higher weight for tighter clusters
                            found_together = False
                            
                            for cluster in clusters:
                                if i in cluster and j in cluster:
                                    # Nodes are in same cluster at this level
                                    density = len(cluster) / (np.mean([
                                        distance_matrix[a, b] 
                                        for a in cluster for b in cluster if a < b
                                    ]) + epsilon)
                                    
                                    # Score based on cluster density and distance
                                    cluster_score = density / (distance_matrix[i, j] + epsilon)
                                    score += weight * cluster_score
                                    found_together = True
                                    break
                            
                            if not found_together:
                                # Penalty for nodes not in same cluster
                                score -= 0.5 * weight
                            
                            weight_sum += weight
                        
                        if weight_sum > 0:
                            normalized_score = 0.5 + np.tanh(score / (weight_sum + epsilon))
                            cluster_scores[i, j] = cluster_scores[j, i] = normalized_score
        except:
            pass
    
    # === 4. Dynamic Regret with Entropy Weighting ===
    regret_scores = np.ones_like(distance_matrix)
    
    if n > 2:
        # Compute distance distribution entropy for each node
        node_entropies = np.zeros(n)
        
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Normalize to probabilities
            probs = 1 / (dists + epsilon)
            probs = probs / (probs.sum() + epsilon)
            
            # Compute entropy
            entropy = -np.sum(probs * np.log(probs + epsilon))
            node_entropies[i] = entropy
        
        # Normalize entropies
        max_entropy = np.max(node_entropies)
        if max_entropy > 0:
            node_entropies = node_entropies / max_entropy
        
        # Dynamic regret with entropy-adaptive order
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Order based on entropy (higher entropy = more uniform distances = higher order)
            order = max(2, min(5, int(3 + 2 * node_entropies[i])))
            order = min(order, n - 1)
            
            if len(dists) >= order + 1:
                # Get order+1 smallest distances
                indices = np.argpartition(dists, order + 1)[:order + 1]
                sorted_indices = indices[np.argsort(dists[indices])]
                sorted_dists = dists[sorted_indices]
                
                best_dist = sorted_dists[0]
                
                for j in range(n):
                    if i != j:
                        d_ij = distance_matrix[i, j]
                        
                        # Find position of j
                        if j in sorted_indices:
                            pos = np.where(sorted_indices == j)[0][0]
                            # Use the order-th distance as alternative
                            alt_idx = min(order, len(sorted_dists) - 1)
                            alt_dist = sorted_dists[alt_idx]
                        else:
                            alt_dist = sorted_dists[min(order, len(sorted_dists) - 1)]
                        
                        if d_ij > 0:
                            # Weighted regret considering entropy
                            regret_weight = 0.5 + 0.5 * node_entropies[i]
                            regret = (alt_dist - d_ij) / (d_ij + epsilon)
                            regret_scores[i, j] = 1.0 + regret_weight * np.tanh(regret)
    
    # === 5. Multi-objective Fusion with Structural Adaptation ===
    # Compute structural features for adaptive weighting
    structural_features = {}
    
    if n > 3:
        # Feature 1: Distance uniformity (entropy of all distances)
        flat_dists = distance_matrix[~np.eye(n, dtype=bool)].flatten()
        probs = flat_dists / (flat_dists.sum() + epsilon)
        structural_features['uniformity'] = -np.sum(probs * np.log(probs + epsilon))
        
        # Feature 2: Cluster tendency (ratio of MST edges to minimum possible edges)
        mst = minimum_spanning_tree(distance_matrix)
        mst_edges = np.count_nonzero(mst.toarray())
        structural_features['clustering'] = mst_edges / (n - 1)
        
        # Feature 3: Distance spread (coefficient of variation)
        mean_dist = np.mean(flat_dists)
        std_dist = np.std(flat_dists)
        structural_features['spread'] = std_dist / (mean_dist + epsilon)
    
    # Adaptive weights based on structural features
    if n <= 15:
        w_inv = 1.0
        w_between = 0.8
        w_cluster = 1.2
        w_regret = 1.0
    elif n <= 60:
        # Adjust based on structural features
        if 'uniformity' in structural_features:
            uniformity = structural_features['uniformity']
            clustering = structural_features['clustering']
            
            w_inv = 1.0 - 0.3 * uniformity
            w_between = 0.7 + 0.3 * clustering
            w_cluster = 1.0 + 0.4 * clustering
            w_regret = 0.8 + 0.4 * uniformity
        else:
            w_inv = 0.9
            w_between = 0.9
            w_cluster = 1.1
            w_regret = 1.0
    else:
        # Large instances
        w_inv = 0.8
        w_between = 1.0
        w_cluster = 1.3
        w_regret = 0.9
    
    # Multiplicative fusion with exponentiation
    heuristics = (inv_dist ** w_inv) * \
                 (betweenness_scores ** w_between) * \
                 (cluster_scores ** w_cluster) * \
                 (regret_scores ** w_regret)
    
    # === 6. MST Reinforcement with Degree Awareness ===
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_array = mst.toarray()
            mst_mask = (mst_array > 0) | (mst_array.T > 0)
            
            # Compute MST node degrees
            mst_degrees = np.sum(mst_mask, axis=1)
            max_degree = np.max(mst_degrees) + epsilon
            
            # Boost MST edges, more for high-degree nodes
            for i in range(n):
                for j in range(n):
                    if mst_mask[i, j]:
                        degree_factor = 0.5 + 0.5 * (mst_degrees[i] + mst_degrees[j]) / (2 * max_degree)
                        heuristics[i, j] *= (1.2 + 0.2 * degree_factor)
        except:
            pass
    
    # === 7. Connectivity-Preserving Sparsification ===
    for i in range(n):
        row = heuristics[i].copy()
        
        # Dynamic threshold based on rank statistics
        if n <= 20:
            keep_k = max(3, n // 2)
        elif n <= 100:
            keep_k = max(5, n // 3)
        else:
            keep_k = max(8, n // 4)
        
        keep_k = min(keep_k, n - 1)
        
        if keep_k > 0:
            # Get threshold for top keep_k edges
            if len(row) > keep_k:
                threshold = np.partition(row, -keep_k)[-keep_k]
            else:
                threshold = np.min(row)
            
            # Preserve MST edges and ensure connectivity
            preserve_mask = np.zeros(n, dtype=bool)
            
            if n > 2:
                try:
                    mst_i = minimum_spanning_tree(distance_matrix).toarray()[i] > 0
                    preserve_mask = preserve_mask | mst_i
                except:
                    pass
            
            # Also preserve edges to nearest neighbors for connectivity
            if n > 1:
                dists_i = distance_matrix[i].copy()
                dists_i[i] = np.inf
                nearest = np.argsort(dists_i)[:min(3, n-1)]
                preserve_mask[nearest] = True
            
            # Apply sparsification
            mask = (row >= threshold) | preserve_mask
            
            # Ensure at least 2 connections
            if np.sum(mask) < 2:
                top_idx = np.argsort(row)[-2:]
                mask[top_idx] = True
            
            row[~mask] = 0
            heuristics[i] = row
    
    # === 8. Symmetric Refinement ===
    # Geometric mean for symmetry (multiplicative fusion)
    heuristics_sym = np.sqrt(heuristics * heuristics.T)
    
    # Add mutual reinforcement
    mutual_reinforcement = np.minimum(heuristics_sym, heuristics_sym.T)
    heuristics_sym += 0.15 * mutual_reinforcement
    
    # Final cleanup
    np.fill_diagonal(heuristics_sym, 0)
    heuristics_sym = np.maximum(heuristics_sym, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_sym)
    if max_val > epsilon:
        heuristics_sym = heuristics_sym / max_val
    
    return heuristics_sym
