import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection that combines multiple factors
    and sparsifies unpromising edges.
    """
    n = len(distance_matrix)
    heuristics = np.zeros_like(distance_matrix)
    
    # Factor 1: Distance-based priority (inverse weighting)
    with np.errstate(divide='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    
    # Factor 2: Nearest neighbor priority
    for i in range(n):
        sorted_indices = np.argsort(distance_matrix[i])
        # Give higher priority to k nearest neighbors
        k = max(3, n // 10)  # Dynamic k based on problem size
        for rank, j in enumerate(sorted_indices[:k]):
            if i != j:
                inv_dist[i, j] *= (1.0 + (k - rank) / k)
    
    # Factor 3: MST (Minimum Spanning Tree) edge bonus
    mst_edges = _compute_mst_edges(distance_matrix)
    mst_bonus = np.zeros_like(distance_matrix)
    for i, j in mst_edges:
        mst_bonus[i, j] = 1.5
        mst_bonus[j, i] = 1.5
    inv_dist *= (1.0 + mst_bonus)
    
    # Factor 4: Proportional regret penalty
    # Penalize edges where the second best option is much worse
    regret_penalty = np.ones_like(distance_matrix)
    for i in range(n):
        # Find two smallest distances from node i (excluding self)
        row = distance_matrix[i].copy()
        row[i] = np.inf
        sorted_vals = np.sort(row)
        if len(sorted_vals) >= 2:
            d1, d2 = sorted_vals[:2]
            regret_ratio = d2 / d1 if d1 > 0 else 1.0
            for j in range(n):
                if i != j and np.abs(distance_matrix[i, j] - d1) < 1e-10:
                    # This is the best edge from i, apply regret penalty
                    regret_penalty[i, j] = min(1.0, 1.0 / regret_ratio)
    
    heuristics = inv_dist * regret_penalty
    
    # Factor 5: Sparsification - keep only top promising edges
    sparsified = np.zeros_like(heuristics)
    k_sparse = max(5, n // 5)  # Keep top edges per node
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf  # Ignore self
        
        # Find threshold for top-k edges
        if n > k_sparse:
            kth_val = np.partition(row, -k_sparse)[-k_sparse]
            mask = row >= kth_val
        else:
            mask = np.ones(n, dtype=bool)
            mask[i] = False
        
        sparsified[i, mask] = heuristics[i, mask]
    
    # Symmetrize the sparsification (if edge is promising in one direction, keep both)
    symmetric_mask = (sparsified > 0) | (sparsified.T > 0)
    final_heuristics = np.zeros_like(heuristics)
    final_heuristics[symmetric_mask] = heuristics[symmetric_mask]
    
    # Normalize for numerical stability
    max_val = np.max(final_heuristics)
    if max_val > 0:
        final_heuristics /= max_val
    
    return final_heuristics

def _compute_mst_edges(distance_matrix: np.ndarray) -> list:
    """Compute edges of Minimum Spanning Tree using Prim's algorithm."""
    n = len(distance_matrix)
    visited = [False] * n
    visited[0] = True
    edges = []
    
    for _ in range(n - 1):
        min_dist = np.inf
        min_edge = None
        
        for i in range(n):
            if visited[i]:
                for j in range(n):
                    if not visited[j] and i != j:
                        if distance_matrix[i, j] < min_dist:
                            min_dist = distance_matrix[i, j]
                            min_edge = (i, j)
        
        if min_edge:
            edges.append(min_edge)
            visited[min_edge[1]] = True
    
    return edges
