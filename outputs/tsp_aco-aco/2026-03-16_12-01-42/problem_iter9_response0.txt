import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge prioritization with adaptive harmonic normalization,
    dynamic regret adaptation, and multiplicative fusion with sparsification.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base inverse distance with stability epsilon ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Normalized Spatial Harmonic Scoring ===
    # Compute harmonic mean of k nearest distances per node
    k_harmonic = max(3, min(15, int(np.sqrt(n))))
    node_harmonic_means = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if len(dists) >= k_harmonic:
            nearest_idx = np.argpartition(dists, k_harmonic)[:k_harmonic]
            nearest_dists = dists[nearest_idx]
            
            # Harmonic mean: k / Σ(1/d)
            if np.any(nearest_dists > 0):
                node_harmonic_means[i] = k_harmonic / np.sum(1 / (nearest_dists + epsilon))
    
    # Combine edge scores harmonically and normalize to [0.5, 1.5]
    harmonic_scores = np.ones_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j and node_harmonic_means[i] > 0 and node_harmonic_means[j] > 0:
                # Harmonic combination of node measures
                combined_harmonic = 2 * node_harmonic_means[i] * node_harmonic_means[j] / \
                                   (node_harmonic_means[i] + node_harmonic_means[j] + epsilon)
                
                # Ratio to actual distance
                ratio = combined_harmonic / (distance_matrix[i, j] + epsilon)
                harmonic_scores[i, j] = 1.0 + 0.5 * np.tanh(np.log(ratio + 1))
    
    # Normalize to [0.5, 1.5] range
    if n > 1:
        off_diag_mask = ~np.eye(n, dtype=bool)
        min_score = np.min(harmonic_scores[off_diag_mask])
        max_score = np.max(harmonic_scores[off_diag_mask])
        if max_score > min_score:
            harmonic_scores[off_diag_mask] = 0.5 + 1.0 * (harmonic_scores[off_diag_mask] - min_score) / \
                                             (max_score - min_score)
    
    np.fill_diagonal(harmonic_scores, 0)
    
    # === 3. Dynamic Regret with Order Adaptation ===
    regret_matrix = np.ones_like(distance_matrix)
    
    if n > 2:
        # Extract off-diagonal distances for statistics
        off_diag_mask = ~np.eye(n, dtype=bool)
        global_distances = distance_matrix[off_diag_mask]
        global_mean = np.mean(global_distances)
        global_std = np.std(global_distances)
        cv = global_std / (global_mean + epsilon)
        
        # Adaptive regret order based on coefficient of variation
        if cv < 0.2:
            regret_order = 2  # Uniform graphs
        elif cv < 0.5:
            regret_order = 3  # Moderate variation
        else:
            regret_order = 4  # High variation
        
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            if len(dists) >= regret_order:
                # Get regret_order-th smallest distance as reference
                ref_idx = np.argpartition(dists, regret_order-1)[regret_order-1]
                ref_dist = dists[ref_idx]
                
                if ref_dist > 0:
                    # Vectorized regret computation
                    for j in range(n):
                        if i != j:
                            d_ij = distance_matrix[i, j]
                            if d_ij < ref_dist:
                                # Penalty based on relative improvement
                                regret_penalty = (ref_dist - d_ij) / (ref_dist + epsilon)
                                # Spread regularization using CV
                                spread_factor = 1.0 / (1.0 + cv * regret_penalty)
                                regret_matrix[i, j] = 1.0 + spread_factor * regret_penalty
    
    # === 4. Lightweight Efficiency Scoring ===
    efficiency_scores = np.ones_like(distance_matrix)
    
    if n > 3:
        # Simple two-step alternative path check
        for i in range(n):
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            
            # Get 3 nearest neighbors for efficiency check
            k_eff = min(3, n-1)
            if len(dists_i) >= k_eff:
                neighbors_i = np.argpartition(dists_i, k_eff)[:k_eff]
                
                for j in range(n):
                    if i != j:
                        d_ij = distance_matrix[i, j]
                        
                        # Check if there's a significantly better two-step path
                        best_alt = np.inf
                        for k in neighbors_i:
                            if k != j:
                                alt_path = distance_matrix[i, k] + distance_matrix[k, j]
                                if alt_path < best_alt:
                                    best_alt = alt_path
                        
                        if best_alt < np.inf and d_ij > 0:
                            efficiency_ratio = best_alt / (d_ij + epsilon)
                            # Boost edges with no efficient alternatives
                            if efficiency_ratio > 1.1:  # Alternative is at least 10% worse
                                efficiency_scores[i, j] = 1.0 + 0.3 * np.tanh(efficiency_ratio - 1.0)
    
    # === 5. Multiplicative Fusion with Adaptive Weights ===
    # Determine graph characteristics for weight adaptation
    if n > 1:
        off_diag_mask = ~np.eye(n, dtype=bool)
        global_distances = distance_matrix[off_diag_mask]
        mean_dist = np.mean(global_distances)
        std_dist = np.std(global_distances)
        cv = std_dist / (mean_dist + epsilon)
    else:
        cv = 1.0
    
    # Adaptive weights based on graph variation
    if cv < 0.2:
        # Uniform distances: emphasize spatial harmony
        w_inv = 0.8
        w_harmonic = 1.2
        w_regret = 0.9
        w_eff = 0.7
    elif cv < 0.5:
        # Moderate variation: balanced approach
        w_inv = 1.0
        w_harmonic = 1.0
        w_regret = 1.1
        w_eff = 0.9
    else:
        # High variation: emphasize regret and efficiency
        w_inv = 1.1
        w_harmonic = 0.9
        w_regret = 1.2
        w_eff = 1.0
    
    # Multiplicative fusion
    heuristics = (inv_dist ** w_inv) * \
                 (harmonic_scores ** w_harmonic) * \
                 (regret_matrix ** w_regret) * \
                 (efficiency_scores ** w_eff)
    
    # === 6. MST Reinforcement (lightweight) ===
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            mst_present = (mst_dense > 0) | (mst_dense.T > 0)
            heuristics[mst_present] *= 1.3  # Moderate boost for MST edges
        except:
            pass
    
    # === 7. Efficiency-Aware Sparsification ===
    for i in range(n):
        row = heuristics[i].copy()
        
        # Determine adaptive keep threshold
        if n > 3:
            # Base threshold: keep edges above mean
            mean_val = np.mean(row[~np.isclose(row, 0)])
            threshold = mean_val
            
            # Adjust based on node's efficiency scores
            eff_mean = np.mean(efficiency_scores[i, ~np.isclose(efficiency_scores[i], 1)])
            if eff_mean > 1.1:  # Node has many efficient direct connections
                threshold *= 1.2  # Be more selective
            elif eff_mean < 0.9:  # Node lacks efficient connections
                threshold *= 0.8  # Be less selective
            
            # Apply threshold
            mask = row >= threshold
            
            # Ensure minimum connectivity (at least 2 edges)
            if np.sum(mask) < 2:
                top2_idx = np.argsort(row)[-2:]
                mask[top2_idx] = True
            
            row[~mask] = 0
            heuristics[i] = row
    
    # === 8. Symmetrization and Final Normalization ===
    # Symmetrize using geometric mean for multiplicative consistency
    heuristics_sym = np.sqrt(heuristics * heuristics.T)
    
    # Final cleanup
    np.fill_diagonal(heuristics_sym, 0)
    heuristics_sym = np.maximum(heuristics_sym, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics_sym)
    if max_val > 0:
        heuristics_sym = heuristics_sym / max_val
    
    return heuristics_sym
