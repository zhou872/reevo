import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge prioritization with centrality-guided adaptive lookahead,
    integrated node importance scoring, and optimized computational efficiency.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base inverse distance ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Centrality-based adaptive parameters ===
    if n > 1:
        # Compute node centrality using harmonic centrality
        centrality = np.zeros(n)
        for i in range(n):
            mask = np.arange(n) != i
            if np.any(mask):
                # Harmonic centrality: sum of reciprocal distances
                centrality[i] = np.sum(1 / (distance_matrix[i, mask] + epsilon))
        
        # Normalize centrality to [0.5, 1.5] range
        if np.max(centrality) > epsilon:
            centrality_norm = 0.5 + (centrality - np.min(centrality)) / (np.max(centrality) - np.min(centrality) + epsilon)
        else:
            centrality_norm = np.ones(n)
    else:
        centrality_norm = np.ones(n)
    
    # === 3. Adaptive lookahead regret with centrality weighting ===
    lookahead_scores = np.ones_like(distance_matrix)
    
    if n > 2:
        # Precompute nearest neighbors for each node
        k_neighbors = max(3, min(8, int(np.sqrt(n))))
        neighbor_indices = np.zeros((n, k_neighbors), dtype=int)
        
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            neighbor_indices[i] = np.argpartition(dists, k_neighbors)[:k_neighbors]
        
        # Compute lookahead regret for each edge
        for i in range(n):
            i_centrality = centrality_norm[i]
            
            for j in range(i + 1, n):
                d_ij = distance_matrix[i, j]
                if d_ij <= epsilon:
                    continue
                
                j_centrality = centrality_norm[j]
                
                # Adaptive lookahead depth based on node centrality
                lookahead_depth = max(2, min(4, int(2 + i_centrality + j_centrality)))
                
                # Get candidate intermediate nodes
                candidates = set()
                for k in neighbor_indices[i][:lookahead_depth]:
                    if k != j:
                        candidates.add(k)
                for k in neighbor_indices[j][:lookahead_depth]:
                    if k != i:
                        candidates.add(k)
                
                # Find best alternative path i->k->j
                best_alternative = np.inf
                for k in candidates:
                    if k != i and k != j:
                        alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                        best_alternative = min(best_alternative, alt_cost)
                
                if best_alternative < np.inf:
                    # Compute regret and centrality-weighted score
                    regret = (best_alternative - d_ij) / (d_ij + epsilon)
                    centrality_factor = 0.5 * (i_centrality + j_centrality)
                    
                    # Transform to heuristic score [0.7, 1.3]
                    lookahead_score = 1.0 + 0.3 * np.tanh(centrality_factor * regret)
                    lookahead_scores[i, j] = lookahead_scores[j, i] = lookahead_score
    
    # === 4. Compact density-aware scoring ===
    density_scores = np.ones_like(distance_matrix)
    
    if n > 2:
        # Compute local density for each node
        local_density = np.zeros(n)
        for i in range(n):
            mask = np.arange(n) != i
            if np.any(mask):
                # Use median distance for robustness
                local_density[i] = np.median(distance_matrix[i, mask])
        
        # Compute density-adjusted scores
        for i in range(n):
            for j in range(i + 1, n):
                # Edge density score based on both nodes' local densities
                density_score = 1.0 + 0.2 * np.exp(-0.5 * (local_density[i] + local_density[j]) / 
                                                  (np.mean(local_density) + epsilon))
                density_scores[i, j] = density_scores[j, i] = density_score
    
    # === 5. Integrated multiplicative fusion ===
    heuristics = inv_dist * lookahead_scores * density_scores
    np.fill_diagonal(heuristics, 0)
    
    # === 6. MST reinforcement with centrality weighting ===
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            mst_present = (mst_dense > 0) | (mst_dense.T > 0)
            
            # Boost MST edges based on node centrality
            boost_matrix = np.ones_like(distance_matrix)
            for i in range(n):
                for j in range(i + 1, n):
                    if mst_present[i, j]:
                        boost = 1.1 + 0.2 * (centrality_norm[i] * centrality_norm[j])
                        boost_matrix[i, j] = boost_matrix[j, i] = boost
            
            heuristics *= boost_matrix
        except:
            pass
    
    # === 7. Centrality-guided sparsification ===
    for i in range(n):
        row = heuristics[i].copy()
        
        # Adaptive sparsity based on centrality and problem size
        if centrality_norm[i] > 1.0:
            # High-centrality nodes keep more connections
            keep_top = max(5, min(15, n // 3))
        else:
            keep_top = max(3, min(10, n // 4))
        
        keep_top = min(keep_top, n - 1)
        if keep_top > 0:
            # Keep top edges based on heuristic values
            threshold = np.partition(row, -keep_top)[-keep_top]
            keep_mask = row >= threshold
            
            # Also preserve MST edges if they exist
            if n > 2:
                try:
                    mst_i = (minimum_spanning_tree(distance_matrix).toarray()[i] > 0)
                    keep_mask = keep_mask | mst_i
                except:
                    pass
            
            # Apply soft thresholding
            mask_strength = (row - threshold) / (np.max(row) - threshold + epsilon)
            mask_strength = np.clip(mask_strength, 0, 1)
            row = row * mask_strength
            
            # Zero out completely unimportant edges
            row[~keep_mask] = 0
            heuristics[i] = row
    
    # === 8. Optimized symmetrization and normalization ===
    # Use weighted symmetrization (favor higher confidence direction)
    heuristics_sym = np.where(heuristics >= heuristics.T, heuristics, heuristics.T)
    np.fill_diagonal(heuristics_sym, 0)
    
    # Final normalization with soft compression
    max_val = np.max(heuristics_sym)
    if max_val > epsilon:
        heuristics_sym = heuristics_sym / max_val
        # Gentle power compression to enhance contrast
        heuristics_sym = np.power(heuristics_sym, 0.85)
    
    return heuristics_sym
