import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (like v1) with epsilon for stability
    epsilon = 1e-10
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)  # No self-edges
    
    # 2. Connectivity factor: how many nearest neighbors share this edge
    # For each node, identify its k nearest neighbors
    k = max(2, int(np.sqrt(n)))  # Adaptive k based on problem size
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k+1 smallest distances (k+1 because self is included)
        nearest = np.argpartition(distance_matrix[i], k+1)[:k+1]
        nearest = nearest[nearest != i]  # Remove self
        if len(nearest) > k:
            nearest = nearest[:k]
        nn_mask[i, nearest] = True
    
    # Symmetrize the mask (edge is promising if either endpoint considers it a nearest neighbor)
    sym_mask = nn_mask | nn_mask.T
    
    # 3. Degree constraint factor: edges that help maintain reasonable vertex degrees
    # Compute initial degree contributions from nearest neighbors
    degree_score = np.zeros_like(distance_matrix)
    for i in range(n):
        # Count how many nodes have i as one of their nearest neighbors
        degree_score[i] = np.sum(nn_mask[:, i])
    
    # Normalize degree scores
    if degree_score.max() > 0:
        degree_score = degree_score / degree_score.max()
    
    # 4. Minimum Spanning Tree (MST) inspired factor
    # Use Prim's algorithm to identify MST edges efficiently
    mst_mask = np.zeros_like(distance_matrix, dtype=bool)
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    
    for _ in range(n - 1):
        # Find minimum edge from visited to unvisited
        min_val = np.inf
        min_i = min_j = -1
        
        # Vectorized search for performance
        visited_indices = np.where(visited)[0]
        unvisited_indices = np.where(~visited)[0]
        
        if len(visited_indices) > 0 and len(unvisited_indices) > 0:
            # Create meshgrid for efficient distance lookup
            vi, ui = np.meshgrid(visited_indices, unvisited_indices, indexing='ij')
            current_dists = distance_matrix[vi, ui]
            
            if current_dists.size > 0:
                min_flat_idx = np.argmin(current_dists)
                min_i = visited_indices[min_flat_idx // len(unvisited_indices)]
                min_j = unvisited_indices[min_flat_idx % len(unvisited_indices)]
                
                mst_mask[min_i, min_j] = True
                mst_mask[min_j, min_i] = True
                visited[min_j] = True
    
    # 5. Combine factors with weights
    heuristics = np.zeros_like(distance_matrix)
    
    # Base score from inverse distance
    heuristics = inv_dist.copy()
    
    # Boost edges that are nearest neighbors (symmetrically)
    heuristics[sym_mask] *= 1.5
    
    # Boost edges that appear in MST
    heuristics[mst_mask] *= 2.0
    
    # Incorporate degree balancing (prefer edges to nodes with lower current degree)
    heuristics *= (1 + 0.3 * (1 - degree_score))
    
    # 6. Strategic sparsification: zero out clearly unpromising edges
    # Keep only edges that are in the top percentiles for each node
    for i in range(n):
        row = heuristics[i].copy()
        # Keep top (k+2) edges per node, or top 20%, whichever is larger
        keep_count = max(k + 2, int(0.2 * n))
        if keep_count < n - 1:
            threshold = np.partition(row, -keep_count)[-keep_count]
            row[row < threshold] = 0
            heuristics[i] = row
    
    # Ensure symmetry of sparsification (if either endpoint keeps it, keep it)
    heuristics = np.maximum(heuristics, heuristics.T * 0.5)
    
    # Final cleanup: zero diagonal
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
