import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge prioritization using regret-based lookahead, 
    adaptive multiplicative fusion, and size-aware component weighting.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base inverse distance ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Regret-based multi-step lookahead ===
    regret_scores = np.ones_like(distance_matrix)
    
    if n > 2:
        # Compute regret for each edge using 2-step lookahead
        for i in range(n):
            # Vectorized computation for node i
            d_i = distance_matrix[i, :].copy()
            d_i[i] = np.inf
            
            # Create matrix of alternative path costs i->k->j
            # S[k,j] = d_ik + d_kj for all k, j
            S = d_i.reshape(n, 1) + distance_matrix  # Broadcasted addition
            
            # Exclude cases where k = i or k = j
            S[i, :] = np.inf
            np.fill_diagonal(S, np.inf)
            
            # Find best alternative for each j
            best_alt = np.min(S, axis=0)
            
            # Compute regret: (best_alternative - direct) / direct
            with np.errstate(divide='ignore', invalid='ignore'):
                regret = (best_alt - d_i) / (d_i + epsilon)
                regret = np.nan_to_num(regret, nan=0.0, posinf=0.0, neginf=0.0)
            
            # Transform regret to score using sigmoid
            regret_score = 1.0 + 1.0 / (1.0 + np.exp(-regret))  # Range: [1, 2]
            regret_scores[i, :] = regret_score
            regret_scores[i, i] = 1.0
    
    # === 3. Density-aware spatial scoring ===
    spatial_scores = np.ones_like(distance_matrix)
    
    if n > 2:
        # Compute local density using median distances
        local_density = np.zeros(n)
        for i in range(n):
            mask = np.arange(n) != i
            if np.any(mask):
                local_density[i] = np.median(distance_matrix[i, mask])
        
        global_density = np.median(local_density) + epsilon
        
        # Compute spatial scores for each edge
        for i in range(n):
            for j in range(i + 1, n):
                density_factor = (local_density[i] + local_density[j]) / (2 * global_density)
                spatial_score = 1.0 + 0.3 * np.exp(-density_factor)
                spatial_scores[i, j] = spatial_scores[j, i] = spatial_score
    
    # === 4. Adaptive weight calculation ===
    # Compute coefficient of variation of distances
    flat_dist = distance_matrix[np.triu_indices(n, k=1)]
    if len(flat_dist) > 0 and np.mean(flat_dist) > epsilon:
        cv = np.std(flat_dist) / np.mean(flat_dist)
    else:
        cv = 0.5  # Default value
    
    # Size-dependent base weights
    size_factor = n / 100.0
    
    # Calculate component weights based on CV and size
    w_inv = 1.2 + 0.4 * np.tanh(cv) - 0.3 * size_factor
    w_regret = 0.8 - 0.2 * np.tanh(cv) + 0.4 * size_factor
    w_spatial = 0.6 + 0.3 * np.tanh(cv) + 0.1 * size_factor
    
    # Clip weights to reasonable ranges
    w_inv = np.clip(w_inv, 0.5, 2.0)
    w_regret = np.clip(w_regret, 0.3, 1.8)
    w_spatial = np.clip(w_spatial, 0.3, 1.5)
    
    # === 5. Multiplicative fusion with exponent weights ===
    # Normalize components to [0, 1] range
    inv_dist_norm = inv_dist / (np.max(inv_dist) + epsilon)
    regret_norm = regret_scores / (np.max(regret_scores) + epsilon)
    spatial_norm = spatial_scores / (np.max(spatial_scores) + epsilon)
    
    # Apply exponent weights and combine
    heuristics = (
        np.power(inv_dist_norm, w_inv) *
        np.power(regret_norm, w_regret) *
        np.power(spatial_norm, w_spatial)
    )
    np.fill_diagonal(heuristics, 0)
    
    # === 6. Size-aware sparsification ===
    if n > 10:
        for i in range(n):
            row = heuristics[i].copy()
            
            # Determine number of edges to keep based on problem size
            if n < 30:
                keep_top = min(8, n - 1)
            elif n < 100:
                keep_top = min(15, n // 4)
            else:
                keep_top = min(25, n // 5)
            
            # Keep top-k edges
            if keep_top > 0:
                threshold = np.partition(row, -keep_top)[-keep_top]
                mask = row >= threshold
                
                # Preserve MST edges if available
                if n > 2:
                    try:
                        mst = minimum_spanning_tree(distance_matrix).toarray()
                        mst_mask = (mst[i, :] > 0) | (mst[:, i] > 0)
                        mask = mask | mst_mask
                    except:
                        pass
                
                # Apply soft thresholding
                row[~mask] *= 0.1
                heuristics[i] = row
    
    # === 7. Symmetrization and final normalization ===
    # Symmetrize by taking maximum of both directions
    heuristics = np.maximum(heuristics, heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(heuristics)
    if max_val > epsilon:
        heuristics = heuristics / max_val
    
    # Apply mild compression to enhance contrast
    heuristics = np.power(heuristics, 0.9)
    
    return heuristics
