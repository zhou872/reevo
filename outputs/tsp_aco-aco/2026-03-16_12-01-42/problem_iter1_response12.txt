```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection.
    Returns a sparse matrix indicating edge promise, with same shape as input.
    """
    n = distance_matrix.shape[0]
    
    # Base inverse distance (with epsilon to avoid division by zero)
    epsilon = 1e-10
    base_heuristic = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(base_heuristic, 0)  # Ignore self-loops
    
    # 1. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst > 0).toarray()
    mst_mask = np.maximum(mst_mask, mst_mask.T)  # Make symmetric
    
    # 2. Mutual nearest neighbors bonus
    k = max(3, int(np.sqrt(n)))  # Adaptive neighborhood size
    nn_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf  # Exclude self
        neighbors = np.argpartition(row, k)[:k]
        nn_mask[i, neighbors] = True
    
    mutual_nn_mask = nn_mask & nn_mask.T  # Both directions
    
    # 3. Combine bonuses
    bonus_matrix = np.ones((n, n))
    bonus_matrix[mst_mask] *= 2.0          # MST edges get 2x bonus
    bonus_matrix[mutual_nn_mask] *= 1.5    # Mutual NN get 1.5x bonus
    
    # Combined heuristic
    heuristic = base_heuristic * bonus_matrix
    
    # 4. Sparsification: keep top percentile
    # Use upper triangle to avoid double counting
    upper_tri_indices = np.triu_indices(n, k=1)
    upper_values = heuristic[upper_tri_indices]
    
    if len(upper_values) > 0:
        # Keep top 15% of edges (adjustable parameter)
        top_percentile = 15
        threshold = np.percentile(upper_values, 100 - top_percentile)
        
        # Create sparse mask
        sparse_mask = np.zeros((n, n), dtype=bool)
        sparse_mask[upper_tri_indices] = upper_values >= threshold
        
        # Make symmetric
        sparse_mask = np.maximum(sparse_mask, sparse_mask.T)
        heuristic[~sparse_mask] = 0
    
    # Ensure symmetry and zero diagonal
    heuristic = (heuristic + heuristic.T) / 2
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
```
