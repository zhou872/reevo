import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with two-step lookahead regret,
    geometric spatial scoring, and instance-adaptive component weighting.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Instance feature extraction ===
    # Extract features to adapt heuristic weights
    np.fill_diagonal(distance_matrix, np.inf)
    
    # Instance size and variability features
    triu_vals = distance_matrix[np.triu_indices(n, 1)]
    mean_dist = np.mean(triu_vals)
    std_dist = np.std(triu_vals)
    cv_dist = std_dist / (mean_dist + epsilon)
    
    # Nearest neighbor statistics
    nearest_dists = np.partition(distance_matrix, 1, axis=1)[:, 1]
    nn_mean = np.mean(nearest_dists)
    nn_std = np.std(nearest_dists)
    nn_cv = nn_std / (nn_mean + epsilon)
    
    # Spatial dispersion measure
    spatial_dispersion = np.std(distance_matrix, axis=1).mean()
    
    np.fill_diagonal(distance_matrix, 0)
    
    # === 2. Two-step lookahead regret ===
    # Compute regret considering two-step alternative paths
    regret_matrix = np.zeros_like(distance_matrix)
    
    # Precompute nearest neighbors for each node
    k_regret = min(8, n - 1)
    nn_indices = np.zeros((n, k_regret), dtype=int)
    nn_distances = np.zeros((n, k_regret))
    
    for i in range(n):
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        idx = np.argsort(dist_i)[:k_regret]
        nn_indices[i] = idx
        nn_distances[i] = dist_i[idx]
    
    for i in range(n):
        for j in range(i + 1, n):
            d_ij = distance_matrix[i, j]
            
            # Find best two-step alternative path i -> k -> j
            best_two_step = np.inf
            
            # Consider top neighbors of i and j
            for k in nn_indices[i][:min(4, k_regret)]:
                if k == j:
                    continue
                for l in nn_indices[j][:min(4, k_regret)]:
                    if l == i or l == k:
                        continue
                    # Path i -> k -> l -> j (two edges)
                    alt_cost = distance_matrix[i, k] + distance_matrix[k, l] + distance_matrix[l, j]
                    best_two_step = min(best_two_step, alt_cost)
            
            # Compute regret as improvement over best alternative
            if np.isfinite(best_two_step):
                # Normalized regret: how much better is direct edge?
                regret = max(0, best_two_step - d_ij) / (mean_dist + epsilon)
                # Scale based on instance variability
                scale = 1.0 + 0.5 * cv_dist
                regret_matrix[i, j] = regret_matrix[j, i] = 1.0 + scale * np.tanh(regret)
            else:
                regret_matrix[i, j] = regret_matrix[j, i] = 1.0
    
    # === 3. Geometric spatial scoring ===
    # Use geometric mean for more balanced spatial assessment
    spatial_scores = np.ones_like(distance_matrix)
    
    # Compute node spatial profiles using geometric mean of nearest distances
    node_geometric = np.zeros(n)
    k_spatial = min(6, n - 1)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest_dists = np.partition(dists, k_spatial)[:k_spatial]
        # Geometric mean (more robust to outliers than arithmetic mean)
        with np.errstate(divide='ignore', invalid='ignore'):
            log_sum = np.sum(np.log(nearest_dists + epsilon))
            node_geometric[i] = np.exp(log_sum / k_spatial)
    
    # Combine endpoint scores using geometric mean
    for i in range(n):
        for j in range(n):
            if i != j:
                # Geometric mean of endpoint geometric means
                spatial_scores[i, j] = np.sqrt(node_geometric[i] * node_geometric[j] + epsilon)
    
    # Normalize spatial scores to [0.5, 1.5] range
    non_zero = spatial_scores > 0
    if np.any(non_zero):
        min_val = np.min(spatial_scores[non_zero])
        max_val = np.max(spatial_scores[non_zero])
        if max_val > min_val:
            spatial_scores = 0.5 + (spatial_scores - min_val) / (max_val - min_val)
    
    # === 4. Instance-adaptive component weights ===
    # Dynamic weighting based on instance characteristics
    size_factor = np.log10(n) / np.log10(100) if n > 1 else 1.0
    
    # Weight regret more for instances with high spatial dispersion
    w_regret = 0.4 + 0.3 * min(1.0, spatial_dispersion / (mean_dist + epsilon))
    
    # Weight spatial scores more for larger instances
    w_spatial = 0.3 + 0.2 * min(1.0, size_factor)
    
    # Base inverse distance weight (complementary)
    w_inverse = 0.3 - 0.1 * min(1.0, cv_dist)
    
    # Normalize weights to sum to 1
    total_weight = w_regret + w_spatial + w_inverse
    w_regret /= total_weight
    w_spatial /= total_weight
    w_inverse /= total_weight
    
    # === 5. Base heuristic components ===
    # Inverse distance (with stability)
    inv_dist = 1.0 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # Mutual nearest neighbor enhancement
    k_mutual = max(3, int(np.sqrt(n)))
    mutual_boost = np.ones_like(distance_matrix)
    
    # Find mutual nearest neighbors
    for i in range(n):
        # Get k nearest neighbors for node i
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        i_nn = np.argsort(dist_i)[:k_mutual]
        
        for j in i_nn:
            # Check if i is in j's k nearest neighbors
            dist_j = distance_matrix[j].copy()
            dist_j[j] = np.inf
            j_nn = np.argsort(dist_j)[:k_mutual]
            
            if i in j_nn:
                # Strong boost for mutual connections
                mutual_boost[i, j] = mutual_boost[j, i] = 1.8
    
    # MST-based connectivity enhancement
    mst_boost = np.ones_like(distance_matrix)
    if n > 2:
        # Get MST edges
        mst = minimum_spanning_tree(distance_matrix)
        mst_dense = mst.toarray()
        mst_edges = np.abs(mst_dense) > 0
        
        # Boost MST edges based on their regret score
        mst_boost[mst_edges] = 1.0 + 0.5 * regret_matrix[mst_edges]
    
    # === 6. Multiplicative factor fusion ===
    # Combine all components with instance-adaptive weights
    base_heuristic = (w_inverse * inv_dist + 
                     w_regret * regret_matrix * inv_dist + 
                     w_spatial * spatial_scores * inv_dist)
    
    # Apply multiplicative boosts
    heuristics = base_heuristic * mutual_boost * mst_boost
    
    # Degree balancing: penalize nodes with too many strong connections
    if n > 3:
        # Compute node "popularity" based on strong heuristic values
        threshold = np.percentile(heuristics[heuristics > 0], 70) if np.any(heuristics > 0) else 0
        strong_edges = heuristics > threshold
        
        for i in range(n):
            popularity = np.sum(strong_edges[i])
            if popularity > k_mutual * 2:
                # Scale down edges from overly popular nodes
                penalty = np.sqrt(k_mutual * 2 / (popularity + epsilon))
                heuristics[i] *= penalty
    
    # === 7. Adaptive sparsification ===
    # Keep only the most promising edges for each node
    adaptive_k = max(3, min(10, n // 3))
    
    for i in range(n):
        row = heuristics[i].copy()
        
        # Sort and identify top edges
        if adaptive_k < n - 1:
            sorted_idx = np.argsort(row)[-adaptive_k:]
            mask = np.zeros(n, dtype=bool)
            mask[sorted_idx] = True
            
            # Always keep mutual nearest neighbors
            for j in range(n):
                if i != j and mutual_boost[i, j] > 1.5:
                    mask[j] = True
            
            # Apply mask
            row[~mask] = 0
            heuristics[i] = row
    
    # === 8. Symmetrization and final normalization ===
    # Ensure symmetry with preference for mutual high scores
    symmetric = np.maximum(heuristics, heuristics.T * 0.7)
    np.fill_diagonal(symmetric, 0)
    
    # Final instance-adaptive normalization
    max_val = np.max(symmetric)
    if max_val > 0:
        # Soft normalization that preserves relative differences
        symmetric = np.tanh(symmetric / (max_val + epsilon) * 1.5)
    
    return symmetric
