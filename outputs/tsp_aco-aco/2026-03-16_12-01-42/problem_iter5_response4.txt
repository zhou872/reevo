import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with node-specific adaptive sparsification
    and improved regret modeling.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base inverse distance with stability epsilon ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Improved regret-based edge uniqueness ===
    # Calculate per-node regret as (second_min - first_min) / first_min
    regret_per_node = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Get two smallest distances
        min_indices = np.argpartition(dists, 2)[:2]
        min_vals = dists[min_indices]
        if len(min_vals) >= 2:
            min1, min2 = min_vals[0], min_vals[1]
            regret_per_node[i] = (min2 - min1) / (min1 + epsilon)
        else:
            regret_per_node[i] = 0
    
    # Symmetric regret weighting for edges
    regret_weight = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Combined regret from both endpoints
                combined_regret = (regret_per_node[i] + regret_per_node[j]) / 2
                # Transform to boost factor (1 + c*regret)
                regret_weight[i, j] = 1.0 + 1.5 * combined_regret
    
    # === 3. Spatial neighborhood affinity ===
    # For each edge, consider combined neighborhood of both endpoints
    k_spatial = min(5, n - 1)
    spatial_scores = np.zeros_like(distance_matrix)
    
    # Pre-compute nearest neighbor lists for all nodes
    nn_lists = []
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        if k_spatial > 0:
            nearest = np.argpartition(dists, k_spatial)[:k_spatial]
            nn_lists.append(set(nearest))
        else:
            nn_lists.append(set())
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            # Combine neighborhoods of i and j (excluding i and j)
            neighborhood = nn_lists[i].union(nn_lists[j])
            neighborhood.discard(i)
            neighborhood.discard(j)
            
            if len(neighborhood) > 0:
                # Calculate average inverse distance to shared neighborhood
                inv_dists_i = [1/(distance_matrix[i, k] + epsilon) for k in neighborhood]
                inv_dists_j = [1/(distance_matrix[j, k] + epsilon) for k in neighborhood]
                avg_inv_i = np.mean(inv_dists_i) if inv_dists_i else 0
                avg_inv_j = np.mean(inv_dists_j) if inv_dists_j else 0
                
                # Edge score proportional to how well-connected endpoints are to shared neighborhood
                spatial_scores[i, j] = (avg_inv_i + avg_inv_j) / 2
            else:
                spatial_scores[i, j] = 1.0  # Neutral when no shared neighbors
    
    # Normalize spatial scores
    if np.max(spatial_scores) > 0:
        spatial_scores = spatial_scores / np.max(spatial_scores)
    
    # === 4. Adaptive node-specific sparsification ===
    # Determine per-node sparsity based on local regret
    base_edges_to_keep = max(3, int(np.sqrt(n)))
    heuristics = inv_dist.copy()
    
    # First pass: compute initial scores
    heuristics *= regret_weight
    heuristics *= (0.5 + 0.5 * spatial_scores)  # Scale spatial scores to [0.5, 1.5]
    
    # Node-specific adaptive sparsification
    for i in range(n):
        row_scores = heuristics[i].copy()
        
        # Determine adaptive k for this node based on regret
        regret_i = regret_per_node[i]
        # High regret nodes need more edge options
        adaptive_k = int(base_edges_to_keep * (1.0 + 0.5 * regret_i))
        adaptive_k = min(adaptive_k, n - 1)
        
        # Sort edge scores
        sorted_indices = np.argsort(row_scores)[::-1]  # Descending
        
        # Create keep mask
        keep_mask = np.zeros(n, dtype=bool)
        
        # Keep top adaptive_k edges
        keep_indices = sorted_indices[:adaptive_k]
        keep_mask[keep_indices] = True
        
        # Also keep mutual nearest neighbors (important for connectivity)
        if i > 0:  # Not necessary for first node
            # Find nodes that have i in their top neighbors
            for j in range(n):
                if i != j and i in nn_lists[j] and j in nn_lists[i]:
                    keep_mask[j] = True
        
        # Ensure at least 2 edges are kept
        if np.sum(keep_mask) < 2:
            top2 = sorted_indices[:2]
            keep_mask[top2] = True
        
        # Apply sparsification
        row_scores[~keep_mask] = 0
        heuristics[i] = row_scores
    
    # === 5. Symmetric consensus ===
    # Encourage edges that are promising from both directions
    symmetric_heuristics = np.zeros_like(heuristics)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Take geometric mean of both directions
                # This favors edges that are good from both perspectives
                if heuristics[i, j] > 0 and heuristics[j, i] > 0:
                    symmetric_heuristics[i, j] = np.sqrt(heuristics[i, j] * heuristics[j, i])
                else:
                    symmetric_heuristics[i, j] = 0
    
    # Ensure diagonal is zero
    np.fill_diagonal(symmetric_heuristics, 0)
    
    # Final normalization
    max_val = np.max(symmetric_heuristics)
    if max_val > 0:
        symmetric_heuristics = symmetric_heuristics / max_val
    
    return symmetric_heuristics
