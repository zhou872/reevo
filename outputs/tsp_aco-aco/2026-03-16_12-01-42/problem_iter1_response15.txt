import numpy as np
import heapq

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic scoring for TSP edges combining multiple factors.
    
    The function computes a priority score for each edge (i,j) considering:
    1. Connectivity-corrected inverse distance (favors edges that connect to well-connected nodes)
    2. Nearest neighbor relationships (sparsifies to promising candidate lists)
    3. Triangle inequality violation penalty (penalizes edges that create detours)
    4. Edge centrality in local neighborhoods
    
    Returns a sparse matrix where only promising edges have non-zero scores.
    """
    n = distance_matrix.shape[0]
    
    # 1. Basic inverse distance with connectivity correction
    # Avoid division by zero for diagonal
    with np.errstate(divide='ignore'):
        inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # Compute node connectivity scores (average inverse distance to all other nodes)
    connectivity = np.sum(inv_dist, axis=1) / (n - 1)
    
    # Connectivity-corrected inverse distance
    # Edges connecting to well-connected nodes get higher scores
    connectivity_matrix = np.outer(connectivity, connectivity)
    base_score = inv_dist * connectivity_matrix
    
    # 2. Nearest neighbor candidate list generation
    # For each node, keep only k nearest neighbors
    k = max(3, int(np.sqrt(n)))  # Adaptive k based on problem size
    candidate_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k+1 smallest distances (k+1 because we'll exclude self)
        nearest = np.argpartition(distance_matrix[i], k+1)[:k+1]
        nearest = nearest[nearest != i]  # Remove self
        candidate_mask[i, nearest] = True
    
    # Make symmetric (if i is in j's candidate list, j should be in i's list)
    candidate_mask = candidate_mask | candidate_mask.T
    
    # 3. Triangle inequality violation penalty
    # For each triangle (i,j,k), penalize edges that would create a detour
    penalty = np.ones_like(distance_matrix)
    
    # Sample triangles to avoid O(n^3) complexity
    sample_size = min(100, n)  # Adaptive sampling
    if n > 3:
        sample_indices = np.random.choice(n, size=sample_size, replace=False)
        
        for i in sample_indices:
            for j in range(n):
                if i == j: continue
                # Find best alternative path i->k->j
                alt_paths = distance_matrix[i, :] + distance_matrix[:, j]
                alt_paths[i] = np.inf  # Exclude direct path
                alt_paths[j] = np.inf
                best_alt = np.min(alt_paths)
                
                # If direct path is much longer than best alternative, penalize
                if distance_matrix[i, j] > 1.3 * best_alt:  # 30% tolerance
                    penalty[i, j] *= 0.7
    
    # 4. Local edge centrality bonus
    # Reward edges that appear in many local minimum spanning trees
    centrality = np.zeros_like(distance_matrix)
    
    # Build local MSTs for sampled nodes
    for seed in range(min(10, n)):
        # Build MST starting from this seed
        visited = np.zeros(n, dtype=bool)
        visited[seed] = True
        edges = []
        
        # Add edges from seed to all other nodes
        for j in range(n):
            if j != seed:
                heapq.heappush(edges, (distance_matrix[seed, j], seed, j))
        
        # Build MST
        mst_edges = []
        while edges and len(mst_edges) < n - 1:
            dist, u, v = heapq.heappop(edges)
            if not visited[v]:
                visited[v] = True
                mst_edges.append((u, v))
                # Add edges from v to unvisited nodes
                for w in range(n):
                    if not visited[w] and w != v:
                        heapq.heappush(edges, (distance_matrix[v, w], v, w))
        
        # Increment centrality for edges in this MST
        for u, v in mst_edges:
            centrality[u, v] += 1
            centrality[v, u] += 1
    
    # Normalize centrality
    if centrality.max() > 0:
        centrality = centrality / centrality.max()
    
    # 5. Combine all factors
    final_scores = base_score * penalty * (1 + 0.5 * centrality)  # Centrality adds up to 50% bonus
    
    # 6. Apply candidate mask and sparsify
    final_scores = final_scores * candidate_mask
    
    # Set scores below threshold to zero
    # Threshold = median of non-zero scores
    non_zero = final_scores[final_scores > 0]
    if len(non_zero) > 0:
        threshold = np.percentile(non_zero, 50)  # Keep top 50% of non-zero scores
        final_scores[final_scores < threshold] = 0
    
    # Normalize to [0, 1] range for consistency
    if final_scores.max() > 0:
        final_scores = final_scores / final_scores.max()
    
    return final_scores
