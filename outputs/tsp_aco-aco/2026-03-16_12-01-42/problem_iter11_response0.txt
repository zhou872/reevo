import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge prioritization with exponential decay regret combination,
    adaptive multiplicative fusion, and spectral ratio penalty scaling.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base inverse distance ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Compute graph variation statistics ===
    if n > 1:
        # Extract off-diagonal elements for variation calculation
        mask = ~np.eye(n, dtype=bool)
        off_diag_dists = distance_matrix[mask].flatten()
        
        # Coefficient of variation for adaptive weighting
        if len(off_diag_dists) > 0 and np.mean(off_diag_dists) > epsilon:
            coef_var = np.std(off_diag_dists) / np.mean(off_diag_dists)
        else:
            coef_var = 1.0
        
        # Spectral ratio for penalty scaling (approximated via graph Laplacian)
        if n > 2:
            # Compute normalized Laplacian
            degree_matrix = np.diag(np.sum(distance_matrix, axis=1))
            laplacian = degree_matrix - distance_matrix
            
            # Get eigenvalues (use efficient method for large n)
            if n <= 100:
                eigenvalues = np.linalg.eigvalsh(laplacian)
                eigenvalues = eigenvalues[eigenvalues > epsilon]
                if len(eigenvalues) >= 2:
                    spectral_ratio = eigenvalues[1] / eigenvalues[-1] if eigenvalues[-1] > epsilon else 1.0
                else:
                    spectral_ratio = 1.0
            else:
                # Approximation for large n
                spectral_ratio = 0.5 + 0.3 * np.log1p(coef_var)
        else:
            spectral_ratio = 1.0
    else:
        coef_var = 1.0
        spectral_ratio = 1.0
    
    # === 3. Size-adaptive parameter configuration ===
    # Dynamic weights based on problem size and graph variation
    if n < 10:
        base_weights = {'inv': 1.2, 'regret': 0.8, 'spatial': 1.0}
    elif n < 50:
        base_weights = {'inv': 1.0, 'regret': 1.0, 'spatial': 1.0}
    else:
        base_weights = {'inv': 0.8, 'regret': 1.2, 'spatial': 0.9}
    
    # Adjust weights by coefficient of variation
    w_inv = base_weights['inv'] + 0.1 * coef_var
    w_regret = base_weights['regret'] + 0.3 * coef_var
    w_spatial = base_weights['spatial'] - 0.2 * coef_var
    
    # === 4. Enhanced regret-based multi-step lookahead ===
    regret_scores = np.ones_like(distance_matrix)
    
    if n > 2:
        # Precompute nearest neighbors for efficiency
        k_neighbors = max(3, min(10, int(np.sqrt(n))))
        neighbor_indices = np.zeros((n, k_neighbors), dtype=int)
        neighbor_dists = np.zeros((n, k_neighbors))
        
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            idx = np.argpartition(dists, k_neighbors)[:k_neighbors]
            neighbor_indices[i] = idx
            neighbor_dists[i] = dists[idx]
        
        # Compute regret scores with exponential decay combination
        for i in range(n):
            for j in range(i + 1, n):
                d_ij = distance_matrix[i, j]
                if d_ij <= epsilon:
                    continue
                
                # Collect 2-step alternatives
                alternatives = []
                
                # Consider neighbors of i
                for idx_k, k in enumerate(neighbor_indices[i]):
                    if k != j:
                        alt_cost = neighbor_dists[i, idx_k] + distance_matrix[k, j]
                        regret = (alt_cost - d_ij) / (d_ij + epsilon)
                        alternatives.append((regret, alt_cost))
                
                # Consider neighbors of j
                for idx_k, k in enumerate(neighbor_indices[j]):
                    if k != i and k not in neighbor_indices[i][:3]:  # Avoid duplicates
                        alt_cost = distance_matrix[i, k] + neighbor_dists[j, idx_k]
                        regret = (alt_cost - d_ij) / (d_ij + epsilon)
                        alternatives.append((regret, alt_cost))
                
                if alternatives:
                    # Sort by regret (ascending)
                    alternatives.sort(key=lambda x: x[0])
                    
                    # Exponential decay combination
                    decay_rate = 0.7
                    weights = [decay_rate ** rank for rank in range(len(alternatives))]
                    total_weight = sum(weights)
                    
                    # Weighted average regret
                    weighted_regret = sum(r * w for (r, _), w in zip(alternatives, weights)) / total_weight
                    
                    # Scale penalty by spectral ratio
                    penalty_scale = 1.0 + 0.5 * spectral_ratio
                    
                    # Transform to score (higher for larger positive regret)
                    regret_score = np.exp(penalty_scale * weighted_regret / (1.0 + abs(weighted_regret)))
                    regret_scores[i, j] = regret_scores[j, i] = np.clip(regret_score, 0.5, 2.0)
    
    # === 5. Spatial scoring with cluster density awareness ===
    spatial_scores = np.ones_like(distance_matrix)
    
    if n > 2:
        # Compute local cluster density using k-nearest neighbors
        cluster_density = np.zeros(n)
        for i in range(n):
            # Use median distance to k nearest neighbors
            k_local = min(5, n - 1)
            if k_local > 0:
                dists = distance_matrix[i].copy()
                dists[i] = np.inf
                nearest_dists = np.partition(dists, k_local)[:k_local]
                cluster_density[i] = np.median(nearest_dists)
            else:
                cluster_density[i] = 1.0
        
        # Normalize densities
        if np.max(cluster_density) > epsilon:
            density_norm = cluster_density / (np.mean(cluster_density) + epsilon)
        else:
            density_norm = np.ones(n)
        
        # Compute spatial scores
        for i in range(n):
            for j in range(i + 1, n):
                # Combine node densities and distance
                density_factor = 0.5 * (density_norm[i] + density_norm[j])
                dist_factor = distance_matrix[i, j] / (np.mean(distance_matrix) + epsilon)
                
                # Spatial score favors medium-density clusters with moderate distances
                spatial_score = np.exp(-0.5 * (density_factor - 1.0)**2) * np.exp(-0.3 * dist_factor)
                spatial_scores[i, j] = spatial_scores[j, i] = np.clip(spatial_score, 0.7, 1.3)
    
    # === 6. Adaptive multiplicative fusion with exponent weights ===
    # Apply exponent weights to components
    inv_dist_exp = inv_dist ** w_inv
    regret_scores_exp = regret_scores ** w_regret
    spatial_scores_exp = spatial_scores ** w_spatial
    
    # Fused heuristic
    heuristics = inv_dist_exp * regret_scores_exp * spatial_scores_exp
    np.fill_diagonal(heuristics, 0)
    
    # === 7. MST reinforcement ===
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            mst_present = (mst_dense > 0) | (mst_dense.T > 0)
            
            # Boost MST edges with size-dependent factor
            mst_boost = 1.0 + 0.3 * np.log1p(n) / (1.0 + coef_var)
            boost_matrix = np.ones_like(distance_matrix)
            boost_matrix[mst_present] = mst_boost
            
            heuristics *= boost_matrix
        except:
            pass
    
    # === 8. Adaptive sparsification ===
    for i in range(n):
        row = heuristics[i].copy()
        
        # Determine number of edges to keep based on problem size
        if n <= 20:
            keep_top = max(3, min(8, n // 2))
        elif n <= 100:
            keep_top = max(5, min(15, n // 3))
        else:
            keep_top = max(8, min(25, n // 4))
        
        keep_top = min(keep_top, n - 1)
        if keep_top > 0:
            # Find threshold for top edges
            threshold = np.partition(row, -keep_top)[-keep_top]
            
            # Apply sigmoid-based soft thresholding
            scale = np.std(row[row > epsilon]) if np.any(row > epsilon) else 1.0
            sigmoid_mask = 1.0 / (1.0 + np.exp(-10.0 * (row - threshold) / (scale + epsilon)))
            
            row = row * sigmoid_mask
            heuristics[i] = row
    
    # === 9. Symmetrization and final normalization ===
    # Symmetric combination (max of both directions)
    heuristics_sym = np.maximum(heuristics, heuristics.T)
    np.fill_diagonal(heuristics_sym, 0)
    
    # Normalize with power compression
    max_val = np.max(heuristics_sym)
    if max_val > epsilon:
        heuristics_sym = heuristics_sym / max_val
        # Adaptive compression based on coefficient of variation
        compression_exp = 0.8 + 0.2 * np.tanh(coef_var - 1.0)
        heuristics_sym = np.power(heuristics_sym, compression_exp)
    
    return heuristics_sym
