import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge prioritization with harmonic normalization,
    dynamic k-regret scoring, and adaptive multi-component fusion.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base Inverse Distance ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Harmonic Spatial Normalization ===
    # Compute harmonic neighborhood statistics per node
    k_harmonic = max(3, min(15, int(np.sqrt(n))))
    node_harmonic_means = np.zeros(n)
    node_harmonic_stds = np.zeros(n)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if len(dists) >= k_harmonic:
            nearest_idx = np.argpartition(dists, k_harmonic)[:k_harmonic]
            nearest_dists = dists[nearest_idx]
            
            # Harmonic mean and std of k-nearest distances
            if np.any(nearest_dists > 0):
                node_harmonic_means[i] = k_harmonic / np.sum(1 / (nearest_dists + epsilon))
                # Approximate harmonic std using log-normalized distances
                log_dists = np.log(nearest_dists + 1)
                node_harmonic_stds[i] = np.std(log_dists)
    
    # Edge scores based on harmonic compatibility
    harmonic_scores = np.ones_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j and node_harmonic_means[i] > 0 and node_harmonic_means[j] > 0:
                # Combined harmonic reference
                combined_mean = 2 * node_harmonic_means[i] * node_harmonic_means[j] / \
                               (node_harmonic_means[i] + node_harmonic_means[j] + epsilon)
                
                # Compatibility score: how well edge matches combined harmonic reference
                ratio = distance_matrix[i, j] / (combined_mean + epsilon)
                # Penalize both too-short and too-long edges relative to harmonic reference
                deviation = np.abs(np.log(ratio + 1))
                # Adjust penalty based on harmonic variance
                variance_factor = 1 + 0.5 * (node_harmonic_stds[i] + node_harmonic_stds[j])
                harmonic_scores[i, j] = np.exp(-deviation * variance_factor)
    
    # === 3. Dynamic K-Regret Scoring ===
    regret_scores = np.ones_like(distance_matrix)
    
    if n > 2:
        # Determine optimal k for regret calculation
        off_diag_mask = ~np.eye(n, dtype=bool)
        all_dists = distance_matrix[off_diag_mask]
        
        if len(all_dists) > 0:
            mean_dist = np.mean(all_dists)
            std_dist = np.std(all_dists)
            cv = std_dist / (mean_dist + epsilon)
            
            # Adaptive k based on graph variation and size
            base_k = max(2, min(5, int(np.sqrt(n))))
            if cv < 0.3:  # Uniform distances
                k_regret = base_k + 1
            elif cv < 0.7:  # Moderate variation
                k_regret = base_k
            else:  # High variation
                k_regret = max(2, base_k - 1)
        else:
            k_regret = 3
        
        # Compute regret scores
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            if len(dists) >= k_regret:
                # Get k-th smallest distance as reference (k-regret)
                kth_idx = np.argpartition(dists, k_regret)[k_regret]
                kth_dist = dists[kth_idx]
                
                if kth_dist > 0:
                    for j in range(n):
                        if i != j:
                            d_ij = distance_matrix[i, j]
                            # Normalized regret: how much better than k-th alternative
                            regret = max(0, kth_dist - d_ij) / (kth_dist + epsilon)
                            # Non-linear mapping for stronger signal
                            regret_scores[i, j] = 1.0 + 2.0 * np.tanh(regret)
    
    # === 4. Two-Step Lookahead Efficiency ===
    efficiency_scores = np.ones_like(distance_matrix)
    
    if n > 3:
        # Precompute nearest neighbors for efficiency
        k_efficiency = min(5, n - 1)
        nearest_indices = np.argsort(distance_matrix + np.eye(n) * np.inf, axis=1)[:, 1:k_efficiency+1]
        
        for i in range(n):
            for j in range(n):
                if i != j:
                    d_ij = distance_matrix[i, j]
                    
                    # Check best two-step alternative via nearest neighbors
                    best_alternative = np.inf
                    for k in nearest_indices[i]:
                        if k != j:
                            alt_path = distance_matrix[i, k] + distance_matrix[k, j]
                            if alt_path < best_alternative:
                                best_alternative = alt_path
                    
                    if best_alternative < np.inf and d_ij > 0:
                        efficiency_ratio = best_alternative / (d_ij + epsilon)
                        # Boost edges with no efficient two-step alternatives
                        if efficiency_ratio > 1.2:
                            efficiency_scores[i, j] = 1.0 + np.log(efficiency_ratio)
    
    # === 5. Multiplicative Fusion with Adaptive Weights ===
    # Analyze graph characteristics for weight adaptation
    if n > 1:
        off_diag_mask = ~np.eye(n, dtype=bool)
        all_dists = distance_matrix[off_diag_mask]
        mean_dist = np.mean(all_dists)
        std_dist = np.std(all_dists)
        cv = std_dist / (mean_dist + epsilon)
        
        # Weight adaptation based on variation and size
        if n < 20:  # Small instances
            w_inv = 1.2
            w_harmonic = 0.9
            w_regret = 1.1
            w_eff = 0.8
        elif cv < 0.3:  # Uniform distances
            w_inv = 0.8
            w_harmonic = 1.3
            w_regret = 0.9
            w_eff = 0.7
        elif cv < 0.7:  # Moderate variation
            w_inv = 1.0
            w_harmonic = 1.0
            w_regret = 1.2
            w_eff = 0.9
        else:  # High variation / clustered
            w_inv = 1.1
            w_harmonic = 0.8
            w_regret = 1.3
            w_eff = 1.0
    else:
        w_inv = w_harmonic = w_regret = w_eff = 1.0
    
    # Multiplicative fusion with exponentiated weights
    heuristics = (inv_dist ** w_inv) * \
                 (harmonic_scores ** w_harmonic) * \
                 (regret_scores ** w_regret) * \
                 (efficiency_scores ** w_eff)
    
    # === 6. MST Reinforcement ===
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            mst_present = (mst_dense > 0) | (mst_dense.T > 0)
            heuristics[mst_present] *= 1.4  # Strong boost for MST edges
        except:
            pass
    
    # === 7. Efficiency-Aware Sparsification ===
    for i in range(n):
        row = heuristics[i].copy()
        
        if n > 3:
            # Determine adaptive sparsity threshold
            row_nonzero = row[~np.isclose(row, 0)]
            if len(row_nonzero) > 0:
                row_mean = np.mean(row_nonzero)
                row_std = np.std(row_nonzero)
                
                # Threshold based on mean and std
                threshold = row_mean
                if row_std > 0:
                    # Adjust based on local efficiency
                    eff_mean = np.mean(efficiency_scores[i, ~np.isclose(efficiency_scores[i], 1)])
                    if eff_mean > 1.5:  # Many efficient alternatives
                        threshold = row_mean + 0.5 * row_std
                    elif eff_mean > 1.2:
                        threshold = row_mean
                    else:  # Few efficient alternatives
                        threshold = row_mean - 0.5 * row_std
                
                # Apply threshold
                mask = row >= threshold
                
                # Ensure minimum connectivity
                if np.sum(mask) < 2:
                    top2_idx = np.argsort(row)[-2:]
                    mask[top2_idx] = True
                
                # Always keep MST edges
                if n > 2:
                    try:
                        mst_edges = np.where(mst_present[i])[0]
                        mask[mst_edges] = True
                    except:
                        pass
                
                row[~mask] = 0
                heuristics[i] = row
    
    # === 8. Symmetrization and Final Normalization ===
    # Geometric mean for multiplicative consistency
    heuristics_sym = np.sqrt(heuristics * heuristics.T)
    np.fill_diagonal(heuristics_sym, 0)
    
    # Row-wise softmax with adaptive temperature
    for i in range(n):
        row = heuristics_sym[i]
        row_max = np.max(row)
        if row_max > 0:
            # Adaptive temperature based on row sparsity
            nonzero_count = np.sum(row > 0)
            if nonzero_count > 1:
                temperature = max(0.5, 2.0 - np.std(row[row > 0]))
            else:
                temperature = 1.0
            
            exp_row = np.exp(row / temperature - np.max(row / temperature))
            heuristics_sym[i] = exp_row / np.sum(exp_row)
    
    # Ensure non-negativity and scale
    heuristics_sym = np.maximum(heuristics_sym, 0)
    global_max = np.max(heuristics_sym)
    if global_max > 0:
        heuristics_sym = heuristics_sym / global_max
    
    return heuristics_sym
