import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    n = distance_matrix.shape[0]
    # 1. Inverse distance
    inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() != 0) | (mst.toarray().T != 0)
    
    # 3. k-nearest neighbors (k=3) bonus, made symmetric
    k = 3
    nn_mask = np.zeros((n, n), dtype=bool)
    for i in range(n):
        indices = np.argsort(distance_matrix[i])[1:k+1]
        nn_mask[i, indices] = True
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Savings inspired by Clarke-Wright
    dist_with_inf = distance_matrix + np.diag([np.inf] * n)
    min_dist = np.min(dist_with_inf, axis=1)
    savings = min_dist[:, None] + min_dist[None, :] - distance_matrix
    np.fill_diagonal(savings, 0)
    
    # Normalize each component to [0,1] for combination
    inv_dist_nonzero = inv_dist[inv_dist != 0]
    max_inv = inv_dist_nonzero.max() if len(inv_dist_nonzero) > 0 else 1
    norm_inv = inv_dist / max_inv
    
    mst_bonus = mst_mask.astype(float)
    nn_bonus = nn_mask.astype(float)
    
    off_diag_indices = np.triu_indices(n, 1)
    savings_off_diag = savings[off_diag_indices]
    if len(savings_off_diag) > 0:
        min_sav = savings_off_diag.min()
        max_sav = savings_off_diag.max()
        if max_sav > min_sav:
            norm_savings = (savings - min_sav) / (max_sav - min_sav)
        else:
            norm_savings = np.zeros_like(savings)
    else:
        norm_savings = np.zeros_like(savings)
    np.fill_diagonal(norm_savings, 0)
    
    # Combine with weights
    w1, w2, w3 = 1.0, 1.0, 1.0
    combined = norm_inv + w1 * mst_bonus + w2 * nn_bonus + w3 * norm_savings
    np.fill_diagonal(combined, 0)
    
    # Sparsify: set values below the median of off-diagonal entries to zero
    off_diag_combined = combined[~np.eye(n, dtype=bool)]
    if len(off_diag_combined) > 0:
        threshold = np.median(off_diag_combined)
        combined[combined < threshold] = 0
    
    return combined
