import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced edge promise heuristics for TSP with adaptive weights and
    lightweight probabilistic lookahead.
    
    Args:
        distance_matrix: Symmetric n×n matrix of pairwise distances.
    
    Returns:
        Heuristic matrix H where H[i,j] indicates promise of edge (i,j).
    """
    n = distance_matrix.shape[0]
    if n < 3:
        # Trivial cases
        H = np.ones_like(distance_matrix)
        np.fill_diagonal(H, 0)
        return H
    
    # Work with a copy, setting diagonal to infinity to avoid self-edges
    D = distance_matrix.copy()
    np.fill_diagonal(D, np.inf)
    
    # === 1. Base inverse distance (normalized) ===
    with np.errstate(divide='ignore'):
        inv_dist = 1.0 / D
    inv_dist[np.isinf(inv_dist)] = 0
    
    # Normalize to [0,1]
    if inv_dist.max() > 0:
        inv_dist = inv_dist / inv_dist.max()
    
    # === 2. Symmetric regret weighting ===
    # For each node, find two nearest neighbors
    nearest = np.partition(D, 2, axis=1)[:, :2]  # (n,2)
    nearest_dists = nearest[:, 0]
    second_dists = nearest[:, 1]
    regret_i = (second_dists - nearest_dists) / (nearest_dists + 1e-12)
    regret_matrix = regret_i[:, None] + regret_i[None, :]  # Symmetric regret
    
    # === 3. Lightweight two-neighbor lookahead ===
    # For each node, get indices of two nearest neighbors
    nearest_idx = np.argpartition(D, 2, axis=1)[:, :2]  # (n,2)
    
    # Precompute distances to nearest neighbors
    dist_to_nn = np.zeros((n, 2))
    for i in range(n):
        dist_to_nn[i, 0] = D[i, nearest_idx[i, 0]]
        dist_to_nn[i, 1] = D[i, nearest_idx[i, 1]]
    
    # Compute lookahead score for each edge (i,j)
    lookahead = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            # Consider two-step paths through i's nearest neighbors (excluding j)
            candidates = []
            for k_idx in range(2):
                k = nearest_idx[i, k_idx]
                if k != j:
                    candidates.append(dist_to_nn[i, k_idx] + D[k, j])
            
            # Consider two-step paths through j's nearest neighbors (excluding i)
            for k_idx in range(2):
                k = nearest_idx[j, k_idx]
                if k != i:
                    candidates.append(dist_to_nn[j, k_idx] + D[i, k])
            
            if candidates:
                best_two_step = min(candidates)
                penalty = best_two_step - D[i, j]
                # Convert penalty to score (small penalty → high score)
                scale = np.mean(nearest_dists[[i, j]]) + 1e-12
                lookahead[i, j] = np.exp(-max(0, penalty) / scale)
    
    # === 4. Adaptive weight learning based on instance characteristics ===
    # Compute instance statistics
    flat_dists = D[D < np.inf]
    dist_cv = np.std(flat_dists) / (np.mean(flat_dists) + 1e-12)  # Coefficient of variation
    dist_skew = np.mean((flat_dists - np.mean(flat_dists))**3) / (np.std(flat_dists)**3 + 1e-12)
    
    # Adapt weights based on statistics
    base_weights = {
        'inv_dist': 0.4,
        'regret': 0.3,
        'lookahead': 0.3
    }
    
    # Adjust for uniformity (high CV means more varied distances)
    cv_factor = np.tanh(dist_cv)  # ∈ [0,1)
    # When distances are more varied, trust inverse distance more
    base_weights['inv_dist'] *= (1.0 + 0.5 * cv_factor)
    base_weights['regret'] *= (1.0 - 0.3 * cv_factor)
    base_weights['lookahead'] *= (1.0 - 0.2 * cv_factor)
    
    # Adjust for skewness (asymmetric distance distribution)
    skew_factor = np.tanh(abs(dist_skew))
    # Skewed distributions benefit more from lookahead
    base_weights['lookahead'] *= (1.0 + 0.4 * skew_factor)
    base_weights['inv_dist'] *= (1.0 - 0.2 * skew_factor)
    
    # Normalize weights
    total = sum(base_weights.values())
    weights = {k: v/total for k, v in base_weights.items()}
    
    # === 5. Combine components ===
    # Normalize components to comparable scales
    if regret_matrix.max() > 0:
        regret_norm = regret_matrix / regret_matrix.max()
    else:
        regret_norm = regret_matrix
    
    if lookahead.max() > 0:
        lookahead_norm = lookahead / lookahead.max()
    else:
        lookahead_norm = lookahead
    
    H = (weights['inv_dist'] * inv_dist +
         weights['regret'] * regret_norm +
         weights['lookahead'] * lookahead_norm)
    
    # === 6. Isolation-aware sparsification ===
    # Compute node isolation (average distance to others)
    node_isolation = np.mean(D, axis=1)
    isolation_factor = node_isolation / (node_isolation.max() + 1e-12)
    
    # Dynamic thresholds based on isolation
    sparsified = np.zeros_like(H)
    for i in range(n):
        # More isolated nodes need more connections
        k = max(2, min(5, int(3 + 2 * isolation_factor[i])))
        
        # Get top-k edges from this node
        row = H[i].copy()
        row[i] = -np.inf  # Exclude self
        top_k_idx = np.argpartition(row, -k)[-k:]
        sparsified[i, top_k_idx] = row[top_k_idx]
    
    # Symmetrize (undirected TSP)
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Keep only edges that are top-k for both endpoints (mutual promising)
    mutual_promising = (sparsified > 0)
    H = H * mutual_promising
    
    # Final threshold: keep edges above adaptive percentile
    non_zero = H[H > 0]
    if len(non_zero) > 0:
        # Higher threshold for larger instances
        percentile = 40 + 10 * np.tanh(n / 50)  # Ranges 40-50 for n=∞
        threshold = np.percentile(non_zero, percentile)
        H[H < threshold] = 0
    
    # Ensure each node has at least 2 connections
    for i in range(n):
        if np.sum(H[i] > 0) < 2:
            # Add back best missing connections
            row = H[i].copy()
            row[i] = -np.inf
            top_idx = np.argsort(row)[-2:]
            for idx in top_idx:
                if H[i, idx] == 0:
                    H[i, idx] = row[idx]
                    H[idx, i] = row[idx]  # Maintain symmetry
    
    np.fill_diagonal(H, 0)
    return H
