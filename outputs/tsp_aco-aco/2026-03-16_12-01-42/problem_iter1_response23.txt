import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Inverse distance weighting (baseline)
    2. Local neighborhood density bonus (edges connecting to popular nodes)
    3. Triangle inequality violation penalty
    4. Edge sparsification by keeping only top promising edges
    """
    n = distance_matrix.shape[0]
    
    # Avoid division by zero for diagonal/zero distances
    safe_dist = distance_matrix.copy()
    safe_dist[safe_dist == 0] = np.inf
    
    # 1. Base inverse distance score
    inv_dist = 1 / safe_dist
    
    # 2. Neighborhood density bonus: nodes with many short connections are hubs
    # Calculate how "attractive" each node is based on its connections
    proximity_threshold = np.percentile(safe_dist[safe_dist < np.inf], 25)
    is_close = (safe_dist < proximity_threshold).astype(float)
    node_popularity = is_close.sum(axis=1)
    
    # Bonus factor: edges connecting to popular nodes get higher scores
    popularity_bonus = np.outer(node_popularity, node_popularity) / (n * n)
    
    # 3. Triangle inequality penalty
    # Penalize edges that violate triangle inequality significantly
    penalty_matrix = np.ones((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if safe_dist[i, j] < np.inf:
                # Find the best alternative path through a third node
                min_detour = np.min(safe_dist[i, :] + safe_dist[:, j])
                if min_detour < np.inf:
                    violation_ratio = safe_dist[i, j] / (min_detour + 1e-10)
                    # Strongly penalize edges much longer than best detour
                    if violation_ratio > 1.5:
                        penalty_matrix[i, j] = penalty_matrix[j, i] = 0.3
                    elif violation_ratio > 1.2:
                        penalty_matrix[i, j] = penalty_matrix[j, i] = 0.7
    
    # 4. Combine factors
    heuristics_matrix = inv_dist * (1 + 0.5 * popularity_bonus) * penalty_matrix
    
    # 5. Edge sparsification: keep only top promising edges
    # Calculate edge ranks within each node's connections
    rank_matrix = np.zeros((n, n))
    for i in range(n):
        valid_indices = np.where(safe_dist[i, :] < np.inf)[0]
        if len(valid_indices) > 0:
            # Rank edges by heuristic score (descending)
            scores = heuristics_matrix[i, valid_indices]
            ranks = np.argsort(np.argsort(-scores))  # 0 is best rank
            # Convert to selection probability
            # Keep approximately top 40% of edges for each node
            keep_threshold = max(3, len(valid_indices) // 2.5)
            selection_prob = np.where(ranks < keep_threshold, 1.0, 0.0)
            rank_matrix[i, valid_indices] = selection_prob
    
    # Make symmetric (keep edge if either endpoint selects it)
    symmetric_mask = np.maximum(rank_matrix, rank_matrix.T)
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * symmetric_mask
    
    # Normalize to similar scale as original distances
    max_val = np.max(heuristics_matrix[heuristics_matrix > 0])
    if max_val > 0:
        heuristics_matrix = heuristics_matrix / max_val
    
    return heuristics_matrix
