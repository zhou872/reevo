import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with learned meta-weighting, refined path regret,
    and adaptive spectral analysis for diverse TSP instances.
    
    Features:
    1. Adaptive meta-weighting based on problem characteristics
    2. Three-level path regret analysis with lookahead optimization
    3. Spectral clustering with dynamic thresholding
    4. Entropy-driven sparsification with connectivity preservation
    
    Args:
        distance_matrix: Square symmetric matrix of pairwise distances
        
    Returns:
        Heuristic matrix with higher values indicating more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result
    
    # === 1. PROBLEM CHARACTERIZATION FOR ADAPTIVE PARAMETERS ===
    # Compute comprehensive instance features
    triu_indices = np.triu_indices(n, k=1)
    distances_upper = distance_matrix[triu_indices]
    
    avg_dist = np.mean(distances_upper)
    std_dist = np.std(distances_upper)
    uniformity = std_dist / (avg_dist + 1e-10)
    
    # Skewness of distance distribution
    skewness = 0 if std_dist == 0 else np.mean((distances_upper - avg_dist)**3) / (std_dist**3 + 1e-10)
    
    # Density estimation
    if n > 10:
        # Approximate spatial density via nearest neighbor distances
        sorted_dists = np.sort(distance_matrix, axis=1)[:, 1:min(6, n)]
        nn_density = 1 / (np.mean(sorted_dists) + 1e-10)
    else:
        nn_density = 1
    
    # === 2. LEARNED META-WEIGHTING FUNCTION ===
    # Adaptive weights based on instance characteristics
    def compute_meta_weights(n: int, uniformity: float, skewness: float, nn_density: float) -> dict:
        """Compute component weights based on problem features"""
        size_factor = np.log(n + 1) / np.log(100)  # Normalized by n=100
        
        # Base weights with instance-adaptive adjustments
        w_inv = 1.0
        
        # Lookahead weight: higher for uniform, lower for skewed distributions
        w_lookahead = 0.7 + 0.3 * (1 - abs(skewness)) * (1 + 0.2 * uniformity)
        
        # Spectral weight: higher when clear structure exists
        w_spectral = 0.4 * (1 - 0.3 * uniformity) * min(1.0, nn_density / 2)
        
        # MST weight: higher for irregular instances
        w_mst = 0.5 - 0.3 * uniformity + 0.2 * abs(skewness)
        
        # NN connectivity weight
        w_nn = 0.6 + 0.2 * size_factor
        
        return {
            'inv': max(0.5, min(2.0, w_inv)),
            'lookahead': max(0.3, min(1.5, w_lookahead)),
            'spectral': max(0.1, min(1.0, w_spectral)),
            'mst': max(0.2, min(1.0, w_mst)),
            'nn': max(0.3, min(1.0, w_nn))
        }
    
    meta_weights = compute_meta_weights(n, uniformity, skewness, nn_density)
    
    # === 3. BASE INVERSE DISTANCE WITH ADAPTIVE SMOOTHING ===
    epsilon = 1e-10 * avg_dist
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 4. SPECTRAL GRAPH ANALYSIS WITH DYNAMIC THRESHOLDING ===
    # Compute normalized Laplacian
    W = np.exp(-distance_matrix / (avg_dist + 1e-10))
    np.fill_diagonal(W, 0)
    D = np.diag(W.sum(axis=1))
    D_inv_sqrt = np.diag(1 / (np.sqrt(np.diag(D)) + 1e-10))
    L_norm = D_inv_sqrt @ (D - W) @ D_inv_sqrt
    
    # Compute eigenvalues and eigenvectors
    try:
        eigvals, eigvecs = np.linalg.eigh(L_norm)
        # Sort eigenvalues
        sort_idx = np.argsort(eigvals)
        eigvals = eigvals[sort_idx]
        eigvecs = eigvecs[:, sort_idx]
        
        # Fiedler vector for clustering
        fiedler_idx = 1 if n > 1 else 0
        fiedler_vec = eigvecs[:, fiedler_idx]
        
        # Adaptive clustering threshold
        if n > 10:
            # Use Otsu-like thresholding on Fiedler vector
            hist, bins = np.histogram(fiedler_vec, bins=min(20, n))
            bin_centers = (bins[:-1] + bins[1:]) / 2
            
            total = np.sum(hist)
            if total > 0:
                norm_hist = hist / total
                cumsum = np.cumsum(norm_hist)
                
                # Find threshold maximizing between-class variance
                max_var = 0
                best_threshold = np.median(fiedler_vec)
                
                for i in range(1, len(norm_hist)):
                    w0 = cumsum[i]
                    w1 = 1 - w0
                    
                    if w0 > 0 and w1 > 0:
                        mean0 = np.sum(norm_hist[:i] * bin_centers[:i]) / w0
                        mean1 = np.sum(norm_hist[i:] * bin_centers[i:]) / w1
                        between_var = w0 * w1 * (mean0 - mean1)**2
                        
                        if between_var > max_var:
                            max_var = between_var
                            best_threshold = bin_centers[i]
                
                cluster_assign = (fiedler_vec > best_threshold).astype(float)
            else:
                cluster_assign = np.zeros(n)
        else:
            cluster_assign = (fiedler_vec > np.median(fiedler_vec)).astype(float)
        
        # Spectral ratio for weight adjustment
        if len(eigvals) >= 3 and eigvals[-1] > 0:
            spectral_ratio = eigvals[1] / eigvals[2]
        else:
            spectral_ratio = 1.0
            
    except np.linalg.LinAlgError:
        cluster_assign = np.zeros(n)
        spectral_ratio = 1.0
    
    # Spectral penalty matrix
    spectral_penalty = np.ones((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if cluster_assign[i] != cluster_assign[j]:
                # Adaptive penalty based on spectral ratio
                penalty_strength = 0.5 + 0.3 * (1 - spectral_ratio)
                penalty_factor = 1 - penalty_strength * meta_weights['spectral']
                spectral_penalty[i, j] = spectral_penalty[j, i] = max(0.2, penalty_factor)
    
    # === 5. ENHANCED THREE-LEVEL PATH REGRET ANALYSIS ===
    # Precompute sorted distances and indices
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_dists = np.sort(distance_matrix, axis=1)
    
    # Parameters for lookahead
    k1 = min(4, n-1)  # First step candidates
    k2 = min(3, n-1)  # Second step candidates
    k3 = min(2, n-1)  # Third step candidates
    
    lookahead_score = np.ones((n, n))
    
    for i in range(n):
        # Best 1-step cost
        best_1step = sorted_dists[i, 1] if n > 1 else 0
        
        # Compute best 2-step and 3-step paths for this node
        best_2step = np.inf
        best_3step = np.inf
        
        # 2-step paths
        for j_idx in range(1, min(k1+1, n)):
            j = sorted_indices[i, j_idx]
            if j == i:
                continue
                
            # Best continuation from j (excluding i)
            for k_idx in range(1, min(k2+1, n)):
                k = sorted_indices[j, k_idx]
                if k != i and k != j:
                    cost_2 = distance_matrix[i, j] + distance_matrix[j, k]
                    best_2step = min(best_2step, cost_2)
        
        # 3-step paths (if enough nodes)
        if n >= 4:
            for j_idx in range(1, min(k1+1, n)):
                j = sorted_indices[i, j_idx]
                if j == i:
                    continue
                    
                for k_idx in range(1, min(k2+1, n)):
                    k = sorted_indices[j, k_idx]
                    if k in [i, j]:
                        continue
                        
                    for l_idx in range(1, min(k3+1, n)):
                        l = sorted_indices[k, l_idx]
                        if l not in [i, j, k]:
                            cost_3 = (distance_matrix[i, j] + 
                                     distance_matrix[j, k] + 
                                     distance_matrix[k, l])
                            best_3step = min(best_3step, cost_3)
        
        # If no valid paths found, use approximations
        if np.isinf(best_2step):
            best_2step = best_1step * 2 if best_1step > 0 else avg_dist * 2
            
        if np.isinf(best_3step):
            best_3step = best_2step * 1.5
        
        # Compute regrets for each edge from i
        for j in range(n):
            if i == j:
                continue
                
            # 1-step regret
            regret_1 = (distance_matrix[i, j] - best_1step) / (best_1step + 1e-10)
            
            # 2-step regret approximation
            # Find best continuation from j
            best_from_j = np.inf
            for k_idx in range(1, min(k2+1, n)):
                k = sorted_indices[j, k_idx]
                if k != i:
                    best_from_j = min(best_from_j, distance_matrix[j, k])
            
            if np.isinf(best_from_j):
                best_from_j = avg_dist
                
            cost_2 = distance_matrix[i, j] + best_from_j
            regret_2 = (cost_2 - best_2step) / (best_2step + 1e-10)
            
            # 3-step regret approximation
            if n >= 4:
                # Estimate third step
                third_options = []
                for k_idx in range(1, min(k2+1, n)):
                    k = sorted_indices[j, k_idx]
                    if k == i:
                        continue
                        
                    # Find best continuation from k
                    for l_idx in range(1, min(k3+1, n)):
                        l = sorted_indices[k, l_idx]
                        if l not in [i, j, k]:
                            third_options.append(distance_matrix[k, l])
                
                if third_options:
                    best_third = min(third_options)
                    cost_3 = distance_matrix[i, j] + best_from_j + best_third
                    regret_3 = (cost_3 - best_3step) / (best_3step + 1e-10)
                else:
                    regret_3 = regret_2
            else:
                regret_3 = regret_2
            
            # Combined regret with adaptive weighting
            w1 = 0.5 + 0.2 * (1 - uniformity)  # Higher weight for uniform instances
            w2 = 0.3
            w3 = 0.2
            
            combined_regret = w1 * regret_1 + w2 * regret_2 + w3 * regret_3
            
            # Apply penalty function
            penalty = np.exp(-0.8 * combined_regret)
            lookahead_score[i, j] = max(0.1, min(1.0, penalty))
    
    # === 6. MST BOTTLENECK ANALYSIS ===
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = (mst + mst.T) > 0
    
    mst_bonus = np.zeros((n, n))
    
    # Build MST adjacency list for path finding
    mst_adj = [[] for _ in range(n)]
    for i in range(n):
        for j in range(i+1, n):
            if mst_edges[i, j]:
                mst_adj[i].append((j, distance_matrix[i, j]))
                mst_adj[j].append((i, distance_matrix[i, j]))
    
    # Function to find maximum edge on MST path between two nodes
    def find_mst_bottleneck(start: int, end: int) -> float:
        """Find maximum edge weight on unique MST path between start and end"""
        if start == end:
            return 0
            
        # BFS to find path and track maximum edge
        visited = [False] * n
        parent = [-1] * n
        max_edge = [0.0] * n
        
        queue = [start]
        visited[start] = True
        
        while queue:
            current = queue.pop(0)
            
            if current == end:
                return max_edge[current]
            
            for neighbor, weight in mst_adj[current]:
                if not visited[neighbor]:
                    visited[neighbor] = True
                    parent[neighbor] = current
                    max_edge[neighbor] = max(max_edge[current], weight)
                    queue.append(neighbor)
        
        # If no path found (shouldn't happen in MST)
        return 0
    
    # Compute bottleneck improvements for non-MST edges
    for i in range(n):
        for j in range(i+1, n):
            if not mst_edges[i, j]:
                bottleneck = find_mst_bottleneck(i, j)
                if bottleneck > 0:
                    improvement = max(0, bottleneck - distance_matrix[i, j])
                    bonus = 0.4 * improvement / (bottleneck + 1e-10)
                    mst_bonus[i, j] = mst_bonus[j, i] = min(0.5, bonus)
    
    # === 7. ADAPTIVE NEAREST NEIGHBOR CONNECTIVITY ===
    k_nn = max(2, int(np.log2(n + 1) * (1 + 0.3 * uniformity)))
    k_nn = min(k_nn, n - 1)
    
    nn_connectivity = np.zeros((n, n))
    for i in range(n):
        nearest = sorted_indices[i, 1:k_nn+1]
        for rank, j in enumerate(nearest):
            # Exponential decay based on rank
            weight = np.exp(-2.0 * rank / k_nn)
            nn_connectivity[i, j] = weight
    
    # Symmetrize and normalize
    nn_connectivity = (nn_connectivity + nn_connectivity.T) / 2
    row_sums = nn_connectivity.sum(axis=1, keepdims=True)
    nn_connectivity = np.divide(nn_connectivity, row_sums + 1e-10)
    
    # === 8. COMBINE COMPONENTS WITH META-WEIGHTS ===
    combined = (
        meta_weights['inv'] * inv_dist *
        (meta_weights['lookahead'] * lookahead_score) *
        (1 + meta_weights['spectral'] * (spectral_penalty - 1)) *
        (1 + meta_weights['mst'] * mst_bonus) *
        (1 + meta_weights['nn'] * nn_connectivity)
    )
    
    np.fill_diagonal(combined, 0)
    
    # === 9. ENTROPY-BASED ADAPTIVE SPARSIFICATION ===
    # Global entropy calculation
    upper_triangle = combined[np.triu_indices(n, k=1)]
    if len(upper_triangle) > 1:
        global_probs = upper_triangle / (upper_triangle.sum() + 1e-10)
        global_entropy = -np.sum(global_probs * np.log(global_probs + 1e-10))
        max_global_entropy = np.log(len(upper_triangle))
        global_entropy_norm = global_entropy / max_global_entropy
    else:
        global_entropy_norm = 0.5
    
    # Create sparsification mask
    sparse_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf
        
        # Local entropy
        row_pos = row[row > 0]
        if len(row_pos) > 1:
            local_probs = row_pos / (row_pos.sum() + 1e-10)
            local_entropy = -np.sum(local_probs * np.log(local_probs + 1e-10))
            max_local_entropy = np.log(len(row_pos))
            local_entropy_norm = local_entropy / max_local_entropy
        else:
            local_entropy_norm = 0.5
        
        # Combined entropy score
        entropy_score = 0.6 * local_entropy_norm + 0.4 * global_entropy_norm
        
        # Adaptive keep ratio based on entropy and problem size
        base_ratio = 0.15 + 0.15 * entropy_score
        size_factor = min(0.3, 15 / n)
        keep_ratio = min(0.35, max(0.1, base_ratio + size_factor))
        
        k_keep = max(2, int(keep_ratio * (n - 1)))
        
        # Select top edges
        if k_keep < n - 1:
            threshold = np.partition(row, -k_keep)[-k_keep]
            sparse_mask[i] = row >= threshold
        else:
            sparse_mask[i] = row > -np.inf
        
        sparse_mask[i, i] = False
    
    # Ensure each node has at least 2 connections
    for i in range(n):
        if sparse_mask[i].sum() < 2:
            row = combined[i].copy()
            row[i] = -np.inf
            row[sparse_mask[i]] = -np.inf
            needed = 2 - sparse_mask[i].sum()
            if needed > 0 and row.max() > -np.inf:
                top_indices = np.argpartition(row, -needed)[-needed:]
                sparse_mask[i, top_indices] = True
    
    # Ensure symmetry and connectivity
    sparse_mask = sparse_mask | sparse_mask.T
    
    # Additional connectivity safeguard
    visited = np.zeros(n, dtype=bool)
    stack = [0]
    visited[0] = True
    
    while stack:
        current = stack.pop()
        neighbors = np.where(sparse_mask[current])[0]
        for neighbor in neighbors:
            if not visited[neighbor]:
                visited[neighbor] = True
                stack.append(neighbor)
    
    # If graph is disconnected, add minimum spanning tree edges
    if not np.all(visited):
        # Use MST edges to connect components
        for i in range(n):
            for j in range(i+1, n):
                if mst_edges[i, j]:
                    sparse_mask[i, j] = sparse_mask[j, i] = True
    
    # Apply sparsification
    heuristics_matrix = combined * sparse_mask
    
    # === 10. FINAL PROCESSING ===
    # Row normalization with smoothing
    for i in range(n):
        row_sum = heuristics_matrix[i].sum()
        if row_sum > 0:
            # Smooth between normalized scores and uniform distribution
            heuristics_matrix[i] = (0.85 * (heuristics_matrix[i] / row_sum) + 
                                   0.15 * (1 / (n - 1)))
        else:
            heuristics_matrix[i] = 1 / (n - 1)
        heuristics_matrix[i, i] = 0
    
    # Final symmetrization
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    return heuristics_matrix
