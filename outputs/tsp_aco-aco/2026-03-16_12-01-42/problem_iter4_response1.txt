import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics incorporating regret-based penalties and
    efficient two-step lookahead with adaptive local scaling.
    
    Args:
        distance_matrix: Symmetric square matrix of pairwise distances between nodes.
        
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges.
        Unpromising edges are set to zero via rank-based sparsification.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Dynamic parameter selection based on problem size
    k = max(2, min(8, int(np.log2(n)) + 1))  # For local scaling
    L = min(5, n-1)  # For lookahead computation (nearest neighbors to consider)
    
    # 2. Compute local density scales using k-nearest neighbors
    local_scales = np.zeros(n)
    sorted_dist_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        neighbor_dists = distance_matrix[i, sorted_dist_indices[i, 1:k+1]]
        local_scales[i] = np.mean(neighbor_dists) if len(neighbor_dists) > 0 else 1.0
    
    # 3. Global structure integration: Minimum Spanning Tree edges
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 4. Compute base heuristic with adaptive scaling
    scale_matrix = np.maximum(local_scales[:, None], local_scales[None, :])
    scaled_dist = distance_matrix / (scale_matrix + 1e-12)
    base_heuristic = np.exp(-scaled_dist ** 2)
    
    # 5. Regret-based penalty computation
    # Compute regret for each node based on alternative edge quality
    regret = np.zeros(n)
    for i in range(n):
        # Get distances to nearest neighbors
        neighbor_dists = distance_matrix[i, sorted_dist_indices[i, 1:4]]  # Top 3
        if len(neighbor_dists) >= 2:
            best = neighbor_dists[0]
            second_best = neighbor_dists[1]
            # Regret measures how much worse the second option is
            regret[i] = max(0, (second_best - best) / (best + 1e-12))
    
    # Apply regret penalty symmetrically: edges from nodes with high regret get boosted
    regret_matrix = 1.0 + (regret[:, None] + regret[None, :]) / 2.0
    
    # 6. Efficient two-step lookahead using precomputed nearest neighbors
    lookahead_score = np.zeros((n, n))
    
    if L >= 1 and n > 3:
        # Precompute nearest neighbor lists for each node
        nearest_lists = [sorted_dist_indices[i, 1:L+1] for i in range(n)]
        
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                # Find best continuation from j (excluding i)
                best_continuation = np.inf
                for k in nearest_lists[j]:
                    if k == i:
                        continue
                    path_score = distance_matrix[i, j] + distance_matrix[j, k]
                    if path_score < best_continuation:
                        best_continuation = path_score
                
                # Find best approach to i (excluding j)
                best_approach = np.inf
                for h in nearest_lists[i]:
                    if h == j:
                        continue
                    path_score = distance_matrix[h, i] + distance_matrix[i, j]
                    if path_score < best_approach:
                        best_approach = path_score
                
                # Only compute score if we found valid paths
                if best_continuation < np.inf and best_approach < np.inf:
                    avg_path = (best_continuation + best_approach) / 2.0
                    # Normalize by local scale
                    local_scale = max(local_scales[i], local_scales[j])
                    lookahead_score[i, j] = np.exp(-(avg_path / local_scale) ** 2)
    
    # 7. Combine components with adaptive weights
    # Weights adjust based on problem size
    if n > 20:
        base_weight = 0.4
        regret_weight = 0.3
        lookahead_weight = 0.3
    else:
        base_weight = 0.5
        regret_weight = 0.25
        lookahead_weight = 0.25
    
    combined_heuristic = (
        base_weight * base_heuristic * regret_matrix +
        lookahead_weight * lookahead_score
    )
    
    # Boost MST edges (diminishing factor with problem size)
    mst_bonus = 0.4 / np.log2(n + 1)
    combined_heuristic[mst_mask] *= (1 + mst_bonus)
    
    # 8. Symmetrize and zero diagonal
    combined_heuristic = (combined_heuristic + combined_heuristic.T) / 2
    np.fill_diagonal(combined_heuristic, 0)
    
    # 9. Rank-based sparsification (keep top O(n) edges)
    if n > 3:
        # Get upper triangular values
        triu_indices = np.triu_indices(n, 1)
        values = combined_heuristic[triu_indices]
        
        if len(values) > 0:
            # Keep approximately 2n edges (adjust based on problem size)
            n_keep = min(2 * n, len(values))
            sorted_values = np.sort(values)[::-1]
            
            if n_keep < len(values):
                threshold = sorted_values[n_keep - 1]
                mask = combined_heuristic < threshold
                combined_heuristic[mask] = 0
    
    # 10. Final normalization
    max_val = np.max(combined_heuristic)
    if max_val > 0:
        combined_heuristic = combined_heuristic / max_val
    
    return combined_heuristic
