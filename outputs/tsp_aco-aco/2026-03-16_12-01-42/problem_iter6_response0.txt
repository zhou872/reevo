import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse import csr_matrix
from scipy.sparse.linalg import eigsh

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection with 5-step lookahead,
    Christofides-informed initialization, and adaptive reinforcement learning.
    
    Features:
    1. 5-step recursive regret lookahead with path continuation analysis
    2. Christofides-derived edge set for initialization
    3. Spectral manifold embedding for multi-scale clustering
    4. Adaptive reservoir sampling for maintaining edge diversity
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result
    
    # ==================== ADAPTIVE PARAMETER TUNING ====================
    avg_dist = np.mean(distance_matrix[np.triu_indices(n, k=1)])
    std_dist = np.std(distance_matrix[np.triu_indices(n, k=1)])
    uniformity = std_dist / (avg_dist + 1e-10)
    
    # Dynamic parameters based on problem characteristics
    k_nn = max(3, int(np.log2(n) * (1.2 + 0.3 * uniformity)))
    lookahead_depth = min(5, max(3, int(np.log(n) / np.log(1.5))))
    spectral_dim = min(8, max(3, n // 20))
    reservoir_size = min(100, max(20, n // 5))
    
    # ==================== CHRISTOFIDES-INFORMED INITIALIZATION ====================
    # Get initial edge set from Christofides algorithm approximation
    christofides_bonus = np.zeros((n, n))
    
    try:
        # 1. Minimum Spanning Tree
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst = mst + mst.T
        
        # 2. Identify odd-degree vertices in MST
        odd_vertices = np.where((mst > 0).sum(axis=1) % 2 == 1)[0]
        
        # 3. Approximate minimum weight perfect matching on odd vertices
        if len(odd_vertices) > 0:
            # Create submatrix for odd vertices
            odd_dist = distance_matrix[np.ix_(odd_vertices, odd_vertices)]
            
            # Greedy matching approximation
            matched = np.zeros(len(odd_vertices), dtype=bool)
            matching_edges = []
            
            for _ in range(len(odd_vertices) // 2):
                # Find minimum unmatched edge
                min_val = np.inf
                min_i, min_j = -1, -1
                
                for i in range(len(odd_vertices)):
                    if matched[i]:
                        continue
                    for j in range(i + 1, len(odd_vertices)):
                        if matched[j]:
                            continue
                        if odd_dist[i, j] < min_val:
                            min_val = odd_dist[i, j]
                            min_i, min_j = i, j
                
                if min_i >= 0 and min_j >= 0:
                    matching_edges.append((odd_vertices[min_i], odd_vertices[min_j]))
                    matched[min_i] = matched[min_j] = True
            
            # Combine MST and matching edges
            christofides_edges = (mst > 0).astype(float)
            for i, j in matching_edges:
                christofides_edges[i, j] = christofides_edges[j, i] = 1.0
            
            # Create Eulerian tour approximation
            # Use depth-first traversal on the multigraph
            eulerian_graph = christofides_edges.copy()
            
            # Create heuristic bonus based on Eulerian circuit edges
            for i in range(n):
                neighbors = np.where(eulerian_graph[i] > 0)[0]
                for j in neighbors:
                    if i < j:
                        # Bonus for edges in the Christofides construction
                        christofides_bonus[i, j] = christofides_bonus[j, i] = 1.5
    except:
        # Fallback to MST edges only
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst = mst + mst.T
        christofides_bonus = (mst > 0).astype(float) * 1.2
    
    # ==================== 5-STEP RECURSIVE REGRET LOOKAHEAD ====================
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_dists = np.sort(distance_matrix, axis=1)
    
    def compute_k_step_best(start_node, k, max_branch=3, visited=None):
        """Recursively compute best k-step path from start_node"""
        if k == 1:
            # Return best edge from start_node (excluding self)
            for j_idx in range(1, n):
                j = sorted_indices[start_node, j_idx]
                if j != start_node:
                    return distance_matrix[start_node, j], [start_node, j]
            return np.inf, []
        
        if visited is None:
            visited = set([start_node])
        
        best_cost = np.inf
        best_path = []
        
        # Consider top branches from current node
        for branch in range(1, min(max_branch + 1, n)):
            next_node = sorted_indices[start_node, branch]
            if next_node in visited:
                continue
            
            new_visited = visited.copy()
            new_visited.add(next_node)
            
            # Recursive call for remaining steps
            remaining_cost, remaining_path = compute_k_step_best(
                next_node, k - 1, max_branch, new_visited)
            
            total_cost = distance_matrix[start_node, next_node] + remaining_cost
            
            if total_cost < best_cost:
                best_cost = total_cost
                best_path = [start_node] + remaining_path
        
        return best_cost, best_path
    
    # Precompute best paths for each depth
    best_paths = {}
    for i in range(n):
        for depth in range(1, min(lookahead_depth + 1, 6)):
            cost, path = compute_k_step_best(i, depth, max_branch=min(3, n // 2))
            best_paths[(i, depth)] = (cost, path)
    
    # Compute lookahead scores with dynamic weights
    lookahead_score = np.ones((n, n))
    
    for i in range(n):
        for j in range(n):
            if i != j:
                regrets = []
                weights = []
                
                # Compute regret for each lookahead depth
                for depth in range(1, min(lookahead_depth + 1, 6)):
                    base_cost, base_path = best_paths[(i, depth)]
                    
                    # Compute alternative cost through edge (i, j)
                    if depth == 1:
                        alt_cost = distance_matrix[i, j]
                    else:
                        # Recursively compute (depth-1)-step from j
                        remaining_cost, remaining_path = compute_k_step_best(
                            j, depth - 1, max_branch=min(3, n // 2), 
                            visited=set([i, j]))
                        alt_cost = distance_matrix[i, j] + remaining_cost
                    
                    # Compute normalized regret
                    regret = (alt_cost - base_cost) / (base_cost + 1e-10)
                    regrets.append(regret)
                    
                    # Dynamic weight based on depth and instance characteristics
                    # Deeper lookahead gets more weight for large/uniform instances
                    weight = np.exp(-0.8 * (depth - 1)) * (1 + 0.5 * uniformity)
                    weights.append(weight)
                
                # Normalize weights
                weights = np.array(weights)
                if weights.sum() > 0:
                    weights = weights / weights.sum()
                
                # Combined regret score
                combined_regret = np.sum(np.array(regrets) * weights)
                
                # Convert to score (higher is better)
                score = np.exp(-1.2 * combined_regret)
                lookahead_score[i, j] = max(0.1, min(1.0, score))
    
    # ==================== SPECTRAL MANIFOLD EMBEDDING ====================
    # Construct similarity matrix
    sigma = avg_dist / 3  # Adaptive bandwidth
    similarity = np.exp(-distance_matrix ** 2 / (2 * sigma ** 2))
    np.fill_diagonal(similarity, 0)
    
    # Normalized Laplacian
    D = np.diag(similarity.sum(axis=1))
    D_inv_sqrt = np.diag(1 / np.sqrt(np.diag(D) + 1e-10))
    L_norm = np.eye(n) - D_inv_sqrt @ similarity @ D_inv_sqrt
    
    # Spectral embedding with multiple eigenvectors
    try:
        if n <= 500:
            eigvals, eigvecs = np.linalg.eigh(L_norm)
        else:
            eigvals, eigvecs = eigsh(L_norm, k=min(spectral_dim + 1, n-1), which='SM')
        
        # Use eigenvectors corresponding to smallest non-zero eigenvalues
        sorted_idx = np.argsort(eigvals)
        embedding = eigvecs[:, sorted_idx[1:spectral_dim+1]]  # Skip first trivial
        
        # Compute manifold distance with adaptive metric
        manifold_dist = np.zeros((n, n))
        for i in range(n):
            for j in range(i+1, n):
                # Mahalanobis-like distance with eigenvalue weighting
                diff = embedding[i] - embedding[j]
                weights = 1 / (np.abs(eigvals[1:spectral_dim+1]) + 1e-10)
                manifold_dist[i, j] = manifold_dist[j, i] = np.sqrt(
                    np.sum(weights * diff ** 2))
        
        # Spectral score: favor edges connecting similar manifold positions
        max_manifold = manifold_dist.max()
        spectral_score = np.exp(-manifold_dist / (0.3 * max_manifold))
        
    except:
        # Fallback to simple clustering based on Fiedler vector
        spectral_score = np.ones((n, n))
    
    # ==================== ADAPTIVE RESERVOIR SAMPLING ====================
    # Maintain diversity of promising edges
    reservoir = np.zeros((reservoir_size, 2), dtype=int)
    reservoir_weights = np.zeros(reservoir_size)
    
    # Initialize reservoir with MST edges
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = np.argwhere(mst > 0)
    mst_edges = mst_edges[mst_edges[:, 0] < mst_edges[:, 1]]
    
    for idx, (i, j) in enumerate(mst_edges[:reservoir_size]):
        reservoir[idx] = [i, j]
        reservoir_weights[idx] = 1.0 / (distance_matrix[i, j] + 1e-10)
    
    # Update reservoir with promising edges from lookahead
    candidate_edges = []
    for i in range(n):
        for j in range(i+1, n):
            score = lookahead_score[i, j] * christofides_bonus[i, j]
            candidate_edges.append((score, i, j))
    
    candidate_edges.sort(reverse=True)
    
    for score, i, j in candidate_edges[:reservoir_size * 2]:
        # Reservoir sampling update
        if np.random.rand() < 0.3:  # Probability of inclusion
            min_idx = np.argmin(reservoir_weights)
            if score > reservoir_weights[min_idx]:
                reservoir[min_idx] = [i, j]
                reservoir_weights[min_idx] = score
    
    # Create reservoir bonus matrix
    reservoir_bonus = np.ones((n, n))
    for (i, j), weight in zip(reservoir, reservoir_weights):
        if weight > 0:
            reservoir_bonus[i, j] = reservoir_bonus[j, i] = 1.0 + 0.5 * weight
    
    # ==================== COMBINE ALL FACTORS ====================
    # Base inverse distance
    inv_dist = 1 / (distance_matrix + 1e-10 * avg_dist)
    np.fill_diagonal(inv_dist, 0)
    
    # Adaptive weights based on instance characteristics
    if n < 50:
        weights = [0.25, 0.30, 0.25, 0.20]  # Small instances: focus on lookahead
    elif uniformity < 0.3:
        weights = [0.20, 0.35, 0.25, 0.20]  # Uniform: emphasis on lookahead and spectral
    else:
        weights = [0.30, 0.25, 0.25, 0.20]  # Irregular: balance all factors
    
    w_inv, w_lookahead, w_spectral, w_reservoir = weights
    
    combined = (
        w_inv * inv_dist +
        w_lookahead * lookahead_score +
        w_spectral * spectral_score +
        w_reservoir * reservoir_bonus
    ) * (1 + 0.8 * christofides_bonus)
    
    # ==================== ADAPTIVE SPARSIFICATION ====================
    # Determine sparsity level based on problem complexity
    complexity = uniformity * np.log(n)
    keep_fraction = 0.15 + 0.10 * np.tanh(1.5 - 0.3 * complexity)
    
    sparsified = np.zeros_like(combined)
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf
        
        # Determine number of edges to keep
        k_keep = max(3, int(keep_fraction * (n - 1)))
        
        if k_keep < n - 1:
            # Keep top k_keep edges
            threshold = np.partition(row, -k_keep)[-k_keep]
            mask = row >= threshold
        else:
            mask = row > -np.inf
        
        # Ensure minimum connectivity
        if mask.sum() < 3:
            # Add strongest edges to reach minimum
            additional = 3 - mask.sum()
            if additional > 0:
                candidates = np.argsort(row)[-additional:]
                mask[candidates] = True
        
        sparsified[i] = combined[i] * mask
    
    # Ensure symmetry
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Final normalization with smoothing
    row_sums = sparsified.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1
    normalized = sparsified / row_sums
    
    # Apply softmax-like transformation
    beta = 2.0  # Temperature parameter
    exp_scores = np.exp(beta * normalized)
    exp_scores[np.eye(n, dtype=bool)] = 0
    
    row_sums_exp = exp_scores.sum(axis=1, keepdims=True)
    row_sums_exp[row_sums_exp == 0] = 1
    final_heuristics = exp_scores / row_sums_exp
    
    # Final symmetrization
    final_heuristics = (final_heuristics + final_heuristics.T) / 2
    np.fill_diagonal(final_heuristics, 0)
    
    return final_heuristics
