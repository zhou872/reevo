import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with improved performance and stability.
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance with stability epsilon
    epsilon = 1e-12
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Enhanced mutual nearest neighbor factor with adaptive k
    k_mutual = max(3, int(np.sqrt(n)))  # Adaptive k for mutual nearest neighbors
    mutual_nn_boost = np.zeros_like(distance_matrix)
    
    # Identify k-nearest neighbors for each node
    nn_indices = np.zeros((n, k_mutual), dtype=int)
    for i in range(n):
        # Get k+1 smallest (including self)
        nearest = np.argpartition(distance_matrix[i], k_mutual + 1)[:k_mutual + 1]
        nearest = nearest[nearest != i]
        if len(nearest) > k_mutual:
            nearest = nearest[:k_mutual]
        nn_indices[i, :len(nearest)] = nearest
    
    # Boost edges where both endpoints are mutual nearest neighbors
    for i in range(n):
        for j_idx in range(k_mutual):
            j = nn_indices[i, j_idx]
            if j == 0 and i == 0:
                continue
            if np.any(nn_indices[j] == i):  # Check if i is in j's nearest neighbors
                mutual_nn_boost[i, j] = 1.6  # Strong boost for mutual connections
    
    # 3. Efficient MST factor using Prim's algorithm with priority tracking
    mst_boost = np.zeros_like(distance_matrix)
    if n > 1:
        visited = np.zeros(n, dtype=bool)
        visited[0] = True
        min_edge_to_tree = distance_matrix[0].copy()
        source_node = np.zeros(n, dtype=int)
        
        for _ in range(n - 1):
            # Find minimum edge from visited to unvisited
            unvisited = ~visited
            if not np.any(unvisited):
                break
                
            min_val = np.inf
            best_j = -1
            
            # Vectorized search among unvisited nodes
            candidates = np.where(unvisited)[0]
            if len(candidates) > 0:
                min_idx = np.argmin(min_edge_to_tree[candidates])
                best_j = candidates[min_idx]
                min_val = min_edge_to_tree[best_j]
            
            if best_j != -1 and min_val < np.inf:
                i = source_node[best_j]
                mst_boost[i, best_j] = 2.0
                mst_boost[best_j, i] = 2.0
                visited[best_j] = True
                
                # Update distances for newly visited node
                for v in range(n):
                    if not visited[v] and distance_matrix[best_j, v] < min_edge_to_tree[v]:
                        min_edge_to_tree[v] = distance_matrix[best_j, v]
                        source_node[v] = best_j
    
    # 4. Improved degree balancing factor
    degree_penalty = np.zeros_like(distance_matrix)
    for i in range(n):
        # Count how many nodes have i as one of their k-nearest neighbors
        degree_count = 0
        for j in range(n):
            if i != j and np.any(nn_indices[j] == i):
                degree_count += 1
        # Normalize and invert: prefer edges to less popular nodes
        norm_degree = min(1.0, degree_count / (k_mutual * 2))
        degree_penalty[i, :] = 1.0 - 0.4 * norm_degree  # Modest penalty for high-degree nodes
    
    # 5. Combine factors with optimized weights
    heuristics = inv_dist.copy()
    heuristics += mutual_nn_boost
    heuristics *= (1.0 + mst_boost)  # MST gives multiplicative boost
    heuristics *= degree_penalty
    
    # 6. Improved adaptive sparsification
    adaptive_k = min(15, max(3, n // 4))  # Adaptive top-k per node
    
    for i in range(n):
        row = heuristics[i].copy()
        # Keep top adaptive_k edges, but ensure we don't remove all connections
        if adaptive_k < n - 1:
            # Find threshold for top adaptive_k edges
            threshold_idx = n - adaptive_k - 1
            threshold = np.partition(row, threshold_idx)[threshold_idx]
            # Zero out edges below threshold, but keep at least 2 connections
            below_threshold = row < threshold
            if np.sum(~below_threshold) < 2:  # If less than 2 edges above threshold
                # Keep top 2 edges
                top2_threshold = np.partition(row, n - 3)[n - 3]
                row[row < top2_threshold] = 0
            else:
                row[below_threshold] = 0
            heuristics[i] = row
    
    # 7. Symmetrize while preserving sparsity pattern
    # Take maximum of both directions, but weight toward mutual decisions
    symmetric_heuristics = np.maximum(heuristics, heuristics.T * 0.7)
    
    # 8. Final cleanup
    np.fill_diagonal(symmetric_heuristics, 0)
    
    # Ensure numerical stability
    symmetric_heuristics = np.maximum(symmetric_heuristics, 0)
    
    return symmetric_heuristics
