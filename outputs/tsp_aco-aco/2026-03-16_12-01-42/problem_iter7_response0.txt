import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with proportional regret penalization
    and optimized multiplicative factor fusion.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        # Simple case: uniform heuristic for trivial instances
        heuristics = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    epsilon = 1e-12
    
    # === 1. Base inverse distance ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Proportional Regret Penalization ===
    # Implementation: For each node, compute the proportional regret
    # regret = (second_best_distance - best_distance) / best_distance
    # Prioritize edges where alternative options are much worse
    regret_factor = np.ones_like(distance_matrix)
    
    for i in range(n):
        # Get distances from node i
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        
        # Find two smallest distances for regret calculation
        sorted_idx = np.argsort(dist_i)[:2]
        d1, d2 = dist_i[sorted_idx[0]], dist_i[sorted_idx[1]]
        
        # Compute proportional regret for each edge from i
        for j in range(n):
            if i == j:
                continue
                
            d_ij = distance_matrix[i, j]
            
            # Find the best alternative to j
            if j == sorted_idx[0]:
                best_alternative = d2
            else:
                best_alternative = d1
            
            # Proportional regret: (alternative - current) / current
            # Higher regret means fewer good alternatives
            if d_ij > epsilon:
                regret = (best_alternative - d_ij) / d_ij
                # Scale regret factor between 0.5 and 2.0
                regret_factor[i, j] = max(0.5, min(2.0, 1.0 + regret))
    
    # === 3. Mutual Nearest Neighbor Boost ===
    # Strongly boost edges that are mutually near in both directions
    k_mutual = max(3, int(np.sqrt(n)))
    mutual_boost = np.ones_like(distance_matrix)
    
    # Find k-nearest neighbors for each node
    nn_indices = np.zeros((n, k_mutual), dtype=int)
    for i in range(n):
        # Get k smallest distances (excluding self)
        nearest = np.argpartition(distance_matrix[i], k_mutual + 1)[:k_mutual + 1]
        nearest = nearest[nearest != i][:k_mutual]
        nn_indices[i, :len(nearest)] = nearest
    
    # Apply mutual nearest neighbor boost
    for i in range(n):
        for j_idx in range(k_mutual):
            j = nn_indices[i, j_idx]
            if j == 0 and i == 0:
                continue
            if np.any(nn_indices[j] == i):
                # Strong boost for reciprocal nearest neighbors
                mutual_boost[i, j] = 2.0
    
    # === 4. Multiplicative Factor Fusion ===
    # Combine components multiplicatively for enhanced differentiation
    heuristics = inv_dist.copy()
    heuristics *= regret_factor
    heuristics *= mutual_boost
    
    # === 5. Adaptive Spatial Sparsification ===
    # Keep only the most promising edges based on local connectivity
    adaptive_k = min(15, max(4, n // 3))
    
    for i in range(n):
        row = heuristics[i].copy()
        
        if adaptive_k < n - 1:
            # Keep top-k edges by heuristic value
            sorted_indices = np.argsort(row)[-adaptive_k:]
            keep_mask = np.zeros(n, dtype=bool)
            keep_mask[sorted_indices] = True
            
            # Always keep mutual nearest neighbors
            for j_idx in range(k_mutual):
                j = nn_indices[i, j_idx]
                if j != 0 or i != 0:
                    if np.any(nn_indices[j] == i):
                        keep_mask[j] = True
            
            # Ensure at least 3 edges are kept
            if np.sum(keep_mask) < 3:
                top_indices = np.argsort(row)[-3:]
                keep_mask[top_indices] = True
            
            # Apply mask: zero out unpromising edges
            row[~keep_mask] = 0
            heuristics[i] = row
    
    # === 6. Symmetrization and Normalization ===
    # Make matrix symmetric with preference for mutual decisions
    symmetric_heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(symmetric_heuristics, 0)
    
    # Normalize to [0, 1] range for consistency
    max_val = np.max(symmetric_heuristics)
    if max_val > epsilon:
        symmetric_heuristics = symmetric_heuristics / max_val
    
    return symmetric_heuristics
