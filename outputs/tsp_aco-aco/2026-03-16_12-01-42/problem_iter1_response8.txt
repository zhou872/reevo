import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Distance-savings hybrid score
    2. Minimum Spanning Tree (MST) based sparsification
    3. Local neighbor density bonus
    4. Symmetry enforcement for undirected graphs
    """
    n = distance_matrix.shape[0]
    if n < 2:
        return np.zeros_like(distance_matrix)
    
    # Add small epsilon to avoid division by zero
    eps = 1e-10
    dist = distance_matrix + eps * np.eye(n)
    
    # 1. Base inverse distance score (like v1 but with damping)
    inv_dist = 1 / (dist ** 1.2)
    
    # 2. Savings heuristic score: s(i,j) = d(i,0) + d(0,j) - d(i,j)
    # Using the node with minimum average distance as pseudo-depot
    avg_distances = np.mean(dist, axis=1)
    depot = np.argmin(avg_distances)
    savings = dist[depot, :][:, np.newaxis] + dist[depot, :][np.newaxis, :] - dist
    savings = np.maximum(savings, 0)  # Negative savings don't make sense
    
    # 3. MST-based sparsification
    mst = minimum_spanning_tree(dist)
    mst_mask = (mst + mst.T) > 0  # Convert to symmetric mask
    mst_bonus = 2.0  # Bonus factor for MST edges
    
    # 4. Local neighbor density: edges connecting to nodes with many nearby alternatives
    # For each node, count edges within certain radius
    radius = np.percentile(dist[dist > 0], 25)  # 25th percentile distance
    neighbor_density = np.sum(dist < radius, axis=1) - 1  # Subtract self
    neighbor_bonus = 1 + 0.1 * (neighbor_density[:, np.newaxis] + neighbor_density[np.newaxis, :])
    
    # 5. Combined score
    combined_score = inv_dist * (1 + 0.5 * savings / (np.max(savings) + eps)) * neighbor_bonus
    combined_score *= np.where(mst_mask, mst_bonus, 1.0)
    
    # 6. Symmetry enforcement for undirected TSP
    combined_score = (combined_score + combined_score.T) / 2
    
    # 7. Aggressive sparsification: keep only top k edges per node
    k = min(5, n - 1)
    for i in range(n):
        row = combined_score[i].copy()
        row[i] = -np.inf  # Ignore self
        threshold = np.sort(row)[-k] if k > 0 else -np.inf
        combined_score[i] = np.where(combined_score[i] >= threshold, combined_score[i], 0)
    
    # Set diagonal to zero
    np.fill_diagonal(combined_score, 0)
    
    return combined_score
