import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection combining multiple structural insights.
    
    Features:
    1. MST Edge Bonus: Edges present in minimum spanning tree get boosted scores
    2. Local Clustering: Favors edges connecting nodes with shared neighbors
    3. Adaptive Sparsification: Zeroes out edges longer than median per node
    4. Directional Correction: Adjusts for asymmetry in local connectivity patterns
    """
    n = distance_matrix.shape[0]
    
    # Base inverse distance (avoid division by zero)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / np.where(distance_matrix > 0, distance_matrix, np.inf)
    np.fill_diagonal(inv_dist, 0)
    
    # 1. Minimum Spanning Tree bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = np.where(mst + mst.T > 0, 1.5, 0)
    
    # 2. Local clustering density score
    # Find k-nearest neighbors for each node (k = sqrt(n))
    k = max(2, int(np.sqrt(n)))
    nearest_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        indices = np.argsort(distance_matrix[i])[1:k+1]
        nearest_mask[i, indices] = True
    
    # Shared neighbor ratio: how many common nearest neighbors
    shared_neighbors = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            shared = np.intersect1d(
                np.where(nearest_mask[i])[0],
                np.where(nearest_mask[j])[0]
            ).size
            shared_neighbors[i, j] = shared_neighbors[j, i] = shared / k
    
    # 3. Distance-based sparsification with adaptive threshold
    # Use median distance per node as threshold
    row_medians = np.median(distance_matrix, axis=1, keepdims=True)
    col_medians = np.median(distance_matrix, axis=0, keepdims=True)
    median_threshold = (row_medians + col_medians.T) / 2
    
    # Sparsification mask
    sparsity_mask = distance_matrix < median_threshold
    
    # 4. Directional connectivity correction
    # Measure asymmetry in local connectivity
    out_density = np.sum(nearest_mask, axis=1, keepdims=True) / (n - 1)
    in_density = np.sum(nearest_mask, axis=0, keepdims=True) / (n - 1)
    directional_correction = 1 + 0.3 * (out_density - in_density.T)
    directional_correction = np.clip(directional_correction, 0.7, 1.3)
    
    # Combine all components
    heuristics = inv_dist * directional_correction
    heuristics *= (1 + 0.5 * shared_neighbors)  # Density bonus
    heuristics += mst_bonus                     # MST edge boost
    
    # Apply sparsification
    heuristics = heuristics * sparsity_mask
    
    # Normalize to comparable scale with original inverse distances
    original_norm = np.mean(np.abs(inv_dist[inv_dist > 0])) if np.any(inv_dist > 0) else 1
    current_norm = np.mean(np.abs(heuristics[heuristics > 0])) if np.any(heuristics > 0) else 1
    
    if current_norm > 0:
        heuristics = heuristics * (original_norm / current_norm)
    
    # Ensure symmetry
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
