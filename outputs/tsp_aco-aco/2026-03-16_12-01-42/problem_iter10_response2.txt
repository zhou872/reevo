import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge prioritization with multi-step path analysis,
    clustering-based structural patterns, and optimized adaptive fusion.
    
    Key Innovations:
    1. 3 and 4-step path efficiency analysis for deeper lookahead
    2. Cluster density and structural scoring for global patterns
    3. Bayesian optimization-inspired weight adaptation
    4. Optimized computational efficiency with vectorization
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Enhanced Base Inverse Distance ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Dynamic Regret with Variance-Adaptive Order ===
    off_diag_mask = ~np.eye(n, dtype=bool)
    global_distances = distance_matrix[off_diag_mask]
    
    if len(global_distances) > 0:
        global_mean = np.mean(global_distances)
        global_std = np.std(global_distances)
        cv = global_std / (global_mean + epsilon)
    else:
        cv = 0.5
        global_mean = 1.0
    
    # Adaptive regret order based on problem characteristics
    regret_matrix = np.ones_like(distance_matrix)
    
    # Determine optimal regret order for each node
    for i in range(n):
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        
        # Node-specific CV for adaptive order selection
        node_mean = np.mean(dists_i[dists_i < np.inf])
        node_std = np.std(dists_i[dists_i < np.inf])
        node_cv = node_std / (node_mean + epsilon)
        
        # Adaptive k based on node and global characteristics
        k_regret = min(max(3, int(3 + 3 * node_cv + 2 * cv)), n - 1)
        
        if len(dists_i) >= k_regret:
            # Get k_regret best distances
            indices = np.argpartition(dists_i, k_regret)[:k_regret]
            best_dists = dists_i[indices]
            
            # Sort the k_regret best distances
            sorted_idx = np.argsort(best_dists)
            sorted_indices = indices[sorted_idx]
            sorted_dists = best_dists[sorted_idx]
            
            for j in range(n):
                if i == j:
                    continue
                
                d_ij = distance_matrix[i, j]
                if j in sorted_indices[:k_regret-1]:
                    # Edge is in top k-1 edges
                    kth_best = sorted_dists[-1]
                    if d_ij > 0:
                        regret = (kth_best - d_ij) / (d_ij + epsilon)
                        # Enhanced sigmoid with adaptive steepness
                        regret_matrix[i, j] = 1.0 + np.tanh(2 * regret)
                else:
                    # Edge is not in top k-1
                    best_excluding_ij = sorted_dists[0] if j != sorted_indices[0] else sorted_dists[1]
                    if d_ij > 0:
                        regret = (best_excluding_ij - d_ij) / (d_ij + epsilon)
                        regret_matrix[i, j] = 1.0 - 0.4 * np.tanh(-2 * regret)
    
    # === 3. Multi-step Path Efficiency Analysis ===
    path_efficiency = np.ones_like(distance_matrix)
    
    if n > 4:
        # Dynamic neighborhood sizes based on problem scale
        k_neighbors = min(max(3, int(np.log2(n) * 2)), 8)
        
        # Precompute k nearest neighbors for all nodes
        neighbor_indices = np.zeros((n, k_neighbors), dtype=int)
        neighbor_dists = np.zeros((n, k_neighbors))
        
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            if len(dists) >= k_neighbors:
                # Get k_neighbors nearest indices
                nearest_idx = np.argpartition(dists, k_neighbors)[:k_neighbors]
                # Sort them by distance
                sorted_order = np.argsort(dists[nearest_idx])
                nearest_idx = nearest_idx[sorted_order]
                neighbor_indices[i] = nearest_idx
                neighbor_dists[i] = dists[nearest_idx]
            else:
                # All other nodes are neighbors
                all_idx = np.arange(n)
                mask = all_idx != i
                nearest_idx = all_idx[mask]
                neighbor_indices[i, :len(nearest_idx)] = nearest_idx
                neighbor_dists[i, :len(nearest_idx)] = dists[nearest_idx]
        
        # Compute path efficiency for all pairs
        for i in range(n):
            for j in range(i+1, n):
                d_ij = distance_matrix[i, j]
                
                # 2-step paths through common neighbors
                best_two_step = np.inf
                for k in neighbor_indices[i]:
                    if k != j:
                        two_step_cost = distance_matrix[i, k] + distance_matrix[k, j]
                        best_two_step = min(best_two_step, two_step_cost)
                
                # 3-step paths through two intermediate nodes
                best_three_step = np.inf
                for k in neighbor_indices[i]:
                    if k != j:
                        for l in neighbor_indices[j]:
                            if l != i and l != k:
                                three_step_cost = (distance_matrix[i, k] + 
                                                  distance_matrix[k, l] + 
                                                  distance_matrix[l, j])
                                best_three_step = min(best_three_step, three_step_cost)
                
                # 4-step paths (deep lookahead)
                best_four_step = np.inf
                if n > 8:
                    for k in neighbor_indices[i]:
                        if k != j:
                            # Limited exploration for efficiency
                            for l_idx in range(min(3, k_neighbors)):
                                l = neighbor_indices[k, l_idx]
                                if l != i and l != j:
                                    for m in neighbor_indices[j]:
                                        if m != i and m != k and m != l:
                                            four_step_cost = (distance_matrix[i, k] +
                                                            distance_matrix[k, l] +
                                                            distance_matrix[l, m] +
                                                            distance_matrix[m, j])
                                            best_four_step = min(best_four_step, four_step_cost)
                
                # Combine multi-step alternatives
                best_alt = min(best_two_step, best_three_step, best_four_step)
                
                if best_alt < np.inf:
                    # Enhanced efficiency ratio with non-linear transformation
                    efficiency_ratio = d_ij / (best_alt + epsilon)
                    
                    if efficiency_ratio <= 0.8:
                        # Strong boost for significantly better edges
                        boost = 1.5 * np.exp(-5 * (efficiency_ratio - 0.8)**2)
                    elif efficiency_ratio <= 1.2:
                        # Moderate scoring for comparable edges
                        boost = 1.0
                    else:
                        # Penalty for inefficient edges
                        boost = np.exp(-2 * (efficiency_ratio - 1.2))
                    
                    path_efficiency[i, j] = boost
                    path_efficiency[j, i] = boost
    
    # === 4. Cluster-based Structural Scoring ===
    cluster_scores = np.ones_like(distance_matrix)
    
    if n > 6:
        # Estimate cluster structure using distance variance patterns
        node_density = np.zeros(n)
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            if len(dists) > 3:
                # Use harmonic mean for density estimation
                k_density = min(5, n-1)
                nearest = np.partition(dists, k_density)[:k_density]
                node_density[i] = k_density / np.sum(1/(nearest + epsilon))
        
        # Normalize densities
        if np.max(node_density) > np.min(node_density):
            norm_density = (node_density - np.min(node_density)) / \
                          (np.max(node_density) - np.min(node_density) + epsilon)
        else:
            norm_density = np.ones(n) * 0.5
        
        # Compute edge scores based on cluster compatibility
        for i in range(n):
            for j in range(i+1, n):
                # Similarity in local density
                density_sim = 1.0 - abs(norm_density[i] - norm_density[j])
                
                # Distance ratio to local minima
                d_ij = distance_matrix[i, j]
                min_i = np.min(distance_matrix[i, distance_matrix[i] > 0])
                min_j = np.min(distance_matrix[j, distance_matrix[j] > 0])
                ratio_i = d_ij / (min_i + epsilon)
                ratio_j = d_ij / (min_j + epsilon)
                
                # Combined structural score
                if ratio_i <= 1.5 and ratio_j <= 1.5:
                    # Good intra-cluster edge
                    cluster_scores[i, j] = 0.7 + 0.6 * density_sim
                else:
                    # Inter-cluster or outlier edge
                    cluster_scores[i, j] = 0.4 + 0.3 * density_sim
                
                cluster_scores[j, i] = cluster_scores[i, j]
    
    # === 5. Bayesian-inspired Weight Adaptation ===
    # Adaptive weights based on problem characteristics
    if n <= 10:
        # Small instances: favor regret and path efficiency
        weights = {'inv': 0.8, 'regret': 1.3, 'path': 1.2, 'cluster': 0.7}
    elif n <= 30:
        # Medium instances: balanced approach
        weights = {'inv': 0.9, 'regret': 1.1, 'path': 1.1, 'cluster': 0.9}
    elif n <= 100:
        # Large instances: favor structural patterns
        weights = {'inv': 1.0, 'regret': 0.9, 'path': 1.0, 'cluster': 1.1}
    else:
        # Very large instances: emphasize efficiency and structure
        weights = {'inv': 0.8, 'regret': 0.8, 'path': 1.2, 'cluster': 1.2}
    
    # Adjust weights based on distance variation
    weights['regret'] *= (0.8 + 0.4 * cv)
    weights['cluster'] *= (0.9 + 0.2 * cv)
    
    # Multiplicative fusion with normalized exponents
    total_weight = sum(weights.values())
    for key in weights:
        weights[key] /= total_weight
    
    # Component fusion with stabilization
    heuristics = (
        np.clip(inv_dist, 0, 1e6) ** weights['inv'] *
        np.clip(regret_matrix, 0.1, 10) ** weights['regret'] *
        np.clip(path_efficiency, 0.1, 10) ** weights['path'] *
        np.clip(cluster_scores, 0.1, 10) ** weights['cluster']
    )
    
    # === 6. Enhanced MST Reinforcement ===
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            mst_mask = (mst_dense > 0) | (mst_dense.T > 0)
            
            # Adaptive MST boost based on problem size
            mst_boost = 1.5 + 0.5 * np.log2(n) / np.log2(100)  # 1.5 to 2.0
            heuristics[mst_mask] *= min(mst_boost, 2.0)
        except:
            pass  # Fallback if MST computation fails
    
    # === 7. Optimized Adaptive Sparsification ===
    for i in range(n):
        row = heuristics[i].copy()
        
        # Determine keep threshold based on distribution
        non_zero = row[row > 0]
        if len(non_zero) > 4:
            # Adaptive quantile selection
            mean_val = np.mean(non_zero)
            std_val = np.std(non_zero)
            
            # Dynamic threshold: mean + alpha*std
            # More aggressive sparsification for larger n
            alpha = max(0.1, 0.5 - n/200)
            threshold = mean_val + alpha * std_val
            
            # Ensure MST edges are kept
            mst_mask_i = np.zeros(n, dtype=bool)
            if n > 2:
                try:
                    mst_i = minimum_spanning_tree(distance_matrix).toarray()[i] > 0
                    mst_mask_i = mst_i
                except:
                    pass
            
            # Keep edges above threshold or MST edges
            keep_mask = (row >= threshold) | mst_mask_i
            
            # Ensure minimum and maximum degrees
            min_degree = max(2, int(np.log2(n + 1)))
            max_degree = min(10, n-1)
            
            if np.sum(keep_mask) < min_degree:
                # Add top edges to reach minimum degree
                top_idx = np.argsort(row)[-min_degree:]
                keep_mask[top_idx] = True
            elif np.sum(keep_mask) > max_degree:
                # Keep only top-k edges, preserving MST
                non_mst_mask = keep_mask & ~mst_mask_i
                non_mst_indices = np.where(non_mst_mask)[0]
                if len(non_mst_indices) > 0:
                    non_mst_values = row[non_mst_indices]
                    keep_count = max_degree - np.sum(mst_mask_i)
                    if keep_count > 0:
                        top_non_mst_idx = np.argsort(non_mst_values)[-keep_count:]
                        keep_mask = np.zeros(n, dtype=bool)
                        keep_mask[mst_mask_i] = True
                        keep_mask[non_mst_indices[top_non_mst_idx]] = True
            
            # Apply sparsification
            row[~keep_mask] = 0
            heuristics[i] = row
    
    # === 8. Final Processing ===
    # Symmetrize by maximum (preserve strongest signals)
    heuristics = np.maximum(heuristics, heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    # Final normalization with enhancement
    max_val = np.max(heuristics)
    if max_val > epsilon:
        heuristics = heuristics / max_val
        
        # Non-linear enhancement to increase differentiation
        # Preserve very small values while enhancing moderate ones
        enhancement_factor = 0.85
        heuristics = np.power(heuristics, enhancement_factor)
        
        # Final scaling to [0,1]
        heuristics = heuristics / np.max(heuristics)
    
    return heuristics
