import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge selection heuristics with adaptive regret scaling,
    dynamic mutual neighbor thresholds, and balanced additive fusion.
    
    Key Improvements:
    1. Adaptive regret scaling based on global distance variance
    2. Dynamic mutual neighbor criteria with distance-based thresholds
    3. Balanced additive component fusion for robustness
    4. Distance variance weighting for spatial coherence
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        heuristics = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    epsilon = 1e-12
    
    # === Global Distance Statistics ===
    # Extract upper triangular distances (excluding diagonal)
    triu_indices = np.triu_indices(n, k=1)
    flat_dists = distance_matrix[triu_indices]
    
    if len(flat_dists) == 0:
        heuristics = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # Compute robust statistics
    dist_mean = np.mean(flat_dists)
    dist_std = np.std(flat_dists)
    dist_cv = dist_std / (dist_mean + epsilon)  # Coefficient of variation
    
    # === 1. Base Inverse Distance with Distance Variance Weighting ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # Weight by global variance: higher variance = stronger inverse distance preference
    # Using a sigmoid function to map CV to weight between 0.5 and 1.5
    inv_dist_weight = 1.0 + 0.5 * np.tanh(2 * dist_cv - 1)
    inv_dist = inv_dist_weight * inv_dist
    
    # === 2. Adaptive Regret Scaling ===
    regret_scores = np.ones_like(distance_matrix)
    
    for i in range(n):
        # Get distances from node i
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        
        # Find two smallest distances
        sorted_idx = np.argsort(dist_i)
        d1_idx, d2_idx = sorted_idx[0], sorted_idx[1]
        d1, d2 = dist_i[d1_idx], dist_i[d2_idx]
        
        # Adaptive k for regret calculation based on problem size and variance
        # Higher variance = consider more alternatives
        k_regret = min(max(4, int(np.sqrt(n) * (1 + dist_cv))), n - 1)
        
        # Get k smallest distances for robust regret calculation
        if n - 1 > k_regret:
            k_smallest_idx = sorted_idx[:k_regret]
            k_smallest = dist_i[k_smallest_idx]
        else:
            k_smallest_idx = sorted_idx[:-1]  # Exclude diagonal
            k_smallest = dist_i[k_smallest_idx]
        
        # Compute mean and std of k smallest distances for local scaling
        if len(k_smallest) > 0:
            local_mean = np.mean(k_smallest)
            local_std = np.std(k_smallest)
            local_cv = local_std / (local_mean + epsilon)
        else:
            local_mean = d1
            local_cv = 0
        
        for j in range(n):
            if i == j:
                continue
                
            d_ij = distance_matrix[i, j]
            
            # Find best alternative to j among k smallest
            if j in k_smallest_idx:
                # Create mask excluding j
                mask = k_smallest_idx != j
                alt_distances = k_smallest[mask]
                if len(alt_distances) > 0:
                    best_alternative = np.min(alt_distances)
                else:
                    best_alternative = d1 if j != d1_idx else d2
            else:
                best_alternative = d1
            
            # Adaptive regret calculation with local scaling
            if d_ij > epsilon:
                base_regret = (best_alternative - d_ij) / d_ij
                
                # Scale regret by local coefficient of variation
                # Higher local variance = stronger regret effect
                regret_scale = 1.0 + 0.5 * np.tanh(local_cv)
                scaled_regret = base_regret * regret_scale
                
                # Transform to score with adaptive bounds
                min_score = 0.5  # Minimum score for poor edges
                max_score = 2.0  # Maximum score for excellent edges
                
                # Sigmoid transformation for bounded, smooth scores
                regret_scores[i, j] = min_score + (max_score - min_score) * (
                    1.0 / (1.0 + np.exp(-2.0 * scaled_regret))
                )
    
    # === 3. Dynamic Mutual Neighbor Criteria ===
    mutual_boost = np.ones_like(distance_matrix)
    
    # Dynamic k based on problem size and distance variance
    k_base = max(3, min(15, int(np.sqrt(n))))
    k_neighbors = min(k_base + int(2 * dist_cv * k_base), n - 1)
    
    # Precompute nearest neighbors for each node
    nn_indices = np.zeros((n, k_neighbors), dtype=int)
    nn_distances = np.zeros((n, k_neighbors))
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if n - 1 > k_neighbors:
            # Get k_neighbors nearest indices
            nearest_idx = np.argpartition(dists, k_neighbors)[:k_neighbors]
            # Sort them by distance
            sorted_order = np.argsort(dists[nearest_idx])
            nearest_idx = nearest_idx[sorted_order]
            nn_indices[i] = nearest_idx
            nn_distances[i] = dists[nearest_idx]
        else:
            # All other nodes are neighbors
            all_idx = np.arange(n)
            mask = all_idx != i
            nearest_idx = all_idx[mask]
            nn_indices[i, :len(nearest_idx)] = nearest_idx
            nn_distances[i, :len(nearest_idx)] = dists[nearest_idx]
    
    # Dynamic mutual neighbor threshold
    # Base threshold as multiple of minimum distance
    threshold_factor = 1.2 + 0.3 * dist_cv  # Higher variance = more lenient threshold
    
    for i in range(n):
        for j_idx in range(k_neighbors):
            j = nn_indices[i, j_idx]
            if j == 0 and i == 0 and n > 1:
                continue
                
            # Check if i is in j's neighborhood
            if np.any(nn_indices[j] == i):
                # Check distance ratio condition
                d_ij = distance_matrix[i, j]
                min_dist_i = nn_distances[i, 0] if nn_distances[i, 0] > 0 else d_ij
                min_dist_j = nn_distances[j, 0] if nn_distances[j, 0] > 0 else d_ij
                
                # Dynamic threshold based on both nodes' minimum distances
                threshold_i = threshold_factor * min_dist_i
                threshold_j = threshold_factor * min_dist_j
                
                # Apply boost if within dynamic thresholds
                if d_ij <= min(threshold_i, threshold_j):
                    # Stronger boost for closer mutual neighbors
                    rank_i = np.where(nn_indices[i] == j)[0]
                    rank_j = np.where(nn_indices[j] == i)[0]
                    
                    if len(rank_i) > 0 and len(rank_j) > 0:
                        rank_i = rank_i[0]
                        rank_j = rank_j[0]
                        # Boost decays with rank
                        rank_penalty = (rank_i + rank_j) / (2 * k_neighbors)
                        boost_strength = 1.8 - 0.6 * rank_penalty
                        mutual_boost[i, j] = max(mutual_boost[i, j], boost_strength)
    
    # === 4. Spatial Coherence Component ===
    spatial_scores = np.ones_like(distance_matrix)
    
    # Compute node spatial signatures using distance variance
    node_variances = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        valid_dists = dists[dists < np.inf]
        if len(valid_dists) > 0:
            node_variances[i] = np.var(valid_dists) / (np.mean(valid_dists) + epsilon)
    
    # Normalize variances
    if np.max(node_variances) > np.min(node_variances):
        var_norm = (node_variances - np.min(node_variances)) / (
            np.max(node_variances) - np.min(node_variances) + epsilon
        )
    else:
        var_norm = np.ones_like(node_variances) * 0.5
    
    # Favor edges between nodes with similar distance variance patterns
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Similarity score based on variance patterns
            var_similarity = 1.0 - np.abs(var_norm[i] - var_norm[j])
            # Adjust spatial score based on similarity
            spatial_scores[i, j] = 0.8 + 0.4 * var_similarity
    
    # === 5. Balanced Additive Fusion ===
    # Adaptive weights based on problem characteristics
    # Higher variance = more weight on regret and spatial components
    base_weight = 0.35
    regret_weight = 0.30 + 0.10 * np.tanh(dist_cv)
    spatial_weight = 0.25 + 0.10 * np.tanh(dist_cv)
    mutual_weight = 0.10
    
    # Normalize weights to sum to 1
    total_weight = base_weight + regret_weight + spatial_weight + mutual_weight
    base_weight /= total_weight
    regret_weight /= total_weight
    spatial_weight /= total_weight
    mutual_weight /= total_weight
    
    # Additive combination with normalized components
    # Normalize each component to [0,1] range first
    def normalize_component(component):
        max_val = np.max(component)
        if max_val > epsilon:
            return component / max_val
        return component
    
    norm_inv_dist = normalize_component(inv_dist)
    norm_regret = normalize_component(regret_scores)
    norm_spatial = normalize_component(spatial_scores)
    norm_mutual = normalize_component(mutual_boost)
    
    # Weighted additive fusion
    heuristics = (
        base_weight * norm_inv_dist +
        regret_weight * norm_regret +
        spatial_weight * norm_spatial +
        mutual_weight * norm_mutual
    )
    
    # === 6. MST Reinforcement ===
    # Add MST edges as a small bonus
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            mst_mask = (mst_dense > 0) | (mst_dense.T > 0)
            mst_bonus = 0.15  # Small additive bonus for MST edges
            heuristics[mst_mask] += mst_bonus
        except:
            pass  # Fallback if MST computation fails
    
    # === 7. Adaptive Sparsification ===
    # Dynamic sparsification based on problem size and characteristics
    density_factor = 0.25 + 0.25 * (1.0 - dist_cv)  # Higher variance = denser
    adaptive_k = int(max(3, min(n - 1, density_factor * n)))
    
    # Keep top edges while preserving mutual neighbors
    for i in range(n):
        row = heuristics[i].copy()
        
        if adaptive_k < n - 1:
            # Identify mutual neighbors to preserve
            mutual_mask = np.zeros(n, dtype=bool)
            for j in range(n):
                if i != j and mutual_boost[i, j] > 1.0:
                    mutual_mask[j] = True
            
            # Keep top adaptive_k edges by heuristic value
            sorted_indices = np.argsort(row)[-adaptive_k:]
            keep_mask = np.zeros(n, dtype=bool)
            keep_mask[sorted_indices] = True
            
            # Ensure mutual neighbors are kept
            keep_mask = keep_mask | mutual_mask
            
            # Ensure at least 3 edges are kept
            if np.sum(keep_mask) < 3:
                top_indices = np.argsort(row)[-3:]
                keep_mask[top_indices] = True
            
            # Apply sparsification
            row[~keep_mask] = 0
            heuristics[i] = row
    
    # === 8. Symmetrization and Final Processing ===
    # Symmetrize with preference for stronger mutual decisions
    symmetric_heuristics = np.zeros_like(heuristics)
    for i in range(n):
        for j in range(n):
            if i == j:
                symmetric_heuristics[i, j] = 0
            else:
                # Weighted average favoring edges with mutual boost
                if mutual_boost[i, j] > 1.0 and mutual_boost[j, i] > 1.0:
                    # Strong mutual connection: equal weighting
                    symmetric_heuristics[i, j] = 0.5 * (heuristics[i, j] + heuristics[j, i])
                else:
                    # Asymmetric connection: favor stronger direction
                    max_val = max(heuristics[i, j], heuristics[j, i])
                    min_val = min(heuristics[i, j], heuristics[j, i])
                    symmetric_heuristics[i, j] = 0.7 * max_val + 0.3 * min_val
    
    # Final normalization
    np.fill_diagonal(symmetric_heuristics, 0)
    max_val = np.max(symmetric_heuristics)
    if max_val > epsilon:
        symmetric_heuristics = symmetric_heuristics / max_val
    
    return symmetric_heuristics
