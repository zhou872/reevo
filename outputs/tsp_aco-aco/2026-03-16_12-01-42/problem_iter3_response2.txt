import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with mutual nearest neighbor emphasis
    and adaptive dynamic sparsification. Returns a matrix where higher values
    indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-10
    
    # Stage 1: Base scoring with inverse distance (stability optimized)
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # Stage 2: Mutual Nearest Neighbor Emphasis (core feature)
    k_mnn = max(3, int(np.sqrt(n)))  # Adaptive k for mutual nearest neighbors
    
    # Create boolean masks for nearest neighbors
    nn_masks = []
    for i in range(n):
        # Get k+1 nearest neighbors (including self)
        nearest = np.argpartition(distance_matrix[i], k_mnn + 1)[:k_mnn + 1]
        nearest = nearest[nearest != i]  # Remove self
        if len(nearest) > k_mnn:
            nearest = nearest[:k_mnn]
        
        mask = np.zeros(n, dtype=bool)
        mask[nearest] = True
        nn_masks.append(mask)
    
    # Mutual nearest neighbor mask: both endpoints consider each other among k nearest
    mnn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        for j in range(i + 1, n):
            if nn_masks[i][j] and nn_masks[j][i]:
                mnn_mask[i, j] = mnn_mask[j, i] = True
    
    # Stage 3: Minimum Spanning Tree identification
    mst_mask = np.zeros_like(distance_matrix, dtype=bool)
    if n > 1:
        visited = np.zeros(n, dtype=bool)
        visited[0] = True
        
        for _ in range(n - 1):
            # Find minimum edge connecting visited to unvisited
            visited_idx = np.where(visited)[0]
            unvisited_idx = np.where(~visited)[0]
            
            if len(visited_idx) > 0 and len(unvisited_idx) > 0:
                # Vectorized search for minimum edge
                v_grid, u_grid = np.meshgrid(visited_idx, unvisited_idx, indexing='ij')
                current_dists = distance_matrix[v_grid, u_grid]
                
                if current_dists.size > 0:
                    min_idx = np.argmin(current_dists)
                    min_i = visited_idx[min_idx // len(unvisited_idx)]
                    min_j = unvisited_idx[min_idx % len(unvisited_idx)]
                    
                    mst_mask[min_i, min_j] = mst_mask[min_j, min_i] = True
                    visited[min_j] = True
    
    # Stage 4: Hybrid scoring with weighted factors
    heuristics = inv_dist.copy()
    
    # Strong boost for mutual nearest neighbors
    heuristics[mnn_mask] *= 2.5
    
    # Moderate boost for MST edges
    heuristics[mst_mask] *= 1.8
    
    # Additional boost for edges connecting low-degree nodes in nearest neighbor graph
    nn_degree = np.zeros(n)
    for i in range(n):
        nn_degree[i] = np.sum([nn_masks[j][i] for j in range(n)])
    
    # Normalize degree scores
    if nn_degree.max() > 0:
        degree_factor = 1 - (nn_degree / nn_degree.max())
    else:
        degree_factor = np.ones(n)
    
    # Apply degree balancing factor
    for i in range(n):
        for j in range(n):
            if i != j:
                heuristics[i, j] *= (1 + 0.2 * (degree_factor[i] + degree_factor[j]) / 2)
    
    # Stage 5: Dynamic three-stage sparsification
    # Step 1: Adaptive top-k per node (ensuring feasibility)
    k_edges = min(10, max(3, n // 4))
    row_masks = []
    
    for i in range(n):
        row = heuristics[i].copy()
        # Keep top k_edges, but ensure at least 2 edges remain
        keep_count = max(2, k_edges)
        if keep_count < n - 1:
            threshold = np.partition(row, -keep_count)[-keep_count]
            mask = row >= threshold
        else:
            mask = np.ones(n, dtype=bool)
        mask[i] = False  # Never keep self-edges
        row_masks.append(mask)
    
    # Step 2: Global percentile threshold
    all_values = heuristics[np.triu_indices(n, k=1)]  # Upper triangle without diagonal
    if len(all_values) > 0:
        global_threshold = np.percentile(all_values, 30)  # Keep top 70%
        global_mask = heuristics >= global_threshold
    else:
        global_mask = np.ones_like(heuristics, dtype=bool)
    
    # Step 3: Combine masks and apply sparsification
    final_mask = np.zeros_like(heuristics, dtype=bool)
    for i in range(n):
        for j in range(n):
            if i != j:
                # Keep edge if: (node i considers it in top-k) AND 
                # (node j considers it in top-k OR it passes global threshold)
                if row_masks[i][j] and (row_masks[j][i] or global_mask[i, j]):
                    final_mask[i, j] = True
    
    # Apply final sparsification
    heuristics = heuristics * final_mask
    
    # Ensure symmetry (if edge is kept in one direction, keep in both)
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # Final diagonal zeroing
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
