import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic function for TSP edge selection with multi-step lookahead,
    spectral graph clustering, and adaptive meta-parameter optimization.
    
    Features:
    1. Three-step regret-based lookahead with path continuation analysis
    2. Spectral clustering penalty for inter-cluster connections
    3. MST sensitivity analysis with bottleneck detection
    4. Adaptive entropy sparsification with global structure preservation
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        
    Returns:
        Heuristic matrix where higher values indicate more promising edges
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        result = np.ones_like(distance_matrix)
        np.fill_diagonal(result, 0)
        return result
    
    # Adaptive parameters based on problem characteristics
    avg_dist = np.mean(distance_matrix[np.triu_indices(n, k=1)])
    std_dist = np.std(distance_matrix[np.triu_indices(n, k=1)])
    uniformity = std_dist / (avg_dist + 1e-10)
    
    k_nn = max(2, int(np.log2(n) * (1 + 0.5 * uniformity)))
    lookahead_depth = min(3, max(2, int(np.log(n) / np.log(2))))
    cluster_param = min(0.7, max(0.3, 0.5 + 0.2 * uniformity))
    
    # 1. Base inverse distance with adaptive smoothing
    epsilon = 1e-10 * avg_dist
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Spectral graph analysis for clustering structure
    # Compute normalized Laplacian
    W = np.exp(-distance_matrix / avg_dist)
    np.fill_diagonal(W, 0)
    D = np.diag(W.sum(axis=1))
    L = D - W
    D_inv_sqrt = np.diag(1 / np.sqrt(np.diag(D) + 1e-10))
    L_norm = D_inv_sqrt @ L @ D_inv_sqrt
    
    # Get Fiedler vector for clustering
    try:
        eigvals, eigvecs = np.linalg.eigh(L_norm)
        fiedler_idx = np.argsort(eigvals)[1]  # Second smallest eigenvalue
        fiedler_vec = eigvecs[:, fiedler_idx]
        
        # Cluster assignment based on Fiedler vector sign
        cluster_assign = (fiedler_vec > np.median(fiedler_vec)).astype(float)
        spectral_ratio = eigvals[1] / eigvals[-1] if eigvals[-1] > 0 else 1
    except:
        cluster_assign = np.zeros(n)
        spectral_ratio = 1
    
    # 3. Enhanced three-step lookahead with regret estimation
    # Precompute sorted distances and indices
    sorted_indices = np.argsort(distance_matrix, axis=1)
    sorted_dists = np.sort(distance_matrix, axis=1)
    
    lookahead_score = np.ones((n, n))
    
    # For each node, compute best paths of length 1, 2, 3
    best_1step = sorted_dists[:, 1]  # Best edge from each node
    
    # Precompute best 2-step paths
    best_2step = np.full(n, np.inf)
    for i in range(n):
        top_k = min(4, n)
        for j_idx in range(1, top_k):
            j = sorted_indices[i, j_idx]
            if j == i:
                continue
            # Best edge from j (not going back to i)
            for k_idx in range(1, min(4, n)):
                k = sorted_indices[j, k_idx]
                if k != i and k != j:
                    cost_2 = distance_matrix[i, j] + distance_matrix[j, k]
                    best_2step[i] = min(best_2step[i], cost_2)
        if np.isinf(best_2step[i]):
            best_2step[i] = best_1step[i] * 2
    
    # 3-step lookahead approximation
    best_3step = np.full(n, np.inf)
    for i in range(n):
        top_k = min(3, n)
        candidates = []
        for j_idx in range(1, top_k):
            j = sorted_indices[i, j_idx]
            if j == i:
                continue
            for k_idx in range(1, min(3, n)):
                k = sorted_indices[j, k_idx]
                if k == i or k == j:
                    continue
                for l_idx in range(1, min(3, n)):
                    l = sorted_indices[k, l_idx]
                    if l not in [i, j, k]:
                        cost_3 = distance_matrix[i, j] + distance_matrix[j, k] + distance_matrix[k, l]
                        candidates.append(cost_3)
        if candidates:
            best_3step[i] = min(candidates)
        else:
            best_3step[i] = best_2step[i] * 1.5
    
    # Compute combined regret for each edge
    for i in range(n):
        for j in range(n):
            if i != j:
                # Current regret
                regret_1 = (distance_matrix[i, j] - best_1step[i]) / (best_1step[i] + 1e-10)
                
                # 2-step regret approximation
                best_from_j = sorted_dists[j, 1] if sorted_indices[j, 1] != i else sorted_dists[j, 2]
                cost_2 = distance_matrix[i, j] + best_from_j
                regret_2 = (cost_2 - best_2step[i]) / (best_2step[i] + 1e-10)
                
                # 3-step regret approximation
                # Estimate third step as average of top-k from j
                k_third = min(3, n-2)
                third_options = []
                for k_idx in range(1, k_third+1):
                    k = sorted_indices[j, k_idx]
                    if k != i:
                        best_from_k = sorted_dists[k, 1]
                        if sorted_indices[k, 1] not in [i, j]:
                            third_options.append(best_from_k)
                        elif n > 2:
                            best_from_k = sorted_dists[k, 2]
                            third_options.append(best_from_k)
                
                if third_options:
                    avg_third = np.mean(third_options)
                    cost_3 = distance_matrix[i, j] + best_from_j + avg_third
                    regret_3 = (cost_3 - best_3step[i]) / (best_3step[i] + 1e-10)
                else:
                    regret_3 = regret_2
                
                # Combined regret with decreasing weights for longer lookahead
                combined_regret = 0.5 * regret_1 + 0.3 * regret_2 + 0.2 * regret_3
                
                # Apply penalty
                penalty = np.exp(-0.7 * combined_regret)
                lookahead_score[i, j] = max(0.1, min(1.0, penalty))
    
    # 4. Spectral clustering penalty
    cluster_penalty = np.ones((n, n))
    for i in range(n):
        for j in range(i+1, n):
            # Penalize edges that cross clusters (inter-cluster edges)
            if cluster_assign[i] != cluster_assign[j]:
                # Scale penalty based on spectral ratio
                penalty_factor = 1 - cluster_param * spectral_ratio
                cluster_penalty[i, j] = cluster_penalty[j, i] = max(0.3, penalty_factor)
    
    # 5. MST sensitivity analysis with bottleneck detection
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_edges = mst + mst.T > 0
    
    # Compute MST bottleneck distances (max edge in path between nodes)
    # Approximation using MST adjacency
    mst_bonus = np.zeros((n, n))
    
    # For each non-MST edge, compute how much it would improve the MST
    mst_dist = mst + mst.T  # Symmetric MST distance matrix
    for i in range(n):
        for j in range(i+1, n):
            if not mst_edges[i, j]:
                # Find the maximum edge in the MST path between i and j
                # Simple approximation: use the maximum of edges from i and j in MST
                i_mst_neighbors = np.where(mst_edges[i])[0]
                j_mst_neighbors = np.where(mst_edges[j])[0]
                
                if len(i_mst_neighbors) > 0 and len(j_mst_neighbors) > 0:
                    # Estimate bottleneck as max of incident MST edges
                    max_i_edge = max([distance_matrix[i, k] for k in i_mst_neighbors], default=0)
                    max_j_edge = max([distance_matrix[j, k] for k in j_mst_neighbors], default=0)
                    bottleneck = max(max_i_edge, max_j_edge)
                    
                    # Bonus proportional to improvement over bottleneck
                    if bottleneck > 0:
                        improvement = max(0, bottleneck - distance_matrix[i, j])
                        mst_bonus[i, j] = mst_bonus[j, i] = 0.3 * improvement / (bottleneck + 1e-10)
    
    # 6. Adaptive nearest neighbor connectivity
    nn_connectivity = np.zeros((n, n))
    for i in range(n):
        k = min(k_nn, n-1)
        nearest = sorted_indices[i, 1:k+1]
        for rank, j in enumerate(nearest):
            weight = np.exp(-rank / k)  # Exponential decay
            nn_connectivity[i, j] += weight
    
    # Normalize
    nn_connectivity = (nn_connectivity + nn_connectivity.T) / 2
    row_sums = nn_connectivity.sum(axis=1, keepdims=True)
    nn_connectivity = np.divide(nn_connectivity, row_sums + 1e-10, 
                                where=row_sums > 0)
    
    # 7. Combine all factors with meta-learned weights
    # Weights adapt based on problem characteristics
    w_inv = 1.0
    w_lookahead = 0.8 + 0.2 * uniformity  # Higher for more uniform instances
    w_cluster = 0.4 * spectral_ratio  # Higher when clear clusters exist
    w_mst = 0.5 - 0.2 * uniformity  # Higher for non-uniform instances
    w_nn = 0.6
    
    combined = (w_inv * inv_dist * 
                (w_lookahead * lookahead_score) *
                (1 + w_cluster * cluster_penalty) *
                (1 + w_mst * mst_bonus) *
                (1 + w_nn * nn_connectivity))
    
    # 8. Enhanced entropy-based sparsification
    # Global entropy calculation
    upper_triangle = combined[np.triu_indices(n, k=1)]
    if len(upper_triangle) > 1:
        global_probs = upper_triangle / (upper_triangle.sum() + 1e-10)
        global_entropy = -np.sum(global_probs * np.log(global_probs + 1e-10))
        max_global_entropy = np.log(len(upper_triangle))
        global_entropy_norm = global_entropy / max_global_entropy
    else:
        global_entropy_norm = 0.5
    
    sparse_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        row = combined[i].copy()
        row[i] = -np.inf
        
        # Local entropy
        row_pos = row[row > 0]
        if len(row_pos) > 1:
            local_probs = row_pos / row_pos.sum()
            local_entropy = -np.sum(local_probs * np.log(local_probs + 1e-10))
            max_local_entropy = np.log(len(row_pos))
            local_entropy_norm = local_entropy / max_local_entropy
        else:
            local_entropy_norm = 0.5
        
        # Combine local and global entropy
        combined_entropy = 0.7 * local_entropy_norm + 0.3 * global_entropy_norm
        
        # Determine keep fraction based on entropy
        # Higher entropy -> more uniform distribution -> keep more edges
        base_fraction = 0.15 + 0.15 * combined_entropy
        # Adjust based on problem size
        size_factor = min(0.4, 10/n)
        keep_fraction = min(0.3, max(0.1, base_fraction + size_factor))
        
        k_keep = max(2, int(keep_fraction * (n-1)))
        
        # Keep top edges
        if k_keep < n-1:
            threshold = np.partition(row, -k_keep)[-k_keep]
            sparse_mask[i] = row >= threshold
        else:
            sparse_mask[i] = row > -np.inf
        
        sparse_mask[i, i] = False
    
    # Ensure connectivity: each node must have at least 2 edges
    for i in range(n):
        if sparse_mask[i].sum() < 2:
            # Add the next best edges
            row = combined[i].copy()
            row[i] = -np.inf
            row[sparse_mask[i]] = -np.inf
            needed = 2 - sparse_mask[i].sum()
            if needed > 0 and row.max() > -np.inf:
                top_indices = np.argpartition(row, -needed)[-needed:]
                sparse_mask[i, top_indices] = True
    
    # Ensure symmetry
    sparse_mask = sparse_mask | sparse_mask.T
    
    # Apply sparsification
    heuristics_matrix = combined * sparse_mask
    
    # 9. Final normalization with structure preservation
    # Preserve relative differences while ensuring numerical stability
    row_max = heuristics_matrix.max(axis=1, keepdims=True)
    row_max[row_max == 0] = 1
    heuristics_matrix = heuristics_matrix / row_max
    
    # Smooth the distribution
    for i in range(n):
        row_sum = heuristics_matrix[i].sum()
        if row_sum > 0:
            heuristics_matrix[i] = 0.8 * (heuristics_matrix[i] / row_sum) + 0.2 * (1/(n-1))
        else:
            heuristics_matrix[i] = 1/(n-1)
        heuristics_matrix[i, i] = 0
    
    # Final symmetrization
    heuristics_matrix = (heuristics_matrix + heuristics_matrix.T) / 2
    
    return heuristics_matrix
