import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with dynamic regret weighting and 
    connectivity-aware sparsification.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Dynamic Regret Weighting ===
    # Adjust regret computation based on local node density
    regret_scores = np.ones_like(distance_matrix)
    
    for i in range(n):
        # Compute local density around node i
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        k_density = min(10, n - 1)
        local_dists = np.partition(dists_i, k_density)[:k_density]
        density_i = np.mean(local_dists)
        
        # Adaptive number of alternatives based on local density
        # Denser regions consider more alternatives
        k_regret = min(max(3, int(8 * (1 - density_i / np.max(local_dists)))), n - 1)
        
        # Get distances to alternatives
        dist_i = dists_i.copy()
        sorted_indices = np.argsort(dist_i)[:k_regret + 1]  # +1 to ensure we have enough
        
        for j in range(n):
            if i == j:
                continue
                
            d_ij = distance_matrix[i, j]
            
            # Find best alternative excluding j
            alternatives = []
            for idx in sorted_indices:
                if idx != j and idx != i:
                    alternatives.append(distance_matrix[i, idx])
                    if len(alternatives) >= k_regret:
                        break
            
            if alternatives:
                best_alt = min(alternatives)
                # Dynamic regret: emphasize edges where alternative cost gap is significant
                # relative to local average distances
                regret_ratio = (best_alt - d_ij) / (d_ij + epsilon)
                # Scale by local density: stronger effect in dense regions
                density_factor = 1.0 + 0.5 * (1.0 - density_i / (np.max(distance_matrix) + epsilon))
                regret_scores[i, j] = 1.0 + 1.5 * max(0, regret_ratio) * density_factor
            else:
                regret_scores[i, j] = 1.0
    
    # === 2. Local Neighborhood Quality Scoring ===
    # Evaluate edges based on quality of local neighborhoods
    local_quality = np.ones_like(distance_matrix)
    
    # Precompute neighborhood statistics
    k_neighborhood = min(6, n - 1)
    neighborhood_stats = np.zeros((n, 2))  # [mean, std] of nearest distances
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        nearest = np.partition(dists, k_neighborhood)[:k_neighborhood]
        neighborhood_stats[i, 0] = np.mean(nearest)
        neighborhood_stats[i, 1] = np.std(nearest) + epsilon
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            d_ij = distance_matrix[i, j]
            
            # Score based on how well edge fits local neighborhoods
            # Good edges connect similar-quality neighborhoods
            mean_i, std_i = neighborhood_stats[i]
            mean_j, std_j = neighborhood_stats[j]
            
            # Edge should be close to average of neighborhood means
            target_mean = (mean_i + mean_j) / 2
            mean_score = 1.0 - 0.3 * min(1.0, abs(d_ij - target_mean) / (target_mean + epsilon))
            
            # Penalize edges connecting nodes with very different neighborhood characteristics
            similarity = 1.0 - 0.2 * min(1.0, abs(mean_i - mean_j) / (np.max([mean_i, mean_j]) + epsilon))
            
            # Combine factors
            local_quality[i, j] = mean_score * similarity
    
    # === 3. Base Inverse Distance with Adaptive Scaling ===
    base_scores = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(base_scores, 0)
    
    # Scale base scores by global statistics
    if n > 1:
        # Use median for robustness against outliers
        median_dist = np.median(distance_matrix[np.triu_indices(n, k=1)])
        scale_factor = 1.0 / (median_dist + epsilon)
        base_scores *= scale_factor
    
    # === 4. Mutual Connectivity Boost ===
    connectivity_boost = np.ones_like(distance_matrix)
    
    # Find mutual nearest neighbors at different scales
    for k in [2, 4, 8]:
        if k >= n:
            continue
            
        nn_masks = np.zeros((n, n), dtype=bool)
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest_indices = np.argpartition(dists, k)[:k]
            nn_masks[i, nearest_indices] = True
        
        # Boost mutual nearest neighbors
        mutual_mask = nn_masks & nn_masks.T
        boost_value = 1.0 + 0.4 / np.log2(k + 1)  # Stronger boost for smaller k
        connectivity_boost[mutual_mask] *= boost_value
    
    # === 5. Combined Scoring ===
    # Multiplicative combination with dynamic weighting
    heuristics = base_scores.copy()
    heuristics *= regret_scores
    heuristics *= local_quality
    heuristics *= connectivity_boost
    
    # === 6. Connectivity-Aware Sparsification ===
    # Ensure each node remains sufficiently connected
    min_edges_per_node = max(2, min(5, int(np.sqrt(n))))
    
    for i in range(n):
        row = heuristics[i].copy()
        
        # Always keep strongest edges
        sorted_indices = np.argsort(row)[::-1]
        keep_mask = np.zeros(n, dtype=bool)
        
        # Keep top edges based on both score and regret
        top_by_score = sorted_indices[:min_edges_per_node]
        keep_mask[top_by_score] = True
        
        # Also keep edges with high regret scores
        regret_row = regret_scores[i].copy()
        regret_row[i] = 0
        regret_threshold = np.percentile(regret_row[regret_row > 0], 75)
        high_regret = np.where(regret_row >= regret_threshold)[0]
        keep_mask[high_regret] = True
        
        # Ensure connectivity to diverse parts of graph
        # Keep edges to nodes with different neighborhood characteristics
        if np.sum(keep_mask) < min_edges_per_node * 2:
            # Add edges to nodes with most different neighborhood stats
            current_mean = neighborhood_stats[i, 0]
            diff_means = np.abs(neighborhood_stats[:, 0] - current_mean)
            diff_means[i] = -np.inf
            diverse_indices = np.argsort(diff_means)[-3:]
            keep_mask[diverse_indices] = True
        
        # Apply sparsification
        row[~keep_mask] *= 0.1  # Drastically reduce rather than zeroing
        heuristics[i] = row
    
    # === 7. Symmetry Enforcement with Preference for Consistency ===
    symmetric_heuristics = (heuristics + heuristics.T) / 2
    
    # Boost edges where both directions agree on high value
    agreement_mask = (heuristics > np.percentile(heuristics, 75)) & \
                     (heuristics.T > np.percentile(heuristics, 75))
    symmetric_heuristics[agreement_mask] *= 1.2
    
    # === 8. Final Processing ===
    np.fill_diagonal(symmetric_heuristics, 0)
    
    # Ensure non-negativity
    symmetric_heuristics = np.maximum(symmetric_heuristics, 0)
    
    # Soft normalization to preserve relative differences
    max_val = np.max(symmetric_heuristics)
    if max_val > 0:
        # Use softmax-like scaling to maintain distribution
        symmetric_heuristics = symmetric_heuristics / (max_val + epsilon)
        # Apply mild compression to reduce extreme values
        symmetric_heuristics = np.power(symmetric_heuristics, 0.9)
    
    return symmetric_heuristics
