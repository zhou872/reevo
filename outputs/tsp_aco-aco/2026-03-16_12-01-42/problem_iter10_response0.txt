import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.cluster import KMeans
from scipy.spatial.distance import cdist

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge prioritization with multi-step bidirectional path analysis,
    explicit cluster-aware scoring, and adaptive parameter optimization.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base Components ===
    # Inverse distance with stability
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # Global statistics for adaptive parameters
    if n > 1:
        off_diag_mask = ~np.eye(n, dtype=bool)
        global_dists = distance_matrix[off_diag_mask]
        mean_dist = np.mean(global_dists)
        std_dist = np.std(global_dists) + epsilon
        cv = std_dist / mean_dist
    else:
        mean_dist = 1.0
        std_dist = 1.0
        cv = 1.0
    
    # === 2. Multi-step Bidirectional Path Analysis ===
    path_scores = np.ones_like(distance_matrix)
    
    if n > 3:
        # Adaptive exploration depth based on problem size
        max_depth = min(4, max(2, int(np.log2(n))))
        k_neighbors = max(3, min(15, n // 4))
        
        # Precompute nearest neighbors for all nodes
        neighbor_indices = np.zeros((n, k_neighbors), dtype=int)
        neighbor_dists = np.zeros((n, k_neighbors))
        
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            if len(dists) >= k_neighbors:
                idx = np.argpartition(dists, k_neighbors)[:k_neighbors]
                neighbor_indices[i] = idx
                neighbor_dists[i] = dists[idx]
        
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                direct_dist = distance_matrix[i, j]
                if direct_dist == 0:
                    continue
                
                # Bidirectional path exploration
                best_alt = np.inf
                
                # Explore paths from i to j
                for depth in range(2, max_depth + 1):
                    # Generate candidate paths using bidirectional BFS-like approach
                    if depth == 2:
                        # Two-step paths: i -> k -> j
                        for k in neighbor_indices[i]:
                            if k != j:
                                alt = distance_matrix[i, k] + distance_matrix[k, j]
                                best_alt = min(best_alt, alt)
                    elif depth == 3:
                        # Three-step paths: i -> k -> l -> j
                        for k in neighbor_indices[i]:
                            for l in neighbor_indices[j]:
                                if k != l and k != j and l != i:
                                    alt = (distance_matrix[i, k] + 
                                          distance_matrix[k, l] + 
                                          distance_matrix[l, j])
                                    best_alt = min(best_alt, alt)
                    elif depth == 4:
                        # Four-step paths with limited branching
                        for k in neighbor_indices[i]:
                            if k == j:
                                continue
                            # Get k's nearest neighbors
                            k_nearest = neighbor_indices[k][:min(5, k_neighbors)]
                            for l in k_nearest:
                                if l == i or l == j:
                                    continue
                                for m in neighbor_indices[j]:
                                    if m == i or m == k or m == l:
                                        continue
                                    alt = (distance_matrix[i, k] + 
                                          distance_matrix[k, l] +
                                          distance_matrix[l, m] + 
                                          distance_matrix[m, j])
                                    best_alt = min(best_alt, alt)
                
                if best_alt < np.inf and best_alt > 0:
                    # Score based on path efficiency ratio
                    efficiency = direct_dist / (best_alt + epsilon)
                    # Exponential decay for highly efficient edges
                    if efficiency < 0.8:  # Direct path is at least 20% better
                        path_scores[i, j] = 1.0 + 0.6 * np.exp(-3 * efficiency)
                    elif efficiency < 1.2:
                        path_scores[i, j] = 1.0 + 0.2 * (1.2 - efficiency)
    
    # === 3. Cluster-Aware Edge Promotion ===
    cluster_scores = np.ones_like(distance_matrix)
    
    if n >= 4:
        # Determine optimal number of clusters
        k_clusters = max(2, min(8, int(np.sqrt(n))))
        
        try:
            # Use coordinates derived from distance matrix via MDS approximation
            # Simple approach: use first few nodes as cluster centers
            if n > 10:
                # Sample-based clustering for efficiency
                sample_size = min(50, n)
                indices = np.random.choice(n, sample_size, replace=False)
                sample_dists = distance_matrix[np.ix_(indices, indices)]
                
                # Perform k-means on sampled distances
                kmeans = KMeans(n_clusters=k_clusters, random_state=42, n_init=10)
                labels_sample = kmeans.fit_predict(sample_dists)
                
                # Assign all nodes to nearest cluster center
                centers = kmeans.cluster_centers_
                all_dists_to_centers = cdist(distance_matrix.mean(axis=1).reshape(-1, 1), centers)
                cluster_labels = np.argmin(all_dists_to_centers, axis=1)
            else:
                # Full clustering for small instances
                kmeans = KMeans(n_clusters=k_clusters, random_state=42, n_init=10)
                cluster_labels = kmeans.fit_predict(distance_matrix)
            
            # Boost intra-cluster edges, penalize inter-cluster based on distance
            for i in range(n):
                for j in range(n):
                    if i == j:
                        continue
                    
                    if cluster_labels[i] == cluster_labels[j]:
                        # Same cluster: promote based on relative position
                        cluster_size = np.sum(cluster_labels == cluster_labels[i])
                        # Normalize by cluster size - smaller clusters get stronger boost
                        boost = 1.5 - 0.5 * min(1.0, cluster_size / max(5, n / k_clusters))
                        cluster_scores[i, j] = boost
                    else:
                        # Different clusters: penalize unless very close
                        # Check if this is a natural bridge between clusters
                        d_ij = distance_matrix[i, j]
                        # Compare to average distance between these clusters
                        mask_i = cluster_labels == cluster_labels[i]
                        mask_j = cluster_labels == cluster_labels[j]
                        inter_distances = distance_matrix[mask_i][:, mask_j]
                        if inter_distances.size > 0:
                            avg_inter = np.mean(inter_distances)
                            if d_ij < 0.7 * avg_inter:  # Good bridge edge
                                cluster_scores[i, j] = 1.2
                            else:
                                cluster_scores[i, j] = 0.8
        except:
            # Fallback to density-based clustering
            for i in range(n):
                # Local density estimate
                k_density = max(3, min(10, n // 5))
                dists = distance_matrix[i].copy()
                dists[i] = np.inf
                nearest = np.partition(dists, k_density)[:k_density]
                local_density = np.mean(nearest)
                
                for j in range(n):
                    if i == j:
                        continue
                    
                    d_ij = distance_matrix[i, j]
                    # Score based on relative density
                    if d_ij < 0.5 * local_density:
                        cluster_scores[i, j] = 1.3
                    elif d_ij < local_density:
                        cluster_scores[i, j] = 1.1
    
    # === 4. Enhanced Regret with Adaptive Order ===
    regret_matrix = np.ones_like(distance_matrix)
    
    if n > 2:
        # Dynamic regret order based on graph properties
        if cv < 0.3:
            regret_order = 2
            regret_weight = 0.8
        elif cv < 0.7:
            regret_order = 3
            regret_weight = 1.0
        else:
            regret_order = min(5, max(3, n // 10))
            regret_weight = 1.2
        
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            if len(dists) >= regret_order:
                # Find regret_order-th best distance
                ref_idx = np.argpartition(dists, regret_order-1)[regret_order-1]
                ref_dist = dists[ref_idx]
                
                if ref_dist > 0:
                    for j in range(n):
                        if i == j:
                            continue
                        
                        d_ij = distance_matrix[i, j]
                        if d_ij < ref_dist:
                            regret_value = (ref_dist - d_ij) / (ref_dist + epsilon)
                            # Apply adaptive weighting
                            regret_matrix[i, j] = 1.0 + regret_weight * np.tanh(regret_value)
    
    # === 5. Dynamic Parameter Optimization ===
    # Compute additional graph metrics for parameter tuning
    if n > 2:
        # Graph sparsity measure
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_length = np.sum(mst[mst > 0])
        graph_density = mst_length / (mean_dist * (n-1) + epsilon)
        
        # Average local clustering coefficient approximation
        avg_clustering = 0
        for i in range(min(20, n)):  # Sample for efficiency
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            if len(dists) >= 3:
                nearest = np.partition(dists, 3)[:3]
                triangle_possible = np.sum(nearest) / (3 * mean_dist + epsilon)
                avg_clustering += min(1.0, triangle_possible)
        avg_clustering = avg_clustering / min(20, n)
    
        # Set adaptive weights based on multiple metrics
        if n < 15:
            w_inv = 1.2
            w_path = 0.9
            w_cluster = 1.1
            w_regret = 1.0
        elif graph_density < 0.3:
            w_inv = 1.0
            w_path = 1.3  # Emphasize path analysis in sparse graphs
            w_cluster = 0.8
            w_regret = 1.1
        elif avg_clustering > 0.6:
            w_inv = 0.9
            w_path = 1.0
            w_cluster = 1.4  # Emphasize clustering in dense graphs
            w_regret = 0.9
        else:
            w_inv = 1.0
            w_path = 1.1
            w_cluster = 1.0
            w_regret = 1.1
    else:
        w_inv = w_path = w_cluster = w_regret = 1.0
    
    # === 6. Multiplicative Fusion ===
    heuristics = (inv_dist ** w_inv) * \
                 (path_scores ** w_path) * \
                 (cluster_scores ** w_cluster) * \
                 (regret_matrix ** w_regret)
    
    # === 7. MST Reinforcement with Mutual Agreement ===
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            mst_present = (mst_dense > 0) | (mst_dense.T > 0)
            
            # Boost MST edges based on mutual agreement
            for i in range(n):
                for j in range(n):
                    if i != j and mst_present[i, j]:
                        # Check if edge is also promising by our heuristics
                        edge_score = heuristics[i, j]
                        row_mean = np.mean(heuristics[i, ~np.eye(n, dtype=bool)[i]])
                        col_mean = np.mean(heuristics[j, ~np.eye(n, dtype=bool)[j]])
                        
                        if edge_score > max(row_mean, col_mean):
                            heuristics[i, j] *= 1.6  # Strong boost for agreed edges
                        else:
                            heuristics[i, j] *= 1.3  # Moderate boost
        except:
            pass
    
    # === 8. Symmetrization with Mutual Agreement Weighting ===
    heuristics_sym = np.zeros_like(heuristics)
    
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            
            # Weighted symmetrization favoring mutual agreement
            h_ij = heuristics[i, j]
            h_ji = heuristics[j, i]
            
            if h_ij > 0 and h_ji > 0:
                # Geometric mean for multiplicative consistency
                base = np.sqrt(h_ij * h_ji)
                # Additional boost for strong mutual agreement
                agreement = min(h_ij, h_ji) / (max(h_ij, h_ji) + epsilon)
                if agreement > 0.7:
                    boost = 1.0 + 0.3 * (agreement - 0.7) / 0.3
                else:
                    boost = 1.0
                heuristics_sym[i, j] = base * boost
            else:
                heuristics_sym[i, j] = max(h_ij, h_ji)
    
    # === 9. Final Processing ===
    np.fill_diagonal(heuristics_sym, 0)
    
    # Adaptive thresholding for sparsity
    if n > 3:
        for i in range(n):
            row = heuristics_sym[i].copy()
            non_zero = row[row > 0]
            if len(non_zero) > 2:
                # Keep top edges based on percentiles
                keep_ratio = max(0.3, min(0.7, 0.6 - 0.01 * n))  # More selective for larger n
                threshold = np.percentile(non_zero, 100 * (1 - keep_ratio))
                mask = row >= threshold
                
                # Ensure minimum connectivity
                if np.sum(mask) < 2:
                    top2 = np.argsort(row)[-2:]
                    mask[top2] = True
                
                heuristics_sym[i, ~mask] = 0
    
    # Normalize to [0, 1]
    max_val = np.max(heuristics_sym)
    if max_val > 0:
        heuristics_sym = heuristics_sym / max_val
    
    return heuristics_sym
