import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with adaptive neighborhood optimization,
    harmonic symmetry fusion, and meta-heuristic weight adaptation.
    
    Parameters
    ----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns
    -------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base proximity scoring ===
    # Use inverse distance with adaptive scaling
    dist_no_diag = distance_matrix.copy()
    np.fill_diagonal(dist_no_diag, np.inf)
    min_dist_per_node = np.min(dist_no_diag, axis=1)
    inv_dist = 1 / (distance_matrix + epsilon * min_dist_per_node[:, np.newaxis])
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Adaptive neighborhood size determination ===
    # Compute local density using median nearest neighbor distance
    if n > 2:
        k_median = min(5, n - 1)
        sorted_dists = np.sort(dist_no_diag, axis=1)[:, :k_median]
        median_nn_dist = np.median(sorted_dists, axis=1)
        global_median = np.median(median_nn_dist)
        
        # Neighborhood size inversely proportional to local density
        base_k = int(np.clip(3 + np.sqrt(n), 4, 15))
        density_ratio = median_nn_dist / (global_median + epsilon)
        k_per_node = np.clip(base_k * (2.0 - np.exp(-density_ratio)), 3, n - 1).astype(int)
    else:
        k_per_node = np.array([n - 1] * n)
    
    # === 3. Adaptive regret scoring ===
    regret_scores = np.zeros_like(distance_matrix)
    
    for i in range(n):
        if k_per_node[i] > 0:
            # Get k nearest neighbors
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            nearest_idx = np.argpartition(dists_i, k_per_node[i])[:k_per_node[i]]
            nearest_dists = dists_i[nearest_idx]
            
            if len(nearest_dists) >= 2:
                # Get two smallest distances among neighbors
                sorted_nearest = np.sort(nearest_dists)
                first_min = sorted_nearest[0]
                second_min = sorted_nearest[1]
                
                # Compute regret for each neighbor
                for j_idx, j in enumerate(nearest_idx):
                    d_ij = distance_matrix[i, j]
                    if d_ij > 0:
                        # Alternative distance is second best if j is the best,
                        # otherwise it's the best distance
                        best_alt = second_min if np.isclose(d_ij, first_min) else first_min
                        regret = (best_alt - d_ij) / (d_ij + epsilon)
                        
                        # Adaptive sigmoid transformation
                        regret_scale = 2.0 / (1.0 + np.exp(-regret)) - 1.0
                        regret_scores[i, j] = max(0, regret_scale)
    
    # === 4. Two-step path efficiency with adaptive lookahead ===
    path_efficiency = np.ones_like(distance_matrix)
    
    for i in range(n):
        if k_per_node[i] > 1:
            # Get i's neighbors
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            neighbors_i = np.argpartition(dists_i, k_per_node[i])[:k_per_node[i]]
            
            for j in neighbors_i:
                if i == j:
                    continue
                    
                d_ij = distance_matrix[i, j]
                
                # Explore intermediate nodes in intersection of neighborhoods
                if k_per_node[j] > 1:
                    dists_j = distance_matrix[j].copy()
                    dists_j[j] = np.inf
                    neighbors_j = np.argpartition(dists_j, k_per_node[j])[:k_per_node[j]]
                    
                    # Consider common neighbors for two-step paths
                    common_neighbors = set(neighbors_i).intersection(set(neighbors_j))
                    if i in common_neighbors:
                        common_neighbors.remove(i)
                    if j in common_neighbors:
                        common_neighbors.remove(j)
                    
                    if common_neighbors:
                        best_two_step = np.inf
                        for k in common_neighbors:
                            two_step = distance_matrix[i, k] + distance_matrix[k, j]
                            if two_step < best_two_step:
                                best_two_step = two_step
                        
                        if best_two_step < np.inf:
                            efficiency_ratio = d_ij / (best_two_step + epsilon)
                            # Exponential penalty for inefficient direct edges
                            path_efficiency[i, j] = np.exp(-max(0, efficiency_ratio - 1.0))
                    else:
                        # No common neighbors - penalize slightly
                        path_efficiency[i, j] = 0.8
    
    # === 5. Harmonic density bonus ===
    density_bonus = np.ones_like(distance_matrix)
    
    for i in range(n):
        if k_per_node[i] > 1:
            # Compute local harmonic mean of distances
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            nearest_idx = np.argpartition(dists_i, k_per_node[i])[:k_per_node[i]]
            nearest_dists = dists_i[nearest_idx]
            
            harmonic_mean = len(nearest_dists) / np.sum(1 / (nearest_dists + epsilon))
            
            # Apply bonus based on how much closer than harmonic mean
            for j in nearest_idx:
                d_ij = distance_matrix[i, j]
                if d_ij > 0:
                    ratio = harmonic_mean / (d_ij + epsilon)
                    # Boost edges significantly closer than local average
                    density_bonus[i, j] = 1.0 + 0.5 * np.tanh(ratio - 1.0)
    
    # === 6. Meta-heuristic weight adaptation ===
    # Adjust component weights based on problem characteristics
    
    # Compute problem statistics
    if n > 1:
        triu_indices = np.triu_indices(n, k=1)
        off_diag_distances = distance_matrix[triu_indices]
        dist_mean = np.mean(off_diag_distances)
        dist_cv = np.std(off_diag_distances) / (dist_mean + epsilon)  # Coefficient of variation
    else:
        dist_cv = 1.0
    
    # Dynamic weights based on problem size and distance distribution
    if n <= 10:
        w_inv = 1.3
        w_regret = 1.2
        w_path = 0.8
        w_density = 0.9
    elif n <= 50:
        # Adjust based on distance variation
        w_inv = 1.0 + 0.3 * np.tanh(1.0 - dist_cv)
        w_regret = 1.1 + 0.2 * np.tanh(dist_cv - 0.5)
        w_path = 0.9
        w_density = 1.0
    else:
        w_inv = 0.9
        w_regret = 1.0
        w_path = 1.1
        w_density = 1.2
    
    # === 7. Composite scoring with weighted geometric mean ===
    # Use geometric mean for more balanced combination
    composite = (inv_dist ** w_inv) * \
                ((1.0 + regret_scores) ** w_regret) * \
                (path_efficiency ** w_path) * \
                (density_bonus ** w_density)
    
    np.fill_diagonal(composite, 0)
    
    # === 8. MST reinforcement ===
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            mst_mask = (mst_dense > 0) | (mst_dense.T > 0)
            
            # Strong reinforcement for MST edges, especially if they score well
            mst_boost = np.where(mst_mask, 1.5 + 0.5 * composite / (np.max(composite) + epsilon), 1.0)
            composite *= mst_boost
        except:
            pass
    
    # === 9. Harmonic symmetrization ===
    # Use harmonic mean for symmetry (more robust to outliers)
    composite_harmonic = 2 * composite * composite.T / (composite + composite.T + epsilon)
    np.fill_diagonal(composite_harmonic, 0)
    
    # === 10. Adaptive sparsification ===
    # Keep top edges per node, ensuring connectivity
    for i in range(n):
        row = composite_harmonic[i].copy()
        
        # Determine number of edges to keep
        keep_count = max(3, min(10, int(np.sqrt(n) + 2)))
        
        if keep_count < n - 1:
            # Keep top edges
            threshold = np.partition(row, -keep_count)[-keep_count]
            keep_mask = row >= threshold
            
            # Ensure MST edges are kept if they exist
            if n > 2:
                try:
                    mst_i = minimum_spanning_tree(distance_matrix).toarray()[i] > 0
                    keep_mask = keep_mask | mst_i
                except:
                    pass
            
            # Ensure at least 2 edges are kept
            if np.sum(keep_mask) < 2:
                top2_idx = np.argsort(row)[-2:]
                keep_mask[top2_idx] = True
            
            row[~keep_mask] = 0
        
        composite_harmonic[i] = row
    
    # === 11. Final normalization ===
    # Ensure symmetry after sparsification
    final_heuristic = (composite_harmonic + composite_harmonic.T) / 2
    np.fill_diagonal(final_heuristic, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(final_heuristic)
    if max_val > 0:
        final_heuristic = final_heuristic / max_val
    
    # Apply softmax-style transformation for better contrast
    final_heuristic = np.exp(3.0 * final_heuristic) / np.sum(np.exp(3.0 * final_heuristic))
    
    return final_heuristic
