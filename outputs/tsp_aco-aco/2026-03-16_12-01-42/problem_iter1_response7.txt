import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    n = distance_matrix.shape[0]
    # set k to be 10% of n, but at least 3 and at most n-1
    k = max(3, int(0.1 * n))
    if k >= n:
        k = n - 1

    # 1. Inverse distance component
    inv_dist = 1 / (distance_matrix + 1e-10)
    np.fill_diagonal(inv_dist, 0)
    if inv_dist.max() - inv_dist.min() > 1e-10:
        inv_dist_norm = (inv_dist - inv_dist.min()) / (inv_dist.max() - inv_dist.min())
    else:
        inv_dist_norm = np.ones_like(inv_dist) * 0.5

    # 2. Savings heuristic component (Clarke-Wright)
    ref_node = 0  # arbitrary reference node
    savings = distance_matrix[:, ref_node][:, None] + distance_matrix[ref_node, :][None, :] - distance_matrix
    np.fill_diagonal(savings, 0)
    if savings.max() - savings.min() > 1e-10:
        savings_norm = (savings - savings.min()) / (savings.max() - savings.min())
    else:
        savings_norm = np.ones_like(savings) * 0.5

    # 3. Mutual nearest neighbor bonus
    is_nearest = np.zeros((n, n), dtype=bool)
    for i in range(n):
        dist = distance_matrix[i].copy()
        dist[i] = np.inf  # ignore self
        # get the k smallest distances indices
        idx = np.argsort(dist)[:k]
        is_nearest[i, idx] = True

    mutual = is_nearest & is_nearest.T
    mutual_bonus = mutual.astype(float)

    # Combine the three components with equal weight
    priority = inv_dist_norm + savings_norm + mutual_bonus

    # Sparsify: keep only edges that are in the top-k nearest neighbors of at least one endpoint
    keep_mask = is_nearest | is_nearest.T
    priority = priority * keep_mask

    # Ensure diagonal is zero
    np.fill_diagonal(priority, 0)

    return priority
