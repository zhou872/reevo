```python
import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors:
    1. Sparsification via k-nearest neighbor masking
    2. Minimum spanning tree connectivity bonus
    3. Distance normalization by local neighborhood density
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # 1. Base inverse distance with regularization
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Sparsification: keep only edges that are in k-nearest neighbors
    k = max(2, min(10, n // 4))  # Adaptive k based on problem size
    mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        nearest = np.argsort(distance_matrix[i])[1:k+1]
        mask[i, nearest] = True
    
    # Make mask symmetric (if j is in i's neighbors, include both directions)
    symmetric_mask = mask | mask.T
    
    # 3. Minimum Spanning Tree connectivity bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)  # Undirected MST edges
    
    # 4. Local density normalization
    local_density = np.zeros(n)
    for i in range(n):
        # Average distance to k nearest neighbors as density measure
        nearest_dists = np.sort(distance_matrix[i])[1:k+1]
        local_density[i] = np.mean(nearest_dists) + epsilon
    
    # Compute density-normalized scores
    density_matrix = np.sqrt(np.outer(local_density, local_density))
    normalized_inv_dist = inv_dist * density_matrix
    
    # 5. Combine all factors
    heuristic = np.zeros_like(distance_matrix)
    
    # Base score for promising edges
    base_score = normalized_inv_dist * symmetric_mask
    
    # MST bonus: double the score for MST edges
    mst_bonus = 2.0
    heuristic = base_score * (1 + (mst_bonus - 1) * mst_mask)
    
    # 6. Final sparsification: zero out remaining unpromising edges
    # Keep only top percent of edges per node
    for i in range(n):
        row = heuristic[i]
        threshold = np.sort(row)[-k]  # k-th highest value
        row[row < threshold] = 0
    
    # Ensure symmetry
    heuristic = np.maximum(heuristic, heuristic.T)
    np.fill_diagonal(heuristic, 0)
    
    return heuristic
```
