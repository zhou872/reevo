import numpy as np
from scipy.spatial.distance import pdist, squareform
from scipy.special import softmax

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection probability estimation.
    
    Features:
    1. Distance Penalty Decay: Uses exponential decay rather than simple inverse
    2. Neighborhood Sparsification: Only considers top-k nearest neighbors per node
    3. Two-Hop Connectivity Scoring: Scores edges based on connectivity to common neighbors
    4. Local Density Bonus: Rewards edges between nodes in sparse regions
    """
    n = distance_matrix.shape[0]
    np.fill_diagonal(distance_matrix, np.inf)  # Avoid self-connections
    
    # 1. Base distance penalty with exponential decay
    # Closer distances get exponentially higher scores
    median_dist = np.median(distance_matrix[distance_matrix != np.inf])
    distance_scores = np.exp(-distance_matrix / (median_dist + 1e-10))
    
    # 2. Neighborhood sparsification - keep only top-k connections per node
    k = max(2, min(10, n // 5))  # Adaptive k based on problem size
    mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get indices of k-nearest neighbors (excluding self)
        neighbor_indices = np.argsort(distance_matrix[i])[:k]
        mask[i, neighbor_indices] = True
    
    # Make mask symmetric (if i is in j's top-k, include edge)
    mask = np.logical_or(mask, mask.T)
    
    # 3. Two-hop connectivity scoring
    # Reward edges that connect to common neighbors
    connectivity_scores = np.zeros((n, n))
    
    # Create adjacency from mask
    adjacency = mask.astype(float)
    np.fill_diagonal(adjacency, 0)
    
    # Two-hop connectivity: A^2 counts common neighbors
    two_hop = adjacency @ adjacency
    np.fill_diagonal(two_hop, 0)
    
    # Normalize two-hop connectivity
    if two_hop.max() > 0:
        connectivity_scores = two_hop / two_hop.max()
    
    # 4. Local density bonus
    # Reward edges between nodes in sparse regions
    local_density = np.zeros(n)
    for i in range(n):
        # Density = average inverse distance to k nearest neighbors
        neighbor_dists = np.sort(distance_matrix[i])[1:k+2]  # Skip self
        local_density[i] = 1.0 / (np.mean(neighbor_dists) + 1e-10)
    
    # Normalize densities
    if local_density.max() > 0:
        local_density = local_density / local_density.max()
    
    # Create density bonus matrix
    density_bonus = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j and mask[i, j]:
                # Higher bonus when both nodes are in sparse areas
                density_bonus[i, j] = (1 - local_density[i]) * (1 - local_density[j])
    
    # Combine all components
    combined_scores = np.zeros_like(distance_matrix)
    
    # Weighted combination
    alpha = 0.5  # Base distance weight
    beta = 0.3   # Connectivity weight
    gamma = 0.2  # Density bonus weight
    
    for i in range(n):
        for j in range(n):
            if i != j and mask[i, j]:
                combined_scores[i, j] = (
                    alpha * distance_scores[i, j] +
                    beta * connectivity_scores[i, j] +
                    gamma * density_bonus[i, j]
                )
            else:
                # Zero for unpromising edges (sparsification)
                combined_scores[i, j] = 0
    
    # Ensure symmetry
    combined_scores = (combined_scores + combined_scores.T) / 2
    
    # Apply softmax per node to get probability-like scores
    result = np.zeros_like(combined_scores)
    for i in range(n):
        # Only consider non-zero entries for this node
        row_nonzero = combined_scores[i] > 0
        if np.any(row_nonzero):
            # Apply softmax only to promising edges
            result[i, row_nonzero] = softmax(combined_scores[i, row_nonzero])
    
    # Normalize so maximum value is 1
    if result.max() > 0:
        result = result / result.max()
    
    np.fill_diagonal(result, 0)  # Ensure diagonal is zero
    np.fill_diagonal(distance_matrix, 0)  # Restore original distance matrix
    
    return result
