import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse import csr_matrix
from scipy.sparse.linalg import eigsh

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristic matrix for TSP edge selection with spectral embedding,
    multi-order regret propagation, and dynamic factor optimization.
    
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # Preprocess distances
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base inverse distance with adaptive scaling
    inv_dist = 1.0 / safe_dist
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Spectral graph embedding for structural insights
    embedding_bonus = np.zeros_like(inv_dist)
    if n >= 4:  # Only compute for non-trivial cases
        try:
            # Construct normalized Laplacian
            similarity = 1.0 / (1.0 + safe_dist)  # Convert to similarity
            D = np.diag(np.sum(similarity, axis=1))
            D_inv_sqrt = np.diag(1.0 / np.sqrt(np.diag(D)))
            L = np.eye(n) - D_inv_sqrt @ similarity @ D_inv_sqrt
            
            # Compute spectral embedding (k=min(5, n-1) dimensions)
            k = min(5, n - 1)
            if n <= 1000:  # Full eigendecomposition for moderate sizes
                eigvals, eigvecs = np.linalg.eigh(L)
                embedding = eigvecs[:, 1:k+1]  # Skip first trivial eigenvector
            else:  # Approximate for large graphs
                embedding, _ = eigsh(L, k=k, which='SM')
            
            # Compute embedding-based distances
            for i in range(n):
                for j in range(i + 1, n):
                    emb_dist = np.linalg.norm(embedding[i] - embedding[j])
                    # Bonus for edges connecting similar structural positions
                    embedding_bonus[i, j] = embedding_bonus[j, i] = np.exp(-emb_dist)
        except:
            # Fallback to simple inverse distance if embedding fails
            embedding_bonus = inv_dist.copy() * 0.5
    
    # 3. Higher-order regret propagation
    regret_factor = np.ones_like(inv_dist)
    order = min(4, n - 1)  # Higher-order regret
    
    for i in range(n):
        sorted_idx = np.argsort(safe_dist[i])
        
        if len(sorted_idx) >= order + 1:
            distances = safe_dist[i, sorted_idx[:order + 1]]
            
            # Compute regret cascade
            base = distances[0]
            regret_weights = np.zeros(order)
            for o in range(1, order + 1):
                regret = (distances[o] - base) / (base + 1e-10)
                regret_weights[o-1] = 1.0 / (1.0 + 2.0 * regret)  # Exponential decay
            
            # Apply weighted regret to edges
            for rank, j in enumerate(sorted_idx):
                if rank == 0:
                    regret_factor[i, j] = 1.2  # Boost for best edge
                elif rank < len(regret_weights):
                    regret_factor[i, j] = regret_weights[rank-1]
                else:
                    # Progressive penalty for worse edges
                    penalty = 0.7 ** (rank - order)
                    regret_factor[i, j] = max(0.1, penalty)
    
    # Symmetrize regret factor
    regret_factor = (regret_factor + regret_factor.T) / 2
    
    # 4. Adaptive mutual nearest neighbor with density awareness
    nn_bonus = np.zeros_like(inv_dist)
    k_nn = max(2, min(8, n // 3))
    
    nn_indices = np.argsort(safe_dist, axis=1)[:, 1:k_nn+1]
    nn_mask = np.zeros_like(inv_dist, dtype=bool)
    
    for i in range(n):
        nn_mask[i, nn_indices[i]] = True
    
    # Mutual nearest neighbor bonus with density weighting
    mutual_nn = nn_mask & nn_mask.T
    
    # Compute local density for each node
    densities = np.zeros(n)
    for i in range(n):
        # Average distance to k nearest neighbors as density measure
        densities[i] = np.mean(safe_dist[i, nn_indices[i]])
    
    # Normalize densities
    if np.max(densities) > 0:
        densities = 1.0 / (1.0 + densities / np.max(densities))
    
    # Apply density-weighted mutual NN bonus
    for i in range(n):
        for j in range(i + 1, n):
            if mutual_nn[i, j]:
                density_weight = (densities[i] + densities[j]) / 2
                nn_bonus[i, j] = nn_bonus[j, i] = 1.5 * density_weight
    
    # 5. Geometric clustering via MST-enhanced connectivity
    cluster_bonus = np.zeros_like(inv_dist)
    
    # Compute MST for base structure
    mst = minimum_spanning_tree(csr_matrix(safe_dist)).toarray()
    mst_mask = mst > 0
    
    # Two-hop neighborhood analysis
    for i in range(n):
        # Direct neighbors from MST
        mst_neighbors = np.where(mst_mask[i])[0]
        
        # Two-hop neighbors through MST
        two_hop = set()
        for neighbor in mst_neighbors:
            two_hop.update(np.where(mst_mask[neighbor])[0])
        two_hop.discard(i)
        
        # Bonus for edges within the same local cluster
        for j in two_hop:
            if i < j:
                # Shared neighbor ratio
                shared = len(set(mst_neighbors) & set(np.where(mst_mask[j])[0]))
                total = len(set(mst_neighbors) | set(np.where(mst_mask[j])[0]))
                if total > 0:
                    cluster_score = shared / total
                    cluster_bonus[i, j] = cluster_bonus[j, i] = cluster_score ** 0.7
    
    # 6. Dynamic factor weighting based on instance characteristics
    # Analyze instance properties
    dist_flat = safe_dist[safe_dist < np.inf]
    dist_mean = np.mean(dist_flat)
    dist_std = np.std(dist_flat)
    
    # Determine weighting strategy
    if dist_std / dist_mean < 0.3:  # Uniform distances
        weights = [0.25, 0.25, 0.25, 0.25]
    elif n < 50:  # Small instance
        weights = [0.3, 0.3, 0.2, 0.2]
    else:  # Large/irregular instance
        weights = [0.2, 0.25, 0.3, 0.25]
    
    # Combine factors with dynamic weights
    w_base, w_embed, w_nn, w_cluster = weights
    
    heuristics = (
        w_base * inv_dist +
        w_embed * embedding_bonus +
        w_nn * nn_bonus +
        w_cluster * cluster_bonus
    )
    
    # Apply regret factor multiplicatively
    heuristics = heuristics * regret_factor
    
    # 7. Adaptive sparsification with connectivity guarantee
    # Determine sparsification level based on instance size
    if n < 100:
        keep_edges = max(3, n // 4)
    else:
        keep_edges = max(5, n // 10)
    
    # Ensure each node has at least 2 connections
    min_edges = 2
    
    for i in range(n):
        row = heuristics[i].copy()
        
        # Keep strongest edges
        if np.count_nonzero(row) > min_edges:
            # Find threshold for top keep_edges
            if keep_edges < n - 1:
                threshold_idx = -keep_edges - 1
                threshold = np.partition(row, threshold_idx)[threshold_idx]
                mask = row >= threshold
                
                # Ensure minimum edges
                if np.sum(mask) < min_edges:
                    # Add strongest edges to reach minimum
                    missing = min_edges - np.sum(mask)
                    candidates = np.argsort(row)[-missing:]
                    mask[candidates] = True
                
                heuristics[i] = row * mask
    
    # Symmetrize and finalize
    heuristics = np.maximum(heuristics, heuristics.T * 0.9)
    
    # Global threshold to remove weak edges
    non_zero = heuristics[heuristics > 0]
    if len(non_zero) > 0:
        global_thresh = np.percentile(non_zero, 30)
        heuristics[heuristics < global_thresh] = 0
    
    # Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val * 10
    
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
