import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection.
    
    Combines multiple factors to determine edge promise:
    1. Inverse distance (baseline)
    2. Penalization for edges connecting to highly connected nodes
    3. Local clustering coefficient to favor edges within clusters
    4. Degree centrality to balance node connectivity
    5. Sparsification by keeping only top edges for each node
    """
    n = distance_matrix.shape[0]
    
    # 1. Baseline inverse distance
    inv_dist = 1 / (distance_matrix + np.eye(n) * 1e-10)  # Avoid division by zero
    
    # 2. Calculate degree centrality for each node
    # Degree here means how "attractive" a node is based on short edges to it
    degree_centrality = np.zeros(n)
    for i in range(n):
        # Count edges that are significantly shorter than average
        avg_dist = np.mean(distance_matrix[i, distance_matrix[i] > 0])
        degree_centrality[i] = np.sum(distance_matrix[i] < 0.7 * avg_dist)
    
    # Normalize degree centrality
    if np.max(degree_centrality) > 0:
        degree_centrality = degree_centrality / np.max(degree_centrality)
    
    # 3. Calculate local clustering coefficient for each node
    # Measure how well nodes cluster together
    clustering_coeff = np.zeros(n)
    for i in range(n):
        # Get neighbors (closest 30% of nodes)
        sorted_indices = np.argsort(distance_matrix[i])
        k = max(2, int(0.3 * n))
        neighbors = sorted_indices[1:k+1]  # Exclude self
        
        if len(neighbors) < 2:
            clustering_coeff[i] = 0
            continue
            
        # Calculate triangles among neighbors
        triangles = 0
        possible = 0
        
        for a_idx in range(len(neighbors)):
            for b_idx in range(a_idx + 1, len(neighbors)):
                a, b = neighbors[a_idx], neighbors[b_idx]
                # Check if a and b are close to each other
                if distance_matrix[a, b] < 0.8 * max(distance_matrix[i, a], distance_matrix[i, b]):
                    triangles += 1
                possible += 1
        
        clustering_coeff[i] = triangles / possible if possible > 0 else 0
    
    # 4. Compute combined heuristic matrix
    heuristic_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(i + 1, n):  # Only upper triangle, matrix will be symmetric
            if distance_matrix[i, j] <= 0:
                continue
                
            # Base score: inverse distance
            score = inv_dist[i, j]
            
            # Penalize if both nodes have high degree (to avoid hubs)
            degree_penalty = 1.0 - 0.3 * (degree_centrality[i] + degree_centrality[j])
            
            # Bonus if nodes belong to same cluster
            cluster_bonus = 1.0
            if clustering_coeff[i] > 0.5 and clustering_coeff[j] > 0.5:
                # Check if i and j have common close neighbors
                common_neighbors = 0
                for k in range(n):
                    if k != i and k != j:
                        if (distance_matrix[i, k] < 1.5 * distance_matrix[i, j] and 
                            distance_matrix[j, k] < 1.5 * distance_matrix[i, j]):
                            common_neighbors += 1
                cluster_bonus = 1.0 + 0.2 * min(common_neighbors / 3, 1.0)
            
            # Combine factors
            heuristic_matrix[i, j] = score * degree_penalty * cluster_bonus
            heuristic_matrix[j, i] = heuristic_matrix[i, j]  # Symmetric
    
    # 5. Sparsification: keep only top promising edges for each node
    sparsified = np.zeros_like(heuristic_matrix)
    
    for i in range(n):
        # For each node, keep edges to its top k neighbors
        k = max(3, int(0.25 * n))  # Keep 25% of edges, at least 3
        
        # Get indices of top k edges (excluding self)
        row_heuristics = heuristic_matrix[i].copy()
        row_heuristics[i] = -np.inf  # Exclude diagonal
        
        # Get indices of top k values
        top_indices = np.argpartition(row_heuristics, -k)[-k:]
        
        # Only keep these edges
        sparsified[i, top_indices] = heuristic_matrix[i, top_indices]
        sparsified[top_indices, i] = heuristic_matrix[top_indices, i]
    
    # Normalize to [0, 1] range for consistency
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
