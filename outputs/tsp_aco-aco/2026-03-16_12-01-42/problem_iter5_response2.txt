import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with adaptive sparsification,
    lookahead spatial averaging, regret-based penalization, and cluster density awareness.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base inverse distance ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Regret-based penalization (vectorized) ===
    k_regret = min(5, n - 1)
    regret_boost = np.ones_like(distance_matrix)
    
    # Compute regret for each node efficiently
    if n > 1:
        # Sort distances for each node
        sorted_indices = np.argsort(distance_matrix + np.eye(n) * np.inf, axis=1)[:, 1:k_regret+1]
        
        # Extract distances to k nearest neighbors
        row_indices = np.arange(n)[:, np.newaxis].repeat(k_regret, axis=1)
        knn_distances = distance_matrix[row_indices, sorted_indices]
        
        # Compute regret for each edge
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                # Find if j is in i's k-nearest
                j_in_knn = (sorted_indices[i] == j).any()
                
                if j_in_knn:
                    # For edges in k-nearest, regret is based on alternatives
                    mask = sorted_indices[i] != j
                    if mask.any():
                        best_alt = knn_distances[i, mask].min()
                        regret = max(0, best_alt - distance_matrix[i, j])
                        avg_dist = knn_distances[i].mean()
                        regret_boost[i, j] = 1.0 + 2.5 * (regret / (avg_dist + epsilon))
                else:
                    # For edges not in k-nearest, use scaled regret
                    d_ij = distance_matrix[i, j]
                    avg_knn = knn_distances[i].mean()
                    regret_ratio = max(0, avg_knn - d_ij) / (avg_knn + epsilon)
                    regret_boost[i, j] = 1.0 + 1.5 * regret_ratio
    
    # === 3. Lookahead spatial averaging with cluster density ===
    spatial_scores = np.zeros_like(distance_matrix)
    
    if n > 1:
        k_spatial = min(7, n - 1)
        
        # Compute local density for each node
        node_density = np.zeros(n)
        for i in range(n):
            dists = np.partition(distance_matrix[i], k_spatial + 1)[:k_spatial + 1]
            node_density[i] = 1.0 / (np.mean(dists[dists > 0]) + epsilon)
        
        # Compute spatial score combining endpoint characteristics
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                # Harmonic mean of inverse endpoint distances (emphasizes balanced connectivity)
                inv_i = 1.0 / (distance_matrix[i, j] + epsilon)
                
                # Consider 2nd order neighbors for lookahead
                if n > 2:
                    # Average distance from i to j's neighbors and vice versa
                    j_neighbors = np.argsort(distance_matrix[j])[1:min(4, n)]
                    i_neighbors = np.argsort(distance_matrix[i])[1:min(4, n)]
                    
                    avg_ij = np.mean([distance_matrix[i, k] for k in j_neighbors if k != i])
                    avg_ji = np.mean([distance_matrix[j, k] for k in i_neighbors if k != j])
                    
                    lookahead_factor = 1.0 / (0.5 * (avg_ij + avg_ji) + epsilon)
                    spatial_scores[i, j] = 0.7 * inv_i + 0.3 * lookahead_factor
                else:
                    spatial_scores[i, j] = inv_i
        
        # Normalize spatial scores by node density
        max_score = spatial_scores.max()
        if max_score > 0:
            spatial_scores = spatial_scores / max_score
            
        for i in range(n):
            for j in range(n):
                if i != j:
                    # Boost edges in dense regions, penalize in sparse regions
                    density_factor = (node_density[i] + node_density[j]) / 2.0
                    spatial_scores[i, j] *= (0.8 + 0.4 * density_factor)
    
    # === 4. Adaptive sparsification with regret-aware threshold ===
    adaptive_k = min(15, max(4, int(np.sqrt(n) * 1.5)))
    
    if n > adaptive_k:
        # Initialize edge importance matrix
        edge_importance = inv_dist.copy()
        edge_importance *= regret_boost
        edge_importance *= spatial_scores
        
        # MST-based importance boost
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_mask = (mst > 0) | (mst.T > 0)
        edge_importance[mst_mask] *= 1.8
        
        # Sparsify while maintaining connectivity
        sparsified = np.zeros_like(edge_importance)
        
        for i in range(n):
            row_importance = edge_importance[i].copy()
            
            # Always keep edges to nearest neighbors based on adaptive regret
            k_keep = min(adaptive_k, n - 1)
            
            # Sort by importance
            top_indices = np.argsort(row_importance)[-k_keep:]
            
            # Adjust threshold based on regret distribution
            if len(top_indices) > 0:
                # Compute regret variance in top candidates
                top_regrets = regret_boost[i, top_indices]
                regret_var = np.var(top_regrets)
                
                # More aggressive sparsification if regret is uniform
                if regret_var < 0.1:
                    # Keep only top adaptive_k/2 edges
                    top_indices = top_indices[-k_keep//2:]
                else:
                    # Keep edges with above-average regret boost
                    mean_regret = top_regrets.mean()
                    top_indices = [idx for idx in top_indices 
                                  if regret_boost[i, idx] > mean_regret * 0.8]
            
            # Ensure at least 2 edges per node for connectivity
            if len(top_indices) < 2:
                top_indices = np.argsort(row_importance)[-2:]
            
            sparsified[i, top_indices] = edge_importance[i, top_indices]
        
        # Symmetrize the sparsified matrix
        sparsified = np.maximum(sparsified, sparsified.T * 0.9)
        
        # Final heuristics matrix
        heuristics = sparsified
    else:
        # For small n, use all edges with importance weighting
        heuristics = inv_dist * regret_boost * spatial_scores
    
    # === 5. Final normalization and cleanup ===
    np.fill_diagonal(heuristics, 0)
    
    # Ensure non-negativity
    heuristics = np.maximum(heuristics, 0)
    
    # Normalize to [0, 1] range
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
