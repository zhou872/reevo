import numpy as np
from scipy.spatial import distance_matrix

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection combining multiple spatial and
    structural features with intelligent sparsification.
    """
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix)
    
    if n <= 2:
        return heuristics
    
    # 1. Nearest neighbor clustering component
    k_nearest = max(2, n // 10)  # Adaptive k
    nn_scores = np.zeros_like(distance_matrix)
    
    for i in range(n):
        # Get k nearest neighbors
        indices = np.argsort(distance_matrix[i])[1:k_nearest+1]
        # Inverse distance weighting with rank penalty
        for rank, j in enumerate(indices, 1):
            nn_scores[i, j] = (1 / distance_matrix[i, j]) * (1 - 0.1 * (rank - 1))
    
    # 2. Savings heuristic (Clarke-Wright inspired)
    savings = np.zeros_like(distance_matrix)
    
    # Compute average distances to all other nodes
    avg_distances = np.sum(distance_matrix, axis=1) / (n - 1)
    
    for i in range(n):
        for j in range(i + 1, n):
            if distance_matrix[i, j] > 0:
                # Classic savings formula with normalization
                saving = avg_distances[i] + avg_distances[j] - distance_matrix[i, j]
                savings[i, j] = saving
                savings[j, i] = saving
    
    # Normalize savings
    if np.max(savings) > 0:
        savings = savings / np.max(savings)
    
    # 3. Spatial connectivity component
    spatial_scores = np.zeros_like(distance_matrix)
    
    # Compute node centrality (closeness)
    closeness = 1 / (np.sum(distance_matrix, axis=1) + 1e-10)
    
    for i in range(n):
        for j in range(i + 1, n):
            if distance_matrix[i, j] > 0:
                # Combine distance inverse with node centrality
                centrality_factor = (closeness[i] + closeness[j]) / 2
                spatial_scores[i, j] = (1 / distance_matrix[i, j]) * centrality_factor
                spatial_scores[j, i] = spatial_scores[i, j]
    
    # Normalize spatial scores
    if np.max(spatial_scores) > 0:
        spatial_scores = spatial_scores / np.max(spatial_scores)
    
    # 4. Combine components with adaptive weights
    w_nn = 0.4
    w_savings = 0.3
    w_spatial = 0.3
    
    heuristics = (w_nn * nn_scores + 
                  w_savings * savings + 
                  w_spatial * spatial_scores)
    
    # 5. Intelligent sparsification
    for i in range(n):
        row = heuristics[i].copy()
        
        # Keep only promising edges based on multiple criteria
        threshold = np.percentile(row[row > 0], 70) if np.any(row > 0) else 0
        
        # Additional condition: require reasonable spatial connectivity
        for j in range(n):
            if i != j:
                # Must be in top 50% of nearest neighbors OR have high savings
                is_near = j in np.argsort(distance_matrix[i])[1:n//2+1]
                has_high_saving = savings[i, j] > np.percentile(savings[savings > 0], 60)
                
                if heuristics[i, j] < threshold and not (is_near and has_high_saving):
                    heuristics[i, j] = 0
    
    # 6. Ensure symmetry and zero diagonal
    heuristics = np.maximum(heuristics, heuristics.T)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
