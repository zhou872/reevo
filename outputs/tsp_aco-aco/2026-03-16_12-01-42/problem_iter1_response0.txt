import numpy as np
from scipy.spatial.distance import cdist

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple spatial and structural factors.
    Returns a matrix of edge promise scores with the same shape as input.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance score (normalized)
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    np.divide(1.0, distance_matrix, out=inv_dist, where=distance_matrix > 0)
    inv_dist -= np.diag(np.diag(inv_dist))  # Zero diagonal
    
    # 2. Two-step lookahead spatial score
    # For each edge (i,j), consider best connections from both endpoints
    lookahead_score = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Find third node k that minimizes (d(i,k) + d(k,j)) - d(i,j)
                mask = ~np.isin(range(n), [i, j])
                if np.any(mask):
                    third_node_penalty = np.min(
                        distance_matrix[i, mask] + distance_matrix[mask, j] - distance_matrix[i, j]
                    )
                    lookahead_score[i, j] = np.exp(-third_node_penalty / distance_matrix[i, j].mean())
    
    # 3. Regret-based penalty
    # Penalize edges that are much longer than the best alternative from each endpoint
    regret_matrix = np.zeros((n, n))
    for i in range(n):
        # Sort distances from node i (excluding self)
        sorted_dists = np.sort(distance_matrix[i, distance_matrix[i] > 0])
        if len(sorted_dists) >= 2:
            second_best = sorted_dists[1] if len(sorted_dists) > 1 else sorted_dists[0]
            for j in range(n):
                if i != j:
                    regret = distance_matrix[i, j] - second_best
                    regret_matrix[i, j] = np.exp(-max(0, regret) / (second_best + 1e-10))
    
    # 4. Cluster density bonus
    # Reward edges connecting nodes in dense regions
    density_bonus = np.zeros((n, n))
    
    # Compute local density for each node (average inverse distance to k nearest neighbors)
    k = max(2, min(5, n // 4))
    for i in range(n):
        dists = distance_matrix[i]
        sorted_idx = np.argsort(dists)[1:k+1]  # Exclude self
        density_i = np.mean(1.0 / (dists[sorted_idx] + 1e-10))
        
        for j in range(n):
            if i != j:
                # Check if j is in i's nearest neighbors
                if j in sorted_idx:
                    density_bonus[i, j] = density_i
    
    # Combine factors with weights
    weights = {
        'inv_dist': 0.4,
        'lookahead': 0.3,
        'regret': 0.2,
        'density': 0.1
    }
    
    # Normalize each component to [0,1] range
    components = {
        'inv_dist': inv_dist / (inv_dist.max() + 1e-10),
        'lookahead': lookahead_score / (lookahead_score.max() + 1e-10),
        'regret': regret_matrix / (regret_matrix.max() + 1e-10),
        'density': density_bonus / (density_bonus.max() + 1e-10)
    }
    
    # Weighted combination
    combined = np.zeros((n, n))
    for key, weight in weights.items():
        combined += weight * components[key]
    
    # 5. Strategic sparsification
    # Keep only promising edges based on multiple criteria
    sparsified = np.zeros((n, n))
    
    # Criterion 1: Top edges per node
    edges_per_node = max(3, min(8, n // 3))
    for i in range(n):
        row_scores = combined[i]
        top_indices = np.argsort(row_scores)[-edges_per_node:]
        sparsified[i, top_indices] = row_scores[top_indices]
    
    # Criterion 2: Mutual top selection (symmetric promising)
    mutual_mask = (sparsified > 0) & (sparsified.T > 0)
    final_heuristic = np.where(mutual_mask, combined, 0)
    
    # Criterion 3: Absolute threshold (keep only scores above median)
    if n > 2:
        median_score = np.median(final_heuristic[final_heuristic > 0])
        final_heuristic[final_heuristic < median_score] = 0
    
    # Ensure diagonal is zero
    np.fill_diagonal(final_heuristic, 0)
    
    return final_heuristic
