import numpy as np
from typing import Tuple
import heapq

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge selection heuristics incorporating Christofides algorithm insights
    and adaptive factor fusion.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        heuristics = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    epsilon = 1e-12
    
    # === 1. Base inverse distance with adaptive scaling ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # Normalize base inverse distances
    inv_dist_max = np.max(inv_dist)
    if inv_dist_max > 0:
        inv_dist = inv_dist / inv_dist_max
    
    # === 2. Christofides-inspired MST + Perfect Matching boost ===
    # Compute MST using Prim's algorithm
    mst_edges = set()
    if n > 1:
        visited = [False] * n
        min_edge = [np.inf] * n
        parent = [-1] * n
        min_edge[0] = 0
        
        for _ in range(n):
            # Find minimum edge to tree
            u = -1
            min_val = np.inf
            for i in range(n):
                if not visited[i] and min_edge[i] < min_val:
                    min_val = min_edge[i]
                    u = i
            
            if u == -1:
                break
                
            visited[u] = True
            if parent[u] != -1:
                mst_edges.add((min(u, parent[u]), max(u, parent[u])))
            
            # Update distances
            for v in range(n):
                if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                    min_edge[v] = distance_matrix[u, v]
                    parent[v] = u
    
    # Identify odd-degree vertices in MST
    degree = [0] * n
    for u, v in mst_edges:
        degree[u] += 1
        degree[v] += 1
    
    odd_vertices = [i for i in range(n) if degree[i] % 2 == 1]
    
    # Greedy minimum weight perfect matching on odd vertices
    matching_edges = set()
    if len(odd_vertices) >= 2:
        used = [False] * len(odd_vertices)
        
        for i in range(len(odd_vertices)):
            if not used[i]:
                # Find closest unmatched odd vertex
                best_j = -1
                best_dist = np.inf
                u = odd_vertices[i]
                
                for j in range(i + 1, len(odd_vertices)):
                    if not used[j]:
                        v = odd_vertices[j]
                        dist = distance_matrix[u, v]
                        if dist < best_dist:
                            best_dist = dist
                            best_j = j
                
                if best_j != -1:
                    v = odd_vertices[best_j]
                    matching_edges.add((min(u, v), max(u, v)))
                    used[i] = True
                    used[best_j] = True
    
    # Create Christofides boost matrix
    christofides_boost = np.ones_like(distance_matrix)
    for u, v in mst_edges:
        boost_val = 2.0 * (1.0 - distance_matrix[u, v] / np.max(distance_matrix[u, distance_matrix[u, :] > 0]))
        christofides_boost[u, v] = max(christofides_boost[u, v], boost_val)
        christofides_boost[v, u] = max(christofides_boost[v, u], boost_val)
    
    for u, v in matching_edges:
        boost_val = 1.5 * (1.0 - distance_matrix[u, v] / np.max(distance_matrix[u, distance_matrix[u, :] > 0]))
        christofides_boost[u, v] = max(christofides_boost[u, v], boost_val)
        christofides_boost[v, u] = max(christofides_boost[v, u], boost_val)
    
    # === 3. Dynamic regret with local competition ===
    # For each node, consider competition among k nearest neighbors
    k_regret = min(10, max(3, n // 10))
    regret_boost = np.ones_like(distance_matrix)
    
    # Pre-sort distances for efficiency
    sorted_indices = np.argsort(distance_matrix + np.eye(n) * np.inf, axis=1)
    
    for i in range(n):
        # Get k+1 nearest neighbors (including self at position 0)
        nearest = sorted_indices[i, :k_regret + 1]
        
        for idx, j in enumerate(nearest):
            if i == j:
                continue
                
            d_ij = distance_matrix[i, j]
            
            # Find best alternative among other nearest neighbors
            alt_distances = []
            for alt_j in nearest:
                if alt_j != i and alt_j != j:
                    alt_distances.append(distance_matrix[i, alt_j])
            
            if alt_distances:
                best_alt = np.min(alt_distances)
                # Competition ratio: how much better is this edge than alternatives
                comp_ratio = best_alt / (d_ij + epsilon)
                # Sigmoid transform to smooth boost
                regret_boost[i, j] = 1.0 + 0.8 * (2.0 / (1.0 + np.exp(-comp_ratio + 1.5)) - 1.0)
    
    # === 4. Multi-scale spatial scoring ===
    # Use multiple neighborhood sizes for robust spatial assessment
    spatial_scores = np.ones_like(distance_matrix)
    
    # Compute spatial signatures at different scales
    scales = [min(5, n-1), min(10, n-1), min(20, n-1)]
    scale_weights = [0.4, 0.3, 0.3]  # Emphasize smaller neighborhoods
    
    for scale, weight in zip(scales, scale_weights):
        if scale >= 2:
            scale_scores = np.zeros_like(distance_matrix)
            
            for i in range(n):
                # Get distances to scale nearest neighbors
                nearest_indices = sorted_indices[i, 1:scale+1]  # Exclude self
                nearest_dists = distance_matrix[i, nearest_indices]
                
                # Compute harmonic mean of distances for this node at this scale
                with np.errstate(divide='ignore', invalid='ignore'):
                    harmonic_mean = len(nearest_dists) / np.sum(1.0 / (nearest_dists + epsilon))
                
                # Score edges based on deviation from harmonic mean
                for j in range(n):
                    if i != j:
                        deviation = abs(distance_matrix[i, j] - harmonic_mean) / (harmonic_mean + epsilon)
                        # Reward edges close to the typical distance at this scale
                        scale_scores[i, j] += weight * np.exp(-deviation)
            
            spatial_scores += scale_scores
    
    # Normalize spatial scores
    spatial_max = np.max(spatial_scores)
    if spatial_max > 0:
        spatial_scores = spatial_scores / spatial_max
    
    # === 5. Adaptive factor fusion ===
    # Learn optimal combination weights from problem characteristics
    heuristics = np.zeros_like(distance_matrix)
    
    # Analyze problem structure to adjust weights
    if n > 10:
        # Estimate problem "hardness" via distance variance
        dist_flat = distance_matrix[np.triu_indices(n, k=1)]
        dist_cv = np.std(dist_flat) / (np.mean(dist_flat) + epsilon)
        
        # Adaptive weights based on coefficient of variation
        if dist_cv < 0.3:  # Uniform distances
            weights = [0.3, 0.4, 0.2, 0.1]  # Emphasize Christofides structure
        elif dist_cv < 0.7:  # Moderate variation
            weights = [0.25, 0.35, 0.25, 0.15]
        else:  # High variation
            weights = [0.2, 0.3, 0.3, 0.2]  # Emphasize local competition
    else:
        weights = [0.25, 0.35, 0.25, 0.15]  # Default weights
    
    # Convex combination of factors
    heuristics = (weights[0] * inv_dist +
                  weights[1] * christofides_boost +
                  weights[2] * regret_boost +
                  weights[3] * spatial_scores)
    
    # === 6. Context-aware sparsification ===
    # Preserve structure while removing weak edges
    adaptive_k = min(15, max(3, n // 5))
    
    for i in range(n):
        row = heuristics[i].copy()
        
        # Always keep Christofides edges
        keep_mask = np.zeros(n, dtype=bool)
        for j in range(n):
            if i != j and christofides_boost[i, j] > 1.2:
                keep_mask[j] = True
        
        # Keep top-k edges by combined score
        if adaptive_k < n - 1:
            # Adjust threshold based on row statistics
            row_mean = np.mean(row[row > 0])
            row_std = np.std(row[row > 0])
            threshold = row_mean + 0.5 * row_std
            
            # Keep edges above threshold or in top-k
            above_threshold = row > threshold
            top_k_indices = np.argsort(row)[-adaptive_k:]
            
            keep_mask = keep_mask | above_threshold
            keep_mask[top_k_indices] = True
        
        # Ensure minimum connectivity
        if np.sum(keep_mask) < 2:
            top2_indices = np.argsort(row)[-2:]
            keep_mask[top2_indices] = True
        
        row[~keep_mask] = 0
        heuristics[i] = row
    
    # === 7. Symmetrization with structural awareness ===
    symmetric_heuristics = np.zeros_like(heuristics)
    
    for i in range(n):
        for j in range(n):
            if i == j:
                symmetric_heuristics[i, j] = 0
            else:
                # Prefer mutual high scores, with bias toward shorter edges
                edge_factor = 1.0
                if distance_matrix[i, j] < np.percentile(distance_matrix[i], 25):
                    edge_factor = 1.2  # Boost short edges
                
                symmetric_heuristics[i, j] = max(
                    heuristics[i, j],
                    heuristics[j, i]
                ) * edge_factor
    
    # === 8. Final refinement ===
    np.fill_diagonal(symmetric_heuristics, 0)
    
    # Apply softmax-like normalization per node
    for i in range(n):
        row = symmetric_heuristics[i]
        row_max = np.max(row)
        if row_max > 0:
            # Softmax with temperature based on row variance
            row_var = np.var(row[row > 0])
            temperature = max(0.1, 1.0 / (1.0 + row_var))
            exp_row = np.exp(row / temperature - np.max(row / temperature))
            symmetric_heuristics[i] = exp_row / np.sum(exp_row)
    
    # Ensure non-negativity and scale to reasonable range
    symmetric_heuristics = np.maximum(symmetric_heuristics, 0)
    global_max = np.max(symmetric_heuristics)
    if global_max > 0:
        symmetric_heuristics = symmetric_heuristics / global_max
    
    return symmetric_heuristics
