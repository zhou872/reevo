import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Local neighborhood sparsification based on relative proximity
    2. Triangle inequality violation detection
    3. Node centrality based on local density
    4. Distance-regularized scoring
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) / (distance_matrix + 1e-10)
    
    heuristics = np.zeros((n, n))
    np.fill_diagonal(heuristics, 0)
    
    # Factor 1: Basic inverse distance (regularized to avoid extreme values)
    inv_dist = 1 / (distance_matrix + 0.1 * np.mean(distance_matrix))
    np.fill_diagonal(inv_dist, 0)
    
    # Factor 2: Local neighborhood sparsification
    # For each node, identify promising connections within local radius
    avg_dist = np.mean(distance_matrix)
    local_threshold = 0.3 * avg_dist
    
    # Create neighborhood mask
    neighborhood_mask = distance_matrix <= local_threshold
    np.fill_diagonal(neighborhood_mask, False)
    
    # Factor 3: Triangle inequality potential
    # Edges that violate triangle inequality are less promising
    tri_potential = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if neighborhood_mask[i, j]:
                # Check if edge (i,j) is shortcut for some k
                min_violation = np.inf
                for k in range(n):
                    if k != i and k != j:
                        violation = distance_matrix[i, j] / (distance_matrix[i, k] + distance_matrix[k, j] + 1e-10)
                        min_violation = min(min_violation, violation)
                tri_potential[i, j] = tri_potential[j, i] = min_violation
    
    # Factor 4: Density-weighted centrality
    # Nodes in dense regions should connect to their nearest neighbors
    local_density = np.zeros(n)
    for i in range(n):
        # Count neighbors within local radius
        local_density[i] = np.sum(neighborhood_mask[i])
    
    density_weight = np.zeros((n, n))
    for i in range(n):
        for j in range(i+1, n):
            if neighborhood_mask[i, j]:
                # Prefer connections between nodes with similar local density
                density_diff = abs(local_density[i] - local_density[j])
                density_weight[i, j] = density_weight[j, i] = 1 / (1 + density_diff)
    
    # Factor 5: Proximity-regularized scoring
    # Combine all factors with appropriate weights
    for i in range(n):
        for j in range(i+1, n):
            if neighborhood_mask[i, j]:
                # Base score from inverse distance
                base_score = inv_dist[i, j]
                
                # Boost if edge satisfies triangle inequality (value < 1)
                tri_boost = 1.5 if tri_potential[i, j] < 0.8 else 0.7
                
                # Density compatibility bonus
                density_bonus = 1 + 0.5 * density_weight[i, j]
                
                # Distance regularity - penalize edges much longer than average
                dist_ratio = distance_matrix[i, j] / avg_dist
                regularity = np.exp(-dist_ratio)
                
                # Combine factors
                score = base_score * tri_boost * density_bonus * regularity
                
                # Apply soft thresholding
                if score > 0.1:
                    heuristics[i, j] = heuristics[j, i] = score
    
    # Ensure numerical stability
    heuristics = np.maximum(heuristics, 0)
    
    # Normalize to make values comparable across instances
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
