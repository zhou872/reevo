import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic function for TSP edge selection probability.
    Returns a sparse matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Inverse distance (baseline)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        np.fill_diagonal(inv_dist, 0)
    
    # 2. Nearest neighbor density score
    # For each node, how many other nodes consider it among their k-nearest neighbors
    k = max(2, int(np.log2(n)))  # Adaptive k
    nn_density = np.zeros((n, n))
    
    for i in range(n):
        # Get indices of k nearest neighbors (excluding self)
        nearest = np.argsort(distance_matrix[i])[1:k+1]
        nn_density[i, nearest] += 1
    
    # Make symmetric and normalize
    nn_density = (nn_density + nn_density.T) / (2 * k)
    
    # 3. Minimum spanning tree connectivity bonus
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = ((mst > 0) | (mst.T > 0)).astype(float)
    
    # 4. Regret penalty: penalize edges that are much worse than the best alternative
    regret_penalty = np.ones((n, n))
    for i in range(n):
        # Sort distances from node i
        sorted_dists = np.sort(distance_matrix[i])
        # Best and second best distances (excluding zero distance to self)
        best_dist = sorted_dists[1]
        second_best = sorted_dists[2] if n > 2 else best_dist
        
        for j in range(n):
            if i != j:
                # Penalty increases with how much worse this edge is than alternatives
                regret = (distance_matrix[i, j] - best_dist) / (second_best - best_dist + 1e-10)
                regret_penalty[i, j] = np.exp(-0.5 * regret)
    
    # 5. Combine factors
    heuristics_matrix = inv_dist * (1 + nn_density) * (1 + 0.3 * mst_bonus) * regret_penalty
    
    # 6. Sparsification: keep only top edges for each node
    k_sparse = min(10, max(3, n // 4))  # Keep more edges for smaller instances
    sparse_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get top k_sparse edges for node i
        top_indices = np.argsort(heuristics_matrix[i])[-k_sparse:]
        sparse_mask[i, top_indices] = True
    
    # Make symmetric (undirected graph)
    sparse_mask = sparse_mask | sparse_mask.T
    
    # Apply sparsification
    heuristics_matrix = heuristics_matrix * sparse_mask
    
    # 7. Local normalization for each node
    for i in range(n):
        row_sum = heuristics_matrix[i].sum()
        if row_sum > 0:
            heuristics_matrix[i] /= row_sum
    
    return heuristics_matrix
