import numpy as np
from scipy.spatial import KDTree
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection combining multiple factors:
    1. Two-hop neighborhood clustering to identify promising local structures
    2. Local minimum spanning tree edges get priority
    3. Sparsification by keeping only top-ranked edges per node
    4. Bonus for edges that align well with local angular structure
    """
    n = distance_matrix.shape[0]
    np.fill_diagonal(distance_matrix, np.inf)  # Ignore self-distances
    
    # 1. Basic inverse distance (baseline)
    inv_dist = 1 / distance_matrix
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Two-hop neighborhood analysis
    # For each node, find the average distance to its k nearest neighbors
    k_nearest = min(5, n - 1)
    two_hop_scores = np.zeros((n, n))
    
    for i in range(n):
        # Get k nearest neighbors
        nearest_indices = np.argsort(distance_matrix[i])[:k_nearest]
        for j in nearest_indices:
            if i == j:
                continue
            # For each neighbor, look at its k nearest neighbors
            neighbor_nearest = np.argsort(distance_matrix[j])[:k_nearest]
            # Edge gets bonus if it connects to a node that is also well-connected
            common_neighbors = len(set(nearest_indices) & set(neighbor_nearest))
            two_hop_scores[i, j] = 1.0 + 0.2 * common_neighbors
    
    # 3. Local Minimum Spanning Tree bonus
    mst_bonus = np.zeros((n, n))
    # Build MST on complete graph
    mst = minimum_spanning_tree(distance_matrix)
    mst_dense = mst.toarray()
    mst_edges = np.where(mst_dense > 0)
    
    for i, j in zip(*mst_edges):
        if i != j:
            # MST edges get significant bonus
            mst_bonus[i, j] = 2.0
            mst_bonus[j, i] = 2.0
    
    # 4. Angular correlation bonus (for geometric TSP instances)
    angular_bonus = np.ones((n, n))
    try:
        # Estimate node positions using multidimensional scaling (crude approximation)
        # This works better when distance matrix approximates Euclidean distances
        from sklearn.manifold import MDS
        if n >= 3:
            embedder = MDS(n_components=2, dissimilarity='precomputed', random_state=42)
            positions = embedder.fit_transform(distance_matrix[:min(n, 50), :min(n, 50)])
            if len(positions) == n:
                for i in range(n):
                    vectors = positions - positions[i]
                    norms = np.linalg.norm(vectors, axis=1)
                    valid = norms > 0
                    if valid.any():
                        # Normalize vectors
                        vectors[valid] = vectors[valid] / norms[valid, None]
                        # For each neighbor j, compute angular correlation
                        for j in range(n):
                            if i != j and distance_matrix[i, j] < np.inf:
                                # Find vector from i to j
                                if norms[j] > 0:
                                    v_ij = vectors[j]
                                    # Find closest vectors to v_ij
                                    dots = np.dot(vectors, v_ij)
                                    dots[i] = -2  # Exclude self
                                    # Bonus if edge aligns with local structure
                                    angular_bonus[i, j] = 1.0 + 0.5 * np.max(dots)
    except:
        # If MDS fails or not available, use uniform bonus
        pass
    
    # Combine all factors
    heuristics_matrix = inv_dist * two_hop_scores * angular_bonus + mst_bonus
    
    # 5. Sparsification: keep only top m edges for each node
    m = min(10, n - 1)  # Keep at most m promising edges per node
    sparsified = np.zeros((n, n))
    
    for i in range(n):
        # Get indices of top m edges from node i
        row = heuristics_matrix[i]
        top_indices = np.argsort(row)[-m:]
        # Keep only these edges
        sparsified[i, top_indices] = row[top_indices]
    
    # Make symmetric (keep the better of the two directions)
    symmetric_heuristics = np.maximum(sparsified, sparsified.T)
    
    # Final cleanup
    np.fill_diagonal(symmetric_heuristics, 0)
    
    return symmetric_heuristics
