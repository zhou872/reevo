import numpy as np
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray, 
                  mst_weight: float = 1.5,
                  neighbor_threshold: float = 0.3,
                  connectivity_factor: float = 0.8) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors.
    
    Args:
        distance_matrix: Square matrix of pairwise distances
        mst_weight: Weight for Minimum Spanning Tree edges
        neighbor_threshold: Percentile for neighbor selection (0-1)
        connectivity_factor: Weight for node connectivity consideration
        
    Returns:
        Heuristic matrix of same shape indicating edge promise
    """
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance weighting
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1 / distance_matrix
        inv_dist[np.isinf(inv_dist)] = 0
        inv_dist[np.isnan(inv_dist)] = 0
    
    # Normalize inverse distances
    if np.any(inv_dist > 0):
        inv_dist = inv_dist / np.max(inv_dist)
    
    # 2. Minimum Spanning Tree edges get bonus
    mst_mask = _compute_mst_edges(distance_matrix)
    
    # 3. Local neighborhood consideration
    neighbor_mask = _compute_neighbor_mask(distance_matrix, neighbor_threshold)
    
    # 4. Connectivity bonus: favor edges connecting well-connected nodes
    connectivity = _compute_connectivity_bonus(distance_matrix)
    
    # Combine all factors
    heuristics = inv_dist.copy()
    
    # Apply MST bonus
    heuristics[mst_mask] *= mst_weight
    
    # Apply neighbor filtering
    heuristics *= neighbor_mask
    
    # Add connectivity bonus
    connectivity_bonus = np.outer(connectivity, connectivity) * connectivity_factor
    np.fill_diagonal(connectivity_bonus, 0)
    heuristics += connectivity_bonus
    
    # Sparsify: zero out unpromising edges (bottom 30%)
    threshold = np.percentile(heuristics[heuristics > 0], 30) if np.any(heuristics > 0) else 0
    heuristics[heuristics < threshold] = 0
    
    # Ensure symmetry
    heuristics = np.maximum(heuristics, heuristics.T)
    
    return heuristics

def _compute_mst_edges(distance_matrix: np.ndarray) -> np.ndarray:
    """Compute Minimum Spanning Tree edges using Prim's algorithm."""
    n = distance_matrix.shape[0]
    mst_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    if n == 0:
        return mst_mask
    
    # Prim's algorithm
    visited = np.zeros(n, dtype=bool)
    visited[0] = True
    edges_added = 0
    
    while edges_added < n - 1:
        min_edge = (None, None)
        min_dist = np.inf
        
        for i in range(n):
            if not visited[i]:
                continue
            for j in range(n):
                if not visited[j] and distance_matrix[i, j] < min_dist and i != j:
                    min_dist = distance_matrix[i, j]
                    min_edge = (i, j)
        
        if min_edge[0] is not None:
            i, j = min_edge
            mst_mask[i, j] = True
            mst_mask[j, i] = True
            visited[j] = True
            edges_added += 1
    
    return mst_mask

def _compute_neighbor_mask(distance_matrix: np.ndarray, 
                          threshold_percentile: float) -> np.ndarray:
    """
    Create mask for promising neighbor edges based on distance percentile.
    """
    n = distance_matrix.shape[0]
    mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get distances from node i to all others
        distances = distance_matrix[i].copy()
        distances[i] = np.inf  # Ignore self
        
        # Calculate threshold for this node
        valid_distances = distances[distances < np.inf]
        if len(valid_distances) == 0:
            continue
            
        threshold = np.percentile(valid_distances, threshold_percentile * 100)
        
        # Mark edges to close neighbors
        close_neighbors = distances <= threshold
        mask[i] = close_neighbors
    
    # Ensure symmetry
    mask = mask | mask.T
    np.fill_diagonal(mask, False)
    
    return mask.astype(float)

def _compute_connectivity_bonus(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Compute connectivity score for each node based on average distance to others.
    """
    n = distance_matrix.shape[0]
    connectivity = np.zeros(n)
    
    for i in range(n):
        # Consider distances to all other nodes
        other_dists = np.delete(distance_matrix[i], i)
        if len(other_dists) > 0:
            # Inverse of average distance: better connected nodes have smaller avg distance
            avg_dist = np.mean(other_dists)
            connectivity[i] = 1 / (avg_dist + 1e-10)
    
    # Normalize connectivity scores
    if np.max(connectivity) > 0:
        connectivity = connectivity / np.max(connectivity)
    
    return connectivity
