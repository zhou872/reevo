import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection with optimized computational efficiency
    and sophisticated scoring mechanisms.
    
    Features:
    1. Vectorized regret computation for improved performance
    2. Harmonic centrality-based edge penalty
    3. Fusion of structural bonuses (MST, mutual nearest neighbors)
    4. Adaptive per-node sparsification with top-K selection
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        Symmetric matrix of distances between nodes.
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape as input, indicating edge promise.
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # 1. Base inverse distance heuristic
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Vectorized regret-based penalty
    regret_penalty = np.ones_like(distance_matrix)
    
    # Find best and second-best distances for each node (vectorized)
    # Replace diagonal with infinity to exclude self
    dist_with_inf = distance_matrix.copy()
    np.fill_diagonal(dist_with_inf, np.inf)
    
    # Get sorted indices and values for each row
    sorted_indices = np.argsort(dist_with_inf, axis=1)
    sorted_values = np.take_along_axis(dist_with_inf, sorted_indices, axis=1)
    
    # Best and second-best distances
    best_dists = sorted_values[:, 0]
    second_best_dists = sorted_values[:, 1] if n > 2 else best_dists
    
    # Compute regret and spread for each edge (vectorized)
    for i in range(n):
        # Regret from node i's perspective
        regret_i = dist_with_inf[i] - best_dists[i]
        spread_i = np.maximum(epsilon, second_best_dists[i] - best_dists[i])
        penalty_i = 1.0 / (1.0 + regret_i / spread_i)
        
        # Regret from node j's perspective (using symmetry)
        regret_j = dist_with_inf[i] - best_dists  # Broadcasting
        spread_j = np.maximum(epsilon, second_best_dists - best_dists)
        penalty_j = 1.0 / (1.0 + regret_j / spread_j)
        
        # Geometric mean for symmetric penalty
        regret_penalty[i] = np.sqrt(penalty_i * penalty_j)
    
    np.fill_diagonal(regret_penalty, 0)
    
    # 3. Harmonic centrality-based penalty
    # Compute harmonic centrality: sum of inverse distances
    harmonic_centrality = np.sum(1 / (distance_matrix + epsilon), axis=1) - 1/(epsilon)  # Subtract self
    
    # Normalize centrality
    if np.max(harmonic_centrality) > np.min(harmonic_centrality):
        centrality_norm = (harmonic_centrality - np.min(harmonic_centrality)) / \
                         (np.max(harmonic_centrality) - np.min(harmonic_centrality))
    else:
        centrality_norm = np.ones_like(harmonic_centrality) * 0.5
    
    # Penalize edges between high-centrality nodes
    centrality_penalty = 1.0 - 0.3 * np.sqrt(np.outer(centrality_norm, centrality_norm))
    np.fill_diagonal(centrality_penalty, 0)
    
    # 4. Efficient MST computation using built-in function
    mst_boost = np.zeros_like(distance_matrix)
    if n > 1:
        # Get MST from scipy (returns sparse matrix)
        mst_sparse = minimum_spanning_tree(distance_matrix)
        mst_dense = mst_sparse.toarray()
        mst_boost = (mst_dense > 0).astype(float) * 1.5  # MST bonus factor
        mst_boost = np.maximum(mst_boost, mst_boost.T)  # Symmetrize
    
    # 5. Mutual nearest neighbors with vectorized computation
    k_mnn = max(3, int(np.sqrt(n)))
    mnn_boost = np.zeros_like(distance_matrix)
    
    if n > k_mnn:
        # Find k nearest neighbors for each node
        nn_indices = np.argpartition(distance_matrix, k_mnn+1, axis=1)[:, 1:k_mnn+1]
        
        # Create adjacency matrix for nearest neighbors
        nn_adj = np.zeros_like(distance_matrix, dtype=bool)
        for i in range(n):
            nn_adj[i, nn_indices[i]] = True
        
        # Mutual nearest neighbors (both directions)
        mutual_nn = nn_adj & nn_adj.T
        mnn_boost[mutual_nn] = 1.3  # Mutual NN bonus factor
    
    # 6. Structural fusion: combine base heuristic with penalties and bonuses
    # Multiplicative combination for normalized components
    base_score = inv_dist * regret_penalty * centrality_penalty
    
    # Additive combination for structural bonuses
    structural_boost = mst_boost + mnn_boost
    heuristics = base_score * (1.0 + 0.5 * structural_boost)  # Scale structural influence
    
    # 7. Adaptive per-node sparsification with top-K selection
    adaptive_k = min(12, max(4, n // 6))  # Adaptive number of edges to keep per node
    
    for i in range(n):
        row = heuristics[i].copy()
        
        # Ensure we don't remove all connections
        if adaptive_k < n - 1:
            # Find threshold for top adaptive_k edges
            threshold_idx = n - adaptive_k - 1
            # Use partition for efficiency (doesn't fully sort)
            partitioned = np.partition(row, threshold_idx)
            threshold = partitioned[threshold_idx]
            
            # Zero out edges below threshold
            below_threshold = row < threshold
            
            # Ensure at least 3 connections remain
            if np.sum(~below_threshold) < 3:
                # Keep top 3 edges
                top3_threshold = np.partition(row, n - 4)[n - 4]
                below_threshold = row < top3_threshold
            
            row[below_threshold] = 0
            heuristics[i] = row
    
    # 8. Symmetrize final heuristic matrix
    # Take geometric mean of both directions to preserve edge importance
    heuristics_sym = np.sqrt(heuristics * heuristics.T)
    np.fill_diagonal(heuristics_sym, 0)
    
    # Ensure non-negativity
    heuristics_sym = np.maximum(heuristics_sym, 0)
    
    return heuristics_sym
