import numpy as np
from scipy.spatial import distance_matrix
import itertools

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection that combines multiple factors
    and sparsifies unpromising edges.
    
    Factors considered:
    1. Direct distance reciprocal (closer nodes are better)
    2. Node connectivity analysis (balanced connectivity is better)
    3. Local clustering tendency
    4. Edge promise based on node centrality
    """
    n = distance_matrix.shape[0]
    
    # Avoid division by zero
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # Factor 1: Basic distance reciprocal (normalized)
    dist_reciprocal = 1.0 / safe_dist
    dist_reciprocal = dist_reciprocal / dist_reciprocal.max()
    
    # Factor 2: Node connectivity score
    # For each node, compute how well-connected it is to others
    connectivity = np.zeros((n, n))
    for i in range(n):
        # Get distances from node i to all others
        dist_from_i = safe_dist[i, :]
        
        # Normalize distances
        if dist_from_i.max() > 0:
            norm_dist = dist_from_i / dist_from_i.max()
            # Connectivity is higher when distances are more balanced
            # (not just connected to nearest neighbors)
            connectivity[i, :] = 1.0 - norm_dist
        else:
            connectivity[i, :] = 0
    
    # Symmetrize connectivity
    connectivity = (connectivity + connectivity.T) / 2
    
    # Factor 3: Local clustering coefficient approximation
    # For each edge (i,j), check how many common neighbors are close to both
    cluster_score = np.zeros((n, n))
    
    # Threshold for considering a neighbor "close"
    dist_threshold = np.percentile(safe_dist[safe_dist < np.inf], 25)
    
    for i in range(n):
        for j in range(i+1, n):
            if i == j:
                continue
                
            # Find common neighbors that are close to both i and j
            close_to_i = safe_dist[i, :] < dist_threshold
            close_to_j = safe_dist[j, :] < dist_threshold
            
            # Common close neighbors
            common_close = np.sum(close_to_i & close_to_j)
            
            # Maximum possible common neighbors (excluding i and j)
            max_possible = n - 2
            
            if max_possible > 0:
                cluster_score[i, j] = cluster_score[j, i] = common_close / max_possible
    
    # Factor 4: Edge centrality (how central the edge is in connecting the graph)
    edge_centrality = np.zeros((n, n))
    
    # Simple centrality: edges that connect to nodes with many close neighbors
    node_centrality = np.zeros(n)
    for i in range(n):
        # Count how many nodes are within median distance
        median_dist = np.median(safe_dist[i, :])
        node_centrality[i] = np.sum(safe_dist[i, :] < median_dist) / (n - 1)
    
    for i in range(n):
        for j in range(i+1, n):
            if i == j:
                continue
            # Edge centrality is geometric mean of node centralities
            edge_centrality[i, j] = edge_centrality[j, i] = np.sqrt(
                node_centrality[i] * node_centrality[j]
            )
    
    # Combine all factors with weights
    weights = {
        'dist': 0.4,        # Distance is most important
        'connectivity': 0.25, # Balanced connectivity
        'cluster': 0.2,     # Local clustering
        'centrality': 0.15   # Global position
    }
    
    # Compute combined score
    combined_score = (
        weights['dist'] * dist_reciprocal +
        weights['connectivity'] * connectivity +
        weights['cluster'] * cluster_score +
        weights['centrality'] * edge_centrality
    )
    
    # Ensure diagonal is zero
    np.fill_diagonal(combined_score, 0)
    
    # Sparsification: Set unpromising edges to zero
    # Strategy: For each node, keep only the top k edges
    
    # Adaptive k: keep more edges for larger graphs, fewer for smaller
    k = max(3, min(n // 3, 10))  # Between 3 and 10, scaling with n
    
    sparsified = np.zeros_like(combined_score)
    
    for i in range(n):
        # Get scores for edges from node i
        node_scores = combined_score[i, :].copy()
        
        # Find top k edges (excluding self)
        top_indices = np.argsort(node_scores)[-k:]
        
        # Only keep these edges
        sparsified[i, top_indices] = node_scores[top_indices]
    
    # Make the matrix symmetric (since TSP edges are undirected)
    # Take the maximum of i->j and j->i
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # Final normalization to [0, 1] range
    max_val = sparsified.max()
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified

    points = np.random.rand(5, 2) * 100
    dist_mat = distance_matrix(points, points)
    
    heuristics_v1_result = 1 / (dist_mat + np.eye(5) * 1e-10)  # Avoid division by zero on diagonal
    np.fill_diagonal(heuristics_v1_result, 0)
    
    heuristics_v2_result = heuristics_v2(dist_mat)
