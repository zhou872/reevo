import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection probability.
    Combines multiple factors to score edges and sparsifies unpromising ones.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance (closer is better)
    inv_dist = np.zeros_like(distance_matrix, dtype=float)
    np.fill_diagonal(inv_dist, 0)
    mask = distance_matrix != 0
    inv_dist[mask] = 1.0 / distance_matrix[mask]
    
    # 2. Local neighborhood factor - edges connecting to nodes with few good alternatives
    # For each node, find average distance to other nodes (excluding itself)
    avg_dist = np.sum(distance_matrix, axis=1) / (n - 1)
    # Nodes with higher average distance are more "isolated"
    isolation_factor = avg_dist.reshape(-1, 1) + avg_dist.reshape(1, -1)
    isolation_bonus = 1.0 / (1.0 + isolation_factor)
    
    # 3. Rank-based penalty - penalize edges that are not top choices for either endpoint
    k = max(3, int(np.sqrt(n)))  # Adaptive sparsity parameter
    
    # Get sorted indices for each row (excluding self)
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    # Create rank matrix where lower rank = closer neighbor
    rank_matrix = np.zeros_like(distance_matrix, dtype=int)
    for i in range(n):
        # Assign ranks: 0 for self, 1 for nearest, 2 for second nearest, etc.
        rank_matrix[i, sorted_indices[i]] = np.arange(n)
    
    # Penalty based on geometric mean of ranks
    rank_penalty = np.sqrt((rank_matrix + 1) * (rank_matrix.T + 1))
    rank_factor = 1.0 / (1.0 + rank_penalty / k)
    
    # 4. Symmetric connectivity bonus - edges that are mutually important
    mutual_rank = np.maximum(rank_matrix, rank_matrix.T)
    symmetric_bonus = 1.0 / (1.0 + mutual_rank)
    
    # 5. Combine all factors with weights
    weights = np.array([2.0, 1.0, 1.5, 0.8])  # Weights for each factor
    weights = weights / np.sum(weights)
    
    # Normalize each factor to [0, 1] range
    factors = []
    
    # Normalize inverse distance
    inv_dist_norm = inv_dist / (np.max(inv_dist) + 1e-10)
    factors.append(inv_dist_norm)
    
    # Normalize isolation bonus
    isolation_norm = isolation_bonus / (np.max(isolation_bonus) + 1e-10)
    factors.append(isolation_norm)
    
    # Rank factor is already in [0, 1]
    factors.append(rank_factor)
    
    # Normalize symmetric bonus
    symmetric_norm = symmetric_bonus / (np.max(symmetric_bonus) + 1e-10)
    factors.append(symmetric_norm)
    
    # Weighted combination
    score_matrix = np.zeros_like(distance_matrix, dtype=float)
    for i, factor in enumerate(factors):
        score_matrix += weights[i] * factor
    
    # 6. Adaptive sparsification - keep only promising edges
    # Calculate adaptive threshold
    sorted_scores = np.sort(score_matrix.flatten())[::-1]
    keep_count = min(2 * n, n * (n - 1) // 4)  # Keep at most 2*n edges or 25% of all edges
    
    if keep_count < len(sorted_scores):
        threshold = sorted_scores[keep_count]
        score_matrix[score_matrix < threshold] = 0
    
    # Ensure diagonal is zero
    np.fill_diagonal(score_matrix, 0)
    
    return score_matrix
