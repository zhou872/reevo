import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection combining multiple factors:
    1. Sparsified reciprocal distance scoring
    2. Nearest-neighbor chain connectivity
    3. Minimum spanning tree alignment
    4. Regional connectivity analysis
    """
    n = len(distance_matrix)
    np.fill_diagonal(distance_matrix, np.inf)  # Ignore self-loops
    
    # Base reciprocal score (like v1)
    recip = 1 / distance_matrix
    
    # 1. Nearest-neighbor chain boost
    # For each node, identify top-k nearest neighbors
    k = max(2, int(np.sqrt(n)))
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        indices = np.argsort(distance_matrix[i])[:k]
        nn_mask[i, indices] = True
        nn_mask[indices, i] = True  # Symmetric
    
    # Boost edges that are mutual nearest neighbors
    mutual_nn = nn_mask & nn_mask.T
    nn_boost = np.zeros_like(distance_matrix)
    nn_boost[mutual_nn] = 2.0  # Strong boost for mutual NN
    
    # 2. MST-based prioritization
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = np.zeros_like(distance_matrix)
    mst_edges = (mst > 0) | (mst.T > 0)
    mst_bonus[mst_edges] = 1.5
    
    # 3. Regional connectivity analysis
    # Penalize edges that connect nodes with very different neighborhood distances
    region_penalty = np.zeros_like(distance_matrix)
    
    # Compute average distance to nearest p neighbors for each node
    p = min(5, n-1)
    avg_local_dist = np.zeros(n)
    for i in range(n):
        nearest_dists = np.sort(distance_matrix[i])[1:p+1]  # Exclude self
        avg_local_dist[i] = np.mean(nearest_dists)
    
    # Penalize edges connecting nodes with very different local densities
    for i in range(n):
        for j in range(i+1, n):
            if distance_matrix[i, j] < np.inf:
                density_ratio = max(avg_local_dist[i], avg_local_dist[j]) / \
                               min(avg_local_dist[i], avg_local_dist[j])
                if density_ratio > 2.0:  # Very different neighborhoods
                    region_penalty[i, j] = region_penalty[j, i] = 0.3
    
    # 4. Combine factors with weighting
    heuristics_matrix = (
        0.4 * recip +           # Base reciprocal
        0.3 * nn_boost +        # Nearest neighbor boost
        0.3 * mst_bonus -       # MST alignment
        0.2 * region_penalty    # Regional penalty
    )
    
    # 5. Sparsification: keep only promising edges
    # For each node, keep top m edges and zero out others
    m = min(8, n-1)
    for i in range(n):
        row = heuristics_matrix[i].copy()
        # Get indices of top m edges (excluding self)
        top_indices = np.argsort(row)[-m:]
        mask = np.zeros(n, dtype=bool)
        mask[top_indices] = True
        mask[i] = False  # Ensure diagonal stays zero
        heuristics_matrix[i, ~mask] = 0
    
    # Ensure symmetry (since TSP edges are undirected)
    heuristics_matrix = np.maximum(heuristics_matrix, heuristics_matrix.T)
    
    # Normalize to [0, 1] range for consistency
    max_val = np.max(heuristics_matrix)
    if max_val > 0:
        heuristics_matrix /= max_val
    
    np.fill_diagonal(heuristics_matrix, 0)  # Explicit zero diagonal
    return heuristics_matrix
