import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced edge promise heuristics for TSP solution sampling.
    Combines multiple factors and sparsifies unpromising edges.
    """
    n = distance_matrix.shape[0]
    np.fill_diagonal(distance_matrix, np.inf)
    
    # 1. Inverse distance (base heuristic)
    with np.errstate(divide='ignore'):
        inv_dist = 1.0 / distance_matrix
    inv_dist[np.isinf(inv_dist)] = 0
    
    # 2. Symmetric regret weighting
    # For each node, find two nearest neighbors
    nearest = np.partition(distance_matrix, 2, axis=1)[:, :2]
    regret = (nearest[:, 1] - nearest[:, 0]) / (nearest[:, 0] + 1e-12)
    regret_matrix = regret[:, None] + regret[None, :]
    
    # 3. Connectivity potential scoring
    # Nodes with higher average distance need stronger connections
    node_isolation = np.mean(distance_matrix, axis=1)
    isolation_factor = (node_isolation[:, None] + node_isolation[None, :]) / 2
    
    # 4. Combine factors with weights
    heuristics = (
        0.5 * inv_dist +                   # Base inverse distance
        0.3 * regret_matrix * inv_dist +   # Regret-weighted
        0.2 * inv_dist / isolation_factor  # Isolation-normalized
    )
    
    # 5. Local neighborhood emphasis
    # Boost edges connecting to k-nearest neighbors
    k = max(3, n // 10)
    nearest_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        indices = np.argsort(distance_matrix[i])[:k]
        nearest_mask[i, indices] = True
    heuristics *= (1.0 + 0.5 * nearest_mask)
    
    # 6. Symmetrize (for undirected TSP)
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 7. Sparsification with relative threshold
    # Keep only top promising edges
    mask = np.zeros_like(heuristics, dtype=bool)
    threshold = np.percentile(heuristics[heuristics > 0], 50)  # Median threshold
    
    # Preserve strong local connections
    local_threshold = 0.7 * np.max(heuristics, axis=1, keepdims=True)
    local_mask = heuristics > local_threshold
    
    # Keep either globally strong or locally strong edges
    mask = (heuristics > threshold) | local_mask
    
    # Ensure each node has at least 2 connections
    for i in range(n):
        if np.sum(mask[i]) < 2:
            indices = np.argsort(heuristics[i])[-2:]
            mask[i, indices] = True
            mask[indices, i] = True
    
    heuristics = heuristics * mask
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
