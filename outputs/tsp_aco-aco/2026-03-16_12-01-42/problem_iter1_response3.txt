import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic for TSP edge selection combining multiple factors.
    Returns a matrix of edge promise indicators with same shape as input.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.ones_like(distance_matrix) / (distance_matrix + 1e-10)
    
    # Initialize heuristic matrix
    H = np.zeros_like(distance_matrix)
    
    # 1. Base inverse distance weighting
    inv_dist = 1 / (distance_matrix + 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Compute nearest neighbors for each node
    k = max(3, min(8, n // 4))  # Adaptive neighbor count
    nearest_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        indices = np.argsort(distance_matrix[i])[1:k+1]
        nearest_mask[i, indices] = True
    
    # 3. Compute regret values (difference between best and second-best edges)
    regret_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        sorted_dists = np.sort(distance_matrix[i])
        if len(sorted_dists) >= 3:
            best, second_best = sorted_dists[1], sorted_dists[2]
            for j in range(n):
                if i != j:
                    # Higher regret means edge to j is much better than alternatives
                    regret = max(0, second_best - distance_matrix[i, j])
                    regret_matrix[i, j] = regret / (second_best + 1e-10)
    
    # 4. Compute cluster density scores
    # Approximate clusters using k-means inspired grouping
    cluster_scores = np.ones_like(distance_matrix)
    if n >= 10:
        # Simple clustering: identify dense regions
        avg_dist = np.mean(distance_matrix)
        for i in range(n):
            # Count neighbors within 1.5x average distance
            close_count = np.sum(distance_matrix[i] < 1.5 * avg_dist) - 1
            for j in range(n):
                if i != j and distance_matrix[i, j] < avg_dist:
                    # Edges within clusters get bonus
                    density_bonus = 1.0 + 0.3 * (close_count / n)
                    cluster_scores[i, j] = density_bonus
    
    # 5. Compute MST connectivity bonus (approximate)
    mst_bonus = np.ones_like(distance_matrix)
    
    # Simple approximation: edges that are part of many short paths
    for i in range(n):
        # Find 2 closest nodes to i
        closest = np.argsort(distance_matrix[i])[1:3]
        for j in range(n):
            if i != j:
                # Check if edge (i,j) connects to i's nearest neighbors
                if j in closest:
                    mst_bonus[i, j] = 1.5
                # Bonus for edges that are shorter than average path through neighbors
                if k >= 2:
                    avg_via_neighbor = np.mean([distance_matrix[i, c] + distance_matrix[c, j] 
                                              for c in closest[:2]]) / 2
                    if distance_matrix[i, j] < 0.8 * avg_via_neighbor:
                        mst_bonus[i, j] = max(mst_bonus[i, j], 1.3)
    
    # 6. Combine all factors with adaptive weights
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            base_score = inv_dist[i, j]
            
            # Apply sparsification: only keep promising edges
            if nearest_mask[i, j] or nearest_mask[j, i]:
                # Edge is in nearest neighbor graph
                neighbor_factor = 1.5
            elif distance_matrix[i, j] < np.percentile(distance_matrix[i], 30):
                # Edge is in top 30% shortest for node i
                neighbor_factor = 1.2
            else:
                neighbor_factor = 0.3  # Strongly demote non-promising edges
            
            # Combine all factors
            score = (base_score * neighbor_factor * 
                    (1.0 + 0.4 * regret_matrix[i, j]) *
                    cluster_scores[i, j] *
                    mst_bonus[i, j])
            
            # Additional penalty for very long edges
            if distance_matrix[i, j] > 2.0 * np.median(distance_matrix):
                score *= 0.2
            
            H[i, j] = score
    
    # 7. Symmetrize and ensure diagonal is zero
    H = (H + H.T) / 2
    np.fill_diagonal(H, 0)
    
    # 8. Final sparsification: zero out weakest edges
    if n > 10:
        # Keep only top edges for each node
        keep_per_node = max(3, n // 3)
        H_sparse = np.zeros_like(H)
        for i in range(n):
            indices = np.argsort(H[i])[-keep_per_node:]
            H_sparse[i, indices] = H[i, indices]
        H = H_sparse
    
    # Normalize to reasonable range
    max_val = np.max(H)
    if max_val > 0:
        H = H / max_val
    
    return H
