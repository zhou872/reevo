import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with adaptive density-scaled regret,
    lightweight path efficiency lookahead, and optimized sparsification preserving
    high-regret connectivity.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        heuristics = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    epsilon = 1e-12
    
    # === 1. Base inverse distance ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Adaptive density-scaled regret ===
    # Compute local density for adaptive scaling
    dist_with_inf = distance_matrix.copy()
    np.fill_diagonal(dist_with_inf, np.inf)
    
    # Adaptive k for density estimation (scales with problem size)
    k_density = max(3, int(np.sqrt(n)))
    
    # Compute local density for each node (average distance to k_density nearest neighbors)
    density = np.zeros(n)
    for i in range(n):
        nearest_dists = np.partition(dist_with_inf[i], k_density)[:k_density]
        density[i] = np.mean(nearest_dists)
    
    # Normalize density to [0, 1]
    max_density = np.max(density) + epsilon
    density_norm = density / max_density
    
    # Compute proportional regret with density-based adaptive scaling
    regret_matrix = np.ones_like(distance_matrix)
    
    for i in range(n):
        # Get distances from node i
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        
        # Find two smallest distances
        sorted_idx = np.argsort(dist_i)[:2]
        d1, d2 = dist_i[sorted_idx[0]], dist_i[sorted_idx[1]]
        
        # Compute density-based scaling factor for node i
        # Lower density (sparser region) -> higher scaling -> emphasize regret more
        scale_i = 1.0 + (1.0 - density_norm[i])
        
        for j in range(n):
            if i == j:
                continue
                
            d_ij = distance_matrix[i, j]
            
            # Find the best alternative to j
            if j == sorted_idx[0]:
                best_alternative = d2
            else:
                best_alternative = d1
            
            if d_ij > epsilon:
                # Base proportional regret
                regret = (best_alternative - d_ij) / d_ij
                
                # Density-scaled regret: emphasize regret in sparse regions
                scaled_regret = regret * scale_i
                
                # Apply sigmoid to bound the effect
                regret_factor = 1.0 + 2.0 / (1.0 + np.exp(-scaled_regret)) - 1.0
                regret_matrix[i, j] = np.clip(regret_factor, 0.5, 2.5)
    
    # === 3. Lightweight one-step lookahead path efficiency ===
    efficiency_factor = np.ones_like(distance_matrix)
    
    # Consider top L neighbors for lookahead (lightweight computation)
    L = min(5, n - 2)
    if L > 0:
        # Get nearest neighbors for each node
        nearest_indices = np.zeros((n, L), dtype=int)
        for i in range(n):
            dist_i = distance_matrix[i].copy()
            dist_i[i] = np.inf
            nearest = np.argpartition(dist_i, L)[:L]
            nearest_indices[i, :] = nearest
        
        # Compute efficiency factor for each edge
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                d_ij = distance_matrix[i, j]
                best_alternative_cost = np.inf
                
                # Check alternative paths through neighbors of i and j
                for k in nearest_indices[i]:
                    if k != j:
                        alt_cost = distance_matrix[i, k] + distance_matrix[k, j]
                        if alt_cost < best_alternative_cost:
                            best_alternative_cost = alt_cost
                
                for k in nearest_indices[j]:
                    if k != i:
                        alt_cost = distance_matrix[j, k] + distance_matrix[k, i]
                        if alt_cost < best_alternative_cost:
                            best_alternative_cost = alt_cost
                
                if best_alternative_cost < np.inf:
                    # Efficiency ratio: direct vs best alternative path
                    efficiency_ratio = d_ij / (best_alternative_cost + epsilon)
                    
                    # Penalize edges with inefficient alternatives
                    # Use smooth penalty function: 1 - exp(-(1 - ratio)) for ratio < 1
                    if efficiency_ratio < 1.0:
                        penalty = 1.0 - np.exp(-(1.0 - efficiency_ratio))
                        efficiency_factor[i, j] = 1.0 - 0.5 * penalty
                    else:
                        # Slight bonus for highly efficient edges
                        efficiency_factor[i, j] = 1.0 + 0.1 * (efficiency_ratio - 1.0)
    
    # === 4. Mutual nearest neighbor boost ===
    mutual_boost = np.ones_like(distance_matrix)
    
    # Adaptive k for mutual neighbor detection
    k_mutual = max(3, int(np.sqrt(n)))
    
    # Find k-nearest neighbors for each node
    nn_indices = np.zeros((n, k_mutual), dtype=int)
    for i in range(n):
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        nearest = np.argpartition(dist_i, k_mutual)[:k_mutual]
        nn_indices[i, :] = nearest
    
    # Apply mutual nearest neighbor boost
    for i in range(n):
        for j in nn_indices[i]:
            if i == j:
                continue
            if np.any(nn_indices[j] == i):
                # Boost proportional to inverse distance and scaled by density
                closeness = 1.0 / (distance_matrix[i, j] + epsilon)
                density_factor = 1.0 + (1.0 - density_norm[i]) * (1.0 - density_norm[j])
                mutual_boost[i, j] = 1.0 + 0.8 * closeness * density_factor
    
    # === 5. Combine components multiplicatively ===
    heuristics = inv_dist.copy()
    heuristics *= regret_matrix
    heuristics *= efficiency_factor
    heuristics *= mutual_boost
    
    # === 6. Optimized sparsification preserving high-regret connectivity ===
    # Compute edge importance scores based on regret-adjusted heuristic
    importance = heuristics * (1.0 + 0.5 * (regret_matrix - 1.0))
    
    # Adaptive sparsification parameters
    base_k = max(3, min(10, n // 4))
    
    # Ensure connectivity by preserving MST edges
    if n > 2:
        mst = minimum_spanning_tree(distance_matrix)
        mst_dense = mst.toarray()
        mst_mask = (mst_dense > 0) | (mst_dense.T > 0)
    
    for i in range(n):
        row_importance = importance[i].copy()
        
        # Always keep:
        # 1. Mutual nearest neighbors
        keep_mask = np.zeros(n, dtype=bool)
        for j in nn_indices[i]:
            if i != j and np.any(nn_indices[j] == i):
                keep_mask[j] = True
        
        # 2. MST edges for connectivity
        if n > 2:
            keep_mask = keep_mask | mst_mask[i]
        
        # 3. Top edges by importance, with adaptive count based on local density
        # Denser regions: keep more edges; sparser regions: keep fewer but high-quality edges
        adaptive_k = int(base_k * (1.0 + 0.5 * density_norm[i]))
        adaptive_k = min(adaptive_k, n - 1)
        
        if adaptive_k > 0:
            # Get top adaptive_k edges not already kept
            non_kept = np.where(~keep_mask)[0]
            if len(non_kept) > 0:
                # Exclude self from consideration
                non_kept = non_kept[non_kept != i]
                if len(non_kept) > 0:
                    # Get top edges among non-kept
                    top_k = min(adaptive_k, len(non_kept))
                    top_indices = non_kept[np.argsort(row_importance[non_kept])[-top_k:]]
                    keep_mask[top_indices] = True
        
        # 4. Ensure minimum connectivity: always keep at least 2 edges
        if np.sum(keep_mask) < 2:
            # Keep top 2 edges by heuristic value
            top2_idx = np.argsort(heuristics[i])[-2:]
            keep_mask[top2_idx] = True
            keep_mask[i] = False  # Exclude self
        
        # Apply sparsification mask
        heuristics[i, ~keep_mask] = 0
    
    # === 7. Symmetrization and normalization ===
    # Use geometric mean for symmetry (preserves zero entries)
    heuristics_sym = np.sqrt(heuristics * heuristics.T + epsilon)
    np.fill_diagonal(heuristics_sym, 0)
    
    # Ensure non-negativity and normalize
    heuristics_sym = np.maximum(heuristics_sym, 0)
    max_val = np.max(heuristics_sym)
    if max_val > epsilon:
        heuristics_sym = heuristics_sym / max_val
    
    return heuristics_sym
