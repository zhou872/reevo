import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with mutual nearest neighbor emphasis
    and dynamic sparsification. Returns a matrix where higher values indicate more
    promising edges for inclusion in TSP solutions.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    epsilon = 1e-12
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # 1. Mutual Nearest Neighbor Emphasis
    k_mutual = max(3, int(np.sqrt(n)))
    mutual_nn_mask = np.zeros((n, n), dtype=bool)
    
    # Get nearest neighbors for each node
    nearest_indices = np.argsort(distance_matrix, axis=1)[:, 1:k_mutual+1]
    
    # Create boolean mask of nearest neighbors
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        nn_mask[i, nearest_indices[i]] = True
    
    # Mutual nearest neighbor: i in j's k_mutual nearest AND j in i's k_mutual nearest
    mutual_nn_mask = nn_mask & nn_mask.T
    
    # 2. Enhanced MST integration using scipy's efficient implementation
    mst_edges = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = mst_edges > 0
    
    # 3. Combine factors with calibrated weights
    heuristics = inv_dist.copy()
    
    # Strong boost for mutual nearest neighbors (bidirectional preference)
    heuristics[mutual_nn_mask] *= 2.5
    
    # Moderate boost for MST edges
    heuristics[mst_mask] *= 1.8
    
    # 4. Dynamic Sparsification with Adaptive Thresholds
    # First, ensure each node has at least 2 connections for feasibility
    min_connections = 2
    
    # Dynamic sparsification thresholds
    for i in range(n):
        row = heuristics[i].copy()
        
        # Keep at least min_connections edges
        if np.count_nonzero(row) > min_connections:
            # Adaptive k: ensure we keep enough edges while being selective
            adaptive_k = min(10, max(3, n // 4, min_connections))
            keep_count = max(min_connections, adaptive_k)
            
            # Find threshold for top keep_count edges
            if keep_count < n - 1:
                # Use partition to efficiently find threshold
                threshold_idx = -keep_count - 1
                threshold = np.partition(row, threshold_idx)[threshold_idx]
                row[row < threshold] = 0
        
        heuristics[i] = row
    
    # 5. Connectivity-preserving sparsification
    # Ensure symmetry: if edge is kept by either endpoint, keep it
    heuristics_sym = np.maximum(heuristics, heuristics.T * 0.8)
    
    # Additional global sparsification: remove edges below global percentile
    non_zero_vals = heuristics_sym[heuristics_sym > 0]
    if len(non_zero_vals) > 0:
        global_threshold = np.percentile(non_zero_vals, 25)  # Keep top 75%
        heuristics_sym[heuristics_sym < global_threshold] = 0
    
    # Final connectivity check: ensure each node has at least min_connections
    for i in range(n):
        row_nonzero = np.nonzero(heuristics_sym[i])[0]
        if len(row_nonzero) < min_connections:
            # Add back strongest missing edges
            row_vals = heuristics[i].copy()
            row_vals[row_nonzero] = 0  # Zero out already selected
            if np.any(row_vals > 0):
                # Add strongest edges to reach min_connections
                needed = min_connections - len(row_nonzero)
                top_indices = np.argsort(row_vals)[-needed:]
                for idx in top_indices:
                    if row_vals[idx] > 0:
                        heuristics_sym[i, idx] = row_vals[idx]
                        heuristics_sym[idx, i] = max(heuristics_sym[idx, i], row_vals[idx] * 0.8)
    
    np.fill_diagonal(heuristics_sym, 0)
    
    return heuristics_sym
