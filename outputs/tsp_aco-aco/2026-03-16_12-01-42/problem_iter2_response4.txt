import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge promise heuristics with two-step lookahead and
    graph structure-adaptive parameter optimization.
    
    Features:
    1. Two-step lookahead path potential scoring
    2. Graph topology-based adaptive weight adjustment
    3. Hybrid sparsification with dynamic thresholds
    4. Regret computation with penalty for distant alternatives
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        # Trivial cases
        heuristics = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    # Preprocess: handle diagonal and ensure positive distances
    working_matrix = distance_matrix.copy()
    np.fill_diagonal(working_matrix, np.inf)
    working_matrix = np.maximum(working_matrix, 1e-12)
    
    # 1. Base inverse distance with stability
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = 1.0 / working_matrix
    inv_dist = np.nan_to_num(inv_dist, nan=0.0, posinf=0.0, neginf=0.0)
    
    # 2. Graph structure analysis for adaptive parameters
    # Compute graph statistics to adapt parameters
    avg_distance = np.mean(working_matrix[working_matrix < np.inf])
    std_distance = np.std(working_matrix[working_matrix < np.inf])
    cv_distance = std_distance / (avg_distance + 1e-12)  # Coefficient of variation
    
    # Adaptive parameters based on graph structure
    k_near = max(3, min(n // 4, int(5 / (1 + cv_distance))))  # More neighbors for uniform graphs
    lookahead_depth = 2 if n > 10 else 1  # Reduce lookahead for small graphs
    
    # 3. Enhanced regret computation with penalty for distant alternatives
    # Find k nearest neighbors for each node
    k_regret = min(5, n-1)
    sorted_indices = np.argsort(working_matrix, axis=1)
    sorted_distances = np.take_along_axis(working_matrix, sorted_indices, axis=1)
    
    # Compute regret as difference between 2nd nearest and nearest
    regret_base = (sorted_distances[:, 1] - sorted_distances[:, 0]) / (sorted_distances[:, 0] + 1e-12)
    
    # Additional penalty if 3rd nearest is much further (alternative quality drop)
    alt_quality = (sorted_distances[:, 2] - sorted_distances[:, 1]) / (sorted_distances[:, 1] + 1e-12)
    regret = regret_base * (1.0 + 0.3 * np.tanh(alt_quality))
    
    regret_matrix = regret[:, None] + regret[None, :]
    
    # 4. Two-step lookahead path potential scoring
    # For each edge (i,j), evaluate potential 3-node paths i-j-k
    lookahead_score = np.zeros_like(working_matrix)
    
    if lookahead_depth == 2 and n > 3:
        # Precompute nearest neighbors for efficient lookahead
        nearest_counts = min(8, n-2)
        nearest_for_lookahead = sorted_indices[:, 1:nearest_counts+1]  # Exclude self
        
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                # Find best continuation from j (excluding i)
                best_continuation = np.inf
                for idx in range(nearest_counts):
                    k = nearest_for_lookahead[j, idx]
                    if k == i:
                        continue
                    # Evaluate path i->j->k
                    path_score = working_matrix[i, j] + working_matrix[j, k]
                    if path_score < best_continuation:
                        best_continuation = path_score
                
                # Also consider path k->i->j for symmetry
                best_approach = np.inf
                for idx in range(nearest_counts):
                    k = nearest_for_lookahead[i, idx]
                    if k == j:
                        continue
                    path_score = working_matrix[k, i] + working_matrix[i, j]
                    if path_score < best_approach:
                        best_approach = path_score
                
                # Combined lookahead score (lower is better, so invert)
                if best_continuation < np.inf and best_approach < np.inf:
                    combined_lookahead = (best_continuation + best_approach) / 2.0
                    lookahead_score[i, j] = 1.0 / (combined_lookahead + 1e-12)
    
    # Normalize lookahead score if available
    if np.max(lookahead_score) > 0:
        lookahead_score = lookahead_score / np.max(lookahead_score)
    
    # 5. MST-based connectivity prior
    mst = minimum_spanning_tree(working_matrix).toarray()
    mst_mask = (mst > 0) | (mst.T > 0)
    
    # 6. Dynamic weight adjustment based on graph structure
    # More weight to lookahead for larger, more complex graphs
    if n > 20:
        lookahead_weight = 0.4
        regret_weight = 0.3
        inv_weight = 0.2
        mst_weight = 0.1
    else:
        lookahead_weight = 0.3
        regret_weight = 0.35
        inv_weight = 0.25
        mst_weight = 0.1
    
    # Adjust weights based on distance distribution uniformity
    if cv_distance < 0.3:  # Very uniform distances
        lookahead_weight *= 1.2
    elif cv_distance > 1.0:  # Highly varied distances
        regret_weight *= 1.2
    
    # 7. Combine components with dynamic weights
    heuristics = (
        inv_weight * inv_dist +
        regret_weight * regret_matrix * inv_dist +
        lookahead_weight * lookahead_score +
        mst_weight * mst_mask
    )
    
    # 8. Local neighborhood emphasis with adaptive k
    nearest_mask = np.zeros_like(working_matrix, dtype=bool)
    for i in range(n):
        indices = sorted_indices[i, 1:k_near+1]  # Exclude self
        nearest_mask[i, indices] = True
    
    # Boost edges in local neighborhood
    heuristics *= (1.0 + 0.6 * nearest_mask)
    
    # 9. Symmetrize for undirected TSP
    heuristics = (heuristics + heuristics.T) / 2.0
    
    # 10. Hybrid sparsification strategy
    # Strategy 1: Percentile-based threshold
    flat_heuristics = heuristics[heuristics > 0]
    if len(flat_heuristics) > 0:
        percentile_threshold = np.percentile(flat_heuristics, 100 - min(40, 1000//n))
        percentile_mask = heuristics > percentile_threshold
    else:
        percentile_mask = np.ones_like(heuristics, dtype=bool)
    
    # Strategy 2: Per-node relative threshold
    row_max = np.max(heuristics, axis=1, keepdims=True)
    relative_threshold = 0.6 * row_max
    relative_mask = heuristics > relative_threshold
    
    # Strategy 3: Ensure minimum connectivity
    min_connections = max(2, min(4, n // 5))
    connectivity_mask = np.zeros_like(heuristics, dtype=bool)
    
    for i in range(n):
        # Get top connections for each node
        row = heuristics[i].copy()
        row[i] = -np.inf  # Exclude self
        top_indices = np.argsort(row)[-min_connections:]
        connectivity_mask[i, top_indices] = True
        connectivity_mask[top_indices, i] = True
    
    # Combine sparsification strategies
    final_mask = (percentile_mask & relative_mask) | connectivity_mask | nearest_mask
    
    # Apply mask
    heuristics = heuristics * final_mask
    
    # 11. Final normalization and cleanup
    np.fill_diagonal(heuristics, 0)
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics = heuristics / max_val
    
    return heuristics
