import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection.
    Combines inverse distance with node centrality measures,
    connectivity patterns, and local density to prioritize
    edges likely to be in optimal tours.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return 1 / (distance_matrix + 1e-10)
    
    # Initialize heuristics matrix
    H = np.zeros((n, n))
    
    # 1. Base inverse distance (normalized)
    inv_dist = 1 / (distance_matrix + 1e-10)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Node centrality factor (nodes with many short connections are more central)
    # Calculate average distance from each node to all others
    avg_dist = np.mean(distance_matrix, axis=1)
    centrality = 1 / (avg_dist + 1e-10)
    centrality = centrality / np.max(centrality)
    
    # Centrality factor promotes edges between central nodes
    centrality_factor = centrality[:, None] * centrality[None, :]
    
    # 3. Nearest-neighbor connectivity pattern
    # For each node, identify top k nearest neighbors
    k = max(2, n // 10)  # Adaptive k based on problem size
    nn_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get indices of k+1 smallest distances (including self)
        indices = np.argpartition(distance_matrix[i], k+1)[:k+1]
        # Remove self from nearest neighbors
        indices = indices[indices != i]
        nn_mask[i, indices] = True
    
    # Symmetrize nearest-neighbor mask
    nn_mask = nn_mask | nn_mask.T
    
    # 4. Two-hop potential: edges that connect to nodes with good subsequent connections
    two_hop_potential = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Look at nodes reachable in two hops through edge (i,j)
                # Consider nodes that are close to both i and j
                mask = np.ones(n, dtype=bool)
                mask[[i, j]] = False
                # Average distance from third node to i and j
                third_node_dist = (distance_matrix[i, mask] + distance_matrix[j, mask]) / 2
                # Good if there are many close third nodes (forming triangles)
                two_hop_potential[i, j] = np.sum(1 / (third_node_dist + 1e-10))
    
    # Normalize two-hop potential
    if np.max(two_hop_potential) > 0:
        two_hop_potential = two_hop_potential / np.max(two_hop_potential)
    
    # 5. Local density bonus
    # For each edge, compute how many other nodes are close to both endpoints
    density_bonus = np.zeros((n, n))
    density_threshold = np.percentile(distance_matrix[distance_matrix > 0], 25)
    
    for i in range(n):
        for j in range(n):
            if i != j:
                # Count nodes within threshold distance of both i and j
                close_to_i = distance_matrix[i] < density_threshold
                close_to_j = distance_matrix[j] < density_threshold
                shared_close = np.sum(close_to_i & close_to_j) - 2  # Exclude i and j
                density_bonus[i, j] = shared_close
    
    # Normalize density bonus
    if np.max(density_bonus) > 0:
        density_bonus = density_bonus / np.max(density_bonus)
    
    # Combine all factors
    H = inv_dist * centrality_factor * (1 + two_hop_potential) * (1 + 0.5 * density_bonus)
    
    # 6. Sparsification: zero out unpromising edges
    # Keep edges that are either:
    # a) In nearest-neighbor set, OR
    # b) Have heuristic value above threshold
    
    # Calculate sparsity threshold (adaptive)
    if nn_mask.sum() > 0:
        nn_vals = H[nn_mask]
        threshold = np.percentile(nn_vals, 25)  # Lower quartile of nearest-neighbor edges
    else:
        non_zero = H[H > 0]
        threshold = np.percentile(non_zero, 50) if len(non_zero) > 0 else 0
    
    # Create sparsity mask
    keep_mask = nn_mask | (H > threshold)
    
    # Apply sparsity mask
    H = H * keep_mask
    
    # Ensure symmetry
    H = np.maximum(H, H.T)
    
    # Set diagonal to zero
    np.fill_diagonal(H, 0)
    
    # Normalize to [0, 1] range for consistency
    if np.max(H) > 0:
        H = H / np.max(H)
    
    return H
