import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.sparse.csgraph import connected_components

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced edge promise heuristics for TSP solution sampling.
    Incorporates clustering-based edge boosting via MST partitioning.
    """
    n = distance_matrix.shape[0]
    # Work on a copy to avoid modifying input
    dist = distance_matrix.copy()
    np.fill_diagonal(dist, np.inf)
    
    # 1. Base inverse distance
    with np.errstate(divide='ignore'):
        inv_dist = 1.0 / dist
    inv_dist[np.isinf(inv_dist)] = 0
    
    # 2. Symmetric regret weighting
    nearest = np.partition(dist, 2, axis=1)[:, :2]
    regret = (nearest[:, 1] - nearest[:, 0]) / (nearest[:, 0] + 1e-12)
    regret_matrix = regret[:, None] + regret[None, :]
    
    # 3. Node isolation normalization
    node_isolation = np.mean(dist, axis=1)
    isolation_factor = (node_isolation[:, None] + node_isolation[None, :]) / 2
    
    # 4. Combine base factors
    heuristics = (
        0.4 * inv_dist +                    # Base inverse distance
        0.3 * regret_matrix * inv_dist +    # Regret-weighted
        0.2 * inv_dist / isolation_factor   # Isolation-normalized
    )
    
    # 5. MST-based clustering bonus
    mst = minimum_spanning_tree(dist).toarray()
    # Make MST symmetric (undirected)
    mst_sym = np.maximum(mst, mst.T)
    
    # Extract MST edges and their weights
    edges = np.argwhere(mst_sym > 0)
    weights = mst_sym[mst_sym > 0]
    
    if len(weights) > 0:
        # Remove longest edges to form clusters (number of clusters ~ sqrt(n))
        k_clusters = max(2, int(np.sqrt(n)))
        if len(weights) >= k_clusters - 1:
            # Sort edges by weight descending and remove (k-1) longest
            sorted_idx = np.argsort(weights)[::-1]
            edges_to_remove = edges[sorted_idx[:k_clusters - 1]]
            
            # Create adjacency without removed edges
            mst_clustered = mst_sym.copy()
            mst_clustered[edges_to_remove[:, 0], edges_to_remove[:, 1]] = 0
            mst_clustered[edges_to_remove[:, 1], edges_to_remove[:, 0]] = 0
            
            # Find connected components (clusters)
            n_components, labels = connected_components(
                mst_clustered, directed=False, return_labels=True
            )
            
            # Bonus for intra-cluster edges
            same_cluster = labels[:, None] == labels[None, :]
            np.fill_diagonal(same_cluster, False)
            heuristics *= (1.0 + 0.3 * same_cluster)
    
    # 6. Local neighborhood emphasis (weaker than v1 to avoid overlap with clustering)
    k = max(3, n // 10)
    nearest_mask = np.zeros_like(dist, dtype=bool)
    for i in range(n):
        indices = np.argsort(dist[i])[:k]
        nearest_mask[i, indices] = True
    heuristics *= (1.0 + 0.3 * nearest_mask)
    
    # 7. Symmetrize
    heuristics = np.maximum(heuristics, heuristics.T)
    
    # 8. Adaptive sparsification
    mask = np.zeros_like(heuristics, dtype=bool)
    
    # Keep globally strong edges (top 50%)
    threshold = np.percentile(heuristics[heuristics > 0], 50)
    global_mask = heuristics > threshold
    
    # Keep locally strong edges (top 30% per node)
    local_threshold = 0.7 * np.max(heuristics, axis=1, keepdims=True)
    local_mask = heuristics > local_threshold
    
    # Keep either globally or locally strong edges
    mask = global_mask | local_mask
    
    # Ensure minimum connectivity: each node connects to at least 2 others
    for i in range(n):
        if np.sum(mask[i]) < 2:
            indices = np.argsort(heuristics[i])[-2:]
            mask[i, indices] = True
            mask[indices, i] = True
    
    heuristics = heuristics * mask
    np.fill_diagonal(heuristics, 0)
    
    # 9. Final normalization
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics /= max_val
    
    return heuristics
