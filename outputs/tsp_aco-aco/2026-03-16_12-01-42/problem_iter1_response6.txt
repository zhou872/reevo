import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge prioritization using local neighborhood scaling
    and adaptive sparsification.
    
    Args:
        distance_matrix: Symmetric square matrix of pairwise distances between nodes.
        
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges.
        Unpromising edges are set to zero via median-based sparsification.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # 1. Determine k for k-nearest neighbors (adaptive to problem size)
    k = min(5, n - 1)
    
    # 2. Compute k-th nearest neighbor distance for each node
    kth_dists = np.zeros(n)
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf  # Ignore self-distance
        kth_dists[i] = np.partition(row, k - 1)[k - 1]  # k-th smallest distance (1-indexed)
    
    # 3. Compute Gaussian kernel weights using local scaling
    heur = np.zeros((n, n))
    for i in range(n):
        for j in range(i + 1, n):  # Utilize symmetry
            if distance_matrix[i, j] == 0:
                continue
            # Scale by the smaller of the two nodes' k-th neighbor distances
            scale = min(kth_dists[i], kth_dists[j])
            # Gaussian kernel: decays with distance relative to local scale
            heur[i, j] = np.exp(-(distance_matrix[i, j] / scale) ** 2)
            heur[j, i] = heur[i, j]  # Symmetric
    
    # 4. Median-based sparsification
    non_diag = heur[~np.eye(n, dtype=bool)]
    if len(non_diag) > 0:
        median_val = np.median(non_diag)
        heur[heur < median_val] = 0
    
    # Ensure diagonal is zero
    np.fill_diagonal(heur, 0)
    
    return heur
