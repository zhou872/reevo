import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection with regret-based penalization
    and lookahead spatial averaging.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base inverse distance with stability epsilon ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Regret-based alternative cost penalization ===
    # For each edge (i,j), compute how much worse the best alternative is
    # This prioritizes edges with fewer good alternatives
    regret_boost = np.zeros_like(distance_matrix)
    
    # Compute regret for each node
    for i in range(n):
        # Get distances from node i to all others
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf  # Ignore self
        
        # Sort distances to find k-best alternatives
        k_regret = min(5, n - 1)  # Consider top 5 alternatives
        sorted_indices = np.argsort(dist_i)[:k_regret]
        sorted_distances = dist_i[sorted_indices]
        
        # Compute regret for each potential edge from i
        for j in range(n):
            if i == j:
                continue
                
            # Distance of edge (i,j)
            d_ij = distance_matrix[i, j]
            
            # Find best alternative to j among the k-best
            # (excluding j itself from consideration)
            alternatives = [d for idx, d in zip(sorted_indices, sorted_distances) 
                          if idx != j]
            
            if alternatives:
                best_alternative = min(alternatives)
                # Regret: how much worse is the best alternative?
                # Higher regret = edge is more unique/valuable
                regret = max(0, best_alternative - d_ij)
                # Normalize by average distance for scale invariance
                avg_dist = np.mean(sorted_distances)
                regret_boost[i, j] = 1.0 + 2.0 * (regret / (avg_dist + epsilon))
            else:
                regret_boost[i, j] = 1.0  # Neutral boost
    
    # === 3. Lookahead spatial averaging ===
    # Weight edges by the average quality of connections from their endpoints
    spatial_scores = np.zeros_like(distance_matrix)
    
    # Pre-compute node spatial characteristics
    node_spatial = np.zeros(n)
    k_spatial = min(7, n - 1)
    
    for i in range(n):
        # Get distances to other nodes
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Take k_spatial nearest neighbors
        nearest_dists = np.partition(dists, k_spatial)[:k_spatial]
        # Spatial score: harmonic mean of nearest distances
        # (emphasizes consistently good connections)
        with np.errstate(divide='ignore', invalid='ignore'):
            node_spatial[i] = k_spatial / np.sum(1 / (nearest_dists + epsilon))
    
    # Combine spatial characteristics of edge endpoints
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Harmonic mean of endpoint spatial scores
            # (edges between well-connected nodes are promising)
            spatial_scores[i, j] = 2 / (1/(node_spatial[i] + epsilon) + 
                                        1/(node_spatial[j] + epsilon))
    
    # Normalize spatial scores to [0.5, 1.5] range
    if np.max(spatial_scores) > 0:
        spatial_min, spatial_max = np.min(spatial_scores[spatial_scores > 0]), \
                                  np.max(spatial_scores[spatial_scores > 0])
        if spatial_max > spatial_min:
            spatial_scores = 0.5 + (spatial_scores - spatial_min) / \
                            (spatial_max - spatial_min)
    
    # === 4. Mutual nearest neighbor factor ===
    k_mutual = max(3, int(np.sqrt(n)))
    mutual_nn_boost = np.zeros_like(distance_matrix)
    
    # Find k-nearest neighbors for each node
    nn_indices = np.zeros((n, k_mutual), dtype=int)
    for i in range(n):
        # Get k+1 smallest (including self)
        nearest = np.argpartition(distance_matrix[i], k_mutual + 1)[:k_mutual + 1]
        nearest = nearest[nearest != i]
        if len(nearest) > k_mutual:
            nearest = nearest[:k_mutual]
        nn_indices[i, :len(nearest)] = nearest
    
    # Boost mutual nearest neighbors
    for i in range(n):
        for j_idx in range(k_mutual):
            j = nn_indices[i, j_idx]
            if j == 0 and i == 0:
                continue
            if np.any(nn_indices[j] == i):
                # Stronger boost for mutual connections with high regret
                mutual_nn_boost[i, j] = 1.8 * regret_boost[i, j]
    
    # === 5. Efficient MST factor ===
    mst_boost = np.zeros_like(distance_matrix)
    if n > 1:
        visited = np.zeros(n, dtype=bool)
        visited[0] = True
        min_edge_to_tree = distance_matrix[0].copy()
        source_node = np.zeros(n, dtype=int)
        
        for _ in range(n - 1):
            unvisited = ~visited
            if not np.any(unvisited):
                break
                
            candidates = np.where(unvisited)[0]
            if len(candidates) > 0:
                min_idx = np.argmin(min_edge_to_tree[candidates])
                best_j = candidates[min_idx]
                
                i = source_node[best_j]
                # MST edges get boost weighted by spatial score
                mst_boost[i, best_j] = 2.0 * spatial_scores[i, best_j]
                mst_boost[best_j, i] = 2.0 * spatial_scores[best_j, i]
                visited[best_j] = True
                
                for v in range(n):
                    if not visited[v] and distance_matrix[best_j, v] < min_edge_to_tree[v]:
                        min_edge_to_tree[v] = distance_matrix[best_j, v]
                        source_node[v] = best_j
    
    # === 6. Degree balancing with popularity penalty ===
    degree_penalty = np.zeros_like(distance_matrix)
    for i in range(n):
        # Count how many nodes have i in their k-nearest
        popularity = 0
        for j in range(n):
            if i != j and np.any(nn_indices[j] == i):
                popularity += 1
        
        # Normalize and apply penalty: highly popular nodes get penalized
        norm_pop = min(1.0, popularity / (k_mutual * 2))
        penalty_val = 1.0 - 0.3 * norm_pop  # Moderate penalty
        degree_penalty[i, :] = penalty_val
    np.fill_diagonal(degree_penalty, 0)
    
    # === 7. Multiplicative factor fusion ===
    # Combine all factors multiplicatively for stronger signal differentiation
    heuristics = inv_dist.copy()
    heuristics *= (1.0 + mutual_nn_boost)  # Additive then multiplicative
    heuristics *= (1.0 + mst_boost)
    heuristics *= regret_boost  # Regret-based adjustment
    heuristics *= spatial_scores  # Spatial lookahead weighting
    heuristics *= degree_penalty  # Degree balancing
    
    # === 8. Adaptive sparsification with regret-aware threshold ===
    adaptive_k = min(12, max(3, n // 4))
    
    for i in range(n):
        row = heuristics[i].copy()
        
        if adaptive_k < n - 1:
            # Sort values and get threshold for top-k
            sorted_row = np.sort(row)
            threshold_idx = n - adaptive_k - 1
            base_threshold = sorted_row[threshold_idx]
            
            # Adjust threshold based on regret: keep edges with high regret even if below threshold
            regret_adjusted = row.copy()
            for j in range(n):
                if i != j:
                    # Boost edges with high regret
                    regret_adjusted[j] *= min(2.0, 1.0 + (regret_boost[i, j] - 1.0) * 0.5)
            
            # Determine which edges to keep
            keep_mask = np.zeros(n, dtype=bool)
            
            # Always keep mutual nearest neighbors
            for j_idx in range(k_mutual):
                j = nn_indices[i, j_idx]
                if j != 0 or i != 0:
                    if np.any(nn_indices[j] == i):
                        keep_mask[j] = True
            
            # Keep top edges by regret-adjusted score
            top_k_indices = np.argsort(regret_adjusted)[-adaptive_k:]
            keep_mask[top_k_indices] = True
            
            # Ensure at least 2 edges are kept
            if np.sum(keep_mask) < 2:
                top2_indices = np.argsort(row)[-2:]
                keep_mask[top2_indices] = True
            
            # Apply mask
            row[~keep_mask] = 0
            heuristics[i] = row
    
    # === 9. Symmetrization with preference for mutual decisions ===
    symmetric_heuristics = np.maximum(heuristics, heuristics.T * 0.8)
    
    # === 10. Final cleanup and normalization ===
    np.fill_diagonal(symmetric_heuristics, 0)
    
    # Ensure non-negativity and scale for consistency
    symmetric_heuristics = np.maximum(symmetric_heuristics, 0)
    if np.max(symmetric_heuristics) > 0:
        symmetric_heuristics = symmetric_heuristics / np.max(symmetric_heuristics)
    
    return symmetric_heuristics
