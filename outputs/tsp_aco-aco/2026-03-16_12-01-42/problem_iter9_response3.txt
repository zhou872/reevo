import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge prioritization with normalized harmonic scoring,
    adaptive regret order, multiplicative fusion, and adaptive sparsification.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base inverse distance ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Normalized Spatial Harmonic Scoring ===
    harmonic_scores = np.ones_like(distance_matrix)
    
    if n > 1:
        # Adaptive neighborhood size
        k_harmonic = max(3, min(10, int(np.sqrt(n))))
        
        # Compute harmonic means for each node
        node_harmonics = np.zeros(n)
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            if len(dists) >= k_harmonic:
                nearest_idx = np.argpartition(dists, k_harmonic)[:k_harmonic]
                nearest_dists = dists[nearest_idx]
                if np.any(nearest_dists > 0):
                    node_harmonics[i] = k_harmonic / np.sum(1 / (nearest_dists + epsilon))
        
        # Compute edge harmonic scores
        for i in range(n):
            for j in range(i + 1, n):
                if node_harmonics[i] > 0 and node_harmonics[j] > 0:
                    # Harmonic combination of node harmonics
                    combined_harmonic = 2 / (1/(node_harmonics[i] + epsilon) + 
                                           1/(node_harmonics[j] + epsilon))
                    
                    # Edge's harmonic score relative to its distance
                    if distance_matrix[i, j] > 0:
                        ratio = combined_harmonic / distance_matrix[i, j]
                        # Normalize to [0.5, 1.5] using logistic transform
                        base_score = 0.5 + 1.0 / (1 + np.exp(-(ratio - 1.0)))
                        harmonic_scores[i, j] = harmonic_scores[j, i] = base_score
    
    # === 3. Dynamic Regret Order Adaptation ===
    regret_scores = np.ones_like(distance_matrix)
    
    if n > 2:
        # Compute coefficient of variation for adaptive order selection
        off_diag_mask = ~np.eye(n, dtype=bool)
        global_dists = distance_matrix[off_diag_mask]
        if len(global_dists) > 0:
            cv = np.std(global_dists) / (np.mean(global_dists) + epsilon)
            
            # Select regret order based on coefficient of variation
            if cv < 0.3:
                regret_order = 2  # Uniform distances
            elif cv < 0.7:
                regret_order = 3  # Moderate variation
            else:
                regret_order = 4  # High variation
        else:
            regret_order = 2
        
        # Compute regret scores
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            if len(dists) >= regret_order:
                # Get regret_order best distances
                k_best_idx = np.argpartition(dists, regret_order)[:regret_order]
                k_best_dists = dists[k_best_idx]
                
                # Sort the k best
                sorted_idx = np.argsort(k_best_dists)
                sorted_dists = k_best_dists[sorted_idx]
                sorted_nodes = k_best_idx[sorted_idx]
                
                # Base regret for each edge
                for rank, j in enumerate(sorted_nodes):
                    if rank == 0:
                        # Best edge gets base score
                        regret_scores[i, j] = 1.2
                    else:
                        # Regret compared to best edge
                        regret = (sorted_dists[rank] - sorted_dists[0]) / (sorted_dists[0] + epsilon)
                        # Apply spread regularization
                        penalty = np.exp(-regret / (cv + 0.5))
                        regret_scores[i, j] = 1.0 + 0.2 * penalty
    
    # === 4. Multiplicative Factor Fusion ===
    # Combine components multiplicatively
    heuristics = inv_dist * harmonic_scores * regret_scores
    np.fill_diagonal(heuristics, 0)
    
    # === 5. MST Reinforcement ===
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            mst_present = (mst_dense > 0) | (mst_dense.T > 0)
            
            # Boost MST edges
            heuristics[mst_present] *= 1.3
        except:
            pass
    
    # === 6. Adaptive Sparsification ===
    for i in range(n):
        row = heuristics[i].copy()
        
        # Determine sparsity level based on problem size
        if n <= 20:
            keep_top = max(5, n // 2)
        elif n <= 100:
            keep_top = max(10, n // 3)
        else:
            keep_top = max(20, n // 4)
        
        # Keep top edges
        keep_top = min(keep_top, n - 1)
        if keep_top > 0:
            threshold = np.partition(row, -keep_top)[-keep_top]
            mask = row >= threshold
            
            # Also preserve MST edges if they exist
            if n > 2:
                try:
                    mst_i = minimum_spanning_tree(distance_matrix).toarray()[i] > 0
                    mask = mask | mst_i
                except:
                    pass
            
            row[~mask] = 0
            heuristics[i] = row
    
    # === 7. Symmetrization and Normalization ===
    # Symmetrize by geometric mean (multiplicative symmetry)
    heuristics_sym = np.sqrt(heuristics * heuristics.T)
    np.fill_diagonal(heuristics_sym, 0)
    
    # Final normalization to [0, 1]
    max_val = np.max(heuristics_sym)
    if max_val > 0:
        heuristics_sym = heuristics_sym / max_val
    
    return heuristics_sym
