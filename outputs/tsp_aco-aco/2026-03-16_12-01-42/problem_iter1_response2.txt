import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Compute heuristic indicators for each edge in a TSP instance.
    
    Parameters:
    - distance_matrix: 2D numpy array of shape (n, n) representing distances between nodes.
    
    Returns:
    - heuristic_matrix: 2D numpy array of same shape, where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    D = distance_matrix.astype(float)
    
    # Step 1: Set diagonal to infinity to avoid self-loops in computations
    np.fill_diagonal(D, np.inf)
    
    # Step 2: Inverse distance as base (handle division by infinity)
    inv_D = 1 / D
    inv_D = np.where(np.isinf(inv_D), 0, inv_D)
    
    # Step 3: Compute average distance per node (excluding self)
    avg_dist = np.array([np.mean(D[i, ~np.isinf(D[i])]) for i in range(n)])
    
    # Step 4: Symmetric local average ratio factor
    # For edge (i,j), use (avg_i + avg_j) / (2 * D[i,j])
    avg_i = avg_dist.reshape(-1, 1)
    avg_j = avg_dist.reshape(1, -1)
    with np.errstate(divide='ignore', invalid='ignore'):
        F_sym = (avg_i + avg_j) / (2 * D)
    F_sym = np.where(np.isinf(F_sym) | np.isnan(F_sym), 0, F_sym)
    
    # Step 5: Nearest neighbor bonus (asymmetric then symmetrized)
    k = min(5, n - 1)
    NN_bonus = np.ones((n, n))
    for i in range(n):
        # Get k smallest distances from i (excluding infinity at diagonal)
        valid_indices = np.where(~np.isinf(D[i]))[0]
        if len(valid_indices) > 0:
            k_actual = min(k, len(valid_indices))
            # Argsort on the valid part
            sorted_idx = np.argsort(D[i, valid_indices])[:k_actual]
            nearest = valid_indices[sorted_idx]
            NN_bonus[i, nearest] = 1.5
    # Symmetrize: if either direction has bonus, both get it
    NN_bonus_sym = np.maximum(NN_bonus, NN_bonus.T)
    
    # Step 6: Minimum Spanning Tree bonus (only if matrix is symmetric)
    def prim_mst(dist_mat):
        """Prim's algorithm for MST on a dense graph."""
        m = dist_mat.shape[0]
        visited = [False] * m
        visited[0] = True
        edges = []
        for _ in range(m - 1):
            min_dist = np.inf
            min_edge = None
            for i in range(m):
                if visited[i]:
                    for j in range(m):
                        if not visited[j] and dist_mat[i, j] < min_dist:
                            min_dist = dist_mat[i, j]
                            min_edge = (i, j)
            if min_edge:
                edges.append(min_edge)
                visited[min_edge[1]] = True
        return edges
    
    # Check symmetry (within relative tolerance)
    is_symmetric = np.allclose(D, D.T, rtol=1e-5)
    MST_bonus = np.ones((n, n))
    if is_symmetric and n > 1:
        mst_edges = prim_mst(D)
        for (i, j) in mst_edges:
            MST_bonus[i, j] = 2.0
            MST_bonus[j, i] = 2.0
    
    # Step 7: Combine factors element-wise
    H = inv_D * F_sym * NN_bonus_sym * MST_bonus
    
    # Step 8: Sparsification based on distance threshold
    # For each edge (i,j), if distance > 2 * max(avg_i, avg_j), set to zero
    threshold_matrix = 2 * np.maximum(avg_i, avg_j)
    # Create a mask of edges to keep (distance not too large)
    mask = D <= threshold_matrix
    # Apply mask (set to zero where mask is False)
    H = H * mask
    # Also set diagonal to zero (already zero from inv_D, but ensure)
    np.fill_diagonal(H, 0)
    
    return H
