import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics with adaptive local scaling, global structure integration,
    and rank-based sparsification.
    
    Args:
        distance_matrix: Symmetric square matrix of pairwise distances between nodes.
        
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges.
        Unpromising edges are set to zero via rank-based sparsification.
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        return np.zeros_like(distance_matrix)
    
    # 1. Dynamic k selection based on problem size
    # Use logarithmic growth with upper bound
    k = max(2, min(8, int(np.log2(n)) + 1))
    
    # 2. Compute local density scales using k-nearest neighbors
    local_scales = np.zeros(n)
    sorted_indices = np.argsort(distance_matrix, axis=1)
    
    for i in range(n):
        # Get distances to k nearest neighbors (excluding self)
        neighbor_dists = distance_matrix[i, sorted_indices[i, 1:k+1]]
        local_scales[i] = np.mean(neighbor_dists) if len(neighbor_dists) > 0 else 1.0
    
    # 3. Global structure integration: Minimum Spanning Tree edges
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_mask = (mst + mst.T) > 0
    
    # 4. Compute base heuristic with adaptive scaling and global bonus
    heur = np.zeros((n, n))
    scale_matrix = np.maximum(local_scales[:, None], local_scales[None, :])
    
    # Gaussian kernel with local scaling
    with np.errstate(divide='ignore', invalid='ignore'):
        scaled_dist = distance_matrix / scale_matrix
        heur = np.exp(-scaled_dist ** 2)
    
    # Boost MST edges with diminishing factor based on problem size
    mst_bonus = 0.5 / np.log2(n + 1)  # Bonus decreases with problem size
    heur[mst_mask] *= (1 + mst_bonus)
    
    # Zero diagonal and ensure symmetry
    np.fill_diagonal(heur, 0)
    heur = (heur + heur.T) / 2
    
    # 5. Rank-based sparsification (more robust than median)
    if n > 3:
        # Get upper triangular values (excluding diagonal)
        triu_indices = np.triu_indices(n, 1)
        values = heur[triu_indices]
        
        if len(values) > 0:
            # Sort values and keep top 2n (approximately O(n) edges)
            n_keep = min(2 * n, len(values))
            sorted_values = np.sort(values)[::-1]  # Descending
            
            if n_keep < len(values):
                threshold = sorted_values[n_keep - 1]
                # Apply sparsification symmetrically
                mask = heur < threshold
                heur[mask] = 0
    
    return heur
