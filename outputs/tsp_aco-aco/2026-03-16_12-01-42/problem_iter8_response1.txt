import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge selection heuristics with adaptive regret windows
    and cluster-aware symmetry breaking for improved performance on
    clustered node distributions.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base inverse distance ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Adaptive regret window based on graph density ===
    # Estimate graph density using coefficient of variation
    if n > 1:
        # Use upper triangle distances
        triu_indices = np.triu_indices(n, k=1)
        dist_values = distance_matrix[triu_indices]
        if len(dist_values) > 0:
            dist_mean = np.mean(dist_values)
            dist_std = np.std(dist_values)
            dist_cv = dist_std / (dist_mean + epsilon)
            
            # Higher CV means more clustered/spread distribution
            # Use larger regret window for uniform distributions
            # Smaller window for clustered distributions
            base_k = max(3, min(10, int(np.sqrt(n))))
            if dist_cv < 0.3:  # Uniform distances
                k_regret = min(base_k + 3, n - 1)
            elif dist_cv < 0.7:  # Moderate variation
                k_regret = base_k
            else:  # High variation (clustered)
                k_regret = max(3, base_k - 2)
        else:
            k_regret = min(5, n - 1)
    else:
        k_regret = 3
    
    # Regret computation with adaptive window
    regret_boost = np.zeros_like(distance_matrix)
    
    # Pre-sort distances for efficiency
    sorted_indices = np.argsort(distance_matrix + np.eye(n) * np.inf, axis=1)
    
    for i in range(n):
        # Get k_regret+1 nearest neighbors
        k_actual = min(k_regret, n - 1)
        nearest = sorted_indices[i, :k_actual + 1]
        nearest = nearest[nearest != i]  # Remove self
        
        if len(nearest) == 0:
            continue
            
        # Compute average distance to nearest neighbors
        nearest_dists = distance_matrix[i, nearest]
        avg_dist = np.mean(nearest_dists)
        
        for j in range(n):
            if i == j:
                continue
                
            d_ij = distance_matrix[i, j]
            
            # Find best alternative among nearest neighbors (excluding j)
            alternatives = [distance_matrix[i, k] for k in nearest if k != j]
            
            if alternatives:
                best_alt = np.min(alternatives)
                # Compute normalized regret
                regret = max(0, best_alt - d_ij)
                # Scale regret by average distance for robustness
                normalized_regret = regret / (avg_dist + epsilon)
                # Non-linear transformation for stronger signal
                regret_boost[i, j] = 1.0 + 2.0 * np.tanh(normalized_regret)
            else:
                regret_boost[i, j] = 1.0
    
    # === 3. Lightweight cluster detection for symmetry breaking ===
    cluster_aware = np.ones_like(distance_matrix)
    
    if n > 10:
        # Simple density-based clustering
        k_density = min(7, n - 1)
        node_density = np.zeros(n)
        
        for i in range(n):
            # Get distances to k_density nearest neighbors
            nearest_indices = sorted_indices[i, 1:k_density + 1]
            nearest_dists = distance_matrix[i, nearest_indices]
            # Use harmonic mean for density (more robust to outliers)
            with np.errstate(divide='ignore', invalid='ignore'):
                node_density[i] = k_density / np.sum(1.0 / (nearest_dists + epsilon))
        
        # Identify potential cluster boundaries
        density_std = np.std(node_density)
        if density_std > epsilon:
            # Normalize densities
            norm_density = (node_density - np.min(node_density)) / \
                          (np.max(node_density) - np.min(node_density) + epsilon)
            
            # Boost edges that connect nodes with similar densities
            # (within same cluster) OR very different densities
            # (inter-cluster connections)
            for i in range(n):
                for j in range(i + 1, n):
                    density_diff = abs(norm_density[i] - norm_density[j])
                    d_ij = distance_matrix[i, j]
                    
                    # Factor 1: Boost edges within same cluster
                    if density_diff < 0.2:  # Similar densities
                        # Strong boost for short intra-cluster edges
                        max_local_dist = np.max([
                            node_density[i], node_density[j]
                        ])
                        intra_factor = 1.0 + 1.5 * np.exp(-d_ij / (max_local_dist + epsilon))
                        cluster_aware[i, j] = cluster_aware[j, i] = intra_factor
                    
                    # Factor 2: Moderate boost for essential inter-cluster connections
                    elif density_diff > 0.6:  # Very different densities
                        # Reward edges that are not excessively long
                        avg_density = (node_density[i] + node_density[j]) / 2
                        inter_factor = 1.0 + 0.8 * np.exp(-d_ij / (2 * avg_density + epsilon))
                        cluster_aware[i, j] = cluster_aware[j, i] = inter_factor
    
    # === 4. Harmonic spatial scoring with adaptive neighborhood ===
    spatial_scores = np.ones_like(distance_matrix)
    
    # Adaptive neighborhood size based on local density variation
    if n > 5:
        for i in range(n):
            # Get distances to all other nodes
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            # Determine adaptive neighborhood size
            local_std = np.std(dists[dists < np.inf])
            avg_dist = np.mean(dists[dists < np.inf])
            local_cv = local_std / (avg_dist + epsilon)
            
            if local_cv < 0.3:  # Uniform neighborhood
                k_spatial = min(10, n - 1)
            else:  # Variable distances
                k_spatial = min(5, n - 1)
            
            # Get k_spatial nearest neighbors
            nearest_indices = np.argsort(dists)[:k_spatial]
            nearest_dists = dists[nearest_indices]
            
            # Compute harmonic mean as typical distance scale
            with np.errstate(divide='ignore', invalid='ignore'):
                harmonic_mean = len(nearest_dists) / np.sum(1.0 / (nearest_dists + epsilon))
            
            # Score edges based on deviation from typical scale
            for j in range(n):
                if i != j:
                    deviation = abs(distance_matrix[i, j] - harmonic_mean) / (harmonic_mean + epsilon)
                    # Reward edges close to typical distance
                    spatial_scores[i, j] *= np.exp(-deviation)
    
    # === 5. Efficient MST factor with cluster awareness ===
    mst_boost = np.ones_like(distance_matrix)
    if n > 1:
        # Prim's algorithm
        visited = np.zeros(n, dtype=bool)
        min_edge = np.full(n, np.inf)
        parent = np.full(n, -1)
        min_edge[0] = 0
        
        for _ in range(n):
            # Find minimum edge to tree
            u = -1
            min_val = np.inf
            for i in range(n):
                if not visited[i] and min_edge[i] < min_val:
                    min_val = min_edge[i]
                    u = i
            
            if u == -1:
                break
                
            visited[u] = True
            if parent[u] != -1:
                v = parent[u]
                # Boost MST edges with cluster awareness
                mst_boost[u, v] = mst_boost[v, u] = 2.0 * cluster_aware[u, v]
            
            # Update distances
            for v in range(n):
                if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                    min_edge[v] = distance_matrix[u, v]
                    parent[v] = u
    
    # === 6. Degree balancing with popularity awareness ===
    k_degree = max(3, int(np.sqrt(n)))
    popularity = np.zeros(n)
    
    # Count how many times each node appears in others' k-nearest
    for i in range(n):
        nearest = sorted_indices[i, 1:k_degree + 1]
        for j in nearest:
            popularity[j] += 1
    
    degree_penalty = np.ones_like(distance_matrix)
    if np.max(popularity) > 0:
        norm_pop = popularity / np.max(popularity)
        for i in range(n):
            # Penalize edges to very popular nodes
            penalty = 1.0 - 0.4 * norm_pop[i]
            degree_penalty[i, :] = penalty
            degree_penalty[:, i] = np.minimum(degree_penalty[:, i], penalty)
    
    np.fill_diagonal(degree_penalty, 0)
    
    # === 7. Multiplicative factor fusion ===
    heuristics = inv_dist.copy()
    heuristics *= regret_boost  # Adaptive regret
    heuristics *= spatial_scores  # Harmonic spatial scoring
    heuristics *= cluster_aware  # Cluster symmetry breaking
    heuristics *= mst_boost  # MST with cluster awareness
    heuristics *= degree_penalty  # Degree balancing
    
    # === 8. Adaptive sparsification with cluster-aware threshold ===
    adaptive_k = min(15, max(3, n // 3))
    
    for i in range(n):
        row = heuristics[i].copy()
        
        if adaptive_k < n - 1 and n > 3:
            # Boost edges that are both high regret AND cluster-appropriate
            adjusted_row = row.copy()
            for j in range(n):
                if i != j:
                    # Combine regret and cluster factors for adjustment
                    cluster_factor = cluster_aware[i, j]
                    regret_factor = min(2.0, 1.0 + (regret_boost[i, j] - 1.0) * 0.5)
                    adjusted_row[j] *= cluster_factor * regret_factor
            
            # Keep top edges by adjusted score
            top_k_indices = np.argsort(adjusted_row)[-adaptive_k:]
            keep_mask = np.zeros(n, dtype=bool)
            keep_mask[top_k_indices] = True
            
            # Always keep MST edges
            for j in range(n):
                if i != j and mst_boost[i, j] > 1.5:
                    keep_mask[j] = True
            
            # Ensure minimum connectivity
            if np.sum(keep_mask) < 2:
                top2_indices = np.argsort(row)[-2:]
                keep_mask[top2_indices] = True
            
            row[~keep_mask] = 0
            heuristics[i] = row
    
    # === 9. Symmetrization with cluster-aware preference ===
    symmetric_heuristics = np.zeros_like(heuristics)
    
    for i in range(n):
        for j in range(n):
            if i == j:
                symmetric_heuristics[i, j] = 0
            else:
                # Prefer mutual decisions, with cluster-aware weighting
                cluster_weight = cluster_aware[i, j]
                symmetric_heuristics[i, j] = max(
                    heuristics[i, j],
                    heuristics[j, i] * 0.9
                ) * cluster_weight
    
    # === 10. Final normalization ===
    np.fill_diagonal(symmetric_heuristics, 0)
    
    # Apply softmax per node with adaptive temperature
    for i in range(n):
        row = symmetric_heuristics[i]
        row_max = np.max(row)
        if row_max > 0:
            # Temperature based on row variance (lower variance = higher temperature)
            row_nonzero = row[row > 0]
            if len(row_nonzero) > 1:
                row_var = np.var(row_nonzero)
                temperature = max(0.5, 2.0 - row_var)
            else:
                temperature = 1.0
            
            exp_row = np.exp(row / temperature - np.max(row / temperature))
            symmetric_heuristics[i] = exp_row / np.sum(exp_row)
    
    # Ensure non-negativity and scale
    symmetric_heuristics = np.maximum(symmetric_heuristics, 0)
    global_max = np.max(symmetric_heuristics)
    if global_max > 0:
        symmetric_heuristics = symmetric_heuristics / global_max
    
    return symmetric_heuristics
