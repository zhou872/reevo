import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from scipy.spatial.distance import squareform
from scipy.cluster.hierarchy import fcluster, linkage

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge heuristics integrating global structural features,
    multi-step lookahead efficiency, and dynamic parameter adaptation.
    
    Parameters
    ----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns
    -------
    np.ndarray
        Heuristic matrix of same shape where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        heuristics = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    epsilon = 1e-12
    
    # === 1. DYNAMIC PARAMETER ADAPTATION ===
    # Analyze instance characteristics for parameter tuning
    off_diag = distance_matrix[~np.eye(n, dtype=bool)]
    global_mean = np.mean(off_diag)
    global_cv = np.std(off_diag) / (global_mean + epsilon)
    
    # Adaptive parameters based on instance size and variation
    if n <= 10:
        k_regret = 2
        L_lookahead = 3
        n_clusters = 1
    elif n <= 50:
        k_regret = 3
        L_lookahead = min(5, n - 2)
        n_clusters = max(2, int(np.sqrt(n)) // 2)
    else:
        k_regret = max(3, min(5, int(np.log2(n))))
        L_lookahead = min(8, n // 10)
        n_clusters = max(3, int(np.sqrt(n)))
    
    # Adjust based on coefficient of variation
    if global_cv < 0.3:
        # Uniform distances: emphasize global structure
        w_inv, w_regret, w_lookahead, w_cluster = 1.0, 0.9, 1.1, 1.2
    elif global_cv < 0.7:
        # Moderate variation: balanced approach
        w_inv, w_regret, w_lookahead, w_cluster = 1.1, 1.0, 1.0, 1.0
    else:
        # High variation: emphasize local regret
        w_inv, w_regret, w_lookahead, w_cluster = 0.9, 1.3, 0.9, 0.8
    
    # === 2. BASE INVERSE DISTANCE WITH ADAPTIVE SCALING ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 3. ADAPTIVE REGRET WITH DENSITY AWARENESS ===
    regret_matrix = np.ones_like(distance_matrix)
    
    # Compute local density for each node
    k_density = max(3, min(10, n // 5))
    dist_with_inf = distance_matrix.copy()
    np.fill_diagonal(dist_with_inf, np.inf)
    
    density = np.zeros(n)
    for i in range(n):
        nearest = np.partition(dist_with_inf[i], k_density)[:k_density]
        density[i] = np.mean(nearest)
    
    density_norm = density / (np.max(density) + epsilon)
    
    # Compute adaptive regret
    for i in range(n):
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        
        # Find k_regret+1 smallest distances
        k_actual = min(k_regret + 1, n - 1)
        if k_actual > 0:
            sorted_idx = np.argsort(dist_i)[:k_actual]
            sorted_dists = dist_i[sorted_idx]
            
            for j in range(n):
                if i == j:
                    continue
                
                d_ij = distance_matrix[i, j]
                if d_ij <= epsilon:
                    continue
                
                # Find best alternative (not j)
                best_alt = np.inf
                for idx, alt_j in enumerate(sorted_idx):
                    if alt_j != j:
                        best_alt = sorted_dists[idx]
                        break
                
                if best_alt < np.inf:
                    # Density-aware regret scaling
                    density_scale = 1.0 + 2.0 * (1.0 - density_norm[i])
                    regret = (best_alt - d_ij) / (d_ij + epsilon)
                    scaled_regret = np.tanh(regret * density_scale)
                    regret_matrix[i, j] = 1.0 + scaled_regret
    
    # === 4. MULTI-STEP LOOKAHEAD PATH EFFICIENCY ===
    efficiency_matrix = np.ones_like(distance_matrix)
    
    if L_lookahead > 0 and n > 3:
        # Precompute nearest neighbors for each node
        nn_indices = np.zeros((n, L_lookahead), dtype=int)
        for i in range(n):
            dist_i = distance_matrix[i].copy()
            dist_i[i] = np.inf
            if L_lookahead < n - 1:
                nn_indices[i] = np.argpartition(dist_i, L_lookahead)[:L_lookahead]
            else:
                nn_indices[i] = np.argsort(dist_i)[:L_lookahead]
        
        # Compute multi-step efficiency
        for i in range(n):
            for j in range(n):
                if i == j:
                    continue
                
                d_ij = distance_matrix[i, j]
                best_alternative = np.inf
                
                # Two-step paths (i -> a -> j)
                for a in nn_indices[i]:
                    if a != j:
                        two_step = distance_matrix[i, a] + distance_matrix[a, j]
                        if two_step < best_alternative:
                            best_alternative = two_step
                
                # Three-step paths (i -> a -> b -> j) for top neighbors
                if L_lookahead >= 2:
                    for a in nn_indices[i][:L_lookahead//2]:
                        if a == j:
                            continue
                        for b in nn_indices[j][:L_lookahead//2]:
                            if b != i and b != a:
                                three_step = (distance_matrix[i, a] + 
                                            distance_matrix[a, b] + 
                                            distance_matrix[b, j])
                                if three_step < best_alternative:
                                    best_alternative = three_step
                
                if best_alternative < np.inf and best_alternative > epsilon:
                    efficiency_ratio = d_ij / best_alternative
                    # Smooth efficiency score
                    if efficiency_ratio < 0.8:
                        efficiency_matrix[i, j] = 1.5
                    elif efficiency_ratio < 1.0:
                        efficiency_matrix[i, j] = 1.0 + 0.5 * (1.0 - efficiency_ratio)
                    else:
                        efficiency_matrix[i, j] = 1.0 / (1.0 + 0.1 * (efficiency_ratio - 1.0))
    
    # === 5. HIERARCHICAL CLUSTERING-BASED BONUS ===
    cluster_bonus = np.ones_like(distance_matrix)
    
    if n_clusters > 1 and n > n_clusters:
        try:
            # Perform hierarchical clustering
            condensed_dist = squareform(distance_matrix, checks=False)
            Z = linkage(condensed_dist, method='average')
            cluster_labels = fcluster(Z, n_clusters, criterion='maxclust')
            
            # Compute intra-cluster and inter-cluster bonuses
            for i in range(n):
                for j in range(n):
                    if i == j:
                        continue
                    
                    if cluster_labels[i] == cluster_labels[j]:
                        # Intra-cluster: boost based on relative position within cluster
                        cluster_mask = cluster_labels == cluster_labels[i]
                        cluster_size = np.sum(cluster_mask)
                        if cluster_size > 1:
                            # Normalize distance within cluster
                            cluster_dists = distance_matrix[i, cluster_mask]
                            cluster_dists = cluster_dists[cluster_dists > 0]
                            if len(cluster_dists) > 0:
                                mean_cluster_dist = np.mean(cluster_dists)
                                rel_dist = distance_matrix[i, j] / (mean_cluster_dist + epsilon)
                                # Boost closer intra-cluster edges more
                                bonus = 1.0 + 0.5 * np.exp(-rel_dist)
                                cluster_bonus[i, j] = bonus
                    else:
                        # Inter-cluster: mild penalty unless it's the best connection
                        # Find minimum distance between these clusters
                        mask_i = cluster_labels == cluster_labels[i]
                        mask_j = cluster_labels == cluster_labels[j]
                        inter_dists = distance_matrix[mask_i][:, mask_j]
                        min_inter = np.min(inter_dists)
                        if distance_matrix[i, j] <= min_inter + epsilon:
                            # Best connection between clusters - slight boost
                            cluster_bonus[i, j] = 1.1
                        else:
                            # Not the best connection - mild penalty
                            cluster_bonus[i, j] = 0.9
        except:
            # Fallback if clustering fails
            pass
    
    # === 6. STRUCTURAL CONVEXITY BONUS ===
    convexity_bonus = np.ones_like(distance_matrix)
    
    if n > 3:
        # Approximate convex hull edges using distance-based heuristic
        # Nodes with high average distance are likely on convex hull
        avg_distances = np.mean(distance_matrix, axis=1)
        hull_threshold = np.percentile(avg_distances, 75)
        hull_nodes = avg_distances >= hull_threshold
        
        # Boost edges between hull nodes
        for i in range(n):
            if hull_nodes[i]:
                for j in range(n):
                    if i != j and hull_nodes[j]:
                        # Bonus proportional to inverse distance
                        bonus = 1.0 + 0.3 / (1.0 + distance_matrix[i, j] / global_mean)
                        convexity_bonus[i, j] = bonus
        
        # Also boost MST edges (structural backbone)
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            mst_mask = (mst_dense > 0) | (mst_dense.T > 0)
            convexity_bonus[mst_mask] *= 1.2
        except:
            pass
    
    # === 7. COMBINE COMPONENTS WITH ADAPTIVE WEIGHTS ===
    heuristics = (inv_dist ** w_inv) * \
                 (regret_matrix ** w_regret) * \
                 (efficiency_matrix ** w_lookahead) * \
                 (cluster_bonus ** w_cluster) * \
                 convexity_bonus
    
    # === 8. CONNECTIVITY-PRESERVING SPARSIFICATION ===
    # Compute importance scores with multiple factors
    importance = heuristics.copy()
    importance *= 1.0 + 0.3 * (regret_matrix - 1.0)  # Emphasize high-regret edges
    importance *= 1.0 + 0.2 * (efficiency_matrix - 1.0)  # Emphasize efficient edges
    
    # Ensure MST connectivity
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            mst_mask = (mst_dense > 0) | (mst_dense.T > 0)
        except:
            mst_mask = np.zeros_like(distance_matrix, dtype=bool)
    else:
        mst_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Adaptive sparsification per node
    for i in range(n):
        row_importance = importance[i].copy()
        row_heuristics = heuristics[i].copy()
        
        # Always keep: MST edges and mutual nearest neighbors
        keep_mask = mst_mask[i].copy()
        
        # Add mutual nearest neighbors (within top L_lookahead)
        if L_lookahead > 0:
            my_nn = set(nn_indices[i][:L_lookahead])
            for j in my_nn:
                if i != j and j < n and i in set(nn_indices[j][:L_lookahead]):
                    keep_mask[j] = True
        
        # Adaptive number of additional edges based on local density
        base_keep = max(3, min(8, n // 5))
        density_factor = 1.0 + density_norm[i]  # Keep more in dense regions
        adaptive_keep = int(base_keep * density_factor)
        
        # Select top additional edges by importance
        candidate_mask = ~keep_mask & (np.arange(n) != i)
        candidates = np.where(candidate_mask)[0]
        
        if len(candidates) > 0:
            k_keep = min(adaptive_keep, len(candidates))
            top_indices = candidates[np.argsort(row_importance[candidates])[-k_keep:]]
            keep_mask[top_indices] = True
        
        # Ensure minimum connectivity
        if np.sum(keep_mask) < 2:
            top2_idx = np.argsort(row_heuristics)[-2:]
            keep_mask[top2_idx] = True
            keep_mask[i] = False
        
        # Apply sparsification
        heuristics[i, ~keep_mask] = 0
    
    # === 9. SYMMETRIZATION AND FINAL NORMALIZATION ===
    # Symmetric geometric mean (preserves zero entries)
    heuristics_sym = np.sqrt(heuristics * heuristics.T + epsilon)
    np.fill_diagonal(heuristics_sym, 0)
    
    # Final reinforcement of mutual high scores
    mutual_high = (heuristics > 0) & (heuristics.T > 0)
    heuristics_sym[mutual_high] *= 1.1
    
    # Ensure non-negativity and normalize to [0, 1]
    heuristics_sym = np.maximum(heuristics_sym, 0)
    max_val = np.max(heuristics_sym)
    if max_val > epsilon:
        heuristics_sym = heuristics_sym / max_val
    
    return heuristics_sym
