import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection with regret-based penalization
    and lookahead spatial averaging.
    
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Avoid division by zero and self-loops
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base inverse distance (closer nodes are generally better)
    inv_dist = 1.0 / safe_dist
    
    # 2. Nearest-neighbor gradient with adaptive k
    k = max(2, min(5, n // 4))
    nearest_neighbor_bonus = np.zeros_like(inv_dist)
    for i in range(n):
        sorted_idx = np.argsort(safe_dist[i])[:k]
        nearest_neighbor_bonus[i, sorted_idx] = 1.0
    
    symmetric_nn = (nearest_neighbor_bonus + nearest_neighbor_bonus.T) / 2
    
    # 3. Local connectivity clustering via shared neighbors
    cluster_bonus = np.zeros_like(inv_dist)
    for i in range(n):
        for j in range(i + 1, n):
            threshold_i = np.percentile(safe_dist[i], 30)
            threshold_j = np.percentile(safe_dist[j], 30)
            near_i = set(np.where(safe_dist[i] <= threshold_i)[0]) - {i}
            near_j = set(np.where(safe_dist[j] <= threshold_j)[0]) - {j}
            common = len(near_i.intersection(near_j))
            cluster_bonus[i, j] = cluster_bonus[j, i] = common / max(1, min(len(near_i), len(near_j)))
    
    # 4. Regret-based penalization: prioritize edges with few good alternatives
    regret_penalty = np.ones_like(inv_dist)
    for i in range(n):
        row = safe_dist[i]
        sorted_idx = np.argsort(row)
        
        if len(sorted_idx) >= 3:
            # Get 3 best distances for each node
            best_dist = row[sorted_idx[0]]
            second_best = row[sorted_idx[1]]
            third_best = row[sorted_idx[2]]
            
            # Calculate regret: high penalty if there are close alternatives
            regret_ratio = (second_best - best_dist) / (third_best - best_dist + 1e-10)
            
            # Apply penalty to edges that are not the best but have close alternatives
            for idx, j in enumerate(sorted_idx):
                if idx == 0:  # Best edge gets no penalty
                    regret_penalty[i, j] = 1.0
                else:
                    # Calculate how much worse this edge is than the best
                    relative_gap = (row[j] - best_dist) / (best_dist + 1e-10)
                    
                    # Apply higher penalty if there are close alternatives (low regret_ratio)
                    penalty = 1.0 - min(0.5, relative_gap * (1.0 - regret_ratio))
                    regret_penalty[i, j] = penalty
    
    # Symmetrize regret penalty
    regret_penalty = (regret_penalty + regret_penalty.T) / 2
    
    # 5. Lookahead spatial averaging: consider neighborhood consistency
    spatial_avg = np.zeros_like(inv_dist)
    lookahead_k = max(3, min(6, n // 3))
    
    for i in range(n):
        # Get lookahead neighbors for node i
        i_neighbors = np.argsort(safe_dist[i])[1:lookahead_k+1]  # Exclude self
        
        for j in range(n):
            if i == j:
                continue
                
            # Get lookahead neighbors for node j
            j_neighbors = np.argsort(safe_dist[j])[1:lookahead_k+1]
            
            # Calculate average inverse distance in the combined neighborhood
            neighborhood = set(i_neighbors) | set(j_neighbors) | {i, j}
            total = 0
            count = 0
            
            for a in neighborhood:
                for b in neighborhood:
                    if a != b:
                        total += 1.0 / safe_dist[a, b]
                        count += 1
            
            if count > 0:
                avg_value = total / count
                # Bonus if this edge is better than neighborhood average
                edge_value = 1.0 / safe_dist[i, j]
                spatial_avg[i, j] = edge_value / (avg_value + 1e-10)
    
    # Normalize spatial averaging factor
    max_spatial = spatial_avg.max()
    if max_spatial > 0:
        spatial_avg = spatial_avg / max_spatial
    
    # 6. Combine all factors with weights
    heuristics = (
        0.3 * inv_dist +           # Base inverse distance
        0.2 * symmetric_nn +       # Mutual nearest-neighbor bonus
        0.15 * cluster_bonus +     # Clustering bonus
        0.2 * regret_penalty +     # Regret-based penalty (acts as multiplier)
        0.15 * spatial_avg         # Lookahead spatial averaging
    )
    
    # Apply regret penalty multiplicatively
    heuristics = heuristics * regret_penalty
    
    # 7. Adaptive sparsification with regret consideration
    keep_percentile = 40
    threshold_matrix = np.zeros_like(heuristics)
    for i in range(n):
        row = heuristics[i]
        # Adjust threshold based on regret: keep more edges if alternatives are poor
        regret_i = 1.0 - regret_penalty[i].min()
        adjusted_percentile = keep_percentile * (1.0 + 0.3 * regret_i)
        adjusted_percentile = min(70, max(30, adjusted_percentile))
        
        threshold = np.percentile(row, 100 - adjusted_percentile)
        threshold_matrix[i] = threshold
    
    # Zero out edges below both nodes' thresholds
    mask = (heuristics >= threshold_matrix) | (heuristics >= threshold_matrix.T)
    heuristics = heuristics * mask
    
    # 8. Final normalization
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val * 10
    
    # Ensure diagonal is zero (no self-loops)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
