import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge heuristics with lightweight lookahead criticality assessment
    and adaptive feature combination. Maintains computational efficiency while
    incorporating global connectivity considerations.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix (non-negative)
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in TSP solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Dynamic Regret with Local Density Scaling ===
    # Emphasize edges where alternative costs are significantly higher
    regret_scores = np.ones_like(distance_matrix)
    
    # Precompute local densities for each node
    k_density = min(10, n - 1) if n > 1 else 1
    densities = np.zeros(n)
    
    for i in range(n):
        if n == 1:
            densities[i] = 1.0
            continue
            
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Get k_density smallest distances
        nearest_dists = np.partition(dists, k_density)[:k_density]
        densities[i] = np.mean(nearest_dists) + epsilon
    
    # Dynamic regret computation
    for i in range(n):
        if n <= 1:
            continue
            
        # Adaptive k based on local density (denser = more alternatives)
        k_regret = min(max(3, int(8 * (1 - densities[i] / (np.max(densities) + epsilon)))), n - 1)
        
        # Get sorted distances for this node
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        sorted_indices = np.argsort(dists_i)[:k_regret + 2]  # Extra for safety
        
        for j in range(n):
            if i == j:
                continue
                
            d_ij = distance_matrix[i, j]
            
            # Find best alternative excluding j
            alternatives = []
            for idx in sorted_indices:
                if idx != j and idx != i:
                    alternatives.append(distance_matrix[i, idx])
                    if len(alternatives) >= k_regret:
                        break
            
            if alternatives:
                best_alt = min(alternatives)
                # Regret ratio: how much worse is the best alternative?
                regret_ratio = max(0, (best_alt - d_ij) / (d_ij + epsilon))
                # Scale by density: stronger effect in dense regions
                density_factor = 1.0 + 0.5 * (densities[i] / (np.median(densities) + epsilon))
                regret_scores[i, j] = 1.0 + 1.5 * regret_ratio * density_factor
            else:
                regret_scores[i, j] = 1.0
    
    # === 2. Mutual K-Nearest Neighbor Boost ===
    # Boost edges that are mutually important at multiple scales
    connectivity_boost = np.ones_like(distance_matrix)
    
    # Multiple scales for robustness
    for k in [2, 3, 5]:
        if k >= n:
            continue
            
        # Find k-nearest neighbors for each node
        nn_masks = np.zeros((n, n), dtype=bool)
        for i in range(n):
            if n == 1:
                continue
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest_indices = np.argpartition(dists, k)[:k]
            nn_masks[i, nearest_indices] = True
        
        # Boost mutual nearest neighbors
        mutual_mask = nn_masks & nn_masks.T
        # Stronger boost for tighter neighborhoods
        boost_value = 1.0 + 0.5 / np.log2(k + 2)
        connectivity_boost[mutual_mask] *= boost_value
    
    # === 3. Lightweight Two-Step Edge Criticality ===
    # Assess how critical each edge is for local connectivity
    criticality_scores = np.ones_like(distance_matrix)
    
    if n > 2:
        for i in range(n):
            # Get two nearest neighbors for node i
            dists_i = distance_matrix[i].copy()
            dists_i[i] = np.inf
            nearest_i = np.argpartition(dists_i, 2)[:2]
            
            for j in range(n):
                if i == j:
                    continue
                    
                d_ij = distance_matrix[i, j]
                
                # Find best two-step alternative path i -> a -> j
                best_two_step = np.inf
                
                # Check via i's nearest neighbors (excluding j)
                for a in nearest_i:
                    if a != j:
                        two_step_cost = distance_matrix[i, a] + distance_matrix[a, j]
                        best_two_step = min(best_two_step, two_step_cost)
                
                # Criticality: how much better is direct edge?
                if best_two_step < np.inf:
                    criticality = max(0, (best_two_step - d_ij) / (d_ij + epsilon))
                    # Normalize criticality score
                    criticality_scores[i, j] = 1.0 + 0.8 * np.tanh(criticality)
    
    # === 4. Base Inverse Distance with Adaptive Scaling ===
    base_scores = 1.0 / (distance_matrix + epsilon)
    np.fill_diagonal(base_scores, 0)
    
    # Scale by global statistics if we have enough nodes
    if n > 2:
        # Use robust statistics
        upper_tri = distance_matrix[np.triu_indices(n, k=1)]
        if len(upper_tri) > 0:
            median_dist = np.median(upper_tri)
            q75_dist = np.percentile(upper_tri, 75)
            scale_factor = q75_dist / (median_dist + epsilon)
            base_scores *= np.clip(scale_factor, 0.5, 2.0)
    
    # === 5. Adaptive Feature Combination ===
    # Combine scores multiplicatively with minor weighting
    heuristics = base_scores.copy()
    heuristics *= regret_scores
    heuristics *= connectivity_boost
    heuristics *= criticality_scores
    
    # === 6. Global Connectivity Consideration ===
    # Ensure we don't ignore potentially important long edges
    if n > 3:
        # Compute MST to identify globally important edges
        mst = minimum_spanning_tree(distance_matrix).toarray()
        mst_binary = (mst > 0) | (mst.T > 0)
        
        # Boost MST edges moderately
        mst_boost = 1.0 + 0.3 * mst_binary
        heuristics *= mst_boost
    
    # === 7. Adaptive Edge Sparsification ===
    # Keep most promising edges while maintaining connectivity
    if n > 2:
        min_edges_per_node = max(2, min(5, int(np.sqrt(n))))
        
        for i in range(n):
            row = heuristics[i].copy()
            row[i] = 0
            
            if np.sum(row > 0) <= min_edges_per_node:
                continue  # Already sparse enough
                
            # Determine adaptive threshold for this node
            non_zero = row[row > 0]
            if len(non_zero) > 0:
                # Keep top edges and those above median
                median_val = np.median(non_zero)
                q75_val = np.percentile(non_zero, 75)
                threshold = max(median_val, q75_val * 0.7)
                
                # Also always keep top k edges
                sorted_indices = np.argsort(row)[::-1]
                top_k_indices = sorted_indices[:min_edges_per_node]
                
                # Create mask
                keep_mask = (row >= threshold) | np.isin(np.arange(n), top_k_indices)
                row[~keep_mask] *= 0.1  # Reduce rather than eliminate
                heuristics[i] = row
    
    # === 8. Symmetry and Final Normalization ===
    # Ensure symmetry and proper scaling
    symmetric_heuristics = (heuristics + heuristics.T) / 2.0
    np.fill_diagonal(symmetric_heuristics, 0)
    
    # Soft normalization preserving relative differences
    max_val = np.max(symmetric_heuristics)
    if max_val > 0:
        symmetric_heuristics = symmetric_heuristics / (max_val + epsilon)
        # Gentle compression to reduce extreme values
        symmetric_heuristics = np.power(symmetric_heuristics, 0.85)
    
    # Ensure non-negativity
    symmetric_heuristics = np.maximum(symmetric_heuristics, 0)
    
    return symmetric_heuristics
