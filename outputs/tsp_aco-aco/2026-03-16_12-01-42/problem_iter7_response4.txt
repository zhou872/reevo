import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with normalized proportional regret
    and multiplicative factor fusion for improved edge prioritization.
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in solutions
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base inverse distance with stability epsilon ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Normalized Proportional Regret Penalization ===
    # Compute regret as (second_best - first_best) / first_best for scale invariance
    regret_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Find two best distances for each node
        if len(dists) >= 2:
            # Get indices of two smallest distances
            indices = np.argpartition(dists, 2)[:2]
            first_best_idx, second_best_idx = indices[0], indices[1]
            
            # Ensure proper ordering
            if dists[second_best_idx] < dists[first_best_idx]:
                first_best_idx, second_best_idx = second_best_idx, first_best_idx
            
            first_best = dists[first_best_idx]
            second_best = dists[second_best_idx]
            
            # Compute normalized regret for the best edge
            if first_best > 0:
                base_regret = (second_best - first_best) / first_best
                # Scale regret to appropriate range
                regret_matrix[i, first_best_idx] = 1.5 + np.tanh(base_regret)
            
            # For other edges, compute their regret relative to the best alternative
            for j in range(n):
                if i == j or j == first_best_idx:
                    continue
                
                d_ij = distance_matrix[i, j]
                # Best alternative is either first_best (if j != first_best) or second_best
                best_alternative = second_best if j == first_best_idx else first_best
                
                if d_ij > 0:
                    edge_regret = (best_alternative - d_ij) / d_ij
                    # Convert to boost factor: higher for edges with high regret (few alternatives)
                    # Negative regret means edge is worse than alternative - penalize
                    regret_matrix[i, j] = 1.0 + 0.5 * np.tanh(edge_regret)
    
    # === 3. Spatial Cluster Density Scoring ===
    # Reward edges within dense clusters and penalize inter-cluster edges
    cluster_scores = np.zeros_like(distance_matrix)
    
    # Adaptive cluster size based on problem size
    k_cluster = max(3, min(10, n // 5))
    
    for i in range(n):
        # Get distances and sort
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Find k_cluster nearest neighbors
        nearest_indices = np.argpartition(dists, k_cluster)[:k_cluster]
        nearest_dists = dists[nearest_indices]
        
        # Compute local density measure (harmonic mean of nearest distances)
        if np.any(nearest_dists > 0):
            harmonic_mean = k_cluster / np.sum(1 / (nearest_dists + epsilon))
            
            # For each neighbor, compute density-adjusted score
            for idx, j in enumerate(nearest_indices):
                if i == j:
                    continue
                
                d_ij = distance_matrix[i, j]
                # Score inversely proportional to distance, scaled by local density
                # Edges in dense regions get higher scores
                density_factor = harmonic_mean / (d_ij + epsilon)
                cluster_scores[i, j] = 1.0 + 0.5 * np.tanh(density_factor - 1.0)
    
    # === 4. Node Centrality Adjustment ===
    # Penalize connections to overly popular nodes to encourage balanced tours
    centrality_penalty = np.ones_like(distance_matrix)
    
    # Compute how often each node appears in others' nearest neighbor lists
    node_popularity = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        # Count how many times i is in top k_nearest of other nodes
        k_check = max(3, min(8, n // 3))
        nearest_to_i = np.argpartition(dists, k_check)[:k_check]
        
        for j in range(n):
            if i != j and j in nearest_to_i:
                node_popularity[j] += 1
    
    # Normalize popularity and apply penalty
    if np.max(node_popularity) > 0:
        norm_popularity = node_popularity / np.max(node_popularity)
        for i in range(n):
            for j in range(n):
                if i != j:
                    # Penalize edges to highly popular nodes
                    penalty = 1.0 - 0.4 * norm_popularity[j]
                    centrality_penalty[i, j] = max(0.6, penalty)
    
    np.fill_diagonal(centrality_penalty, 0)
    
    # === 5. Edge Neighborhood Consistency ===
    # Boost edges where endpoints have similar neighborhoods
    consistency_scores = np.ones_like(distance_matrix)
    
    k_consistency = max(3, min(7, n // 4))
    
    for i in range(n):
        # Get i's neighborhood
        dists_i = distance_matrix[i].copy()
        dists_i[i] = np.inf
        neighbors_i = set(np.argpartition(dists_i, k_consistency)[:k_consistency])
        
        for j in range(n):
            if i == j:
                continue
            
            # Get j's neighborhood
            dists_j = distance_matrix[j].copy()
            dists_j[j] = np.inf
            neighbors_j = set(np.argpartition(dists_j, k_consistency)[:k_consistency])
            
            # Jaccard similarity of neighborhoods
            intersection = len(neighbors_i.intersection(neighbors_j))
            union = len(neighbors_i.union(neighbors_j))
            
            if union > 0:
                similarity = intersection / union
                # Boost edges between nodes with similar neighborhoods
                consistency_scores[i, j] = 1.0 + 0.3 * similarity
    
    # === 6. Multiplicative Factor Fusion ===
    # Combine all components multiplicatively for strong differentiation
    heuristics = inv_dist.copy()
    heuristics *= regret_matrix  # Regret-based prioritization
    heuristics *= cluster_scores  # Cluster density awareness
    heuristics *= centrality_penalty  # Popularity balancing
    heuristics *= consistency_scores  # Neighborhood consistency
    
    # === 7. Adaptive Sparsification ===
    # Keep only promising edges while preserving connectivity
    for i in range(n):
        row = heuristics[i].copy()
        
        # Determine adaptive threshold
        if n > 3:
            # Keep top k edges, where k scales with problem size
            keep_k = max(3, min(15, int(np.sqrt(n) * 2)))
            
            # Sort and threshold
            sorted_indices = np.argsort(row)
            threshold = row[sorted_indices[-keep_k]]
            
            # Also keep edges with very high regret (unique connections)
            regret_threshold = np.percentile(regret_matrix[i][regret_matrix[i] > 0], 70)
            regret_mask = regret_matrix[i] > regret_threshold
            
            # Apply sparsification
            mask = (row >= threshold) | regret_mask
            # Ensure at least 2 connections per node
            if np.sum(mask) < 2:
                top2 = sorted_indices[-2:]
                mask[top2] = True
            
            row[~mask] = 0
            heuristics[i] = row
    
    # === 8. Symmetrization and Finalization ===
    # Average directional scores with preference for mutual high scores
    symmetric_heuristics = (heuristics + heuristics.T) / 2
    
    # Additional boost for edges that both endpoints consider promising
    mutual_boost = np.minimum(heuristics, heuristics.T)
    symmetric_heuristics += 0.3 * mutual_boost
    
    # Final cleanup
    np.fill_diagonal(symmetric_heuristics, 0)
    symmetric_heuristics = np.maximum(symmetric_heuristics, 0)
    
    # Normalize to [0, 1] range
    max_val = np.max(symmetric_heuristics)
    if max_val > 0:
        symmetric_heuristics = symmetric_heuristics / max_val
    
    return symmetric_heuristics
