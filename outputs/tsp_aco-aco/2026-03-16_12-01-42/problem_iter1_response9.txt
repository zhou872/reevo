import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic matrix for TSP edge selection.
    Returns a matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    if n <= 1:
        return np.zeros_like(distance_matrix)
    
    # Avoid division by zero and self-loops
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    
    # 1. Base inverse distance (closer nodes are generally better)
    inv_dist = 1.0 / safe_dist
    
    # 2. Nearest-neighbor gradient: emphasize edges where one node is among 
    # the other's k-nearest neighbors
    k = max(2, min(5, n // 4))  # Adaptive k based on problem size
    nearest_neighbor_bonus = np.zeros_like(inv_dist)
    for i in range(n):
        # Get k smallest distances from node i (excluding self)
        sorted_idx = np.argsort(safe_dist[i])[:k]
        nearest_neighbor_bonus[i, sorted_idx] = 1.0
    
    # Symmetric bonus: edges where both nodes consider each other near
    symmetric_nn = (nearest_neighbor_bonus + nearest_neighbor_bonus.T) / 2
    
    # 3. Local connectivity clustering: nodes that share many near neighbors
    # might belong to the same cluster
    cluster_bonus = np.zeros_like(inv_dist)
    for i in range(n):
        for j in range(i + 1, n):
            # Count how many common near neighbors (within 30th percentile)
            threshold_i = np.percentile(safe_dist[i], 30)
            threshold_j = np.percentile(safe_dist[j], 30)
            near_i = set(np.where(safe_dist[i] <= threshold_i)[0]) - {i}
            near_j = set(np.where(safe_dist[j] <= threshold_j)[0]) - {j}
            common = len(near_i.intersection(near_j))
            cluster_bonus[i, j] = cluster_bonus[j, i] = common / max(1, min(len(near_i), len(near_j)))
    
    # 4. Combine factors with weights
    heuristics = (
        0.5 * inv_dist +           # Base inverse distance
        0.3 * symmetric_nn +       # Mutual nearest-neighbor bonus
        0.2 * cluster_bonus        # Clustering bonus
    )
    
    # 5. Sparsification: zero out unpromising edges
    # Keep only edges in the top percentile for each node
    keep_percentile = 40  # Keep top 40% most promising edges per node
    threshold_matrix = np.zeros_like(heuristics)
    for i in range(n):
        row = heuristics[i]
        threshold = np.percentile(row, 100 - keep_percentile)
        threshold_matrix[i] = threshold
    
    # Zero out edges below both nodes' thresholds
    mask = (heuristics >= threshold_matrix) | (heuristics >= threshold_matrix.T)
    heuristics = heuristics * mask
    
    # 6. Normalize to reasonable scale
    max_val = heuristics.max()
    if max_val > 0:
        heuristics = heuristics / max_val * 10
    
    # Ensure diagonal is zero (no self-loops)
    np.fill_diagonal(heuristics, 0)
    
    return heuristics
