import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with improved factors and sparsification.
    Returns a symmetric matrix where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12  # Increased for numerical stability
    
    # 1. Base reciprocal distance (more stable than pure inverse)
    with np.errstate(divide='ignore', invalid='ignore'):
        recip_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(recip_dist, 0)
    
    # 2. Mutual Nearest Neighbor Emphasis (as per knowledge graph suggestion)
    k = max(3, int(np.sqrt(n)))  # Adaptive k
    mutual_nn_mask = np.zeros((n, n), dtype=bool)
    
    for i in range(n):
        # Get k nearest neighbors (excluding self)
        nearest = np.argsort(distance_matrix[i])[1:k+1]
        mutual_nn_mask[i, nearest] = True
    
    # Only keep edges where BOTH endpoints consider each other in top-k
    mutual_nn_mask = mutual_nn_mask & mutual_nn_mask.T
    
    # 3. Minimum Spanning Tree factor (more efficient implementation)
    mst_mask = np.zeros((n, n), dtype=bool)
    mst = minimum_spanning_tree(distance_matrix)
    mst_mask = (mst.toarray() > 0) | (mst.toarray().T > 0)
    
    # 4. Dynamic sparsification with adaptive thresholds (as per knowledge graph)
    # Global percentile threshold
    global_thresh = np.percentile(recip_dist[recip_dist > 0], 70)  # Keep top 30%
    
    # Per-node local thresholds ensuring minimum connectivity
    sparsified = np.zeros_like(recip_dist)
    for i in range(n):
        row_vals = recip_dist[i].copy()
        row_vals[i] = 0  # Ignore diagonal
        
        # Keep at least 2 edges per node (ensuring feasibility)
        min_keep = max(2, k)
        if np.sum(row_vals > 0) > min_keep:
            # Local threshold: keep edges above both global and local percentiles
            local_thresh = max(global_thresh, np.percentile(row_vals[row_vals > 0], 50))
            keep_mask = row_vals >= local_thresh
            
            # Ensure we keep at least min_keep edges
            if np.sum(keep_mask) < min_keep:
                top_indices = np.argsort(row_vals)[-min_keep:]
                keep_mask = np.zeros(n, dtype=bool)
                keep_mask[top_indices] = True
            
            sparsified[i] = row_vals * keep_mask
    
    # Symmetrize sparsification
    sparsified = np.maximum(sparsified, sparsified.T)
    
    # 5. Combine factors with multiplicative boosts
    heuristics = sparsified.copy()
    
    # Strong boost for mutual nearest neighbors
    heuristics[mutual_nn_mask] *= 3.0
    
    # Strong boost for MST edges
    heuristics[mst_mask] *= 2.5
    
    # 6. Degree-based penalty to encourage balanced tours
    # Compute approximate degree from current heuristic values
    with np.errstate(divide='ignore', invalid='ignore'):
        degree_estimate = np.sum(heuristics > 0, axis=1)
        degree_penalty = 1.0 / (1.0 + np.clip(degree_estimate - 2, 0, None))  # Penalize nodes with >2 connections
        degree_matrix = degree_penalty[:, None] * degree_penalty[None, :]
    
    heuristics *= degree_matrix
    
    # 7. Adaptive top-k sparsification (as per knowledge graph)
    final_k = min(15, max(3, n // 3))
    final_matrix = np.zeros_like(heuristics)
    
    for i in range(n):
        row = heuristics[i].copy()
        if np.sum(row > 0) > final_k:
            threshold = np.sort(row[row > 0])[-final_k]
            row[row < threshold] = 0
        final_matrix[i] = row
    
    # Final symmetrization
    final_matrix = np.maximum(final_matrix, final_matrix.T * 0.7)  # Asymmetric tolerance
    np.fill_diagonal(final_matrix, 0)
    
    return final_matrix
