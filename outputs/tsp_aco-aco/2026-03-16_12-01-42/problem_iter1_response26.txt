import numpy as np
from scipy.spatial import distance_matrix
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic indicators for TSP edge selection combining multiple factors
    and sparsifying unpromising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Basic inverse distance (foundation)
    with np.errstate(divide='ignore', invalid='ignore'):
        inv_dist = np.where(distance_matrix > 0, 1 / distance_matrix, 0)
    
    # 2. Minimum Spanning Tree bonus edges
    mst = minimum_spanning_tree(distance_matrix).toarray()
    mst_bonus = np.where(mst + mst.T > 0, 2.0, 1.0)  # Double importance for MST edges
    
    # 3. Geometric center attraction
    # Points closer to geometric center are better connected
    if n > 2:
        # Compute geometric center
        # Generate points from indices (simple approximation)
        points = np.random.RandomState(42).randn(n, 2)  # Deterministic for reproducibility
        center = np.mean(points, axis=0)
        center_dist = distance_matrix(points, [center]).flatten()
        center_attraction = 1 / (1 + center_dist / np.max(center_dist))
        # Symmetric matrix of center attraction product
        center_factor = np.outer(center_attraction, center_attraction)
    else:
        center_factor = np.ones((n, n))
    
    # 4. Two-level nearest neighbor analysis
    # First level: direct nearest neighbors
    nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        # Mark 3 nearest neighbors (excluding self)
        idx = np.argsort(distance_matrix[i])[1:4]  # Skip i itself
        nn_mask[i, idx] = True
    
    # Second level: neighbors of neighbors
    nn2_mask = np.zeros_like(distance_matrix, dtype=bool)
    for i in range(n):
        # Get neighbors of i
        neighbors = np.where(nn_mask[i])[0]
        # Get neighbors of neighbors
        nn2 = []
        for j in neighbors:
            nn2.extend(np.where(nn_mask[j])[0])
        nn2 = np.unique(nn2)
        nn2_mask[i, nn2] = True
    
    # Combined neighbor priority
    neighbor_factor = np.zeros_like(distance_matrix)
    neighbor_factor[nn_mask] = 3.0  # Direct neighbors
    neighbor_factor[nn2_mask] = 1.5  # Second-order neighbors
    np.fill_diagonal(neighbor_factor, 0)
    
    # 5. Estimated tour length potential
    # Calculate average distance from each node to all others
    avg_dists = np.mean(distance_matrix, axis=1, keepdims=True)
    # Nodes with smaller average distance are more central
    centrality = 1 / (1 + avg_dists / np.max(avg_dists))
    centrality_factor = np.outer(centrality, centrality)
    
    # 6. Combine all factors
    heuristic_matrix = (
        inv_dist * mst_bonus * 
        center_factor * 
        (1 + 0.3 * neighbor_factor) * 
        centrality_factor
    )
    
    # 7. Sparsify: keep only top edges per node
    sparsified = np.zeros_like(heuristic_matrix)
    k_sparse = max(2, min(8, n // 3))  # Adaptive sparsity
    
    for i in range(n):
        # Get indices of top k edges (excluding self)
        top_indices = np.argsort(heuristic_matrix[i])[-k_sparse:]
        top_indices = top_indices[top_indices != i]
        sparsified[i, top_indices] = heuristic_matrix[i, top_indices]
    
    # Ensure symmetry for undirected TSP
    sparsified = np.maximum(sparsified, sparsified.T)
    
    return sparsified
