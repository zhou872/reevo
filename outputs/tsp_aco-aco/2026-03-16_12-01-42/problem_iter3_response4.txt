import numpy as np
from typing import Optional

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection with mutual nearest neighbor emphasis
    and adaptive sparsification strategies.
    
    Args:
        distance_matrix: Symmetric n×n matrix with distances between nodes.
        
    Returns:
        Heuristic matrix of same shape where higher values indicate more promising edges.
    """
    n = distance_matrix.shape[0]
    
    # 1. Base inverse distance with epsilon for numerical stability
    epsilon = 1e-12
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # 2. Mutual nearest neighbor detection with adaptive k
    k = max(3, int(np.sqrt(n)))  # Adaptive mutual neighbor count
    mutual_nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    oneway_nn_mask = np.zeros_like(distance_matrix, dtype=bool)
    
    # Precompute nearest neighbors for all nodes
    nearest_indices = []
    for i in range(n):
        # Get k+1 smallest distances (including self)
        nearest = np.argpartition(distance_matrix[i], k+1)[:k+1]
        nearest = nearest[nearest != i]
        if len(nearest) > k:
            nearest = nearest[:k]
        nearest_indices.append(set(nearest))
    
    # Identify mutual and one-way nearest neighbors
    for i in range(n):
        for j in range(i+1, n):
            if j in nearest_indices[i] and i in nearest_indices[j]:
                mutual_nn_mask[i, j] = mutual_nn_mask[j, i] = True
            elif j in nearest_indices[i] or i in nearest_indices[j]:
                oneway_nn_mask[i, j] = oneway_nn_mask[j, i] = True
    
    # 3. Minimum Spanning Tree edges using Prim's algorithm
    mst_mask = np.zeros_like(distance_matrix, dtype=bool)
    visited = np.zeros(n, dtype=bool)
    min_edge = np.full(n, np.inf)
    parent = np.full(n, -1)
    
    # Start from node 0
    min_edge[0] = 0
    
    for _ in range(n):
        # Find unvisited node with minimum edge weight
        u = -1
        min_val = np.inf
        for v in range(n):
            if not visited[v] and min_edge[v] < min_val:
                min_val = min_edge[v]
                u = v
        
        if u == -1:
            break
        
        visited[u] = True
        
        # Add edge to MST if not the starting node
        if parent[u] != -1:
            mst_mask[u, parent[u]] = mst_mask[parent[u], u] = True
        
        # Update minimum edges for unvisited neighbors
        for v in range(n):
            if not visited[v] and distance_matrix[u, v] < min_edge[v]:
                min_edge[v] = distance_matrix[u, v]
                parent[v] = u
    
    # 4. Degree balancing factor - prefer edges connecting to nodes with lower degree centrality
    degree_scores = np.zeros(n)
    for i in range(n):
        # Count number of mutual nearest neighbors for each node
        degree_scores[i] = np.sum(mutual_nn_mask[i])
    
    if degree_scores.max() > 0:
        # Normalize and invert so lower-degree nodes get higher scores
        degree_scores = 1.0 - (degree_scores / degree_scores.max())
    
    # 5. Combine factors with emphasis on mutual nearest neighbors
    heuristics = inv_dist.copy()
    
    # Strong boost for mutual nearest neighbors (both directions)
    heuristics[mutual_nn_mask] *= 2.5
    
    # Moderate boost for one-way nearest neighbors
    heuristics[oneway_nn_mask] *= 1.8
    
    # Strong boost for MST edges (connectivity backbone)
    heuristics[mst_mask] *= 2.2
    
    # Incorporate degree balancing - prefer edges to nodes with fewer mutual connections
    for i in range(n):
        for j in range(n):
            if i != j:
                avg_degree_score = (degree_scores[i] + degree_scores[j]) / 2
                heuristics[i, j] *= (1.0 + 0.4 * avg_degree_score)
    
    # 6. Adaptive sparsification with top-k retention
    # Determine adaptive k per node
    base_k = min(10, max(3, n // 4))
    
    for i in range(n):
        row = heuristics[i].copy()
        
        # Keep at least 2 edges to maintain connectivity possibility
        keep_count = max(2, base_k)
        
        if keep_count < n:
            # Find threshold for top keep_count edges
            threshold_idx = n - keep_count - 1
            threshold = np.partition(row, threshold_idx)[threshold_idx]
            row[row < threshold] = 0
            heuristics[i] = row
        
        # Ensure diagonal remains zero
        heuristics[i, i] = 0
    
    # 7. Symmetrize the sparsified matrix
    # Keep edge if either endpoint considers it promising (with preference to stronger direction)
    symmetric_heuristics = np.maximum(heuristics, heuristics.T * 0.7)
    
    # Final cleanup: ensure diagonal is zero
    np.fill_diagonal(symmetric_heuristics, 0)
    
    return symmetric_heuristics
