import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced TSP edge selection heuristics incorporating path-based lookahead,
    meta-learning-inspired parameter adjustment, and ensemble techniques.
    
    Key Improvements:
    1. Two-step path lookahead for edge context evaluation
    2. Self-adjusting meta-parameters based on problem characteristics
    3. Ensemble fusion of complementary heuristic strategies
    4. Hierarchical mutual neighborhood reinforcement
    
    Parameters:
    -----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix
        
    Returns:
    --------
    np.ndarray
        Heuristic matrix of same shape, where higher values indicate
        more promising edges for inclusion in solutions
    """
    n = distance_matrix.shape[0]
    if n <= 2:
        heuristics = np.ones_like(distance_matrix)
        np.fill_diagonal(heuristics, 0)
        return heuristics
    
    epsilon = 1e-12
    
    # === Meta-parameter self-adjustment based on problem characteristics ===
    # Analyze distance distribution to set adaptive parameters
    flat_dists = distance_matrix[np.triu_indices(n, k=1)]
    distance_stats = {
        'mean': np.mean(flat_dists),
        'std': np.std(flat_dists),
        'cv': np.std(flat_dists) / (np.mean(flat_dists) + epsilon)
    }
    
    # Dynamic parameter adjustment based on problem statistics
    base_k = max(3, min(15, int(np.sqrt(n) * (1 + distance_stats['cv']))))
    lookahead_depth = min(3, max(2, int(2 * distance_stats['cv'] + 1)))
    
    # === 1. Two-step path lookahead scoring ===
    def compute_path_lookahead_scores(depth: int) -> np.ndarray:
        """Compute edge scores considering multi-step paths"""
        path_scores = np.zeros_like(distance_matrix)
        
        for i in range(n):
            # Consider paths of length up to 'depth'
            for j in range(n):
                if i == j:
                    continue
                    
                # Base: direct edge
                base_quality = 1 / (distance_matrix[i, j] + epsilon)
                
                # Lookahead: consider paths i -> j -> k
                lookahead_qualities = []
                for k in range(n):
                    if k == i or k == j:
                        continue
                    
                    # Two-step path quality (inverse of total distance)
                    two_step_quality = 2 / (distance_matrix[i, j] + distance_matrix[j, k] + epsilon)
                    
                    if depth >= 3:
                        # Three-step path i -> j -> k -> l (return to start consideration)
                        for l in range(n):
                            if l == i or l == j or l == k:
                                continue
                            three_step_quality = 3 / (distance_matrix[i, j] + 
                                                     distance_matrix[j, k] + 
                                                     distance_matrix[k, l] + epsilon)
                            lookahead_qualities.append(three_step_quality)
                    else:
                        lookahead_qualities.append(two_step_quality)
                
                # Use maximum lookahead quality as bonus
                if lookahead_qualities:
                    lookahead_bonus = np.max(lookahead_qualities) / (base_quality + epsilon)
                    # Normalized sigmoid to prevent excessive values
                    path_scores[i, j] = base_quality * (1 + 0.5 * np.tanh(lookahead_bonus - 1))
                else:
                    path_scores[i, j] = base_quality
        
        np.fill_diagonal(path_scores, 0)
        return path_scores
    
    path_scores = compute_path_lookahead_scores(lookahead_depth)
    
    # === 2. Ensemble of complementary heuristics ===
    ensemble_weights = {'path': 0.4, 'regret': 0.25, 'spatial': 0.2, 'mst': 0.15}
    
    # Component A: Enhanced regret with adaptive window
    regret_scores = np.ones_like(distance_matrix)
    k_regret = min(max(4, base_k), n-1)
    
    for i in range(n):
        dist_i = distance_matrix[i].copy()
        dist_i[i] = np.inf
        
        # Get k_regret nearest distances
        if n > k_regret:
            k_nearest = np.partition(dist_i, k_regret)[:k_regret]
            k_nearest_indices = np.argpartition(dist_i, k_regret)[:k_regret]
        else:
            k_nearest = dist_i[dist_i < np.inf]
            k_nearest_indices = np.where(dist_i < np.inf)[0]
        
        for j in range(n):
            if i == j:
                continue
                
            d_ij = distance_matrix[i, j]
            
            # Find alternative edges excluding current edge
            alternative_mask = k_nearest_indices != j
            if np.any(alternative_mask):
                alt_distances = k_nearest[alternative_mask]
                if len(alt_distances) > 0:
                    best_alt = np.min(alt_distances)
                    regret = max(0, (best_alt - d_ij) / (d_ij + epsilon))
                    # Adaptive regret transformation based on distance statistics
                    regret_transform = np.exp(-regret / (1 + distance_stats['cv']))
                    regret_scores[i, j] = 1 + regret_transform
                else:
                    regret_scores[i, j] = 2.0
            else:
                regret_scores[i, j] = 2.0
    
    # Component B: Spatial coherence with adaptive neighborhood
    spatial_scores = np.ones_like(distance_matrix)
    k_spatial = min(max(5, base_k * 2), n-1)
    
    # Compute node spatial signatures
    node_signatures = np.zeros(n)
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        if n > k_spatial:
            spatial_dists = np.partition(dists, k_spatial)[:k_spatial]
            # Use robust statistical measures
            with np.errstate(divide='ignore', invalid='ignore'):
                node_signatures[i] = np.median(spatial_dists) * (1 + np.std(spatial_dists) / 
                                                                (np.mean(spatial_dists) + epsilon))
        else:
            valid_dists = dists[dists < np.inf]
            if len(valid_dists) > 0:
                node_signatures[i] = np.median(valid_dists)
    
    # Normalize signatures
    if np.max(node_signatures) > np.min(node_signatures):
        signatures_norm = (node_signatures - np.min(node_signatures)) / \
                         (np.max(node_signatures) - np.min(node_signatures) + epsilon)
    else:
        signatures_norm = np.ones_like(node_signatures) * 0.5
    
    # Compute spatial scores favoring edges between similar nodes
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
            # Similarity-based scoring: favor edges between nodes with similar spatial contexts
            similarity = 1 - np.abs(signatures_norm[i] - signatures_norm[j])
            spatial_scores[i, j] = 0.7 + 0.6 * similarity
    
    # Component C: MST-based structural reinforcement
    mst_scores = np.ones_like(distance_matrix)
    if n > 2:
        mst_sparse = minimum_spanning_tree(distance_matrix)
        mst_dense = mst_sparse.toarray()
        mst_mask = (mst_dense > 0) | (mst_dense.T > 0)
        mst_scores[mst_mask] = 1.6  # MST bonus
    
    # === 3. Hierarchical mutual neighborhood reinforcement ===
    # Level 1: Immediate mutual neighbors
    k_mutual1 = min(max(3, base_k // 2), n-1)
    # Level 2: Extended mutual neighbors
    k_mutual2 = min(max(5, base_k), n-1)
    
    # Build neighborhood hierarchies
    nn_sets_level1 = [set() for _ in range(n)]
    nn_sets_level2 = [set() for _ in range(n)]
    
    for i in range(n):
        dists = distance_matrix[i].copy()
        dists[i] = np.inf
        
        # Level 1 neighbors
        if n-1 > k_mutual1:
            neighbors1 = set(np.argpartition(dists, k_mutual1)[:k_mutual1])
        else:
            neighbors1 = set(np.arange(n)) - {i}
        nn_sets_level1[i] = neighbors1
        
        # Level 2 neighbors
        if n-1 > k_mutual2:
            neighbors2 = set(np.argpartition(dists, k_mutual2)[:k_mutual2])
        else:
            neighbors2 = set(np.arange(n)) - {i}
        nn_sets_level2[i] = neighbors2
    
    # Apply hierarchical mutual reinforcement
    mutual_boost = np.ones_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i == j:
                continue
                
            mutual_level = 0
            if j in nn_sets_level1[i] and i in nn_sets_level1[j]:
                mutual_level = 2  # Strong mutual reinforcement
            elif j in nn_sets_level2[i] and i in nn_sets_level2[j]:
                mutual_level = 1  # Moderate mutual reinforcement
            
            if mutual_level == 2:
                mutual_boost[i, j] = 1.8
            elif mutual_level == 1:
                mutual_boost[i, j] = 1.4
            elif j in nn_sets_level1[i] or i in nn_sets_level1[j]:
                mutual_boost[i, j] = 1.2
    
    # === 4. Ensemble fusion ===
    # Combine weighted components
    base_heuristics = (
        ensemble_weights['path'] * path_scores +
        ensemble_weights['regret'] * regret_scores +
        ensemble_weights['spatial'] * spatial_scores +
        ensemble_weights['mst'] * mst_scores
    )
    
    # Apply mutual boost multiplicatively
    heuristics = base_heuristics * mutual_boost
    np.fill_diagonal(heuristics, 0)
    
    # === 5. Connectivity-aware adaptive sparsification ===
    # Dynamic sparsification based on problem size and distance characteristics
    density_factor = 0.3 + 0.4 * (1 - distance_stats['cv'])  # Higher CV → sparser
    adaptive_k = int(max(3, min(n-1, density_factor * n)))
    
    for i in range(n):
        row = heuristics[i].copy()
        
        if adaptive_k < n - 1:
            # Calculate enhanced edge importance considering mutual neighborhoods
            enhanced_scores = row.copy()
            for j in range(n):
                if i == j:
                    continue
                
                # Boost scores for edges in mutual neighborhoods
                if j in nn_sets_level1[i] and i in nn_sets_level1[j]:
                    enhanced_scores[j] *= 1.5
                elif j in nn_sets_level2[i] and i in nn_sets_level2[j]:
                    enhanced_scores[j] *= 1.3
                
                # Penalize edges that would create very long paths
                if lookahead_depth >= 2:
                    # Check if this edge leads to isolated nodes
                    other_nodes = set(range(n)) - {i, j}
                    avg_dist_to_others = np.mean([distance_matrix[j, k] for k in other_nodes])
                    if avg_dist_to_others > 1.5 * distance_stats['mean']:
                        enhanced_scores[j] *= 0.8
            
            # Keep top adaptive_k edges by enhanced score
            top_indices = np.argsort(enhanced_scores)[-adaptive_k:]
            
            # Ensure connectivity: always keep mutual level-1 neighbors
            mutual_level1 = [j for j in nn_sets_level1[i] if i in nn_sets_level1[j]]
            for j in mutual_level1:
                if j not in top_indices:
                    # Replace weakest kept edge
                    min_kept_idx = top_indices[np.argmin(enhanced_scores[top_indices])]
                    if enhanced_scores[j] > enhanced_scores[min_kept_idx]:
                        top_indices = np.setdiff1d(top_indices, [min_kept_idx])
                        top_indices = np.append(top_indices, j)
            
            # Apply sparsification
            keep_mask = np.zeros(n, dtype=bool)
            keep_mask[top_indices] = True
            row[~keep_mask] = 0
            heuristics[i] = row
    
    # === 6. Symmetrization with lookahead-aware weighting ===
    symmetric_heuristics = np.zeros_like(heuristics)
    for i in range(n):
        for j in range(n):
            if i == j:
                symmetric_heuristics[i, j] = 0
            else:
                # Weighted average favoring stronger mutual decisions
                if j in nn_sets_level1[i] and i in nn_sets_level1[j]:
                    weight_i, weight_j = 0.7, 0.7
                elif j in nn_sets_level2[i] and i in nn_sets_level2[j]:
                    weight_i, weight_j = 0.6, 0.6
                else:
                    weight_i, weight_j = 0.55, 0.45
                
                symmetric_heuristics[i, j] = (
                    weight_i * heuristics[i, j] + 
                    weight_j * heuristics[j, i]
                )
    
    # === 7. Final normalization and validation ===
    np.fill_diagonal(symmetric_heuristics, 0)
    symmetric_heuristics = np.maximum(symmetric_heuristics, 0)
    
    # Ensure each node has at least 2 non-zero connections
    for i in range(n):
        non_zero = np.nonzero(symmetric_heuristics[i])[0]
        if len(non_zero) < 2 and n > 2:
            # Add connections to nearest neighbors
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            nearest = np.argsort(dists)[:2]
            for j in nearest:
                if symmetric_heuristics[i, j] == 0:
                    symmetric_heuristics[i, j] = 0.1 * np.max(symmetric_heuristics[i])
                    symmetric_heuristics[j, i] = max(symmetric_heuristics[j, i], 
                                                     symmetric_heuristics[i, j])
    
    # Normalize to [0, 1] range
    max_val = np.max(symmetric_heuristics)
    if max_val > 0:
        symmetric_heuristics = symmetric_heuristics / max_val
    
    return symmetric_heuristics
