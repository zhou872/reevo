import numpy as np
from scipy.spatial.distance import pdist, squareform

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection with improved lookahead,
    refined mutual neighbor criteria, and balanced global-local factors.
    
    Core innovations:
    1. Two-step lookahead regret estimation with path completion cost
    2. Adaptive mutual neighbor criteria using distance ratios
    3. Dynamic weighting between global MST and local nearest neighbors
    4. Multi-stage sparsification with connectivity preservation
    """
    n = distance_matrix.shape[0]
    
    # Base inverse distance with adaptive regularization
    eps = 1e-12
    avg_dist = np.mean(distance_matrix[distance_matrix > 0])
    inv_dist = 1 / (distance_matrix + eps * avg_dist)
    np.fill_diagonal(inv_dist, 0)
    
    # 1. REFINED MUTUAL NEIGHBOR CRITERIA WITH DISTANCE RATIOS
    # Adaptive k based on instance density
    k_mutual = max(3, min(15, int(np.sqrt(n) * 1.5)))
    
    # Compute neighbor rankings
    neighbor_ranks = np.zeros((n, n), dtype=int)
    for i in range(n):
        ranks = np.argsort(distance_matrix[i])
        neighbor_ranks[i] = ranks
    
    # Enhanced mutual scoring with ratio-based threshold
    mutual_score = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(i + 1, n):
            # Check if j is in i's top k_mutual neighbors
            j_rank_i = np.where(neighbor_ranks[i] == j)[0][0]
            i_rank_j = np.where(neighbor_ranks[j] == i)[0][0]
            
            if j_rank_i < k_mutual and i_rank_j < k_mutual:
                # Ratio-based scoring: more valuable if both ranks are high
                rank_score = (1 - j_rank_i/k_mutual) * (1 - i_rank_j/k_mutual)
                
                # Distance ratio test: prefer edges where distance is much smaller
                # than to other nearby nodes
                avg_alt_i = np.mean(distance_matrix[i, neighbor_ranks[i, 1:4]])
                avg_alt_j = np.mean(distance_matrix[j, neighbor_ranks[j, 1:4]])
                dist_ratio = min(distance_matrix[i,j]/avg_alt_i, distance_matrix[i,j]/avg_alt_j)
                
                # Combined score with diminishing returns for very close ratios
                mutual_score[i,j] = mutual_score[j,i] = rank_score * (1.0 / (1.0 + np.log1p(dist_ratio)))
    
    # 2. MST FACTOR WITH ADAPTIVE BOOST
    mst_factor = np.zeros_like(distance_matrix)
    if n > 1:
        # Run Prim's algorithm
        in_mst = np.zeros(n, dtype=bool)
        min_edge = np.full(n, np.inf)
        source = np.full(n, -1, dtype=int)
        
        # Start from node 0
        min_edge[0] = 0
        
        for _ in range(n):
            # Find minimum edge to add
            u = -1
            min_val = np.inf
            for v in range(n):
                if not in_mst[v] and min_edge[v] < min_val:
                    min_val = min_edge[v]
                    u = v
            
            if u == -1:
                break
            
            in_mst[u] = True
            
            # Mark MST edge
            if source[u] != -1:
                mst_factor[u, source[u]] = mst_factor[source[u], u] = 1.0
            
            # Update distances
            for v in range(n):
                if not in_mst[v] and distance_matrix[u, v] < min_edge[v]:
                    min_edge[v] = distance_matrix[u, v]
                    source[v] = u
    
    # 3. TWO-STEP LOOKAHEAD REGRET ESTIMATION
    lookahead_score = np.zeros_like(distance_matrix)
    
    # Precompute second-best options for each node
    second_best = np.zeros((n, n), dtype=float)
    for i in range(n):
        sorted_dists = np.sort(distance_matrix[i])
        # Second smallest (smallest is 0 to self)
        second_best[i] = sorted_dists[1] if len(sorted_dists) > 1 else 0
    
    # Estimate completion cost for each potential edge
    for i in range(n):
        for j in range(i + 1, n):
            if distance_matrix[i, j] == 0:
                continue
                
            # Estimate minimum completion cost from both ends
            # Consider best continuations from j (excluding i) and from i (excluding j)
            mask_i = np.ones(n, dtype=bool)
            mask_i[[i, j]] = False
            mask_j = np.ones(n, dtype=bool)
            mask_j[[i, j]] = False
            
            if np.any(mask_i) and np.any(mask_j):
                # Best continuation from j
                best_from_j = np.min(distance_matrix[j, mask_i])
                # Best continuation from i
                best_from_i = np.min(distance_matrix[i, mask_j])
                
                # Two-step path cost estimate
                est_path_cost = distance_matrix[i, j] + (best_from_i + best_from_j) / 2
                
                # Compare with alternative: use second-best from i to somewhere else
                # plus second-best from that node to j
                regret = est_path_cost - (second_best[i] + second_best[j])
                
                # Convert regret to score (lower regret = higher score)
                lookahead_score[i, j] = lookahead_score[j, i] = 1.0 / (1.0 + max(0, regret) / avg_dist)
    
    # 4. DYNAMIC FACTOR WEIGHTING
    # Analyze instance characteristics
    density_factor = np.mean(inv_dist[inv_dist > 0]) / np.max(inv_dist)
    size_factor = min(1.0, n / 100.0)
    
    # Adjust weights based on instance properties
    w_mutual = 0.35 + 0.15 * (1 - density_factor)  # Higher for sparse instances
    w_mst = 0.25 + 0.15 * size_factor  # Higher for larger instances
    w_lookahead = 0.30 + 0.10 * density_factor  # Higher for dense instances
    w_inv_dist = 0.10  # Base factor
    
    # Normalize weights
    total = w_mutual + w_mst + w_lookahead + w_inv_dist
    w_mutual /= total
    w_mst /= total
    w_lookahead /= total
    w_inv_dist /= total
    
    # 5. COMBINE FACTORS WITH DYNAMIC WEIGHTS
    # Normalize each factor to [0,1] range
    def normalize_matrix(mat):
        max_val = np.max(mat)
        return mat / max_val if max_val > 0 else mat
    
    mutual_norm = normalize_matrix(mutual_score)
    mst_norm = normalize_matrix(mst_factor)
    lookahead_norm = normalize_matrix(lookahead_score)
    inv_dist_norm = normalize_matrix(inv_dist)
    
    # Weighted combination
    combined = (
        w_inv_dist * inv_dist_norm +
        w_mutual * mutual_norm +
        w_mst * mst_norm +
        w_lookahead * lookahead_norm
    )
    
    # 6. MULTI-STAGE SPARSIFICATION WITH CONNECTIVITY PRESERVATION
    # Stage 1: Keep top edges per node based on adaptive threshold
    sparsified = combined.copy()
    adaptive_top_k = max(3, min(10, n // 5))
    
    for i in range(n):
        row = sparsified[i].copy()
        # Keep top adaptive_top_k edges
        if adaptive_top_k < n - 1:
            threshold_idx = n - adaptive_top_k - 1
            threshold = np.partition(row, threshold_idx)[threshold_idx]
            row[row < threshold] = 0
            sparsified[i] = row
    
    # Stage 2: Ensure graph connectivity by keeping MST edges
    sparsified[mst_factor > 0] = np.maximum(sparsified[mst_factor > 0], 0.5)
    
    # Stage 3: Symmetrize with preference for mutual decisions
    symmetric = np.maximum(sparsified, sparsified.T * 0.8)
    
    # Stage 4: Final cleanup and scaling
    np.fill_diagonal(symmetric, 0)
    max_val = np.max(symmetric)
    if max_val > 0:
        symmetric = symmetric / max_val
    
    # Ensure all nodes have at least 2 connections
    for i in range(n):
        if np.sum(symmetric[i] > 0) < 2:
            # Find top 2 edges (excluding diagonal)
            row = symmetric[i].copy()
            row[i] = -np.inf
            top2_idx = np.argpartition(-row, 2)[:2]
            symmetric[i, top2_idx] = np.maximum(symmetric[i, top2_idx], 0.3)
            symmetric[top2_idx, i] = np.maximum(symmetric[top2_idx, i], 0.3)
    
    return symmetric
