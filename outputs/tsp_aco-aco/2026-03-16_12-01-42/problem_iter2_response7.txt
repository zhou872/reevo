import numpy as np
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced heuristics for TSP edge selection featuring adaptive mechanisms.
    
    Core innovations:
    1. Adaptive regret-based lookahead that considers both immediate and future costs
    2. Dynamic factor weighting based on instance characteristics
    3. Multi-scale triangle inequality evaluation
    4. Context-aware edge sparsification
    """
    n = distance_matrix.shape[0]
    
    # Edge case handling
    if n <= 2:
        base = np.ones_like(distance_matrix)
        np.fill_diagonal(base, 0)
        return base
    
    # 1. ADAPTIVE REGRET-BASED EDGE PRIORITIZATION
    # Compute regret as the difference between best and second-best connections
    def compute_regret_matrix(dist_mat: np.ndarray) -> np.ndarray:
        """Calculate regret-based scores for each edge"""
        regret_mat = np.zeros((n, n))
        
        for i in range(n):
            # Get sorted distances for node i (excluding self)
            valid_distances = dist_mat[i].copy()
            valid_distances[i] = np.inf
            sorted_indices = np.argsort(valid_distances)
            
            if len(sorted_indices) >= 2:
                # First and second best distances
                best_dist = valid_distances[sorted_indices[0]]
                second_best_dist = valid_distances[sorted_indices[1]]
                
                # Regret for each potential edge
                for j in range(n):
                    if j != i:
                        current_dist = dist_mat[i, j]
                        if np.isclose(current_dist, best_dist):
                            # This is the best edge - high priority
                            regret = second_best_dist - best_dist
                        else:
                            # Not the best edge - calculate potential loss
                            regret = max(0, current_dist - best_dist)
                        
                        # Normalize by best distance
                        if best_dist > 0:
                            regret_mat[i, j] = 1.0 / (1.0 + regret / best_dist)
        
        return (regret_mat + regret_mat.T) / 2  # Symmetrize
    
    # 2. DYNAMIC FACTOR WEIGHTING ENSEMBLE
    def compute_factor_weights(n: int, dist_mat: np.ndarray) -> dict:
        """Determine optimal weights for different heuristic factors"""
        # Analyze instance characteristics
        mean_dist = np.mean(dist_mat[dist_mat > 0])
        std_dist = np.std(dist_mat[dist_mat > 0])
        sparsity = std_dist / mean_dist if mean_dist > 0 else 1.0
        
        # Adaptive weights based on instance properties
        weights = {
            'regret': 0.4 + 0.2 * np.log1p(n) / np.log1p(50),  # Increases with n
            'inverse_dist': 0.3 - 0.1 * min(1.0, sparsity),    # Decreases with sparsity
            'triangle': 0.2 + 0.1 * (1.0 - min(1.0, sparsity)), # Increases with uniformity
            'centrality': 0.1 + 0.1 * min(1.0, n / 100)        # Increases with n
        }
        
        # Normalize to sum to 1
        total = sum(weights.values())
        return {k: v/total for k, v in weights.items()}
    
    # 3. MULTI-SCALE TRIANGLE INEQUALITY EVALUATION
    def compute_triangle_scores(dist_mat: np.ndarray) -> np.ndarray:
        """Evaluate triangle inequality at multiple scales"""
        tri_scores = np.ones((n, n))
        
        # Use efficient matrix operations
        for i in range(n):
            for j in range(i+1, n):
                if dist_mat[i, j] > 0:
                    # Consider multiple alternative paths
                    alt_paths = dist_mat[i, :] + dist_mat[:, j]
                    alt_paths[i] = alt_paths[j] = np.inf  # Exclude self
                    
                    # Multi-scale evaluation
                    best_alt = np.min(alt_paths)
                    if best_alt < np.inf:
                        ratio = dist_mat[i, j] / best_alt
                        
                        # Gradual penalty based on violation severity
                        if ratio < 1.0:
                            # Edge satisfies triangle inequality - bonus
                            tri_scores[i, j] = 1.2
                        elif ratio < 1.2:
                            tri_scores[i, j] = 1.0
                        elif ratio < 1.5:
                            tri_scores[i, j] = 0.7
                        elif ratio < 2.0:
                            tri_scores[i, j] = 0.4
                        else:
                            tri_scores[i, j] = 0.1
                        
                        tri_scores[j, i] = tri_scores[i, j]
        
        return tri_scores
    
    # 4. CONTEXT-AWARE EDGE SPARSIFICATION
    def adaptive_sparsification(scores: np.ndarray, dist_mat: np.ndarray) -> np.ndarray:
        """Intelligently prune edges while preserving promising connections"""
        sparsified = scores.copy()
        
        # Determine sparsity level based on instance size
        target_edges = min(n * 3, (n * (n - 1)) // 4)  # Adaptive target
        
        # Create edge list with scores
        edges = []
        for i in range(n):
            for j in range(i+1, n):
                edges.append((scores[i, j], i, j))
        
        # Sort by score
        edges.sort(reverse=True)
        
        # Create selection mask
        selection_mask = np.zeros((n, n), dtype=bool)
        selected_count = 0
        
        for score, i, j in edges:
            if selected_count < target_edges:
                selection_mask[i, j] = selection_mask[j, i] = True
                selected_count += 1
            else:
                # Keep only if it's among the best for both nodes
                node_i_best = np.argsort(-scores[i])[:min(5, n-1)]
                node_j_best = np.argsort(-scores[j])[:min(5, n-1)]
                
                if j in node_i_best or i in node_j_best:
                    selection_mask[i, j] = selection_mask[j, i] = True
        
        # Apply mask
        sparsified[~selection_mask] = 0
        
        return sparsified
    
    # MAIN COMPUTATION PIPELINE
    
    # Compute individual factors
    regret_scores = compute_regret_matrix(distance_matrix)
    
    # Inverse distance (regularized)
    safe_dist = distance_matrix.copy()
    np.fill_diagonal(safe_dist, np.inf)
    inv_dist = 1.0 / (safe_dist + 0.01 * np.mean(safe_dist[safe_dist < np.inf]))
    
    # Triangle inequality scores
    triangle_scores = compute_triangle_scores(distance_matrix)
    
    # Node centrality (local density)
    # Adaptive threshold based on instance size
    k_neighbors = max(3, int(np.sqrt(n)))
    centrality = np.zeros(n)
    for i in range(n):
        # Distance to k-th nearest neighbor
        sorted_dists = np.sort(distance_matrix[i])
        if len(sorted_dists) > k_neighbors:
            centrality[i] = 1.0 / (1.0 + sorted_dists[k_neighbors])
    
    centrality_bonus = np.outer(centrality, centrality)
    
    # Get dynamic weights
    weights = compute_factor_weights(n, distance_matrix)
    
    # Combine factors with dynamic weights
    combined = (
        weights['regret'] * regret_scores +
        weights['inverse_dist'] * inv_dist +
        weights['triangle'] * triangle_scores +
        weights['centrality'] * centrality_bonus
    )
    
    # Apply context-aware sparsification
    sparsified = adaptive_sparsification(combined, distance_matrix)
    
    # Final normalization and cleanup
    np.fill_diagonal(sparsified, 0)
    max_val = np.max(sparsified)
    if max_val > 0:
        sparsified = sparsified / max_val
    
    return sparsified
