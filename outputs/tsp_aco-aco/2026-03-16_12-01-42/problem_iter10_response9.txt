import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Advanced TSP edge prioritization combining robust spatial statistics,
    adaptive regret, lightweight centrality, and data-driven fusion.
    
    Parameters
    ----------
    distance_matrix : np.ndarray
        n x n symmetric distance matrix with non-negative entries
        
    Returns
    -------
    np.ndarray
        Heuristic matrix of same shape, higher values indicate
        more promising edges for TSP inclusion
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-12
    
    # === 1. Base inverse distance ===
    inv_dist = 1 / (distance_matrix + epsilon)
    np.fill_diagonal(inv_dist, 0)
    
    # === 2. Robust Hybrid Spatial Scoring ===
    spatial = np.ones_like(distance_matrix)
    if n > 2:
        k_spatial = max(3, min(10, int(np.sqrt(n))))
        
        # Precompute per-node spatial statistics
        harm_means = np.zeros(n)
        geom_means = np.zeros(n)
        
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            if len(dists) >= k_spatial:
                # Get k nearest neighbors
                nearest_idx = np.argpartition(dists, k_spatial)[:k_spatial]
                nearest_dists = dists[nearest_idx]
                
                # Harmonic mean
                if np.any(nearest_dists > 0):
                    harm_means[i] = k_spatial / np.sum(1 / (nearest_dists + epsilon))
                
                # Geometric mean (log-sum-exp for numerical stability)
                valid_dists = nearest_dists[nearest_dists > 0]
                if len(valid_dists) > 0:
                    geom_means[i] = np.exp(np.mean(np.log(valid_dists + epsilon)))
        
        # Combine harmonic and geometric means per node
        valid_nodes = (harm_means > 0) & (geom_means > 0)
        if np.any(valid_nodes):
            # Use geometric mean of harmonic and geometric means
            combined_means = np.sqrt(harm_means * geom_means)
            
            # Compute edge spatial scores
            for i in range(n):
                if valid_nodes[i]:
                    for j in range(n):
                        if i != j and valid_nodes[j]:
                            # Harmonic mean of combined node measures
                            h_mean = 2 * combined_means[i] * combined_means[j] / \
                                    (combined_means[i] + combined_means[j] + epsilon)
                            
                            # Compare with actual distance
                            ratio = h_mean / (distance_matrix[i, j] + epsilon)
                            # Sigmoid transformation for bounded output
                            spatial[i, j] = 1.0 / (1.0 + np.exp(-np.log(ratio + 1)))
            
            # Normalize off-diagonal entries to [0.5, 1.5]
            off_diag = ~np.eye(n, dtype=bool)
            if np.any(spatial[off_diag] > 0):
                min_val = np.min(spatial[off_diag])
                max_val = np.max(spatial[off_diag])
                if max_val > min_val:
                    spatial[off_diag] = 0.5 + 1.0 * (spatial[off_diag] - min_val) / (max_val - min_val)
    
    np.fill_diagonal(spatial, 0)
    
    # === 3. Adaptive Regret with Continuous Order ===
    regret = np.ones_like(distance_matrix)
    
    if n > 2:
        # Compute graph statistics
        off_diag = ~np.eye(n, dtype=bool)
        global_dists = distance_matrix[off_diag]
        mean_dist = np.mean(global_dists)
        std_dist = np.std(global_dists)
        cv = std_dist / (mean_dist + epsilon)
        
        # Continuous regret order based on coefficient of variation
        # Maps CV ∈ [0,∞) to order ∈ [2,5]
        regret_order = 2.0 + 3.0 * (1.0 - np.exp(-cv))
        
        for i in range(n):
            dists = distance_matrix[i].copy()
            dists[i] = np.inf
            
            if len(dists) > 1:
                # Use fractional order: linear interpolation between floors
                order_floor = int(np.floor(regret_order))
                order_ceil = order_floor + 1
                
                # Get distances at order positions
                if order_floor < len(dists):
                    idx_floor = np.argpartition(dists, order_floor-1)[order_floor-1]
                    ref_floor = dists[idx_floor]
                else:
                    ref_floor = np.max(dists)
                
                if order_ceil < len(dists):
                    idx_ceil = np.argpartition(dists, order_ceil-1)[order_ceil-1]
                    ref_ceil = dists[idx_ceil]
                else:
                    ref_ceil = ref_floor
                
                # Linear interpolation
                alpha = regret_order - order_floor
                ref_dist = (1 - alpha) * ref_floor + alpha * ref_ceil
                
                if ref_dist > 0:
                    for j in range(n):
                        if i != j:
                            d_ij = distance_matrix[i, j]
                            if d_ij < ref_dist:
                                penalty = (ref_dist - d_ij) / (ref_dist + epsilon)
                                # Adaptive smoothing based on graph uniformity
                                smooth = 1.0 / (1.0 + 0.5 * cv * penalty)
                                regret[i, j] = 1.0 + smooth * penalty
    
    # === 4. Lightweight Centrality Bonus ===
    centrality = np.ones_like(distance_matrix)
    
    if n > 2:
        # Compute closeness centrality (inverse average distance)
        node_centrality = np.zeros(n)
        for i in range(n):
            dists = distance_matrix[i]
            mask = np.arange(n) != i
            if np.any(mask):
                avg_dist = np.mean(dists[mask])
                if avg_dist > 0:
                    node_centrality[i] = 1.0 / avg_dist
        
        # Normalize centrality scores to [0.5, 1.5]
        if np.max(node_centrality) > 0:
            node_centrality = 0.5 + 1.0 * (node_centrality - np.min(node_centrality)) / \
                             (np.max(node_centrality) - np.min(node_centrality) + epsilon)
        
        # Apply centrality bonus to edges
        for i in range(n):
            for j in range(n):
                if i != j:
                    # Geometric mean of node centralities
                    cent_geom = np.sqrt(node_centrality[i] * node_centrality[j])
                    # Boost edges connecting nodes with complementary centrality
                    # (balances exploration/exploitation)
                    centrality[i, j] = 0.8 + 0.4 * np.tanh(cent_geom - 1.0)
    
    # === 5. Efficiency Scoring (Two-Step Lookahead) ===
    efficiency = np.ones_like(distance_matrix)
    
    if n > 3:
        # Precompute nearest neighbors for efficiency check
        k_eff = min(4, n-1)
        neighbor_indices = np.argsort(distance_matrix, axis=1)[:, 1:k_eff+1]
        
        for i in range(n):
            for j in range(n):
                if i != j:
                    d_ij = distance_matrix[i, j]
                    
                    # Check alternative two-step paths through i's neighbors
                    best_alt = np.inf
                    for k in neighbor_indices[i]:
                        if k != j:
                            alt = distance_matrix[i, k] + distance_matrix[k, j]
                            if alt < best_alt:
                                best_alt = alt
                    
                    if best_alt < np.inf and d_ij > 0:
                        ratio = best_alt / d_ij
                        # Only boost edges with no efficient alternatives
                        if ratio > 1.15:
                            efficiency[i, j] = 1.0 + 0.25 * np.tanh(ratio - 1.0)
    
    # === 6. Data-Driven Multiplicative Fusion ===
    # Compute adaptive weights based on graph characteristics
    if n > 2:
        off_diag = ~np.eye(n, dtype=bool)
        global_dists = distance_matrix[off_diag]
        mean_dist = np.mean(global_dists)
        std_dist = np.std(global_dists)
        cv = std_dist / (mean_dist + epsilon)
        
        # Additional statistic: spatial skewness
        skewness = np.mean((global_dists - mean_dist)**3) / (std_dist**3 + epsilon)
        
        # Weight adaptation based on CV and skewness
        # High CV: emphasize regret and centrality
        # High positive skew: emphasize spatial harmony
        w_inv = 0.9 + 0.2 * np.tanh(cv - 1.0)
        w_spatial = 1.1 - 0.3 * np.tanh(skewness)
        w_regret = 1.0 + 0.3 * np.tanh(cv - 0.5)
        w_centrality = 0.8 + 0.4 * np.tanh(cv)
        w_efficiency = 0.9 + 0.2 * np.tanh(1.0 - cv)
    else:
        w_inv = w_spatial = w_regret = w_centrality = w_efficiency = 1.0
    
    # Multiplicative fusion with exponent weights
    heuristics = (inv_dist ** w_inv) * \
                 (spatial ** w_spatial) * \
                 (regret ** w_regret) * \
                 (centrality ** w_centrality) * \
                 (efficiency ** w_efficiency)
    
    # === 7. MST Reinforcement ===
    if n > 2:
        try:
            mst = minimum_spanning_tree(distance_matrix)
            mst_dense = mst.toarray()
            mst_present = (mst_dense > 0) | (mst_dense.T > 0)
            # Adaptive MST boost based on graph size
            mst_boost = 1.2 + 0.2 * np.tanh(20.0 / n)
            heuristics[mst_present] *= mst_boost
        except:
            pass
    
    # === 8. Adaptive Sparsification ===
    for i in range(n):
        row = heuristics[i].copy()
        
        if n > 3:
            # Determine keep threshold based on row statistics
            nonzero_mask = row > 0
            if np.any(nonzero_mask):
                row_nonzero = row[nonzero_mask]
                q75 = np.percentile(row_nonzero, 75)
                q25 = np.percentile(row_nonzero, 25)
                iqr = q75 - q25
                
                # Dynamic threshold: median + 0.5*IQR
                threshold = np.median(row_nonzero) + 0.5 * iqr
                
                # Adjust based on node's spatial characteristics
                if i < len(spatial):
                    spatial_i = np.mean(spatial[i, nonzero_mask])
                    threshold *= (0.9 + 0.2 * spatial_i)
                
                # Apply threshold
                mask = row >= threshold
                
                # Ensure minimum connectivity (2-4 edges based on n)
                min_edges = max(2, min(4, n // 5))
                if np.sum(mask) < min_edges:
                    top_idx = np.argsort(row)[-min_edges:]
                    mask[top_idx] = True
                
                row[~mask] = 0
                heuristics[i] = row
    
    # === 9. Symmetrization and Final Processing ===
    # Symmetrize using geometric mean for multiplicative consistency
    heuristics_sym = np.sqrt(heuristics * heuristics.T)
    np.fill_diagonal(heuristics_sym, 0)
    
    # Final normalization with enhanced contrast
    heuristics_sym = np.maximum(heuristics_sym, 0)
    max_val = np.max(heuristics_sym)
    if max_val > 0:
        heuristics_sym = heuristics_sym / max_val
    
    # Soft thresholding to emphasize top edges
    heuristics_sym = np.power(heuristics_sym, 0.8)
    
    return heuristics_sym
