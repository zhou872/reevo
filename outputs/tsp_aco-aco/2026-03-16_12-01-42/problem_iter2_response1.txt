import numpy as np
from scipy.sparse.csgraph import minimum_spanning_tree
from sklearn.manifold import MDS
from typing import Tuple

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristics for TSP edge selection incorporating lookahead simulations
    and adaptive ensemble methods for improved global optimization.
    
    Key innovations:
    1. Two-step lookahead simulation to estimate edge impact on future path segments
    2. Adaptive ensemble weighting based on instance characteristics
    3. Local optimum escape mechanism via escape bonus
    4. Cluster-aware density correction for non-uniform node distributions
    """
    n = distance_matrix.shape[0]
    if n < 3:
        return np.ones_like(distance_matrix) - np.eye(n)
    
    # Initialize result matrix
    heuristics = np.zeros_like(distance_matrix)
    np.fill_diagonal(distance_matrix, np.inf)
    
    # Calculate instance characteristics for adaptive weighting
    instance_stats = _analyze_instance(distance_matrix, n)
    
    # Component 1: Baseline inverse distance with dynamic scaling
    inv_dist = np.zeros_like(distance_matrix)
    mask = distance_matrix > 0
    inv_dist[mask] = 1.0 / distance_matrix[mask]
    
    # Component 2: Two-step lookahead simulation
    lookahead_scores = _compute_lookahead_scores(distance_matrix, n, instance_stats)
    
    # Component 3: MST-based structural bonus
    mst_bonus = _compute_mst_bonus(distance_matrix, n)
    
    # Component 4: Regret penalty with local optimum escape
    regret_scores = _compute_regret_scores(distance_matrix, n, instance_stats)
    
    # Component 5: Cluster density correction
    density_correction = _compute_density_correction(distance_matrix, n)
    
    # Adaptive ensemble combination based on instance characteristics
    weights = _compute_adaptive_weights(instance_stats)
    
    # Combine components with adaptive weights
    components = [
        inv_dist * density_correction,
        lookahead_scores,
        mst_bonus,
        regret_scores
    ]
    
    for i, component in enumerate(components):
        heuristics += weights[i] * component
    
    # Symmetrize and normalize
    heuristics = (heuristics + heuristics.T) / 2
    np.fill_diagonal(heuristics, 0)
    
    # Adaptive sparsification based on instance density
    sparsity_factor = max(0.1, min(0.4, instance_stats['sparsity']))
    k_keep = max(3, int(n * sparsity_factor))
    
    heuristics = _adaptive_sparsify(heuristics, k_keep, n)
    
    # Final normalization
    max_val = np.max(heuristics)
    if max_val > 0:
        heuristics /= max_val
    
    return heuristics

def _analyze_instance(distance_matrix: np.ndarray, n: int) -> dict:
    """Analyze TSP instance characteristics for adaptive parameter tuning."""
    stats = {}
    
    # Calculate sparsity metric
    avg_dist = np.mean(distance_matrix[distance_matrix < np.inf])
    std_dist = np.std(distance_matrix[distance_matrix < np.inf])
    stats['sparsity'] = std_dist / (avg_dist + 1e-10)
    
    # Check for geometric structure
    if n >= 10:
        try:
            # Sample points for MDS to detect Euclidean structure
            sample_size = min(50, n)
            indices = np.random.choice(n, sample_size, replace=False)
            sample_dist = distance_matrix[np.ix_(indices, indices)]
            
            mds = MDS(n_components=2, dissimilarity='precomputed', 
                     random_state=42, max_iter=100)
            _ = mds.fit_transform(sample_dist)
            stress = mds.stress_ / (sample_size * sample_size)
            stats['is_geometric'] = stress < 0.2
        except:
            stats['is_geometric'] = False
    else:
        stats['is_geometric'] = False
    
    # Measure cluster tendency
    nn_distances = np.partition(distance_matrix, 3, axis=1)[:, :3]
    avg_nn_dist = np.mean(nn_distances)
    stats['cluster_tendency'] = np.std(nn_distances) / (avg_nn_dist + 1e-10)
    
    return stats

def _compute_lookahead_scores(distance_matrix: np.ndarray, n: int, 
                            stats: dict) -> np.ndarray:
    """Compute two-step lookahead simulation scores for each edge."""
    scores = np.zeros((n, n))
    
    # For each potential edge (i,j), simulate two-step lookahead
    for i in range(n):
        row_i = distance_matrix[i].copy()
        row_i[i] = np.inf
        
        # Get candidate edges from i
        candidates_i = np.argsort(row_i)[:min(10, n-1)]
        
        for j in candidates_i:
            if i == j:
                continue
            
            # Simulate two-step lookahead: i -> j -> best_k
            # Exclude i and j from consideration for k
            row_j = distance_matrix[j].copy()
            row_j[i] = np.inf
            row_j[j] = np.inf
            
            if np.all(row_j == np.inf):
                continue
                
            # Find best continuation from j
            k = np.argmin(row_j)
            lookahead_cost = distance_matrix[i, j] + distance_matrix[j, k]
            
            # Compare with alternative from i
            row_i_excl_j = row_i.copy()
            row_i_excl_j[j] = np.inf
            alt_k = np.argmin(row_i_excl_j)
            alt_cost = distance_matrix[i, alt_k]
            
            if np.min(row_i_excl_j) < np.inf:
                # Score based on relative advantage
                advantage = alt_cost / lookahead_cost if lookahead_cost > 0 else 1.0
                scores[i, j] = min(2.0, max(0.5, advantage))
            else:
                scores[i, j] = 1.0
    
    return scores

def _compute_mst_bonus(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Compute MST-based bonus with additional connectivity analysis."""
    bonus = np.ones((n, n))
    
    # Standard MST bonus
    mst = minimum_spanning_tree(distance_matrix)
    mst_dense = mst.toarray()
    mst_edges = np.where(mst_dense > 0)
    
    for i, j in zip(*mst_edges):
        if i != j:
            bonus[i, j] = 1.5
            bonus[j, i] = 1.5
    
    # Additional bonus for edges that connect MST components in nearest neighbor graph
    k_nn = min(5, n-1)
    for i in range(n):
        nn_indices = np.argsort(distance_matrix[i])[1:k_nn+1]
        for j in nn_indices:
            if mst_dense[i, j] == 0 and mst_dense[j, i] == 0:
                # Check if edge bridges different MST subtrees in NN graph
                i_nn = set(np.argsort(distance_matrix[i])[1:3])
                j_nn = set(np.argsort(distance_matrix[j])[1:3])
                if len(i_nn & j_nn) == 0:
                    bonus[i, j] *= 1.2
    
    return bonus

def _compute_regret_scores(distance_matrix: np.ndarray, n: int, 
                          stats: dict) -> np.ndarray:
    """Compute regret scores with local optimum escape mechanism."""
    scores = np.ones((n, n))
    
    # Base regret calculation
    for i in range(n):
        row = distance_matrix[i].copy()
        row[i] = np.inf
        
        if np.all(row == np.inf):
            continue
            
        # Find two best alternatives
        sorted_indices = np.argsort(row)
        if len(sorted_indices) >= 2:
            d1_idx, d2_idx = sorted_indices[:2]
            d1, d2 = row[d1_idx], row[d2_idx]
            
            if d1 > 0:
                regret_ratio = d2 / d1
                # Original regret penalty
                regret_penalty = min(1.0, 1.0 / regret_ratio)
                
                # Add escape bonus for edges that are not the absolute minimum
                # but have good alternatives from their endpoint
                for j in range(n):
                    if i != j and distance_matrix[i, j] < np.inf:
                        if np.abs(distance_matrix[i, j] - d1) < 1e-10:
                            # This is the best edge
                            scores[i, j] = regret_penalty
                        else:
                            # Not the best edge - check if it has escape potential
                            row_j = distance_matrix[j].copy()
                            row_j[i] = np.inf
                            row_j[j] = np.inf
                            
                            if np.any(row_j < np.inf):
                                j_best = np.min(row_j)
                                if j_best < distance_matrix[i, j] * 0.8:
                                    # Good escape opportunity
                                    scores[i, j] = min(1.5, 1.0 + (d1 / distance_matrix[i, j]))
    
    return scores

def _compute_density_correction(distance_matrix: np.ndarray, n: int) -> np.ndarray:
    """Apply cluster-aware density correction."""
    correction = np.ones((n, n))
    
    # Estimate local densities using k-nearest neighbors
    k_density = min(7, n-1)
    densities = np.zeros(n)
    
    for i in range(n):
        nn_dists = np.partition(distance_matrix[i], k_density)[:k_density+1]
        densities[i] = 1.0 / (np.mean(nn_dists[nn_dists > 0]) + 1e-10)
    
    # Normalize densities
    if np.max(densities) > 0:
        densities = densities / np.max(densities)
    
    # Apply correction: penalize edges between dense areas and sparse areas
    for i in range(n):
        for j in range(n):
            if i != j and distance_matrix[i, j] < np.inf:
                density_diff = abs(densities[i] - densities[j])
                if density_diff > 0.5:  # Significant density difference
                    # Favor edges within similar density regions
                    correction[i, j] = 1.0 - 0.3 * density_diff
    
    return correction

def _compute_adaptive_weights(stats: dict) -> np.ndarray:
    """Compute adaptive weights for ensemble combination."""
    weights = np.ones(4)
    
    # Weight adjustment based on instance characteristics
    if stats.get('is_geometric', False):
        # For geometric instances, emphasize lookahead and density correction
        weights = [0.3, 0.4, 0.2, 0.1]
    elif stats['cluster_tendency'] > 0.5:
        # For clustered instances, emphasize regret and MST
        weights = [0.2, 0.2, 0.4, 0.2]
    else:
        # Balanced weights for mixed instances
        weights = [0.25, 0.3, 0.25, 0.2]
    
    # Adjust for sparsity
    if stats['sparsity'] > 0.7:
        # Sparse instances: emphasize baseline and lookahead
        weights = [weights[0] * 1.3, weights[1] * 1.2, 
                  weights[2] * 0.8, weights[3] * 0.7]
    
    # Normalize
    weights = np.array(weights) / np.sum(weights)
    return weights

def _adaptive_sparsify(heuristics: np.ndarray, k_keep: int, n: int) -> np.ndarray:
    """Adaptively sparsify the heuristics matrix while preserving connectivity."""
    sparsified = np.zeros_like(heuristics)
    
    for i in range(n):
        row = heuristics[i].copy()
        row[i] = -np.inf
        
        # Keep top k_keep edges, but ensure at least 2 edges per node
        k_actual = max(2, min(k_keep, np.sum(row > 0)))
        
        if k_actual > 0:
            threshold = np.partition(row, -k_actual)[-k_actual]
            mask = row >= threshold
            sparsified[i, mask] = heuristics[i, mask]
    
    # Ensure graph connectivity by checking degree
    degrees = np.sum(sparsified > 0, axis=1)
    for i in range(n):
        if degrees[i] < 2:
            # Add back the best missing edge
            row = heuristics[i].copy()
            row[i] = -np.inf
            row[sparsified[i] > 0] = -np.inf
            if np.any(row > 0):
                best_idx = np.argmax(row)
                sparsified[i, best_idx] = heuristics[i, best_idx]
    
    # Symmetrize: keep edge if promising in either direction
    symmetric_mask = (sparsified > 0) | (sparsified.T > 0)
    final_heuristics = np.zeros_like(heuristics)
    final_heuristics[symmetric_mask] = heuristics[symmetric_mask]
    
    return final_heuristics
