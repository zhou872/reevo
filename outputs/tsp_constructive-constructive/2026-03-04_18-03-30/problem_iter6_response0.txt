import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Advanced node selection for TSP constructive heuristics using three-step look-ahead
    with dynamic regret calculation and adaptive weight adjustment.
    
    Args:
        current_node: Index of current node.
        destination_node: Index of final destination node.
        unvisited_nodes: Set of nodes not yet visited.
        distance_matrix: 2D numpy array of pairwise distances.
    
    Returns:
        Index of the next node to visit.
    """
    # Early termination cases
    if not unvisited_nodes:
        return destination_node
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Dynamic weight adjustment based on problem size and progress
    n_total = distance_matrix.shape[0]
    progress = 1.0 - len(unvisited_nodes) / (n_total - 1)  # 0 to 1
    
    # Weights shift from future-looking to immediate cost as tour progresses
    if n_total < 30:  # Small problems: balanced approach
        w_imm = 0.5
        w_future = 0.35
        w_regret = 0.15
    else:  # Large problems: progress-dependent weights
        # Early stages: focus more on future potential
        # Late stages: focus more on immediate connections
        w_imm = 0.3 + 0.4 * progress  # 0.3 → 0.7
        w_future = 0.6 - 0.4 * progress  # 0.6 → 0.2
        w_regret = 0.1  # Constant regret importance
    
    # Normalize weights
    total = w_imm + w_future + w_regret
    w_imm /= total
    w_future /= total
    w_regret /= total
    
    # Pre-compute distances for efficiency
    n_unvisited = len(unvisited_nodes)
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # Skip destination until it's the only node left
        if candidate == destination_node and n_unvisited > 1:
            continue
        
        # 1. Immediate cost component
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Three-step look-ahead future cost estimation
        future_cost = 0
        if n_unvisited > 1:
            # Create remaining nodes after candidate
            remaining = unvisited_nodes - {candidate}
            
            if len(remaining) == 1:
                # Only one node left after candidate
                last_node = next(iter(remaining))
                future_cost = (distance_matrix[candidate][last_node] +
                             distance_matrix[last_node][destination_node])
            else:
                # Estimate future cost using three-step look-ahead
                future_costs = []
                
                for next_node in remaining:
                    # Second step: candidate → next_node
                    cost_to_next = distance_matrix[candidate][next_node]
                    
                    # Third step: estimate minimum from next_node
                    remaining_after_next = remaining - {next_node}
                    if remaining_after_next:
                        # Find best node to go to after next_node
                        min_third_step = float('inf')
                        for third_node in remaining_after_next:
                            cost_third = distance_matrix[next_node][third_node]
                            # Estimate remaining path from third_node
                            remaining_after_third = remaining_after_next - {third_node}
                            if remaining_after_third:
                                # Use minimum distance to any remaining node + to destination
                                min_to_rest = min(distance_matrix[third_node][node] 
                                                for node in remaining_after_third)
                                est_rest = min_to_rest + 0.5 * distance_matrix[third_node][destination_node]
                            else:
                                # Go directly to destination
                                est_rest = distance_matrix[third_node][destination_node]
                            
                            total_third_step = cost_third + est_rest
                            min_third_step = min(min_third_step, total_third_step)
                    else:
                        # Go directly to destination from next_node
                        min_third_step = distance_matrix[next_node][destination_node]
                    
                    future_costs.append(cost_to_next + min_third_step)
                
                # Use the minimum future cost
                future_cost = min(future_costs)
        
        # 3. Dynamic regret calculation
        regret = 0
        if n_unvisited > 1:
            # Calculate average distance from candidate to remaining nodes
            other_nodes = unvisited_nodes - {candidate}
            avg_dist_to_others = np.mean([distance_matrix[candidate][node] 
                                        for node in other_nodes])
            
            # Calculate minimum distance from current to any unvisited
            min_current_dist = min(distance_matrix[current_node][node] 
                                 for node in unvisited_nodes)
            
            # Dynamic regret: combines opportunity cost and connectivity
            regret = (avg_dist_to_others * 0.7 + 
                     max(0, immediate_cost - min_current_dist) * 0.3)
        
        # 4. Combine components with weights
        # Scale components for better balance
        scale_imm = 1.0
        scale_future = 0.8 / max(1, n_unvisited ** 0.5)  # Diminish with remaining nodes
        scale_regret = 0.5
        
        score = (w_imm * immediate_cost * scale_imm +
                w_future * future_cost * scale_future +
                w_regret * regret * scale_regret)
        
        # 5. Special consideration for nodes that are good stepping stones
        # Favor nodes that are well-connected to remaining nodes
        if n_unvisited > 2:
            # Calculate connectivity score (lower is better)
            connectivity = np.std([distance_matrix[candidate][node] 
                                 for node in unvisited_nodes if node != candidate])
            # Small penalty for high variance connections
            score *= (1.0 + 0.05 * connectivity)
        
        # Update best node
        if score < best_score:
            best_score = score
            best_node = candidate
    
    # Fallback: return the closest node if no candidate selected
    if best_node is None:
        # Find closest unvisited node (excluding destination if other options exist)
        available = [node for node in unvisited_nodes 
                    if not (node == destination_node and n_unvisited > 1)]
        if available:
            return min(available, key=lambda x: distance_matrix[current_node][x])
        return destination_node
    
    return best_node
