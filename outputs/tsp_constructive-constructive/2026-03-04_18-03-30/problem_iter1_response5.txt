import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """Select next node using two-step lookahead with adaptive scoring."""
    
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Adaptive parameters based on search progress
    progress = 1 - (len(unvisited_nodes) / distance_matrix.shape[0])
    lookahead_depth = max(1, int(3 * progress))  # Deeper lookahead early, shallower later
    diversity_weight = 0.3 * (1 - progress)  # Higher early, lower late
    
    candidates = list(unvisited_nodes)
    scores = []
    
    for candidate in candidates:
        # Immediate cost
        immediate_cost = distance_matrix[current_node][candidate]
        
        # Lookahead: simulate next step after candidate
        remaining_nodes = unvisited_nodes - {candidate}
        
        if len(remaining_nodes) == 0:
            # Last node before destination
            future_cost = distance_matrix[candidate][destination_node]
            total_cost = immediate_cost + future_cost
            scores.append((candidate, total_cost))
            continue
            
        # Evaluate several potential next steps
        lookahead_costs = []
        test_nodes = list(remaining_nodes)
        
        # Limit lookahead to avoid combinatorial explosion
        if len(test_nodes) > lookahead_depth:
            # Select diverse test nodes
            test_nodes = sorted(test_nodes, 
                              key=lambda x: distance_matrix[candidate][x])[:lookahead_depth]
        
        for test_node in test_nodes:
            # Two-step cost: current -> candidate -> test_node
            two_step = immediate_cost + distance_matrix[candidate][test_node]
            
            # Estimate remaining path cost from test_node
            future_remaining = remaining_nodes - {test_node}
            
            if len(future_remaining) == 0:
                # Only destination remains
                future_estimate = distance_matrix[test_node][destination_node]
            else:
                # Estimate using nearest neighbor and average
                nearest_to_dest = min(distance_matrix[node][destination_node] 
                                    for node in future_remaining)
                avg_to_remaining = np.mean([distance_matrix[test_node][node] 
                                          for node in future_remaining])
                
                # Blend estimates based on remaining nodes count
                future_estimate = (0.6 * nearest_to_dest + 
                                 0.4 * avg_to_remaining * len(future_remaining))
            
            lookahead_costs.append(two_step + future_estimate)
        
        # Use minimum lookahead cost for this candidate
        best_lookahead = min(lookahead_costs)
        
        # Diversity bonus: prefer nodes that are far from recently chosen ones
        if len(scores) > 0:
            avg_distance_to_others = np.mean([distance_matrix[candidate][s[0]] 
                                            for s in scores[:3] if len(scores) >= 3])
            diversity_bonus = diversity_weight * avg_distance_to_others
        else:
            diversity_bonus = 0
        
        # Final score (lower is better)
        final_score = best_lookahead - diversity_bonus
        scores.append((candidate, final_score))
    
    # Return candidate with minimum score
    return min(scores, key=lambda x: x[1])[0]
