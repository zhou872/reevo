import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Advanced node selection for TSP constructive heuristics with adaptive weighting,
    three-step look-ahead, and dynamic regret calculation.
    
    Features:
    1. Adaptive weights that adjust based on construction progress
    2. Three-step look-ahead for better future cost estimation
    3. Dynamic regret calculation considering opportunity costs
    4. Progressive depth adjustment based on remaining nodes
    """
    
    # Single node case
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Adaptive parameters based on construction progress
    total_nodes = distance_matrix.shape[0]
    visited_count = total_nodes - len(unvisited_nodes) - 1  # -1 for current node
    progress = visited_count / total_nodes
    
    # Weights adapt based on construction stage
    # Early stages: focus more on future costs and regret
    # Late stages: focus more on immediate costs
    w_immediate = 0.3 + 0.4 * progress  # Increases from 0.3 to 0.7
    w_future = 0.5 - 0.3 * progress     # Decreases from 0.5 to 0.2
    w_regret = 0.2 + 0.2 * (1 - progress)  # Decreases from 0.4 to 0.2
    
    scores = {}
    
    for candidate in unvisited_nodes:
        # 1. Immediate cost
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Look-ahead evaluation
        remaining_after_candidate = unvisited_nodes - {candidate}
        
        if not remaining_after_candidate:
            # Last node before destination
            future_estimate = distance_matrix[candidate][destination_node]
            regret = 0
        else:
            # Dynamic look-ahead depth based on remaining nodes
            if len(remaining_after_candidate) <= 2:
                look_ahead_depth = 1  # Shallow look-ahead for few remaining nodes
            elif len(remaining_after_candidate) <= 5:
                look_ahead_depth = 2  # Medium look-ahead
            else:
                look_ahead_depth = 3  # Deep look-ahead
            
            look_ahead_costs = []
            
            for next_candidate in remaining_after_candidate:
                # Two-step cost
                cost_to_next = distance_matrix[candidate][next_candidate]
                
                # Estimate additional costs with look-ahead
                if look_ahead_depth >= 2:
                    remaining_after_next = remaining_after_candidate - {next_candidate}
                    
                    if not remaining_after_next:
                        # Direct to destination
                        additional_cost = distance_matrix[next_candidate][destination_node]
                    elif look_ahead_depth == 2:
                        # Two-step look-ahead: choose best third node
                        additional_cost = min(
                            distance_matrix[next_candidate][third_candidate]
                            for third_candidate in remaining_after_next
                        )
                    else:
                        # Three-step look-ahead: candidate -> next -> third -> best remaining
                        three_step_costs = []
                        for third_candidate in remaining_after_next:
                            remaining_after_third = remaining_after_next - {third_candidate}
                            cost_to_third = distance_matrix[next_candidate][third_candidate]
                            
                            if not remaining_after_third:
                                # Third -> destination
                                cost_from_third = distance_matrix[third_candidate][destination_node]
                            else:
                                # Best from third to remaining or destination
                                cost_from_third = min(
                                    min(distance_matrix[third_candidate][node] 
                                        for node in remaining_after_third),
                                    distance_matrix[third_candidate][destination_node]
                                )
                            
                            three_step_costs.append(cost_to_third + cost_from_third)
                        
                        additional_cost = min(three_step_costs) if three_step_costs else 0
                    
                    look_ahead_costs.append(cost_to_next + additional_cost)
                else:
                    # One-step look-ahead
                    look_ahead_costs.append(cost_to_next)
            
            # Future estimate: minimum look-ahead cost
            future_estimate = min(look_ahead_costs) if look_ahead_costs else 0
            
            # 3. Dynamic regret calculation
            # Calculate opportunity cost considering different future paths
            if len(look_ahead_costs) >= 2:
                sorted_costs = np.sort(look_ahead_costs)
                best_cost = sorted_costs[0]
                second_best = sorted_costs[1] if len(sorted_costs) > 1 else best_cost
                
                # Regret as the cost difference between choosing this candidate vs alternatives
                # Adjusted by the candidate's immediate cost advantage
                immediate_advantage = immediate_cost - min(
                    distance_matrix[current_node][other] 
                    for other in unvisited_nodes if other != candidate
                )
                
                regret = (second_best - best_cost) * (1 + 0.5 * immediate_advantage / (immediate_cost + 1e-10))
            else:
                regret = 0
        
        # 4. Destination consideration
        dest_cost = distance_matrix[candidate][destination_node]
        w_destination = 0.1 * (1 - progress)  # Less important as we progress
        
        # Combined score (lower is better)
        # Note: regret is positive, we subtract it because higher regret means better opportunity
        combined_score = (
            w_immediate * immediate_cost +
            w_future * future_estimate +
            w_destination * dest_cost -
            w_regret * regret
        )
        
        scores[candidate] = combined_score
    
    # Select node with minimum score
    return min(scores, key=scores.get)
