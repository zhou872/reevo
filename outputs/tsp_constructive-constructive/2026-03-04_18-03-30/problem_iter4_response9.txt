import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int,
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Enhanced node selection with progress-adaptive weighting and cluster-awareness.
    
    Combines:
    1. Immediate proximity to current node (scaled by progress)
    2. Cluster centrality (average distance to other unvisited nodes)
    3. Cluster connectivity (minimum distance to other unvisited nodes)
    4. Destination guidance (progress-aware)
    
    Features:
    - Adaptive weights based on construction progress
    - Balanced exploration/exploitation trade-off
    - Lightweight cluster detection without complex computations
    """
    
    # Base case: only destination remains
    if unvisited_nodes == {destination_node}:
        return destination_node
    
    # Calculate progress through construction
    total_nodes = distance_matrix.shape[0]
    remaining = len(unvisited_nodes)
    progress = 1.0 - (remaining / total_nodes) if total_nodes > 0 else 0.5
    
    # Adaptive weights based on construction phase
    if progress < 0.3:  # Early phase: explore clusters
        w_current = 0.40      # Immediate distance
        w_avg = -0.35         # Cluster centrality (negative: prefer central nodes)
        w_min = -0.20         # Cluster connectivity (negative: prefer well-connected)
        w_dest = 0.05         # Destination guidance
    elif progress < 0.7:  # Middle phase: balance
        w_current = 0.45
        w_avg = -0.30
        w_min = -0.15
        w_dest = 0.10
    else:  # Late phase: focus on completion
        w_current = 0.50
        w_avg = -0.25
        w_min = -0.10
        w_dest = 0.15
    
    # Normalize weights
    weight_sum = abs(w_current) + abs(w_avg) + abs(w_min) + abs(w_dest)
    w_current /= weight_sum
    w_avg /= weight_sum
    w_min /= weight_sum
    w_dest /= weight_sum
    
    # Pre-filter candidates (avoid destination unless necessary)
    candidates = unvisited_nodes - {destination_node}
    if not candidates:  # Only destination remains
        return destination_node
    
    scores = {}
    # Pre-compute destination distances for all candidates
    dest_distances = {node: distance_matrix[node, destination_node] for node in candidates}
    
    for node in candidates:
        # 1. Immediate cost
        immediate_cost = distance_matrix[current_node, node]
        
        # 2. Cluster centrality: average distance to other unvisited nodes
        other_nodes = unvisited_nodes - {node}
        if other_nodes:
            distances_to_others = [distance_matrix[node, other] for other in other_nodes]
            avg_distance = np.mean(distances_to_others)
            min_distance = np.min(distances_to_others) if len(distances_to_others) > 0 else 0
        else:
            avg_distance = 0
            min_distance = 0
        
        # 3. Destination guidance with progress scaling
        # Early: mild preference, Late: stronger preference
        dest_penalty = dest_distances[node] * (0.5 + 0.5 * progress)
        
        # Combined score (minimization)
        score = (w_current * immediate_cost +
                w_avg * avg_distance +
                w_min * min_distance +
                w_dest * dest_penalty)
        
        scores[node] = score
    
    # Return best candidate, fall back to destination if no candidates
    return min(scores, key=scores.get) if scores else destination_node
