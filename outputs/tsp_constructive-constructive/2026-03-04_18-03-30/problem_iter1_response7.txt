import numpy as np
from typing import Set, Tuple

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select next node using a two-step lookahead heuristic with cluster awareness.
    
    Features:
    1. Two-step lookahead: Evaluates current->candidate->next sequences
    2. Cluster awareness: Penalizes nodes that would isolate remaining unvisited nodes
    3. Dynamic weighting: Adjusts weights based on progress and proximity to destination
    4. Greedy-lookahead hybrid: Balances immediate gain with future accessibility
    """
    
    # Dynamic weights adjust based on progress
    unvisited_count = len(unvisited_nodes)
    progress = 1.0 - (unvisited_count / len(distance_matrix))
    
    # Base weights that change with progress
    w_current = 0.4 * (1.0 - 0.3 * progress)  # Less emphasis on immediate distance as we progress
    w_lookahead = 0.3 + 0.2 * progress         # More emphasis on lookahead as we progress
    w_cluster = 0.2 + 0.1 * progress          # More emphasis on clusters as we progress
    w_destination = 0.1 + 0.2 * (1.0 - progress)  # More emphasis on destination early on
    
    # Normalize weights
    total_weight = w_current + w_lookahead + w_cluster + w_destination
    w_current /= total_weight
    w_lookahead /= total_weight
    w_cluster /= total_weight
    w_destination /= total_weight
    
    # For clustering penalty, compute centroid of remaining unvisited nodes
    if len(unvisited_nodes) > 1:
        # Simple centroid approximation using distances
        cluster_penalty_factors = {}
        for node in unvisited_nodes:
            # Sum of distances to all other unvisited nodes
            total_dist = sum(distance_matrix[node][other] 
                           for other in unvisited_nodes if other != node)
            cluster_penalty_factors[node] = total_dist / (len(unvisited_nodes) - 1)
    
    best_score = float('inf')
    best_node = None
    
    for candidate in unvisited_nodes:
        # 1. Immediate cost from current to candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Two-step lookahead: best option from candidate
        remaining_after_candidate = unvisited_nodes - {candidate}
        if remaining_after_candidate:
            # Find best next step from candidate
            lookahead_costs = []
            for next_candidate in remaining_after_candidate:
                # Consider the sequence: current -> candidate -> next_candidate
                seq_cost = (immediate_cost + 
                          distance_matrix[candidate][next_candidate])
                lookahead_costs.append(seq_cost)
            
            # Use the 30th percentile to avoid being too greedy
            lookahead_cost = np.percentile(lookahead_costs, 30)
        else:
            lookahead_cost = immediate_cost
        
        # 3. Cluster penalty: penalize nodes that isolate remaining clusters
        if len(unvisited_nodes) > 1:
            # If we pick this node, how isolated are the remaining nodes from each other?
            remaining_isolation = 0
            remaining_list = list(remaining_after_candidate)
            
            if len(remaining_list) >= 2:
                # Sample some pairs to estimate connectivity
                sample_pairs = min(5, len(remaining_list) * (len(remaining_list) - 1) // 2)
                for i in range(sample_pairs):
                    if i < len(remaining_list):
                        for j in range(i + 1, min(i + 3, len(remaining_list))):
                            remaining_isolation += distance_matrix[remaining_list[i]][remaining_list[j]]
                
                if sample_pairs > 0:
                    remaining_isolation /= sample_pairs
                
                # Also consider distance from candidate to remaining centroid
                avg_dist_to_remaining = sum(distance_matrix[candidate][node] 
                                          for node in remaining_list) / len(remaining_list)
                cluster_penalty = (remaining_isolation * 0.7 + avg_dist_to_remaining * 0.3)
            else:
                cluster_penalty = 0
        else:
            cluster_penalty = 0
        
        # 4. Destination attraction (inversely proportional to distance)
        if candidate == destination_node:
            # If this is the destination and we still have other nodes, 
            # only choose it if we're very close or it's the last node
            if len(unvisited_nodes) > 1:
                dest_attraction = immediate_cost * 2  # Penalize going to destination too early
            else:
                dest_attraction = immediate_cost
        else:
            # Attraction based on how much closer this gets us to destination
            current_to_dest = distance_matrix[current_node][destination_node]
            cand_to_dest = distance_matrix[candidate][destination_node]
            
            # Positive if candidate is closer to destination than current node
            dest_progress = current_to_dest - cand_to_dest
            dest_attraction = -dest_progress  # Negative because we want to minimize
        
        # Combine all factors with dynamic weights
        score = (w_current * immediate_cost +
                w_lookahead * lookahead_cost +
                w_cluster * cluster_penalty +
                w_destination * dest_attraction)
        
        # Additional heuristic: if we're close to destination, prefer nodes
        # that don't take us too far away
        if progress > 0.7:  # In last 30% of nodes
            current_to_dest = distance_matrix[current_node][destination_node]
            cand_to_dest = distance_matrix[candidate][destination_node]
            if cand_to_dest > current_to_dest * 1.5:  # Taking us much further from destination
                score *= 1.2  # Penalize
        
        if score < best_score:
            best_score = score
            best_node = candidate
    
    # Fallback: if no node selected, use nearest neighbor
    if best_node is None:
        best_node = min(unvisited_nodes, 
                       key=lambda x: distance_matrix[current_node][x])
    
    return best_node
