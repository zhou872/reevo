import numpy as np
from typing import Set, Tuple

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Enhanced node selection for TSP constructive heuristics with two-step look-ahead
    and cluster-aware penalty.
    """
    if not unvisited_nodes:
        raise ValueError("No unvisited nodes available")
    
    # Adaptive threshold based on problem scale
    n = len(distance_matrix)
    threshold = 0.7 if n > 20 else 0.9  # More exploration for larger problems
    
    # Dynamic weights based on remaining nodes
    remaining_ratio = len(unvisited_nodes) / n
    c1 = 0.4 * remaining_ratio  # Immediate cost weight decreases as tour progresses
    c2 = 0.3 * (1 - remaining_ratio)  # Centroid importance increases
    c3 = 0.2
    c4 = 0.1
    c5 = 0.3  # Two-step look-ahead weight
    
    # Calculate centroid of unvisited nodes (approximation)
    centroid_distances = np.zeros(n)
    if len(unvisited_nodes) > 1:
        for node in unvisited_nodes:
            # Average distance to other unvisited nodes (cluster cohesion)
            others = [distance_matrix[node][other] 
                     for other in unvisited_nodes if other != node]
            centroid_distances[node] = np.mean(others) if others else 0
    
    scores = {}
    for node in unvisited_nodes:
        # 1. Immediate cost
        immediate_cost = distance_matrix[current_node][node]
        
        # 2. Two-step look-ahead: minimal cost for next step
        remaining_after = unvisited_nodes - {node}
        if remaining_after:
            # Best possible next move from this candidate
            next_step_costs = [distance_matrix[node][next_node] 
                              for next_node in remaining_after]
            two_step_cost = min(next_step_costs) if next_step_costs else 0
        else:
            # Last node before destination
            two_step_cost = distance_matrix[node][destination_node]
        
        # 3. Cluster penalty (avoid isolating nodes)
        cluster_penalty = centroid_distances[node]
        
        # 4. Distance to other unvisited nodes statistics
        other_dists = [distance_matrix[node][other] 
                      for other in unvisited_nodes if other != node]
        avg_dist = np.mean(other_dists) if other_dists else 0
        std_dist = np.std(other_dists) if len(other_dists) > 1 else 0
        
        # 5. Distance to final destination
        dest_dist = distance_matrix[node][destination_node]
        
        # Composite score (lower is better)
        score = (c1 * immediate_cost + 
                 c5 * two_step_cost +
                 c2 * cluster_penalty -
                 c3 * avg_dist + 
                 c4 * std_dist - 
                 c4 * dest_dist)
        
        scores[node] = score
    
    # Adaptive selection: occasional exploration
    if np.random.random() > threshold:
        # Deterministic greedy selection
        next_node = min(scores, key=scores.get)
    else:
        # Probabilistic selection favoring good but not necessarily best
        scores_list = list(scores.items())
        weights = [1/(score + 1e-6) for _, score in scores_list]  # Convert to weights
        probs = np.array(weights) / sum(weights)
        next_node = np.random.choice([n for n, _ in scores_list], p=probs)
    
    return next_node
