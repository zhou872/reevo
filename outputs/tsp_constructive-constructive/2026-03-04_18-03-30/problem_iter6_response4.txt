import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, 
                        destination_node: int, 
                        unvisited_nodes: Set[int], 
                        distance_matrix: np.ndarray) -> int:
    """
    Select the next node to visit using an enhanced deterministic heuristic.
    
    The algorithm balances four key components:
    1. Immediate cost to candidate node
    2. Two-step lookahead with weighted regret
    3. Cluster coherence to maintain spatial efficiency
    4. Destination guidance for path completion
    
    Parameters:
    -----------
    current_node : int
        Current node index
    destination_node : int
        Final destination node index
    unvisited_nodes : Set[int]
        Set of unvisited node indices
    distance_matrix : np.ndarray
        Distance matrix between nodes (n x n)
    
    Returns:
    --------
    int : Next node to visit (deterministic choice)
    """
    if not unvisited_nodes:
        return destination_node
    
    # Normalize distances for consistent scaling
    max_dist = np.max(distance_matrix)
    min_dist = np.min(distance_matrix[distance_matrix > 0])
    if max_dist > min_dist:
        def normalize(d):
            return (d - min_dist) / (max_dist - min_dist)
    else:
        def normalize(d):
            return d / max_dist if max_dist > 0 else 0.0
    
    # Fixed weights for deterministic behavior
    WEIGHT_IMMEDIATE = 0.35    # Current to candidate distance
    WEIGHT_LOOKAHEAD = 0.30    # Two-step lookahead consideration
    WEIGHT_CLUSTER = 0.20      # Cluster coherence
    WEIGHT_DESTINATION = 0.15  # Destination guidance (negative weight)
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        if candidate == destination_node and len(unvisited_nodes) > 1:
            # Only choose destination as last node
            continue
        
        # 1. Immediate cost (normalized)
        immediate_cost = normalize(distance_matrix[current_node][candidate])
        
        # 2. Two-step lookahead with weighted regret
        remaining_after_candidate = unvisited_nodes - {candidate}
        
        if remaining_after_candidate:
            # Distances from candidate to remaining nodes
            cand_to_remaining = [distance_matrix[candidate][r] for r in remaining_after_candidate]
            
            # Lookahead: minimum distance from candidate to remaining nodes
            lookahead_min = min(cand_to_remaining) if cand_to_remaining else 0
            
            # Lightweight regret: difference between best and second best
            regret = 0.0
            if len(cand_to_remaining) >= 2:
                sorted_dists = sorted(cand_to_remaining)
                regret = sorted_dists[1] - sorted_dists[0]
                # Scale regret by number of alternatives
                regret *= (1.0 + 0.1 * np.log1p(len(cand_to_remaining)))
            
            # Combined lookahead with regret (normalized)
            lookahead_cost = normalize(lookahead_min + 0.3 * regret)
        else:
            # Last node before destination
            lookahead_cost = normalize(distance_matrix[candidate][destination_node])
        
        # 3. Cluster coherence (average distance to remaining cluster)
        cluster_penalty = 0.0
        if remaining_after_candidate:
            avg_to_cluster = np.mean([distance_matrix[candidate][r] 
                                     for r in remaining_after_candidate])
            cluster_penalty = normalize(avg_to_cluster)
        
        # 4. Destination guidance (closer to destination is better)
        dest_distance = normalize(distance_matrix[candidate][destination_node])
        
        # Combined score (lower is better)
        # Note: dest_distance is subtracted as we want to move toward destination
        score = (WEIGHT_IMMEDIATE * immediate_cost +
                 WEIGHT_LOOKAHEAD * lookahead_cost +
                 WEIGHT_CLUSTER * cluster_penalty -
                 WEIGHT_DESTINATION * dest_distance)
        
        # Update best node
        if score < best_score:
            best_score = score
            best_node = candidate
    
    # Fallback: if best_node is None (shouldn't happen), return destination
    return best_node if best_node is not None else destination_node
