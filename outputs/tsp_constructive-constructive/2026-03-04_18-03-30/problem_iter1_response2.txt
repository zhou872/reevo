import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Enhanced node selection with two-step lookahead and adaptive scoring.
    Balances immediate gains with future consequences through lookahead simulation.
    """
    
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Adaptive parameters based on remaining nodes
    remaining_ratio = len(unvisited_nodes) / (distance_matrix.shape[0] - 1)
    
    # Weights shift from exploration to exploitation as path builds
    w_immediate = 0.4 - 0.2 * remaining_ratio  # Decreases as we progress
    w_future = 0.3 + 0.2 * remaining_ratio     # Increases as we progress
    w_centrality = 0.2
    w_destination = 0.1
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # Immediate cost from current to candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # ----- TWO-STEP LOOKAHEAD -----
        # Simulate one step ahead from candidate
        future_costs = []
        remaining_after_candidate = unvisited_nodes - {candidate}
        
        if remaining_after_candidate:
            # Look one step ahead from candidate
            for next_candidate in remaining_after_candidate:
                # Cost from candidate to next_candidate
                cost1 = distance_matrix[candidate][next_candidate]
                
                # Estimated remaining cost from next_candidate
                # Using minimum distance to remaining nodes as heuristic
                remaining_nodes = remaining_after_candidate - {next_candidate}
                if remaining_nodes:
                    min_remaining = min(
                        distance_matrix[next_candidate][node] 
                        for node in remaining_nodes
                    )
                else:
                    min_remaining = distance_matrix[next_candidate][destination_node]
                
                future_costs.append(cost1 + 0.5 * min_remaining)
            
            future_cost = np.mean(future_costs) + 0.5 * np.std(future_costs)
        else:
            # Only candidate remains
            future_cost = distance_matrix[candidate][destination_node]
        
        # ----- CENTRALITY MEASURE -----
        # How centrally located is the candidate among unvisited nodes
        centrality_score = 0
        if len(unvisited_nodes) > 1:
            candidate_distances = [
                distance_matrix[candidate][node] 
                for node in unvisited_nodes if node != candidate
            ]
            centrality_score = np.std(candidate_distances)
        
        # ----- DESTINATION PROXIMITY -----
        dest_proximity = distance_matrix[candidate][destination_node]
        
        # ----- COMPOSITE SCORE -----
        score = (
            w_immediate * immediate_cost +
            w_future * future_cost +
            w_centrality * centrality_score -
            w_destination * dest_proximity  # Negative: prefer nodes closer to destination
        )
        
        # Tie-breaking: prefer nodes with more connections to remaining nodes
        if abs(score - best_score) < 1e-6:  # Tie case
            candidate_connectivity = sum(
                1 for node in remaining_after_candidate 
                if distance_matrix[candidate][node] < np.median(distance_matrix[candidate])
            )
            current_connectivity = sum(
                1 for node in remaining_after_candidate 
                if distance_matrix[best_node][node] < np.median(distance_matrix[best_node])
            ) if best_node else 0
            
            if candidate_connectivity > current_connectivity:
                best_score = score
                best_node = candidate
        elif score < best_score:
            best_score = score
            best_node = candidate
    
    return best_node
