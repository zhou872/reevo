import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Select the next node using two-step look-ahead with adaptive clustering.
    
    Features:
    1. Two-step look-ahead evaluates immediate and subsequent move costs
    2. Cluster-aware heuristic penalizes moves that isolate remaining nodes
    3. Adaptive weights based on remaining nodes count
    4. Deterministic greedy selection
    """
    if not unvisited_nodes:
        return destination_node
        
    # Adaptive weights based on remaining nodes
    n_remaining = len(unvisited_nodes)
    lookahead_weight = 0.3 + 0.4 * np.exp(-n_remaining / 10)  # More lookahead early
    cluster_weight = 0.4 * (1 - np.exp(-n_remaining / 5))     # More clustering late
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # Immediate cost
        immediate_cost = distance_matrix[current_node][candidate]
        
        # Two-step look-ahead: best next move from candidate
        remaining_after = unvisited_nodes - {candidate}
        if remaining_after:
            # Find minimal second step from candidate
            second_step_costs = [distance_matrix[candidate][next_node] 
                               for next_node in remaining_after]
            min_second_step = min(second_step_costs)
        else:
            # Last node before destination
            min_second_step = distance_matrix[candidate][destination_node]
        
        # Cluster penalty: measure how isolated remaining nodes would be
        if remaining_after:
            # Calculate centroid of remaining nodes (approximation)
            centroid_distances = []
            for node in remaining_after:
                # Average distance from node to other remaining nodes
                distances = [distance_matrix[node][other] 
                           for other in remaining_after if other != node]
                if distances:
                    centroid_distances.append(np.mean(distances))
            
            if centroid_distances:
                avg_cluster_compactness = np.mean(centroid_distances)
                # Penalty for moving away from cluster
                candidate_to_cluster = np.mean([distance_matrix[candidate][node] 
                                              for node in remaining_after])
                cluster_penalty = candidate_to_cluster - avg_cluster_compactness
            else:
                cluster_penalty = 0
        else:
            cluster_penalty = 0
        
        # Destination consideration (encourage moving toward destination when appropriate)
        dest_factor = distance_matrix[candidate][destination_node] / np.max(distance_matrix)
        
        # Final score with adaptive weights
        score = (immediate_cost + 
                lookahead_weight * min_second_step +
                cluster_weight * max(0, cluster_penalty) +
                0.1 * dest_factor)
        
        # Deterministic greedy selection with tie-breaking by node index
        if score < best_score or (abs(score - best_score) < 1e-9 and candidate < best_node):
            best_score = score
            best_node = candidate
    
    return best_node
