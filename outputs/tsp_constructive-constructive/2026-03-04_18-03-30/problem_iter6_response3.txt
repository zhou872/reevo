import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select the next node using adaptive two-step look-ahead with enhanced caching.
    
    Args:
        current_node: Current position node index
        destination_node: Final destination node index
        unvisited_nodes: Set of unvisited node indices
        distance_matrix: 2D array of distances between nodes
    
    Returns:
        Next node index to visit
    """
    if not unvisited_nodes:
        return destination_node
    
    n = len(unvisited_nodes)
    total_nodes = distance_matrix.shape[0]
    
    # Adaptive weights based on remaining nodes
    # More weight to lookahead when many nodes remain, more weight to immediate cost when few remain
    lookahead_weight = 0.7 * (n / total_nodes) + 0.3  # Ranges from 0.3 to 1.0
    regret_weight = 0.4 * (1 - n / total_nodes) + 0.1  # Ranges from 0.1 to 0.5
    terminal_weight = 0.2 * (1 - n / total_nodes)  # Ranges from 0 to 0.2
    
    # Cache nearest neighbor distances for each node to unvisited nodes
    # This avoids repeated min computations
    nearest_cache = {}
    for node in unvisited_nodes.union({current_node, destination_node}):
        # Find the minimum distance from this node to any unvisited node (excluding itself)
        min_dist = float('inf')
        for other in unvisited_nodes:
            if node != other:
                dist = distance_matrix[node][other]
                if dist < min_dist:
                    min_dist = dist
        # Also consider distance to destination for terminal nodes
        dest_dist = distance_matrix[node][destination_node]
        nearest_cache[node] = (min_dist, dest_dist)
    
    scores = {}
    
    for candidate in unvisited_nodes:
        # Immediate cost: current -> candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # Two-step look-ahead with adaptive weighting
        remaining_nodes = unvisited_nodes - {candidate}
        
        if remaining_nodes:
            # Calculate lookahead costs for each possible next node from candidate
            lookahead_costs = []
            lookahead_details = []
            
            for next_node in remaining_nodes:
                # Direct cost: candidate -> next_node
                step_cost = distance_matrix[candidate][next_node]
                
                # Estimate next step from next_node using cached nearest neighbor
                next_remaining = remaining_nodes - {next_node}
                if next_remaining:
                    # Use cached nearest distance for next_node to remaining nodes
                    min_third_step, _ = nearest_cache[next_node]
                    # Adjust if the cached nearest was the candidate (now removed)
                    if min_third_step == distance_matrix[next_node][candidate]:
                        # Recompute minimum excluding candidate
                        min_third_step = min(distance_matrix[next_node][n] 
                                           for n in next_remaining)
                else:
                    # If no more unvisited, use destination
                    _, dest_dist = nearest_cache[next_node]
                    min_third_step = dest_dist
                
                total_lookahead = step_cost + lookahead_weight * min_third_step
                lookahead_costs.append(total_lookahead)
                lookahead_details.append((next_node, total_lookahead))
            
            # Enhanced regret calculation with normalization
            if len(lookahead_costs) >= 2:
                sorted_costs = sorted(lookahead_costs)
                best_lookahead = sorted_costs[0]
                
                # Consider top 3 alternatives or half of available options
                k = min(3, max(2, len(sorted_costs) // 2))
                top_k_avg = np.mean(sorted_costs[:k])
                
                # Normalize regret by the range of lookahead costs
                cost_range = max(lookahead_costs) - min(lookahead_costs)
                if cost_range > 0:
                    regret = (best_lookahead - top_k_avg) / cost_range
                else:
                    regret = 0
            else:
                # Only one option available
                best_lookahead = lookahead_costs[0]
                regret = 0
            
            # Terminal cost with adaptive weight
            terminal_cost = terminal_weight * distance_matrix[candidate][destination_node]
            
            # Combined score with adaptive weights
            # Negative regret because lower regret (more negative) is better
            score = immediate_cost + best_lookahead - regret_weight * regret + terminal_cost
            
        else:
            # Only one node left to visit
            lookahead_cost = distance_matrix[candidate][destination_node]
            score = immediate_cost + lookahead_cost
        
        scores[candidate] = score
    
    # Deterministic greedy selection
    return min(scores, key=scores.get)
