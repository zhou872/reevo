import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select the next node using a two-step lookahead mechanism with adaptive weights.
    
    The algorithm considers:
    1. Immediate cost to candidate
    2. Expected future cost after visiting candidate (two-step lookahead)
    3. Distance to destination (importance grows as fewer nodes remain)
    4. Node centrality within remaining unvisited nodes
    """
    
    # Base case: only destination remains
    if unvisited_nodes == {destination_node}:
        return destination_node
    
    # Adaptive parameters based on search progress
    n_remaining = len(unvisited_nodes)
    progress = 1.0 - (n_remaining / len(distance_matrix))
    
    # Weights adapt based on search progress
    w_current = 0.5 - 0.3 * progress  # Less important as we progress
    w_lookahead = 0.3 + 0.4 * progress  # More important as we progress
    w_destination = 0.1 + 0.3 * progress  # More important near the end
    w_centrality = 0.1  # Constant small weight for centrality
    
    # Pre-compute normalized distances for efficiency
    max_dist = np.max(distance_matrix)
    min_dist = np.min(distance_matrix[distance_matrix > 0])
    
    def normalize(d):
        return (d - min_dist) / (max_dist - min_dist + 1e-10)
    
    # Candidate filtering: only consider top k closest nodes for efficiency
    candidate_nodes = list(unvisited_nodes - {destination_node})
    if len(candidate_nodes) > 10:
        # Get distances from current node to all candidates
        cand_dists = [distance_matrix[current_node][c] for c in candidate_nodes]
        # Select top 10 closest candidates
        top_indices = np.argsort(cand_dists)[:10]
        candidate_nodes = [candidate_nodes[i] for i in top_indices]
    
    scores = {}
    
    for node in candidate_nodes:
        # 1. Immediate cost (normalized)
        immediate_cost = normalize(distance_matrix[current_node][node])
        
        # 2. Two-step lookahead: minimum expected cost from node to any other unvisited then to destination
        remaining_after_node = unvisited_nodes - {node, destination_node}
        
        if remaining_after_node:
            # Calculate all possible two-step paths: node -> next -> destination
            two_step_costs = []
            for next_node in remaining_after_node:
                cost = (normalize(distance_matrix[node][next_node]) + 
                       normalize(distance_matrix[next_node][destination_node]))
                two_step_costs.append(cost)
            lookahead_cost = np.min(two_step_costs) if two_step_costs else 0
        else:
            # If this would be the last node before destination
            lookahead_cost = normalize(distance_matrix[node][destination_node])
        
        # 3. Destination proximity (normalized)
        dest_cost = normalize(distance_matrix[node][destination_node])
        
        # 4. Node centrality: average distance to other unvisited nodes
        other_unvisited = unvisited_nodes - {node}
        if other_unvisited:
            centrality = np.mean([normalize(distance_matrix[node][other]) 
                                 for other in other_unvisited])
        else:
            centrality = 0
        
        # Combined score (lower is better)
        score = (w_current * immediate_cost +
                w_lookahead * lookahead_cost +
                w_destination * dest_cost +
                w_centrality * centrality)
        
        # Small random tie-breaking
        scores[node] = score * (1.0 + np.random.random() * 0.001)
    
    # Return node with minimum score
    return min(scores, key=scores.get)
