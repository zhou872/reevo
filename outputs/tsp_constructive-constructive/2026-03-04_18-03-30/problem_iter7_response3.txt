import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Select the next node to visit using a deterministic multi-heuristic approach.
    Combines immediate cost, two-step lookahead, cluster penalty, and destination proximity.
    """
    # Handle trivial case
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Fixed weights as per suggestions
    w_immediate = 0.40    # Immediate cost weight
    w_lookahead = 0.25    # Two-step lookahead weight
    w_cluster = 0.20      # Cluster penalty weight
    w_destination = 0.15  # Destination proximity weight
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # 1. Immediate cost: current_node -> candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Two-step lookahead: candidate -> other -> destination
        # Find minimum two-step cost from candidate through another unvisited node to destination
        other_nodes = unvisited_nodes - {candidate}
        if other_nodes:
            # Consider all two-step paths: candidate -> other -> destination
            two_step_costs = [
                distance_matrix[candidate][other] + distance_matrix[other][destination_node]
                for other in other_nodes
            ]
            lookahead_cost = min(two_step_costs)
        else:
            # No other nodes: direct to destination
            lookahead_cost = distance_matrix[candidate][destination_node]
        
        # 3. Cluster penalty: average distance from candidate to other unvisited nodes
        if other_nodes:
            cluster_distances = [distance_matrix[candidate][other] for other in other_nodes]
            cluster_penalty = np.mean(cluster_distances)
        else:
            cluster_penalty = 0
        
        # 4. Destination proximity: candidate -> destination
        destination_proximity = distance_matrix[candidate][destination_node]
        
        # Combined score (lower is better)
        score = (
            w_immediate * immediate_cost +
            w_lookahead * lookahead_cost +
            w_cluster * cluster_penalty +
            w_destination * destination_proximity
        )
        
        # Update best node
        if score < best_score:
            best_score = score
            best_node = candidate
    
    return best_node
