import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Enhanced node selection with dynamic weight adaptation and improved cluster awareness.
    
    Features:
    1. Dynamic weights: Adjusts scoring components based on remaining nodes ratio
    2. Centroid-based clustering: Uses weighted centroids for better cluster detection
    3. Efficient two-step lookahead: Samples diverse nodes for future cost estimation
    4. Adaptive exploration: Balances exploration vs exploitation throughout the tour
    """
    
    # Base case: only one node remains
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    total_nodes = distance_matrix.shape[0]
    remaining_ratio = len(unvisited_nodes) / (total_nodes - 1)
    
    # Dynamic weight adjustment based on remaining nodes
    # Early stages: focus on exploration and cluster cohesion
    # Late stages: focus on immediate cost and destination proximity
    w_immediate = 0.25 + 0.25 * (1 - remaining_ratio)  # Increases as tour progresses
    w_lookahead = 0.35 - 0.15 * (1 - remaining_ratio)  # Decreases as tour progresses
    w_cluster = 0.25 + 0.10 * remaining_ratio         # Higher early, lower late
    w_destination = 0.15 + 0.10 * (1 - remaining_ratio)  # Increases as tour progresses
    
    best_node = None
    best_score = float('inf')
    
    # Precompute centroid of remaining nodes for cluster analysis
    remaining_list = list(unvisited_nodes)
    if len(remaining_list) >= 2:
        # Calculate weighted centroid based on distances
        centroid_distances = np.zeros(len(remaining_list))
        for i, node_i in enumerate(remaining_list):
            total_dist = 0
            for node_j in remaining_list:
                if node_i != node_j:
                    total_dist += distance_matrix[node_i][node_j]
            centroid_distances[i] = total_dist
        
        # Nodes with smaller total distances are more central
        min_dist_idx = np.argmin(centroid_distances)
        centroid_node = remaining_list[min_dist_idx]
    
    for candidate in unvisited_nodes:
        # 1. Immediate cost from current node to candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Two-step lookahead with adaptive sampling
        remaining_after = unvisited_nodes - {candidate}
        
        if not remaining_after:
            # Only destination remains
            future_cost = distance_matrix[candidate][destination_node]
        else:
            # Sample diverse nodes for lookahead estimation
            sample_size = min(5, len(remaining_after))
            other_nodes = list(remaining_after)
            
            # Get distances from candidate to other nodes
            dist_to_candidate = [(node, distance_matrix[candidate][node]) 
                                 for node in other_nodes]
            dist_to_candidate.sort(key=lambda x: x[1])
            
            # Sample closest, farthest, and middle nodes
            if len(dist_to_candidate) <= sample_size:
                sampled_nodes = [node for node, _ in dist_to_candidate]
            else:
                indices = [0, 
                          len(dist_to_candidate) // 4,
                          len(dist_to_candidate) // 2,
                          3 * len(dist_to_candidate) // 4,
                          -1][:sample_size]
                sampled_nodes = [dist_to_candidate[i][0] for i in indices]
            
            # Estimate future costs through sampled nodes
            lookahead_costs = []
            for next_candidate in sampled_nodes:
                step_cost = distance_matrix[candidate][next_candidate]
                future_remaining = remaining_after - {next_candidate}
                
                if not future_remaining:
                    remaining_est = distance_matrix[next_candidate][destination_node]
                else:
                    # Quick estimate: average distance to remaining nodes
                    remaining_dists = [distance_matrix[next_candidate][node] 
                                     for node in future_remaining]
                    avg_remaining = np.mean(remaining_dists)
                    remaining_est = avg_remaining * 0.7 + distance_matrix[next_candidate][destination_node] * 0.3
                
                lookahead_costs.append(step_cost + remaining_est)
            
            future_cost = np.mean(lookahead_costs)
        
        # 3. Improved cluster penalty using centroid-based approach
        if len(unvisited_nodes) > 1 and 'centroid_node' in locals():
            # Calculate distance from candidate to cluster centroid
            centroid_distance = distance_matrix[candidate][centroid_node]
            
            # Calculate average distance among remaining nodes
            other_nodes = [node for node in unvisited_nodes if node != candidate]
            if len(other_nodes) >= 2:
                # Sample inter-distances among remaining nodes
                sample_pairs = min(10, len(other_nodes) * (len(other_nodes) - 1) // 2)
                inter_distances = []
                for i in range(min(sample_pairs, len(other_nodes))):
                    for j in range(i + 1, min(i + 1 + sample_pairs, len(other_nodes))):
                        inter_distances.append(distance_matrix[other_nodes[i]][other_nodes[j]])
                
                if inter_distances:
                    avg_inter_distance = np.mean(inter_distances)
                    # Normalize cluster penalty
                    cluster_penalty = centroid_distance / (avg_inter_distance + 1e-8)
                else:
                    cluster_penalty = 1.0
            else:
                cluster_penalty = 1.0
        else:
            cluster_penalty = 1.0
        
        # 4. Destination proximity
        destination_cost = distance_matrix[candidate][destination_node]
        
        # 5. Composite score with dynamic weights
        score = (w_immediate * immediate_cost +
                w_lookahead * future_cost +
                w_cluster * cluster_penalty +
                w_destination * destination_cost)
        
        # Tie-breaking: prefer nodes with better connectivity to remaining nodes
        if abs(score - best_score) < 1e-9:
            if best_node is None:
                best_score = score
                best_node = candidate
            else:
                # Count connections below average distance
                candidate_conn = sum(1 for node in remaining_after 
                                   if distance_matrix[candidate][node] < np.mean(distance_matrix[candidate]))
                best_conn = sum(1 for node in remaining_after 
                              if distance_matrix[best_node][node] < np.mean(distance_matrix[best_node]))
                
                if candidate_conn > best_conn:
                    best_score = score
                    best_node = candidate
        elif score < best_score:
            best_score = score
            best_node = candidate
    
    return best_node
