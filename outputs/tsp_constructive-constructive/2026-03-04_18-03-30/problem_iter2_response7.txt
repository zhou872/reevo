import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select the next node using a simplified, efficient heuristic.
    
    Combines:
    1. Immediate proximity to current node
    2. Cluster connectivity (average distance to other unvisited nodes)
    3. Cluster diversity (standard deviation of distances)
    4. Destination proximity (with reduced weight)
    
    All weights are fixed for consistency and efficiency.
    """
    # Base case: only destination remains
    if unvisited_nodes == {destination_node}:
        return destination_node
    
    # Fixed weights for reliable performance
    w_current = 0.45      # Weight for immediate distance
    w_avg = -0.35         # Negative: prefer nodes central to remaining cluster
    w_std = 0.15          # Positive: prefer nodes with diverse connections
    w_dest = -0.05        # Small negative: mild preference toward destination
    
    scores = {}
    
    # Process each candidate (excluding destination until necessary)
    candidates = unvisited_nodes - {destination_node}
    
    for node in candidates:
        # 1. Immediate cost
        immediate_cost = distance_matrix[current_node, node]
        
        # 2. Cluster connectivity: average distance to other unvisited nodes
        other_nodes = unvisited_nodes - {node}
        if other_nodes:
            distances_to_others = [distance_matrix[node, other] for other in other_nodes]
            avg_distance = np.mean(distances_to_others)
            std_distance = np.std(distances_to_others) if len(distances_to_others) > 1 else 0
        else:
            avg_distance = 0
            std_distance = 0
        
        # 3. Destination proximity
        dest_distance = distance_matrix[node, destination_node]
        
        # Combined score (minimization)
        score = (w_current * immediate_cost +
                w_avg * avg_distance +
                w_std * std_distance +
                w_dest * dest_distance)
        
        scores[node] = score
    
    # Return best candidate, fall back to destination if no candidates
    return min(scores, key=scores.get) if scores else destination_node
