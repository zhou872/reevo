import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int,
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Improved node selection with two-step lookahead, adaptive weighting,
    regret heuristic, and centrality scoring.
    
    Args:
        current_node: Current node index
        destination_node: Final destination node index
        unvisited_nodes: Set of unvisited node indices
        distance_matrix: 2D array of distances between nodes
    
    Returns:
        Index of next node to visit
    """
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Calculate progress ratio
    total_nodes = distance_matrix.shape[0]
    remaining_ratio = len(unvisited_nodes) / (total_nodes - 1)  # Exclude starting node
    
    # Adaptive weights based on progress
    if remaining_ratio > 0.75:  # Early phase
        w_immediate = 0.2
        w_lookahead = 0.3
        w_centrality = 0.3
        w_destination = 0.1
        w_regret = 0.1
    elif remaining_ratio > 0.5:  # Early-mid phase
        w_immediate = 0.25
        w_lookahead = 0.25
        w_centrality = 0.25
        w_destination = 0.15
        w_regret = 0.1
    elif remaining_ratio > 0.25:  # Late-mid phase
        w_immediate = 0.3
        w_lookahead = 0.2
        w_centrality = 0.2
        w_destination = 0.2
        w_regret = 0.1
    else:  # Late phase
        w_immediate = 0.4
        w_lookahead = 0.1
        w_centrality = 0.1
        w_destination = 0.3
        w_regret = 0.1
    
    # Scale factor for normalization
    scale = np.mean(distance_matrix[distance_matrix > 0])  # Exclude zeros
    
    best_score = float('inf')
    best_node = None
    
    for candidate in unvisited_nodes:
        # Skip destination if other nodes remain
        if candidate == destination_node and len(unvisited_nodes) > 1:
            continue
        
        # 1. Immediate cost
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Two-step lookahead with improved estimation
        lookahead_costs = []
        remaining_after_candidate = unvisited_nodes - {candidate}
        
        for next_candidate in remaining_after_candidate:
            # Cost from candidate to next_candidate
            cost1 = distance_matrix[candidate][next_candidate]
            
            # Remaining nodes after two steps
            remaining_nodes = remaining_after_candidate - {next_candidate}
            
            if remaining_nodes:
                # Get distances from next_candidate to remaining nodes
                distances = [distance_matrix[next_candidate][node] 
                           for node in remaining_nodes]
                distances.sort()
                
                # Use combination of min and average of top k
                min_dist = distances[0]
                top_k = min(3, len(distances))
                avg_top_k = np.mean(distances[:top_k])
                
                # Future estimate with improved heuristic
                future_estimate = cost1 + 0.6 * min_dist + 0.4 * avg_top_k
            else:
                # Last step before destination
                future_estimate = cost1 + distance_matrix[next_candidate][destination_node]
            
            lookahead_costs.append(future_estimate)
        
        # Calculate lookahead score (mean with std adjustment)
        if lookahead_costs:
            lookahead_score = np.mean(lookahead_costs) + 0.5 * np.std(lookahead_costs)
        else:
            lookahead_score = distance_matrix[candidate][destination_node]
        
        # 3. Centrality measure (average distance to unvisited nodes)
        if len(unvisited_nodes) > 1:
            candidate_distances = [distance_matrix[candidate][node] 
                                 for node in unvisited_nodes if node != candidate]
            centrality = np.mean(candidate_distances)
        else:
            centrality = 0
        
        # 4. Destination proximity
        dest_proximity = distance_matrix[candidate][destination_node]
        
        # 5. Regret heuristic (difference between 2nd and 1st nearest)
        if len(unvisited_nodes) > 1:
            # Get distances to all unvisited nodes except candidate
            regret_distances = [distance_matrix[candidate][node] 
                              for node in unvisited_nodes if node != candidate]
            if len(regret_distances) >= 2:
                regret_distances.sort()
                regret = regret_distances[1] - regret_distances[0]
            else:
                regret = 0
        else:
            regret = 0
        
        # Normalized composite score
        score = (w_immediate * immediate_cost / scale +
                 w_lookahead * lookahead_score / scale +
                 w_centrality * centrality / scale +
                 w_destination * dest_proximity / scale +
                 w_regret * (-regret) / scale)  # Negative because we want high regret
        
        # Tie-breaking with connectivity
        if abs(score - best_score) < 1e-10:  # Tie case
            if best_node is not None:
                # Count how many remaining nodes are relatively close to candidate
                candidate_close = sum(1 for node in remaining_after_candidate 
                                    if distance_matrix[candidate][node] < np.median(distance_matrix[candidate]))
                best_close = sum(1 for node in remaining_after_candidate 
                               if distance_matrix[best_node][node] < np.median(distance_matrix[best_node]))
                if candidate_close > best_close:
                    best_score = score
                    best_node = candidate
        elif score < best_score:
            best_score = score
            best_node = candidate
    
    # Fallback
    if best_node is None:
        best_node = min(unvisited_nodes)
    
    return best_node
