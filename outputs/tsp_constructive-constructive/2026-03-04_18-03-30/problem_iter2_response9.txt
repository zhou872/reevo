import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select next node using adaptive multi-criteria scoring with online learning.
    
    Features:
    1. Predictive lookahead with variable depth based on state complexity
    2. Gradient-based weight adaptation using immediate reward feedback
    3. Online learning of scoring parameters via exponential smoothing
    4. State-aware exploration-exploitation balancing
    """
    
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Initialize learning state (simulated persistent memory via function attributes)
    if not hasattr(select_next_node_v2, "learning_state"):
        select_next_node_v2.learning_state = {
            "weights": np.array([0.35, 0.30, 0.20, 0.15]),  # Immediate, Future, Connect, Dest
            "performance_history": [],
            "alpha": 0.1,  # Learning rate
            "exploration_rate": 0.2,
            "temperature": 1.0
        }
    
    state = select_next_node_v2.learning_state
    n = distance_matrix.shape[0]
    progress = 1.0 - len(unvisited_nodes) / (n - 1)
    
    # Dynamic temperature for softmax exploration (cooling schedule)
    state["temperature"] = max(0.1, 1.0 - progress * 0.9)
    
    # Adaptive lookahead depth based on remaining complexity
    lookahead_depth = min(3, max(1, int(3 * (1 - progress))))
    
    candidates = []
    scores = []
    components = []  # Store individual component values for learning
    
    # Filter candidates (avoid destination unless it's the last)
    for candidate in unvisited_nodes:
        if candidate == destination_node and len(unvisited_nodes) > 1:
            continue
            
        remaining = unvisited_nodes - {candidate}
        
        # 1. Immediate cost component
        immediate = distance_matrix[current_node][candidate]
        
        # 2. Predictive lookahead component (variable depth)
        if remaining:
            # Generate k-step lookahead predictions
            lookahead_costs = []
            
            # Heuristic: consider closest 'lookahead_depth' nodes from candidate
            close_nodes = sorted(remaining, 
                               key=lambda x: distance_matrix[candidate][x])[:lookahead_depth]
            
            for next_node in close_nodes:
                # Two-step cost
                two_step = immediate + distance_matrix[candidate][next_node]
                
                # Estimate remaining path from next_node
                further_remaining = remaining - {next_node}
                if not further_remaining:
                    future_est = distance_matrix[next_node][destination_node]
                else:
                    # Dual estimation: nearest neighbor + MST approximation
                    nearest_dist = min(distance_matrix[next_node][node] 
                                     for node in further_remaining)
                    mst_est = np.mean([distance_matrix[next_node][node] 
                                      for node in further_remaining]) * 0.7
                    future_est = nearest_dist * 0.4 + mst_est * 0.6
                
                lookahead_costs.append(two_step + future_est)
            
            future_cost = np.mean(lookahead_costs) if lookahead_costs else 0
        else:
            future_cost = distance_matrix[candidate][destination_node]
        
        # 3. Network connectivity component
        if remaining:
            # Measure candidate's centrality in remaining network
            candidate_dists = [distance_matrix[candidate][node] for node in remaining]
            mean_dist = np.mean(candidate_dists)
            std_dist = np.std(candidate_dists)
            
            # Good connector: low mean distance with moderate variance
            if std_dist > 0:
                connectivity_score = mean_dist / (std_dist + 1e-10)
            else:
                connectivity_score = mean_dist
        else:
            connectivity_score = 0
        
        # 4. Destination progress component
        if candidate != destination_node:
            curr_to_dest = distance_matrix[current_node][destination_node]
            cand_to_dest = distance_matrix[candidate][destination_node]
            progress_gain = (curr_to_dest - cand_to_dest) / (curr_to_dest + 1e-10)
        else:
            progress_gain = 1.0
        
        # Store component values (normalized)
        max_dist = np.max(distance_matrix)
        components.append([
            immediate / max_dist,
            future_cost / max_dist if max_dist > 0 else 0,
            connectivity_score / max_dist if max_dist > 0 else 0,
            max(0, min(1, progress_gain))  # Clamp to [0,1]
        ])
        
        # Composite score with current weights
        score = np.dot(state["weights"], components[-1])
        candidates.append(candidate)
        scores.append(score)
    
    if not candidates:  # All candidates were destination node
        return destination_node
    
    # Exploration-exploitation with softmax
    if np.random.random() < state["exploration_rate"]:
        # Explore: choose random candidate with probability inverse to score
        probs = np.exp(-np.array(scores) / state["temperature"])
        probs = probs / np.sum(probs)
        chosen_idx = np.random.choice(len(candidates), p=probs)
    else:
        # Exploit: choose best candidate
        chosen_idx = np.argmin(scores)
    
    chosen_node = candidates[chosen_idx]
    
    # Online learning: update weights based on immediate reward
    if len(state["performance_history"]) > 0:
        # Calculate reward: inverse of immediate cost (normalized)
        immediate_cost = distance_matrix[current_node][chosen_node]
        reward = 1.0 / (1.0 + immediate_cost / np.mean(distance_matrix))
        
        # Gradient update: adjust weights toward better components
        chosen_components = components[chosen_idx]
        gradient = chosen_components - np.mean(chosen_components)
        
        # Update weights with momentum
        state["weights"] += state["alpha"] * reward * gradient
        state["weights"] = np.clip(state["weights"], 0.05, 0.6)  # Keep reasonable bounds
        state["weights"] = state["weights"] / np.sum(state["weights"])  # Normalize
        
        # Track performance for adaptive exploration
        state["performance_history"].append(reward)
        if len(state["performance_history"]) > 10:
            state["performance_history"].pop(0)
        
        # Adjust exploration rate based on recent performance variance
        if len(state["performance_history"]) >= 5:
            perf_std = np.std(state["performance_history"])
            state["exploration_rate"] = max(0.05, min(0.3, perf_std * 2))
    
    return chosen_node
