import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Enhanced node selection heuristic with adaptive weights and simplified regret.
    
    Args:
        current_node: Index of current node
        destination_node: Index of final destination node
        unvisited_nodes: Set of nodes not yet visited
        distance_matrix: 2D array of pairwise distances
    
    Returns:
        Index of next node to visit
    """
    if not unvisited_nodes:
        return destination_node
    
    # Adaptive weights based on problem size and progress
    n_total = distance_matrix.shape[0]
    progress = 1.0 - len(unvisited_nodes) / (n_total - 1)
    
    # Base weights that adapt to problem size and progress
    if n_total < 20:  # Small problems
        w1, w2, w3, w4 = 0.4, 0.3, 0.2, 0.1
    elif n_total < 100:  # Medium problems
        w1 = 0.35 + 0.1 * progress  # Immediate cost weight increases
        w2 = 0.3 - 0.1 * progress   # Global view weight decreases
        w3 = 0.25
        w4 = 0.1
    else:  # Large problems
        w1 = 0.4 + 0.2 * progress   # Strong focus on immediate cost late
        w2 = 0.3 - 0.15 * progress  # Less global view as tour progresses
        w3 = 0.25 - 0.05 * progress
        w4 = 0.1 - 0.05 * progress
    
    # Normalize weights to sum to 1
    total = w1 + w2 + w3 + w4
    w1, w2, w3, w4 = w1/total, w2/total, w3/total, w4/total
    
    scores = {}
    
    # Pre-compute min distance from current to any unvisited (for regret)
    if len(unvisited_nodes) > 1:
        min_current_dist = min(distance_matrix[current_node][n] for n in unvisited_nodes)
    else:
        min_current_dist = 0
    
    for node in unvisited_nodes:
        # Immediate cost component
        immediate_cost = distance_matrix[current_node][node]
        
        # Simplified regret: penalty for choosing much worse than best immediate option
        regret_penalty = max(0, immediate_cost - min_current_dist) / (immediate_cost + 1e-10)
        
        # Global distribution statistics
        distances_to_others = [distance_matrix[node][other] 
                              for other in unvisited_nodes if other != node]
        
        if distances_to_others:
            avg_distance = np.mean(distances_to_others)
            std_distance = np.std(distances_to_others)
            # Coefficient of variation (std/mean) - measures relative dispersion
            coeff_var = std_distance / (avg_distance + 1e-10)
        else:
            avg_distance = distance_matrix[node][destination_node]
            std_distance = 0
            coeff_var = 0
        
        # Destination proximity
        dest_distance = distance_matrix[node][destination_node]
        
        # Combined score - lower is better
        score = (w1 * immediate_cost + 
                 w2 * regret_penalty * immediate_cost +  # Regret-weighted immediate cost
                 w3 * avg_distance - 
                 w4 * coeff_var -  # Prefer nodes with varied distances to others
                 0.05 * dest_distance)  # Small constant bias toward destination
        
        scores[node] = score
    
    # Return the node with minimum score
    return min(scores, key=scores.get)
