import numpy as np
import random
from typing import Set
from collections import defaultdict

def select_next_node_v2(current_node: int, 
                       destination_node: int, 
                       unvisited_nodes: Set[int], 
                       distance_matrix: np.ndarray,
                       lookahead_depth: int = 3,
                       exploration_rate: float = 0.05,
                       learning_rate: float = 0.1) -> int:
    """
    Advanced node selection with configurable lookahead depth, adaptive learning,
    and hybrid stochastic-deterministic strategy.
    
    Args:
        current_node: Current node index
        destination_node: Final destination node index
        unvisited_nodes: Set of unvisited node indices
        distance_matrix: 2D array of distances between nodes
        lookahead_depth: Depth of lookahead search (default: 3)
        exploration_rate: Probability of random exploration (default: 0.05)
        learning_rate: Rate for updating component weights (default: 0.1)
    
    Returns:
        Index of next node to visit
    """
    if not unvisited_nodes:
        raise ValueError("No unvisited nodes available")
    
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Stochastic exploration phase
    if random.random() < exploration_rate and len(unvisited_nodes) > 1:
        candidate_nodes = list(unvisited_nodes)
        if destination_node in candidate_nodes and len(candidate_nodes) > 1:
            candidate_nodes.remove(destination_node)
        return random.choice(candidate_nodes)
    
    total_nodes = distance_matrix.shape[0]
    progress_ratio = 1 - len(unvisited_nodes) / total_nodes
    
    # Initialize adaptive weights based on progress
    base_weights = initialize_weights(progress_ratio)
    
    # Calculate heuristic components with multi-step lookahead
    component_scores = calculate_components(
        current_node, destination_node, unvisited_nodes, 
        distance_matrix, lookahead_depth, progress_ratio
    )
    
    # Learn optimal weight adjustments from problem characteristics
    weight_adjustments = learn_weight_adjustments(
        component_scores, distance_matrix, progress_ratio
    )
    
    # Apply adjustments with learning rate
    final_weights = {}
    for component in base_weights:
        adjusted = base_weights[component] + learning_rate * weight_adjustments.get(component, 0)
        final_weights[component] = max(0.05, min(0.8, adjusted))
    
    # Normalize weights
    weight_sum = sum(final_weights.values())
    normalized_weights = {k: v/weight_sum for k, v in final_weights.items()}
    
    # Calculate composite scores
    scores = {}
    for candidate in unvisited_nodes:
        if candidate == destination_node and len(unvisited_nodes) > 1:
            continue
            
        score = 0
        for component in normalized_weights:
            score += normalized_weights[component] * component_scores[component][candidate]
        
        # Add tiny deterministic tie-breaker based on node index
        scores[candidate] = score + candidate * 1e-10
    
    # Deterministic selection with fallback
    if scores:
        return min(scores, key=scores.get)
    else:
        return min(unvisited_nodes)

def initialize_weights(progress_ratio: float) -> dict:
    """Initialize component weights based on solution progress."""
    if progress_ratio < 0.3:
        return {'immediate': 0.2, 'lookahead': 0.4, 'regret': 0.3, 'dest_connect': 0.1}
    elif progress_ratio < 0.7:
        return {'immediate': 0.4, 'lookahead': 0.3, 'regret': 0.2, 'dest_connect': 0.1}
    else:
        return {'immediate': 0.5, 'lookahead': 0.2, 'regret': 0.1, 'dest_connect': 0.2}

def calculate_components(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray,
                        depth: int, progress_ratio: float) -> dict:
    """Calculate all heuristic component scores for each candidate."""
    candidates = list(unvisited_nodes)
    components = {
        'immediate': {},
        'lookahead': {},
        'regret': {},
        'dest_connect': {}
    }
    
    # Precompute statistics for normalization
    non_zero_dists = distance_matrix[distance_matrix > 0]
    dist_mean = np.mean(non_zero_dists)
    dist_std = np.std(non_zero_dists) if np.std(non_zero_dists) > 0 else 1
    
    for candidate in candidates:
        # Immediate cost component
        components['immediate'][candidate] = distance_matrix[current_node][candidate] / dist_mean
        
        # Multi-step lookahead component
        lookahead_cost = calculate_lookahead_cost(
            current_node, candidate, destination_node, 
            unvisited_nodes, distance_matrix, depth
        )
        components['lookahead'][candidate] = lookahead_cost / dist_mean
        
        # Regret component (opportunity cost)
        regret = calculate_regret(
            candidate, unvisited_nodes, distance_matrix, depth
        )
        components['regret'][candidate] = regret / dist_mean
        
        # Destination connectivity component
        components['dest_connect'][candidate] = distance_matrix[candidate][destination_node] / dist_mean
    
    return components

def calculate_lookahead_cost(start_node: int, first_node: int, dest_node: int,
                            unvisited: Set[int], dist_mat: np.ndarray, 
                            depth: int) -> float:
    """Calculate minimum path cost over depth-limited lookahead."""
    if depth <= 1 or len(unvisited) <= 2:
        return dist_mat[start_node][first_node]
    
    remaining = unvisited - {first_node}
    
    def dfs(current: int, visited: Set, current_depth: int, current_cost: float) -> float:
        if current_depth >= depth or len(visited) == len(unvisited):
            # Add cost to destination from last node
            if dest_node not in visited:
                return current_cost + dist_mat[current][dest_node]
            return current_cost
        
        best_cost = float('inf')
        next_candidates = list(unvisited - visited)
        
        # Prune to top candidates based on progress
        prune_factor = max(3, len(next_candidates) // 4)
        if len(next_candidates) > prune_factor:
            next_candidates = sorted(
                next_candidates, 
                key=lambda x: dist_mat[current][x]
            )[:prune_factor]
        
        for next_node in next_candidates:
            new_visited = visited | {next_node}
            new_cost = current_cost + dist_mat[current][next_node]
            total_cost = dfs(next_node, new_visited, current_depth + 1, new_cost)
            best_cost = min(best_cost, total_cost)
        
        return best_cost
    
    return dfs(first_node, {start_node, first_node}, 1, dist_mat[start_node][first_node])

def calculate_regret(candidate: int, unvisited: Set[int], 
                    dist_mat: np.ndarray, depth: int) -> float:
    """Calculate regret of not choosing alternative paths."""
    if len(unvisited) <= 2:
        return 0
    
    # Calculate alternative costs for other candidates
    other_candidates = list(unvisited - {candidate})
    
    if len(other_candidates) < 2:
        return 0
    
    # Sample alternative paths (limited for efficiency)
    sample_size = min(5, len(other_candidates))
    sampled_alternatives = random.sample(other_candidates, sample_size)
    
    alternative_costs = []
    for alt in sampled_alternatives:
        # Simplified lookahead for alternatives
        remaining = unvisited - {alt}
        if remaining:
            min_next = min(dist_mat[alt][node] for node in remaining)
            cost = dist_mat[alt][candidate] + min_next  # Cost to reach candidate via alternative
        else:
            cost = dist_mat[alt][candidate]
        alternative_costs.append(cost)
    
    if alternative_costs:
        avg_alternative = np.mean(alternative_costs)
        return avg_alternative
    return 0

def learn_weight_adjustments(component_scores: dict, dist_mat: np.ndarray,
                            progress_ratio: float) -> dict:
    """Learn optimal weight adjustments from component correlations."""
    adjustments = defaultdict(float)
    
    # Analyze component correlations with problem characteristics
    n = dist_mat.shape[0]
    dist_variability = np.std(dist_mat[dist_mat > 0]) / np.mean(dist_mat[dist_mat > 0])
    
    # Adjust based on problem structure
    if dist_variability > 1.5:  # High variability
        adjustments['regret'] += 0.1
        adjustments['lookahead'] += 0.05
    else:  # Low variability
        adjustments['immediate'] += 0.1
        adjustments['dest_connect'] += 0.05
    
    # Adjust based on progress phase
    if progress_ratio < 0.3:
        adjustments['lookahead'] += 0.15
        adjustments['regret'] += 0.1
    elif progress_ratio > 0.7:
        adjustments['immediate'] += 0.15
        adjustments['dest_connect'] += 0.1
    
    return adjustments
