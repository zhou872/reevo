import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select the next node using a two-step look-ahead with regret-based heuristic.
    
    For each candidate node, we consider:
    1. Direct cost from current to candidate
    2. Minimum cost from candidate to any remaining node (look-ahead)
    3. Regret: difference between best and average of top alternative look-ahead costs
    4. Distance to final destination
    """
    
    # Weights for different components (sum to 1)
    w_current = 0.35      # Weight for current→candidate distance
    w_lookahead = 0.30    # Weight for look-ahead cost
    w_regret = 0.25       # Weight for regret term (higher regret = better)
    w_destination = 0.10  # Weight for distance to destination
    
    # For regret calculation: consider top k alternatives
    k_alternatives = min(3, len(unvisited_nodes) - 1) if len(unvisited_nodes) > 1 else 1
    
    scores = {}
    
    for candidate in unvisited_nodes:
        # 1. Direct cost from current to candidate
        direct_cost = distance_matrix[current_node][candidate]
        
        # 2. Look-ahead: minimum cost from candidate to remaining nodes (excluding candidate)
        remaining = unvisited_nodes - {candidate}
        
        if remaining:
            # Get distances from candidate to all remaining nodes
            future_distances = [distance_matrix[candidate][node] for node in remaining]
            
            # Two-step look-ahead: minimum distance to next possible node
            lookahead_cost = min(future_distances)
            
            # 3. Calculate regret: difference between best and average of top k alternatives
            if len(future_distances) >= k_alternatives:
                # Sort distances ascending (smallest first)
                sorted_distances = np.sort(future_distances)
                # Take top k smallest distances
                top_k_distances = sorted_distances[:k_alternatives]
                # Regret = average of top k - best (we want to minimize missing good opportunities)
                regret = np.mean(top_k_distances) - top_k_distances[0]
            else:
                # If not enough alternatives, use available ones
                regret = np.mean(future_distances) - min(future_distances)
        else:
            # If no remaining nodes, next step is destination
            lookahead_cost = distance_matrix[candidate][destination_node]
            regret = 0
        
        # 4. Distance from candidate to destination
        dest_cost = distance_matrix[candidate][destination_node]
        
        # Combine scores (lower is better)
        # Note: regret is subtracted because higher regret means we should choose this node
        # to avoid missing a good opportunity later
        score = (w_current * direct_cost + 
                 w_lookahead * lookahead_cost - 
                 w_regret * regret + 
                 w_destination * dest_cost)
        
        scores[candidate] = score
    
    # Return node with minimum score
    return min(scores, key=scores.get)
