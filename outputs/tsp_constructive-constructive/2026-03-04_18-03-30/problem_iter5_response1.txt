import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Enhanced node selection for TSP constructive heuristics using two-step look-ahead
    and regret-based opportunity cost calculation.
    """
    # If only one node remains, return it
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Parameters for balancing immediate vs future costs
    immediate_weight = 0.6
    future_weight = 0.4
    
    scores = {}
    
    for candidate in unvisited_nodes:
        # Immediate cost: current node to candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # Calculate look-ahead costs for this candidate
        look_ahead_costs = []
        remaining_after_candidate = unvisited_nodes - {candidate}
        
        if remaining_after_candidate:
            # For each potential next node after candidate, calculate two-step cost
            for next_candidate in remaining_after_candidate:
                # Cost from candidate to next candidate
                cost_to_next = distance_matrix[candidate][next_candidate]
                
                # Estimate additional cost from next candidate (minimum to remaining nodes)
                remaining_after_next = remaining_after_candidate - {next_candidate}
                if remaining_after_next:
                    min_future_cost = min(
                        distance_matrix[next_candidate][future_node] 
                        for future_node in remaining_after_next
                    )
                else:
                    # If no nodes remain after next, cost to return to destination
                    min_future_cost = distance_matrix[next_candidate][destination_node]
                
                look_ahead_costs.append(cost_to_next + min_future_cost)
            
            # Use minimum look-ahead cost as the future estimate
            future_estimate = min(look_ahead_costs) if look_ahead_costs else 0
            
            # Calculate regret: opportunity cost of not choosing this candidate
            # Regret = difference between candidate's look-ahead cost and average of alternatives
            avg_look_ahead = np.mean(look_ahead_costs)
            regret = avg_look_ahead - future_estimate
            
        else:
            # If this is the last node before destination
            future_estimate = distance_matrix[candidate][destination_node]
            regret = 0
        
        # Combine immediate cost, future estimate, and regret
        # Lower immediate cost and future estimate are better, higher regret is better
        combined_score = (
            immediate_weight * immediate_cost +
            future_weight * future_estimate -
            0.3 * regret  # Subtract because we're minimizing and want to favor high regret
        )
        
        scores[candidate] = combined_score
    
    # Select node with minimum combined score
    return min(scores, key=scores.get)
