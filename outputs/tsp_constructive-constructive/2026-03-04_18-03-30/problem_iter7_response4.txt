import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select the next node using a deterministic multi-heuristic approach.
    
    Args:
        current_node: Current node index
        destination_node: Final destination node index
        unvisited_nodes: Set of unvisited node indices
        distance_matrix: 2D array of distances between nodes
    
    Returns:
        Next node to visit
    """
    # Fixed heuristic weights (deterministic, no randomization)
    w1, w2, w3, w4 = 0.40, 0.25, 0.20, 0.15  # immediate, two-step, cluster, destination
    
    scores = {}
    
    for candidate in unvisited_nodes:
        # 1. Immediate cost: distance from current to candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Two-step lookahead: min distance from candidate to next possible node
        # Only consider if there are other unvisited nodes besides candidate
        other_unvisited = unvisited_nodes - {candidate}
        if other_unvisited:
            two_step_cost = min(distance_matrix[candidate][next_candidate] 
                               for next_candidate in other_unvisited)
        else:
            two_step_cost = distance_matrix[candidate][destination_node]  # direct to destination
        
        # 3. Cluster penalty: average distance from candidate to other unvisited
        # Penalizes nodes that are isolated from the remaining unvisited nodes
        if len(other_unvisited) > 0:
            cluster_distances = [distance_matrix[candidate][other] 
                                 for other in other_unvisited]
            cluster_penalty = np.mean(cluster_distances)
        else:
            cluster_penalty = 0  # No cluster penalty if no other unvisited nodes
        
        # 4. Destination proximity: distance from candidate to final destination
        destination_cost = distance_matrix[candidate][destination_node]
        
        # Combine all heuristics with deterministic weights
        score = (w1 * immediate_cost + 
                 w2 * two_step_cost + 
                 w3 * cluster_penalty + 
                 w4 * destination_cost)
        
        scores[candidate] = score
    
    # Greedy deterministic selection: choose node with minimum combined score
    return min(scores, key=scores.get)
