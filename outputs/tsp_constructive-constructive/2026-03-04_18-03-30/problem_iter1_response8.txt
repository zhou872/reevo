import numpy as np
from typing import Set, Tuple

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select the next node using a two-step look-ahead mechanism that considers
    both immediate and forward path quality.
    
    Args:
        current_node: Current node index
        destination_node: Final destination node index (starting point)
        unvisited_nodes: Set of unvisited node indices
        distance_matrix: 2D array of distances between nodes
    
    Returns:
        Index of the next node to visit
    """
    if not unvisited_nodes:
        return destination_node
    
    # Weights for the composite score
    w_immediate = 0.35      # Weight for immediate cost
    w_lookahead = 0.40      # Weight for look-ahead cost  
    w_balance = 0.15        # Weight for edge balance
    w_destination = 0.10    # Weight for destination proximity
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # 1. Immediate cost: current → candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Look-ahead cost: candidate → best next node → remaining nodes
        lookahead_cost = _compute_lookahead_cost(
            candidate, destination_node, unvisited_nodes, distance_matrix
        )
        
        # 3. Edge balance: compare candidate edge with average tour edges
        balance_score = _compute_edge_balance(
            current_node, candidate, destination_node, 
            unvisited_nodes, distance_matrix
        )
        
        # 4. Destination proximity: how close candidate is to final destination
        dest_proximity = distance_matrix[candidate][destination_node]
        
        # Composite score (lower is better)
        score = (w_immediate * immediate_cost +
                 w_lookahead * lookahead_cost +
                 w_balance * balance_score +
                 w_destination * dest_proximity)
        
        if score < best_score:
            best_score = score
            best_node = candidate
    
    return best_node


def _compute_lookahead_cost(candidate: int, destination_node: int,
                           unvisited_nodes: Set[int], 
                           distance_matrix: np.ndarray) -> float:
    """
    Compute two-step look-ahead cost: from candidate to best next node,
    then estimate remaining path quality.
    """
    remaining_nodes = unvisited_nodes - {candidate}
    
    if not remaining_nodes:
        # Only destination remains
        return distance_matrix[candidate][destination_node]
    
    # Find best next node from candidate's perspective
    best_next_cost = float('inf')
    best_next_rem_cost = float('inf')
    
    for next_candidate in remaining_nodes:
        # Cost: candidate → next_candidate
        first_step = distance_matrix[candidate][next_candidate]
        
        # Estimate remaining cost after next_candidate
        remaining_after_next = remaining_nodes - {next_candidate}
        rem_est = _estimate_remaining_cost(
            next_candidate, destination_node, remaining_after_next, distance_matrix
        )
        
        total = first_step + rem_est
        
        if total < best_next_cost:
            best_next_cost = total
            best_next_rem_cost = rem_est
    
    # Blend immediate lookahead with forward estimate
    return 0.6 * best_next_cost + 0.4 * best_next_rem_cost


def _estimate_remaining_cost(from_node: int, destination_node: int,
                            remaining_nodes: Set[int], 
                            distance_matrix: np.ndarray) -> float:
    """
    Estimate cost to complete tour from from_node through remaining_nodes
    to destination_node using a quick approximation.
    """
    if not remaining_nodes:
        return distance_matrix[from_node][destination_node]
    
    # Quick MST-based estimation
    nodes_list = list(remaining_nodes) + [from_node, destination_node]
    n = len(nodes_list)
    
    if n <= 1:
        return 0
    
    # Create submatrix for relevant nodes
    sub_dist = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            sub_dist[i, j] = distance_matrix[nodes_list[i]][nodes_list[j]]
    
    # Approximate with average of two heuristics
    # 1. Nearest neighbor chain from current node
    nn_cost = _nearest_neighbor_estimate(from_node, destination_node, 
                                        remaining_nodes, distance_matrix)
    
    # 2. Minimum spanning tree approximation (double MST cost)
    mst_est = _mst_estimate(nodes_list, sub_dist)
    
    return 0.7 * nn_cost + 0.3 * mst_est


def _nearest_neighbor_estimate(start: int, end: int,
                              nodes: Set[int], 
                              distance_matrix: np.ndarray) -> float:
    """Estimate path cost using greedy nearest neighbor."""
    cost = 0
    current = start
    unvisited = set(nodes)
    
    while unvisited:
        # Find nearest unvisited node
        nearest = min(unvisited, key=lambda x: distance_matrix[current][x])
        cost += distance_matrix[current][nearest]
        current = nearest
        unvisited.remove(nearest)
    
    # Return to end
    cost += distance_matrix[current][end]
    return cost


def _mst_estimate(nodes: list, dist_matrix: np.ndarray) -> float:
    """Estimate tour cost using minimum spanning tree (double for TSP bound)."""
    n = len(nodes)
    if n <= 1:
        return 0
    
    # Prim's algorithm for MST
    in_mst = [False] * n
    min_edge = [float('inf')] * n
    min_edge[0] = 0
    mst_cost = 0
    
    for _ in range(n):
        # Find minimum edge vertex not in MST
        u = min((v for v in range(n) if not in_mst[v]), 
                key=lambda v: min_edge[v], default=-1)
        if u == -1:
            break
            
        in_mst[u] = True
        mst_cost += min_edge[u]
        
        # Update min_edge for adjacent vertices
        for v in range(n):
            if not in_mst[v] and dist_matrix[u, v] < min_edge[v]:
                min_edge[v] = dist_matrix[u, v]
    
    # Double MST cost gives upper bound for TSP
    return 2 * mst_cost


def _compute_edge_balance(current: int, candidate: int, destination: int,
                         unvisited_nodes: Set[int], 
                         distance_matrix: np.ndarray) -> float:
    """
    Compute edge balance score to avoid very long/short edges in the tour.
    """
    # Current candidate edge length
    edge_len = distance_matrix[current][candidate]
    
    # Average of possible edges in the remaining tour
    remaining = list(unvisited_nodes - {candidate})
    if remaining:
        edge_lengths = []
        
        # Candidate to remaining nodes
        for node in remaining:
            edge_lengths.append(distance_matrix[candidate][node])
        
        # Some remaining node to destination
        if remaining:
            to_dest = min(distance_matrix[node][destination] for node in remaining)
            edge_lengths.append(to_dest)
        
        avg_len = np.mean(edge_lengths)
        std_len = np.std(edge_lengths)
        
        # Penalize edges that are much longer/shorter than average
        if avg_len > 0:
            deviation = abs(edge_len - avg_len) / avg_len
            balance = deviation + 0.5 * (std_len / avg_len)
            return balance
    
    return 0
