import numpy as np
from typing import Set, Tuple

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select the next node using a look-ahead mechanism that considers:
    1. Immediate cost to candidate
    2. Future cost from candidate to remaining nodes
    3. Connectivity of candidate to unvisited network
    4. Progress toward destination
    """
    
    # Adaptive weights based on search progress
    progress = 1.0 - len(unvisited_nodes) / (distance_matrix.shape[0] - 1)
    
    # Weights evolve during search
    if progress < 0.3:
        # Early phase: prioritize exploration
        w_immediate, w_future, w_connect, w_dest = 0.25, 0.35, 0.25, 0.15
    elif progress < 0.7:
        # Middle phase: balance exploration and destination approach
        w_immediate, w_future, w_connect, w_dest = 0.35, 0.30, 0.20, 0.15
    else:
        # Late phase: prioritize reaching destination
        w_immediate, w_future, w_connect, w_dest = 0.40, 0.20, 0.15, 0.25
    
    best_score = float('inf')
    best_node = None
    
    for candidate in unvisited_nodes:
        if candidate == destination_node and len(unvisited_nodes) > 1:
            # Don't return to destination until all nodes visited
            continue
            
        # 1. Immediate cost
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Look-ahead: minimum future cost from candidate
        remaining_nodes = unvisited_nodes - {candidate}
        if remaining_nodes:
            # Best immediate move from candidate
            min_future = min(distance_matrix[candidate][other] 
                           for other in remaining_nodes)
            # Second best for robustness
            future_costs = sorted(distance_matrix[candidate][other] 
                                for other in remaining_nodes)
            avg_future = np.mean(future_costs[:min(3, len(future_costs))])
            future_cost = 0.6 * min_future + 0.4 * avg_future
        else:
            # Last node before destination
            future_cost = distance_matrix[candidate][destination_node]
        
        # 3. Connectivity assessment
        if len(remaining_nodes) >= 2:
            # Calculate how well candidate connects unvisited nodes
            candidate_distances = [distance_matrix[candidate][n] for n in remaining_nodes]
            connectivity = np.std(candidate_distances) / (np.mean(candidate_distances) + 1e-10)
            # Low std dev means candidate is equally distant from all unvisited (good hub)
            connectivity_score = 1.0 / (1.0 + connectivity)
        else:
            connectivity_score = 0.5
        
        # 4. Destination progress
        if candidate != destination_node:
            # Distance from candidate to destination
            dest_dist = distance_matrix[candidate][destination_node]
            # Current distance to destination
            current_dest_dist = distance_matrix[current_node][destination_node]
            # Progress: how much closer we get to destination
            dest_progress = (current_dest_dist - dest_dist) / (current_dest_dist + 1e-10)
        else:
            dest_progress = 1.0
        
        # Normalize components (min-max scaling within current candidates)
        # We'll normalize during comparison using relative scores
        
        # Composite score (lower is better)
        score = (w_immediate * immediate_cost +
                 w_future * future_cost -
                 w_connect * connectivity_score * np.mean(distance_matrix) -
                 w_dest * dest_progress * np.max(distance_matrix))
        
        # Tie-breaker: prefer node with smaller index for deterministic behavior
        if score < best_score - 1e-10 or (abs(score - best_score) < 1e-10 and candidate < best_node):
            best_score = score
            best_node = candidate
    
    # Fallback if all nodes skipped (shouldn't happen)
    if best_node is None:
        best_node = min(unvisited_nodes)
    
    return best_node
