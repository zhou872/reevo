import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, 
                        destination_node: int, 
                        unvisited_nodes: Set[int], 
                        distance_matrix: np.ndarray) -> int:
    """
    Select the next node using adaptive three-step lookahead with dynamic weight adjustment.
    
    Key improvements over v1:
    1. Three-step lookahead for better long-term planning
    2. Adaptive weights based on remaining nodes and spatial distribution
    3. Enhanced cluster coherence using minimum spanning tree heuristic
    4. Progressive focus on destination as tour completes
    
    Parameters:
    -----------
    current_node : int
        Current node index
    destination_node : int
        Final destination node index
    unvisited_nodes : Set[int]
        Set of unvisited node indices
    distance_matrix : np.ndarray
        Distance matrix between nodes (n x n)
    
    Returns:
    --------
    int : Next node to visit (deterministic choice)
    """
    if not unvisited_nodes:
        return destination_node
    
    n_total = distance_matrix.shape[0]
    n_remaining = len(unvisited_nodes)
    progress_ratio = 1.0 - n_remaining / n_total
    
    # Adaptive weights based on progress and spatial distribution
    WEIGHT_IMMEDIATE = 0.35 - 0.15 * progress_ratio
    WEIGHT_THREE_STEP = 0.30 + 0.10 * progress_ratio
    WEIGHT_CLUSTER = 0.25 - 0.15 * progress_ratio
    WEIGHT_DESTINATION = 0.10 + 0.20 * progress_ratio
    
    # Normalize weights to sum to 1
    weight_sum = WEIGHT_IMMEDIATE + WEIGHT_THREE_STEP + WEIGHT_CLUSTER + WEIGHT_DESTINATION
    WEIGHT_IMMEDIATE /= weight_sum
    WEIGHT_THREE_STEP /= weight_sum
    WEIGHT_CLUSTER /= weight_sum
    WEIGHT_DESTINATION /= weight_sum
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # 1. Immediate cost (current to candidate)
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Three-step lookahead: minimum cost path candidate -> next -> next2
        remaining_after_candidate = unvisited_nodes - {candidate}
        
        if len(remaining_after_candidate) >= 2:
            # Calculate best three-step path from candidate
            three_step_cost = float('inf')
            
            # Efficient approximation: consider top 3 nearest nodes for next step
            nearest_to_candidate = sorted(remaining_after_candidate, 
                                         key=lambda n: distance_matrix[candidate][n])[:3]
            
            for next_node in nearest_to_candidate:
                remaining_after_next = remaining_after_candidate - {next_node}
                
                if remaining_after_next:
                    # Find best two-step from next_node (next -> next2)
                    best_next2 = min(remaining_after_next,
                                    key=lambda n: distance_matrix[next_node][n])
                    remaining_after_next2 = remaining_after_next - {best_next2}
                    
                    if remaining_after_next2:
                        # Third step: minimum from next2 to any remaining node
                        min_third_step = min(distance_matrix[best_next2][n] 
                                           for n in remaining_after_next2)
                        candidate_three_step = (distance_matrix[candidate][next_node] +
                                               distance_matrix[next_node][best_next2] +
                                               min_third_step)
                    else:
                        # Only destination remains after next2
                        candidate_three_step = (distance_matrix[candidate][next_node] +
                                               distance_matrix[next_node][best_next2] +
                                               distance_matrix[best_next2][destination_node])
                    
                    three_step_cost = min(three_step_cost, candidate_three_step)
                else:
                    # Only destination remains after next
                    three_step_cost = min(three_step_cost,
                                         distance_matrix[candidate][next_node] +
                                         distance_matrix[next_node][destination_node])
            
            if three_step_cost == float('inf'):
                three_step_cost = distance_matrix[candidate][destination_node]
        elif remaining_after_candidate:
            # Only one node remains after candidate
            next_node = next(iter(remaining_after_candidate))
            three_step_cost = (distance_matrix[candidate][next_node] +
                              distance_matrix[next_node][destination_node])
        else:
            # Candidate is the last unvisited node
            three_step_cost = distance_matrix[candidate][destination_node]
        
        # 3. Enhanced cluster coherence (minimum spanning tree heuristic)
        if remaining_after_candidate:
            # Calculate MST-like penalty: sum of distances from candidate to
            # its k nearest neighbors in remaining set (k = min(3, |remaining|))
            k = min(3, len(remaining_after_candidate))
            nearest_distances = sorted([distance_matrix[candidate][n] 
                                      for n in remaining_after_candidate])[:k]
            cluster_penalty = sum(nearest_distances) / k
        else:
            cluster_penalty = 0
        
        # 4. Destination guidance with progress awareness
        dest_distance = distance_matrix[candidate][destination_node]
        # Scale destination influence by progress
        dest_influence = dest_distance * (1.0 + 2.0 * progress_ratio)
        
        # Combined score (lower is better)
        score = (WEIGHT_IMMEDIATE * immediate_cost +
                 WEIGHT_THREE_STEP * three_step_cost +
                 WEIGHT_CLUSTER * cluster_penalty +
                 WEIGHT_DESTINATION * dest_influence)
        
        # Tie-breaking: prefer nodes closer to destination when scores are close
        if abs(score - best_score) < 1e-6:
            if dest_distance < distance_matrix[best_node][destination_node]:
                best_score = score
                best_node = candidate
        elif score < best_score:
            best_score = score
            best_node = candidate
    
    return best_node
