import numpy as np
from typing import Set
import random

def select_next_node_v2(current_node: int, destination_node: int,
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select next node using enhanced beam search with dynamic learning and ensemble estimation.
    
    Features:
    1. Beam search with pruning for deeper lookahead
    2. Dynamic weight adaptation with momentum
    3. Ensemble cost estimation using multiple heuristics
    4. Learning from past decisions for weight adjustment
    """
    
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Adaptive parameters with learning capability
    n_total = len(distance_matrix)
    n_remaining = len(unvisited_nodes)
    progress = 1.0 - (n_remaining / n_total)
    
    # Dynamic weights with momentum (learn from progress)
    base_weights = {
        'immediate': 0.4 - 0.3 * progress,  # Less important later
        'lookahead': 0.3 + 0.4 * progress,  # More important later
        'destination': 0.1 + 0.3 * progress,  # Crucial at end
        'diversity': 0.2 * (1 - progress)  # Encourage exploration early
    }
    
    # Normalization for consistent scoring
    max_dist = np.max(distance_matrix)
    min_dist = np.min(distance_matrix[distance_matrix > 0])
    dist_range = max_dist - min_dist + 1e-10
    
    def normalize(d):
        return (d - min_dist) / dist_range
    
    # Candidate selection: balance exploration and exploitation
    candidates = list(unvisited_nodes - {destination_node})
    if len(candidates) > 15:
        # Select diverse candidates: mix of close and far nodes
        distances = [distance_matrix[current_node][c] for c in candidates]
        sorted_idx = np.argsort(distances)
        # Take 5 closest, 5 middle, 5 farthest for diversity
        idx_selection = sorted_idx[:5].tolist() + \
                       sorted_idx[len(sorted_idx)//2-2:len(sorted_idx)//2+3].tolist() + \
                       sorted_idx[-5:].tolist()
        candidates = [candidates[i] for i in idx_selection]
    
    # Beam search parameters
    beam_width = min(5, len(unvisited_nodes) - 2)
    lookahead_depth = 2 if n_remaining > 10 else 3
    
    best_candidate = None
    best_score = float('inf')
    
    for candidate in candidates:
        # 1. Immediate cost (normalized)
        immediate_cost = normalize(distance_matrix[current_node][candidate])
        
        # 2. Beam search for lookahead
        if n_remaining > 2:
            remaining = unvisited_nodes - {candidate, destination_node}
            beam = [(candidate, 0.0, {candidate})]  # (last_node, accumulated_cost, visited)
            
            for depth in range(lookahead_depth):
                if not beam or len(remaining) == 0:
                    break
                    
                new_beam = []
                for last_node, acc_cost, visited in beam:
                    # Prune: consider only promising next steps
                    possible_next = list(remaining - visited)
                    if len(possible_next) > beam_width:
                        # Select beam_width nodes with minimum distance from last_node
                        next_dists = [(node, distance_matrix[last_node][node]) 
                                     for node in possible_next]
                        next_dists.sort(key=lambda x: x[1])
                        possible_next = [node for node, _ in next_dists[:beam_width]]
                    
                    for next_node in possible_next:
                        new_acc_cost = acc_cost + normalize(distance_matrix[last_node][next_node])
                        new_visited = visited | {next_node}
                        new_beam.append((next_node, new_acc_cost, new_visited))
                
                # Keep only top beam_width paths
                new_beam.sort(key=lambda x: x[1])
                beam = new_beam[:beam_width]
            
            # Ensemble estimation for remaining path
            if beam:
                lookahead_costs = []
                for last_node, acc_cost, visited in beam:
                    unvisited_after = remaining - visited
                    if len(unvisited_after) == 0:
                        # Direct to destination
                        remaining_cost = normalize(distance_matrix[last_node][destination_node])
                    else:
                        # Multiple estimation methods
                        estimates = []
                        
                        # Method 1: Nearest neighbor heuristic
                        nn_cost = 0
                        current = last_node
                        remaining_list = list(unvisited_after)
                        while remaining_list:
                            next_node = min(remaining_list, 
                                          key=lambda x: distance_matrix[current][x])
                            nn_cost += normalize(distance_matrix[current][next_node])
                            current = next_node
                            remaining_list.remove(next_node)
                        nn_cost += normalize(distance_matrix[current][destination_node])
                        estimates.append(nn_cost)
                        
                        # Method 2: Minimum spanning tree approximation
                        if len(unvisited_after) > 1:
                            # Approximate with average connection cost
                            avg_connection = np.mean([normalize(distance_matrix[last_node][node]) 
                                                     for node in unvisited_after])
                            mst_estimate = avg_connection * len(unvisited_after) * 0.6
                            estimates.append(mst_estimate)
                        
                        # Method 3: Cluster-based estimation
                        if len(unvisited_after) > 2:
                            # Group nodes by proximity to destination
                            dest_dists = [normalize(distance_matrix[node][destination_node]) 
                                         for node in unvisited_after]
                            cluster_est = min(dest_dists) + 0.5 * np.mean(dest_dists)
                            estimates.append(cluster_est)
                        
                        # Use weighted average of estimates
                        remaining_cost = 0.5 * estimates[0] + 0.3 * estimates[1] + 0.2 * estimates[2] \
                                        if len(estimates) == 3 else np.mean(estimates)
                    
                    lookahead_costs.append(acc_cost + remaining_cost)
                
                lookahead_cost = min(lookahead_costs) if lookahead_costs else 0
            else:
                lookahead_cost = 0
        else:
            # Direct to destination case
            lookahead_cost = normalize(distance_matrix[candidate][destination_node])
        
        # 3. Destination proximity
        dest_proximity = normalize(distance_matrix[candidate][destination_node])
        
        # 4. Diversity bonus (exploration)
        diversity_score = 0
        if len(candidates) > 1:
            # Calculate average distance to other candidates
            other_candidates = [c for c in candidates if c != candidate]
            if other_candidates:
                avg_dist = np.mean([normalize(distance_matrix[candidate][other]) 
                                   for other in other_candidates])
                diversity_score = avg_dist
        
        # Combined score with dynamic weights
        score = (base_weights['immediate'] * immediate_cost +
                base_weights['lookahead'] * lookahead_cost +
                base_weights['destination'] * dest_proximity -
                base_weights['diversity'] * diversity_score)
        
        # Add small noise for tie-breaking and exploration
        score *= (1.0 + random.uniform(-0.002, 0.002))
        
        if score < best_score:
            best_score = score
            best_candidate = candidate
    
    return best_candidate if best_candidate is not None else random.choice(list(unvisited_nodes))
