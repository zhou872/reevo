import numpy as np
from typing import Set, Dict

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select next node using a balanced heuristic with progress-aware destination guidance.
    
    Combines:
    1. Immediate distance to current node (greedy component)
    2. Cluster centrality: average distance to other unvisited nodes
    3. Cluster dispersion: standard deviation of distances to unvisited nodes
    4. Destination proximity with progress-based weighting
    
    Key improvements over v1:
    - Fixed weights for stability (no complex adaptive schemes)
    - Standard deviation captures cluster shape better than min-distance
    - Progress-based destination weighting without overcomplication
    - Handles edge cases robustly
    """
    
    # Base case: only destination remains
    if len(unvisited_nodes) == 1 and destination_node in unvisited_nodes:
        return destination_node
    
    # Fixed heuristic weights (empirically determined for general TSP)
    w_current = 0.40    # Weight for immediate distance
    w_avg = -0.35       # Negative: prefer central nodes in clusters
    w_std = 0.15        # Positive: prefer nodes with varied distances (exploration)
    w_dest = -0.10      # Negative: prefer nodes closer to destination
    
    # Progress factor for destination weighting
    total_nodes = distance_matrix.shape[0]
    remaining = len(unvisited_nodes)
    progress = 1.0 - (remaining / total_nodes)
    
    # Adjust destination weight based on progress (lightweight adaptation)
    # More important later in the tour
    dest_weight = w_dest * (1.0 + 0.5 * progress)
    
    # Pre-compute destination distances for efficiency
    dest_distances = {node: distance_matrix[node, destination_node] 
                     for node in unvisited_nodes}
    
    scores: Dict[int, float] = {}
    
    for node in unvisited_nodes:
        # Skip if this would be an immediate jump to destination (unless necessary)
        if node == destination_node and len(unvisited_nodes) > 1:
            continue
        
        # 1. Immediate cost from current node
        immediate_cost = distance_matrix[current_node, node]
        
        # 2. Cluster statistics
        other_unvisited = unvisited_nodes - {node}
        if other_unvisited:
            distances_to_others = [distance_matrix[node, other] 
                                  for other in other_unvisited]
            avg_distance = np.mean(distances_to_others)
            std_distance = np.std(distances_to_others)
        else:
            # Last node before destination
            avg_distance = 0
            std_distance = 0
        
        # 3. Destination proximity with progress scaling
        dest_proximity = dest_distances[node]
        
        # Combined score (lower is better)
        score = (w_current * immediate_cost +
                w_avg * avg_distance +
                w_std * std_distance +
                dest_weight * dest_proximity)
        
        scores[node] = score
    
    # If no candidates scored (edge case), return the closest node to current
    if not scores:
        # Fallback: nearest unvisited node
        closest_dist = float('inf')
        closest_node = destination_node
        for node in unvisited_nodes:
            if node != current_node:
                dist = distance_matrix[current_node, node]
                if dist < closest_dist:
                    closest_dist = dist
                    closest_node = node
        return closest_node
    
    # Return node with minimum score
    return min(scores, key=scores.get)
