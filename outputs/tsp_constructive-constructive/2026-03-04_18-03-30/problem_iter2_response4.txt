import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Enhanced node selection with efficient two-step lookahead and adaptive multi-factor scoring.
    Dynamically balances exploration vs. exploitation based on remaining nodes.
    """
    
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Calculate adaptive parameters
    total_nodes = distance_matrix.shape[0]
    remaining_ratio = len(unvisited_nodes) / (total_nodes - 1)
    
    # Dynamic weights: shift from exploration to exploitation
    # Early: more weight on centrality and exploration
    # Late: more weight on immediate cost and destination
    w_immediate = 0.3 + 0.3 * (1 - remaining_ratio)  # Increases as we progress
    w_future = 0.4 - 0.2 * (1 - remaining_ratio)     # Decreases as we progress
    w_centrality = 0.2 + 0.1 * remaining_ratio      # Higher early, lower late
    w_destination = 0.1 + 0.2 * (1 - remaining_ratio) # Increases as we progress
    
    # Adaptive sampling for lookahead
    lookahead_samples = max(3, min(8, int(5 + 5 * remaining_ratio)))
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # 1. Immediate cost
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Efficient two-step lookahead with sampling
        remaining_after = unvisited_nodes - {candidate}
        
        if not remaining_after:
            # Only destination remains
            future_cost = distance_matrix[candidate][destination_node]
        else:
            # Sample representative nodes for lookahead
            if len(remaining_after) > lookahead_samples:
                # Sample diverse nodes: closest, farthest, and random
                other_nodes = list(remaining_after)
                dist_to_candidate = [(node, distance_matrix[candidate][node]) 
                                     for node in other_nodes]
                dist_to_candidate.sort(key=lambda x: x[1])
                
                # Take closest, farthest, and random samples
                sample_size = min(lookahead_samples, len(other_nodes))
                step = max(1, len(other_nodes) // sample_size)
                sample_indices = list(range(0, len(other_nodes), step))[:sample_size]
                sampled_nodes = [dist_to_candidate[i][0] for i in sample_indices]
            else:
                sampled_nodes = list(remaining_after)
            
            # Evaluate two-step lookahead for each sampled node
            lookahead_costs = []
            for next_candidate in sampled_nodes:
                # Cost: candidate -> next_candidate
                step_cost = distance_matrix[candidate][next_candidate]
                
                # Estimate remaining cost from next_candidate
                future_remaining = remaining_after - {next_candidate}
                
                if not future_remaining:
                    # Only destination left after next_candidate
                    remaining_est = distance_matrix[next_candidate][destination_node]
                else:
                    # Use fast heuristic: minimum spanning tree approximation
                    # Estimate with average distance to remaining nodes plus distance to destination
                    avg_to_remaining = np.mean([distance_matrix[next_candidate][node] 
                                              for node in future_remaining])
                    min_to_dest = min(distance_matrix[node][destination_node] 
                                    for node in future_remaining)
                    
                    # Weighted estimate that considers number of remaining nodes
                    remaining_est = (0.5 * avg_to_remaining * len(future_remaining) + 
                                   0.5 * min_to_dest)
                
                lookahead_costs.append(step_cost + remaining_est)
            
            # Use robust estimate: weighted average favoring better paths
            sorted_costs = sorted(lookahead_costs)
            weights = np.exp(-0.5 * np.arange(len(sorted_costs)))  # Exponential decay weights
            weights /= weights.sum()
            future_cost = np.dot(sorted_costs, weights)
        
        # 3. Centrality measure among unvisited nodes
        if len(unvisited_nodes) > 1:
            # Calculate mean distance to other unvisited nodes
            other_nodes = [node for node in unvisited_nodes if node != candidate]
            mean_dist = np.mean([distance_matrix[candidate][node] for node in other_nodes])
            
            # Normalize by overall distances for better scale invariance
            all_dists = distance_matrix[candidate]
            scale_factor = np.mean(all_dists[all_dists > 0])  # Avoid zero distances
            
            if scale_factor > 0:
                centrality = mean_dist / scale_factor
            else:
                centrality = 1.0
        else:
            centrality = 1.0
        
        # 4. Destination proximity
        dest_dist = distance_matrix[candidate][destination_node]
        
        # 5. Composite score
        score = (w_immediate * immediate_cost +
                w_future * future_cost +
                w_centrality * centrality +
                w_destination * dest_dist)
        
        # Tie-breaking with connectivity measure
        if abs(score - best_score) < 1e-9:
            # Prefer node with more short connections to remaining nodes
            candidate_connections = sum(1 for node in remaining_after 
                                      if distance_matrix[candidate][node] < np.median(distance_matrix[candidate]))
            best_connections = sum(1 for node in remaining_after 
                                 if distance_matrix[best_node][node] < np.median(distance_matrix[best_node]))
            
            if candidate_connections > best_connections:
                best_score = score
                best_node = candidate
        elif score < best_score:
            best_score = score
            best_node = candidate
    
    return best_node
