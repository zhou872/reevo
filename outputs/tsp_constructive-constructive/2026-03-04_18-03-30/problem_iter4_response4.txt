import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select next node using adaptive multi-criteria scoring with enhanced regret and progress-aware weighting.
    
    Args:
        current_node: Index of current node.
        destination_node: Index of final destination node (typically starting node in TSP).
        unvisited_nodes: Set of nodes not yet visited.
        distance_matrix: 2D numpy array of pairwise distances between nodes.
    
    Returns:
        Index of the next node to visit.
    """
    # Handle trivial cases
    if not unvisited_nodes:
        return destination_node
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Problem size and progress calculation
    n_total = distance_matrix.shape[0]
    progress = 1.0 - len(unvisited_nodes) / (n_total - 1)  # Progress from 0 to 1
    
    # Adaptive weight scheduling based on problem size and progress
    if n_total < 20:  # Small problems: balanced weights
        w_imm = 0.4
        w_reg = 0.3
        w_glob = 0.2
        w_dest = 0.1
    else:  # Medium/large problems: progress-dependent weights
        # Weights shift focus to immediate cost and destination as tour progresses
        w_imm = 0.35 + 0.15 * progress  # Increased focus on immediate cost
        w_reg = 0.25 - 0.10 * progress  # Reduced regret emphasis over time
        w_glob = 0.25 - 0.05 * progress  # Reduced global view
        w_dest = 0.15 + 0.00 * progress  # Constant destination bias (can adjust if needed)
    
    # Normalize weights to sum to 1
    total_weight = w_imm + w_reg + w_glob + w_dest
    w_imm /= total_weight
    w_reg /= total_weight
    w_glob /= total_weight
    w_dest /= total_weight
    
    # Pre-compute minimum distance from current node to any unvisited node for regret calculation
    min_current_dist = min(distance_matrix[current_node][node] for node in unvisited_nodes)
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # Skip destination if it's not the last node (to avoid premature return)
        if candidate == destination_node and len(unvisited_nodes) > 1:
            continue
        
        # 1. Immediate cost component
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Regret penalty component: penalize choosing much worse than best immediate option
        if immediate_cost > 0:
            regret_penalty = max(0.0, immediate_cost - min_current_dist) / immediate_cost
        else:
            regret_penalty = 0.0
        
        # 3. Global connectivity component: statistical analysis of distances to other unvisited nodes
        other_nodes = [node for node in unvisited_nodes if node != candidate]
        if other_nodes:
            distances_to_others = [distance_matrix[candidate][node] for node in other_nodes]
            avg_distance = np.mean(distances_to_others)
            std_distance = np.std(distances_to_others)
            coeff_var = std_distance / (avg_distance + 1e-10)  # Coefficient of variation
        else:
            # Only candidate left, so connectivity is not applicable
            avg_distance = 0.0
            coeff_var = 0.0
        
        # 4. Destination proximity component
        dest_distance = distance_matrix[candidate][destination_node]
        
        # Combined score calculation (lower is better)
        # All components are used in their natural units; weights handle scaling
        score = (w_imm * immediate_cost +
                 w_reg * regret_penalty * immediate_cost +  # Regret-weighted cost
                 w_glob * avg_distance -
                 w_glob * 0.3 * coeff_var +  # Prefer higher dispersion (embedded in global weight)
                 w_dest * dest_distance)  # Bias toward nodes closer to destination
        
        # Update best node
        if score < best_score:
            best_score = score
            best_node = candidate
    
    # Fallback: if no candidate was considered (e.g., only destination left), return destination
    if best_node is None:
        return destination_node
    
    return best_node
