import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, 
                       destination_node: int, 
                       unvisited_nodes: Set[int], 
                       distance_matrix: np.ndarray) -> int:
    """
    Enhanced node selection with adaptive multi-criteria scoring and progress-based tuning.
    
    Args:
        current_node: Current node index
        destination_node: Final destination node index
        unvisited_nodes: Set of unvisited node indices
        distance_matrix: 2D array of distances between nodes
    
    Returns:
        Index of next node to visit
    """
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Avoid returning to destination prematurely
    if destination_node in unvisited_nodes and len(unvisited_nodes) > 1:
        unvisited_nodes = unvisited_nodes - {destination_node}
    
    total_nodes = distance_matrix.shape[0]
    progress = 1.0 - (len(unvisited_nodes) / total_nodes)
    
    # Progress-based adaptive weighting
    # Early phase: Explore and build structure
    # Late phase: Focus on completion and destination
    if progress < 0.3:
        w_direct = 0.4      # Direct distance to candidate
        w_cluster = 0.3     # Neighborhood connectivity
        w_future = 0.2      # Future path estimation
        w_dest = 0.1        # Destination alignment
    elif progress < 0.7:
        w_direct = 0.35
        w_cluster = 0.25
        w_future = 0.25
        w_dest = 0.15
    else:
        w_direct = 0.45
        w_cluster = 0.15
        w_future = 0.15
        w_dest = 0.25
    
    # Normalize weights
    total = w_direct + w_cluster + w_future + w_dest
    w_direct /= total
    w_cluster /= total
    w_future /= total
    w_dest /= total
    
    # Scale factor for normalization
    scale = np.mean(distance_matrix[distance_matrix > 0])
    
    # Sample candidates for efficiency in large sets
    candidates = list(unvisited_nodes)
    if len(candidates) > 20:
        # Prioritize closer nodes but maintain diversity
        distances = [distance_matrix[current_node][node] for node in candidates]
        indices = np.argsort(distances)[:15]
        candidates = [candidates[i] for i in indices]
    
    best_score = float('inf')
    best_node = None
    
    for candidate in candidates:
        # 1. Direct distance score (normalized)
        direct_dist = distance_matrix[current_node][candidate]
        direct_score = direct_dist / scale
        
        # 2. Neighborhood connectivity score
        # Average distance to other unvisited nodes (lower is better)
        other_nodes = unvisited_nodes - {candidate}
        if other_nodes:
            cluster_dists = [distance_matrix[candidate][node] for node in other_nodes]
            # Use 25th percentile to focus on close neighbors
            cluster_score = np.percentile(cluster_dists, 25) / scale
        else:
            cluster_score = distance_matrix[candidate][destination_node] / scale
        
        # 3. Future path estimation
        # Estimate the rest of the path from this candidate
        remaining_nodes = other_nodes
        if remaining_nodes:
            # Minimum spanning tree heuristic approximation
            # Use the average of: min distance to remaining + avg distance among remaining
            min_to_remaining = min([distance_matrix[candidate][node] 
                                   for node in remaining_nodes], default=0)
            
            # Average distance among remaining nodes (approximates connectivity)
            if len(remaining_nodes) > 1:
                avg_among_remaining = np.mean([
                    distance_matrix[a][b] 
                    for i, a in enumerate(remaining_nodes) 
                    for b in list(remaining_nodes)[i+1:]
                ])
            else:
                avg_among_remaining = 0
            
            # Distance to destination from remaining nodes (weighted by progress)
            avg_to_dest = np.mean([distance_matrix[node][destination_node] 
                                  for node in remaining_nodes])
            
            future_estimate = (min_to_remaining + 0.3 * avg_among_remaining + 
                             0.2 * avg_to_dest)
            future_score = future_estimate / scale
        else:
            future_score = distance_matrix[candidate][destination_node] / scale
        
        # 4. Destination alignment score
        # How well does this move align with reaching destination eventually
        current_to_dest = distance_matrix[current_node][destination_node]
        cand_to_dest = distance_matrix[candidate][destination_node]
        
        # Progress toward destination (positive if moving closer)
        dest_progress = current_to_dest - cand_to_dest
        
        # Weight by progress phase
        dest_weight = 0.5 + 0.5 * progress
        dest_score = (dest_progress * dest_weight) / scale
        
        # 5. Composite score
        # Lower scores are better
        score = (w_direct * direct_score +
                w_cluster * cluster_score +
                w_future * future_score -
                w_dest * dest_score)  # Negative because we want to move toward destination
        
        # Tie-breaking: prefer nodes with better local connectivity
        if abs(score - best_score) < 1e-10 and best_node is not None:
            # Count immediate neighbors within median distance
            cand_close = sum(1 for node in other_nodes 
                           if distance_matrix[candidate][node] < np.median(distance_matrix[candidate]))
            best_close = sum(1 for node in other_nodes 
                           if distance_matrix[best_node][node] < np.median(distance_matrix[best_node]))
            
            if cand_close > best_close:
                best_score = score
                best_node = candidate
        elif score < best_score:
            best_score = score
            best_node = candidate
    
    # Fallback to nearest neighbor
    if best_node is None:
        best_node = min(unvisited_nodes, 
                       key=lambda x: distance_matrix[current_node][x])
    
    return best_node
