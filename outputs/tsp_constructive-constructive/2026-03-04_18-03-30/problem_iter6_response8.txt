import numpy as np
import random
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Enhanced node selection with progress-adaptive stochastic diversification.
    
    Features:
    1. Progress-aware stochastic selection balancing exploration/exploitation
    2. Efficient cluster-awareness using distance statistics
    3. Adaptive temperature controlling randomness
    4. Multi-criteria scoring with dynamic weights
    
    Strategy: Uses softmax selection with temperature that decreases with progress.
    Early phases explore diverse options; later phases converge to greedy.
    """
    
    # Base cases
    if unvisited_nodes == {destination_node}:
        return destination_node
    
    # Calculate progress
    total_nodes = distance_matrix.shape[0]
    remaining = len(unvisited_nodes)
    progress = 1.0 - (remaining / total_nodes) if total_nodes > 0 else 0.5
    
    # Pre-filter candidates (avoid destination until necessary)
    candidates = list(unvisited_nodes - {destination_node})
    if not candidates:
        return destination_node
    
    # Adaptive temperature: high early (more exploration), low late (more exploitation)
    temperature = max(0.1, 1.0 - progress * 0.9)  # Ranges from 1.0 to 0.1
    
    # Dynamic weights based on progress
    if progress < 0.3:  # Exploration phase
        w_immediate = 0.35
        w_centrality = -0.35  # Negative: prefer central nodes
        w_connectivity = -0.15
        w_destination = 0.15
    elif progress < 0.7:  # Balanced phase
        w_immediate = 0.40
        w_centrality = -0.30
        w_connectivity = -0.10
        w_destination = 0.20
    else:  # Exploitation phase
        w_immediate = 0.50
        w_centrality = -0.20
        w_connectivity = -0.05
        w_destination = 0.25
    
    # Calculate scores for all candidates
    scores = []
    other_nodes = unvisited_nodes - {destination_node}
    
    for node in candidates:
        # 1. Immediate cost
        immediate_cost = distance_matrix[current_node, node]
        
        # 2. Cluster centrality (average distance to other unvisited nodes)
        if len(other_nodes) > 1:
            # Exclude current node from other_nodes for this calculation
            others = other_nodes - {node}
            if others:
                avg_distance = np.mean([distance_matrix[node, other] for other in others])
            else:
                avg_distance = 0
        else:
            avg_distance = 0
        
        # 3. Minimum distance to other unvisited nodes (connectivity)
        if len(other_nodes) > 1:
            others = other_nodes - {node}
            if others:
                min_distance = np.min([distance_matrix[node, other] for other in others])
            else:
                min_distance = 0
        else:
            min_distance = 0
        
        # 4. Destination guidance with progress scaling
        dest_penalty = distance_matrix[node, destination_node] * (0.3 + 0.7 * progress)
        
        # Combined score
        score = (w_immediate * immediate_cost +
                w_centrality * avg_distance +
                w_connectivity * min_distance +
                w_destination * dest_penalty)
        
        scores.append(score)
    
    # Convert scores to probabilities using softmax with temperature
    scores_array = np.array(scores)
    
    # Handle negative scores by shifting
    min_score = np.min(scores_array)
    if min_score < 0:
        scores_array = scores_array - min_score + 1e-8
    
    # Apply temperature scaling
    scaled_scores = -scores_array / temperature  # Negative because lower scores are better
    
    # Softmax probabilities (higher probability for lower original scores)
    exp_scores = np.exp(scaled_scores - np.max(scaled_scores))  # Numerical stability
    probabilities = exp_scores / np.sum(exp_scores)
    
    # Stochastic selection based on probabilities
    # With small probability, force random exploration even in late phases
    if random.random() < 0.05 * (1.0 - progress):  # Decreasing random chance
        return random.choice(candidates)
    
    # Otherwise, use probability-weighted selection
    selected_idx = np.random.choice(len(candidates), p=probabilities)
    return candidates[selected_idx]
