import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select the next node using an adaptive regret heuristic with two-step look-ahead.
    
    This implementation focuses on:
    1. Adaptive regret threshold based on remaining nodes
    2. Simple two-step look-ahead with weighted future costs
    3. Stage-aware balancing between exploration and exploitation
    4. Computational efficiency with O(n²) worst-case complexity
    
    Args:
        current_node: Current position node index
        destination_node: Final destination node index
        unvisited_nodes: Set of unvisited node indices
        distance_matrix: 2D array of distances between nodes
    
    Returns:
        Next node index to visit
    """
    if not unvisited_nodes:
        return destination_node
    
    # Base case: only one node remains
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Calculate tour stage (remaining ratio)
    total_nodes = distance_matrix.shape[0]
    remaining_ratio = len(unvisited_nodes) / total_nodes
    
    # Adaptive regret parameters based on tour stage
    # Early tour: high regret weight (explore clusters)
    # Late tour: low regret weight (exploit greedy choices)
    regret_weight = 0.4 + 0.3 * remaining_ratio  # Decreases from 0.7 to 0.4
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # Immediate cost: current -> candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # Two-step look-ahead analysis
        remaining_after = unvisited_nodes - {candidate}
        
        if not remaining_after:
            # Only destination remains
            future_cost = distance_matrix[candidate][destination_node]
            regret = 0
        else:
            # Calculate lookahead costs for all possible next steps from candidate
            lookahead_costs = []
            
            for next_node in remaining_after:
                # Cost: candidate -> next_node
                lookahead_cost = distance_matrix[candidate][next_node]
                
                # Estimate third step cost
                remaining_third = remaining_after - {next_node}
                if remaining_third:
                    # Minimum distance from next_node to any remaining node
                    third_step_cost = min(distance_matrix[next_node][n] 
                                        for n in remaining_third)
                else:
                    # Go to destination
                    third_step_cost = distance_matrix[next_node][destination_node]
                
                # Weighted two-step cost (more weight on immediate next step)
                total_cost = lookahead_cost + 0.3 * third_step_cost
                lookahead_costs.append(total_cost)
            
            # Find best lookahead cost
            lookahead_costs.sort()
            best_lookahead = lookahead_costs[0]
            
            # Calculate adaptive regret
            if len(lookahead_costs) >= 2:
                # Average of top 2-3 alternatives (excluding best)
                k = min(3, len(lookahead_costs) - 1)
                avg_alternatives = np.mean(lookahead_costs[1:1+k])
                regret = best_lookahead - avg_alternatives
            else:
                regret = 0
            
            # Terminal cost consideration (weighted by tour stage)
            # More important early in tour to avoid paint-yourself-into-corner
            terminal_weight = 0.2 * (1 - remaining_ratio)
            terminal_cost = terminal_weight * distance_matrix[candidate][destination_node]
            
            future_cost = best_lookahead + terminal_cost
        
        # Composite score with adaptive regret
        score = immediate_cost + future_cost - regret_weight * regret
        
        # Tie-breaking: prefer nodes with better immediate connections
        if abs(score - best_score) < 1e-9:
            # If scores are equal, prefer node with smaller immediate cost
            if immediate_cost < distance_matrix[current_node][best_node]:
                best_score = score
                best_node = candidate
        elif score < best_score:
            best_score = score
            best_node = candidate
    
    return best_node
