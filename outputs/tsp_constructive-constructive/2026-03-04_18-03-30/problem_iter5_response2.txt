import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Select the next node using two-step look-ahead with enhanced regret heuristic
    and adaptive weighting.
    """
    if not unvisited_nodes:
        raise ValueError("No unvisited nodes available")

    # Adaptive weights based on remaining nodes
    n_remaining = len(unvisited_nodes)
    # Early stage: focus more on future connectivity
    # Late stage: focus more on immediate cost
    if n_remaining > len(distance_matrix) * 0.7:
        w_current, w_future, w_regret, w_dest = 0.3, 0.4, 0.2, 0.1
    elif n_remaining > len(distance_matrix) * 0.3:
        w_current, w_future, w_regret, w_dest = 0.5, 0.3, 0.1, 0.1
    else:
        w_current, w_future, w_regret, w_dest = 0.7, 0.1, 0.1, 0.1

    scores = {}
    look_ahead_costs = {}
    
    # Calculate two-step look-ahead for each candidate
    for node in unvisited_nodes:
        # Step 1: cost to reach candidate node
        cost_to_node = distance_matrix[current_node][node]
        
        # Step 2: minimum cost from candidate to remaining nodes
        remaining_after_node = unvisited_nodes - {node}
        if remaining_after_node:
            min_next_cost = min(distance_matrix[node][next_node] 
                              for next_node in remaining_after_node)
        else:
            # Last node: cost to return to destination
            min_next_cost = distance_matrix[node][destination_node]
        
        look_ahead_costs[node] = cost_to_node + min_next_cost
    
    # Enhanced regret calculation
    if len(look_ahead_costs) > 1:
        # Get top k alternative costs (k = min(3, remaining_nodes-1))
        k = min(3, len(look_ahead_costs) - 1)
        sorted_costs = sorted(look_ahead_costs.values())
        top_k_costs = sorted_costs[:k]
        avg_top_k = np.mean(top_k_costs)
        
        # Best look-ahead cost
        best_cost = min(look_ahead_costs.values())
        
        # Regret: difference between best and average of top alternatives
        base_regret = avg_top_k - best_cost
        
        for node in unvisited_nodes:
            # Normalized regret contribution
            node_cost = look_ahead_costs[node]
            regret_contribution = avg_top_k - node_cost if node_cost != best_cost else base_regret
            
            # Future connectivity: average distance to other unvisited nodes
            other_nodes = unvisited_nodes - {node}
            if other_nodes:
                avg_to_others = np.mean([distance_matrix[node][other] 
                                       for other in other_nodes])
            else:
                avg_to_others = distance_matrix[node][destination_node]
            
            # Destination consideration
            dest_cost = distance_matrix[node][destination_node]
            
            # Calculate composite score
            score = (w_current * distance_matrix[current_node][node] +
                    w_future * avg_to_others +
                    w_regret * regret_contribution -
                    w_dest * dest_cost)
            
            scores[node] = score
    else:
        # Only one node left
        node = next(iter(unvisited_nodes))
        scores[node] = (w_current * distance_matrix[current_node][node] -
                       w_dest * distance_matrix[node][destination_node])

    # Deterministic greedy selection
    next_node = min(scores, key=scores.get)
    return next_node
