import numpy as np
import random

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select next node using look-ahead mechanism with probabilistic exploration."""
    
    def path_completion_cost(start_node: int, remaining_nodes: set, dist_mat: np.ndarray, dest: int) -> float:
        """Estimate cost to complete path from start_node through remaining_nodes to dest."""
        if not remaining_nodes:
            return dist_mat[start_node][dest]
        
        # Nearest neighbor heuristic for completion estimate
        total = 0
        current = start_node
        nodes_left = set(remaining_nodes)
        
        while nodes_left:
            next_node = min(nodes_left, key=lambda x: dist_mat[current][x])
            total += dist_mat[current][next_node]
            current = next_node
            nodes_left.remove(next_node)
        
        total += dist_mat[current][dest]
        return total
    
    # Look-ahead parameters
    look_ahead_nodes = min(3, len(unvisited_nodes))
    exploration_rate = 0.2
    centrality_weight = 0.3
    completion_weight = 0.4
    
    # Calculate centrality scores for all nodes
    centrality_scores = {}
    for node in unvisited_nodes:
        # Measure how central this node is to unvisited nodes
        distances_to_others = [distance_matrix[node][other] for other in unvisited_nodes if other != node]
        centrality_scores[node] = np.mean(distances_to_others) if distances_to_others else 0
    
    # Evaluate candidates with look-ahead
    candidate_scores = {}
    
    for node in unvisited_nodes:
        # Immediate cost
        immediate_cost = distance_matrix[current_node][node]
        
        # Completion estimate after choosing this node
        remaining_after = unvisited_nodes - {node}
        completion_cost = path_completion_cost(node, remaining_after, distance_matrix, destination_node)
        
        # Centrality penalty (avoid isolated nodes early)
        centrality_penalty = centrality_scores[node]
        
        # Look-ahead: Evaluate k-step forward
        look_ahead_cost = 0
        if len(unvisited_nodes) > 1:
            # Select top look_ahead_nodes candidates for deeper evaluation
            test_remaining = remaining_after.copy()
            test_current = node
            
            for step in range(look_ahead_nodes):
                if not test_remaining:
                    break
                
                # Find next candidate in look-ahead
                candidates = list(test_remaining)
                if len(candidates) > 5:
                    candidates = random.sample(candidates, 5)  # Sample for efficiency
                
                # Evaluate immediate next step
                next_candidate = min(candidates, 
                                   key=lambda x: distance_matrix[test_current][x] + 
                                               0.5 * path_completion_cost(x, test_remaining - {x}, 
                                                                         distance_matrix, destination_node))
                
                look_ahead_cost += distance_matrix[test_current][next_candidate] / (step + 2)  # Discount future steps
                test_current = next_candidate
                test_remaining.remove(next_candidate)
        
        # Combined score
        total_score = (immediate_cost + 
                      completion_weight * completion_cost +
                      centrality_weight * centrality_penalty +
                      0.2 * look_ahead_cost)
        
        candidate_scores[node] = total_score
    
    # Probabilistic selection for exploration
    if random.random() < exploration_rate and len(unvisited_nodes) > 1:
        # Randomly select from top 30% of candidates
        sorted_nodes = sorted(candidate_scores.keys(), key=lambda x: candidate_scores[x])
        top_percentile = max(1, len(sorted_nodes) // 3)
        return random.choice(sorted_nodes[:top_percentile])
    else:
        # Greedy selection
        return min(candidate_scores, key=candidate_scores.get)
