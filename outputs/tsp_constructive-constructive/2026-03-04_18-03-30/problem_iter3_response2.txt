import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select the next node to visit using a sophisticated multi-criteria heuristic
    with two-step look-ahead and cluster-awareness.
    """
    if not unvisited_nodes:
        raise ValueError("Unvisited nodes set is empty")
    
    # Precompute centroids for cluster awareness
    def compute_centroid_penalty(candidate: int, remaining: Set[int]) -> float:
        """Penalize nodes that isolate remaining clusters"""
        if not remaining:
            return 0.0
        
        # Approximate cluster centroid using distance sampling
        sample_size = min(5, len(remaining))
        sample_nodes = np.random.choice(list(remaining), size=sample_size, replace=False)
        
        # Average distance from candidate to sample nodes
        avg_dist_to_cluster = np.mean([distance_matrix[candidate][n] for n in sample_nodes])
        
        # Penalize candidates far from the remaining cluster
        return avg_dist_to_cluster * 0.15
    
    scores = {}
    remaining_after_candidate = unvisited_nodes.copy()
    
    for candidate in unvisited_nodes:
        # Immediate cost
        immediate_cost = distance_matrix[current_node][candidate]
        
        # Two-step look-ahead: consider best next move from candidate
        remaining_after_candidate.remove(candidate)
        if remaining_after_candidate:
            # Find minimum cost from candidate to any remaining node (excluding destination if present)
            next_costs = [distance_matrix[candidate][next_node] 
                         for next_node in remaining_after_candidate]
            lookahead_cost = min(next_costs) if next_costs else 0
        else:
            # If no nodes left after candidate, cost to destination
            lookahead_cost = distance_matrix[candidate][destination_node]
        
        # Cluster penalty
        cluster_penalty = compute_centroid_penalty(candidate, remaining_after_candidate)
        
        # Multi-criteria weighted score
        score = (0.35 * immediate_cost + 
                 0.25 * lookahead_cost + 
                 0.20 * cluster_penalty - 
                 0.20 * distance_matrix[candidate][destination_node])
        
        scores[candidate] = score
        
        # Restore candidate for next iteration
        remaining_after_candidate.add(candidate)
    
    # Deterministic greedy selection
    return min(scores.items(), key=lambda x: x[1])[0]
