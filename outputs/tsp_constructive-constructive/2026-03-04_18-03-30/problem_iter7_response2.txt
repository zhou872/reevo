import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node using two-step lookahead with focus on destination proximity.
    
    Args:
        current_node: The current node index
        destination_node: The final destination node index (starting point for TSP)
        unvisited_nodes: Set of unvisited node indices
        distance_matrix: 2D array of distances between nodes
        
    Returns:
        int: The next node to visit
    """
    # Fixed weights for deterministic selection
    w_immediate = 0.35      # Weight for immediate cost to candidate
    w_lookahead = 0.30      # Weight for two-step lookahead cost
    w_dest_proximity = 0.20 # Weight for proximity to final destination
    w_dispersion = 0.15     # Weight for neighborhood dispersion
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # Immediate cost: current → candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # Two-step lookahead: candidate → next best
        # Find minimum additional distance from candidate to any other unvisited node
        # (excluding candidate and destination if it's in unvisited)
        other_unvisited = [n for n in unvisited_nodes if n != candidate]
        if other_unvisited:
            # Best possible next step from candidate
            next_costs = [distance_matrix[candidate][n] for n in other_unvisited]
            lookahead_cost = min(next_costs)
        else:
            # If candidate is the last unvisited, then directly to destination
            lookahead_cost = distance_matrix[candidate][destination_node]
        
        # Destination proximity: how close candidate is to final destination
        dest_dist = distance_matrix[candidate][destination_node]
        
        # Neighborhood dispersion: spread of distances from candidate to remaining nodes
        # Encourages selecting nodes that are centrally located among remaining nodes
        if len(unvisited_nodes) > 1:
            remaining_nodes = [n for n in unvisited_nodes if n != candidate]
            distances_to_remaining = [distance_matrix[candidate][n] for n in remaining_nodes]
            dispersion = np.std(distances_to_remaining) if distances_to_remaining else 0
        else:
            dispersion = 0
        
        # Combined score (lower is better)
        score = (w_immediate * immediate_cost + 
                 w_lookahead * lookahead_cost + 
                 w_dest_proximity * dest_dist + 
                 w_dispersion * dispersion)
        
        # Track best node
        if score < best_score:
            best_score = score
            best_node = candidate
    
    return best_node
