import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, 
                        destination_node: int, 
                        unvisited_nodes: Set[int], 
                        distance_matrix: np.ndarray) -> int:
    """
    Select the next node to visit using a deterministic, spatially-aware heuristic.
    
    The algorithm combines:
    1. Immediate nearest-neighbor cost
    2. Two-step lookahead to avoid dead ends
    3. Cluster awareness to maintain spatial coherence
    4. Destination proximity to guide towards completion
    
    Parameters:
    -----------
    current_node : int
        Current node index
    destination_node : int
        Final destination node index
    unvisited_nodes : Set[int]
        Set of unvisited node indices
    distance_matrix : np.ndarray
        Distance matrix between nodes (n x n)
    
    Returns:
    --------
    int : Next node to visit (deterministic choice)
    """
    if not unvisited_nodes:
        return destination_node
    
    # Fixed weights for robust performance (tuned for TSP)
    WEIGHT_IMMEDIATE = 0.40    # Current to candidate distance
    WEIGHT_LOOKAHEAD = 0.25    # Two-step lookahead consideration
    WEIGHT_CLUSTER = 0.20      # Cluster coherence penalty
    WEIGHT_DESTINATION = 0.15  # Destination proximity (negative weight)
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # 1. Immediate cost (nearest neighbor component)
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Two-step lookahead: minimum cost from candidate to any remaining node
        remaining_after_candidate = unvisited_nodes - {candidate}
        if remaining_after_candidate:
            # Find minimum distance from candidate to remaining nodes
            lookahead_cost = min(distance_matrix[candidate][n] 
                                for n in remaining_after_candidate)
        else:
            # Last node before destination
            lookahead_cost = distance_matrix[candidate][destination_node]
        
        # 3. Cluster awareness: penalize nodes that isolate remaining clusters
        cluster_penalty = 0.0
        if remaining_after_candidate:
            # Average distance from candidate to remaining nodes
            # Higher values mean candidate is far from the cluster
            avg_dist = np.mean([distance_matrix[candidate][n] 
                               for n in remaining_after_candidate])
            cluster_penalty = avg_dist
        
        # 4. Destination proximity (negative term - closer is better)
        dest_distance = distance_matrix[candidate][destination_node]
        
        # Combined score (lower is better)
        score = (WEIGHT_IMMEDIATE * immediate_cost +
                 WEIGHT_LOOKAHEAD * lookahead_cost +
                 WEIGHT_CLUSTER * cluster_penalty -
                 WEIGHT_DESTINATION * dest_distance)
        
        # Update best node
        if score < best_score:
            best_score = score
            best_node = candidate
    
    return best_node
