import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node using a two-step look-ahead mechanism with adaptive weights."""
    # Adaptive weights based on remaining nodes
    n_remaining = len(unvisited_nodes)
    if n_remaining == 1:
        # If only one node left, we must choose it
        return next(iter(unvisited_nodes))
    
    # Adaptive weight adjustment
    progress = 1.0 - (n_remaining / (len(distance_matrix) - 1))
    w_current = 0.4 + 0.2 * progress  # Increases as we progress
    w_future = 0.6 - 0.2 * progress   # Decreases as we progress
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # Current cost to reach candidate
        current_cost = distance_matrix[current_node][candidate]
        
        # Estimate future cost from candidate
        future_cost = estimate_future_cost(candidate, destination_node, unvisited_nodes, distance_matrix)
        
        # Combined score with adaptive weights
        score = w_current * current_cost + w_future * future_cost
        
        if score < best_score:
            best_score = score
            best_node = candidate
    
    return best_node

def estimate_future_cost(candidate: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> float:
    """Estimate the future cost from candidate node using two-step look-ahead."""
    remaining = unvisited_nodes - {candidate}
    
    if not remaining:
        # Direct to destination if no nodes left
        return distance_matrix[candidate][destination_node]
    
    # Two-step look-ahead: candidate -> next -> destination
    future_costs = []
    
    for next_node in remaining:
        # First step: candidate to next_node
        first_step = distance_matrix[candidate][next_node]
        
        # Estimate remaining path from next_node
        remaining_after = remaining - {next_node}
        if not remaining_after:
            # If next_node is the last unvisited, then go to destination
            second_step = distance_matrix[next_node][destination_node]
        else:
            # Estimate minimum connection from next_node to remaining nodes
            min_connection = min(distance_matrix[next_node][node] for node in remaining_after)
            
            # Estimate remaining path length (simplified)
            avg_distance = np.mean([distance_matrix[i][j] for i in remaining_after for j in remaining_after if i != j])
            second_step = min_connection + avg_distance * 0.5
        
        future_costs.append(first_step + second_step)
    
    # Return average of top 30% best future costs (optimistic but not greedy)
    k = max(1, int(len(future_costs) * 0.3))
    best_costs = sorted(future_costs)[:k]
    return np.mean(best_costs)
