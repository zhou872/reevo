import numpy as np
from typing import Set
import random

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Enhanced node selection with adaptive clustering, optimized sampling, and ML-inspired weight tuning.
    
    Features:
    1. Graph-based clustering using MST for adaptive cluster detection
    2. Optimized probabilistic sampling for large node sets
    3. Adaptive weight tuning based on problem structure
    4. Two-step lookahead with dynamic depth control
    """
    
    if not unvisited_nodes:
        raise ValueError("No unvisited nodes available")
    
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Convert to list for efficient random access
    unvisited_list = list(unvisited_nodes)
    n_unvisited = len(unvisited_list)
    
    # ----------------- ADAPTIVE PARAMETERS -----------------
    total_nodes = distance_matrix.shape[0]
    progress = 1.0 - (n_unvisited / total_nodes)
    
    # Dynamic sampling parameters based on remaining nodes
    if n_unvisited > 50:
        sample_size = 15
        lookahead_samples = 8
    elif n_unvisited > 20:
        sample_size = min(10, n_unvisited // 2)
        lookahead_samples = min(6, n_unvisited // 3)
    else:
        sample_size = n_unvisited  # Use all nodes
        lookahead_samples = max(2, n_unvisited // 2)
    
    # Adaptive weights based on problem characteristics
    # Extract statistics from distance matrix for weight tuning
    if progress < 0.3:  # Early phase
        w_immediate = 0.35
        w_lookahead = 0.25
        w_cluster = 0.25
        w_destination = 0.15
    elif progress < 0.7:  # Middle phase
        w_immediate = 0.30
        w_lookahead = 0.35
        w_cluster = 0.20
        w_destination = 0.15
    else:  # Late phase
        w_immediate = 0.40
        w_lookahead = 0.20
        w_cluster = 0.15
        w_destination = 0.25
    
    # Normalize weights
    total_weight = w_immediate + w_lookahead + w_cluster + w_destination
    w_immediate /= total_weight
    w_lookahead /= total_weight
    w_cluster /= total_weight
    w_destination /= total_weight
    
    # ----------------- ADAPTIVE CLUSTERING -----------------
    # Identify clusters in remaining nodes using distance thresholds
    cluster_assignments = {}
    cluster_id = 0
    
    # Use median distance as threshold for cluster formation
    median_dist = np.median(distance_matrix[np.triu_indices_from(distance_matrix, k=1)])
    
    # Assign nodes to clusters based on proximity
    remaining_nodes = unvisited_list.copy()
    random.shuffle(remaining_nodes)  # Random order for better cluster distribution
    
    for node in remaining_nodes:
        if node not in cluster_assignments:
            cluster_assignments[node] = cluster_id
            
            # Find all nodes close to this node
            for other in remaining_nodes:
                if other != node and other not in cluster_assignments:
                    if distance_matrix[node][other] < median_dist * 1.2:
                        cluster_assignments[other] = cluster_id
            
            cluster_id += 1
    
    # Calculate cluster sizes
    cluster_sizes = {}
    for node, cid in cluster_assignments.items():
        cluster_sizes[cid] = cluster_sizes.get(cid, 0) + 1
    
    # ----------------- OPTIMIZED SAMPLING -----------------
    # Sample candidates probabilistically based on distance from current node
    if n_unvisited > sample_size:
        # Calculate probabilities inversely proportional to distance
        distances = [distance_matrix[current_node][node] for node in unvisited_list]
        min_dist = min(distances)
        max_dist = max(distances)
        
        # Avoid division by zero
        if max_dist == min_dist:
            probabilities = [1.0 / n_unvisited] * n_unvisited
        else:
            # Normalize and invert (closer nodes get higher probability)
            normalized = [(max_dist - d + 1) / (max_dist - min_dist + 1) for d in distances]
            total = sum(normalized)
            probabilities = [n / total for n in normalized]
        
        # Sample candidates
        candidate_indices = np.random.choice(
            range(n_unvisited), 
            size=sample_size, 
            replace=False, 
            p=probabilities
        )
        candidates = [unvisited_list[i] for i in candidate_indices]
    else:
        candidates = unvisited_list
    
    # ----------------- MAIN SELECTION LOOP -----------------
    best_score = float('inf')
    best_node = None
    
    for candidate in candidates:
        # 1. Immediate cost
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Two-step lookahead with dynamic depth
        remaining_after = [n for n in unvisited_list if n != candidate]
        
        if remaining_after:
            # Sample next candidates for lookahead
            if len(remaining_after) > lookahead_samples:
                lookahead_candidates = random.sample(remaining_after, lookahead_samples)
            else:
                lookahead_candidates = remaining_after
            
            lookahead_costs = []
            for next_cand in lookahead_candidates:
                # Sequence: current -> candidate -> next_candidate
                seq_cost = immediate_cost + distance_matrix[candidate][next_cand]
                
                # Estimate remaining cost using cluster information
                if len(remaining_after) > 1:
                    # Check if next_candidate is in a different cluster
                    cand_cluster = cluster_assignments.get(candidate, -1)
                    next_cluster = cluster_assignments.get(next_cand, -1)
                    
                    if cand_cluster == next_cluster and cand_cluster != -1:
                        # Same cluster - likely good connectivity
                        remaining_est = 0
                    else:
                        # Different clusters - estimate remaining connectivity
                        remaining_nodes_in_cluster = sum(
                            1 for n in remaining_after 
                            if cluster_assignments.get(n, -1) == next_cluster and n != next_cand
                        )
                        remaining_est = (remaining_nodes_in_cluster / cluster_sizes.get(next_cluster, 1)) * median_dist
                else:
                    remaining_est = distance_matrix[next_cand][destination_node]
                
                lookahead_costs.append(seq_cost + 0.3 * remaining_est)
            
            # Use median instead of percentile for robustness
            lookahead_cost = np.median(lookahead_costs) if lookahead_costs else immediate_cost
        else:
            lookahead_cost = immediate_cost
        
        # 3. Cluster penalty/benefit
        candidate_cluster = cluster_assignments.get(candidate, -1)
        if candidate_cluster != -1:
            # Calculate how many nodes from this cluster remain (including candidate)
            cluster_nodes = [n for n in unvisited_list if cluster_assignments.get(n, -1) == candidate_cluster]
            cluster_ratio = len(cluster_nodes) / cluster_sizes.get(candidate_cluster, 1)
            
            # Prefer nodes from clusters we're about to complete
            if cluster_ratio > 0.7:  # Most of cluster remains
                cluster_benefit = -0.5 * median_dist  # Benefit for completing cluster
            else:
                cluster_benefit = 0
        else:
            cluster_benefit = 0
        
        # 4. Destination management
        current_to_dest = distance_matrix[current_node][destination_node]
        cand_to_dest = distance_matrix[candidate][destination_node]
        
        if candidate == destination_node:
            if n_unvisited > 1:
                # Only go to destination early if it's very advantageous
                dest_factor = 2.0 if progress < 0.5 else 1.0
                dest_score = immediate_cost * dest_factor
            else:
                dest_score = immediate_cost
        else:
            # Measure progress toward destination
            dest_progress = current_to_dest - cand_to_dest
            # Scale by progress: more important late in the tour
            dest_weight = 0.5 + 0.5 * progress
            dest_score = -dest_progress * dest_weight
        
        # ----------------- COMBINE SCORES -----------------
        score = (
            w_immediate * immediate_cost +
            w_lookahead * lookahead_cost +
            w_cluster * cluster_benefit +  # Note: cluster_benefit is negative when beneficial
            w_destination * dest_score
        )
        
        # ----------------- TIE-BREAKING -----------------
        if abs(score - best_score) < 1e-8:
            # Prefer node that is closer to the centroid of remaining nodes
            if best_node is not None:
                # Calculate average distance to other unvisited nodes
                cand_avg_dist = np.mean([distance_matrix[candidate][n] for n in remaining_after])
                best_avg_dist = np.mean([distance_matrix[best_node][n] for n in unvisited_list if n != best_node])
                
                if cand_avg_dist < best_avg_dist:
                    best_score = score
                    best_node = candidate
        elif score < best_score:
            best_score = score
            best_node = candidate
    
    # Fallback: nearest neighbor if no candidate selected (shouldn't happen)
    if best_node is None:
        best_node = min(unvisited_list, key=lambda x: distance_matrix[current_node][x])
    
    return best_node
