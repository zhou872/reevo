import numpy as np
from typing import Set
import heapq

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Enhanced node selection for TSP constructive heuristic with adaptive multi-step
    lookahead, dynamic clustering, and online learning of heuristic weights.
    
    Features:
    1. Adaptive multi-step lookahead (2-4 steps based on remaining nodes)
    2. Dynamic cluster preservation using spectral clustering principles
    3. Online weight adaptation via reinforcement of successful patterns
    4. Intelligent candidate pruning with quality-diversity balance
    """
    
    # Base case: only destination remains
    if unvisited_nodes == {destination_node}:
        return destination_node
    
    # Initialize adaptive parameters
    n_total = len(distance_matrix)
    n_remaining = len(unvisited_nodes)
    progress = 1.0 - (n_remaining / n_total)
    
    # Adaptive lookahead depth: deeper when fewer nodes remain
    lookahead_depth = min(4, max(2, int(4 * (1 - progress) + 2)))
    
    # Dynamic weights initialized (will be adjusted during evaluation)
    weights = {
        'immediate': 0.35 - 0.25 * progress,
        'lookahead': 0.25 + 0.35 * progress,
        'cluster': 0.20 + 0.15 * progress,
        'destination': 0.20 - 0.10 * progress
    }
    
    # Normalize weights
    total = sum(weights.values())
    for key in weights:
        weights[key] /= total
    
    # Pre-compute normalized distances for efficient comparisons
    valid_distances = distance_matrix[distance_matrix > 0]
    min_dist = np.min(valid_distances) if len(valid_distances) > 0 else 1
    max_dist = np.max(distance_matrix)
    
    def normalize(d):
        return (d - min_dist) / (max_dist - min_dist + 1e-10)
    
    # Candidate selection with quality-diversity balance
    candidates = list(unvisited_nodes - {destination_node})
    
    if len(candidates) > 15:
        # Get base scores using simple greedy criterion
        base_scores = []
        for node in candidates:
            immediate = normalize(distance_matrix[current_node][node])
            to_dest = normalize(distance_matrix[node][destination_node])
            base_score = 0.7 * immediate + 0.3 * to_dest
            base_scores.append((base_score, node))
        
        # Select top 10 by quality
        base_scores.sort()
        top_by_quality = [node for _, node in base_scores[:10]]
        
        # Add 5 diverse candidates based on distance from already selected
        remaining = [node for _, node in base_scores[10:]]
        if remaining:
            diverse_candidates = []
            for _ in range(min(5, len(remaining))):
                if not diverse_candidates:
                    diverse_candidates.append(remaining.pop(0))
                else:
                    # Find node farthest from already selected diverse candidates
                    max_min_dist = -1
                    best_node = None
                    for node in remaining:
                        min_dist_to_diverse = min(normalize(distance_matrix[node][d]) 
                                                for d in diverse_candidates)
                        if min_dist_to_diverse > max_min_dist:
                            max_min_dist = min_dist_to_diverse
                            best_node = node
                    if best_node:
                        diverse_candidates.append(best_node)
                        remaining.remove(best_node)
            candidates = top_by_quality + diverse_candidates
    
    # Track successful patterns for weight adaptation
    pattern_scores = []
    
    # Evaluate each candidate
    best_score = float('inf')
    best_node = None
    
    for candidate in candidates:
        # 1. Immediate cost (normalized)
        immediate_cost = normalize(distance_matrix[current_node][candidate])
        
        # 2. Multi-step lookahead with beam search
        remaining_after = unvisited_nodes - {candidate}
        
        # Adjust lookahead depth based on remaining nodes
        current_depth = min(lookahead_depth, len(remaining_after) + 1)
        
        if current_depth > 1 and len(remaining_after) > 0:
            # Use beam search for efficient multi-step lookahead
            beam_width = min(3, len(remaining_after))
            beam = [(immediate_cost, [candidate], remaining_after)]
            
            for step in range(1, current_depth):
                new_beam = []
                for cost, path, remaining in beam:
                    last_node = path[-1]
                    
                    if not remaining:
                        continue
                    
                    # Evaluate next steps from last node
                    next_options = []
                    for next_node in remaining:
                        if next_node == destination_node and len(remaining) > 1:
                            continue  # Avoid destination too early
                        
                        step_cost = normalize(distance_matrix[last_node][next_node])
                        new_cost = cost + step_cost
                        new_path = path + [next_node]
                        new_remaining = remaining - {next_node}
                        next_options.append((new_cost, new_path, new_remaining))
                    
                    # Keep best options in beam
                    next_options.sort(key=lambda x: x[0])
                    new_beam.extend(next_options[:beam_width])
                
                if new_beam:
                    beam = new_beam
                else:
                    break
            
            # Evaluate final cost to destination for each beam path
            lookahead_costs = []
            for cost, path, remaining in beam:
                last_node = path[-1]
                if remaining:
                    # Estimate remaining cost using nearest neighbor heuristic
                    estimated_remaining = 0
                    current = last_node
                    temp_remaining = set(remaining)
                    
                    while temp_remaining:
                        next_node = min(temp_remaining, 
                                      key=lambda x: normalize(distance_matrix[current][x]))
                        estimated_remaining += normalize(distance_matrix[current][next_node])
                        current = next_node
                        temp_remaining.remove(next_node)
                    
                    # Add cost to destination
                    final_cost = cost + estimated_remaining + normalize(distance_matrix[current][destination_node])
                else:
                    final_cost = cost + normalize(distance_matrix[last_node][destination_node])
                
                lookahead_costs.append(final_cost)
            
            # Use 30th percentile to avoid extreme paths
            lookahead_cost = np.percentile(lookahead_costs, 30) if lookahead_costs else immediate_cost
        else:
            lookahead_cost = immediate_cost
        
        # 3. Dynamic cluster preservation
        if len(unvisited_nodes) > 2:
            # Compute candidate's centrality in remaining graph
            remaining_without_candidate = unvisited_nodes - {candidate}
            
            if len(remaining_without_candidate) >= 2:
                # Compute average distance to other remaining nodes
                avg_dist_to_others = np.mean([normalize(distance_matrix[candidate][other])
                                            for other in remaining_without_candidate])
                
                # Compute connectivity of remaining subgraph
                connectivity_score = 0
                sample_size = min(5, len(remaining_without_candidate))
                sample_nodes = list(remaining_without_candidate)[:sample_size]
                
                for i in range(sample_size):
                    for j in range(i+1, sample_size):
                        connectivity_score += normalize(distance_matrix[sample_nodes[i]][sample_nodes[j]])
                
                if sample_size >= 2:
                    connectivity_score /= (sample_size * (sample_size - 1) / 2)
                
                # Cluster penalty: high if candidate is central AND removing it breaks connectivity
                cluster_penalty = avg_dist_to_others * 0.4 + connectivity_score * 0.6
            else:
                cluster_penalty = 0
        else:
            cluster_penalty = 0
        
        # 4. Destination-oriented guidance
        if candidate == destination_node:
            # Only choose destination early if it's very advantageous
            if n_remaining > 2:
                dest_attraction = immediate_cost * 3  # Heavy penalty for early destination
            else:
                dest_attraction = immediate_cost
        else:
            # Consider progress toward destination
            current_to_dest = normalize(distance_matrix[current_node][destination_node])
            cand_to_dest = normalize(distance_matrix[candidate][destination_node])
            
            # Favor nodes that make progress toward destination
            dest_progress = current_to_dest - cand_to_dest
            
            # Also consider if this is a detour
            direct_vs_detour = immediate_cost / (current_to_dest + 1e-10)
            
            dest_attraction = cand_to_dest * 0.7 - dest_progress * 0.2 + direct_vs_detour * 0.1
        
        # Combine components with current weights
        score = (weights['immediate'] * immediate_cost +
                weights['lookahead'] * lookahead_cost +
                weights['cluster'] * cluster_penalty +
                weights['destination'] * dest_attraction)
        
        # Store pattern for weight adaptation
        pattern_scores.append({
            'node': candidate,
            'score': score,
            'components': [immediate_cost, lookahead_cost, cluster_penalty, dest_attraction]
        })
        
        # Small random tie-breaking
        score *= (1.0 + np.random.random() * 0.001)
        
        if score < best_score:
            best_score = score
            best_node = candidate
    
    # Online weight adaptation based on top candidates
    if len(pattern_scores) >= 3:
        top_patterns = sorted(pattern_scores, key=lambda x: x['score'])[:3]
        
        # Analyze which components were strong in good candidates
        component_importance = np.zeros(4)
        for pattern in top_patterns:
            # Lower component values are better for all components
            for i in range(4):
                component_importance[i] += 1.0 / (pattern['components'][i] + 1e-10)
        
        # Normalize and update weights slightly
        if np.sum(component_importance) > 0:
            component_importance /= np.sum(component_importance)
            
            # Blend old weights with learned importance (slow adaptation)
            learning_rate = 0.1
            keys = ['immediate', 'lookahead', 'cluster', 'destination']
            for i, key in enumerate(keys):
                weights[key] = (1 - learning_rate) * weights[key] + learning_rate * component_importance[i]
            
            # Renormalize
            total = sum(weights.values())
            for key in weights:
                weights[key] /= total
    
    # Fallback: nearest neighbor if no candidate selected
    if best_node is None:
        best_node = min(unvisited_nodes, 
                       key=lambda x: distance_matrix[current_node][x])
    
    return best_node
