import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select next node using adaptive regret-weighted heuristic with improved exploration
    and path length estimation.
    
    Parameters:
    -----------
    current_node : int
        Current node index
    destination_node : int
        Final destination node index
    unvisited_nodes : Set[int]
        Set of unvisited node indices
    distance_matrix : np.ndarray
        Distance matrix between nodes
    
    Returns:
    --------
    int : Next node to visit
    """
    if not unvisited_nodes:
        return destination_node
    
    # Adaptive parameters based on search progress
    n_total = distance_matrix.shape[0] - 1  # Excluding current node
    progress = 1.0 - (len(unvisited_nodes) / n_total)
    
    # Dynamic weights that adapt based on progress
    # More emphasis on immediate cost and destination proximity as we progress
    alpha = 0.4 + 0.3 * progress  # Weight for immediate cost (40-70%)
    beta = 0.4 - 0.2 * progress   # Weight for look-ahead (40-20%)
    gamma = 0.15 + 0.1 * progress # Weight for regret (15-25%)
    delta = 0.05 + 0.05 * progress # Weight for destination (5-10%)
    
    # Adaptive exploration parameters
    # More exploration early, focused exploitation late
    exploration_prob = max(0.05, 0.3 * (1.0 - progress))
    exploration_width = max(0.2, 0.5 * (1.0 - progress))  # Width of candidate pool for exploration
    
    candidate_scores = {}
    
    for candidate in unvisited_nodes:
        # Immediate cost from current to candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # Look-ahead: best possible move after candidate
        remaining = unvisited_nodes - {candidate}
        
        if remaining:
            # Find costs from candidate to all remaining nodes
            look_ahead_costs = [distance_matrix[candidate][next_node] for next_node in remaining]
            best_look_ahead = min(look_ahead_costs)
            
            # Enhanced regret calculation
            if len(look_ahead_costs) > 1:
                sorted_costs = sorted(look_ahead_costs)
                # Regret as difference between best and average of top alternatives
                if len(sorted_costs) >= 3:
                    top_avg = np.mean(sorted_costs[1:3])  # Average of 2nd and 3rd best
                    regret = top_avg - sorted_costs[0]
                else:
                    regret = sorted_costs[1] - sorted_costs[0]
            else:
                regret = 0
                
            # Estimate minimum path length from candidate through remaining nodes to destination
            # Using minimum spanning tree approximation (nearest neighbor chain)
            if len(remaining) > 1:
                # Create sorted list of distances from candidate to remaining nodes
                remaining_list = list(remaining)
                distances_to_remaining = [(dist, node) for node, dist in 
                                         zip(remaining_list, [distance_matrix[candidate][n] for n in remaining_list])]
                distances_to_remaining.sort()
                
                # Estimate path length using nearest neighbor chain
                estimated_path_length = 0
                last_node = candidate
                remaining_set = set(remaining)
                
                while remaining_set:
                    # Find nearest unvisited node from last node
                    nearest_dist = float('inf')
                    nearest_node = None
                    for node in remaining_set:
                        dist = distance_matrix[last_node][node]
                        if dist < nearest_dist:
                            nearest_dist = dist
                            nearest_node = node
                    
                    estimated_path_length += nearest_dist
                    last_node = nearest_node
                    remaining_set.remove(nearest_node)
                
                # Add distance from last node to destination
                estimated_path_length += distance_matrix[last_node][destination_node]
                path_estimate = estimated_path_length / len(unvisited_nodes)  # Normalize
            else:
                path_estimate = best_look_ahead + distance_matrix[list(remaining)[0]][destination_node]
        else:
            # If candidate is last unvisited node
            best_look_ahead = distance_matrix[candidate][destination_node]
            regret = 0
            path_estimate = best_look_ahead
        
        # Distance to destination
        dest_dist = distance_matrix[candidate][destination_node]
        
        # Combined score with dynamic weights
        # Lower score is better (minimizing total path length)
        score = (
            alpha * immediate_cost +
            beta * path_estimate -
            gamma * regret +
            delta * dest_dist
        )
        
        candidate_scores[candidate] = score
    
    # Adaptive exploration mechanism
    if np.random.random() < exploration_prob and len(candidate_scores) > 1:
        # Sort candidates by score
        sorted_candidates = sorted(candidate_scores.keys(), key=lambda x: candidate_scores[x])
        
        # Determine exploration pool size based on current progress
        pool_size = max(2, int(len(sorted_candidates) * exploration_width))
        exploration_pool = sorted_candidates[:pool_size]
        
        # Weighted random selection within pool (better candidates have higher probability)
        if len(exploration_pool) > 1:
            # Convert scores to probabilities (lower scores = higher probability)
            pool_scores = [candidate_scores[node] for node in exploration_pool]
            max_score = max(pool_scores)
            min_score = min(pool_scores)
            
            if max_score > min_score:
                # Inverse linear mapping: lower score = higher probability
                weights = [max_score - score + (max_score - min_score) * 0.1 
                          for score in pool_scores]
                weights = np.array(weights) / sum(weights)
                return np.random.choice(exploration_pool, p=weights)
        
        return np.random.choice(exploration_pool)
    
    # Return the best candidate (exploitation)
    return min(candidate_scores, key=candidate_scores.get)
