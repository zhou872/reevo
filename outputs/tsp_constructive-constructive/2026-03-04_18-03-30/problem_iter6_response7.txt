import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select next node using a hybrid adaptive heuristic with dynamic lookahead and regret-diversification.
    
    Combines:
    1. Nearest Neighbor baseline with adaptive weights
    2. Dynamic lookahead depth based on remaining nodes
    3. Regret with diversification (avoiding local minima)
    4. Progress-aware component blending
    
    Args:
        current_node: Index of current node.
        destination_node: Index of final destination node.
        unvisited_nodes: Set of nodes not yet visited.
        distance_matrix: 2D numpy array of pairwise distances.
    
    Returns:
        Index of the next node to visit.
    """
    # Trivial cases
    if not unvisited_nodes:
        return destination_node
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Problem parameters
    n_total = distance_matrix.shape[0]
    n_remaining = len(unvisited_nodes)
    progress = 1.0 - n_remaining / (n_total - 1)
    
    # Hybrid strategy selection based on problem stage
    if progress < 0.3:  # Early stage: exploration focus
        strategy = "exploration"
    elif progress < 0.7:  # Middle stage: balanced
        strategy = "balanced"
    else:  # Late stage: exploitation focus
        strategy = "exploitation"
    
    # Dynamic lookahead depth - adapts to remaining nodes
    base_depth = min(3, n_remaining - 1)
    if strategy == "exploration":
        lookahead_depth = base_depth + 1
    elif strategy == "exploitation":
        lookahead_depth = max(1, base_depth - 1)
    else:
        lookahead_depth = base_depth
    
    # Adaptive component weights based on strategy and progress
    if strategy == "exploration":
        w_imm = 0.25  # Lower immediate focus
        w_reg = 0.35  # Higher regret focus
        w_look = 0.25  # Moderate lookahead
        w_dest = 0.15  # Lower destination focus
    elif strategy == "exploitation":
        w_imm = 0.45  # Higher immediate focus
        w_reg = 0.15  # Lower regret focus
        w_look = 0.20  # Lower lookahead
        w_dest = 0.20  # Higher destination focus
    else:  # balanced
        w_imm = 0.35 + 0.05 * progress  # Increasing with progress
        w_reg = 0.30 - 0.05 * progress  # Decreasing with progress
        w_look = 0.20 - 0.05 * progress  # Decreasing with progress
        w_dest = 0.15 + 0.05 * progress  # Increasing with progress
    
    # Normalize weights
    total = w_imm + w_reg + w_look + w_dest
    w_imm, w_reg, w_look, w_dest = [w/total for w in [w_imm, w_reg, w_look, w_dest]]
    
    # Pre-compute key statistics
    current_to_unvisited = [distance_matrix[current_node][n] for n in unvisited_nodes]
    min_immediate = min(current_to_unvisited) if current_to_unvisited else 0
    avg_immediate = np.mean(current_to_unvisited) if current_to_unvisited else 0
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # Avoid premature return to destination
        if candidate == destination_node and n_remaining > 1:
            continue
        
        # 1. Immediate cost component (Nearest Neighbor hybrid)
        immediate_cost = distance_matrix[current_node][candidate]
        nn_score = immediate_cost / (avg_immediate + 1e-10)
        
        # 2. Dynamic lookahead component
        remaining_after_candidate = unvisited_nodes - {candidate}
        lookahead_score = 0
        
        if remaining_after_candidate and lookahead_depth > 0:
            # Calculate average cost of lookahead paths
            lookahead_costs = []
            for next_candidate in list(remaining_after_candidate)[:min(5, len(remaining_after_candidate))]:
                # Simulate one more step
                cost = distance_matrix[candidate][next_candidate]
                
                # Consider cost from that node to remaining nodes
                remaining_after_next = remaining_after_candidate - {next_candidate}
                if remaining_after_next:
                    future_costs = [distance_matrix[next_candidate][n] for n in remaining_after_next]
                    cost += min(future_costs) if future_costs else 0
                else:
                    cost += distance_matrix[next_candidate][destination_node]
                
                lookahead_costs.append(cost)
            
            if lookahead_costs:
                lookahead_score = np.mean(lookahead_costs)
        
        # 3. Enhanced regret with diversification
        regret_score = 0
        if len(remaining_after_candidate) >= 2:
            # Calculate distances from candidate to remaining nodes
            candidate_to_remaining = [distance_matrix[candidate][n] for n in remaining_after_candidate]
            
            # Sort distances
            sorted_dists = np.sort(candidate_to_remaining)
            
            # Base regret: difference between best and average
            base_regret = np.mean(sorted_dists[:min(3, len(sorted_dists))]) - sorted_dists[0]
            
            # Diversification factor: avoid clusters by preferring nodes with varied distances
            if len(sorted_dists) >= 3:
                dist_std = np.std(sorted_dists[:3])
                dist_mean = np.mean(sorted_dists[:3])
                diversification = dist_std / (dist_mean + 1e-10) if dist_mean > 0 else 0
            else:
                diversification = 0
            
            # Combined regret score (higher is worse)
            regret_score = base_regret * (1.0 - 0.3 * diversification)
        
        # 4. Destination-aware component
        dest_distance = distance_matrix[candidate][destination_node]
        dest_score = dest_distance
        
        # 5. Progress bonus: favor nodes that are central when many remain
        if strategy == "exploration" and len(remaining_after_candidate) >= 3:
            # Calculate average distance from candidate to other unvisited nodes
            centrality = np.mean([distance_matrix[candidate][n] for n in remaining_after_candidate])
            progress_bonus = 0.1 * centrality
        else:
            progress_bonus = 0
        
        # Combined score (lower is better)
        score = (w_imm * nn_score +
                 w_look * lookahead_score +
                 w_reg * regret_score +
                 w_dest * dest_score -
                 progress_bonus)
        
        # Update best node
        if score < best_score:
            best_score = score
            best_node = candidate
    
    # Fallback: if no candidate found (e.g., only destination left)
    if best_node is None:
        return destination_node if destination_node in unvisited_nodes else next(iter(unvisited_nodes))
    
    return best_node
