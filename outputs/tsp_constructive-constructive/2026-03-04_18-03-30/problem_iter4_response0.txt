import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, 
                        destination_node: int,
                        unvisited_nodes: Set[int],
                        distance_matrix: np.ndarray) -> int:
    """
    Select next node using improved constructive heuristic with efficient path estimation.
    
    Features:
    1. Robust path estimation using weighted combination of heuristics
    2. Adaptive weights based on remaining nodes and progress
    3. Controlled exploration with candidate pruning
    4. Efficient computation with minimal overhead
    
    Args:
        current_node: Current position in the tour
        destination_node: Final destination to return to
        unvisited_nodes: Set of nodes yet to be visited
        distance_matrix: Square matrix of distances between nodes
    
    Returns:
        Next node to visit
    """
    
    # Base case: only destination remains
    if not unvisited_nodes:
        return destination_node
    if unvisited_nodes == {destination_node}:
        return destination_node
    
    # Dynamic parameters based on remaining nodes
    n_total = distance_matrix.shape[0]
    n_remaining = len(unvisited_nodes)
    progress = 1.0 - (n_remaining / n_total)
    
    # Adaptive weights - simpler and more intuitive
    # Early: focus more on exploration and lookahead
    # Late: focus more on immediate costs and destination
    if progress < 0.3:  # Early stage
        immediate_weight = 0.35
        lookahead_weight = 0.45
        regret_weight = 0.15
        dest_weight = 0.05
        exploration_rate = 0.25
    elif progress < 0.7:  # Middle stage
        immediate_weight = 0.45
        lookahead_weight = 0.35
        regret_weight = 0.12
        dest_weight = 0.08
        exploration_rate = 0.15
    else:  # Late stage
        immediate_weight = 0.60
        lookahead_weight = 0.20
        regret_weight = 0.10
        dest_weight = 0.10
        exploration_rate = 0.05
    
    # Efficient candidate pruning
    candidate_nodes = list(unvisited_nodes - {destination_node})
    
    # Prune to manageable number if too many
    if len(candidate_nodes) > 20:
        # Get distances to candidates
        candidate_dists = [distance_matrix[current_node][c] for c in candidate_nodes]
        
        # Select diverse candidates: closest 10 + 5 random others
        sorted_idx = np.argsort(candidate_dists)
        closest_candidates = [candidate_nodes[i] for i in sorted_idx[:10]]
        
        # Add random candidates for exploration
        remaining_candidates = [c for c in candidate_nodes if c not in closest_candidates]
        if remaining_candidates:
            random_candidates = np.random.choice(remaining_candidates, 
                                               min(5, len(remaining_candidates)), 
                                               replace=False)
            candidate_nodes = closest_candidates + list(random_candidates)
        else:
            candidate_nodes = closest_candidates
    
    # Pre-compute frequently used values
    max_dist = np.max(distance_matrix)
    min_dist = np.min(distance_matrix[distance_matrix > 0])
    dist_range = max_dist - min_dist + 1e-10
    
    def normalize(d: float) -> float:
        """Normalize distance to [0, 1] range."""
        return (d - min_dist) / dist_range
    
    scores = {}
    
    for candidate in candidate_nodes:
        # Immediate cost (normalized)
        immediate_cost = normalize(distance_matrix[current_node][candidate])
        
        # Remaining nodes after choosing candidate
        remaining = unvisited_nodes - {candidate, destination_node}
        
        # Path estimation using multiple heuristics
        if remaining:
            # 1. Nearest neighbor estimate
            distances_to_remaining = [distance_matrix[candidate][r] for r in remaining]
            nearest_dist = min(distances_to_remaining) if distances_to_remaining else 0
            
            # 2. Average distance estimate
            avg_dist = np.mean(distances_to_remaining) if distances_to_remaining else 0
            
            # 3. Minimal spanning tree approximation
            # Simple approximation: average of nearest and average distances
            mst_estimate = 0.6 * nearest_dist + 0.4 * avg_dist
            
            # Scale by number of remaining nodes
            path_estimate = mst_estimate * (len(remaining) ** 0.85)
            
            # Add cost to reach destination from estimated last node
            # Simple heuristic: use average distance to destination
            if remaining:
                avg_to_dest = np.mean([distance_matrix[r][destination_node] for r in remaining])
                path_estimate += avg_to_dest
            
            # Enhanced regret calculation
            if len(distances_to_remaining) > 1:
                sorted_dists = sorted(distances_to_remaining)
                # Regret: difference between best and second best options
                regret = sorted_dists[1] - sorted_dists[0]
                # Scale regret by number of alternatives
                if len(sorted_dists) > 2:
                    regret *= (1 + 0.2 * np.log(len(sorted_dists)))
            else:
                regret = 0
        else:
            # No remaining nodes except destination
            path_estimate = normalize(distance_matrix[candidate][destination_node])
            regret = 0
        
        # Distance to destination
        dest_dist = normalize(distance_matrix[candidate][destination_node])
        
        # Combined score (lower is better)
        score = (immediate_weight * immediate_cost +
                lookahead_weight * normalize(path_estimate) -
                regret_weight * normalize(regret) +
                dest_weight * dest_dist)
        
        scores[candidate] = score
    
    # Controlled exploration mechanism
    if np.random.random() < exploration_rate and len(scores) > 2:
        # Sort candidates by score
        sorted_candidates = sorted(scores.items(), key=lambda x: x[1])
        
        # Select from top 30% of candidates
        cutoff = max(2, int(len(sorted_candidates) * 0.3))
        exploration_pool = [c for c, _ in sorted_candidates[:cutoff]]
        
        # Weighted random selection favoring better candidates
        pool_scores = [scores[c] for c in exploration_pool]
        max_score = max(pool_scores)
        min_score = min(pool_scores)
        
        if max_score > min_score:
            # Exponential weighting: better candidates have much higher probability
            weights = [np.exp(-(score - min_score) / (max_score - min_score + 1e-10))
                      for score in pool_scores]
            weights = np.array(weights) / sum(weights)
            return np.random.choice(exploration_pool, p=weights)
        else:
            return np.random.choice(exploration_pool)
    
    # Greedy selection (exploitation)
    return min(scores, key=scores.get)
