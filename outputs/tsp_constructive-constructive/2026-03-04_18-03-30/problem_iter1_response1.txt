import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """Select next node using look-ahead mechanism with weighted regret and two-step evaluation."""
    
    if not unvisited_nodes:
        return destination_node
    
    # Dynamic parameters based on search progress
    progress = 1.0 - (len(unvisited_nodes) / (distance_matrix.shape[0] - 1))
    alpha = 0.5 * (1.0 + progress)  # Weight for current step (increases with progress)
    beta = 0.3 * (1.0 - progress)   # Weight for look-ahead (decreases with progress)
    gamma = 0.2                     # Weight for regret component
    
    candidate_scores = {}
    
    for candidate in unvisited_nodes:
        # Immediate cost from current to candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # Look-ahead: best possible move after candidate
        remaining = unvisited_nodes - {candidate}
        if remaining:
            # Find minimum cost from candidate to any remaining node
            look_ahead_costs = [distance_matrix[candidate][next_node] for next_node in remaining]
            best_look_ahead = min(look_ahead_costs)
            
            # Regret: difference between best and second best look-ahead
            if len(look_ahead_costs) > 1:
                sorted_costs = sorted(look_ahead_costs)
                regret = sorted_costs[1] - sorted_costs[0]
            else:
                regret = 0
        else:
            # If candidate is last unvisited node, look-ahead is cost to destination
            best_look_ahead = distance_matrix[candidate][destination_node]
            regret = 0
        
        # Two-step cost: current to candidate plus look-ahead
        two_step_cost = immediate_cost + best_look_ahead
        
        # Distance to destination (prefer nodes closer to final destination)
        dest_dist = distance_matrix[candidate][destination_node]
        
        # Combined score with dynamic weights
        # Lower is better (we want to minimize total path length)
        score = (
            alpha * immediate_cost + 
            beta * best_look_ahead - 
            gamma * regret + 
            0.1 * dest_dist  # Small weight for destination proximity
        )
        
        candidate_scores[candidate] = score
    
    # Dynamic threshold for randomization (exploration vs exploitation)
    threshold = 0.2 * progress  # Less random as we progress
    if np.random.random() < threshold and len(candidate_scores) > 1:
        # Occasionally choose a random good candidate (not the worst)
        sorted_candidates = sorted(candidate_scores.keys(), key=lambda x: candidate_scores[x])
        # Random choice from top 30% of candidates
        top_k = max(1, len(sorted_candidates) // 3)
        return np.random.choice(sorted_candidates[:top_k])
    
    # Normally return the best candidate
    return min(candidate_scores, key=candidate_scores.get)
