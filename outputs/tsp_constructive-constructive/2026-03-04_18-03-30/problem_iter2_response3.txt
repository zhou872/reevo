import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select next node using enhanced multi-step lookahead with simulated annealing.
    
    Features:
    1. Multi-step lookahead (up to 3 steps) with adaptive depth based on remaining nodes
    2. Simulated annealing for temperature-controlled exploration
    3. Candidate pruning using k-nearest neighbors for efficiency
    4. Enhanced regret mechanism considering multiple alternatives
    5. Adaptive weights based on search progress
    
    Args:
        current_node: Current position in the tour
        destination_node: Final destination to return to
        unvisited_nodes: Set of nodes yet to be visited
        distance_matrix: Square matrix of distances between nodes
    
    Returns:
        Next node to visit
    """
    
    # Base case: only destination remains
    if unvisited_nodes == {destination_node}:
        return destination_node
    
    # Dynamic parameters based on search progress
    n_total = len(distance_matrix)
    n_remaining = len(unvisited_nodes)
    progress = 1.0 - (n_remaining / n_total)
    
    # Adaptive lookahead depth: deeper when fewer nodes remain
    lookahead_depth = min(3, n_remaining - 1)
    
    # Simulated annealing temperature (decays with progress)
    initial_temp = 1.0
    temperature = initial_temp * (1.0 - progress**2)
    
    # Adaptive weights
    alpha = 0.4 * (1.0 + progress)  # Immediate cost weight
    beta = 0.3 * (1.0 - 0.7 * progress)  # Lookahead weight
    gamma = 0.2  # Regret weight
    delta = 0.1  # Destination proximity weight
    
    # Candidate selection using k-nearest neighbors
    candidate_nodes = list(unvisited_nodes - {destination_node})
    
    # Use nearest neighbor list for efficiency
    if len(candidate_nodes) > 20:
        # Get distances from current node to all candidates
        cand_dists = [distance_matrix[current_node][c] for c in candidate_nodes]
        # Select top 20 closest candidates (more for better quality)
        top_indices = np.argsort(cand_dists)[:20]
        candidate_nodes = [candidate_nodes[i] for i in top_indices]
    
    # Precompute normalized distances once for efficiency
    max_dist = np.max(distance_matrix)
    min_dist = np.min(distance_matrix[distance_matrix > 0])
    
    def normalize(d):
        return (d - min_dist) / (max_dist - min_dist + 1e-10)
    
    scores = {}
    
    for candidate in candidate_nodes:
        # 1. Immediate cost (normalized)
        immediate_cost = normalize(distance_matrix[current_node][candidate])
        
        # 2. Multi-step lookahead evaluation
        remaining_after_candidate = unvisited_nodes - {candidate, destination_node}
        
        if remaining_after_candidate and lookahead_depth > 0:
            lookahead_costs = []
            
            # Consider different paths for lookahead
            for next_node in list(remaining_after_candidate)[:min(5, len(remaining_after_candidate))]:
                path_cost = normalize(distance_matrix[candidate][next_node])
                remaining_after_next = remaining_after_candidate - {next_node}
                
                # Recursive lookahead if depth > 1
                if lookahead_depth > 1 and remaining_after_next:
                    # Find best continuation from next_node
                    best_continuation = float('inf')
                    for third_node in list(remaining_after_next)[:min(3, len(remaining_after_next))]:
                        continuation_cost = normalize(distance_matrix[next_node][third_node])
                        # Add cost to destination
                        to_dest = normalize(distance_matrix[third_node][destination_node])
                        total_continuation = continuation_cost + to_dest
                        best_continuation = min(best_continuation, total_continuation)
                    
                    if best_continuation < float('inf'):
                        path_cost += best_continuation
                    else:
                        # Fall back to direct to destination
                        path_cost += normalize(distance_matrix[next_node][destination_node])
                else:
                    # Single step lookahead
                    path_cost += normalize(distance_matrix[next_node][destination_node])
                
                lookahead_costs.append(path_cost)
            
            if lookahead_costs:
                sorted_costs = sorted(lookahead_costs)
                best_lookahead = sorted_costs[0]
                
                # Enhanced regret: consider multiple alternatives
                if len(sorted_costs) > 1:
                    # Weighted regret considering top 3 alternatives
                    regret_sum = sum(sorted_costs[1:4]) if len(sorted_costs) >= 4 else sum(sorted_costs[1:])
                    avg_alternative = regret_sum / (len(sorted_costs[1:4]) if len(sorted_costs) >= 4 else len(sorted_costs[1:]))
                    regret = avg_alternative - best_lookahead
                else:
                    regret = 0
            else:
                best_lookahead = normalize(distance_matrix[candidate][destination_node])
                regret = 0
        else:
            # Direct to destination if no intermediate nodes
            best_lookahead = normalize(distance_matrix[candidate][destination_node])
            regret = 0
        
        # 3. Destination proximity
        dest_cost = normalize(distance_matrix[candidate][destination_node])
        
        # 4. Combined score (lower is better)
        score = (alpha * immediate_cost +
                beta * best_lookahead -
                gamma * regret +
                delta * dest_cost)
        
        scores[candidate] = score
    
    # Simulated annealing selection
    if temperature > 0.01 and len(scores) > 1:
        # Convert scores to probabilities using Boltzmann distribution
        min_score = min(scores.values())
        exp_scores = {node: np.exp(-(score - min_score) / temperature) 
                     for node, score in scores.items()}
        total = sum(exp_scores.values())
        probabilities = [exp_scores[node] / total for node in scores.keys()]
        
        # Random selection based on probabilities
        return np.random.choice(list(scores.keys()), p=probabilities)
    else:
        # Greedy selection when temperature is low
        return min(scores, key=scores.get)
