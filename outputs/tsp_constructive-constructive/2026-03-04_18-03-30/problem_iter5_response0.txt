import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Select the next node using a two-step look-ahead with enhanced regret heuristic.
    
    Args:
        current_node: Current position node index
        destination_node: Final destination node index
        unvisited_nodes: Set of unvisited node indices
        distance_matrix: 2D array of distances between nodes
    
    Returns:
        Next node index to visit
    """
    if not unvisited_nodes:
        return destination_node
    
    scores = {}
    
    for candidate in unvisited_nodes:
        # Immediate cost: current -> candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # Two-step look-ahead: candidate -> best next node
        remaining_nodes = unvisited_nodes - {candidate}
        
        if remaining_nodes:
            # For each possible next step from candidate
            lookahead_costs = []
            for next_node in remaining_nodes:
                lookahead_cost = distance_matrix[candidate][next_node]
                # Simple look-ahead to the node after next (third step)
                next_remaining = remaining_nodes - {next_node}
                if next_remaining:
                    # Estimate cost from next_node to nearest unvisited
                    min_third_step = min(distance_matrix[next_node][n] for n in next_remaining)
                else:
                    # If no more unvisited, go to destination
                    min_third_step = distance_matrix[next_node][destination_node]
                
                total_lookahead = lookahead_cost + 0.3 * min_third_step  # Weighted lookahead
                lookahead_costs.append(total_lookahead)
            
            # Enhanced regret calculation
            if len(lookahead_costs) >= 2:
                sorted_costs = sorted(lookahead_costs)
                best_lookahead = sorted_costs[0]
                
                # Consider top 3 alternatives for better regret calculation
                k = min(3, len(sorted_costs))
                top_k_avg = np.mean(sorted_costs[:k])
                regret = best_lookahead - top_k_avg
            else:
                # Only one option available
                best_lookahead = lookahead_costs[0]
                regret = 0
            
            # Terminal cost: candidate -> destination (weighted less)
            terminal_cost = 0.2 * distance_matrix[candidate][destination_node]
            
            # Combined score: immediate cost + best lookahead + regret + terminal cost
            score = immediate_cost + best_lookahead - 0.5 * regret + terminal_cost
            
        else:
            # Only one node left to visit
            lookahead_cost = distance_matrix[candidate][destination_node]
            score = immediate_cost + lookahead_cost
        
        scores[candidate] = score
    
    # Deterministic greedy selection
    return min(scores, key=scores.get)
