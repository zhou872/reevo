import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select the next node using a two-step lookahead mechanism with adaptive exploration.
    
    Strategy:
    1. For each candidate node, simulate the optimal next step after it
    2. Balance between immediate gain and future flexibility
    3. Use adaptive threshold to control exploration vs exploitation
    4. Incorporate regret heuristic for nodes that would become hard to reach later
    """
    
    if not unvisited_nodes:
        return destination_node
    
    # Adaptive parameters based on remaining nodes
    n_remaining = len(unvisited_nodes)
    exploration_rate = 0.3 + 0.4 * np.exp(-n_remaining / 10)  # More exploratory early
    lookahead_weight = 0.6 if n_remaining > 5 else 0.8  # Stronger lookahead when many nodes left
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # Immediate cost to reach candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # Regret: cost if we don't visit candidate now
        # Minimum additional cost if we skip candidate now and visit it later
        regret_cost = 0
        if n_remaining > 1:
            # Find the node that would be visited after candidate if we choose it
            temp_unvisited = unvisited_nodes.copy()
            temp_unvisited.remove(candidate)
            
            # Estimate the additional cost to visit candidate later
            # This is the minimum extra distance from any future node to candidate
            min_extra_cost = float('inf')
            for future_node in temp_unvisited:
                for other_node in temp_unvisited:
                    if other_node != future_node:
                        # Cost to detour to candidate from future_node via other_node
                        detour_cost = (distance_matrix[future_node][candidate] + 
                                      distance_matrix[candidate][other_node] - 
                                      distance_matrix[future_node][other_node])
                        min_extra_cost = min(min_extra_cost, detour_cost)
            
            regret_cost = min_extra_cost if min_extra_cost < float('inf') else 0
        
        # Two-step lookahead: simulate optimal next move after candidate
        lookahead_cost = 0
        if n_remaining > 1:
            # Remove candidate from consideration for next step
            remaining_after_candidate = unvisited_nodes.copy()
            remaining_after_candidate.remove(candidate)
            
            # Find the best next node after visiting candidate
            min_next_cost = float('inf')
            for next_candidate in remaining_after_candidate:
                # Cost of going candidate -> next_candidate
                step_cost = distance_matrix[candidate][next_candidate]
                
                # Estimate remaining cost after next_candidate (simplified)
                remaining_after_next = remaining_after_candidate.copy()
                remaining_after_next.remove(next_candidate)
                
                if remaining_after_next:
                    # Approximate future cost as average distance to remaining nodes
                    avg_future_cost = np.mean([
                        distance_matrix[next_candidate][node] 
                        for node in remaining_after_next
                    ])
                else:
                    # If next_candidate is the last unvisited, cost to destination
                    avg_future_cost = distance_matrix[next_candidate][destination_node]
                
                total_next_cost = step_cost + 0.3 * avg_future_cost
                min_next_cost = min(min_next_cost, total_next_cost)
            
            lookahead_cost = min_next_cost
        
        # Terminal cost: distance from candidate to destination (if candidate were last)
        terminal_cost = distance_matrix[candidate][destination_node]
        
        # Calculate final score with weighted components
        score = (immediate_cost + 
                 lookahead_weight * lookahead_cost +
                 0.2 * regret_cost +
                 0.1 * terminal_cost)
        
        # Exploration: occasionally choose suboptimal nodes to avoid local minima
        if np.random.random() < exploration_rate / n_remaining:
            # Adjust score randomly within a bound
            score *= (0.8 + 0.4 * np.random.random())
        
        if score < best_score:
            best_score = score
            best_node = candidate
    
    # Fallback: if something went wrong, use nearest neighbor
    if best_node is None:
        best_node = min(unvisited_nodes, 
                       key=lambda x: distance_matrix[current_node][x])
    
    return best_node
