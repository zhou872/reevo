import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node to visit using two-step look-ahead with enhanced regret heuristic."""
    
    # Step 1: Calculate two-step costs for each candidate
    two_step_costs = {}
    for candidate in unvisited_nodes:
        # First step: current_node -> candidate
        first_step = distance_matrix[current_node][candidate]
        
        # Second step: look ahead to best possible next node from candidate
        remaining_nodes = unvisited_nodes - {candidate}
        
        if remaining_nodes:
            # Find minimum distance from candidate to any remaining node
            min_next_step = min(distance_matrix[candidate][next_node] for next_node in remaining_nodes)
            two_step_costs[candidate] = first_step + min_next_step
        else:
            # If candidate is the last unvisited node, go directly to destination
            two_step_costs[candidate] = first_step + distance_matrix[candidate][destination_node]
    
    # Step 2: Calculate enhanced regret for each candidate
    regrets = {}
    for candidate in unvisited_nodes:
        remaining_nodes = unvisited_nodes - {candidate}
        
        if remaining_nodes:
            # Calculate average of top alternative look-ahead costs
            alt_costs = []
            for next_candidate in remaining_nodes:
                # Two-step cost via alternative candidate
                alt_first_step = distance_matrix[current_node][next_candidate]
                remaining_after_alt = remaining_nodes - {next_candidate}
                
                if remaining_after_alt:
                    alt_second_step = min(distance_matrix[next_candidate][node] for node in remaining_after_alt)
                else:
                    alt_second_step = distance_matrix[next_candidate][destination_node]
                
                alt_costs.append(alt_first_step + alt_second_step)
            
            # Sort alternative costs and take top k (k = min(3, len(alt_costs)))
            k = min(3, len(alt_costs))
            top_alt_costs = sorted(alt_costs)[:k]
            avg_top_alt = np.mean(top_alt_costs) if top_alt_costs else 0
            
            # Regret: difference between current candidate's cost and average of top alternatives
            regrets[candidate] = avg_top_alt - two_step_costs.get(candidate, float('inf'))
        else:
            regrets[candidate] = 0  # No alternatives, zero regret
    
    # Step 3: Combined score balancing immediate cost and regret
    # Normalize both components for fair comparison
    if two_step_costs:
        min_cost = min(two_step_costs.values())
        max_cost = max(two_step_costs.values())
        cost_range = max_cost - min_cost if max_cost > min_cost else 1
        
        min_regret = min(regrets.values())
        max_regret = max(regrets.values())
        regret_range = max_regret - min_regret if max_regret > min_regret else 1
        
        scores = {}
        for candidate in unvisited_nodes:
            # Normalize cost (lower is better) and regret (higher is better)
            norm_cost = (two_step_costs[candidate] - min_cost) / cost_range
            norm_regret = (regrets[candidate] - min_regret) / regret_range
            
            # Balance: 60% weight on cost, 40% on regret
            scores[candidate] = 0.6 * norm_cost - 0.4 * norm_regret
    else:
        # Fallback to simple greedy if something went wrong
        scores = {node: distance_matrix[current_node][node] for node in unvisited_nodes}
    
    # Step 4: Deterministic greedy selection
    return min(scores, key=scores.get)
