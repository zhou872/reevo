import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """Select the next node using two-step look-ahead and adaptive weighting."""
    
    # Base case: only one unvisited node left
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Adaptive weights based on remaining nodes
    remaining_ratio = len(unvisited_nodes) / (distance_matrix.shape[0] - 1)
    
    # Weight adjustment: prioritize different strategies at different stages
    if remaining_ratio > 0.7:
        # Early stage: focus on global structure
        w_immediate, w_lookahead, w_cluster, w_dest = 0.3, 0.4, 0.2, 0.1
    elif remaining_ratio > 0.3:
        # Middle stage: balance
        w_immediate, w_lookahead, w_cluster, w_dest = 0.4, 0.3, 0.2, 0.1
    else:
        # Late stage: focus on completing tour
        w_immediate, w_lookahead, w_cluster, w_dest = 0.5, 0.2, 0.1, 0.2
    
    scores = {}
    
    for candidate in unvisited_nodes:
        # Immediate cost to candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # Two-step look-ahead: best next move from candidate
        remaining = unvisited_nodes - {candidate}
        if remaining:
            # Find the best next node from candidate
            next_costs = [
                distance_matrix[candidate][next_node] + 
                min([distance_matrix[next_node][other] for other in (remaining - {next_node})] or [0])
                for next_node in remaining
            ]
            lookahead_cost = min(next_costs)
        else:
            lookahead_cost = distance_matrix[candidate][destination_node]
        
        # Cluster penalty: penalize nodes that isolate remaining clusters
        cluster_penalty = 0
        if len(remaining) >= 2:
            # Calculate average distance from candidate to remaining nodes
            avg_to_remaining = np.mean([distance_matrix[candidate][node] for node in remaining])
            
            # Calculate average distance among remaining nodes
            remaining_list = list(remaining)
            n_remaining = len(remaining_list)
            if n_remaining > 1:
                intra_cluster_dists = []
                for i in range(n_remaining):
                    for j in range(i + 1, n_remaining):
                        intra_cluster_dists.append(distance_matrix[remaining_list[i]][remaining_list[j]])
                avg_intra_cluster = np.mean(intra_cluster_dists) if intra_cluster_dists else 0
                
                # Penalize if candidate is far from remaining cluster
                if avg_intra_cluster > 0:
                    cluster_penalty = avg_to_remaining / avg_intra_cluster
        
        # Destination proximity (penalize going too close to destination early)
        dest_distance = distance_matrix[candidate][destination_node]
        
        # Composite score (lower is better)
        score = (
            w_immediate * immediate_cost +
            w_lookahead * lookahead_cost +
            w_cluster * cluster_penalty +
            w_dest * (dest_distance if remaining_ratio > 0.2 else -dest_distance)  # Attract to destination late in tour
        )
        
        scores[candidate] = score
    
    # Return node with minimum score
    return min(scores, key=scores.get)
