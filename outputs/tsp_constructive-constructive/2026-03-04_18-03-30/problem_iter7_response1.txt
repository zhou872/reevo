import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Select the next node using a deterministic multi-heuristic approach 
    with two-step lookahead and destination awareness.
    
    Parameters:
    - current_node: Current node index
    - destination_node: Final destination node index
    - unvisited_nodes: Set of unvisited node indices
    - distance_matrix: 2D array of distances between nodes
    
    Returns:
    - Next node index to visit
    """
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    scores = {}
    
    for node in unvisited_nodes:
        # 1. Immediate cost (40%)
        immediate_cost = distance_matrix[current_node][node]
        
        # 2. Two-step lookahead (25%)
        # Look ahead to the next potential step after this node
        remaining_nodes = [n for n in unvisited_nodes if n != node]
        if remaining_nodes:
            # Find minimum cost from current node to node to some other unvisited node
            lookahead_costs = []
            for next_node in remaining_nodes:
                # Cost of current->node->next_node
                cost = immediate_cost + distance_matrix[node][next_node]
                lookahead_costs.append(cost)
            two_step_cost = min(lookahead_costs)
        else:
            # If no other nodes, just use distance to destination
            two_step_cost = immediate_cost + distance_matrix[node][destination_node]
        
        # 3. Cluster penalty (20%)
        # Penalize nodes that are far from other unvisited nodes
        if len(unvisited_nodes) > 1:
            distances_to_others = [distance_matrix[node][other] 
                                  for other in unvisited_nodes if other != node]
            cluster_penalty = np.mean(distances_to_others)
        else:
            cluster_penalty = 0
        
        # 4. Destination proximity (15%)
        # Consider distance to final destination
        dest_proximity = distance_matrix[node][destination_node]
        
        # Weighted combination
        score = (0.40 * immediate_cost + 
                 0.25 * two_step_cost + 
                 0.20 * cluster_penalty + 
                 0.15 * dest_proximity)
        
        scores[node] = score
    
    # Return node with minimum score (deterministic greedy)
    return min(scores.items(), key=lambda x: x[1])[0]
