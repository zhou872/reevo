import numpy as np
from typing import Set

def select_next_node_v2(current_node: int, destination_node: int, 
                       unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Select next node using refined heuristics with progressive intensification.
    
    Enhanced features:
    1. Two-phase adaptive weighting with smoother transitions
    2. Improved cluster cohesion measurement using MST approximation
    3. Progressive intensification from exploration to exploitation
    4. Dynamic candidate filtering based on remaining nodes
    
    Args:
        current_node: Current position in the tour
        destination_node: Final destination to return to
        unvisited_nodes: Set of nodes yet to be visited
        distance_matrix: Square matrix of distances between nodes
    
    Returns:
        Next node to visit
    """
    
    # Base case: only one unvisited node left
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Dynamic parameters based on search progress
    n_total = len(distance_matrix)
    n_remaining = len(unvisited_nodes)
    progress = 1.0 - (n_remaining / n_total)
    
    # Two-phase adaptive weighting with smoother transitions
    if progress < 0.4:  # Early phase: focus on exploration and structure
        w_immediate = 0.35 - 0.15 * (progress / 0.4)
        w_lookahead = 0.40 - 0.10 * (progress / 0.4)
        w_cluster = 0.20 + 0.10 * (progress / 0.4)
        w_dest = 0.05
    else:  # Late phase: focus on completion and exploitation
        late_progress = (progress - 0.4) / 0.6
        w_immediate = 0.50 + 0.10 * late_progress
        w_lookahead = 0.30 - 0.15 * late_progress
        w_cluster = 0.10 - 0.05 * late_progress
        w_dest = 0.10 + 0.10 * late_progress
    
    # Normalize weights to sum to 1
    total_weight = w_immediate + w_lookahead + w_cluster + w_dest
    w_immediate /= total_weight
    w_lookahead /= total_weight
    w_cluster /= total_weight
    w_dest /= total_weight
    
    # Dynamic candidate filtering - balance between quality and efficiency
    candidate_nodes = list(unvisited_nodes - {destination_node})
    
    # If many candidates, filter to most promising ones
    if len(candidate_nodes) > 25:
        # Use combination of distance and centrality
        scores = []
        for candidate in candidate_nodes:
            # Score based on proximity and cluster connectivity
            immediate = distance_matrix[current_node][candidate]
            
            # Estimate cluster connectivity
            remaining = unvisited_nodes - {candidate, destination_node}
            if remaining:
                # Average distance to remaining nodes (excluding destination)
                avg_to_remaining = np.mean([distance_matrix[candidate][node] 
                                          for node in remaining if node != destination_node])
                score = 0.7 * immediate + 0.3 * avg_to_remaining
            else:
                score = immediate
            
            scores.append(score)
        
        # Select top 25 candidates
        top_indices = np.argsort(scores)[:25]
        candidate_nodes = [candidate_nodes[i] for i in top_indices]
    
    scores = {}
    
    for candidate in candidate_nodes:
        # 1. Immediate cost
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Enhanced lookahead with MST approximation
        remaining = unvisited_nodes - {candidate, destination_node}
        
        if remaining:
            # Estimate remaining tour cost using Prim's algorithm approximation
            mst_estimate = 0
            nodes_in_mst = set()
            # Start with the node closest to candidate
            start_node = min(remaining, key=lambda x: distance_matrix[candidate][x])
            nodes_in_mst.add(start_node)
            
            while len(nodes_in_mst) < len(remaining):
                min_edge = float('inf')
                next_node_to_add = None
                
                # Find minimum edge connecting MST to remaining nodes
                for node_in_mst in nodes_in_mst:
                    for node_outside in remaining - nodes_in_mst:
                        edge_cost = distance_matrix[node_in_mst][node_outside]
                        if edge_cost < min_edge:
                            min_edge = edge_cost
                            next_node_to_add = node_outside
                
                if next_node_to_add:
                    mst_estimate += min_edge
                    nodes_in_mst.add(next_node_to_add)
            
            # Add cost from last MST node to destination
            if nodes_in_mst:
                last_to_dest = min(distance_matrix[node][destination_node] 
                                 for node in nodes_in_mst)
                mst_estimate += last_to_dest
            
            # Two-step lookahead with MST estimate
            lookahead_cost = mst_estimate
        else:
            lookahead_cost = distance_matrix[candidate][destination_node]
        
        # 3. Improved cluster cohesion measurement
        remaining_nodes = unvisited_nodes - {candidate}
        cluster_penalty = 0
        
        if len(remaining_nodes) >= 2:
            # Calculate candidate's distance to remaining nodes
            cand_to_remaining = [distance_matrix[candidate][node] 
                               for node in remaining_nodes]
            
            # Calculate distances among remaining nodes
            remaining_list = list(remaining_nodes)
            intra_dists = []
            for i in range(len(remaining_list)):
                for j in range(i + 1, len(remaining_list)):
                    intra_dists.append(distance_matrix[remaining_list[i]][remaining_list[j]])
            
            if intra_dists:
                avg_cand_to_remaining = np.mean(cand_to_remaining)
                avg_intra_cluster = np.mean(intra_dists)
                
                # Penalize if candidate is far from cluster relative to cluster density
                if avg_intra_cluster > 0:
                    # Use normalized ratio to avoid extreme values
                    ratio = avg_cand_to_remaining / avg_intra_cluster
                    cluster_penalty = min(ratio, 3.0)  # Cap at 3x
                else:
                    cluster_penalty = 1.0
        
        # 4. Destination proximity with progressive adjustment
        dest_distance = distance_matrix[candidate][destination_node]
        
        # Early: avoid destination, Late: attract to destination
        if progress < 0.7:
            dest_effect = dest_distance * (1.0 + 0.5 * (0.7 - progress) / 0.7)
        else:
            dest_effect = dest_distance * (1.0 - 0.5 * (progress - 0.7) / 0.3)
        
        # 5. Combined score (lower is better)
        score = (w_immediate * immediate_cost +
                w_lookahead * lookahead_cost +
                w_cluster * cluster_penalty +
                w_dest * dest_effect)
        
        scores[candidate] = score
    
    # Progressive intensification: add small random element in early phases
    if progress < 0.3 and len(scores) > 1:
        # Gradually reduce randomness as we progress
        randomness_factor = 0.1 * (1.0 - progress / 0.3)
        
        if np.random.random() < randomness_factor:
            # Occasionally choose a non-optimal node for exploration
            sorted_candidates = sorted(scores.keys(), key=lambda x: scores[x])
            # Choose from top 30% of candidates
            top_n = max(2, int(len(sorted_candidates) * 0.3))
            return np.random.choice(sorted_candidates[:top_n])
    
    # Return node with minimum score
    return min(scores, key=scores.get)
