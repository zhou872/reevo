import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Select the next node to visit using a multi-heuristic approach with two-step lookahead.
    
    The scoring combines:
    1. Immediate cost to candidate node (40%)
    2. Two-step lookahead cost (30%)
    3. Destination proximity penalty (20%)
    4. Cluster penalty (10%)
    """
    
    # Fixed weights for heuristic combination
    w_immediate = 0.4
    w_lookahead = 0.3
    w_destination = 0.2
    w_cluster = 0.1
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # 1. Immediate cost: distance from current to candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Two-step lookahead: minimum cost from candidate to any remaining node
        remaining_nodes = unvisited_nodes - {candidate}
        
        if remaining_nodes:
            # Calculate the minimum distance from candidate to any other unvisited node
            min_next_step = min(distance_matrix[candidate][next_node] 
                              for next_node in remaining_nodes)
        else:
            # If no other unvisited nodes, use distance to destination
            min_next_step = distance_matrix[candidate][destination_node]
        
        # 3. Destination proximity: how close candidate is to destination
        # Negative weight encourages moving toward destination
        dest_proximity = distance_matrix[candidate][destination_node]
        
        # 4. Cluster penalty: penalize nodes that are far from other unvisited nodes
        # This encourages visiting isolated nodes earlier
        if remaining_nodes:
            avg_dist_to_others = np.mean([distance_matrix[candidate][other] 
                                        for other in remaining_nodes])
        else:
            avg_dist_to_others = 0
        
        # Combine all components into a single score
        score = (w_immediate * immediate_cost +
                 w_lookahead * min_next_step +
                 w_destination * dest_proximity +
                 w_cluster * avg_dist_to_others)
        
        # Update best candidate
        if score < best_score:
            best_score = score
            best_node = candidate
    
    return best_node
