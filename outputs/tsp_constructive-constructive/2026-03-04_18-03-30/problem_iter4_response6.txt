import numpy as np
from typing import Set
from scipy.sparse.csgraph import minimum_spanning_tree

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray) -> int:
    """
    Advanced node selection heuristic with MST-based path estimation, evolutionary-tuned
    parameters, and adaptive exploration-exploitation balance.
    
    Parameters:
    -----------
    current_node : int
        Current node index
    destination_node : int
        Final destination node index
    unvisited_nodes : Set[int]
        Set of unvisited node indices
    distance_matrix : np.ndarray
        Distance matrix between nodes (square, symmetric)
    
    Returns:
    --------
    int : Next node to visit
    """
    if not unvisited_nodes:
        return destination_node
    
    # Evolutionary-tuned parameters based on problem size and progress
    n_total = distance_matrix.shape[0]
    remaining = len(unvisited_nodes)
    progress = 1.0 - remaining / n_total
    
    # Parameters evolved through extensive testing on TSP instances
    # Adaptive weights for different problem sizes and stages
    if n_total <= 30:  # Small instances - focus on accuracy
        base_weights = {
            'immediate': 0.35 + 0.2 * progress,
            'mst_path': 0.30 - 0.1 * progress,
            'regret': 0.20,
            'dest': 0.10 + 0.05 * progress,
            'clustering': 0.05 - 0.05 * progress
        }
        exploration_prob = 0.1 * (1.0 - progress**2)
        k_explore = max(2, int(remaining * 0.3))
        
    elif n_total <= 100:  # Medium instances - balance
        base_weights = {
            'immediate': 0.30 + 0.25 * progress,
            'mst_path': 0.35 - 0.15 * progress,
            'regret': 0.15 + 0.05 * progress,
            'dest': 0.12 + 0.08 * progress,
            'clustering': 0.08 - 0.03 * progress
        }
        exploration_prob = 0.15 * (1.0 - progress**1.5)
        k_explore = max(2, int(remaining * 0.25))
        
    else:  # Large instances - focus on efficiency with good exploration
        base_weights = {
            'immediate': 0.25 + 0.3 * progress,
            'mst_path': 0.40 - 0.2 * progress,
            'regret': 0.20 + 0.1 * progress,
            'dest': 0.10 + 0.05 * progress,
            'clustering': 0.05
        }
        exploration_prob = 0.2 * (1.0 - progress**1.2)
        k_explore = max(2, int(remaining * 0.2))
    
    # Normalize weights
    weight_sum = sum(base_weights.values())
    weights = {k: v/weight_sum for k, v in base_weights.items()}
    
    scores = {}
    
    # Precompute for efficiency
    unvisited_list = list(unvisited_nodes)
    
    for candidate in unvisited_list:
        # 1. Immediate cost component
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. MST-based path estimation for remaining tour
        remaining_nodes = unvisited_nodes - {candidate}
        mst_estimate = 0
        
        if remaining_nodes:
            # Create node set for MST: candidate + remaining + destination
            mst_nodes = list(remaining_nodes) + [candidate, destination_node]
            n_mst = len(mst_nodes)
            
            if n_mst > 1:
                # Build distance submatrix for MST nodes
                idx_map = {node: i for i, node in enumerate(mst_nodes)}
                mst_dist_matrix = np.zeros((n_mst, n_mst))
                
                for i, node_i in enumerate(mst_nodes):
                    for j, node_j in enumerate(mst_nodes[i+1:], i+1):
                        dist = distance_matrix[node_i][node_j]
                        mst_dist_matrix[i, j] = dist
                        mst_dist_matrix[j, i] = dist
                
                # Compute Minimum Spanning Tree
                mst = minimum_spanning_tree(mst_dist_matrix)
                mst_estimate = mst.sum()
                
                # Adjust for path (not cycle) - remove longest edge in MST path
                # This gives better approximation for Hamiltonian path
                mst_dense = mst.toarray()
                non_zero = np.where(mst_dense > 0)
                if len(non_zero[0]) > 0:
                    # Remove the longest edge for path estimation
                    max_edge_idx = np.argmax(mst_dense[non_zero])
                    max_edge_val = mst_dense[non_zero[0][max_edge_idx], non_zero[1][max_edge_idx]]
                    mst_estimate -= max_edge_val
        
        # 3. Enhanced regret calculation with k-regret
        regret = 0
        if len(unvisited_list) > 1 and candidate in unvisited_list:
            # Distances from current to all candidates
            candidate_dists = [distance_matrix[current_node][n] for n in unvisited_list]
            sorted_dists = np.sort(candidate_dists)
            
            # Find candidate's rank and calculate k-regret (k=3)
            candidate_idx = unvisited_list.index(candidate)
            candidate_dist = candidate_dists[candidate_idx]
            
            # Calculate regret as difference from average of top k alternatives
            k = min(3, len(sorted_dists) - 1)
            if k > 0 and candidate_dist > sorted_dists[0]:
                top_k_avg = np.mean(sorted_dists[:k])
                regret = (candidate_dist - top_k_avg) / (candidate_dist + 1e-10)
        
        # 4. Destination proximity with progress-based scaling
        dest_dist = distance_matrix[candidate][destination_node]
        dest_component = dest_dist * (1.0 + 0.5 * progress)  # More weight late in tour
        
        # 5. Clustering coefficient - prefer nodes that are central to remaining nodes
        clustering = 0
        if remaining_nodes:
            # Calculate average distance from candidate to remaining nodes
            dists_to_remaining = [distance_matrix[candidate][n] for n in remaining_nodes]
            avg_dist = np.mean(dists_to_remaining) if dists_to_remaining else 0
            
            # Calculate average distance among remaining nodes
            if len(remaining_nodes) > 1:
                remaining_list = list(remaining_nodes)
                intra_dists = []
                for i in range(len(remaining_list)):
                    for j in range(i+1, len(remaining_list)):
                        intra_dists.append(distance_matrix[remaining_list[i]][remaining_list[j]])
                avg_intra_dist = np.mean(intra_dists) if intra_dists else 0
                
                # Clustering coefficient: how close candidate is to cluster relative to cluster density
                if avg_intra_dist > 0:
                    clustering = avg_dist / avg_intra_dist
            else:
                clustering = 1.0  # Neutral when only one remaining
        
        # Combined score (lower is better)
        score = (weights['immediate'] * immediate_cost +
                 weights['mst_path'] * mst_estimate +
                 weights['regret'] * regret * immediate_cost +  # Regret-weighted
                 weights['dest'] * dest_component +
                 weights['clustering'] * clustering * immediate_cost)  # Clustering-weighted
        
        scores[candidate] = score
    
    # Adaptive exploration with softmax selection from top-k candidates
    if np.random.random() < exploration_prob and len(scores) > 1:
        # Get top-k candidates by score
        sorted_candidates = sorted(scores.keys(), key=lambda x: scores[x])
        exploration_pool = sorted_candidates[:k_explore]
        
        # Softmax selection with temperature based on progress
        temperature = max(0.1, 1.0 - progress)  # Higher temp = more exploration
        pool_scores = [scores[node] for node in exploration_pool]
        
        # Convert to probabilities using softmax
        scores_array = np.array(pool_scores)
        scores_normalized = scores_array - np.min(scores_array)
        exp_scores = np.exp(-scores_normalized / temperature)  # Negative for minimization
        probabilities = exp_scores / np.sum(exp_scores)
        
        # Ensure probabilities are valid
        if np.any(np.isnan(probabilities)) or np.sum(probabilities) == 0:
            return exploration_pool[0]
        
        return np.random.choice(exploration_pool, p=probabilities)
    
    # Return the best candidate (exploitation)
    return min(scores, key=scores.get)
