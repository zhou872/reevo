import numpy as np
import random
from typing import Set

def select_next_node_v2(current_node: int, 
                        destination_node: int, 
                        unvisited_nodes: Set[int], 
                        distance_matrix: np.ndarray,
                        exploration_rate: float = 0.1,
                        weight_noise: float = 0.05) -> int:
    """
    Enhanced node selection with dynamic weight tuning and stochastic exploration.
    
    Args:
        current_node: Current node index
        destination_node: Final destination node index
        unvisited_nodes: Set of unvisited node indices
        distance_matrix: 2D array of distances between nodes
        exploration_rate: Probability of random exploration (default: 0.1)
        weight_noise: Maximum random perturbation to weights (default: 0.05)
    
    Returns:
        Index of next node to visit
    """
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # With exploration_rate probability, perform random exploration
    if random.random() < exploration_rate and len(unvisited_nodes) > 1:
        # Avoid selecting destination unless it's the only option
        candidate_nodes = list(unvisited_nodes)
        if destination_node in candidate_nodes and len(candidate_nodes) > 1:
            candidate_nodes.remove(destination_node)
        return random.choice(candidate_nodes)
    
    # Calculate problem characteristics for dynamic weight tuning
    total_nodes = distance_matrix.shape[0]
    progress_ratio = 1 - len(unvisited_nodes) / total_nodes
    
    # Extract problem statistics for normalization
    non_zero_dists = distance_matrix[distance_matrix > 0]
    dist_mean = np.mean(non_zero_dists)
    dist_std = np.std(non_zero_dists)
    
    # Dynamic weight adjustment based on progress and problem characteristics
    # Early phase: focus on exploration and global structure
    # Late phase: focus on immediate cost and destination
    if progress_ratio < 0.3:  # Early phase
        base_weights = {
            'immediate': 0.3,
            'average': 0.35,
            'std': 0.2,
            'destination': 0.15
        }
    elif progress_ratio < 0.7:  # Middle phase
        base_weights = {
            'immediate': 0.4,
            'average': 0.3,
            'std': 0.15,
            'destination': 0.15
        }
    else:  # Late phase
        base_weights = {
            'immediate': 0.5,
            'average': 0.2,
            'std': 0.1,
            'destination': 0.2
        }
    
    # Add small random perturbation to weights to avoid local optima
    perturbed_weights = {}
    for key, weight in base_weights.items():
        perturbation = random.uniform(-weight_noise, weight_noise)
        perturbed_weights[key] = max(0.05, weight + perturbation)
    
    # Normalize weights to sum to 1
    weight_sum = sum(perturbed_weights.values())
    normalized_weights = {k: v/weight_sum for k, v in perturbed_weights.items()}
    
    best_score = float('inf')
    best_node = None
    
    for candidate in unvisited_nodes:
        # Skip destination if other nodes remain (except in final step)
        if candidate == destination_node and len(unvisited_nodes) > 1:
            continue
        
        # Calculate heuristic components
        immediate_cost = distance_matrix[current_node][candidate] / dist_mean
        
        # Average distance to other unvisited nodes
        other_nodes = unvisited_nodes - {candidate}
        if other_nodes:
            avg_distances = np.mean([distance_matrix[candidate][node] for node in other_nodes])
            avg_normalized = avg_distances / dist_mean
        else:
            avg_normalized = 0
        
        # Standard deviation of distances to other unvisited nodes
        if len(other_nodes) > 1:
            std_distances = np.std([distance_matrix[candidate][node] for node in other_nodes])
            std_normalized = std_distances / dist_std if dist_std > 0 else 0
        else:
            std_normalized = 0
        
        # Distance to destination
        dest_distance = distance_matrix[candidate][destination_node] / dist_mean
        
        # Composite score with dynamic weights
        score = (
            normalized_weights['immediate'] * immediate_cost +
            normalized_weights['average'] * avg_normalized +
            normalized_weights['std'] * std_normalized +
            normalized_weights['destination'] * dest_distance
        )
        
        # Add small random tie-breaking noise (even during exploitation)
        score += random.uniform(0, 1e-10)
        
        if score < best_score:
            best_score = score
            best_node = candidate
    
    # Fallback: if no node was selected (e.g., only destination remains)
    if best_node is None:
        best_node = min(unvisited_nodes)
    
    return best_node
