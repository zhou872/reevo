import numpy as np

def select_next_node_v2(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """
    Selects the next node with two-step look-ahead and cluster-awareness.
    
    Features:
    1. Two-step look-ahead: Evaluates immediate cost plus best next-step cost
    2. Cluster-aware: Penalizes nodes that isolate remaining clusters using centroid approximation
    3. Deterministic greedy: Always selects the minimum score without randomization
    """
    
    # Weights for different components (sum to 1.0)
    w_immediate = 0.35      # Immediate cost from current to candidate
    w_lookahead = 0.35      # Best next-step cost from candidate
    w_cluster = 0.20        # Cluster isolation penalty
    w_destination = 0.10    # Distance to final destination
    
    best_node = None
    best_score = float('inf')
    
    for candidate in unvisited_nodes:
        # 1. Immediate cost: current_node -> candidate
        immediate_cost = distance_matrix[current_node][candidate]
        
        # 2. Two-step look-ahead: candidate -> best_next
        remaining_after_candidate = unvisited_nodes - {candidate}
        if remaining_after_candidate:
            # Find minimum cost from candidate to any other unvisited node
            lookahead_costs = [distance_matrix[candidate][next_node] 
                              for next_node in remaining_after_candidate]
            lookahead_cost = min(lookahead_costs)
        else:
            # If candidate is the last unvisited node, cost to destination
            lookahead_cost = distance_matrix[candidate][destination_node]
        
        # 3. Cluster-aware heuristic: Penalize nodes that isolate remaining clusters
        if len(remaining_after_candidate) >= 2:
            # Sample distances from candidate to other unvisited nodes
            sample_size = min(5, len(remaining_after_candidate))
            sampled_nodes = list(remaining_after_candidate)[:sample_size]
            
            # Calculate average distance from candidate to sampled nodes
            sampled_distances = [distance_matrix[candidate][n] for n in sampled_nodes]
            avg_distance = np.mean(sampled_distances)
            
            # Calculate average inter-distance among sampled nodes (cluster cohesion)
            inter_distances = []
            for i in range(len(sampled_nodes)):
                for j in range(i+1, len(sampled_nodes)):
                    inter_distances.append(distance_matrix[sampled_nodes[i]][sampled_nodes[j]])
            
            if inter_distances:  # Avoid division by zero
                avg_inter_distance = np.mean(inter_distances)
                # Isolation factor: how much farther candidate is from cluster vs cluster's internal distances
                cluster_penalty = avg_distance / (avg_inter_distance + 1e-8)
            else:
                cluster_penalty = 1.0
        else:
            cluster_penalty = 1.0  # Neutral when few nodes remain
        
        # 4. Destination cost: candidate -> destination_node
        destination_cost = distance_matrix[candidate][destination_node]
        
        # Calculate weighted score (lower is better)
        score = (w_immediate * immediate_cost +
                 w_lookahead * lookahead_cost +
                 w_cluster * cluster_penalty +
                 w_destination * destination_cost)
        
        if score < best_score:
            best_score = score
            best_node = candidate
    
    return best_node
