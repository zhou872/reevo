import numpy as np
import random
from typing import Set, Dict, Tuple
import heapq
from collections import defaultdict

class AdaptiveParameterManager:
    """Reinforcement learning inspired parameter adaptation"""
    
    def __init__(self, n_nodes: int):
        self.n_nodes = n_nodes
        self.performance_history = []
        self.weight_history = []
        self.learning_rate = 0.1
        self.momentum = 0.9
        
        # Initial adaptive weights
        self.weights = {
            'immediate': 0.35,
            'lookahead': 0.30,
            'diversity': 0.20,
            'destination': 0.15
        }
        
        # Weight adjustment deltas (learned through experience)
        self.delta = {k: 0.0 for k in self.weights}
        
    def update_weights(self, tour_length: float, progress: float) -> None:
        """Reinforcement learning update based on performance"""
        if not self.performance_history:
            self.performance_history.append(tour_length)
            self.weight_history.append(self.weights.copy())
            return
            
        # Calculate performance improvement
        prev_length = self.performance_history[-1]
        improvement = (prev_length - tour_length) / prev_length if prev_length > 0 else 0
        
        # Adjust weights based on improvement and progress
        progress_factor = 1.0 - abs(0.5 - progress)  # Highest at start and end
        
        # Learning rule: reinforce what worked well
        if improvement > 0:
            # Successful strategy, reinforce current weights
            adjustment = self.learning_rate * improvement * progress_factor
            for key in self.weights:
                self.delta[key] = self.momentum * self.delta[key] + adjustment
                self.weights[key] = min(0.9, max(0.1, self.weights[key] + self.delta[key]))
        else:
            # Unsuccessful, explore new weight combinations
            for key in self.weights:
                self.delta[key] = self.momentum * self.delta[key] - 0.01 * progress_factor
                self.weights[key] = min(0.9, max(0.1, self.weights[key] + self.delta[key]))
        
        # Normalize weights
        total = sum(self.weights.values())
        for key in self.weights:
            self.weights[key] /= total
        
        # Store history
        self.performance_history.append(tour_length)
        self.weight_history.append(self.weights.copy())
        
    def get_weights(self, progress: float) -> Dict[str, float]:
        """Get current weights with progress-based modulation"""
        base_weights = self.weights.copy()
        
        # Progress-based adjustments
        progress_mod = {
            'immediate': 0.8 - 0.6 * progress,
            'lookahead': 0.4 + 1.2 * progress,
            'diversity': 1.2 - 1.0 * progress,
            'destination': 0.2 + 0.8 * progress
        }
        
        # Apply modulation
        modulated = {}
        for key in base_weights:
            modulated[key] = base_weights[key] * progress_mod[key]
        
        # Normalize
        total = sum(modulated.values())
        for key in modulated:
            modulated[key] /= total
        
        return modulated

class EfficientCache:
    """Optimized caching for distance computations"""
    
    def __init__(self, distance_matrix: np.ndarray):
        self.distance_matrix = distance_matrix
        self.n_nodes = len(distance_matrix)
        self.pair_cache = {}
        self.mst_cache = {}
        self.nn_cache = {}
        
    def get_distance(self, a: int, b: int) -> float:
        """Get cached distance with lazy computation"""
        key = (min(a, b), max(a, b))
        if key not in self.pair_cache:
            self.pair_cache[key] = self.distance_matrix[a][b]
        return self.pair_cache[key]
    
    def estimate_mst_cost(self, nodes: Set[int]) -> float:
        """Fast MST cost estimation using caching"""
        key = tuple(sorted(nodes))
        if key in self.mst_cache:
            return self.mst_cache[key]
        
        if len(nodes) <= 1:
            return 0.0
        
        # Prim's algorithm with binary heap
        remaining = set(nodes)
        start = next(iter(remaining))
        visited = {start}
        remaining.remove(start)
        
        heap = []
        for node in remaining:
            dist = self.get_distance(start, node)
            heapq.heappush(heap, (dist, node))
        
        total_cost = 0.0
        while heap and remaining:
            dist, node = heapq.heappop(heap)
            if node not in visited:
                visited.add(node)
                remaining.remove(node)
                total_cost += dist
                
                # Add new edges
                for next_node in remaining:
                    if next_node not in visited:
                        new_dist = self.get_distance(node, next_node)
                        heapq.heappush(heap, (new_dist, next_node))
        
        self.mst_cache[key] = total_cost
        return total_cost

def select_next_node_v2(current_node: int, destination_node: int, 
                        unvisited_nodes: Set[int], distance_matrix: np.ndarray,
                        param_manager: AdaptiveParameterManager = None,
                        cache: EfficientCache = None) -> int:
    """
    Enhanced node selection with RL-based adaptation and computational optimizations.
    
    Features:
    1. Reinforcement learning weight adaptation based on tour performance
    2. Efficient beam search with adaptive pruning
    3. Multi-heuristic ensemble with optimized computation
    4. Advanced caching for repeated distance calculations
    """
    
    if len(unvisited_nodes) == 1:
        return next(iter(unvisited_nodes))
    
    # Initialize adaptive components if not provided
    if param_manager is None:
        param_manager = AdaptiveParameterManager(len(distance_matrix))
    if cache is None:
        cache = EfficientCache(distance_matrix)
    
    # Calculate progress for adaptive behavior
    n_total = len(distance_matrix)
    n_remaining = len(unvisited_nodes)
    progress = 1.0 - (n_remaining / n_total)
    
    # Get dynamically adapted weights
    weights = param_manager.get_weights(progress)
    
    # Normalization parameters
    max_dist = np.max(distance_matrix)
    min_dist = np.min(distance_matrix[distance_matrix > 0])
    dist_range = max_dist - min_dist + 1e-10
    
    def normalize(d: float) -> float:
        return (d - min_dist) / dist_range
    
    # Candidate pruning with adaptive beam width
    candidates = list(unvisited_nodes)
    beam_width = max(3, min(8, int(10 * (1 - progress))))
    
    if len(candidates) > beam_width * 3:
        # Strategic candidate selection
        current_dists = [(c, cache.get_distance(current_node, c)) for c in candidates]
        current_dists.sort(key=lambda x: x[1])
        
        # Select diverse candidates
        selected = []
        selected.extend([c for c, _ in current_dists[:beam_width]])  # Closest
        selected.extend([c for c, _ in current_dists[-beam_width:]])  # Farthest
        
        # Add random candidates for exploration
        middle = len(candidates) // 2
        if len(candidates) > beam_width * 2:
            selected.extend(random.sample(candidates[middle-beam_width:middle+beam_width], 
                                        min(beam_width, len(candidates)-middle*2)))
        
        candidates = list(set(selected))
    
    # Main evaluation with efficient lookahead
    best_candidate = None
    best_score = float('inf')
    
    for candidate in candidates:
        # 1. Immediate cost (cached)
        immediate_cost = normalize(cache.get_distance(current_node, candidate))
        
        # 2. Efficient beam search lookahead
        if n_remaining > 2:
            remaining = unvisited_nodes - {candidate, destination_node}
            lookahead_cost = _efficient_beam_lookahead(candidate, remaining, destination_node, 
                                                      beam_width, progress, cache, normalize)
        else:
            # Direct to destination case
            lookahead_cost = normalize(cache.get_distance(candidate, destination_node))
        
        # 3. Destination proximity
        dest_proximity = normalize(cache.get_distance(candidate, destination_node))
        
        # 4. Diversity score (encourage exploration early)
        diversity_score = 0.0
        if progress < 0.7:  # Only apply diversity early in tour
            other_candidates = [c for c in candidates if c != candidate]
            if other_candidates:
                # Use cached distances
                avg_dist = np.mean([normalize(cache.get_distance(candidate, other)) 
                                  for other in random.sample(other_candidates, 
                                                           min(5, len(other_candidates)))])
                diversity_score = avg_dist
        
        # 5. Path quality estimation using cached MST
        if n_remaining > 3 and progress < 0.8:
            remaining = unvisited_nodes - {candidate}
            mst_estimate = cache.estimate_mst_cost(remaining)
            path_quality = normalize(mst_estimate) * (1.0 - progress)
        else:
            path_quality = 0.0
        
        # Combined score with adaptive weights
        score = (weights['immediate'] * immediate_cost +
                weights['lookahead'] * lookahead_cost +
                weights['destination'] * dest_proximity -
                weights['diversity'] * diversity_score +
                0.1 * path_quality)  # Small path quality influence
        
        # Add minimal noise for tie-breaking
        score *= (1.0 + random.uniform(-0.001, 0.001))
        
        if score < best_score:
            best_score = score
            best_candidate = candidate
    
    return best_candidate if best_candidate is not None else random.choice(list(unvisited_nodes))

def _efficient_beam_lookahead(start_node: int, remaining_nodes: Set[int], 
                             destination: int, beam_width: int, progress: float,
                             cache: EfficientCache, normalize) -> float:
    """Optimized beam search for lookahead evaluation"""
    
    if not remaining_nodes:
        return normalize(cache.get_distance(start_node, destination))
    
    # Initialize beam with start node
    beam = [(0.0, start_node, frozenset())]  # (cost, last_node, visited)
    
    # Adaptive lookahead depth
    lookahead_depth = min(3, len(remaining_nodes))
    if len(remaining_nodes) > 10:
        lookahead_depth = 2
    if progress > 0.8:  # Shallower lookahead near end
        lookahead_depth = min(2, lookahead_depth)
    
    for depth in range(lookahead_depth):
        new_beam = []
        
        for acc_cost, last_node, visited in beam:
            # Get unvisited nodes after this path
            unvisited = remaining_nodes - set(visited)
            
            if not unvisited:
                # Path has visited all remaining nodes
                final_cost = acc_cost + normalize(cache.get_distance(last_node, destination))
                new_beam.append((final_cost, last_node, visited))
                continue
            
            # Select promising next nodes
            possible_next = list(unvisited)
            if len(possible_next) > beam_width:
                # Quick evaluation based on distance and destination proximity
                scores = []
                for node in possible_next:
                    dist = normalize(cache.get_distance(last_node, node))
                    dest_dist = normalize(cache.get_distance(node, destination))
                    score = dist * 0.7 + dest_dist * 0.3
                    scores.append((score, node))
                
                scores.sort(key=lambda x: x[0])
                possible_next = [node for _, node in scores[:beam_width]]
            
            for next_node in possible_next:
                new_cost = acc_cost + normalize(cache.get_distance(last_node, next_node))
                new_visited = frozenset(visited | {next_node})
                new_beam.append((new_cost, next_node, new_visited))
        
        # Prune beam
        if not new_beam:
            break
            
        new_beam.sort(key=lambda x: x[0])
        beam = new_beam[:beam_width]
    
    # Estimate remaining path for each beam entry
    if not beam:
        # Fallback: nearest neighbor estimate
        return _fast_nearest_neighbor_estimate(start_node, remaining_nodes, destination, cache, normalize)
    
    final_costs = []
    for acc_cost, last_node, visited in beam:
        unvisited_after = remaining_nodes - set(visited)
        
        if not unvisited_after:
            # Direct to destination
            total_cost = acc_cost + normalize(cache.get_distance(last_node, destination))
        else:
            # Fast estimation of remaining path
            if len(unvisited_after) <= 3:
                # Exact computation for small sets
                best_cost = float('inf')
                for path in _generate_paths(last_node, unvisited_after, destination, cache, normalize):
                    best_cost = min(best_cost, acc_cost + path)
                total_cost = best_cost
            else:
                # Fast heuristic estimation
                estimate = _fast_path_estimate(last_node, unvisited_after, destination, cache, normalize)
                total_cost = acc_cost + estimate
        
        final_costs.append(total_cost)
    
    return min(final_costs) if final_costs else float('inf')

def _fast_nearest_neighbor_estimate(start: int, nodes: Set[int], 
                                   destination: int, cache: EfficientCache, normalize) -> float:
    """Quick nearest neighbor path estimation"""
    total = 0.0
    current = start
    remaining = set(nodes)
    
    while remaining:
        next_node = min(remaining, key=lambda x: cache.get_distance(current, x))
        total += normalize(cache.get_distance(current, next_node))
        current = next_node
        remaining.remove(next_node)
    
    total += normalize(cache.get_distance(current, destination))
    return total

def _generate_paths(start: int, nodes: Set[int], destination: int, 
                   cache: EfficientCache, normalize):
    """Generate all possible paths for small node sets"""
    if not nodes:
        yield normalize(cache.get_distance(start, destination))
        return
    
    for node in nodes:
        remaining = nodes - {node}
        cost = normalize(cache.get_distance(start, node))
        for rest_cost in _generate_paths(node, remaining, destination, cache, normalize):
            yield cost + rest_cost

def _fast_path_estimate(start: int, nodes: Set[int], destination: int,
                       cache: EfficientCache, normalize) -> float:
    """Fast path estimation using multiple heuristics"""
    node_list = list(nodes)
    
    # Method 1: MST-based estimate
    mst_cost = cache.estimate_mst_cost(set(node_list))
    mst_est = normalize(mst_cost) * 1.2  # MST approximation factor
    
    # Method 2: Average connection
    start_dists = [normalize(cache.get_distance(start, node)) for node in node_list]
    dest_dists = [normalize(cache.get_distance(node, destination)) for node in node_list]
    avg_est = (np.mean(start_dists) + np.mean(dest_dists)) * 0.5 * len(node_list)
    
    # Method 3: Cluster approximation
    if len(node_list) > 2:
        # Group by proximity to start and destination
        cluster_score = (min(start_dists) + min(dest_dists)) * 0.6 + \
                       (np.mean(start_dists) + np.mean(dest_dists)) * 0.4
    else:
        cluster_score = avg_est
    
    # Weighted ensemble
    return 0.4 * mst_est + 0.4 * avg_est + 0.2 * cluster_score
